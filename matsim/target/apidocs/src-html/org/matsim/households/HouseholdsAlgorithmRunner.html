020package org.matsim.households;
021
022import java.util.ArrayList;
023import java.util.List;
024
025import org.apache.log4j.Logger;
026import org.matsim.core.utils.misc.Counter;
027import org.matsim.households.algorithms.HouseholdAlgorithm;
028
029
030/**
031 * @author jjoubert
032 * @author dgrether
033 *
034 */
035public class HouseholdsAlgorithmRunner {
036        
037        private static final Logger log = Logger.getLogger(HouseholdsAlgorithmRunner.class);
038        
039        private final List<HouseholdAlgorithm> householdAlgorithms = new ArrayList<HouseholdAlgorithm>();
040
041        /**
042         * Run all the algorithms added to the container. 
043         */
044        public final void runAlgorithms(Households hh){
045                        for(int i = 0; i < this.householdAlgorithms.size(); i++){
046                                HouseholdAlgorithm algorithm = this.householdAlgorithms.get(i);
047                                log.info("Running algorithm " + algorithm.getClass().getName());
048                                Counter c = new Counter ("  household # ");
049                                for(Household household : hh.getHouseholds().values()){
050                                        algorithm.run(household);
051                                        c.incCounter();
052                                }
053                                c.printCounter();
054                                log.info("Done running algorithm.");
055                        }
056        }
057
058        public void runAlgorithms(Household h){
059                for(HouseholdAlgorithm algorithm : this.householdAlgorithms){
060                        log.info("Running algorithm " + algorithm.getClass().getName() + " on household " + h.getId());
061                        algorithm.run(h);
062                }               
063        }
064        
065        
066        /**
067         * Removes all the algorithms from the Households container.
068         */
069        public final void clearAlgorithms(){
070                this.householdAlgorithms.clear();
071        }
072        
073        /**
074         * Removes the first instance found of the algorithm from the list. It is 
075         * possible that the same algorithm can appear multiple times in the list.
076         * @param algorithm
077         * @return
078         */
079        public boolean removeAlgorithm(final HouseholdAlgorithm algorithm){
080                return this.householdAlgorithms.remove(algorithm);
081        }
082        
083        
084        /**
085         * Add the algorithm to the container. Algorithms will be executed in the
086         * same sequence in which they are added.
087         * @param algorithm
088         */
089        public final void addAlgorithm(final HouseholdAlgorithm algorithm){
090                this.householdAlgorithms.add(algorithm);
091        }
092}
