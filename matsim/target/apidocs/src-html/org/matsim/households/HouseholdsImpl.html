020package org.matsim.households;
021
022import java.util.LinkedHashMap;
023import java.util.Map;
024
025import org.matsim.api.core.v01.Id;
026import org.matsim.utils.objectattributes.ObjectAttributes;
027
028/**
029 * Basic implementation of the Households container, a pure data class
030 * 
031 * @author dgrether
032 */
033public class HouseholdsImpl implements Households {
034        
035//      private final static Logger log = Logger.getLogger(HouseholdImpl.class);
036
037        private HouseholdsFactory factory;
038
039        private Map<Id<Household>, Household> households;
040        
041        private final ObjectAttributes householdAttributes = new ObjectAttributes();
042        
043        public HouseholdsImpl(){
044                this.households = new LinkedHashMap<Id<Household>, Household>();
045                this.factory = new HouseholdsFactoryImpl();
046        }
047        
048        
049        /**
050         * Adds the household to the container. If streaming is set, the household
051         * is added, the algorithms are run on the household, and it is subsequently
052         * removed. If not, the household is added incrementally to the container.
053         * @param household
054         * @throws IllegalArgumentException if the container already includes the 
055         *              {@link Id} of the household being added.
056         */
057        public final void addHousehold(final Household household){
058                /* Validate that a household with the same Id does not exist yet. */
059                if(this.getHouseholds().containsKey(household.getId())){
060                        throw new IllegalArgumentException("Household with Id " + household.getId() + 
061                                        " already exisits.");
062                }
063                this.households.put(household.getId(), household);
064        }
065        
066        @Override
067        public HouseholdsFactory getFactory() {
068                return this.factory;
069        }
070
071        public void setFactory(HouseholdsFactory factory) {
072                this.factory = factory;
073        }
074        
075        @Override
076        public Map<Id<Household>, Household> getHouseholds() {
077                return this.households;
078        }
079
080        @Override
081        public ObjectAttributes getHouseholdAttributes() {
082                return this.householdAttributes;
083        }
084
085}
