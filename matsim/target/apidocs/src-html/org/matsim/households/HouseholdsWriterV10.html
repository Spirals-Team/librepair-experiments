040/**
041 * @author dgrether
042 *
043 */
044public class HouseholdsWriterV10 extends MatsimXmlWriter implements HouseholdAlgorithm{
045
046        private List<Tuple<String, String>> atts = new ArrayList<>();
047        private Households households;
048        private final Map<Class<?>,AttributeConverter<?>> attributeConverters = new HashMap<>();
049
050        public HouseholdsWriterV10(Households households) {
051                this.households = households;
052        }
053
054        public <T> void putAttributeConverter(Class<T> clazz, AttributeConverter<T> converter) {
055                this.attributeConverters.put( clazz , converter );
056        }
057
058        public void putAttributeConverters( final Map<Class<?>, AttributeConverter<?>> converters ) {
059                this.attributeConverters.putAll( converters );
060        }
061
062        public void writeFile(String filename) throws UncheckedIOException {
063                this.openFileAndWritePreamble(filename);
064                this.writeHouseholds(this.households);
065                this.writeEndAndCloseFile();
066        }
067
068        public void writeV1(String filename) {
069
070        }
071
072        /*package*/ void openFileAndWritePreamble(String filename){
073                this.openFile(filename);
074                this.writeXmlHead();
075                this.writeHeader();
076        }
077        
078        /*package*/ void writeEndAndCloseFile(){
079                
080                this.writeEndTag(HouseholdsSchemaV10Names.HOUSEHOLDS);
081                this.close();
082        }
083        
084        
085        private void writeHeader(){
086                atts.clear();
087                atts.add(createTuple(XMLNS, MatsimXmlWriter.MATSIM_NAMESPACE));
088                atts.add(createTuple(XMLNS + ":xsi", DEFAULTSCHEMANAMESPACELOCATION));
089                atts.add(createTuple("xsi:schemaLocation", MATSIM_NAMESPACE + " " + DEFAULT_DTD_LOCATION + "households_v1.0.xsd"));
090                this.writeStartTag(HouseholdsSchemaV10Names.HOUSEHOLDS, atts);
091        }
092        
093        private void writeHouseholds(Households basicHouseholds) throws UncheckedIOException {
094                Counter counter = new Counter("[HouseholdsWriter] dumped household # ");
095                for (Household h : basicHouseholds.getHouseholds().values()) {
096                        this.writeHousehold(h);
097                        counter.incCounter();
098                }
099                counter.printCounter();
100        }
101
102        /*package*/ void writeHousehold(Household h) throws UncheckedIOException {
103                this.atts.clear();
104                atts.add(createTuple(HouseholdsSchemaV10Names.ID, h.getId().toString()));
105                this.writeStartTag(HouseholdsSchemaV10Names.HOUSEHOLD, atts);
106                if ((h.getMemberIds() != null) && !h.getMemberIds().isEmpty()){
107                        this.writeMembers(h.getMemberIds());
108                }
109                if ((h.getVehicleIds() != null) && !h.getVehicleIds().isEmpty()) {
110                        this.writeStartTag(HouseholdsSchemaV10Names.VEHICLES, null);
111                        for (Id<Vehicle> id : h.getVehicleIds()){
112                                atts.clear();
113                                atts.add(createTuple(HouseholdsSchemaV10Names.REFID, id.toString()));
114                                this.writeStartTag(HouseholdsSchemaV10Names.VEHICLEDEFINITIONID, atts, true);
115                        }
116                        this.writeEndTag(HouseholdsSchemaV10Names.VEHICLES);
117                }
118                if (h.getIncome() != null){
119                        this.writeIncome(h.getIncome());
120                }
121
122                AttributesXmlWriterDelegate attributesWriter = new AttributesXmlWriterDelegate();
123                attributesWriter.putAttributeConverters(this.attributeConverters);
124                try {
125                        this.writer.write(NL);
126                } catch (IOException e) {
127                        e.printStackTrace();
128                }
129                attributesWriter.writeAttributes( "\t\t" , this.writer , h.getAttributes() );
130
131                this.writeEndTag(HouseholdsSchemaV10Names.HOUSEHOLD);
132        }
133
134        private void writeIncome(Income income) throws UncheckedIOException {
135                atts.clear();
136                if (income.getCurrency() != null) {
137                        atts.add(createTuple(HouseholdsSchemaV10Names.CURRENCY,income.getCurrency()));
138                }
139                atts.add(createTuple(HouseholdsSchemaV10Names.PERIOD, income.getIncomePeriod().toString()));
140                this.writeStartTag(HouseholdsSchemaV10Names.INCOME, atts);
141                this.writeContent(Double.toString(income.getIncome()), true);
142                this.writeEndTag(HouseholdsSchemaV10Names.INCOME);
143        }
144
145        private void writeMembers(List<Id<Person>> memberIds) throws UncheckedIOException {
146                this.writeStartTag(HouseholdsSchemaV10Names.MEMBERS, null);
147                for (Id<Person> id : memberIds){
148                        atts.clear();
149                        atts.add(createTuple(HouseholdsSchemaV10Names.REFID, id.toString()));
150                        this.writeStartTag(HouseholdsSchemaV10Names.PERSONID, atts, true);
151                }
152                this.writeEndTag(HouseholdsSchemaV10Names.MEMBERS);
153        }
154
155        @Override
156        public void run(Household household) {
157                writeHousehold(household);
158        }
