033/**
034 * Represents a unique identifier.
035 * 
036 * Note that Ids should not contain any whitespace characters (spaces, tabs, newlines, ...), 
037 * as this may lead to problems when Ids are written to file and read back in.
038 * 
039 *  @author mrieser / Senozon AG
040 */
041public abstract class Id<T> implements Comparable<Id<T>> {
042
043        private final static Map<Class<?>, Map<String, Id<?>>> cache = new ConcurrentHashMap<Class<?>, Map<String, Id<?>>>();
044        
045        
046        public static <T> Id<T> create(final long key, final Class<T> type) {
047                return create(Long.toString(key), type);
048        }
049        
050        public static <T> Id<T> create(final Id<?> id, final Class<T> type) {
051                if (id == null) {
052                        return null;
053                }
054                return create(id.toString(), type);
055        }
056        
057        /**
058         * This method supports a cache where ids are stored and re-used per type.   
059         */
060        public static <T> Id<T> create(final String key, final Class<T> type) {
061                Map<String, Id<?>> map = cache.get(type);
062                if (map == null) {
063                        map = new ConcurrentHashMap<String, Id<?>>();
064                        cache.put(type, map);
065                }
066                Gbl.assertNotNull(key);
067                Id<?> id = map.get(key);
068                if (id == null) {
069                        id = new IdImpl<T>(key);
070                        map.put(key, id);
071                }
072                
073                return (Id<T>) id;
074        }
075        
076        /**
077         * @return <code>0</code> when the two objects being compared are the same objects, other values according to their ids being compared to each other.
078         * 
079         * @throws IllegalArgumentException when the two objects being compared have the same id, but are not the same object because this means they must have different generic types
080         */
081        @Override
082        public int compareTo(Id<T> o) throws IllegalArgumentException {
083                int res = this.toString().compareTo(o.toString());
084//              if (res == 0) {   // FIXME temporary relax the check until the Id migration has taken place
085//                      if (equals(o)) {
086//                              return 0;
087//                      }
088//                      throw new IllegalArgumentException("The ids are equal but of different types.");
089//              }
090                return res;
091        }
092        
093        @Override
094        public boolean equals(Object obj) {
095                if (obj instanceof Id) {
096                        return this.compareTo((Id) obj) == 0;
097                }
098                return false;
099//              return this == obj; // FIXME temporary relax the check until the Id migration has taken place
100                // all other objects have to be different by definition, as long as the cache is correctly implemented
101        }
102
103        
104        /**
105         * The default implementation to be used for Ids.
106         * Have this as a separate class instead of integrated into the Id class
107         * to allow for future optimization of Ids.
108         * 
109         * @author mrieser
110         *
111         * @param <T>
112         */
113        private static class IdImpl<T> extends Id<T> {
114
115                private final String id; 
116                
117                /*package*/ IdImpl(final String id) {
118                        this.id = id;
119                }
120
121                @Override
122                public int hashCode() {
123                        return this.id.hashCode();
124                }
125                
126                @Override
127                public String toString() {
128                        return this.id;
129                }
130        }
131
132        // helper classes for some common cases:
133        public static Id<Person> createPersonId( final long key ) {
134                return create( key, Person.class ) ;
135        }
136        public static Id<Person> createPersonId( final Id<?> id ) {
137                return create( id, Person.class ) ;
138        }
139        public static Id<Person> createPersonId( final String str ) {
140                return create( str, Person.class ) ;
141        }
142        public static Id<Link> createLinkId( final long key ) {
143                return create( key, Link.class ) ;
144        }
145        public static Id<Link> createLinkId( final Id<?> id ) {
146                return create( id, Link.class ) ;
147        }
148        public static Id<Link> createLinkId( final String str ) {
149                return create( str, Link.class ) ;
150        }
151        public static Id<Node> createNodeId( final long key ) {
152                return create( key, Node.class ) ;
153        }
154        public static Id<Node> createNodeId( final Id<?> id ) {
155                return create( id, Node.class ) ;
156        }
157        public static Id<Node> createNodeId( final String str ) {
158                return create( str, Node.class ) ;
159        }
160        public static Id<Vehicle> createVehicleId( final long key ) {
161                return create( key, Vehicle.class ) ;
162        }
163        public static Id<Vehicle> createVehicleId( final Id<?> id ) {
164                return create( id, Vehicle.class ) ;
165        }
166        public static Id<Vehicle> createVehicleId( final String str ) {
167                return create( str, Vehicle.class ) ;
168        }
