020package org.matsim.api.core.v01.network;
021
022import java.util.Set;
023
024import org.matsim.api.core.v01.BasicLocation;
025import org.matsim.core.utils.misc.Time;
026import org.matsim.utils.objectattributes.attributable.Attributable;
027
028/**
029 * This interface deliberately does NOT have a back pointer ...
030 * ... since, at this level, one should be able to get the relevant container from
031 * the context.
032 * (This becomes clear if you think about a nodeId/linkId given by person.)
033 */
034public interface Link extends BasicLocation<Link>, Attributable {
035
036
037        /**
038         * Sets this link's non-<code>null</code> upstream node.
039         *
040         * @param node
041         *            the <code>BasicNodeI</code> to be set
042         *
043         * @return <true> if <code>node</code> has been set and <code>false</code>
044         *         otherwise
045         *
046         * @throws IllegalArgumentException
047         *             if <code>node</code> is <code>null</code>
048         */
049        public boolean setFromNode(Node node);
050
051        /**
052         * Sets this link's non-<code>null</code> downstream node.
053         *
054         * @param node
055         *            the <code>BasicNodeI</code> to be set
056         *
057         * @return <code>true</code> if <code>node</code> has been set and
058         *         <code>false</code> otherwise
059         *
060         * @throws IllegalArgumentException
061         *             if <code>node</code> is <code>null</code>
062         */
063        public boolean setToNode(Node node);
064
065        /**
066         * @return this link's downstream node
067         */
068        public Node getToNode();
069
070        /**
071         * @return this link's upstream node
072         */
073        public Node getFromNode();
074
075
076        public double getLength();
077
078        public double getNumberOfLanes();
079
080        public double getNumberOfLanes(double time);
081
082        public double getFreespeed();
083
084        public double getFreespeed(final double time);
085
086        /**
087         * This method returns the capacity as set in the xml defining the network. Be aware
088         * that this capacity is not normalized in time, it depends on the period set
089         * in the network file (the capperiod attribute).
090         * @return the capacity per network's capperiod timestep
091         *
092         * @see Network#getCapacityPeriod()
093         */
094        public double getCapacity();
095
096        /**
097         * This method returns the capacity as set in the xml defining the network. Be aware
098         * that this capacity is not normalized in time, it depends on the period set
099         * in the network file (the capperiod attribute).
100         * @param time the time at which the capacity is requested. Use {@link Time#UNDEFINED_TIME} to get the default value.
101         * @return the capacity per network's capperiod timestep
102         *
103         * @see Network#getCapacityPeriod()
104         */
105        public double getCapacity(double time);
106
107        public void setFreespeed(double freespeed);
108
109        public void setLength(double length);
110
111        public void setNumberOfLanes(double lanes);
112
113        public void setCapacity(double capacity);
114
115        public void setAllowedModes(Set<String> modes);
116
117        /**
118         * @return an <b>immutable</b> set containing all transport modes that are allowed on that link.
119         */
120        public Set<String> getAllowedModes();
121
122        double getFlowCapacityPerSec();
123
124        double getFlowCapacityPerSec(double time);
125
126}
