001package org.matsim.api.core.v01.population;
002
003import java.util.List;
004
005import org.matsim.api.core.v01.Identifiable;
006
007public interface HasPlansAndId<T extends BasicPlan, I> extends Identifiable<I> {
008
009        /**
010         * Seems that <? extends T> is actually more restrictive than <T>, i.e. we may later switch from 
011         * <? extends T> to <T>, but not the other way around.
012         * <p></p>
013         * Practically, with <? extends T>, you cannot do getPlans().add( new MyPlans() ), where MyPlans is derived from T.
014         */
015        public abstract List<? extends T> getPlans();
016
017        /**
018         * adds the plan to the Person's List of plans and
019         * sets the reference to this person in the Plan instance.
020         */
021        public abstract boolean addPlan(T p);
022        
023        public abstract boolean removePlan(T p);
024
025        public abstract T getSelectedPlan();
026
027        public abstract void setSelectedPlan(T selectedPlan);
028
029        public abstract T createCopyOfSelectedPlanAndMakeSelected() ;
030
031}
