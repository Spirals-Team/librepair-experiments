021package org.matsim.api.core.v01.events;
022
023import java.util.Map;
024
025import org.matsim.api.core.v01.Id;
026import org.matsim.api.core.v01.population.Person;
027import org.matsim.core.api.internal.HasPersonId;
028
029/**
030 * This event specifies that an agent has gained (or paid) some money.
031 * Scoring functions should handle these Events by adding the amount somehow
032 * to the score (this can include some kind of normalization with the
033 * agent's income or something similar).
034 *
035 * @author mrieser
036 */
037public final class PersonMoneyEvent extends Event implements HasPersonId {
038
039        public static final String ATTRIBUTE_AMOUNT = "amount";
040
041        public static final String EVENT_TYPE = "personMoney";
042        public static final String ATTRIBUTE_PERSON = "person";
043
044        private final Id<Person> personId;
045        private final double amount;
046
047        /**
048         * Creates a new event describing that the given <tt>agent</tt> has <em>gained</em>
049         * some money at the specified <tt>time</tt>. Positive values for <tt>amount</tt>
050         * mean the agent has gained money, negative values that the agent has paid money.
051         *
052         * @param time
053         * @param agentId
054         * @param amount
055         */
056        public PersonMoneyEvent(final double time, final Id<Person> agentId, final double amount) {
057                super(time);
058                this.personId = agentId;
059                this.amount = amount;
060        }
061
062        public Id<Person> getPersonId() {
063                return this.personId;
064        }
065        
066        public double getAmount() {
067                return this.amount;
068        }
069
070        @Override
071        public String getEventType() {
072                return EVENT_TYPE;
073        }
074        
075        @Override
076        public Map<String, String> getAttributes() {
077                Map<String, String> attr = super.getAttributes();
078                attr.put(ATTRIBUTE_AMOUNT, Double.toString(this.amount));
079                attr.put(ATTRIBUTE_PERSON, this.personId.toString());
080                return attr;
081        }
082}
