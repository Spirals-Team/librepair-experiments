020package org.matsim.utils.eventsfilecomparison;
021
022import java.util.Map;
023import java.util.Map.Entry;
024import java.util.concurrent.CyclicBarrier;
025
026import org.apache.log4j.Logger;
027import org.matsim.utils.eventsfilecomparison.EventsFileComparator.Result;
028
029/**
030 * This class checks if two events files are semantic equivalent. The order of the events does not matter as long as
031 * they are chronologically sorted.
032 *
033 * @author mrieser
034 * @author laemmel
035 */
036public final class EventsFileComparator {
037        private EventsFileComparator() {} // not meant to be instantiated
038
039        private static final Logger log = Logger.getLogger(EventsFileComparator.class);
040
041        @Deprecated // use Result enum
042        public static final int CODE_FILES_ARE_EQUAL = 0;
043        @Deprecated // use Result enum
044        public static final int CODE_DIFFERENT_NUMBER_OF_TIMESTEPS = -1;
045        @Deprecated // use Result enum
046        public static final int CODE_DIFFERENT_TIMESTEPS = -2;
047        @Deprecated // use Result enum
048        public static final int CODE_MISSING_EVENT = -3;
049        @Deprecated // use Result enum
050        public static final int CODE_WRONG_EVENT_COUNT = -4;
051        
052        public static enum Result { FILES_ARE_EQUAL, DIFFERENT_NUMBER_OF_TIMESTEPS,
053                DIFFERENT_TIMESTEPS, MISSING_EVENT, WRONG_EVENT_COUNT }
054
055        public static void main(String[] args) {
056                if (args.length != 2) {
057                        System.out.println("Error: expected 2 events files as input arguments but found " + args.length);
058                        System.out.println("Syntax: EventsFileComparator eventsFile1 eventsFile2");
059                } else {
060                        String filename1 = args[0];
061                        String filename2 = args[1];
062                        
063                        EventsFileComparator.compareAndReturnInt(filename1, filename2);                 
064                }
065        }
066        
067        /**
068         * Compares two Events files. This method is thread-safe.
069         *
070         * @param filename1
071         * @param filename2
072         * @return <code>0</code> if the events files are equal, or some error code (see constants) if not.
073         */
074        @Deprecated // prefer the variant returning a Result enum.  kai, nov'17
075        public static int compareAndReturnInt(final String filename1, final String filename2) {
076                Result result = compare( filename1, filename2 ) ;
077                switch ( result ) {
078                case DIFFERENT_NUMBER_OF_TIMESTEPS:
079                        return -1 ; 
080                case DIFFERENT_TIMESTEPS:
081                        return -2 ; 
082                case FILES_ARE_EQUAL:
083                        return 0 ;
084                case MISSING_EVENT:
085                        return -3 ; 
086                case WRONG_EVENT_COUNT:
087                        return -4 ;
088                default:
089                        throw new RuntimeException("unknown Result code") ; 
090                }
091        }
092        public static Result compare(final String filename1, final String filename2) {
093                EventsComparator comparator = new EventsComparator();
094                CyclicBarrier doComparison = new CyclicBarrier(2, comparator);
095                Worker w1 = new Worker(filename1, doComparison);
096                Worker w2 = new Worker(filename2, doComparison);
097                comparator.setWorkers(w1, w2);
098                w1.start();
099                w2.start();
100
101                try {
102                        w1.join();
103                } catch (InterruptedException e) {
104                        e.printStackTrace();
105                }
106                try {
107                        w2.join();
108                } catch (InterruptedException e) {
109                        e.printStackTrace();
110                }
111
112                Result retCode = comparator.retCode;
113                if (retCode == Result.FILES_ARE_EQUAL) {
114                        log.info("Event files are semantic equivalent.");
115                } else {
116                        log.warn("Event files differ.");
117                }
118                return retCode;
119        }
120
121        /*package*/ static class EventsComparator implements Runnable {
122
123                private Worker worker1 = null;
124                private Worker worker2 = null;
125                /*package*/ volatile Result retCode = null ;
126
127                /*package*/ void setWorkers(final Worker w1, final Worker w2) {
128                        this.worker1 = w1;
129                        this.worker2 = w2;
130                }
131
132                @Override
133                public void run() {
134                        if (this.worker1.getCurrentTime() != this.worker2.getCurrentTime()) {
135                                log.warn("Differnt time steps in event files!");
136                                setExitCode(Result.DIFFERENT_TIMESTEPS);
137                                return;
138                        }
139
140                        if (this.worker1.isFinished() != this.worker2.isFinished()) {
141                                log.warn("Events files have different number of time steps!");
142                                setExitCode(Result.DIFFERENT_NUMBER_OF_TIMESTEPS);
143                                return;
144                        }
145
146                        Map<String, Counter> map1 = this.worker1.getEventsMap();
147                        Map<String, Counter> map2 = this.worker2.getEventsMap();
148
149                        // check that map2 contains all keys of map1, with the same values
150                        for (Entry<String, Counter> e : map1.entrySet()) {
151                                Counter c = map2.get(e.getKey());
152                                if (c == null) {
153                                        log.warn("Missing event:" ) ;
154                                        log.warn( e.getKey() );
155                                        log.warn("in events file:" + worker2.getEventsFile());
156                                        setExitCode(Result.MISSING_EVENT);
157                                        return;
158                                }
159                                if (c.getCount() != e.getValue().getCount()) {
160                                        log.warn("Wrong event count for: " + e.getKey() + "\n" + e.getValue().getCount() + " times in file:" + worker1.getEventsFile()
161                                                        + "\n" + c.getCount() + " times in file:" + worker2.getEventsFile());
162                                        setExitCode(Result.WRONG_EVENT_COUNT);
163                                        return;
164                                }
165                        }
166
167                        // also check that map1 contains all keys of map2
168                        for (Entry<String, Counter> e : map2.entrySet()) {
169                                Counter c = map1.get(e.getKey());
170                                if (c == null) {
171                                        log.warn("Missing event:" + e.getKey() + "\nin events file:" + worker1.getEventsFile());
172                                        setExitCode(Result.MISSING_EVENT);
173                                        return;
174                                }
175                        }
176
177                        if (this.worker1.isFinished()) {
178                                setExitCode(Result.FILES_ARE_EQUAL);
179                        }
180                }
181
182                private void setExitCode(final Result errCode) {
183                        this.retCode= errCode;
184                        if (errCode != Result.FILES_ARE_EQUAL) {
185                                this.worker1.interrupt();
186                                this.worker2.interrupt();
187                        }
188                }
189        }
190
191}
