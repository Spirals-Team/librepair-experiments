020package org.matsim.utils.gis.matsim2esri.network;

022import java.util.ArrayList;
023import java.util.Collection;
024
025import org.apache.log4j.Logger;
026import org.geotools.feature.simple.SimpleFeatureBuilder;
027import org.geotools.feature.simple.SimpleFeatureTypeBuilder;
028import org.matsim.api.core.v01.network.Network;
029import org.matsim.api.core.v01.network.Node;
030import org.matsim.core.config.ConfigUtils;
031import org.matsim.core.network.NetworkUtils;
032import org.matsim.core.network.io.MatsimNetworkReader;
033import org.matsim.core.scenario.MutableScenario;
034import org.matsim.core.scenario.ScenarioUtils;
035import org.matsim.core.utils.geometry.geotools.MGC;
036import org.matsim.core.utils.gis.ShapeFileWriter;
037import org.opengis.feature.simple.SimpleFeature;
038import org.opengis.referencing.crs.CoordinateReferenceSystem;
039
040import com.vividsolutions.jts.geom.Point;
041
042/**
043 * Simple class to convert the nodess of MATSim network files to ESRI shape files. The nodes could be written points
044 * For a basic example please have a look at the <code>main</code> method.
045 * Can also be called as Nodes2ESRIShape inputNetwork.xml output.shp .
046 *
047 * <p> <strong>Keywords:</strong> converter, network, nodes, esri, shp, matsim </p>
048 *
049 * @author laemmel
050 */
051public class Nodes2ESRIShape {
052
053        private final static Logger log = Logger.getLogger(Nodes2ESRIShape.class);
054
055        private final Network network;
056        private final String filename;
057        private SimpleFeatureBuilder builder;
058
059        
060        public Nodes2ESRIShape(final Network network, final String filename, final String coordinateSystem) {
061                this(network, filename, MGC.getCRS(coordinateSystem));
062        }
063        
064        public Nodes2ESRIShape(Network network, String filename, CoordinateReferenceSystem crs) {
065                this.network = network;
066                this.filename = filename;
067                initFeatureType(crs);
068        }
069
070        public void write() {
071                Collection<SimpleFeature> features = new ArrayList<SimpleFeature>();
072                for (Node node : NetworkUtils.getSortedNodes(this.network)) {
073                        features.add(getFeature(node));
074                }
075                ShapeFileWriter.writeGeometries(features, this.filename);
076
077        }
078
079        private SimpleFeature getFeature(Node node) {
080                Point p = MGC.coord2Point(node.getCoord());
081                try {
082                        return this.builder.buildFeature(null, new Object[]{p,node.getId().toString()});
083                } catch (IllegalArgumentException e) {
084                        throw new RuntimeException(e);
085                }
086        }
087
088        private void initFeatureType(final CoordinateReferenceSystem crs) {
089                SimpleFeatureTypeBuilder typeBuilder = new SimpleFeatureTypeBuilder();
090                typeBuilder.setName("node");
091                typeBuilder.setCRS(crs);
092                typeBuilder.add("location", Point.class);
093                typeBuilder.add("ID", String.class);
094
095                this.builder = new SimpleFeatureBuilder(typeBuilder.buildFeatureType());
096        }
097
098        public static void main(String [] args) {
099                String netfile = null ;
100                String outputFile = null ;
101
102                if ( args.length == 0 ) {
103                        netfile = "./examples/equil/network.xml";
104//              String netfile = "./test/scenarios/berlin/network.xml.gz";
105
106                        outputFile = "./plans/networkNodes.shp";
107                } else if ( args.length == 2 ) {
108                        netfile = args[0] ;
109                        outputFile = args[1] ;
110                } else {
111                        log.error("Arguments cannot be interpreted.  Aborting ...") ;
112                        System.exit(-1) ;
113                }
114
115                MutableScenario scenario = (MutableScenario) ScenarioUtils.createScenario(ConfigUtils.createConfig());
116//              scenario.getConfig().global().setCoordinateSystem("DHDN_GK4");
117
118                log.info("loading network from " + netfile);
119                final Network network = scenario.getNetwork();
120                new MatsimNetworkReader(scenario.getNetwork()).readFile(netfile);
121                log.info("done.");
122
123                new Nodes2ESRIShape(network,outputFile, "DHDN_GK4").write();
124        }
125
126}
