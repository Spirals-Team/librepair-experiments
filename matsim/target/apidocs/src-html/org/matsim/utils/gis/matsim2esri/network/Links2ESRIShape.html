021package org.matsim.utils.gis.matsim2esri.network;
022
023import java.util.ArrayList;
024import java.util.Collection;
025
026import org.apache.log4j.Logger;
027import org.matsim.api.core.v01.Scenario;
028import org.matsim.api.core.v01.network.Link;
029import org.matsim.api.core.v01.network.Network;
030import org.matsim.core.config.ConfigUtils;
031import org.matsim.core.network.NetworkUtils;
032import org.matsim.core.network.io.MatsimNetworkReader;
033import org.matsim.core.scenario.ScenarioUtils;
034import org.matsim.core.utils.geometry.geotools.MGC;
035import org.matsim.core.utils.gis.ShapeFileWriter;
036import org.opengis.feature.simple.SimpleFeature;
037import org.opengis.referencing.crs.CoordinateReferenceSystem;
038
039/**
040 * Simple class to convert the links of MATSim network files to ESRI shape files. The network can be written either
041 * as line strings or as polygons. Furthermore the width of the links could be calculated according to
042 * freespeed, lanes or capacity. For some basic examples please have a look at the <code>main</code> method.
043 * Can also be called as Links2ESRIShape inputNetwork.xml outputAsLines.shp outputAsPolygons.shp .
044 *
045 * <p> <strong>Keywords:</strong> converter, network, links, esri, shp, matsim </p>
046 *
047 * @author laemmel
048 */
049public class Links2ESRIShape {
050
051        private static Logger log = Logger.getLogger(Links2ESRIShape.class);
052
053        private final FeatureGenerator featureGenerator;
054        private final Network network;
055        private final String filename;
056
057
058        public Links2ESRIShape(final Network network, final String filename, final String coordinateSystem) {
059                this(network, filename, new FeatureGeneratorBuilderImpl(network, coordinateSystem));
060        }
061
062        public Links2ESRIShape(final Network network, final String filename, final FeatureGeneratorBuilder builder) {
063                this.network = network;
064                this.filename = filename;
065                this.featureGenerator = builder.createFeatureGenerator();
066
067        }
068
069        public void write() {
070                log.info("creating features...");
071                Collection<SimpleFeature> features = new ArrayList<SimpleFeature>();
072                for (Link link : NetworkUtils.getSortedLinks(this.network)) {
073                        features.add(this.featureGenerator.getFeature(link));
074                }
075                log.info("writing features to shape file... " + this.filename);
076                ShapeFileWriter.writeGeometries(features, this.filename);
077                log.info("done writing shape file.");
078        }
079
080        public static void main(final String [] args) {
081                String netfile = null ;
082                String outputFileLs = null ;
083                String outputFileP = null ;
084                String defaultCRS = "DHDN_GK4";
085                boolean commonWealth = false; //to render Commonwealth networks correctly (e.g. drive on left-hand side of the road)
086                if ( args.length == 0 ) {
087                        netfile = "./examples/equil/network.xml";
088//              String netfile = "./test/scenarios/berlin/network.xml.gz";
089
090                        outputFileLs = "./plans/networkLs.shp";
091                        outputFileP = "./plans/networkP.shp";
092                } else if ( args.length == 3 ) {
093                        netfile = args[0] ;
094                        outputFileLs = args[1] ;
095                        outputFileP  = args[2] ;
096                } else if ( args.length == 4 ) {
097                        netfile = args[0] ;
098                        outputFileLs = args[1] ;
099                        outputFileP  = args[2] ;
100                        defaultCRS   = args[3] ;
101                } else if ( args.length == 5 ) {
102                        netfile = args[0] ;
103                        outputFileLs = args[1] ;
104                        outputFileP  = args[2] ;
105                        defaultCRS   = args[3] ;
106                        commonWealth = Boolean.parseBoolean(args[4]) ;
107                } else {
108                        log.error("Arguments cannot be interpreted.  Aborting ...") ;
109                        System.exit(-1) ;
110                }
111
112                Scenario scenario = ScenarioUtils.createScenario(ConfigUtils.createConfig());
113                scenario.getConfig().global().setCoordinateSystem(defaultCRS);
114
115                log.info("loading network from " + netfile);
116                final Network network = scenario.getNetwork();
117                new MatsimNetworkReader(scenario.getNetwork()).readFile(netfile);
118                log.info("done.");
119
120                FeatureGeneratorBuilderImpl builder = new FeatureGeneratorBuilderImpl(network, defaultCRS);
121                builder.setFeatureGeneratorPrototype(LineStringBasedFeatureGenerator.class);
122                builder.setWidthCoefficient(0.5);
123                builder.setWidthCalculatorPrototype(LanesBasedWidthCalculator.class);
124                new Links2ESRIShape(network,outputFileLs, builder).write();
125
126                CoordinateReferenceSystem crs = MGC.getCRS(defaultCRS);
127                builder.setWidthCoefficient((commonWealth ? -1 : 1) * 0.003);
128                builder.setFeatureGeneratorPrototype(PolygonFeatureGenerator.class);
129                builder.setWidthCalculatorPrototype(CapacityBasedWidthCalculator.class);
130                builder.setCoordinateReferenceSystem(crs);
131                new Links2ESRIShape(network,outputFileP, builder).write();
132
133        }
134
135}
