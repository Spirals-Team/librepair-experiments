059/**
060 * Simple class to convert MATSim plans to ESRI shape files. Activities will be converted into points and
061 * legs will be converted into line strings. Parameters as defined in the population xml file will be added
062 * as attributes to the shape files. There are also some parameters to configure this converter, please
063 * consider the corresponding setters in this class.
064 *
065 * @author laemmel
066 */
067public class SelectedPlans2ESRIShape {
068        private final CoordinateReferenceSystem crs;
069        private final Population population;
070        private double outputSample = 1;
071        private double actBlurFactor = 0;
072        private double legBlurFactor = 0;
073        private final String outputDir;
074        private boolean writeActs = true;
075        private boolean writeLegs = true;
076        private ArrayList<Plan> outputSamplePlans;
077        private SimpleFeatureBuilder actBuilder;
078        private SimpleFeatureBuilder legBuilder;
079        private final GeometryFactory geofac;
080        private final Network network;
081
082        public SelectedPlans2ESRIShape(final Population population, final Network network, final CoordinateReferenceSystem crs, final String outputDir) {
083                this.population = population;
084                this.network = network;
085                this.crs = crs;
086                this.outputDir = outputDir;
087                this.geofac = new GeometryFactory();
088                initFeatureType();
089        }
090
091        public void setOutputSample(final double sample) {
092                this.outputSample = sample;
093        }
094
095        public void setWriteActs(final boolean writeActs) {
096                this.writeActs = writeActs;
097        }
098
099        public void setWriteLegs(final boolean writeLegs) {
100                this.writeLegs = writeLegs;
101        }
102
103        public void setActBlurFactor(final double actBlurFactor) {
104                this.actBlurFactor = actBlurFactor;
105        }
106
107        public void setLegBlurFactor(final double legBlurFactor) {
108                this.legBlurFactor  = legBlurFactor;
109        }
110
111        public void write() {
112                try {
113                        drawOutputSample();
114                        if (this.writeActs) {
115                                writeActs();
116                        } 
117                        if (this.writeLegs) {
118                                writeLegs();
119                        }
120                } catch (IOException e) {
121                        throw new RuntimeException(e);
122                }
123        }
124
125        private void drawOutputSample() {
126                this.outputSamplePlans = new ArrayList<Plan>();
127                for (Person pers : PopulationUtils.getSortedPersons(this.population).values()) {
128                        if (MatsimRandom.getRandom().nextDouble() <= this.outputSample) {
129                                this.outputSamplePlans.add(pers.getSelectedPlan());
130                        }
131                }
132        }
133
134        private void writeActs() throws IOException {
135                String outputFile = this.outputDir + "/acts.shp";
136                ArrayList<SimpleFeature> fts = new ArrayList<SimpleFeature>();
137                for (Plan plan : this.outputSamplePlans) {
138                        String id = plan.getPerson().getId().toString();
139                        for (PlanElement pe : plan.getPlanElements()) {
140                                if (pe instanceof Activity) {
141                                        Activity act = (Activity) pe;
142                                        fts.add(getActFeature(id, act));
143                                }
144                        }
145                }
146
147                ShapeFileWriter.writeGeometries(fts, outputFile);
148        }
149
150        private void writeLegs() throws IOException {
151                String outputFile = this.outputDir + "/legs.shp";
152                ArrayList<SimpleFeature> fts = new ArrayList<SimpleFeature>();
153                for (Plan plan : this.outputSamplePlans) {
154                        String id = plan.getPerson().getId().toString();
155                        for (PlanElement pe : plan.getPlanElements()) {
156                                if (pe instanceof Leg) {
157                                        Leg leg = (Leg) pe;
158                                        if (leg.getRoute() instanceof NetworkRoute) {
159                                                if (RouteUtils.calcDistanceExcludingStartEndLink((NetworkRoute) leg.getRoute(), this.network) > 0) {
160                                                        fts.add(getLegFeature(leg, id));
161                                                }
162                                        } else if (leg.getRoute().getDistance() > 0) {
163                                                fts.add(getLegFeature(leg, id));
164                                        }
165                                }
166                        }
167                }
168                ShapeFileWriter.writeGeometries(fts, outputFile);
169        }
170
171        private SimpleFeature getActFeature(final String id, final Activity act) {
172                String type = act.getType();
173                String linkId = act.getLinkId().toString();
174                Double startTime = act.getStartTime();
175                Double endTime = act.getEndTime();
176                double rx = MatsimRandom.getRandom().nextDouble() * this.actBlurFactor;
177                double ry = MatsimRandom.getRandom().nextDouble() * this.actBlurFactor;
178                Coord cc = this.network.getLinks().get(act.getLinkId()).getCoord();
179                Coord c = new Coord(cc.getX() + rx, cc.getY() + ry);
180                
181                try {
182                        return this.actBuilder.buildFeature(null, new Object [] {MGC.coord2Point(c), id, type, linkId, startTime, endTime});
183                } catch (IllegalArgumentException e) {
184                        e.printStackTrace();
185                }
186
187                return null;
188        }
189
190        private SimpleFeature getLegFeature(final Leg leg, final String id) {
191                if (!(leg.getRoute() instanceof NetworkRoute)) {
192                        return null;
193                }
194                String mode = leg.getMode();
195                Double depTime = leg.getDepartureTime();
196                Double travTime = leg.getTravelTime();
197                Double dist = RouteUtils.calcDistanceExcludingStartEndLink((NetworkRoute) leg.getRoute(), this.network);
198
199                List<Id<Link>> linkIds = ((NetworkRoute) leg.getRoute()).getLinkIds();
200                Coordinate [] coords = new Coordinate[linkIds.size() + 1];
201                for (int i = 0; i < linkIds.size(); i++) {
202                        Link link = this.network.getLinks().get(linkIds.get(i));
203                        Coord c = link.getFromNode().getCoord();
204                        double rx = MatsimRandom.getRandom().nextDouble() * this.legBlurFactor;
205                        double ry = MatsimRandom.getRandom().nextDouble() * this.legBlurFactor;
206                        Coordinate cc = new Coordinate(c.getX()+rx,c.getY()+ry);
207                        coords[i] = cc;
208                }
209
210                Link link = this.network.getLinks().get(linkIds.get(linkIds.size() - 1));
211                Coord c = link.getToNode().getCoord();
212                double rx = MatsimRandom.getRandom().nextDouble() * this.legBlurFactor;
213                double ry = MatsimRandom.getRandom().nextDouble() * this.legBlurFactor;
214                Coordinate cc = new Coordinate(c.getX()+rx,c.getY()+ry);
215                coords[linkIds.size()] = cc;
216
217                LineString ls = this.geofac.createLineString(coords);
218
219                try {
220                        return this.legBuilder.buildFeature(null, new Object [] {ls,id,mode,depTime,travTime,dist});
221                } catch (IllegalArgumentException e) {
222                        e.printStackTrace();
223                }
224
225                return null;
226        }
227
228
229        private void initFeatureType() {
230                SimpleFeatureTypeBuilder actBuilder = new SimpleFeatureTypeBuilder();
231                actBuilder.setName("activity");
232                actBuilder.setCRS(this.crs);
233                actBuilder.add("the_geom", Point.class);
234                actBuilder.add("PERS_ID", String.class);
235                actBuilder.add("TYPE", String.class);
236                actBuilder.add("LINK_ID", String.class);
237                actBuilder.add("START_TIME", Double.class);
238                actBuilder.add("END_TIME", Double.class);
239                
240                SimpleFeatureTypeBuilder legBuilder = new SimpleFeatureTypeBuilder();
241                legBuilder.setName("leg");
242                legBuilder.setCRS(this.crs);
243                legBuilder.add("the_geom", LineString.class);
244                legBuilder.add("PERS_ID", String.class);
245                legBuilder.add("MODE", String.class);
246                legBuilder.add("DEP_TIME", Double.class);
247                legBuilder.add("TRAV_TIME", Double.class);
248                legBuilder.add("DIST", Double.class);
249
250                this.actBuilder = new SimpleFeatureBuilder(actBuilder.buildFeatureType());
251                this.legBuilder = new SimpleFeatureBuilder(legBuilder.buildFeatureType());
252        }
253
254        public static void main(final String [] args) {
255                // FIXME hard-coded file names; does this class really need a main-method?
256                final String populationFilename = "./examples/equil/plans100.xml";
257                final String networkFilename = "./examples/equil/network.xml";
258                //              final String populationFilename = "./test/scenarios/berlin/plans_hwh_1pct.xml.gz";
259                //              final String networkFilename = "./test/scenarios/berlin/network.xml.gz";
260
261                final String outputDir = "./plans/";
262                new File(outputDir).mkdir();
263
264                MutableScenario scenario = (MutableScenario) ScenarioUtils.createScenario(ConfigUtils.createConfig());
265                new MatsimNetworkReader(scenario.getNetwork()).readFile(networkFilename);
266                new PopulationReader(scenario).readFile(populationFilename);
267
268                CoordinateReferenceSystem crs = MGC.getCRS("DHDN_GK4");
269                SelectedPlans2ESRIShape sp = new SelectedPlans2ESRIShape(scenario.getPopulation(), scenario.getNetwork(), crs, outputDir);
270                sp.setOutputSample(0.05);
271                sp.setActBlurFactor(100);
272                sp.setLegBlurFactor(100);
273                sp.setWriteActs(true);
274                sp.setWriteLegs(true);
275
276                sp.write();
277        }
