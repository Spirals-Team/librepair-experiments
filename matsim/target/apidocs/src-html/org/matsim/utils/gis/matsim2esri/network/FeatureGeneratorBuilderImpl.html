021package org.matsim.utils.gis.matsim2esri.network;
022
023import java.lang.reflect.Constructor;
024import java.lang.reflect.InvocationTargetException;
025
026import org.matsim.api.core.v01.network.Network;
027import org.matsim.core.utils.geometry.geotools.MGC;
028import org.opengis.referencing.crs.CoordinateReferenceSystem;
029
030/**
031 * Design thoughts:<ul>
032 * <li> I would argue that basing this on reflection is software design overkill.  In my understanding, reflection is useful to decouple
033 * pieces of software, i.e. be able to insert a class (a constructor) into compiled code (e.g. a jar file) if I do not have access to the source
034 * code of the jar file.  This is, however, not the situation here.  kai, mar'14
035 * <li> Practically, the current design is in my way.  I need additional information inside the width calculator, because I do not want to
036 * calculate width based on a network property, but on something else (speed differences).  So now I will have to abuse "capacity"
037 * for the quantity of interest. kai, mar'14
038 * </ul
039 */
040public class FeatureGeneratorBuilderImpl implements FeatureGeneratorBuilder {
041
042
043        private final Network network;
044        private CoordinateReferenceSystem crs;
045
046        private Constructor<? extends FeatureGenerator> featureGeneratorPrototypeContructor;
047        private static final Class[] FEATURE_GENERATOR_PROTOTYPECONSTRUCTOR =  { WidthCalculator.class, CoordinateReferenceSystem.class};
048
049        private Constructor<? extends WidthCalculator> widthCalculatorPrototypeContructor;
050
051        private double widthCoefficient = 1;
052        private static final Class[] WIDTH_CALCULATOR_PROTOTYPECONSTRUCTOR =  { Network.class, Double.class};
053
054
055        public FeatureGeneratorBuilderImpl(final Network network, final String coordinateSystem) {
056                this.network = network;
057                this.crs = MGC.getCRS(coordinateSystem);
058                try {
059                        this.featureGeneratorPrototypeContructor = PolygonFeatureGenerator.class.getConstructor(FEATURE_GENERATOR_PROTOTYPECONSTRUCTOR);
060                } catch (SecurityException e) {
061                        e.printStackTrace();
062                } catch (NoSuchMethodException e) {
063                        e.printStackTrace();
064                }
065
066                try {
067                        this.widthCalculatorPrototypeContructor = LanesBasedWidthCalculator.class.getConstructor(WIDTH_CALCULATOR_PROTOTYPECONSTRUCTOR);
068                } catch (SecurityException e) {
069                        e.printStackTrace();
070                } catch (NoSuchMethodException e) {
071                        e.printStackTrace();
072                }
073        }
074
075        @Override
076        public FeatureGenerator createFeatureGenerator() {
077
078                WidthCalculator widthCalc = createWidthCalculator();
079                FeatureGenerator ret;
080                Exception ex;
081                try {
082                        ret = this.featureGeneratorPrototypeContructor.newInstance(widthCalc, this.crs);
083                        return ret;
084                } catch (IllegalArgumentException e) {
085                        ex = e;
086                } catch (InstantiationException e) {
087                        ex = e;
088                } catch (IllegalAccessException e) {
089                        ex = e;
090                } catch (InvocationTargetException e) {
091                        ex = e;
092                }
093                throw new RuntimeException(
094                                "Could not instantiate feature generator from prototype! " + this.featureGeneratorPrototypeContructor.getDeclaringClass().getCanonicalName(),
095                                ex);
096        }
097
098        private WidthCalculator createWidthCalculator() {
099                WidthCalculator ret;
100                Exception ex;
101                try {
102                        ret = this.widthCalculatorPrototypeContructor.newInstance(this.network, this.widthCoefficient);
103                        return ret;
104                } catch (IllegalArgumentException e) {
105                        ex = e;
106                } catch (InstantiationException e) {
107                        ex = e;
108                } catch (IllegalAccessException e) {
109                        ex = e;
110                } catch (InvocationTargetException e) {
111                        ex = e;
112                }
113                throw new RuntimeException(
114                                "Could not instantiate width calculator from prototype!",
115                                ex);
116
117        }
118
119
120        public void setWidthCoefficient(final double coef) {
121                this.widthCoefficient  = coef;
122        }
123
124        public void setCoordinateReferenceSystem(final CoordinateReferenceSystem crs) {
125                this.crs = crs;
126        }
127
128        public void setWidthCalculatorPrototype(final Class<? extends WidthCalculator> prototype) {
129
130                try {
131                        Constructor<? extends WidthCalculator> c = prototype.getConstructor(WIDTH_CALCULATOR_PROTOTYPECONSTRUCTOR);
132                        if (null != c) {
133                                this.widthCalculatorPrototypeContructor = c;
134                        }
135                        else {
136                                throw new IllegalArgumentException("Wrong prototype constructor!");
137                        }
138                } catch (SecurityException e) {
139                        e.printStackTrace();
140                } catch (NoSuchMethodException e) {
141                        e.printStackTrace();
142                }
143        }
144
145        public void setFeatureGeneratorPrototype(final Class<? extends FeatureGenerator> prototype) {
146
147                try {
148                        Constructor<? extends FeatureGenerator> c = prototype.getConstructor(FEATURE_GENERATOR_PROTOTYPECONSTRUCTOR);
149                        if (null != c) {
150                                this.featureGeneratorPrototypeContructor = c;
151                        }
152                        else {
153                                throw new IllegalArgumentException("Wrong prototype constructor!");
154                        }
155                } catch (SecurityException e) {
156                        e.printStackTrace();
157                } catch (NoSuchMethodException e) {
158                        e.printStackTrace();
159                }
160        }
161
162}
