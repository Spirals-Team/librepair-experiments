047/**
048 * Calculates a least-cost-path tree using Dijkstra's algorithm  for calculating a shortest-path
049 * tree, given a node as root of the tree.
050 *
051 * 
052 * @author balmermi, mrieser
053 */
054public class LeastCostPathTree {
055
056        // ////////////////////////////////////////////////////////////////////
057        // member variables
058        // ////////////////////////////////////////////////////////////////////
059
060        private Node origin1 = null;
061        private double dTime = Time.UNDEFINED_TIME;
062        
063        private final TravelTime ttFunction;
064        private final TravelDisutility tcFunction;
065        private HashMap<Id<Node>, NodeData> nodeData = null;
066        
067        private final Vehicle VEHICLE = VehicleUtils.getFactory().createVehicle(Id.create("theVehicle", Vehicle.class), VehicleUtils.getDefaultVehicleType());
068        private final Person PERSON = PopulationUtils.getFactory().createPerson(Id.create("thePerson", Person.class));
069
070        // ////////////////////////////////////////////////////////////////////
071        // constructors
072        // ////////////////////////////////////////////////////////////////////
073
074        public LeastCostPathTree(TravelTime tt, TravelDisutility tc) {
075                this.ttFunction = tt;
076                this.tcFunction = tc;
077        }
078
079        public void calculate(final Network network, final Node origin, final double time) {
080                this.origin1 = origin;
081                this.dTime = time;
082                
083                this.nodeData = new HashMap<Id<Node>, NodeData>((int) (network.getNodes().size() * 1.1), 0.95f);
084                NodeData d = new NodeData();
085                d.time = time;
086                d.cost = 0;
087                this.nodeData.put(origin.getId(), d);
088
089                ComparatorCost comparator = new ComparatorCost(this.nodeData);
090                PriorityQueue<Node> pendingNodes = new PriorityQueue<Node>(500, comparator);
091                relaxNode(origin, pendingNodes);
092                while (!pendingNodes.isEmpty()) {
093                        Node n = pendingNodes.poll();
094                        relaxNode(n, pendingNodes);
095                }
096        }
097
098        // ////////////////////////////////////////////////////////////////////
099        // inner classes
100        // ////////////////////////////////////////////////////////////////////
101
102        public static class NodeData {
103                private Id<Node> prevId = null;
104                private double cost = Double.MAX_VALUE;
105                private double time = 0;
106
107        /*package*/ void visit(final Id<Node> comingFromNodeId, final double cost1, final double time1) {
108                        this.prevId = comingFromNodeId;
109                        this.cost = cost1;
110                        this.time = time1;
111                }
112
113                public double getCost() {
114                        return this.cost;
115                }
116
117                public double getTime() {
118                        return this.time;
119                }
120
121                public Id<Node> getPrevNodeId() {
122                        return this.prevId;
123                }
124        }
125
126        /*package*/ static class ComparatorCost implements Comparator<Node> {
127                protected Map<Id<Node>, ? extends NodeData> nodeData;
128
129                ComparatorCost(final Map<Id<Node>, ? extends NodeData> nodeData) {
130                        this.nodeData = nodeData;
131                }
132
133                @Override
134                public int compare(final Node n1, final Node n2) {
135                        double c1 = getCost(n1);
136                        double c2 = getCost(n2);
137                        if (c1 < c2)
138                                return -1;
139                        if (c1 > c2)
140                                return +1;
141                        return n1.getId().compareTo(n2.getId());
142                }
143
144                protected double getCost(final Node node) {
145                        return this.nodeData.get(node.getId()).getCost();
146                }
147        }
148
149        // ////////////////////////////////////////////////////////////////////
150        // get methods
151        // ////////////////////////////////////////////////////////////////////
152
153        public final Map<Id<Node>, NodeData> getTree() {
154                return this.nodeData;
155        }
156
157        /**
158         * @return Returns the root of the calculated tree, or <code>null</code> if no tree was calculated yet.
159         */
160        public final Node getOrigin() {
161                return this.origin1;
162        }
163
164        public final double getDepartureTime() {
165                return this.dTime;
166        }
167
168        // ////////////////////////////////////////////////////////////////////
169        // private methods
170        // ////////////////////////////////////////////////////////////////////
171
172        private void relaxNode(final Node n, PriorityQueue<Node> pendingNodes) {
173                NodeData nData = nodeData.get(n.getId());
174                double currTime = nData.getTime();
175                double currCost = nData.getCost();
176                for (Link l : n.getOutLinks().values()) {
177                        Node nn = l.getToNode();
178                        NodeData nnData = nodeData.get(nn.getId());
179                        if (nnData == null) {
180                                nnData = new NodeData();
181                                this.nodeData.put(nn.getId(), nnData);
182                        }
183                        double visitCost = currCost + tcFunction.getLinkTravelDisutility(l, currTime, PERSON, VEHICLE);
184                        double visitTime = currTime + ttFunction.getLinkTravelTime(l, currTime, PERSON, VEHICLE);
185                        
186                        
187                        if (visitCost < nnData.getCost()) {
188                                pendingNodes.remove(nn);
189                                nnData.visit(n.getId(), visitCost, visitTime);
190                                additionalComputationsHook( l, currTime ) ;
191                                pendingNodes.add(nn);
192                        }
193                }
194        }
195
196        protected void additionalComputationsHook( Link link, double currTime ) {
197                // left empty for inheritance
198        }
199
200        // ////////////////////////////////////////////////////////////////////
201        // main method
202        // ////////////////////////////////////////////////////////////////////
203
204        public static void main(String[] args) {
205                MutableScenario scenario = (MutableScenario) ScenarioUtils.createScenario(ConfigUtils.createConfig());
206                Network network = scenario.getNetwork();
207                new MatsimNetworkReader(scenario.getNetwork()).readFile("../../input/network.xml");
208
209                TravelTimeCalculator ttc = new TravelTimeCalculator(network, 60, 30 * 3600, scenario.getConfig().travelTimeCalculator());
210                LeastCostPathTree st = new LeastCostPathTree(ttc.getLinkTravelTimes(), new RandomizingTimeDistanceTravelDisutilityFactory( TransportMode.car, scenario.getConfig().planCalcScore() ).createTravelDisutility(ttc.getLinkTravelTimes()));
211                Node origin = network.getNodes().get(Id.create(1, Node.class));
212                st.calculate(network, origin, 8*3600);
213                Map<Id<Node>, NodeData> tree = st.getTree();
214                for (Map.Entry<Id<Node>, NodeData> e : tree.entrySet()) {
215                        Id<Node> id = e.getKey();
216                        NodeData d = e.getValue();
217                        if (d.getPrevNodeId() != null) {
218                                System.out.println(id + "\t" + d.getTime() + "\t" + d.getCost() + "\t" + d.getPrevNodeId());
219                        } else {
220                                System.out.println(id + "\t" + d.getTime() + "\t" + d.getCost() + "\t" + "0");
221                        }
222                }
223        }
