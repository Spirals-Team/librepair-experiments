001package org.matsim.utils.objectattributes.attributable;
002
003import java.util.*;
004
005/**
006 * This class is optimized for memory footprint and query time, at the expense of insertion time.
007 *
008 * @author thibautd
009 */
010public final class Attributes {
011        // there are potentially lots of instance of this class, containing typically few mappings each.
012        // to minimize memory footprint, values are stored in arrays, kept as short as possible.
013        // This makes insertion costly, but query can be kept efficient even when the number of mappings
014        // increases, using binary search. This should be fine, as the typical usage is to set once and
015        // access often. Replacing a value is also efficient.
016        //
017        // In addition, as lots of classes implement Attributable, there might be a large number of empty attributes,
018        // which would result in unnecessary memory overhead if each attribute would use new instances of empty arrays
019        // (Which are essentially immutable objects), hence the two "empty" constants (idea from Marcel Rieser, see MATSIM-811)
020        private static final String[] EMPTY_KEYS = new String[0];
021        private static final Object[] EMPTY_VALUES = new Object[0];
022
023        private String[] keys = EMPTY_KEYS;
024        private Object[] values = EMPTY_VALUES;
025
026        @Override
027        public String toString() {
028                StringBuilder stb = new StringBuilder() ;
029                for ( int i=0; i < keys.length; i++ ) {
030                        String subkey = keys[ i ];
031                        stb.append("{ key=").append(subkey);
032                        stb.append("; object=").append( values[ i ].toString());
033                        stb.append( " }" );
034                }
035                return stb.toString() ;
036        }
037
038        public Object putAttribute( final String attribute, final Object value) {
039                final int insertion = Arrays.binarySearch( keys , attribute );
040
041                if ( insertion >= 0 ) {
042                        final Object prev = values[ insertion ];
043                        values[ insertion ] = value;
044                        return prev;
045                }
046
047                final int newIndex = -insertion - 1;
048
049                keys = Arrays.copyOf( keys , keys.length + 1 );
050                values = Arrays.copyOf( values , values.length + 1 );
051
052                for ( int i=keys.length - 2; i >= newIndex; i-- ) {
053                        keys[ i + 1 ] = keys[ i ];
054                        values[ i + 1 ] = values[ i ];
055                }
056
057                keys[newIndex] = attribute;
058                values[newIndex] = value;
059
060                return null;
061        }
062
063        public Object getAttribute( final String attribute) {
064                final int insertion = Arrays.binarySearch( keys , attribute );
065
066                if ( insertion < 0 ) return null;
067
068                return values[ insertion ];
069        }
070
071        public Object removeAttribute( final String attribute ) {
072                final int insertion = Arrays.binarySearch( keys , attribute );
073
074                if ( insertion < 0 ) return null;
075
076                final Object prev = values[ insertion ];
077
078                for ( int i=insertion; i < keys.length - 1; i++ ) {
079                        keys[ i ] = keys[ i + 1 ];
080                        values[ i ] = values[ i + 1 ];
081                }
082
083                keys = Arrays.copyOf( keys , keys.length - 1 );
084                values = Arrays.copyOf( values , values.length - 1 );
085
086                return prev;
087        }
088
089        public void clear() {
090                keys = EMPTY_KEYS;
091                values = EMPTY_VALUES;
092        }
093
094        /**
095         * Returns a view of the mappings stored by this object as an immutable Map. Behavior is undefined if the mappings
096         * are modified after this method was called.
097         *
098         * It is mostly provided to allow iterating through all attributes for writing to file.
099         *
100         * @return a map that represents the mappings stored in this object
101         */
102        public Map<String, Object> getAsMap() {
103                return new AbstractMap<String, Object>() {
104                        @Override
105                        public Set<Entry<String, Object>> entrySet() {
106                                return new AbstractSet<Entry<String, Object>>() {
107                                        @Override
108                                        public Iterator<Entry<String, Object>> iterator() {
109                                                return new EntryIterator();
110                                        }
111
112                                        @Override
113                                        public int size() {
114                                                return keys.length;
115                                        }
116                                };
117                        }
118                };
119        }
120
121        public int size() {
122                return keys.length;
123        }
124
125        public boolean isEmpty() {
126                return size() == 0;
127        }
128
129        private class EntryIterator implements Iterator<Map.Entry<String, Object>> {
130                private int index = 0;
131
132                @Override
133                public boolean hasNext() {
134                        return index < keys.length;
135                }
136
137                @Override
138                public Map.Entry<String, Object> next() {
139                        Map.Entry<String, Object> entry = new AbstractMap.SimpleEntry<>(keys[index], values[index]) ;
140                        index++;
141                        return entry;
142                }
143        }
144}
