001package org.matsim.utils.objectattributes.attributable;
002
003import java.util.*;
004
005/**
006 * This class is optimized for memory footprint and query time, at the expense of insertion time.
007 *
008 * @author thibautd
009 */
010public final class Attributes {
011        // there are potentially lots of instance of this class, containing typically few mappings each.
012        // to minimize memory footprint, values are stored in arrays, kept as short as possible.
013        // This makes insertion costly, but query can be kept efficient even when the number of mappings
014        // increases, using binary search. This should be fine, as the typical usage is to set once and
015        // access often. Replacing a value is also efficient.
016        private String[] keys = new String[0];
017        private Object[] values = new Object[0];
018
019        @Override
020        public String toString() {
021                StringBuilder stb = new StringBuilder() ;
022                for ( int i=0; i < keys.length; i++ ) {
023                        String subkey = keys[ i ];
024                        stb.append("{ key=").append(subkey);
025                        stb.append("; object=").append( values[ i ].toString());
026                        stb.append( " }" );
027                }
028                return stb.toString() ;
029        }
030
031        public Object putAttribute( final String attribute, final Object value) {
032                final int insertion = Arrays.binarySearch( keys , attribute );
033
034                if ( insertion >= 0 ) {
035                        final Object prev = values[ insertion ];
036                        values[ insertion ] = value;
037                        return prev;
038                }
039
040                final int newIndex = -insertion - 1;
041
042                keys = Arrays.copyOf( keys , keys.length + 1 );
043                values = Arrays.copyOf( values , values.length + 1 );
044
045                for ( int i=keys.length - 2; i >= newIndex; i-- ) {
046                        keys[ i + 1 ] = keys[ i ];
047                        values[ i + 1 ] = values[ i ];
048                }
049
050                keys[newIndex] = attribute;
051                values[newIndex] = value;
052
053                return null;
054        }
055
056        public Object getAttribute( final String attribute) {
057                final int insertion = Arrays.binarySearch( keys , attribute );
058
059                if ( insertion < 0 ) return null;
060
061                return values[ insertion ];
062        }
063
064        public Object removeAttribute( final String attribute ) {
065                final int insertion = Arrays.binarySearch( keys , attribute );
066
067                if ( insertion < 0 ) return null;
068
069                final Object prev = values[ insertion ];
070
071                for ( int i=insertion; i < keys.length - 1; i++ ) {
072                        keys[ i ] = keys[ i + 1 ];
073                        values[ i ] = values[ i + 1 ];
074                }
075
076                keys = Arrays.copyOf( keys , keys.length - 1 );
077                values = Arrays.copyOf( values , values.length - 1 );
078
079                return prev;
080        }
081
082        public void clear() {
083                keys = new String[ 0 ];
084                values = new Object[ 0 ];
085        }
086
087        /**
088         * Returns a view of the mappings stored by this object as an immutable Map. Behavior is undefined if the mappings
089         * are modified after this method was called.
090         *
091         * It is mostly provided to allow iterating through all attributes for writing to file.
092         *
093         * @return a map that represents the mappings stored in this object
094         */
095        public Map<String, Object> getAsMap() {
096                return new AbstractMap<String, Object>() {
097                        @Override
098                        public Set<Entry<String, Object>> entrySet() {
099                                return new AbstractSet<Entry<String, Object>>() {
100                                        @Override
101                                        public Iterator<Entry<String, Object>> iterator() {
102                                                return new EntryIterator();
103                                        }
104
105                                        @Override
106                                        public int size() {
107                                                return keys.length;
108                                        }
109                                };
110                        }
111                };
112        }
113
114        public int size() {
115                return keys.length;
116        }
117
118        public boolean isEmpty() {
119                return size() == 0;
120        }
121
122        private class EntryIterator implements Iterator<Map.Entry<String, Object>> {
123                private int index = 0;
124
125                @Override
126                public boolean hasNext() {
127                        return index < keys.length;
128                }
129
130                @Override
131                public Map.Entry<String, Object> next() {
132                        Map.Entry<String, Object> entry = new AbstractMap.SimpleEntry<>(keys[index], values[index]) ;
133                        index++;
134                        return entry;
135                }
136        }
137}
