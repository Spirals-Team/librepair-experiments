019/**
020 * This class is meant to be used as a delegate by any reader that reads an {@link Attributable} object
021 * @author thibautd
022 */
023public class AttributesXmlReaderDelegate {
024        private final static Logger log = Logger.getLogger(AttributesXmlReaderDelegate.class);
025        private final ObjectAttributesConverter converter = new ObjectAttributesConverter();
026
027        private Attributes currentAttributes = null;
028        private String currentAttribute = null;
029        private String currentAttributeClass = null;
030
031        /*package*/ final static String TAG_ATTRIBUTES = "attributes";
032        /*package*/ final static String TAG_ATTRIBUTE = "attribute";
033        /*package*/ final static String ATTR_ATTRIBUTENAME = "name";
034        /*package*/ final static String ATTR_ATTRIBUTECLASS = "class";
035
036        public void startTag(String name,
037                                                 org.xml.sax.Attributes atts,
038                                                 Stack<String> context,
039                                                 Attributes currentAttributes ) {
040                if (TAG_ATTRIBUTE.equals(name)) {
041                        this.currentAttribute = atts.getValue(ATTR_ATTRIBUTENAME);
042                        this.currentAttributeClass = atts.getValue(ATTR_ATTRIBUTECLASS);
043                } else if (TAG_ATTRIBUTES.equals(name)) {
044                        this.currentAttributes = currentAttributes;
045                }
046        }
047
048        public void endTag(String name, String content, Stack<String> context) {
049                if (TAG_ATTRIBUTE.equals(name)) {
050                        Object o = converter.convert(this.currentAttributeClass, content);
051                        if (o == null) return;
052                        this.currentAttributes.putAttribute( this.currentAttribute, o);
053                }
054        }
055
056        public Attributes getCurrentAttributes() {
057                return currentAttributes;
058        }
059
060        /**
061         * Sets the converter for reading attributes of the specified class.
062         *
063         * @param clazz
064         * @param converter
065         * @return the previously registered converter for this class, or <code>null</code> if none was set before.
066         */
067        public AttributeConverter<?> putAttributeConverter(final Class<?> clazz, final AttributeConverter<?> converter) {
068                return this.converter.putAttributeConverter(clazz, converter);
069        }
070
071        public void putAttributeConverters( final Map<Class<?>, AttributeConverter<?>> converters ) {
072                for ( Map.Entry<Class<?>, AttributeConverter<?>> e : converters.entrySet() ) {
073                        putAttributeConverter( e.getKey() , e.getValue() );
074                }
075        }
076
077        /**
078         * Removes the converter for reading attributes of the specified class.
079         *
080         * @param clazz
081         * @return the previously registered converter for this class, of <code>null</code> if none was set.
082         */
083        public AttributeConverter<?> removeAttributeConverter(final Class<?> clazz) {
084                return this.converter.removeAttributeConverter(clazz);
085        }
086}
