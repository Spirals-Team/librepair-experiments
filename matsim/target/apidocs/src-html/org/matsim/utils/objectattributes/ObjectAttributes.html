020package org.matsim.utils.objectattributes;
021
022import java.util.IdentityHashMap;
023import java.util.LinkedHashMap;
024import java.util.Map;
025import java.util.Map.Entry;
026
027import org.matsim.core.api.internal.MatsimExtensionPoint;
028
029/**
030 * A simple helper class to store arbitrary attributes (identified by Strings) for
031 * arbitrary objects (identified by String-Ids). Note that this implementation uses
032 * large amounts of memory for storing many attributes for many objects, it is not
033 * heavily optimized.
034 * <p></p>
035 * <em>This class is not thread-safe.</em>
036 * <p></p>
037 * More information can be found in the package's Javadoc.
038 * <p></p>
039 * Example(s):<ul>
040 * <li> {@link RunObjectAttributesExample}
041 * </ul>
042 * @author mrieser
043 */
044public class ObjectAttributes implements MatsimExtensionPoint {
045
046        /*package*/ Map<String, Map<String, Object>> attributes = new LinkedHashMap<String, Map<String, Object>>(1000);
047        
048        @Override
049        public String toString() {
050                StringBuilder stb = new StringBuilder() ;
051                for ( Entry<String, Map<String,Object>> entry : attributes.entrySet() ) {
052                        String key = entry.getKey() ;
053                        stb.append("key=").append(key);
054                        Map<String,Object> map = entry.getValue() ;
055                        for ( Entry<String,Object> ee : map.entrySet() ) {
056                                String subkey = ee.getKey();
057                                stb.append("; subkey=").append(subkey);
058                                stb.append("; object=").append(ee.getValue().toString());
059                        }
060                        stb.append("\n") ;
061                }
062                return stb.toString() ;
063        }
064
065        public Object putAttribute(final String objectId, final String attribute, final Object value) {
066                Map<String, Object> attMap = this.attributes.get(objectId);
067                if (attMap == null) {
068                        attMap = new IdentityHashMap<String, Object>(5);
069                        this.attributes.put(objectId, attMap);
070                }
071                return attMap.put(attribute.intern(), value);
072        }
073
074        public Object getAttribute(final String objectId, final String attribute) {
075                Map<String, Object> attMap = this.attributes.get(objectId);
076                if (attMap == null) {
077                        return null;
078                }
079                return attMap.get(attribute.intern());
080        }
081
082        public Object removeAttribute(final String objectId, final String attribute) {
083                Map<String, Object> attMap = this.attributes.get(objectId);
084                if (attMap == null) {
085                        return null;
086                }
087                return attMap.remove(attribute.intern());
088        }
089
090        public void removeAllAttributes(final String objectId) {
091                this.attributes.remove(objectId);
092        }
093
094        /**
095         * Deletes all attributes of all objects, and all objects-ids.
096         */
097        public void clear() {
098                this.attributes.clear();
099        }
100
101}
