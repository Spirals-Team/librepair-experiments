001package org.matsim.utils.objectattributes.attributeconverters;
002
003import org.apache.log4j.Logger;
004import org.matsim.api.core.v01.Coord;
005import org.matsim.utils.objectattributes.AttributeConverter;
006
007public class CoordConverter implements AttributeConverter<Coord> {
008        private final Logger log = Logger.getLogger(CoordConverter.class);
009
010        @Override
011        public Coord convert(String value) {
012                String s = value.replace("(", "");
013                s = s.replace(")", "");
014                String[] sa = s.split(";");
015                return new Coord(Double.parseDouble(sa[0]), Double.parseDouble(sa[1]));
016        }
017
018        @Override
019        public String convertToString(Object o) {
020                if(!(o instanceof Coord)){
021                        log.error("Object is not of type Coord: " + o.getClass().toString());
022                        return null;
023                }
024                Coord c = (Coord)o;
025                
026                return String.format("(%.2f;%.2f)", c.getX(), c.getY());
027        }
028
029}
