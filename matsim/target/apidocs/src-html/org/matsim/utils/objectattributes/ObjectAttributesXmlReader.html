020package org.matsim.utils.objectattributes;
021
022import static org.matsim.utils.objectattributes.ObjectAttributesXmlWriter.ATTR_ATTRIBUTECLASS;
023import static org.matsim.utils.objectattributes.ObjectAttributesXmlWriter.ATTR_ATTRIBUTENAME;
024import static org.matsim.utils.objectattributes.ObjectAttributesXmlWriter.ATTR_OBJECTID;
025import static org.matsim.utils.objectattributes.ObjectAttributesXmlWriter.TAG_ATTRIBUTE;
026import static org.matsim.utils.objectattributes.ObjectAttributesXmlWriter.TAG_OBJECT;
027
028import java.util.HashMap;
029import java.util.HashSet;
030import java.util.Map;
031import java.util.Set;
032import java.util.Stack;
033
034import com.google.inject.Inject;
035import org.apache.log4j.Logger;
036import org.matsim.core.utils.io.MatsimXmlParser;
037import org.matsim.utils.objectattributes.attributeconverters.BooleanConverter;
038import org.matsim.utils.objectattributes.attributeconverters.DoubleConverter;
039import org.matsim.utils.objectattributes.attributeconverters.FloatConverter;
040import org.matsim.utils.objectattributes.attributeconverters.IntegerConverter;
041import org.matsim.utils.objectattributes.attributeconverters.LongConverter;
042import org.matsim.utils.objectattributes.attributeconverters.StringConverter;
043import org.xml.sax.Attributes;
044import org.xml.sax.SAXException;

046/**
047 * Reads object attributes from a file. The reader supports attributes of type {@link String},
048 * {@link Integer}, {@link Double} and {@link Boolean} out of the box. Other types must be manually
049 * supported by implementing {@link AttributeConverter} and adding it to the reader with
050 * {@link #putAttributeConverter(Class, AttributeConverter)}.
051 *
052 * @author mrieser
053 */
054public class ObjectAttributesXmlReader extends MatsimXmlParser {
055        private final static Logger log = Logger.getLogger(ObjectAttributesXmlReader.class);
056        private final ObjectAttributesConverter converter = new ObjectAttributesConverter();
057        private final ObjectAttributes attributes;
058        private boolean readCharacters = false;
059        private String currentObject = null;
060        private String currentAttribute = null;
061        private String currentAttributeClass = null;
062        private long count = 0;
063
064
065        public ObjectAttributesXmlReader(final ObjectAttributes attributes) {
066                this.attributes = attributes;
067                super.setValidating(false);
068        }
069
070        @Override
071        public void startTag(String name, Attributes atts, Stack<String> context) {
072                if (TAG_ATTRIBUTE.equals(name)) {
073                        this.currentAttribute = atts.getValue(ATTR_ATTRIBUTENAME);
074                        this.currentAttributeClass = atts.getValue(ATTR_ATTRIBUTECLASS);
075                        this.readCharacters = true;
076                } else if (TAG_OBJECT.equals(name)) {
077                        this.currentObject = atts.getValue(ATTR_OBJECTID);
078                }
079        }
080
081        @Override
082        public void endTag(String name, String content, Stack<String> context) {
083                if (TAG_ATTRIBUTE.equals(name)) {
084                        this.readCharacters = false;
085
086                        Object o = converter.convert(this.currentAttributeClass, content);
087                        this.attributes.putAttribute(this.currentObject, this.currentAttribute, o);
088                } else if (TAG_OBJECT.equals(name)) {
089                        if (this.count % 100000 == 0) {
090                                log.info("reading object #" + this.count);
091                        }
092                        this.count++;
093                        this.currentObject = null;
094                }
095        }
096
097        @Override
098        public void characters(char[] ch, int start, int length) throws SAXException {
099                if (this.readCharacters) {
100                        super.characters(ch, start, length);
101                }
102                // ignore characters to prevent OutOfMemoryExceptions
103                /* non-validating files contain empty tags with attributes,
104                 * but without the dtd or schema, all whitespace between tags is handled
105                 * by characters and added up by super.characters, consuming huge
106                 * amount of memory when large files are read in.
107                 */
108        }
109
110        /**
111         * Sets the converter for reading attributes of the specified class.
112         *
113         * @param clazz
114         * @param converter
115         * @return the previously registered converter for this class, or <code>null</code> if none was set before.
116         */
117        public AttributeConverter<?> putAttributeConverter(final Class<?> clazz, final AttributeConverter<?> converter) {
118                return this.converter.putAttributeConverter(clazz, converter);
119        }
120
121        @Inject
122        public void putAttributeConverters( final Map<Class<?>, AttributeConverter<?>> converters ) {
123                this.converter.putAttributeConverters(converters);
124        }
125
126        /**
127         * Removes the converter for reading attributes of the specified class.
128         *
129         * @param clazz
130         * @return the previously registered converter for this class, of <code>null</code> if none was set.
131         */
132        public AttributeConverter<?> removeAttributeConverter(final Class<?> clazz) {
133                return this.converter.removeAttributeConverter(clazz);
134        }
135
136}
