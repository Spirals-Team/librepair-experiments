025/**
026 * @author thibautd
027 */
028public class AttributesXmlWriterDelegate {
029        private final ObjectAttributesConverter converter = new ObjectAttributesConverter();
030
031        public final void writeAttributes(final String indentation, final BufferedWriter writer, final Attributes attributes) {
032                if (attributes.size() == 0) {
033                        return;
034                }
035
036                try {
037                        writer.write(indentation);
038                        writer.write("<attributes>");
039                        writer.newLine();
040
041                        // write attributes
042                        for (Map.Entry<String, Object> objAttribute : attributes.getAsMap().entrySet()) {
043                                Class<?> clazz = objAttribute.getValue().getClass();
044                                String converted = converter.convertToString(objAttribute.getValue());
045                                if (converted != null) {
046                                        writer.write(indentation + "\t");
047                                        writer.write("<attribute name=\"" + objAttribute.getKey() + "\" ");
048                                        writer.write("class=\"" + clazz.getCanonicalName() + "\" >");
049                                        writer.write(converted);
050                                        writer.write("</attribute>");
051                                        writer.newLine();
052                                }
053                        }
054
055                        writer.write(indentation);
056                        writer.write("</attributes>");
057                        writer.newLine();
058                } catch (IOException e) {
059                        throw new UncheckedIOException(e);
060                }
061        }
062
063        public void putAttributeConverters(Map<Class<?>, AttributeConverter<?>> converters) {
064                this.converter.putAttributeConverters(converters);
065        }
066}
