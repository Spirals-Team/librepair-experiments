001package org.matsim.utils.objectattributes.attributable;
002
003import org.matsim.utils.objectattributes.AttributeConverter;
004
005import java.util.Map;
006
007/**
008 * @author thibautd
009 */
010public class AttributesUtils {
011        /**
012         * Adds the mappings from "from" to "to". Nothing is done to copy the Object themselves,
013         * which should be fine for 99.9% of the usecases of Attributes (value objects)
014         */
015        public static void copyTo( Attributes from , Attributes to ) {
016                for ( Map.Entry<String, Object> entry : from.getAsMap().entrySet() ) {
017                        to.putAttribute( entry.getKey() , entry.getValue() );
018                }
019        }
020
021        /**
022         * Adds the mappings from "from" to "to". Nothing is done to copy the Object themselves,
023         * which should be fine for 99.9% of the usecases of Attributes (value objects)
024         */
025        public static <T extends Attributable> void copyAttributesFromTo( T from , T to ) {
026                copyTo( from.getAttributes() , to.getAttributes() );
027        }
028
029        /**
030         * @deprecated use {@link Attributes#isEmpty()} instead
031         * @param attributes collection of attributes
032         * @return <code>true</code> if the attributes collection does not contain any attribute
033         */
034        @Deprecated
035        public static boolean isEmpty(Attributes attributes) {
036                return attributes.size() == 0;
037        }
038}
