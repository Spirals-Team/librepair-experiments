041/**
042 * Writes object attributes to a file.
043 *
044 * @author mrieser
045 */
046public class ObjectAttributesXmlWriter extends MatsimXmlWriter {
047
048        private final static Logger log = Logger.getLogger(ObjectAttributesXmlWriter.class);
049
050        /*package*/ final static String TAG_OBJECT_ATTRIBUTES = "objectAttributes";
051        /*package*/ final static String TAG_OBJECT = "object";
052        /*package*/ final static String TAG_ATTRIBUTE = "attribute";
053        /*package*/ final static String ATTR_OBJECTID = "id";
054        /*package*/ final static String ATTR_ATTRIBUTENAME = "name";
055        /*package*/ final static String ATTR_ATTRIBUTECLASS = "class";
056
057
058
059        private final ObjectAttributes attributes;
060        private final ObjectAttributesConverter converter = new ObjectAttributesConverter();
061
062        public ObjectAttributesXmlWriter(final ObjectAttributes attributes) {
063                this.attributes = attributes;
064        }
065
066        public void writeFile(final String filename) throws UncheckedIOException {
067                openFile(filename);
068                writeXmlHead();
069                writeDoctype(TAG_OBJECT_ATTRIBUTES, "http://matsim.org/files/dtd/objectattributes_v1.dtd");
070                writeStartTag(TAG_OBJECT_ATTRIBUTES, null);
071                List<Tuple<String, String>> xmlAttributes = new LinkedList<Tuple<String, String>>();
072                for (Map.Entry<String, Map<String, Object>> entry : this.attributes.attributes.entrySet()) {
073                        xmlAttributes.add(super.createTuple(ATTR_OBJECTID, entry.getKey()));
074                        writeStartTag(TAG_OBJECT, xmlAttributes);
075                        xmlAttributes.clear();
076                        // sort attributes by name
077                        Map<String, Object> objAttributes = new TreeMap<String, Object>();
078                        for (Map.Entry<String, Object> objAttribute : entry.getValue().entrySet()) {
079                                objAttributes.put(objAttribute.getKey(), objAttribute.getValue());
080                        }
081                        // write attributes
082                        for (Map.Entry<String, Object> objAttribute : objAttributes.entrySet()) {
083                                Class<?> clazz = objAttribute.getValue().getClass();
084                                String value = converter.convertToString(objAttribute.getValue());
085                                if (value != null) {
086                                        xmlAttributes.add(super.createTuple(ATTR_ATTRIBUTENAME, objAttribute.getKey()));
087                                        xmlAttributes.add(super.createTuple(ATTR_ATTRIBUTECLASS, clazz.getCanonicalName()));
088                                        writeStartTag(TAG_ATTRIBUTE, xmlAttributes);
089                                        xmlAttributes.clear();
090                                        writeContent(value, false);
091                                        writeEndTag(TAG_ATTRIBUTE);
092                                }
093                        }
094                        writeEndTag(TAG_OBJECT);
095                }
096                writeEndTag(TAG_OBJECT_ATTRIBUTES);
097                close();
098        }
099
100        /**
101         * Sets the converter for reading attributes of the specified class.
102         *
103         * @param clazz
104         * @param converter
105         * @return the previously registered converter for this class, or <code>null</code> if none was set before.
106         */
107        public AttributeConverter putAttributeConverter(final Class<?> clazz, final AttributeConverter converter) {
108                return this.converter.putAttributeConverter(clazz, converter);
109        }
110
111        @Inject
112        public void putAttributeConverters( final Map<Class<?>, AttributeConverter<?>> converters ) {
113                this.converter.putAttributeConverters(converters);
114        }
115
116        /**
117         * Removes the converter for reading attributes of the specified class.
118         *
119         * @param clazz
120         * @return the previously registered converter for this class, of <code>null</code> if none was set.
121         */
122        public AttributeConverter removeAttributeConverter(final Class<?> clazz) {
123                return this.converter.removeAttributeConverter(clazz);
124        }
125
126}
