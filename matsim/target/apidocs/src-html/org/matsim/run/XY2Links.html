021package org.matsim.run;
022
023import java.util.Iterator;
024
025import org.matsim.core.config.Config;
026import org.matsim.core.config.ConfigUtils;
027import org.matsim.core.gbl.MatsimRandom;
028import org.matsim.core.network.io.MatsimNetworkReader;
029import org.matsim.core.population.PopulationUtils;
030import org.matsim.core.population.io.StreamingPopulationReader;
031import org.matsim.core.population.io.StreamingPopulationWriter;
032import org.matsim.core.scenario.MutableScenario;
033import org.matsim.core.scenario.ScenarioUtils;
034import org.matsim.core.utils.misc.ArgumentParser;
035
036/**
037 * Assigns each activity in each plan of each person in the population a link
038 * where the activity takes place based on the coordinates given for the activity.
039 * This tool is used for mapping a new demand/population to a network for the first time.
040 *
041 * @author mrieser
042 */
043public class XY2Links {
044
045        private Config config;
046        private String configfile = null;
047        private String plansfile = null;
048
049        /**
050         * Parses all arguments and sets the corresponding members.
051         *
052         * @param args
053         */
054        private void parseArguments(final String[] args) {
055                if (args.length == 0) {
056                        System.out.println("Too few arguments.");
057                        printUsage();
058                        System.exit(1);
059                }
060                Iterator<String> argIter = new ArgumentParser(args).iterator();
061                String arg = argIter.next();
062                if (arg.equals("-h") || arg.equals("--help")) {
063                        printUsage();
064                        System.exit(0);
065                } else {
066                        this.configfile = arg;
067                        this.plansfile = argIter.next();
068                        if (argIter.hasNext()) {
069                                System.out.println("Too many arguments.");
070                                printUsage();
071                                System.exit(1);
072                        }
073                }
074        }
075
076        private void printUsage() {
077                System.out.println();
078                System.out.println("XY2Links");
079                System.out.println("Reads a plans-file and assignes each activity in each plan of each person");
080                System.out.println("a link based on the coordinates given in the activity. The modified plans/");
081                System.out.println("persons are then written out to file again.");
082                System.out.println();
083                System.out.println("usage: XY2Links [OPTIONS] configfile plansfile");
084                System.out.println("       The following parameters must be given in the config-file:");
085                System.out.println("       - network.inputNetworkFile");
086                System.out.println("       - plans.inputPlansFile");
087                System.out.println();
088                System.out.println("Options:");
089                System.out.println("-h, --help:     Displays this message.");
090                System.out.println();
091                System.out.println("----------------");
092                System.out.println("2008, matsim.org");
093                System.out.println();
094        }
095
096        /** Starts the assignment of links to activities.
097         *
098         * @param args command-line arguments
099         */
100        public void run(final String[] args) {
101                parseArguments(args);
102                this.config = ConfigUtils.loadConfig(this.configfile);
103                MatsimRandom.reset(this.config.global().getRandomSeed());
104                MutableScenario scenario = ScenarioUtils.createMutableScenario(this.config);
105                new MatsimNetworkReader(scenario.getNetwork()).readFile(this.config.network().getInputFile());
106                this.config = scenario.getConfig();
107
108                StreamingPopulationReader reader = new StreamingPopulationReader( scenario ) ;
109                final StreamingPopulationWriter plansWriter = new StreamingPopulationWriter();
110                plansWriter.startStreaming(this.plansfile);
111                reader.addAlgorithm(new org.matsim.core.population.algorithms.XY2Links(scenario));
112                reader.addAlgorithm(plansWriter);
113                reader.readFile(this.config.plans().getInputFile());
114                PopulationUtils.printPlansCount(reader) ;
115                plansWriter.closeStreaming();
116
117                System.out.println("done.");
118        }
119
120        /**
121         * Main method to start the assignment of links to activities.
122         *
123         * @param args Array of arguments, usually passed on the command line.
124         */
125        public static void main(final String[] args) {
126                new XY2Links().run(args);
127        }
128
129}
