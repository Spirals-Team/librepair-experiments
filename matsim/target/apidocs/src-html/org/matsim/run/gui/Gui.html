374        public void startMATSim() {
375                progressBar.setVisible(true);
376                progressBar.setEnabled(true);
377                this.btnStartMatsim.setEnabled(false);
378
379                new Thread(new Runnable() {
380                        @Override
381                        public void run() {
382                                String classpath = System.getProperty("java.class.path");
383                                String[] cpParts = classpath.split(File.pathSeparator);
384                                StringBuilder absoluteClasspath = new StringBuilder();
385                                for (String cpPart : cpParts) {
386                                        if (absoluteClasspath.length() > 0) {
387                                                absoluteClasspath.append(File.pathSeparatorChar);
388                                        }
389                                        absoluteClasspath.append(new File(cpPart).getAbsolutePath());
390                                }
391                                String[] cmdArgs = new String[] {
392                                                txtJvmlocation.getText(),
393                                                "-cp",
394                                                absoluteClasspath.toString(),
395                                                "-Xmx" + txtRam.getText() + "m",
396                                                Gui.this.mainClass,
397                                                txtConfigfilename.getText()
398                                };
399                                Gui.this.textStdOut.setText("");
400                                Gui.this.textErrOut.setText("");
401                                Gui.this.exeRunner = ExeRunner.run(cmdArgs, Gui.this.textStdOut, Gui.this.textErrOut, new File(txtConfigfilename.getText()).getParent());
402                                Gui.this.btnStartMatsim.setText("Stop MATSim");
403                                Gui.this.btnStartMatsim.setEnabled(true);
404                                int exitcode = exeRunner.waitForFinish();
405                                Gui.this.exeRunner = null;
406
407                                SwingUtilities.invokeLater(new Runnable() {
408                                        @Override
409                                        public void run() {
410                                                progressBar.setVisible(false);
411                                                btnStartMatsim.setText("Start MATSim");
412                                                btnStartMatsim.setEnabled(true);
413                                        }
414                                });
415
416                                if (exitcode != 0) {
417                                        Gui.this.textStdOut.append("\n");
418                                        Gui.this.textStdOut.append("The simulation did not run properly. Error/Exit code: " + exitcode);
419                                        Gui.this.textStdOut.setCaretPosition(Gui.this.textStdOut.getDocument().getLength());
420                                        Gui.this.textErrOut.append("\n");
421                                        Gui.this.textErrOut.append("The simulation did not run properly. Error/Exit code: " + exitcode);
422                                        Gui.this.textErrOut.setCaretPosition(Gui.this.textStdOut.getDocument().getLength());
423                                        throw new RuntimeException("There was a problem running MATSim. exit code: " + exitcode);
424                                }
425                        }
426                }).start();
427                
428        }

430        public void loadConfigFile(final File configFile) {
431                this.configFile = configFile;
432                String configFilename = configFile.getAbsolutePath();
433                
434                Config config = ConfigUtils.createConfig();
435                try {
436                        ConfigUtils.loadConfig(config, configFilename);
437                } catch (Exception e) {
438                        Gui.this.textStdOut.setText("");
439                        Gui.this.textStdOut.append("The configuration file could not be loaded. Error message:\n");
440                        Gui.this.textStdOut.append(e.getMessage());
441                        Gui.this.textErrOut.setText("");
442                        Gui.this.textErrOut.append("The configuration file could not be loaded. Error message:\n");
443                        Gui.this.textErrOut.append(e.getMessage());
444                        return;
445                }
446                txtConfigfilename.setText(configFilename);
447                
448                File par = configFile.getParentFile();
449                File outputDir = new File(par, config.controler().getOutputDirectory());
450                try {
451                        txtOutput.setText(outputDir.getCanonicalPath());
452                } catch (IOException e1) {
453                        txtOutput.setText(outputDir.getAbsolutePath());
454                }
455                
456                btnStartMatsim.setEnabled(true);
457                btnEdit.setEnabled(true);
458        }

460        public void stopMATSim() {
461                ExeRunner runner = this.exeRunner;
462                if (runner != null) {
463                        runner.killProcess();
464                        SwingUtilities.invokeLater(new Runnable() {
465                                @Override
466                                public void run() {
467                                        progressBar.setVisible(false);
468                                        btnStartMatsim.setText("Start MATSim");
469                                        btnStartMatsim.setEnabled(true);
470                                        
471                                        Gui.this.textStdOut.append("\n");
472                                        Gui.this.textStdOut.append("The simulation was stopped forcefully.");
473                                        Gui.this.textStdOut.setCaretPosition(Gui.this.textStdOut.getDocument().getLength());
474                                        Gui.this.textErrOut.append("\n");
475                                        Gui.this.textErrOut.append("The simulation was stopped forcefully.");
476                                        Gui.this.textErrOut.setCaretPosition(Gui.this.textStdOut.getDocument().getLength());
477                                }
478                        });
479                }
480        }
