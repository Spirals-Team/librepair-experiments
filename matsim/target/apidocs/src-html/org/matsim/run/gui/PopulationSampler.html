132        private void createSample() {
133                final String srcFilename = PopulationSampler.this.txtPath.getText();
134                File srcFile = new File(srcFilename);
135                if (!srcFile.exists()) {
136                        JOptionPane.showMessageDialog(null,
137                            "The specified file could not be found: " + srcFilename,
138                            "File not found!",
139                            JOptionPane.ERROR_MESSAGE);
140                        return;
141                }
142
143                final String namePart = srcFilename.substring(0, srcFilename.toLowerCase(Locale.ROOT).lastIndexOf(".xml"));
144                final int percentage = ((Integer) PopulationSampler.this.pctSpinner.getValue()).intValue();
145                final double samplesize = percentage / 100.0;
146
147                JFileChooser chooser = new SaveFileSaver();
148                chooser.setCurrentDirectory(srcFile.getParentFile());
149                chooser.setSelectedFile(new File(srcFile.getParentFile(), namePart + "." + Integer.toString(percentage) + "pct.xml.gz"));
150                int saveResult = chooser.showSaveDialog(PopulationSampler.this);
151                if (saveResult == JFileChooser.APPROVE_OPTION) {
152                        File destFile = chooser.getSelectedFile();
153                        try {
154                                createSample(srcFile, null, samplesize, destFile);
155                        } catch (RuntimeException | IOException ex) {
156                                destFile.delete();
157                                JOptionPane.showMessageDialog(null,
158                                    "<html>It looks like the population file cannot be parsed without a network file.<br />Please select a matching network file in the next dialog.</html>",
159                                    "Problems creating pouplation sample",
160                                    JOptionPane.WARNING_MESSAGE);
161                                
162                                JFileChooser netChooser = new JFileChooser();
163                                netChooser.setCurrentDirectory(srcFile.getParentFile());
164                                int result = netChooser.showOpenDialog(PopulationSampler.this);
165                                if (result == JFileChooser.APPROVE_OPTION) {
166                                        File networkFile = netChooser.getSelectedFile();
167                                        try {
168                                                createSample(srcFile, networkFile, samplesize, destFile);
169                                        } catch (RuntimeException | IOException ex2) {
170                                                destFile.delete();
171                                                JOptionPane.showMessageDialog(null,
172                                                    "The population sample cannot be created, as not all necessary data is available.",
173                                                    "Cannot create population sample",
174                                                    JOptionPane.ERROR_MESSAGE);
175                                        }
176                                }
177                        }
178                }               
179                
180        }

182        private static void createSample(final File inputPopulationFile, final File networkFile, final double samplesize, final File outputPopulationFile) throws RuntimeException, IOException {
183                MutableScenario sc = ScenarioUtils.createMutableScenario(ConfigUtils.createConfig());
184                
185                if (networkFile != null) {
186                        try (FileInputStream fis = new FileInputStream(networkFile);
187                                        BufferedInputStream is = (networkFile.getName().toLowerCase(Locale.ROOT).endsWith(".gz")) ? 
188                                                        new BufferedInputStream(new UnicodeInputStream(new GZIPInputStream(fis))) :
189                                                                new BufferedInputStream(new UnicodeInputStream(fis))
190                                        ) {
191                                AsyncFileInputProgressDialog gui = new AsyncFileInputProgressDialog(fis, "Loading Network…");
192                                try {
193                                        new MatsimNetworkReader(sc.getNetwork()).parse(is);
194                                } finally {
195                                        gui.close();
196                                }
197                        }
198                }
199                
200//              Population pop = (Population) sc.getPopulation();
201                StreamingPopulationReader reader = new StreamingPopulationReader( sc ) ;
202                StreamingDeprecated.setIsStreaming(reader, true);
203                
204                StreamingPopulationWriter writer = null;
205                try {
206                
207                        writer = new StreamingPopulationWriter(null, samplesize);
208                        writer.startStreaming(outputPopulationFile.getAbsolutePath());
209                        final PersonAlgorithm algo = writer;
210                        
211                        reader.addAlgorithm(algo);
212                        
213                        try (FileInputStream fis = new FileInputStream(inputPopulationFile);
214                                BufferedInputStream is = (inputPopulationFile.getName().toLowerCase(Locale.ROOT).endsWith(".gz")) ? 
215                                                new BufferedInputStream(new UnicodeInputStream(new GZIPInputStream(fis))) :
216                                                new BufferedInputStream(new UnicodeInputStream(fis))
217                                ) {
218                                AsyncFileInputProgressDialog gui = new AsyncFileInputProgressDialog(fis, "Creating Population Sample…");
219                                try {
220//                                      new MatsimPopulationReader(sc).parse(is);
221                                        reader.parse(is);
222                                } finally {
223                                        gui.close();
224                                }
225                        }
226                } catch (NullPointerException e) {
227                        throw new RuntimeException(e);
228                } finally {
229                        if (writer != null) {
230                                writer.closeStreaming();
231                        }
232                }
233                
234        }
