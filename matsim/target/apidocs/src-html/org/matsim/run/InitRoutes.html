048/**
049 * Assigns for each leg of each plan of each person an initial (freespeed) route.
050 * All given activities must have a link assigned already (use XY2Links).
051 *
052 * @author balmermi
053 * @author mrieser
054 */
055public class InitRoutes {
056
057        //////////////////////////////////////////////////////////////////////
058        // member variables
059        //////////////////////////////////////////////////////////////////////
060
061        private Config config;
062        private String configfile = null;
063        private String plansfile = null;
064
065        //////////////////////////////////////////////////////////////////////
066        // parse methods
067        //////////////////////////////////////////////////////////////////////
068
069        /**
070         * Parses all arguments and sets the corresponding members.
071         *
072         * @param args
073         */
074        private void parseArguments(final String[] args) {
075                if (args.length == 0) {
076                        System.out.println("Too few arguments.");
077                        printUsage();
078                        System.exit(1);
079                }
080                Iterator<String> argIter = new ArgumentParser(args).iterator();
081                String arg = argIter.next();
082                if (arg.equals("-h") || arg.equals("--help")) {
083                        printUsage();
084                        System.exit(0);
085                } else {
086                        this.configfile = arg;
087                        this.plansfile = argIter.next();
088                        if (argIter.hasNext()) {
089                                System.out.println("Too many arguments.");
090                                printUsage();
091                                System.exit(1);
092                        }
093                }
094        }
095
096        //////////////////////////////////////////////////////////////////////
097        // print methods
098        //////////////////////////////////////////////////////////////////////
099
100        private void printUsage() {
101                System.out.println();
102                System.out.println("InitRoutes");
103                System.out.println("Reads a plans-file and assignes each leg in each plan of each person");
104                System.out.println("a an initial route (freespeed) based on the given netowrk. The modified plans/");
105                System.out.println("persons are then written out to file again.");
106                System.out.println();
107                System.out.println("usage: InitRoutes [OPTIONS] configfile");
108                System.out.println("       The following parameters must be given in the config-file:");
109                System.out.println("       - network.inputNetworkFile");
110                System.out.println("       - plans.inputPlansFile");
111                System.out.println("       - plans.outputPlansFile");
112                System.out.println();
113                System.out.println("Options:");
114                System.out.println("-h, --help:     Displays this message.");
115                System.out.println();
116                System.out.println("----------------");
117                System.out.println("2008, matsim.org");
118                System.out.println();
119        }
120
121        //////////////////////////////////////////////////////////////////////
122        // run method
123        //////////////////////////////////////////////////////////////////////
124
125        public void run(final String[] args) {
126                parseArguments(args);
127                this.config = ConfigUtils.loadConfig(this.configfile);
128                MatsimRandom.reset(config.global().getRandomSeed());
129                final MutableScenario scenario = ScenarioUtils.createMutableScenario(config);
130//              final Population plans = PopulationUtils.createStreamingPopulation( config.plans(), null );
131                StreamingPopulationReader reader = new StreamingPopulationReader( scenario ) ;
132
133                new MatsimNetworkReader(scenario.getNetwork()).readFile(config.network().getInputFile());
134
135                final StreamingPopulationWriter plansWriter = new StreamingPopulationWriter();
136                Gbl.assertNotNull(this.plansfile);
137                plansWriter.startStreaming(this.plansfile);
138                final FreespeedTravelTimeAndDisutility timeCostCalc = new FreespeedTravelTimeAndDisutility(config.planCalcScore());
139                com.google.inject.Injector injector = Injector.createInjector(scenario.getConfig(), new AbstractModule() {
140                        @Override
141                        public void install() {
142                        install(AbstractModule.override(Arrays.asList(new TripRouterModule()), new AbstractModule() {
143                                @Override
144                                public void install() {
145                                install(new ScenarioByInstanceModule(scenario));
146                                addTravelTimeBinding("car").toInstance(timeCostCalc);
147                                addTravelDisutilityFactoryBinding("car").toInstance(new TravelDisutilityFactory() {
148                                        @Override
149                                        public TravelDisutility createTravelDisutility(TravelTime timeCalculator) {
150                                                return timeCostCalc;
151                                        }
152                                });
153                                }
154                        }));
155                        }
156                });
157                reader.addAlgorithm(new PlanRouter(injector.getInstance(TripRouter.class), null));
158                reader.addAlgorithm(plansWriter);
159                reader.readFile(this.config.plans().getInputFile());
160                PopulationUtils.printPlansCount(reader) ;
161                plansWriter.closeStreaming();
162
163                System.out.println("done.");
164        }
165
166        //////////////////////////////////////////////////////////////////////
167        // main method
168        //////////////////////////////////////////////////////////////////////
169
170        public static void main(final String[] args) {
171                new InitRoutes().run(args);
172        }
