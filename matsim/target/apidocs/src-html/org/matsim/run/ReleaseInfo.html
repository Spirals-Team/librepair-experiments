021package org.matsim.run;
022
023import java.io.BufferedReader;
024import java.io.IOException;
025import java.io.InputStreamReader;
026import java.net.URL;
027
028/**
029 * This is just a simple class to use as main class in jar-file releases.
030 *
031 * @author mrieser
032 */
033public class ReleaseInfo {
034
035        public static void main(final String[] args) {
036
037                // try to load the svn-revision used to build and the build-date from the information in the jar-file
038                String revision = null;
039                String date = null;
040                URL url = ReleaseInfo.class.getResource("/revision.txt");
041                if (url != null) {
042                        try (BufferedReader reader = new BufferedReader(new InputStreamReader(url.openStream()))) {
043                                revision = reader.readLine();
044                                date = reader.readLine();
045                        } catch (IOException e) {
046                                e.printStackTrace();
047                        }
048                }
049
050                // output copyright-message
051                System.out.println();
052                System.out.println("MATSim");
053                System.out.println("    Multi-Agent Transport Simulation Toolkit");
054                if (revision == null) {
055                        System.out.println("    Build: unknown");
056                } else {
057                        System.out.println("    Build: " + revision + " (" + date + ")");
058                }
059
060                System.out.println();
061                System.out.println("Copyright (C) 2012 by");
062                System.out.println("    Kay W. Axhausen, Michael Balmer, Christoph Dobler, Thibaut Dubernet,");
063                System.out.println("    Dominik Grether, Andreas Horni, Gregor Laemmel, Nicolas Lefebvre,");
064                System.out.println("    Fabrice Marchal, Konrad Meister, Kai Nagel, Andreas Neumann,");
065                System.out.println("    Marcel Rieser, David Strippgen, Rashid Waraich, Michael Zilske,");
066                System.out.println("    Technische Universitaet Berlin (TU-Berlin) and");
067                System.out.println("    Swiss Federal Institute of Technology Zurich (ETHZ)");
068                System.out.println();
069                System.out.println("This program is distributed under the Gnu Public License (GPL) 2 and");
070                System.out.println("comes WITHOUT ANY WARRANTY.");
071                System.out.println("Please see the files WARRANTY, LICENSE and COPYING in the distribution.");
072                System.out.println();
073        }
074
075}
