023package org.matsim.analysis;
024
025import org.jfree.chart.ChartFactory;
026import org.jfree.chart.ChartUtilities;
027import org.jfree.chart.JFreeChart;
028import org.jfree.chart.axis.CategoryAxis;
029import org.jfree.chart.axis.NumberAxis;
030import org.jfree.chart.plot.PlotOrientation;
031import org.jfree.chart.plot.XYPlot;
032import org.jfree.data.xy.XYSeries;
033import org.jfree.data.xy.XYSeriesCollection;
034import org.matsim.core.utils.io.UncheckedIOException;
035
036import java.awt.*;
037import java.io.File;
038import java.io.IOException;
039
040public class LegHistogramChart {
041    static JFreeChart getGraphic(final LegHistogram.DataFrame dataFrame, final String mode, int iteration) {
042                final XYSeriesCollection xyData = new XYSeriesCollection();
043                final XYSeries departuresSerie = new XYSeries("departures", false, true);
044                final XYSeries arrivalsSerie = new XYSeries("arrivals", false, true);
045                final XYSeries onRouteSerie = new XYSeries("en route", false, true);
046                int onRoute = 0;
047                for (int i = 0; i < dataFrame.countsDep.length; i++) {
048                        onRoute = onRoute + dataFrame.countsDep[i] - dataFrame.countsArr[i] - dataFrame.countsStuck[i];
049                        double hour = i*dataFrame.binSize / 60.0 / 60.0;
050                        departuresSerie.add(hour, dataFrame.countsDep[i]);
051                        arrivalsSerie.add(hour, dataFrame.countsArr[i]);
052                        onRouteSerie.add(hour, onRoute);
053                }
054
055                xyData.addSeries(departuresSerie);
056                xyData.addSeries(arrivalsSerie);
057                xyData.addSeries(onRouteSerie);
058
059        final JFreeChart chart = ChartFactory.createXYStepChart(
060                "Leg Histogram, " + mode + ", it." + iteration,
061                "time", "# persons",
062                xyData,
063                PlotOrientation.VERTICAL,
064                true,   // legend
065                false,   // tooltips
066                false   // urls
067        );
068
069                XYPlot plot = chart.getXYPlot();
070
071                final CategoryAxis axis1 = new CategoryAxis("hour");
072                axis1.setTickLabelFont(new Font("SansSerif", Font.PLAIN, 7));
073                plot.setDomainAxis(new NumberAxis("time"));
074
075                plot.getRenderer().setSeriesStroke(0, new BasicStroke(2.0f));
076                plot.getRenderer().setSeriesStroke(1, new BasicStroke(2.0f));
077                plot.getRenderer().setSeriesStroke(2, new BasicStroke(2.0f));
078                plot.setBackgroundPaint(Color.white);
079                plot.setRangeGridlinePaint(Color.gray);
080                plot.setDomainGridlinePaint(Color.gray);
081
082                return chart;
083        }
084
085    /**
086         * Writes a graphic showing the number of departures, arrivals and vehicles
087         * en route of all legs/trips to the specified file.
088         *
089         * @param legHistogram
090     * @param filename
091         *
092         */
093        public static void writeGraphic(LegHistogram legHistogram, final String filename) {
094                try {
095            ChartUtilities.saveChartAsPNG(new File(filename), getGraphic(legHistogram.getAllModesData(), "all", legHistogram.getIteration()), 1024, 768);
096                } catch (IOException e) {
097            throw new UncheckedIOException(e);
098                }
099        }
100
101    /**
102         * Writes a graphic showing the number of departures, arrivals and vehicles
103         * en route of all legs/trips with the specified transportation mode to the
104         * specified file.
105         *
106         * @param legHistogram
107     * @param filename
108         * @param legMode
109         *
110         */
111        public static void writeGraphic(LegHistogram legHistogram, final String filename, final String legMode) {
112                try {
113            ChartUtilities.saveChartAsPNG(new File(filename), getGraphic(legHistogram.getDataForMode(legMode), legMode, legHistogram.getIteration()), 1024, 768);
114                } catch (IOException e) {
115                        throw new UncheckedIOException(e);
116                }
117        }
118}
