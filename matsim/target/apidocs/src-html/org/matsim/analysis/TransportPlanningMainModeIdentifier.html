001package org.matsim.analysis;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import org.matsim.api.core.v01.TransportMode;
007import org.matsim.api.core.v01.population.Leg;
008import org.matsim.api.core.v01.population.PlanElement;
009import org.matsim.core.router.MainModeIdentifier;
010
011/**
012 * Attempt to provide a {@link MainModeIdentifier} that is justified from the transport engineering perspective, not from the
013 * software perspective.
014 * 
015 * @author nagel
016 *
017 */
018public final class TransportPlanningMainModeIdentifier implements MainModeIdentifier {
019        // I am against other people changing this lightheartedly since some of my analysis depends on it.  So either please discuss,
020        // or use your own variant.  kai, sep'16
021
022        // yyyyyy some variant of this (maybe not exactly this one here) should be able to deal with multi-leg trips.  kai/amit, mar'17
023
024        private final List<String> modeHierarchy = new ArrayList<>() ;
025
026        public TransportPlanningMainModeIdentifier() {
027                modeHierarchy.add( TransportMode.access_walk ) ;
028                modeHierarchy.add( TransportMode.egress_walk ) ;
029                modeHierarchy.add( TransportMode.transit_walk ) ;
030                modeHierarchy.add( "undefined" ) ;
031                modeHierarchy.add( TransportMode.other ) ;
032                modeHierarchy.add( TransportMode.transit_walk ) ;
033                modeHierarchy.add( TransportMode.walk ) ;
034                modeHierarchy.add( TransportMode.bike ) ;
035                modeHierarchy.add( TransportMode.pt ) ;
036                modeHierarchy.add( TransportMode.ride ) ;
037                modeHierarchy.add( TransportMode.car ) ;
038                
039                // NOTE: This hierarchical stuff is not so great: is park-n-ride a car trip or a pt trip?  Could weigh it by distance, or by time spent
040                // in respective mode.  Or have combined modes as separate modes.  In any case, can't do it at the leg level, since it does not
041                // make sense to have the system calibrate towards something where we have counted the car and the pt part of a multimodal
042                // trip as two separate trips. kai, sep'16
043        }
044
045        @Override public String identifyMainMode( List<? extends PlanElement> planElements ) {
046                int mainModeIndex = -1 ;
047                for ( PlanElement pe : planElements ) {
048                        if ( pe instanceof Leg ) {
049                                Leg leg = (Leg) pe ;
050                                int index = modeHierarchy.indexOf( leg.getMode() ) ;
051                                if ( index < 0 ) {
052                                        throw new RuntimeException("unknown mode=" + leg.getMode() ) ;
053                                }
054                                if ( index > mainModeIndex ) {
055                                        mainModeIndex = index ;
056                                }
057                        }
058                }
059                if ( mainModeIndex <= modeHierarchy.indexOf( TransportMode.transit_walk ) ) {
060                        mainModeIndex = modeHierarchy.indexOf( TransportMode.walk ) ;
061                }
062                return modeHierarchy.get( mainModeIndex ) ;
063        }
064}
