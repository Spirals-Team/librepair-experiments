021package org.matsim.analysis;
022
023import java.util.HashMap;
024import java.util.Map;
025import java.util.Set;
026import java.util.TreeSet;
027
028import org.apache.log4j.Logger;
029import org.matsim.api.core.v01.Id;
030import org.matsim.api.core.v01.events.LinkLeaveEvent;
031import org.matsim.api.core.v01.events.VehicleEntersTrafficEvent;
032import org.matsim.api.core.v01.events.handler.LinkLeaveEventHandler;
033import org.matsim.api.core.v01.events.handler.VehicleEntersTrafficEventHandler;
034import org.matsim.api.core.v01.network.Link;
035import org.matsim.api.core.v01.network.Network;
036import org.matsim.vehicles.Vehicle;
037import org.matsim.core.api.experimental.events.EventsManager;
038
039import javax.inject.Inject;
040
041/**
042 * Counts the number of vehicles leaving a link, aggregated into time bins of a specified size.
043 *
044 * @author mrieser
045 */
046public class VolumesAnalyzer implements LinkLeaveEventHandler, VehicleEntersTrafficEventHandler {
047
048        private final static Logger log = Logger.getLogger(VolumesAnalyzer.class);
049        private final int timeBinSize;
050        private final int maxTime;
051        private final int maxSlotIndex;
052        private final Map<Id<Link>, int[]> links;
053        
054        // for multi-modal support
055        private final boolean observeModes;
056        private final Map<Id<Vehicle>, String> enRouteModes;
057        private final Map<Id<Link>, Map<String, int[]>> linksPerMode;
058
059        @Inject
060        VolumesAnalyzer(Network network, EventsManager eventsManager) {
061                this(3600, 24 * 3600 - 1, network);
062                eventsManager.addHandler(this);
063        }
064
065        public VolumesAnalyzer(final int timeBinSize, final int maxTime, final Network network) {
066                this(timeBinSize, maxTime, network, true);
067        }
068        
069        public VolumesAnalyzer(final int timeBinSize, final int maxTime, final Network network, boolean observeModes) {
070                this.timeBinSize = timeBinSize;
071                this.maxTime = maxTime;
072                this.maxSlotIndex = (this.maxTime/this.timeBinSize) + 1;
073                this.links = new HashMap<>((int) (network.getLinks().size() * 1.1), 0.95f);
074                
075                this.observeModes = observeModes;
076                if (this.observeModes) {
077                        this.enRouteModes = new HashMap<>();
078                        this.linksPerMode = new HashMap<>((int) (network.getLinks().size() * 1.1), 0.95f);
079                } else {
080                        this.enRouteModes = null;
081                        this.linksPerMode = null;
082                }
083        }
084        
085        @Override
086        public void handleEvent(VehicleEntersTrafficEvent event) {
087                if (observeModes) {
088                        enRouteModes.put(event.getVehicleId(), event.getNetworkMode());
089                }
090        }
091        
092        @Override
093        public void handleEvent(final LinkLeaveEvent event) {
094                int[] volumes = this.links.get(event.getLinkId());
095                if (volumes == null) {
096                        volumes = new int[this.maxSlotIndex + 1]; // initialized to 0 by default, according to JVM specs
097                        this.links.put(event.getLinkId(), volumes);
098                }
099                int timeslot = getTimeSlotIndex(event.getTime());
100                volumes[timeslot]++;
101                
102                if (observeModes) {
103                        Map<String, int[]> modeVolumes = this.linksPerMode.get(event.getLinkId());
104                        if (modeVolumes == null) {
105                                modeVolumes = new HashMap<>();
106                                this.linksPerMode.put(event.getLinkId(), modeVolumes);
107                        }
108                        String mode = enRouteModes.get(event.getVehicleId());
109                        volumes = modeVolumes.get(mode);
110                        if (volumes == null) {
111                                volumes = new int[this.maxSlotIndex + 1]; // initialized to 0 by default, according to JVM specs
112                                modeVolumes.put(mode, volumes);
113                        }
114                        volumes[timeslot]++;
115                }
116        }
117
118        private int getTimeSlotIndex(final double time) {
119                if (time > this.maxTime) {
120                        return this.maxSlotIndex;
121                }
122                return ((int)time / this.timeBinSize);
123        }
124
125        /**
126         * @param linkId
127         * @return Array containing the number of vehicles leaving the link <code>linkId</code> per time bin,
128         *              starting with time bin 0 from 0 seconds to (timeBinSize-1)seconds.
129         */
130        public int[] getVolumesForLink(final Id<Link> linkId) {
131                return this.links.get(linkId);
132        }
133        
134        /**
135         * @param linkId
136         * @param mode
137         * @return Array containing the number of vehicles using the specified mode leaving the link 
138         *      <code>linkId</code> per time bin, starting with time bin 0 from 0 seconds to (timeBinSize-1)seconds.
139         */
140        public int[] getVolumesForLink(final Id<Link> linkId, String mode) {
141                if (observeModes) {
142                        Map<String, int[]> modeVolumes = this.linksPerMode.get(linkId);
143                        if (modeVolumes != null) return modeVolumes.get(mode);
144                } 
145                return null;
146        }
147
148        /**
149         *
150         * @return The size of the arrays returned by calls to the {@link #getVolumesForLink(Id)} and the {@link #getVolumesForLink(Id, String)}
151         * methods.
152         */
153        public int getVolumesArraySize() {
154                return this.maxSlotIndex + 1;
155        }
156        
157        /*
158         * This procedure is only working if (hour % timeBinSize == 0)
159         * 
160         * Example: 15 minutes bins
161         *  ___________________
162         * |  0 | 1  | 2  | 3  |
163         * |____|____|____|____|
164         * 0   900 1800  2700 3600
165                ___________________
166         * |      hour 0           |
167         * |___________________|
168         * 0                              3600
169         * 
170         * hour 0 = bins 0,1,2,3
171         * hour 1 = bins 4,5,6,7
172         * ...
173         * 
174         * getTimeSlotIndex = (int)time / this.timeBinSize => jumps at 3600.0!
175         * Thus, starting time = (hour = 0) * 3600.0
176         */
177        public double[] getVolumesPerHourForLink(final Id<Link> linkId) {
178                if (3600.0 % this.timeBinSize != 0) log.error("Volumes per hour and per link probably not correct!");
179                
180                double[] volumes = new double[24];
181                
182                int[] volumesForLink = this.getVolumesForLink(linkId);
183                if (volumesForLink == null) return volumes;
184
185                int slotsPerHour = (int)(3600.0 / this.timeBinSize);
186                for (int hour = 0; hour < 24; hour++) {
187                        double time = hour * 3600.0;
188                        for (int i = 0; i < slotsPerHour; i++) {
189                                volumes[hour] += volumesForLink[this.getTimeSlotIndex(time)];
190                                time += this.timeBinSize;
191                        }
192                }
193                return volumes;
194        }
195
196        public double[] getVolumesPerHourForLink(final Id<Link> linkId, String mode) {
197                if (observeModes) {
198                        if (3600.0 % this.timeBinSize != 0) log.error("Volumes per hour and per link probably not correct!");
199                        
200                        double [] volumes = new double[24];
201                        for (int hour = 0; hour < 24; hour++) {
202                                volumes[hour] = 0.0;
203                        }
204                        
205                        int[] volumesForLink = this.getVolumesForLink(linkId, mode);
206                        if (volumesForLink == null) return volumes;
207        
208                        int slotsPerHour = (int)(3600.0 / this.timeBinSize);
209                        for (int hour = 0; hour < 24; hour++) {
210                                double time = hour * 3600.0;
211                                for (int i = 0; i < slotsPerHour; i++) {
212                                        volumes[hour] += volumesForLink[this.getTimeSlotIndex(time)];
213                                        time += this.timeBinSize;
214                                }
215                        }
216                        return volumes;
217                }
218                return null;
219        }
220        
221        /**
222         * @return Set of Strings containing all modes for which counting-values are available.
223         */
224        public Set<String> getModes() {
225                Set<String> modes = new TreeSet<>();
226                
227                for (Map<String, int[]> map : this.linksPerMode.values()) {
228                        modes.addAll(map.keySet());
229                }
230                
231                return modes;
232        }
233        
234        /**
235         * @return Set of Strings containing all link ids for which counting-values are available.
236         */
237        public Set<Id<Link>> getLinkIds() {
238                return this.links.keySet();
239        }
240
241        @Override
242        public void reset(final int iteration) {
243                this.links.clear();
244                if (observeModes) {
245                        this.linksPerMode.clear();
246                        this.enRouteModes.clear();
247                }
248        }
249}
