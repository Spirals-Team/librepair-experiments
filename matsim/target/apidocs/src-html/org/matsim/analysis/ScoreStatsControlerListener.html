062/**
063 * Calculates at the end of each iteration the following statistics:
064 * <ul>
065 * <li>average score of the selected plan</li>
066 * <li>average of the score of the worst plan of each agent</li>
067 * <li>average of the score of the best plan of each agent</li>
068 * <li>average of the average score of all plans of each agent</li>
069 * </ul>
070 * Plans with undefined scores
071 * are not included in the statistics. The calculated values are written to a file, each iteration on
072 * a separate line.
073 *
074 * @author mrieser
075 */
076public class ScoreStatsControlerListener implements StartupListener, IterationEndsListener, ShutdownListener, ScoreStats {
077
078        public static final String FILENAME_SCORESTATS = "scorestats";
079
080        public static enum ScoreItem { worst, best, average, executed } ;
081
082        final private Population population;
083        final private BufferedWriter out;
084        final private String fileName;
085        
086        private final boolean createPNG;
087        private final ControlerConfigGroup controlerConfigGroup;
088
089        Map<ScoreItem,Map< Integer, Double>> scoreHistory = new HashMap<>() ;
090        private int minIteration = 0;
091
092        private final static Logger log = Logger.getLogger(ScoreStatsControlerListener.class);
093
094        @Inject
095        ScoreStatsControlerListener(ControlerConfigGroup controlerConfigGroup, Population population1, OutputDirectoryHierarchy controlerIO,
096                        PlanCalcScoreConfigGroup scoreConfig, Provider<TripRouter> tripRouterFactory ) {
097                this.controlerConfigGroup = controlerConfigGroup;
098                this.population = population1;
099                this.fileName = controlerIO.getOutputFilename(FILENAME_SCORESTATS);
100                this.createPNG = controlerConfigGroup.isCreateGraphs();
101                this.out = IOUtils.getBufferedWriter(this.fileName + ".txt");
102                try {
103                        this.out.write("ITERATION\tavg. EXECUTED\tavg. WORST\tavg. AVG\tavg. BEST\n");
104                } catch (IOException e) {
105                        throw new UncheckedIOException(e);
106                }
107        }
108
109        @Override
110        public void notifyStartup(final StartupEvent event) {
111                this.minIteration = controlerConfigGroup.getFirstIteration();
112                //              int maxIter = controlerConfigGroup.getLastIteration();
113                //              int iterations = maxIter - this.minIteration;
114                //              if (iterations > 5000) iterations = 5000; // limit the history size
115                for ( ScoreItem item : ScoreItem.values() ) {
116                        scoreHistory.put( item, new TreeMap<Integer,Double>() ) ;
117                }
118        }
119
120        @Override
121        public void notifyIterationEnds(final IterationEndsEvent event) {
122                collectScoreInfo(event);
123        }
124
125        private void collectScoreInfo(final IterationEndsEvent event) {
126                double sumScoreWorst = 0.0;
127                double sumScoreBest = 0.0;
128                double sumAvgScores = 0.0;
129                double sumExecutedScores = 0.0;
130                int nofScoreWorst = 0;
131                int nofScoreBest = 0;
132                int nofAvgScores = 0;
133                int nofExecutedScores = 0;
134
135                for (Person person : this.population.getPersons().values()) {
136                        Plan worstPlan = null;
137                        Plan bestPlan = null;
138                        double worstScore = Double.POSITIVE_INFINITY;
139                        double bestScore = Double.NEGATIVE_INFINITY;
140                        double sumScores = 0.0;
141                        double cntScores = 0;
142                        for (Plan plan : person.getPlans()) {
143
144                                if (plan.getScore() == null) {
145                                        continue;
146                                }
147                                double score = plan.getScore();
148
149                                // worst plan
150                                if (worstPlan == null) {
151                                        worstPlan = plan;
152                                        worstScore = score;
153                                } else if (score < worstScore) {
154                                        worstPlan = plan;
155                                        worstScore = score;
156                                }
157
158                                // best plan
159                                if (bestPlan == null) {
160                                        bestPlan = plan;
161                                        bestScore = score;
162                                } else if (score > bestScore) {
163                                        bestPlan = plan;
164                                        bestScore = score;
165                                }
166
167                                // avg. score
168                                sumScores += score;
169                                cntScores++;
170
171                                // executed plan?
172                                if (PersonUtils.isSelected(plan)) {
173                                        sumExecutedScores += score;
174                                        nofExecutedScores++;
175                                }
176                        }
177
178                        if (worstPlan != null) {
179                                nofScoreWorst++;
180                                sumScoreWorst += worstScore;
181                        }
182                        if (bestPlan != null) {
183                                nofScoreBest++;
184                                sumScoreBest += bestScore;
185                        }
186                        if (cntScores > 0) {
187                                sumAvgScores += (sumScores / cntScores);
188                                nofAvgScores++;
189                        }
190                }
191                log.info("-- avg. score of the executed plan of each agent: " + (sumExecutedScores / nofExecutedScores));
192                log.info("-- avg. score of the worst plan of each agent: " + (sumScoreWorst / nofScoreWorst));
193                log.info("-- avg. of the avg. plan score per agent: " + (sumAvgScores / nofAvgScores));
194                log.info("-- avg. score of the best plan of each agent: " + (sumScoreBest / nofScoreBest));
195
196                try {
197                        this.out.write(event.getIteration() + "\t" + (sumExecutedScores / nofExecutedScores) + "\t" +
198                                        (sumScoreWorst / nofScoreWorst) + "\t" + (sumAvgScores / nofAvgScores) + "\t" + (sumScoreBest / nofScoreBest) + "\n");
199                        this.out.flush();
200                } catch (IOException e) {
201                        e.printStackTrace();
202                }
203
204//              int index = event.getIteration() - this.minIteration;
205
206                this.scoreHistory.get( ScoreItem.worst ).put( event.getIteration(), sumScoreWorst / nofScoreWorst ) ;
207                this.scoreHistory.get( ScoreItem.best ).put( event.getIteration(), sumScoreBest / nofScoreBest ) ;
208                this.scoreHistory.get( ScoreItem.average ).put( event.getIteration(), sumAvgScores / nofAvgScores ) ;
209                this.scoreHistory.get( ScoreItem.executed ).put( event.getIteration(), sumExecutedScores / nofExecutedScores ) ;
210
211                if (this.createPNG && event.getIteration() > this.minIteration) {
212                        // create chart when data of more than one iteration is available.
213                        XYLineChart chart = new XYLineChart("Score Statistics", "iteration", "score");
214//                      double[] iterations = new double[index + 1];
215//                      for (int i = 0; i <= index; i++) {
216//                              iterations[i] = i + this.minIteration;
217//                      }
218                        chart.addSeries("avg. worst score", this.scoreHistory.get( ScoreItem.worst ) ) ;
219                        chart.addSeries("avg. best score", this.scoreHistory.get( ScoreItem.best) );
220                        chart.addSeries("avg. of plans' average score", this.scoreHistory.get( ScoreItem.average) );
221                        chart.addSeries("avg. executed score", this.scoreHistory.get( ScoreItem.executed ) );
222                        chart.addMatsimLogo();
223                        chart.saveAsPng(this.fileName + ".png", 800, 600);
224                }
225        }
226
227        @Override
228        public void notifyShutdown(final ShutdownEvent controlerShudownEvent) {
229                try {
230                        this.out.close();
231                } catch (IOException e) {
232                        e.printStackTrace();
233                }
234
235        }
236
237        @Override
238        public Map<ScoreItem, Map<Integer, Double>> getScoreHistory() {
239                return Collections.unmodifiableMap( this.scoreHistory ) ;
240        }
