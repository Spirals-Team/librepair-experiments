021package org.matsim.analysis;
022
023import org.apache.log4j.Logger;
024import org.matsim.api.core.v01.Id;
025import org.matsim.api.core.v01.events.ActivityEndEvent;
026import org.matsim.api.core.v01.events.ActivityStartEvent;
027import org.matsim.api.core.v01.events.PersonArrivalEvent;
028import org.matsim.api.core.v01.events.PersonDepartureEvent;
029import org.matsim.api.core.v01.events.handler.ActivityEndEventHandler;
030import org.matsim.api.core.v01.events.handler.ActivityStartEventHandler;
031import org.matsim.api.core.v01.events.handler.PersonArrivalEventHandler;
032import org.matsim.api.core.v01.events.handler.PersonDepartureEventHandler;
033import org.matsim.api.core.v01.population.Person;
034import org.matsim.core.api.experimental.events.EventsManager;
035import org.matsim.core.utils.io.IOUtils;
036import org.matsim.core.utils.io.UncheckedIOException;
037import org.matsim.core.utils.misc.Time;
038
039import javax.inject.Inject;
040import java.io.BufferedWriter;
041import java.io.IOException;
042import java.util.HashMap;
043import java.util.Map;
044import java.util.TreeMap;
045
046/**
047 * @author mrieser
048 *
049 * Calculates the distribution of legs-durations, e.g. how many legs took at
050 * most 5 minutes, how many between 5 and 10 minutes, and so on.
051 * Also calculates the average trip duration.
052 * Trips ended because of vehicles being stuck are not counted.
053 */
054public class CalcLegTimes implements PersonDepartureEventHandler, PersonArrivalEventHandler, 
055        ActivityEndEventHandler, ActivityStartEventHandler {
056
057        private final static Logger log = Logger.getLogger(CalcLegTimes.class);
058        
059        private static final int SLOT_SIZE = 300;       // 5-min slots
060        private static final int MAXINDEX = 12; // slots 0..11 are regular slots, slot 12 is anything above
061
062        private final Map<Id<Person>, Double> agentDepartures = new HashMap<>();
063        private final Map<Id<Person>, Double> agentArrivals = new HashMap<>();
064        private final Map<String, int[]> legStats = new TreeMap<>();
065        private final Map<Id<Person>, String> previousActivityTypes = new HashMap<>();
066        private double sumTripDurations = 0;
067        private int sumTrips = 0;
068
069        @Inject
070        CalcLegTimes(EventsManager eventsManager) {
071                eventsManager.addHandler(this);
072        }
073
074        public CalcLegTimes() {
075
076        }
077
078        @Override
079        public void handleEvent(ActivityEndEvent event) {
080                this.previousActivityTypes.put(event.getPersonId(), event.getActType());
081        }
082        
083        @Override
084        public void handleEvent(final PersonDepartureEvent event) {
085                this.agentDepartures.put(event.getPersonId(), event.getTime());
086        }
087
088        @Override
089        public void handleEvent(final PersonArrivalEvent event) {
090                this.agentArrivals.put(event.getPersonId(), event.getTime());
091        }
092
093
094        @Override
095        public void handleEvent(ActivityStartEvent event) {
096                Double depTime = this.agentDepartures.remove(event.getPersonId());
097                Double arrTime = this.agentArrivals.remove(event.getPersonId());
098                if (depTime != null) {
099                        double travTime = arrTime - depTime;
100                        String fromActType = previousActivityTypes.remove(event.getPersonId());
101                        String toActType = event.getActType();
102                        String legType = fromActType + "---" + toActType;
103                        int[] stats = this.legStats.get(legType);
104                        if (stats == null) {
105                                stats = new int[MAXINDEX+1];
106                                for (int i = 0; i <= MAXINDEX; i++) {
107                                        stats[i] = 0;
108                                }
109                                this.legStats.put(legType, stats);
110                        }
111                        stats[getTimeslotIndex(travTime)]++;
112
113                        this.sumTripDurations += travTime;
114                        this.sumTrips++;
115                }
116        }
117
118        
119        @Override
120        public void reset(final int iteration) {
121                this.previousActivityTypes.clear();
122                this.agentDepartures.clear();
123                this.legStats.clear();
124                this.sumTripDurations = 0;
125                this.sumTrips = 0;
126        }
127
128        public Map<String, int[]> getLegStats() {
129                return this.legStats;
130        }
131
132        public static int getTimeslotIndex(final double time_s) {
133                int idx = (int)(time_s / SLOT_SIZE);
134                if (idx > MAXINDEX) idx = MAXINDEX;
135                return idx;
136        }
137
138        public double getAverageTripDuration() {
139                return (this.sumTripDurations / this.sumTrips);
140        }
141
142        public void writeStats(final String filename) {
143                try (BufferedWriter legStatsFile = IOUtils.getBufferedWriter(filename)) {
144                        writeStats(legStatsFile);
145                } catch (IOException e) {
146                        log.error(e);
147                }
148        }
149
150        public void writeStats(final java.io.Writer out) throws UncheckedIOException {
151                try {
152                boolean first = true;
153                for (Map.Entry<String, int[]> entry : this.legStats.entrySet()) {
154                        String key = entry.getKey();
155                        int[] counts = entry.getValue();
156                        if (first) {
157                                first = false;
158                                out.write("pattern");
159                                for (int i = 0; i < counts.length; i++) {
160                                        out.write("\t" + (i*SLOT_SIZE/60) + "+");
161                                }
162                                out.write("\n");
163                        }
164                        out.write(key);
165            for (int count : counts) {
166                out.write("\t" + count);
167            }
168                        out.write("\n");
169                }
170                out.write("\n");
171                if (this.sumTrips == 0) {
172                        out.write("average trip duration: no trips!");
173                } else {
174                        out.write("average trip duration: "
175                                        + (this.sumTripDurations / this.sumTrips) + " seconds = "
176                                        + Time.writeTime(((int)(this.sumTripDurations / this.sumTrips))));
177                }
178                out.write("\n");
179                } catch (IOException e) {
180                        throw new UncheckedIOException(e);
181                } finally {
182                        try {
183                                out.flush();
184                        } catch (IOException e) {
185                                log.error(e);
186                        }
187                }
188        }
189}
