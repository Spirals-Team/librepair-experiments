036        private static final double MINIMUMFREESPEED = 3;
037
038        public ChangeableNetworkCreator(){
039                this.networkChangeEvents = new ArrayList<NetworkChangeEvent>();

041        }

049        private void run() {
050                prepareScen();
051                tcc =  readEvents();
052                createNetworkChangeEvents(sc.getNetwork(),tcc);
053                new NetworkChangeEventsWriter().write(CHANGEFILE, networkChangeEvents);
054        }
055        
056        public void createNetworkChangeEvents(Network network, TravelTimeCalculator tcc2) {
057                for (Link l : network.getLinks().values()){
058                        if (l.getId().toString().startsWith("pt")) continue;
059                        
060                        double length = l.getLength();
061                        double previousTravelTime=l.getLength()/l.getFreespeed()        ;       
062                        
063                        for (double time = 0; time<ENDTIME ; time = time+TIMESTEP){
064                                
065                                double newTravelTime = tcc2.getLinkTravelTimes().getLinkTravelTime(l, time, null, null);
066                                if (newTravelTime != previousTravelTime){
067                                        NetworkChangeEvent nce = new NetworkChangeEvent(time);
068                                        nce.addLink(l);
069                                        double newFreespeed = length / newTravelTime;
070                                        if (newFreespeed < MINIMUMFREESPEED) newFreespeed = MINIMUMFREESPEED;
071                                        ChangeValue freespeedChange = new ChangeValue(ChangeType.ABSOLUTE_IN_SI_UNITS, newFreespeed);
072                                        nce.setFreespeedChange(freespeedChange);
073                                        
074                                        
075                                        this.networkChangeEvents.add(nce);
076                                        previousTravelTime= newTravelTime;
077                                }
078                        }
079                }
080        }
081        
082
083        private void prepareScen() {
084                
085                sc = ScenarioUtils.createScenario(ConfigUtils.createConfig());
086                new MatsimNetworkReader(sc.getNetwork()).readFile(NETWORKFILE);
087        }
088
089        private TravelTimeCalculator readEvents (){
090                EventsManager manager = EventsUtils.createEventsManager();
091        
092                TravelTimeCalculatorConfigGroup ttccg = new TravelTimeCalculatorConfigGroup();
093                TravelTimeCalculator tc = new TravelTimeCalculator(sc.getNetwork(), ttccg);
094                manager.addHandler(tc);
095                new MatsimEventsReader(manager).readFile(EVENTSFILE);
096                return tc;
097        }
098
099        public List<NetworkChangeEvent> getNetworkChangeEvents() {
100                return networkChangeEvents;
101        }
