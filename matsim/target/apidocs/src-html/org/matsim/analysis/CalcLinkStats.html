039/**
040 * Calculates the average link volumes and travel times over any number of iterations.
041 *
042 * @author mrieser

046        private final static Logger log = Logger.getLogger(CalcLinkStats.class);

048        private static class LinkData {
049                public final double[][] volumes;
050                public final double[][] ttimes;
051
052                public LinkData(final double[][] linksVolumes, final double[][] linksTTimes) {
053                        this.volumes = linksVolumes.clone();
054                        this.ttimes = linksTTimes.clone();
055                }
056        }

058        private double volScaleFactor = 1.0;
059
060        private int count = 0;
061        private final Map<Id<Link>, LinkData> linkData;
062        private final int nofHours;
063        private final Network network;
064
065        private static final int MIN = 0;
066        private static final int MAX = 1;
067        private static final int SUM = 2;
068        private static final int NOF_STATS = 3;
069
070        @Inject
071        public CalcLinkStats(final Network network) {
072                this.network = network;
073                this.linkData = new TreeMap<>();
074                this.nofHours = 24;
075                reset();
076        }

085                this(network);

089        public void addData(final VolumesAnalyzer analyzer, final TravelTime ttimes) {
090                this.count++;
091                // TODO verify ttimes has hourly timeBin-Settings
092
093                // go through all links
094                for (Id<Link> linkId : this.linkData.keySet()) {
095                        
096                        // retrieve link from link ID
097                        Link link = this.network.getLinks().get(linkId);
098                        
099                        // get the volumes for the link ID from the analyzier
100                        double[] volumes = analyzer.getVolumesPerHourForLink(linkId);
101                        
102                        // get the destination container for the data from link data (could have gotten this through iterator right away)
103                        LinkData data = this.linkData.get(linkId);
104                        
105                        // prepare the sum variables (for volumes);
106                        long sumVolumes = 0; // daily (0-24) sum
107                        
108                        // go through all hours:
109                        for (int hour = 0; hour < this.nofHours; hour++) {
110                                
111                                // get travel time for hour
112                                double ttime = ttimes.getLinkTravelTime(link, hour*3600, null, null);
113                                
114                                // add for daily sum:
115                                sumVolumes += volumes[hour];
116
117                                // the following has something to do with the fact that we are doing this for multiple iterations.  So there are variations.
118                                // this collects min and max.  There is, however, no good control over how many iterations this is collected.
119                                if (this.count == 1) {
120                                        data.volumes[MIN][hour] = volumes[hour];
121                                        data.volumes[MAX][hour] = volumes[hour];
122                                        data.ttimes[MIN][hour] = ttime;
123                                        data.ttimes[MAX][hour] = ttime;
124                                } else {
125                                        if (volumes[hour] < data.volumes[MIN][hour]) data.volumes[MIN][hour] = volumes[hour];
126                                        if (volumes[hour] > data.volumes[MAX][hour]) data.volumes[MAX][hour] = volumes[hour];
127                                        if (ttime < data.ttimes[MIN][hour]) data.ttimes[MIN][hour] = ttime;
128                                        if (ttime > data.ttimes[MAX][hour]) data.ttimes[MAX][hour] = ttime;
129                                }
130                                
131                                // this is the regular summing up for each hour
132                                data.volumes[SUM][hour] += volumes[hour];
133                                data.ttimes[SUM][hour] += volumes[hour] * ttime;
134                        }
135                        // dataVolumes[.][nofHours] are daily (0-24) values
136                        if (this.count == 1) {
137                                data.volumes[MIN][this.nofHours] = sumVolumes;
138                                data.volumes[SUM][this.nofHours] = sumVolumes;
139                                data.volumes[MAX][this.nofHours] = sumVolumes;
140                        } else {
141                                if (sumVolumes < data.volumes[MIN][this.nofHours]) data.volumes[MIN][this.nofHours] = sumVolumes;
142                                data.volumes[SUM][this.nofHours] += sumVolumes;
143                                if (sumVolumes > data.volumes[MAX][this.nofHours]) data.volumes[MAX][this.nofHours] = sumVolumes;
144                        }
145                }
146        }
147
148        public void reset() {
149                this.linkData.clear();
150                this.count = 0;
151                log.info( " resetting `count' to zero.  This info is here since we want to check when this" +
152                                " is happening during normal simulation runs.  kai, jan'11") ;
153
154                // initialize our data-table
155                for (Link link : this.network.getLinks().values()) {
156                        LinkData data = new LinkData(new double[NOF_STATS][this.nofHours + 1], new double[NOF_STATS][this.nofHours]);
157                        this.linkData.put(link.getId(), data);
158                }
159
160        }

162        public void writeFile(final String filename) {
163                BufferedWriter out = null;
164                try {
165                        out = IOUtils.getBufferedWriter(filename);
166
167                        // write header
168                        out.write("LINK\tORIG_ID\tFROM\tTO\tLENGTH\tFREESPEED\tCAPACITY");
169                        for (int i = 0; i < this.nofHours; i++) {
170                                out.write("\tHRS" + i + "-" + (i+1) + "min");
171                                out.write("\tHRS" + i + "-" + (i+1) + "avg");
172                                out.write("\tHRS" + i + "-" + (i+1) + "max");
173                        }
174                        out.write("\tHRS0-" + this.nofHours + "min");
175                        out.write("\tHRS0-" + this.nofHours + "avg");
176                        out.write("\tHRS0-" + this.nofHours + "max");
177                        for (int i = 0; i < this.nofHours; i++) {
178                                out.write("\tTRAVELTIME" + i + "-" + (i+1) + "min");
179                                out.write("\tTRAVELTIME" + i + "-" + (i+1) + "avg");
180                                out.write("\tTRAVELTIME" + i + "-" + (i+1) + "max");
181                        }
182                        out.write("\n");
183
184                        // write data
185                        for (Map.Entry<Id<Link>, LinkData> entry : this.linkData.entrySet()) {
186                                Id<Link> linkId = entry.getKey();
187                                LinkData data = entry.getValue();
188                                Link link = this.network.getLinks().get(linkId);
189
190                                out.write(linkId.toString());
191                                out.write("\t"); // origId, no longer supported
192                                out.write("\t" + link.getFromNode().getId().toString());
193                                out.write("\t" + link.getToNode().getId().toString());
194                                out.write("\t" + Double.toString(link.getLength()));
195                                out.write("\t" + Double.toString(link.getFreespeed()));
196                                out.write("\t" + Double.toString(link.getCapacity()));
197
198                                // HRS0-1, HRS1-2, ... HRS23-24
199//                              int[] sum = {0, 0, 0};
200                                for (int i = 0; i < this.nofHours; i++) {
201                                        out.write("\t" + Double.toString(data.volumes[MIN][i]));
202//                                      sum[MIN] = sum[MIN] + data.volumes[MIN][i];
203                                        out.write("\t" + Double.toString((data.volumes[SUM][i]) / this.count));
204//                                      sum[SUM] = sum[SUM] + data.volumes[SUM][i];
205                                        out.write("\t" + Double.toString(data.volumes[MAX][i]));
206//                                      sum[MAX] = sum[MAX] + data.volumes[MAX][i];
207                                }
208
209                                // HRS0-nofHours
210                                out.write("\t" + Double.toString(data.volumes[MIN][this.nofHours]));
211                                out.write("\t" + Double.toString((data.volumes[SUM][this.nofHours]) / this.count));
212                                out.write("\t" + Double.toString(data.volumes[MAX][this.nofHours]));
213
214                                // TRAVELTIME0-1, TRAVELTIME1-2, ... TRAVELTIME23-24
215                                for (int i = 0; i < this.nofHours; i++) {
216                                        String ttimesMin = Double.toString(data.ttimes[MIN][i]);
217                                        out.write("\t" + ttimesMin);
218                                        if (data.volumes[SUM][i] == 0) {
219                                                // nobody traveled along the link in this hour, so we cannot calculate an average
220                                                // use the value available or the minimum instead (min and max should be the same, =freespeed)
221                                                double ttsum = data.ttimes[SUM][i];
222                                                if (ttsum != 0.0) {
223                                                        out.write("\t" + Double.toString(ttsum));
224                                                } else {
225                                                        out.write("\t" + ttimesMin);
226                                                }
227                                        } else {
228                                                double ttsum = data.ttimes[SUM][i];
229                                                if (ttsum == 0) {
230                                                        out.write("\t" + ttimesMin);
231                                                } else {
232                                                        out.write("\t" + Double.toString(ttsum / data.volumes[SUM][i]));
233                                                }
234                                        }
235                                        out.write("\t" + Double.toString(data.ttimes[MAX][i]));
236                                }
237                                out.write("\n");
238                        }
239
240                } catch (IOException e) {
241                        e.printStackTrace();
242                }
243                finally {
244                        if (out != null) {
245                                try {
246                                        out.close();
247                                } catch (IOException e) {
248                                        log.warn("Could not close output-stream.", e);
249                                }
250                        }
251                }
252        }

254        public void readFile(final String filename) {
255                // start with a clean, empty data structure
256                reset();
257
258                BufferedReader reader = null;
259                try {
260                        reader = IOUtils.getBufferedReader(filename);
261
262                        // read header
263                        String header = reader.readLine();
264                        if (header == null) {
265                                // it seems there is no data in this file...
266                                return;
267                        }
268                        // ignore the header, but if there was a header, set the count to 1
269                        this.count = 1;
270
271                        // read lines
272                        String line = reader.readLine();
273                        while (line != null) {
274                                String[] parts = StringUtils.explode(line, '\t');
275                                if (parts.length == 154) {
276                                        Id<Link> linkId = Id.create(parts[0], Link.class);
277                                        LinkData data = this.linkData.get(linkId);
278                                        if (data == null) {
279                                                System.err.println("CalcLinkStats.readFile(); unknown link: " + linkId.toString());
280                                        } else {
281                                                int baseTTimes;
282                                                for (int i = 0; i < this.nofHours; i++) {
283                                                        data.volumes[MIN][i] = Double.parseDouble(parts[7 + i*3]);
284                                                        data.volumes[MIN][i] *= this.volScaleFactor;
285                                                        data.volumes[SUM][i] = Double.parseDouble(parts[8 + i*3]);
286                                                        data.volumes[SUM][i] *= this.volScaleFactor;
287                                                        data.volumes[MAX][i] = Double.parseDouble(parts[9 + i*3]);
288                                                        data.volumes[MAX][i] *= this.volScaleFactor;
289                                                        baseTTimes = 7 + (this.nofHours+1)*3;
290                                                        data.ttimes[MIN][i] = Double.parseDouble(parts[baseTTimes + i*3]);
291                                                        if (data.volumes[SUM][i] == 0) {
292                                                                data.ttimes[SUM][i] = Double.parseDouble(parts[baseTTimes + i*3 + 1]);
293                                                        } else {
294                                                                data.ttimes[SUM][i] = Double.parseDouble(parts[baseTTimes + i*3 + 1]) * data.volumes[SUM][i];
295                                                        }
296                                                        data.ttimes[MAX][i] = Double.parseDouble(parts[baseTTimes + i*3 + 2]);
297                                                }
298                                                data.volumes[MIN][this.nofHours] = Double.parseDouble(parts[7 + this.nofHours*3]);
299                                                data.volumes[MIN][this.nofHours] *= this.volScaleFactor;
300                                                data.volumes[SUM][this.nofHours] = Double.parseDouble(parts[8 + this.nofHours*3]);
301                                                data.volumes[SUM][this.nofHours] *= this.volScaleFactor;
302                                                data.volumes[MAX][this.nofHours] = Double.parseDouble(parts[9 + this.nofHours*3]);
303                                                data.volumes[MAX][this.nofHours] *= this.volScaleFactor;
304                                        }
305                                }
306                                else if (parts.length == 153) {
307                                        String linkId = parts[0];
308                                        LinkData data = this.linkData.get(Id.create(linkId, Link.class));
309                                        if (data == null) {
310                                                System.err.println("CalcLinkStats.readFile(); unknown link: " + linkId);
311                                        } else {
312                                                int baseTTimes;
313                                                for (int i = 0; i < this.nofHours; i++) {
314                                                        data.volumes[MIN][i] = Double.parseDouble(parts[6 + i*3]);
315                                                        data.volumes[MIN][i] *= this.volScaleFactor;
316                                                        data.volumes[SUM][i] = Integer.parseInt(parts[7 + i*3]);
317                                                        data.volumes[SUM][i] *= this.volScaleFactor;
318                                                        data.volumes[MAX][i] = Double.parseDouble(parts[8 + i*3]);
319                                                        data.volumes[MAX][i] *= this.volScaleFactor;
320                                                        baseTTimes = 6 + (this.nofHours+1)*3;
321                                                        data.ttimes[MIN][i] = Double.parseDouble(parts[baseTTimes + i*3]);
322                                                        if (data.volumes[SUM][i] == 0) {
323                                                                data.ttimes[SUM][i] = Double.parseDouble(parts[baseTTimes + i*3 + 1]);
324                                                        } else {
325                                                                data.ttimes[SUM][i] = Double.parseDouble(parts[baseTTimes + i*3 + 1]) * data.volumes[SUM][i];
326                                                        }
327                                                        data.ttimes[MAX][i] = Double.parseDouble(parts[baseTTimes + i*3 + 2]);
328                                                }
329                                                data.volumes[MIN][this.nofHours] = Double.parseDouble(parts[6 + this.nofHours*3]);
330                                                data.volumes[MIN][this.nofHours] *= this.volScaleFactor;
331                                                data.volumes[SUM][this.nofHours] = Double.parseDouble(parts[7 + this.nofHours*3]);
332                                                data.volumes[SUM][this.nofHours] *= this.volScaleFactor;
333                                                data.volumes[MAX][this.nofHours] = Double.parseDouble(parts[8 + this.nofHours*3]);
334                                                data.volumes[MAX][this.nofHours] *= this.volScaleFactor;
335                                        }
336                                }
337                                else {
338                                        System.err.println("CalcLinkStats.readFile(); line cannot be parsed: " + line + " number of colums is: " + parts.length);
339                                        break;
340                                }
341                                line = reader.readLine();
342                        }
343
344                } catch (IOException e) {
345                        e.printStackTrace();
346                } finally {
347                        if (reader != null) {
348                                try { reader.close(); }
349                                catch (IOException e) {
350                                        log.warn("Could not close input-stream.", e);
351                                }
352                        }
353                }
354        }

356        /**
357         * @param linkId
358         * @return if no data is available, an array with length 0 is returned.
359         */
360        public double[] getAvgLinkVolumes(final Id<Link> linkId) {
361                LinkData data = this.linkData.get(linkId);
362                if (data == null) {
363                        return new double[0];
364                }
365                if (this.count == 0) {
366                        return new double[0];
367                }
368                double[] volumes = new double[this.nofHours];
369                for (int i = 0; i < this.nofHours; i++) {
370                        volumes[i] = (data.volumes[SUM][i]) / (this.count);
371                }
372                return volumes;
373        }

377         * @return if no data is available, an array with length 0 is returned.
378         * 
379         * The method reflects the (wrong) logic of what is done when writing the output and should eventually be deleted or modified.

382        @Deprecated
383        protected double[] getAvgTravelTimes(final Id<Link> linkId) {
384                LinkData data = this.linkData.get(linkId);
385                if (data == null) {
386                        return new double[0];
387                }
388                if (this.count == 0) {
389                        return new double[0];
390                }
391                double[] ttimesMin = new double[this.nofHours];
392                double[] ttimesSum = new double[this.nofHours];
393                double[] volumes = new double[this.nofHours];
394                
395                double[] avgTTimes = new double[this.nofHours];
396                
397                for (int i = 0; i < this.nofHours; i++) {
398                        volumes[i] = (data.volumes[SUM][i]) / (this.count);
399                        ttimesMin[i] = (data.ttimes[MIN][i]) / (this.count);
400                        ttimesSum[i] = (data.ttimes[SUM][i]) / (this.count);
401
402                        if (volumes[i] == 0.) {
403                                avgTTimes[i] = ttimesMin[i];
404                        } else {
405                                avgTTimes[i] = ttimesSum[i] / volumes[i];
406                        }
407                }
408                return avgTTimes;
409        }
