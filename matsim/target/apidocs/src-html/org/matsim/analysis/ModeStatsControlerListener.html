058/**
059 * Calculates at the end of each iteration mode statistics, based on the main mode identifier of a trip chain.
060 * For multi-modal trips, this is only as accurate as your main mode identifier.
061 * The calculated values are written to a file, each iteration on
062 * a separate line.
063 *
064 * @author mrieser
065 */
066public class ModeStatsControlerListener implements StartupListener, IterationEndsListener, 
067ShutdownListener {
068
069        public static final String FILENAME_MODESTATS = "modestats";
070
071        final private Population population;
072        
073        final private BufferedWriter modeOut ;
074        final private String modeFileName ;
075
076        private final boolean createPNG;
077        private final ControlerConfigGroup controlerConfigGroup;
078
079        Map<String,Map<Integer,Double>> modeHistories = new HashMap<>() ;
080        private int minIteration = 0;
081        private final Provider<TripRouter> tripRouterFactory;
082        private StageActivityTypes stageActivities;
083        private MainModeIdentifier mainModeIdentifier;
084        private Map<String,Double> modeCnt = new TreeMap<>() ;
085        
086        private final Set<String> modes;
087
088        private final static Logger log = Logger.getLogger(ModeStatsControlerListener.class);
089
090        @Inject
091        ModeStatsControlerListener(ControlerConfigGroup controlerConfigGroup, Population population1, OutputDirectoryHierarchy controlerIO,
092                        PlanCalcScoreConfigGroup scoreConfig, Provider<TripRouter> tripRouterFactory ) {
093                this.controlerConfigGroup = controlerConfigGroup;
094                this.population = population1;
095                this.modeFileName = controlerIO.getOutputFilename( FILENAME_MODESTATS ) ;
096                this.createPNG = controlerConfigGroup.isCreateGraphs();
097                this.modeOut = IOUtils.getBufferedWriter(this.modeFileName + ".txt");
098                try {
099                        this.modeOut.write("Iteration");
100                        this.modes = new TreeSet<>();
101                        this.modes.addAll(scoreConfig.getAllModes());
102                        for ( String mode : modes ) {
103                                this.modeOut.write("\t" + mode);
104                        }
105                        this.modeOut.write("\n"); ;
106                } catch (IOException e) {
107                        throw new UncheckedIOException(e);
108                }
109                this.tripRouterFactory = tripRouterFactory;
110        }
111
112        @Override
113        public void notifyStartup(final StartupEvent event) {
114                this.minIteration = controlerConfigGroup.getFirstIteration();
115                TripRouter tripRouter = tripRouterFactory.get();
116                this.stageActivities = tripRouter.getStageActivityTypes() ;
117                this.mainModeIdentifier = tripRouter.getMainModeIdentifier() ;
118        }
119
120        @Override
121        public void notifyIterationEnds(final IterationEndsEvent event) {
122                collectModeShareInfo(event) ;
123        }
124
125        private void collectModeShareInfo(final IterationEndsEvent event) {
126                for (Person person : this.population.getPersons().values()) {
127                        Plan plan = person.getSelectedPlan() ;
128                        List<Trip> trips = TripStructureUtils.getTrips(plan, stageActivities) ;
129                        for ( Trip trip : trips ) {
130                                String mode = this.mainModeIdentifier.identifyMainMode( trip.getTripElements() ) ;
131                                // yy as stated elsewhere, the "computer science" mode identification may not be the same as the "transport planning" 
132                                // mode identification.  Maybe revise.  kai, nov'16
133                                
134                                Double cnt = this.modeCnt.get( mode );
135                                if ( cnt==null ) {
136                                        cnt = 0. ;
137                                }
138                                this.modeCnt.put( mode, cnt + 1 ) ;
139                        }
140                }
141
142                double sum = 0 ;
143                for ( Double val : this.modeCnt.values() ) {
144                        sum += val ;
145                }
146                
147                try {
148                        this.modeOut.write( String.valueOf(event.getIteration()) ) ;
149                        for ( String mode : modes ) {
150                                Double cnt = this.modeCnt.get(mode) ;
151                                double share = 0. ;
152                                if ( cnt!=null ) {
153                                        share = cnt/sum;
154                                }
155                                log.info("-- mode share of mode " + mode + " = " + share );
156                                this.modeOut.write( "\t" + share ) ;
157                                
158                                Map<Integer, Double> modeHistory = this.modeHistories.get(mode) ;
159                                if ( modeHistory == null ) {
160                                        modeHistory = new TreeMap<>() ;
161                                        this.modeHistories.put(mode, modeHistory) ;
162                                }
163                                modeHistory.put( event.getIteration(), share ) ;
164                                
165                        }
166                        this.modeOut.write("\n");
167                        this.modeOut.flush();
168                } catch (IOException e) {
169                        e.printStackTrace();
170                }
171
172
173                // yyyy the following does not work!!
174                // Why? The charts seem to be useful (JB, April 2017)
175                if (this.createPNG && event.getIteration() > this.minIteration) {
176                        // create chart when data of more than one iteration is available.
177                        XYLineChart chart = new XYLineChart("Mode Statistics", "iteration", "mode");
178                        for ( Entry<String, Map<Integer, Double>> entry : this.modeHistories.entrySet() ) {
179                                String mode = entry.getKey() ;
180                                Map<Integer, Double> history = entry.getValue() ;
181//                              log.warn( "about to add the following series:" ) ;
182//                              for ( Entry<Integer, Double> item : history.entrySet() ) {
183//                                      log.warn( item.getKey() + " -- " + item.getValue() );
184//                              }
185                                chart.addSeries(mode, history ) ;
186                        }
187                        chart.addMatsimLogo();
188                        chart.saveAsPng(this.modeFileName + ".png", 800, 600);
189                }
190                modeCnt.clear();
191        }
192
193        @Override
194        public void notifyShutdown(final ShutdownEvent controlerShudownEvent) {
195                try {
196                        this.modeOut.close();
197                } catch (IOException e) {
198                        e.printStackTrace();
199                }
200
201        }
202}
