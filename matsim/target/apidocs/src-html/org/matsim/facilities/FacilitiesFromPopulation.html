020package org.matsim.facilities;
021
022import java.util.HashMap;
023import java.util.HashSet;
024import java.util.Map;
025import java.util.Set;
026import org.apache.log4j.Logger;
027import org.matsim.api.core.v01.Coord;
028import org.matsim.api.core.v01.Id;
029import org.matsim.api.core.v01.network.Link;
030import org.matsim.api.core.v01.network.Network;
031import org.matsim.api.core.v01.population.Activity;
032import org.matsim.api.core.v01.population.Person;
033import org.matsim.api.core.v01.population.Plan;
034import org.matsim.api.core.v01.population.PlanElement;
035import org.matsim.api.core.v01.population.Population;
036import org.matsim.core.config.groups.FacilitiesConfigGroup;
037import org.matsim.core.config.groups.PlanCalcScoreConfigGroup;
038import org.matsim.core.config.groups.PlanCalcScoreConfigGroup.ActivityParams;
039import org.matsim.core.network.NetworkUtils;
040
041/**
042 * Generates {@link ActivityFacility}s from the {@link Activity Activities} in a population
043 * and assigns the activity facilities as the activity locations in the population while
044 * removing the old locations (coord and link) from the population.
045 * If an activity already has an ActivityFacility assigned, the ActivityFacility is overwritten.
046 * If an activity only has a coordinate, different behavior is possible, see
047 * {@link #setAssignLinksToFacilitiesIfMissing(boolean, Network)}.
048 *
049 * @author mrieser
050 */
051public class FacilitiesFromPopulation {
052
053        private final static Logger log = Logger.getLogger(FacilitiesFromPopulation.class);
054
055        private final ActivityFacilities facilities;
056        private boolean oneFacilityPerLink = true;
057        private String idPrefix = "";
058        private Network network = null;
059        private boolean removeLinksAndCoordinates = true;
060        private PlanCalcScoreConfigGroup planCalcScoreConfigGroup = null;
061        private boolean addEmptyActivityOptions = false;
062
063        public FacilitiesFromPopulation(final ActivityFacilities facilities) {
064                this.facilities = facilities;
065        }
066
067        public FacilitiesFromPopulation(final ActivityFacilities facilities, final FacilitiesConfigGroup facilityConfigGroup) {
068                this(facilities);
069                this.oneFacilityPerLink = facilityConfigGroup.isOneFacilityPerLink();
070                this.idPrefix = facilityConfigGroup.getIdPrefix();
071                this.removeLinksAndCoordinates = facilityConfigGroup.isRemovingLinksAndCoordinates();
072                this.addEmptyActivityOptions = facilityConfigGroup.isAddEmptyActivityOption();
073        }
074
075        /**
076         * Sets whether all activities on a link should be collected within one ActivityFacility.
077         * Default is <code>true</code>. If set to <code>false</code>, for each coordinate
078         * found in the population's activities a separate ActivityFacility will be created.
079         *
080         * @param oneFacilityPerLink
081         */
082        public void setOneFacilityPerLink(final boolean oneFacilityPerLink) {
083                this.oneFacilityPerLink = oneFacilityPerLink;
084        }
085
086        public void setIdPrefix(final String prefix) {
087                this.idPrefix = prefix;
088        }
089
090        /**
091         * In the case that a facility has no link assigned, the ActivityFacility can be assigned to the closest link.
092         * If there should be only one ActivityFacility per link (see {@link #setOneFacilityPerLink(boolean)}),
093         * and if no link-assignment should be done, then a new ActivityFacility will be created at that coordinate
094         * and the facility will not be assigned to a link, essentially breaking the contract of
095         * {@link #setOneFacilityPerLink(boolean)}.
096         *
097         * @param doAssignment
098         * @param network
099         */
100        public void setAssignLinksToFacilitiesIfMissing(final boolean doAssignment, final Network network) {
101                if (doAssignment && network == null) {
102                        throw new IllegalArgumentException("Network cannot be null if assignment should be done.");
103                }
104                this.network = doAssignment ? network : null;
105        }
106
107        /**
108         * If set to <code>true</code> (which is the default), the link and coordinate attributes
109         * are nulled in the activities, as this information is now available via the facility.
110         *
111         * @param doRemoval
112         */
113        public void setRemoveLinksAndCoordinates(final boolean doRemoval) {
114                this.removeLinksAndCoordinates = doRemoval;
115        }
116
117        public void assignOpeningTimes(final boolean doAssignment, final PlanCalcScoreConfigGroup calcScoreConfigGroup) {
118                if (doAssignment && calcScoreConfigGroup == null) {
119                        throw new IllegalArgumentException("Config must not be null if opening times should be assigned.");
120                }
121                this.planCalcScoreConfigGroup = doAssignment ? calcScoreConfigGroup : null;
122        }
123
124        public void run(final Population population) {
125                handleActivities(population);
126                if (this.planCalcScoreConfigGroup != null ) {
127                        if (this.addEmptyActivityOptions) {
128                                this.assignOpeningTimes();
129                        } else{
130                                log.error("Cannot assign opening times to activity facilities because switch to add empty activity option to activity facilities is set to false.");
131                        }
132                }
133        }
134
135        private void handleActivities(final Population population) {
136                int idxCounter = 0;
137                ActivityFacilitiesFactory factory = this.facilities.getFactory();
138                Map<Id<Link>, ActivityFacility> facilitiesPerLinkId = new HashMap<>();
139                Map<Coord, ActivityFacility> facilitiesPerCoordinate = new HashMap<>();
140
141                for (Person person : population.getPersons().values()) {
142                        for (Plan plan : person.getPlans()) {
143                                for (PlanElement pe : plan.getPlanElements()) {
144                                        if (pe instanceof Activity) {
145                                                Activity a = (Activity) pe;
146
147                                                Coord c = a.getCoord();
148                                                Id<Link> linkId = a.getLinkId();
149                                                ActivityFacility facility = null;
150
151                                                if (linkId == null && this.network != null) {
152                                                        linkId = NetworkUtils.getNearestLinkExactly(this.network, c).getId();
153                                                }
154
155                                                if (this.oneFacilityPerLink && linkId != null) {
156                                                        facility = facilitiesPerLinkId.get(linkId);
157                                                        if (facility == null) {
158                                                                facility = factory.createActivityFacility(Id.create(this.idPrefix + linkId.toString(), ActivityFacility.class), c, linkId);
159                                                                this.facilities.addActivityFacility(facility);
160                                                                facilitiesPerLinkId.put(linkId, facility);
161                                                        }
162                                                } else {
163                                                        facility = facilitiesPerCoordinate.get(c);
164                                                        if (facility == null) {
165                                                                facility = factory.createActivityFacility(Id.create(this.idPrefix + idxCounter++, ActivityFacility.class), c, linkId);
166                                                                this.facilities.addActivityFacility(facility);
167                                                                facilitiesPerCoordinate.put(c, facility);
168                                                        }
169                                                }
170
171                                                if (this.addEmptyActivityOptions) {
172                                                        String actType = a.getType();
173                                                        ActivityOption option = facility.getActivityOptions().get(actType);
174                                                        if (option == null) {
175                                                                option = factory.createActivityOption(actType);
176                                                                facility.addActivityOption(option);
177                                                        }
178                                                }
179
180                                                a.setFacilityId(facility.getId());
181                                                if (this.removeLinksAndCoordinates) {
182                                                        a.setLinkId(null);
183                                                        a.setCoord(null);
184                                                }
185                                        }
186                                }
187                        }
188                }
189        }
190
191        private void assignOpeningTimes() {
192                Set<String> missingActTypes = new HashSet<>();
193                for (ActivityFacility af : this.facilities.getFacilities().values()) {
194                        for (ActivityOption ao : af.getActivityOptions().values()) {
195                                String actType = ao.getType();
196                                ActivityParams params = this.planCalcScoreConfigGroup.getActivityParams(actType);
197                                if (params == null) {
198                                        if (missingActTypes.add(actType)) {
199                                                log.error("No information for activity type " + actType + " found in given configuration.");
200                                        }
201                                } else {
202                                        ao.addOpeningTime(new OpeningTimeImpl(params.getOpeningTime(), params.getClosingTime()));
203                                }
204                        }
205                }
206        }
207
208        public void setAddEmptyActivityOptions(boolean addEmptyActivityOptions) {
209                this.addEmptyActivityOptions = addEmptyActivityOptions;
210        }
211}
