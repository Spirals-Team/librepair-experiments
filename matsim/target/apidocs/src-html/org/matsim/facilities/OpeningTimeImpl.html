021package org.matsim.facilities;
022
023import org.matsim.core.utils.misc.Time;
024
025public class OpeningTimeImpl implements OpeningTime {
026
027        public static enum TimeRelation { 
028                START_AND_END_BEFORE, 
029                START_AND_END_AFTER, 
030                START_AND_END_WITHIN, 
031                START_BEFORE_END_WITHIN,
032                START_WITHIN_END_AFTER,
033                START_BEFORE_END_AFTER
034        }
035        
036        private double startTime;
037        private double endTime;
038        
039        public OpeningTimeImpl(final double startTime, final double endTime) {
040                this.startTime = startTime;
041                this.endTime = endTime;
042                this.validateTimes();
043        }
044
045        //////////////////////////////////////////////////////////////////////
046        //
047        // interface implementation
048        //
049        //////////////////////////////////////////////////////////////////////
050
051        @Override
052        public int compareTo(OpeningTime other) {
053                // two functionalities in one:
054                // 1. the earlier start_time comes before the other. If they're the same,
055                //    the end times decides which comes first
056                // 2. the meaning of the return value. See the ASCII figures for that.
057                if (this.startTime > other.getEndTime()) {         // this:       |-----|
058                        return -6;                                       // other: |--|
059                }
060                else if (this.startTime == other.getEndTime()) {   // this:       |-----|
061                        return -5;                                       // other: |----|
062                }
063                else if (this.startTime > other.getStartTime()) {
064                        if (this.endTime > other.getEndTime()) {         // this:       |-----|
065                                return -4;                                     // other: |--------|
066                        }
067                        else if (this.endTime == other.getEndTime()) {   // this:       |-----|
068                                return -3;                                     // other: |----------|
069                        }
070                        else {                                        // this:       |-----|
071                                return -2;                                  // other: |---------------|
072                        }
073                }
074                else if (this.startTime == other.getStartTime()) {
075                        if (this.endTime > other.getEndTime()) {         // this:       |-----|
076                                return -1;                                     // other:      |---|
077                        }
078                        else if (this.endTime == other.getEndTime()) {   // this:       |-----|
079                                return 0;                                      // other:      |-----|
080                        }
081                        else {                                           // this:       |-----|
082                                return 3;                                      // other:      |----------|
083                        }
084                }
085                else if (this.endTime > other.getEndTime()) {      // this:       |-----|
086                        return 2;                                        // other:        |-|
087                }
088                else if (this.endTime == other.getEndTime()) {     // this:       |-----|
089                        return 1;                                     // other:        |---|
090                }
091                else if (this.endTime > other.getStartTime()) {    // this:       |-----|
092                        return 4;                                     // other:        |--------|
093                }
094                else if (this.endTime == other.getStartTime()) {   // this:       |-----|
095                        return 5;                                     // other:            |----|
096                }
097                else {                                          // this:       |-----|
098                        return 6;                                     // other:              |--|
099                }
100        }
101
102        //////////////////////////////////////////////////////////////////////
103        // query methods
104        //////////////////////////////////////////////////////////////////////
105
106        @Override
107        public final boolean equals(final Object o) {
108                if (o instanceof OpeningTimeImpl) {
109                        OpeningTimeImpl other = (OpeningTimeImpl)o;
110                        if ((other.startTime == this.startTime) && (other.endTime == this.endTime)) {
111                                return true;
112                        }
113                }
114                return false;
115        }
116
117        @Override
118        public final int hashCode() {
119                /* equals() checks day, startTime and endTime, so we should include those into hashCode as well */
120                return (Double.valueOf(this.startTime).hashCode()
121                                + Double.valueOf(this.endTime).hashCode());
122        }
123
124        private final void validateTimes() {
125                if (this.startTime != Time.UNDEFINED_TIME && this.endTime != Time.UNDEFINED_TIME && this.startTime >= this.endTime) {
126                        throw new RuntimeException(this + "[startTime=" + this.startTime + " >= endTime=" + this.endTime + " not allowed]");
127                }
128        }
129
130        //////////////////////////////////////////////////////////////////////
131        // set methods
132        //////////////////////////////////////////////////////////////////////
133
134        @Override
135        public final void setStartTime(final double start_time) {
136                this.startTime = start_time;
137                this.validateTimes();
138        }
139
140        @Override
141        public final void setEndTime(final double end_time) {
142                this.endTime = end_time;
143                this.validateTimes();
144        }
145        
146        //////////////////////////////////////////////////////////////////////
147        // get methods
148        //////////////////////////////////////////////////////////////////////
149
150        @Override
151        public final double getStartTime() {
152                return this.startTime;
153        }
154
155        @Override
156        public final double getEndTime() {
157                return this.endTime;
158        }
159
160        //////////////////////////////////////////////////////////////////////
161        // print methods
162        //////////////////////////////////////////////////////////////////////
163
164        @Override
165        public final String toString() {
166                return "[startTime=" + Time.writeTime(this.startTime) + "]" +
167                                "[endTime=" + Time.writeTime(this.endTime) + "]";
168        }
169
170}
