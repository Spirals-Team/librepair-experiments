021package org.matsim.facilities;
022
023import java.util.Collection;
024import java.util.Iterator;
025import java.util.SortedSet;
026import java.util.TreeSet;
027
028public class ActivityOptionImpl implements ActivityOption {
029
030        private final String type;
031        private Double capacity = Double.valueOf(Integer.MAX_VALUE);
032        private SortedSet<OpeningTime> openingTimes = new TreeSet<>();
033
034        public ActivityOptionImpl(final String type) {
035                this.type = type;
036        }
037
038        public void clearOpeningTimes() {
039                this.openingTimes.clear();
040        }
041        
042        @Override
043        public void addOpeningTime(OpeningTime opentime) {
044                if (openingTimes.isEmpty()) {
045                        openingTimes.add(opentime);
046                        return;
047                }
048                TreeSet<OpeningTime> new_o_set = new TreeSet<>();
049                Iterator<OpeningTime> o_it = openingTimes.iterator();
050                while (o_it.hasNext()) {
051                        OpeningTime o = o_it.next();
052                        int merge_type = o.compareTo(opentime); // see Opentime for the meaning
053                        if ((merge_type == -6) || (merge_type == 6)) {
054                                // complete disjoint
055                                new_o_set.add(o);
056                                new_o_set.add(opentime);
057                        }
058                        else if ((merge_type >= -1) && (merge_type <= 2)) {
059                                // opentime is subset of o
060                                new_o_set.add(o);
061                        }
062                        else if ((merge_type == -3) || (merge_type == -2) || (merge_type == 3)) {
063                                // o is subset of opentime
064                                new_o_set.add(opentime);
065                        }
066                        else { // union of the two opentimes
067                                if ((merge_type == -5) || (merge_type == -4)) {
068                                        // start_time of opentime and endtime of o
069                                        opentime.setEndTime(o.getEndTime());
070                                        new_o_set.add(opentime);
071                                }
072                                else if ((merge_type == 4) || (merge_type == 5)) {
073                                        // start_time of o and endtime of opentime
074                                        opentime.setStartTime(o.getStartTime());
075                                        new_o_set.add(opentime);
076                                }
077                                else {
078                                        throw new RuntimeException("[Something is wrong]");
079                                }
080                        }
081                }
082                this.openingTimes.clear();
083                this.openingTimes.addAll(new_o_set);
084        }
085        
086        public void setOpeningTimes(Collection<OpeningTime> times) {
087                this.clearOpeningTimes();
088                for (OpeningTime t : times) {
089                        this.addOpeningTime(t);
090                }
091        }
092
093        @Override
094        public final void setCapacity(double capacity) {
095                if (capacity < 0) {
096                        throw new NumberFormatException("A capacity of an activity must be >= 0.");
097                }
098                this.capacity = capacity;
099        }
100
101        @Override
102        public final String getType() {
103                return this.type;
104        }
105
106        @Override
107        public final double getCapacity() {
108                return this.capacity;
109        }
110
111        @Override
112        public final SortedSet<OpeningTime> getOpeningTimes() {
113                return this.openingTimes;
114        }
115
116        @Override
117        public final String toString() {
118                return "[type=" + this.type + "]" +
119                                "[capacity=" + this.capacity + "]" +
120                                "[nof_opentimes=" + this.getOpeningTimes().size() + "]";
121        }
122
123}
