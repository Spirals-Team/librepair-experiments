021package org.matsim.facilities.algorithms;
022
023import java.util.Iterator;
024
025import org.matsim.api.core.v01.Coord;
026import org.matsim.facilities.ActivityFacilities;
027import org.matsim.facilities.ActivityFacility;
028import org.matsim.facilities.ActivityOption;
029import org.matsim.facilities.ActivityOptionImpl;
030
031public class FacilitiesSummary {
032
033        public FacilitiesSummary() {
034                super();
035        }
036
037        public void run(ActivityFacilities facilities) {
038                System.out.println("    running " + this.getClass().getName() + " algorithm...");
039                int f_cnt = 0;
040                int act_cnt = 0;
041//              Coord min_coord = new Coord(Double.POSITIVE_INFINITY, Double.POSITIVE_INFINITY);
042//              Coord max_coord = new Coord(Double.NEGATIVE_INFINITY, Double.NEGATIVE_INFINITY);
043                double min_coordX = Double.POSITIVE_INFINITY;
044                double min_coordY = Double.POSITIVE_INFINITY;
045                double max_coordX = Double.NEGATIVE_INFINITY;
046                double max_coordY = Double.NEGATIVE_INFINITY;
047                //            home,work,education,shop,leisure
048                int caps[] = {0   ,   0,        0,   0,      0};
049                int unlimit_cap_cnt = 0;
050                for (ActivityFacility f : facilities.getFacilities().values()) {
051                        f_cnt++;
052                        if (f.getCoord().getX() > max_coordX) { max_coordX=f.getCoord().getX(); }
053                        if (f.getCoord().getY() > max_coordY) { max_coordY=f.getCoord().getY(); }
054                        if (f.getCoord().getX() < min_coordX) { min_coordX=f.getCoord().getX(); }
055                        if (f.getCoord().getY() < min_coordY) { min_coordY=f.getCoord().getY(); }
056
057                        Iterator<? extends ActivityOption> a_it = f.getActivityOptions().values().iterator();
058                        while (a_it.hasNext()) {
059                                ActivityOptionImpl a = (ActivityOptionImpl) a_it.next();
060                                act_cnt++;
061                                if (a.getCapacity() != Integer.MAX_VALUE) {
062                                        if (a.getType().equals("home")) {
063                                                caps[0] += a.getCapacity();
064                                        }
065                                        else if (a.getType().equals("work")) {
066                                                caps[1] += a.getCapacity();
067                                        }
068                                        else if (a.getType().equals("education")) {
069                                                caps[2] += a.getCapacity();
070                                        }
071                                        else if (a.getType().equals("shop")) {
072                                                caps[3] += a.getCapacity();
073                                        }
074                                        else if (a.getType().equals("leisure")) {
075                                                caps[4] += a.getCapacity();
076                                        }
077                                        else {
078                                                throw new RuntimeException("ERROR: in " + this.getClass().getName() +
079                                                                                                                         " in run(Facilities facilities):" + 
080                                                                                                                         " do not know type = " + a.getType());
081                                        }
082                                }
083                                else {
084                                        unlimit_cap_cnt++;
085                                }
086                        }
087                }
088                System.out.println("      Number of Facilities:             " + f_cnt);
089                System.out.println("      Number of Activities:             " + act_cnt);
090                System.out.println("      Min Coord:                        " + min_coordX + " " + min_coordY );
091                System.out.println("      Max Coord:                        " + max_coordX + " " + max_coordY ) ;
092                System.out.println("      total home cap:                   " + caps[0]);
093                System.out.println("      total work cap:                   " + caps[1]);
094                System.out.println("      total education cap:              " + caps[2]);
095                System.out.println("      total shop cap:                   " + caps[3]);
096                System.out.println("      total leisure cap:                " + caps[4]);
097                System.out.println("      total acts with unlimited cap:    " + unlimit_cap_cnt);
098
099                System.out.println("    done.");
100        }
101}
