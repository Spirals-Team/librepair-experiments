021package org.matsim.facilities.algorithms;
022
023import java.util.Iterator;
024
025import org.matsim.core.utils.misc.Time;
026import org.matsim.facilities.ActivityFacilitiesImpl;
027import org.matsim.facilities.ActivityFacility;
028import org.matsim.facilities.ActivityOption;
029import org.matsim.facilities.ActivityOptionImpl;
030import org.matsim.facilities.OpeningTimeImpl;
031
032public class FacilitiesDefineCapAndOpentime {
033
034        private final double TIME_0000 = Time.parseTime("00:00");
035        private final double TIME_0800 = Time.parseTime("08:00");
036        private final double TIME_0830 = Time.parseTime("08:30");
037        private final double TIME_0900 = Time.parseTime("09:00");
038        private final double TIME_1200 = Time.parseTime("12:00");
039        private final double TIME_1300 = Time.parseTime("13:00");
040        private final double TIME_1330 = Time.parseTime("13:30");
041        private final double TIME_1700 = Time.parseTime("17:00");
042        private final double TIME_1800 = Time.parseTime("18:00");
043        private final double TIME_2000 = Time.parseTime("20:00");
044        private final double TIME_2400 = Time.parseTime("24:00");
045        
046        private final int nof_persons;
047
048        public FacilitiesDefineCapAndOpentime(final int nof_persons) {
049                super();
050                this.nof_persons = nof_persons;
051                if (nof_persons <= 0) {
052                        throw new RuntimeException("[nof_persons=" + nof_persons + " not allowed]");
053                }
054        }
055
056        public void run(ActivityFacilitiesImpl facilities) {
057                System.out.println("    running " + this.getClass().getName() + " algorithm...");
058
059                int home_cnt = 0;
060                int work_cnt = 0;
061                int educ_cnt = 0;
062                int shop_cnt = 0;
063                int leis_cnt = 0;
064
065                for (ActivityFacility f : facilities.getFacilities().values()) {
066                        Iterator<String> at_it = f.getActivityOptions().keySet().iterator();
067                        while (at_it.hasNext()) {
068                                String at = at_it.next();
069                                if (at.equals("home")) { home_cnt++; }
070                                else if (at.equals("work")) { work_cnt++; }
071                                else if (at.equals("education")) { educ_cnt++; }
072                                else if (at.equals("shop")) { shop_cnt++; }
073                                else if (at.equals("leisure")) { leis_cnt++; }
074                                else { throw new RuntimeException("[something is wrong!]"); }
075                        }
076                }
077                System.out.println("      home_cnt = " + home_cnt);
078                System.out.println("      work_cnt = " + work_cnt);
079                System.out.println("      educ_cnt = " + educ_cnt);
080                System.out.println("      shop_cnt = " + shop_cnt);
081                System.out.println("      leis_cnt = " + leis_cnt);
082
083                for (ActivityFacility f : facilities.getFacilities().values()) {
084                        Iterator<? extends ActivityOption> a_it = f.getActivityOptions().values().iterator();
085                        while (a_it.hasNext()) {
086                                ActivityOptionImpl a = (ActivityOptionImpl) a_it.next();
087                                if (a.getType().equals("home")) {
088                                        a.setCapacity(this.nof_persons/home_cnt);
089                                        a.addOpeningTime(new OpeningTimeImpl(TIME_0000, TIME_2400));
090                                }
091                                else if (a.getType().equals("work")) {
092                                        a.setCapacity(this.nof_persons/work_cnt);
093                                        a.addOpeningTime(new OpeningTimeImpl(TIME_0800, TIME_1800));
094                                }
095                                else if (a.getType().equals("education")) {
096                                        a.setCapacity(this.nof_persons/educ_cnt);
097                                        a.addOpeningTime(new OpeningTimeImpl(TIME_0800, TIME_1200));
098                                        a.addOpeningTime(new OpeningTimeImpl(TIME_1300, TIME_1700));
099                                }
100                                else if (a.getType().equals("shop")) {
101                                        a.setCapacity(this.nof_persons/shop_cnt);
102                                        a.addOpeningTime(new OpeningTimeImpl(TIME_0830, TIME_1200));
103                                        a.addOpeningTime(new OpeningTimeImpl(TIME_1330, TIME_2000));
104                                        a.addOpeningTime(new OpeningTimeImpl(TIME_0900, TIME_1700));
105                                }
106                                else if (a.getType().equals("leisure")) {
107                                        a.setCapacity(this.nof_persons/leis_cnt);
108                                        a.addOpeningTime(new OpeningTimeImpl(TIME_1700, TIME_2400));
109                                        a.addOpeningTime(new OpeningTimeImpl(TIME_2000, TIME_2400));
110                                }
111                                else { throw new RuntimeException("[something is wrong!]"); }
112                        }
113                }
114
115                System.out.println("    done.");
116        }
117}
