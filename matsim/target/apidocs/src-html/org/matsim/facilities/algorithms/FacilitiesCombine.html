021package org.matsim.facilities.algorithms;
022
023import java.util.Iterator;
024import java.util.Map;
025import java.util.SortedSet;
026import java.util.TreeMap;
027
028import org.matsim.api.core.v01.Id;
029import org.matsim.facilities.ActivityFacilities;
030import org.matsim.facilities.ActivityFacility;
031import org.matsim.facilities.ActivityFacilityImpl;
032import org.matsim.facilities.ActivityOption;
033import org.matsim.facilities.ActivityOptionImpl;
034import org.matsim.facilities.OpeningTime;
035
036public class FacilitiesCombine {
037
038        //////////////////////////////////////////////////////////////////////
039        // constructors
040        //////////////////////////////////////////////////////////////////////
041
042        public FacilitiesCombine() {
043                super();
044        }
045
046        //////////////////////////////////////////////////////////////////////
047        // private methods
048        //////////////////////////////////////////////////////////////////////
049
050        private final void combine(ActivityFacility f,ActivityFacilityImpl f2) {
051                System.out.println("      Combining f_id=" + f.getId() + " into f2_id=" + f2.getId());
052                Iterator<? extends ActivityOption> a_it = f.getActivityOptions().values().iterator();
053                while (a_it.hasNext()) {
054                        ActivityOptionImpl a = (ActivityOptionImpl) a_it.next();
055                        if (f2.getActivityOptions().get(a.getType()) == null) {
056                                ActivityOptionImpl a2 = f2.createAndAddActivityOption(a.getType());
057                                a2.setCapacity(a.getCapacity());
058                        }
059                        else {
060                                ActivityOptionImpl a2 = (ActivityOptionImpl) f2.getActivityOptions().get(a.getType());
061                                double cap2 = a2.getCapacity();
062                                double cap = a.getCapacity();
063                                if ((cap < Integer.MAX_VALUE) && (cap2 < Integer.MAX_VALUE)) { a2.setCapacity(cap + cap2); }
064                                else { a2.setCapacity(Integer.MAX_VALUE); }
065                        }
066                        SortedSet<OpeningTime> ts = a.getOpeningTimes();
067                        Iterator<OpeningTime> o_it = ts.iterator();
068                        while (o_it.hasNext()) {
069                                OpeningTime o = o_it.next();
070                                f2.getActivityOptions().get(a.getType()).addOpeningTime(o);
071                        }
072                }
073                if (Integer.parseInt(f2.getId().toString()) > Integer.parseInt(f.getId().toString())) {
074                        System.out.println("      => assigning f_id="+f.getId()+" to f2.");
075                        throw new RuntimeException("Can't set ids anymore.");
076                }
077        }
078        
079        //////////////////////////////////////////////////////////////////////
080        // run method
081        //////////////////////////////////////////////////////////////////////
082
083        public void run(final ActivityFacilities facilities) {
084                System.out.println("    running " + this.getClass().getName() + " algorithm...");
085                System.out.println("      # facilities = " + facilities.getFacilities().size());
086
087                // TreeMap<XCOORD,TreeMap<YCOORD,FACILITY>>
088                TreeMap<Double,TreeMap<Double,ActivityFacilityImpl>> facs = new TreeMap<Double, TreeMap<Double,ActivityFacilityImpl>>();
089
090                for (ActivityFacility f : facilities.getFacilities().values()) {
091                        Double x = f.getCoord().getX();
092                        Double y = f.getCoord().getY();
093                        if (facs.containsKey(x)) { // same x coord
094                                TreeMap<Double,ActivityFacilityImpl> tree = facs.get(x);
095                                if (tree.containsKey(y)) { // and same y coord
096                                        ActivityFacilityImpl f2 = tree.get(y);
097                                        this.combine(f,f2);
098                                }
099                                else { // not the same y coord
100                                        tree.put(y,(ActivityFacilityImpl) f);
101                                }
102                        }
103                        else { // not the same x coord
104                                TreeMap<Double,ActivityFacilityImpl> tree = new TreeMap<Double, ActivityFacilityImpl>();
105                                tree.put(y,(ActivityFacilityImpl) f);
106                                facs.put(x,tree);
107                        }
108                }
109
110                Map<Id<ActivityFacility>, ? extends ActivityFacility> fs = facilities.getFacilities();
111                fs.clear();
112
113                Iterator<TreeMap<Double,ActivityFacilityImpl>> t_it = facs.values().iterator();
114                while (t_it.hasNext()) {
115                        TreeMap<Double,ActivityFacilityImpl> t = t_it.next();
116                        Iterator<ActivityFacilityImpl> ff_it = t.values().iterator();
117                        while (ff_it.hasNext()) {
118                                ActivityFacilityImpl ff = ff_it.next();
119                                facilities.addActivityFacility(ff);
120                        }
121                }
122
123                System.out.println("      # facilities = " + facilities.getFacilities().size());
124                System.out.println("    done.");
125        }
126}
