021package org.matsim.facilities;
022
023import org.apache.log4j.Logger;
024import org.matsim.api.core.v01.Coord;
025import org.matsim.api.core.v01.Id;
026import org.matsim.api.core.v01.Scenario;
027import org.matsim.api.core.v01.network.Link;
028import org.matsim.core.utils.geometry.CoordinateTransformation;
029import org.matsim.core.utils.geometry.transformations.IdentityTransformation;
030import org.matsim.core.utils.io.MatsimXmlParser;
031import org.matsim.core.utils.misc.Time;
032import org.matsim.utils.objectattributes.AttributeConverter;
033import org.matsim.utils.objectattributes.attributable.AttributesXmlReaderDelegate;
034import org.xml.sax.Attributes;
035
036import java.util.Map;
037import java.util.Stack;
038
039/**
040 * A reader for facilities-files of MATSim according to <code>facilities_v1.dtd</code>.
041 *
042 * @author mrieser
043 * @author balmermi
044 */
045public class FacilitiesReaderMatsimV1 extends MatsimXmlParser {
046
047    private final static String FACILITIES = "facilities";
048    private final static String FACILITY = "facility";
049    private final static String ACTIVITY = "activity";
050    private final static String CAPACITY = "capacity";
051    private final static String OPENTIME = "opentime";
052    private static final String ATTRIBUTES = "attributes";
053    private static final String ATTRIBUTE = "attribute";
054
055    private final ActivityFacilities facilities;
056    private final ActivityFacilitiesFactory factory;
057    private final AttributesXmlReaderDelegate attributesReader;
058    private ActivityFacility currfacility = null;
059    private ActivityOption curractivity = null;
060    private org.matsim.utils.objectattributes.attributable.Attributes currAttributes;
061
062    private final CoordinateTransformation coordinateTransformation;
063
064    public FacilitiesReaderMatsimV1(final Scenario scenario) {
065        this(new IdentityTransformation(), scenario);
066    }
067
068    public FacilitiesReaderMatsimV1(
069            final CoordinateTransformation coordinateTransformation,
070            final Scenario scenario) {
071        this.coordinateTransformation = coordinateTransformation;
072        this.facilities = scenario.getActivityFacilities();
073        this.factory = this.facilities.getFactory();
074        this.attributesReader = new AttributesXmlReaderDelegate();
075        this.currAttributes = null;
076    }
077
078    public void putAttributeConverter(Class<?> clazz, AttributeConverter<?> converter) {
079        this.attributesReader.putAttributeConverter(clazz, converter);
080    }
081
082    public void putAttributeConverters(Map<Class<?>, AttributeConverter<?>> converters) {
083        this.attributesReader.putAttributeConverters(converters);
084    }
085
086    @Override
087    public void startTag(final String name, final org.xml.sax.Attributes atts, final Stack<String> context) {
088        if (FACILITIES.equals(name)) {
089            startFacilities(atts);
090        } else if (FACILITY.equals(name)) {
091            startFacility(atts);
092        } else if (ACTIVITY.equals(name)) {
093            startActivity(atts);
094        } else if (CAPACITY.equals(name)) {
095            startCapacity(atts);
096        } else if (OPENTIME.equals(name)) {
097            startOpentime(atts);
098        } else if (ATTRIBUTE.equals(name)) {
099            this.attributesReader.startTag(name, atts, context, this.currAttributes);
100        } else if (ATTRIBUTES.equals(name)) {
101            currAttributes = this.currfacility.getAttributes();
102            attributesReader.startTag(name, atts, context, currAttributes);
103        }
104    }
105
106    @Override
107    public void endTag(final String name, final String content, final Stack<String> context) {
108        if (FACILITY.equals(name)) {
109            this.currfacility = null;
110        } else if (ACTIVITY.equals(name)) {
111            this.curractivity = null;
112        } else if (ATTRIBUTES.equalsIgnoreCase(name)) {
113            this.currAttributes = null;
114        } else if (ATTRIBUTE.equalsIgnoreCase(name)) {
115            this.attributesReader.endTag(name, content, context);
116        }
117    }
118
119    private void startFacilities(final Attributes atts) {
120        this.facilities.setName(atts.getValue("name"));
121        if (atts.getValue("aggregation_layer") != null) {
122            Logger.getLogger(FacilitiesReaderMatsimV1.class).warn("aggregation_layer is deprecated.");
123        }
124    }
125
126    private void startFacility(final Attributes atts) {
127        this.currfacility =
128                this.factory.createActivityFacility(
129                        Id.create(atts.getValue("id"), ActivityFacility.class),
130                        coordinateTransformation.transform(
131                                new Coord(
132                                        Double.parseDouble(atts.getValue("x")),
133                                        Double.parseDouble(atts.getValue("y")))));
134        this.facilities.addActivityFacility(this.currfacility);
135        String value = atts.getValue("linkId");
136        if (value != null) {
137            ((ActivityFacilityImpl) this.currfacility).setLinkId(Id.create(value, Link.class));
138        }
139        ((ActivityFacilityImpl) this.currfacility).setDesc(atts.getValue("desc"));
140    }
141
142    private void startActivity(final Attributes atts) {
143        this.curractivity = this.factory.createActivityOption(atts.getValue("type"));
144        this.currfacility.addActivityOption(this.curractivity);
145    }
146
147    private void startCapacity(final Attributes atts) {
148        double cap = Double.parseDouble(atts.getValue("value"));
149        this.curractivity.setCapacity(cap);
150    }
151
152    private void startOpentime(final Attributes atts) {
153        this.curractivity.addOpeningTime(new OpeningTimeImpl(Time.parseTime(atts.getValue("start_time")), Time.parseTime(atts.getValue("end_time"))));
154    }
155
156
157}
