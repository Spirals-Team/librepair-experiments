084        public final ActivityFacilityImpl createAndAddFacility(final Id<ActivityFacility> id, final Coord center, final Id<Link> linkId) {
085                if (this.facilities.containsKey(id)) {
086                        throw new IllegalArgumentException("Facility with id=" + id + " already exists.");
087                }
088                ActivityFacilityImpl f = new ActivityFacilityImpl(id, center, linkId);
089                this.facilities.put(f.getId(),f);
090
091                // show counter
092                if (this.facilities.size() % this.nextMsg == 0) {
093                        this.nextMsg *= 2;
094                        log.info("    facility # " + this.facilities.size() );
095                }
096
097                return f;
098        }

110        @Override
111        public final TreeMap<Id<ActivityFacility>, ActivityFacility> getFacilitiesForActivityType(final String act_type) {
112                TreeMap<Id<ActivityFacility>, ActivityFacility> facs = new TreeMap<>();
113                Iterator<ActivityFacility> iter = this.facilities.values().iterator();
114                while (iter.hasNext()){
115                        ActivityFacility f = iter.next();
116                        Map<String, ? extends ActivityOption> a = f.getActivityOptions();
117                        if(a.containsKey(act_type)){
118                                facs.put(f.getId(),f);
119                        }
120                }
121                return facs;
122        }

149        @Override
150        public String toString() {
151                StringBuilder stb = new StringBuilder() ;
152                stb.append( super.toString() + "\n" ) ;
153                stb.append( "[number of facilities=" + this.facilities.size() + "]\n" ) ;
154                for ( Entry<Id<ActivityFacility>,? extends ActivityFacility> entry : this.facilities.entrySet() ) {
155                        final ActivityFacility fac = entry.getValue();
156                        stb.append( "[key=" + entry.getKey().toString() + "; value=" + fac.toString() + "]\n" ) ;
157                }
158                
159                return stb.toString() ;
160        }

162        synchronized private void buildQuadTree() {
163                /* the method must be synchronized to ensure we only build one quadTree
164                 * in case that multiple threads call a method that requires the quadTree.
165                 */
166                if (this.facilitiesQuadTree != null) {
167                        return;
168                }
169                double startTime = System.currentTimeMillis();
170                double minx = Double.POSITIVE_INFINITY;
171                double miny = Double.POSITIVE_INFINITY;
172                double maxx = Double.NEGATIVE_INFINITY;
173                double maxy = Double.NEGATIVE_INFINITY;
174                for ( ActivityFacility n : this.facilities.values()) {
175                        if (n.getCoord().getX() < minx) { minx = n.getCoord().getX(); }
176                        if (n.getCoord().getY() < miny) { miny = n.getCoord().getY(); }
177                        if (n.getCoord().getX() > maxx) { maxx = n.getCoord().getX(); }
178                        if (n.getCoord().getY() > maxy) { maxy = n.getCoord().getY(); }
179                }
180                minx -= 1.0;
181                miny -= 1.0;
182                maxx += 1.0;
183                maxy += 1.0;
184                // yy the above four lines are problematic if the coordinate values are much smaller than one. kai, oct'15
185
186                log.info("building QuadTree for nodes: xrange(" + minx + "," + maxx + "); yrange(" + miny + "," + maxy + ")");
187                QuadTree<ActivityFacility> quadTree = new QuadTree<>(minx, miny, maxx, maxy);
188                for (ActivityFacility n : this.facilities.values()) {
189                        quadTree.put(n.getCoord().getX(), n.getCoord().getY(), n);
190                }
191                /* assign the quadTree at the very end, when it is complete.
192                 * otherwise, other threads may already start working on an incomplete quadtree
193                 */
194                this.facilitiesQuadTree = quadTree;
195                log.info("Building QuadTree took " + ((System.currentTimeMillis() - startTime) / 1000.0) + " seconds.");
196        }

199        /**
200         * finds the node nearest to <code>coord</code>
201         *
202         * @param coord the coordinate to which the closest node should be found
203         * @return the closest node found, null if none
204         */
205        @Override public ActivityFacility getNearestFacility(final Coord coord) {
206                if (this.facilitiesQuadTree == null) { buildQuadTree(); }
207                return this.facilitiesQuadTree.getClosest(coord.getX(), coord.getY());
208        }
209
210        /**
211         * finds the nodes within distance to <code>coord</code>
212         *
213         * @param coord the coordinate around which nodes should be located
214         * @param distance the maximum distance a node can have to <code>coord</code> to be found
215         * @return all nodes within distance to <code>coord</code>
216         */
217        @Override public Collection<ActivityFacility> getNearestFacilities(final Coord coord, final double distance) {
218                if (this.facilitiesQuadTree == null) { buildQuadTree(); }
219                return this.facilitiesQuadTree.getDisk(coord.getX(), coord.getY(), distance);
220        }
