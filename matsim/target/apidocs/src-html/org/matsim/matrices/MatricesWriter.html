021package org.matsim.matrices;
022
023import java.io.IOException;
024import java.util.ArrayList;
025import java.util.Iterator;
026
027import org.matsim.core.api.internal.MatsimWriter;
028import org.matsim.core.utils.io.MatsimXmlWriter;
029
030public class MatricesWriter extends MatsimXmlWriter implements MatsimWriter {
031
032        //////////////////////////////////////////////////////////////////////
033        // member variables
034        //////////////////////////////////////////////////////////////////////
035
036        private MatricesWriterHandler handler = null;
037        private final Matrices matrices;
038        private String dtd;
039
040        //////////////////////////////////////////////////////////////////////
041        // constructors
042        //////////////////////////////////////////////////////////////////////
043
044        public MatricesWriter(final Matrices matrices) {
045                super();
046                this.matrices = matrices;
047                // always write out in newest version, currently v1
048                this.dtd = "http://matsim.org/files/dtd/matrices_v1.dtd";
049                this.handler = new MatricesWriterHandlerImplV1();
050        }
051
052        //////////////////////////////////////////////////////////////////////
053        // write methods
054        //////////////////////////////////////////////////////////////////////
055
056        @Override
057        public final void write(final String filename) {
058                try {
059                        openFile(filename);
060                        writeXmlHead();
061                        writeDoctype("matrices", this.dtd);
062                        this.handler.startMatrices(this.matrices, this.writer);
063                        this.handler.writeSeparator(this.writer);
064                        Iterator<Matrix> m_it = this.matrices.getMatrices().values().iterator();
065                        while (m_it.hasNext()) {
066                                Matrix m = m_it.next();
067                                this.handler.startMatrix(m, this.writer);
068                                Iterator<ArrayList<Entry>> eal_it = m.getFromLocations().values().iterator();
069                                while (eal_it.hasNext()) {
070                                        ArrayList<Entry> eal = eal_it.next();
071                                        Iterator<Entry> e_it = eal.iterator();
072                                        while (e_it.hasNext()) {
073                                                Entry e = e_it.next();
074                                                this.handler.startEntry(e, this.writer);
075                                                this.handler.endEntry(this.writer);
076                                        }
077                                }
078                                this.handler.endMatrix(this.writer);
079                                this.handler.writeSeparator(this.writer);
080                                this.writer.flush();
081                        }
082                        this.handler.endMatrices(this.writer);
083                        close();
084                }
085                catch (IOException e) {
086                        throw new RuntimeException(e);
087                }
088        }
089
090        //////////////////////////////////////////////////////////////////////
091        // print methods
092        //////////////////////////////////////////////////////////////////////
093
094        @Override
095        public final String toString() {
096                return super.toString();
097        }
098}
