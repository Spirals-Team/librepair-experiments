021package org.matsim.matrices;
022
023import java.util.Stack;
024
025import org.matsim.core.utils.io.MatsimXmlParser;
026import org.xml.sax.Attributes;
027
028/**
029 * A reader for matrices files of MATSim according to <code>matrices_v1.dtd</code>.
030 *
031 * @author mrieser
032 * @author balmermi
033 */
034public class MatricesReaderMatsimV1<T> extends MatsimXmlParser {
035
036        private final static String MATRICES = "matrices";
037        private final static String MATRIX = "matrix";
038        private final static String ENTRY = "entry";
039
040        private Matrices matrices;
041        private Matrix currMatrix = null;
042
043        public MatricesReaderMatsimV1(final Matrices matrices) {
044                this.matrices = matrices;
045        }
046
047        @Override
048        public void startTag(final String name, final Attributes atts, final Stack<String> context) {
049                if (MATRICES.equals(name)) {
050                        startMatrices(atts);
051                } else if (MATRIX.equals(name)) {
052                        startMatrix(atts);
053                } else if (ENTRY.equals(name)) {
054                        startEntry(atts);
055                } else {
056                        throw new RuntimeException(this + "[tag=" + name + " not known or not supported]");
057                }
058        }
059
060        @Override
061        public void endTag(final String name, final String content, final Stack<String> context) {
062                if (MATRICES.equals(name)) {
063                        this.matrices = null;
064                } else if (MATRIX.equals(name)) {
065                        this.currMatrix = null;
066                }
067        }
068
069        private void startMatrices(final Attributes atts) {
070                this.matrices.setName(atts.getValue("name"));
071        }
072
073        private void startMatrix(final Attributes atts) {
074                this.currMatrix = this.matrices.createMatrix(atts.getValue("id"),
075                                atts.getValue("desc"));
076        }
077
078        private void startEntry(final Attributes  atts) {
079                this.currMatrix.createAndAddEntry(atts.getValue("from_id"), atts.getValue("to_id"), Double.parseDouble(atts.getValue("value")));
080        }
081
082}
