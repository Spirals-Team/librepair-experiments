021package org.matsim.matrices;
022
023import java.util.ArrayList;
024import java.util.List;
025import java.util.Map;
026import java.util.TreeMap;
027
028import org.apache.log4j.Logger;
029
030public final class Matrix {
031
032        //////////////////////////////////////////////////////////////////////
033        // member variables
034        //////////////////////////////////////////////////////////////////////
035
036        private final String id;
037        private String desc = null;
038
039        // double data structure. for fast access via from_location or via to_location
040        private final TreeMap<String,ArrayList<Entry>> fromLocs = new TreeMap<>();
041        private final TreeMap<String,ArrayList<Entry>> toLocs = new TreeMap<>();
042
043        private static final Logger log = Logger.getLogger(Matrix.class);
044
045        private long counter = 0;
046        private long nextMsg = 1;
047
048        //////////////////////////////////////////////////////////////////////
049        // Constructors
050        //////////////////////////////////////////////////////////////////////
051
052        public Matrix(final String id, final String desc) {
053                if (id == null) {
054                        throw new NullPointerException("id must not be null");
055                }
056                this.id = id;
057                this.desc = desc;
058        }
059
060        //////////////////////////////////////////////////////////////////////
061        // create methods
062        //////////////////////////////////////////////////////////////////////
063
064        public final Entry createAndAddEntry(final String fromLocId, final String toLocId, final double value) {
065
066                // create an entry
067                Entry e = new Entry(fromLocId, toLocId, value);
068
069                // add it to the from location data structure
070                if (!this.fromLocs.containsKey(fromLocId)) {
071                        this.fromLocs.put(fromLocId, new ArrayList<Entry>());
072                }
073                ArrayList<Entry> fe = this.fromLocs.get(fromLocId);
074                fe.add(e);
075
076                // add it to the to location data structure
077                if (!this.toLocs.containsKey(toLocId)) {
078                        this.toLocs.put(toLocId, new ArrayList<Entry>());
079                }
080                ArrayList<Entry> te = this.toLocs.get(toLocId);
081                te.add(e);
082
083                // show counter
084                this.counter++;
085                if (this.counter % this.nextMsg == 0) {
086                        this.nextMsg *= 2;
087                        log.info("Matrix id=" + this.id + ": entry # " + this.counter);
088                }
089
090                // return the new entry
091                return e;
092        }
093
094        //////////////////////////////////////////////////////////////////////
095        // set/add methods
096        //////////////////////////////////////////////////////////////////////
097
098        public final Entry setEntry(final String fromLocId, final String toLocId, final double value) {
099                Entry e = getEntry(fromLocId, toLocId);
100                if (e == null) {
101                        return createAndAddEntry(fromLocId, toLocId, value);
102                }
103                e.setValue(value);
104                return e;
105        }
106
107        public final void setDesc(final String desc) {
108                this.desc = desc;
109        }
110
111        //////////////////////////////////////////////////////////////////////
112        // remove methods
113        //////////////////////////////////////////////////////////////////////
114
115        public final void removeEntry(final String from_loc, final String to_loc) {
116                Entry entry = getEntry(from_loc, to_loc);
117                ArrayList<Entry> from_loc_entries = this.fromLocs.get(from_loc);
118                ArrayList<Entry> to_loc_entries = this.toLocs.get(to_loc);
119
120                if ((from_loc_entries == null) || (to_loc_entries == null)) {
121                        throw new RuntimeException("entry with from_loc_id=" + from_loc +
122                                     " and to_loc_id=" + to_loc + " does not exist!" +
123                                     " Inconsistent data strucutre!!!");
124                }
125
126                if (!from_loc_entries.remove(entry)) {
127                        throw new RuntimeException("entry with from_loc_id=" + from_loc +
128                             " and to_loc_id=" + to_loc + " does not exist!" +
129                             " Inconsistent data strucutre!!!");
130                }
131                if (from_loc_entries.isEmpty()) {
132                        this.fromLocs.remove(from_loc);
133                }
134
135                if (!to_loc_entries.remove(entry)) {
136                        throw new RuntimeException("entry with from_loc_id=" + from_loc +
137                             " and to_loc_id=" + to_loc + " does not exist!" +
138                             " Inconsistent data strucutre!!!");
139                }
140                if (to_loc_entries.isEmpty()) {
141                        this.toLocs.remove(to_loc);
142                }
143                log.info("entry " + entry.toString() + " removed.");
144        }
145
146        //////////////////////////////////////////////////////////////////////
147        // get methods
148        //////////////////////////////////////////////////////////////////////
149
150        public final String getId() {
151                return this.id;
152        }
153
154        public final String getDesc() {
155                return this.desc;
156        }
157
158        public final Map<String, ArrayList<Entry>> getFromLocations() {
159                return this.fromLocs;
160        }
161
162        public final Map<String, ArrayList<Entry>> getToLocations() {
163                return this.toLocs;
164        }
165
166        public final List<Entry> getFromLocEntries(final String fromLocationId) {
167                return this.fromLocs.get(fromLocationId);
168        }
169
170        public final List<Entry> getToLocEntries(final String toLocationId) {
171                return this.toLocs.get(toLocationId);
172        }
173
174        public final Entry getEntry(final String from, final String to) {
175                ArrayList<Entry> fe = this.fromLocs.get(from);
176                if (fe == null) return null;
177                for (int i=0; i<fe.size(); i++) {
178                        Entry e = fe.get(i);
179                        if (e.getToLocation().equals(to)) {
180                                return e;
181                        }
182                }
183                return null;
184        }
185
186        //////////////////////////////////////////////////////////////////////
187        // print methods
188        //////////////////////////////////////////////////////////////////////
189
190        @Override
191        public final String toString() {
192                return "[id=" + this.id + "]" +
193                                "[desc=" + this.desc + "]" +
194                                "[nof_from_locs=" + this.fromLocs.size() + "]" +
195                                "[nof_to_locs=" + this.toLocs.size() + "]";
196        }
197}
