182        /**
183         * This method initializes the styles for the different icons used.
184         */
185        private void createStyles() {
186
187                this.redCircleStyle = this.kmlObjectFactory.createStyleType(); this.redCircleStyle.setId("redCircleStyle");
188                this.redCrossStyle = this.kmlObjectFactory.createStyleType(); this.redCrossStyle.setId("redCrossStyle");
189                this.redMinusStyle = this.kmlObjectFactory.createStyleType(); this.redMinusStyle.setId("redMinusStyle");
190                this.yellowCircleStyle = this.kmlObjectFactory.createStyleType(); this.yellowCircleStyle.setId("yellowCircleStyle");
191                this.yellowCrossStyle = this.kmlObjectFactory.createStyleType(); this.yellowCrossStyle.setId("yellowCrossStyle");
192                this.yellowMinusStyle = this.kmlObjectFactory.createStyleType(); this.yellowMinusStyle.setId("yellowMinusStyle");
193                this.greenCircleStyle = this.kmlObjectFactory.createStyleType(); this.greenCircleStyle.setId("greenCircleStyle");
194                this.greenCrossStyle = this.kmlObjectFactory.createStyleType(); this.greenCrossStyle.setId("greenCrossStyle");
195                this.greenMinusStyle = this.kmlObjectFactory.createStyleType(); this.greenMinusStyle.setId("greenMinusStyle");
196                this.greyCircleStyle = this.kmlObjectFactory.createStyleType(); this.greyCircleStyle.setId("greyCircleStyle");
197                this.greyCrossStyle = this.kmlObjectFactory.createStyleType(); this.greyCrossStyle.setId("greyCrossStyle");
198                this.greyMinusStyle = this.kmlObjectFactory.createStyleType(); this.greyMinusStyle.setId("greyMinusStyle");
199
200                byte[] red = new byte[]{(byte) 0xFF, (byte) 0x0F, (byte) 0x0F, (byte) 0xBE};
201                byte[] green = new byte[]{(byte) 0xFF, (byte) 0x14, (byte) 0xDC, (byte) 0x0A};
202                byte[] yellow = new byte[]{(byte) 0xFF, (byte) 0x14, (byte) 0xE6, (byte) 0xE6};
203                byte[] grey = new byte[]{(byte) 0xFF, (byte) 0x42, (byte) 0x42, (byte) 0x42};
204
205                HashMap<StyleType, byte[]> colors = new HashMap<StyleType, byte[]>();
206                colors.put(this.redCircleStyle, red);
207                colors.put(this.redCrossStyle, red);
208                colors.put(this.redMinusStyle, red);
209                colors.put(this.yellowCircleStyle, yellow);
210                colors.put(this.yellowCrossStyle, yellow);
211                colors.put(this.yellowMinusStyle, yellow);
212                colors.put(this.greenCircleStyle, green);
213                colors.put(this.greenCrossStyle, green);
214                colors.put(this.greenMinusStyle, green);
215                colors.put(this.greyCircleStyle, grey);
216                colors.put(this.greyCrossStyle, grey);
217                colors.put(this.greyMinusStyle, grey);
218
219                HashMap<StyleType, String> hrefs = new HashMap<StyleType, String>();
220                hrefs.put(this.redCircleStyle, CIRCLEICON);
221                hrefs.put(this.redCrossStyle, CROSSICON);
222                hrefs.put(this.redMinusStyle, MINUSICON);
223                hrefs.put(this.yellowCircleStyle, CIRCLEICON);
224                hrefs.put(this.yellowCrossStyle, CROSSICON);
225                hrefs.put(this.yellowMinusStyle, MINUSICON);
226                hrefs.put(this.greenCircleStyle, CIRCLEICON);
227                hrefs.put(this.greenCrossStyle, CROSSICON);
228                hrefs.put(this.greenMinusStyle, MINUSICON);
229                hrefs.put(this.greyCircleStyle, CIRCLEICON);
230                hrefs.put(this.greyCrossStyle, CROSSICON);
231                hrefs.put(this.greyMinusStyle, MINUSICON);
232
233                for (StyleType styleType : new StyleType[]{
234                                this.redCircleStyle, this.redCrossStyle, this.redMinusStyle, this.yellowCircleStyle, this.yellowCrossStyle, this.yellowMinusStyle,
235                                this.greenCircleStyle, this.greenCrossStyle, this.greenMinusStyle, this.greyCircleStyle, this.greyCrossStyle, this.greyMinusStyle}) {
236
237                        IconStyleType icon = this.kmlObjectFactory.createIconStyleType();
238                        icon.setColor(
239                                        new byte[]{
240                                                        colors.get(styleType)[0],
241                                                        colors.get(styleType)[1],
242                                                        colors.get(styleType)[2],
243                                                        colors.get(styleType)[3]});
244                        icon.setScale(ICONSCALE);
245
246                        LinkType link = this.kmlObjectFactory.createLinkType();
247                        link.setHref(hrefs.get(styleType));
248                        icon.setIcon(link);
249
250                        styleType.setIconStyle(icon);
251
252                        this.mainDoc.getAbstractStyleSelectorGroup().add(this.kmlObjectFactory.createStyle(styleType));
253                }
254        }

256        /**
257         * Writes the data to the file at the path given as String
258         *
259         * @param filename
260         */
261        @Override
262        public void writeFile(final String filename) {
263                log.info("Writing google earth file to " + filename);
264                // init kml
265                this.mainKml = this.kmlObjectFactory.createKmlType();
266                this.mainDoc = this.kmlObjectFactory.createDocumentType();
267                this.mainKml.setAbstractFeatureGroup(this.kmlObjectFactory.createDocument(mainDoc));
268
269                // create the styles and the folders
270                createStyles();
271                // create a folder
272                this.mainFolder = this.kmlObjectFactory.createFolderType();
273                this.mainFolder.setName("Comparison, Iteration " + this.iterationNumber);
274                this.mainDoc.getAbstractFeatureGroup().add(this.kmlObjectFactory.createFolder(this.mainFolder));
275                // the writer
276                this.writer = new KMZWriter(filename);
277
278                try {
279                        //try to create the legend
280                        this.mainFolder.getAbstractFeatureGroup().add(this.kmlObjectFactory.createScreenOverlay(createLegend()));
281                        this.mainFolder.getAbstractFeatureGroup().add(this.kmlObjectFactory.createScreenOverlay(createLegendNormalized()));
282                        this.mainFolder.getAbstractFeatureGroup().add(this.kmlObjectFactory.createScreenOverlay(createLegendGEH()));
283                } catch (IOException e) {
284                        log.error("Cannot add legend to the KMZ file.", e);
285                }
286                try {
287                        //add the matsim logo to the kml
288                        this.mainFolder.getAbstractFeatureGroup().add(this.kmlObjectFactory.createScreenOverlay(MatsimKMLLogo.writeMatsimKMLLogo(writer)));
289                } catch (IOException e) {
290                        log.error("Cannot add logo to the KMZ file.", e);
291                }
292
293                try {
294                        // copy required icons to the kmz
295                        this.writer.addNonKMLFile(MatsimResource.getAsInputStream("icons/circle.png"), CIRCLEICON);
296                        this.writer.addNonKMLFile(MatsimResource.getAsInputStream("icons/plus.png"), CROSSICON);
297                        this.writer.addNonKMLFile(MatsimResource.getAsInputStream("icons/minus.png"), MINUSICON);
298                } catch (IOException e) {
299                        log.error("Could not copy copy plus-/minus-icons to the KMZ.", e);
300                }
301
302                // prepare folders for simRealPerHour-Graphs (top-left, xy-plots)
303                FolderType simRealFolder = this.kmlObjectFactory.createFolderType();
304                simRealFolder.setName("XY Comparison Plots");
305                this.mainFolder.getAbstractFeatureGroup().add(this.kmlObjectFactory.createFolder(simRealFolder));
306
307                // error graphs and awtv graph
308                {
309                        ScreenOverlayType errorGraph = createBiasErrorGraph(filename);
310                        errorGraph.setVisibility(Boolean.TRUE);
311                        this.mainFolder.getAbstractFeatureGroup().add(this.kmlObjectFactory.createScreenOverlay(errorGraph));
312                }
313                {
314                        ScreenOverlayType errorGraph = createBiasNormalizedErrorGraph(filename);
315                        errorGraph.setVisibility(Boolean.FALSE);
316                        this.mainFolder.getAbstractFeatureGroup().add(this.kmlObjectFactory.createScreenOverlay(errorGraph));
317                }
318                {
319                        ScreenOverlayType errorGraph = createBoxPlotErrorGraph();
320                        if (errorGraph != null) {
321                                errorGraph.setVisibility(Boolean.FALSE);
322                                this.mainFolder.getAbstractFeatureGroup().add(this.kmlObjectFactory.createScreenOverlay(errorGraph));
323                        }
324                }
325                {
326                        ScreenOverlayType errorGraph = createBoxPlotNormalizedErrorGraph();
327                        if (errorGraph != null) {
328                                errorGraph.setVisibility(Boolean.FALSE);
329                                this.mainFolder.getAbstractFeatureGroup().add(this.kmlObjectFactory.createScreenOverlay(errorGraph));
330                        }
331                }
332                {
333                        ScreenOverlayType awtv = null;
334                        try {
335                                awtv=this.createAWTVGraph();
336                        } catch (Exception ee) {
337                                log.warn("generating awtv (average weekday traffic volumes) graph failed; printing stacktrace but continuing anyways ...");
338                                for (int ii = 0; ii < ee.getStackTrace().length; ii++) {
339                                        log.info(ee.getStackTrace()[ii].toString());
340                                }
341                        }
342                        if (awtv != null) {
343                                awtv.setVisibility(Boolean.FALSE);
344                                this.mainFolder.getAbstractFeatureGroup().add(this.kmlObjectFactory.createScreenOverlay(awtv));
345                        }
346                }
347
348                // link graphs
349                this.createCountsLoadCurveGraphs();
350                this.createCountsGEHCurveGraphs();
351
352                // hourly data...
353                FolderType folderRelative = this.kmlObjectFactory.createFolderType();
354                folderRelative.setName("Relative Values");
355                this.mainFolder.getAbstractFeatureGroup().add(this.kmlObjectFactory.createFolder(folderRelative));
356                
357                FolderType folderNormalizedRelative = this.kmlObjectFactory.createFolderType();
358                folderNormalizedRelative.setName("Normalized Relative Values");
359                folderNormalizedRelative.setVisibility(Boolean.FALSE);
360                this.mainFolder.getAbstractFeatureGroup().add(this.kmlObjectFactory.createFolder(folderNormalizedRelative));
361                
362                FolderType folderGEH = kmlObjectFactory.createFolderType();
363                folderGEH.setName("GEH Values");
364                folderGEH.setVisibility(Boolean.FALSE);
365                this.mainFolder.getAbstractFeatureGroup().add(this.kmlObjectFactory.createFolder(folderGEH));
366                
367                for (int h = 1; h < 25; h++) {
368                        // the timespan for this hour
369                        TimeSpanType timespan = this.kmlObjectFactory.createTimeSpanType();
370                        timespan.setBegin("1999-01-01T" + Time.writeTime(((h - 1) * 3600)));
371                        timespan.setEnd("1999-01-01T" + Time.writeTime((h * 3600)));
372
373                        // first add the xyplot ("SimRealPerHourGraph") as overlay
374                        this.addCountsSimRealPerHourGraphs(simRealFolder, h, timespan);
375
376                        // add the placemarks for the links in this hour
377                        FolderType subfolderRelative = this.kmlObjectFactory.createFolderType();
378                        subfolderRelative.setName(createFolderName(h));
379                        subfolderRelative.setAbstractTimePrimitiveGroup(this.kmlObjectFactory.createTimeSpan(timespan));
380                        folderRelative.getAbstractFeatureGroup().add(this.kmlObjectFactory.createFolder(subfolderRelative));
381
382                        FolderType subfolderNormalizedRelative = this.kmlObjectFactory.createFolderType();
383                        subfolderNormalizedRelative.setName(createFolderName(h));
384                        subfolderNormalizedRelative.setAbstractTimePrimitiveGroup(this.kmlObjectFactory.createTimeSpan(timespan));
385                        folderNormalizedRelative.getAbstractFeatureGroup().add(this.kmlObjectFactory.createFolder(subfolderNormalizedRelative));
386                        
387                        FolderType subfolderGEH = this.kmlObjectFactory.createFolderType();
388                        subfolderGEH.setName(createFolderName(h));
389                        subfolderGEH.setAbstractTimePrimitiveGroup(this.kmlObjectFactory.createTimeSpan(timespan));
390                        subfolderGEH.setVisibility(Boolean.FALSE);
391                        folderGEH.getAbstractFeatureGroup().add(this.kmlObjectFactory.createFolder(subfolderGEH));
392                        
393                        writeLinkData(this.countComparisonFilter.getCountsForHour(Integer.valueOf(h)), subfolderRelative, subfolderNormalizedRelative, subfolderGEH);
394                }
395                
396                finish();
397        }

399        /**
400         * Creates the string for the foldername
401         * @param timestep
402         * @return a timestep specific standard string
403         */
404        private String createFolderName(final int timestep) {
405                StringBuilder buffer = new StringBuilder(30);
406                buffer.append("Traffic from ");
407                buffer.append(this.timestepToString(timestep - 1));
408                buffer.append(" to ");
409                buffer.append(this.timestepToString(timestep));
410                buffer.append(" o'clock");
411                return buffer.toString();
412        }

519        /**
520         * This method writes all the data for each of the links/counts to the kml
521         * document.
522         *
523         * @param countSimComparisonList provides "the data"
524         * @param folder The folder to which to add the data in the kml-file.
525         */
526        private void writeLinkData(final List<CountSimComparison> countSimComparisonList, final FolderType folder, final FolderType folderNormalized, final FolderType folderGEH) {
527                PlacemarkType placemark;
528                double relativeError;
529                double normalizedRelativeError;
530                double gehValue;
531                PointType point;
532                for (CountSimComparison csc : countSimComparisonList) {
533                        Id<T> itemId = csc.getId();
534                        Coord coord = null;
535//                      String description = csc.getCsId();
536                        
537                        if (this.counts == null) {                      
538                                Link link = this.network.getLinks().get(itemId);

608                        
609                        // circle placemark for geh
610                        {
611                                // build placemark
612                                placemark = createPlacemark(itemId.toString(), csc, relativeError, normalizedRelativeError, gehValue, csc.getHour(), this.countsGEHCurveGraphMap.get(itemId.toString()));
613//                              if (description != null) placemark.setName(description);
614                                placemark.setVisibility(Boolean.FALSE);

628        /**
629         * Calculates the position of a placemark in a way that it is 40 % of the link
630         * length away from the node where the link starts.
631         *
632         * @param l
633         * @return the CoordI instance
634         */
635        private Coord calculatePlacemarkPosition(final Link l) {
636                Coord coordFrom = l.getFromNode().getCoord();
637                Coord coordTo = l.getToNode().getCoord();
638                double xDiff = coordTo.getX() - coordFrom.getX();
639                double yDiff = coordTo.getY() - coordFrom.getY();
640                double length = Math.sqrt((xDiff*xDiff) + (yDiff*yDiff));
641                double scale = 0.4;
642                scale = l.getLength() * scale;
643                Coord vec = new Coord(coordFrom.getX() + (xDiff * scale / length), coordFrom.getY() + (yDiff * scale / length));
644                return vec;
645        }
