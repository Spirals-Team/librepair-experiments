059        public CountsLoadCurveGraph(final List<CountSimComparison> ccl, final int iteration, final String filename){
060                super(ccl, iteration, filename, filename);
061                this.dataset0 = new DefaultCategoryDataset();
062                // TRRE:  this.dataset1 = new DefaultCategoryDataset();
063        }
064
065        public void clearDatasets(){
066                this.dataset0.clear();
067                // TRRE: this.dataset1.clear();
068        }
069
070        public void add2LoadCurveDataSets(final CountSimComparison cc ) {
071                String matsim_series = "Sim Volumes";
072                String real_series = "Count Volumes";
073
074                String h=Integer.toString(cc.getHour());
075
076                this.dataset0.addValue(cc.getSimulationValue(),matsim_series, h);
077                this.dataset0.addValue(cc.getCountValue(),real_series,h);
078
079                //relative error
080                // TRRE: this.dataset1.addValue(cc.calculateRelativeError(),"Signed Rel. Error",Integer.toString(cc.getHour()));
081        }//add2LoadCurveDataSets
082
083
084        @Override
085        public JFreeChart createChart(final int nbr) {
086                String title = this.getChartTitle() + ", Iteration: " + this.iteration_;
087                this.chart_ = ChartFactory.createBarChart(title, "Hour", "Volumes [veh/h]", this.dataset0,
088                                PlotOrientation.VERTICAL, true, // legend?
089                                true, // tooltips?
090                                false // URLs?
091                                );
092                CategoryPlot plot = this.chart_.getCategoryPlot();
093//              plot.getRangeAxis().setRange(0.0, 10000.0); // do not set a fixed range for the single link graphs
094                plot.getDomainAxis().setCategoryLabelPositions(CategoryLabelPositions.UP_45);
095
096                // BarRenderer renderer=(BarRenderer) plot.getRenderer();
097                BarRenderer renderer = new BarRenderer();
098                renderer.setSeriesOutlinePaint(0, Color.black);
099                renderer.setSeriesOutlinePaint(1, Color.black);
100                renderer.setSeriesPaint(0, Color.getHSBColor((float) 0.62, (float) 0.56, (float) 0.93));
101                // Color.orange gives a dirty yellow!
102                renderer.setSeriesPaint(1, Color.getHSBColor((float) 0.1, (float) 0.79, (float) 0.89));
103                renderer.setSeriesToolTipGenerator(0, new StandardCategoryToolTipGenerator());
104                renderer.setSeriesToolTipGenerator(1, new StandardCategoryToolTipGenerator());
105                renderer.setItemMargin(0.0);
106                
107                // configure plot with light colors instead of the default 3D
108                renderer.setShadowVisible(false);  
109                renderer.setBarPainter( new StandardBarPainter() ); 
110                this.chart_.setBackgroundPaint(Color.getHSBColor((float) 0.0, (float) 0.0, (float) 0.93));
111                plot.setBackgroundPaint(Color.white);        
112                plot.setRangeGridlinePaint(Color.gray);         
113                plot.setRangeGridlinesVisible(true);      
114
115                plot.setRenderer(0, renderer);
116                plot.setDomainAxisLocation(AxisLocation.BOTTOM_OR_RIGHT);
117                // TRRE:  plot.setDataset(1, this.dataset1);
118                plot.mapDatasetToRangeAxis(1, 1);
119
120                final CategoryAxis axis1 = plot.getDomainAxis();
121                axis1.setCategoryMargin(0.25); // leave a gap of 25% between categories
122
123                /*  TRRE: 
124                final ValueAxis axis2 = new NumberAxis("Signed Rel. Error [%]");
125                plot.setRangeAxis(1, axis2);
126
127                final LineAndShapeRenderer renderer2 = new LineAndShapeRenderer();
128                renderer2.setSeriesToolTipGenerator(0, new StandardCategoryToolTipGenerator());
129                renderer2.setSeriesShape(0, new Rectangle2D.Double(-1.5, -1.5, 3.0, 3.0));
130                renderer2.setSeriesPaint(0, Color.black);
131                renderer2.setBaseStroke(new BasicStroke(2.5f));
132                plot.setRenderer(1, renderer2);
133                plot.setDatasetRenderingOrder(DatasetRenderingOrder.FORWARD);
134                */
135                
136                return this.chart_;
137        }
