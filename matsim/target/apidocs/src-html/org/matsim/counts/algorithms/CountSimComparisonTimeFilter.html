021package org.matsim.counts.algorithms;
022
023import java.util.List;
024import java.util.Vector;
025
026import org.matsim.counts.CountSimComparison;
027
028/**
029 * This class can be used to return filtered views on a List of
030 * CountSimComparison objects.
031 *
032 * @author dgrether
033 *
034 */
035public class CountSimComparisonTimeFilter {
036        /**
037         * this array maps the time steps to Lists of CountSimComparison objects
038         */
039        private final List<CountSimComparison>[] countSimComparisonTimeMap;
040
041        /**
042         * The list containing the comparisons
043         */
044        private final List<CountSimComparison> countSimComparisons;
045
046        /**
047         * Fills the Array for each hour with the appropriate CountSimComparison objects
048         * from the list given as parameter.
049         * @param countSimComparisons
050         *
051         */
052        public CountSimComparisonTimeFilter(
053                        final List<CountSimComparison> countSimComparisons) {
054                this.countSimComparisons = countSimComparisons;
055                int countsPerHour = this.countSimComparisons.size() / 24;
056                // initialize array
057                this.countSimComparisonTimeMap = new List[24];
058                for (int i = 0; i < 24; i++) {
059                        this.countSimComparisonTimeMap[i] = new Vector<CountSimComparison>(countsPerHour);
060                }
061                // and add the data
062                for (CountSimComparison csc : this.countSimComparisons) {
063                        this.countSimComparisonTimeMap[csc.getHour() - 1].add(csc);
064                }
065        }
066
067        /**
068         * @param timefilter
069         *          A time step to filter the counts. Time may be null to avoid
070         *          filtering. A value in 1..24, 1 for 0 a.m. to 1 a.m., 2 for 1 a.m.
071         *          to 2 a.m. If null nothing is filtered.
072         *
073         * @return All counts if the parameter is null, else a subset of all counts for the given hour
074         */
075        public List<CountSimComparison> getCountsForHour(final Integer timefilter) {
076                // only need to do this once
077                if (timefilter == null) {
078                        return this.countSimComparisons;
079                }
080                return this.countSimComparisonTimeMap[timefilter.intValue() - 1];
081        }
082}
