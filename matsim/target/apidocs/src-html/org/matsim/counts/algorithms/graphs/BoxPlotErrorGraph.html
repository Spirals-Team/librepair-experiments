021package org.matsim.counts.algorithms.graphs;
022
023import java.awt.Color;
024import java.awt.Font;
025import java.util.ArrayList;
026import java.util.List;
027
028import org.jfree.chart.JFreeChart;
029import org.jfree.chart.axis.CategoryAxis;
030import org.jfree.chart.axis.CategoryLabelPositions;
031import org.jfree.chart.axis.NumberAxis;
032import org.jfree.chart.labels.BoxAndWhiskerToolTipGenerator;
033import org.jfree.chart.plot.CategoryPlot;
034import org.jfree.chart.plot.CombinedDomainCategoryPlot;
035import org.jfree.chart.renderer.category.BoxAndWhiskerRenderer;
036import org.jfree.data.statistics.DefaultBoxAndWhiskerCategoryDataset;
037import org.matsim.counts.CountSimComparison;
038
039public final class BoxPlotErrorGraph extends CountsGraph {
040
041        public BoxPlotErrorGraph(final List<CountSimComparison> ccl, final int iteration, final String filename,
042                        final String chartTitle) {
043                super(ccl, iteration, filename, chartTitle);
044        }
045
046        @SuppressWarnings("unchecked")
047        @Override
048        public JFreeChart createChart(final int nbr) {
049
050                DefaultBoxAndWhiskerCategoryDataset dataset0 = new DefaultBoxAndWhiskerCategoryDataset();
051                DefaultBoxAndWhiskerCategoryDataset dataset1 = new DefaultBoxAndWhiskerCategoryDataset();
052
053                final ArrayList<Double>[] listRel = new ArrayList[24];
054                final ArrayList<Double>[] listAbs = new ArrayList[24];
055
056                // init
057                for (int i = 0; i < 24; i++) {
058                        listRel[i] = new ArrayList<Double>();
059                        listAbs[i] = new ArrayList<Double>();
060                }
061
062                // add the values of all counting stations to each hour
063                for (CountSimComparison cc : this.ccl_) {
064                        int hour = cc.getHour() - 1;
065                        listRel[hour].add(cc.calculateRelativeError() * 100);
066                        listAbs[hour].add(cc.getSimulationValue() - cc.getCountValue());
067                }
068
069                // add the collected values to the graph / dataset
070                for (int i = 0; i < 24; i++) {
071                        dataset0.add(listRel[i], "Rel Error", Integer.toString(i + 1));
072                        dataset1.add(listAbs[i], "Abs Error", Integer.toString(i + 1));
073                }
074
075                String title = "Iteration: " + this.iteration_;
076
077                final CombinedDomainCategoryPlot plot = new CombinedDomainCategoryPlot();
078
079                final CategoryAxis xAxis = new CategoryAxis("Hour");
080                final NumberAxis yAxis0 = new NumberAxis("Signed Rel. Error [%]");
081                final NumberAxis yAxis1 = new NumberAxis("Signed Abs. Error [veh]");
082                yAxis0.setAutoRangeIncludesZero(false);
083                yAxis1.setAutoRangeIncludesZero(false);
084
085                final BoxAndWhiskerRenderer renderer = new BoxAndWhiskerRenderer();
086                renderer.setFillBox(false);
087                renderer.setSeriesPaint(0, Color.blue);
088                renderer.setSeriesToolTipGenerator(0, new BoxAndWhiskerToolTipGenerator());
089
090                CategoryPlot subplot0 = new CategoryPlot(dataset0, xAxis, yAxis0, renderer);
091                CategoryPlot subplot1 = new CategoryPlot(dataset1, xAxis, yAxis1, renderer);
092
093                plot.add(subplot0);
094                plot.add(subplot1);
095
096                final CategoryAxis axis1 = new CategoryAxis("hour");
097                axis1.setTickLabelFont(new Font("SansSerif", Font.PLAIN, 7));
098                axis1.setCategoryLabelPositions(CategoryLabelPositions.UP_45);
099                plot.setDomainAxis(axis1);
100
101                this.chart_ = new JFreeChart(title, new Font("SansSerif", Font.BOLD, 14), plot, false);
102                return this.chart_;
103        }
104}
