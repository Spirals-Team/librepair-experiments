021package org.matsim.counts.algorithms.graphs;
022
023import java.awt.Font;
024import java.util.List;
025
026import org.jfree.chart.ChartFactory;
027import org.jfree.chart.JFreeChart;
028import org.jfree.chart.axis.AxisLocation;
029import org.jfree.chart.axis.CategoryAxis;
030import org.jfree.chart.axis.NumberAxis;
031import org.jfree.chart.axis.ValueAxis;
032import org.jfree.chart.labels.StandardCategoryToolTipGenerator;
033import org.jfree.chart.plot.CategoryPlot;
034import org.jfree.chart.plot.DatasetRenderingOrder;
035import org.jfree.chart.plot.PlotOrientation;
036import org.jfree.chart.renderer.category.LineAndShapeRenderer;
037import org.jfree.data.category.DefaultCategoryDataset;
038import org.matsim.counts.ComparisonErrorStatsCalculator;
039import org.matsim.counts.CountSimComparison;
040
041public final class BiasErrorGraph extends CountsGraph {
042
043        private ComparisonErrorStatsCalculator errorStats;
044        
045        public BiasErrorGraph(final List<CountSimComparison> ccl, final int iteration, final String filename,
046                        final String chartTitle) {
047                super(ccl, iteration, filename, chartTitle);
048        }
049
050        @Override
051        public JFreeChart createChart(final int nbr) {
052                DefaultCategoryDataset dataset0 = new DefaultCategoryDataset();
053                DefaultCategoryDataset dataset1 = new DefaultCategoryDataset();
054
055                this.errorStats = new ComparisonErrorStatsCalculator(this.ccl_);
056
057                double[] meanRelError = this.errorStats.getMeanRelError();
058//              double[] meanAbsError = this.errorStats.getMeanAbsError();
059                double[] meanBias = this.errorStats.getMeanBias();
060
061                for (int h = 0; h < 24; h++) {
062                        meanRelError[h] *= 100;
063                        dataset0.addValue(meanRelError[h], "Mean rel error", Integer.toString(h + 1));
064//                      dataset1.addValue(meanAbsError[h], "Mean abs error", Integer.toString(h + 1));
065                        dataset1.addValue(meanBias[h], "Mean bias", Integer.toString(h + 1));
066                }
067
068                this.chart_ = ChartFactory.createLineChart("", "Hour", "Mean rel error [%]", dataset0, PlotOrientation.VERTICAL,
069                                true, // legend?
070                                true, // tooltips?
071                                false // URLs?
072                                );
073                CategoryPlot plot = this.chart_.getCategoryPlot();
074                plot.setDomainAxisLocation(AxisLocation.BOTTOM_OR_RIGHT);
075                plot.setDataset(1, dataset1);
076                plot.mapDatasetToRangeAxis(1, 1);
077
078                final LineAndShapeRenderer renderer = new LineAndShapeRenderer();
079                renderer.setSeriesToolTipGenerator(0, new StandardCategoryToolTipGenerator());
080                plot.setRenderer(0, renderer);
081
082                final CategoryAxis axis1 = new CategoryAxis("Hour");
083                axis1.setTickLabelFont(new Font("SansSerif", Font.PLAIN, 7));
084                plot.setDomainAxis(axis1);
085
086//              final ValueAxis axis2 = new NumberAxis("Mean abs {bias, error} [veh/h]");
087                final ValueAxis axis2 = new NumberAxis("Mean bias [veh/h]");
088                plot.setRangeAxis(1, axis2);
089
090                final ValueAxis axis3 = plot.getRangeAxis(0);
091                axis3.setRange(0.0, 100.0);
092
093                final LineAndShapeRenderer renderer2 = new LineAndShapeRenderer();
094                renderer2.setSeriesToolTipGenerator(0, new StandardCategoryToolTipGenerator());
095                renderer2.setSeriesToolTipGenerator(1, new StandardCategoryToolTipGenerator());
096//              renderer2.setSeriesPaint(0, Color.black);
097                plot.setRenderer(1, renderer2);
098                plot.setDatasetRenderingOrder(DatasetRenderingOrder.REVERSE);
099
100                return this.chart_;
101        }
102        
103        public double[] getMeanRelError() {
104                if (this.errorStats == null) {
105                        throw new RuntimeException("Object not initialized correctly. Call createChart(..) first!");
106                }
107                return this.errorStats.getMeanRelError();
108        }
109
110        public double[] getMeanBias() {
111                if (this.errorStats == null) {
112                        throw new RuntimeException("Object not initialized correctly. Call createChart(..) first!");
113                }
114                return this.errorStats.getMeanBias();
115        }
116}
