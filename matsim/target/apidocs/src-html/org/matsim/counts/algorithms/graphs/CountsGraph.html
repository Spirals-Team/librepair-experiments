021package org.matsim.counts.algorithms.graphs;
022
023import java.util.List;
024
025import org.jfree.chart.JFreeChart;
026import org.matsim.counts.CountSimComparison;
027
028
029public abstract class CountsGraph {
030        protected List<CountSimComparison> ccl_;
031        protected JFreeChart chart_;
032        protected int iteration_;
033        /**
034         * The title of the chart, may contain whitespaces
035         */
036        private String chartTitle_;
037        /**
038         * the filename of the chart should not contain any whitespaces 
039         */
040        private String filename;
041
042        public CountsGraph() {
043        }
044        
045        public CountsGraph(final List<CountSimComparison> ccl, final int iteration, final String filename, final String chartTitle) {   
046                this.ccl_=ccl;
047                this.iteration_=iteration;
048                this.chartTitle_=chartTitle;
049                this.filename = filename;
050        }       
051        
052        public String getChartTitle() {
053                return this.chartTitle_;
054        }
055        public JFreeChart getChart() {
056                return this.chart_;
057        }
058        public void setChartTitle(final String chartTitle) {
059                this.chartTitle_=chartTitle;
060        }
061                
062        public abstract JFreeChart createChart(int nbr);
063
064        
065        /**
066         * @return the filename
067         */
068        public String getFilename() {
069                return this.filename;
070        }
071        
072        /**
073         * @param filename the filename to set
074         */
075        public void setFilename(final String filename) {
076                this.filename = filename;
077        }       
078}
