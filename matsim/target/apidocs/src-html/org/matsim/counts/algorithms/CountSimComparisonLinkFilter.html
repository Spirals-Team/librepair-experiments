021package org.matsim.counts.algorithms;
022
023import java.util.Collections;
024import java.util.Hashtable;
025import java.util.Iterator;
026import java.util.List;
027import java.util.Vector;
028
029import org.matsim.api.core.v01.Id;
030import org.matsim.api.core.v01.network.Link;
031import org.matsim.counts.CountSimComparison;
032
033/**
034 * This class can be used to return filtered views on a List of
035 * CountSimComparison objects.
036 *
037 * @author anhorni
038 *
039 */
040public class CountSimComparisonLinkFilter {
041
042        private Hashtable<Id<Link>, List<CountSimComparison>> countSimComparisonLinkMap;
043        
044        /**
045         * The list containing the comparisons
046         */
047        private final List<CountSimComparison> countSimComparisons;
048
049        /**
050         * Fills the hash table for each link with the appropriate CountSimComparison objects
051         * from the list given as parameter.
052         * @param countSimComparisons
053         *
054         */
055        public CountSimComparisonLinkFilter(final List<CountSimComparison> countSimComparisons) {
056                this.countSimComparisons = countSimComparisons;
057                this.countSimComparisonLinkMap=new Hashtable<>();
058                
059                if (this.countSimComparisons.size() < 1) {
060                        return;
061                }
062
063                List<CountSimComparison> countSimComparisonsPerLink=new Vector<CountSimComparison>();
064                Id<Link> prevId=this.countSimComparisons.get(0).getId();
065                
066                Iterator<CountSimComparison> csc_it = this.countSimComparisons.iterator();              
067                while (csc_it.hasNext()) {
068                        CountSimComparison csc= csc_it.next();  
069                                                
070                        // how to compare IdIs ? Id:  ... this.id.compareTo(id.toString());
071                        if (csc.getId().compareTo(prevId)==0) {
072                                countSimComparisonsPerLink.add(csc);
073                        }
074                
075                        // first element of next link id OR last element of last link id
076                        if (!(csc.getId().compareTo(prevId)==0) || (!csc_it.hasNext())) {
077                                countSimComparisonLinkMap.put(prevId, countSimComparisonsPerLink);
078                                countSimComparisonsPerLink=new Vector<CountSimComparison>();
079                                countSimComparisonsPerLink.add(csc);
080                        }
081                        prevId=csc.getId();
082                }
083        }
084
085        /**
086         * @param linkfilter
087         * @return All counts if the parameter is null, else a subset of all counts for the given link
088         */
089        public List<CountSimComparison> getCountsForLink(final Id<Link> linkfilter) {
090                // only need to do this once
091                if (linkfilter == null) {
092                        return this.countSimComparisons;
093                }
094                return this.countSimComparisonLinkMap.get(linkfilter);
095        }
096        
097        public double getAggregatedCountValue(final Id<Link> linkfilter) {
098                Iterator<CountSimComparison> csc_it = this.countSimComparisonLinkMap.get(linkfilter).iterator();                
099                double countValue=0.0;
100                while (csc_it.hasNext()) {
101                        CountSimComparison csc= csc_it.next();
102                        countValue+=csc.getCountValue();
103                }
104                return countValue;
105        }
106        
107        public double getAggregatedSimValue(final Id<Link> linkfilter) {
108                Iterator<CountSimComparison> csc_it = this.countSimComparisonLinkMap.get(linkfilter).iterator();                
109                double simValue=0.0;
110                while (csc_it.hasNext()) {
111                        CountSimComparison csc= csc_it.next();
112                        simValue+=csc.getSimulationValue();
113                }       
114                return simValue;
115        }
116        
117        // sorting ok for IdIs ?
118        public Vector<Id<Link>> getLinkIds() {
119                Vector<Id<Link>> linkIds = new Vector<Id<Link>>(countSimComparisonLinkMap.keySet());
120            Collections.sort(linkIds);    
121            return linkIds;
122        }
123}
