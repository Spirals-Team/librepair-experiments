058        public void outputHtml(){
059                new File(this.iterPath_+"/png").mkdir();
060        for (CountsGraph cg : this.cg_list_) {
061            writeHtml(cg, this.iterPath_, false);
062        }
063                writeHtml(null, this.iterPath_, true);
064                try {
065                        new File(this.iterPath_+"/div").mkdir();
066                        copyResourceToFile("style1.css", this.iterPath_ + "/div/style1.css");
067                        copyResourceToFile("logo.png", this.iterPath_ + "/div/logo.png");
068                        copyResourceToFile("overlib.js", this.iterPath_ + "/div/overlib.js");
069                        copyResourceToFile("title.png", this.iterPath_ + "/div/title.png");
070                } catch (IOException e) {
071                        System.out.println(e.toString());
072                        e.printStackTrace();
073                }
074        }
075
076        private void copyResourceToFile(final String resourceFilename, final String destinationFilename) throws IOException {
077                InputStream inStream = null;
078                OutputStream outStream = null;
079                try {
080                        inStream = MatsimResource.getAsInputStream(resourceFilename);
081                        outStream = new FileOutputStream(destinationFilename);
082                        IOUtils.copyStream(inStream, outStream);
083                } finally {
084                        if (inStream != null) {
085                                inStream.close();
086                        }
087                        if (outStream != null) {
088                                outStream.close();
089                        }
090                }
091        }
