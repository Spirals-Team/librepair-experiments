021package org.matsim.counts;
022
023import org.matsim.core.api.internal.MatsimWriter;
024import org.matsim.core.utils.geometry.CoordinateTransformation;
025import org.matsim.core.utils.geometry.transformations.IdentityTransformation;
026import org.matsim.core.utils.io.MatsimXmlWriter;
027
028import java.io.IOException;
029import java.io.Serializable;
030import java.util.Collections;
031import java.util.Comparator;
032import java.util.List;
033import java.util.Vector;
034
035public class CountsWriter extends MatsimXmlWriter implements MatsimWriter {
036
037        private final CountsWriterHandler handler;
038        private final Counts counts;
039
040        public CountsWriter(final Counts counts) {
041                this( new IdentityTransformation() , counts );
042        }
043
044        public CountsWriter(
045                        final CoordinateTransformation coordinateTransformation,
046                        final Counts counts) {
047                this.counts = counts;
048
049                // use the newest writer-version by default
050                this.handler = new CountsWriterHandlerImplV1(coordinateTransformation);
051        }
052
053        @Override
054        public final void write(final String filename) {
055                try {
056                        openFile(filename);
057
058                        // write custom header
059                        writeXmlHead();
060
061                        this.handler.startCounts(this.counts, this.writer);
062                        this.handler.writeSeparator(this.writer);
063                        
064                        List<Count> countsTemp = new Vector<Count>();
065                        countsTemp.addAll(this.counts.getCounts().values());
066                        Collections.sort(countsTemp, new CountComparator());
067
068            for (Count c : countsTemp) {
069                List<Volume> volumesTemp = new Vector<Volume>();
070                volumesTemp.addAll(c.getVolumes().values());
071                Collections.sort(volumesTemp, new VolumeComparator());
072                this.handler.startCount(c, this.writer);
073                for (Volume v : volumesTemp) {
074                    this.handler.startVolume(v, this.writer);
075                    this.handler.endVolume(this.writer);
076                }
077                this.handler.endCount(this.writer);
078                this.handler.writeSeparator(this.writer);
079                this.writer.flush();
080            }
081                        this.handler.endCounts(this.writer);
082                        close();
083                }
084                catch (IOException e) {
085                        throw new RuntimeException(e);
086                }
087        }
088
089        @Override
090        public final String toString() {
091                return super.toString();
092        }
093        
094        static class VolumeComparator implements Comparator<Volume>, Serializable {
095                private static final long serialVersionUID = 1L;
096
097                @Override
098                public int compare(final Volume v1, final Volume v2) {
099                        return Double.compare(v1.getHourOfDayStartingWithOne(), v2.getHourOfDayStartingWithOne());
100                }
101        }
102        
103        static class CountComparator implements Comparator<Count>, Serializable {
104                private static final long serialVersionUID = 1L;
105
106                @Override
107                public int compare(final Count c1, final Count c2) {
108                        
109                        // This seemed to be intended for a rather special case with a rather special
110                        // numbering of counts...  michaz mar 10
111                        
112//                      int i1 = Integer.parseInt(c1.getCsId().substring(c1.getCsId().length()-3));
113//                      int i2 = Integer.parseInt(c2.getCsId().substring(c2.getCsId().length()-3));
114//                      if (i1 < i2) return -1;
115//                      else if (i1 > i2) return 1;
116//                      else return 0;
117                        
118                        
119                        return c1.getCsLabel().compareTo(c2.getCsLabel());
120                }
121        }
122}
