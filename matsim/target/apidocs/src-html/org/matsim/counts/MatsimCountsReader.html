021package org.matsim.counts;
022
023import java.util.Stack;
024
025import org.apache.log4j.Logger;
026import org.matsim.core.utils.geometry.CoordinateTransformation;
027import org.matsim.core.utils.geometry.transformations.IdentityTransformation;
028import org.matsim.core.utils.io.MatsimXmlParser;
029import org.xml.sax.Attributes;
030
031/**
032 * A reader for counts-files of MATSim. This reader recognizes the format of the counts-file and uses
033 * the correct reader for the specific counts-version, without manual setting.
034 *
035 * @author mrieser
036 */
037public class MatsimCountsReader extends MatsimXmlParser {
038
039        private final static Logger log = Logger.getLogger(MatsimCountsReader.class);
040        private final static String COUNTS_V1 = "counts_v1.xsd";
041
042        private final Counts counts;
043        private MatsimXmlParser delegate = null;
044
045        private final CoordinateTransformation coordinateTransformation;
046
047        /**
048         * Creates a new reader for MATSim counts files.
049         *
050         * @param counts The Counts-object to store the configuration settings in.
051         */
052        public MatsimCountsReader(final Counts counts) {
053                this( new IdentityTransformation() , counts );
054        }
055
056        /**
057         * Creates a new reader for MATSim counts files.
058         *
059         * @param coordinateTransformation transformation from the CRS of the file to the internal CRS for MATSim
060         * @param counts The Counts-object to store the configuration settings in.
061         */
062        public MatsimCountsReader(
063                        final CoordinateTransformation coordinateTransformation,
064                        final Counts counts) {
065                this.coordinateTransformation = coordinateTransformation;
066                this.counts = counts;
067        }
068
069        @Override
070        public void startTag(final String name, final Attributes atts, final Stack<String> context) {
071                this.delegate.startTag(name, atts, context);
072        }
073
074        @Override
075        public void endTag(final String name, final String content, final Stack<String> context) {
076                this.delegate.endTag(name, content, context);
077        }
078
079        @Override
080        protected void setDoctype(final String doctype) {
081                super.setDoctype(doctype);
082                // Currently the only counts-type is v1
083                if (COUNTS_V1.equals(doctype)) {
084                        this.delegate = new CountsReaderMatsimV1( coordinateTransformation , this.counts);
085                        log.info("using counts_v1-reader.");
086                } else {
087                        throw new IllegalArgumentException("Doctype \"" + doctype + "\" not known.");
088                }
089        }
090
091}
