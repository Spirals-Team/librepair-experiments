021package org.matsim.counts;
022
023import java.util.HashMap;
024
025import org.matsim.api.core.v01.Coord;
026import org.matsim.api.core.v01.Id;
027import org.matsim.api.core.v01.Identifiable;
028
029public class Count<T> implements Identifiable<T> {
030
031        private final Id<T> linkId;
032        private String stationName;
033
034        private final HashMap<Integer,Volume> volumes = new HashMap<>();
035        private Coord coord;
036
037
038        protected Count(final Id<T> linkId2, final String stationName) {
039                this.linkId = linkId2;
040                this.stationName = stationName;
041        }
042
043        /**
044         * Creates and adds a {@link Volume} to the {@link Count}ing station.
045         * @param h indicating the hour-of-day. <b><i>Note: the hours for a counting 
046         *              station must be from 1-24, and <b><i>not</i></b> from 0-23, 
047         *              otherwise the {@link MatsimCountsReader} will throw an error.
048         *              </i></b>
049         * @param val the total number of vehicles counted during hour <code>h</code>.
050         * @return the {@link Count}ing station's {@link Volume}.
051         */
052        public final Volume createVolume(final int h, final double val) {
053                if ( h < 1 ) {
054                        throw new RuntimeException( "counts start at 1, not at 0.  If you have a use case where you need to go below one, "
055                                        + "let us know and we think about it, but so far we had numerous debugging sessions because someone inserted counts at 0.") ;
056                }
057                // overkill?
058                Volume v = new Volume(h,val);
059                this.volumes.put(Integer.valueOf(h), v);
060                return v;
061        }
062
063        public final void setCsId(final String cs_id) {
064                this.stationName = cs_id;
065        }
066
067        @Override
068        public final Id<T> getId() {
069                return this.linkId;
070        }
071
072        public final String getCsLabel() {
073                return this.stationName;
074        }
075
076        public final Volume getMaxVolume() {
077                Volume v_max = null;
078                double max = -1.0;
079                for (Volume v : this.volumes.values()) {
080                        if (v.getValue() > max) { max = v.getValue(); v_max = v; }
081                }
082                return v_max;
083        }
084
085        public final Volume getVolume(final int h) {
086                return this.volumes.get(Integer.valueOf(h));
087        }
088
089        public final HashMap<Integer,Volume> getVolumes() {
090                return this.volumes;
091        }
092
093        public void setCoord(final Coord coord) {
094                this.coord = coord;
095        }
096
097        /** @return Returns the exact coordinate, where this counting station is
098         * located, or null if no exact location is available.
099         **/
100        public Coord getCoord() {
101                return this.coord;
102        }
103
104        @Override
105        public final String toString() {
106                return "[Loc_id=" + this.linkId + "]" +
107                "[cs_id=" + this.stationName + "]" +
108                "[nof_volumes=" + this.volumes.size() + "]";
109        }
110
111}
