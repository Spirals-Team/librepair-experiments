021package org.matsim.counts;
022
023import org.matsim.api.core.v01.Id;
024import org.matsim.api.core.v01.network.Link;
025
026/**
027 * Classes implementing this interface can be used to access the data
028 * needed to compare traffic counts and simulation traffic at a specific
029 * link for a one hour time step.
030 * 
031 * @author dgrether
032 */
033public interface CountSimComparison<T> {
034
035        /**
036         * @return The Id of the link
037         */
038        public Id<T> getId();
039
040        /**
041         * 
042         * @return The Id of the count station
043         */
044        public String getCsId();
045        
046        /**
047         * The time at which the data was measured.
048         * @return A value in 1..24, 1 means 0 - 1 am, 2 means 1 - 2 am and so on
049         */
050        public int getHour();
051
052        /**
053         * @return The real traffic amount
054         */
055        public double getCountValue();
056
057        /**
058         * @return The traffic amount of the simulation
059         */
060        public double getSimulationValue();
061
062        /**
063         * Calculates the relative error.
064         * @return the relative error
065         */
066        public double calculateRelativeError();
067
068        /**
069         * Calculates the normalized relative error.
070         * @return the normalized relative error
071         */
072        public double calculateNormalizedRelativeError();
073        
074        /**
075         * Calculates the GEH value
076         * @return the GEH value
077         */
078        public double calculateGEHValue();      
079}
