021package org.matsim.pt.transitSchedule;
022
023import org.matsim.pt.transitSchedule.api.TransitRouteStop;
024import org.matsim.pt.transitSchedule.api.TransitStopFacility;
025
026
027/**
028 * Describes the stop within a route of a transit line. Specifies also at
029 * what time a headway is expected at the stop as offset from the route start.
030 *
031 * @author mrieser
032 */
033public class TransitRouteStopImpl implements TransitRouteStop {
034
035        private TransitStopFacility stop;
036        private final double departureOffset;
037        private final double arrivalOffset;
038        private boolean awaitDepartureTime = false;
039
040        protected TransitRouteStopImpl(final TransitStopFacility stop, final double arrivalDelay, final double departureDelay) {
041                this.stop = stop;
042                this.departureOffset = departureDelay;
043                this.arrivalOffset = arrivalDelay;
044        }
045
046        @Override
047        public TransitStopFacility getStopFacility() {
048                return this.stop;
049        }
050
051        @Override
052        public void setStopFacility(final TransitStopFacility stopFacility) {
053                this.stop = stopFacility;
054        }
055
056        @Override
057        public double getDepartureOffset() {
058                return this.departureOffset;
059        }
060
061        @Override
062        public double getArrivalOffset() {
063                return this.arrivalOffset;
064        }
065
066        @Override
067        public boolean isAwaitDepartureTime() {
068                return this.awaitDepartureTime;
069        }
070
071        @Override
072        public void setAwaitDepartureTime(final boolean awaitDepartureTime) {
073                this.awaitDepartureTime = awaitDepartureTime;
074        }
075
076        /**
077         * TransitRouteStops are typical Value Objects, so we consider two stops equal if they are equal field-wise.
078         * 
079         */
080        @Override
081        public boolean equals(Object obj) {
082                if (!(obj instanceof TransitRouteStopImpl)) {
083                        return false;
084                }
085                TransitRouteStopImpl other = (TransitRouteStopImpl) obj;
086                if (this.stop == null) {
087                        if (other.getStopFacility() != null) {
088                                return false;
089                        }
090                } else {
091                        if (!stop.equals(other.getStopFacility())) {
092                                return false;
093                        }
094                }
095                if (this.departureOffset != other.getDepartureOffset()) {
096                        return false;
097                } 
098                if (this.arrivalOffset != other.getArrivalOffset()) {
099                        return false;
100                }
101                if (this.awaitDepartureTime != other.isAwaitDepartureTime()) {
102                        return false;
103                }
104                return true;
105        }
106
107        @Override
108        public int hashCode() {
109                return stop.hashCode();
110        }
111        
112        @Override
113        public String toString() {
114                return "[TransitRouteStop stop=" + this.stop.getId() + " offset=" + this.departureOffset +" ]";
115        }
116
117}
