049/**
050 * Writes a transit schedule to a XML file in the format described by <code>transitSchedule_v2.dtd</code>.
051 *
052 * @author mrieser
053 */
054public class TransitScheduleWriterV2 extends MatsimXmlWriter implements MatsimSomeWriter {
055
056        private final CoordinateTransformation coordinateTransformation;
057        private final TransitSchedule schedule;
058        private final AttributesXmlWriterDelegate attributesWriter = new AttributesXmlWriterDelegate();
059
060        public TransitScheduleWriterV2(final TransitSchedule schedule) {
061                this(new IdentityTransformation() , schedule);
062        }
063
064        public TransitScheduleWriterV2(
065                        final CoordinateTransformation coordinateTransformation,
066                        final TransitSchedule schedule) {
067                this.coordinateTransformation = coordinateTransformation;
068                this.schedule = schedule;
069        }
070
071        public void write(OutputStream stream) throws UncheckedIOException {
072                this.openOutputStream(stream);
073                try {
074                        writeData();
075                } catch (IOException e) {
076                        throw new UncheckedIOException(e);
077                }
078        }
079
080        public void write(final String filename) throws UncheckedIOException {
081                this.openFile(filename);
082                try {
083                        writeData();
084                } catch (IOException e) {
085                        throw new UncheckedIOException(e);
086                }
087        }
088
089        private void writeData() throws IOException, UncheckedIOException {
090                this.writeXmlHead();
091                this.writeDoctype(Constants.TRANSIT_SCHEDULE, "http://www.matsim.org/files/dtd/transitSchedule_v2.dtd");
092                this.writeStartTag(Constants.TRANSIT_SCHEDULE, null);
093                this.writer.write(NL);
094                this.attributesWriter.writeAttributes( "\t" , this.writer , this.schedule.getAttributes() );
095
096                this.writeTransitStops();
097                this.writeMinimalTransferTimes();
098                for (TransitLine line : this.schedule.getTransitLines().values()) {
099                        writeTransitLine(line);
100                }
101                this.writeEndTag(Constants.TRANSIT_SCHEDULE);
102                this.close();
103        }
104
105        private void writeTransitStops() throws IOException, UncheckedIOException {
106                this.writeStartTag(Constants.TRANSIT_STOPS, null);
107
108                List<Tuple<String, String>> attributes = new ArrayList<>(5);
109                for (TransitStopFacility stop : this.schedule.getFacilities().values()) {
110                        attributes.clear();
111                        attributes.add(createTuple(Constants.ID, stop.getId().toString()));
112                        final Coord coord = this.coordinateTransformation.transform( stop.getCoord() );
113                        attributes.add(createTuple("x", coord.getX()));
114                        attributes.add(createTuple("y", coord.getY()));
115                        if (coord.hasZ()) {
116                                attributes.add(createTuple("z", coord.getZ()));
117                        }
118                        if (stop.getLinkId() != null) {
119                                attributes.add(createTuple("linkRefId", stop.getLinkId().toString()));
120                        }
121                        if (stop.getName() != null) {
122                                attributes.add(createTuple("name", stop.getName()));
123                        }
124                        if (stop.getStopAreaId() != null) {
125                                attributes.add(createTuple(Constants.STOP_AREA_ID, stop.getStopAreaId().toString()));
126                        }
127                        attributes.add(createTuple("isBlocking", stop.getIsBlockingLane()));
128                        if (AttributesUtils.isEmpty(stop.getAttributes())) {
129                                this.writeStartTag(Constants.STOP_FACILITY, attributes, true);
130                        } else {
131                                this.writeStartTag(Constants.STOP_FACILITY, attributes, false);
132                                if (!AttributesUtils.isEmpty(stop.getAttributes())) {
133                                        this.writer.write(NL);
134                                        this.attributesWriter.writeAttributes("\t\t\t", this.writer, stop.getAttributes());
135                                }
136                                this.writeEndTag(Constants.STOP_FACILITY);
137
138                        }
139                }
140
141                this.writeEndTag(Constants.TRANSIT_STOPS);
142        }
143
144        private void writeMinimalTransferTimes() {
145                List<Tuple<String, String>> attributes = new ArrayList<>(5);
146                MinimalTransferTimes.MinimalTransferTimesIterator iter = this.schedule.getMinimalTransferTimes().iterator();
147                if (iter.hasNext()) {
148                        this.writeStartTag(Constants.MINIMAL_TRANSFER_TIMES, attributes);
149                        while (iter.hasNext()) {
150                                iter.next();
151                                attributes.clear();
152                                attributes.add(createTuple(Constants.FROM_STOP, iter.getFromStopId().toString()));
153                                attributes.add(createTuple(Constants.TO_STOP, iter.getToStopId().toString()));
154                                attributes.add(createTuple(Constants.TRANSFER_TIME, iter.getSeconds()));
155                                this.writeStartTag(Constants.RELATION, attributes, true);
156                        }
157                        this.writeEndTag(Constants.MINIMAL_TRANSFER_TIMES);
158                }
159        }
160
161        private void writeTransitLine(final TransitLine line) throws IOException,  UncheckedIOException {
162                List<Tuple<String, String>> attributes = new ArrayList<>(1);
163                attributes.add(createTuple(Constants.ID, line.getId().toString()));
164                if (line.getName() != null) {
165                        attributes.add(createTuple(Constants.NAME, line.getName()));
166                }
167                this.writeStartTag(Constants.TRANSIT_LINE, attributes);
168                if (!AttributesUtils.isEmpty(line.getAttributes())) {
169                        this.writer.write(NL);
170                        this.attributesWriter.writeAttributes("\t\t", this.writer, line.getAttributes());
171                }
172
173                for (TransitRoute route : line.getRoutes().values()) {
174                        writeTransitRoute(route);
175                }
176
177                this.writeEndTag(Constants.TRANSIT_LINE);
178        }
179
180        private void writeTransitRoute(final TransitRoute route) throws IOException, UncheckedIOException {
181                List<Tuple<String, String>> attributes = new ArrayList<>(1);
182                attributes.add(createTuple(Constants.ID, route.getId().toString()));
183                this.writeStartTag(Constants.TRANSIT_ROUTE, attributes);
184
185                if (!AttributesUtils.isEmpty(route.getAttributes())) {
186                        this.writer.write(NL);
187                        this.attributesWriter.writeAttributes("\t\t\t", this.writer, route.getAttributes());
188                }
189
190                if (route.getDescription() != null) {
191                        this.writeStartTag(Constants.DESCRIPTION, null);
192                        this.writeContent(route.getDescription(), false);
193                        this.writeEndTag(Constants.DESCRIPTION);
194                }
195
196                this.writeStartTag(Constants.TRANSPORT_MODE, null);
197                this.writeContent(route.getTransportMode(), false);
198                this.writeEndTag(Constants.TRANSPORT_MODE);
199
200                this.writeRouteProfile(route.getStops());
201                this.writeRoute(route.getRoute());
202                this.writeDepartures(route.getDepartures());
203
204                this.writeEndTag(Constants.TRANSIT_ROUTE);
205        }
206
207        private void writeRouteProfile(final List<TransitRouteStop> stops) throws UncheckedIOException {
208                this.writeStartTag(Constants.ROUTE_PROFILE, null);
209
210                // optimization: only create one List for multiple departures
211                List<Tuple<String, String>> attributes = new ArrayList<>(4);
212                for (TransitRouteStop stop : stops) {
213                        attributes.clear();
214                        attributes.add(createTuple(Constants.REF_ID, stop.getStopFacility().getId().toString()));
215                        if (stop.getArrivalOffset() != Time.UNDEFINED_TIME) {
216                                attributes.add(createTimeTuple(Constants.ARRIVAL_OFFSET, stop.getArrivalOffset()));
217                        }
218                        if (stop.getDepartureOffset() != Time.UNDEFINED_TIME) {
219                                attributes.add(createTimeTuple(Constants.DEPARTURE_OFFSET, stop.getDepartureOffset()));
220                        }
221                        attributes.add(createTuple(Constants.AWAIT_DEPARTURE, String.valueOf(stop.isAwaitDepartureTime())));
222                        this.writeStartTag(Constants.STOP, attributes, true);
223                }
224
225                this.writeEndTag(Constants.ROUTE_PROFILE);
226        }
227
228        private void writeRoute(final NetworkRoute route) throws UncheckedIOException {
229                if (route != null) {
230                        this.writeStartTag(Constants.ROUTE, null);
231
232                        // optimization: only create one List for multiple departures
233                        List<Tuple<String, String>> attributes = new ArrayList<>(1);
234                        attributes.add(createTuple(Constants.REF_ID, route.getStartLinkId().toString()));
235                        this.writeStartTag(Constants.LINK, attributes, true);
236
237                        for (Id<Link> linkId : route.getLinkIds()) {
238                                attributes.clear();
239                                attributes.add(createTuple(Constants.REF_ID, linkId.toString()));
240                                this.writeStartTag(Constants.LINK, attributes, true);
241                        }
242
243                        attributes.clear();
244                        attributes.add(createTuple(Constants.REF_ID, route.getEndLinkId().toString()));
245                        this.writeStartTag(Constants.LINK, attributes, true);
246
247                        this.writeEndTag(Constants.ROUTE);
248                }
249        }
250
251        private void writeDepartures(final Map<Id<Departure>, Departure> departures) throws IOException, UncheckedIOException {
252                this.writeStartTag(Constants.DEPARTURES, null);
253
254                // optimization: only create one List for multiple departures
255                List<Tuple<String, String>> attributes = new ArrayList<>(3);
256
257                for (Departure dep : departures.values()) {
258                        attributes.clear();
259                        attributes.add(createTuple(Constants.ID, dep.getId().toString()));
260                        attributes.add(createTimeTuple(Constants.DEPARTURE_TIME, dep.getDepartureTime()));
261                        if (dep.getVehicleId() != null) {
262                                attributes.add(createTuple(Constants.VEHICLE_REF_ID, dep.getVehicleId().toString()));
263                        }
264                        if (AttributesUtils.isEmpty(dep.getAttributes())) {
265                                this.writeStartTag(Constants.DEPARTURE, attributes, true);
266                        } else {
267                                this.writeStartTag(Constants.DEPARTURE, attributes, false);
268                                this.writer.write(NL);
269                                this.attributesWriter.writeAttributes("\t\t\t\t\t", this.writer, dep.getAttributes());
270                                this.writeEndTag(Constants.DEPARTURE);
271                        }
272
273                }
274
275                this.writeEndTag(Constants.DEPARTURES);
276        }
