021package org.matsim.pt.transitSchedule.api;
022
023import org.matsim.api.core.v01.Coord;
024import org.matsim.api.core.v01.Id;
025import org.matsim.api.core.v01.network.Link;
026import org.matsim.facilities.Facility;
027import org.matsim.utils.objectattributes.attributable.Attributable;
028
029/**
030 * A facility (infrastructure) describing a public transport stop.
031 *
032 * @author mrieser
033 */
034public interface TransitStopFacility extends Facility<TransitStopFacility>, Attributable {
035
036        boolean getIsBlockingLane();
037
038        void setLinkId(final Id<Link> linkId);
039
040        /**
041         * Sets a human name for the stop facility, e.g. to be displayed
042         * on vehicles or at the stops' locations. The name can be
043         * <code>null</code> to delete a previously assigned name.
044         *
045         * @param name
046         */
047        void setName(final String name);
048
049        /**
050         * @return name of the stop facility. Can be <code>null</code>.
051         */
052        String getName();
053
054        Id<TransitStopArea> getStopAreaId();
055
056        void setStopAreaId(Id<TransitStopArea> stopAreaId);
057
058        void setCoord(Coord coord);
059        
060}
