021package org.matsim.pt.transitSchedule;
022
023import java.util.ArrayList;
024import java.util.List;
025import java.util.Map;
026
027import org.matsim.api.core.v01.Coord;
028import org.matsim.api.core.v01.Id;
029import org.matsim.api.core.v01.network.Link;
030import org.matsim.core.api.internal.MatsimSomeWriter;
031import org.matsim.core.population.routes.NetworkRoute;
032import org.matsim.core.utils.collections.Tuple;
033import org.matsim.core.utils.geometry.CoordinateTransformation;
034import org.matsim.core.utils.geometry.transformations.IdentityTransformation;
035import org.matsim.core.utils.io.MatsimXmlWriter;
036import org.matsim.core.utils.io.UncheckedIOException;
037import org.matsim.core.utils.misc.Time;
038import org.matsim.pt.transitSchedule.api.Departure;
039import org.matsim.pt.transitSchedule.api.TransitLine;
040import org.matsim.pt.transitSchedule.api.TransitRoute;
041import org.matsim.pt.transitSchedule.api.TransitRouteStop;
042import org.matsim.pt.transitSchedule.api.TransitSchedule;
043import org.matsim.pt.transitSchedule.api.TransitStopFacility;
044
045/**
046 * Writes a transit schedule to a XML file in the format described by <code>transitSchedule_v1.dtd</code>.
047 *
048 * @author mrieser
049 */
050public class TransitScheduleWriterV1 extends MatsimXmlWriter implements MatsimSomeWriter {
051
052        private final CoordinateTransformation coordinateTransformation;
053
054        private final TransitSchedule schedule;
055
056        public TransitScheduleWriterV1(final TransitSchedule schedule) {
057                this( new IdentityTransformation() , schedule );
058        }
059
060        public TransitScheduleWriterV1(
061                        final CoordinateTransformation coordinateTransformation,
062                        final TransitSchedule schedule) {
063                this.coordinateTransformation = coordinateTransformation;
064                this.schedule = schedule;
065        }
066
067        public void write(final String filename) throws UncheckedIOException {
068                this.openFile(filename);
069                this.writeXmlHead();
070                this.writeDoctype(Constants.TRANSIT_SCHEDULE, "http://www.matsim.org/files/dtd/transitSchedule_v1.dtd");
071                this.writeStartTag(Constants.TRANSIT_SCHEDULE, null);
072
073                this.writeTransitStops();
074                for (TransitLine line : this.schedule.getTransitLines().values()) {
075                        writeTransitLine(line);
076                }
077                this.writeEndTag(Constants.TRANSIT_SCHEDULE);
078                this.close();
079        }
080
081        private void writeTransitStops() throws UncheckedIOException {
082                this.writeStartTag(Constants.TRANSIT_STOPS, null);
083
084                List<Tuple<String, String>> attributes = new ArrayList<Tuple<String, String>>(5);
085                for (TransitStopFacility stop : this.schedule.getFacilities().values()) {
086                        attributes.clear();
087                        attributes.add(this.createTuple(Constants.ID, stop.getId().toString()));
088                        final Coord coord = coordinateTransformation.transform( stop.getCoord() );
089                        attributes.add(this.createTuple("x", coord.getX()));
090                        attributes.add(this.createTuple("y", coord.getY()));
091                        if (stop.getLinkId() != null) {
092                                attributes.add(this.createTuple("linkRefId", stop.getLinkId().toString()));
093                        }
094                        if (stop.getName() != null) {
095                                attributes.add(this.createTuple("name", stop.getName()));
096                        }
097                        attributes.add(this.createTuple("isBlocking", stop.getIsBlockingLane()));
098                        this.writeStartTag(Constants.STOP_FACILITY, attributes, true);
099                }
100
101                this.writeEndTag(Constants.TRANSIT_STOPS);
102        }
103
104        private void writeTransitLine(final TransitLine line) throws UncheckedIOException {
105                List<Tuple<String, String>> attributes = new ArrayList<Tuple<String, String>>(1);
106                attributes.add(this.createTuple(Constants.ID, line.getId().toString()));
107                if (line.getName() != null) {
108                        attributes.add(this.createTuple(Constants.NAME, line.getName()));
109                }
110                this.writeStartTag(Constants.TRANSIT_LINE, attributes);
111
112                for (TransitRoute route : line.getRoutes().values()) {
113                        writeTransitRoute(route);
114                }
115
116                this.writeEndTag(Constants.TRANSIT_LINE);
117        }
118
119        private void writeTransitRoute(final TransitRoute route) throws UncheckedIOException {
120                List<Tuple<String, String>> attributes = new ArrayList<Tuple<String, String>>(1);
121                attributes.add(this.createTuple(Constants.ID, route.getId().toString()));
122                this.writeStartTag(Constants.TRANSIT_ROUTE, attributes);
123
124                if (route.getDescription() != null) {
125                        this.writeStartTag(Constants.DESCRIPTION, null);
126                        this.writeContent(route.getDescription(), false);
127                        this.writeEndTag(Constants.DESCRIPTION);
128                }
129
130                this.writeStartTag(Constants.TRANSPORT_MODE, null);
131                this.writeContent(route.getTransportMode(), false);
132                this.writeEndTag(Constants.TRANSPORT_MODE);
133
134                this.writeRouteProfile(route.getStops());
135                this.writeRoute(route.getRoute());
136                this.writeDepartures(route.getDepartures());
137
138                this.writeEndTag(Constants.TRANSIT_ROUTE);
139        }
140
141        private void writeRouteProfile(final List<TransitRouteStop> stops) throws UncheckedIOException {
142                this.writeStartTag(Constants.ROUTE_PROFILE, null);
143
144                // optimization: only create one List for multiple departures
145                List<Tuple<String, String>> attributes = new ArrayList<Tuple<String, String>>(4);
146                for (TransitRouteStop stop : stops) {
147                        attributes.clear();
148                        attributes.add(this.createTuple(Constants.REF_ID, stop.getStopFacility().getId().toString()));
149                        if (stop.getArrivalOffset() != Time.UNDEFINED_TIME) {
150                                attributes.add(this.createTimeTuple(Constants.ARRIVAL_OFFSET, stop.getArrivalOffset()));
151                        }
152                        if (stop.getDepartureOffset() != Time.UNDEFINED_TIME) {
153                                attributes.add(this.createTimeTuple(Constants.DEPARTURE_OFFSET, stop.getDepartureOffset()));
154                        }
155                        attributes.add(this.createTuple(Constants.AWAIT_DEPARTURE, String.valueOf(stop.isAwaitDepartureTime())));
156                        this.writeStartTag(Constants.STOP, attributes, true);
157                }
158
159                this.writeEndTag(Constants.ROUTE_PROFILE);
160        }
161
162        private void writeRoute(final NetworkRoute route) throws UncheckedIOException {
163                if (route != null) {
164                        this.writeStartTag(Constants.ROUTE, null);
165
166                        // optimization: only create one List for multiple departures
167                        List<Tuple<String, String>> attributes = new ArrayList<Tuple<String, String>>(1);
168                        attributes.add(this.createTuple(Constants.REF_ID, route.getStartLinkId().toString()));
169                        this.writeStartTag(Constants.LINK, attributes, true);
170
171                        for (Id<Link> linkId : route.getLinkIds()) {
172                                attributes.clear();
173                                attributes.add(this.createTuple(Constants.REF_ID, linkId.toString()));
174                                this.writeStartTag(Constants.LINK, attributes, true);
175                        }
176
177                        attributes.clear();
178                        attributes.add(this.createTuple(Constants.REF_ID, route.getEndLinkId().toString()));
179                        this.writeStartTag(Constants.LINK, attributes, true);
180
181                        this.writeEndTag(Constants.ROUTE);
182                }
183        }
184
185        private void writeDepartures(final Map<Id<Departure>, Departure> departures) throws UncheckedIOException {
186                this.writeStartTag(Constants.DEPARTURES, null);
187
188                // optimization: only create one List for multiple departures
189                List<Tuple<String, String>> attributes = new ArrayList<Tuple<String, String>>(3);
190
191                for (Departure dep : departures.values()) {
192                        attributes.clear();
193                        attributes.add(this.createTuple(Constants.ID, dep.getId().toString()));
194                        attributes.add(this.createTimeTuple(Constants.DEPARTURE_TIME, dep.getDepartureTime()));
195                        if (dep.getVehicleId() != null) {
196                                attributes.add(this.createTuple(Constants.VEHICLE_REF_ID, dep.getVehicleId().toString()));
197                        }
198                        this.writeStartTag(Constants.DEPARTURE, attributes, true);
199                }
200
201                this.writeEndTag(Constants.DEPARTURES);
202        }
203}
