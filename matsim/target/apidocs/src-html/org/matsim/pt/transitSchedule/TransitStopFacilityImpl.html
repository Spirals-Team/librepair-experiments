021package org.matsim.pt.transitSchedule;
022
023import java.util.Map;
024
025import org.matsim.api.core.v01.Coord;
026import org.matsim.api.core.v01.Customizable;
027import org.matsim.api.core.v01.Id;
028import org.matsim.api.core.v01.network.Link;
029import org.matsim.core.scenario.CustomizableUtils;
030import org.matsim.pt.transitSchedule.api.TransitStopArea;
031import org.matsim.pt.transitSchedule.api.TransitStopFacility;
032import org.matsim.utils.objectattributes.attributable.Attributes;
033
034/**
035 * A facility (infrastructure) describing a public transport stop.
036 *
037 * @author mrieser
038 */
039public class TransitStopFacilityImpl implements TransitStopFacility {
040        
041        private final Id<TransitStopFacility> id;
042        private Id<TransitStopArea> stopAreaId;
043        private Coord coord;
044        private Id<Link> linkId = null;
045        private final boolean isBlockingLane;
046        private String name = null;
047        private Customizable customizableDelegate;
048        private final Attributes attributes = new Attributes();
049
050        protected TransitStopFacilityImpl(final Id<TransitStopFacility> id, final Coord coord, final boolean isBlockingLane) {
051                this.id = id;
052                this.coord = coord;
053                this.isBlockingLane = isBlockingLane;
054        }
055
056        @Override
057        public void setLinkId(final Id<Link> linkId) {
058                this.linkId = linkId;
059        }
060
061        @Override
062        public Id<Link> getLinkId() {
063                return this.linkId;
064        }
065
066        @Override
067        public Coord getCoord() {
068                return this.coord;
069        }
070        
071        @Override
072        public void setCoord(Coord coord) {
073                this.coord = coord;
074        }
075
076        @Override
077        public Id<TransitStopFacility> getId() {
078                return this.id;
079        }
080
081        @Override
082        public boolean getIsBlockingLane() {
083                return this.isBlockingLane;
084        }
085
086        @Override
087        public void setName(final String name) {
088                this.name = name;
089        }
090
091        @Override
092        public String getName() {
093                return this.name;
094        }
095
096        @Override
097        public String toString() {
098                return "TransitStopFacilityImpl_" + this.id;
099        }
100
101        @Override
102        public Id<TransitStopArea> getStopAreaId() {
103                return this.stopAreaId;
104        }
105
106        @Override
107        public void setStopAreaId(Id<TransitStopArea> stopAreaId) {
108                this.stopAreaId = stopAreaId;
109        }
110        
111        @Override
112        public Map<String, Object> getCustomAttributes() {
113                if (this.customizableDelegate == null) {
114                        this.customizableDelegate = CustomizableUtils.createCustomizable();
115                }
116                return this.customizableDelegate.getCustomAttributes();
117        }
118
119        @Override
120        public Attributes getAttributes() {
121                return this.attributes;
122        }
123}
