021package org.matsim.pt.transitSchedule.api;
022
023import org.matsim.core.api.internal.MatsimSomeWriter;
024import org.matsim.core.utils.geometry.CoordinateTransformation;
025import org.matsim.core.utils.geometry.transformations.IdentityTransformation;
026import org.matsim.core.utils.io.UncheckedIOException;
027import org.matsim.pt.transitSchedule.TransitScheduleWriterV1;
028import org.matsim.pt.transitSchedule.TransitScheduleWriterV2;
029
030/**
031 * Writes {@link TransitSchedule}s to file in one of the
032 * supported file formats.
033 *
034 * @author mrieser
035 */
036public class TransitScheduleWriter implements MatsimSomeWriter {
037
038        private final TransitSchedule schedule;
039        private final CoordinateTransformation transformation;
040
041        public TransitScheduleWriter(final TransitSchedule schedule) {
042                this( new IdentityTransformation() , schedule );
043        }
044
045        public TransitScheduleWriter(
046                        final CoordinateTransformation transformation,
047                        final TransitSchedule schedule) {
048                this.transformation = transformation;
049                this.schedule = schedule;
050        }
051
052        /**
053         * Writes the transit schedule to the specified file in the most
054         * current file format (currently V2).
055         *
056         * @param filename
057         * @throws UncheckedIOException
058         * @see {@link #writeFileV2(String)}
059         */
060        public void writeFile(final String filename) throws UncheckedIOException {
061                writeFileV2(filename);
062        }
063
064        /**
065         * Writes the transit schedule to the specified file in the file
066         * format specified by <tt>transitSchedule_v1.dtd</tt>
067         *
068         * @param filename
069         * @throws UncheckedIOException
070         */
071        public void writeFileV1(final String filename) throws UncheckedIOException {
072                new TransitScheduleWriterV1(this.transformation, this.schedule).write(filename);
073        }
074
075        /**
076         * Writes the transit schedule to the specified file in the file
077         * format specified by <tt>transitSchedule_v2.dtd</tt>
078         *
079         * @param filename
080         * @throws UncheckedIOException
081         */
082        public void writeFileV2(final String filename) throws UncheckedIOException {
083                new TransitScheduleWriterV2(this.transformation, this.schedule).write(filename);
084        }
085}
