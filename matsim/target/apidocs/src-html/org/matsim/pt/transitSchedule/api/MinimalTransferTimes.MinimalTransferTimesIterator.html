020package org.matsim.pt.transitSchedule.api;
021
022import org.matsim.api.core.v01.Id;
023
024/**
025 * Container class to manage minimal transfer times between two {@link TransitStopFacility}.
026 * Transfer times are *not* bidirectional and *not* transitive. This means:
027 * <ul>
028 *     <li>A transfer time defined from stop A to stop B does not define a transfer time from stop B to stop A.</li>
029 *     <li>A transfer time defined from stop A to stop B, and another defined from stop B to stop C, does not define a transfer time from stop A to stop C.</li>
030 * </ul>
031 * Both cases, the reverse transfer from stop B to stop A, and the indirect transfer from stop A to stop C via stop B,
032 * must explicitly be set in order to be returned in {@link #get(Id, Id)}.
033 *
034 * @author mrieser / SBB
035 */
036public interface MinimalTransferTimes {
037
038        /**
039         * Sets the minimal transfer time in seconds needed to transfer from <code>fromStop</code> to <code>toStop</code>.
040         * @param fromStop
041         * @param toStop
042         * @param seconds the minimal transfer time, in seconds
043         * @return the minimal transfer time previously assigned between the two stops, <code>Double.NaN</code> if none was specified.
044         */
045        double set(Id<TransitStopFacility> fromStop, Id<TransitStopFacility> toStop, double seconds);
046
047        /**
048         * @param fromStop
049         * @param toStop
050         * @return the minimal transfer time between the two stops if defined, <code>Double.NaN</code> if none is set.
051         */
052        double get(Id<TransitStopFacility> fromStop, Id<TransitStopFacility> toStop);
053
054        /**
055         * @param fromStop
056         * @param toStop
057         * @param defaultSeconds
058         * @return the minimal transfer time between the two stops if defined, <code>defaultSeconds</code> if none is set.
059         */
060        double get(Id<TransitStopFacility> fromStop, Id<TransitStopFacility> toStop, double defaultSeconds);
061
062        /**
063         * Removes the minimal transfer time between the two stops if there was one set.
064         * @param fromStop
065         * @param toStop
066         * @return the previously set minimal transfer time, or <code>Double.NaN</code> if none was set.
067         */
068        double remove(Id<TransitStopFacility> fromStop, Id<TransitStopFacility> toStop);
069
070        /**
071         * @return an iterator to iterate over all minimal transfer times set.
072         */
073        MinimalTransferTimesIterator iterator();
074
075        interface MinimalTransferTimesIterator {
076                boolean hasNext();
077
078                /**
079                 * advances the iterator to the next element
080                 */
081                void next();
082
083                /**
084                 * @return the Id of the fromStop of the current iterator element.
085                 */
086                Id<TransitStopFacility> getFromStopId();
087
088                /**
089                 * @return the Id of the toStop of the current iterator element.
090                 */
091                Id<TransitStopFacility> getToStopId();
092
093                /**
094                 * @return the minimal transfer time in seconds of the current iterator element.
095                 */
096                double getSeconds();
097        }
098}
