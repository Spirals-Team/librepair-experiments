051/**
052 * Reads a transit schedule from a XML file in the format described by <code>transitSchedule_v1.dtd</code>.
053 *
054 * @author mrieser
055 */
056public class TransitScheduleReaderV2 extends MatsimXmlParser {
057
058        private final TransitSchedule schedule;
059        private final RouteFactories routeFactory;
060
061        private TransitLine currentTransitLine = null;
062        private TempTransitRoute currentTransitRoute = null;
063        private TempRoute currentRouteProfile = null;
064        private Departure currentDeparture = null;
065
066        private final AttributesXmlReaderDelegate attributesDelegate = new AttributesXmlReaderDelegate();
067        private org.matsim.utils.objectattributes.attributable.Attributes currentAttributes = null;
068
069        private final CoordinateTransformation coordinateTransformation;
070        private final StringCache cache = new StringCache();
071
072        public TransitScheduleReaderV2(final TransitSchedule schedule, final RouteFactories routeFactory) {
073                this( new IdentityTransformation() , schedule , routeFactory );
074        }
075
076        public TransitScheduleReaderV2(final Scenario scenario) {
077                this( scenario.getTransitSchedule(),
078                                scenario.getPopulation().getFactory().getRouteFactories() );
079        }
080
081        public TransitScheduleReaderV2(
082                        final CoordinateTransformation coordinateTransformation,
083                        final Scenario scenario) {
084                this( coordinateTransformation,
085                                scenario.getTransitSchedule(),
086                                scenario.getPopulation().getFactory().getRouteFactories() );
087        }
088
089        public TransitScheduleReaderV2(
090                        CoordinateTransformation coordinateTransformation,
091                        TransitSchedule schedule,
092                        RouteFactories routeFactory) {
093                this.coordinateTransformation = coordinateTransformation;
094                this.schedule = schedule;
095                this.routeFactory = routeFactory;
096        }
097
098        @Override
099        public void startTag(final String name, final Attributes atts, final Stack<String> context) {
100                if (Constants.STOP_FACILITY.equals(name)) {
101                        boolean isBlocking = Boolean.parseBoolean(atts.getValue(Constants.IS_BLOCKING));
102                        Coord coord = atts.getValue(Constants.Z) == null ?
103                                        new Coord(Double.parseDouble(atts.getValue(Constants.X)), Double.parseDouble(atts.getValue(Constants.Y))) :
104                                        new Coord(Double.parseDouble(atts.getValue(Constants.X)), Double.parseDouble(atts.getValue(Constants.Y)), Double.parseDouble(atts.getValue(Constants.Z)));
105                        TransitStopFacility stop =
106                                        this.schedule.getFactory().createTransitStopFacility(
107                                                        Id.create(atts.getValue(Constants.ID), TransitStopFacility.class),
108                                                        this.coordinateTransformation.transform(coord),
109                                                        isBlocking);
110                        this.currentAttributes = stop.getAttributes();
111                        if (atts.getValue(Constants.LINK_REF_ID) != null) {
112                                Id<Link> linkId = Id.create(atts.getValue(Constants.LINK_REF_ID), Link.class);
113                                stop.setLinkId(linkId);
114                        }
115                        if (atts.getValue(Constants.NAME) != null) {
116                                stop.setName(this.cache.get(atts.getValue(Constants.NAME)));
117                        }
118                        if (atts.getValue(Constants.STOP_AREA_ID) != null) {
119                                stop.setStopAreaId(Id.create(atts.getValue(Constants.STOP_AREA_ID), TransitStopArea.class));
120                        }
121                        this.schedule.addStopFacility(stop);
122                } else if (Constants.TRANSIT_LINE.equals(name)) {
123                        Id<TransitLine> id = Id.create(atts.getValue(Constants.ID), TransitLine.class);
124                        this.currentTransitLine = this.schedule.getFactory().createTransitLine(id);
125                        this.currentAttributes = this.currentTransitLine.getAttributes();
126                        if (atts.getValue(Constants.NAME) != null) {
127                                this.currentTransitLine.setName(atts.getValue(Constants.NAME));
128                        }
129                        this.schedule.addTransitLine(this.currentTransitLine);
130                } else if (Constants.TRANSIT_ROUTE.equals(name)) {
131                        Id<TransitRoute> id = Id.create(atts.getValue(Constants.ID), TransitRoute.class);
132                        this.currentTransitRoute = new TempTransitRoute(id);
133                        this.currentAttributes = this.currentTransitRoute.attributes;
134                } else if (Constants.DEPARTURE.equals(name)) {
135                        Id<Departure> id = Id.create(atts.getValue(Constants.ID), Departure.class);
136                        this.currentDeparture = new DepartureImpl(id, Time.parseTime(atts.getValue("departureTime")));
137                        this.currentAttributes = this.currentDeparture.getAttributes();
138                        String vehicleRefId = atts.getValue(Constants.VEHICLE_REF_ID);
139                        if (vehicleRefId != null) {
140                                this.currentDeparture.setVehicleId(Id.create(vehicleRefId, Vehicle.class));
141                        }
142                        this.currentTransitRoute.departures.put(id, this.currentDeparture);
143                } else if (Constants.ROUTE_PROFILE.equals(name)) {
144                        this.currentRouteProfile = new TempRoute();
145                } else if (Constants.LINK.equals(name)) {
146                        String linkStr = atts.getValue(Constants.REF_ID);
147                        if (!linkStr.contains(" ")) {
148                                this.currentRouteProfile.addLink(Id.create(linkStr, Link.class));
149                        } else {
150                                String[] links = linkStr.split(" ");
151                                for (int i = 0; i < links.length; i++) {
152                                        this.currentRouteProfile.addLink(Id.create(links[i], Link.class));
153                                }
154                        }
155                } else if (Constants.STOP.equals(name)) {
156                        Id<TransitStopFacility> id = Id.create(atts.getValue(Constants.REF_ID), TransitStopFacility.class);
157                        TransitStopFacility facility = this.schedule.getFacilities().get(id);
158                        if (facility == null) {
159                                throw new RuntimeException("no stop/facility with id " + atts.getValue(Constants.REF_ID));
160                        }
161                        TempStop stop = new TempStop(facility);
162                        String arrival = atts.getValue(Constants.ARRIVAL_OFFSET);
163                        String departure = atts.getValue(Constants.DEPARTURE_OFFSET);
164                        if (arrival != null) {
165                                stop.arrival = Time.parseTime(arrival);
166                        }
167                        if (departure != null) {
168                                stop.departure = Time.parseTime(departure);
169                        }
170                        stop.awaitDeparture = Boolean.parseBoolean(atts.getValue(Constants.AWAIT_DEPARTURE));
171                        this.currentTransitRoute.stops.add(stop);
172                } else if (Constants.RELATION.equals(name)) {
173                        Id<TransitStopFacility> fromStop = Id.create(atts.getValue(Constants.FROM_STOP), TransitStopFacility.class);
174                        Id<TransitStopFacility> toStop = Id.create(atts.getValue(Constants.TO_STOP), TransitStopFacility.class);
175                        double transferTime = Time.parseTime(atts.getValue(Constants.TRANSFER_TIME));
176                        this.schedule.getMinimalTransferTimes().set(fromStop, toStop, transferTime);
177                } else if (Constants.ATTRIBUTE.equals(name)) {
178                        this.attributesDelegate.startTag(name, atts, context, this.currentAttributes);
179                } else if (Constants.ATTRIBUTES.equals(name)) {
180                        this.attributesDelegate.startTag(name, atts, context, this.currentAttributes);
181                } else if (Constants.TRANSIT_SCHEDULE.equals(name)) {
182                        this.currentAttributes = this.schedule.getAttributes();
183                }
184        }
185
186        @Override
187        public void endTag(final String name, final String content, final Stack<String> context) {
188                if (Constants.DEPARTURE.equals(name)) {
189                        this.currentDeparture = null;
190                }
191                if (Constants.DESCRIPTION.equals(name) && Constants.TRANSIT_ROUTE.equals(context.peek())) {
192                        this.currentTransitRoute.description = content;
193                } else if (Constants.TRANSPORT_MODE.equals(name)) {
194                        this.currentTransitRoute.mode = content.intern();
195                } else if (Constants.DEPARTURE.equals(name)) {
196                        this.currentDeparture = null;
197                } else if (Constants.TRANSIT_ROUTE.equals(name)) {
198                        List<TransitRouteStop> stops = new ArrayList<>(this.currentTransitRoute.stops.size());
199                        for (TempStop tStop : this.currentTransitRoute.stops) {
200                                TransitRouteStopImpl routeStop = new TransitRouteStopImpl(tStop.stop, tStop.arrival, tStop.departure);
201                                stops.add(routeStop);
202                                routeStop.setAwaitDepartureTime(tStop.awaitDeparture);
203                        }
204                        NetworkRoute route = null;
205                        if (this.currentRouteProfile.firstLinkId != null) {
206                                if (this.currentRouteProfile.lastLinkId == null) {
207                                        this.currentRouteProfile.lastLinkId = this.currentRouteProfile.firstLinkId;
208                                }
209                                route = this.routeFactory.createRoute(NetworkRoute.class, this.currentRouteProfile.firstLinkId, this.currentRouteProfile.lastLinkId);
210                                route.setLinkIds(this.currentRouteProfile.firstLinkId, this.currentRouteProfile.linkIds, this.currentRouteProfile.lastLinkId);
211                        }
212                        TransitRoute transitRoute = this.schedule.getFactory().createTransitRoute(this.currentTransitRoute.id, route, stops, this.currentTransitRoute.mode);
213                        transitRoute.setDescription(this.currentTransitRoute.description);
214                        for (Departure departure : this.currentTransitRoute.departures.values()) {
215                                transitRoute.addDeparture(departure);
216                        }
217                        AttributesUtils.copyTo(this.currentTransitRoute.attributes, transitRoute.getAttributes());
218                        this.currentTransitLine.addRoute(transitRoute);
219                        this.currentTransitRoute = null;
220                } else if (Constants.TRANSIT_LINE.equals(name)) {
221                        this.currentTransitLine = null;
222                } else if (Constants.ATTRIBUTE.equals(name)) {
223                        this.attributesDelegate.endTag(name, content, context);
224                } else if (Constants.ATTRIBUTES.equals(name)) {
225                        this.currentAttributes = null;
226                }
227        }
228
229        private static class TempTransitRoute {
230                protected final Id<TransitRoute> id;
231                protected final org.matsim.utils.objectattributes.attributable.Attributes attributes = new org.matsim.utils.objectattributes.attributable.Attributes();
232                protected String description = null;
233                protected Map<Id<Departure>, Departure> departures = new LinkedHashMap<>();
234                /*package*/ List<TempStop> stops = new ArrayList<>();
235                /*package*/ String mode = null;
236
237                protected TempTransitRoute(final Id<TransitRoute> id) {
238                        this.id = id;
239                }
240        }
241
242        private static class TempStop {
243                protected final TransitStopFacility stop;
244                protected double departure = Time.UNDEFINED_TIME;
245                protected double arrival = Time.UNDEFINED_TIME;
246                protected boolean awaitDeparture = false;
247
248                protected TempStop(final TransitStopFacility stop) {
249                        this.stop = stop;
250                }
251        }
252
253        private static class TempRoute {
254                /*package*/ List<Id<Link>> linkIds = new ArrayList<>();
255                /*package*/ Id<Link> firstLinkId = null;
256                /*package*/ Id<Link> lastLinkId = null;
257
258                protected TempRoute() {
259                        // public constructor for private inner class
260                }
261
262                protected void addLink(final Id<Link> linkId) {
263                        if (this.firstLinkId == null) {
264                                this.firstLinkId = linkId;
265                        } else if (this.lastLinkId == null) {
266                                this.lastLinkId = linkId;
267                        } else {
268                                this.linkIds.add(this.lastLinkId);
269                                this.lastLinkId = linkId;
270                        }
271                }
272
273        }
274
275        private static class StringCache {
276
277                private ConcurrentHashMap<String, String> cache = new ConcurrentHashMap<>(10000);
278
279                /**
280                 * Returns the cached version of the given String. If the strings was
281                 * not yet in the cache, it is added and returned as well.
282                 *
283                 * @param string
284                 * @return cached version of string
285                 */
286                public String get(final String string) {
287                        if (string == null) {
288                                return null;
289                        }
290                        String s = this.cache.putIfAbsent(string, string);
291                        if (s == null) {
292                                return string;
293                        }
294                        return s;
295                }
296        }
