020package org.matsim.pt.config;
021
022import java.util.Map;
023
024import org.matsim.core.config.ReflectiveConfigGroup;
025
026public class TransitRouterConfigGroup extends ReflectiveConfigGroup {
027
028        public static final String GROUP_NAME = "transitRouter";
029
030        /*package*/ static final String SEARCH_RADIUS = "searchRadius";
031        /*package*/ static final String EXTENSION_RADIUS = "extensionRadius";
032        /*package*/ static final String MAX_BEELINE_WALK_CONNECTION_DISTANCE = "maxBeelineWalkConnectionDistance";
033        /*package*/ static final String ADDITIONAL_TRANSFER_TIME = "additionalTransferTime";
034
035        private double searchRadius = 1000.0;
036        private double extensionRadius = 200.0;
037        private double maxBeelineWalkConnectionDistance = 100.0;
038        private double additionalTransferTime = 0.0;
039
040        private double directWalkFactor = 1. ;
041
042        public TransitRouterConfigGroup() {
043                super(GROUP_NAME);
044        }
045
046        private static final String DIRECT_WALK_FACTOR="directWalkFactor" ;
047        static final String DIRECT_WALK_FACTOR_CMT = "Factor with which direct walk generalized cost is multiplied before it is compared "
048                        + "to the pt generalized cost.  Set to a very high value to reduce direct walk results." ;
049        @Override
050        public Map<String, String> getComments() {
051                Map<String, String> comments = super.getComments();
052                comments.put(SEARCH_RADIUS, "the radius in which stop locations are searched, given a start or target coordinate");
053                comments.put(EXTENSION_RADIUS, "step size to increase searchRadius if no stops are found");
054                comments.put(MAX_BEELINE_WALK_CONNECTION_DISTANCE, "maximum beeline distance between stops that agents could transfer to by walking");
055                comments.put(ADDITIONAL_TRANSFER_TIME, "additional time the router allocates when a line switch happens. Can be interpreted as a 'safety' time that agents need to safely transfer from one line to another");
056                comments.put(DIRECT_WALK_FACTOR, DIRECT_WALK_FACTOR_CMT);
057                return comments;
058        }
059
060        @StringSetter( SEARCH_RADIUS )
061        public void setSearchRadius(final double searchRadius) {
062                testForLocked() ;
063                this.searchRadius = searchRadius;
064        }
065
066        @StringGetter( SEARCH_RADIUS )
067        public double getSearchRadius() {
068                return this.searchRadius;
069        }
070
071        @StringSetter( EXTENSION_RADIUS )
072        public void setExtensionRadius(final double extensionRadius) {
073                testForLocked() ;
074                this.extensionRadius = extensionRadius;
075        }
076
077        @StringGetter( EXTENSION_RADIUS )
078        public double getExtensionRadius() {
079                return this.extensionRadius;
080        }
081
082        @StringSetter( MAX_BEELINE_WALK_CONNECTION_DISTANCE )
083        public void setMaxBeelineWalkConnectionDistance(final double maxBeelineWalkConnectionDistance) {
084                testForLocked() ;
085                this.maxBeelineWalkConnectionDistance = maxBeelineWalkConnectionDistance;
086        }
087
088        @StringGetter( MAX_BEELINE_WALK_CONNECTION_DISTANCE )
089        public double getMaxBeelineWalkConnectionDistance() {
090                return this.maxBeelineWalkConnectionDistance;
091        }
092
093        @StringSetter( ADDITIONAL_TRANSFER_TIME )
094        public void setAdditionalTransferTime(final double additionalTransferTime) {
095                testForLocked() ;
096                this.additionalTransferTime = additionalTransferTime;
097        }
098
099        @StringGetter( ADDITIONAL_TRANSFER_TIME )
100        public double getAdditionalTransferTime() {
101                return this.additionalTransferTime;
102        }
103
104        /**
105         * {@value #DIRECT_WALK_FACTOR_CMT}
106         */
107        @StringGetter(DIRECT_WALK_FACTOR)
108        public double getDirectWalkFactor() {
109                return this.directWalkFactor ;
110        }
111        /**
112         * {@value #DIRECT_WALK_FACTOR_CMT}
113         */
114        @StringSetter(DIRECT_WALK_FACTOR)
115        public final void setDirectWalkFactor(double directWalkFactor) {
116                this.directWalkFactor = directWalkFactor;
117        }
118}
