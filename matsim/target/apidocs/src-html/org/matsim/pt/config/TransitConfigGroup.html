021package org.matsim.pt.config;
022
023import java.net.MalformedURLException;
024import java.net.URL;
025import java.util.Collections;
026import java.util.HashSet;
027import java.util.LinkedHashSet;
028import java.util.Map;
029import java.util.Set;
030
031import org.matsim.api.core.v01.TransportMode;
032import org.matsim.core.config.ConfigGroup;
033import org.matsim.core.config.ReflectiveConfigGroup;
034import org.matsim.core.config.ReflectiveConfigGroup.StringGetter;
035import org.matsim.core.utils.collections.CollectionUtils;
036
037/**
038 * @author mrieser
039 */
040public class TransitConfigGroup extends ReflectiveConfigGroup {
041
042        public static final String GROUP_NAME = "transit";
043
044        /*package*/ static final String TRANSIT_SCHEDULE_FILE = "transitScheduleFile";
045        private static final String TRANSIT_LINES_ATTRIBUTES = "transitLinesAttributesFile";
046        private static final String TRANSIT_STOPS_ATTRIBUTES = "transitStopsAttributesFile";
047        /*package*/ static final String VEHICLES_FILE = "vehiclesFile";
048        /*package*/ static final String TRANSIT_MODES = "transitModes";
049        private static final String SCHEDULE_CRS = "inputScheduleCRS";
050
051        private String transitScheduleFile = null;
052        private String vehiclesFile = null;
053        private String transitLinesAttributesFile = null;
054        private String transitStopsAttributesFile = null;
055        private String inputScheduleCRS = null;
056
057        private Set<String> transitModes;
058        
059        // ---
060        private static final String USE_TRANSIT = "useTransit";
061        private boolean useTransit = false;
062
063        // ---
064
065        public TransitConfigGroup() {
066                super(GROUP_NAME);
067                Set<String> modes = new LinkedHashSet<>();
068                modes.add(TransportMode.pt);
069                this.transitModes = Collections.unmodifiableSet(modes);
070        }
071
072        @StringSetter( TRANSIT_MODES )
073        private void setTransitModes( final String value ) {
074                this.transitModes = Collections.unmodifiableSet(CollectionUtils.stringToSet(value));
075        }
076
077        @StringGetter( TRANSIT_MODES )
078        private String getTransitModeString() {
079                boolean isFirst = true;
080                StringBuilder str = new StringBuilder();
081                for (String mode : this.transitModes) {
082                        if (!isFirst) {
083                                str.append(',');
084                        }
085                        str.append(mode);
086                        isFirst = false;
087                }
088                return str.toString();
089        }
090
091        @Override
092        public Map<String, String> getComments() {
093                Map<String, String> comments = super.getComments();
094                comments.put(TRANSIT_SCHEDULE_FILE, "Input file containing the transit schedule to be simulated.");
095                comments.put(VEHICLES_FILE, "Input file containing the vehicles used by the departures in the transit schedule.");
096                comments.put(TRANSIT_MODES, "Comma-separated list of transportation modes that are handled as transit. Defaults to 'pt'.");
097                comments.put(TRANSIT_LINES_ATTRIBUTES, "Optional input file containing additional attributes for transit lines, stored as ObjectAttributes.");
098                comments.put(TRANSIT_STOPS_ATTRIBUTES, "Optional input file containing additional attributes for transit stop facilities, stored as ObjectAttributes.");
099                comments.put(USE_TRANSIT, "Set this parameter to true if transit should be simulated, false if not.");
100
101                comments.put( SCHEDULE_CRS , "The Coordinates Reference System in which the coordinates are expressed in the input file." +
102                                " At import, the coordinates will be converted to the coordinate system defined in \"global\", and will" +
103                                "be converted back at export. If not specified, no conversion happens." );
104                return comments;
105        }
106
107        @StringSetter( TRANSIT_SCHEDULE_FILE )
108        public void setTransitScheduleFile(final String filename) {
109                this.testForLocked();
110                this.transitScheduleFile = filename;
111        }
112
113        @StringGetter( TRANSIT_SCHEDULE_FILE )
114        public String getTransitScheduleFile() {
115                return this.transitScheduleFile;
116        }
117
118        public URL getTransitScheduleFileURL(URL context) {
119                return ConfigGroup.getInputFileURL(context, getTransitScheduleFile() ) ;
120        }
121
122        @StringSetter( VEHICLES_FILE )
123        public void setVehiclesFile(final String filename) {
124                this.vehiclesFile = filename;
125        }
126
127        @StringGetter( VEHICLES_FILE )
128        public String getVehiclesFile() {
129                return this.vehiclesFile;
130        }
131
132        public URL getVehiclesFileURL(URL context) {
133                return ConfigGroup.getInputFileURL(context, getVehiclesFile() ) ;
134        }
135
136        public void setTransitModes(final Set<String> modes) {
137                this.transitModes = Collections.unmodifiableSet(new HashSet<>(modes));
138        }
139
140        public Set<String> getTransitModes() {
141                return this.transitModes;
142        }
143
144        @StringGetter( TRANSIT_LINES_ATTRIBUTES )
145        public String getTransitLinesAttributesFile() {
146                return transitLinesAttributesFile;
147        }
148        
149        @StringSetter( TRANSIT_LINES_ATTRIBUTES )
150        public void setTransitLinesAttributesFile(final String transitLinesAttributesFile) {
151                this.transitLinesAttributesFile = transitLinesAttributesFile;
152        }
153
154        @StringGetter( TRANSIT_STOPS_ATTRIBUTES )
155        public String getTransitStopsAttributesFile() {
156                return this.transitStopsAttributesFile;
157        }
158        public URL getTransitStopsAttributesFileURL(URL context) {
159                return ConfigGroup.getInputFileURL(context, getTransitStopsAttributesFile()) ;
160        }
161        
162        @StringSetter( TRANSIT_STOPS_ATTRIBUTES )
163        public void setTransitStopsAttributesFile(final String transitStopsAttributesFile) {
164                this.transitStopsAttributesFile = transitStopsAttributesFile;
165        }
166        
167        @StringGetter( USE_TRANSIT )
168        public boolean isUseTransit() {
169                return this.useTransit;
170        }
171        @StringSetter( USE_TRANSIT )
172        public void setUseTransit( boolean val ) {
173                this.testForLocked();
174                this.useTransit = val ;
175        }
176
177
178        @StringGetter( SCHEDULE_CRS )
179        public String getInputScheduleCRS() {
180                return inputScheduleCRS;
181        }
182
183        @StringSetter( SCHEDULE_CRS )
184        public void setInputScheduleCRS(String inputScheduleCRS) {
185                this.inputScheduleCRS = inputScheduleCRS;
186        }
187}
