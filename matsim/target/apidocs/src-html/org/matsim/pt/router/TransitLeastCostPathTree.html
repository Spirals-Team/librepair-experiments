043/**
044 * This class is based on and similar to org.matsim.pt.router.MultiNodeDijkstra
045 *
046 * In contrast to org.matsim.pt.router.MultiNodeDijkstra, however, it stores the last
047 * LeastCostPathTree. It is, therefore, much faster than it, in cases where many routes
048 * starting with the same fromNode are calculated subsequently as every route from that
049 * fromCoord can be retrieved now without having to compute the tree again.
050 *
051 * Call createLeastCostPathTree(fromNodes, person, fromCoord) in order to create and cache the
052 * LeastCostPathTree. Every following call of getPath(toNodes) will return the leastCostPath
053 * from the fromNodes to the toNodes.
054 *
055 * The fromCoord you passed in createLeastCostPathTree operates as a primary key to determine
056 * if you are working with the LeastCostPathTree you think you are. You can just pass the
057 * original requested origin here.
058 *
059 * Please keep in mind, that you are responsible to order all queries in a way such that all
060 * those originating in the same fromCoord are done subsequently. Otherwise, the mentioned
061 * efficiency gain with not take effect.
062 *
063 * @author gthunig
064 */
065public class TransitLeastCostPathTree {

090        public TransitLeastCostPathTree(final Network network, final TransitTravelDisutility costFunction,
091                                                                        final TravelTime timeFunction,
092                                                                        final Map<Node, InitialNode> fromNodes, final Person person) {
093                this.network = network;
094                this.costFunction = costFunction;
095                this.timeFunction = timeFunction;
096
097                this.nodeData = new HashMap<>((int)(network.getNodes().size() * 1.1), 0.95f);
098
099                //create tree
100                this.resetNetworkVisited();
101                this.person = person;
102                this.customDataManager.reset();
103                this.fromNodes = fromNodes;
104
105                pendingNodes = (RouterPriorityQueue<Node>) createRouterPriorityQueue();
106                for (Map.Entry<Node, InitialNode> entry : fromNodes.entrySet()) {
107                        DijkstraNodeData data = getData(entry.getKey());
108                        visitNode(entry.getKey(), data, pendingNodes, entry.getValue().initialTime, entry.getValue().initialCost, null);
109                }
110
111                // do the real work
112                while (pendingNodes.size() > 0) {
113                        Node outNode = pendingNodes.poll();
114                        relaxNode(outNode, pendingNodes);
115                }
116        }

118        public TransitLeastCostPathTree(final Network network, final TransitTravelDisutility costFunction,
119                                                                        final TravelTime timeFunction,
120                                                                        final Map<Node, InitialNode> fromNodes, final Map<Node, InitialNode> toNodes,
121                                                                        final Person person) {
122                this.network = network;
123                this.costFunction = costFunction;
124                this.timeFunction = timeFunction;
125
126                this.nodeData = new HashMap<>((int)(network.getNodes().size() * 1.1), 0.95f);
127
128                //create tree
129                this.resetNetworkVisited();
130                this.person = person;
131                this.customDataManager.reset();
132                this.fromNodes = fromNodes;
133
134                pendingNodes = (RouterPriorityQueue<Node>) createRouterPriorityQueue();
135                for (Map.Entry<Node, InitialNode> entry : fromNodes.entrySet()) {
136                        DijkstraNodeData data = getData(entry.getKey());
137                        visitNode(entry.getKey(), data, pendingNodes, entry.getValue().initialTime, entry.getValue().initialCost, null);
138                }
139
140                expandNodeData(toNodes);
141        }

149        /**
150         * Resets all nodes in the network as if they have not been visited yet.
151         */
152        private void resetNetworkVisited() {
153                for (Node node : this.network.getNodes().values()) {
154                        DijkstraNodeData data = getData(node);
155                        data.resetVisited();
156                }
157        }

159        private void expandNodeData(final Map<Node, InitialNode> toNodes) {
160                Set<Node> endNodes = new HashSet<>(toNodes.keySet());
161                double minCost = Double.POSITIVE_INFINITY;
162
163                // do the real work
164                while (endNodes.size() > 0) {
165                        Node outNode = pendingNodes.poll();
166
167                        if (outNode == null) {
168                                // seems we have no more nodes left, but not yet reached all endNodes...
169                                endNodes.clear();
170                        } else {
171                                DijkstraNodeData data = getData(outNode);
172                                boolean isEndNode = endNodes.remove(outNode);
173                                if (isEndNode) {
174                                        InitialNode initData = toNodes.get(outNode);
175                                        double cost = data.getCost() + initData.initialCost;
176                                        if (cost < minCost) {
177                                                minCost = cost;
178                                        }
179                                }
180                                if (data.getCost() > minCost) {
181                                        endNodes.clear(); // we can't get any better now
182                                } else {
183                                        relaxNode(outNode, pendingNodes);
184                                }
185                        }
186                }
187        }

189        /**
190         * Method to request the passenger route from the (cached) fromNodes to the passed toNodes.
191         * Should only be requested after calling createTransitLeastCostPathTree().
192         *
193         * @param toNodes
194         *          The nodes that are the next stops to the toCoord and you like to route to.
195         *
196         * @return
197         *          the transitPassengerRoute between the fromNode and the toNode.
198         *          Will be null if the route could not be found.
199         */
200        public TransitPassengerRoute getTransitPassengerRoute(final Map<Node, InitialNode> toNodes) {
201                //find the best node
202                double minCost = Double.POSITIVE_INFINITY;
203                Node minCostNode = null;
204                for (Map.Entry<Node,InitialNode> e : toNodes.entrySet()) {
205                        Node currentNode = e.getKey();
206                        DijkstraNodeData r = this.nodeData.get(currentNode.getId());
207                        if (r == null) {
208                                expandNodeData(toNodes);
209                        }
210                        DijkstraNodeData data = getData(currentNode);
211                        InitialNode initData = e.getValue();
212                        double cost = data.getCost() + initData.initialCost;
213                        if (data.getCost() != 0.0 || fromNodes.containsKey(currentNode)) {
214                                if (cost < minCost) {
215                                        minCost = cost;
216                                        minCostNode = currentNode;
217                                }
218                        }
219                }
220
221                if (minCostNode == null) {
222                        return null;
223                }
224
225                // now construct route segments, which are required for TransitPassengerRoute
226                List<RouteSegment> routeSegments = new ArrayList<>();
227
228                TransitRouterNetworkLink link = (TransitRouterNetworkLink) getData(minCostNode).getPrevLink();
229                TransitRouterNetworkLink downstreamLink = null;
230                Node previousFromNode = minCostNode;
231                double transferCost = 0.;
232
233                while (link != null) {
234                        TransitRouterNetworkNode fromNode = link.fromNode;
235                        TransitRouterNetworkNode toNode = link.toNode;
236
237                        double travelTime = getData(toNode).getTime() - getData(fromNode).getTime();
238                        Id<TransitLine> transitLineId = null;
239                        Id<TransitRoute> routeId = null;
240
241                        boolean isTransferLeg = false;
242                        if (link.line==null) isTransferLeg = true;
243                        else {
244                                transitLineId = link.line.getId();
245                                routeId = link.route.getId();
246                        }
247
248                        if (downstreamLink==null && isTransferLeg) {
249                                // continuous transfers: see TransitRouterImplTest.testDoubleWalk.
250                                // Another possibility is that trip start with transfer itself, see TransitRouterImplTest.testDoubleWalkOnly.
251                                double tempTravelTime = 0.;
252                                TransitStopFacility toStop = null;
253
254                                if (routeSegments.size()==0) { // very first leg starts with transfer
255                                        toStop = toNode.stop.getStopFacility();
256                                } else {
257                                        RouteSegment routeSegment = routeSegments.remove(0);
258                                        travelTime = routeSegment.travelTime;
259                                        toStop = routeSegment.toStop;
260                                }
261
262                                routeSegments.add(0,
263                                                new RouteSegment(fromNode.stop.getStopFacility(),
264                                                toStop,
265                                                tempTravelTime+travelTime,
266                                                transitLineId,
267                                                routeId));
268                                //not sure, if transferCost will be included for every transfer or not. Currently, transfer cost will be accumulated for every transfer. Amit Sep'17
269                        } else if (downstreamLink == null // very first pt leg or first pt leg after transfer
270                                        || isTransferLeg ) {
271                                routeSegments.add(0, new RouteSegment( fromNode.stop.getStopFacility(),
272                                                toNode.stop.getStopFacility(),
273                                                travelTime,
274                                                transitLineId,
275                                                routeId
276                                ));
277                        } else if (downstreamLink.line.getId() == link.line.getId() && downstreamLink.route.getId() == link.route.getId() ){
278                                //same route --> update the top routeSegment
279                                RouteSegment routeSegment = routeSegments.remove(0);
280                                routeSegments.add(0, new RouteSegment(fromNode.stop.getStopFacility(),
281                                                routeSegment.toStop,
282                                                routeSegment.travelTime+travelTime,
283                                                transitLineId,
284                                                routeId));
285                        }
286
287                        if (isTransferLeg) {
288                                // transfer cost
289                                if ( ! (this.costFunction instanceof  TransitRouterNetworkTravelTimeAndDisutility) ) {
290                                        throw new RuntimeException("TransitTravelDisutility is not instance of "+TransitRouterNetworkTravelTimeAndDisutility.class.getSimpleName()
291                                        +". An acc ");
292                                }
293
294                                transferCost += ((TransitRouterNetworkTravelTimeAndDisutility) this.costFunction).defaultTransferCost(link,
295                                                Time.UNDEFINED_TIME,null,null);
296
297                                downstreamLink = null;
298                        } else {
299                                downstreamLink = link;
300                        }
301
302                        previousFromNode = fromNode;
303                        link = (TransitRouterNetworkLink) getData(fromNode).getPrevLink();
304                }
305
306                DijkstraNodeData startNodeData = getData(previousFromNode);
307                DijkstraNodeData toNodeData = getData(minCostNode);
308
309                double cost = toNodeData.getCost() - startNodeData.getCost()
310                                + this.fromNodes.get(previousFromNode).initialCost
311                                + toNodes.get(minCostNode).initialCost
312                                + transferCost;
313
314                // if there is no connection found, getPath(...) return a path with nothing in it, however, I (and AN) think that it should throw null. Amit Sep'17
315                if (routeSegments.size()==0) return null;
316                else return new TransitPassengerRoute(cost, routeSegments);
317        }

319        /**
320         * Method to request the path from the (cached) fromNodes to the passed toNodes.
321         * Should only be requested after calling createTransitLeastCostPathTree().
322         *
323         * @param toNodes
324         *          The nodes that are the next stops to the toCoord and you like to route to.
325         *
326         * @return
327         *          the leastCostPath between the fromNode and the toNode.
328         *          Will be null if the path could not be found.
329         */
330        @SuppressWarnings("unchecked")
331        public Path getPath(final Map<Node, InitialNode> toNodes) {
332
333                //find the best node
334                double minCost = Double.POSITIVE_INFINITY;
335                Node minCostNode = null;
336                for (Map.Entry<Node, InitialNode> e : toNodes.entrySet()) {
337                        Node currentNode = e.getKey();
338                        DijkstraNodeData r = this.nodeData.get(currentNode.getId());
339                        if (r == null) {
340                                expandNodeData(toNodes);
341                        }
342                        DijkstraNodeData data = getData(currentNode);
343                        InitialNode initData = e.getValue();
344                        double cost = data.getCost() + initData.initialCost;
345                        if (data.getCost() != 0.0 || fromNodes.containsKey(currentNode)) {
346                                if (cost < minCost) {
347                                        minCost = cost;
348                                        minCostNode = currentNode;
349                                }
350                        }
351                }
352
353                if (minCostNode == null) {
354                        return null;
355                }
356
357                // now construct the path
358                List<Node> nodes = new LinkedList<>();
359                List<Link> links = new LinkedList<>();
360
361                nodes.add(0, minCostNode);
362                Link tmpLink = getData(minCostNode).getPrevLink();
363                while (tmpLink != null) {
364                        links.add(0, tmpLink);
365                        nodes.add(0, tmpLink.getFromNode());
366                        tmpLink = getData(tmpLink.getFromNode()).getPrevLink();
367                }
368
369                DijkstraNodeData startNodeData = getData(nodes.get(0));
370                DijkstraNodeData toNodeData = getData(minCostNode);
371
372                return new Path(nodes, links, toNodeData.getTime() - startNodeData.getTime(),
373                                toNodeData.getCost() - startNodeData.getCost());
374        }

384        /**
385         * Inserts the given Node n into the pendingNodes queue and updates its time
386         * and cost information.
387         *
388         * @param n
389         *            The Node that is revisited.
390         * @param data
391         *            The data for n.
392         * @param pendingNodes
393         *            The nodes visited and not processed yet.
394         * @param time
395         *            The time of the visit of n.
396         * @param cost
397         *            The accumulated cost at the time of the visit of n.
398         * @param outLink
399         *            The node from which we came visiting n.
400         */
401        protected void visitNode(final Node n, final DijkstraNodeData data,
402                        final RouterPriorityQueue<Node> pendingNodes, final double time, final double cost,
403                        final Link outLink) {
404                data.visit(outLink, cost, time, getIterationId());
405                pendingNodes.add(n, getPriority(data));
406        }

408        /**
409         * Expands the given Node in the routing algorithm; may be overridden in
410         * sub-classes.
411         *
412         * @param outNode
413         *            The Node to be expanded.
414         * @param pendingNodes
415         *            The set of pending nodes so far.
416         */
417        protected void relaxNode(final Node outNode, final RouterPriorityQueue<Node> pendingNodes) {
418
419                DijkstraNodeData outData = getData(outNode);
420                double currTime = outData.getTime();
421                double currCost = outData.getCost();
422                for (Link l : outNode.getOutLinks().values()) {
423                        relaxNodeLogic(l, pendingNodes, currTime, currCost);
424                }
425        }

427        /**
428         * Logic that was previously located in the relaxNode(...) method.
429         * By doing so, the FastDijkstra can overwrite relaxNode without copying the logic.
430         */
431        /*package*/ void relaxNodeLogic(final Link l, final RouterPriorityQueue<Node> pendingNodes,
432                        final double currTime, final double currCost) {
433                addToPendingNodes(l, l.getToNode(), pendingNodes, currTime, currCost);
434        }

436        /**
437         * Adds some parameters to the given Node then adds it to the set of pending
438         * nodes.
439         *
440         * @param l
441         *            The link from which we came to this Node.
442         * @param n
443         *            The Node to add to the pending nodes.
444         * @param pendingNodes
445         *            The set of pending nodes.
446         * @param currTime
447         *            The time at which we started to traverse l.
448         * @param currCost
449         *            The cost at the time we started to traverse l.
450         * @return true if the node was added to the pending nodes, false otherwise
451         *              (e.g. when the same node already has an earlier visiting time).
452         */
453        protected boolean addToPendingNodes(final Link l, final Node n,
454                        final RouterPriorityQueue<Node> pendingNodes, final double currTime,
455                        final double currCost) {
456
457        this.customDataManager.initForLink(l);
458        double travelTime = this.timeFunction.getLinkTravelTime(l, currTime, this.person, this.vehicle);
459        double travelCost = this.costFunction.getLinkTravelDisutility(l, currTime, this.person, this.vehicle, this.customDataManager);
460        DijkstraNodeData data = getData(n);
461        double nCost = data.getCost();
462        if (!data.isVisited(getIterationId())) {
463            visitNode(n, data, pendingNodes, currTime + travelTime, currCost + travelCost, l);
464            this.customDataManager.storeTmpData();
465            return true;
466        }
467                double totalCost = currCost + travelCost;
468        if (totalCost < nCost) {
469            revisitNode(n, data, pendingNodes, currTime + travelTime, totalCost, l);
470            this.customDataManager.storeTmpData();
471            return true;
472        }
473
474                return false;
475        }

477        /**
478         * Changes the position of the given Node n in the pendingNodes queue and
479         * updates its time and cost information.
480         *
481         * @param n
482         *            The Node that is revisited.
483         * @param data
484         *            The data for n.
485         * @param pendingNodes
486         *            The nodes visited and not processed yet.
487         * @param time
488         *            The time of the visit of n.
489         * @param cost
490         *            The accumulated cost at the time of the visit of n.
491         * @param outLink
492         *            The link from which we came visiting n.
493         */
494        void revisitNode(final Node n, final DijkstraNodeData data,
495                        final RouterPriorityQueue<Node> pendingNodes, final double time, final double cost,
496                        final Link outLink) {
497                pendingNodes.remove(n);
498
499                data.visit(outLink, cost, time, getIterationId());
500                pendingNodes.add(n, getPriority(data));
501        }

503        /**
504         * The value used to sort the pending nodes during routing.
505         * This implementation compares the total effective travel cost
506         * to sort the nodes in the pending nodes queue during routing.
507         */
508        private double getPriority(final DijkstraNodeData data) {
509                return data.getCost();
510        }

512        /**
513         * Returns the data for the given node. Creates a new NodeData if none exists
514         * yet.
515         *
516         * @param n
517         *            The Node for which to return the data.
518         * @return The data for the given Node
519         */
520        protected DijkstraNodeData getData(final Node n) {
521                DijkstraNodeData r = this.nodeData.get(n.getId());
522                if (null == r) {
523                        r = new DijkstraNodeData();
524                        this.nodeData.put(n.getId(), r);
525                }
526                return r;
527        }
