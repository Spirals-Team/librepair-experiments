038/**
039 * Not thread-safe because MultiNodeDijkstra is not. Does not expect the TransitSchedule to change once constructed! michaz '13
040 *
041 * @author mrieser
042 */
043public class TransitRouterImpl extends AbstractTransitRouter implements TransitRouter {
044
045    private final TransitRouterNetwork transitNetwork;
046    private final TravelTime travelTime;
047    private final TransitTravelDisutility travelDisutility;
048    private final PreparedTransitSchedule preparedTransitSchedule;
049
050    public TransitRouterImpl(final TransitRouterConfig trConfig, final TransitSchedule schedule) {
051        super(trConfig);
052        this.transitNetwork = TransitRouterNetwork.createFromSchedule(schedule,
053                trConfig.getBeelineWalkConnectionDistance());
054        this.preparedTransitSchedule = new PreparedTransitSchedule(schedule);
055        TransitRouterNetworkTravelTimeAndDisutility transitRouterNetworkTravelTimeAndDisutility = new TransitRouterNetworkTravelTimeAndDisutility(
056                trConfig,
057                new PreparedTransitSchedule(schedule));
058        this.travelDisutility = transitRouterNetworkTravelTimeAndDisutility;
059        this.travelTime = transitRouterNetworkTravelTimeAndDisutility;
060        setTransitTravelDisutility(this.travelDisutility);
061    }
062
063    public TransitRouterImpl(
064            final TransitRouterConfig trConfig,
065            final PreparedTransitSchedule preparedTransitSchedule,
066            final TransitRouterNetwork routerNetwork,
067            final TravelTime travelTime,
068            final TransitTravelDisutility travelDisutility) {
069
070        super(trConfig, travelDisutility);
071
072        this.transitNetwork = routerNetwork;
073        this.preparedTransitSchedule = preparedTransitSchedule;
074        this.travelDisutility = travelDisutility;
075        this.travelTime = travelTime;
076    }
077
078    private Map<Node, InitialNode> locateWrappedNearestTransitNodes(Person person, Coord coord, double departureTime) {
079        Collection<TransitRouterNetwork.TransitRouterNetworkNode> nearestNodes = getTransitRouterNetwork().getNearestNodes(
080                coord,
081                this.getConfig().getSearchRadius());
082        if (nearestNodes.size() < 2) {
083            // also enlarge search area if only one stop found, maybe a second one is near the border of the search area
084            TransitRouterNetwork.TransitRouterNetworkNode nearestNode = this.getTransitRouterNetwork()
085                                                                            .getNearestNode(coord);
086            if (nearestNode != null) { // transit schedule might be completely empty!
087                double distance = CoordUtils.calcEuclideanDistance(coord,
088                        nearestNode.stop.getStopFacility().getCoord());
089                nearestNodes = this.getTransitRouterNetwork()
090                                   .getNearestNodes(coord, distance + this.getConfig().getExtensionRadius());
091            }
092        }
093        Map<Node, InitialNode> wrappedNearestNodes = new LinkedHashMap<>();
094        for (TransitRouterNetwork.TransitRouterNetworkNode node : nearestNodes) {
095            Coord toCoord = node.stop.getStopFacility().getCoord();
096            double initialTime = getWalkTime(person, coord, toCoord);
097            double initialCost = getWalkDisutility(person, coord, toCoord);
098            wrappedNearestNodes.put(node, new InitialNode(initialCost, initialTime + departureTime));
099        }
100        return wrappedNearestNodes;
101    }
102
103    @Override
104    public List<Leg> calcRoute(final Facility<?> fromFacility, final Facility<?> toFacility, final double departureTime, final Person person) {
105        // find possible start stops
106        Map<Node, InitialNode> wrappedFromNodes = this.locateWrappedNearestTransitNodes(person,
107                fromFacility.getCoord(),
108                departureTime);
109        // find possible end stops
110        Map<Node, InitialNode> wrappedToNodes = this.locateWrappedNearestTransitNodes(person,
111                toFacility.getCoord(),
112                departureTime);
113
114        TransitPassengerRoute transitPassengerRoute = null;
115
116        TransitLeastCostPathTree tree = new TransitLeastCostPathTree(getTransitRouterNetwork(),
117                getTravelDisutility(),
118                getTravelTime(),
119                wrappedFromNodes,
120                wrappedToNodes,
121                person);
122        // yyyyyy This sounds like it is doing the full tree.  But I think it is not. Kai, nov'16
123
124        // find routes between start and end stop
125        transitPassengerRoute = tree.getTransitPassengerRoute(wrappedToNodes);
126
127        if (transitPassengerRoute == null) {
128//                              return null; // yyyyyy why not return the direct walk leg?? kai/dz, mar'17
129            return this.createDirectWalkLegList(null, fromFacility.getCoord(), toFacility.getCoord());
130        }
131        double pathCost = transitPassengerRoute.getTravelCost();
132
133        double directWalkCost = getWalkDisutility(person, fromFacility.getCoord(), toFacility.getCoord());
134
135        if (directWalkCost * getConfig().getDirectWalkFactor() < pathCost) {
136            return this.createDirectWalkLegList(null, fromFacility.getCoord(), toFacility.getCoord());
137        }
138        return convertPassengerRouteToLegList(departureTime,
139                transitPassengerRoute,
140                fromFacility.getCoord(),
141                toFacility.getCoord(),
142                person);
143    }
144

146        return transitNetwork;
147    }
148
149    TravelTime getTravelTime() {
150        return travelTime;
151    }
152
153    PreparedTransitSchedule getPreparedTransitSchedule() {
154        return preparedTransitSchedule;
155    }
