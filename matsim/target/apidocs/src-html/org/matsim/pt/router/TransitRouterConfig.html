032/**
033 * Design decisions:<ul>
034 * <li>At this point, only those variables that are needed elsewhere (in particular in the scoring) are set from the
035 * config.  All other variables are considered internal computational variables.  kai, feb'11
036 * </ul>
037 *
038 */
039public class TransitRouterConfig implements MatsimParameters {
040
041        /**
042         * The distance in meters in which stop facilities should be searched for
043         * around the start and end coordinate.
044         */
045        private double searchRadius = 1000.0;
046
047        /**
048         * If no stop facility is found around start or end coordinate (see
049         * {@link #searchRadius}), the nearest stop location is searched for
050         * and the distance from start/end coordinate to this location is
051         * extended by the given amount.<br />
052         * If only one stop facility is found within {@link #searchRadius},
053         * the radius is also extended in the hope to find more stop
054         * facilities (e.g. in the opposite direction of the already found
055         * stop).
056         */
057        private double extensionRadius = 200.0;
058
059        /**
060         * The distance in meters that agents can walk to get from one stop to
061         * another stop of a nearby transit line.
062         * <p></p>
063         * Is this really needed?  If the marg utl of walk is correctly set, this should come out automagically.
064         * kai, feb'11
065         * This value is used to generate the walk connections between stop facilities. If they are used,
066         * depends on the scoring/cost calculation. But when they are missing, they cannot be used at all.
067         * mrieser, mar'11
068         */
069        private double beelineWalkConnectionDistance = 100.0;
070
071        /**
072         * The minimum time needed for a transfer is calculated based on the distance and the beeline walk speed
073         * between two stop facilities. Due to passengers probably not being able to immediately (=in the same
074         * second) leave a transit vehicle, or a vehicle being delayed by a small amount of time, an additional
075         * "savety" time can be added to transfers when searching for connecting trips. This could help to find
076         * "better" transfer connections such as they can indeed be realized by the simulation. This value only
077         * affects the routing process, not the simulation itself.
078         */
079        private double additionalTransferTime = 0.0;
080
081        private double beelineWalkSpeed; // meter / second
082
083        private double marginalUtilityOfTravelTimeWalk_utl_s;
084
085        private double marginalUtilityOfTravelTimeTransit_utl_s;
086
087        private double marginalUtilityOfWaitingPt_utl_s;
088
089        private double marginalUtilityOfTravelDistanceWalk_utl_m;
090        
091        private double marginalUtilityOfTravelDistanceTransit_utl_m;
092
093        private double utilityOfLineSwitch_utl;
094
095        private Double beelineDistanceFactor;
096
097        private final double directWalkFactor ;
098
099        public TransitRouterConfig(final Config config) {
100                this(config.planCalcScore(), config.plansCalcRoute(), config.transitRouter(), config.vspExperimental());
101        }
102        
103        public TransitRouterConfig(final PlanCalcScoreConfigGroup pcsConfig, final PlansCalcRouteConfigGroup pcrConfig, 
104                        final TransitRouterConfigGroup trConfig, final VspExperimentalConfigGroup vspConfig ) 
105        {
106                pcsConfig.setLocked(); pcrConfig.setLocked() ; trConfig.setLocked() ; vspConfig.setLocked() ;
107                
108                if (pcsConfig.getScoringParametersPerSubpopulation().size()>1){
109                        Logger.getLogger(getClass()).warn("More than one subpopulation is used in plansCalcScore. "
110                                        + "This is not currently implemented in the TransitRouter (but should work for scoring),"
111                                        + " so the values for the \"default\" subpopulation will be used. (jb, Feb 2018)");
112                }
113                
114                // walk:
115                this.beelineDistanceFactor = pcrConfig.getModeRoutingParams().get( TransportMode.walk ).getBeelineDistanceFactor();
116
117                this.beelineWalkSpeed = pcrConfig.getTeleportedModeSpeeds().get(TransportMode.walk)
118                                / beelineDistanceFactor ;
119                
120                this.marginalUtilityOfTravelTimeWalk_utl_s = pcsConfig.getModes().get(TransportMode.walk).getMarginalUtilityOfTraveling() /3600.0 - pcsConfig.getPerforming_utils_hr()/3600. ;
121                
122                this.marginalUtilityOfTravelDistanceWalk_utl_m = pcsConfig.getMarginalUtilityOfMoney() *
123                                pcsConfig.getModes().get(TransportMode.walk).getMonetaryDistanceRate() +
124                                pcsConfig.getModes().get(TransportMode.walk).getMarginalUtilityOfDistance();
125                
126                // pt:
127                this.marginalUtilityOfTravelTimeTransit_utl_s = pcsConfig.getModes().get(TransportMode.pt).getMarginalUtilityOfTraveling() /3600.0 - pcsConfig.getPerforming_utils_hr()/3600. ;
128
129                this.marginalUtilityOfTravelDistanceTransit_utl_m = pcsConfig.getMarginalUtilityOfMoney() *
130                                pcsConfig.getModes().get(TransportMode.pt).getMonetaryDistanceRate() +
131                                pcsConfig.getModes().get(TransportMode.pt).getMarginalUtilityOfDistance();
132
133                this.marginalUtilityOfWaitingPt_utl_s = pcsConfig.getMarginalUtlOfWaitingPt_utils_hr() / 3600.0 - pcsConfig.getPerforming_utils_hr()/3600. ;
134
135                this.utilityOfLineSwitch_utl = pcsConfig.getUtilityOfLineSwitch();
136
137                // router:
138                this.setSearchRadius(trConfig.getSearchRadius());
139                this.setExtensionRadius(trConfig.getExtensionRadius());
140                this.setBeelineWalkConnectionDistance(trConfig.getMaxBeelineWalkConnectionDistance());
141                this.setAdditionalTransferTime(trConfig.getAdditionalTransferTime());
142                this.directWalkFactor = trConfig.getDirectWalkFactor() ;
143        }
