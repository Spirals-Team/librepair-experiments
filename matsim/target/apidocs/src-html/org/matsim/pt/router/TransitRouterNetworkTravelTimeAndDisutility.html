034/**
035 * TravelTime and TravelCost calculator to be used with the transit network used for transit routing.
036 *
037 * <em>This class is NOT thread-safe!</em>
038 *
039 * @author mrieser
040 */
041public class TransitRouterNetworkTravelTimeAndDisutility implements TravelTime, TransitTravelDisutility {
042
043        final static double MIDNIGHT = 24.0*3600;
044
045        protected final TransitRouterConfig config;
046        private Link previousLink = null;
047        private double previousTime = Double.NaN;
048        private double cachedTravelTime = Double.NaN;
049
050        private final PreparedTransitSchedule preparedTransitSchedule;
051
052        /*
053         * If this constructor is used, every instance use its own PreparedTransitSchedule which might
054         * consume a lot of memory.
055         * 
056         * cdobler, nov'12
057         */
058        @Deprecated
059        public TransitRouterNetworkTravelTimeAndDisutility(final TransitRouterConfig config) {
060                this(config, new PreparedTransitSchedule());
061        }
062        
063        public TransitRouterNetworkTravelTimeAndDisutility(final TransitRouterConfig config, PreparedTransitSchedule preparedTransitSchedule) {
064                this.config = config;
065                this.preparedTransitSchedule = preparedTransitSchedule;
066        }
067
068        @Override
069        public double getLinkTravelDisutility(final Link link, final double time, final Person person, final Vehicle vehicle, final CustomDataManager dataManager) {
070                double cost;
071                if (((TransitRouterNetworkLink) link).getRoute() == null) {
072                        // "route" here means "pt route".  If no pt route is attached, it means that it is a transfer link.
073
074                        cost = defaultTransferCost(link, time, person, vehicle);
075                        
076                } else {
077                        double offVehWaitTime = offVehicleWaitTime(link, time);         
078                        double inVehTime = getLinkTravelTime(link,time, person, vehicle) - offVehWaitTime;              
079                        cost = - inVehTime       * this.config.getMarginalUtilityOfTravelTimePt_utl_s() 
080                               -offVehWaitTime   * this.config.getMarginalUtilityOfWaitingPt_utl_s()
081                               -link.getLength() * this.config.getMarginalUtilityOfTravelDistancePt_utl_m();
082
083                }
084                return cost;
085        }
086        
087        /**
088         * method to allow inclusion of offVehicleWaitTime without code replication.  kai, oct'12
089         * 
090         * @param link
091         * @param time
092         * @return
093         */
094        protected double offVehicleWaitTime(final Link link, final double time) {
095                double offVehWaitTime=0;
096                double nextVehArrivalTime = getVehArrivalTime(link, time);
097                if (time < nextVehArrivalTime){ // it means the agent waits outside the veh                             
098                        offVehWaitTime = nextVehArrivalTime-time;
099                }
100                return offVehWaitTime;
101
102        }
103
104        /**
105         * convenience method for derived classes in order to bring Manuel's version closer to this one here.
106         * kai, oct'12
107         */
108        protected final double defaultTransferCost(final Link link, final double time,
109                        final Person person, final Vehicle vehicle) {
110                double cost;
111                double transfertime = getLinkTravelTime(link, time, person, vehicle);
112                double waittime = this.config.getAdditionalTransferTime();
113                
114                // say that the effective walk time is the transfer time minus some "buffer"
115                double walktime = transfertime - waittime;
116                // (this looks like walktime might become negative.  But at least in the default version, the additional transfer time is always
117                // _added_ to the walk time (see getLinkTravelTime below) so at least in that case this cannot happen.  kai, triggered by cd, may'16)
118                if ( walktime < 0. ) {
119                        throw new RuntimeException( "negative walk time; should not happen; needs to be repaired" ) ;
120                }
121                
122                double walkDistance = link.getLength();
123                
124                // weigh this "buffer" not with the walk time disutility, but with the wait time disutility:
125                // (note that this is the same "additional disutl of wait" as in the scoring function.  Its default is zero.
126                // only if you are "including the opportunity cost of time into the router", then the disutility of waiting will
127                // be the same as the marginal opprotunity cost of time).  kai, nov'11
128                cost = - walktime * this.config.getMarginalUtilityOfTravelTimeWalk_utl_s()
129                       - walkDistance * this.config.getMarginalUtilityOfTravelDistanceWalk_utl_m() 
130                       - waittime * this.config.getMarginalUtilityOfWaitingPt_utl_s()
131                       - this.config.getUtilityOfLineSwitch_utl();
132                return cost;
133        }
134        
135        @Override
136        public double getLinkTravelTime(final Link link, final double time, Person person, Vehicle vehicle) {
137                if ((link == this.previousLink) && (time == this.previousTime)) {
138                        return this.cachedTravelTime;
139                }
140                this.previousLink = link;
141                this.previousTime = time;
142
143                TransitRouterNetworkLink wrapped = (TransitRouterNetworkLink) link;
144                TransitRouteStop fromStop = wrapped.fromNode.stop;
145                TransitRouteStop toStop = wrapped.toNode.stop;
146                if (wrapped.route != null) {
147                        // (agent stays on the same route, so use transit line travel time)
148                        
149                        // get the next departure time:
150                        double bestDepartureTime = preparedTransitSchedule.getNextDepartureTime(wrapped.route, fromStop, time);
151
152                        // the travel time on the link is 
153                        //   the time until the departure (``dpTime - now'')
154                        //   + the travel time on the link (there.arrivalTime - here.departureTime)
155                        // But quite often, we only have the departure time at the next stop.  Then we use that:
156                        double arrivalOffset = (toStop.getArrivalOffset() != Time.UNDEFINED_TIME) ? toStop.getArrivalOffset() : toStop.getDepartureOffset();
157                        double time2 = (bestDepartureTime - time) + (arrivalOffset - fromStop.getDepartureOffset());
158                        if (time2 < 0) {
159                                // ( this can only happen, I think, when ``bestDepartureTime'' is after midnight but ``time'' was before )
160                                time2 += MIDNIGHT;
161                        }
162                        this.cachedTravelTime = time2;
163                        return time2;
164                }
165                // different transit routes, so it must be a line switch
166                double distance = wrapped.getLength();
167                double time2 = distance / this.config.getBeelineWalkSpeed() + this.config.getAdditionalTransferTime();
168                this.cachedTravelTime = time2;
169                return time2;
170        }
171        
172        //variables for caching offVehWaitTime
173        Link previousWaitLink;
174        double previousWaitTime;
175        double cachedVehArrivalTime;
176        
177        public double getVehArrivalTime(final Link link, final double now){
178                if ((link == this.previousWaitLink) && (now == this.previousWaitTime)) {
179                        return this.cachedVehArrivalTime;
180                }
181                this.previousWaitLink = link;
182                this.previousWaitTime = now;
183                
184                //first find out vehicle arrival time to fromStop according to transit schedule
185                TransitRouterNetworkLink wrapped = (TransitRouterNetworkLink) link;
186                if (wrapped.getRoute() == null) { 
187                        throw new RuntimeException("should not happen") ;
188                }
189                TransitRouteStop fromStop = wrapped.fromNode.stop;
190                
191                double nextDepartureTime = preparedTransitSchedule.getNextDepartureTime(wrapped.getRoute(), fromStop, now);
192                
193                double fromStopArrivalOffset = (fromStop.getArrivalOffset() != Time.UNDEFINED_TIME) ? fromStop.getArrivalOffset() : fromStop.getDepartureOffset();
194                double vehWaitAtStopTime = fromStop.getDepartureOffset() - fromStopArrivalOffset; //time in which the veh stops at station
195                double vehArrivalTime = nextDepartureTime - vehWaitAtStopTime;
196                cachedVehArrivalTime = vehArrivalTime ;
197                return vehArrivalTime ;         
198        }

200        @Override
201        public double getWalkTravelDisutility(Person person, Coord coord, Coord toCoord) {
202                //  getMarginalUtilityOfTravelTimeWalk INCLUDES the opportunity cost of time.  kai, dec'12
203                double timeCost = - getWalkTravelTime(person, coord, toCoord) * config.getMarginalUtilityOfTravelTimeWalk_utl_s() ;
204                // (sign: margUtl is negative; overall it should be positive because it is a cost.)
205                
206                double distanceCost = - CoordUtils.calcEuclideanDistance(coord,toCoord) * 
207                                config.getBeelineDistanceFactor() * config.getMarginalUtilityOfTravelDistanceWalk_utl_m();
208                // (sign: same as above)
209                
210                return timeCost + distanceCost ;
211        }
212
213        @Override
214        public double getWalkTravelTime(Person person, Coord coord, Coord toCoord) {
215                double distance = CoordUtils.calcEuclideanDistance(coord, toCoord);
216                double initialTime = distance / config.getBeelineWalkSpeed();
217                return initialTime;
218        }
