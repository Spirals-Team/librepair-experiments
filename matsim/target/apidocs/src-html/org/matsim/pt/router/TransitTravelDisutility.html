020package org.matsim.pt.router;
021
022import org.matsim.api.core.v01.Coord;
023import org.matsim.api.core.v01.network.Link;
024import org.matsim.api.core.v01.population.Person;
025import org.matsim.vehicles.Vehicle;
026
027/**
028 * @author mrieser / senozon
029 */
030public interface TransitTravelDisutility {
031
032        /**
033         * Returns the disutility to travel on the specified link at the specified time.
034         *
035         * @param link The link for which the travel disutility is calculated.
036         * @param time The departure time (in seconds since 00:00) at the beginning of the link for which the disutility is calculated.
037         * @param person The person that wants to travel along the link. Note that this parameter can be <code>null</code>!
038         * @param vehicle The vehicle with which the person wants to travel along the link. Note that this parameter can be <code>null</code>!
039         * @param dataManager A helper class to enable the cost calculator store arbitrary data during one routing request.
040         * @return The disutility to travel over the link <code>link</code>, departing at time <code>time</code>.
041         */
042        public double getLinkTravelDisutility(final Link link, final double time, final Person person, final Vehicle vehicle, final CustomDataManager dataManager);
043        
044        /**
045         * This is used for walking to and from the nearest transit stop from the start and end location,
046         * as well as for the "direct" walk from start to finish without using a pt line at all.
047         * It is not used for transfer links (these are handled by the transitTravelDisutility).
048         */
049        double getWalkTravelTime(Person person, Coord coord, Coord toCoord);
050
051        double getWalkTravelDisutility(Person person, Coord coord, Coord toCoord);
052        
053        
054}
