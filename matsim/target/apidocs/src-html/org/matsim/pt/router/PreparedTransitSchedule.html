020package org.matsim.pt.router;
021
022import java.util.Arrays;
023import java.util.Map;
024import java.util.concurrent.ConcurrentHashMap;
025
026import org.matsim.pt.transitSchedule.api.Departure;
027import org.matsim.pt.transitSchedule.api.TransitRoute;
028import org.matsim.pt.transitSchedule.api.TransitRouteStop;
029import org.matsim.pt.transitSchedule.api.TransitSchedule;
030
031/**
032 * 
033 * Allows fast queries of a TransitSchedule for the next departure of a given route at a given stop, from a given point
034 * in time. If you need further queries of a TransitSchedule, put them here!
035 * 
036 * (I renamed this class and put the TransitSchedule in the constructor to make the purpose clear. michaz '13)
037 * 
038 * Thread-safe.
039 * 
040 * @author mrieser
041 *
042 */
043public class PreparedTransitSchedule {
044        
045        /*
046         * This needs to be a ConcurrentHashMap since multiple threads might add
047         * data concurrently. Alternatively, the map could be filled with data
048         * before getNextDepartureTime(...) - then all concurrent accesses would be
049         * read only.
050         * cdobler, nov'12
051         */
052        private final Map<TransitRoute, double[]> sortedDepartureCache = new ConcurrentHashMap<TransitRoute, double[]>();
053
054    /*
055     * Conceptually, an instance of this class wraps a TransitSchedule to optimize a function of it.
056     */
057        public PreparedTransitSchedule(TransitSchedule schedule) {
058                
059        }
060
061        @Deprecated
062        /*
063         * See other constructor.
064         */
065        public PreparedTransitSchedule() {
066
067        }
068        
069        public double getNextDepartureTime(final TransitRoute route, final TransitRouteStop stop, final double depTime) {
070        
071                double earliestDepartureTimeAtTerminus = depTime - stop.getDepartureOffset();
072                // This shifts my time back to the terminus.
073        
074                if (earliestDepartureTimeAtTerminus >= TransitRouterNetworkTravelTimeAndDisutility.MIDNIGHT) {
075                        earliestDepartureTimeAtTerminus = earliestDepartureTimeAtTerminus % TransitRouterNetworkTravelTimeAndDisutility.MIDNIGHT;
076                }
077                if (earliestDepartureTimeAtTerminus < 0) {
078                        // this may happen when depTime < departureOffset, e.g. I want to start at 24:03, but the bus departs at 23:55 at terminus
079                        earliestDepartureTimeAtTerminus += TransitRouterNetworkTravelTimeAndDisutility.MIDNIGHT;
080                }
081        
082                // this will search for the terminus departure that corresponds to my departure at the stop:
083                double[] cache = sortedDepartureCache.get(route);
084                if (cache == null) {
085                        cache = new double[route.getDepartures().size()];
086                        int i = 0;
087                        for (Departure dep : route.getDepartures().values()) {
088                                cache[i++] = dep.getDepartureTime();
089                        }
090                        Arrays.sort(cache);
091                        sortedDepartureCache.put(route, cache);
092                }
093                int pos = Arrays.binarySearch(cache, earliestDepartureTimeAtTerminus);
094                if (pos < 0) {
095                        // (if the departure time is not found _exactly_, binarySearch returns (-(insertion point) - 1).  That is
096                        // retval = -(insertion point) - 1  or insertion point = -(retval+1) .
097                        // This will, in fact, be the normal situation, so it is important to understand this.)
098                        pos = -(pos + 1);
099                }
100                if (pos >= cache.length) {
101                        pos = 0; // there is no later departure time, take the first in the morning
102                }
103                double bestDepartureTime = cache[pos];
104                // (departure time at terminus)
105        
106                bestDepartureTime += stop.getDepartureOffset();
107                // (resulting departure time at stop)
108                
109                while (bestDepartureTime < depTime) {
110                        bestDepartureTime += TransitRouterNetworkTravelTimeAndDisutility.MIDNIGHT;
111                        // (add enough "MIDNIGHT"s until we are _after_ the desired departure time)
112                }
113                return bestDepartureTime;
114        }
115}
