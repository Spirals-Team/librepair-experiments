021package org.matsim.pt.router;
022
023import java.util.List;
024
025import org.matsim.api.core.v01.TransportMode;
026import org.matsim.api.core.v01.population.Activity;
027import org.matsim.api.core.v01.population.Leg;
028import org.matsim.api.core.v01.population.Plan;
029import org.matsim.api.core.v01.population.PlanElement;
030import org.matsim.api.core.v01.population.Plan;
031import org.matsim.core.population.PopulationUtils;
032import org.matsim.core.population.algorithms.PlanAlgorithm;
033import org.matsim.pt.PtConstants;
034
035/**
036 * Removes all transit activities (like "pt -interaction") as well as the legs
037 * following those activities. In addition, all legs with mode "transit_walk"
038 * are set to mode "pt" to be routed again with the transit. 
039 * 
040 * @see PtConstants#TRANSIT_ACTIVITY_TYPE
041 * 
042 * @author mrieser
043 */
044public class TransitActsRemover implements PlanAlgorithm {
045
046        public void run(final Plan plan) {
047                List<PlanElement> planElements = plan.getPlanElements();
048                for (int i = 0, n = planElements.size(); i < n; i++) {
049                        PlanElement pe = planElements.get(i);
050                        if (pe instanceof Activity) {
051                                Activity act = (Activity) pe;
052                                if (PtConstants.TRANSIT_ACTIVITY_TYPE.equals(act.getType())) {
053                                        final int index = i;
054                                        PopulationUtils.removeActivity(((Plan) plan), index);
055                                        n -= 2;
056                                        i--; // i will be incremented again in next loop-iteration, so we'll check the next act
057                                }
058                        } else if (pe instanceof Leg) {
059                                Leg leg = (Leg) pe;
060                                if (TransportMode.transit_walk.equals(leg.getMode())) {
061                                        leg.setMode(TransportMode.pt);
062                                        leg.setRoute(null);
063                                }
064                        }
065                }
066        }
067
068}
