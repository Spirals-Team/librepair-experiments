020package org.matsim.pt.router;
021
022import java.util.HashMap;
023
024import org.matsim.api.core.v01.network.Link;
025import org.matsim.api.core.v01.network.Node;
026
027/**
028 * A helper class to store custom data for {@link TransitTravelDisutility} which can be used
029 * to e.g. keep track of paid fares during the routing process.
030 * The stored data will be invalidated for each new routing request.
031 * 
032 * @author mrieser / senozon
033 */
034public class CustomDataManager {
035
036        private final HashMap<Node, Object> data = new HashMap<Node, Object>();
037        private Node fromNode = null;
038        private Node toNode = null;
039
040        private Object tmpToNodeData = null;
041        
042        public void setToNodeCustomData(final Object data) {
043                this.tmpToNodeData = data;
044        }
045
046        /**
047         * @param node
048         * @return the stored data for the given node, or <code>null</code> if there is no data stored yet.
049         */
050        public Object getFromNodeCustomData() {
051                return this.data.get(this.fromNode);
052        }
053
054        public void initForLink(final Link link) {
055                this.fromNode = link.getFromNode();
056                this.toNode = link.getToNode();
057                this.tmpToNodeData = null;
058        }
059        
060        public void storeTmpData() {
061                if (this.tmpToNodeData != null) {
062                        this.data.put(this.toNode, this.tmpToNodeData);
063                }
064        }
065        
066        public void reset() {
067                this.data.clear();
068                this.fromNode = null;
069                this.toNode = null;
070                this.tmpToNodeData = null;
071        }
072
073}
