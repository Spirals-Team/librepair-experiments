001package org.matsim.pt;
002
003import java.util.List;
004
005import org.matsim.api.core.v01.Id;
006import org.matsim.api.core.v01.network.Link;
007import org.matsim.api.core.v01.network.Network;
008import org.matsim.api.core.v01.network.Node;
009import org.matsim.core.config.groups.PlanCalcScoreConfigGroup;
010import org.matsim.core.network.NetworkUtils;
011import org.matsim.core.population.routes.NetworkRoute;
012import org.matsim.core.population.routes.RouteUtils;
013import org.matsim.core.router.DijkstraFactory;
014import org.matsim.core.router.costcalculators.FreespeedTravelTimeAndDisutility;
015import org.matsim.core.router.util.LeastCostPathCalculator;
016import org.matsim.core.router.util.LeastCostPathCalculator.Path;
017
018public class UmlaufInterpolator {
019
020        private final Network network;
021    private final LeastCostPathCalculator routingAlgo;
022
023        public UmlaufInterpolator(Network network, final PlanCalcScoreConfigGroup config) {
024                super();
025                this.network = network;
026        FreespeedTravelTimeAndDisutility travelTimes = new FreespeedTravelTimeAndDisutility(config);
027                this.routingAlgo = new DijkstraFactory().createPathCalculator(network, travelTimes, travelTimes);
028        }
029
030        public void addUmlaufStueckToUmlauf(UmlaufStueck umlaufStueck, Umlauf umlauf) {
031                List<UmlaufStueckI> umlaufStueckeOfThisUmlauf = umlauf.getUmlaufStuecke();
032                if (! umlaufStueckeOfThisUmlauf.isEmpty()) {
033                        UmlaufStueckI previousUmlaufStueck = umlaufStueckeOfThisUmlauf.get(umlaufStueckeOfThisUmlauf.size() - 1);
034                        NetworkRoute previousCarRoute = previousUmlaufStueck.getCarRoute();
035                        Id<Link> fromLinkId = previousCarRoute.getEndLinkId();
036                        Id<Link> toLinkId = umlaufStueck.getCarRoute().getStartLinkId();
037                        if (!fromLinkId.equals(toLinkId)) {
038                                insertWenden(fromLinkId, toLinkId, umlauf);
039                        }
040                }
041                umlaufStueckeOfThisUmlauf.add(umlaufStueck);
042        }
043
044        private void insertWenden(Id<Link> fromLinkId, Id<Link> toLinkId, Umlauf umlauf) {
045                Node startNode = this.network.getLinks().get(fromLinkId).getToNode();
046                Node endNode = this.network.getLinks().get(toLinkId).getFromNode();
047                double depTime = 0.0;
048                Path wendenPath = routingAlgo.calcLeastCostPath(startNode, endNode, depTime, null, null);
049                if (wendenPath == null) {
050                        throw new RuntimeException("No route found from node "
051                                        + startNode.getId() + " to node " + endNode.getId() + ".");
052                }
053                NetworkRoute route = RouteUtils.createLinkNetworkRouteImpl(fromLinkId, toLinkId);
054                route.setLinkIds(fromLinkId, NetworkUtils.getLinkIds(wendenPath.links), toLinkId);
055                umlauf.getUmlaufStuecke().add(new Wenden(route));
056        }
057
058}
