021package org.matsim.pt.analysis;
022
023import java.util.HashMap;
024import java.util.HashSet;
025import java.util.Map;
026import java.util.Set;
027
028import org.matsim.api.core.v01.Id;
029import org.matsim.api.core.v01.events.PersonEntersVehicleEvent;
030import org.matsim.api.core.v01.events.PersonLeavesVehicleEvent;
031import org.matsim.api.core.v01.events.handler.PersonEntersVehicleEventHandler;
032import org.matsim.api.core.v01.events.handler.PersonLeavesVehicleEventHandler;
033import org.matsim.pt.transitSchedule.api.Departure;
034import org.matsim.pt.transitSchedule.api.TransitRoute;
035
036/**
037 * Keeps track of the total number of passengers entering and leaving a transit
038 * vehicle along a given route, on all Departures.
039 *
040 * @author mrieser
041 */
042public class RouteOccupancy implements PersonEntersVehicleEventHandler, PersonLeavesVehicleEventHandler {
043
044        private static final Integer Int1 = Integer.valueOf(1);
045
046        private final TransitRoute transitRoute;
047        private final VehicleTracker tracker;
048        private Set<Id> routeVehicles = null;
049        private final Map<Id, Integer> enteringPassengers = new HashMap<Id, Integer>();
050        private final Map<Id, Integer> leavingPassengers = new HashMap<Id, Integer>();
051
052        public RouteOccupancy(final TransitRoute transitRoute, final VehicleTracker tracker) {
053                this.transitRoute = transitRoute;
054                this.tracker = tracker;
055        }
056
057        public void handleEvent(final PersonEntersVehicleEvent event) {
058                if (this.routeVehicles == null) {
059                        collectVehiclesInfo();
060                }
061                if (this.routeVehicles.contains(event.getVehicleId())) {
062                        Id facilityId = this.tracker.getFacilityIdForVehicle(event.getVehicleId());
063                        Integer count = this.enteringPassengers.get(facilityId);
064                        if (count == null) {
065                                this.enteringPassengers.put(facilityId, Int1);
066                        } else {
067                                this.enteringPassengers.put(facilityId, Integer.valueOf(1 + count.intValue()));
068                        }
069                }
070        }
071
072        public void handleEvent(final PersonLeavesVehicleEvent event) {
073                if (this.routeVehicles == null) {
074                        collectVehiclesInfo();
075                }
076                if (this.routeVehicles.contains(event.getVehicleId())) {
077                        Id facilityId = this.tracker.getFacilityIdForVehicle(event.getVehicleId());
078                        Integer count = this.leavingPassengers.get(facilityId);
079                        if (count == null) {
080                                this.leavingPassengers.put(facilityId, Int1);
081                        } else {
082                                this.leavingPassengers.put(facilityId, Integer.valueOf(1 + count.intValue()));
083                        }
084                }
085        }
086
087        public void reset(final int iteration) {
088                this.routeVehicles = null;
089                this.enteringPassengers.clear();
090                this.leavingPassengers.clear();
091        }
092
093        public int getNumberOfEnteringPassengers(final Id stopFacilityId) {
094                Integer count = this.enteringPassengers.get(stopFacilityId);
095                if (count == null) {
096                        return 0;
097                }
098                return count.intValue();
099        }
100
101        public int getNumberOfLeavingPassengers(final Id stopFacilityId) {
102                Integer count = this.leavingPassengers.get(stopFacilityId);
103                if (count == null) {
104                        return 0;
105                }
106                return count.intValue();
107        }
108
109        /**
110         * Lazy initialization, as the vehicle info may not be available from the beginning.
111         */
112        private void collectVehiclesInfo() {
113                Set<Id> set = new HashSet<Id>(this.transitRoute.getDepartures().size()*2);
114
115                for (Departure departure : this.transitRoute.getDepartures().values()) {
116                        if (departure.getVehicleId() != null) {
117                                set.add(departure.getVehicleId());
118                        }
119                }
120
121                /* try to make it thread-safe by assigning class-member at the end.
122                 * if two threads enter this method, nothing bad should happen,
123                 * as both threads should generated the same initialization.
124                 */
125                this.routeVehicles = set;
126        }
127
128}
