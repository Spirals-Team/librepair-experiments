050/**
051 * Collects data to create Route-Time-Diagrams based on the actual simulation.
052 * A Route-Time-Diagram shows the position along one transit route of one or
053 * more vehicles over the lapse of time.
054 *
055 * @author mrieser
056 */
057public class RouteTimeDiagram implements VehicleArrivesAtFacilityEventHandler, VehicleDepartsAtFacilityEventHandler {
058
059        /**
060         * Map containing for each vehicle a list of positions, stored as StopFacility Ids and the time.
061         */
062        private final Map<Id, List<Tuple<Id, Double>>> positions = new HashMap<Id, List<Tuple<Id, Double>>>();
063
064        public void handleEvent(final VehicleArrivesAtFacilityEvent event) {
065                List<Tuple<Id, Double>> list = this.positions.get(event.getVehicleId());
066                if (list == null) {
067                        list = new ArrayList<Tuple<Id, Double>>();
068                        this.positions.put(event.getVehicleId(), list);
069                }
070                list.add(new Tuple<Id, Double>(event.getFacilityId(), Double.valueOf(event.getTime())));
071        }
072
073        public void handleEvent(final VehicleDepartsAtFacilityEvent event) {
074                List<Tuple<Id, Double>> list = this.positions.get(event.getVehicleId());
075                if (list == null) {
076                        list = new ArrayList<Tuple<Id, Double>>();
077                        this.positions.put(event.getVehicleId(), list);
078                }
079                list.add(new Tuple<Id, Double>(event.getFacilityId(), Double.valueOf(event.getTime())));
080        }
081
082        public void reset(final int iteration) {
083                this.positions.clear();
084        }
085
086        public void writeData() {
087                for (List<Tuple<Id, Double>> list : this.positions.values()) {
088                        for (Tuple<Id, Double> info : list) {
089                                System.out.println(info.getFirst().toString() + "\t" + info.getSecond().toString());
090                        }
091                        System.out.println();
092                }
093        }
094
095        public void createGraph(final String filename, final TransitRoute route) {
096
097                HashMap<Id, Integer> stopIndex = new HashMap<Id, Integer>();
098                int idx = 0;
099                for (TransitRouteStop stop : route.getStops()) {
100                        stopIndex.put(stop.getStopFacility().getId(), idx);
101                        idx++;
102                }
103
104                HashSet<Id> vehicles = new HashSet<Id>();
105                for (Departure dep : route.getDepartures().values()) {
106                        vehicles.add(dep.getVehicleId());
107                }
108
109                XYSeriesCollection dataset = new XYSeriesCollection();
110                int numSeries = 0;
111                double earliestTime = Double.POSITIVE_INFINITY;
112                double latestTime = Double.NEGATIVE_INFINITY;
113
114                for (Map.Entry<Id, List<Tuple<Id, Double>>> entry : this.positions.entrySet()) {
115                        if (vehicles.contains(entry.getKey())) {
116                                XYSeries series = new XYSeries("t", false, true);
117                                for (Tuple<Id, Double> pos : entry.getValue()) {
118                                        Integer stopIdx = stopIndex.get(pos.getFirst());
119                                        if (stopIdx != null) {
120                                                double time = pos.getSecond().doubleValue();
121                                                series.add(stopIdx.intValue(), time);
122                                                if (time < earliestTime) {
123                                                        earliestTime = time;
124                                                }
125                                                if (time > latestTime) {
126                                                        latestTime = time;
127                                                }
128                                        }
129                                }
130                                dataset.addSeries(series);
131                                numSeries++;
132
133                        }
134                }
135
136                JFreeChart c = ChartFactory.createXYLineChart("Route-Time Diagram, Route = " + route.getId(), "stops", "time",
137                                dataset, PlotOrientation.VERTICAL,
138                                false, // legend?
139                                false, // tooltips?
140                                false // URLs?
141                                );
142                c.setBackgroundPaint(new Color(1.0f, 1.0f, 1.0f, 1.0f));
143
144                XYPlot p  = (XYPlot) c.getPlot();
145
146                p.getRangeAxis().setInverted(true);
147                p.getRangeAxis().setRange(earliestTime, latestTime);
148                XYItemRenderer renderer = p.getRenderer();
149                for (int i = 0; i < numSeries; i++) {
150                        renderer.setSeriesPaint(i, Color.black);
151                }
152
153                try {
154                        ChartUtilities.saveChartAsPNG(new File(filename), c, 1024, 768, null, true, 9);
155                } catch (IOException e) {
156                        e.printStackTrace();
157                }
158        }
