044/**
045 * Calculates the number of passenger that are in a transit vehicle when
046 * the vehicle departs a stop location.
047 *
048 * @author mrieser
049 */
050public class TransitLoad implements TransitDriverStartsEventHandler, VehicleArrivesAtFacilityEventHandler, VehicleDepartsAtFacilityEventHandler, PersonEntersVehicleEventHandler, PersonLeavesVehicleEventHandler {
051
052        private final Map<Id<TransitLine>, LineData> lineData = new HashMap<>();
053
054        private final Map<Id<Vehicle>, Id<TransitStopFacility>> vehicleFacilityMap = new HashMap<>();
055        private final Map<Id<Vehicle>, VehicleData> vehicleData = new HashMap<>();
056
057        public TransitLoad() {
058        }
059
060        public int getLoadAtDeparture(final TransitLine line, final TransitRoute route, final TransitStopFacility stopFacility, final Departure departure) {
061                int nOfPassengers = 0;
062                for (TransitRouteStop stop : route.getStops()) {
063                        StopInformation si = getStopInformation(line.getId(), route.getId(), stop.getStopFacility().getId(), departure.getId(), false);
064                        if (si != null) {
065                                nOfPassengers -= si.nOfLeaving;
066                                nOfPassengers += si.nOfEntering;
067                        }
068                        if (stop.getStopFacility() == stopFacility) {
069                                return nOfPassengers;
070                        }
071                }
072                return -1;
073        }
074
075        public StopInformation getDepartureStopInformation(final TransitLine line, final TransitRoute route, final TransitStopFacility stopFacility, final Departure departure) {
076                return getStopInformation(line.getId(), route.getId(), stopFacility.getId(), departure.getId(), false);
077        }
078
079        @Override
080        public void handleEvent(TransitDriverStartsEvent event) {
081                this.vehicleData.put(event.getVehicleId(), new VehicleData(event.getVehicleId(), event.getTransitLineId(), event.getTransitRouteId(), event.getDepartureId()));
082        }
083
084        @Override
085        public void handleEvent(final VehicleArrivesAtFacilityEvent event) {
086                this.vehicleFacilityMap.put(event.getVehicleId(), event.getFacilityId());
087                VehicleData vData = this.vehicleData.get(event.getVehicleId());
088                if (vData != null) {
089                        StopInformation si = getStopInformation(vData.lineId, vData.routeId, this.vehicleFacilityMap.get(event.getVehicleId()), vData.departureId, true);
090                        si.arrivalTime = event.getTime();
091                }
092        }
093
094        @Override
095        public void handleEvent(final VehicleDepartsAtFacilityEvent event) {
096                Id<TransitStopFacility> stopId = this.vehicleFacilityMap.remove(event.getVehicleId());
097                VehicleData vData = this.vehicleData.get(event.getVehicleId());
098                if (vData != null) {
099                        StopInformation si = getStopInformation(vData.lineId, vData.routeId, stopId, vData.departureId, true);
100                        si.departureTime = event.getTime();
101                }
102        }
103
104        @Override
105        public void handleEvent(final PersonEntersVehicleEvent event) {
106                VehicleData vData = this.vehicleData.get(event.getVehicleId());
107                if (vData != null) {
108                        StopInformation si = getStopInformation(vData.lineId, vData.routeId, this.vehicleFacilityMap.get(event.getVehicleId()), vData.departureId, true);
109                        si.nOfEntering++;
110                }
111        }
112
113        @Override
114        public void handleEvent(final PersonLeavesVehicleEvent event) {
115                VehicleData vData = this.vehicleData.get(event.getVehicleId());
116                if (vData != null) {
117                        StopInformation si = getStopInformation(vData.lineId, vData.routeId, this.vehicleFacilityMap.get(event.getVehicleId()), vData.departureId, true);
118                        si.nOfLeaving++;
119                }
120        }
121
122        @Override
123        public void reset(final int iteration) {
124                this.vehicleFacilityMap.clear();
125                this.vehicleData.clear();
126        }
127
128        private StopInformation getStopInformation(final Id<TransitLine> lineId, final Id<TransitRoute> routeId, final Id<TransitStopFacility> stopFacilityId, final Id<Departure> departureId, final boolean createIfMissing) {
129                LineData ld = this.lineData.get(lineId);
130                if (ld == null) {
131                        if (createIfMissing) {
132                                ld = new LineData();
133                                this.lineData.put(lineId, ld);
134                        } else {
135                                return null;
136                        }
137                }
138
139                RouteData rd = ld.routeData.get(routeId);
140                if (rd == null) {
141                        if (createIfMissing) {
142                                rd = new RouteData();
143                                ld.routeData.put(routeId, rd);
144                        } else {
145                                return null;
146                        }
147                }
148
149                StopData sd = rd.stopData.get(stopFacilityId);
150                if (sd == null) {
151                        if (createIfMissing) {
152                                sd = new StopData();
153                                rd.stopData.put(stopFacilityId, sd);
154                        } else {
155                                return null;
156                        }
157                }
158
159                StopInformation si = sd.departureData.get(departureId);
160                if (si == null) {
161                        if (createIfMissing) {
162                                si = new StopInformation();
163                                sd.departureData.put(departureId, si);
164                        } else {
165                                return null;
166                        }
167                }
168                return si;
169        }
170
171        private static class VehicleData {
172                public final Id<Vehicle> vehicleId;
173                public final Id<TransitLine> lineId;
174                public final Id<TransitRoute> routeId;
175                public final Id<Departure> departureId;
176
177                public VehicleData(final Id<Vehicle> vehicleId, final Id<TransitLine> lineId, final Id<TransitRoute> routeId, final Id<Departure> departureId) {
178                        this.vehicleId = vehicleId;
179                        this.lineId = lineId;
180                        this.routeId = routeId;
181                        this.departureId = departureId;
182                }
183        }
184
185        /*package*/ static class LineData {
186                public final Map<Id<TransitRoute>, RouteData> routeData = new HashMap<>();
187        }
188
189        /*package*/ static class RouteData {
190                public final Map<Id<TransitStopFacility>, StopData> stopData = new HashMap<>(); // use stopFacilityId as key
191        }
192
193        /*package*/ static class StopData {
194                public final Map<Id<Departure>, StopInformation> departureData = new HashMap<>(); // use departure id as key
195        }
196
197        public static class StopInformation {
198                public short nOfEntering = 0;
199                public short nOfLeaving = 0;
200                public double arrivalTime = Time.UNDEFINED_TIME;
201                public double departureTime = Time.UNDEFINED_TIME;
202        }
