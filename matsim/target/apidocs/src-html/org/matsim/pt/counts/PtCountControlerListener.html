053
054public class PtCountControlerListener implements StartupListener, IterationEndsListener,
055BeforeMobsimListener, AfterMobsimListener  {
056
057        private EventsManager eventsManager;
058        private OutputDirectoryHierarchy controlerIO;
059        private IterationStopWatch iterationStopWatch;
060        private Network network;
061
062        private static enum CountType { Boarding, Alighting, Occupancy }
063
064        private static final Logger log = Logger.getLogger(PtCountControlerListener.class);
065
066        /*
067         * String used to identify the operation in the IterationStopWatch.
068         */
069        public static final String OPERATION_COMPAREPTCOUNTS = "compare with pt counts";
070
071        private final static String MODULE_NAME = "ptCounts";
072        // yy the above should be removed; the commands should be replaced by the "typed" commands.  kai, oct'10
073
074        private final Config config;
075        private final Counts boardCounts, alightCounts,occupancyCounts;
076        private final OccupancyAnalyzer occupancyAnalyzer;
077
078        @Inject
079        PtCountControlerListener(final Config config, EventsManager eventsManager, OutputDirectoryHierarchy controlerIO, IterationStopWatch iterationStopWatch, Network network) {
080                this.eventsManager = eventsManager;
081                this.controlerIO = controlerIO;
082                this.iterationStopWatch = iterationStopWatch;
083                this.network = network;
084                log.info("Using pt counts.");
085                this.config = config;
086                this.boardCounts = new Counts();
087                this.alightCounts = new Counts();
088                this.occupancyCounts = new Counts();
089                this.occupancyAnalyzer = new OccupancyAnalyzer(3600, 24 * 3600 - 1) ;
090        }
091
092        @Override
093        public void notifyStartup(final StartupEvent controlerStartupEvent) {
094                PtCountsConfigGroup ptCounts = this.config.ptCounts();
095                String boardCountsFilename = ptCounts.getBoardCountsFileName();
096                String alightCountsFilename = ptCounts.getAlightCountsFileName();
097                String occupancyCountsFilename = ptCounts.getOccupancyCountsFileName();
098                if (boardCountsFilename == null || alightCountsFilename == null || occupancyCountsFilename == null) {
099                        throw new RuntimeException("for pt counts, at this point all three files must be given!");
100                }
101                new MatsimCountsReader(this.alightCounts).readFile(alightCountsFilename);
102                new MatsimCountsReader(this.boardCounts).readFile(boardCountsFilename);
103                new MatsimCountsReader(this.occupancyCounts).readFile(occupancyCountsFilename);
104        }
105
106        @Override
107        public void notifyBeforeMobsim(BeforeMobsimEvent event) {
108                int iter = event.getIteration();
109                if ( isActiveInThisIteration( iter ) ) {
110                        occupancyAnalyzer.reset(iter);
111                        eventsManager.addHandler(occupancyAnalyzer);
112                }
113        }
114
115        @Override
116        public void notifyAfterMobsim(AfterMobsimEvent event) {
117                int it = event.getIteration();
118                if ( isActiveInThisIteration( it ) ) {
119                        eventsManager.removeHandler(occupancyAnalyzer);
120                        occupancyAnalyzer.write(controlerIO.getIterationFilename(it, "occupancyAnalysis.txt"));
121                }
122        }
123
124        private boolean isActiveInThisIteration( int iter ) {
125                return iter % config.ptCounts().getPtCountsInterval() == 0 && iter >= config.controler().getFirstIteration();
126        }
127
128        @Override
129        public void notifyIterationEnds(final IterationEndsEvent event) {
130                int iter = event.getIteration();
131                if ( isActiveInThisIteration( iter ) ) {
132
133                        iterationStopWatch.beginOperation(OPERATION_COMPAREPTCOUNTS);
134
135                        double countsScaleFactor = Double.parseDouble(this.config.getParam(MODULE_NAME, "countsScaleFactor"));
136
137                        Map<CountType,CountsComparisonAlgorithm> cca = new HashMap<CountType,CountsComparisonAlgorithm>();
138                        cca.put( CountType.Boarding, new CountsComparisonAlgorithm(new CountsComparisonAlgorithm.VolumesForId() {
139
140                                @Override
141                                public double[] getVolumesForStop(Id<TransitStopFacility> locationId) {
142                                        return copyFromIntArray(occupancyAnalyzer.getBoardVolumesForStop(locationId));
143                                }
144                        }, this.boardCounts, network, countsScaleFactor)) ;
145                        cca.put( CountType.Alighting, new CountsComparisonAlgorithm(new CountsComparisonAlgorithm.VolumesForId() {
146
147                                @Override
148                                public double[] getVolumesForStop(Id<TransitStopFacility> locationId) {
149                                        return copyFromIntArray(occupancyAnalyzer.getAlightVolumesForStop(locationId));
150                                }
151                        }, this.alightCounts, network, countsScaleFactor) ) ;
152                        cca.put( CountType.Occupancy, new CountsComparisonAlgorithm(new CountsComparisonAlgorithm.VolumesForId() {
153
154                                @Override
155                                public double[] getVolumesForStop(Id<TransitStopFacility> locationId) {
156                                        return copyFromIntArray(occupancyAnalyzer.getOccupancyVolumesForStop(locationId));
157                                }
158                        }, this.occupancyCounts, network, countsScaleFactor) ) ;
159
160                        String distanceFilterStr = this.config.findParam(MODULE_NAME, "distanceFilter");
161                        String distanceFilterCenterNodeId = this.config.findParam(MODULE_NAME, "distanceFilterCenterNode");
162                        
163                        if ((distanceFilterStr != null) && (distanceFilterCenterNodeId != null)) {
164
165                                
166                                for ( CountsComparisonAlgorithm algo : cca.values() ) {
167                                        algo.setCountCoordUsingDistanceFilter(Double.parseDouble(distanceFilterStr), distanceFilterCenterNodeId) ;
168                                }
169                        }
170                        for ( CountsComparisonAlgorithm algo : cca.values() ) {
171                                algo.setCountsScaleFactor(countsScaleFactor);
172                                algo.run();
173                        }
174
175                        String outputFormat = this.config.findParam(MODULE_NAME, "outputformat");
176                        if (outputFormat.contains("kml") || outputFormat.contains("all")) {
177
178                                String filename = controlerIO.getIterationFilename(iter, "ptcountscompare.kmz");
179
180                                // yy would be good to also just pass a collection/a map but we ain't there. kai, dec'13
181                                Map< CountType, List<CountSimComparison> > comparisons = new HashMap< > () ;
182                                for ( CountType countType : CountType.values()  ) {
183                                        if ( cca.get( countType ) != null ) {
184                                                comparisons.put( countType, cca.get(countType).getComparison() ) ;
185                                        }
186                                }
187                                final CoordinateTransformation coordTransform = TransformationFactory.getCoordinateTransformation(this.config.global().getCoordinateSystem(),TransformationFactory.WGS84);
188                                // the following should work since comparisons.get(...) for material that does not exist should return null, which should be a valid 
189                                // parameter.  kai, dec'13
190//                              PtCountSimComparisonKMLWriter kmlWriterOrig = new PtCountSimComparisonKMLWriter(comparisons.get(CountType.Boarding), 
191//                                              comparisons.get(CountType.Alighting), comparisons.get(CountType.Occupancy),
192//                                              coordTransform,
193//                                              this.boardCounts, this.alightCounts,occupancyCounts);
194                                
195                                CountSimComparisonKMLWriter kmlWriter = new CountSimComparisonKMLWriter(comparisons.get(CountType.Occupancy), this.occupancyCounts, coordTransform, "ptCountsOccup") ;
196
197                                kmlWriter.setIterationNumber(iter);
198                                kmlWriter.writeFile(filename);
199                        }
200                        if (outputFormat.contains("txt") || outputFormat.contains("all")) {
201                                for ( Entry<CountType,CountsComparisonAlgorithm> entry : cca.entrySet() ) {
202                                        CountsComparisonAlgorithm algo = entry.getValue() ;
203                                        new PtCountSimComparisonTableWriter(algo.getComparison()).write(controlerIO.getIterationFilename(iter, "simCountCompare" + entry.getKey().toString() + ".txt"));
204                                }
205                        }
206
207                        iterationStopWatch.endOperation(OPERATION_COMPAREPTCOUNTS);
208                }
209        }
210
211        private static double[] copyFromIntArray(int[] source) {
212            double[] dest = new double[source.length];
213            for(int i=0; i<source.length; i++) {
214                dest[i] = source[i];
215            }
216            return dest;
217        }
