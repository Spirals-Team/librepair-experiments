046/**
047 * @author yChen
048 * @author mrieser / senozon
049 */
050public class OccupancyAnalyzer implements PersonEntersVehicleEventHandler, PersonLeavesVehicleEventHandler,
051                VehicleArrivesAtFacilityEventHandler, VehicleDepartsAtFacilityEventHandler, TransitDriverStartsEventHandler {
052
053        private static final Logger log = Logger.getLogger(OccupancyAnalyzer.class);
054
055        private final int timeBinSize, maxSlotIndex;
056        private final double maxTime;
057        /** Map< stopFacilityId,value[]> */
058        private Map<Id<TransitStopFacility>, int[]> boards, alights, occupancies;
059
060        /** Map< vehId,stopFacilityId> */
061        private final Map<Id<Vehicle>, Id<TransitStopFacility>> vehStops = new HashMap<>();
062        /** Map<vehId,passengersNo. in Veh> */
063        private final Map<Id<Vehicle>, Integer> vehPassengers = new HashMap<>();
064        private StringBuffer occupancyRecord = new StringBuffer("time\tvehId\tStopId\tno.ofPassengersInVeh\n");
065        private final Set<Id<Person>> transitDrivers = new HashSet<>();
066        private final Set<Id<Vehicle>> transitVehicles = new HashSet<>();
067
068        public OccupancyAnalyzer(final int timeBinSize, final double maxTime) {
069                log.setLevel( Level.INFO ) ;
070
071                this.timeBinSize = timeBinSize;
072                this.maxTime = maxTime;
073                this.maxSlotIndex = ((int) this.maxTime) / this.timeBinSize + 1;
074                this.boards = new HashMap<>();
075                this.alights = new HashMap<>();
076                this.occupancies = new HashMap<>();
077        }
078
079        public void setBoards(Map<Id<TransitStopFacility>, int[]> boards) {
080                this.boards = boards;
081        }
082
083        public void setAlights(Map<Id<TransitStopFacility>, int[]> alights) {
084                this.alights = alights;
085        }
086
087        public void setOccupancies(Map<Id<TransitStopFacility>, int[]> occupancies) {
088                this.occupancies = occupancies;
089        }
090
091        public int getTimeSlotIndex(final double time) {
092                if (time > this.maxTime) {
093                        return this.maxSlotIndex;
094                }
095                return ((int) time / this.timeBinSize);
096        }
097
098        @Override
099        public void reset(int iteration) {
100                this.boards.clear();
101                this.alights.clear();
102                this.occupancies.clear();
103                this.vehStops.clear();
104                this.vehPassengers.clear();
105                this.occupancyRecord = new StringBuffer("time\tvehId\tStopId\tno.ofPassengersInVeh\n");
106                this.transitDrivers.clear();
107                this.transitVehicles.clear();
108        }
109
110        @Override
111        public void handleEvent(TransitDriverStartsEvent event) {
112                this.transitDrivers.add(event.getDriverId());
113                this.transitVehicles.add(event.getVehicleId());
114        }
115        
116        @Override
117        public void handleEvent(PersonEntersVehicleEvent event) {
118                if (this.transitDrivers.contains(event.getPersonId()) || !this.transitVehicles.contains(event.getVehicleId())) {
119                        return; // ignore transit drivers or persons entering non-transit vehicles
120                }
121                
122                Id<Vehicle> vehId = event.getVehicleId();
123                Id<TransitStopFacility> stopId = this.vehStops.get(vehId);
124                double time = event.getTime();
125                // --------------------------getOns---------------------------
126                int[] getOn = this.boards.get(stopId);
127                if (getOn == null) {
128                        getOn = new int[this.maxSlotIndex + 1];
129                        this.boards.put(stopId, getOn);
130                }
131                getOn[getTimeSlotIndex(time)]++;
132                // ------------------------veh_passenger---------------------------
133                Integer nPassengers = this.vehPassengers.get(vehId);
134                this.vehPassengers.put(vehId, (nPassengers != null) ? (nPassengers + 1) : 1);
135                this.occupancyRecord.append("time :\t").append(time).append(" veh :\t").append(vehId).append(" has Passenger\t").append(this.vehPassengers.get(vehId)).append(" \tat stop :\t").append(stopId).append(" ENTERING PERSON :\t").append(event.getPersonId()).append("\n");
136        }
137
138        @Override
139        public void handleEvent(PersonLeavesVehicleEvent event) {
140                if (this.transitDrivers.contains(event.getPersonId()) || !this.transitVehicles.contains(event.getVehicleId())) {
141                        return; // ignore transit drivers or persons entering non-transit vehicles
142                }
143                
144                Id<Vehicle> vehId = event.getVehicleId();
145                Id<TransitStopFacility> stopId = this.vehStops.get(vehId);
146                double time = event.getTime();
147                // --------------------------getDowns---------------------------
148                int[] getDown = this.alights.get(stopId);
149                if (getDown == null) {
150                        getDown = new int[this.maxSlotIndex + 1];
151                        this.alights.put(stopId, getDown);
152                }
153                getDown[getTimeSlotIndex(time)]++;
154                // ------------------------veh_passenger---------------------------
155                Integer nPassengers = this.vehPassengers.get(vehId);
156                if (nPassengers == null) {
157                        log.error( "tests for `null' but exception says 'negative'???  kai, oct'10 ") ;
158                        throw new RuntimeException("negative passenger-No. in vehicle?");
159                }
160                this.vehPassengers.put(vehId, nPassengers - 1);
161                if (this.vehPassengers.get(vehId).intValue() == 0) {
162                        this.vehPassengers.remove(vehId);
163                }
164
165                Integer passengers = this.vehPassengers.get(vehId);
166                this.occupancyRecord.append("time :\t").append(time).append(" veh :\t").append(vehId).append(" has Passenger\t").append((passengers != null) ? passengers : 0).append("\n");
167        }
168
169        @Override
170        public void handleEvent(VehicleArrivesAtFacilityEvent event) {
171                Id<TransitStopFacility> stopId = event.getFacilityId();
172                this.vehStops.put(event.getVehicleId(), stopId);
173        }
174
175        @Override
176        public void handleEvent(VehicleDepartsAtFacilityEvent event) {
177                Id<TransitStopFacility> stopId = event.getFacilityId();
178                Id<Vehicle> vehId = event.getVehicleId();
179                this.vehStops.remove(vehId);
180                // -----------------------occupancy--------------------------------
181                int[] occupancyAtStop = this.occupancies.get(stopId);
182
183                if (occupancyAtStop == null) {
184                        // no previous departure from this stop, therefore no occupancy record yet.  Create this:
185                        occupancyAtStop = new int[this.maxSlotIndex + 1];
186                        this.occupancies.put(stopId, occupancyAtStop);
187                }
188
189                Integer noPassengersInVeh = this.vehPassengers.get(vehId);
190
191                if (noPassengersInVeh != null) {
192                        occupancyAtStop[this.getTimeSlotIndex(event.getTime())] += noPassengersInVeh;
193
194                        this.occupancyRecord.append(event.getTime());
195                        this.occupancyRecord.append("\t");
196                        this.occupancyRecord.append(vehId);
197                        this.occupancyRecord.append("\t");
198                        this.occupancyRecord.append(stopId);
199                        this.occupancyRecord.append("\t");
200                        this.occupancyRecord.append(noPassengersInVeh);
201                        this.occupancyRecord.append("\n");
202                }
203        }
204
205        /**
206         * @param stopId
207         * @return Array containing the number of agents boarding at the stop
208         *         <code>stopId</code> per time bin, starting with time bin 0 from 0
209         *         seconds to (timeBinSize-1)seconds.
210         */
211        public int[] getBoardVolumesForStop(final Id<TransitStopFacility> stopId) {
212                int[] values = this.boards.get(stopId);
213                if (values == null) {
214                        return new int[this.maxSlotIndex + 1];
215                }
216                return values;
217        }
218
219        /**
220         * @param stopId
221         * @return Array containing the number of agents alighting at the stop
222         *         {@code stopId} per time bin, starting with time bin 0 from 0
223         *         seconds to (timeBinSize-1)seconds.
224         */
225        public int[] getAlightVolumesForStop(final Id<TransitStopFacility> stopId) {
226                int[] values = this.alights.get(stopId);
227                if (values == null) {
228                        return new int[this.maxSlotIndex + 1];
229                }
230                return values;
231        }
232
233        /**
234         * @param stopId
235         * @return Array containing the number of passengers in bus after the
236         *         transfer at the stop {@code stopId} per time bin, starting with
237         *         time bin 0 from 0 seconds to (timeBinSize-1)seconds.
238         */
239        public int[] getOccupancyVolumesForStop(final Id<TransitStopFacility> stopId) {
240                int[] values = this.occupancies.get(stopId);
241                if (values == null) {
242                        return new int[this.maxSlotIndex + 1];
243                }
244                return values;
245        }
246
247        /**
248         * @return Set of {@code Id}s containing all stop ids, where the agents
249         *         boarded, for which counting-values are available.
250         */
251        public Set<Id<TransitStopFacility>> getBoardStopIds() {
252                return this.boards.keySet();
253        }
254
255        /**
256         * @return Set of {@code Id}s containing all stop ids, where the agents
257         *         alit, for which counting-values are available.
258         */
259        public Set<Id<TransitStopFacility>> getAlightStopIds() {
260                return this.alights.keySet();
261        }
262
263        public Set<Id<TransitStopFacility>> getOccupancyStopIds() {
264                return this.occupancies.keySet();
265        }
266
267        /**
268         * @return Set of {@code Id}s containing all stop ids, where the agents alit
269         *         or/and boarded, for which counting-values are available.
270         */
271        public Set<Id<TransitStopFacility>> getAllStopIds() {
272                Set<Id<TransitStopFacility>> allStopIds = new TreeSet<>();
273                allStopIds.addAll(getBoardStopIds());
274                allStopIds.addAll(getAlightStopIds());
275                allStopIds.addAll(getOccupancyStopIds());
276                return allStopIds;
277        }

279        public void write(String filename) {
280                SimpleWriter writer = new SimpleWriter(filename);
281                // write filehead
282                writer.write("stopId\t");
283                for (int i = 0; i < 24; i++) {
284                        writer.write("bo" + i + "-" + (i + 1) + "\t");
285                }
286                for (int i = 0; i < 24; i++) {
287                        writer.write("al" + i + "-" + (i + 1) + "\t");
288                }
289                for (int i = 0; i < 24; i++) {
290                        writer.write("oc" + i + "-" + (i + 1) + "\t");
291                }
292                writer.writeln();
293                // write content
294                for (Id<TransitStopFacility> stopId : getAllStopIds()) {
295                        writer.write(stopId + "\t");
296
297                        int[] board = this.boards.get(stopId);
298                        if (board == null){
299                                log.debug("stopId:\t" + stopId + "\thas null boards!");
300                        }
301                        for (int i = 0; i < 24; i++) {
302                                writer.write((board != null ? board[i] : 0) + "\t");
303                        }
304
305                        int[] alight = this.alights.get(stopId);
306                        if (alight == null) {
307                                log.debug("stopId:\t" + stopId + "\thas null alights!");
308                        }
309                        for (int i = 0; i < 24; i++) {
310                                writer.write((alight != null ? alight[i] : 0) + "\t");
311                        }
312
313                        int[] ocuppancy = this.occupancies.get(stopId);
314                        if (ocuppancy == null) {
315                                log.debug("stopId:\t" + stopId + "\tthere aren't passengers in Bus after the transfer!");
316                        }
317                        for (int i = 0; i < 24; i++) {
318                                writer.write((ocuppancy != null ? ocuppancy[i] : 0) + "\t");
319                        }
320                        writer.writeln();
321                }
322                writer.write(this.occupancyRecord.toString());
323                writer.close();
324        }
