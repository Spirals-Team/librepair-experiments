001package org.matsim.pt.counts;
002
003import java.util.List;
004
005import org.matsim.api.core.v01.Id;
006import org.matsim.core.api.internal.MatsimWriter;
007import org.matsim.counts.CountSimComparison;
008import org.matsim.pt.transitSchedule.api.TransitStopFacility;
009
010public class PtCountSimComparisonTableWriter implements MatsimWriter {
011
012        private List<CountSimComparison> countSimCompList;
013
014        public PtCountSimComparisonTableWriter(List<CountSimComparison> countSimCompList) {
015                this.countSimCompList = countSimCompList;
016        }
017
018        @Override
019        public void write(String filename) {
020                SimpleWriter simpleWriter = null;
021                try {
022                        simpleWriter = new SimpleWriter(filename);
023                        Id<TransitStopFacility> lastStopId = null;
024                        // Sorted by stopId
025                        for (CountSimComparison count : countSimCompList) {
026                                Id<TransitStopFacility> stopId = Id.create(count.getId(), TransitStopFacility.class);
027                                if (!stopId.equals(lastStopId)) {
028                                        simpleWriter.write("StopId :\t");
029                                        simpleWriter.write(stopId.toString());
030                                        simpleWriter.write("\nhour\tsimVal\tscaledSimVal\tcountVal\n");
031                                        lastStopId = stopId;
032                                }
033                                simpleWriter.write(count.getHour());
034                                simpleWriter.write('\t');
035
036                                double countValue = count.getCountValue();
037                                double simValue = count.getSimulationValue();
038
039                                simpleWriter.write(simValue);
040                                simpleWriter.write('\t');
041
042                                simpleWriter.write(simValue);
043                                simpleWriter.write('\t');
044                                simpleWriter.write(countValue);
045                                simpleWriter.write('\n');
046
047                        }
048                } finally {
049                        if (simpleWriter != null) {
050                                simpleWriter.close();
051                        }
052                }
053        }
054        
055}
