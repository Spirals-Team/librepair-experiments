001package org.matsim.pt;
002
003import org.matsim.core.population.routes.NetworkRoute;
004import org.matsim.pt.transitSchedule.api.Departure;
005import org.matsim.pt.transitSchedule.api.TransitLine;
006import org.matsim.pt.transitSchedule.api.TransitRoute;
007
008public class UmlaufStueck implements UmlaufStueckI {
009
010        private TransitLine line;
011        private TransitRoute route;
012        private Departure departure;
013        
014        public UmlaufStueck(TransitLine line, TransitRoute route,
015                        Departure departure) {
016                this.line = line;
017                this.route = route;
018                this.departure = departure;
019        }
020
021        @Override
022        public TransitLine getLine() {
023                return line;
024        }
025
026        @Override
027        public TransitRoute getRoute() {
028                return route;
029        }
030
031        @Override
032        public Departure getDeparture() {
033                return departure;
034        }
035
036        @Override
037        public NetworkRoute getCarRoute() {
038                return route.getRoute();
039        }
040
041        @Override
042        public boolean isFahrt() {
043                return true;
044        }
045        
046
047}
