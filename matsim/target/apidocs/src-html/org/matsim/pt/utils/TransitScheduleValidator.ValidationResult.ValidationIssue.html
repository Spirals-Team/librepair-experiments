043/**
044 * An abstract class offering a number of static methods to validate several aspects of transit schedules.
045 * 
046 * @author mrieser
047 */
048public abstract class TransitScheduleValidator {
049
050        private TransitScheduleValidator() {
051                // this class should not be instantiated
052        }
053        
054        /**
055         * Checks that the links specified for a network route really builds a complete route that can be driven along.
056         *
057         * @param schedule
058         * @param network
059         * @return
060         */
061        public static ValidationResult validateNetworkRoutes(final TransitSchedule schedule, final Network network) {
062                ValidationResult result = new ValidationResult();
063                if (network == null || network.getLinks().size() == 0) {
064                        result.addWarning("Cannot validate network routes: No network given!");
065                        return result;
066                }
067
068                for (TransitLine line : schedule.getTransitLines().values()) {
069                        for (TransitRoute route : line.getRoutes().values()) {
070                                NetworkRoute netRoute = route.getRoute();
071                                if (netRoute == null) {
072                                        result.addIssue(new ValidationResult.ValidationIssue(ValidationResult.Severity.ERROR, "Transit line " + line.getId() + ", route " + route.getId() + " has no network route.", ValidationResult.Type.OTHER, Collections.singleton(route.getId())));
073                                } else {
074                                        Link prevLink = network.getLinks().get(netRoute.getStartLinkId());
075                                        for (Id<Link> linkId : netRoute.getLinkIds()) {
076                                                Link link = network.getLinks().get(linkId);
077                                                if (link == null) {
078                                                        result.addIssue(new ValidationResult.ValidationIssue(ValidationResult.Severity.ERROR, "Transit line " + line.getId() + ", route " + route.getId() +
079                                                                        " contains a link that is not part of the network: " + linkId, ValidationResult.Type.OTHER, Collections.singleton(route.getId())));
080                                                } else if (prevLink != null && !prevLink.getToNode().equals(link.getFromNode())) {
081                                                        result.addIssue(new ValidationResult.ValidationIssue(ValidationResult.Severity.ERROR, "Transit line " + line.getId() + ", route " + route.getId() +
082                                                                        " has inconsistent network route, e.g. between link " + prevLink.getId() + " and " + linkId, ValidationResult.Type.OTHER, Collections.singleton(route.getId())));
083                                                }
084                                                prevLink = link;
085                                        }
086                                }
087                        }
088                }
089                return result;
090        }
091
092        /**
093         * Checks that all the listed stops in a route appear in that order when driving along the network route
094         *
095         * @param schedule
096         * @param network
097         * @return
098         */
099        public static ValidationResult validateStopsOnNetworkRoute(final TransitSchedule schedule, final Network network) {
100                ValidationResult result = new ValidationResult();
101                if (network == null || network.getLinks().size() == 0) {
102                        result.addWarning("Cannot validate stops on network route: No network given!");
103                        return result;
104                }
105
106                for (TransitLine line : schedule.getTransitLines().values()) {
107                        for (TransitRoute route : line.getRoutes().values()) {
108                                NetworkRoute netRoute = route.getRoute();
109                                if (netRoute == null) {
110                                        result.addIssue(new ValidationResult.ValidationIssue(ValidationResult.Severity.ERROR, "Transit line " + line.getId() + ", route " + route.getId() + " has no network route.", ValidationResult.Type.OTHER, Collections.singleton(route.getId())));
111                                } else {
112                                        List<Id<Link>> linkIds = new ArrayList<>();
113                                        linkIds.add(netRoute.getStartLinkId());
114                                        linkIds.addAll(netRoute.getLinkIds());
115                                        linkIds.add(netRoute.getEndLinkId());
116                                        Iterator<Id<Link>> linkIdIterator = linkIds.iterator();
117                                        Id<Link> nextLinkId = linkIdIterator.next();
118                                        boolean error = false;
119                                        for (TransitRouteStop stop : route.getStops()) {
120                                                Id<Link> linkRefId = stop.getStopFacility().getLinkId();
121
122                                                while (!linkRefId.equals(nextLinkId)) {
123                                                        if (linkIdIterator.hasNext()) {
124                                                                nextLinkId = linkIdIterator.next();
125                                                        } else {
126                                                                result.addIssue(new ValidationResult.ValidationIssue(ValidationResult.Severity.ERROR, "Transit line " + line.getId() + ", route " + route.getId() + ": Stop " + stop.getStopFacility().getId() + " cannot be reached along network route.", ValidationResult.Type.ROUTE_HAS_UNREACHABLE_STOP, Collections.singletonList(stop.getStopFacility().getId())));
127                                                                error = true;
128                                                                break;
129                                                        }
130                                                }
131                                                if (error) {
132                                                        break;
133                                                }
134
135                                        }
136                                }
137                        }
138                }
139                return result;
140        }
141
142        public static ValidationResult validateUsedStopsHaveLinkId(final TransitSchedule schedule) {
143                ValidationResult result = new ValidationResult();
144                for (TransitLine line : schedule.getTransitLines().values()) {
145                        for (TransitRoute route : line.getRoutes().values()) {
146                                for (TransitRouteStop stop : route.getStops()) {
147                                        Id<Link> linkId = stop.getStopFacility().getLinkId();
148                                        if (linkId == null) {
149                                                result.addIssue(new ValidationResult.ValidationIssue(ValidationResult.Severity.ERROR,"Transit Stop Facility " + stop.getStopFacility().getId() + " has no linkId, but is used by transit line " + line.getId() + ", route " + route.getId(), ValidationResult.Type.HAS_NO_LINK_REF, Collections.singleton(stop.getStopFacility().getId())));
150                                        }
151                                }
152                        }
153                }
154                return result;
155        }
156
157        public static ValidationResult validateAllStopsExist(final TransitSchedule schedule) {
158                ValidationResult result = new ValidationResult();
159                for (TransitLine line : schedule.getTransitLines().values()) {
160                        for (TransitRoute route : line.getRoutes().values()) {
161                                for (TransitRouteStop stop : route.getStops()) {
162                                        if (stop.getStopFacility() == null) {
163                                                result.addIssue(new ValidationResult.ValidationIssue(ValidationResult.Severity.ERROR, "Transit line " + line.getId() + ", route " + route.getId() + " contains a stop (dep-offset=" + stop.getDepartureOffset() + ") without stop-facility. Most likely, a wrong id was specified in the file.", ValidationResult.Type.HAS_MISSING_STOP_FACILITY, Collections.singletonList(route.getId())));
164                                        } else if (schedule.getFacilities().get(stop.getStopFacility().getId()) == null) {
165                                                result.addIssue(new ValidationResult.ValidationIssue(ValidationResult.Severity.ERROR, "Transit line " + line.getId() + ", route " + route.getId() + " contains a stop (stop-facility " + stop.getStopFacility().getId() + ") that is not contained in the list of all stop facilities.", ValidationResult.Type.HAS_MISSING_STOP_FACILITY, Collections.singletonList(route.getId())));
166                                        }
167                                }
168                        }
169                }
170                return result;
171        }
172        
173        public static ValidationResult validateOffsets(final TransitSchedule schedule) {
174                ValidationResult result = new ValidationResult();
175
176                for (TransitLine line : schedule.getTransitLines().values()) {
177                        for (TransitRoute route : line.getRoutes().values()) {
178                                ArrayList<TransitRouteStop> stops = new ArrayList<TransitRouteStop>(route.getStops());
179                                int stopCount = stops.size();
180                                
181                                if (stopCount > 0) {
182                                        TransitRouteStop stop = stops.get(0);
183                                        if (stop.getDepartureOffset() == Time.UNDEFINED_TIME) {
184                                                result.addError("Transit line " + line.getId() + ", route " + route.getId() + ": The first stop does not contain any departure offset.");
185                                        }
186                                        
187                                        for (int i = 1; i < stopCount - 1; i++) {
188                                                stop = stops.get(i);
189                                                if (stop.getDepartureOffset() == Time.UNDEFINED_TIME) {
190                                                        result.addError("Transit line " + line.getId() + ", route " + route.getId() + ": Stop " + i + " does not contain any departure offset.");
191                                                }
192                                        }
193                                        
194                                        stop = stops.get(stopCount - 1);
195                                        if (stop.getArrivalOffset() == Time.UNDEFINED_TIME) {
196                                                result.addError("Transit line " + line.getId() + ", route " + route.getId() + ": The last stop does not contain any arrival offset.");
197                                        }
198                                } else {
199                                        result.addWarning("Transit line " + line.getId() + ", route " + route.getId() + ": The route has not stops assigned, looks suspicious.");
200                                }
201                                
202                        }
203                }
204                
205                return result;
206        }
207
208        public static ValidationResult validateAll(final TransitSchedule schedule, final Network network) {
209                ValidationResult v = validateUsedStopsHaveLinkId(schedule);
210                v.add(validateNetworkRoutes(schedule, network));
211                try {
212                        v.add(validateStopsOnNetworkRoute(schedule, network));
213                } catch (NullPointerException e) {
214                        v.addError("Exception during 'validateStopsOnNetworkRoute'. Most likely something is wrong in the file, but it cannot be specified in more detail." + Arrays.toString(e.getStackTrace()));
215                }
216                v.add(validateAllStopsExist(schedule));
217                v.add(validateOffsets(schedule));
218                return v;
219        }
220        
221        public static void printResult(final ValidationResult result) {
222                if (result.isValid()) {
223                        System.out.println("Schedule appears valid!");
224                } else {
225                        System.out.println("Schedule is NOT valid!");
226                }
227                if (result.getErrors().size() > 0) {
228                        System.out.println("Validation errors:");
229                        for (String e : result.getErrors()) {
230                                System.out.println(e);
231                        }
232                }
233                if (result.getWarnings().size() > 0) {
234                        System.out.println("Validation warnings:");
235                        for (String w : result.getWarnings()) {
236                                System.out.println(w);
237                        }
238                }
239        }

241        /**
242         * @param args [0] path to transitSchedule.xml, [1] path to network.xml (optional)
243         * @throws IOException
244         * @throws SAXException
245         * @throws ParserConfigurationException
246         */
247        public static void main(String[] args) throws IOException, SAXException, ParserConfigurationException {
248                if (args.length > 2 || args.length < 1) {
249                        System.err.println("Usage: TransitScheduleValidator transitSchedule.xml [network.xml]");
250                        return;
251                }
252                
253                MutableScenario s = (MutableScenario) ScenarioUtils.createScenario(ConfigUtils.createConfig());
254                s.getConfig().transit().setUseTransit(true);
255                TransitSchedule ts = s.getTransitSchedule();
256                Network net = s.getNetwork();
257
258                if (args.length > 1) {
259                        new MatsimNetworkReader(s.getNetwork()).readFile(args[1]);
260                }
261                new TransitScheduleReader(s).readFile(args[0]);
262
263                ValidationResult v = validateAll(ts, net);
264                printResult(v);
265        }
266
267        public static class ValidationResult {
268
269                public enum Severity {
270                        WARNING, ERROR;
271                }
272
273                public enum Type {
274                        HAS_MISSING_STOP_FACILITY, HAS_NO_LINK_REF, ROUTE_HAS_UNREACHABLE_STOP, OTHER;
275                }
276
277                public static class ValidationIssue<T> {
278                        private final Severity severity;
279                        private final String message;
280                        private final Type errorCode;
281                        private final Collection<Id<T>> entities;
282
283                        public ValidationIssue(Severity severity, String message, Type errorCode, Collection<Id<T>> entities) {
284                                this.severity = severity;
285                                this.message = message;
286                                this.errorCode = errorCode;
287                                this.entities = entities;
288                        }
289
290                        public Severity getSeverity() {
291                                return severity;
292                        }
293
294                        public String getMessage() {
295                                return message;
296                        }
297
298                        public Type getErrorCode() {
299                                return errorCode;
300                        }
301
302                        public Collection<Id<T>> getEntities() {
303                                return entities;
304                        }
305
306                }
307
308                private boolean isValid = true;
309                private final List<ValidationIssue> issues = new ArrayList<>();
310
311                public boolean isValid() {
312                        return this.isValid;
313                }
314
315                public List<String> getWarnings() {
316                        List<String> result = new ArrayList<>();
317                        for (ValidationIssue issue : this.issues) {
318                                if (issue.severity == Severity.WARNING) {
319                                        result.add(issue.getMessage());
320                                }
321                        }
322                        return Collections.unmodifiableList(result);
323                }
324
325                public List<String> getErrors() {
326                        List<String> result = new ArrayList<>();
327                        for (ValidationIssue issue : this.issues) {
328                                if (issue.severity == Severity.ERROR) {
329                                        result.add(issue.getMessage());
330                                }
331                        }
332                        return Collections.unmodifiableList(result);
333                }
334
335                public List<ValidationIssue> getIssues() {
336                        return Collections.unmodifiableList(this.issues);
337                }
338
339                public void addWarning(final String warning) {
340                        this.issues.add(new ValidationIssue(Severity.WARNING, warning, Type.OTHER, Collections.<Id<?>>emptyList()));
341                }
342
343                public void addError(final String error) {
344                        this.issues.add(new ValidationIssue(Severity.ERROR, error, Type.OTHER, Collections.<Id<?>>emptyList()));
345                        this.isValid = false;
346                }
347
348                public void addIssue(final ValidationIssue issue) {
349                        this.issues.add(issue);
350                }
351
352                public void add(final ValidationResult otherResult) {
353                        this.issues.addAll(otherResult.getIssues());
354                        this.isValid = this.isValid && otherResult.isValid;
355                }
356        }
