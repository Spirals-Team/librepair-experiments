047/**
048 * Builds a network where transit vehicles can drive along and assigns the correct
049 * links to the transit stop facilities and routes of transit lines. As each transit
050 * stop facility can only be connected to at most one link, the algorithm is forced
051 * to duplicated transit stop facilities in certain cases to build the network.
052 *
053 * @author mrieser
054 */
055public class CreatePseudoNetwork {
056
057        private final TransitSchedule schedule;
058        private final Network network;
059        private final String prefix;
060
061        private final Map<Tuple<Node, Node>, Link> links = new HashMap<Tuple<Node, Node>, Link>();
062        private final Map<Tuple<Node, Node>, TransitStopFacility> stopFacilities = new HashMap<Tuple<Node, Node>, TransitStopFacility>();
063        private final Map<TransitStopFacility, Node> nodes = new HashMap<TransitStopFacility, Node>();
064        private final Map<TransitStopFacility, List<TransitStopFacility>> facilityCopies = new HashMap<TransitStopFacility, List<TransitStopFacility>>();
065
066        private long linkIdCounter = 0;
067
068        private final Set<String> transitModes = Collections.singleton(TransportMode.pt);
069
070        public CreatePseudoNetwork(final TransitSchedule schedule, final Network network, final String networkIdPrefix) {
071                this.schedule = schedule;
072                this.network = network;
073                this.prefix = networkIdPrefix;
074        }
075
076        public void createNetwork() {
077
078                List<Tuple<TransitLine, TransitRoute>> toBeRemoved = new LinkedList<Tuple<TransitLine, TransitRoute>>();
079
080                for (TransitLine tLine : this.schedule.getTransitLines().values()) {
081                        for (TransitRoute tRoute : tLine.getRoutes().values()) {
082                                ArrayList<Id<Link>> routeLinks = new ArrayList<Id<Link>>();
083                                TransitRouteStop prevStop = null;
084                                for (TransitRouteStop stop : tRoute.getStops()) {
085                                        Link link = getNetworkLink(prevStop, stop);
086                                        routeLinks.add(link.getId());
087                                        prevStop = stop;
088                                }
089
090                                if (routeLinks.size() > 0) {
091                                        NetworkRoute route = RouteUtils.createNetworkRoute(routeLinks, this.network);
092                                        tRoute.setRoute(route);
093                                } else {
094                                        System.err.println("Line " + tLine.getId() + " route " + tRoute.getId() + " has less than two stops. Removing this route from schedule.");
095                                        toBeRemoved.add(new Tuple<TransitLine, TransitRoute>(tLine, tRoute));
096                                }
097                        }
098                }
099
100                for (Tuple<TransitLine, TransitRoute> remove : toBeRemoved) {
101                        remove.getFirst().removeRoute(remove.getSecond());
102                }
103        }
104
105        private Link getNetworkLink(final TransitRouteStop fromStop, final TransitRouteStop toStop) {
106                TransitStopFacility fromFacility = (fromStop == null) ? toStop.getStopFacility() : fromStop.getStopFacility();
107                TransitStopFacility toFacility = toStop.getStopFacility();
108
109                Node fromNode = this.nodes.get(fromFacility);
110                if (fromNode == null) {
111                        fromNode = this.network.getFactory().createNode(Id.create(this.prefix + toFacility.getId(), Node.class), fromFacility.getCoord());
112                        this.network.addNode(fromNode);
113                        this.nodes.put(toFacility, fromNode);
114                }
115
116                Node toNode = this.nodes.get(toFacility);
117                if (toNode == null) {
118                        toNode = this.network.getFactory().createNode(Id.create(this.prefix + toFacility.getId(), Node.class), toFacility.getCoord());
119                        this.network.addNode(toNode);
120                        this.nodes.put(toFacility, toNode);
121                }
122
123                Tuple<Node, Node> connection = new Tuple<Node, Node>(fromNode, toNode);
124                Link link = this.links.get(connection);
125                if (link == null) {
126                        link = createAndAddLink(fromNode, toNode, connection);
127
128                        if (toFacility.getLinkId() == null) {
129                                toFacility.setLinkId(link.getId());
130                                this.stopFacilities.put(connection, toFacility);
131                        } else {
132                                List<TransitStopFacility> copies = this.facilityCopies.get(toFacility);
133                                if (copies == null) {
134                                        copies = new ArrayList<TransitStopFacility>();
135                                        this.facilityCopies.put(toFacility, copies);
136                                }
137                                Id<TransitStopFacility> newId = Id.create(toFacility.getId().toString() + "." + Integer.toString(copies.size() + 1), TransitStopFacility.class);
138                                TransitStopFacility newFacility = this.schedule.getFactory().createTransitStopFacility(newId, toFacility.getCoord(), toFacility.getIsBlockingLane());
139                                newFacility.setStopAreaId(Id.create(toFacility.getId(), TransitStopArea.class));
140                                newFacility.setLinkId(link.getId());
141                                newFacility.setName(toFacility.getName());
142                                copies.add(newFacility);
143                                this.nodes.put(newFacility, toNode);
144                                this.schedule.addStopFacility(newFacility);
145                                toStop.setStopFacility(newFacility);
146                                this.stopFacilities.put(connection, newFacility);
147                        }
148                } else {
149                        toStop.setStopFacility(this.stopFacilities.get(connection));
150                }
151                return link;
152        }
153
154        private Link createAndAddLink(Node fromNode, Node toNode,
155                        Tuple<Node, Node> connection) {
156                Link link;
157                link = this.network.getFactory().createLink(Id.create(this.prefix + this.linkIdCounter++, Link.class), fromNode, toNode);
158                if (fromNode == toNode) {
159                        link.setLength(50);
160                } else {
161                        link.setLength(CoordUtils.calcEuclideanDistance(fromNode.getCoord(), toNode.getCoord()));
162                }
163                link.setFreespeed(30.0 / 3.6);
164                link.setCapacity(500);
165                link.setNumberOfLanes(1);
166                this.network.addLink(link);
167                link.setAllowedModes(this.transitModes);
168                this.links.put(connection, link);
169                return link;
170        }
171
172        public Link getLinkBetweenStops(final TransitStopFacility fromStop, final TransitStopFacility toStop) {
173                Node fromNode = this.nodes.get(fromStop);
174                Node toNode = this.nodes.get(toStop);
175                Tuple<Node, Node> connection = new Tuple<Node, Node>(fromNode, toNode);
176                return this.links.get(connection);
177        }
