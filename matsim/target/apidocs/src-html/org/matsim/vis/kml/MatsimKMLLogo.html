021package org.matsim.vis.kml;
022
023import java.io.IOException;
024
025import org.matsim.core.gbl.MatsimResource;
026
027import net.opengis.kml.v_2_2_0.LinkType;
028import net.opengis.kml.v_2_2_0.ObjectFactory;
029import net.opengis.kml.v_2_2_0.ScreenOverlayType;
030import net.opengis.kml.v_2_2_0.UnitsEnumType;
031import net.opengis.kml.v_2_2_0.Vec2Type;
032
033/**
034 * A ScreenOverlay representing the MATSim logo
035 * 
036 * @author dgrether
037 */
038public class MatsimKMLLogo {
039        /**
040         * Writes the logo file to the kmz and creates the ScreenOverlay with the logo
041         * @param writer
042         * @return ScreenOverlay containing the MATSim logo to be added to the KMZ file
043         * @throws IOException if the file with the logo could not be read and written
044         */
045        public static ScreenOverlayType writeMatsimKMLLogo(final KMZWriter writer) throws IOException {
046                
047                writer.addNonKMLFile(MatsimResource.getAsInputStream("matsim_logo_transparent.png"), "matsimLogo.png");
048                
049                ObjectFactory kmlObjectFactory = new ObjectFactory();
050
051                ScreenOverlayType matsimLogo = kmlObjectFactory.createScreenOverlayType();
052                
053                LinkType icon = kmlObjectFactory.createLinkType();
054                icon.setHref("./matsimLogo.png");
055                matsimLogo.setIcon(icon);
056                matsimLogo.setName("Matsim Logo");
057                Vec2Type overlayXY = kmlObjectFactory.createVec2Type();
058                overlayXY.setX(1.0);
059                overlayXY.setY(-0.7);
060                overlayXY.setXunits(UnitsEnumType.FRACTION);
061                overlayXY.setYunits(UnitsEnumType.FRACTION);
062                matsimLogo.setOverlayXY(overlayXY);
063                Vec2Type screenXY = kmlObjectFactory.createVec2Type();
064                screenXY.setX(0.85);
065                screenXY.setY(25.0);
066                screenXY.setXunits(UnitsEnumType.FRACTION);
067                screenXY.setYunits(UnitsEnumType.PIXELS);
068                matsimLogo.setScreenXY(screenXY);
069                Vec2Type sizeXY = kmlObjectFactory.createVec2Type();
070                sizeXY.setX(0.14);
071                sizeXY.setY(0.0);
072                sizeXY.setXunits(UnitsEnumType.FRACTION);
073                sizeXY.setYunits(UnitsEnumType.FRACTION);
074                matsimLogo.setSize(sizeXY);
075                matsimLogo.setDrawOrder(Integer.MAX_VALUE);
076                
077                return matsimLogo;
078        }
079
080}
