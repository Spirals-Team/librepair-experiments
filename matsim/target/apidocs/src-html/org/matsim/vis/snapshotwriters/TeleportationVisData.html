021package org.matsim.vis.snapshotwriters;
022
023import org.matsim.api.core.v01.Coord;
024import org.matsim.api.core.v01.Id;
025import org.matsim.api.core.v01.population.Person;
026
027/**
028 *
029 * @author dgrether
030 * 
031 *
032 */
033public class TeleportationVisData implements AgentSnapshotInfo {
034
035        private final double startX;
036        private final double startY;
037        private double currentX;
038        private double currentY;
039        private final double starttime;
040        private final Id<Person> agentId;
041        private int userDefined;
042        private double colorval;
043        private AgentState state = AgentSnapshotInfo.AgentState.PERSON_OTHER_MODE;
044        private int intX;
045        private int intY;
046        private final double endX;
047        private final double endY;
048        private final double travelTime;
049        private final static int offset = 100;
050
051        public TeleportationVisData(double now, Id<Person> personId, Coord fromCoord, Coord toCoord, double travelTime ) {
052                this.starttime = now;
053                this.travelTime = travelTime ;
054                this.agentId = personId;
055                this.startX = fromCoord.getX();
056                this.startY = fromCoord.getY();
057                this.endX = toCoord.getX();
058                this.endY = toCoord.getY();
059                this.currentX = startX;
060                this.currentY = startY;
061                
062                // the following is there to somewhat shift teleported agents.  So that they are not exactly on top of each other when they have
063                // exactly the same dp time and destination ... as may happen e.g. for teleported transit walk and illustrative examples. kai, apr'16
064                String idstr = personId.toString();
065                int hashCode = idstr.hashCode();
066                intX = hashCode%offset;
067                hashCode -= intX;
068                hashCode /= offset;
069                intY = hashCode%offset;
070        }
071
072        @Override
073        public double getEasting() {
074                return this.currentX;
075        }
076
077        @Override
078        public double getNorthing() {
079                return this.currentY;
080        }
081
082        @Override
083        public Id<Person> getId(){
084                return this.agentId;
085        }
086
087        public final void updatePosition(double time) {
088                double frac = (time - starttime) / travelTime ;
089                this.currentX = (1.-frac) * this.startX + frac * this.endX + 0.1*(intX-offset/2) ;
090                this.currentY = (1.-frac) * this.startY + frac * this.endY + 0.1*(intY-offset/2) ;
091        }
092
093        @Override
094        public AgentState getAgentState() {
095                return this.state;
096        }
097
098        @Override
099        public double getAzimuth() {
100                throw new UnsupportedOperationException();
101        }
102
103        @Override
104        public double getColorValueBetweenZeroAndOne() {
105                return this.colorval;
106        }
107
108        @Override
109        public int getUserDefined() {
110                return this.userDefined;
111        }
112
113        @Override
114        public void setAgentState(AgentState state) {
115                this.state = state;
116        }
117
118        @Override
119        public void setColorValueBetweenZeroAndOne(double tmp) {
120                this.colorval = tmp;
121        }
122        @Override
123        public void setUserDefined(int tmp) {
124                this.userDefined = tmp;
125        }
126}
