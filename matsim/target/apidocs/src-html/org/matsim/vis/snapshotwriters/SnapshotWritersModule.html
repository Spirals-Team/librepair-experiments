023package org.matsim.vis.snapshotwriters;
024
025import org.matsim.core.config.Config;
026import org.matsim.core.config.groups.ControlerConfigGroup;
027import org.matsim.core.controler.AbstractModule;
028import org.matsim.core.mobsim.framework.listeners.MobsimListener;
029import org.matsim.core.replanning.ReplanningContext;
030
031import javax.inject.Inject;
032import javax.inject.Provider;
033import java.util.Collection;
034
035public class SnapshotWritersModule extends AbstractModule {
036        @Override
037        public void install() {
038                if (getConfig().controler().getSnapshotFormat().contains("googleearth")) {
039                        addSnapshotWriterBinding().toProvider(KMLSnapshotWriterFactory.class);
040                }
041                if (getConfig().controler().getSnapshotFormat().contains("transims")) {
042                        addSnapshotWriterBinding().toProvider(TransimsSnapshotWriterFactory.class);
043                }
044                if (getConfig().controler().getWriteSnapshotsInterval() != 0) {
045
046                        addMobsimListenerBinding().toProvider(SnapshotWriterManagerProvider.class);
047
048                }
049        }
050
051        private static class SnapshotWriterManagerProvider implements Provider<MobsimListener> {
052
053                private final Config config;
054                private final ControlerConfigGroup controlerConfigGroup;
055                private final ReplanningContext iterationContext;
056                private final Collection<com.google.inject.Provider<SnapshotWriter>> snapshotWriters;
057
058                @Inject
059                private SnapshotWriterManagerProvider(Config config, ControlerConfigGroup controlerConfigGroup,
060                                                                                          ReplanningContext iterationContext,
061                                                                                          Collection<com.google.inject.Provider<SnapshotWriter>> snapshotWriters) {
062                        this.config = config;
063                        this.controlerConfigGroup = controlerConfigGroup;
064                        this.iterationContext = iterationContext;
065                        this.snapshotWriters = snapshotWriters;
066                }
067
068                @Override
069                public MobsimListener get() {
070                        if (iterationContext.getIteration() % controlerConfigGroup.getWriteSnapshotsInterval() == 0) {
071                                SnapshotWriterManager manager = new SnapshotWriterManager(config);
072                                for (com.google.inject.Provider<SnapshotWriter> snapshotWriter : this.snapshotWriters) {
073                                        manager.addSnapshotWriter(snapshotWriter.get());
074                                }
075                                return manager;
076                        } else {
077                                return new NoopMobsimListener();
078                        }
079                }
080
081                private class NoopMobsimListener implements MobsimListener {
082                }
083        }}
