021package org.matsim.vis.snapshotwriters;
022
023import org.matsim.core.config.Config;
024import org.matsim.core.config.groups.ExternalMobimConfigGroup;
025import org.matsim.core.mobsim.framework.ObservableMobsim;
026import org.matsim.core.mobsim.framework.events.MobsimAfterSimStepEvent;
027import org.matsim.core.mobsim.framework.events.MobsimBeforeCleanupEvent;
028import org.matsim.core.mobsim.framework.events.MobsimInitializedEvent;
029import org.matsim.core.mobsim.framework.listeners.MobsimAfterSimStepListener;
030import org.matsim.core.mobsim.framework.listeners.MobsimBeforeCleanupListener;
031import org.matsim.core.mobsim.framework.listeners.MobsimInitializedListener;
032import org.matsim.core.mobsim.qsim.interfaces.Netsim;
033
034import java.util.ArrayList;
035import java.util.Collection;
036import java.util.List;
037
038public class SnapshotWriterManager implements MobsimBeforeCleanupListener, MobsimAfterSimStepListener, MobsimInitializedListener {
039        
040        private final List<SnapshotWriter> snapshotWriters = new ArrayList<SnapshotWriter>();
041        
042        /** time since last snapshot */
043        private double snapshotTime = 0.0;
044
045        final private int snapshotPeriod;
046
047        public SnapshotWriterManager(Config config) {
048                snapshotPeriod = findSnapshotPeriod(config);
049        }
050
051        // yuck
052        private int findSnapshotPeriod(Config config) {
053                if (config.qsim() != null) {
054                        return (int) config.qsim().getSnapshotPeriod();
055//              } else if (config.getModule(SimulationConfigGroup.GROUP_NAME) != null) {
056//                      return (int) ((SimulationConfigGroup) config.getModule(SimulationConfigGroup.GROUP_NAME)).getSnapshotPeriod();
057                } else {
058                        return 1;
059                }
060        }
061
062        @Override
063        public void notifyMobsimInitialized(MobsimInitializedEvent e) {
064                Netsim mobsim = (Netsim) e.getQueueSimulation();
065                this.snapshotTime = Math.floor(mobsim.getSimTimer().getSimStartTime()
066                                / this.snapshotPeriod)
067                                * this.snapshotPeriod;
068                if (this.snapshotTime < mobsim.getSimTimer().getSimStartTime()) {
069                        this.snapshotTime += this.snapshotPeriod;
070                }
071        }
072
073        @Override
074        public void notifyMobsimBeforeCleanup(MobsimBeforeCleanupEvent e) {
075                closeSnapshotWriters();
076        }
077
078        private void closeSnapshotWriters() {
079                for (SnapshotWriter writer : this.snapshotWriters) {
080                        writer.finish();
081                }
082        }
083
084        @Override
085        public void notifyMobsimAfterSimStep(MobsimAfterSimStepEvent e) {
086                double time = e.getSimulationTime();
087                if (time >= this.snapshotTime) {
088                        this.snapshotTime += this.snapshotPeriod;
089                        doSnapshot(time, (VisMobsim) e.getQueueSimulation());
090                }
091        }
092        
093        private void doSnapshot(final double time, VisMobsim visMobsim) {
094                if (!this.snapshotWriters.isEmpty()) {
095                        Collection<AgentSnapshotInfo> positions = new ArrayList<AgentSnapshotInfo>();
096                        for (VisLink link : visMobsim.getVisNetwork().getVisLinks().values()) {
097                                link.getVisData().addAgentSnapshotInfo(positions);
098                        }
099                        
100                        // We do not put non-network agents in movies.
101                        // Otherwise, we would add snapshots from visMobsim.getNonNetworkAgentSnapshots() here.
102                        
103                        for (SnapshotWriter writer : this.snapshotWriters) {
104                                writer.beginSnapshot(time);
105                                for (AgentSnapshotInfo position : positions) {
106                                        writer.addAgent(position);
107                                }
108                                writer.endSnapshot();
109                        }
110                }
111        }
112
113        public final void addSnapshotWriter(SnapshotWriter snapshotWriter) {
114                this.snapshotWriters.add(snapshotWriter);
115        }
116
117}
