020package org.matsim.vis.snapshotwriters;
021
022
023/**
024 * @author dgrether
025 *
026 */
027public class SnapshotLinkWidthCalculator {
028        /**
029         * Distance (in m) between the two innermost opposing lanes. Setting this to a larger number (e.g. 30) lets you see the
030         * direction in which cars are going. (Setting this to a negative number probably gives you "driving on the left", but lanes
031         * will be wrong (could be fixed), and, more importantly, the simulation still assumes "driving on the right" when locations
032         * (with coordinates) are connected to links.) TODO should be configurable
033         */
034        private double widthOfMedian = 30. ; // default
035
036        private double laneWidth = 3.75; //default in NetworkImpl and network_v1.dtd
037        
038        public void setLaneWidth( double dd ) {
039                if ( !Double.isNaN(dd) ) {
040                        // ( seems that this can be NaN when coming from reading a regular network. kai, jul'16 )
041                        laneWidth = dd ;
042                }
043        }
044        
045        public double getLaneWidth(){
046                return laneWidth;
047        }
048        
049        public void setLinkWidthForVis(double linkWidthCorrectionFactor){
050                this.widthOfMedian = linkWidthCorrectionFactor;
051        }
052        
053        public double calculateLinkWidth(double nrOfLanes){
054//              System.err.println(widthOfMedian + " "  + laneWidth + " " + nrOfLanes);
055                return widthOfMedian + laneWidth * nrOfLanes; 
056        }
057        
058        double calculateLanePosition(double lane){
059                return 0.5 * widthOfMedian + laneWidth * lane;
060        }
061        
062}
