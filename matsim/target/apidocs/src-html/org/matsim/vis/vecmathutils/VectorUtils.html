020package org.matsim.vis.vecmathutils;
021
022import java.awt.geom.Point2D;
023
024import org.matsim.core.utils.collections.Tuple;
025
026/**
027 * Util class containing stateless vecmath utility methods frequently needed for visualization code.
028 * @author dgrether
029 * @author tthunig
030 * 
031 */
032public class VectorUtils {
033
034        /**
035         * Scales a vector given by start and end point by a factor. Changes both - start and end coordinate of the vector.
036         * @param scaleFactor a scale factor in [0..1]
037         * @return a tuple with the scaled start point as first and the end point as
038         *         second entry
039         */
040        public static Tuple<Point2D.Double, Point2D.Double> scaleVector(final Point2D.Double start, final Point2D.Double end,
041                        double scaleFactor) {
042                
043                return calculateScaledVector(start, end, scaleFactor);
044        }
045
046        /**
047         * Scales a vector given by start and end point by a factor. Changes only the start coordinate of the vector.
048         * @param scaleFactor a scale factor in [0..1]
049         * @return a tuple with the scaled start point as first and the old end point as
050         *         second entry
051         * @deprecated not used yet
052         */
053        @Deprecated
054        public static Tuple<Point2D.Double, Point2D.Double> scaleVectorStart(final Point2D.Double start, final Point2D.Double end,
055                        double scaleFactor) {
056                
057                Tuple<Point2D.Double, Point2D.Double> scaledVector = calculateScaledVector(start, end, scaleFactor);
058                return new Tuple<Point2D.Double, Point2D.Double>(scaledVector.getFirst(), end);
059        }
060        
061        /**
062         * Scales a vector given by start and end point by a factor. Changes only the end coordinate of the vector.
063         * @param scaleFactor a scale factor in [0..1]
064         * @return a tuple with the old start point as first and the scaled end point as
065         *         second entry
066         * @deprecated not used yet
067         */
068        @Deprecated
069        public static Tuple<Point2D.Double, Point2D.Double> scaleVectorEnd(final Point2D.Double start, final Point2D.Double end,
070                        double scaleFactor) {
071                
072                Tuple<Point2D.Double, Point2D.Double> scaledVector = calculateScaledVector(start, end, scaleFactor);
073                return new Tuple<Point2D.Double, Point2D.Double>(start, scaledVector.getSecond());
074        }
075        
076        
077        private static Tuple<Point2D.Double, Point2D.Double> calculateScaledVector(
078                        final Point2D.Double start, final Point2D.Double end,
079                        double scaleFactor) {
080                //norm
081                Point2D.Double.Double delta = new Point2D.Double.Double(end.x - start.x, end.y - start.y);
082                double length = Math.sqrt(Math.pow(delta.x, 2) + Math.pow(delta.y, 2));
083                Point2D.Double.Double deltaNorm = new Point2D.Double.Double(delta.x / length, delta.y / length);
084                //scale
085                double scaledLength = length * scaleFactor;
086                double offset = (length - scaledLength) / 2;
087                Point2D.Double scaledEnd = new Point2D.Double(start.x + ((length - offset) * deltaNorm.x), start.y
088                                + ((length - offset) * deltaNorm.y));
089                Point2D.Double scaledStart = new Point2D.Double(start.x
090                                + (offset * deltaNorm.x), start.y
091                                + (offset * deltaNorm.y));
092                Tuple<Point2D.Double, Point2D.Double> scaledVector = new Tuple<Point2D.Double, Point2D.Double>(scaledStart, scaledEnd);
093                return scaledVector;
094        }
095}
