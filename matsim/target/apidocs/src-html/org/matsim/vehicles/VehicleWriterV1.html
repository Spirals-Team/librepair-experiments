020package org.matsim.vehicles;
021
022import java.util.ArrayList;
023import java.util.List;
024import java.util.Map;
025
026import org.apache.log4j.Logger;
027import org.matsim.api.core.v01.Id;
028import org.matsim.core.utils.collections.Tuple;
029import org.matsim.core.utils.io.MatsimXmlWriter;
030import org.matsim.core.utils.io.UncheckedIOException;
031
032/**
033 * @author dgrether
034 * @author jwjoubert
035 */
036public class VehicleWriterV1 extends MatsimXmlWriter {
037  
038        private static final Logger log = Logger.getLogger(VehicleWriterV1.class);
039        
040        private List<Tuple<String, String>> atts = new ArrayList<Tuple<String, String>>();
041        private Map<Id<VehicleType>, VehicleType> vehicleTypes;
042        private Map<Id<Vehicle>, Vehicle> vehicles;
043
044        
045        public VehicleWriterV1(Vehicles vehicles) {
046                this.vehicleTypes = vehicles.getVehicleTypes();
047                this.vehicles = vehicles.getVehicles();
048        }
049        
050        public void writeFile(String filename) throws UncheckedIOException {
051                this.openFile(filename);
052                this.writeXmlHead();
053                this.writeRootElement();
054                this.close();
055                log.info("Vehicles written to: " + filename);
056        }
057
058        private void writeRootElement() throws UncheckedIOException {
059                atts.clear();
060                atts.add(this.createTuple(XMLNS, MatsimXmlWriter.MATSIM_NAMESPACE));
061                atts.add(this.createTuple(XMLNS + ":xsi", DEFAULTSCHEMANAMESPACELOCATION));
062                atts.add(this.createTuple("xsi:schemaLocation", MATSIM_NAMESPACE + " " + DEFAULT_DTD_LOCATION + "vehicleDefinitions_v1.0.xsd"));
063                this.writeStartTag(VehicleSchemaV1Names.VEHICLEDEFINITIONS, atts);
064                this.writeVehicleTypes(this.vehicleTypes);
065                this.writeVehicles(this.vehicles);
066                this.writeEndTag(VehicleSchemaV1Names.VEHICLEDEFINITIONS);
067        }
068
069        private void writeVehicles(Map<Id<Vehicle>, Vehicle> veh) throws UncheckedIOException {
070                for (Vehicle v : veh.values()) {
071                        atts.clear();
072                        atts.add(this.createTuple(VehicleSchemaV1Names.ID, v.getId().toString()));
073                        atts.add(this.createTuple(VehicleSchemaV1Names.TYPE, v.getType().getId().toString()));
074                        this.writeStartTag(VehicleSchemaV1Names.VEHICLE, atts, true);
075                }
076        }
077
078        private void writeVehicleTypes(Map<Id<VehicleType>, VehicleType> vts) throws UncheckedIOException {
079                for (VehicleType vt : vts.values()) {
080                        atts.clear();
081                        atts.add(this.createTuple(VehicleSchemaV1Names.ID, vt.getId().toString()));
082                        this.writeStartTag(VehicleSchemaV1Names.VEHICLETYPE, atts);
083                        if (vt.getDescription() != null) {
084                                this.writeStartTag(VehicleSchemaV1Names.DESCRIPTION, null);
085                                this.writeContent(vt.getDescription(), true);
086                                this.writeEndTag(VehicleSchemaV1Names.DESCRIPTION);
087                        }
088                        if (vt.getCapacity() != null) {
089                                this.writeCapacity(vt.getCapacity());
090                        }
091                        if (!Double.isNaN(vt.getLength())){
092                                atts.clear();
093                                atts.add(this.createTuple(VehicleSchemaV1Names.METER, Double.toString(vt.getLength())));
094                                this.writeStartTag(VehicleSchemaV1Names.LENGTH, atts, true);
095                        }
096                        if (!Double.isNaN(vt.getWidth())){
097                                atts.clear();
098                                atts.add(this.createTuple(VehicleSchemaV1Names.METER, Double.toString(vt.getWidth())));
099                                this.writeStartTag(VehicleSchemaV1Names.WIDTH, atts, true);
100                        }
101                        if (!Double.isNaN(vt.getMaximumVelocity()) && !Double.isInfinite(vt.getMaximumVelocity())){
102                                atts.clear();
103                                atts.add(this.createTuple(VehicleSchemaV1Names.METERPERSECOND, Double.toString(vt.getMaximumVelocity())));
104                                this.writeStartTag(VehicleSchemaV1Names.MAXIMUMVELOCITY, atts, true);
105                        }
106                        if (vt.getEngineInformation() != null) {
107                                this.writeEngineInformation(vt.getEngineInformation());
108                        }
109                        atts.clear();
110                        atts.add(this.createTuple(VehicleSchemaV1Names.SECONDSPERPERSON, vt.getAccessTime()));
111                        this.writeStartTag(VehicleSchemaV1Names.ACCESSTIME, atts, true);
112                        atts.clear();
113      atts.add(this.createTuple(VehicleSchemaV1Names.SECONDSPERPERSON, vt.getEgressTime()));
114      this.writeStartTag(VehicleSchemaV1Names.EGRESSTIME, atts, true);
115      atts.clear();
116      atts.add(this.createTuple(VehicleSchemaV1Names.MODE, vt.getDoorOperationMode().toString()));
117      this.writeStartTag(VehicleSchemaV1Names.DOOROPERATION, atts, true);
118      atts.clear();
119      atts.add(this.createTuple(VehicleSchemaV1Names.PCE, vt.getPcuEquivalents()));
120      this.writeStartTag(VehicleSchemaV1Names.PASSENGERCAREQUIVALENTS, atts, true);
121                        this.writeEndTag(VehicleSchemaV1Names.VEHICLETYPE);
122                }
123        }
124
125        private void writeEngineInformation(EngineInformation ei) throws UncheckedIOException {
126                this.writeStartTag(VehicleSchemaV1Names.ENGINEINFORMATION, null);
127                this.writeStartTag(VehicleSchemaV1Names.FUELTYPE, null);
128                this.writeContent(ei.getFuelType().toString(), false);
129                this.writeEndTag(VehicleSchemaV1Names.FUELTYPE);
130                atts.clear();
131                atts.add(this.createTuple(VehicleSchemaV1Names.LITERPERMETER, Double.toString(ei.getGasConsumption())));
132                this.writeStartTag(VehicleSchemaV1Names.GASCONSUMPTION, atts, true);
133                this.writeEndTag(VehicleSchemaV1Names.ENGINEINFORMATION);
134        }
135
136        private void writeCapacity(VehicleCapacity cap) throws UncheckedIOException {
137                this.writeStartTag(VehicleSchemaV1Names.CAPACITY, null);
138                if (cap.getSeats() != null) {
139                        atts.clear();
140                        atts.add(this.createTuple(VehicleSchemaV1Names.PERSONS, cap.getSeats()));
141                        this.writeStartTag(VehicleSchemaV1Names.SEATS, atts, true);
142                }
143                if (cap.getStandingRoom() != null) {
144                        atts.clear();
145                        atts.add(this.createTuple(VehicleSchemaV1Names.PERSONS, cap.getStandingRoom()));
146                        this.writeStartTag(VehicleSchemaV1Names.STANDINGROOM, atts, true);
147                }
148                if (cap.getFreightCapacity() != null) {
149                        this.writeFreightCapacity(cap.getFreightCapacity());
150                }
151                this.writeEndTag(VehicleSchemaV1Names.CAPACITY);
152        }
153
154        private void writeFreightCapacity(FreightCapacity fc) throws UncheckedIOException {
155                this.writeStartTag(VehicleSchemaV1Names.FREIGHTCAPACITY, null);
156                atts.clear();
157                atts.add(this.createTuple(VehicleSchemaV1Names.CUBICMETERS, Double.toString(fc.getVolume())));
158                this.writeStartTag(VehicleSchemaV1Names.VOLUME, atts, true);
159                this.writeEndTag(VehicleSchemaV1Names.FREIGHTCAPACITY);
160        }
161        
162}
