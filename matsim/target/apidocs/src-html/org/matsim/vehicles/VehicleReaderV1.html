020package org.matsim.vehicles;
021
022import java.util.Stack;
023
024import org.apache.log4j.Logger;
025import org.matsim.api.core.v01.Id;
026import org.matsim.core.utils.io.MatsimXmlParser;
027import org.matsim.core.utils.io.UncheckedIOException;
028import org.matsim.vehicles.EngineInformation.FuelType;
029import org.matsim.vehicles.VehicleType.DoorOperationMode;
030import org.xml.sax.Attributes;
031
032/**
033 * @author dgrether
034 */
035public class VehicleReaderV1 extends MatsimXmlParser {
036
037        private final static Logger log = Logger.getLogger(VehicleReaderV1.class);
038        
039        private final Vehicles vehicles;
040        private final VehiclesFactory builder;
041        private VehicleType currentVehType = null;
042        private VehicleCapacity currentCapacity = null;
043        private FreightCapacity currentFreightCap = null;
044        private EngineInformation.FuelType currentFuelType = null;
045        private double currentGasConsumption = Double.NaN;
046
047        public VehicleReaderV1(final Vehicles vehicles) {
048                this.vehicles = vehicles;
049                this.builder = this.vehicles.getFactory();
050        }
051
052        @Override
053        public void endTag(final String name, final String content, final Stack<String> context) {
054                if (VehicleSchemaV1Names.DESCRIPTION.equalsIgnoreCase(name) && (content.trim().length() > 0)){
055                        this.currentVehType.setDescription(content.trim());
056                }
057                else if (VehicleSchemaV1Names.ENGINEINFORMATION.equalsIgnoreCase(name)){
058                        EngineInformation currentEngineInfo = this.builder.createEngineInformation(this.currentFuelType, this.currentGasConsumption);
059                        this.currentVehType.setEngineInformation(currentEngineInfo);
060                        this.currentFuelType = null;
061                        this.currentGasConsumption = Double.NaN;
062                }
063                else if (VehicleSchemaV1Names.FUELTYPE.equalsIgnoreCase(name)){
064                        this.currentFuelType = this.parseFuelType(content.trim());
065                }
066                else if (VehicleSchemaV1Names.FREIGHTCAPACITY.equalsIgnoreCase(name)){
067                        this.currentCapacity.setFreightCapacity(this.currentFreightCap);
068                        this.currentFreightCap = null;
069                }
070                else if (VehicleSchemaV1Names.CAPACITY.equalsIgnoreCase(name)) {
071                        this.currentVehType.setCapacity(this.currentCapacity);
072                        this.currentCapacity = null;
073                }
074                else if (VehicleSchemaV1Names.VEHICLETYPE.equalsIgnoreCase(name)){
075                        this.vehicles.addVehicleType(this.currentVehType);
076                        this.currentVehType = null;
077                }
078
079        }
080
081        private FuelType parseFuelType(final String content) {
082                if (FuelType.gasoline.toString().equalsIgnoreCase(content)){
083                        return FuelType.gasoline;
084                }
085                else if (FuelType.diesel.toString().equalsIgnoreCase(content)){
086                        return FuelType.diesel;
087                }
088                else if (FuelType.electricity.toString().equalsIgnoreCase(content)){
089                        return FuelType.electricity;
090                }
091                else if (FuelType.biodiesel.toString().equalsIgnoreCase(content)){
092                        return FuelType.biodiesel;
093                }
094                else {
095                        throw new IllegalArgumentException("Fuel type: " + content + " is not supported!");
096                }
097        }
098        
099        private DoorOperationMode parseDoorOperationMode(final String modeString){
100                if (DoorOperationMode.serial.toString().equalsIgnoreCase(modeString)){
101                        return DoorOperationMode.serial;
102                }
103                else if (DoorOperationMode.parallel.toString().equalsIgnoreCase(modeString)){
104                        return DoorOperationMode.parallel;
105                }
106                else {
107                        throw new IllegalArgumentException("Door operation mode " + modeString + " is not supported");
108                }
109        }
110
111        @Override
112        public void startTag(final String name, final Attributes atts, final Stack<String> context) {
113                if (VehicleSchemaV1Names.VEHICLETYPE.equalsIgnoreCase(name)) {
114                        this.currentVehType = this.builder.createVehicleType(Id.create(atts.getValue(VehicleSchemaV1Names.ID), VehicleType.class));
115                }
116                else if (VehicleSchemaV1Names.LENGTH.equalsIgnoreCase(name)){
117                        this.currentVehType.setLength(Double.parseDouble(atts.getValue(VehicleSchemaV1Names.METER)));
118                }
119                else if (VehicleSchemaV1Names.WIDTH.equalsIgnoreCase(name)) {
120                        this.currentVehType.setWidth(Double.parseDouble(atts.getValue(VehicleSchemaV1Names.METER)));
121                }
122                else if (VehicleSchemaV1Names.MAXIMUMVELOCITY.equalsIgnoreCase(name)) {
123                        double val = Double.parseDouble(atts.getValue(VehicleSchemaV1Names.METERPERSECOND));
124                        if (val == 1.0) {
125                                log.warn("The vehicle type's maximum velocity is set to 1.0 meter per second, is this really intended? vehicletype = " + this.currentVehType.getId().toString());
126                        }
127                        this.currentVehType.setMaximumVelocity(val);
128                }
129                else if (VehicleSchemaV1Names.CAPACITY.equalsIgnoreCase(name)) {
130                        this.currentCapacity = this.builder.createVehicleCapacity();
131                }
132                else if(VehicleSchemaV1Names.SEATS.equalsIgnoreCase(name)){
133                        this.currentCapacity.setSeats(Integer.valueOf(atts.getValue(VehicleSchemaV1Names.PERSONS)));
134                }
135                else if (VehicleSchemaV1Names.STANDINGROOM.equalsIgnoreCase(name)){
136                        this.currentCapacity.setStandingRoom(Integer.valueOf(atts.getValue(VehicleSchemaV1Names.PERSONS)));
137                }
138                else if (VehicleSchemaV1Names.FREIGHTCAPACITY.equalsIgnoreCase(name)){
139                        this.currentFreightCap = this.builder.createFreigthCapacity();
140                }
141                else if (VehicleSchemaV1Names.VOLUME.equalsIgnoreCase(name)){
142                        this.currentFreightCap.setVolume(Double.parseDouble(atts.getValue(VehicleSchemaV1Names.CUBICMETERS)));
143                }
144                else if (VehicleSchemaV1Names.GASCONSUMPTION.equalsIgnoreCase(name)){
145                        this.currentGasConsumption = Double.parseDouble(atts.getValue(VehicleSchemaV1Names.LITERPERMETER));
146                }
147                else if (VehicleSchemaV1Names.VEHICLE.equalsIgnoreCase(name)){
148                        Id<VehicleType> typeId = Id.create(atts.getValue(VehicleSchemaV1Names.TYPE), VehicleType.class);
149                        VehicleType type = this.vehicles.getVehicleTypes().get(typeId);
150                        if (type == null) {
151                                log.error("VehicleType " + typeId + " does not exist.");
152                        }
153                        String idString = atts.getValue(VehicleSchemaV1Names.ID);
154                        Id<Vehicle> id = Id.create(idString, Vehicle.class);
155                        Vehicle v = this.builder.createVehicle(id, type);
156                        this.vehicles.addVehicle(v);
157                }
158                else if (VehicleSchemaV1Names.ACCESSTIME.equalsIgnoreCase(name)){
159                  this.currentVehType.setAccessTime(Double.parseDouble(atts.getValue(VehicleSchemaV1Names.SECONDSPERPERSON)));
160                }
161                else if (VehicleSchemaV1Names.EGRESSTIME.equalsIgnoreCase(name)){
162                  this.currentVehType.setEgressTime(Double.parseDouble(atts.getValue(VehicleSchemaV1Names.SECONDSPERPERSON)));
163                }
164                else if (VehicleSchemaV1Names.DOOROPERATION.equalsIgnoreCase(name)){
165                        this.currentVehType.setDoorOperationMode(this.parseDoorOperationMode(atts.getValue(VehicleSchemaV1Names.MODE)));
166                }
167                else if (VehicleSchemaV1Names.PASSENGERCAREQUIVALENTS.equalsIgnoreCase(name)){
168                        this.currentVehType.setPcuEquivalents(Double.parseDouble(atts.getValue(VehicleSchemaV1Names.PCE)));
169                }
170        }
171
172}
