020package org.matsim.vehicles;
021
022import org.matsim.api.core.v01.Id;
023
024/**
025 * @author dgrether
026 */
027public class VehicleTypeImpl implements VehicleType {
028
029        private double width = 1.0;
030        private double maxVelocity = Double.POSITIVE_INFINITY; // default: constrained only by the link speed
031        private double length = 7.5;
032    private double pcuEquivalents = 1.0;
033    private double flowEfficiencyFactor = 1.0;
034        private EngineInformation engineInformation;
035        private String description;
036        private VehicleCapacity capacity;
037        
038        /**
039         * default from xml schema
040         */
041        private double accessTime = 1.0;
042        
043        /**
044         * default from xml schema
045         */
046        private double egressTime = 1.0;
047
048        private Id<VehicleType> id;
049
050        private DoorOperationMode doorOperationMode = DoorOperationMode.serial;
051
052        public VehicleTypeImpl(Id<VehicleType> typeId) {
053                this.id = typeId;
054        }
055
056        @Override
057        public void setCapacity(VehicleCapacity capacity) {
058                this.capacity = capacity;
059        }
060
061        @Override
062        public void setDescription(String desc) {
063                this.description = desc;
064        }
065
066        @Override
067        public void setEngineInformation(EngineInformation engineInformation) {
068                this.engineInformation = engineInformation;
069        }
070
071        @Override
072        public void setLength(double length) {
073                this.length = length;
074        }
075
076        @Override
077        public void setMaximumVelocity(double meterPerSecond) {
078                this.maxVelocity = meterPerSecond;
079        }
080
081        @Override
082        public void setWidth(double width) {
083                this.width = width;
084        }
085
086        @Override
087        public double getWidth() {
088                return width;
089        }
090
091        @Override
092        public double getMaximumVelocity() {
093                return maxVelocity;
094        }
095
096        @Override
097        public double getLength() {
098                return length;
099        }
100
101        @Override
102        public EngineInformation getEngineInformation() {
103                return engineInformation;
104        }
105
106        @Override
107        public String getDescription() {
108                return description;
109        }
110
111        @Override
112        public VehicleCapacity getCapacity() {
113                return capacity;
114        }
115
116        @Override
117        public Id<VehicleType> getId() {
118                return id;
119        }
120
121        @Override
122        public double getAccessTime() {
123                return this.accessTime;
124        }
125
126        @Override
127        public double getEgressTime() {
128                return this.egressTime;
129        }
130
131        @Override
132        public void setAccessTime(double seconds) {
133                this.accessTime = seconds;
134        }
135
136        @Override
137        public void setEgressTime(double seconds) {
138                this.egressTime = seconds;
139        }
140
141        @Override
142        public DoorOperationMode getDoorOperationMode() {
143                return this.doorOperationMode;
144        }
145
146        @Override
147        public void setDoorOperationMode(DoorOperationMode mode) {
148                this.doorOperationMode = mode;
149        }
150
151    @Override
152    public double getPcuEquivalents() {
153        return pcuEquivalents;
154    }
155
156    @Override
157    public void setPcuEquivalents(double pcuEquivalents) {
158        this.pcuEquivalents = pcuEquivalents;
159    }
160
161    @Override
162    public double getFlowEfficiencyFactor() {
163        return flowEfficiencyFactor;
164    }
165
166    @Override
167    public void setFlowEfficiencyFactor(double flowEfficiencyFactor) {
168        this.flowEfficiencyFactor = flowEfficiencyFactor;
169    }
170}
