020package org.matsim.vehicles;
021
022import org.matsim.api.core.v01.Id;
023
024
025/**
026 * @author dgrether
027 */
028public interface VehicleType {
029        
030        public enum DoorOperationMode {serial, parallel}
031
032        public void setDescription(String desc);
033
034        public void setLength(double length);
035
036        public void setWidth(double width);
037
038        public void setMaximumVelocity(double meterPerSecond);
039
040        public void setEngineInformation(EngineInformation currentEngineInfo);
041
042        public void setCapacity(VehicleCapacity capacity);
043
044        public double getWidth();
045
046        public double getMaximumVelocity();
047        
048        public double getLength();
049        
050        public EngineInformation getEngineInformation();
051        
052        public String getDescription();
053        
054        /**
055         * Comments:<ul>
056         * <li> What happens with multi-vehicle trains?  I assume they need to be defined as a single vehicle?  kai, jul'11
057         * </ul>
058         */
059        public VehicleCapacity getCapacity();
060        
061        public Id<VehicleType> getId();
062        
063        /**
064         * Comments:<ul>
065         * <li> In my understanding, ``access'' time is the time to reach the public transit system (e.g. by walking).  
066         * See, e.g., http://en.wikipedia.org/wiki/Public_Transport_Accessibility_Level .  It is thus not the same as
067         * the time for one person to enter the vehicle.  Or is there an alternative definition somewhere? kai, jul'11
068         * <li> Even if this is understood as the time per person to enter the vehicle, it needs to be clear that this is, at
069         * best, the emtpy vehicle entering time, and there may be additional functions computing longer vehicle entering times 
070         * when the vehicle is full. kai, jul'11
071         * <li> Finally, the time needs to be understood as capacitated.  I.e. when there are enough doors so that 9 passengers
072         * can enter the vehicle per second, then this time should be 1/9=0.111. kai, jul'11
073         * </ul>
074         */
075        @Deprecated
076        public double getAccessTime();
077        
078        /**
079         * See comments under getter.
080         */
081        @Deprecated
082        public void setAccessTime(double seconds);
083        
084        /**
085         * Comments:<ul>
086         * <li> In my understanding, ``egress'' time is either the time from the vehicle to the final destination
087         * (e.g. http://www.worldtransitresearch.info/research/1507/ ), or 
088         * the time that is necessary to evacuate the vehicle.  Both definitions are decidedly not the same as the 
089         * time for one person to leave the vehicle.  kai, jul'11  
090         * <li> Also see remarks under getAccessTime() .
091         * </ul>
092         */
093        @Deprecated
094        public double getEgressTime();
095        
096        /**
097         * See comments under getter.
098         */
099        @Deprecated
100        public void setEgressTime(double seconds);
101        
102        public DoorOperationMode getDoorOperationMode();
103        
104        public void setDoorOperationMode(DoorOperationMode mode);
105        
106        public double getPcuEquivalents();
107
108        public void setPcuEquivalents(double pcuEquivalents);
109
110    public double getFlowEfficiencyFactor();
111
112        public void setFlowEfficiencyFactor(double flowEfficiencyFactor);
113}
