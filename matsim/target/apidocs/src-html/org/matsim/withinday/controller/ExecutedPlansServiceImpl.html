049/**
050 * Take the plans that the agents have after within-day replanning, and write them to file.
051 * <br>
052 * Not the same as the {@link ExperiencedPlansService}, since that is based on the events.
053 * 
054 * @author (of documentation) nagel
055  */
056public class ExecutedPlansServiceImpl implements AfterMobsimListener, ExecutedPlansService {
057        // I renamed this from ExperiencedPlansWriter into ExecutedPlansWriter since we also have an ExperiencedPlansService that
058        // reconstructs experienced plans from events. kai, jun'16
059        
060        private static final Logger log = Logger.getLogger( ExecutedPlansServiceImpl.class );
061
062        public static final String EXECUTEDPLANSFILE = "executedPlans.xml.gz";
063
064        private Population experiencedPopulation ;
065
066        private Scenario scenario;
067
068        private MobsimDataProvider mobsimDataProvider;
069
070        private OutputDirectoryHierarchy controlerIO;
071        
072        @Inject
073        ExecutedPlansServiceImpl( Scenario scenario, MobsimDataProvider mobsimDataProvider, OutputDirectoryHierarchy controlerIO ) {
074                this.scenario = scenario;
075                this.mobsimDataProvider = mobsimDataProvider;
076                this.controlerIO = controlerIO;
077        }
078        
079        @Override
080        public void notifyAfterMobsim(AfterMobsimEvent event) {
081                Gbl.assertNotNull(scenario);
082                final Config config = scenario.getConfig();
083                Scenario experiencedScenario = ScenarioUtils.createScenario(config);
084                experiencedPopulation = experiencedScenario.getPopulation(); 
085                                
086                for (Person person : scenario.getPopulation().getPersons().values()) {
087                        
088                        MobsimAgent agent = this.mobsimDataProvider.getAgent(person.getId());
089                        
090                        if (agent == null || !(agent instanceof PersonDriverAgentImpl)) experiencedPopulation.addPerson(person);
091                        else {
092                                Person experiencedPerson = experiencedPopulation.getFactory().createPerson(person.getId());
093                                
094                                // add experienced plan
095                                final Plan plan = WithinDayAgentUtils.getModifiablePlan(agent);
096                                experiencedPerson.addPlan(plan);
097                                experiencedPerson.setSelectedPlan(plan);
098                                
099                                // copy attributes
100                                PersonUtils.setAge(experiencedPerson, PersonUtils.getAge(person));
101                                PersonUtils.setCarAvail(experiencedPerson, PersonUtils.getCarAvail(person));
102                                PersonUtils.setEmployed(experiencedPerson, PersonUtils.isEmployed(person));
103                                PersonUtils.setLicence(experiencedPerson, PersonUtils.getLicense(person));
104                                PersonUtils.setSex(experiencedPerson, PersonUtils.getSex(person));
105
106                                experiencedPopulation.addPerson(experiencedPerson);
107                        }
108                }
109
110                // yy write this in every iteration in order to be consistent with previous design.  I think this should be changed.  kai, jun'16
111                String outputFile = controlerIO.getIterationFilename(event.getIteration(), EXECUTEDPLANSFILE);
112                writeExecutedPlans( outputFile ) ;
113        }
114
115        @Override
116        public void writeExecutedPlans(String outputFile) {
117                final Config config = scenario.getConfig();
118                final String inputCRS = config.plans().getInputCRS();
119                final String internalCRS = config.global().getCoordinateSystem();
120
121                if ( inputCRS == null ) {
122                        new PopulationWriter(experiencedPopulation, scenario.getNetwork()).write(outputFile);
123                }
124                else {
125                        log.info( "re-projecting \"experienced\" population from "+internalCRS+" to "+inputCRS+" for export" );
126
127                        final CoordinateTransformation transformation =
128                                        TransformationFactory.getCoordinateTransformation(
129                                                        internalCRS,
130                                                        inputCRS );
131
132                        new PopulationWriter(transformation, experiencedPopulation, scenario.getNetwork()).write(outputFile);
133                }
134        }
135
136        @Override
137        public Map<Id<Person>, Plan> getExecutedPlans() {
138                Map<Id<Person>,Plan> map = new HashMap<>() ;
139                for ( Person pp : this.experiencedPopulation.getPersons().values() ) {
140                        map.put( pp.getId(), pp.getSelectedPlan() ) ;
141                }
142                return map ;
143        }
144}
