021package org.matsim.withinday.controller;
022
023import org.apache.log4j.Logger;
024import org.matsim.api.core.v01.Scenario;
025import org.matsim.core.api.experimental.events.EventsManager;
026import org.matsim.core.controler.events.StartupEvent;
027import org.matsim.core.controler.listener.StartupListener;
028import org.matsim.core.mobsim.framework.listeners.FixedOrderSimulationListener;
029import org.matsim.core.router.TripRouter;
030import org.matsim.core.router.costcalculators.TravelDisutilityFactory;
031import org.matsim.core.router.util.LeastCostPathCalculatorFactory;
032import org.matsim.core.router.util.TravelTime;
033import org.matsim.pt.router.TransitRouter;
034import org.matsim.withinday.mobsim.MobsimDataProvider;
035import org.matsim.withinday.mobsim.WithinDayEngine;
036import org.matsim.withinday.replanning.identifiers.tools.ActivityReplanningMap;
037import org.matsim.withinday.replanning.identifiers.tools.LinkReplanningMap;
038import org.matsim.withinday.trafficmonitoring.EarliestLinkExitTimeProvider;
039import org.matsim.withinday.trafficmonitoring.WithinDayTravelTime;
040
041import javax.inject.Inject;
042import javax.inject.Provider;
043import java.util.Collections;
044import java.util.HashMap;
045import java.util.Map;
046import java.util.Map.Entry;
047import java.util.Set;
048
049/**
050 * Attempt to realize functionality provided by WithinDayController by
051 * a ControlerListener (similar to what is done by the MultiModalControlerListener).
052 * 
053 * Note: this class has to be registered as Controller Listener!
054 * 
055 * @author cdobler
056 */
057public class WithinDayControlerListener implements StartupListener {
058
059        private static final Logger log = Logger.getLogger(WithinDayControlerListener.class);
060
061        private int numReplanningThreads = 0;
062        @Inject private WithinDayTravelTime WithinDayTravelTime;
063        @Inject private ActivityReplanningMap activityReplanningMap;
064        @Inject private LinkReplanningMap linkReplanningMap;
065        @Inject private MobsimDataProvider mobsimDataProvider;
066        @Inject private EarliestLinkExitTimeProvider earliestLinkExitTimeProvider;
067
068        @Inject private EventsManager eventsManager;
069        @Inject private Scenario scenario;
070
071        @Inject private WithinDayEngine withinDayEngine;
072        @Inject private FixedOrderSimulationListener fosl;
073        private final Map<String, TravelTime> multiModalTravelTimes = new HashMap<String, TravelTime>();
074
075        public void setNumberOfReplanningThreads(int threads) {
076                throw new RuntimeException(this.getClass().toString() + " configuration has already been locked!");
077        }
078
079        public void setTravelDisutilityFactory(TravelDisutilityFactory travelDisutilityFactory) {
080                throw new RuntimeException(this.getClass().toString() + " configuration has already been locked!");
081        }
082
083        public void setLeastCostPathCalculatorFactory(LeastCostPathCalculatorFactory leastCostPathCalculatorFactory) {
084                throw new RuntimeException(this.getClass().toString() + " configuration has already been locked!");
085        }
086
087        public void setTransitRouterFactory(Provider<TransitRouter> transitRouterFactory) {
088                throw new RuntimeException(this.getClass().toString() + " configuration has already been locked!");
089        }
090        
091        public void setModesAnalyzedByWithinDayTravelTime(Set<String> modes) {
092                throw new RuntimeException(this.getClass().toString() + " configuration has already been locked!");
093        }
094        
095        public Map<String, TravelTime> getMultiModalTravelTimes() {
096                return Collections.unmodifiableMap(this.multiModalTravelTimes);
097        }
098        
099        public void addMultiModalTravelTimes(Map<String, TravelTime> multiModalTravelTimes) {
100                for (Entry<String, TravelTime> entry : multiModalTravelTimes.entrySet()) {
101                        this.addMultiModalTravelTime(entry.getKey(), entry.getValue());
102                }
103        }
104        
105        public void addMultiModalTravelTime(String mode, TravelTime travelTime) {
106                throw new RuntimeException(this.getClass().toString() + " configuration has already been locked!");
107        }
108
109        public WithinDayTravelTime getWithinDayTravelTime() {
110                return this.WithinDayTravelTime;
111        }
112
113        public EarliestLinkExitTimeProvider getEarliestLinkExitTimeProvider() {
114                return this.earliestLinkExitTimeProvider;
115        }
116
117        public void setWithinDayTripRouterFactory(Provider<TripRouter> tripRouterFactory) {
118                throw new RuntimeException(this.getClass().toString() + " configuration has already been locked!");
119        }
120
121        public FixedOrderSimulationListener getFixedOrderSimulationListener() {
122                return this.fosl;
123        }
124        /*
125         * ===================================================================
126         */
127
128        /*
129         * When the Controller Startup Event is created, the EventsManager
130         * has already been initialized. Therefore we can initialize now
131         * all Objects, that have to be registered at the EventsManager.
132         */
133        @Override
134        public void notifyStartup(StartupEvent event) {
135                this.initWithinDayEngine(this.numReplanningThreads);
136                this.createAndInitMobsimDataProvider();
137                this.createAndInitActivityReplanningMap();
138                this.createAndInitLinkReplanningMap();
139        }
140
141        /*
142         * ===================================================================
143         * creation and initialization methods
144         * ===================================================================
145         */
146        
147        /*
148         * TODO: Add a Within-Day Group to the Config. Then this method
149         * can be called on startup.
150         */
151        private void initWithinDayEngine(int numOfThreads) {
152                log.info("Initialize WithinDayEngine");
153        }
154
155        private void createAndInitMobsimDataProvider() {
156                this.getFixedOrderSimulationListener().addSimulationListener(this.mobsimDataProvider);
157        }
158
159        private void createAndInitActivityReplanningMap() {
160                fosl.addSimulationListener(activityReplanningMap);
161        }
162        
163        private void createAndInitLinkReplanningMap() {
164                this.fosl.addSimulationListener(linkReplanningMap);
165        }
166
167}
