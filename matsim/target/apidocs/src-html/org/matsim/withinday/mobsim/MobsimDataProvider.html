086        private QSim qSim;
087        
088        @Inject MobsimDataProvider(){}
089        
090        @Override
091        public final void notifyMobsimInitialized(MobsimInitializedEvent e) {
092                qSim = (QSim) e.getQueueSimulation();
093        }
094
095        public final Map<Id<Person>, MobsimAgent> getAgents() {
096                return this.qSim.getAgents() ;
097        }
098        
099        public final MobsimAgent getAgent(Id<Person> agentId) {
100                return this.getAgents().get(agentId);
101        }
102        
103        public final Map<Id<Vehicle>, MobsimVehicle> getVehicles() {
104                return this.qSim.getVehicles() ;
105        }
106        
107        public final MobsimVehicle getVehicle(Id<Vehicle> vehicleId) {
108                return this.getVehicles().get(vehicleId);
109        }
110        
111        public final Collection<MobsimVehicle> getEnrouteVehiclesOnLink(Id<Link> linkId) {
112                return this.qSim.getNetsimNetwork().getNetsimLink(linkId).getAllNonParkedVehicles();
113        }
114        
115        public final MobsimVehicle getDriversVehicle(Id<Person> driverId) {
116                MobsimAgent mobsimAgent = this.getAgents().get(driverId);
117                if (mobsimAgent == null) return null;
118                
119                DriverAgent driver = (DriverAgent) mobsimAgent;
120                return driver.getVehicle();
121        }
122        
123        public final MobsimAgent getVehiclesDriver(Id<Vehicle> vehicleId) {
124                MobsimVehicle mobsimVehicle = this.getVehicles().get(vehicleId);
125                if (mobsimVehicle == null) return null;
126                else return mobsimVehicle.getDriver();
127        }
128}
