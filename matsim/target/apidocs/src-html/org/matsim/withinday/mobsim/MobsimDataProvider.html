083        private QSim qSim;
084
085        @Override
086        public final void notifyMobsimInitialized(MobsimInitializedEvent e) {
087                qSim = (QSim) e.getQueueSimulation();
088        }
089
090        public final Map<Id<Person>, MobsimAgent> getAgents() {
091                return this.qSim.getAgents() ;
092        }
093        
094        public final MobsimAgent getAgent(Id<Person> agentId) {
095                return this.getAgents().get(agentId);
096        }
097        
098        public final Map<Id<Vehicle>, MobsimVehicle> getVehicles() {
099                return this.qSim.getVehicles() ;
100        }
101        
102        public final MobsimVehicle getVehicle(Id<Vehicle> vehicleId) {
103                return this.getVehicles().get(vehicleId);
104        }
105        
106        public final Collection<MobsimVehicle> getEnrouteVehiclesOnLink(Id<Link> linkId) {
107                return this.qSim.getNetsimNetwork().getNetsimLink(linkId).getAllNonParkedVehicles();
108        }
109        
110        public final MobsimVehicle getDriversVehicle(Id<Person> driverId) {
111                MobsimAgent mobsimAgent = this.getAgents().get(driverId);
112                if (mobsimAgent == null) return null;
113                
114                DriverAgent driver = (DriverAgent) mobsimAgent;
115                return driver.getVehicle();
116        }
117        
118        public final MobsimAgent getVehiclesDriver(Id<Vehicle> vehicleId) {
119                MobsimVehicle mobsimVehicle = this.getVehicles().get(vehicleId);
120                if (mobsimVehicle == null) return null;
121                else return mobsimVehicle.getDriver();
122        }
123}
