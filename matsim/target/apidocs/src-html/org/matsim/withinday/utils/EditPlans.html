045        public boolean addActivityAtEnd(MobsimAgent agent, Activity activity, String routingMode) {
046                log.debug("entering addActivityAtEnd with routingMode=" + routingMode) ;
047
048                Plan plan = WithinDayAgentUtils.getModifiablePlan(agent);
049                List<PlanElement> planElements = plan.getPlanElements();
050                
051                boolean retVal1 = false;
052                
053                if (isAtRealActivity(agent)) {
054                        retVal1 = planElements.add(pf.createLeg(routingMode));
055                }
056                
057                final boolean retVal = planElements.add(activity);
058                // (need the terminating activity in order to find the current trip. kai, nov'17)
059                
060                if (!isAtRealActivity(agent)) {
061                        retVal1 = editTrips.replanCurrentTrip(agent,mobsim.getSimTimer().getTimeOfDay(),routingMode);
062                }
063                
064                
065                WithinDayAgentUtils.resetCaches(agent);
066                this.mobsim.rescheduleActivityEnd(agent);
067                return (retVal1 && retVal);
068        }
069        public PlanElement removeActivity(MobsimAgent agent, int index, String mode) {
070                Plan plan = WithinDayAgentUtils.getModifiablePlan(agent) ;
071                List<PlanElement> planElements = plan.getPlanElements() ;
072
073                checkIfNotInPastOrCurrent(agent, index);
074
075                final Trip tripBefore = TripStructureUtils.findTripEndingAtActivity( (Activity) planElements.get(index),plan,tripRouter.getStageActivityTypes() );
076                final Trip tripAfter = TripStructureUtils.findTripStartingAtActivity( (Activity)planElements.get(index),plan,tripRouter.getStageActivityTypes() );
077                if ( mode==null ) {
078                        final String mainModeBefore = tripRouter.getMainModeIdentifier().identifyMainMode( tripBefore.getTripElements() );
079                        final String mainModeAfter = tripRouter.getMainModeIdentifier().identifyMainMode( tripAfter.getTripElements() );
080                        if ( mainModeBefore.equals( mainModeAfter ) ) {
081                                mode = mainModeBefore ;
082                        } else {
083                                throw new ReplanningException("mode not given and mode before removed activity != mode after removed activity; don't know which mode to use") ;
084                        }
085                }
086                PlanElement pe = planElements.remove(index) ; 
087                if ( checkIfTripHasAlreadyStarted( agent, tripBefore.getTripElements() ) ) {
088                        editTrips.replanCurrentTrip(agent, mobsim.getSimTimer().getTimeOfDay() , mode);
089                } else {
090                        editTrips.insertEmptyTrip(plan, tripBefore.getOriginActivity(), tripAfter.getDestinationActivity(), mode ) ;
091                }
092                WithinDayAgentUtils.resetCaches(agent);
093                this.mobsim.rescheduleActivityEnd(agent);
094                return pe ;
095        }

116                }

146                        }

148                // trip after (if any):
149                if ( index < planElements.size()-1 ) {
150                        Trip tripAfterAct = TripStructureUtils.findTripStartingAtActivity(origAct,plan,tripRouter.getStageActivityTypes());
151                        Gbl.assertIf( tripAfterAct!=null ); // there could also just be a sequence of activities?!
152                        if ( downstreamMode==null ) {
153                                final String currentMainMode = this.tripRouter.getMainModeIdentifier().identifyMainMode( tripAfterAct.getTripElements() );
154                                EditTrips.insertEmptyTrip(plan, newAct, tripAfterAct.getDestinationActivity(), currentMainMode, pf);
155                        } else {
156                                EditTrips.insertEmptyTrip(plan, newAct, tripAfterAct.getDestinationActivity(), downstreamMode, pf);
157                        }
158                }
159                WithinDayAgentUtils.resetCaches(agent);
160                this.mobsim.rescheduleActivityEnd(agent);
161                return origAct ;
162        }
163        public void insertActivity(MobsimAgent agent, int index, Activity activity, String upstreamMode, String downstreamMode ) {
164                Plan plan = WithinDayAgentUtils.getModifiablePlan(agent) ;
165                List<PlanElement> planElements = plan.getPlanElements() ;
166
167                checkIfNotInPastOrCurrent(agent, index) ;
168                planElements.add( index, activity ) ;
169                {
170                        // activity before:
171                        Activity actBefore = findRealActBefore(agent, index);
172                        if ( actBefore != null ) {
173                                if ( EditPlans.indexOfPlanElement(agent, actBefore) < WithinDayAgentUtils.getCurrentPlanElementIndex(agent) ) {
174                                        // we are already under way
175                                        editTrips.replanCurrentTrip(agent, this.mobsim.getSimTimer().getTimeOfDay(), upstreamMode );
176                                } else {
177                                        // we are not yet under way; inserting empty trip:
178                                        EditTrips.insertEmptyTrip(plan, actBefore, activity, upstreamMode, pf ) ;
179                                }
180                        }
181                }
182                {
183                        // activity after:
184                        Activity actAfter = findRealActAfter(agent, index);
185                        if ( actAfter != null ) {
186                                EditTrips.insertEmptyTrip(plan, activity, actAfter, downstreamMode, pf ) ;
187                        }
188                }                       
189                WithinDayAgentUtils.resetCaches(agent);
190                this.mobsim.rescheduleActivityEnd(agent);
191        }

244        private void checkIfNotStageActivity(Activity origAct) {
245                if( this.tripRouter.getStageActivityTypes().isStageActivity(origAct.getType()) ){
246                        throw new ReplanningException("trying to replace a helper activity (stage activity) by a real activity; this is not possible") ;
247                }
248        }
249        private static boolean checkIfTripHasAlreadyStarted(MobsimAgent agent, final List<PlanElement> currentTripElements) {
250                return currentTripElements.contains( ((PlanAgent)agent).getCurrentPlanElement() ) ;
251        }
252        private static void checkIfNotInPastOrCurrent(MobsimAgent agent, int index) {
253                final Integer currentIndex = WithinDayAgentUtils.getCurrentPlanElementIndex(agent);
254
255                // make sure we are not yet beyond the to-be-replanned activity:
256                if ( index <= currentIndex ) {
257                        throw new ReplanningException("trying to replace an activity that lies in the past or is current; this is not possible") ;
258                }
259        }
260        private static void checkIfSameMode(String upstreamMode, final String currentMode) {
261                if ( upstreamMode!=null && !upstreamMode.equals(currentMode) ) {
262                        throw new ReplanningException( "cannot change mode in trip that has already started.  Don't set the mode in the request, "
263                                        + "or somehow make the agent to abort the current trip first." ) ;
264                }
265        }

280        public int findIndexOfRealActAfter(MobsimAgent agent, int index) {
281                Plan plan = WithinDayAgentUtils.getModifiablePlan(agent) ;
282                List<PlanElement> planElements = plan.getPlanElements() ;
283
284                int theIndex = -1 ;
285                for ( int ii=planElements.size()-1 ; ii>index; ii-- ) {
286                        if ( planElements.get(ii) instanceof Activity ) {
287                                Activity act = (Activity) planElements.get(ii) ;
288                                if ( !this.tripRouter.getStageActivityTypes().isStageActivity( act.getType() ) ) {
289                                        theIndex = ii ;
290                                }
291                        }
292                }
293                return theIndex ;
294        }
