063        /*
064         * Design thoughts while adding access/egress walk legs to car trips:
065         * . Matsim has a computer science router (LeastCostPathCalculator, from node to node) and a behavioral router (RoutingModule, from
066         *    facility to facility).
067         * . For something like replanCurrentRoute it does not really make sense to start at a facility.
068         * . So it seems that for replanCurrentRoute, the computer science router makes more sense.
069         * . This is, however, not true for all the relocate calls, since they also need to reroute the egress leg.
070         * yyyy So, in short: all the relocate methods need to be adapted.  For "future", this is easy.  For "current", the splicing needs to be re-done.  kai, dec'15
071         * 
072         * Hm, next thought: It is not fully clear how relocateXxx has worked in the past, since also there the succeeding activity would 
073         * have needed adaptation. So in the end the method should just do what it promises (i.e. modify a single leg), and not
074         * think too much about its context.  (In the end, will presumably retrofit the mobsim in a way that it is able to process also the
075         * old car trips.)
076         */

103         * Re-locates a future route. The route is given by its leg. It just replaces the route for the given leg, without looking after

111        public boolean relocateFutureLegRoute(Leg leg, Id<Link> fromLinkId, Id<Link> toLinkId, Person person ) {
112
113                Link fromLink = network.getLinks().get(fromLinkId);
114                Link toLink = network.getLinks().get(toLinkId);
115
116                Vehicle vehicle = null ;
117                Node startNode = fromLink.getToNode() ;
118                Node endNode = toLink.getFromNode() ;
119                double starttime = leg.getDepartureTime() ;
120                Path path = pathCalculator.calcLeastCostPath(startNode, endNode, starttime, person, vehicle) ;
121
122                if (path == null) throw new RuntimeException("No route found from node " + startNode.getId() + " to node " + endNode.getId() + ".");
123                NetworkRoute route = this.routeFactories.createRoute(NetworkRoute.class, fromLink.getId(), toLink.getId());
124                route.setLinkIds(fromLink.getId(), NetworkUtils.getLinkIds(path.links), toLink.getId());
125                route.setTravelTime((int) path.travelTime); // yyyy why int?  kai, dec'15
126                route.setTravelCost(path.travelCost);
127                route.setDistance(RouteUtils.calcDistance(route,1.,1., this.network));
128                leg.setRoute(route);
129                
130                return true;
131        }

134                Plan plan = WithinDayAgentUtils.getModifiablePlan(agent) ;
135                PlanElement pe = plan.getPlanElements().get( WithinDayAgentUtils.getCurrentPlanElementIndex(agent)) ;

140                this.replanCurrentLegRoute((Leg)pe, ((HasPerson)agent).getPerson(), currentLinkIndex, now ) ;
141                WithinDayAgentUtils.resetCaches(agent);

153        public static boolean relocateFutureLegRoute(Leg leg, Id<Link> fromLinkId, Id<Link> toLinkId, Person person, Network network, TripRouter tripRouter) {
154                // TripRouter variant; everything else uses the PathCalculator
155                
156                Link fromLink = network.getLinks().get(fromLinkId);
157                Link toLink = network.getLinks().get(toLinkId);
158
159                Facility<ActivityFacility> fromFacility = new LinkWrapperFacility(fromLink);
160                Facility<ActivityFacility> toFacility = new LinkWrapperFacility(toLink);
161
162                List<? extends PlanElement> planElements = tripRouter.calcRoute(leg.getMode(), fromFacility, toFacility, leg.getDepartureTime(), person);
163
164                if (planElements.size() != 1) {
165                        throw new RuntimeException("Expected a list of PlanElements containing exactly one element, " +
166                                        "but the returned list contained " + planElements.size() + " elements."); 
167                }
168
169                Leg newLeg = (Leg) planElements.get(0);
170
171                leg.setTravelTime(newLeg.getTravelTime());
172                leg.setRoute(newLeg.getRoute());
173
174                return true;
175        }

232         * Re-locates a current route. The route is given by its leg.  Does not look after plans consistency!

234         * @return true when replacing the route worked, false when something went wrong

239        public boolean relocateCurrentLegRoute(Leg leg, Person person, int currentLinkIndex, Id<Link> toLinkId, double time ) {
240
241                Route route = leg.getRoute();
242
243                // if the route type is not supported:
244                if (!(route instanceof NetworkRoute)) {
245                        log.warn( "route not instance of NetworkRoute");
246                        return false;
247                }
248
249                NetworkRoute oldRoute = (NetworkRoute) route;
250
251                /*
252                 *  Create a List that contains all links of a route, including the Start- and EndLinks.
253                 *  Get the Id of the current Link.
254                 */
255                List<Id<Link>> oldLinkIds = getRouteLinkIds(oldRoute);
256                Id<Link> currentLinkId = oldLinkIds.get(currentLinkIndex);
257
258                final Link startLink = network.getLinks().get(currentLinkId);
259                final Link endLink = network.getLinks().get(toLinkId);
260
261                Vehicle vehicle = null ;
262                Path path = this.pathCalculator.calcLeastCostPath(startLink.getToNode(), endLink.getFromNode(), time, person, vehicle) ;
263                
264                spliceNewPathIntoOldRoute(currentLinkIndex, toLinkId, oldRoute, NetworkUtils.getLinkIds(path.links), currentLinkId) ;
265
266                return true;
267        }

269        static void spliceNewPathIntoOldRoute(int currentLinkIndex, Id<Link> toLinkId, NetworkRoute oldRoute,
270                                                                                  List<Id<Link>> newLinksIds, Id<Link> currentLinkId) {
271                List<Id<Link>> oldLinkIds = oldRoute.getLinkIds();;
272                List<Id<Link>> resultingLinkIds = new ArrayList<>();
273                
274                // REMEMBER: the "currentLinkIndex" does not point to the current position, but one beyond. Probably
275                // because there is a starting link, which is not part of getLinkIds().
276
277                /*
278                 * Get those Links which have already been passed.
279                 * oldLinkIds contains also the startLinkId, which should not
280                 * be part of the List - it is set separately. Therefore we start
281                 * at index 1.  CD, 201X
282                 * --> I cannot confirm this from the code.  Maybe it was like that in earlier times?  kai, feb'18
283                 */
284                if (currentLinkIndex > 0) {
285//                      log.warn("oldLinkIds.size = " + oldLinkIds.size() + "; currentLinkIndex = " + currentLinkIndex ) ;
286//                      resultingLinkIds.addAll(oldLinkIds.subList(1, currentLinkIndex + 1));
287                        resultingLinkIds.addAll(oldLinkIds.subList(0, currentLinkIndex-1));
288                        // (1) I cannot confirm the "starting at 1" from the code, as stated above.  Maybe it was like that in earlier code.
289                        // (2) [0,current-1[ now means [0,current-2].  Since "current-1" is the current link, current-2 is one before.
290                        // This will then be compensated below.
291                }
292                if ( !oldRoute.getStartLinkId().equals( currentLinkId ) ) {
293                        // (this happens if the agent is still on the departure link: that is not part of getLinkIds().  Otherwise:   )
294                        resultingLinkIds.add(currentLinkId);
295                }
296                
297                // Merge old and new Route.
298                /*
299                 * Edit cdobler 25.5.2010
300                 * If the new leg ends at the current Link, we have to
301                 * remove that linkId from the linkIds List - it is stored
302                 * in the endLinkId field of the route.
303                 * --> This is essentially the addition of the "currentLinkId" above, which
304                 * could now just be suppressed in this situation.  kai, feb'18
305                 * --> I just tried that, but at least the obvious approach does not pass the tests. kai, feb'18
306                 */
307//              if (newLinkIds.size() > 0 && path.links.size()>0 && newLinkIds.get(newLinkIds.size() - 1).equals( path.links.get( path.links.size()-1 ) ) ) {
308                if (resultingLinkIds.size() > 0 && newLinksIds.size()>0
309                                        && resultingLinkIds.get(resultingLinkIds.size() - 1).equals( newLinksIds.get( newLinksIds.size()-1 ) )
310                                ) {
311                        resultingLinkIds.remove( resultingLinkIds.size()-1 );
312                }
313                
314                resultingLinkIds.addAll( newLinksIds ) ;
315                
316                StringBuilder strb = new StringBuilder() ;
317                for ( int ii= Math.max(0,currentLinkIndex-2) ; ii < Math.min( resultingLinkIds.size(), currentLinkIndex+3) ; ii++ ) {
318                        strb.append("-") ;
319                        strb.append( resultingLinkIds.get(ii) ) ;
320                        strb.append("-") ;
321                }
322                log.info( "linkIds at join: " + strb ) ;
323                
324                // Overwrite old Route
325                oldRoute.setLinkIds(oldRoute.getStartLinkId(), resultingLinkIds, toLinkId );
326        }

385        private static List<Id<Link>> getRouteLinkIds(Route route) {
386                List<Id<Link>> linkIds = new ArrayList<>();
387
388                if (route instanceof NetworkRoute) {
389                        NetworkRoute networkRoute = (NetworkRoute) route;
390                        linkIds.add(networkRoute.getStartLinkId());
391                        linkIds.addAll(networkRoute.getLinkIds());
392                        linkIds.add(networkRoute.getEndLinkId());
393                } else {
394                        throw new RuntimeException("Currently only NetworkRoutes are supported for Within-Day Replanning!");
395                }
396
397                return linkIds;
398        }
