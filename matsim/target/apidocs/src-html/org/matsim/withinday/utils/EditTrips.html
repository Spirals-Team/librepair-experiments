055        public Trip findTripAtPlanElement(MobsimAgent agent, PlanElement pe) {
056                log.debug("plan element to be found=" + pe ) ;
057                List<Trip> trips = TripStructureUtils.getTrips( WithinDayAgentUtils.getModifiablePlan(agent), tripRouter.getStageActivityTypes() ) ;
058                for ( Trip trip : trips ) {
059                        for ( PlanElement te : trip.getTripElements() ) {
060                                log.debug("trip element to be compared with=" + te ) ;
061                                if ( te==pe ) {
062                                        log.debug("found trip element") ;
063                                        return trip;
064                                }
065                        }
066                }
067                throw new ReplanningException("trip not found") ;
068        }

072        // current trip:
073        public final boolean replanCurrentTrip(MobsimAgent agent, double now, String routingMode ) {
074                log.debug("entering replanCurrentTrip with routingMode=" + routingMode) ;
075
076                // I will treat that in the way that it will make the trip consistent with the activities.  So if the activity in the
077                // plan has changed, the trip will go to a new location.
078                
079                // what matters is the external interface, which is the method call.  Everything below is internal so it does not matter so much.
080
081                Trip trip = findCurrentTrip( agent ) ;
082                final PlanElement currentPlanElement = WithinDayAgentUtils.getCurrentPlanElement(agent) ;
083                final List<PlanElement> tripElements = trip.getTripElements();
084                int tripElementsIndex = tripElements.indexOf( currentPlanElement ) ;
085//              final String currentMode = tripRouter.getMainModeIdentifier().identifyMainMode( tripElements ) ;
086
087                if ( currentPlanElement instanceof Activity ) {
088                        // we are on a stage activity.  Take it from there:
089                        replanCurrentTripFromStageActivity(tripElements, tripElementsIndex, agent, routingMode);
090                } else {
091                        // we are on a leg
092                        replanCurrentTripFromLeg(trip.getDestinationActivity(), currentPlanElement, routingMode, now, agent);
093                }
094                WithinDayAgentUtils.resetCaches(agent);
095                return true ;
096        }
097        private void replanCurrentTripFromLeg(Activity newAct, final PlanElement currentPlanElement, final String routingMode,
098                                                                                  double now, MobsimAgent agent) {
099                log.debug("entering replanCurrentTripFromLeg");
100                Leg currentLeg = (Leg) currentPlanElement ;
101                if ( currentLeg.getRoute() instanceof NetworkRoute ) {
102                        replanCurrentLegWithNetworkRoute(newAct, routingMode, currentLeg, now, agent);
103                } else {
104                        throw new ReplanningException("not implemented") ;
105                        // Does not feel so hard: 
106                        // * with teleported legs, push forward to next facility
107                        // * with passenger legs, push forward to next possibility of exit
108                }
109                WithinDayAgentUtils.resetCaches(agent);
110        }
111        private void replanCurrentLegWithNetworkRoute(Activity newAct, String mainMode, Leg currentLeg, double now, 
112                        MobsimAgent agent) {
113                log.debug("entering replanCurrentLegWithNetworkRoute") ;
114                
115                Plan plan = WithinDayAgentUtils.getModifiablePlan(agent) ;
116                List<PlanElement> planElements = plan.getPlanElements() ;
117                Person person = plan.getPerson() ;
118                
119                // (1) get new trip from current position to new activity:
120                List<? extends PlanElement> newTripElements = newTripToNewActivity(newAct, mainMode, now, agent, person, scenario);
121
122                // (2) prune the new trip up to the current leg:
123                pruneUpToCurrentLeg(currentLeg, newTripElements);
124
125                // (2) modify current route:
126                replaceRemainderOfCurrentRoute(currentLeg, newTripElements, agent);
127
128                // (3) remove remainder of old trip after current leg in plan:
129                int pos = WithinDayAgentUtils.getCurrentPlanElementIndex(agent) + 1 ;
130
131                while ( !planElements.get(pos).equals(newAct) ) {
132                        planElements.remove(pos) ;
133                }
134
135                // (4) insert new trip after current leg:
136                for ( int ijk = 1 ; ijk < newTripElements.size() ; ijk++ ) {
137                        planElements.add( pos, newTripElements.get(ijk) ) ;
138                }
139                WithinDayAgentUtils.resetCaches(agent);
140        }
141        private List<? extends PlanElement> newTripToNewActivity(Activity newAct, String mainMode, double now, 
142                        MobsimAgent agent, Person person, Scenario scenario) {
143                log.debug("entering newTripToNewActivity") ;
144                Link currentLink = scenario.getNetwork().getLinks().get( agent.getCurrentLinkId() ) ;
145                Facility<?> fromFacility = new LinkWrapperFacility( currentLink ) ; 
146                Facility<?> toFacility = scenario.getActivityFacilities().getFacilities().get( newAct.getFacilityId() ) ;
147                if ( toFacility == null ) {
148                        toFacility = new ActivityWrapperFacility( newAct ) ;
149                }
150                List<? extends PlanElement> newTrip = tripRouter.calcRoute(mainMode, fromFacility, toFacility, now, person) ;
151                return newTrip;
152        }

230        /**
231         * In contrast to the other replanFutureLegRoute(...) method, the leg at the given index is replaced
232         * by a new one. This is e.g. necessary when replacing a pt trip which might consists of multiple legs
233         * and pt_interaction activities.  
234         */
235        public static boolean replanFutureTrip(Trip trip, Plan plan, String routingMode, double departureTime, TripRouter tripRouter) {
236                Person person = plan.getPerson();
237
238                Facility<?> fromFacility = new ActivityWrapperFacility( trip.getOriginActivity() ) ;
239                Facility<?> toFacility = new ActivityWrapperFacility( trip.getDestinationActivity() ) ;
240
241                final List<? extends PlanElement> newTrip = tripRouter.calcRoute(routingMode, fromFacility, toFacility, departureTime, person);
242                
243                log.debug("new trip:" + newTrip ) ;
244                for ( PlanElement pe : newTrip ) {
245                        log.debug( pe ) ;
246                }
247
248                TripRouter.insertTrip(plan, trip.getOriginActivity(), newTrip, trip.getDestinationActivity());
249
250                return true;
251        }
