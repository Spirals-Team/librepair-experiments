021package org.matsim.withinday.replanning.identifiers.interfaces;
022
023import java.util.Collections;
024import java.util.LinkedHashSet;
025import java.util.Set;
026
027import org.matsim.api.core.v01.Id;
028import org.matsim.api.core.v01.population.Person;
029import org.matsim.core.mobsim.framework.MobsimAgent;
030
031/**
032 * Identifies agents that need a replanning of their scheduled plan.
033 * <p></p>
034 * Used to be called "Identifier" until approx. mar'15, but was renamed since "Identifier", albeit correct, conflicts with "Id"
035 * which also stands for "Identifier".
036 * 
037 * @author cdobler
038 */
039public abstract class AgentSelector {
040        
041        private AgentSelectorFactory identifierFactory;
042        private final Set<AgentFilter> agentFilters = new LinkedHashSet<AgentFilter>();
043        
044        public abstract Set<MobsimAgent> getAgentsToReplan(double time);
045        
046        public final void addAgentFilter(AgentFilter agentFilter) {
047                this.agentFilters.add(agentFilter);
048        }
049        
050        public final boolean removeAgentFilter(AgentFilter agentFilter) {
051                return this.agentFilters.remove(agentFilter);
052        }
053
054        public final Set<AgentFilter> getAgentFilters() {
055                return Collections.unmodifiableSet(agentFilters);
056        }
057        
058        public final void applyFilters(Set<Id<Person>> set, double time) {
059                for (AgentFilter agentFilter : agentFilters) agentFilter.applyAgentFilter(set, time);
060        }
061        
062        public final boolean applyFilters(Id<Person> id, double time) {
063                for (AgentFilter agentFilter : agentFilters) {
064                        if(!agentFilter.applyAgentFilter(id, time)) return false;
065                }
066                return true;
067        }
068        
069        public final void setAgentSelectorFactory(AgentSelectorFactory factory) {
070                this.identifierFactory = factory;
071        }
072        
073        public final AgentSelectorFactory getAgentSelectorFactory() {
074                return identifierFactory;
075        }
076}
