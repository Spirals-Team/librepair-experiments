021package org.matsim.withinday.replanning.identifiers;
022
023import java.util.Map;
024import java.util.Set;
025import java.util.TreeSet;
026
027import org.matsim.api.core.v01.Id;
028import org.matsim.api.core.v01.population.Person;
029import org.matsim.core.mobsim.framework.MobsimAgent;
030import org.matsim.withinday.mobsim.MobsimDataProvider;
031import org.matsim.withinday.replanning.identifiers.interfaces.DuringLegAgentSelector;
032import org.matsim.withinday.replanning.identifiers.tools.LinkReplanningMap;
033
034public class LegStartedIdentifier extends DuringLegAgentSelector {
035
036        private final LinkReplanningMap linkReplanningMap;
037        private final MobsimDataProvider mobsimDataProvider;
038        
039        // use the Factory!
040        /*package*/ LegStartedIdentifier(LinkReplanningMap linkReplanningMap, MobsimDataProvider mobsimDataProvider) {
041                this.linkReplanningMap = linkReplanningMap;
042                this.mobsimDataProvider = mobsimDataProvider;
043        }
044        
045        @Override
046        public Set<MobsimAgent> getAgentsToReplan(double time) {                
047                Map<Id<Person>, MobsimAgent> mapping = this.mobsimDataProvider.getAgents();
048                Set<MobsimAgent> agentsToReplan = new TreeSet<MobsimAgent>(new ById());
049
050                /*
051                 * Identify those leg performing agents that should be replanned.
052                 * Add them to a set of MobsimAgents.
053                 */
054                for (Id<Person> agentId : this.linkReplanningMap.getLegStartedAgents()) {
055                        if (this.applyFilters(agentId, time)) agentsToReplan.add(mapping.get(agentId));
056                }
057                
058                return agentsToReplan;
059        }
060
061}
