037/**
038 * The NextLegReplanner can be used while an agent is performing an activity. The
039 * replanner creates a new trip from the current activity to the next main activity 
040 * in the agent's plan.
041 * 
042 * In fact this should be renamed to NextTripReplanner. cdobler, apr'14
043 */
044public class NextLegReplanner extends WithinDayDuringActivityReplanner {
045
046        private final TripRouter tripRouter;
047        
048        /*package*/ NextLegReplanner(Id<WithinDayReplanner> id, Scenario scenario, ActivityEndRescheduler internalInterface, TripRouter tripRouter) {
049                super(id, scenario, internalInterface);
050                this.tripRouter = tripRouter;
051        }
052
053        @Override
054        public boolean doReplanning(MobsimAgent withinDayAgent) {
055                
056                Plan executedPlan = WithinDayAgentUtils.getModifiablePlan(withinDayAgent);
057
058                // If we don't have an executed plan
059                if (executedPlan == null) return false;
060
061                // Get the activity currently performed by the agent as well as the subsequent trip.
062                Activity currentActivity = (Activity) WithinDayAgentUtils.getCurrentPlanElement(withinDayAgent);
063                Trip trip = TripStructureUtils.findTripStartingAtActivity(currentActivity, executedPlan, this.tripRouter.getStageActivityTypes() );
064
065                // If there is no trip after the activity.
066                if (trip == null) return false;
067                
068                String mainMode = this.tripRouter.getMainModeIdentifier().identifyMainMode(trip.getTripElements());
069                double departureTime = EditTrips.getDepartureTime(trip);
070                new EditTrips(this.tripRouter, scenario).replanFutureTrip(trip, executedPlan, mainMode, departureTime);
071                
072                return true;
073        }
074
075}
