021package org.matsim.withinday.replanning.identifiers.tools;
022
023import java.util.Collections;
024import java.util.HashSet;
025import java.util.Map.Entry;
026import java.util.Set;
027
028import org.apache.log4j.Logger;
029import org.matsim.api.core.v01.Id;
030import org.matsim.api.core.v01.events.ActivityEndEvent;
031import org.matsim.api.core.v01.events.ActivityStartEvent;
032import org.matsim.api.core.v01.events.PersonStuckEvent;
033import org.matsim.api.core.v01.events.handler.ActivityEndEventHandler;
034import org.matsim.api.core.v01.events.handler.ActivityStartEventHandler;
035import org.matsim.api.core.v01.events.handler.PersonStuckEventHandler;
036import org.matsim.api.core.v01.population.Person;
037import org.matsim.core.api.experimental.events.EventsManager;
038import org.matsim.core.mobsim.framework.events.MobsimAfterSimStepEvent;
039import org.matsim.core.mobsim.framework.listeners.MobsimAfterSimStepListener;
040import org.matsim.withinday.trafficmonitoring.EarliestLinkExitTimeProvider;
041
042import javax.inject.Inject;
043
044/**
045 * This Module is used by a CurrentLegReplanner. It calculates the time
046 * when an agent should do CurrentLegReplanning.
047 * <p></p>
048 * The time is estimated as following:
049 * When a LinkEnterEvent is thrown the Replanning Time is set to
050 * the current time + the FreeSpeed Travel Time. This guarantees that
051 * the replanning will be done while the agent is on the Link. After that
052 * time, the agent might be already in the outgoing queue of a QLink
053 * where not all replanning operations are possible anymore (the agent
054 * can e.g. not insert an Activity on its current link anymore).
055 * <p></p>
056 * <p>
057 * The replanning interval (multiple replannings on the same link when
058 * an agent is stuck on a link due to a traffic jam) has been removed
059 * since it cannot be guaranteed that all replanning operations are
060 * valid anymore.
061 * </p>
062 * 
063 * @author cdobler
064 */
065public class LinkReplanningMap implements PersonStuckEventHandler, ActivityStartEventHandler, ActivityEndEventHandler, 
066                MobsimAfterSimStepListener {
067
068        private static final Logger log = Logger.getLogger(LinkReplanningMap.class);
069
070        private final EarliestLinkExitTimeProvider earliestLinkExitTimeProvider;
071
072        /*
073         * EXACT... replanning is scheduled for the current time step (time == replanning time)
074         * RESTRICTED ... available replanning operations are restricted (time > replanning time)
075         * UNRESTRICTED ... replanning operations are not restricted (time <= replanning time)
076         */
077        private enum TimeFilterMode {
078                EXACT, RESTRICTED, UNRESTRICTED
079        }
080
081        private final Set<Id<Person>> legJustStartedAgents;
082        private double currentTime = 0.0;
083
084        @Inject
085        public LinkReplanningMap(EarliestLinkExitTimeProvider earliestLinkExitTimeProvider, EventsManager eventsManager) {
086
087                eventsManager.addHandler(this);
088                log.info("Note that the LinkReplanningMap has to be registered as an EventHandler and a SimulationListener!");
089                this. earliestLinkExitTimeProvider = earliestLinkExitTimeProvider;
090                
091                this.legJustStartedAgents = new HashSet<Id<Person>>();
092        }
093
094        @Override
095        public void handleEvent(PersonStuckEvent event) {
096                this.legJustStartedAgents.remove(event.getPersonId());
097        }
098        
099        @Override
100        public void handleEvent(ActivityEndEvent event) {
101                checkTime(event.getTime());
102                this.legJustStartedAgents.add(event.getPersonId());     
103        }
104
105        @Override
106        public void handleEvent(ActivityStartEvent event) {
107                this.legJustStartedAgents.remove(event.getPersonId());
108        }
109        
110        @Override
111        public void notifyMobsimAfterSimStep(MobsimAfterSimStepEvent e) {
112                checkTime(e.getSimulationTime());
113        }
114        
115        /*
116         * If time > currentTime, then legJustStartedAgents is not up to date
117         * anymore, therefore we clear all entries and update currentTime.
118         */
119        private void checkTime(double time) {
120                if (time > currentTime) {
121                        this.currentTime = time;
122                        this.legJustStartedAgents.clear();
123                }
124        }
125        
126        @Override
127        public void reset(int iteration) {
128                currentTime = 0.0;
129                this.legJustStartedAgents.clear();
130        }
131        
132        /**
133         * @param time
134         * @return a list of agents who might need a replanning
135         */
136        public Set<Id<Person>> getReplanningAgents(final double time) {
137                
138                Set<Id<Person>> set = this.earliestLinkExitTimeProvider.getEarliestLinkExitTimesPerTimeStep(time);
139                if (set != null) return Collections.unmodifiableSet(set);
140                else return new HashSet<>();
141        }
142
143        /**
144         * @param time
145         * @return a list of agents who might need an unrestricted replanning and use the given transport mode
146         */
147        public Set<Id<Person>> getUnrestrictedReplanningAgents(final double time) {
148                return this.filterAgents(time, TimeFilterMode.UNRESTRICTED);
149        }
150        
151        /**
152         * @param time
153         * @return a list of agents who might need a restricted replanning and use the given transport mode
154         */
155        public Set<Id<Person>> getRestrictedReplanningAgents(final double time) {
156                return this.filterAgents(time, TimeFilterMode.RESTRICTED);
157        }
158        
159        private Set<Id<Person>> filterAgents(final double time, final TimeFilterMode timeMode) {
160                
161                Set<Id<Person>> set = new HashSet<>();
162                
163                Set<Entry<Double, Set<Id<Person>>>> entries = this.earliestLinkExitTimeProvider.getEarliestLinkExitTimesPerTimeStep().entrySet();
164                
165                for (Entry<Double, Set<Id<Person>>> entry : entries) {
166                        double earliestLinkExitTime = entry.getKey();
167                        
168                        // check time
169                        if (timeMode == TimeFilterMode.RESTRICTED) {
170                                if (time <= earliestLinkExitTime) continue;
171                        } else if (timeMode == TimeFilterMode.UNRESTRICTED) {
172                                if (time > earliestLinkExitTime) continue;
173                        } else {
174                                throw new RuntimeException("Unexpected TimeFilterMode was found: " + timeMode.toString());
175                        }
176                        
177                        // non of the checks fails therefore add agents to the set
178                        set.addAll(entry.getValue());
179                }
180
181                return set;
182        }
183        
184        /**
185         * @return A list of all agents that are currently performing a leg. Note that
186         * some of them might be limited in the available replanning operations! 
187         */
188        public Set<Id<Person>> getLegPerformingAgents() {
189                return Collections.unmodifiableSet(this.earliestLinkExitTimeProvider.getEarliestLinkExitTimes().keySet());
190        }
191
192        /**
193         * @return A list of all agents that have just started a leg. Note that
194         * they cannot end their leg on their current link!
195         */
196        public Set<Id<Person>> getLegStartedAgents() {
197                return Collections.unmodifiableSet(this.legJustStartedAgents);
198        }
199}
