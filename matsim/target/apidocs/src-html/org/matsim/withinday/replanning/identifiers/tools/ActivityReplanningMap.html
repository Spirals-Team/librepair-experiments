045import javax.inject.Inject;
046import java.util.*;
047import java.util.concurrent.ConcurrentHashMap;
048
049/**
050 * This class tracks agents and their activity end times. It can be used to identify
051 * those agents which are going to end their activities in the current time step.
052 * Doing so allows one to e.g. extend their activities. Moreover, it can return a set
053 * containing all agents currently performing activites. 
054 */
055public class ActivityReplanningMap implements PersonStuckEventHandler,
056                ActivityStartEventHandler, ActivityEndEventHandler, ReplanningEventHandler,
057                MobsimInitializedListener, MobsimAfterSimStepListener {
058
059        private static final Logger log = Logger.getLogger(ActivityReplanningMap.class);
060
061        private final MobsimDataProvider mobsimDataProvider;
062        
063        /*
064         * Agents that have started an activity in the current time step. We store the
065         * MobsimAgents and their Ids (which are contained in the activity start
066         * event). By doing so, the lookup Id -> MobsimAgent is performed by the
067         * EventHandler which is parallelized.
068         * However, this will result in few unnecessary lookups for agents who end their
069         * activity in the same time step as they start it. 
070         */
071        private final Map<Id<Person>, MobsimAgent> startingAgents;      // PersonId
072
073        /*
074         * Contains activity end times of agents that are currently performing an activity.
075         * This is required in case an agent changes its activity end time. To be able to
076         * remove the agent from the activityPerformingAgents we need to know its original
077         * activity end time.
078         */
079        private final Map<Id<Person>, Double> activityEndTimes; // scheduled activity end times
080        
081        /*
082         * Contains a map for each time bin (a bin equals a time step in the mobility simulation).
083         * The set contains all agents ending their activity in that time bin.
084         */
085        private final Map<Integer, Map<Id<Person>, MobsimAgent>> activityPerformingAgents;
086        
087        // package protected to be accessible for test case
088        /*package*/ double simStartTime = Time.UNDEFINED_TIME;
089        /*package*/ double timeStepSize = Time.UNDEFINED_TIME;
090
091        @Inject
092        public ActivityReplanningMap(MobsimDataProvider mobsimDataProvider, EventsManager eventsManager) {
093                eventsManager.addHandler(this);
094                log.info("Note that the ActivityReplanningMap has to be registered as an EventHandler and a SimulationListener!");
095
096                this.mobsimDataProvider = mobsimDataProvider;
097
098                this.startingAgents = new HashMap<>();
099                this.activityEndTimes = new HashMap<>();
100                
101                this.activityPerformingAgents = new ConcurrentHashMap<Integer, Map<Id<Person>, MobsimAgent>>();
102        }
103
104        /*
105         * When the simulation starts the agents are all performing an activity. We collect
106         * them and check their activity end time.
107         */
108        @Override
109        public void notifyMobsimInitialized(MobsimInitializedEvent e) {
110
111                MobsimTimer mobsimTimer = ((QSim) e.getQueueSimulation()).getSimTimer();
112                this.simStartTime = mobsimTimer.getSimStartTime();
113                this.timeStepSize = mobsimTimer.getSimTimestepSize();
114
115                this.activityPerformingAgents.clear();
116                
117                for (MobsimAgent mobsimAgent : this.mobsimDataProvider.getAgents().values()) {
118                        
119                        // get the agent's activity end time and mark it as currently performing an Activity
120                        double activityEndTime = mobsimAgent.getActivityEndTime();
121                        
122                        // add the agent to the collections
123                        this.activityEndTimes.put(mobsimAgent.getId(), activityEndTime);
124                        
125                        int bin = this.getTimeBin(activityEndTime);
126                        Map<Id<Person>, MobsimAgent> map = getMapForTimeBin(bin);
127                        map.put(mobsimAgent.getId(), mobsimAgent);
128                }
129                
130        }
131        
132        /*
133         * The Activity Start Events are thrown before the Activity Start has been handled
134         * by the Simulation. As a result the Departure Time is not set at that time.
135         * We have to wait until the SimStep has been fully simulated and retrieve
136         * the activity departure times then.
137         */
138        @Override
139        public void notifyMobsimAfterSimStep(MobsimAfterSimStepEvent e) {
140                
141                double now = e.getSimulationTime();
142                for (MobsimAgent mobsimAgent : startingAgents.values()) {
143
144                        double departureTime = mobsimAgent.getActivityEndTime();
145
146                        /*
147                         * If it is the last scheduled Activity the departureTime is -infinity.
148                         * Otherwise we select the agent for a replanning.
149                         */
150                        if (departureTime >= now) {
151                                this.activityEndTimes.put(mobsimAgent.getId(), departureTime);
152                                int bin = this.getTimeBin(mobsimAgent.getActivityEndTime());
153                                Map<Id<Person>, MobsimAgent> map = getMapForTimeBin(bin);
154                                map.put(mobsimAgent.getId(), mobsimAgent);
155                        } else {
156                                log.warn("Departure time is in the past - ignoring activity!");
157                        }
158                }
159                this.startingAgents.clear();
160                
161                /*
162                 * Remove current time bin from activityPerformingAgents map. They have been handled
163                 * in the current time step.
164                 */
165                this.activityPerformingAgents.remove(this.getTimeBin(now));
166        }
167
168        /*
169         * We collect departures in time bins. Each bin contains all departures of one
170         * time step of the mobility simulation.
171         * 
172         * Method is package protected to be accessible for test case.
173         */
174        /*package*/ int getTimeBin(double time) {
175                
176                double timeAfterSimStart = time - simStartTime;
177                
178                /*
179                 * Agents who end their first activity before the simulation has started
180                 * will depart in the first time step. 
181                 */
182                if (timeAfterSimStart <= 0.0) return 0;
183
184                /*
185                 * Calculate the bin for the given time. Increase it by one if the result
186                 * of the modulo operation is > 0. If it is 0, it is the last time value
187                 * which is part of the previous bin.
188                 */
189                int bin = (int) (timeAfterSimStart / timeStepSize);
190                if (timeAfterSimStart % timeStepSize != 0.0) bin++;
191                
192                return bin;
193        }
194        
195        private Map<Id<Person>, MobsimAgent> getMapForTimeBin(int bin) {
196                Map<Id<Person>, MobsimAgent> map = this.activityPerformingAgents.get(bin);
197                if (map == null) {
198                        map = new HashMap<>();
199                        this.activityPerformingAgents.put(bin, map);
200                }
201                return map;
202        }
203        
204        /*
205         * Collect the agents that are starting an Activity in the current time step.
206         * Do the agent lookup here since this can be executed parallel to the mobsim. 
207         */
208        @Override
209        public void handleEvent(ActivityStartEvent event) {
210                Id<Person> agentId = event.getPersonId();
211                this.startingAgents.put(agentId, this.mobsimDataProvider.getAgent(agentId));
212        }
213
214        /*
215         * Nothing to do here?
216         * Agents should be removed from the map if their replanning time has come.
217         */
218        @Override
219        public void handleEvent(ActivityEndEvent event) {
220                Id<Person> agentId = event.getPersonId();
221                this.startingAgents.remove(agentId);
222                
223                Double activityEndTime = this.activityEndTimes.remove(agentId);
224                if (activityEndTime != null) {
225                        Map<Id<Person>, MobsimAgent> map;
226                        // remove
227                        map = this.getMapForTimeBin(this.getTimeBin(activityEndTime));
228                        map.remove(agentId);
229                }
230        }
231
232        /*
233         * Nothing to do here?
234         * Agents should not be removed from the simulation if they are performing an activity.
235         */
236        @Override
237        public void handleEvent(PersonStuckEvent event) {
238        }
239        
240        @Override
241        public void handleEvent(ReplanningEvent event) {
242                
243                // check whether the agent is performing an activity
244                Double activityEndTime = this.activityEndTimes.get(event.getPersonId());
245                if (activityEndTime != null) {
246                        
247                        // check whether the agent has changed its planned departure time
248                        MobsimAgent agent = this.mobsimDataProvider.getAgent(event.getPersonId());
249                        if (activityEndTime != agent.getActivityEndTime()) {
250                                // Update the agent's activity end time.
251                                this.activityEndTimes.put(agent.getId(), agent.getActivityEndTime());
252                                
253                                // Update the activity performing agents map. To do so, remove old entry and add new one.
254                                Map<Id<Person>, MobsimAgent> map;                               
255                                // remove
256                                map = this.getMapForTimeBin(this.getTimeBin(activityEndTime));
257                                map.remove(agent.getId());
258                                
259                                // add
260                                map = this.getMapForTimeBin(this.getTimeBin(agent.getActivityEndTime()));
261                                map.put(agent.getId(), agent);
262                        }
263                }
264        }
265        
266        /**
267         * Returns a set containing the Ids of all agents that are currently performing an activity.
268         */
269        public Set<Id<Person>> getActivityPerformingAgents() {
270                return Collections.unmodifiableSet(this.activityEndTimes.keySet());
271        }
272        
273        /**
274         * Returns a Collection containing all agents that are going to end their activity in the 
275         * time step belongs to the given time. Typically, this is the current simulation time. 
276         * For times in the past an empty set is returned.
277         * Since in data structure in the background the MobsimAgents are available, we return them
278         * instead of only their Ids as the getActivityPerformingAgents() method does.
279         */
280        public Collection<MobsimAgent> getActivityEndingAgents(double time) {
281                return Collections.unmodifiableCollection(this.getMapForTimeBin(this.getTimeBin(time)).values());
282        }
283
284        @Override
285        public void reset(int iteration) {
286                this.startingAgents.clear();
287                this.activityEndTimes.clear();
288        }
289}
