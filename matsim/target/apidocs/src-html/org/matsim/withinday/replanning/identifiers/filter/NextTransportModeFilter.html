021package org.matsim.withinday.replanning.identifiers.filter;
022
023import java.util.Iterator;
024import java.util.List;
025import java.util.Map;
026import java.util.Set;
027
028import org.matsim.api.core.v01.Id;
029import org.matsim.api.core.v01.population.Leg;
030import org.matsim.api.core.v01.population.Person;
031import org.matsim.api.core.v01.population.PlanElement;
032import org.matsim.core.mobsim.framework.MobsimAgent;
033import org.matsim.core.mobsim.qsim.agents.WithinDayAgentUtils;
034import org.matsim.withinday.replanning.identifiers.interfaces.AgentFilter;
035
036/**
037 * Remove all agents from the set that...
038 * <ul>
039 *      <li>do not perform an activity.</li>
040 *      <li>do not use one of the modes included in the given set of modes for their next trip.</li>
041 * </ul>
042 * 
043 * @author cdobler
044 */
045public class NextTransportModeFilter implements AgentFilter {
046
047        private final Map<Id<Person>, MobsimAgent> agents;
048        private final Set<String> modes;
049        
050        // use the factory
051        /*package*/ NextTransportModeFilter(Map<Id<Person>, MobsimAgent> agents, Set<String> modes) {
052                this.agents = agents;
053                this.modes = modes;
054        }
055        
056        @Override
057        public void applyAgentFilter(Set<Id<Person>> set, double time) {
058                Iterator<Id<Person>> iter = set.iterator();
059                
060                while (iter.hasNext()) {
061                        Id<Person> id = iter.next();
062                        if (!this.applyAgentFilter(id, time)) iter.remove();
063                }
064        }
065
066        @Override
067        public boolean applyAgentFilter(Id<Person> id, double time) {
068                MobsimAgent agent = this.agents.get(id);
069                
070                if (!(agent.getState() == MobsimAgent.State.ACTIVITY)) return false;
071                
072                int planElementIndex = WithinDayAgentUtils.getCurrentPlanElementIndex(agent);
073                List<PlanElement> planElements = WithinDayAgentUtils.getModifiablePlan(agent).getPlanElements();
074                
075                List<PlanElement> subList = planElements.subList(planElementIndex, planElements.size());
076                Iterator<PlanElement> iter = subList.iterator();
077                while (iter.hasNext()) {
078                        PlanElement planElement = iter.next();
079                        if (planElement instanceof Leg) {
080                                Leg leg = (Leg) planElement;
081                                if (modes.contains(leg.getMode())) return true;
082                                else return false;      // mode not in set of valid modes
083                        }
084                }
085                // no next leg was found
086                return false;
087        }
088}
