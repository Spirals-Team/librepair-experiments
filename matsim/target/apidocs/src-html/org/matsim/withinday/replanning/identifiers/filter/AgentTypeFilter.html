021package org.matsim.withinday.replanning.identifiers.filter;
022
023import java.util.Iterator;
024import java.util.Map;
025import java.util.Set;
026
027import org.matsim.api.core.v01.Id;
028import org.matsim.api.core.v01.population.Person;
029import org.matsim.core.mobsim.framework.MobsimAgent;
030import org.matsim.withinday.replanning.identifiers.interfaces.AgentFilter;
031
032public class AgentTypeFilter implements AgentFilter {
033
034        private final Map<Id<Person>, MobsimAgent> agents;
035        private final Set<Class<?>> includedAgentTypes;
036        
037        // use the factory
038        /*package*/ AgentTypeFilter(Map<Id<Person>, MobsimAgent> agents, Set<Class<?>> includedAgentTypes) {
039                this.agents = agents;
040                this.includedAgentTypes = includedAgentTypes;
041        }
042        
043        @Override
044        public void applyAgentFilter(Set<Id<Person>> set, double time) {
045                Iterator<Id<Person>> iter = set.iterator();
046                
047                while (iter.hasNext()) {
048                        Id<Person> id = iter.next();
049                        if (!this.applyAgentFilter(id, time)) iter.remove();
050                }
051        }
052        
053        @Override
054        public boolean applyAgentFilter(Id<Person> id, double time) {
055                MobsimAgent agent = this.agents.get(id);
056                
057                if (!includedAgentTypes.contains(agent.getClass())) return false;
058                else return true;
059        }
060}
