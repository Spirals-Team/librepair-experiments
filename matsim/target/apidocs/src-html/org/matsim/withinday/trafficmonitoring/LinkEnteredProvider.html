021package org.matsim.withinday.trafficmonitoring;
022
023import java.util.Collections;
024import java.util.Map;
025import java.util.concurrent.ConcurrentHashMap;
026
027import org.matsim.api.core.v01.Id;
028import org.matsim.api.core.v01.events.LinkEnterEvent;
029import org.matsim.api.core.v01.events.PersonArrivalEvent;
030import org.matsim.api.core.v01.events.PersonStuckEvent;
031import org.matsim.api.core.v01.events.VehicleLeavesTrafficEvent;
032import org.matsim.api.core.v01.events.VehicleEntersTrafficEvent;
033import org.matsim.api.core.v01.events.handler.LinkEnterEventHandler;
034import org.matsim.api.core.v01.events.handler.PersonArrivalEventHandler;
035import org.matsim.api.core.v01.events.handler.PersonStuckEventHandler;
036import org.matsim.api.core.v01.events.handler.VehicleLeavesTrafficEventHandler;
037import org.matsim.api.core.v01.network.Link;
038import org.matsim.api.core.v01.events.handler.VehicleEntersTrafficEventHandler;
039import org.matsim.api.core.v01.population.Person;
040import org.matsim.core.events.algorithms.Vehicle2DriverEventHandler;
041import org.matsim.core.mobsim.framework.events.MobsimAfterSimStepEvent;
042import org.matsim.core.mobsim.framework.listeners.MobsimAfterSimStepListener;
043
044/**
045 * Returns all agents that have entered a new link in the last time step.
046 * Agents who just ended an activity are NOT included, since they...
047 * <ul>
048 *      <li>do not produce a link entered event.</li>
049 *      <li>are limited in their possible replanning operations.</li>
050 * </ul>
051 * 
052 * @author cdobler
053 */
054public class LinkEnteredProvider implements LinkEnterEventHandler, PersonArrivalEventHandler, PersonStuckEventHandler,
055                MobsimAfterSimStepListener, VehicleEntersTrafficEventHandler, VehicleLeavesTrafficEventHandler {
056
057        private Map<Id<Person>, Id<Link>> linkEnteredAgents = new ConcurrentHashMap<>();        // <agentId, linkId>
058        private Map<Id<Person>, Id<Link>> lastTimeStepLinkEnteredAgents = new ConcurrentHashMap<>();    // <agentId, linkId>
059        
060        private Vehicle2DriverEventHandler delegate = new Vehicle2DriverEventHandler();
061        
062        public Map<Id<Person>, Id<Link>> getLinkEnteredAgentsInLastTimeStep() {
063                return Collections.unmodifiableMap(this.lastTimeStepLinkEnteredAgents);
064        }
065        
066        @Override
067        public void reset(int iteration) {
068                this.linkEnteredAgents.clear();
069                this.lastTimeStepLinkEnteredAgents.clear();
070                delegate.reset(iteration);
071        }
072
073        @Override
074        public void handleEvent(PersonStuckEvent event) {
075                this.linkEnteredAgents.remove(event.getPersonId());
076        }
077
078        /*
079         * Not sure whether the QSim allows an agent to enter a link and start an activity 
080         * in the same time step. If not, this could be removed. 
081         */
082        @Override
083        public void handleEvent(PersonArrivalEvent event) {
084                this.linkEnteredAgents.remove(event.getPersonId());
085        }
086
087        @Override
088        public void handleEvent(LinkEnterEvent event) {
089                this.linkEnteredAgents.put(delegate.getDriverOfVehicle(event.getVehicleId()), event.getLinkId());
090        }
091
092        @Override
093        public void notifyMobsimAfterSimStep(MobsimAfterSimStepEvent e) {
094                this.lastTimeStepLinkEnteredAgents = linkEnteredAgents;
095                this.linkEnteredAgents = new ConcurrentHashMap<>();
096        }
097
098        @Override
099        public void handleEvent(VehicleLeavesTrafficEvent event) {
100                delegate.handleEvent(event);
101        }
102
103        @Override
104        public void handleEvent(VehicleEntersTrafficEvent event) {
105                delegate.handleEvent(event);
106        }
107
108}
