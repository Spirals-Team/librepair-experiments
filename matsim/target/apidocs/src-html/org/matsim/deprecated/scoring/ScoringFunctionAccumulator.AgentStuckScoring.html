032/**
033 * 
034 * This is a scoring function which is configurable with sum terms for Activities, Legs, money spent, and becoming stuck.
035 * You can create an instance of this class from your own ScoringFunctionFactory and add different sum terms to it,
036 * for example depending on a configuration option.
037 * 
038 * The sum terms are instances which implement BasicScoring as well as one of ActivityScoring, LegScoring, MoneyScoring or
039 * AgentStuckScoring. They are notified about these events, and after finish() is called on them, they must be able to answer
040 * getScore(). Instances of BasicScoring, like ScoringFunctions, must be able to onIterationStarts() themselves between iterations, though
041 * this is currently not used (fresh instances are created).
042 * 
043 * Note that the startActivity() and endActivity() as well as starLeg() and endLeg() are redundant. The start time as well as
044 * the end time is contained in the Activity and the Leg. This is for historical reasons, when the Activity and the Leg which 
045 * were passed were instances from the Plan.
046 * 
047 * It is by no means necessary to use this class for a ScoringFunction. You are encouraged to just implement ScoringFunction.
048 * 
049 * @see <a href="http://www.matsim.org/node/263">http://www.matsim.org/node/263</a>
050 * @author rashid_waraich
051 */
052@Deprecated // preferably use SumScoringFunction.  kai, oct'13
053public final class ScoringFunctionAccumulator implements ScoringFunction {
054        
055        @Deprecated // preferably use SumScoringFunction.  kai, oct'13
056        public interface BasicScoring {
057                @Deprecated // preferably use SumScoringFunction.  kai, oct'13
058                public void finish();
059                @Deprecated // preferably use SumScoringFunction.  kai, oct'13
060                public double getScore();
061                @Deprecated // preferably use SumScoringFunction.  kai, oct'13
062                public void reset();
063        }
064        
065        @Deprecated // preferably use SumScoringFunction.  kai, oct'13
066        public interface ActivityScoring extends BasicScoring {
067                @Deprecated // preferably use SumScoringFunction.  kai, oct'13
068                void startActivity(final double time, final Activity act);
069                @Deprecated // preferably use SumScoringFunction.  kai, oct'13
070                void endActivity(final double time, final Activity act);
071        }
072
073        @Deprecated // preferably use SumScoringFunction.  kai, oct'13
074        public interface LegScoring extends BasicScoring {
075                @Deprecated // preferably use SumScoringFunction.  kai, oct'13
076                void startLeg(final double time, final Leg leg);
077                @Deprecated // preferably use SumScoringFunction.  kai, oct'13
078                void endLeg(final double time);
079        }
080        
081        @Deprecated // preferably use SumScoringFunction.  kai, oct'13
082        public interface MoneyScoring extends BasicScoring {
083                @Deprecated // preferably use SumScoringFunction.  kai, oct'13
084                void addMoney(final double amount);
085        }
086
087        @Deprecated // preferably use SumScoringFunction.  kai, oct'13
088        public interface AgentStuckScoring extends BasicScoring {
089                @Deprecated // preferably use SumScoringFunction.  kai, oct'13
090                void agentStuck(final double time);
091        }
092        
093        @Deprecated // preferably use SumScoringFunction.  kai, oct'13
094        public interface ArbitraryEventScoring extends BasicScoring {
095                @Deprecated // preferably use SumScoringFunction.  kai, oct'13
096                void handleEvent( final Event event ) ;
097        }
098        
099        private static Logger log = Logger.getLogger(ScoringFunctionAccumulator.class);
100
101        private ArrayList<BasicScoring> basicScoringFunctions = new ArrayList<BasicScoring>();
102        private ArrayList<ActivityScoring> activityScoringFunctions = new ArrayList<ActivityScoring>();
103        private ArrayList<MoneyScoring> moneyScoringFunctions = new ArrayList<MoneyScoring>();
104        private ArrayList<LegScoring> legScoringFunctions = new ArrayList<LegScoring>();
105        private ArrayList<AgentStuckScoring> agentStuckScoringFunctions = new ArrayList<AgentStuckScoring>();
106        private ArrayList<ArbitraryEventScoring> arbtraryEventScoringFunctions = new ArrayList<ArbitraryEventScoring>() ;
107        
108        @Deprecated // preferably use SumScoringFunction.  kai, oct'13
109        public ScoringFunctionAccumulator() {} // empty constructor so I can find where this is called.  kai, dec'12
110
111        @Override
112        public final void handleActivity(Activity activity) {
113        if (activity.getStartTime() != Time.UNDEFINED_TIME) {
114            startActivity(activity.getStartTime(), activity);
115        }
116        if (activity.getEndTime() != Time.UNDEFINED_TIME) {
117            endActivity(activity.getEndTime(), activity);
118        }
119    }
120
121        @Override
122    public final void handleLeg(Leg leg) {
123        startLeg(leg.getDepartureTime(), leg);
124        endLeg(leg.getDepartureTime() + leg.getTravelTime());
125    }
126        
127        @Override
128        public void addMoney(double amount) {
129                for (MoneyScoring moneyScoringFunction : moneyScoringFunctions) {
130                        moneyScoringFunction.addMoney(amount);
131                }
132        }
133
134        @Override
135        public void agentStuck(double time) {
136                for (AgentStuckScoring agentStuckScoringFunction : agentStuckScoringFunctions) {
137                        agentStuckScoringFunction.agentStuck(time);
138                }
139        }
140
141        @Override
142        public void handleEvent(Event event) {
143                for ( ArbitraryEventScoring eventScoringFunction : this.arbtraryEventScoringFunctions ) {
144                        eventScoringFunction.handleEvent(event) ;
145                }
146        }
147
148        @Deprecated // use handleActivity instead
149        public void startActivity(double time, Activity act) {
150                for (ActivityScoring activityScoringFunction : activityScoringFunctions) {
151                        activityScoringFunction.startActivity(time, act);
152                }
153        }
154
155        @Deprecated // use handleActivity instead
156        public void endActivity(double time, Activity act) {
157                for (ActivityScoring activityScoringFunction : activityScoringFunctions) {
158                        activityScoringFunction.endActivity(time, act);
159                }
160        }
161        
162        @Deprecated // use handleLeg instead
163        public void startLeg(double time, Leg leg) {
164                for (LegScoring legScoringFunction : legScoringFunctions) {
165                        legScoringFunction.startLeg(time, leg);
166                }
167        }
168        
169        @Deprecated // use handleLeg instead
170        public void endLeg(double time) {
171                for (LegScoring legScoringFunction : legScoringFunctions) {
172                        legScoringFunction.endLeg(time);
173                }
174        }
175
176        @Override
177        public void finish() {
178                for (BasicScoring basicScoringFunction : basicScoringFunctions) {
179                        basicScoringFunction.finish();
180                }
181        }
182
183        /**
184         * Add the score of all functions.
185         */
186        @Override
187        public double getScore() {
188                double score = 0.0;
189                for (BasicScoring basicScoringFunction : basicScoringFunctions) {
190            double contribution = basicScoringFunction.getScore();
191                        if (log.isTraceEnabled()) {
192                                log.trace("Contribution of scoring function: " + basicScoringFunction.getClass().getName() + " is: " + contribution);
193                        }
194            score += contribution;
195                }
196                return score;
197        }
198
199        @Deprecated
200        /** 
201         * Just create a fresh instance instead!
202         */
203        public void reset() {
204                for (BasicScoring basicScoringFunction : basicScoringFunctions) {
205                        basicScoringFunction.reset();
206                }
207        }
208
209        @Deprecated
210        /**
211         * Excuse me? Can't you keep track of these from the outside?
212         * @return
213         */
214        public ArrayList<ActivityScoring> getActivityScoringFunctions() {
215                return activityScoringFunctions;
216        }
217
218        /**
219         * add the scoring function the list of functions, it implemented the
220         * interfaces.
221         * 
222         * @param scoringFunction
223         */
224        @Deprecated // preferably use SumScoringFunction.  kai, oct'13
225        public void addScoringFunction(BasicScoring scoringFunction) {
226                basicScoringFunctions.add(scoringFunction);
227
228                if (scoringFunction instanceof ActivityScoring) {
229                        activityScoringFunctions.add((ActivityScoring) scoringFunction);
230                }
231
232                if (scoringFunction instanceof AgentStuckScoring) {
233                        agentStuckScoringFunctions.add((AgentStuckScoring) scoringFunction);
234                }
235
236                if (scoringFunction instanceof LegScoring) {
237                        legScoringFunctions.add((LegScoring) scoringFunction);
238                }
239
240                if (scoringFunction instanceof MoneyScoring) {
241                        moneyScoringFunctions.add((MoneyScoring) scoringFunction);
242                }
243                
244                if (scoringFunction instanceof ArbitraryEventScoring ) {
245                        this.arbtraryEventScoringFunctions.add((ArbitraryEventScoring) scoringFunction) ;
246                }
247
248        }
