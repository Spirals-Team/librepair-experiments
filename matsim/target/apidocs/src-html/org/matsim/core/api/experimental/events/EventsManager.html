020package org.matsim.core.api.experimental.events;
021
022import org.matsim.api.core.v01.events.Event;
023import org.matsim.core.events.handler.EventHandler;
024import org.matsim.core.mobsim.framework.Steppable;
025
026/**
027 * This should be split into two interfaces:
028 * an API (add/removeHandler) and an SPI (Service Provider Interface)
029 */
030public interface EventsManager {
031
032        public void processEvent(final Event event);
033
034        public void addHandler(final EventHandler handler);
035        
036        public void removeHandler(final EventHandler handler);
037
038        public void resetHandlers(int iteration);
039        
040        /**
041         * Called before the first event is sent for processing. Allows to initialize internal
042         * data structures used to process events.
043         */
044        public void initProcessing();
045
046        /**
047         * Called by a {@link Steppable} Mobsim after each {@link doSimStep} call. Parallel implementations
048         * of an EventsManager can then ensure that all events of the sim step have been processed.
049         */
050        public void afterSimStep(double time);
051        
052        /**
053         * Called after the last event is sent for processing. The method must only return when all
054         * events are completely processing (in case they are not directly processed in
055         * {@link #processEvent(Event)}). Can be used to clean up internal data structures used
056         * to process events.
057         */
058        public void finishProcessing();
059
060}
