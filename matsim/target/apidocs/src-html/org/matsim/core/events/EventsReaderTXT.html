022import org.matsim.api.core.v01.Id;
023import org.matsim.api.core.v01.events.*;
024import org.matsim.api.core.v01.network.Link;

029import org.matsim.vehicles.Vehicle;
030
031import java.io.BufferedReader;
032import java.io.IOException;
033
034import static org.matsim.core.events.algorithms.EventWriterTXT.Number.*;
035import static org.matsim.core.events.algorithms.EventWriterTXT.Number.ActivityEnd;
036
037/**
038 * Created by laemmel on 16/11/15.
039 */
040public final class EventsReaderTXT {
041
042        private final EventsManager em;
043
044        public EventsReaderTXT(EventsManager em) {
045                this.em = em;
046        }
047
048        public void runEventsFile(String file) throws IOException {
049
050
051                BufferedReader br = IOUtils.getBufferedReader(file);
052                String l = br.readLine();
053                l = br.readLine();//skip first line
054                while (l != null) {
055                        String[] expl = StringUtils.explode(l, '\t');
056                        double time = Double.parseDouble(expl[0]);
057                        Id<Vehicle> vehicleId = Id.create(expl[1], Vehicle.class);
058                        Id<Link> linkId = Id.createLinkId(expl[3]);
059                        String type = expl[5];
060
061
062                        Event e = null;
063                        int typeIndex = Integer.parseInt(type);
064                        // (switch does not work with integer input. kai, mar'18)
065                        if (typeIndex == ActivityEnd.ordinal()) {
066                                e = new ActivityEndEvent(time, Id.createPersonId(vehicleId), linkId, null, null);
067                        } else if (typeIndex == PersonDeparture.ordinal()) {
068                                e = new PersonDepartureEvent(time, Id.createPersonId(vehicleId), linkId, "car");
069                        } else if (typeIndex == VehicleEntersTraffic.ordinal()) {
070                                e = new VehicleEntersTrafficEvent(time, Id.createPersonId(vehicleId), linkId, vehicleId, "car", 0);
071                        } else if (typeIndex == LinkLeave.ordinal()) {
072                                e = new LinkLeaveEvent(time, vehicleId, linkId);
073                        } else if (typeIndex == LinkEnter.ordinal()) {
074                                e = new LinkEnterEvent(time, vehicleId, linkId);
075                        } else if (typeIndex == PersonArrival.ordinal()) {
076                                e = new PersonArrivalEvent(time, Id.createPersonId(vehicleId), linkId, "car");
077                        } else if (typeIndex == ActivityStart.ordinal()) {
078                                e = new ActivityStartEvent(time, Id.createPersonId(vehicleId), linkId, null, null);
079                        } else {
080                                throw new RuntimeException("Unsupported event type:" + l);
081                        }
082                        this.em.processEvent(e);
083                        l = br.readLine();
084                }
085                br.close();
086        }
087}
