076        public SnapshotGenerator(final Network network, final double snapshotPeriod, final QSimConfigGroup config) {
077                this.network = network;
078                int initialCapacity = (int)(network.getLinks().size()*1.1);
079                this.eventLinks = new HashMap<>(initialCapacity, 0.95f);
080                this.linkList = new ArrayList<>(initialCapacity);
081                this.eventAgents = new HashMap<>(1000, 0.95f);
082                this.snapshotPeriod = snapshotPeriod;
083                this.capCorrectionFactor = config.getFlowCapFactor() / network.getCapacityPeriod();
084                this.storageCapFactor = config.getStorageCapFactor();
085                this.snapshotStyle = config.getSnapshotStyle();
086                
087                if (! Double.isNaN( config.getLinkWidthForVis() )){
088                        this.linkWidthCalculator.setLinkWidthForVis( config.getLinkWidthForVis() );
089                } 
090                if (! Double.isNaN(network.getEffectiveLaneWidth())){
091                        this.linkWidthCalculator.setLaneWidth(network.getEffectiveLaneWidth());
092                }
093
094                reset(-1);
095        }

137        @Override
138        public void handleEvent(final PersonStuckEvent event) {
139                testForSnapshot(event.getTime());
140                if (event.getLinkId() != null) { // link id is optional - agent can be teleporting or whatever.
141                        this.eventLinks.get(event.getLinkId()).stuck(getEventAgent(event.getPersonId(), event.getTime()));
142                }
143        }

145        @Override
146        public void reset(final int iteration) {
147                this.eventLinks.clear();
148                for (Link link : this.network.getLinks().values()) {
149                        final double effectiveCellSize;
150                        effectiveCellSize = this.network.getEffectiveCellSize();
151                        this.eventLinks.put(link.getId(), new EventLink(link, this.capCorrectionFactor, effectiveCellSize, this.storageCapFactor));
152                }
153                this.linkList.clear();
154                this.linkList.addAll(eventLinks.values());
155                this.eventAgents.clear();
156                this.lastSnapshotIndex = -1;
157                
158                delegate.reset(iteration);
159        }

161        private EventAgent getEventAgent(final Id<Person> id, double time) {
162                EventAgent agent = this.eventAgents.get(id);
163                if (agent == null) {
164                        agent = new EventAgent(id, time);
165                        this.eventAgents.put(id, agent);
166                }
167                agent.time = time;
168                return agent;
169        }
170
171        private void testForSnapshot(final double time) {
172                int snapshotIndex = (int) (time / this.snapshotPeriod);
173                if (this.lastSnapshotIndex == -1) {
174                        this.lastSnapshotIndex = snapshotIndex;
175                }
176                while (snapshotIndex > this.lastSnapshotIndex) {
177                        this.lastSnapshotIndex++;
178                        double snapshotTime = this.lastSnapshotIndex * this.snapshotPeriod;
179                        doSnapshot(snapshotTime);
180                }
181        }
182
183        private void doSnapshot(final double time) {
184                if (time >= skipUntil) {
185                        if (!this.snapshotWriters.isEmpty()) {
186                                Collection<AgentSnapshotInfo> positions = getVehiclePositions(time);
187                                for (SnapshotWriter writer : this.snapshotWriters) {
188                                        writer.beginSnapshot(time);
189                                        for (AgentSnapshotInfo position : positions) {
190                                                writer.addAgent(position);
191                                        }
192                                        writer.endSnapshot();
193                                }
194                        }
195                }
196        }
197
198        private Collection<AgentSnapshotInfo> getVehiclePositions(final double time) {
199                Collection<AgentSnapshotInfo> positions = new ArrayList<>();
200                if (this.snapshotStyle == SnapshotStyle.queue) {
201                        for (EventLink link : this.linkList) {
202                                link.getVehiclePositionsQueue(positions, time, this.snapshotInfoFactory);
203                        }
204                } else if (this.snapshotStyle == SnapshotStyle.equiDist) {
205                        for (EventLink link : this.linkList) {
206                                link.getVehiclePositionsEquil(positions, time, this.snapshotInfoFactory);
207                        }
208                } else {
209                        // log statement to clarify: why only two snapshot styles. Amit Mar'17
210                        log.warn("Cannot generate snapshots offline (e.g., from events) for "+this.snapshotStyle
211                                        + ". This snapshot style is supported during simulation only.");
212                        throw new RuntimeException("The snapshotStyle \"" + this.snapshotStyle + "\" is not supported.");
213                }
214                return positions;
215        }
216
217        public final void finish() {
218                for (SnapshotWriter writer : this.snapshotWriters) {
219                        writer.finish();
220                }
221        }

223        /**
224         *
225         * Allow this SnapshotGenerator to skip all snapshots up to, but not including, a give timestep.
226         * This is especially useful for interactive settings where a user may fast-forward.
227         * Snapshot generation is one of the most expensive parts of mobility simulations, so this saves a lot of time.
228         *
229         * @param when The earliest timestep at which the caller will be interested in snapshots again.
230         */
231        public final void skipUntil(final double when) {
232                this.skipUntil = when;
233        }

235        private static class EventLink {
236                private final Link link;
237                private final List<EventAgent> drivingQueue;
238                private final List<EventAgent> parkingQueue;
239                private final List<EventAgent> waitingQueue;
240                private final List<EventAgent> buffer;
241
242                private final double euklideanDist;
243                private final double freespeedTravelTime;
244                private final double spaceCap;
245                private final double timeCap;
246                private final double storageCapFactor;
247                private final double inverseTimeCap;
248
249                private final double effectiveCellSize;
250
251                private EventLink(final Link link2, final double capCorrectionFactor, final double effectiveCellSize, final double storageCapFactor) {
252                        this.link = link2;
253                        this.drivingQueue = new ArrayList<>();
254                        this.parkingQueue = new ArrayList<>();
255                        this.waitingQueue = new ArrayList<>();
256                        this.buffer = new ArrayList<>();
257                        this.euklideanDist = CoordUtils.calcEuclideanDistance(link2.getFromNode().getCoord(), link2.getToNode().getCoord());
258                        this.freespeedTravelTime = Math.ceil( this.link.getLength() / this.link.getFreespeed() ) + 1; 
259                        this.timeCap = this.link.getCapacity() * capCorrectionFactor;
260                        this.storageCapFactor = storageCapFactor;
261                        this.inverseTimeCap = 1.0 / this.timeCap;
262                        this.effectiveCellSize = effectiveCellSize;
263                        this.spaceCap = (this.link.getLength() * this.link.getNumberOfLanes()) / this.effectiveCellSize * storageCapFactor;
264                }
265
266                private void enter(final EventAgent agent) {
267                        if (agent.currentLink != null) {
268                                agent.currentLink.stuck(agent); // use stuck to remove it from wherever it is
269                        }
270                        agent.currentLink = this;
271                        this.drivingQueue.add(agent);
272                }
273
274                private void leave(final EventAgent agent) {
275                        this.drivingQueue.remove(agent);
276                        this.buffer.remove(agent);
277                        agent.currentLink = null;
278                }
279
280                private void arrival(final EventAgent agent) {
281                        this.buffer.remove(agent);
282                        this.drivingQueue.remove(agent);
283                        this.parkingQueue.add(agent);
284                }
285
286                private void departure(final EventAgent agent) {
287                        agent.currentLink = this;
288                        this.parkingQueue.remove(agent);
289                        this.waitingQueue.add(agent);
290                }
291
292                private void wait2link(final EventAgent agent) {
293                        this.waitingQueue.remove(agent);
294                        this.buffer.add(agent);
295                }
296
297                private void stuck(final EventAgent agent) {
298                        // vehicles can be anywhere when they get stuck
299                        this.drivingQueue.remove(agent);
300                        this.parkingQueue.remove(agent);
301                        this.waitingQueue.remove(agent);
302                        this.buffer.remove(agent);
303                        agent.currentLink = null;
304                }
305
306                /**
307                 * Calculates the positions of all vehicles on this link according to the queue-logic: Vehicles are placed on the link
308                 * according to the ratio between the free-travel time and the time the vehicles are already on the link. If they could
309                 * have left the link already (based on the time), the vehicles start to build a traffic-jam (queue) at the end of the link.
310                 *
311                 * @param positions A collection where the calculated positions can be stored.
312                 * @param time The current timestep
313                 */
314                private void getVehiclePositionsQueue(final Collection<AgentSnapshotInfo> positions, final double time, AgentSnapshotInfoFactory snapshotInfoFactory) {
315                        double queueEnd = this.link.getLength(); // the length of the queue jammed vehicles build at the end of the link
316                        double vehLen = Math.min(       // the length of a vehicle in visualization
317                                        this.euklideanDist / this.spaceCap, // all vehicles must have place on the link
318                                        this.effectiveCellSize / this.storageCapFactor); // a vehicle should not be larger than it's actual size
319
320                        // put all cars in the buffer one after the other
321                        for (EventAgent agent : this.buffer) {
322
323                                int lane = 1 + (agent.intId % NetworkUtils.getNumberOfLanesAsInt(time, this.link));
324
325                                int cmp = (int) (agent.time + this.freespeedTravelTime + this.inverseTimeCap + 2.0);
326                                double speed = (time > cmp) ? 0.0 : this.link.getFreespeed(time);
327                                agent.speed = speed;
328
329                                AgentSnapshotInfo position = snapshotInfoFactory.createAgentSnapshotInfo(agent.id, this.link, queueEnd/* + NetworkLayer.CELL_LENGTH*/, lane);
330                                position.setColorValueBetweenZeroAndOne( agent.speed) ;
331                                position.setAgentState(AgentSnapshotInfo.AgentState.PERSON_DRIVING_CAR);
332                                positions.add(position);
333                                queueEnd -= vehLen;
334                        }
335
336                        /* place other driving cars according the following rule:
337                         * - calculate the time how long the vehicle is on the link already
338                         * - calculate the position where the vehicle should be if it could drive with freespeed
339                         * - if the position is already within the congestion queue, add it to the queue with slow speed
340                         * - if the position is not within the queue, just place the car with free speed at that place
341                         */
342                        double lastDistance = Integer.MAX_VALUE;
343                        for (EventAgent agent : this.drivingQueue) {
344                                double travelTime = time - agent.time;
345                                double distanceOnLink = (this.freespeedTravelTime == 0.0 ? 0.0 : ((travelTime / this.freespeedTravelTime) * this.euklideanDist));
346                                if (distanceOnLink > queueEnd) { // vehicle is already in queue
347                                        distanceOnLink = queueEnd;
348                                        queueEnd -= vehLen;
349                                }
350                                if (distanceOnLink >= lastDistance) {
351                                        /* we have a queue, so it should not be possible that one vehicles overtakes another.
352                                         * additionally, if two vehicles entered at the same time, they would be drawn on top of each other.
353                                         * we don't allow this, so in this case we put one after the other. Theoretically, this could lead to
354                                         * vehicles placed at negative distance when a lot of vehicles all enter at the same time on an empty
355                                         * link. not sure what to do about this yet... just setting them to 0 currently.
356                                         */
357                                        distanceOnLink = lastDistance - vehLen;
358                                        if (distanceOnLink < 0) distanceOnLink = 0.0;
359                                }
360                                int cmp = (int) (agent.time + this.freespeedTravelTime + this.inverseTimeCap + 2.0);
361                                double speed = (time > cmp) ? 0.0 : this.link.getFreespeed(time);
362                                agent.speed = speed;
363                                int lane = 1 + (agent.intId % NetworkUtils.getNumberOfLanesAsInt(org.matsim.core.utils.misc.Time.UNDEFINED_TIME, this.link));
364                                AgentSnapshotInfo position = snapshotInfoFactory.createAgentSnapshotInfo(agent.id, this.link, distanceOnLink/* + NetworkLayer.CELL_LENGTH*/, lane);
365                                position.setColorValueBetweenZeroAndOne( agent.speed) ;
366                                position.setAgentState(AgentSnapshotInfo.AgentState.PERSON_DRIVING_CAR);
367                                positions.add(position);
368                                lastDistance = distanceOnLink;
369                        }
370
371                        /* Put the vehicles from the waiting list in positions.
372                         * Their actual position doesn't matter, so they are just placed
373                         * to the coordinates of the from node */
374                        int lane = NetworkUtils.getNumberOfLanesAsInt(org.matsim.core.utils.misc.Time.UNDEFINED_TIME, this.link) + 1; // place them next to the link
375                        for (EventAgent agent : this.waitingQueue) {
376                                AgentSnapshotInfo position = snapshotInfoFactory.createAgentSnapshotInfo(agent.id, this.link, this.effectiveCellSize, lane);
377                                position.setColorValueBetweenZeroAndOne( 0.0) ;
378                                position.setAgentState(AgentSnapshotInfo.AgentState.PERSON_AT_ACTIVITY);
379                                positions.add(position);
380                        }
381
382                        /* put the vehicles from the parking list in positions
383                         * their actual position doesn't matter, so they are just placed
384                         * to the coordinates of the from node */
385                        lane = NetworkUtils.getNumberOfLanesAsInt(org.matsim.core.utils.misc.Time.UNDEFINED_TIME, this.link) + 2; // place them next to the link
386                        for (EventAgent agent : this.parkingQueue) {
387                                AgentSnapshotInfo position = snapshotInfoFactory.createAgentSnapshotInfo(agent.id, this.link, this.effectiveCellSize, lane);
388                                position.setColorValueBetweenZeroAndOne(0.0) ;
389                                position.setAgentState(AgentSnapshotInfo.AgentState.PERSON_AT_ACTIVITY);
390                                positions.add(position);
391                        }
392                }
393
394                /**
395                 * Calculates the positions of all vehicles on this link so that there is always the same distance between following cars.
396                 * A single vehicle will be placed at the middle (0.5) of the link, two cars will be placed at positions 0.25 and 0.75,
397                 * three cars at positions 0.16, 0.50, 0.83, and so on.
398                 *
399                 * @param positions A collection where the calculated positions can be stored.
400                 * @param time The current timestep
401                 */
402                private void getVehiclePositionsEquil(final Collection<AgentSnapshotInfo> positions, final double time, AgentSnapshotInfoFactory snapshotInfoFactory) {
403                        int bufferSize = this.buffer.size();
404                        int drivingQueueSize = this.drivingQueue.size();
405                        int waitingQueueSize = this.waitingQueue.size();
406                        int parkingQueueSize = this.parkingQueue.size();
407                        if (bufferSize + drivingQueueSize + waitingQueueSize + parkingQueueSize > 0 ) {
408                                int cnt = bufferSize + drivingQueueSize;
409                                int nLanes = NetworkUtils.getNumberOfLanesAsInt(time, this.link);
410                                double linkLength = this.link.getLength();
411                                if (cnt > 0) {
412                                        double cellSize = linkLength / cnt;
413                                        double distFromFromNode = linkLength - cellSize / 2.0;
414                                        double freespeed = this.link.getFreespeed(time);
415
416                                        // the cars in the buffer
417                                        for (EventAgent agent : this.buffer) {
418                                                agent.lane = 1 + agent.intId % nLanes;
419                                                int cmp = (int) (agent.time + this.freespeedTravelTime + this.inverseTimeCap + 2.0);
420                                                if (time > cmp) {
421                                                        agent.speed = 0.0;
422                                                } else {
423                                                        agent.speed = freespeed;
424                                                }
425                                                AgentSnapshotInfo position = snapshotInfoFactory.createAgentSnapshotInfo(agent.id, this.link, distFromFromNode, agent.lane);
426                                                position.setColorValueBetweenZeroAndOne( agent.speed) ;
427                                                position.setAgentState(AgentSnapshotInfo.AgentState.PERSON_DRIVING_CAR);
428                                                positions.add(position);
429                                                distFromFromNode -= cellSize;
430                                        }
431
432                                        // the cars in the drivingQueue
433                                        for (EventAgent agent : this.drivingQueue) {
434                                                agent.lane = 1 + agent.intId % nLanes;
435                                                int cmp = (int) (agent.time + this.freespeedTravelTime + this.inverseTimeCap + 2.0);
436                                                if (time > cmp) {
437                                                        agent.speed = 0.0;
438                                                } else {
439                                                        agent.speed = freespeed;
440                                                }
441                                                AgentSnapshotInfo position = snapshotInfoFactory.createAgentSnapshotInfo(agent.id, this.link, distFromFromNode, agent.lane);
442                                                position.setColorValueBetweenZeroAndOne( agent.speed) ;
443                                                position.setAgentState(AgentSnapshotInfo.AgentState.PERSON_DRIVING_CAR);
444                                                positions.add(position);
445                                                distFromFromNode -= cellSize;
446                                        }
447                                }
448
449                                // the cars in the waitingQueue
450                                // the actual position doesn't matter, so they're just placed next to the link at the end
451                                if (waitingQueueSize > 0) {
452                                        int lane = nLanes + 2;
453                                        double cellSize = Math.min(this.effectiveCellSize, linkLength / waitingQueueSize);
454                                        double distFromFromNode = linkLength - cellSize / 2.0;
455                                        for (EventAgent agent : this.waitingQueue) {
456                                                agent.lane = lane;
457                                                agent.speed = 0.0;
458                                                AgentSnapshotInfo position = snapshotInfoFactory.createAgentSnapshotInfo(agent.id, this.link, distFromFromNode, agent.lane);
459                                                position.setColorValueBetweenZeroAndOne( agent.speed) ;
460                                                position.setAgentState(AgentSnapshotInfo.AgentState.PERSON_AT_ACTIVITY);
461                                                positions.add(position);
462                                                distFromFromNode -= cellSize;
463                                        }
464                                }
465
466                                // the cars in the parkingQueue
467                                // the actual position  doesn't matter, so they're distributed next to the link
468                                if (parkingQueueSize > 0) {
469                                        int lane = nLanes + 4;
470                                        double cellSize = linkLength / parkingQueueSize;
471                                        double distFromFromNode = linkLength - cellSize / 2.0;
472                                        for (EventAgent agent : this.parkingQueue) {
473                                                agent.lane = lane;
474                                                agent.speed = 0.0;
475                                                AgentSnapshotInfo position = snapshotInfoFactory.createAgentSnapshotInfo(agent.id, this.link, distFromFromNode, agent.lane);
476                                                position.setColorValueBetweenZeroAndOne( agent.speed) ;
477                                                position.setAgentState(AgentSnapshotInfo.AgentState.PERSON_AT_ACTIVITY);
478                                                positions.add(position);
479                                                distFromFromNode -= cellSize;
480                                        }
481                                }
482                        }
483                }
484        }

486        private static class EventAgent implements Comparable<EventAgent> {
487                protected final Id<Person> id;
488                protected final int intId;
489                protected double time;
490                protected EventLink currentLink = null;
491                protected double speed = 0.0;
492                protected int lane = 1;
493                protected EventAgent(final Id<Person> id, final double time) {
494                        this.id = id;
495                        this.time = time;
496                        this.intId = id.hashCode();
497                }
498
499                @Override
500                public int compareTo(final EventAgent o) {
501                        return this.id.compareTo(o.id);
502                }
503
504                @Override
505                public boolean equals(final Object o) {
506                        if (o instanceof EventAgent) {
507                                return this.id.equals(((EventAgent) o).id);
508                        }
509                        return false;
510                }
511
512                @Override
513                public int hashCode() {
514                        return this.id.hashCode();
515                }
516
517        }
518
519        @Override
520        public void handleEvent(VehicleLeavesTrafficEvent event) {
521                delegate.handleEvent(event);
522        }
