062public final class EventsReaderXMLv1 extends MatsimXmlParser {
063
064        public interface CustomEventMapper<T extends Event> /* extends Function<GenericEvent, T> */ {
065                T apply(GenericEvent event);
066        }
067
068        static public final String EVENT = "event";
069
070        private final EventsManager events;
071        private final Map<String, CustomEventMapper> customEventMappers = new HashMap<>();
072
073        public EventsReaderXMLv1(final EventsManager events) {
074                this.events = events;
075                this.setValidating(false);// events-files have no DTD, thus they cannot validate
076        }
077
078        public void addCustomEventMapper(String eventType, CustomEventMapper cem) {
079                customEventMappers.put(eventType, cem);
080        }
081
082        @Override
083        public void startTag(final String name, final Attributes atts, final Stack<String> context) {
084                if (EVENT.equals(name)) {
085                        startEvent(atts);
086                }
087        }
088
089        @Override
090        public void characters(char[] ch, int start, int length) throws SAXException {
091                // ignore characters to prevent OutOfMemoryExceptions
092                /* the events-file only contains empty tags with attributes,
093                 * but without the dtd or schema, all whitespace between tags is handled
094                 * by characters and added up by super.characters, consuming huge
095                 * amount of memory when large events-files are read in.
096                 */
097        }
098
099        @Override
100        public void endTag(final String name, final String content, final Stack<String> context) {
101        }
102
103        private void startEvent(final Attributes atts) {
104                double time = Double.parseDouble(atts.getValue("time"));
105                String eventType = atts.getValue("type");
106
107                // === material related to wait2link below here ===
108                if (LinkLeaveEvent.EVENT_TYPE.equals(eventType)) {
109                        this.events.processEvent(new LinkLeaveEvent(time, 
110                                        Id.create(atts.getValue(LinkLeaveEvent.ATTRIBUTE_VEHICLE), Vehicle.class), 
111                                        Id.create(atts.getValue(LinkLeaveEvent.ATTRIBUTE_LINK), Link.class)
112                                        // had driver id in previous version
113                                        ));
114                } else if (LinkEnterEvent.EVENT_TYPE.equals(eventType)) {
115                        this.events.processEvent(new LinkEnterEvent(time, 
116                                        Id.create(atts.getValue(LinkEnterEvent.ATTRIBUTE_VEHICLE), Vehicle.class), 
117                                        Id.create(atts.getValue(LinkEnterEvent.ATTRIBUTE_LINK), Link.class)
118                                        // had driver id in previous version
119                                        ));
120                } else if (VehicleEntersTrafficEvent.EVENT_TYPE.equals(eventType) ) {
121                        // (this is the new version, marked by the new events name)
122
123                        this.events.processEvent(new VehicleEntersTrafficEvent(time, 
124                                        Id.create(atts.getValue(VehicleEntersTrafficEvent.ATTRIBUTE_DRIVER), Person.class), 
125                                        Id.create(atts.getValue(VehicleEntersTrafficEvent.ATTRIBUTE_LINK), Link.class), 
126                                        Id.create(atts.getValue(VehicleEntersTrafficEvent.ATTRIBUTE_VEHICLE), Vehicle.class),
127                                        atts.getValue(VehicleEntersTrafficEvent.ATTRIBUTE_NETWORKMODE), 
128                                        Double.parseDouble( atts.getValue( VehicleEntersTrafficEvent.ATTRIBUTE_POSITION) )
129                                        ));
130                } else if ( "wait2link".equals(eventType) ) {
131                        // (this is the old version, marked by the old events name)
132
133                        // retrofit vehicle Id:
134                        Id<Vehicle> vehicleId ;
135                        if ( atts.getValue(VehicleEntersTrafficEvent.ATTRIBUTE_VEHICLE) != null ) {
136                                vehicleId = Id.create( atts.getValue(VehicleEntersTrafficEvent.ATTRIBUTE_VEHICLE), Vehicle.class ) ;
137                        } else {
138                                // for the old events type, we set the vehicle id to the driver id if the vehicle id does not exist:
139                                vehicleId = Id.create(atts.getValue(VehicleEntersTrafficEvent.ATTRIBUTE_DRIVER), Vehicle.class);
140                        }
141                        // retrofit position:
142                        double position ;
143                        if ( atts.getValue( VehicleEntersTrafficEvent.ATTRIBUTE_POSITION)!=null ) {
144                                position = Double.parseDouble( atts.getValue( VehicleEntersTrafficEvent.ATTRIBUTE_POSITION) ) ;
145                        } else {
146                                position = 1.0 ;
147                        }
148                        this.events.processEvent(new VehicleEntersTrafficEvent(time, 
149                                        Id.create(atts.getValue(VehicleEntersTrafficEvent.ATTRIBUTE_DRIVER), Person.class), 
150                                        Id.create(atts.getValue(VehicleEntersTrafficEvent.ATTRIBUTE_LINK), Link.class), 
151                                        vehicleId,
152                                        atts.getValue(VehicleEntersTrafficEvent.ATTRIBUTE_NETWORKMODE), 
153                                        position
154                                        ));
155                } else if (VehicleLeavesTrafficEvent.EVENT_TYPE.equals(eventType)) {
156                        this.events.processEvent(new VehicleLeavesTrafficEvent(time, 
157                                        Id.create(atts.getValue(VehicleLeavesTrafficEvent.ATTRIBUTE_DRIVER), Person.class), 
158                                        Id.create(atts.getValue(VehicleLeavesTrafficEvent.ATTRIBUTE_LINK), Link.class), 
159                                        atts.getValue(VehicleLeavesTrafficEvent.ATTRIBUTE_VEHICLE) == null ? null : Id.create(atts.getValue(VehicleLeavesTrafficEvent.ATTRIBUTE_VEHICLE), Vehicle.class), 
160                                        atts.getValue(VehicleLeavesTrafficEvent.ATTRIBUTE_NETWORKMODE), 
161                                        Double.parseDouble( atts.getValue( VehicleLeavesTrafficEvent.ATTRIBUTE_POSITION) )
162                                        ));
163                }
164                // === material related to wait2link above here
165                else if (ActivityEndEvent.EVENT_TYPE.equals(eventType)) {
166                        this.events.processEvent(new ActivityEndEvent(
167                                        time, 
168                                        Id.create(atts.getValue(ActivityEndEvent.ATTRIBUTE_PERSON), Person.class), 
169                                        Id.create(atts.getValue(ActivityEndEvent.ATTRIBUTE_LINK), Link.class), 
170                                        atts.getValue(ActivityEndEvent.ATTRIBUTE_FACILITY) == null ? null : Id.create(atts.getValue(ActivityEndEvent.ATTRIBUTE_FACILITY), ActivityFacility.class), 
171                                        atts.getValue(ActivityEndEvent.ATTRIBUTE_ACTTYPE)));
172                } else if (ActivityStartEvent.EVENT_TYPE.equals(eventType)) {
173                        this.events.processEvent(new ActivityStartEvent(time, Id.create(atts.getValue(ActivityStartEvent.ATTRIBUTE_PERSON), Person.class), Id.create(atts.getValue(ActivityStartEvent.ATTRIBUTE_LINK), Link.class), atts.getValue(ActivityStartEvent.ATTRIBUTE_FACILITY) == null ? null : Id.create(atts.getValue(ActivityStartEvent.ATTRIBUTE_FACILITY), ActivityFacility.class), atts.getValue(ActivityStartEvent.ATTRIBUTE_ACTTYPE)));
174                } else if (PersonArrivalEvent.EVENT_TYPE.equals(eventType)) {
175                        String legMode = atts.getValue(PersonArrivalEvent.ATTRIBUTE_LEGMODE);
176                        String mode = legMode == null ? null : legMode.intern();
177                        this.events.processEvent(new PersonArrivalEvent(time, Id.create(atts.getValue(PersonArrivalEvent.ATTRIBUTE_PERSON), Person.class), Id.create(atts.getValue(PersonArrivalEvent.ATTRIBUTE_LINK), Link.class), mode));
178                } else if (PersonDepartureEvent.EVENT_TYPE.equals(eventType)) {
179                        String legMode = atts.getValue(PersonDepartureEvent.ATTRIBUTE_LEGMODE);
180                        String mode = legMode == null ? null : legMode.intern();
181                        this.events.processEvent(new PersonDepartureEvent(time, Id.create(atts.getValue(PersonDepartureEvent.ATTRIBUTE_PERSON), Person.class), Id.create(atts.getValue(PersonDepartureEvent.ATTRIBUTE_LINK), Link.class), mode));
182                } else if (PersonStuckEvent.EVENT_TYPE.equals(eventType)) {
183                        String legMode = atts.getValue(PersonStuckEvent.ATTRIBUTE_LEGMODE);
184                        String mode = legMode == null ? null : legMode.intern();
185                        String linkIdString = atts.getValue(PersonStuckEvent.ATTRIBUTE_LINK);
186                        Id<Link> linkId = linkIdString == null ? null : Id.create(linkIdString, Link.class); // linkId is optional
187                        this.events.processEvent(new PersonStuckEvent(time, Id.create(atts.getValue(PersonStuckEvent.ATTRIBUTE_PERSON), Person.class), linkId, mode));
188                } else if (VehicleAbortsEvent.EVENT_TYPE.equals(eventType)) {
189                        String linkIdString = atts.getValue(VehicleAbortsEvent.ATTRIBUTE_LINK);
190                        Id<Link> linkId = linkIdString == null ? null : Id.create(linkIdString, Link.class);
191                        this.events.processEvent(new VehicleAbortsEvent(time, Id.create(atts.getValue(VehicleAbortsEvent.ATTRIBUTE_VEHICLE), Vehicle.class), linkId));
192                }else if (PersonMoneyEvent.EVENT_TYPE.equals(eventType) || "agentMoney".equals(eventType)) {
193                        this.events.processEvent(new PersonMoneyEvent(time, Id.create(atts.getValue(PersonMoneyEvent.ATTRIBUTE_PERSON), Person.class), Double.parseDouble(atts.getValue(PersonMoneyEvent.ATTRIBUTE_AMOUNT))));
194                } else if (PersonEntersVehicleEvent.EVENT_TYPE.equals(eventType)) {
195                        String personString = atts.getValue(PersonEntersVehicleEvent.ATTRIBUTE_PERSON);
196                        String vehicleString = atts.getValue(PersonEntersVehicleEvent.ATTRIBUTE_VEHICLE);
197                        this.events.processEvent(new PersonEntersVehicleEvent(time, Id.create(personString, Person.class), Id.create(vehicleString, Vehicle.class)));
198                } else if (PersonLeavesVehicleEvent.EVENT_TYPE.equals(eventType)) {
199                        Id<Person> pId = Id.create(atts.getValue(PersonLeavesVehicleEvent.ATTRIBUTE_PERSON), Person.class);
200                        Id<Vehicle> vId = Id.create(atts.getValue(PersonLeavesVehicleEvent.ATTRIBUTE_VEHICLE), Vehicle.class);
201                        this.events.processEvent(new PersonLeavesVehicleEvent(time, pId, vId));
202                } else if (TeleportationArrivalEvent.EVENT_TYPE.equals(eventType)) {
203                        this.events.processEvent(new TeleportationArrivalEvent(
204                                        time, 
205                                        Id.create(atts.getValue(TeleportationArrivalEvent.ATTRIBUTE_PERSON), Person.class), 
206                                        Double.parseDouble(atts.getValue(TeleportationArrivalEvent.ATTRIBUTE_DISTANCE))));
207                } else if (VehicleArrivesAtFacilityEvent.EVENT_TYPE.equals(eventType)) {
208                        String delay = atts.getValue(VehicleArrivesAtFacilityEvent.ATTRIBUTE_DELAY);
209                        this.events.processEvent(new VehicleArrivesAtFacilityEvent(time, Id.create(atts.getValue(VehicleArrivesAtFacilityEvent.ATTRIBUTE_VEHICLE), Vehicle.class), Id.create(atts.getValue(VehicleArrivesAtFacilityEvent.ATTRIBUTE_FACILITY), TransitStopFacility.class), delay == null ? 0.0 : Double.parseDouble(delay)));
210                } else if (VehicleDepartsAtFacilityEvent.EVENT_TYPE.equals(eventType)) {
211                        String delay = atts.getValue(VehicleDepartsAtFacilityEvent.ATTRIBUTE_DELAY);
212                        this.events.processEvent(new VehicleDepartsAtFacilityEvent(time, Id.create(atts.getValue(VehicleArrivesAtFacilityEvent.ATTRIBUTE_VEHICLE), Vehicle.class), Id.create(atts.getValue(VehicleArrivesAtFacilityEvent.ATTRIBUTE_FACILITY), TransitStopFacility.class), delay == null ? 0.0 : Double.parseDouble(delay)));
213                } else if (TransitDriverStartsEvent.EVENT_TYPE.equals(eventType)) {
214                        this.events.processEvent(new TransitDriverStartsEvent(time, Id.create(atts.getValue(TransitDriverStartsEvent.ATTRIBUTE_DRIVER_ID), Person.class), Id.create(atts.getValue(TransitDriverStartsEvent.ATTRIBUTE_VEHICLE_ID), Vehicle.class), Id.create(atts.getValue(TransitDriverStartsEvent.ATTRIBUTE_TRANSIT_LINE_ID), TransitLine.class), Id.create(atts.getValue(TransitDriverStartsEvent.ATTRIBUTE_TRANSIT_ROUTE_ID), TransitRoute.class), Id.create(atts.getValue(TransitDriverStartsEvent.ATTRIBUTE_DEPARTURE_ID), Departure.class)));
215                } else if (BoardingDeniedEvent.EVENT_TYPE.equals(eventType)){
216                        Id<Person> personId = Id.create(atts.getValue(BoardingDeniedEvent.ATTRIBUTE_PERSON_ID), Person.class);
217                        Id<Vehicle> vehicleId = Id.create(atts.getValue(BoardingDeniedEvent.ATTRIBUTE_VEHICLE_ID), Vehicle.class);
218                        this.events.processEvent(new BoardingDeniedEvent(time, personId, vehicleId));
219                } else if (AgentWaitingForPtEvent.EVENT_TYPE.equals(eventType)){
220                        Id<Person> agentId = Id.create(atts.getValue(AgentWaitingForPtEvent.ATTRIBUTE_AGENT), Person.class);
221                        Id<TransitStopFacility> waitStopId = Id.create(atts.getValue(AgentWaitingForPtEvent.ATTRIBUTE_WAITSTOP), TransitStopFacility.class);
222                        Id<TransitStopFacility> destinationStopId = Id.create(atts.getValue(AgentWaitingForPtEvent.ATTRIBUTE_DESTINATIONSTOP), TransitStopFacility.class);
223                        this.events.processEvent(new AgentWaitingForPtEvent(time, agentId, waitStopId, destinationStopId));
224                } else {
225                        GenericEvent event = new GenericEvent(eventType, time);
226                        for ( int ii=0; ii<atts.getLength(); ii++ ) {
227                                String key = atts.getLocalName(ii);
228                                if ( key.equals("time") || key.equals("type") ) {
229                                        continue;
230                                }
231                                String value = atts.getValue(ii);
232                                event.getAttributes().put(key, value);
233                        }
234                        CustomEventMapper cem = customEventMappers.get(eventType);
235                        if (cem != null) {
236                                this.events.processEvent(cem.apply(event));
237                        } else {
238                                this.events.processEvent(event);
239                        }
240                }
241        }
