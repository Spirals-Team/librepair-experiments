001package org.matsim.core.events;
002
003import org.matsim.core.api.experimental.events.EventsManager;
004import org.matsim.core.config.Config;
005import org.matsim.core.config.ConfigUtils;
006import org.matsim.core.controler.Injector;
007
008public class EventsUtils {
009
010    public static EventsManager createEventsManager() {
011        return new EventsManagerImpl();
012    }
013
014    public static EventsManager createEventsManager(Config config) {
015        return Injector.createInjector(config, new EventsManagerModule())
016                .getInstance(EventsManager.class);
017    }
018
019    /**
020     * The SimStepParallelEventsManagerImpl can handle events from multiple threads.
021     * The (Parallel)EventsMangerImpl cannot, therefore it has to be wrapped into a
022     * SynchronizedEventsManagerImpl.
023     */
024    public static EventsManager getParallelFeedableInstance(EventsManager events) {
025        if (events instanceof SimStepParallelEventsManagerImpl) {
026                return events;
027        } else if (events instanceof ParallelEventsManager) {
028                return events;
029        }
030        else if (events instanceof SynchronizedEventsManagerImpl) {
031                return events;
032        } else {
033                return new SynchronizedEventsManagerImpl(events);
034        }
035    }
036    public static void readEvents( EventsManager events, String filename ) {
037        new MatsimEventsReader(events).readFile(filename) ;
038        }
039}
