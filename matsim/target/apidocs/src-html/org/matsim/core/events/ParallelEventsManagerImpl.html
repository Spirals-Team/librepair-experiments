021package org.matsim.core.events;
022
023import java.lang.Thread.UncaughtExceptionHandler;
024import java.util.concurrent.atomic.AtomicBoolean;
025
026import org.apache.log4j.Logger;
027import org.matsim.api.core.v01.events.Event;
028import org.matsim.core.api.experimental.events.EventsManager;
029import org.matsim.core.config.Config;
030import org.matsim.core.events.handler.EventHandler;
031
032import javax.inject.Inject;
033
034/**
035 *
036 * ParallelEvents allows parallelization for events handling. Usage: First
037 * create an object of this class. Before each iteration, call initProcessing.
038 * After each iteration, call finishProcessing. This has already been
039 * incorporated into the Controller.
040 *
041 * Usage via config.xml:
042 *
043 * <pre>
044 * <module name="parallelEventHandling">
045 *  <param name="numberOfThreads" value="2" />
046 * </module>
047 * </pre>
048 *
049 * optionally you can also specify the estimated number of events per iteration:
050 *
051 * <pre>
052 *  <param name="estimatedNumberOfEvents" value="10000000" />
053 * </pre>
054 *
055 * (not really needed, but can make performance slightly faster in larger
056 * simulations).
057 *
058 * @see <a href="http://www.matsim.org/node/238">http://www.matsim.org/node/238</a>
059 * @author rashid_waraich
060 *
061 */
062public final class ParallelEventsManagerImpl implements EventsManager {
063
064        private boolean parallelMode = true;
065        private int numberOfThreads;
066        private EventsManagerImpl[] events = null;
067        private ProcessEventThread[] eventsProcessThread = null;
068        private Thread[] threads = null;
069        private int numberOfAddedEventsHandler = 0;
070        private final AtomicBoolean hadException = new AtomicBoolean(false);
071        private final ExceptionHandler uncaughtExceptionHandler = new ExceptionHandler(hadException);
072
073        private final static Logger log = Logger.getLogger(ParallelEventsManagerImpl.class);
074
075        // this number should be set in the following way:
076        // if the number of events is estimated as x, then this number
077        // could be set to x/10
078        // the higher this parameter, the less locks are used, but
079        // the more the time buffer between the simulation and events handling
080        // for small simulations, the default value is ok and it even works
081        // quite well for larger simulations with 10 million events
082        private int preInputBufferMaxLength = 100000;
083
084        @Inject
085        ParallelEventsManagerImpl(Config config) {
086                if (config.parallelEventHandling().getEstimatedNumberOfEvents() != null) {
087                        preInputBufferMaxLength = (int) (config.parallelEventHandling().getEstimatedNumberOfEvents() / 10);
088                }
089                init(config.parallelEventHandling().getNumberOfThreads());
090        }
091
092        /**
093         * @param numberOfThreads
094         *            - specify the number of threads used for the events handler
095         */
096        public ParallelEventsManagerImpl(int numberOfThreads) {
097                init(numberOfThreads);
098        }
099
100        /**
101         *
102         * @param numberOfThreads
103         * @param estimatedNumberOfEvents
104         *            Only use this constructor for larger simulations (20M+
105         *            events).
106         */
107        public ParallelEventsManagerImpl(int numberOfThreads, long estimatedNumberOfEvents) {
108                preInputBufferMaxLength = (int) (estimatedNumberOfEvents / 10 );
109                init(numberOfThreads);
110        }
111
112        @Override
113        public void processEvent(final Event event) {
114                if (parallelMode) {
115                        for (int i = 0; i < eventsProcessThread.length; i++) {
116                                eventsProcessThread[i].processEvent(event);
117                        }
118                } else {
119                        for (int i = 0; i < eventsProcessThread.length; i++) {
120                                eventsProcessThread[i].getEvents().processEvent(event);
121                        }
122                }
123        }
124
125        @Override
126        public void addHandler(final EventHandler handler) {
127                synchronized (this) {
128                        log.info("adding Event-Handler " + handler.getClass().getName() + " to thread " + numberOfAddedEventsHandler);
129                        events[numberOfAddedEventsHandler].addHandler(handler);
130                        numberOfAddedEventsHandler = (numberOfAddedEventsHandler + 1) % numberOfThreads;
131                }
132        }
133
134        @Override
135        public void resetHandlers(final int iteration) {
136                synchronized (this) {
137                        for (int i = 0; i < events.length; i++) {
138                                events[i].resetHandlers(iteration);
139                        }
140                }
141        }
142
143        @Override
144        public void removeHandler(final EventHandler handler) {
145                synchronized (this) {
146                        for (int i = 0; i < events.length; i++) {
147                                events[i].removeHandler(handler);
148                        }
149                }
150        }
151
152        private void printEventHandlers() {
153                synchronized (this) {
154                        for (int i = 0; i < events.length; i++) {
155                                log.info("registered event handlers for thread " + i + ":");
156                                events[i].printEventHandlers();
157                        }
158                }
159        }
160
161        private void init(int numberOfThreads) {
162                this.numberOfThreads = numberOfThreads;
163                this.events = new EventsManagerImpl[numberOfThreads];
164                this.eventsProcessThread = new ProcessEventThread[numberOfThreads];
165                this.threads = new Thread[numberOfThreads];
166                // the additional 1 is for the simulation barrier
167                for (int i = 0; i < numberOfThreads; i++) {
168                        events[i] = (EventsManagerImpl) EventsUtils.createEventsManager();
169                }
170        }
171
172        // When one simulation iteration is finish, it must call this method,
173        // so that it can communicate to the threads, that the simulation is
174        // finished and that it can await the event handler threads.
175
176        // after call to this method, all event processing is done not in parallel
177        // anymore
178        @Override
179        public void finishProcessing() {
180                for (int i = 0; i < eventsProcessThread.length; i++) {
181                        eventsProcessThread[i].close();
182                }
183
184                try {
185                        for (Thread t : this.threads) {
186                                t.join();
187                        }
188                } catch (InterruptedException e) {
189                        e.printStackTrace();
190                }
191                
192                // list which threads had which handlers to debug performance issues
193                printEventHandlers();
194
195                /*
196                 * introduction of the parallel mode variable was required, because of
197                 * the following reason: previously no event handling was possible after
198                 * the end of the simulation. e.g. adding money events in the after
199                 * mobsim controler listener would not be invoked by parallelEventHandling
200                 */
201
202                parallelMode = false;
203
204                if (this.hadException.get()) {
205                        throw new RuntimeException("Exception while processing events. Cannot guarantee that all events have been fully processed.");
206                }
207        }
208
209        // create event handler threads
210        // prepare for next iteration
211        @Override
212        public void initProcessing() {
213                // reset this class, so that it can be reused for the next iteration
214                for (int i = 0; i < numberOfThreads; i++) {
215                        this.eventsProcessThread[i] = new ProcessEventThread(events[i], preInputBufferMaxLength);
216                        this.threads[i] = new Thread(eventsProcessThread[i], "Events-" + i);
217                        this.threads[i].setUncaughtExceptionHandler(this.uncaughtExceptionHandler);
218                        this.threads[i].start();
219                }
220                
221                // (re-)activate parallel mode while the mobsim is running
222                this.parallelMode = true;
223        }
224
225        /**
226         * @author mrieser
227         */
228        private static class ExceptionHandler implements UncaughtExceptionHandler {
229
230                private final AtomicBoolean hadException;
231
232                public ExceptionHandler(final AtomicBoolean hadException) {
233                        this.hadException = hadException;
234                }
235
236                @Override
237                public void uncaughtException(Thread t, Throwable e) {
238                        log.error("Thread " + t.getName() + " died with exception while handling events.", e);
239                        this.hadException.set(true);
240                }
241
242        }
243
244        @Override
245        public void afterSimStep(double time) {
246                // nothing to do in this implementation
247        }
248
249}
