022package org.matsim.core.events;
023
024import java.util.HashMap;
025import java.util.Map;
026import java.util.Stack;
027
028import org.matsim.api.core.v01.Id;
029import org.matsim.api.core.v01.events.LinkEnterEvent;
030import org.matsim.api.core.v01.events.LinkLeaveEvent;
031import org.matsim.api.core.v01.events.PersonArrivalEvent;
032import org.matsim.api.core.v01.events.PersonDepartureEvent;
033import org.matsim.api.core.v01.events.PersonLeavesVehicleEvent;
034import org.matsim.api.core.v01.events.VehicleEntersTrafficEvent;
035import org.matsim.api.core.v01.events.VehicleLeavesTrafficEvent;
036import org.matsim.api.core.v01.network.Link;
037import org.matsim.api.core.v01.population.Person;
038import org.matsim.core.api.experimental.events.EventsManager;
039import org.matsim.core.utils.io.MatsimXmlParser;
040import org.matsim.vehicles.Vehicle;
041import org.xml.sax.Attributes;
042
043/**
044 * This class reads and completes event files regarding vehicle ids.
045 * 
046 * If the Wait2LinkEvents contain vehicle ids different from null, this ids are added to the LinkEnter- and LinkLeaveEvents.
047 * If not, the person id is used as vehicle id in Wait2Link-, LinkLeave- and LinkEnterEvents.
048 * 
049 * Additionally, this class adds VehicleLeavesTrafficEvents if they are not existing.
050 * 
051 * @author tthunig
052 *
053 */
054public final class EventsConverterXML extends MatsimXmlParser{
055
056        private static final String EVENT = "event";
057        private static final String ATTRIBUTE_PERSON = "person";
058
059        private final EventsManager events;
060        private final EventsReaderXMLv1 basicEventsReader;
061        private boolean containsVehicleLeavesTrafficEvents = false;
062        private Map<Id<Person>, Id<Vehicle>> driverToVeh = new HashMap<>();
063        private Map<Id<Person>, String> personToLegMode = new HashMap<>();
064
065        public EventsConverterXML(final EventsManager events) {
066                this.events = events;
067                this.setValidating(false);// events-files have no DTD, thus they cannot validate
068                this.basicEventsReader = new EventsReaderXMLv1(events);
069        }
070
071
072        @Override
073        public void startTag(String name, Attributes atts, Stack<String> context) {
074                if ( EVENT.equals(name)) {
075                        // (we are essentially ignoring the xml header)
076                        
077                        double time = Double.parseDouble(atts.getValue("time"));
078                        String eventType = atts.getValue("type");
079
080                        switch (eventType){
081                        case PersonDepartureEvent.EVENT_TYPE:
082                                // remember person to (leg) mode relation
083                                personToLegMode.put(Id.createPersonId(atts.getValue(PersonDepartureEvent.ATTRIBUTE_PERSON)), atts.getValue(PersonDepartureEvent.ATTRIBUTE_LEGMODE));
084                                this.basicEventsReader.startTag(name, atts, context);
085                                break;
086                        case VehicleEntersTrafficEvent.EVENT_TYPE:
087                        case "wait2link":
088                                // (assumes that the reader has already converted the wait2link events)
089                                Id<Person> driverId = Id.createPersonId(atts.getValue(VehicleEntersTrafficEvent.ATTRIBUTE_DRIVER));
090                                Id<Vehicle> vehicleId;
091                                if (atts.getValue(VehicleEntersTrafficEvent.ATTRIBUTE_VEHICLE) == null || atts.getValue(VehicleEntersTrafficEvent.ATTRIBUTE_VEHICLE).equals("null")){
092                                        // use the person id as vehicle id
093                                        vehicleId = Id.create(driverId, Vehicle.class);
094                                } else {
095                                        vehicleId = Id.create(atts.getValue(VehicleEntersTrafficEvent.ATTRIBUTE_VEHICLE), Vehicle.class);
096                                }
097                                assert vehicleId != null;
098
099                                String networkMode = atts.getValue(VehicleEntersTrafficEvent.ATTRIBUTE_NETWORKMODE);
100                                // use the (leg) mode from the departure event if this event does not contain a (network) mode
101                                if (networkMode == null){
102                                        networkMode = personToLegMode.get(driverId);
103                                }
104                                
105                                // remember driver to vehicle relation
106                                driverToVeh.put(driverId, vehicleId);                           
107                                // process event with correct vehicleId
108                                this.events.processEvent(new VehicleEntersTrafficEvent(time, driverId, Id.createLinkId(atts.getValue(VehicleEntersTrafficEvent.ATTRIBUTE_LINK)), 
109                                                vehicleId, networkMode, 1.0));
110                                break;
111                        case LinkEnterEvent.EVENT_TYPE:
112                                if (atts.getValue(LinkEnterEvent.ATTRIBUTE_VEHICLE) == null || atts.getValue(LinkEnterEvent.ATTRIBUTE_VEHICLE).equals("null")){
113                                        // create an link enter event with the correct vehicle id
114                                        this.events.processEvent(new LinkEnterEvent(time, driverToVeh.get(Id.createPersonId(atts.getValue(ATTRIBUTE_PERSON))), 
115                                                        Id.create(atts.getValue(LinkEnterEvent.ATTRIBUTE_LINK), Link.class)));
116                                } else {
117                                        // the event already contains the vehicle id and can be processed normally
118                                        this.basicEventsReader.startTag(name, atts, context);
119                                }
120                                break;
121                        case LinkLeaveEvent.EVENT_TYPE:
122                                if (atts.getValue(LinkLeaveEvent.ATTRIBUTE_VEHICLE) == null || atts.getValue(LinkLeaveEvent.ATTRIBUTE_VEHICLE).equals("null")){
123                                        // create an link leave event with the correct vehicle id
124                                        final Id<Person> personId = Id.createPersonId(atts.getValue(ATTRIBUTE_PERSON));
125                                        assert personId != null;
126                                        final Id<Link> linkId = Id.create(atts.getValue(LinkLeaveEvent.ATTRIBUTE_LINK), Link.class);
127                                        assert linkId != null;
128                                        vehicleId = driverToVeh.get(personId);
129                                        assert vehicleId != null;
130                                        this.events.processEvent(new LinkLeaveEvent(time, vehicleId, linkId));
131                                } else {
132                                        // the event already contains the vehicle id and can be processed normally
133                                        this.basicEventsReader.startTag(name, atts, context);
134                                }
135                                break;
136                        case VehicleLeavesTrafficEvent.EVENT_TYPE:
137                                this.containsVehicleLeavesTrafficEvents  = true;
138                                this.basicEventsReader.startTag(name, atts, context);
139                                break;
140                        case PersonLeavesVehicleEvent.EVENT_TYPE:
141                                if (driverToVeh.containsKey(Id.createPersonId(atts.getValue(PersonLeavesVehicleEvent.ATTRIBUTE_PERSON))) && !this.containsVehicleLeavesTrafficEvents){
142                                        /* if the person is a driver and the vehicle leaves traffic event is missing ...
143                                         * ... process the event later (during person arrival) to be able to collect 
144                                         * some information about the missing vehicle leaves traffic event before */
145                                } else {
146                                        // process normally
147                                        this.basicEventsReader.startTag(name, atts, context);
148                                }
149                                break;
150                        case PersonArrivalEvent.EVENT_TYPE:
151                                Id<Person> personId = Id.createPersonId(atts.getValue(PersonArrivalEvent.ATTRIBUTE_PERSON));
152                                Id<Link> linkId = Id.createLinkId(atts.getValue(PersonArrivalEvent.ATTRIBUTE_LINK));
153                                String mode = atts.getValue(PersonArrivalEvent.ATTRIBUTE_LEGMODE);
154
155                                // create a vehicle leaves traffic event if it is missing
156                                if (driverToVeh.containsKey(personId) && !this.containsVehicleLeavesTrafficEvents){
157//                                      Id<Vehicle> vehicleIdOfDriver = driverToVeh.get(personId);
158                                        // remove the person-car-mapping, otherwise every following (even non-vehicle) leg produces leave traffic/vehicle events 
159                                        Id<Vehicle> vehicleIdOfDriver = driverToVeh.remove(personId); 
160
161                                        this.events.processEvent(new VehicleLeavesTrafficEvent(time, personId, linkId, vehicleIdOfDriver, mode, 1.0));
162                                        this.events.processEvent(new PersonLeavesVehicleEvent(time, personId, vehicleIdOfDriver));
163                                }
164                                // process the PersonArrivalEvent normally, independently of missing VehicleLeavesTrafficEvents
165                                this.basicEventsReader.startTag(name, atts, context);
166                                break;
167                        default:
168                                this.basicEventsReader.startTag(name, atts, context);
169                                break;
170                        }
171                }
172        }
173
174        @Override
175        public void endTag(String name, String content, Stack<String> context) {
176        }
177
178}
