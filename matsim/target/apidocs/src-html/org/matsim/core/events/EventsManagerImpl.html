068/**
069 * EventHandling
070 * <ol>
071 * <li>Create a new class MyEventClass extends Event</li>
072 * <li>Create a new interface MyEventHandlerI extends EventHandler</li>
073 * <li>add method public void handleEvent(MyEvent event) to it</li>
074 * <li>ready to go, just implement the interface somewhere and add a
075 * HandlerObject with a call to <code>Events.addHandler(HandlerObject)</code></li>
076 * <li>(optional) add an appropriate line in callHandlerFast() for speeding
077 * up execution!</li>
078 * </ol>
079 *
080 * @author dstrippgen
081 * @author mrieser
082 */
083public final class EventsManagerImpl implements EventsManager {
084
085        private static final Logger log = Logger.getLogger(EventsManagerImpl.class);
086
087        static private class HandlerData {
088
089                protected Class<?> eventklass;
090                protected ArrayList<EventHandler> handlerList = new ArrayList<EventHandler>(5);
091                protected Method method;
092                protected HandlerData(final Class<?> eventklass, final Method method) {
093                        this.eventklass = eventklass;
094                        this.method = method;
095                }
096                protected void removeHandler(final EventHandler handler) {
097                        this.handlerList.remove(handler);
098                }
099        }
100
101        static private class HandlerInfo {
102                protected final Class<?> eventClass;
103                protected final EventHandler eventHandler;
104                protected final Method method;
105
106                protected HandlerInfo(final Class<?> eventClass, final EventHandler eventHandler, final Method method) {
107                        this.eventClass = eventClass;
108                        this.eventHandler = eventHandler;
109                        this.method = method;
110                }
111        }
112
113        private final List<HandlerData> handlerData = new ArrayList<HandlerData>();
114
115        private final Map<Class<?>, HandlerInfo[]> cacheHandlers = new ConcurrentHashMap<Class<?>, HandlerInfo[]>(15);
116
117        private long counter = 0;
118        private long nextCounterMsg = 1;
119
120        private HandlerData findHandler(final Class<?> evklass) {
121                for (HandlerData handler : this.handlerData) {
122                        if (handler.eventklass == evklass) {
123                                return handler;
124                        }
125                }
126                return null;
127        }
128
129        @Override
130        public void processEvent(final Event event) {
131                this.counter++;
132                if (this.counter == this.nextCounterMsg) {
133                        this.nextCounterMsg *= 4;
134                        log.info(" event # " + this.counter);
135                }
136                computeEvent(event);
137        }

139
140        @Override
141        public void addHandler (final EventHandler handler) {
142                Set<Class<?>> addedHandlers = new HashSet<Class<?>>();
143                Class<?> test = handler.getClass();
144                log.info("adding Event-Handler: " + test.getName());
145                while (test != Object.class) {
146                        for (Class<?> theInterface: test.getInterfaces()) {
147                                if (!addedHandlers.contains(theInterface)) {
148                                        log.info("  " + theInterface.getName());
149                                        addHandlerInterfaces(handler, theInterface);
150                                        addedHandlers.add(theInterface);
151                                }
152                        }
153                        test = test.getSuperclass();
154                }
155                this.cacheHandlers.clear();
156                log.info("");
157        }
158
159        @Override
160        public void removeHandler(final EventHandler handler) {
161                log.info("removing Event-Handler: " + handler.getClass().getName());
162                for (HandlerData handlerList : this.handlerData) {
163                        handlerList.removeHandler(handler);
164                }
165                this.cacheHandlers.clear();
166        }
167
168        @Override
169        public void resetHandlers(final int iteration) {
170                log.info("resetting Event-Handlers");
171                this.counter = 0;
172                this.nextCounterMsg = 1;
173                Set<EventHandler> resetHandlers = new HashSet<EventHandler>();
174                for (HandlerData handlerdata : this.handlerData) {
175                        for (EventHandler handler : handlerdata.handlerList) {
176                                if (!resetHandlers.contains(handler)) {
177                                        log.info("  " + handler.getClass().getName());
178                                        handler.reset(iteration);
179                                        resetHandlers.add(handler);
180                                }
181                        }
182                }
183        }

200        private void addHandlerInterfaces(final EventHandler handler, final Class<?> handlerClass) {
201                Method[] classmethods = handlerClass.getMethods();
202                for (Method method : classmethods) {
203                        if (method.getName().equals("handleEvent")) {
204                                Class<?>[] params = method.getParameterTypes();
205                                if (params.length == 1) {
206                                        Class<?> eventClass = params[0];
207                                        log.info("    > " + eventClass.getName());
208                                        HandlerData dat = findHandler(eventClass);
209                                        if (dat == null) {
210                                                dat = new HandlerData(eventClass, method);
211                                                this.handlerData.add(dat);
212                                        }
213                                        dat.handlerList.add(handler);
214                                }
215                        }
216                }
217        }
218
219        private void computeEvent(final Event event) {
220                for (HandlerInfo info : getHandlersForClass(event.getClass())) {
221                        synchronized(info.eventHandler) {
222                                if (callHandlerFast(info.eventClass, event, info.eventHandler)) {
223                                        continue;
224                                }
225                                try {
226                                        info.method.invoke(info.eventHandler, event);
227                                } catch (IllegalArgumentException | IllegalAccessException e) {
228                                        throw new RuntimeException("problem invoking EventHandler " + info.eventHandler.getClass().getCanonicalName() + " for event-class " + info.eventClass.getCanonicalName(), e);
229                                } catch (InvocationTargetException e) {
230                                        throw new RuntimeException("problem invoking EventHandler " + info.eventHandler.getClass().getCanonicalName() + " for event-class " + info.eventClass.getCanonicalName(), e.getCause());
231                                }
232                        }
233                }
234        }
235
236        private HandlerInfo[] getHandlersForClass(final Class<?> eventClass) {
237                Class<?> klass = eventClass;
238                HandlerInfo[] cache = this.cacheHandlers.get(eventClass);
239                if (cache != null) {
240                        return cache;
241                }
242
243                ArrayList<HandlerInfo> info = new ArrayList<HandlerInfo>();
244                // first search in class-hierarchy
245                while (klass != Object.class) {
246                        HandlerData dat = findHandler(klass);
247                        if (dat != null) {
248                                for(EventHandler handler: dat.handlerList) {
249                                        info.add(new HandlerInfo(klass, handler, dat.method));
250                                }
251                        }
252                        klass = klass.getSuperclass();
253                }
254                // now search in implemented interfaces
255                for (Class<?> intfc : getAllInterfaces(eventClass)) {
256                        HandlerData dat = findHandler(intfc);
257                        if (dat != null) {
258                                for(EventHandler handler: dat.handlerList) {
259                                        info.add(new HandlerInfo(intfc, handler, dat.method));
260                                }
261                        }
262                }
263
264                cache = info.toArray(new HandlerInfo[info.size()]);
265                this.cacheHandlers.put(eventClass, cache);
266                return cache;
267        }
268
269        private Set<Class<?>> getAllInterfaces(final Class<?> klass) {
270                Set<Class<?>> intfs = new HashSet<Class<?>>();
271                for (Class<?> intf : klass.getInterfaces()) {
272                        intfs.add(intf);
273                        intfs.addAll(getAllInterfaces(intf));
274                }
275                if (!klass.isInterface()) {
276                        Class<?> superclass = klass.getSuperclass();
277                        while (superclass != Object.class) {
278                                intfs.addAll(getAllInterfaces(superclass));
279                                superclass = superclass.getSuperclass();
280                        }
281                }
282                return intfs;
283        }

285        // this method is purely for performance reasons and need not be implemented
286        private boolean callHandlerFast(final Class<?> klass, final Event ev, final EventHandler handler) {
287                if (klass == LinkLeaveEvent.class) {
288                        ((LinkLeaveEventHandler)handler).handleEvent((LinkLeaveEvent)ev);
289                        return true;
290                } else if (klass == LinkEnterEvent.class) {
291                        ((LinkEnterEventHandler)handler).handleEvent((LinkEnterEvent)ev);
292                        return true;
293                } else if (klass == VehicleEntersTrafficEvent.class) {
294                        ((VehicleEntersTrafficEventHandler)handler).handleEvent((VehicleEntersTrafficEvent)ev);
295                        return true;
296                } else if (klass == PersonArrivalEvent.class) {
297                        ((PersonArrivalEventHandler)handler).handleEvent((PersonArrivalEvent)ev);
298                        return true;
299                } else if (klass == PersonDepartureEvent.class) {
300                        ((PersonDepartureEventHandler)handler).handleEvent((PersonDepartureEvent)ev);
301                        return true;
302                } else if (klass == ActivityEndEvent.class) {
303                        ((ActivityEndEventHandler)handler).handleEvent((ActivityEndEvent)ev);
304                        return true;
305                } else if (klass == ActivityStartEvent.class) {
306                        ((ActivityStartEventHandler)handler).handleEvent((ActivityStartEvent)ev);
307                        return true;
308                } else if (klass == TransitDriverStartsEvent.class) {
309                        ((TransitDriverStartsEventHandler) handler).handleEvent((TransitDriverStartsEvent) ev);
310                        return true;
311                } else if (klass == PersonStuckEvent.class) {
312                        ((PersonStuckEventHandler)handler).handleEvent((PersonStuckEvent)ev);
313                        return true;
314                } else if (klass == PersonMoneyEvent.class) {
315                        ((PersonMoneyEventHandler)handler).handleEvent((PersonMoneyEvent)ev);
316                        return true;
317                } else if (klass == AgentWaitingForPtEvent.class) {
318                        ((AgentWaitingForPtEventHandler)handler).handleEvent((AgentWaitingForPtEvent)ev);
319                        return true;
320                } else if (klass == PersonEntersVehicleEvent.class) {
321                        ((PersonEntersVehicleEventHandler)handler).handleEvent((PersonEntersVehicleEvent)ev);
322                        return true;
323                } else if (klass == PersonLeavesVehicleEvent.class) {
324                        ((PersonLeavesVehicleEventHandler)handler).handleEvent((PersonLeavesVehicleEvent)ev);
325                        return true;
326                } else if (klass == VehicleDepartsAtFacilityEvent.class) {
327                        ((VehicleDepartsAtFacilityEventHandler) handler).handleEvent((VehicleDepartsAtFacilityEvent) ev);
328                        return true;
329                } else if (klass == VehicleArrivesAtFacilityEvent.class) {
330                        ((VehicleArrivesAtFacilityEventHandler) handler).handleEvent((VehicleArrivesAtFacilityEvent) ev);
331                        return true;
332                } else if (klass == Event.class) {
333                        ((BasicEventHandler)handler).handleEvent(ev);
334                        return true;
335                }
336                return false;
337        }

339        public void printEventHandlers() {
340                log.info("currently registered event-handlers:");
341                for (HandlerData handlerType : this.handlerData) {
342                        log.info("+ " + handlerType.eventklass.getName());
343                        for (EventHandler handler : handlerType.handlerList) {
344                                log.info("  - " + handler.getClass().getName());
345                        }
346                }
347        }
