052public class EventWriterTXT implements EventWriter, ActivityEndEventHandler, ActivityStartEventHandler, PersonArrivalEventHandler, 
053                PersonDepartureEventHandler, PersonStuckEventHandler, PersonMoneyEventHandler, 
054                VehicleEntersTrafficEventHandler, LinkEnterEventHandler, LinkLeaveEventHandler {
055        
056        /* Implement all the different event handlers by its own. Future event types will no longer be
057         * suitable to be written to a TXT-format file, but will have additional attributes that need to be
058         * stored in XML. Explicitly listing the event handlers makes sure only events are written to TXT that
059         * can also correctly be read in again, and helps fix the test cases during introduction of the new,
060         * additional events.
061         */
062        
063        private BufferedWriter out = null;
064        private double lastTime = Double.NaN;
065        private String timeString = null;
066        
067        private Map<Id<Vehicle>, Id<Person>> vehicleToDriverMap = new HashMap<>();
068
069        public EventWriterTXT(final String filename) {
070                init(filename);
071        }
072
073        @Override
074        public void closeFile() {
075                if (this.out != null) {
076                        try {
077                                this.out.close();
078                                this.out = null;
079                        } catch (IOException e) {
080                                e.printStackTrace();
081                        }
082                }
083        }
084
085        public void init(final String outfilename) {
086                if (this.out != null) {
087                        try {
088                                this.out.close();
089                                this.out = null;
090                        } catch (IOException e) {
091                                e.printStackTrace();
092                        }
093                }
094                try {
095                        this.out = IOUtils.getBufferedWriter(outfilename);
096                        String eventsTxtFile = "T_GBL\t";
097                        eventsTxtFile +=  "VEH_ID\t";
098                        eventsTxtFile +=  "LEG_NR\t";
099                        eventsTxtFile +=  "LINK_ID\t";
100                        eventsTxtFile +=  "FROM_NODE_ID\t";
101                        eventsTxtFile +=  "EVENT_FLAG\t";
102                        eventsTxtFile +=  "DESCRIPTION\n";
103                        this.out.write(eventsTxtFile);
104                } catch (IOException e) {
105                        e.printStackTrace();
106                }
107        }
108
109        @Override
110        public void reset(final int iter) {
111                closeFile();
112        }
113
114        private void writeLine(final double time, final Id<Person> agentId, final Id<Link> linkId, final int flag, final String description) {
115                try {
116                        this.out.write(getTimeString(time));
117                        if (agentId != null) {
118                                this.out.write(agentId.toString());
119                        }
120                        this.out.write('\t');
121                        // nothing to be written for leg-nr
122                        this.out.write('\t');
123                        if (linkId != null) {
124                                this.out.write(linkId.toString());
125                        }
126                        this.out.write('\t');
127                        this.out.write('0'); // from-node-id
128                        this.out.write('\t');
129                        this.out.write(Integer.toString(flag));
130                        this.out.write('\t');
131                        if (description != null) {
132                                this.out.write(description);
133                        }
134                        this.out.write('\n');
135                } catch (IOException e) {
136                        e.printStackTrace();
137                }
138        }
139
140        /**
141         * Returns the passed time as Seconds, including a trailing tab-character.
142         * Internally caches the returned result to speed up writing many events with the same time.
143         * This may no longer be useful if times are switched to double.
144         *
145         * @param time
146         * @return
147         */
148        private String getTimeString(final double time) {
149                if (time != this.lastTime) {
150                        this.lastTime = time;
151                        this.timeString = Long.toString((long) this.lastTime) + "\t";
152                }
153                return this.timeString;
154        }
155
156        @Override
157        public void handleEvent(ActivityEndEvent event) {
158                writeLine(event.getTime(), event.getPersonId(), event.getLinkId(), Number.ActivityEnd.ordinal(), ActivityEndEvent.EVENT_TYPE + " " + event.getActType());
159        }
160
161        @Override
162        public void handleEvent(ActivityStartEvent event) {
163                writeLine(event.getTime(), event.getPersonId(), event.getLinkId(), Number.ActivityStart.ordinal(), ActivityStartEvent.EVENT_TYPE + " " + event.getActType());
164        }
165
166        @Override
167        public void handleEvent(PersonArrivalEvent event) {
168                writeLine(event.getTime(), event.getPersonId(), event.getLinkId(), Number.PersonArrival.ordinal(), PersonArrivalEvent.EVENT_TYPE);
169        }
170
171        @Override
172        public void handleEvent(PersonDepartureEvent event) {
173                writeLine(event.getTime(), event.getPersonId(), event.getLinkId(), Number.PersonDeparture.ordinal(), PersonDepartureEvent.EVENT_TYPE);
174        }
175        
176        public enum Number{ /*0:*/ PersonArrival, dummy, LinkLeave, PersonStuck, /*4:*/ VehicleEntersTraffic,
177                /*5:*/ LinkEnter, PersonDeparture, ActivityStart, ActivityEnd, /*9:*/ PersonMoney } ;
178        
179        @Override
180        public void handleEvent(PersonStuckEvent event) {
181                writeLine(event.getTime(), event.getPersonId(), event.getLinkId(), Number.PersonStuck.ordinal(), PersonStuckEvent.EVENT_TYPE);
182        }
183
184        @Override
185        public void handleEvent(PersonMoneyEvent event) {
186                writeLine(event.getTime(), event.getPersonId(), null, Number.PersonMoney.ordinal(), "agentMoney" + "\t" + event.getAmount());
187        }
188
189        @Override
190        public void handleEvent(VehicleEntersTrafficEvent event) {
191                writeLine(event.getTime(), event.getPersonId(), event.getLinkId(), Number.VehicleEntersTraffic.ordinal(), VehicleEntersTrafficEvent.EVENT_TYPE);
192                
193                vehicleToDriverMap.put(event.getVehicleId(), event.getPersonId());
194        }
195
196        @Override
197        public void handleEvent(LinkEnterEvent event) {
198                writeLine(event.getTime(), vehicleToDriverMap.get(event.getVehicleId()), event.getLinkId(), Number.LinkEnter.ordinal(), LinkEnterEvent.EVENT_TYPE);
199        }
200
201        @Override
202        public void handleEvent(LinkLeaveEvent event) {
203                writeLine(event.getTime(), vehicleToDriverMap.get(event.getVehicleId()), event.getLinkId(), Number.LinkLeave.ordinal(), LinkLeaveEvent.EVENT_TYPE);
204        }
