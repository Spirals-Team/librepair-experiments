021package org.matsim.core.events;
022
023import java.io.InputStream;
024import java.util.Locale;
025import java.util.Stack;
026
027import org.apache.log4j.Logger;
028import org.matsim.core.api.experimental.events.EventsManager;
029import org.matsim.core.api.internal.MatsimReader;
030import org.matsim.core.utils.io.MatsimXmlParser;
031import org.xml.sax.Attributes;
032import org.xml.sax.SAXException;
033
034/**
035 * A reader for events-files of MATSim. This reader recognizes the format of the events-file and uses
036 * the correct reader for the specific events-version, without manual setting.
037 *
038 * @author mrieser
039 */
040public final class MatsimEventsReader implements MatsimReader {
041
042        private final static Logger log = Logger.getLogger(MatsimEventsReader.class);
043        private final EventsManager events;
044
045        /**
046         * Creates a new reader for MATSim events files.
047         *
048         * @param events The Events-object that handles the events.
049         */
050        public MatsimEventsReader(final EventsManager events) {
051                this.events = events;
052        }
053
054        /**
055         * Parses the specified events file.
056         *
057         * @param filename The name of the file to parse.
058         */
059        @Override
060        public void readFile(final String filename) {
061                String lcFilename = filename.toLowerCase(Locale.ROOT);
062                if (lcFilename.endsWith(".xml") || lcFilename.endsWith(".xml.gz")) {
063                        new XmlEventsReader(this.events).readFile(filename);
064                } else if (lcFilename.endsWith(".txt") || lcFilename.endsWith(".txt.gz")) {
065                        throw new RuntimeException("text events are no longer supported. Please use MATSim 0.6.1 or earlier to read text events.");
066                } else {
067                        throw new IllegalArgumentException("Cannot recognize the format of the events-file " + filename);
068                }
069        }
070
071        public void readStream(final InputStream stream) {
072                new XmlEventsReader(this.events).parse(stream);
073        }
074
075        private static class XmlEventsReader extends MatsimXmlParser {
076
077                final EventsManager events;
078                private final static String EVENTS_V1 = "events_v1.dtd";
079                private MatsimXmlParser delegate = null;
080
081                public XmlEventsReader(final EventsManager events) {
082                        this.events = events;
083                        this.setValidating(false); // events-files have no DTD, thus they cannot validate
084                        setDoctype("events_v1.dtd"); // manually set a doctype, otherwise delegate would not be initialized
085                }
086
087                @Override
088                public void startTag(final String name, final Attributes atts, final Stack<String> context) {
089                        this.delegate.startTag(name, atts, context);
090                }
091
092                @Override
093                public void characters(char[] ch, int start, int length) throws SAXException {
094                        this.delegate.characters(ch, start, length);
095                }
096
097                @Override
098                public void endTag(final String name, final String content, final Stack<String> context) {
099                        this.delegate.endTag(name, content, context);
100                }
101
102                @Override
103                protected void setDoctype(final String doctype) {
104                        super.setDoctype(doctype);
105                        // Currently the only events-type is v1
106                        if (EVENTS_V1.equals(doctype)) {
107                                this.delegate = new EventsReaderXMLv1(this.events);
108                                log.info("using events_v1-reader.");
109                        } else {
110                                throw new IllegalArgumentException("Doctype \"" + doctype + "\" not known.");
111                        }
112                }
113        }
114}
