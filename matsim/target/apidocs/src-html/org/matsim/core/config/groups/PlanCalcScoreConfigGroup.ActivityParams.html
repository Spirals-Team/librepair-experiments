159        @Override
160        public void addParam(final String key, final String value) {
161                if (key.startsWith("monetaryDistanceCostRate")) {
162                        throw new RuntimeException("Please use monetaryDistanceRate (without `cost').  Even better, use config v2, "
163                                        + "mode-parameters (see output of any recent run), and mode-specific monetary " + "distance rate.");
164                } else if (WAITING_PT.equals(key)) {
165                        setMarginalUtlOfWaitingPt_utils_hr(Double.parseDouble(value));
166                }
167
168                // backward compatibility: underscored
169                else if (key.startsWith("activityType_")) {
170                        ActivityParams actParams = getActivityTypeByNumber(key.substring("activityType_".length()));
171
172                        actParams.setActivityType(value);
173                        getScoringParameters(null).removeParameterSet(actParams);
174                        addActivityParams(actParams);
175                } else if (key.startsWith("activityPriority_")) {
176                        ActivityParams actParams = getActivityTypeByNumber(key.substring("activityPriority_".length()));
177                        actParams.setPriority(Double.parseDouble(value));
178                } else if (key.startsWith("activityTypicalDuration_")) {
179                        ActivityParams actParams = getActivityTypeByNumber(key.substring("activityTypicalDuration_".length()));
180                        actParams.setTypicalDuration(Time.parseTime(value));
181                } else if (key.startsWith("activityMinimalDuration_")) {
182                        ActivityParams actParams = getActivityTypeByNumber(key.substring("activityMinimalDuration_".length()));
183                        actParams.setMinimalDuration(Time.parseTime(value));
184                } else if (key.startsWith("activityOpeningTime_")) {
185                        ActivityParams actParams = getActivityTypeByNumber(key.substring("activityOpeningTime_".length()));
186                        actParams.setOpeningTime(Time.parseTime(value));
187                } else if (key.startsWith("activityLatestStartTime_")) {
188                        ActivityParams actParams = getActivityTypeByNumber(key.substring("activityLatestStartTime_".length()));
189                        actParams.setLatestStartTime(Time.parseTime(value));
190                } else if (key.startsWith("activityEarliestEndTime_")) {
191                        ActivityParams actParams = getActivityTypeByNumber(key.substring("activityEarliestEndTime_".length()));
192                        actParams.setEarliestEndTime(Time.parseTime(value));
193                } else if (key.startsWith("activityClosingTime_")) {
194                        ActivityParams actParams = getActivityTypeByNumber(key.substring("activityClosingTime_".length()));
195                        actParams.setClosingTime(Time.parseTime(value));
196                } else if (key.startsWith("scoringThisActivityAtAll_")) {
197                        ActivityParams actParams = getActivityTypeByNumber(key.substring("scoringThisActivityAtAll_".length()));
198                        actParams.setScoringThisActivityAtAll(Boolean.parseBoolean(value));
199                } else if (key.startsWith("traveling_")) {
200                        ModeParams modeParams = getOrCreateModeParams(key.substring("traveling_".length()));
201                        modeParams.setMarginalUtilityOfTraveling(Double.parseDouble(value));
202                } else if (key.startsWith("marginalUtlOfDistance_")) {
203                        ModeParams modeParams = getOrCreateModeParams(key.substring("marginalUtlOfDistance_".length()));
204                        modeParams.setMarginalUtilityOfDistance(Double.parseDouble(value));
205                } else if (key.startsWith("monetaryDistanceRate_")) {
206                        ModeParams modeParams = getOrCreateModeParams(key.substring("monetaryDistanceRate_".length()));
207                        modeParams.setMonetaryDistanceRate(Double.parseDouble(value));
208                } else if ("monetaryDistanceRateCar".equals(key)) {
209                        ModeParams modeParams = getOrCreateModeParams(TransportMode.car);
210                        modeParams.setMonetaryDistanceRate(Double.parseDouble(value));
211                } else if ("monetaryDistanceRatePt".equals(key)) {
212                        ModeParams modeParams = getOrCreateModeParams(TransportMode.pt);
213                        modeParams.setMonetaryDistanceRate(Double.parseDouble(value));
214                } else if (key.startsWith("constant_")) {
215                        ModeParams modeParams = getOrCreateModeParams(key.substring("constant_".length()));
216                        modeParams.setConstant(Double.parseDouble(value));
217                }
218
219                // backward compatibility: "typed" traveling
220                else if ("traveling".equals(key)) {
221                        this.getModes().get(TransportMode.car).setMarginalUtilityOfTraveling(Double.parseDouble(value));
222                } else if ("travelingPt".equals(key)) {
223                        this.getModes().get(TransportMode.pt).setMarginalUtilityOfTraveling(Double.parseDouble(value));
224                } else if ("travelingWalk".equals(key)) {
225                        this.getModes().get(TransportMode.walk).setMarginalUtilityOfTraveling(Double.parseDouble(value));
226                } else if ("travelingOther".equals(key)) {
227                        this.getModes().get(TransportMode.other).setMarginalUtilityOfTraveling(Double.parseDouble(value));
228                } else if ("travelingBike".equals(key)) {
229                        this.getModes().get(TransportMode.bike).setMarginalUtilityOfTraveling(Double.parseDouble(value));
230                }
231
232                // backward compatibility: "typed" util of distance
233                else if ("marginalUtlOfDistanceCar".equals(key)) {
234                        this.getModes().get(TransportMode.car).setMarginalUtilityOfDistance(Double.parseDouble(value));
235                } else if ("marginalUtlOfDistancePt".equals(key)) {
236                        this.getModes().get(TransportMode.pt).setMarginalUtilityOfDistance(Double.parseDouble(value));
237                } else if ("marginalUtlOfDistanceWalk".equals(key)) {
238                        this.getModes().get(TransportMode.walk).setMarginalUtilityOfDistance(Double.parseDouble(value));
239                } else if ("marginalUtlOfDistanceOther".equals(key)) {
240                        this.getModes().get(TransportMode.other).setMarginalUtilityOfDistance(Double.parseDouble(value));
241                }
242
243                // backward compatibility: "typed" constants
244                else if ("constantCar".equals(key)) {
245                        getModes().get(TransportMode.car).setConstant(Double.parseDouble(value));
246                } else if ("constantWalk".equals(key)) {
247                        getModes().get(TransportMode.walk).setConstant(Double.parseDouble(value));
248                } else if ("constantOther".equals(key)) {
249                        getModes().get(TransportMode.other).setConstant(Double.parseDouble(value));
250                } else if ("constantPt".equals(key)) {
251                        getModes().get(TransportMode.pt).setConstant(Double.parseDouble(value));
252                } else if ("constantBike".equals(key)) {
253                        getModes().get(TransportMode.bike).setConstant(Double.parseDouble(value));
254                }
255
256                // old-fashioned scoring parameters: default subpopulation
257                else if (Arrays
258                                .asList(LATE_ARRIVAL, EARLY_DEPARTURE, PERFORMING, MARGINAL_UTL_OF_MONEY, UTL_OF_LINE_SWITCH, WAITING)
259                                .contains(key)) {
260                        getScoringParameters(null).addParam(key, value);
261                }
262
263                else {
264                        delegate.addParam(key, value);
265                }
266        }

459        public void addScoringParameters(final ScoringParameterSet params) {
460                final ScoringParameterSet previous = this.getScoringParameters(params.getSubpopulation());
461
462                if (previous != null) {
463                        log.info("scoring parameters for subpopulation " + previous.getSubpopulation() + " were just replaced.");
464
465                        final boolean removed = removeParameterSet(previous);
466                        if (!removed)
467                                throw new RuntimeException("problem replacing scoring params ");
468                }
469
470                super.addParameterSet(params);
471        }

500        @Override
501        protected void checkParameterSet(final ConfigGroup module) {
502                switch (module.getName()) {
503                case ScoringParameterSet.SET_TYPE:
504                        if (!(module instanceof ScoringParameterSet)) {
505                                throw new RuntimeException("wrong class for " + module);
506                        }
507                        final String s = ((ScoringParameterSet) module).getSubpopulation();
508                        if (getScoringParameters(s) != null) {
509                                throw new IllegalStateException("already a parameter set for subpopulation " + s);
510                        }
511                        break;
512                default:
513                        throw new IllegalArgumentException(module.getName());
514                }
515        }

517        @Override
518        protected final void checkConsistency(final Config config) {
519                super.checkConsistency(config);
520                if (getScoringParametersPerSubpopulation().size()>1){
521                        if (!getScoringParametersPerSubpopulation().containsKey(PlanCalcScoreConfigGroup.DEFAULT_SUBPOPULATION)){
522                                throw new RuntimeException("Using several subpopulations in "+PlanCalcScoreConfigGroup.GROUP_NAME+" requires defining a \""+PlanCalcScoreConfigGroup.DEFAULT_SUBPOPULATION+" \" subpopulation."
523                                                + " Otherwise, crashes can be expected.");
524                        }
525                }
526                if (config.plansCalcRoute().isInsertingAccessEgressWalk()) {
527                        // adding the interaction activities that result from access/egress
528                        // routing. this is strictly speaking
529                        // not a consistency check, but I don't know a better place where to
530                        // add this. kai, jan'18
531                        for (ScoringParameterSet scoringParameterSet : this.getScoringParametersPerSubpopulation().values()) {
532                                for (String mode : config.plansCalcRoute().getNetworkModes()) {
533                                        String interactionActivityType = mode + " interaction";
534                                        ActivityParams set = scoringParameterSet.getActivityParamsPerType().get(interactionActivityType);
535                                        if (set == null) {
536                                                ActivityParams params = new ActivityParams();
537                                                params.setActivityType(interactionActivityType);
538                                                params.setScoringThisActivityAtAll(false);
539                                                scoringParameterSet.addActivityParams(params);
540                                        }
541                                }
542
543                        }
544                }
545
546                for (ActivityParams params : this.getActivityParams()) {
547                        if (params.isScoringThisActivityAtAll() && Time.isUndefinedTime(params.getTypicalDuration())) {
548                                throw new RuntimeException("In activity type=" + params.getActivityType()
549                                                + ", the typical duration is undefined.  This will lead to errors that are difficult to debug, "
550                                                + "so rather aborting here.");
551                        }
552                }
553
554        }

1336                /**
1337                 * Checks whether all the settings make sense or if there are some
1338                 * problems with the parameters currently set. Currently, this checks
1339                 * that for at least one activity type opening AND closing times are
1340                 * defined.
1341                 */
1342                @Override
1343                public void checkConsistency(Config config) {
1344                        super.checkConsistency(config);
1345                        
1346                        
1347                        boolean hasOpeningAndClosingTime = false;
1348                        boolean hasOpeningTimeAndLatePenalty = false;
1349
1350        
1351                        // This cannot be done in ActivityParams (where it would make more
1352                        // sense),
1353                        // because some global properties are also checked
1354                        for (ActivityParams actType : this.getActivityParams()) {
1355                                if (actType.isScoringThisActivityAtAll()) {
1356                                        // (checking consistency only if activity is scored at all)
1357
1358                                        if ((actType.getOpeningTime() != Time.UNDEFINED_TIME)
1359                                                        && (actType.getClosingTime() != Time.UNDEFINED_TIME)) {
1360                                                hasOpeningAndClosingTime = true;
1361                                        }
1362                                        if ((actType.getOpeningTime() != Time.UNDEFINED_TIME) && (getLateArrival_utils_hr() < -0.001)) {
1363                                                hasOpeningTimeAndLatePenalty = true;
1364                                        }
1365                                        if (actType.getOpeningTime() == 0. && actType.getClosingTime() > 24. * 3600 - 1) {
1366                                                log.error("it looks like you have an activity type with opening time set to 0:00 and closing "
1367                                                                + "time set to 24:00. This is most probably not the same as not setting them at all.  "
1368                                                                + "In particular, activities which extend past midnight may not accumulate scores.");
1369                                        }
1370                                }
1371                        }
1372                        if (!hasOpeningAndClosingTime && !hasOpeningTimeAndLatePenalty) {
1373                                log.info("NO OPENING OR CLOSING TIMES DEFINED!\n\n"
1374                                                + "There is no activity type that has an opening *and* closing time (or opening time and late penalty) defined.\n"
1375                                                + "This usually means that the activity chains can be shifted by an arbitrary\n"
1376                                                + "number of hours without having an effect on the score of the plans, and thus\n"
1377                                                + "resulting in wrong results / traffic patterns.\n"
1378                                                + "If you are using MATSim without time adaptation, you can ignore this warning.\n\n");
1379                        }
1380                        if (this.getMarginalUtlOfWaiting_utils_hr() != 0.0) {
1381                                log.warn("marginal utl of wait set to: " + this.getMarginalUtlOfWaiting_utils_hr()
1382                                                + ". Setting this different from zero is "
1383                                                + "discouraged since there is already the marginal utility of time as a resource. The parameter was also used "
1384                                                + "in the past for pt routing; if you did that, consider setting the new "
1385                                                + "parameter waitingPt instead.");
1386                        }
1387                }
