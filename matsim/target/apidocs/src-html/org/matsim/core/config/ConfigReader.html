034/**
035 * A reader for config-files of MATSim. This reader recognizes the format of the config-file and uses
036 * the correct reader for the specific config-version, without manual setting.
037 *
038 * @author mrieser
039 */
040public final class ConfigReader extends MatsimXmlParser {
041
042        private final static Logger log = Logger.getLogger(ConfigReader.class);
043
044        private final static String CONFIG_V1 = "config_v1.dtd";
045        private final static String CONFIG_V2 = "config_v2.dtd";
046
047        private final Config config;
048        private MatsimXmlParser delegate = null;
049
050        private String localDtd;
051
052        /**
053         * Creates a new reader for MATSim configuration files.
054         *
055         * @param config The Config-object to store the configuration settings in. 
056         * <p></p>
057         * As far as I can tell, pre-existing settings in config are preserved
058         * except if they are explicitly overwritten by entries in the config file.  The MATSim4UrbanSim initialization sequence depends on this
059         * behavior.  kn, mar'13 
060         */
061        public ConfigReader(final Config config) {
062                this.config = config;
063        }
064
065        @Override
066        public void startTag(final String name, final Attributes atts, final Stack<String> context) {
067                this.delegate.startTag(name, atts, context);
068        }
069
070        @Override
071        public void endTag(final String name, final String content, final Stack<String> context) {
072                this.delegate.endTag(name, content, context);
073        }
074
075        /**
076         * Parses the specified config file, and uses the given dtd file as a local copy to use as dtd
077         * if the one specified in the config file cannot be found.
078         *
079         * @param filename The name of the file to parse.
080         * @param dtdFilename The name of a (local) dtd-file to be used for validating the config file.
081         * @throws UncheckedIOException e.g. if the file cannot be found
082         */
083        public void readFile(final String filename, final String dtdFilename) throws UncheckedIOException {
084                // yyyyyy if this is a necessary/useful method, I would prefer it in the superclass.  kai, jul'16
085                
086                log.info("trying to read config from " + filename);
087          this.localDtd = dtdFilename;
088                readFile(filename);
089                this.localDtd = null;
090        }
091
092        @Override
093        protected void setDoctype(final String doctype) {
094                super.setDoctype(doctype);
095                // Currently the only config-type is v1
096                if (CONFIG_V1.equals(doctype)) {
097                        this.delegate = new ConfigReaderMatsimV1(this.config);
098                        log.info("using config_v1-reader.");
099                }
100                else if ( CONFIG_V2.equals( doctype ) ) {
101                        this.delegate = new ConfigReaderMatsimV2( this.config );
102                        log.info( "using config_v2-reader" );
103                }
104                else {
105                        throw new IllegalArgumentException("Doctype \"" + doctype + "\" not known.");
106                }
107        }
