196        @Override
197        public String getValue(final String key) {
198                // first check if the parameter is in "underscored" form
199                if ( key.startsWith(MODULE)
200                                || key.startsWith(MODULE_PROBABILITY)
201                                || key.startsWith(MODULE_DISABLE_AFTER_ITERATION)
202                                || key.startsWith(MODULE_EXE_PATH)
203                                || key.startsWith(MODULE_SUBPOPULATION) ) {
204                        throw new IllegalArgumentException( "getting underscored parameter "+key+" is not allowed anymore. The supported way to get those parameters is via parameter sets." );
205                }
206
207                return delegate.getValue( key );
208        }
209
210        @Override
211        public void addParam(final String key, final String value) {
212                // adding underscore parameters is still supported for backward compatibility.
213                if (key != null && key.startsWith(MODULE)) {
214                        StrategySettings settings = getStrategySettings(Id.create(key.substring(MODULE.length()), StrategySettings.class), true);
215                        settings.addParam( "strategyName" , value);
216                }
217                else if (key != null && key.startsWith(MODULE_PROBABILITY)) {
218                        StrategySettings settings = getStrategySettings(Id.create(key.substring(MODULE_PROBABILITY.length()), StrategySettings.class), true);
219                        settings.addParam( "weight" , value );
220                }
221                else if (key != null && key.startsWith(MODULE_DISABLE_AFTER_ITERATION)) {
222                        StrategySettings settings = getStrategySettings(Id.create(key.substring(MODULE_DISABLE_AFTER_ITERATION.length()), StrategySettings.class), true);
223                        settings.setDisableAfter(Integer.parseInt(value));
224                        settings.addParam( "disableAfterIteration" , value );
225                }
226                else if (key != null && key.startsWith(MODULE_EXE_PATH)) {
227                        StrategySettings settings = getStrategySettings(Id.create(key.substring(MODULE_EXE_PATH.length()), StrategySettings.class), true);
228                        settings.addParam( "executionPath" , value );
229                }
230                else if (key != null && key.startsWith(MODULE_SUBPOPULATION)) {
231                        StrategySettings settings = getStrategySettings(Id.create(key.substring(MODULE_SUBPOPULATION.length()), StrategySettings.class), true);
232                        settings.addParam( "subpopulation" , value );
233                }
234                else {
235                        delegate.addParam( key , value );
236                }
237        }

239        private StrategySettings getStrategySettings(final Id<StrategySettings> index, final boolean createIfMissing) {
240                StrategySettings settings = null;
241
242                // should be in a map, but it is difficult to keep consistency with the
243                // delegate...
244                for ( StrategySettings s : getStrategySettings() ) {
245                        if ( !s.getId().equals( index ) ) continue;
246                        if ( settings != null ) throw new IllegalStateException( "several settings with id "+index );
247                        settings = s;
248                }
249
250                if (settings == null && createIfMissing) {
251                        settings = new StrategySettings(index);
252                        addStrategySettings( settings );
253                }
254
255                return settings;
256        }

283        @Override
284        protected void checkParameterSet(final ConfigGroup set) {
285                switch ( set.getName() ) {
286                        case StrategySettings.SET_NAME:
287                                if ( !(set instanceof StrategySettings) ) {
288                                        throw new RuntimeException( set+" is not an instance of StrategySettings" );
289                                }
290                                break;
291                        default:
292                                throw new IllegalArgumentException( "unknown set type "+set.getName() );
293                }
294        }
295
296        @Override
297        public ConfigGroup createParameterSet(final String type) {
298                switch ( type ) {
299                        case StrategySettings.SET_NAME:
300                                return new StrategySettings( );
301                        default:
302                                throw new IllegalArgumentException( "unknown set type "+type );
303                }
304        }
