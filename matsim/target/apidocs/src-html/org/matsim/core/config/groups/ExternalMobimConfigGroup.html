021package org.matsim.core.config.groups;
022
023import java.util.Map;
024
025import org.apache.log4j.Logger;
026import org.matsim.core.config.ReflectiveConfigGroup;
027import org.matsim.core.utils.misc.Time;
028
029public final class ExternalMobimConfigGroup extends ReflectiveConfigGroup {
030
031        @SuppressWarnings("unused")
032        private final static Logger log = Logger.getLogger(ExternalMobimConfigGroup.class);
033
034        public static final String GROUP_NAME = "externalMobsim";
035
036        private static final String START_TIME = "startTime";
037        private static final String END_TIME = "endTime";
038        private static final String EXTERNAL_EXE = "externalExe";
039        private static final String TIMEOUT = "timeout";
040
041        private double startTime = Time.UNDEFINED_TIME;
042        private double endTime = Time.UNDEFINED_TIME;
043        private String externalExe = null;
044        private int timeOut = 3600;
045
046        public ExternalMobimConfigGroup() {
047                super(GROUP_NAME);
048        }
049
050        
051        @Override
052        public final Map<String, String> getComments() {
053                Map<String,String> map = super.getComments();
054                return map ;
055        }
056
057
058        @StringSetter(START_TIME)
059        public void setStartTime(final String startTime) {
060                this.setStartTime( Time.parseTime(startTime) ) ;
061        }
062        public void setStartTime(final double startTime) {
063                this.startTime = startTime;
064        }
065
066        @StringGetter(START_TIME)
067        String getStartTimeAsString() {
068                return Time.writeTime(this.startTime) ;
069        }
070        public double getStartTime() {
071                return this.startTime;
072        }
073
074        @StringSetter(END_TIME)
075        public void setEndTime(final String startTime) {
076                this.setEndTime( Time.parseTime(startTime) );
077        }
078        public void setEndTime(final double endTime) {
079                this.endTime = endTime;
080        }
081
082        @StringGetter(END_TIME)
083        String getEndTimeAsString() {
084                return Time.writeTime(this.endTime ) ;
085        }
086        public double getEndTime() {
087                return this.endTime;
088        }
089
090        @StringSetter( EXTERNAL_EXE )
091        public void setExternalExe(final String externalExe) {
092                this.externalExe = externalExe;
093        }
094        @StringGetter( EXTERNAL_EXE )
095        public String getExternalExe() {
096                return this.externalExe;
097        }
098        @StringSetter( TIMEOUT )        
099        public void setExternalTimeOut(final int timeOut) {
100                this.timeOut = timeOut;
101        }
102        @StringGetter( TIMEOUT )        
103        public int getExternalTimeOut() {
104                return this.timeOut;
105        }
106
107}
