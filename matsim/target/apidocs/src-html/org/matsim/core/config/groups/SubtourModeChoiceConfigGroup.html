020package org.matsim.core.config.groups;
021
022import java.util.Map;
023
024import org.matsim.api.core.v01.TransportMode;
025import org.matsim.core.config.ReflectiveConfigGroup;
026import org.matsim.core.replanning.modules.SubtourModeChoice;
027import org.matsim.core.utils.misc.StringUtils;
028
029/**
030 * @author thibautd
031 */
032public final class SubtourModeChoiceConfigGroup extends ReflectiveConfigGroup {
033
034        public static final String GROUP_NAME = "subtourModeChoice";
035        
036        public final static String MODES = "modes";
037        public final static String CHAINBASEDMODES = "chainBasedModes";
038        public final static String CARAVAIL = "considerCarAvailability";
039        private static final String BEHAVIOR = "behavior";
040        
041        private String[] chainBasedModes = new String[] { TransportMode.car, TransportMode.bike };
042        private String[] allModes = new String[] { TransportMode.car, TransportMode.pt, TransportMode.bike, TransportMode.walk };
043        // default is false for backward compatibility
044        private boolean considerCarAvailability = false;
045        private SubtourModeChoice.Behavior behavior = SubtourModeChoice.Behavior.fromSpecifiedModesToSpecifiedModes ;
046        
047        private double probaForRandomSingleTripMode = 0. ; // yyyyyy backwards compatibility setting; should be change. kai, may'18
048        
049        public SubtourModeChoiceConfigGroup() {
050                super(GROUP_NAME);
051        }
052
053        @StringGetter( MODES )
054        private String getModesString() {
055                return toString( allModes );
056        }
057
058        @StringGetter( CHAINBASEDMODES )
059        private String getChainBaseModesString() {
060                return toString( chainBasedModes );
061        }
062
063        private static String toString( final String[] modes ) {
064                // (not same as toString() because of argument!)
065                
066                StringBuilder b = new StringBuilder();
067
068                if (modes.length > 0) b.append( modes[ 0 ] );
069                for (int i=1; i < modes.length; i++) {
070                        b.append( ',' );
071                        b.append( modes[ i ] );
072                }
073
074                return b.toString();
075        }
076
077        private static String[] toArray( final String modes ) {
078                String[] parts = StringUtils.explode(modes, ',');
079
080                for (int i = 0, n = parts.length; i < n; i++) {
081                        parts[i] = parts[i].trim().intern();
082                }
083
084                return parts;
085        }
086
087        @StringSetter( MODES )
088        private void setModes( final String value ) {
089                setModes( toArray( value ) );
090        }
091
092        @StringSetter( CHAINBASEDMODES )
093        private void setChainBasedModes( final String value ) {
094                setChainBasedModes( toArray( value ) );
095        }
096
097        @Override
098        public Map<String, String> getComments() {
099                Map<String, String> comments = super.getComments();
100                comments.put(MODES, "Defines all the modes available, including chain-based modes, seperated by commas" );
101                comments.put(CHAINBASEDMODES, "Defines the chain-based modes, seperated by commas" );
102                comments.put(CARAVAIL, "Defines whether car availability must be considered or not. A agent has no car only if it has no license, or never access to a car" );
103                {
104                        String msg = "Only for backwards compatibility.  Defines if only trips from modes list should change mode, or all trips.  Options: " ;
105                        for ( SubtourModeChoice.Behavior behavior : SubtourModeChoice.Behavior.values() ) {
106                                msg += behavior.name() + " " ;
107                        }
108                        comments.put(BEHAVIOR, msg);
109                }
110                return comments;
111        }
112
113        /* direct access */
114
115        public void setModes( final String[] modes ) {
116                this.allModes = modes;
117        }
118
119        public String[] getModes() {
120                return this.allModes;
121        }
122
123        public void setChainBasedModes( final String[] modes ) {
124                this.chainBasedModes = modes;
125        }
126
127        public String[] getChainBasedModes() {
128                return this.chainBasedModes;
129        }
130
131        @StringSetter( CARAVAIL )
132        public void setConsiderCarAvailability(final boolean value) {
133                this.considerCarAvailability = value;
134        }
135
136        @StringGetter( CARAVAIL )
137        public boolean considerCarAvailability() {
138                return considerCarAvailability;
139        }
140        @StringSetter( BEHAVIOR )
141        public final void setBehavior( SubtourModeChoice.Behavior behavior ) {
142                this.behavior = behavior ;
143        }
144        @StringGetter( BEHAVIOR )
145        public final SubtourModeChoice.Behavior getBehavior() {
146                return this.behavior ;
147        }
148        
149        public double getProbaForRandomSingleTripMode() {
150                return this.probaForRandomSingleTripMode;
151        }
152        public void setProbaForRandomSingleTripMode(double probaForRandomSingleTripMode) {
153                this.probaForRandomSingleTripMode = probaForRandomSingleTripMode;
154        }
155        
156        
157}
