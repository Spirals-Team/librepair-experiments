020package org.matsim.core.config.groups;
021
022import java.util.Map;
023
024import org.matsim.api.core.v01.TransportMode;
025import org.matsim.core.config.ReflectiveConfigGroup;
026import org.matsim.core.utils.misc.StringUtils;
027
028/**
029 * @author thibautd
030 */
031public final class SubtourModeChoiceConfigGroup extends ReflectiveConfigGroup {
032
033        public static final String GROUP_NAME = "subtourModeChoice";
034        
035        public final static String MODES = "modes";
036        public final static String CHAINBASEDMODES = "chainBasedModes";
037        public final static String CARAVAIL = "considerCarAvailability";
038        
039        private String[] chainBasedModes = new String[] { TransportMode.car, TransportMode.bike };
040        private String[] allModes = new String[] { TransportMode.car, TransportMode.pt, TransportMode.bike, TransportMode.walk };
041        // default is false for backward compatibility
042        private boolean considerCarAvailability = false;
043
044        public SubtourModeChoiceConfigGroup() {
045                super(GROUP_NAME);
046        }
047
048        @StringGetter( MODES )
049        private String getModesString() {
050                return toString( allModes );
051        }
052
053        @StringGetter( CHAINBASEDMODES )
054        private String getChainBaseModesString() {
055                return toString( chainBasedModes );
056        }
057
058        private static String toString( final String[] modes ) {
059                // (not same as toString() because of argument!)
060                
061                StringBuilder b = new StringBuilder();
062
063                if (modes.length > 0) b.append( modes[ 0 ] );
064                for (int i=1; i < modes.length; i++) {
065                        b.append( ',' );
066                        b.append( modes[ i ] );
067                }
068
069                return b.toString();
070        }
071
072        private static String[] toArray( final String modes ) {
073                String[] parts = StringUtils.explode(modes, ',');
074
075                for (int i = 0, n = parts.length; i < n; i++) {
076                        parts[i] = parts[i].trim().intern();
077                }
078
079                return parts;
080        }
081
082        @StringSetter( MODES )
083        private void setModes( final String value ) {
084                setModes( toArray( value ) );
085        }
086
087        @StringSetter( CHAINBASEDMODES )
088        private void setChainBasedModes( final String value ) {
089                setChainBasedModes( toArray( value ) );
090        }
091
092        @Override
093        public Map<String, String> getComments() {
094                Map<String, String> comments = super.getComments();
095                comments.put(MODES, "Defines all the modes available, including chain-based modes, seperated by commas" );
096                comments.put(CHAINBASEDMODES, "Defines the chain-based modes, seperated by commas" );
097                comments.put(CARAVAIL, "Defines whether car availability must be considered or not. A agent has no car only if it has no license, or never access to a car" );
098                return comments;
099        }
100
101        /* direct access */
102
103        public void setModes( final String[] modes ) {
104                this.allModes = modes;
105        }
106
107        public String[] getModes() {
108                return this.allModes;
109        }
110
111        public void setChainBasedModes( final String[] modes ) {
112                this.chainBasedModes = modes;
113        }
114
115        public String[] getChainBasedModes() {
116                return this.chainBasedModes;
117        }
118
119        @StringSetter( CARAVAIL )
120        public void setConsiderCarAvailability(final boolean value) {
121                this.considerCarAvailability = value;
122        }
123
124        @StringGetter( CARAVAIL )
125        public boolean considerCarAvailability() {
126                return considerCarAvailability;
127        }
128}
