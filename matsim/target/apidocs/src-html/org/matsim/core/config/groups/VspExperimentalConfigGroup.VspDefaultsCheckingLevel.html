136                for ( VspDefaultsCheckingLevel option : VspDefaultsCheckingLevel.values() ) {
137                        options.append(option + " | ") ;
138                }
139                map.put( VSP_DEFAULTS_CHECKING_LEVEL,

141                "nothing, logfile infos, logfile warnings, or aborts.  Members of VSP should use `abort' or talk to kai.") ;
