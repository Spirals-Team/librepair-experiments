036/**
037 * A module using reflection for easy implementation of config groups.
038 * <br>
039 * <br>
040 * This class takes care of all the housekeeping tasks that you normally have
041 * to manually implement when extending {@link ConfigGroup} (usually forgetting half of it).
042 * <br>
043 * <br>
044 * For each field in the xml file, just implement a setter taking one of the following types as parameter:
045 * <ul>
046 * <li> a String,</li>
047 * <li> a primitive data type or a "primitive wrapper" type (such as {@link Double}
048 * or {@link Integer}),</li>
049 * <li> an enumeration type</li>
050 * </ul>
051 * and a getter returning an Object or primitive type which String representation
052 * must be written to the xml file, and annotate them with the {@link StringSetter}
053 * and {@link StringGetter} annotation types.
054 * <br>
055 * Those annotations take a mandatory String argument, which is the parameter name
056 * in the xml.
057 * <br>
058 * <br>
059 * In most of the cases (for Strings, primitive types and enums),
060 * annotating the actual setters and getters is fine.
061 * Sometimes (When the {@link Object#toString()} method returns the right String representation,
062 * for instance for {@link Id}s), one needs to implement a specific
063 * "String" setter, but can annotate the normal getter.
064 * Sometimes (for instance when handling collections) one needs specific conversion bothways.
065 * In those cases, just separate actual setters/getters and "string" setters/getters.
066 * <br>
067 * Note that there is no restriction on access modifiers for those methods: they
068 * can even be private, if for instance you do not want to confuse the users with dozens
069 * of methods of the type "<tt>getStringRepresentationOfParameterX()</tt>" or
070 * "<tt>setParameterXFromStringRepresentation( String v )</tt>".
071 * <br>
072 * A commented example can be found at {@link tutorial.programming.reflectiveConfigGroup.MyConfigGroup}.
073 * <br>
074 * <br>
075 * If something is wrong (missing setter or getter, wrong parameter or return type),
076 * an {@link InconsistentModuleException} will be thrown at construction.
077 *
078 * @author thibautd
079 */
080public abstract class ReflectiveConfigGroup extends ConfigGroup implements MatsimExtensionPoint {
081        private static final Logger log =
082                Logger.getLogger(ReflectiveConfigGroup.class);
083
084        private final boolean storeUnknownParameters;
085
086        private final Map<String, Method> setters;
087        private final Map<String, Method> stringGetters;
088
089        // /////////////////////////////////////////////////////////////////////////
090        // Construction
091        // /////////////////////////////////////////////////////////////////////////
092        /**
093         * Creates an instance which will crash if an unknown parameter name
094         * is given.
095         *
096         * @param name the name of the module in the config file.
097         */
098        public ReflectiveConfigGroup(final String name) {
099                this( name , false );
100        }
101
102        /**
103         * Creates an instance, giving the choice on whether unknown parameter names result
104         * in a crash or just in the parameter value being stored as a String.
105         *
106         * @param name the name of the module in the config file.
107         * @param storeUnknownParametersAsStrings if true, when no annotated getter
108         * or setter is found for a parameter name, the parameters are stored using
109         * the default {@link ConfigGroup} behavior. This is not that safe, so be careful.
110         */
111        public ReflectiveConfigGroup(
112                        final String name,
113                        final boolean storeUnknownParametersAsStrings) {
114                super(name);
115                this.storeUnknownParameters = storeUnknownParametersAsStrings;
116                setters = getSetters();
117                stringGetters = getStringGetters();
118
119                if ( !setters.keySet().equals( stringGetters.keySet() ) ) {
120                        throw new InconsistentModuleException( "setters and getters inconsistent" );
121                }
122
123                checkConvertNullAnnotations();
124        }
125
126        private void checkConvertNullAnnotations() {
127                final Class<? extends ReflectiveConfigGroup> c = getClass();
128
129                final Method[] allMethods = c.getDeclaredMethods();
130
131                for (Method m : allMethods) {
132                        final StringGetter annotation = m.getAnnotation( StringGetter.class );
133                        if ( annotation != null ) {
134                                final Method g = getStringGetters().get( annotation.value() );
135
136                                if ( m.isAnnotationPresent( DoNotConvertNull.class ) != g.isAnnotationPresent( DoNotConvertNull.class ) ) {
137                                        throw new InconsistentModuleException( "Inconsistent annotation of getter and setter with ConvertNull in "+getClass().getName() );
138                                }
139                        }
140                }
141        }
142
143        private Map<String, Method> getStringGetters() {
144                final Map<String, Method> gs = new HashMap<String, Method>();
145                final Class<? extends ReflectiveConfigGroup> c = getClass();
146
147                final Method[] allMethods = c.getDeclaredMethods();
148
149                for (Method m : allMethods) {
150                        final StringGetter annotation = m.getAnnotation( StringGetter.class );
151                        if ( annotation != null ) {
152                                checkGetterValidity( m );
153                                final Method old = gs.put( annotation.value() , m );
154                                if ( old != null ) {
155                                        throw new InconsistentModuleException( "several string getters for "+annotation.value() );
156                                }
157                        }
158                }
159
160                return gs;
161        }
162
163        private static void checkGetterValidity(final Method m) {
164                if ( m.getParameterTypes().length > 0 ) {
165                        throw new InconsistentModuleException( "getter "+m+" has parameters" );
166                }
167
168                if ( m.getReturnType().equals( Void.TYPE ) ) {
169                        throw new InconsistentModuleException( "getter "+m+" has void return type" );
170                }
171        }
172
173        private Map<String, Method> getSetters() {
174                final Map<String, Method> ss = new HashMap<String, Method>();
175                final Class<? extends ReflectiveConfigGroup> c = getClass();
176
177                final Method[] allMethods = c.getDeclaredMethods();
178
179                for (Method m : allMethods) {
180                        final StringSetter annotation = m.getAnnotation( StringSetter.class );
181                        if ( annotation != null ) {
182                                checkSetterValidity( m );
183                                final Method old = ss.put( annotation.value() , m );
184                                if ( old != null ) {
185                                        throw new InconsistentModuleException( "several string setters for "+annotation.value() );
186                                }
187                        }
188                }
189
190                return ss;
191        }
192
193        private static void checkSetterValidity(final Method m) {
194                final Class<?>[] params = m.getParameterTypes();
195
196                if (params.length != 1) {
197                        throw new InconsistentModuleException( "setter "+m+" has "+params.length+" parameters instead of one" );
198                }
199
200                final Collection<Class<?>> allowedParameterTypes =
201                        Arrays.<Class<?>>asList(
202                                        String.class,
203                                        Float.class, Double.class, Integer.class, Long.class, Boolean.class, Character.class, Byte.class, Short.class,
204                                        Float.TYPE, Double.TYPE, Integer.TYPE, Long.TYPE, Boolean.TYPE, Character.TYPE, Byte.TYPE, Short.TYPE);
205        if ( !allowedParameterTypes.contains( params[ 0 ] ) && !params[ 0 ].isEnum() ) {
206                        throw new InconsistentModuleException( "setter "+m+" gets a "+params[ 0 ]+". Valid types are String, primitive types and their wrapper classes, and enumerations. "+
207                                        "Other types are fine as parameters, but you will need to implement conversion strategies in the String setters." );
208                }
209        }

214        @Override
215        public final void addParam(
216                        final String param_name,
217                        final String value) {
218                final Method setter = setters.get( param_name );
219
220                if (setter == null) {
221                        if ( !storeUnknownParameters ) {
222                                throw new IllegalArgumentException(
223                                                "Module "+getName()+" of type "+getClass().getName()+
224                                                " doesn't accept unkown parameters. Parameter "+param_name+
225                                                " is not part of the valid parameters: "+setters.keySet() );
226                        }
227                        log.warn( "unknown parameter "+param_name+" for group "+getName()+". Here are the valid parameter names: "+setters.keySet() );
228                        log.warn( "Only the string value will be remembered" );
229                        super.addParam( param_name , value );
230                        return;
231                }
232
233                try {
234                        invokeSetter( setter , value );
235                        log.trace( "value "+value+" successfully set for field "+param_name+" for group "+getName() );
236                }
237                catch (InvocationTargetException e) {
238                        // this exception wraps Throwables intercepted in the invocation of the setter.
239                        // Avoid multiple wrappings (exception wrapped in InvocationTargetException
240                        // itself wrapped in a RuntimeException), as it makes error messages
241                        // messy.
242                        final Throwable cause = e.getCause();
243                        if ( cause instanceof RuntimeException ) {
244                                throw (RuntimeException) cause;
245                        }
246
247                        if ( cause instanceof Error ) {
248                                throw (Error) cause;
249                        }
250
251                        throw new RuntimeException( cause );
252                }
253                catch (IllegalAccessException e) {
254                        throw new RuntimeException( e );
255                }
256        }
257
258        private void invokeSetter(
259                        final Method setter,
260                        final String value) throws IllegalAccessException, InvocationTargetException {
261                // do not care about access modifier:
262                // if a method is tagged with the StringSetter
263                // annotation, we are supposed to access it.
264                // This *is* safe.
265                final boolean accessible = setter.isAccessible();
266                setter.setAccessible( true );
267
268                final Class<?>[] params = setter.getParameterTypes();
269                assert params.length == 1; // already checked at constr.
270
271                final Class<?> type = params[ 0 ];
272
273                if ( value.equals( "null" ) && !setter.isAnnotationPresent( DoNotConvertNull.class ) ) {
274                        setter.invoke( this , new Object[]{ null } );
275                }
276                else if ( type.equals( String.class ) ) {
277                        setter.invoke( this , value );
278                }
279                else if ( type.equals( Float.class ) || type.equals( Float.TYPE ) ) {
280                        setter.invoke( this , Float.parseFloat( value ) );
281                }
282                else if ( type.equals( Double.class ) || type.equals( Double.TYPE ) ) {
283                        setter.invoke( this , Double.parseDouble( value ) );
284                }
285                else if ( type.equals( Integer.class ) || type.equals( Integer.TYPE ) ) {
286                        setter.invoke( this , Integer.parseInt( value ) );
287                }
288                else if ( type.equals( Long.class ) || type.equals( Long.TYPE ) ) {
289                        setter.invoke( this , Long.parseLong( value ) );
290                }
291                else if ( type.equals( Boolean.class ) || type.equals( Boolean.TYPE ) ) {
292                        setter.invoke( this , Boolean.parseBoolean( value ) );
293                }
294                else if ( type.equals( Character.class ) || type.equals( Character.TYPE ) ) {
295                        if ( value.length() != 1 ) throw new IllegalArgumentException( value+" is not a single char!" );
296                        setter.invoke( this , value.toCharArray()[ 0 ] );
297                }
298                else if ( type.equals( Byte.class ) || type.equals( Byte.TYPE ) ) {
299                        setter.invoke( this , Byte.parseByte( value ) );
300                }
301                else if ( type.equals( Short.class ) || type.equals( Short.TYPE ) ) {
302                        setter.invoke( this , Short.parseShort( value ) );
303                }
304                else if ( type.isEnum() ) {
305                        try {
306                                setter.invoke(
307                                                this,
308                                                Enum.valueOf(
309                                                        type.asSubclass( Enum.class ),
310                                                        value ) );
311                        }
312                        catch (IllegalArgumentException e) {
313                                // happens when the string does not correspond to any enum values.
314                                // Surprisingly, the default error message does not print the possible
315                                // values: do it here, so that the user gets an idea of what went wrong
316                                final StringBuilder comment =
317                                        new StringBuilder(
318                                                        "Error trying to set value "+value+
319                                                        " for type "+type.getName()+
320                                                        ": possible values are " );
321
322                                final Object[] consts = type.getEnumConstants();
323                                for ( int i = 0; i < consts.length; i++ ) {
324                                        comment.append( consts[ i ].toString() );
325                                        if ( i < consts.length - 1 ) comment.append( ", " );
326                                }
327
328                                throw new IllegalArgumentException( comment.toString() , e );
329                        }
330                }
331                else {
332                        throw new RuntimeException( "no method to handle type "+type );
333                }
334
335                setter.setAccessible( accessible );
336        }

338        @Override
339        public final String getValue(final String param_name) {
340                final Method getter = stringGetters.get( param_name );
341
342                try {
343                        if (getter != null) {
344                                // do not care about access modifier:
345                                // if a method is tagged with the StringGetter
346                                // annotation, we are supposed to access it.
347                                // This *is* safe.
348                                final boolean accessible = getter.isAccessible();
349                                getter.setAccessible( true );
350                                final Object result = getter.invoke( this );
351                                getter.setAccessible( accessible );
352
353                                if ( result == null ) {
354                                        if ( getter.isAnnotationPresent( DoNotConvertNull.class ) ) {
355                                                log.error( "getter for parameter "+param_name+" of module "+getName()+" returned null." );
356                                                log.error( "This is not allowed for this getter." );
357
358                                                throw new NullPointerException( "getter for parameter "+param_name+" of module "+getClass().getName()+" ("+getName()+") returned null." );
359                                        }
360                                        return null;
361                                }
362
363                                final String value = ""+result;
364
365                                if ( value.equals( "null" ) && !getter.isAnnotationPresent( DoNotConvertNull.class ) ) {
366                                        throw new RuntimeException( "parameter "+param_name+" understands null pointers for IO. As a consequence, the \"null\" String is not a valid value for "+getter.getName() );
367                                }
368
369                                return value;
370                        }
371                }
372                catch (InvocationTargetException e) {
373                        // this exception wraps Throwables intercepted in the invocation of the getter.
374                        // Avoid multiple wrappings (exception wrapped in InvocationTargetException
375                        // itself wrapped in a RuntimeException), as it makes error messages
376                        // messy.
377                        final Throwable cause = e.getCause();
378                        if ( cause instanceof RuntimeException ) {
379                                throw (RuntimeException) cause;
380                        }
381
382                        if ( cause instanceof Error ) {
383                                throw (Error) cause;
384                        }
385
386                        throw new RuntimeException( cause );
387                }
388                catch (IllegalAccessException e) {
389                        throw new RuntimeException( e );
390                }
391
392                if ( !storeUnknownParameters ) {
393                        throw new IllegalArgumentException(
394                                        "Module "+getName()+" of type "+getClass().getName()+
395                                        " doesn't store unkown parameters. Parameter "+param_name+
396                                        " is not part of the valid parameters: "+stringGetters.keySet() );
397                }
398
399                log.warn( "no getter found for param "+param_name+": trying parent method" );
400                return super.getValue( param_name );
401        }
402
403        @Override
404        public final Map<String, String> getParams() {
405                final Map<String, String> map = super.getParams();
406
407                for (String f : setters.keySet()) {
408                        addParameterToMap( map , f );
409                }
410
411                return map;
412        }

414        /**
415         * Comments for parameters which setter get an enum type are automatically generated,
416         * containing a list of possible values. They can be overriden by subclasses without
417         * problems.
418         *
419         * <br>
420         * it is recommended for subclasses to get this map using <tt>super.getComments()</tt>
421         * and fill it with additional comments, rather than generate an empty map.
422         */
423        @Override
424        public Map<String, String> getComments() {
425                // generate some default comments.
426                final Map<String, String> comments = super.getComments();
427
428                for ( Map.Entry<String, Method> entry : setters.entrySet() ) {
429                        final String paramName = entry.getKey();
430                        if ( comments.containsKey( paramName ) ) {
431                                // at the time of implementation, this is not possible,
432                                // but who knows? Do not override something already there.
433                                continue;
434                        }
435
436                        final Method setter = entry.getValue();
437
438                        final Class<?>[] params = setter.getParameterTypes();
439                        assert params.length == 1; // already checked at constr.
440
441                        final Class<?> type = params[ 0 ];
442
443                        if ( type.isEnum() ) {
444                                // generate an automatic comment containing the possible values for enum types
445                                final StringBuilder comment = new StringBuilder( "Possible values: " );
446                                final Object[] consts = type.getEnumConstants();
447
448                                for ( int i = 0; i < consts.length; i++ ) {
449                                        comment.append( consts[ i ].toString() );
450                                        if ( i < consts.length - 1 ) comment.append( ", " );
451                                }
452
453                                comments.put( paramName , comment.toString() );
454                        }
455                }
456
457                return comments;
458        }

463        /**
464         * use to annotate the methods which should be used to read the string
465         * values.
466         * See the class description for a description of the valid signature of the
467         * annotated method.
468         */
469        @Documented
470        @Retention( RetentionPolicy.RUNTIME )
471        public static @interface StringSetter {
472                /**
473                 * the name of the field in the XML document
474                 */
475                String value();
476        }
477
478        /**
479         * use to annotate the methods which should be used to get the string
480         * values.
481         * the methods must take no parameter and return an Object or primitive
482         * data type which string representations is the one which should appear in the
483         * xml.
484         */
485        @Documented
486        @Retention( RetentionPolicy.RUNTIME )
487        public static @interface StringGetter {
488                /**
489                 * the name of the field in the XML document
490                 */
491                String value();
492        }

494        /**
495         * Setters for which the "null" string should NOT be converted
496         * to the <tt>null</tt> pointer, and getter from which a <tt>null</tt>
497         * pointer should NOT be accepted and converted to the "null" string,
498         * should be annotated with this.
499         * <br>
500         * Note that both the setter and the getter for a given parameter,
501         * or none of them, must be annotated. If not, an {@link InconsistentModuleException}
502         * will be thrown.
503         */
504        @Documented
505        @Retention( RetentionPolicy.RUNTIME )
506        public static @interface DoNotConvertNull {}
507
508        public static class InconsistentModuleException extends RuntimeException {
509                private static final long serialVersionUID = 1L;
510
511                private InconsistentModuleException(final String msg) {
512                        super( msg );
513                }
514        }
