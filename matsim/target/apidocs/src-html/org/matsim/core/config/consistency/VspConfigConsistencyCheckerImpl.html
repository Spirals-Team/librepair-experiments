021package org.matsim.core.config.consistency;
022
023import org.apache.log4j.Level;
024import org.apache.log4j.Logger;
025import org.matsim.api.core.v01.TransportMode;
026import org.matsim.core.config.Config;
027import org.matsim.core.config.groups.ControlerConfigGroup.EventsFileFormat;
028import org.matsim.core.config.groups.PlanCalcScoreConfigGroup.ActivityParams;
029import org.matsim.core.config.groups.PlanCalcScoreConfigGroup.ModeParams;
030import org.matsim.core.config.groups.PlansConfigGroup;
031import org.matsim.core.config.groups.QSimConfigGroup.TrafficDynamics;
032import org.matsim.core.config.groups.QSimConfigGroup.VehiclesSource;
033import org.matsim.core.config.groups.StrategyConfigGroup.StrategySettings;
034import org.matsim.core.config.groups.VspExperimentalConfigGroup.VspDefaultsCheckingLevel;
035import org.matsim.core.replanning.strategies.DefaultPlanStrategiesModule.DefaultStrategy;
036import org.matsim.pt.PtConstants;
037
038import java.util.Collection;
039import java.util.Set;
040
041/**
042 * @author nagel
043 *
044 */
045public final class VspConfigConsistencyCheckerImpl implements ConfigConsistencyChecker {
046        private static Logger log = Logger.getLogger(VspConfigConsistencyCheckerImpl.class) ;
047        
048        public VspConfigConsistencyCheckerImpl() {
049                // empty.  only here to find out where it is called.
050        }
051
052        @Override
053        public void checkConsistency(Config config) {
054                Level lvl ;
055                
056                switch ( config.vspExperimental().getVspDefaultsCheckingLevel() ) {
057                case ignore:
058                        log.info( "NOT running vsp config consistency check because vsp defaults checking level is set to IGNORE"); 
059                        return ;
060                case info:
061                        lvl = Level.INFO ;
062                        break ;
063                case warn:
064                        lvl = Level.WARN ;
065                        break;
066                case abort:
067                        lvl = Level.WARN ;
068                        break;
069                default:
070                        throw new RuntimeException("not implemented");
071                }
072                log.info("running checkConsistency ...");
073                
074                boolean problem = false ; // ini
075                
076                // yy: sort the config groups alphabetically
077                
078                // === global:
079                
080                if ( config.global().isInsistingOnDeprecatedConfigVersion() ) {
081                        problem = true ;
082                        System.out.flush();
083                        log.log( lvl, "you are insisting on config v1.  vsp default is using v2." ) ;
084                }
085                
086                // === controler:
087                
088                Set<EventsFileFormat> formats = config.controler().getEventsFileFormats();
089                if ( !formats.contains(EventsFileFormat.xml) ) {
090                        problem = true ;
091                        System.out.flush() ;
092                        log.log( lvl, "did not find xml as one of the events file formats. vsp default is using xml events.");
093                }
094
095                // === location choice:
096                
097                boolean usingLocationChoice = false ;
098                if ( config.getModule("locationchoice")!=null ) {
099                        usingLocationChoice = true ;
100                }
101                
102                if ( usingLocationChoice ) {
103                        final String samplePercent = config.findParam("locationchoice", "destinationSamplePercent");
104                        if ( samplePercent!=null && !samplePercent.equals("100.") ) {
105                                problem = true ;
106                                System.out.flush() ;
107                                log.error("vsp will not accept location choice destination sample percent other than 100 until the corresponding warning in " +
108                                                "DestinationSampler is resolved.  kai, jan'13") ;
109                        }
110//                      if ( !config.locationchoice().getProbChoiceExponent().equals("1.") ) {
111//                              //                      problem = true ;
112//                              log.error("vsp will not accept location choice prob choice exponents other than 1 until the corresponding warning in " +
113//                              "ChoiceSet is resolved.  kai, jan'13") ;
114//                      }
115                        if ( !config.vspExperimental().isUsingOpportunityCostOfTimeForLocationChoice() ) {
116                                problem = true ;
117                                System.out.flush() ;
118                                log.error("vsp will not accept location choice without including opportunity cost of time into the approximation. kai,jan'13") ;
119                        }
120                }
121                
122                // === planCalcScore:
123                
124                // use beta_brain=1 // added as of nov'12
125                if ( config.planCalcScore().getBrainExpBeta() != 1. ) {
126                        problem = true ;
127                        System.out.flush() ;
128                        log.log( lvl, "You are using a brainExpBeta != 1; vsp default is 1.  (Different values may cause conceptual " +
129                                        "problems during paper writing.) This means you have to add the following lines to your config file: ") ;
130                        log.log( lvl, "<module name=\"planCalcScore\">");
131                        log.log( lvl, " <param name=\"BrainExpBeta\" value=\"1.0\" />");
132                        log.log( lvl, "</module>");
133                }
134                                
135                // added apr'15:
136                for ( ActivityParams params : config.planCalcScore().getActivityParams() ) {
137                        if ( PtConstants.TRANSIT_ACTIVITY_TYPE.equals( params.getActivityType() ) ) {
138                                // they have typicalDurationScoreComputation==relative, but are not scored anyways. benjamin/kai, nov'15
139                                continue ;
140                        }
141                        switch( params.getTypicalDurationScoreComputation() ) {
142                        case relative:
143                                break;
144                        case uniform:
145//                              problem = true ;
146                                log.log( lvl,  "found `typicalDurationScoreComputation == uniform' for activity type " + params.getActivityType() + "; vsp should use `relative'. ") ;
147                                break;
148                        default:
149                                throw new RuntimeException("unexpected setting; aborting ... ") ;
150                        }
151                }
152                for ( ModeParams params : config.planCalcScore().getModes().values() ) {
153                        if ( params.getMonetaryDistanceRate() > 0. ) {
154                                problem = true ;
155                                System.out.flush() ;
156                                log.error("found monetary distance rate for mode " + params.getMode() + " > 0.  You probably want a value < 0 here.\n" ) ;
157                        }
158                        if ( params.getMonetaryDistanceRate() < -0.01 ) {
159                                System.out.flush() ;
160                                log.error("found monetary distance rate for mode " + params.getMode() + " < -0.01.  -0.01 per meter means -10 per km.  You probably want to divide your value by 1000." ) ;
161                        }
162                }
163                
164                if ( config.planCalcScore().getModes().get(TransportMode.car).getMonetaryDistanceRate() > 0 ) {
165                        problem = true ;
166                }
167                final ModeParams modeParamsPt = config.planCalcScore().getModes().get(TransportMode.pt);
168                if ( modeParamsPt!=null && modeParamsPt.getMonetaryDistanceRate() > 0 ) {
169                        problem = true ;
170                        System.out.flush() ;
171                        log.error("found monetary distance cost rate pt > 0.  You probably want a value < 0 here.  " +
172                                        "This is a bug and may be changed eventually.  kai, jun'11") ;
173                }
174                if ( config.planCalcScore().getMarginalUtilityOfMoney() < 0. ) {
175                        problem = true ;
176                        System.out.flush() ;
177                        log.error("found marginal utility of money < 0.  You almost certainly want a value > 0 here. " ) ;
178                }
179                // added aug'13:
180                if ( config.planCalcScore().getMarginalUtlOfWaiting_utils_hr() != 0. ) {
181                        problem = true ;
182                        System.out.flush() ;
183                        log.error("found marginal utility of waiting != 0.  vsp default is setting this to 0. " ) ;
184                }
185                
186                // added oct'17:
187                if ( config.planCalcScore().getFractionOfIterationsToStartScoreMSA() == null ) {
188                        problem = true ;
189                        System.out.flush() ;
190                        log.log( lvl, "You are not setting fractionOfIterationsToStartScoreMSA; vsp default is to set this to something like 0.8.  " +
191                                        "This means you have to add the following lines to your config file: ") ;
192                        log.log( lvl, "<module name=\"planCalcScore\">");
193                        log.log( lvl, " <param name=\"fractionOfIterationsToStartScoreMSA\" value=\"0.8\" />");
194                        log.log( lvl, "</module>");
195                }
196
197                // === plans:
198                
199                // added before nov'12
200                if ( !config.plans().isRemovingUnneccessaryPlanAttributes() ) {
201                        problem = true ;
202                        System.out.flush() ;
203                        log.log( lvl, "You are not removing unnecessary plan attributes; vsp default is to do that.") ;
204                }
205                
206                PlansConfigGroup.ActivityDurationInterpretation actDurInterpr =  config.plans().getActivityDurationInterpretation()  ;
207                if ( actDurInterpr == PlansConfigGroup.ActivityDurationInterpretation.endTimeOnly ) {
208                        // added jan'13
209                        log.log( lvl, PlansConfigGroup.ActivityDurationInterpretation.endTimeOnly + " is deprecated. Use " + PlansConfigGroup.ActivityDurationInterpretation.tryEndTimeThenDuration + " instead.") ;
210                        problem = true;
211                        // added before nov'12
212                        if( config.transit().isUseTransit()) {
213                                problem = true;
214                                System.out.flush() ;
215                                log.error("You are using " + config.plans().getActivityDurationInterpretation() + " as activityDurationInterpretation in " +
216                                                "conjunction with the matsim transit module. This is not working at all as pt interaction activities never have an end time and " +
217                                                "thus will never end!");
218                        }
219                }
220                
221                // added jan'13
222                if ( actDurInterpr == PlansConfigGroup.ActivityDurationInterpretation.minOfDurationAndEndTime ) {
223                        problem = true ;
224                        System.out.flush() ;
225                        log.log( lvl, "You are using ActivityDurationInterpretation " + config.plans().getActivityDurationInterpretation() + " ; vsp default is to use " +
226                                        PlansConfigGroup.ActivityDurationInterpretation.tryEndTimeThenDuration + 
227                                        "This means you have to add the following lines into the vspExperimental section of your config file: ") ;
228                        log.log( lvl,  "   <param name=\"activityDurationInterpretation\" value=\"" + PlansConfigGroup.ActivityDurationInterpretation.tryEndTimeThenDuration + "\" />" ) ;
229                        log.log( lvl, "Please report if this causes odd results (this will simplify many code maintenance issues, but is unfortunately not well tested).") ;
230                }
231
232                // === plansCalcRoute:
233                
234                // added feb'16
235                if ( !config.plansCalcRoute().isInsertingAccessEgressWalk() ) {
236                        log.log( lvl, "found `plansCalcRoute.insertingAccessEgressWalk==false'; vsp should use `true' or talk to Kai. " ) ;
237                }
238                
239                // === qsim:
240                
241                // added jun'16
242                if ( config.qsim().getUsePersonIdForMissingVehicleId() ) {
243                        log.log( lvl, "found qsim.usePersonIdForMissingVehicleId==true; this is only for backwards compatibility and should rather be set to false") ;
244                }
245                
246                // added feb'16
247                if ( !config.qsim().isUsingTravelTimeCheckInTeleportation() ) {
248                        log.log( lvl, "found `qsim.usingTravelTimeCheckInTeleporation==false'; vsp should try out `true' and report." ) ;
249                }
250                
251                // added apr'15
252//              if ( !config.qsim().isUsingFastCapacityUpdate() ) {
253//                      log.log( lvl,  " found 'qsim.usingFastCapacityUpdate==false'; vsp should try out `true' and report. ") ;
254//              }
255                switch( config.qsim().getTrafficDynamics() ) {
256                case withHoles:
257                case kinematicWaves:
258                        break;
259                case queue:
260                default:
261                        log.log( lvl,  " found 'qsim.trafficDynamics==" + config.qsim().getTrafficDynamics() + "'; vsp standard is`" 
262                                        + TrafficDynamics.kinematicWaves + "'." ) ;
263                        break;
264                }
265                
266                if ( config.qsim()!=null && config.qsim().isRemoveStuckVehicles() ) {
267                        problem = true ;
268                        System.out.flush() ;
269                        log.log( lvl, "found that the qsim is removing stuck vehicles.  vsp default is setting this to false.");
270                }
271                
272                // === strategy:
273                
274                boolean found = false ;
275                Collection<StrategySettings> settingsColl = config.strategy().getStrategySettings();
276                for ( StrategySettings settings : settingsColl ) {
277                        if ( settings.getStrategyName().equalsIgnoreCase("ChangeExpBeta") ) {
278                                found = true ;
279                        }
280                }
281                if ( !found ) {
282                        problem = true ;
283                        System.out.flush() ;
284                        log.log( lvl, "You have no strategy configured that uses ChangeExpBeta. vsp default is to use ChangeExpBeta at least in one strategy.");
285                }
286                
287                // added may'16
288                if ( config.strategy().getFractionOfIterationsToDisableInnovation()==Double.POSITIVE_INFINITY ) {
289                        problem = true ;
290                        System.out.flush() ;
291                        log.log( lvl, "You have not set fractionOfIterationsToDisableInnovation; vsp default is to set this to 0.8 or similar.  Add the following config lines:" ) ;
292                        log.log( lvl, "<module name=\"strategy\">");
293                        log.log( lvl, " <param name=\"fractionOfIterationsToDisableInnovation\" value=\"0.8\" />");
294                        log.log( lvl, "</module>");
295                }
296                
297                // added nov'15
298                boolean usingTimeMutator = false ;
299                for ( StrategySettings it : config.strategy().getStrategySettings() ) {
300                        if ( DefaultStrategy.TimeAllocationMutator.toString().equals( it.getName() ) ) {
301                                usingTimeMutator = true ;
302                                break ;
303                        }
304                }
305                if ( usingTimeMutator ) {
306                        // added before nov'12
307                        if ( config.timeAllocationMutator().getMutationRange() < 7200 ) {
308                                problem = true ;
309                                System.out.flush() ;
310                                log.log( lvl, "timeAllocationMutator mutationRange < 7200; vsp default is 7200.  This means you have to add the following lines to your config file: ") ;
311                                log.log( lvl, "<module name=\"TimeAllocationMutator\">");
312                                log.log( lvl, " <param name=\"mutationRange\" value=\"7200.0\" />");
313                                log.log( lvl, "</module>");
314                        }
315                        // added jan'14
316                        if ( config.timeAllocationMutator().isAffectingDuration() ) {
317                                //                      problem = true ;
318                                System.out.flush() ;
319                                log.log( lvl, "timeAllocationMutator is affecting duration; vsp default is to not do that.  This will be more strictly" +
320                                                " enforced in the future. This means you have to add the following lines to your config file: ") ;
321                                log.log( lvl, "<module name=\"TimeAllocationMutator\">");
322                                log.log( lvl, " <param name=\"affectingDuration\" value=\"false\" />");
323                                log.log( lvl, "</module>");
324                        }
325                }
326                
327                // === interaction between config groups:
328                boolean containsModeChoice = false ;
329                for ( StrategySettings settings : config.strategy().getStrategySettings() ) {
330                        if ( settings.getStrategyName().contains("Mode") ) {
331                                containsModeChoice = true ;
332                        }
333                }
334                
335                // added jun'16
336                if ( config.qsim().getVehiclesSource()==VehiclesSource.fromVehiclesData 
337                                && config.qsim().getUsePersonIdForMissingVehicleId() 
338                                && containsModeChoice 
339                                && config.qsim().getMainModes().size() > 1 ) 
340                {
341                        problem = true ;
342                        log.log( lvl, "You can't use more than one main (=vehicular) mode while using the agent ID as missing vehicle ID ... "
343                                        + "because in this case the person can only have one vehicle and thus cannot switch to a different vehicle type." ) ;
344                }
345
346                // === zzz:
347                
348                if ( problem && config.vspExperimental().getVspDefaultsCheckingLevel() == VspDefaultsCheckingLevel.abort ) {
349                        String str = "found a situation that leads to vsp-abort.  aborting ..." ; 
350                        System.out.flush() ;
351                        log.fatal( str ) ; 
352                        throw new RuntimeException( str ) ;
353                }
354                
355        }
