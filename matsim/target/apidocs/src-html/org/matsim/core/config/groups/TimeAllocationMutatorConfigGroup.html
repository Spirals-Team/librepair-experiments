021package org.matsim.core.config.groups;
022
023import java.util.Collection;
024import java.util.Map;
025
026import org.matsim.core.config.ConfigGroup;
027import org.matsim.core.config.ReflectiveConfigGroup;
028
029public final class TimeAllocationMutatorConfigGroup extends ReflectiveConfigGroup {
030
031        public static final String GROUP_NAME = "TimeAllocationMutator";
032                
033        public TimeAllocationMutatorConfigGroup() {
034                super(GROUP_NAME);
035        }
036
037        @Override
038        public Map<String, String> getComments() {
039                Map<String, String> comments = super.getComments();
040                comments.put(MUTATION_RANGE, "Default:1800.0; Defines how many seconds a time mutation can maximally shift a time.");
041                comments.put(MUTATION_AFFECTS_DURATION, "Default:true; Defines whether time mutation changes an activity's duration.");
042                comments.put(USE_INDIVIDUAL_SETTINGS_FOR_SUBPOPULATIONS, "false; Use individual settings for each subpopulation. "
043                                + "If enabled but no settings are found, regular settings are uses as fallback.");
044                return comments;
045        }
046        
047        // ---
048
049        private static final String MUTATION_RANGE = "mutationRange";
050        private double mutationRange = 1800.0;
051        @StringGetter(MUTATION_RANGE)
052        public double getMutationRange() {
053                return this.mutationRange;
054        }
055        @StringSetter(MUTATION_RANGE)
056        public void setMutationRange(final double val) {
057                this.mutationRange = val;
058        }
059        
060        // ---
061        
062        private static final String MUTATION_AFFECTS_DURATION = "mutationAffectsDuration";
063        private boolean affectingDuration = true;
064        @StringGetter(MUTATION_AFFECTS_DURATION)
065        public boolean isAffectingDuration() {
066                return this.affectingDuration;
067        }
068        @StringSetter(MUTATION_AFFECTS_DURATION)
069        public void setAffectingDuration(boolean affectingDuration) {
070                this.affectingDuration = affectingDuration;
071        }
072        
073        // ---
074        
075        private static final String USE_INDIVIDUAL_SETTINGS_FOR_SUBPOPULATIONS = "useIndividualSettingsForSubpopulations";
076        private boolean useIndividualSettingsForSubpopulations = false;
077        @StringGetter(USE_INDIVIDUAL_SETTINGS_FOR_SUBPOPULATIONS)
078        public boolean isUseIndividualSettingsForSubpopulations() {
079                return this.useIndividualSettingsForSubpopulations;
080        }
081        @StringSetter(USE_INDIVIDUAL_SETTINGS_FOR_SUBPOPULATIONS)
082        public void setUseIndividualSettingsForSubpopulations(boolean useIndividualSettingsForSubpopulations) {
083                this.useIndividualSettingsForSubpopulations = useIndividualSettingsForSubpopulations;
084        }
085        
086        // ---
087        
088        public TimeAllocationMutatorSubpopulationSettings getTimeAllocationMutatorSubpopulationSettings(String subpopulation) {
089                
090                if (subpopulation == null) return null;
091                
092                Collection<? extends ConfigGroup> configGroups = this.getParameterSets(TimeAllocationMutatorSubpopulationSettings.SET_NAME);
093                for (ConfigGroup group : configGroups) {
094                        if (group instanceof TimeAllocationMutatorSubpopulationSettings) {
095                                TimeAllocationMutatorSubpopulationSettings subpopulationSettings = (TimeAllocationMutatorSubpopulationSettings) group;
096                                if (subpopulation.equals(subpopulationSettings.subpopulation)) return subpopulationSettings;
097                        }
098                }
099                
100                return null;
101        }
102        
103        @Override
104        public ConfigGroup createParameterSet(final String type) {
105                switch (type) {
106                        case TimeAllocationMutatorSubpopulationSettings.SET_NAME:
107                                return new TimeAllocationMutatorSubpopulationSettings();
108                        default:
109                                throw new IllegalArgumentException("unknown set type '" + type + "'");
110                }
111        }
112        
113        @Override
114        public void addParameterSet(final ConfigGroup set) {
115                switch (set.getName()) {
116                        case TimeAllocationMutatorSubpopulationSettings.SET_NAME:
117                                super.addParameterSet(set);
118                                break;
119                        default:
120                                throw new IllegalArgumentException( set.getName() );
121                }
122        }
123        
124        public static class TimeAllocationMutatorSubpopulationSettings extends ReflectiveConfigGroup {
125                
126                public static final String SET_NAME = "subpopulationSettings";
127                private static final String MUTATION_RANGE = "mutationRange";
128                private static final String MUTATION_AFFECTS_DURATION = "mutationAffectsDuration";
129                private static final String SUBPOPULATION = "subpopulation";
130                
131                private double mutationRange = 1800.0;
132                private boolean affectingDuration = true;
133                private String subpopulation = null;
134                
135                public TimeAllocationMutatorSubpopulationSettings() {
136                        super(SET_NAME);
137                }
138                
139                @Override
140                public final Map<String, String> getComments() {
141                        Map<String,String> comments = super.getComments();
142                        comments.put(MUTATION_RANGE, "Default:1800.0; Defines how many seconds a time mutation can maximally shift a time.");
143                        comments.put(MUTATION_AFFECTS_DURATION, "Default:true; Defines whether time mutation changes an activity's duration.");
144                        comments.put(SUBPOPULATION, "Subpopulation to which the values from this parameter set are applied.");
145                        return comments;
146                }
147
148                @StringGetter(MUTATION_RANGE)
149                public double getMutationRange() {
150                        return this.mutationRange;
151                }
152                
153                @StringSetter(MUTATION_RANGE)
154                public void setMutationRange(final double val) {
155                        this.mutationRange = val;
156                }
157                
158                @StringGetter(MUTATION_AFFECTS_DURATION)
159                public boolean isAffectingDuration() {
160                        return affectingDuration;
161                }
162                
163                @StringSetter(MUTATION_AFFECTS_DURATION)
164                public void setAffectingDuration(boolean affectingDuration) {
165                        this.affectingDuration = affectingDuration;
166                }
167
168                @StringSetter(SUBPOPULATION)
169                public void setSubpopulation(final String subpopulation) {
170                        this.subpopulation = subpopulation;
171                }
172
173                @StringGetter(SUBPOPULATION)
174                public String getSubpopulation() {
175                        return this.subpopulation;
176                }
177        }
178}
