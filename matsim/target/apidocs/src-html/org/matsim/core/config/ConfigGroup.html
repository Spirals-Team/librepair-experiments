057        public ConfigGroup(final String name) {
058                this.name = name;
059                this.params = new TreeMap<>();
060        }
061
062        public void addParam(final String paramName, final String value) {
063                if (this.params.containsKey(paramName)) {
064                        log.info(this.toString() + "[paramName=" + paramName + ",oldValue=" + this.params.get(paramName) + ",value=" + value + " value replaced]");
065                }
066                this.params.put(paramName, value);
067        }
068
069        /**
070         * Little helper for subclasses (i.e. the ConfigGroups). This method adds the value of the parameter
071         * to the given map only if the getValue() method of this Module doesn't return
072         * null (Java null-type) or the String representation of null, i.e. "null" or "NULL".
073         * If the value is null, the string "null" is added to the map to document the parameter.
074         *
075         * @param map
076         * @param paramName
077         */
078        protected void addParameterToMap(final Map<String, String> map, final String paramName) {
079                String value = this.getValue(paramName);
080                if (!((value == null) || value.equalsIgnoreCase("null"))) {
081                        map.put(paramName, value);
082                } else {
083                        map.put(paramName, "null");
084                }
085        }
086
087        /** Check if the set values go well together. This method is usually called after reading the
088         * configuration from a file. If an inconsistency is found, a warning or error should be issued
089         * and (optionally) a RuntimeException being thrown.
090         * @param config TODO
091         */
092        protected void checkConsistency(Config config) {
093                // (I added Config as a parameter, since there are many occasions where the validity of a ConfigGroup can only be checked when other
094                // material is known.  Could put all of this in the "global" config consistency checker, but if it conceptually belongs into the
095                // ConfigGroup, I think it is easier to have it more local.  Wasn't a big problem, since this method is _only_ called from the global config
096                // itself, which obviously can just pass on a "this" pointer.  kai, jan'17)
097                
098                // default: just call this method on parameter sets
099                for ( Collection<? extends ConfigGroup> sets : getParameterSets().values() ) {
100                        for ( ConfigGroup set : sets ) set.checkConsistency(config);
101                }
102        }

122                return new HashMap<>();
123        }
124
125        @Override
126        public final String toString() {
127                StringBuilder str = new StringBuilder();
128                for ( Entry<String, String> entry : this.getParams().entrySet() ) {
129                        str.append('[');
130                        str.append(entry.getKey());
131                        str.append('=');
132                        str.append(entry.getValue());
133                        str.append(']');
134                }
135                return "[name=" + this.getName() + "]" +
136                                "[nOfParams=" + this.getParams().size() + "]" + str.toString();
137        }

165        public void addParameterSet(final ConfigGroup set) {
166                checkParameterSet( set );
167                Collection<ConfigGroup> parameterSets = parameterSetsPerType.get( set.getName() );
168
169                if ( parameterSets == null ) {
170                        parameterSets = new ArrayList<>();
171                        parameterSetsPerType.put( set.getName() ,  parameterSets );
172                }
173
174                parameterSets.add( set );
175        }
176
177        public boolean removeParameterSet( final ConfigGroup set ) {
178                final Collection<ConfigGroup> parameterSets = parameterSetsPerType.get( set.getName() );
179                return parameterSets != null ?
180                        parameterSets.remove( set ) :
181                        false;
182        }

195        /**
196         * Useful for instance if default values are provided but should be cleared if
197         * user provides values.
198         */
199        protected final Collection<? extends ConfigGroup> clearParameterSetsForType( final String type ) {
200                return parameterSetsPerType.remove( type );
201        }
202
203        public final Collection<? extends ConfigGroup> getParameterSets(final String type) {
204                final Collection<ConfigGroup> sets = parameterSetsPerType.get( type );
205                return sets == null ?
206                        Collections.<ConfigGroup>emptySet() :
207                        Collections.unmodifiableCollection( sets );
208        }

210        public final Map<String, ? extends Collection<? extends ConfigGroup>> getParameterSets() {
211                // TODO: immutabilize (including lists)
212                // maybe done with what I did below?  kai, sep'16
213                
214                //              return parameterSetsPerType;
215                
216                Map<String, Collection<ConfigGroup>> parameterSetsPerType2 = new TreeMap<>() ;
217                for ( Entry<String, Collection<ConfigGroup>> entry : parameterSetsPerType.entrySet() ) {
218                        parameterSetsPerType2.put( entry.getKey(), Collections.unmodifiableCollection(entry.getValue()) ) ;
219                }
220                return Collections.unmodifiableMap( parameterSetsPerType2 ) ;
221        }
222
223        public final boolean isLocked() {
224                return locked;
225        }
226
227        public void setLocked() {
228                // need to have this non-final to be able to override in order to set delegates.  kai, jun'15
229                this.locked = true ;
230                for ( Collection<ConfigGroup> parameterSets : this.parameterSetsPerType.values() ) {
231                        for ( ConfigGroup parameterSet : parameterSets ) {
232                                parameterSet.setLocked(); 
233                        }
234                }
235        }
236        
237        public final void testForLocked() {
238                if ( locked ) {
239                        throw new RuntimeException( "This config group is locked since material from this config group has already been used.") ;
240                }
241        }

243        public static URL getInputFileURL(URL context, String filename) {
244                if (filename.startsWith("~" + File.separator)) {
245                        filename = System.getProperty("user.home") + filename.substring(1);
246                }
247                return IOUtils.newUrl(context, filename);
248        }
