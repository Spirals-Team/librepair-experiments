021package org.matsim.core.config.groups;
022
023import org.matsim.core.config.ConfigGroup;
024import org.matsim.core.config.ReflectiveConfigGroup;
025
026import java.net.URL;
027
028import java.util.Map;
029
030public final class NetworkConfigGroup extends ReflectiveConfigGroup {
031
032        public static final String GROUP_NAME = "network";
033
034        private static final String INPUT_FILE= "inputNetworkFile";
035
036        private static final String TIME_VARIANT_NETWORK = "timeVariantNetwork";
037        private static final String CHANGE_EVENTS_INPUT_FILE = "inputChangeEventsFile";
038
039        private static final String LANEDEFINITIONSINPUTFILE = "laneDefinitionsFile";
040        private static final String INPUT_CRS = "inputCRS";
041
042        private String inputFile = null;
043
044        private String inputCRS = null;
045
046        private String changeEventsInputFile = null;
047
048        private boolean timeVariantNetwork = false;
049
050        private String laneDefinitionsFile = null;
051
052        public NetworkConfigGroup() {
053                super(NetworkConfigGroup.GROUP_NAME);
054        }
055
056        @Override
057        public Map<String,String> getComments() {
058                final Map<String,String> comments = super.getComments();
059
060                comments.put( INPUT_CRS , "The Coordinates Reference System in which the coordinates are expressed in the input file." +
061                                " At import, the coordinates will be converted to the coordinate system defined in \"global\", and will" +
062                                "be converted back at export. If not specified, no conversion happens." );
063
064                return comments;
065        }
066
067        /* direct access */
068
069        @StringGetter( INPUT_FILE )
070        public String getInputFile() {
071                return this.inputFile;
072        }
073        @StringSetter( INPUT_FILE )
074        public void setInputFile(final String inputFile) {
075                this.inputFile = inputFile;
076        }
077
078        public URL getInputFileURL(URL context) {
079                return ConfigGroup.getInputFileURL(context, this.inputFile);
080        }
081
082        @StringSetter( CHANGE_EVENTS_INPUT_FILE )
083        public void setChangeEventsInputFile(final String changeEventsInputFile) {
084                this.changeEventsInputFile = changeEventsInputFile;
085        }
086        
087        @StringGetter( CHANGE_EVENTS_INPUT_FILE )
088        public String getChangeEventsInputFile() {
089                return changeEventsInputFile;
090        }
091        
092        
093        public URL getChangeEventsInputFileUrl(URL context) {
094                return ConfigGroup.getInputFileURL(context, this.changeEventsInputFile);
095        }
096
097        @StringSetter( TIME_VARIANT_NETWORK )
098        public void setTimeVariantNetwork(final boolean timeVariantNetwork) {
099                testForLocked();
100                this.timeVariantNetwork = timeVariantNetwork;
101        }
102        @StringGetter( TIME_VARIANT_NETWORK )
103        public boolean isTimeVariantNetwork() {
104                return this.timeVariantNetwork;
105        }
106
107        @StringSetter( LANEDEFINITIONSINPUTFILE )
108        public void setLaneDefinitionsFile(final String laneDefinitions) {
109                this.laneDefinitionsFile = laneDefinitions;
110        }
111
112        @StringGetter( LANEDEFINITIONSINPUTFILE )
113        public String getLaneDefinitionsFile(){
114                return this.laneDefinitionsFile;
115        }
116
117
118        @StringGetter( INPUT_CRS )
119        public String getInputCRS() {
120                return inputCRS;
121        }
122
123        @StringSetter( INPUT_CRS )
124        public void setInputCRS(String inputCRS) {
125                this.inputCRS = inputCRS;
126        }
127}
