001package org.matsim.core.config.groups;
002
003
004import org.matsim.api.core.v01.TransportMode;
005import org.matsim.core.config.ReflectiveConfigGroup;
006import org.matsim.core.utils.misc.StringUtils;
007
008import java.util.Map;
009
010public class ChangeModeConfigGroup extends ReflectiveConfigGroup {
011
012        public final static String CONFIG_MODULE = "changeMode";
013        public final static String CONFIG_PARAM_MODES = "modes";
014        public final static String CONFIG_PARAM_IGNORECARAVAILABILITY = "ignoreCarAvailability";
015        public final static String MODE_SWITCH_BEHAVIOR = "modeSwitchBehavior";
016
017
018        public enum Behavior { fromAllModesToSpecifiedModes, fromSpecifiedModesToSpecifiedModes }
019        private Behavior behavior = Behavior.fromSpecifiedModesToSpecifiedModes ;
020
021        private String[] modes = new String[] { TransportMode.car, TransportMode.pt };
022        private boolean ignoreCarAvailability = true;
023        public ChangeModeConfigGroup() {
024                super(CONFIG_MODULE);
025        }
026
027        public String[] getModes() {
028                return modes;
029        }
030
031        @StringGetter( CONFIG_PARAM_MODES )
032        private String getModesString() {
033                return toString( modes );
034        }
035
036        @StringSetter( CONFIG_PARAM_MODES )
037        private void setModes( final String value ) {
038                setModes( toArray( value ) );
039        }
040
041        public void setModes( final String[] modes ) {
042                this.modes = modes;
043        }
044
045        @StringSetter( CONFIG_PARAM_IGNORECARAVAILABILITY )
046        public void setIgnoreCarAvailability(final boolean value) {
047                this.ignoreCarAvailability = value;
048        }
049
050        @StringGetter( CONFIG_PARAM_IGNORECARAVAILABILITY )
051        public boolean getIgnoreCarAvailability() {
052                return ignoreCarAvailability;
053        }
054
055        @StringGetter(MODE_SWITCH_BEHAVIOR)
056        public Behavior getBehavior() {
057                return behavior;
058        }
059
060        @StringSetter(MODE_SWITCH_BEHAVIOR)
061        public void setBehavior(Behavior behavior) {
062                this.behavior = behavior;
063        }
064
065        @Override
066        public Map<String, String> getComments() {
067                Map<String, String> comments = super.getComments();
068                comments.put(CONFIG_PARAM_MODES, "Defines all the modes available, including chain-based modes, seperated by commas" );
069                comments.put(CONFIG_PARAM_IGNORECARAVAILABILITY, "Defines whether car availability is considered be considered or not. An agent has no car only if it has no license, or never access to a car. Default: true" );
070                comments.put(MODE_SWITCH_BEHAVIOR,"Defines the mode switch behavior. Possible values "+Behavior.values().toString()+" Default: fromSpecifiedModesToSpecifiedModes.");
071                return comments;
072        }
073
074        private static String toString(final String[] modes ) {
075                // (not same as toString() because of argument!)
076
077                StringBuilder b = new StringBuilder();
078
079                if (modes.length > 0) b.append( modes[ 0 ] );
080                for (int i=1; i < modes.length; i++) {
081                        b.append( ',' );
082                        b.append( modes[ i ] );
083                }
084
085                return b.toString();
086        }
087
088        private static String[] toArray( final String modes ) {
089                String[] parts = StringUtils.explode(modes, ',');
090
091                for (int i = 0, n = parts.length; i < n; i++) {
092                        parts[i] = parts[i].trim().intern();
093                }
094
095                return parts;
096        }
097
098
099}
