001package org.matsim.core.config.groups;
002
003
004import org.matsim.api.core.v01.TransportMode;
005import org.matsim.core.config.ReflectiveConfigGroup;
006import org.matsim.core.utils.misc.StringUtils;
007
008public class ChangeModeConfigGroup extends ReflectiveConfigGroup {
009
010//      public final static String CONFIG_MODULE = "changeLegMode";
011        public final static String CONFIG_MODULE = "changeMode";
012        public final static String CONFIG_PARAM_MODES = "modes";
013        public final static String CONFIG_PARAM_IGNORECARAVAILABILITY = "ignoreCarAvailability";
014
015        private String[] modes = new String[] { TransportMode.car, TransportMode.pt };
016        private boolean ignoreCarAvailability = true;
017
018        public ChangeModeConfigGroup() {
019                super(CONFIG_MODULE);
020        }
021
022        public String[] getModes() {
023                return modes;
024        }
025
026        @StringGetter( CONFIG_PARAM_MODES )
027        private String getModesString() {
028                return toString( modes );
029        }
030
031        @StringSetter( CONFIG_PARAM_MODES )
032        private void setModes( final String value ) {
033                setModes( toArray( value ) );
034        }
035
036        public void setModes( final String[] modes ) {
037                this.modes = modes;
038        }
039
040        @StringSetter( CONFIG_PARAM_IGNORECARAVAILABILITY )
041        public void setIgnoreCarAvailability(final boolean value) {
042                this.ignoreCarAvailability = value;
043        }
044
045        @StringGetter( CONFIG_PARAM_IGNORECARAVAILABILITY )
046        public boolean getIgnoreCarAvailability() {
047                return ignoreCarAvailability;
048        }
049
050        private static String toString( final String[] modes ) {
051                // (not same as toString() because of argument!)
052
053                StringBuilder b = new StringBuilder();
054
055                if (modes.length > 0) b.append( modes[ 0 ] );
056                for (int i=1; i < modes.length; i++) {
057                        b.append( ',' );
058                        b.append( modes[ i ] );
059                }
060
061                return b.toString();
062        }
063
064        private static String[] toArray( final String modes ) {
065                String[] parts = StringUtils.explode(modes, ',');
066
067                for (int i = 0, n = parts.length; i < n; i++) {
068                        parts[i] = parts[i].trim().intern();
069                }
070
071                return parts;
072        }
073
074
075}
