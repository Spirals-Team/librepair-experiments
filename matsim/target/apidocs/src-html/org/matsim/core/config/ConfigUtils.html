074        public static Config loadConfig(final URL url, ConfigGroup... customModules) throws UncheckedIOException {
075                Gbl.assertNotNull(url);
076                
077                Config config = new Config();
078                config.addCoreModules();
079
080                for (ConfigGroup customModule : customModules) {
081                        config.addModule(customModule);
082                }
083                
084                new ConfigReader(config).parse(url);
085                config.setContext(url);
086                return config;
087        }

090        /**
091         * This does (hopefully) overwrite config settings if they are defined in the file.  So you can do
092         * <pre>
093         * Config config = ConfigUtils.createConfig() ;
094         * config.xxx().setYyy() ; // set some defaults for your application.
095         * ConfigUtils.loadConfig( config, filename ) ; // read user-defined options
096         * config.aaa().bbb() ; // set config options which you don't want the user to potentially overwrite.
097         * ...
098         * </pre>  
099         */
100        public static void loadConfig(final Config config, final String filename) throws UncheckedIOException {
101                if (config.global() == null) {
102                        // if the config does not exist yet, we interpret the file name as is:
103                        config.addCoreModules();
104                        new ConfigReader(config).readFile(filename);
105                } else {
106                        // if the config does already exist, this can be used to read a second config file to override
107                        // settings from before, e.g. using a base config and then several case study configs.
108                        // In this case, using the above syntax generates inconsistent behavior between
109                        // gui and command line: command line takes the config file from the java root,
110                        // while the gui takes it from the config file root.  The following syntax should
111                        // now also take it from the config file root when it is called from the command line
112                        // (same as in other places: we are making the command line behavior
113                        // and GUI behavior consistent).
114                        // kai, jan'18
115//                      URL url = ConfigGroup.getInputFileURL(config.getContext(), filename);;
116//                      new ConfigReader(config).parse(url) ;
117                        // yyyyyy the above probably works, but has ramifications across many test
118                        // cases.  Need to discuss first (and then find some time again).
119                        // See MATSIM-776 and MATSIM-777.  kai, feb'18
120                        
121                        new ConfigReader(config).readFile(filename);
122                }
123        }

142        /**
143         * Adds the <code>pathPrefix</code> in front of all supported file-paths
144         * in core-config groups in the configuration, except when the paths are
145         * specified as absolute paths. This is useful e.g. if the configuration
146         * contains file-paths relative to the location of the config-file.
147         * The currently supported file-paths are:
148         * <ul>
149         *  <li>global.localDtdBase</li>
150         *  <li>controler.outputDirectory</li>
151         *  <li>network.inputNetworkFile</li>
152         *  <li>plans.inputPlansFile</li>
153         *  <li>facilities.inputFacilitiesFile</li>
154         *  <li>counts.inputCountsFile</li>
155         *  <li>world.inputWorldFile</li>
156         *  <li>households.inputFile</li>
157         *  <li>roadpricing.tollLinksFile</li>
158         * </ul>
159         *
160         * @param config
161         * @param pathPrefix
162         */
163        public static void modifyFilePaths(final Config config, final String pathPrefix) {
164                String prefix = pathPrefix;
165                if (!prefix.endsWith("/") && !prefix.endsWith(File.separator)) {
166                        prefix = prefix + File.separator;
167                }
168                config.controler().setOutputDirectory(getAbsolutePath(prefix, config.controler().getOutputDirectory()));
169                config.network().setInputFile(getAbsolutePath(prefix, config.network().getInputFile()));
170                config.plans().setInputFile(getAbsolutePath(prefix, config.plans().getInputFile()));
171                config.facilities().setInputFile(getAbsolutePath(prefix, config.facilities().getInputFile()));
172                config.counts().setInputFile(getAbsolutePath(prefix, config.counts().getCountsFileName()));
173                config.households().setInputFile(getAbsolutePath(prefix, config.households().getInputFile()));
174        }
175
176        private static String getAbsolutePath(final String prefix, final String path) {
177                if (path == null) {
178                        return null;
179                }
180                File file = new File(path);
181                if (file.exists() && path.equals(file.getAbsolutePath())) {
182                        return path;
183                }
184                /* even if the file exists at the given location, its path
185                 * seems not to be specified as absolute path, thus we have to
186                 * interpret it as a relative path and add the prefix to it.
187                 */
188                String absolutePath = prefix + path;
189                return absolutePath;
190        }

192        public static Id<StrategySettings> createAvailableStrategyId(Config config) {
193                long maxStrategyId = 0;
194                Iterator<StrategySettings> iterator = config.strategy().getStrategySettings().iterator();
195                while(iterator.hasNext()){
196                        maxStrategyId = Math.max(maxStrategyId, Long.parseLong(iterator.next().getId().toString()));
197                }
198                return Id.create(maxStrategyId + 1, StrategySettings.class);
199        }

202        /** 
203         * Convenience method to all addOrGetModule with only two arguments.  
204         * 
205         * @param config
206         * @param moduleClass
207         * @return instance of moduleClass inside config
208         */
209        public static <T extends ConfigGroup> T addOrGetModule( Config config, Class<T> moduleClass ) {
210                String groupName;
211                try {
212                        groupName = moduleClass.newInstance().getName();
213                } catch (InstantiationException | IllegalAccessException e) {
214                        throw new RuntimeException(e) ;
215                }
216                return addOrGetModule( config, groupName, moduleClass ) ;
217        }

219        /**
220         * This is a refactoring device to remove former core config groups from the core config.
221         * Instructions: If you want to remove e.g. Config.vspExperimental(), replace that
222         * method with
223         * ConfigUtils.addOrGetModule(this, VspExperimentalConfigGroup.GROUP_NAME, VspExperimentalConfigGroup.class)
224         * and then hit Refactor/Inline.
225         */
226        public static <T extends ConfigGroup> T addOrGetModule(Config config, String groupName, Class<T> moduleClass) {
227                ConfigGroup module = config.getModule(groupName);
228                if (module == null || module.getClass() == ConfigGroup.class) {
229                        try {
230                                module = moduleClass.newInstance();
231                                config.addModule(module);
232                        } catch (InstantiationException | IllegalAccessException e) {
233                                throw new RuntimeException(e);
234                        }
235        }
236                return moduleClass.cast(module);
237        }
