021package org.matsim.core.config.groups;
022
023import java.util.Map;
024
025import org.matsim.core.config.ReflectiveConfigGroup;
026
027/**
028 * @author nagel
029 */
030public final class PtCountsConfigGroup extends ReflectiveConfigGroup {
031
032        public static final String GROUP_NAME = "ptCounts";
033
034        private static final String OUTPUTFORMAT = "outputformat";
035        private static final String DISTANCEFILTER = "distanceFilter";
036        private static final String DISTANCEFILTERCENTERNODE = "distanceFilterCenterNode";
037        private static final String OCCUPANCY_COUNTS_INPUT_FILENAME = "inputOccupancyCountsFile";
038        private static final String BOARD_COUNTS_INPUT_FILENAME = "inputBoardCountsFile";
039        private static final String ALIGHT_COUNTS_INPUT_FILENAME = "inputAlightCountsFile";
040        private static final String COUNTSSCALEFACTOR = "countsScaleFactor";
041        private static final String PT_COUNTS_INTERVAL = "ptCountsInterval" ;
042
043        private String outputFormat;
044
045        /**
046         * the distance filter in m
047         */
048        private Double distanceFilter;
049
050        /**
051         * the id of the node used as center for the distance filter
052         */
053        private String distanceFilterCenterNode;
054
055        /**
056         * the path to the file with the counts
057         */
058        private String occupancyCountsFileName = null;
059        private String boardCountsFileName = null;
060        private String alightCountsFileName = null;
061        /**
062         * the scaling for the counts
063         */
064        private double countsScaleFactor = 1.0;
065        private int ptCountsInterval = 10 ;
066
067        public PtCountsConfigGroup() {
068                super(GROUP_NAME);
069        }
070
071        @Override
072        public Map<String, String> getComments() {
073                Map<String, String> comments = super.getComments();
074                comments.put(OUTPUTFORMAT, CountsConfigGroup.COUNTS_OUTPUTFORMAT_COMMENT ) ;
075                comments.put(DISTANCEFILTER,  CountsConfigGroup.COUNTS_DISTANCEFILTER_COMMENT ) ;
076                comments.put(DISTANCEFILTERCENTERNODE, CountsConfigGroup.COUNTS_DISTANCEFILTERCENTERNODE_COMMENT ) ;
077                comments.put(OCCUPANCY_COUNTS_INPUT_FILENAME, "input file containing the occupancy counts for pt" ) ;
078                comments.put(ALIGHT_COUNTS_INPUT_FILENAME, "input file containing the alighting (getting off) counts for pt" ) ;
079                comments.put(BOARD_COUNTS_INPUT_FILENAME, "input file containing the boarding (getting on) counts for pt" ) ;
080                comments.put(COUNTSSCALEFACTOR, CountsConfigGroup.COUNTSSCALEFACTOR_COMMENT ) ;
081                comments.put(PT_COUNTS_INTERVAL, "every how many iterations (starting with 0) counts comparisons are generated" );
082                return comments;
083        }
084
085        @StringGetter( OUTPUTFORMAT )
086        public String getOutputFormat() {
087                return outputFormat;
088        }
089
090        @StringSetter( OUTPUTFORMAT )
091        public void setOutputFormat(String outputFormat) {
092                this.outputFormat = outputFormat;
093        }
094
095        @StringGetter( DISTANCEFILTER )
096        public Double getDistanceFilter() {
097                return distanceFilter;
098        }
099
100        @StringSetter( DISTANCEFILTER )
101        public void setDistanceFilter(Double distanceFilter) {
102                this.distanceFilter = distanceFilter;
103        }
104
105        @StringGetter( DISTANCEFILTERCENTERNODE )
106        public String getDistanceFilterCenterNode() {
107                return distanceFilterCenterNode;
108        }
109
110        @StringSetter( DISTANCEFILTERCENTERNODE )
111        public void setDistanceFilterCenterNode(String distanceFilterCenterNode) {
112                this.distanceFilterCenterNode = distanceFilterCenterNode;
113        }
114
115        @StringGetter( OCCUPANCY_COUNTS_INPUT_FILENAME )
116        public String getOccupancyCountsFileName() {
117                return occupancyCountsFileName;
118        }
119
120        @StringSetter( OCCUPANCY_COUNTS_INPUT_FILENAME )
121        public void setOccupancyCountsFileName(String occupancyCountsFileName) {
122                this.occupancyCountsFileName = occupancyCountsFileName;
123        }
124
125        @StringGetter( BOARD_COUNTS_INPUT_FILENAME )
126        public String getBoardCountsFileName() {
127                return boardCountsFileName;
128        }
129
130        @StringSetter( BOARD_COUNTS_INPUT_FILENAME )
131        public void setBoardCountsFileName(String boardCountsFileName) {
132                this.boardCountsFileName = boardCountsFileName;
133        }
134
135        @StringGetter( ALIGHT_COUNTS_INPUT_FILENAME )
136        public String getAlightCountsFileName() {
137                return alightCountsFileName;
138        }
139
140        @StringSetter( ALIGHT_COUNTS_INPUT_FILENAME )
141        public void setAlightCountsFileName(String alightCountsFileName) {
142                this.alightCountsFileName = alightCountsFileName;
143        }
144
145        @StringGetter( COUNTSSCALEFACTOR )
146        public double getCountsScaleFactor() {
147                return countsScaleFactor;
148        }
149
150        @StringSetter( COUNTSSCALEFACTOR )
151        public void setCountsScaleFactor(double countsScaleFactor) {
152                this.countsScaleFactor = countsScaleFactor;
153        }
154
155        @StringGetter( PT_COUNTS_INTERVAL )
156        public int getPtCountsInterval() {
157                return ptCountsInterval;
158        }
159
160        @StringSetter( PT_COUNTS_INTERVAL )
161        public void setPtCountsInterval(int ptCountsInterval) {
162                this.ptCountsInterval = ptCountsInterval;
163        }
164
165
166}
