021package org.matsim.core.config.groups;
022
023import java.net.URL;
024import java.util.Locale;
025import java.util.Map;
026
027import org.matsim.api.core.v01.TransportMode;
028import org.matsim.core.config.ConfigGroup;
029import org.matsim.core.config.ReflectiveConfigGroup;
030
031/**
032 * @author dgrether
033 * @author mrieser
034 */
035public final class CountsConfigGroup extends ReflectiveConfigGroup {
036
037        public static final String GROUP_NAME = "counts";
038
039        private static final String OUTPUTFORMAT = "outputformat";
040        private static final String DISTANCEFILTER = "distanceFilter";
041        private static final String DISTANCEFILTERCENTERNODE = "distanceFilterCenterNode";
042        private static final String COUNTSINPUTFILENAME = "inputCountsFile";
043        private static final String COUNTSSCALEFACTOR = "countsScaleFactor";
044        private static final String WRITECOUNTSINTERVAL = "writeCountsInterval";
045        private static final String AVERAGECOUNTSOVERITERATIONS = "averageCountsOverIterations";
046        private static final String ANALYZEDMODES = "analyzedModes";
047        private static final String FILTERMODES = "filterModes";
048        private static final String INPUT_CRS = "inputCRS";
049        
050        private String outputFormat = "txt";
051
052        /**
053         * the distance filter in [length unit defined by coordinates] 
054         */
055        private Double distanceFilter;
056
057        /**
058         * the id of the node used as center for the distance filter
059         */
060        private String distanceFilterCenterNode;
061
062        /**
063         * the path to the file with the counts
064         */
065        private String inputFile = null;
066        /**
067         * the scaling for the counts
068         */
069        private double countsScaleFactor = 1.0;
070        
071        private int writeCountsInterval = 10;
072        private int averageCountsOverIterations = 5;
073
074        private String analyzedModes = TransportMode.car;
075        private boolean filterModes = false;
076        private String inputCRS = null;
077
078        public CountsConfigGroup() {
079                super(GROUP_NAME);
080        }
081
082        // the following are public so they can be re-used in PtCountsComparisonConfigGroup.  Once that group is moved into the same
083        // package, they can be made package-private.  kai, oct'10
084        public static final String COUNTS_OUTPUTFORMAT_COMMENT = "possible values: `html', `kml', `txt', `all'"  ;
085        public static final String COUNTS_DISTANCEFILTER_COMMENT = "distance to distanceFilterCenterNode to include counting stations. The unit of distance is "
086                + "the Euclidean distance implied by the coordinate system" ;
087        public static final String COUNTS_DISTANCEFILTERCENTERNODE_COMMENT = "node id for center node of distance filter" ;
088        public static final String COUNTSINPUTFILENAME_COMMENT = "input file name to counts package" ;
089        public static final String COUNTSSCALEFACTOR_COMMENT = "factor by which to re-scale the simulated values.  necessary when "
090                + "simulation runs with something different from 100%.  needs to be adapted manually" ;
091        
092        @Override
093        public Map<String, String> getComments() {
094                Map<String, String> comments = super.getComments();
095                comments.put(OUTPUTFORMAT, COUNTS_OUTPUTFORMAT_COMMENT ) ;
096                comments.put(DISTANCEFILTER,  COUNTS_DISTANCEFILTER_COMMENT ) ;
097                comments.put(DISTANCEFILTERCENTERNODE, COUNTS_DISTANCEFILTERCENTERNODE_COMMENT ) ;
098                comments.put(COUNTSINPUTFILENAME, COUNTSINPUTFILENAME_COMMENT ) ;
099                comments.put(COUNTSSCALEFACTOR, COUNTSSCALEFACTOR_COMMENT ) ;
100                comments.put(WRITECOUNTSINTERVAL, "Specifies how often the counts comparison should be calculated and written.");
101                comments.put(AVERAGECOUNTSOVERITERATIONS, "Specifies over how many iterations the link volumes should be averaged that are used for the " +
102                                "counts comparison. Use 1 or 0 to only use the link volumes of a single iteration. This values cannot be larger than the value specified for " + WRITECOUNTSINTERVAL);
103                comments.put(ANALYZEDMODES, "Transport modes that will be respected for the counts comparison. 'car' is default, which " +
104                                "includes also bussed from the pt simulation module. Use this parameter in combination with 'filterModes' = true!");
105                comments.put(FILTERMODES, "If true, link counts from legs performed on modes not included in the 'analyzedModes' parameter are ignored.");
106
107                comments.put( INPUT_CRS , "The Coordinates Reference System in which the coordinates are expressed in the input file." +
108                                " At import, the coordinates will be converted to the coordinate system defined in \"global\", and will" +
109                                "be converted back at export. If not specified, no conversion happens." );
110                return comments;
111        }
112
113        @StringGetter( OUTPUTFORMAT )
114        public String getOutputFormat() {
115                return this.outputFormat;
116        }
117
118        @StringSetter( OUTPUTFORMAT )
119        public void setOutputFormat(final String outputFormat) {
120                this.outputFormat = outputFormat;
121        }
122
123        @StringGetter( DISTANCEFILTER )
124        public Double getDistanceFilter() {
125                return this.distanceFilter;
126        }
127
128        @StringSetter( DISTANCEFILTER )
129        public void setDistanceFilter(final Double distanceFilter) {
130                this.distanceFilter = distanceFilter;
131        }
132
133        @StringGetter( DISTANCEFILTERCENTERNODE )
134        public String getDistanceFilterCenterNode() {
135                return this.distanceFilterCenterNode;
136        }
137
138        @StringSetter( DISTANCEFILTERCENTERNODE )
139        public void setDistanceFilterCenterNode(final String distanceFilterCenterNode) {
140                this.distanceFilterCenterNode = distanceFilterCenterNode;
141        }
142
143        /**
144         * @return the filename of the counts file to be read in
145         */
146        @StringGetter( COUNTSINPUTFILENAME )
147        public String getCountsFileName() {
148                return this.inputFile;
149        }
150
151        /**
152         * @param countsFileName the filename of the counts file to be read in
153         */
154        @StringSetter( COUNTSINPUTFILENAME )
155        public void setInputFile(final String countsFileName) {
156                this.inputFile = countsFileName;
157        }
158
159        public URL getCountsFileURL(URL context) {
160                return ConfigGroup.getInputFileURL(context, this.inputFile);
161        }
162
163        @StringGetter( COUNTSSCALEFACTOR )
164        public double getCountsScaleFactor() {
165                return this.countsScaleFactor;
166        }
167
168        @StringSetter( COUNTSSCALEFACTOR )
169        public void setCountsScaleFactor(final double countsScaleFactor) {
170                this.countsScaleFactor = countsScaleFactor;
171        }
172        
173        @StringGetter( WRITECOUNTSINTERVAL )
174        public int getWriteCountsInterval() {
175                return writeCountsInterval;
176        }
177        
178        @StringSetter( WRITECOUNTSINTERVAL )
179        public void setWriteCountsInterval(int writeCountsInterval) {
180                this.writeCountsInterval = writeCountsInterval;
181        }
182        
183        @StringGetter( AVERAGECOUNTSOVERITERATIONS )
184        public int getAverageCountsOverIterations() {
185                return averageCountsOverIterations;
186        }
187        
188        @StringSetter( AVERAGECOUNTSOVERITERATIONS )
189        public void setAverageCountsOverIterations(int averageCountsOverIterations) {
190                this.averageCountsOverIterations = averageCountsOverIterations;
191        }
192        
193        @StringGetter( FILTERMODES )
194        public boolean isFilterModes() {
195                return this.filterModes;
196        }
197        
198        @StringSetter( FILTERMODES )
199        public void setFilterModes(final boolean filterModes) {
200                this.filterModes = filterModes;
201        }
202        
203        @StringGetter( ANALYZEDMODES )
204        public String getAnalyzedModes() {
205                return this.analyzedModes;
206        }
207        
208        @StringSetter( ANALYZEDMODES )
209        public void setAnalyzedModes(final String analyzedModes) {
210                this.analyzedModes = analyzedModes.toLowerCase(Locale.ROOT);
211        }
212
213        @StringGetter( INPUT_CRS )
214        public String getInputCRS() {
215                return inputCRS;
216        }
217
218        @StringSetter( INPUT_CRS )
219        public void setInputCRS(String inputCRS) {
220                this.inputCRS = inputCRS;
221        }
222
223}
