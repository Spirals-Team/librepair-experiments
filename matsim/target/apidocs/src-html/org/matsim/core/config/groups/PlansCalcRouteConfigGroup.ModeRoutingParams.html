113                @Override
114                public void checkConsistency(Config config) {
115                        if ( mode == null ) throw new RuntimeException( "mode for parameter set "+this+" is null!" );
116
117                        if ( teleportedModeSpeed == null && teleportedModeFreespeedFactor == null ) {
118                                throw new RuntimeException( "no teleported mode speed nor freespeed factor defined for mode "+mode );
119                        }
120
121                        if ( teleportedModeSpeed != null && teleportedModeFreespeedFactor != null ) {
122                                // this should not happen anyway as the setters forbid it
123                                throw new RuntimeException( "both teleported mode speed or freespeed factor are set for mode "+mode );
124                        }
125                }

283        @Override
284        protected void checkParameterSet( final ConfigGroup module ) {
285                switch ( module.getName() ) {
286                        case ModeRoutingParams.SET_TYPE:
287                                if ( !(module instanceof ModeRoutingParams) ) {
288                                        throw new RuntimeException( "unexpected class for module "+module );
289                                }
290                                break;
291                        default:
292                                throw new IllegalArgumentException( module.getName() );
293                }

296        @Override
297        public void addParameterSet(final ConfigGroup set) {
298                if ( set.getName().equals( ModeRoutingParams.SET_TYPE ) && !this.acceptModeParamsWithoutClearing ) {
299                        clearParameterSetsForType( set.getName() );
300                        this.acceptModeParamsWithoutClearing = true;
301                        
302                }
303                ModeRoutingParams pars = (ModeRoutingParams) set ;
304                // for the time being pushing the "global" factor into the local ones if they are not initialized by
305                // themselves.  Necessary for some tests; maybe we should eventually disable them.  kai, feb'15
306                if ( pars.getBeelineDistanceFactor()== null ) {
307                        pars.setBeelineDistanceFactor( this.beelineDistanceFactor );
308                }
309                super.addParameterSet( set );
310        }

317        public void removeModeRoutingParams( String key ) {
318                testForLocked() ;
319                for ( ConfigGroup pars : getParameterSets( ModeRoutingParams.SET_TYPE ) ) {
320                        final String mode = ((ModeRoutingParams) pars).getMode();
321                        if ( key.equals(mode) ) {
322                                this.removeParameterSet(pars) ;
323                                break ;
324                        }
325                }
326        }

328        public Map<String, ModeRoutingParams> getModeRoutingParams() {
329                final Map<String, ModeRoutingParams> map = new LinkedHashMap< >();
330
331                for ( ConfigGroup pars : getParameterSets( ModeRoutingParams.SET_TYPE ) ) {
332                        if ( this.isLocked() ) {
333                                pars.setLocked(); 
334                        }
335                        final String mode = ((ModeRoutingParams) pars).getMode();
336                        final ModeRoutingParams old = map.put( mode , (ModeRoutingParams)       pars );
337                        if ( old != null ) throw new IllegalStateException( "several parameter sets for mode "+mode );
338                }
339
340                return map;
341        }

343        public ModeRoutingParams getOrCreateModeRoutingParams(final String mode) {
344                ModeRoutingParams pars = getModeRoutingParams().get( mode );
345
346                if ( pars == null ) {
347                        pars = (ModeRoutingParams) createParameterSet( ModeRoutingParams.SET_TYPE );
348                        pars.setMode( mode );
349                        addParameterSet( pars );
350                }
351                if ( this.isLocked() ) {
352                        pars.setLocked(); 
353                }
354
355                return pars;
356        }

363        @Override
364        public void addParam(final String key, final String value) {
365                if( value.equals( "null" ) ) return; // old behavior of reader: keep defaults if null
366                if (PT_SPEED_FACTOR.equals(key)) {
367                        setTeleportedModeFreespeedFactor(TransportMode.pt, Double.parseDouble(value));
368                } else if (BEELINE_DISTANCE_FACTOR.equals(key)) {
369                        setBeelineDistanceFactor(Double.parseDouble(value));
370                } else if (PT_SPEED.equals(key)) {
371                        setTeleportedModeSpeed(TransportMode.pt, Double.parseDouble(value));
372                } else if (WALK_SPEED.equals(key)) {
373                        setTeleportedModeSpeed(TransportMode.walk, Double.parseDouble(value));
374                } else if (BIKE_SPEED.equals(key)) {
375                        setTeleportedModeSpeed(TransportMode.bike, Double.parseDouble(value));
376                } else if (UNDEFINED_MODE_SPEED.equals(key)) {
377                        setTeleportedModeSpeed(UNDEFINED, Double.parseDouble(value));
378                } else if (NETWORK_MODES.equals(key)) {
379                        setNetworkModes(Arrays.asList(CollectionUtils.stringToArray(value)));
380                } else if (key.startsWith(TELEPORTED_MODE_SPEEDS)) {
381                        setTeleportedModeSpeed(key.substring(TELEPORTED_MODE_SPEEDS.length()), Double.parseDouble(value));
382                } else if (key.startsWith(TELEPORTED_MODE_FREESPEED_FACTORS)) {
383                        setTeleportedModeFreespeedFactor(key.substring(TELEPORTED_MODE_FREESPEED_FACTORS.length()), Double.parseDouble(value));
384                } else {
385                        throw new IllegalArgumentException(key);
386                }
387        }
