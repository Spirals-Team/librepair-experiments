083        //Vehicles of size (in PCU) smaller than or equal to this threshold will be allowed to enter the buffer even
084        //if flowcap_accumulate <= 0.
085        //The default value is 0.0 meaning that all vehicles of non-zero sizes will be let into buffer only
086        //if flowcap_accumulate > 0.
087        //
088        //Flow capacity easing prevents buses from waiting long time for entering the buffer in sub-sampled scenarios
089        //For instance, for 10% scenario, a car (representing 10 cars) has a size of 1.0 PCU, a bus (representing 1 bus)
090        //has a size of 0.3 PCU, and link flow capacities are reduced to about 0.1 of the original capacity.
091        //(1) If pcuThresholdForFlowCapacityEasing == 0, all buses moving just behind private cars wait long times before
092        //entering the buffer (recovering the flow capacity accumulator in a 10% scenario takes approx. 10 times longer than
093        //in the 100% scenario).
094        //(2) If pcuThresholdForFlowCapacityEasing == 0.3, buses at the front of the queue immediately enter the buffer
095        //(once they arrive at the end of a link). If they (one or more buses) have been queued behind a private car,
096        //they all leave the link at the same time as the preceding private car.
097        private double pcuThresholdForFlowCapacityEasing = 0.0;

244                                options += behav + " " ;
245                        }
246                        map.put(VEHICLE_BEHAVIOR, "Defines what happens if an agent wants to depart, but the specified vehicle is not available. " +

248                }
249                map.put(MAIN_MODE, "[comma-separated list] Defines which modes are congested modes. Technically, these are the modes that " +
250                                "the departure handler of the netsimengine handles.  Effective cell size, effective lane width, flow capacity " +

267                                + "flows are updated only if an agent wants to enter the link or an agent is added to buffer. "
268                                + "Default is true.") ;
269                map.put(USE_LANES, "Set this parameter to true if lanes should be used, false if not.");
270                {       
271                        StringBuilder stb = new StringBuilder() ;
272                        for ( VehiclesSource src : VehiclesSource.values() ) {
273                                stb.append(" ").append( src.toString() ) ;
274                        }
275                        map.put( VEHICLES_SOURCE, "If vehicles should all be the same default vehicle, or come from the vehicles file, "

279                map.put(IS_SEEP_MODE_STORAGE_FREE, "If link dynamics is set as "+ LinkDynamics.SeepageQ+", set to true if seep mode do not consumes any space on the link. Default is false.");

282                                "this will be overridden if vehicle source is "+ VehiclesSource.fromVehiclesData+".");

284        }
