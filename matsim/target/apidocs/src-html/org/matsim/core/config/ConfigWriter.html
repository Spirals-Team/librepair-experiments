021package org.matsim.core.config;
022
023import java.io.BufferedWriter;
024import java.io.IOException;
025import java.util.Iterator;
026
027import org.matsim.core.api.internal.MatsimWriter;
028import org.matsim.core.utils.io.MatsimXmlWriter;
029import org.matsim.core.utils.io.UncheckedIOException;
030
031public final class ConfigWriter extends MatsimXmlWriter implements MatsimWriter {
032        
033        public static enum Verbosity {all, minimal } 
034
035        //////////////////////////////////////////////////////////////////////
036        // member variables
037        //////////////////////////////////////////////////////////////////////
038
039        private final Config config;
040        private ConfigWriterHandler handler = null;
041        private String dtd = null;
042
043        //////////////////////////////////////////////////////////////////////
044        // constructors
045        //////////////////////////////////////////////////////////////////////
046        public ConfigWriter(final Config config, final Verbosity verbosity ) {
047                this.config = config;
048                // always write the latest version, currently v2
049                this.dtd = "http://www.matsim.org/files/dtd/config_v2.dtd";
050                this.handler = new ConfigWriterHandlerImplV2(verbosity);
051        }
052        public ConfigWriter(final Config config) {
053                this( config, Verbosity.all ) ;
054        }
055
056        //////////////////////////////////////////////////////////////////////
057        // write methods
058        //////////////////////////////////////////////////////////////////////
059
060        public final void writeStream(final java.io.Writer writer) {
061                try {
062                        this.writer = new BufferedWriter(writer);
063                        write();
064                        this.writer.flush();
065                        this.writer = null;
066                } catch (IOException e) {
067                        throw new UncheckedIOException(e);
068                }
069        }
070
071        public final void writeStream(final java.io.Writer writer, final String newline) {
072                try {
073                        final String formerNewLine = this.handler.setNewline(newline);
074                        this.writer = new BufferedWriter(writer);
075                        write();
076                        this.writer.flush();
077                        this.writer = null;
078                        this.handler.setNewline( formerNewLine );
079                } catch (IOException e) {
080                        throw new UncheckedIOException(e);
081                }
082        }
083
084        @Override
085        public final void write(final String filename) throws UncheckedIOException {
086                openFile(filename);
087                write();
088                close();
089        }
090
091        public final void writeFileV1(final String filename) {
092                this.dtd = "http://www.matsim.org/files/dtd/config_v1.dtd";
093                this.handler = new ConfigWriterHandlerImplV1();
094                write( filename );
095        }
096        
097        public final void writeFileV2(final String filename) {
098                this.dtd = "http://www.matsim.org/files/dtd/config_v2.dtd";
099                this.handler = new ConfigWriterHandlerImplV2(Verbosity.all);
100                write( filename );
101        }
102
103        private void write() {
104                try {
105                        writeXmlHead();
106                        writeDoctype("config", this.dtd);
107
108                        this.handler.startConfig(this.config, this.writer);
109                        this.handler.writeSeparator(this.writer);
110                        for (ConfigGroup m : this.config.getModules().values()) {
111                                this.handler.writeModule(m, this.writer);
112                                this.handler.writeSeparator(this.writer);
113                        }
114                        this.handler.endConfig(this.writer);
115                        this.writer.flush();
116                }
117                catch (IOException e) {
118                        throw new UncheckedIOException(e);
119                }
120        }
121
122}
