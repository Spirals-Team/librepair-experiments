021package org.matsim.core.config.groups;
022
023import java.util.Map;
024
025import org.matsim.core.config.ReflectiveConfigGroup;
026
027/**
028 * @author nagel
029 *
030 */
031public final class ParallelEventHandlingConfigGroup extends ReflectiveConfigGroup {
032
033        public static final String GROUP_NAME = "parallelEventHandling";
034
035        private final static String NUMBER_OF_THREADS = "numberOfThreads";
036        private Integer numberOfThreads = null;
037        public final static String NUMBER_OF_THREADS_COMMENT = "Number of threads for parallel events handler. _null_ means the framework decides by itself. 0 is currently not possible.";
038
039        private final static String ESTIMATED_NUMBER_OF_EVENTS = "estimatedNumberOfEvents";
040        private Long estimatedNumberOfEvents = null;
041
042        private final static String SYNCHRONIZE_ON_SIMSTEPS = "synchronizeOnSimSteps"; 
043        private Boolean synchronizeOnSimSteps = true;
044        
045        private final static String ONE_THREAD_PER_HANDLER = "oneThreadPerHandler"; 
046        private Boolean oneThreadPerHandler = false;
047        
048        private boolean locked = false;
049
050        public ParallelEventHandlingConfigGroup() {
051                super(GROUP_NAME);
052        }
053
054        @Override
055        public Map<String, String> getComments() {
056                Map<String, String> comments = super.getComments();
057                comments.put(NUMBER_OF_THREADS, NUMBER_OF_THREADS_COMMENT);
058                comments.put(ESTIMATED_NUMBER_OF_EVENTS, "Estimated number of events during mobsim run. An optional optimization hint for the framework.");
059                comments.put(SYNCHRONIZE_ON_SIMSTEPS, "If enabled, it is ensured that all events that are created during a time step of the mobility simulation are processed "
060                                + "before the next time step is simulated. E.g. neccessary when within-day replanning is used.");
061                comments.put(ONE_THREAD_PER_HANDLER, "If enabled, each event handler is assigned to its own thread. Note that enabling this feature disabled the " + NUMBER_OF_THREADS + " option! "
062                                + "This feature is still experimental!");
063                return comments;
064        }
065
066        /**
067         * {@value #NUMBER_OF_THREADS_COMMENT}
068         */
069        @StringGetter( NUMBER_OF_THREADS )
070        public Integer getNumberOfThreads() {
071                return numberOfThreads;
072        }
073
074        /**
075         * {@value #NUMBER_OF_THREADS_COMMENT}
076         * 
077         * @param numberOfThreads
078         */
079        @StringSetter( NUMBER_OF_THREADS )
080        public void setNumberOfThreads(Integer numberOfThreads) {
081                if ( !this.locked ) {
082                        this.numberOfThreads = numberOfThreads;
083                } else {
084                        throw new RuntimeException("it is too late in the control flow to modify this parameter");
085                }
086        }
087
088        @StringGetter( ESTIMATED_NUMBER_OF_EVENTS )
089        public Long getEstimatedNumberOfEvents() {
090                return estimatedNumberOfEvents;
091        }
092
093        @StringSetter( ESTIMATED_NUMBER_OF_EVENTS )
094        public void setEstimatedNumberOfEvents(Long estimatedNumberOfEvents) {
095                if ( !this.locked ) {
096                        this.estimatedNumberOfEvents = estimatedNumberOfEvents;
097                } else {
098                        throw new RuntimeException("it is too late in the control flow to modify this parameter");
099                }
100        }
101        
102        @StringGetter( SYNCHRONIZE_ON_SIMSTEPS )
103        public Boolean getSynchronizeOnSimSteps() {
104                return this.synchronizeOnSimSteps;
105        }
106
107        @StringSetter( SYNCHRONIZE_ON_SIMSTEPS )
108        public void setSynchronizeOnSimSteps(Boolean synchronizeOnSimSteps) {
109                if ( !this.locked ) {
110                        this.synchronizeOnSimSteps = synchronizeOnSimSteps;
111                } else {
112                        throw new RuntimeException("it is too late in the control flow to modify this parameter");
113                }
114        }
115
116        @StringGetter( ONE_THREAD_PER_HANDLER )
117        public Boolean getOneThreadPerHandler() {
118                return this.oneThreadPerHandler;
119        }
120
121        @StringSetter( ONE_THREAD_PER_HANDLER )
122        public void setOneThreadPerHandler(Boolean oneThreadPerHandler) {
123                if ( !this.locked ) {
124                        this.oneThreadPerHandler = oneThreadPerHandler;
125                } else {
126                        throw new RuntimeException("it is too late in the control flow to modify this parameter");
127                }
128        }
129        
130        public void makeLocked() {
131                this.locked = true;
132        }
133}
