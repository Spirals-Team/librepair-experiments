046        @Override
047        public void checkConsistency(final Config config) {
048                ConfigConsistencyCheckerImpl.checkScenarioFeaturesEnabled(config);
049                ConfigConsistencyCheckerImpl.checkEventsFormatLanesSignals(config);
050                ConfigConsistencyCheckerImpl.checkTravelTimeCalculationRoutingConfiguration(config);
051                ConfigConsistencyCheckerImpl.checkLaneDefinitionRoutingConfiguration(config);
052                ConfigConsistencyCheckerImpl.checkPlanCalcScore(config);
053                ConfigConsistencyCheckerImpl.checkTransit(config);
054        }

056        /*package*/ static void checkPlanCalcScore(final Config c) {
057                ModeParams ptModeParams = c.planCalcScore().getModes().get(TransportMode.pt);
058                if (ptModeParams!=null && ptModeParams.getMarginalUtilityOfTraveling() > 0) {
059                        log.warn(PlanCalcScoreConfigGroup.GROUP_NAME + ".travelingPt is > 0. This values specifies a utility. " +
060                                        "Typically, this should be a disutility, i.e. have a negative value.");
061                }
062                ModeParams carModeParams = c.planCalcScore().getModes().get(TransportMode.car);
063                if (carModeParams!=null && carModeParams.getMarginalUtilityOfTraveling() > 0) {
064                        log.warn(PlanCalcScoreConfigGroup.GROUP_NAME + ".traveling is > 0. This values specifies a utility. " +
065                        "Typically, this should be a disutility, i.e. have a negative value.");
066                }
067                ModeParams bikeModeParams = c.planCalcScore().getModes().get(TransportMode.bike);
068                if (bikeModeParams!=null && bikeModeParams.getMarginalUtilityOfTraveling() > 0) {
069                        log.warn(PlanCalcScoreConfigGroup.GROUP_NAME + ".travelingBike is > 0. This values specifies a utility. " +
070                        "Typically, this should be a disutility, i.e. have a negative value.");
071                }
072                ModeParams walkModeParams = c.planCalcScore().getModes().get(TransportMode.walk);
073                if (walkModeParams!=null && walkModeParams.getMarginalUtilityOfTraveling() > 0) {
074                        log.warn(PlanCalcScoreConfigGroup.GROUP_NAME + ".travelingWalk is > 0. This values specifies a utility. " +
075                        "Typically, this should be a disutility, i.e. have a negative value.");
076                }
077                
078                        
079                ActivityParams ptAct = c.planCalcScore().getActivityParams(PtConstants.TRANSIT_ACTIVITY_TYPE) ;
080                if ( ptAct != null ) {
081//                      if ( ptAct.getClosingTime()!=0. && ptAct.getClosingTime()!=Time.UNDEFINED_TIME ) {
082//                              if ( !c.vspExperimental().isAbleToOverwritePtInteractionParams()==true ) {
083//                                      throw new RuntimeException("setting the pt interaction activity closing time away from 0/undefined is not allowed because it breaks pt scoring." +
084//                                      " If you need this anyway (for backwards compatibility reasons), you can allow this by a parameter in VspExperimentalConfigGroup.") ;
085//                              }
086//                      }
087                        if ( ptAct.isScoringThisActivityAtAll() ) {
088                                if ( !c.vspExperimental().isAbleToOverwritePtInteractionParams()==true ) {
089                                        throw new RuntimeException("Scoring " + ptAct.getActivityType() + " is not allowed because it breaks pt scoring." +
090                                        " If you need this anyway (for backwards compatibility reasons), you can allow this by a parameter in VspExperimentalConfigGroup.") ;
091                                }
092                        }
093                }
094                
095        }

097        private static void checkEventsFormatLanesSignals(final Config c) {
098                if (c.qsim().isUseLanes()) {
099                        if (!c.controler().getEventsFileFormats().contains(EventsFileFormat.xml)){
100                                log.error("Xml events are not enabled, but lanes and possibly signal systems" +
101                                                "are enalbed. Events from this features will only be written to the xml format, consider" +
102                                                "to add xml events in the controler config module");
103                        }
104                }
105        }
106
107        private static void checkScenarioFeaturesEnabled(final Config c) {
108                if (! ("qsim".equals(c.controler().getMobsim()) ||  c.qsim() != null)){
109                  log.warn("The signal system implementation is only supported by the org.matsim.ptproject.qsim mobility simulation that is not activated. Please make sure you are using the correct" +
110                                "mobility simulation. This warning can be ingored if a customized mobility simulation developed outside of org.matsim is used and set correctly.");
111                }
112        }
113
114
115        private static void checkTravelTimeCalculationRoutingConfiguration(final Config config){
116                if (config.controler().isLinkToLinkRoutingEnabled() &&
117                                !config.travelTimeCalculator().isCalculateLinkToLinkTravelTimes()){
118                        throw new IllegalStateException("LinkToLinkRouting is activated in config and" +
119                                        " link to link traveltime calculation is not enabled but required!");
120                }
121
122                if (config.travelTimeCalculator().isCalculateLinkTravelTimes() &&
123                                config.travelTimeCalculator().isCalculateLinkToLinkTravelTimes() &&
124                                !config.controler().isLinkToLinkRoutingEnabled()) {
125                        log.warn("Config enables link travel time calculation and link to link " +
126                                        "travel time calculation. This requires at least twice as much memory as " +
127                                        "if only one method is used, however it might be necessary to enable " +
128                                        "a certain module configuration.");
129                }
130
131                if (!config.travelTimeCalculator().isCalculateLinkTravelTimes()){
132                        log.warn("Link travel time calculation is switched off, be aware that this optimization" +
133                                        "might not work with all modules. ");
134                }
135
136                if (config.travelTimeCalculator().isCalculateLinkToLinkTravelTimes() &&
137                                config.qsim().isRemoveStuckVehicles()){
138                        throw new IllegalStateException("Link to link travel time calculation is not" +
139                                        "available if using the remove stuck vehicles option!");
140                }
141
142                if ( config.controler().isLinkToLinkRoutingEnabled() &&
143                                config.controler().getRoutingAlgorithmType() != RoutingAlgorithmType.Dijkstra ) {
144                        log.warn("We don't know if non-Dijkstra routing works together with LinkToLink routing.");
145                }
146                
147        }
148
149
150        private static void checkLaneDefinitionRoutingConfiguration(final Config config) {
151                if ((config.qsim().isUseLanes()) &&
152                    !config.controler().isLinkToLinkRoutingEnabled()){
153                        log.warn("Using lanes without enabling linktolinkrouting might not lead to expected simulation results");
154                }
155        }
156
157        private static void checkTransit(final Config config) {
158                if ( config.transit().isUseTransit() && config.transit().getVehiclesFile()==null ) {
159                        log.warn("Your are using Transit but have not provided a transit vehicles file. This most likely won't work.");
160                }
161        }
