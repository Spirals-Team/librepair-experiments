020package org.matsim.core.network.algorithms;
021
022import java.util.HashSet;
023import java.util.Map;
024import java.util.Set;
025
026import org.matsim.api.core.v01.Id;
027import org.matsim.api.core.v01.network.Link;
028import org.matsim.api.core.v01.network.Network;
029import org.matsim.api.core.v01.network.NetworkFactory;
030import org.matsim.api.core.v01.network.Node;
031import org.matsim.core.network.NetworkUtils;
032import org.matsim.utils.objectattributes.attributable.AttributesUtils;
033
034/**
035 * This class extracts a subnetwork from a given network containing only
036 * those links where at least one of the given transport modes are allowed.
037 * The resulting network will not contain any links where none of the
038 * specified transport modes are allowed. In addition, all links in the
039 * resulting network will have at most those modes specified for the
040 * extraction, additional modes are removed from the allowed set for each
041 * link.<br />
042 * This class makes no guarantee that the resulting network is strongly
043 * connected, not even when the input network was strongly connected.
044 *
045 * @author mrieser
046 */
047public final class TransportModeNetworkFilter {
048
049        private final Network fullNetwork;
050
051        public TransportModeNetworkFilter(final Network fullNetwork) {
052                this.fullNetwork = fullNetwork;
053        }
054
055        /**
056         * Extracts a subnetwork containing only links with the specified modes.
057         * 
058         * I had to extend this method in order to keep the nodes in the same order as in
059         * the input network. Otherwise, some router tests failed since for some from-to-pairs,
060         * multiple routes with the same costs were found. In that case, the outcome depends
061         * on the order of the nodes and links in the network. This problem might occur also
062         * in other places, therefore, I fixed it here.
063         * cdobler, sep'17
064         *
065         * @param subNetwork the network object where to store the extracted subnetwork
066         * @param extractModes set of modes that should be contained in the subnetwork
067         */
068        public void filter(final Network subNetwork, final Set<String> extractModes) {  
069                NetworkFactory factory = subNetwork.getFactory();
070
071                // first, clone all nodes to ensure their order is not changed
072                for (Node node : this.fullNetwork.getNodes().values()) {
073                        Node newNode = factory.createNode(node.getId(), node.getCoord());
074                        AttributesUtils.copyAttributesFromTo(node, newNode);
075                        subNetwork.addNode(newNode);
076                }
077
078                // second, create clones of the links allowing the extracted modes
079                Set<Id<Node>> nodesToInclude = new HashSet<>();
080                for (Link link : this.fullNetwork.getLinks().values()) {
081                        Set<String> intersection = new HashSet<>(extractModes);
082                        intersection.retainAll(link.getAllowedModes());
083                        if (intersection.size() > 0) {
084                                Id<Node> fromId = link.getFromNode().getId();
085                                Id<Node> toId = link.getToNode().getId();
086                                Node fromNode2 = subNetwork.getNodes().get(fromId);
087                                Node toNode2 = subNetwork.getNodes().get(toId);
088                                nodesToInclude.add(fromId);
089                                nodesToInclude.add(toId);
090
091                                Link link2 = factory.createLink(link.getId(), fromNode2, toNode2);
092                                link2.setAllowedModes(intersection);
093                                link2.setCapacity(link.getCapacity());
094                                link2.setFreespeed(link.getFreespeed());
095                                link2.setLength(link.getLength());
096                                link2.setNumberOfLanes(link.getNumberOfLanes());
097                                NetworkUtils.setType(link2, NetworkUtils.getType(link));
098                                AttributesUtils.copyAttributesFromTo(link, link2);
099                                subNetwork.addLink(link2);
100                        }
101                }
102                
103                // third, remove all nodes that are not used by the valid links
104                Set<Id<Node>> nodesToRemove = new HashSet<>();
105                for (Node node : this.fullNetwork.getNodes().values()) {
106                        if (!nodesToInclude.contains(node.getId())) nodesToRemove.add(node.getId());
107                }
108                for (Id<Node> nodeId : nodesToRemove) subNetwork.removeNode(nodeId);
109        }
110}
