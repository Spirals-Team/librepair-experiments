021package org.matsim.core.network.algorithms;
022
023import java.io.FileInputStream;
024
025import org.apache.log4j.Logger;
026import org.geotools.data.shapefile.dbf.DbaseFileReader;
027import org.matsim.api.core.v01.Id;
028import org.matsim.api.core.v01.network.Link;
029import org.matsim.api.core.v01.network.Network;
030import org.matsim.core.api.internal.NetworkRunnable;
031import org.matsim.core.network.io.NetworkReaderTeleatlas;
032import org.matsim.core.utils.io.IOUtils;
033
034/**
035 * Adds additional speed restrictions to a MATSim {@link Network network} created
036 * by {@link NetworkReaderTeleatlas}. The input speed restriction DBF file is based on
037 * <strong>Tele Atlas MultiNet Shapefile 4.3.2.1 Format Specifications
038 * document version Final v1.0, June 2007</strong>.
039 *
040 * @author balmermi
041 */
042public final class NetworkTeleatlasAddSpeedRestrictions implements NetworkRunnable {
043
044        //////////////////////////////////////////////////////////////////////
045        // member variables
046        //////////////////////////////////////////////////////////////////////
047
048        private final static Logger log = Logger.getLogger(NetworkTeleatlasAddSpeedRestrictions.class);
049
050        /**
051         * path and name to the Tele Atlas MultiNet speed restriction (sr) DBF file
052         */
053        private final String srDbfFileName;
054
055        private static final String SR_ID_NAME = "ID";
056        private static final String SR_SPEED_NAME = "SPEED";
057        private static final String SR_VALDIR_NAME = "VALDIR";
058        private static final String SR_VERIFIED_NAME = "VERIFIED";
059
060        //////////////////////////////////////////////////////////////////////
061        // constructors
062        //////////////////////////////////////////////////////////////////////
063
064        /**
065         * To add speed restrictions to a Tele Atlas MultiNet {@link Network network}.
066         *
067         * @param srDbfFileName Tele Atlas MultiNet speed restriction DBF file
068         */
069        public NetworkTeleatlasAddSpeedRestrictions(final String srDbfFileName) {
070                log.info("init " + this.getClass().getName() + " module...");
071                this.srDbfFileName = srDbfFileName;
072                log.info("done.");
073        }
074
075        //////////////////////////////////////////////////////////////////////
076        // run method
077        //////////////////////////////////////////////////////////////////////
078
079        /**
080         * Reading and assigning speed restrictions to the {@link Link links} of a {@link Network network}.
081         *
082         * <p>It uses the following attributes from the Tele Atlas MultiNet speed restriction DBF file:
083         * <ul>
084         *   <li>{@link #SR_ID_NAME} (Feature Identification)</li>
085         *   <li>{@link #SR_SPEED_NAME} (Speed Restriction)</li>
086         *   <li>
087         *     {@link #SR_VALDIR_NAME} (Validity Direction)
088         *     <ul>
089         *       <li>1: Valid in Both Directions</li>
090         *       <li>2: Valid Only in Positive Direction</li>
091         *       <li>3: Valid Only in Negative Direction</li>
092         *     </ul>
093         *   </li>
094         *   <li>
095         *     {@link #SR_VERIFIED_NAME} (Verified)
096         *     <ul>
097         *       <li>0: Not Verified (default)</li>
098         *       <li>1: Verified</li>
099         *     </ul>
100         *   </li>
101         * </ul></p>
102         * <p><b>Conversion rules:</b>
103         * <ul>
104         *   <li>speed restrictions that are not verified will be ignored.</li>
105         *   <li>speed restrictions will be assigned in given directions ({@link #SR_VALDIR_NAME})</li>
106         *   <li>speed restrictions will be ignored if the corresponding {@link Link link} is not found (produces a trace message).</li>
107         *   <li>speed restrictions will not be assigned to the {@link Link link} if it already contains a speed that
108         *   is lower then the one from the speed restrictions file.</li>
109         * </ul></p>
110         *
111         * @param network
112         * @throws RuntimeException with another Exception in it in the case something goes wrong
113         */
114        @Override
115        public void run(final Network network) {
116                try {
117                        run2(network);
118                } catch (Exception e) {
119                        throw new RuntimeException(e);
120                }
121        }
122
123        private void run2(final Network network) throws Exception {
124                log.info("running " + this.getClass().getName() + " module...");
125                try (FileInputStream fis = new FileInputStream(this.srDbfFileName)) {
126                        DbaseFileReader r = new DbaseFileReader(fis.getChannel(), true, IOUtils.CHARSET_WINDOWS_ISO88591);
127                        // getting header indices
128                        int srIdNameIndex = -1;
129                        int srSpeedNameIndex = -1;
130                        int srValDirNameIndex = -1;
131                        int srVerifiedNameIndex = -1;
132                        for (int i=0; i<r.getHeader().getNumFields(); i++) {
133                                if (r.getHeader().getFieldName(i).equals(SR_ID_NAME)) { srIdNameIndex = i; }
134                                if (r.getHeader().getFieldName(i).equals(SR_SPEED_NAME)) { srSpeedNameIndex = i; }
135                                if (r.getHeader().getFieldName(i).equals(SR_VALDIR_NAME)) { srValDirNameIndex = i; }
136                                if (r.getHeader().getFieldName(i).equals(SR_VERIFIED_NAME)) { srVerifiedNameIndex = i; }
137                        }
138                        if (srIdNameIndex < 0) { throw new NoSuchFieldException("Field name '"+SR_ID_NAME+"' not found."); }
139                        if (srSpeedNameIndex < 0) { throw new NoSuchFieldException("Field name '"+SR_SPEED_NAME+"' not found."); }
140                        if (srValDirNameIndex < 0) { throw new NoSuchFieldException("Field name '"+SR_VALDIR_NAME+"' not found."); }
141                        if (srVerifiedNameIndex < 0) { throw new NoSuchFieldException("Field name '"+SR_VERIFIED_NAME+"' not found."); }
142                        log.trace("  FieldName-->Index:");
143                        log.trace("    "+SR_ID_NAME+"-->"+srIdNameIndex);
144                        log.trace("    "+SR_SPEED_NAME+"-->"+srSpeedNameIndex);
145                        log.trace("    "+SR_VALDIR_NAME+"-->"+srValDirNameIndex);
146                        log.trace("    "+SR_VERIFIED_NAME+"-->"+srVerifiedNameIndex);
147        
148                        int srCnt = 0;
149                        int srIgnoreCnt = 0;
150                        while (r.hasNext()) {
151                                Object[] entries = r.readEntry();
152                                int verified = Integer.parseInt(entries[srVerifiedNameIndex].toString());
153                                // assign only verfied speed restricitons
154                                if (verified == 1) {
155                                        int valdir = Integer.parseInt(entries[srValDirNameIndex].toString());
156                                        String id = entries[srIdNameIndex].toString();
157                                        if (valdir == 1) {
158                                                // Valid in Both Directions
159                                                Link ftLink = network.getLinks().get(Id.create(id+"FT", Link.class));
160                                                Link tfLink = network.getLinks().get(Id.create(id+"TF", Link.class));
161                                                if ((ftLink == null) || (tfLink == null)) { log.trace("  linkid="+id+", valdir="+valdir+": at least one link not found. Ignoring and proceeding anyway..."); srIgnoreCnt++; }
162                                                else {
163                                                        double speed = Double.parseDouble(entries[srSpeedNameIndex].toString())/3.6;
164                                                        // assigning speed restriction only if given freespeed is higher
165                                                        if (speed < ftLink.getFreespeed()) { ftLink.setFreespeed(speed); srCnt++; } else { srIgnoreCnt++; }
166                                                        if (speed < tfLink.getFreespeed()) { tfLink.setFreespeed(speed); srCnt++; } else { srIgnoreCnt++; }
167                                                }
168                                        }
169                                        else if (valdir == 2) {
170                                                // Valid Only in Positive Direction
171                                                Link ftLink = network.getLinks().get(Id.create(id+"FT", Link.class));
172                                                if (ftLink == null) { log.trace("  linkid="+id+", valdir="+valdir+": link not found. Ignoring and proceeding anyway..."); srIgnoreCnt++; }
173                                                else {
174                                                        double speed = Double.parseDouble(entries[srSpeedNameIndex].toString())/3.6;
175                                                        // assigning speed restriction only if given freespeed is higher
176                                                        if (speed < ftLink.getFreespeed()) { ftLink.setFreespeed(speed); srCnt++; } else { srIgnoreCnt++; }
177                                                }
178                                        }
179                                        else if (valdir == 3) {
180                                                // Valid Only in Negative Direction
181                                                Link tfLink = network.getLinks().get(Id.create(id+"TF", Link.class));
182                                                if (tfLink == null) { log.trace("  linkid="+id+", valdir="+valdir+": link not found. Ignoring and proceeding anyway..."); srIgnoreCnt++; }
183                                                else {
184                                                        double speed = Double.parseDouble(entries[srSpeedNameIndex].toString())/3.6;
185                                                        // assigning speed restriction only if given freespeed is higher
186                                                        if (speed < tfLink.getFreespeed()) { tfLink.setFreespeed(speed); srCnt++; } else { srIgnoreCnt++; }
187                                                }
188                                        }
189                                        else { throw new IllegalArgumentException("linkid="+id+": valdir="+valdir+" not known."); }
190                                }
191                        }
192                        log.info("  "+srCnt+" links with restricted speed assigned.");
193                        log.info("  "+srIgnoreCnt+" speed restrictions ignored (while verified = 1).");
194                        log.info("done.");
195                        r.close();
196                }
197        }
198
199        //////////////////////////////////////////////////////////////////////
200        // print methods
201        //////////////////////////////////////////////////////////////////////
202
203        /**
204         * prints the variable settings to the STDOUT
205         *
206         * @param prefix a prefix for each line of the STDOUT
207         */
208        public final void printInfo(final String prefix) {
209                System.out.println(prefix+"configuration of "+this.getClass().getName()+":");
210                System.out.println(prefix+"  speed restrictions:");
211                System.out.println(prefix+"    srDbfFileName:    "+this.srDbfFileName);
212                System.out.println(prefix+"    SR_ID_NAME:       "+SR_ID_NAME);
213                System.out.println(prefix+"    SR_SPEED_NAME:    "+SR_SPEED_NAME);
214                System.out.println(prefix+"    SR_VALDIR_NAME:   "+SR_VALDIR_NAME);
215                System.out.println(prefix+"    SR_VERIFIED_NAME: "+SR_VERIFIED_NAME);
216                System.out.println(prefix+"done.");
217        }
218}
