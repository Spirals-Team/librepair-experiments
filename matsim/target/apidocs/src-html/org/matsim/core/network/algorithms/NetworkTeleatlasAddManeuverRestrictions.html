045/**
046 * Adds maneuver restrictions to a MATSim {@link Network network} created
047 * by {@link NetworkReaderTeleatlas}. The input maneuver shape file and the maneuver path DBF file
048 * is based on <strong>Tele Atlas MultiNet Shapefile 4.3.2.1 Format Specifications
049 * document version Final v1.0, June 2007</strong>.
050 *
051 * @author balmermi
052 */
053public final class NetworkTeleatlasAddManeuverRestrictions implements NetworkRunnable {
054
055        //////////////////////////////////////////////////////////////////////
056        // member variables
057        //////////////////////////////////////////////////////////////////////
058
059        private final static Logger log = Logger.getLogger(NetworkTeleatlasAddManeuverRestrictions.class);
060
061        /**
062         * path and name to the Tele Atlas MultiNet maneuver (mn) Shape file
063         */
064        private final String mnShpFileName; // teleatlas maneuvers shape file name
065
066        /**
067         * path and name to the Tele Atlas MultiNet maneuver path (mp) DBF file
068         */
069        private final String mpDbfFileName; // teleatlas maneuver paths dbf file name
070
071        /**
072         * option flag: if set, expanded {@link Node nodes} (nodes that contains maneuver restrictions)
073         * will not include new virtual {@link Link links} providing a u-turn maneuver at the
074         * junction.
075         */
076        public boolean removeUTurns = false;
077
078        /**
079         * option parameter: defines the radius how much a {@link Node node} will be expanded.
080         * <p><b>Default:</b> <code>0.000030</code> (appropriate for world coordinate system WGS84)</p>
081         *
082         * @see NetworkExpandNode
083         */
084        public double expansionRadius = 0.000030; // WGS84
085
086        /**
087         * option parameter: defines the offset of a in- and out-link pair of an expanded {@link Node node}.
088         * <p><b>Default:</b> <code>0.000005</code> (appropriate for world coordinate system WGS84)</p>
089         *
090         * @see NetworkExpandNode
091         */
092        public double linkSeparation = 0.000005; // WGS84
093
094        private static final String MN_ID_NAME = "ID";
095        private static final String MN_FEATTYP_NAME = "FEATTYP";
096        private static final String MN_JNCTID_NAME = "JNCTID";
097
098        private static final String MP_ID_NAME = "ID";
099        private static final String MP_SEQNR_NAME = "SEQNR";
100        private static final String MP_TRPELID_NAME = "TRPELID";
101
102        //////////////////////////////////////////////////////////////////////
103        // constructors
104        //////////////////////////////////////////////////////////////////////
105
106        /**
107         * To create maneuver restrictions to a Tele Atlas MultiNet {@link Network network}.
108         *
109         * @param mnShpFileName Tele Atlas MultiNet maneuver Shape file
110         * @param mpDbfFileName Tele Atlas MultiNet maneuver path DBF file
111         */
112        public NetworkTeleatlasAddManeuverRestrictions(final String mnShpFileName, final String mpDbfFileName) {
113                log.info("init " + this.getClass().getName() + " module...");
114                this.mnShpFileName = mnShpFileName;
115                this.mpDbfFileName = mpDbfFileName;
116                log.info("done.");
117        }
118
119        //////////////////////////////////////////////////////////////////////
120        // run method
121        //////////////////////////////////////////////////////////////////////
122
123        /**
124         * Reading and assigning (expanding {@link Node nodes}) maneuver restrictions to the {@link Network network}.
125         *
126         * <p>It uses the following attributes from the Tele Atlas MultiNet maneuver Shape file:
127         * <ul>
128         *   <li>{@link #MN_ID_NAME} (Feature Identification)</li>
129         *   <li>
130         *     {@link #MN_FEATTYP_NAME} (Feature Type)
131         *     <ul>
132         *       <li>9401: Bifurcation</li>
133         *       <li>2104: Priority Maneuver</li>
134         *       <li>2103: Prohibited Maneuver</li>
135         *       <li>2102: Restricted Maneuver</li>
136         *       <li>2101: Calculated/Derived Prohibited Maneuver</li>
137         *     </ul>
138         *   </li>
139         *   <li>{@link #MN_JNCTID_NAME} (Junction Identification of the Location of the Maneuver Sign)</li>
140         * </ul></p>
141         * <p>And it uses the following attributes from the Tele Atlas MultiNet maneuver path DBF file:
142         * <ul>
143         *   <li>{@link #MP_ID_NAME} (Feature Identification)</li>
144         *   <li>{@link #MP_SEQNR_NAME} (Transportation Element Sequential Number within the Maneuver)</li>
145         *   <li>{@link #MP_TRPELID_NAME} (Transportation Element Identification)</li>
146         * </ul></p>
147         * <p><b>Conversion rules:</b>
148         * <ul>
149         *   <li>The maneuver types 'Bifurcation' (9401) and 'Priority Maneuver' (2104) are ignored</li>
150         *   <li>Maneuver restriction sequences that contain links that are not incident to the given
151         *   junction are ignored</li>
152         *   <li>Maneuver restrictions will be modelled in the natwork topology by expanding
153         *   {@link Node nodes} with {@link NetworkExpandNode node expanding procedure}.</li>
154         * </ul></p>
155         *
156         * @param network MATSim {@link Network network} created by {@link NetworkReaderTeleatlas}.
157         * @throws RuntimeException with another Exception in it in the case something goes wrong
158         */
159        @Override
160        public void run(final Network network) {
161                try {
162                        run2(network);
163                } catch (Exception e) {
164                        throw new RuntimeException(e);
165                }
166        }
167
168        private void run2(final Network network) throws Exception {
169                log.info("running " + this.getClass().getName() + " module...");
170                NetworkExpandNode neModule = new NetworkExpandNode(network, expansionRadius, this.linkSeparation);
171
172                TreeMap<String, TreeMap<Integer,Id<Link>>> mSequences = new TreeMap<>();
173                
174                try (FileInputStream fis = new FileInputStream(this.mpDbfFileName)) {
175                        DbaseFileReader r = new DbaseFileReader(fis.getChannel(), true, IOUtils.CHARSET_WINDOWS_ISO88591);
176                        // get header indices
177                        int mpIdNameIndex = -1;
178                        int mpSeqNrNameIndex = -1;
179                        int mpTrpelIDNameIndex = -1;
180                        for (int i=0; i<r.getHeader().getNumFields(); i++) {
181                                if (r.getHeader().getFieldName(i).equals(MP_ID_NAME)) { mpIdNameIndex = i; }
182                                if (r.getHeader().getFieldName(i).equals(MP_SEQNR_NAME)) { mpSeqNrNameIndex = i; }
183                                if (r.getHeader().getFieldName(i).equals(MP_TRPELID_NAME)) { mpTrpelIDNameIndex = i; }
184                        }
185                        if (mpIdNameIndex < 0) { throw new NoSuchFieldException("Field name '"+MP_ID_NAME+"' not found."); }
186                        if (mpSeqNrNameIndex < 0) { throw new NoSuchFieldException("Field name '"+MP_SEQNR_NAME+"' not found."); }
187                        if (mpTrpelIDNameIndex < 0) { throw new NoSuchFieldException("Field name '"+MP_TRPELID_NAME+"' not found."); }
188                        log.trace("  FieldName-->Index:");
189                        log.trace("    "+MP_ID_NAME+"-->"+mpIdNameIndex);
190                        log.trace("    "+MP_SEQNR_NAME+"-->"+mpSeqNrNameIndex);
191                        log.trace("    "+MP_TRPELID_NAME+"-->"+mpTrpelIDNameIndex);
192        
193                        // create mp data structure
194                        // TreeMap<mpId,TreeMap<mpSeqNr,linkId>>
195                        log.info("  parsing meneuver paths dbf file...");
196                        while (r.hasNext()) {
197                                Object[] entries = r.readEntry();
198                                String mpId = entries[mpIdNameIndex].toString();
199                                int mpSeqNr = Integer.parseInt(entries[mpSeqNrNameIndex].toString());
200                                Id<Link> linkId = Id.create(entries[mpTrpelIDNameIndex].toString(), Link.class);
201                                TreeMap<Integer,Id<Link>> mSequence = mSequences.get(mpId);
202                                if (mSequence == null) {
203                                        mSequence = new TreeMap<>();
204                                        mSequences.put(mpId, mSequence);
205                                }
206                                if (mSequence.put(mpSeqNr,linkId) != null) {
207                                        fis.close();
208                                        throw new IllegalArgumentException(MP_ID_NAME+"="+mpId+": "+MP_SEQNR_NAME+" "+mpSeqNr+" already exists.");
209                                }
210                        }
211                        log.info("    "+mSequences.size()+" maneuvers sequences stored.");
212                        log.info("  done.");
213                        r.close();
214                }
215
216                // store the maneuver list of the nodes
217                // TreeMap<NodeId,ArrayList<Tuple<MnId,MnFeatType>>>
218                log.info("  parsing meneuver shape file...");
219                TreeMap<Id<Node>, ArrayList<Tuple<String, Integer>>> maneuvers = new TreeMap<>();
220                SimpleFeatureSource fs = ShapeFileReader.readDataFile(this.mnShpFileName);
221                SimpleFeatureIterator fIt = fs.getFeatures().features();
222                while (fIt.hasNext()) {
223                        SimpleFeature f = fIt.next();
224                        int featType = Integer.parseInt(f.getAttribute(MN_FEATTYP_NAME).toString());
225                        if ((featType == 2103) || (featType == 2102) || (featType == 2101)) {
226                                // keep 'Prohibited Maneuver' (2103), 'Restricted Maneuver' (2102) and 'Calculated/Derived Prohibited Maneuver' (2101)
227                                Id<Node> nodeId = Id.create(f.getAttribute(MN_JNCTID_NAME).toString(), Node.class);
228                                ArrayList<Tuple<String, Integer>> ms = maneuvers.get(nodeId);
229                                if (ms == null) {
230                                        ms = new ArrayList<>();
231                                }
232                                Tuple<String, Integer> m = new Tuple<>(f.getAttribute(MN_ID_NAME).toString(), featType);
233                                ms.add(m);
234                                maneuvers.put(nodeId, ms);
235                        }
236                        else if ((featType == 9401) || (featType == 2104)) {
237                                //ignore 'Bifurcation' (9401) and 'Priority Maneuver' (2104)
238                        }
239                        else {
240                                throw new IllegalArgumentException("mnId="+f.getAttribute(MN_ID_NAME)+": "+MN_FEATTYP_NAME+"="+featType+" not known.");
241                        }
242                }
243                fIt.close();
244                log.info("    "+maneuvers.size()+" nodes with maneuvers stored.");
245                log.info("  done.");
246
247                // create a maneuver matrix for each given node and
248                // expand those nodes
249                log.info("  expand nodes according to the given manveuvers...");
250                int nodesIgnoredCnt = 0;
251                int nodesAssignedCnt = 0;
252                int maneuverIgnoredCnt = 0;
253                int maneuverAssignedCnt = 0;
254                int virtualNodesCnt = 0;
255                int virtualLinksCnt = 0;
256                for (Map.Entry<Id<Node>, ArrayList<Tuple<String, Integer>>> entry : maneuvers.entrySet()) {
257                        Id<Node> nodeId = entry.getKey();
258                        if (network.getNodes().get(nodeId) == null) { 
259                                log.trace("  nodeid="+nodeId+": maneuvers exist for that node but node is missing. Ignoring and proceeding anyway..."); 
260                                nodesIgnoredCnt++;
261                        } else {
262                                // node found
263                                Node n = network.getNodes().get(nodeId);
264                                // init maneuver matrix
265                                // TreeMap<fromLinkId,TreeMap<toLinkId,turnAllowed>>
266                                TreeMap<Id<Link>, TreeMap<Id<Link>, Boolean>> mmatrix = new TreeMap<>();
267                                // assign maneuvers for given node to the matrix
268                                ArrayList<Tuple<String, Integer>> ms = entry.getValue();
269                                for (Tuple<String, Integer> m : ms) {
270                                        // get maneuver path sequence for given maneuver
271                                        TreeMap<Integer,Id<Link>> mSequence = mSequences.get(m.getFirst());
272                                        if (mSequence == null) { throw new Exception("nodeid="+nodeId+"; mnId="+m.getFirst()+": no maneuver sequence given."); }
273                                        if (mSequence.size() < 2) { throw new Exception("nodeid="+nodeId+"; mnId="+m.getFirst()+": mSequenceSize="+mSequence.size()+" not alowed!"); }
274                                        // get the first element of the sequence, defining the start link for the maneuver
275                                        Id<Link> firstLinkid = mSequence.values().iterator().next();
276                                        // go through each other element (target link of the maneuver) of the sequence by sequence number
277                                        for (Id<Link> otherLinkId : mSequence.values()) {
278                                                // get the start link and the target link of the maneuver
279                                                Link inLink = n.getInLinks().get(Id.create(firstLinkid+"FT", Link.class));
280                                                if (inLink == null) {
281                                                        inLink = n.getInLinks().get(Id.create(firstLinkid+"TF", Link.class));
282                                                }
283                                                Link outLink = n.getOutLinks().get(Id.create(otherLinkId+"FT", Link.class));
284                                                if (outLink == null) { 
285                                                        outLink = n.getOutLinks().get(Id.create(otherLinkId+"TF", Link.class));
286                                                }
287                                                if ((inLink != null) && (outLink != null)) {
288                                                        // start and target link found and they are incident to the given node
289                                                        if (m.getSecond() == 2102) {
290                                                                // restricted maneuver: given start and target link path is allowed to drive
291                                                                // store it to the matrix
292                                                                TreeMap<Id<Link>, Boolean> outLinkMap = mmatrix.get(inLink.getId());
293                                                                if (outLinkMap == null) {
294                                                                        outLinkMap = new TreeMap<>();
295                                                                }
296                                                                outLinkMap.put(outLink.getId(), Boolean.TRUE);
297                                                                mmatrix.put(inLink.getId(),outLinkMap);
298                                                        }
299                                                        else {
300                                                                // prohibited maneuver: given start and target link path is not allowed to drive
301                                                                // store it to the matrix
302                                                                TreeMap<Id<Link>,Boolean> outLinkMap = mmatrix.get(inLink.getId());
303                                                                if (outLinkMap == null) {
304                                                                        outLinkMap = new TreeMap<>();
305                                                                }
306                                                                outLinkMap.put(outLink.getId(), Boolean.FALSE);
307                                                                mmatrix.put(inLink.getId(),outLinkMap);
308                                                        }
309                                                        maneuverAssignedCnt++;
310                                                }
311                                                else { maneuverIgnoredCnt++; }
312                                        }
313                                }
314                                // complete the matrix
315                                for (TreeMap<Id<Link>, Boolean> fromLinkEntry : mmatrix.values()) {
316                                        // detect inlinks with restricted maneuvers
317                                        boolean hasRestrictedManeuver = false;
318                                        for (Boolean b : fromLinkEntry.values()) {
319                                                if (b) { hasRestrictedManeuver = true; }
320                                        }
321                                        // add missing toLink maneuvers
322                                        for (Id<Link> toLinkId : n.getOutLinks().keySet()) {
323                                                if (!fromLinkEntry.containsKey(toLinkId)) {
324                                                        fromLinkEntry.put(toLinkId, !hasRestrictedManeuver);
325                                                }
326                                        }
327                                }
328                                // add allowed maneuvers for fromLinks which were not assigned yet.
329                                for (Id<Link> fromLinkId : n.getInLinks().keySet()) {
330                                        if (!mmatrix.containsKey(fromLinkId)) {
331                                                mmatrix.put(fromLinkId, new TreeMap<Id<Link>, Boolean>());
332                                                for (Id<Link> toLinkId : n.getOutLinks().keySet()) {
333                                                        mmatrix.get(fromLinkId).put(toLinkId, Boolean.TRUE);
334                                                }
335                                        }
336                                }
337                                // remove all U-turns from the matrix
338                                if (this.removeUTurns) {
339                                        for (Id<Link> fromLinkId : n.getInLinks().keySet()) {
340                                                String str1 = fromLinkId.toString().substring(0,fromLinkId.toString().length()-2);
341                                                for (Id<Link> toLinkId : n.getOutLinks().keySet()) {
342                                                        String str2 = toLinkId.toString().substring(0,toLinkId.toString().length()-2);
343                                                        if (str1.equals(str2)) {
344                                                                mmatrix.get(fromLinkId).put(toLinkId, Boolean.FALSE);
345                                                        }
346                                                }
347                                        }
348                                }
349                                // create arraylist with turn tuples
350                                ArrayList<TurnInfo> turns = new ArrayList<>();
351                                for (Map.Entry<Id<Link>, TreeMap<Id<Link>, Boolean>> fromLinkEntry : mmatrix.entrySet()) {
352                                        Id<Link> fromLinkId = fromLinkEntry.getKey();
353                                        for (Map.Entry<Id<Link>, Boolean> toLinkEntry : fromLinkEntry.getValue().entrySet()) {
354                                                if (toLinkEntry.getValue()) {
355                                                        turns.add(new TurnInfo(fromLinkId, toLinkEntry.getKey()));
356                                                }
357                                        }
358                                }
359                                // expand the node
360                                Tuple<List<Node>, List<Link>> t = neModule.expandNode(nodeId, turns);
361                                virtualNodesCnt += t.getFirst().size();
362                                virtualLinksCnt += t.getSecond().size();
363                                nodesAssignedCnt++;
364                        }
365                }
366                log.info("    "+nodesAssignedCnt+" nodes expanded.");
367                log.info("    "+maneuverAssignedCnt+" maneuvers assigned.");
368                log.info("    "+virtualNodesCnt+" new nodes created.");
369                log.info("    "+virtualLinksCnt+" new links created.");
370                log.info("    "+nodesIgnoredCnt+" nodes with given maneuvers (2103, 2102 or 2101) ignored.");
371                log.info("    "+maneuverIgnoredCnt+" maneuvers ignored (while node was found).");
372                log.info("  done.");
373                log.info("done.");
374        }
