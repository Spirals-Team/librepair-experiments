021package org.matsim.core.network.algorithms;
022
023import java.util.HashSet;
024import java.util.Set;
025
026import org.apache.log4j.Logger;
027import org.matsim.api.core.v01.Coord;
028import org.matsim.api.core.v01.network.Network;
029import org.matsim.api.core.v01.network.Node;
030import org.matsim.core.api.internal.NetworkRunnable;
031import org.matsim.core.utils.geometry.CoordUtils;
032
033public final class NetworkScenarioCut implements NetworkRunnable {
034
035        private enum CutType {RECTANGLE, CIRCLE}
036
037    private final CutType cutType;
038        
039        private final double minX;
040        private final double maxX;
041        private final double minY;
042        private final double maxY;
043        
044        private final double radius;
045        private final Coord center;
046
047        private final static Logger log = Logger.getLogger(NetworkScenarioCut.class);
048
049        public NetworkScenarioCut(final Coord min, final Coord max) {
050                super();
051                
052                this.cutType = CutType.RECTANGLE;
053                
054                this.minX = min.getX();
055                this.maxX = max.getX();
056                this.minY = min.getY();
057                this.maxY = max.getY();
058                
059                this.radius = Double.MAX_VALUE;
060                this.center = null;
061        }
062
063        public NetworkScenarioCut(final Coord center, final double radius) {
064                super();
065                
066                this.cutType = CutType.CIRCLE;
067                
068                this.center = center;
069                this.radius = radius;
070                
071                this.minX = Double.MIN_VALUE;
072                this.maxX = Double.MAX_VALUE;
073                this.minY = Double.MIN_VALUE;
074                this.maxY = Double.MAX_VALUE;
075        }
076        
077        @Override
078        public void run(final Network network) {
079                
080                Set<Node> nodesToRemove; 
081                
082                if (this.cutType == CutType.RECTANGLE) nodesToRemove = rectangularCut(network);
083                else if (this.cutType == CutType.CIRCLE) nodesToRemove = circularCut(network);
084                else return;
085                
086                int nofLinksRemoved = 0;
087                for (Node n : nodesToRemove) {
088                        nofLinksRemoved += n.getInLinks().size() + n.getOutLinks().size();
089                        network.removeNode(n.getId());
090                }
091                
092                log.info("number of nodes removed: "+nodesToRemove.size());
093                log.info("number of links removed: "+nofLinksRemoved);
094                log.info("number of nodes remaining: "+network.getNodes().size());
095                log.info("number of links remaining: "+network.getLinks().size());
096        }
097        
098        private Set<Node> rectangularCut(Network network) {
099                Set<Node> nodesToRemove = new HashSet<>();
100                for (Node n : network.getNodes().values()) {
101                        Coord coord = n.getCoord();
102                        double x = coord.getX();
103                        double y = coord.getY();
104                        if (!((x < this.maxX) && (this.minX < x) && (y < this.maxY) && (this.minY < y))) {
105                                nodesToRemove.add(n);
106                        }
107                }
108                return nodesToRemove;
109        }
110        
111        private Set<Node> circularCut(Network network) {
112                Set<Node> nodesToRemove = new HashSet<>();
113                for (Node n : network.getNodes().values()) {
114                        Coord coord = n.getCoord();
115                        double distance = CoordUtils.calcEuclideanDistance(coord, center);
116                        if (distance > radius) {
117                                nodesToRemove.add(n);
118                        }
119                }
120                return nodesToRemove;
121        }
122}
