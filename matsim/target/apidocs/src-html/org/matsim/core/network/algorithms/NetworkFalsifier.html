021package org.matsim.core.network.algorithms;
022
023import org.matsim.api.core.v01.Coord;
024import org.matsim.api.core.v01.network.Link;
025import org.matsim.api.core.v01.network.Network;
026import org.matsim.api.core.v01.network.Node;
027import org.matsim.core.api.internal.NetworkRunnable;
028import org.matsim.core.gbl.MatsimRandom;
029import org.matsim.core.utils.geometry.CoordUtils;
030
031/**
032 * Falsifies a network, so it can more legally be redistributed, by moving the nodes by a random amount
033 * north or south and east or west from their original point, but at most <code>distance</code> away.
034 * Additionally, the link length will be set to the euclidean distance between the from- and to-node.
035 *
036 * @author mrieser
037 */
038public final class NetworkFalsifier implements NetworkRunnable {
039
040        private final double distance;
041
042        public NetworkFalsifier(double distance) {
043                this.distance = distance;
044        }
045
046        @Override
047        public void run(Network network) {
048                double maxDistance = this.distance * 2.0;
049                for (Node node : network.getNodes().values()) {
050                        Coord coord = node.getCoord();
051                        node.setCoord( new Coord(coord.getX() + (MatsimRandom.getRandom().nextDouble() - 0.5) *  maxDistance,
052                                        coord.getY() + (MatsimRandom.getRandom().nextDouble() - 0.5) * maxDistance) ) ;
053                }
054
055                for (Link link : network.getLinks().values()) {
056                        Coord fromCoord = link.getFromNode().getCoord();
057                        Coord toCoord = link.getToNode().getCoord();
058                        link.setLength(CoordUtils.calcEuclideanDistance(fromCoord, toCoord));
059                }
060        }
061
062}
