021package org.matsim.core.network.algorithms;
022
023import org.matsim.api.core.v01.Coord;
024import org.matsim.api.core.v01.network.Link;
025import org.matsim.api.core.v01.network.Network;
026import org.matsim.api.core.v01.network.Node;
027import org.matsim.core.api.internal.NetworkRunnable;
028import org.matsim.core.utils.misc.Time;
029
030public final class NetworkSummary implements NetworkRunnable {
031
032        private int network_capacity = 0;
033        private double minX = Double.POSITIVE_INFINITY;
034        private double maxX = Double.NEGATIVE_INFINITY;
035        private double minY = Double.POSITIVE_INFINITY;
036        private double maxY = Double.NEGATIVE_INFINITY;
037
038        public NetworkSummary() {
039                super();
040        }
041
042        @Override
043        public void run(final Network network) {
044                System.out.println("    running " + this.getClass().getName() + " algorithm...");
045
046                int min_node_id = Integer.MAX_VALUE;
047                int max_node_id = Integer.MIN_VALUE;
048                int node_cnt = 0;
049                int min_link_id = Integer.MAX_VALUE;
050                int max_link_id = Integer.MIN_VALUE;
051                int link_cnt = 0;
052
053                for (Node node : network.getNodes().values()) {
054                        node_cnt++;
055                        int node_getID = Integer.parseInt(node.getId().toString());
056                        if (min_node_id > node_getID) { min_node_id = node_getID; }
057                        if (max_node_id < node_getID) { max_node_id = node_getID; }
058                        double x = node.getCoord().getX();
059                        double y = node.getCoord().getY();
060                        if (x > this.maxX) { this.maxX = x; }
061                        if (x < this.minX) { this.minX = x; }
062                        if (y > this.maxY) { this.maxY = y; }
063                        if (y < this.minY) { this.minY = y; }
064                }
065
066                double cellSize = network.getEffectiveCellSize();
067                for (Link link : network.getLinks().values()) {
068                        link_cnt++;
069                        int link_getID = Integer.parseInt(link.getId().toString());
070                        if (min_link_id > link_getID) { min_link_id = link_getID; }
071                        if (max_link_id < link_getID) { max_link_id = link_getID; }
072                        this.network_capacity += Math.ceil(link.getLength()/cellSize);
073                }
074
075                System.out.println("      network summary:");
076//              System.out.println("        name             = " + network.getName());
077                System.out.println("        capperiod        = " + Time.writeTime(network.getCapacityPeriod()));
078                System.out.println("        network_capacity = " + this.network_capacity + " cells");
079                System.out.println("      nodes summary:");
080                System.out.println("        number of nodes = " + node_cnt);
081                System.out.println("        min node id     = " + min_node_id);
082                System.out.println("        max node id     = " + max_node_id);
083                System.out.println("      links summary:");
084                System.out.println("        number of links = " + link_cnt);
085                System.out.println("        min link id     = " + min_link_id);
086                System.out.println("        max link id     = " + max_link_id);
087
088                System.out.println("    done.");
089        }
090
091        public final int getNetworkCapacity() {
092                return this.network_capacity;
093        }
094
095        public final Coord getMinCoord() {
096                return new Coord(this.minX, this.minY);
097        }
098
099        public final Coord getMaxCoord() {
100                return new Coord(this.maxX, this.maxY);
101        }
102}
