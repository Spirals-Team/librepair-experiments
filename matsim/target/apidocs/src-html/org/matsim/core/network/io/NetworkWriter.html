021package org.matsim.core.network.io;
022
023import java.io.IOException;
024import java.util.HashMap;
025import java.util.Map;
026
027import org.apache.log4j.Logger;
028import org.matsim.api.core.v01.network.Link;
029import org.matsim.api.core.v01.network.Network;
030import org.matsim.api.core.v01.network.Node;
031import org.matsim.core.api.internal.MatsimWriter;
032import org.matsim.core.network.NetworkUtils;
033import org.matsim.core.utils.geometry.CoordinateTransformation;
034import org.matsim.core.utils.geometry.transformations.IdentityTransformation;
035import org.matsim.core.utils.io.MatsimXmlWriter;
036import org.matsim.core.utils.io.UncheckedIOException;
037import org.matsim.utils.objectattributes.AttributeConverter;
038
039public final class NetworkWriter extends MatsimXmlWriter implements MatsimWriter {
040        
041        private static final Logger log = Logger.getLogger(NetworkWriter.class);
042        
043        private final Network network;
044        private final CoordinateTransformation transformation;
045        private final Map<Class<?>,AttributeConverter<?>> converters = new HashMap<>();
046
047        public NetworkWriter(final Network network) {
048                this( new IdentityTransformation() , network );
049        }
050
051        public NetworkWriter(
052                        final CoordinateTransformation transformation,
053                        final Network network) {
054                this.transformation = transformation;
055                this.network = network;
056        }
057
058        public void putAttributeConverters(final Map<Class<?>, AttributeConverter<?>> converters) {
059                this.converters.putAll( converters );
060        }
061
062        public void putAttributeConverter(Class<?> clazz , AttributeConverter<?> converter) {
063                this.converters.put(  clazz , converter );
064        }
065
066        @Override
067        public void write(final String filename) {
068                log.info("Writing network to file: " + filename  + "...");
069                // always write out in newest version, currently v2
070                writeFileV2(filename);
071                log.info("done.");
072        }
073
074        public void writeFileV1(final String filename) {
075                String dtd = "http://www.matsim.org/files/dtd/network_v1.dtd";
076                NetworkWriterHandler handler = new NetworkWriterHandlerImplV1(transformation);
077
078                writeFile( dtd , handler , filename );
079        }
080
081        public void writeFileV2(final String filename) {
082                String dtd = "http://www.matsim.org/files/dtd/network_v2.dtd";
083                NetworkWriterHandlerImplV2 handler = new NetworkWriterHandlerImplV2(transformation);
084
085                handler.putAttributeConverters( converters );
086
087                writeFile( dtd , handler , filename );
088        }
089
090        private void writeFile(
091                        final String dtd,
092                        final NetworkWriterHandler handler,
093                        final String filename) {
094
095                try {
096                        openFile(filename);
097                        writeXmlHead();
098                        writeDoctype("network", dtd);
099
100                        handler.startNetwork(network, this.writer);
101                        handler.writeSeparator(this.writer);
102                        handler.startNodes(network, this.writer);
103                        for (Node n : NetworkUtils.getSortedNodes(network)) {
104                                handler.startNode(n, this.writer);
105                                handler.endNode(this.writer);
106                        }
107                        handler.endNodes(this.writer);
108                        handler.writeSeparator(this.writer);
109                        handler.startLinks(network, this.writer);
110                        for (Link l : NetworkUtils.getSortedLinks(network)) {
111                                handler.startLink(l, this.writer);
112                                handler.endLink(this.writer);
113                        }
114                        handler.endLinks(this.writer);
115                        handler.writeSeparator(this.writer);
116                        handler.endNetwork(this.writer);
117                        this.writer.close();
118                }
119                catch (IOException e) {
120                        throw new UncheckedIOException(e);
121                }
122        }
123}
