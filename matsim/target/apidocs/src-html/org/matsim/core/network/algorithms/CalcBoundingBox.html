021package org.matsim.core.network.algorithms;
022
023import org.matsim.api.core.v01.network.Network;
024import org.matsim.api.core.v01.network.Node;
025import org.matsim.core.api.internal.NetworkRunnable;
026
027/**
028 * Calculates the bounding box (=maximal extent) of a network.
029 *
030 * @author mrieser
031 */
032public final class CalcBoundingBox implements NetworkRunnable {
033
034        private double minX = Double.POSITIVE_INFINITY;
035        private double minY = Double.POSITIVE_INFINITY;
036        private double maxX = Double.NEGATIVE_INFINITY;
037        private double maxY = Double.NEGATIVE_INFINITY;
038
039        @Override
040        public void run(final Network network) {
041                this.minX = Double.POSITIVE_INFINITY;
042                this.minY = Double.POSITIVE_INFINITY;
043                this.maxX = Double.NEGATIVE_INFINITY;
044                this.maxY = Double.NEGATIVE_INFINITY;
045                for (Node n : network.getNodes().values()) {
046                        if (n.getCoord().getX() < this.minX) { this.minX = n.getCoord().getX(); }
047                        if (n.getCoord().getY() < this.minY) { this.minY = n.getCoord().getY(); }
048                        if (n.getCoord().getX() > this.maxX) { this.maxX = n.getCoord().getX(); }
049                        if (n.getCoord().getY() > this.maxY) { this.maxY = n.getCoord().getY(); }
050                }
051        }
052
053        public double getMinX() {
054                return this.minX;
055        }
056
057        public double getMaxX() {
058                return this.maxX;
059        }
060
061        public double getMinY() {
062                return this.minY;
063        }
064
065        public double getMaxY() {
066                return this.maxY;
067        }
068
069}
