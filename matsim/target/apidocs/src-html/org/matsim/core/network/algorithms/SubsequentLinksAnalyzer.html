021package org.matsim.core.network.algorithms;
022
023import org.matsim.api.core.v01.Coord;
024import org.matsim.api.core.v01.Id;
025import org.matsim.api.core.v01.network.Link;
026import org.matsim.api.core.v01.network.Network;
027import org.matsim.api.core.v01.network.Node;
028import org.matsim.core.network.LinkIdComparator;
029
030import java.util.*;
031
032/**
033 * Analyzes the network to find for each link the "logical subsequent link",
034 * i.e. the link that most people are likely to continue driving on when
035 * crossing the next node.  This could depend on the geometry (assuming it
036 * is more likely to continue straight on instead of making a U-turn), the
037 * capacity of links (likely to continue on links with the same or larger
038 * capacity instead of using small bumpy roads), the speed limit, other
039 * attributes or combinations there of. This code only looks at geometry.
040 *
041 * @author ychen
042 * @author mrieser
043 */
044public final class SubsequentLinksAnalyzer {
045
046        private final Network network;
047
048        /** Stores the logical subsequent link (value) for a given link (key). */
049        private final TreeMap<Id<Link>, Id<Link>> subsequentLinks = new TreeMap<>();
050
051        public SubsequentLinksAnalyzer(final Network network) {
052                this.network = network;
053                compute();
054        }
055
056        /**
057         * @return a map, giving for each link (key in map) the computed subsequent link (value in map).
058         */
059        public Map<Id<Link>, Id<Link>> getSubsequentLinks() {
060                return this.subsequentLinks;
061        }
062
063        /**
064         * calculates the "default next" link of the current link with respect to
065         * geometry and Capacity (depending on BETA) and writes the result in
066         * ssLinks.
067         */
068        private void compute() {
069                Map<Link, Double> absDeltaThetas = new TreeMap<>(new LinkIdComparator());
070                for (Link l : this.network.getLinks().values()) {
071                        Node from = l.getFromNode();
072                        Node to = l.getToNode();
073                        Coord cFrom = from.getCoord();
074                        Coord cTo = to.getCoord();
075                        double xTo = cTo.getX();
076                        double yTo = cTo.getY();
077                        double thetaL = Math.atan2(yTo - cFrom.getY(), xTo - cFrom.getX());
078                        Collection<? extends Link> outLinks = to.getOutLinks().values();
079                        absDeltaThetas.clear();
080                        if (outLinks.size() > 1) {
081
082                                for (Link out : outLinks) {
083                                        Coord cOut = out.getToNode().getCoord();
084                                        double deltaTheta = Math.atan2(cOut.getY() - yTo, cOut.getX()   - xTo) - thetaL;
085                                        while (deltaTheta < -Math.PI) {
086                                                deltaTheta += 2.0 * Math.PI;
087                                        }
088                                        while (deltaTheta > Math.PI) {
089                                                deltaTheta -= 2.0 * Math.PI;
090                                        }
091                                        absDeltaThetas.put(out, Math.abs(deltaTheta));
092                                }
093                                this.subsequentLinks.put(l.getId(), computeSubsequentLink(absDeltaThetas).getId());
094                        } else if (outLinks.size() == 1) {
095                                this.subsequentLinks.put(l.getId(), outLinks.iterator().next().getId());
096                        }
097                }
098        }
099
100        /**
101         * @param thetas map of outgoing-link and theta of that link. must include at least one entry, otherwise it won't work!
102         * @return the link with the smallest theta, if multiple links have the same smallest theta, the one with the biggest capacity is returned.
103         */
104        private Link computeSubsequentLink(final Map<Link, Double> thetas) {
105                List<Link> minThetaOutLinks = new ArrayList<>();
106
107                minThetaOutLinks.clear();
108                double absMin = Collections.min(thetas.values());
109                for (Map.Entry<Link, Double> entry : thetas.entrySet()) {
110                        if (absMin == entry.getValue()) {
111                                minThetaOutLinks.add(entry.getKey());
112                        }
113                }
114                if (minThetaOutLinks.size() == 1) {
115                        return minThetaOutLinks.get(0);
116                }
117
118                double maxCapacity = Double.NEGATIVE_INFINITY;
119                Link maxCapLink = null;
120                for (Link link : minThetaOutLinks) {
121                        double cap = link.getCapacity();
122                        if (cap > maxCapacity)  {
123                                maxCapacity = cap;
124                                maxCapLink = link;
125                        }
126                }
127                return maxCapLink;
128        }
129
130}
