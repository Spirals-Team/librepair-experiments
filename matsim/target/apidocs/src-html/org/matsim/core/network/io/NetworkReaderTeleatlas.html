039/**
040 * A reader for TeleAtlas network description files. The reader is based on
041 * <strong>Tele Atlas MultiNet Shapefile 4.3.2.1 Format Specifications document
042 * version Final v1.0, June 2007</strong> and uses the input shape files:
043 * <ul>
044 * <li><em>jc.shp</em>: junction shape file (typically called xyz________jc.shp)
045 * </li>
046 * <li><em>nw.shp</em>: junction shape file (typically called xyz________nw.shp)
047 * </li>
048 * </ul>
049 * 
050 * @author balmermi
051 */
052public final class NetworkReaderTeleatlas implements MatsimSomeReader {
053

063         * path and name to the Tele Atlas MultiNet junction (jc) Shapefile
064         */
065        private final String jcShpFileName; // teleatlas junction shape file name
066
067        /**
068         * path and name to the Tele Atlas MultiNet network (nw) Shapefile
069         */
070        private final String nwShpFileName; // teleatlas network shape file name

072        /**
073         * option flag: if set, the reader ignores all network links with
074         * <p>
075         * <code>{@link #LINK_FRCTYP_NAME} == 8</code>
076         * </p>
077         * <p>
078         * <b>Default:</b> <code>false</code>
079         * </p>
080         */
081        public boolean ignoreFrcType8 = false;
082
083        /**
084         * option flag: if set, the reader ignores all network links with
085         * <p>
086         * <code>{@link #LINK_FRCTYP_NAME} == 8 && {@link #LINK_ONEWAY_NAME} == "N"</code>
087         * </p>
088         * <p>
089         * <b>Default:</b> <code>false</code>
090         * </p>
091         */
092        public boolean ignoreFrcType7onewayN = false;
093
094        /**
095         * option parameter: the reader redefines the number of lanes (see permlanes
096         * in {@link Link}) if the value of
097         * <code>{@link #LINK_LANES_NAME}<1</code>. In that case the link gets 2 lanes
098         * for <code>{@link #LINK_FRCTYP_NAME}</code> less or equal this parameter, 1
099         * lane otherwise.
100         * <p>
101         * <b>Default:</b> <code>3</code>
102         * </p>
103         */
104        public int maxFrcTypeForDoubleLaneLink = 3;
105
106        /**
107         * option parameter [in km/h]: Tele Atlas MultiNet Shapefile does not define
108         * capacities for links. Therefore, link flow capacities must be derived from
109         * the given data. Below the given parameter the link gets capacity = 1000
110         * [veh/h] per lane assigned, 2000 [veh/h] otherwise.
111         * <p>
112         * <b>Default:</b> <code>40</code>(km/h)
113         * </p>
114         */
115        public int minSpeedForNormalCapacity = 40; // km/h

125        private static final String LINK_TJNCTID_NAME = "T_JNCTID";

174        public void read() throws IOException {
175                log.info("reading nodes from Junction Shape file '" + this.jcShpFileName + "'...");
176                this.readNodesFromJCshp();
177                log.info("done.");
178                log.info("reading links from Network Shape file '" + this.nwShpFileName + "'...");
179                this.readLinksFromNWshp();
180                log.info("done.");
181                // TODO balmermi: adding date and check if you get more info from the input
182                // files to include into the name
183                network.setName("teleatlas");
184        }

188        /**
189         * Reads Tele Atlas MultiNet junction Shapefile given in
190         * <code>{@link #jcShpFileName}</code>.
191         * 
192         * <p>
193         * It uses the following feature attributes:
194         * <ul>
195         * <li>center coordinate</li>
196         * <li><code>{@link #NODE_ID_NAME} (Feature Identification)</code></li>
197         * <li><code>{@link #NODE_FEATTYP_NAME} (Feature Type)</code></li>
198         * <ul>
199         * <li><code>4120: Junction</code></li>
200         * <li><code>4220: Railway Element Junction</code></li>
201         * </ul>
202         * <li><code>{@link #NODE_JNCTTYP_NAME} (Junction Type)</code></li>
203         * <ul>
204         * <li><code>0: Junction (default)</code></li>
205         * <li><code>2: Bifurcation</code></li>
206         * <li><code>3: Railway Crossing</code></li>
207         * <li><code>4: Country Border Crossing</code></li>
208         * <li><code>5: Ferry Operated by Train Crossing</code></li>
209         * <li><code>6: Internal Data Set Border Crossing</code></li>
210         * </ul>
211         * </ul>
212         * </p>
213         * The MATSim {@link Node#type} is set as
214         * 
215         * <pre>
216         * <code>{@link Node#type} = {@link #NODE_FEATTYP_NAME}+"-"+{@link #NODE_JNCTTYP_NAME}</code>
217         * </pre>
218         *
219         */
220        private void readNodesFromJCshp() throws IOException {
221                int nCnt = network.getNodes().size();
222                SimpleFeatureSource fs = ShapeFileReader.readDataFile(jcShpFileName);
223                SimpleFeatureIterator fIt = fs.getFeatures().features();
224                while (fIt.hasNext()) {
225                        SimpleFeature f = fIt.next();
226                        // get node attributes
227                        // Coordinate c = f.getBounds().centre();
228                        BoundingBox bb = f.getBounds();
229                        Coord c = new Coord((bb.getMinX() + bb.getMaxX()) / 2.0, (bb.getMinY() + bb.getMaxY()) / 2.0);
230
231                        Object id = f.getAttribute(NODE_ID_NAME);
232                        int feattyp = Integer.parseInt(f.getAttribute(NODE_FEATTYP_NAME).toString());
233                        if ((feattyp != 4120) && (feattyp != 4220)) {
234                                throw new IllegalArgumentException(NODE_ID_NAME + "=" + id + ": " + NODE_FEATTYP_NAME + "=" + feattyp + " not allowed.");
235                        }
236                        int jncttyp = Integer.parseInt(f.getAttribute(NODE_JNCTTYP_NAME).toString());
237                        if ((jncttyp < 0) || (jncttyp == 1) || (jncttyp > 6)) {
238                                throw new IllegalArgumentException(NODE_ID_NAME + "=" + id + ": " + NODE_JNCTTYP_NAME + "=" + jncttyp + " not allowed.");
239                        }
240                        if (id == null) {
241                                throw new IllegalArgumentException("In " + jcShpFileName + ": There is at least one feature that does not have an ID set.");
242                        }
243                        String type = feattyp + "-" + jncttyp;
244                        Node n = network.getFactory().createNode(Id.create(id.toString(), Node.class), c);
245                        final String type1 = type;
246                        Node r = n;
247                        NetworkUtils.setType(r,type1);
248                        network.addNode(n);
249                }
250                fIt.close();
251                nCnt = network.getNodes().size() - nCnt;
252                log.info("  " + nCnt + " nodes added to the network.");
253        }

257        /**
258         * Reads Tele Atlas MultiNet network Shapefile given in
259         * <code>{@link #nwShpFileName}</code>.
260         * 
261         * <p>
262         * It uses the following feature attributes:
263         * <ul>
264         * <li><code>{@link #LINK_ID_NAME} (Feature Identification)</code></li>
265         * <li><code>{@link #LINK_FEATTYP_NAME} (Feature Type)</code></li>
266         * <ul>
267         * <li><code>4110: Road Element</code></li>
268         * <li><code>4130: Ferry Connection Element</code></li>
269         * <li><code>4165: Address Area Boundary Element</code></li>
270         * </ul>
271         * <li><code>{@link #LINK_FERRYTYP_NAME} (Ferry Type)</code></li>
272         * <ul>
273         * <li><code>0: No Ferry (default)</code></li>
274         * <li><code>1: Ferry Operated by Ship or Hovercraft</code></li>
275         * <li><code>2: Ferry Operated by Train</code></li>
276         * </ul>
277         * <li>
278         * <code>{@link #LINK_FJNCTID_NAME} (From (Start) Junction Identification)</code>
279         * </li>
280         * <li>
281         * <code>{@link #LINK_TJNCTID_NAME} (To (End) Junction Identification)</code></li>
282         * <li><code>{@link #LINK_LENGTH_NAME} (Feature Length (meters))</code></li>
283         * <li><code>{@link #LINK_FRCTYP_NAME} (Functional Road Class)</code></li>
284         * <li><code>{@link #LINK_FERRYTYP_NAME} (Ferry Type)</code></li>
285         * <ul>
286         * <li><code>-1: Not Applicable (for {@link #LINK_FEATTYP_NAME} 4165)</code></li>
287         * <li><code>0: Motorway, Freeway, or Other Major Road</code></li>
288         * <li><code>1: a Major Road Less Important than a Motorway</code></li>
289         * <li><code>2: Other Major Road</code></li>
290         * <li><code>3: Secondary Road</code></li>
291         * <li><code>4: Local Connecting Road</code></li>
292         * <li><code>5: Local Road of High Importance</code></li>
293         * <li><code>6: Local Road</code></li>
294         * <li><code>7: Local Road of Minor Importance</code></li>
295         * <li><code>8: Other Road</code></li>
296         * </ul>
297         * <li><code>{@link #LINK_ONEWAY_NAME} (Direction of Traffic Flow)</code></li>
298         * <ul>
299         * <li><code>Blank: Open in Both Directions (default)</code></li>
300         * <li><code>FT: Open in Positive Direction</code></li>
301         * <li><code>N: Closed in Both Directions</code></li>
302         * <li><code>TF: Open in Negative Direction</code></li>
303         * </ul>
304         * <li>
305         * <code>{@link #LINK_SPEED_NAME} (Calculated Average Speed (kilometers per hour))</code>
306         * </li>
307         * <li><code>{@link #LINK_LANES_NAME} (Number of Lanes)</code></li>
308         * </ul>
309         * 
310         * <b>Conversion rules:</b>
311         * <ul>
312         * <li>Links that refer to not existing from- or to-link will be ignored
313         * (produces a wanring message)</li>
314         * <li>Links with {@link #LINK_FEATTYP_NAME} not equal to 4110 or 4130 will be
315         * ignored</li>
316         * <li>Links with {@link #LINK_FRCTYP_NAME} less than zero will be ignored</li>
317         * <li>Links with {@link #LINK_FRCTYP_NAME} equals 8 will be ignored if
318         * {@link #ignoreFrcType8} flag is set</li>
319         * <li>Links with {@link #LINK_FRCTYP_NAME} equals 7 and
320         * {@link #LINK_ONEWAY_NAME} equals "N" will be ignored if
321         * {@link #ignoreFrcType7onewayN} is set</li>
322         * <li>Links with {@link #LINK_LANES_NAME} less than 1 will get lanes
323         * according to the rule defined by {@link #maxFrcTypeForDoubleLaneLink}</li>
324         * <li>Links get capacities according to the rule defined by
325         * {@link #minSpeedForNormalCapacity}</li>
326         * <li>Directed links in the MATSim network DB will be created according to
327         * the information in {@link #LINK_ONEWAY_NAME} (two links if " " or "N", one
328         * link for "TF" or "FT"). The link id is defined as
329         * <code>{@link #LINK_ID_NAME}+"TF"</code>,
330         * <code>{@link #LINK_ID_NAME}+"FT"</code> resp.</li>
331         * <li>The {@link Link#type} is set as:
332         * 
333         * <pre>
334         * <code>{@link Link#type} = {@link #LINK_FRCTYP_NAME}+"-"+{@link #LINK_FEATTYP_NAME}+"-"+{@link #LINK_FERRYTYP_NAME}</code>
335         * </pre>
336         * 
337         * </li>
338         * </ul>
339         * </p>
340         * 
341         * @throws IOException
342         */
343        private void readLinksFromNWshp() throws IOException {
344                int lCnt = network.getLinks().size();
345                int ignoreCnt = 0;
346                SimpleFeatureSource fs = ShapeFileReader.readDataFile(this.nwShpFileName);
347                SimpleFeatureIterator fIt = fs.getFeatures().features();
348                while (fIt.hasNext()) {
349                        SimpleFeature f = fIt.next();
350                        boolean ignore = false;
351                        // get link attributes
352                        Object id = f.getAttribute(LINK_ID_NAME);
353                        int featTyp = Integer.parseInt(f.getAttribute(LINK_FEATTYP_NAME).toString());
354                        if ((featTyp != 4110) && (featTyp != 4130) && (featTyp != 4165)) {
355                                throw new IllegalArgumentException(LINK_ID_NAME + "=" + id + ": " + LINK_FEATTYP_NAME + "=" + featTyp + " not allowed.");
356                        }
357                        int ferryType = Integer.parseInt(f.getAttribute(LINK_FERRYTYP_NAME).toString());
358                        if ((ferryType < 0) || (ferryType > 2)) {
359                                throw new IllegalArgumentException(LINK_ID_NAME + "=" + id + ": " + LINK_FERRYTYP_NAME + "=" + ferryType + " not allowed.");
360                        }
361                        Id<Node> fromJunctionId = Id.create(f.getAttribute(LINK_FJNCTID_NAME).toString(), Node.class);
362                        Id<Node> toJunctionId = Id.create(f.getAttribute(LINK_TJNCTID_NAME).toString(), Node.class);
363                        double length = Double.parseDouble(f.getAttribute(LINK_LENGTH_NAME).toString());
364                        int linksType = Integer.parseInt(f.getAttribute(LINK_FRCTYP_NAME).toString());
365                        if ((linksType < -1) || (linksType > 8)) {
366                                throw new IllegalArgumentException(LINK_ID_NAME + "=" + id + ": " + LINK_FRCTYP_NAME + "=" + linksType + " not allowed.");
367                        }
368                        String oneway = f.getAttribute(LINK_ONEWAY_NAME).toString();
369                        if (!oneway.equals(" ") && !oneway.equals("FT") && !oneway.equals("TF") && !oneway.equals("N")) {
370                                throw new IllegalArgumentException(LINK_ID_NAME + "=" + id + ": " + LINK_ONEWAY_NAME + "=" + oneway + " not allowed.");
371                        }
372                        double speed = Double.parseDouble(f.getAttribute(LINK_SPEED_NAME).toString());
373                        double lanes = Double.parseDouble(f.getAttribute(LINK_LANES_NAME).toString());
374                        // ignore link where from node or to node is missing
375                        Node fNode = network.getNodes().get(fromJunctionId);
376                        Node tNode = network.getNodes().get(toJunctionId);
377                        if ((fNode == null) || (tNode == null)) {
378                                log.warn("  linkId=" + id.toString()
379                                                + ": at least one of the two junctions do not exist. Ignoring and proceeding anyway...");
380                                ignore = true;
381                        }
382                        // ignore link that is not a 'Road Element' (4110) or a 'Ferry Connection Element' (4130)
383                        // There are 'Address Area Boundary Element' (4165) links that will be ignored
384                        if ((featTyp != 4110) && (featTyp != 4130)) {
385                                log.trace("  linkId=" + id.toString() + ": ignoring " + LINK_FEATTYP_NAME + "=" + featTyp + ".");
386                                ignore = true;
387                        }
388                        // ignore links FRC types = -1 [Not Applicable (for FeatTyp 4165)]
389                        if (linksType < 0) {
390                                log.trace("  linkId=" + id.toString() + ": ignoring " + LINK_FRCTYP_NAME + "=" + linksType + ".");
391                                ignore = true;
392                        }
393                        // option flag: ignore links FRC type = 8 [Other Road]
394                        if (this.ignoreFrcType8 && (7 < linksType)) {
395                                log.trace("  linkId=" + id.toString() + ": ignoring " + LINK_FRCTYP_NAME + "=" + linksType + ".");
396                                ignore = true;
397                        }
398                        // ignore links FRC types = 7 [Local Road of Minor Importance] that are ONEWAY = N [Closed in Both Directions]
399                        // links with type < 7 also contains ONEWAY = N but should be open anyway....
400                        if (this.ignoreFrcType7onewayN && ((linksType == 7) && oneway.equals("N"))) {
401                                log.trace("  linkId=" + id.toString() + ": ignoring " + LINK_FRCTYP_NAME + "=" + linksType + " with " + LINK_ONEWAY_NAME
402                                                + "=" + oneway);
403                                ignore = true;
404                        }
405
406                        // simple rule for number of lanes (a lot of them are = 0, so we need to define something)
407                        if (lanes < 1) {
408                                if (linksType <= this.maxFrcTypeForDoubleLaneLink) {
409                                        lanes = 2;
410                                } else {
411                                        lanes = 1;
412                                }
413                        }
414                        // simple rule for setting capacities
415                        double cap;
416                        if (speed < minSpeedForNormalCapacity) {
417                                cap = lanes * 1000;
418                        } else {
419                                cap = lanes * 2000;
420                        }
421
422                        if (ignore) {
423                                ignoreCnt++;
424                        } else {
425                                if (oneway.equals(" ") || oneway.equals("N")) {
426                                        Link l = network.getFactory().createLink(Id.create(id.toString() + "FT", Link.class), fNode, tNode);
427                                        l.setLength(length);
428                                        l.setFreespeed(speed / 3.6);
429                                        l.setCapacity(cap);
430                                        l.setNumberOfLanes(lanes);
431                                        NetworkUtils.setOrigId(l, id.toString() ) ;
432                                        NetworkUtils.setType(l, linksType + "-" + featTyp + "-" + ferryType);
433                                        l = network.getFactory().createLink(Id.create(id.toString() + "TF", Link.class), tNode, fNode);
434                                        l.setLength(length);
435                                        l.setFreespeed(speed / 3.6);
436                                        l.setCapacity(cap);
437                                        l.setNumberOfLanes(lanes);
438                                        NetworkUtils.setOrigId(l, id.toString() ) ;
439                                        NetworkUtils.setType(l, linksType + "-" + featTyp + "-" + ferryType);
440                                } else if (oneway.equals("FT")) {
441                                        Link l = network.getFactory().createLink(Id.create(id.toString() + oneway, Link.class), fNode, tNode);
442                                        l.setLength(length);
443                                        l.setFreespeed(speed / 3.6);
444                                        l.setCapacity(cap);
445                                        l.setNumberOfLanes(lanes);
446                                        NetworkUtils.setOrigId(l, id.toString() ) ;
447                                        NetworkUtils.setType(l, linksType + "-" + featTyp + "-" + ferryType);
448                                } else if (oneway.equals("TF")) {
449                                        Link l = network.getFactory().createLink(Id.create(id.toString() + oneway, Link.class), tNode, fNode);
450                                        l.setLength(length);
451                                        l.setFreespeed(speed / 3.6);
452                                        l.setCapacity(cap);
453                                        l.setNumberOfLanes(lanes);
454                                        NetworkUtils.setOrigId(l, id.toString() ) ;
455                                        NetworkUtils.setType(l, linksType + "-" + featTyp + "-" + ferryType);
456                                } else {
457                                        throw new IllegalArgumentException("linkId=" + id.toString() + ": " + LINK_ONEWAY_NAME + "=" + oneway + " not known!");
458                                }
459                        }
460                }
461                fIt.close();
462
463                network.setCapacityPeriod(3600.0);
464
465                lCnt = network.getLinks().size() - lCnt;
466                log.info("  " + lCnt + " links added to the network layer.");
467                log.info("  " + ignoreCnt + " links ignored from the input shape file.");
468        }

477         * @param prefix

484                System.out.println(prefix + "    ignoreFrcType7onewayN:       " + ignoreFrcType7onewayN);

486                System.out.println(prefix + "    minSpeedForNormalCapacity:   " + minSpeedForNormalCapacity);
487                System.out.println(prefix + "  junction shape:");

492                System.out.println(prefix + "  network shape:");
493                System.out.println(prefix + "    nwShpFileName:      " + nwShpFileName);
494                System.out.println(prefix + "    LINK_ID_NAME:       " + LINK_ID_NAME);
495                System.out.println(prefix + "    LINK_FEATTYP_NAME:  " + LINK_FEATTYP_NAME);
496                System.out.println(prefix + "    LINK_FERRYTYP_NAME: " + LINK_FERRYTYP_NAME);
497                System.out.println(prefix + "    LINK_FJNCTID_NAME:  " + LINK_FJNCTID_NAME);
498                System.out.println(prefix + "    LINK_TJNCTID_NAME:  " + LINK_TJNCTID_NAME);
