145        @Override
146        public void endTag(String name, String content, Stack<String> context) {
147                if(name.equalsIgnoreCase(NETWORK_CHANGE_EVENT_TAG)) {
148                        events.add(currentEvent);
149                        currentEvent = null;
150                }
151        }
152
153        @Override
154        public void startTag(String name, Attributes atts, Stack<String> context) {
155                /*
156                 * NetworkChangeEvent
157                 */
158                if(name.equalsIgnoreCase(NETWORK_CHANGE_EVENT_TAG)) {
159                        String value = atts.getValue(START_TIME_TAG);
160                        if(value != null) {
161                                currentEvent = new NetworkChangeEvent(Time.parseTime(value));
162                        } else {
163                                currentEvent = null;
164                                log.warn("A start time must be defined!");
165                        }
166                /*
167                 * Links
168                 */
169                } else if(name.equalsIgnoreCase(LINK_TAG) && currentEvent != null) {
170                        String value = atts.getValue(REF_ID_TAG);
171                        if(value != null) {
172                                Link link = network.getLinks().get(Id.create(value, Link.class));
173                                if(link != null)
174                                        currentEvent.addLink(link);
175                                else
176                                        log.warn(String.format("Link %1$s not found!", value));
177                        }
178                /*
179                 * flow capacity changes 
180                 */
181                } else if(name.equalsIgnoreCase(FLOW_CAPACITY_TAG) && currentEvent != null) {
182                        currentEvent.setFlowCapacityChange(newNetworkChangeType(atts
183                                        .getValue(CHANGE_TYPE_TAG), atts.getValue(VALUE_TAG)));
184                /*
185                 * freespeed change
186                 */
187                } else if(name.equalsIgnoreCase(FREESPEED_TAG) && currentEvent != null) {
188                        currentEvent.setFreespeedChange(newNetworkChangeType(atts
189                                        .getValue(CHANGE_TYPE_TAG), atts.getValue(VALUE_TAG)));
190                /*
191                 * lanes changes
192                 */
193                } else if(name.equalsIgnoreCase(LANES_TAG) && currentEvent != null) {
194                        currentEvent.setLanesChange(newNetworkChangeType(atts
195                                        .getValue(CHANGE_TYPE_TAG), atts.getValue(VALUE_TAG)));
196                }
197                
198        }
199
200        static final String ABSOLUTE_VALUE = "absolute";
201        static final String FACTOR_VALUE = "scaleFactor";
202        static final String OFFSET_VALUE = "offset" ;
203        // the enum is different from the xml value, since the "in_SI_units" was considered important, but changing it also
204        // in the xml was considered too messy.  kai, nov'17
205        private static ChangeValue newNetworkChangeType(String typeStr, String valueStr) {
206                if(typeStr != null && valueStr != null) {
207                        double value = Double.parseDouble(valueStr);
208                        if(typeStr.equalsIgnoreCase(ABSOLUTE_VALUE))
209                                return new ChangeValue(ChangeType.ABSOLUTE_IN_SI_UNITS, value);
210                        else if(typeStr.equalsIgnoreCase(FACTOR_VALUE))
211                                return new ChangeValue(ChangeType.FACTOR, value);
212                        else if(typeStr.equalsIgnoreCase(OFFSET_VALUE))
213                                return new ChangeValue(ChangeType.OFFSET_IN_SI_UNITS, value);
214                        else {
215                                log.warn(String.format(
216                                        "The change type %1$s is not allowed. Alowed: %2$s %3$s %4$s",
217                                        typeStr, ABSOLUTE_VALUE, FACTOR_VALUE, OFFSET_VALUE ));
218                                return null;
219                        }
220                } else {
221                        log.warn("Change type and value must be specified!");
222                        return null;
223                }
224        }
