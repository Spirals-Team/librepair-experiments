008/**
009 * An optimized data structure to answer nearest-neighbor queries for links in a
010 * network. Implementation is based on the idea of a MX-CIF quadtree (Kedem,
011 * 1981).
012 * <p>
013 * "The MX-CIF quadtree of Kedem (1981) (see also Abel and Smith 1983) is a
014 * region-based representation where each rectangle is associated with the
015 * quadtree node corresponding to the smallest block which contains it in its
016 * entirety. Subdivision ceases whenever a node's block contains no rectangles.
017 * Alternatively, subdivision can also cease once a quadtree block is smaller
018 * than a predetermined threshold size." (H. Samet, Hierarchical spatial data
019 * structures, www.cs.umd.edu/~hjs/pubs/SametSSD89.pdf)
020 * </p>
021 * After having looked at the code for 5 min, I cannot, however, tell if this is
022 * exactly what the code is doing, and if so, which version. kai, jul'12
023 *
024 * The implementation splits nodes whenever a node contains more than one link
025 * of which at least one can be assigned to a child node. According to the
026 * description above, the leaf nodes would always be empty. This implementation
027 * splits nodes whenever it is necessary, but not before, leading to leaf nodes
028 * that actually may contain elements.  marcel, jul'12
029 *
030 * @author mrieser / senozon
031 */
032public final class LinkQuadTree {
033
034        private final QuadTreeNode top;
035
036        public LinkQuadTree(final double minX, final double minY, final double maxX, final double maxY) {
037                this.top = new QuadTreeNode(minX, minY, maxX, maxY);
038        }
039
040        public void put(final Link link) {
041                this.top.put(new LinkWrapper(link));
042        }
043
044        public Link getNearest(final double x, final double y) {
045                LinkWrapper w = this.top.getNearest(x, y, new MutableDouble(Double.POSITIVE_INFINITY));
046                if (w == null) {
047                        return null;
048                }
049                return w.link;
050        }
051
052        public void remove(final Link link) {
053                this.top.remove(new LinkWrapper(link));
054        }
055
056        public double getMinEasting() {
057                return this.top.minX;
058        }
059
060        public double getMaxEasting() {
061                return this.top.maxX;
062        }
063
064        public double getMinNorthing() {
065                return this.top.minY;
066        }
067
068        public double getMaxNorthing() {
069                return this.top.maxY;
070        }

072        private static class QuadTreeNode {
073
074//              private final static int NO_CHILD = -1;
075//              private final static int CHILD_NW = 0;
076//              private final static int CHILD_NE = 1;
077//              private final static int CHILD_SE = 2;
078//              private final static int CHILD_SW = 3;
079                
080                private enum ChildPosition {CHILD_NW, CHILD_NE, CHILD_SE, CHILD_SW,NO_CHILD } ;
081
082                // I replaced the "int" by an enum since I find it easier to read, and I needed/wanted to understand the code.  If this causes
083                // computational performance losses, we need to change it back.  kai, sep'16
084                
085                public final double minX;
086                public final double minY;
087                public final double maxX;
088                public final double maxY;
089
090                private final ArrayList<LinkWrapper> links = new ArrayList<>(3);
091                private QuadTreeNode[] children = null;
092
093                public QuadTreeNode(final double minX, final double minY, final double maxX, final double maxY) {
094                        this.minX = Math.min(minX, maxX);
095                        this.minY = Math.min(minY, maxY);
096                        this.maxX = Math.max(minX, maxX);
097                        this.maxY = Math.max(minY, maxY);
098                }
099
100                public void put(final LinkWrapper w) {
101                        if (this.children == null && this.links.isEmpty()) {
102                                // (means quadtree node neither has children nor contains a link yet)
103
104                                this.links.add(w);
105                                // (node now contains this link)
106                        } else {
107                                ChildPosition pos = getChildPosition(w);
108                                if (pos == ChildPosition.NO_CHILD) {
109                                        // (i.e. link extends over more than one child node, so the current node is the smallest one that fully contains the link.
110                                        this.links.add(w);
111                                } else {
112                                        if (this.children == null) {
113                                                split();
114                                        }
115                                        this.children[pos.ordinal()].put(w);
116                                }
117                        }
118                }
119
120                public boolean remove(final LinkWrapper w) {
121                        ChildPosition pos = getChildPosition(w);
122                        if (pos == ChildPosition.NO_CHILD || this.children == null) {
123                                for (int i = 0, n = this.links.size(); i < n; i++) {
124                                        LinkWrapper w2 = this.links.get(i);
125                                        if (w2.link.equals(w.link)) {
126                                                this.links.remove(i);
127                                                return true;
128                                        }
129                                }
130                        } else {
131                                return this.children[pos.ordinal()].remove(w);
132                        }
133                        return false;
134                }
135
136                /**
137                 * @param x
138                 * @param y
139                 * @param bestDistanceIndicator
140                 * @return instance of LinkWrapper that is closest to (x,y) if closer than bestPseudoDistance, null otherwise.  Adjust
141                 * bestPseudoDistance in first case
142                 */
143                public LinkWrapper getNearest(final double x, final double y, final MutableDouble bestDistanceIndicator) {
144                        LinkWrapper closest = null;
145                        
146                        // go through all links in current quadtree node (= box):
147                        for (LinkWrapper w : this.links) {
148                                double tmp = calcLineSegmentDistanceIndicator(x, y, w.link);
149                                if (tmp < bestDistanceIndicator.value) {
150                                        bestDistanceIndicator.value = tmp;
151                                        closest = w;
152                                }
153                        }
154                        
155                        // if we have child nodes (= child boxes) ...
156                        if (this.children != null) { 
157
158                                // ... find the correct one ...
159                                ChildPosition childPos = this.getChildPosition(x, y); 
160
161                                if (childPos != ChildPosition.NO_CHILD) {
162                                        // (not sure we can have children, but none in the correct "direction"; depends on exact behavior of "split")
163                                        
164                                        // Find child in correct "direction" and do recursive call:
165                                        LinkWrapper tmp = this.children[childPos.ordinal()].getNearest(x, y, bestDistanceIndicator);
166                                        if (tmp != null) {
167                                                closest = tmp;
168                                        }
169
170                                        // now also go through all _other_ children ...
171                                        for ( ChildPosition c : ChildPosition.values() ) {
172                                                if (c != childPos && c != ChildPosition.NO_CHILD ) {
173                                                        QuadTreeNode child = this.children[c.ordinal()];
174                                                        
175                                                        // For those other children, check if their bounding box is so close that we also need to look at them:
176                                                        if (child.calcDistanceIndicator(x, y) < bestDistanceIndicator.value) {
177                                                                
178                                                                // Only if the answer is yes, do a recursive call for actual LinkWrapper instances:
179                                                                tmp = child.getNearest(x, y, bestDistanceIndicator);
180                                                                if (tmp != null) {
181                                                                        closest = tmp;
182                                                                }
183                                                        }
184                                                }
185                                        }
186                                }
187                        }
188
189                        return closest;
190                }
191
192                private void split() {
193                        double centerX = (minX + maxX) / 2;
194                        double centerY = (minY + maxY) / 2;
195                        this.children = new QuadTreeNode[4];
196                        this.children[ChildPosition.CHILD_NW.ordinal()] = new QuadTreeNode(this.minX, centerY, centerX, this.maxY);
197                        this.children[ChildPosition.CHILD_NE.ordinal()] = new QuadTreeNode(centerX, centerY, this.maxX, this.maxY);
198                        this.children[ChildPosition.CHILD_SE.ordinal()] = new QuadTreeNode(centerX, this.minY, this.maxX, centerY);
199                        this.children[ChildPosition.CHILD_SW.ordinal()] = new QuadTreeNode(this.minX, this.minY, centerX, centerY);
200
201                        List<LinkWrapper> keep = new ArrayList<>(this.links.size() / 2);
202                        for (LinkWrapper w : this.links) {
203                                ChildPosition pos = getChildPosition(w);
204                                if (pos == ChildPosition.NO_CHILD) {
205                                        keep.add(w);
206                                } else {
207                                        this.children[pos.ordinal()].put(w);
208                                        // (seems to me that this cannot happen to more than one link since the quad tree node will split
209                                        // as soon as a second link is added that can be assigned to a child node.  kai, jul'12)
210                                        // (h√§?  kai, aug'16)
211                                }
212                        }
213                        this.links.clear();
214                        this.links.ensureCapacity(keep.size() + 5);
215                        this.links.addAll(keep);
216                }

218                private ChildPosition getChildPosition(final LinkWrapper w) {
219                        // center of the bounding box of this quad tree node:
220                        double centerX = (minX + maxX) / 2;
221                        double centerY = (minY + maxY) / 2;
222
223                        if (w.maxX < centerX && w.minY >= centerY) {
224                                // (bounding box of link lies fully to left and above the center of the quadtree node)
225                                return ChildPosition.CHILD_NW;
226                        }
227                        if (w.minX >= centerX && w.minY >= centerY) {
228                                return ChildPosition.CHILD_NE;
229                        }
230                        if (w.minX >= centerX && w.maxY < centerY) {
231                                return ChildPosition.CHILD_SE;
232                        }
233                        if (w.maxX < centerX && w.maxY < centerY) {
234                                return ChildPosition.CHILD_SW;
235                        }
236                        return ChildPosition.NO_CHILD;
237                        // (happens when bounding box of link overlaps more than one child node.. i.e. in particular with long links)
238                }
239
240                /**
241                 * @param x
242                 * @param y
243                 * @return quad tree child node (=box) we need to look at if we are at (x,y)
244                 */
245                private ChildPosition getChildPosition(final double x, final double y) {
246                        double centerX = (minX + maxX) / 2;
247                        double centerY = (minY + maxY) / 2;
248                        if (x < centerX && y >= centerY) {
249                                return ChildPosition.CHILD_NW;
250                        }
251                        if (x >= centerX && y >= centerY) {
252                                return ChildPosition.CHILD_NE;
253                        }
254                        if (x >= centerX && y < centerY) {
255                                return ChildPosition.CHILD_SE;
256                        }
257                        if (x < centerX && y < centerY) {
258                                return ChildPosition.CHILD_SW;
259                        }
260                        throw new RuntimeException("should never get here since (x,y) has to be _somewhere_ with respect to centerX and centerY") ;
261                }

263                /**
264                 * Calculates the distance of a given point to the border of the
265                 * rectangle. If the point lies within the rectangle, the distance
266                 * is zero.
267                 *
268                 * @param x left-right location
269                 * @param y up-down location
270                 * @return distance to border, 0 if inside rectangle or on border
271                 */
272                private double calcDistanceIndicator(final double x, final double y) {
273                        double distanceX;
274                        double distanceY;
275
276                        if (this.minX <= x && x <= this.maxX) {
277                                distanceX = 0;
278                        } else {
279                                distanceX = Math.min(Math.abs(this.minX - x), Math.abs(this.maxX - x));
280                        }
281                        if (this.minY <= y && y <= this.maxY) {
282                                distanceY = 0;
283                        } else {
284                                distanceY = Math.min(Math.abs(this.minY - y), Math.abs(this.maxY - y));
285                        }
286
287                        return distanceX * distanceX + distanceY * distanceY; 
288                        // (no Math.sqrt(), as it's only used to compare to each other, thus distance "indicator")
289                }
290
291        }

293        private static double calcLineSegmentDistanceIndicator(final double x, final double y, final Link link) {
294
295                double fx = link.getFromNode().getCoord().getX();
296                double fy = link.getFromNode().getCoord().getY();
297                double lineDX = link.getToNode().getCoord().getX() - fx;
298                double lineDY = link.getToNode().getCoord().getY() - fy;
299
300                if ((lineDX == 0.0) && (lineDY == 0.0)) {
301                        // the line segment is a point without dimension
302                        return calcDistanceIndicator(fx, fy, x, y);
303                }
304
305                double u = ((x - fx)*lineDX + (y - fy)*lineDY) / (lineDX*lineDX + lineDY*lineDY);
306
307                if (u <= 0) {
308                        // (x | y) is not on the line segment, but before lineFrom
309                        return calcDistanceIndicator(fx, fy, x, y);
310                }
311                if (u >= 1) {
312                        // (x | y) is not on the line segment, but after lineTo
313                        return calcDistanceIndicator(fx + lineDX, fy + lineDY, x, y);
314                }
315                return calcDistanceIndicator(fx + u*lineDX, fy + u*lineDY, x, y);
316
317        }
318
319        private static double calcDistanceIndicator(final double fromX, final double fromY, final double toX, final double toY) {
320                double xDiff = toX - fromX;
321                double yDiff = toY - fromY;
322                return (xDiff*xDiff) + (yDiff*yDiff); 
323                // (no Math.sqrt(), as it's only used to compare to each other, thus distance "indicator")
324        }

326        private static class LinkWrapper {
327
328                /*package*/ final double minX;
329                /*package*/ final double minY;
330                /*package*/ final double maxX;
331                /*package*/ final double maxY;
332
333                /*package*/ final Link link;
334
335                public LinkWrapper(final Link link) {
336                        double fx = link.getFromNode().getCoord().getX();
337                        double fy = link.getFromNode().getCoord().getY();
338                        double tx = link.getToNode().getCoord().getX();
339                        double ty = link.getToNode().getCoord().getY();
340
341                        if (fx == tx) {
342                                // enforce minimal extent
343                                this.minX = fx - fx*1e-8; // make it adaptive within the number of significant digits
344                                this.maxX = fx + fx*1e-8; // make it adaptive within the number of significant digits
345                        } else {
346                                this.minX = Math.min(fx, tx);
347                                this.maxX = Math.max(fx, tx);
348                        }
349                        if (fy == ty) {
350                                this.minY = fy - fy*1e-8; // make it adaptive within the number of significant digits
351                                this.maxY = fy + fy*1e-8; // make it adaptive within the number of significant digits
352                        } else {
353                                this.minY = Math.min(fy, ty);
354                                this.maxY = Math.max(fy, ty);
355                        }
356
357                        this.link = link;
358                }
359        }
