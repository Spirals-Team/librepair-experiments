034/**
035 * A Procedure to expand a node of the {@link Network network}.
036 *
037 * @author balmermi
038 * @author mrieser / senozon
039 */
040public final class NetworkExpandNode {
041
042        private final Network network;
043        private double expRadius = 1.0;
044        private double offset = 0.0;
045        private double dist = 1.0; // cache a value that is often used in the method expandNode
046        
047        public NetworkExpandNode(final Network network, final double expRadius, final double offset) {
048                if (network == null) {
049                        throw new IllegalArgumentException("network must not be null.");
050                }
051                this.network = network;
052                this.setExpRadius(expRadius);
053                this.setOffset(offset);
054        }
055        
056        /**
057         * @param expRadius the expansion radius. If zero, all new nodes have the same coordinate
058         * and the new links with have length equals zero
059         */
060    void setExpRadius(final double expRadius) {
061                if (Double.isNaN(expRadius)) {
062                        throw new IllegalArgumentException("expansion radius must not be NaN.");
063                }
064                this.expRadius = expRadius;
065                this.dist = Math.sqrt(this.expRadius * this.expRadius - this.offset * this.offset);
066        }
067
068        /**
069         * @param offset the offset between a link pair with the same incident nodes. If zero, the two new
070         * nodes created for that link pair will have the same coordinates
071         */
072    void setOffset(final double offset) {
073                if (Double.isNaN(offset)) {
074                        throw new IllegalArgumentException("expansion offset must not be NaN.");
075                }
076                this.offset = offset;
077                this.dist = Math.sqrt(this.expRadius * this.expRadius - this.offset * this.offset);
078        }
079
080        /**
081         * Expands the specified {@link Node node} that is part of the {@link Network network} and
082         * adds turning restrictions/maneuvers to it. 
083         *
084         * <p>It is done in the following way:
085         * <ol>
086         * <li>creates for each in- and out-link a new node with
087         * <ul>
088         * <li><code>new_nodeId = nodeId+"-"+index; index=[0..#incidentLinks]</code></li>
089         * <li><code>new_coord</code> with distance <code>r</code> to the given node
090         * in direction of the corresponding incident Link of the given node with
091         * offset <code>e</code>.</li>
092         * </ul>
093         * <pre>
094         * {@code
095         * <-----12------         <----21-------
096         *
097         *            x-0 o     o 1-5
098         *                   O nodeId = x
099         *            x-1 o     o x-4
100         *             x-2 o   o x-3
101         * ------11----->         -----22------>
102         *               |       ^
103         *               |       |
104         *              32       31
105         *               |       |
106         *               |       |
107         *               v       |
108         * }
109         * </pre>
110         * </li>
111         * <li>connects each incident link of the given node with the corresponding <code>new_node</code>
112         * <pre>
113         * {@code
114         * <-----12------ o     o <----21-------
115         *                   O
116         * ------11-----> o     o -----22------>
117         *                 o   o
118         *                 |   ^
119         *                 |   |
120         *                32   31
121         *                 |   |
122         *                 |   |
123         *                 v   |
124         * }
125         * </pre>
126         * </li>
127         * <li>removes the given node from the network
128         * <pre>
129         * {@code
130         * <-----12------ o     o <----21-------
131         *
132         * ------11-----> o     o -----22------>
133         *                 o   o
134         *                 |   ^
135         *                 |   |
136         *                32   31
137         *                 |   |
138         *                 |   |
139         *                 v   |
140         * }
141         * </pre>
142         * </li>
143         * <li>inter-connects the <code>new_node</code>s with new links as defined in the
144         * <code>turns</code> list, with:<br>
145         * <ul>
146         * <li><code>new_linkId = fromLinkId+"-"+index; index=[0..#turn-tuples]</code></li>
147         * <li>length equals the Euclidean distance</li>
148         * <li>freespeed, capacity, permlanes, origId and type are equals to the attributes of the fromLink.</li>
149         * </ul>
150         * <pre>
151         * {@code
152         * <-----12------ o <--21-0-- o <----21-------
153         *
154         * ------11-----> o --11-1--> o -----22------>
155         *                 \         ^
156         *              11-2\       /31-3
157         *                   \     /
158         *                    v   /
159         *                    o   o
160         *                    |   ^
161         *                    |   |
162         *                   32   31
163         *                    |   |
164         *                    |   |
165         *                    v   |
166         * }
167         * </pre>
168         * </li>
169         * </ol>
170         * </p>
171         *
172         * @param nodeId the {@link Id} of the {@link Node} to expand
173         * @param turns The {@link List} of allowed turns at the given {@link Node node}.
174         * @return The {@link Tuple} of {@link List lists} containing the newly created
175         * {@link Node nodes} and {@link Link links}.
176         */
177        public final Tuple<List<Node>,List<Link>> expandNode(final Id<Node> nodeId, final List<TurnInfo> turns) {
178                double e = this.offset;
179                Node node = network.getNodes().get(nodeId);
180                if (node == null) {
181                        throw new IllegalArgumentException("nodeid="+nodeId+": not found in the network.");
182                }
183                if (turns == null) {
184                        throw new IllegalArgumentException("nodeid="+nodeId+": turn list not defined!");
185                }
186
187        for (TurnInfo turn1 : turns) {
188            Id<Link> first = turn1.getFromLinkId();
189            if (first == null) {
190                throw new IllegalArgumentException("given list contains 'null' values.");
191            }
192            if (!node.getInLinks().containsKey(first)) {
193                throw new IllegalArgumentException("nodeid=" + nodeId + ", linkid=" + first + ": link not an inlink of given node.");
194            }
195            Id<Link> second = turn1.getToLinkId();
196            if (second == null) {
197                throw new IllegalArgumentException("given list contains 'null' values.");
198            }
199            if (!node.getOutLinks().containsKey(second)) {
200                throw new IllegalArgumentException("nodeid=" + nodeId + ", linkid=" + second + ": link not an outlink of given node.");
201            }
202        }
203
204                // remove the node
205                Map<Id<Link>,Link> inlinks = new TreeMap<>(node.getInLinks());
206                Map<Id<Link>,Link> outlinks = new TreeMap<>(node.getOutLinks());
207                if (network.removeNode(node.getId()) == null) { throw new RuntimeException("nodeid="+nodeId+": Failed to remove node from the network."); }
208
209                ArrayList<Node> newNodes = new ArrayList<>(inlinks.size()+outlinks.size());
210                ArrayList<Link> newLinks = new ArrayList<>(turns.size());
211                // add new nodes and connect them with the in and out links
212                int nodeIdCnt = 0;
213                double d = this.dist;
214                for (Link inlink : inlinks.values()) {
215                        Coord c = node.getCoord();
216                        Coord p = inlink.getFromNode().getCoord();
217                        Coord cp = new Coord(p.getX() - c.getX(), p.getY() - c.getY());
218                        double lcp = Math.sqrt(cp.getX()*cp.getX()+cp.getY()*cp.getY());
219                        if (Math.abs(lcp) < 1e-8) {
220                                // c and p seem to lay on top of each other, leading to Double.NaN in some calculations
221                                lcp = d;
222                        }
223                        double dx = cp.getX() / lcp;
224                        double dy = cp.getY() / lcp;
225                        double x = c.getX() + d * dx - e * dy;
226                        double y = c.getY() + d * dy + e * dx;
227
228                        Node n = network.getFactory().createNode(Id.create(node.getId()+"-"+nodeIdCnt, Node.class), new Coord(x, y));
229                        network.addNode(n);
230                        newNodes.add(n);
231                        nodeIdCnt++;
232                        Link l = network.getFactory().createLink(inlink.getId(), inlink.getFromNode(), n);
233                        l.setLength(inlink.getLength());
234                        l.setFreespeed(inlink.getFreespeed());
235                        l.setCapacity(inlink.getCapacity());
236                        l.setNumberOfLanes(inlink.getNumberOfLanes());
237                        l.setAllowedModes(inlink.getAllowedModes());
238                        NetworkUtils.setOrigId( ((Link) l), NetworkUtils.getOrigId( ((Link) inlink) ) ) ;
239                        NetworkUtils.setType( ((Link) l), NetworkUtils.getType(((Link) inlink)));
240                        network.addLink(l);
241                }
242                for (Link outlink : outlinks.values()) {
243                        Coord c = node.getCoord();
244                        Coord p = outlink.getToNode().getCoord();
245                        Coord cp = new Coord(p.getX() - c.getX(), p.getY() - c.getY());
246                        double lcp = Math.sqrt(cp.getX()*cp.getX()+cp.getY()*cp.getY());
247                        if (Math.abs(lcp) < 1e-8) {
248                                // c and p seem to lay on top of each other, leading to Double.NaN in some calculations
249                                lcp = d;
250                        }
251                        double dx = cp.getX() / lcp;
252                        double dy = cp.getY() / lcp;
253                        double x = c.getX() + d * dx + e * dy;
254                        double y = c.getY() + d * dy - e * dx;
255                        Node n = network.getFactory().createNode(Id.create(node.getId()+"-"+nodeIdCnt, Node.class), new Coord(x, y));
256                        network.addNode(n);
257                        newNodes.add(n);
258                        nodeIdCnt++;
259                        Link l = network.getFactory().createLink(outlink.getId(), n, outlink.getToNode());
260                        l.setLength(outlink.getLength());
261                        l.setFreespeed(outlink.getFreespeed());
262                        l.setCapacity(outlink.getCapacity());
263                        l.setNumberOfLanes(outlink.getNumberOfLanes());
264                        l.setAllowedModes(outlink.getAllowedModes());
265                        if (outlink instanceof Link) {
266                                NetworkUtils.setOrigId( ((Link) l), NetworkUtils.getOrigId( ((Link) outlink) ) ) ;
267                                NetworkUtils.setType( ((Link) l), NetworkUtils.getType(((Link) outlink)));
268                        }
269                        network.addLink(l);
270                }
271
272                // add virtual links for the turn restrictions
273                for (int i=0; i<turns.size(); i++) {
274                        TurnInfo turn = turns.get(i);
275                        Link fromLink = network.getLinks().get(turn.getFromLinkId());
276                        Link toLink = network.getLinks().get(turn.getToLinkId());
277                        Link l = network.getFactory().createLink(Id.create(fromLink.getId()+"-"+i, Link.class), fromLink.getToNode(), toLink.getFromNode());
278                        double dist = CoordUtils.calcEuclideanDistance(toLink.getFromNode().getCoord(), fromLink.getToNode().getCoord());
279                        if (dist < 0.1 * this.expRadius) {
280                                // mostly the case when nodes are on top of each other
281                                // use a small length, but not really hard-coded because of different coordinate systems ("1" is not always 1 meter)
282                                dist = 0.1 * this.expRadius;
283                        }
284                        l.setLength(dist);
285                        l.setFreespeed(fromLink.getFreespeed());
286                        l.setCapacity(fromLink.getCapacity());
287                        l.setNumberOfLanes(fromLink.getNumberOfLanes());
288                        if (turn.getModes() == null) {
289                                l.setAllowedModes(fromLink.getAllowedModes());
290                        } else {
291                                l.setAllowedModes(turn.getModes());
292                        }
293                        NetworkUtils.setOrigId( ((Link) l), NetworkUtils.getOrigId( ((Link) fromLink) ) ) ;
294                        NetworkUtils.setType( ((Link) l), NetworkUtils.getType(((Link) fromLink)));
295                        network.addLink(l);
296                        newLinks.add(l);
297                }
298                return new Tuple<List<Node>, List<Link>>(newNodes,newLinks);
299        }

301        /**
302         * Checks whether it makes sense to expand the node with turning links or not. For example, it might 
303         * not make sense to expand the node if no real restrictions are given, but just all possible options
304         * are listed -- the exact same turning options could be available without expanding the node.
305         * Thus, it usually only makes sense to expand a node if {@link #turnsAreSameAsSingleNode(Id, List, boolean)}
306         * returns <code>false</code>.
307         * The algorithm can optionally ignore u-turns, that means that it is not checked if all u-turns are
308         * allowed according to the given turning options. 
309         * 
310         * 
311         * @param nodeId
312         * @param turns the allowed turning options
313         * @param ignoreUTurns set this to <code>true</code> if u-turns are excluded from the check
314         * @return <code>true</code> if the list of explicit turns given results in the same turning options available as when the node is not expanded, <code>false</code> otherwise. 
315         */
316        public boolean turnsAreSameAsSingleNode(final Id<Node> nodeId, final List<TurnInfo> turns, final boolean ignoreUTurns) {
317                Node node = this.network.getNodes().get(nodeId);
318
319                // first create list of all possible turning options
320                Map<Id<Link>, Map<Id<Link>, Set<String>>> allTurns = new HashMap<>();
321                
322                for (Link inLink : node.getInLinks().values()) {
323                        Map<Id<Link>, Set<String>> t2 = new HashMap<>();
324                        for (Link outLink : node.getOutLinks().values()) {
325                                if (inLink.getFromNode() != outLink.getToNode() || !ignoreUTurns) {
326                                        HashSet<String> modes = new HashSet<>();
327                                        modes.addAll(inLink.getAllowedModes());
328                                        modes.retainAll(outLink.getAllowedModes()); // modes contains now all useful modes
329                                        t2.put(outLink.getId(), modes);
330                                }
331                        }
332                        allTurns.put(inLink.getId(), t2);
333                }
334                
335                /* now compare the given turning options and remove them from all possible options.
336                 * if there remain some options in the "all possibles", we know that the given turns
337                 * act indeed as restrictions. */
338                for (TurnInfo ti : turns) {
339                        Id<Link> from = ti.fromLinkId;
340                        Id<Link> to = ti.toLinkId;
341                        Map<Id<Link>, Set<String>> t2 = allTurns.get(from);
342                        if (t2 != null) {
343                                Set<String> modes = t2.get(to);
344                                if (modes != null) {
345                                        if (ti.getModes() == null) {
346                                                // no specific modes are set, so remove all modes from the inLink
347                                                // because the modes are the intersection of modes from inLink and outLink, we can just clear it
348                                                modes.clear();
349                                        } else {
350                                                modes.removeAll(ti.getModes());
351                                        }
352                                }
353                        }
354                }
355                
356                // now do the check
357                for (Map<Id<Link>, Set<String>> m : allTurns.values()) {
358                        for (Set<String> modes : m.values()) {
359                                if (modes.size() > 0) {
360                                        return false;
361                                }
362                        }
363                }
364                
365                return true;
366        }
367        
368        public static class TurnInfo {
369                
370                private final Id<Link> fromLinkId;
371                private final Id<Link> toLinkId;
372                private final Set<String> modes;
373                
374                public TurnInfo(final Id<Link> fromLinkId, final Id<Link> toLinkId) {
375                        this.fromLinkId = fromLinkId;
376                        this.toLinkId = toLinkId;
377                        this.modes = null;
378                }
379                
380                public TurnInfo(final Id<Link> fromLinkId, final Id<Link> toLinkId, final Set<String> modes) {
381                        this.fromLinkId = fromLinkId;
382                        this.toLinkId = toLinkId;
383                        this.modes = modes;
384                }
385                
386                public Id<Link> getFromLinkId() {
387                        return this.fromLinkId;
388                }
389                
390                public Id<Link> getToLinkId() {
391                        return this.toLinkId;
392                }
393                
394                public Set<String> getModes() {
395                        return this.modes;
396                }
397
398                @Override
399                public boolean equals(Object obj) {
400                        if (!(obj instanceof TurnInfo)) {
401                                return false;
402                        }
403                        TurnInfo ti = (TurnInfo) obj;
404                        return (ti.fromLinkId.equals(this.fromLinkId))
405                                        && (ti.toLinkId.equals(this.toLinkId))
406                                        && ((ti.modes == null && this.modes == null)
407                                                        || (ti.modes != null && ti.modes.equals(this.modes))
408                                                        );
409                }
410                
411                @Override
412                public int hashCode() {
413                        return this.fromLinkId.hashCode() & this.toLinkId.hashCode() & this.modes.hashCode();
414                }
415                
416        }
