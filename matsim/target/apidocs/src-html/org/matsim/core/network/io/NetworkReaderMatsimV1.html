021package org.matsim.core.network.io;
022
023import org.apache.log4j.Logger;
024import org.matsim.api.core.v01.Coord;
025import org.matsim.api.core.v01.Id;
026import org.matsim.api.core.v01.network.Link;
027import org.matsim.api.core.v01.network.Network;
028import org.matsim.api.core.v01.network.Node;
029import org.matsim.core.network.NetworkUtils;
030import org.matsim.core.utils.geometry.CoordinateTransformation;
031import org.matsim.core.utils.geometry.transformations.IdentityTransformation;
032import org.matsim.core.utils.io.MatsimXmlParser;
033import org.matsim.core.utils.misc.StringUtils;
034import org.matsim.core.utils.misc.Time;
035import org.xml.sax.Attributes;
036
037import java.util.HashSet;
038import java.util.Set;
039import java.util.Stack;
040
041/**
042 * A reader for network-files of MATSim according to <code>network_v1.dtd</code>.
043 *
044 * @author mrieser
045 */
046public final class NetworkReaderMatsimV1 extends MatsimXmlParser {
047
048        private final static String NETWORK = "network";
049        private final static String LINKS = "links";
050        private final static String NODES = "nodes";
051        private final static String NODE = "node";
052        private final static String LINK = "link";
053
054        private final Network network;
055        
056        private boolean hasElevation = false;
057        private boolean hasNoElevation = false;
058
059        // final or settable?
060        private final CoordinateTransformation transformation;
061
062        private final static Logger log = Logger.getLogger(NetworkReaderMatsimV1.class);
063
064        public NetworkReaderMatsimV1(Network network) {
065                this( new IdentityTransformation() , network );
066        }
067
068        public NetworkReaderMatsimV1(
069                        final CoordinateTransformation transformation,
070                        final Network network) {
071                this.transformation = transformation;
072                this.network = network;
073        }
074
075        @Override
076        public void startTag(final String name, final Attributes atts, final Stack<String> context) {
077                if (NODE.equals(name)) {
078                        startNode(atts);
079                } else if (LINK.equals(name)) {
080                        startLink(atts);
081                } else if (NETWORK.equals(name)) {
082                        startNetwork(atts);
083                } else if (LINKS.equals(name)) {
084                        startLinks(atts);
085                }
086        }
087
088        @Override
089        public void endTag(final String name, final String content, final Stack<String> context) {
090                /* Check if there is a mixture of nodes WITH and WITHOUT elevation. */
091                /* FIXME Note: I would like this to rather crash here (jwjoubert, Sep'16) */
092                if(NODES.equals(name)){
093                        if(hasElevation && hasNoElevation){
094                                log.warn("There is a mixture of nodes WITH and WITHOUT elevation! " + 
095                                                  "You will likely run into problems when doing coordinate calculations.");
096                        }
097                }
098        }
099
100        private void startNetwork(final Attributes atts) {
101                if (atts.getValue("type") != null) {
102                        log.info("Attribute 'type' is deprecated. There's always only ONE network, where the links and nodes define, which " +
103                                        "transportation mode is allowed to use it (for the future)");
104                }
105                this.network.setName(atts.getValue("name"));
106                if (atts.getValue("capDivider") != null) {
107                        log.warn("capDivider defined. it will be used but should be gone eventually. " +
108                                        "-- This is a weird comment, since the matsim public api tells to put this into the network rather than" +
109                                        " into the ``links''.  kai, jun'11");
110                        String capperiod = atts.getValue("capDivider") + ":00:00";
111                        this.network.setCapacityPeriod(Time.parseTime(capperiod));
112                }
113        }
114
115        private void startLinks(final Attributes atts) {
116                double capacityPeriod = 3600.0; //the default of one hour
117                String capperiod = atts.getValue("capperiod");
118                if (capperiod != null) {
119                        capacityPeriod = Time.parseTime(capperiod);
120                }
121                else {
122                        log.warn("capperiod was not defined. Using default value of " + Time.writeTime(capacityPeriod) + ".");
123                }
124                this.network.setCapacityPeriod(capacityPeriod);
125
126                String effectivecellsize = atts.getValue("effectivecellsize");
127                if (effectivecellsize == null){
128                        this.network.setEffectiveCellSize(7.5); // we use a default cell size of 7.5 meters
129                } else {
130                        this.network.setEffectiveCellSize(Double.parseDouble(effectivecellsize));
131                }
132
133                String effectivelanewidth = atts.getValue("effectivelanewidth");
134                if (effectivelanewidth == null) {
135                        this.network.setEffectiveLaneWidth(3.75); // the default lane width is 3.75
136                } else {
137                        this.network.setEffectiveLaneWidth(Double.parseDouble(effectivelanewidth));
138                }
139
140                if ((atts.getValue("capPeriod") != null) || (atts.getValue("capDivider") != null) || (atts.getValue("capdivider") != null)) {
141                        log.warn("Found capPeriod, capDivider and/or capdivider in the links element.  They will be ignored, since they " +
142                                        "should be set in the network element. -- This is a weird warning, since setting them in the " +
143                                        "network element also produces a warning.");
144                        log.warn("At this point, it seems that, in network.xml, one sets capperiod in the `links' section, but in the " +
145                                        "matsim api, the corresponding entry belongs into the `network' object. kai, jun'11") ;
146                }
147        }
148
149        private void startNode(final Attributes atts) {
150                boolean hasZ = atts.getValue("z") == null ? false : true;
151                Coord c;
152                if(hasZ){
153                        c = transformation.transform(new Coord(
154                                        Double.parseDouble(atts.getValue("x")),
155                                        Double.parseDouble(atts.getValue("y")),
156                                        Double.parseDouble(atts.getValue("z"))));
157                        hasElevation = true;
158                } else{
159                        c = transformation.transform(new Coord(
160                                        Double.parseDouble(atts.getValue("x")),
161                                        Double.parseDouble(atts.getValue("y"))));
162                        hasNoElevation = true;
163                }
164                
165                final Node node =
166                                this.network.getFactory().createNode(
167                                                Id.create(atts.getValue("id"), Node.class), c);
168                this.network.addNode(node);
169
170                NetworkUtils.setType(node,atts.getValue("type"));
171                // (did not have a null check when I found it.  kai, jul'16) 
172
173                if (atts.getValue("origid") != null) {
174                        NetworkUtils.setOrigId( node, atts.getValue("origid") ) ;
175                }
176        }
177
178        private void startLink(final Attributes atts) {
179                final String fromNodeStr = atts.getValue("from");
180                Node fromNode = this.network.getNodes().get(Id.create(fromNodeStr, Node.class));
181                if ( fromNode==null ) {
182                        throw new RuntimeException("node id given by link cannot be dereferenced; node label=" + fromNodeStr ) ;
183                }
184                final String toNodeStr = atts.getValue("to");
185                Node toNode = this.network.getNodes().get(Id.create(toNodeStr, Node.class));
186                if ( toNode==null ) {
187                        throw new RuntimeException("node id given by link cannot be dereferenced; node label=" + toNodeStr ) ;
188                }
189                Link l = this.network.getFactory().createLink(Id.create(atts.getValue("id"), Link.class), fromNode, toNode);
190                l.setLength(Double.parseDouble(atts.getValue("length")));
191                l.setFreespeed(Double.parseDouble(atts.getValue("freespeed")));
192                l.setCapacity(Double.parseDouble(atts.getValue("capacity")));
193                l.setNumberOfLanes(Double.parseDouble(atts.getValue("permlanes")));
194                this.network.addLink(l);
195                NetworkUtils.setOrigId(l, atts.getValue("origid") ) ;
196                NetworkUtils.setType(l, atts.getValue("type"));
197                if (atts.getValue("modes") != null) {
198                        String[] strModes = StringUtils.explode(atts.getValue("modes"), ',');
199                        if ((strModes.length == 1) && strModes[0].isEmpty()) {
200                                l.setAllowedModes(new HashSet<>());
201                        } else {
202                                Set<String> modes = new HashSet<>();
203                                for (String strMode : strModes) {
204                                        modes.add(strMode.trim().intern());
205                                }
206                                l.setAllowedModes(modes);
207                        }
208                }
209                if (atts.getValue("volume") != null) {
210                        log.info("Attribute volume for element link is deprecated.");
211                }
212                if (atts.getValue("nt_category") != null) {
213                        log.info("Attribute nt_category for element link is deprecated.");
214                }
215                if (atts.getValue("nt_type") != null) {
216                        log.info("Attribute nt_type for element link is deprecated.");
217                }
218        }
