021package org.matsim.core.network.algorithms;
022
023import java.util.ArrayList;
024import java.util.HashMap;
025import java.util.Iterator;
026import java.util.List;
027import java.util.Map;
028import java.util.TreeMap;
029
030import org.apache.log4j.Logger;
031import org.matsim.api.core.v01.Id;
032import org.matsim.api.core.v01.network.Link;
033import org.matsim.api.core.v01.network.Network;
034import org.matsim.api.core.v01.network.Node;
035import org.matsim.core.api.internal.NetworkRunnable;
036
037/**
038 * Ensures that each link in the network can be reached by any other link.
039 * Links that cannot be reached by some other links, or links from which it
040 * is not possible to reach all other links, are removed from the network.
041 * Nodes with no incoming or outgoing links are removed as well from the
042 * network.
043 *
044 * @author mrieser
045 * @author balmermi
046 */
047public final class NetworkCleaner implements NetworkRunnable {
048
049        private static final Logger log = Logger.getLogger(NetworkCleaner.class);
050
051        /**
052         * Finds the cluster of nodes of which <code>startNode</code> is part of. The cluster
053         * contains all nodes which can be reached starting at <code>startNode</code>
054         * and from where it is also possible to return again to <code>startNode</code>.
055         *
056         * @param startNode the node to start building the cluster
057         * @param network the network the startNode is part of
058         * @return cluster of nodes of which <code>startNode</code> is part of
059         */
060        private Map<Id<Node>, Node> findCluster(final Node startNode, final Network network) {
061
062                final Map<Node, DoubleFlagRole> nodeRoles = new HashMap<>(network.getNodes().size());
063
064                ArrayList<Node> pendingForward = new ArrayList<>();
065                ArrayList<Node> pendingBackward = new ArrayList<>();
066
067                TreeMap<Id<Node>, Node> clusterNodes = new TreeMap<>();
068                clusterNodes.put(startNode.getId(), startNode);
069                DoubleFlagRole r = getDoubleFlag(startNode, nodeRoles);
070                r.forwardFlag = true;
071                r.backwardFlag = true;
072
073                pendingForward.add(startNode);
074                pendingBackward.add(startNode);
075
076                // step through the network in forward mode
077                while (pendingForward.size() > 0) {
078                        int idx = pendingForward.size() - 1;
079                        Node currNode = pendingForward.remove(idx); // get the last element to prevent object shifting in the array
080                        for (Link link : currNode.getOutLinks().values()) {
081                                Node node = link.getToNode();
082                                r = getDoubleFlag(node, nodeRoles);
083                                if (!r.forwardFlag) {
084                                        r.forwardFlag = true;
085                                        pendingForward.add(node);
086                                }
087                        }
088                }
089
090                // now step through the network in backward mode
091                while (pendingBackward.size() > 0) {
092                        int idx = pendingBackward.size()-1;
093                        Node currNode = pendingBackward.remove(idx); // get the last element to prevent object shifting in the array
094                        for (Link link : currNode.getInLinks().values()) {
095                                Node node = link.getFromNode();
096                                r = getDoubleFlag(node, nodeRoles);
097                                if (!r.backwardFlag) {
098                                        r.backwardFlag = true;
099                                        pendingBackward.add(node);
100                                        if (r.forwardFlag) {
101                                                // the node can be reached forward and backward, add it to the cluster
102                                                clusterNodes.put(node.getId(), node);
103                                        }
104                                }
105                        }
106                }
107
108                return clusterNodes;
109        }
110
111        /**
112         * Searches the biggest cluster in the given Network. The Network is not modified.
113         */
114        public Map<Id<Node>, Node> searchBiggestCluster(Network network) {
115                final Map<Id<Node>, Node> visitedNodes = new TreeMap<>();
116                Map<Id<Node>, Node> biggestCluster = new TreeMap<>();
117
118                log.info("running " + this.getClass().getName() + " algorithm...");
119
120                // search the biggest cluster of nodes in the network
121                log.info("  checking " + network.getNodes().size() + " nodes and " +
122                                network.getLinks().size() + " links for dead-ends...");
123                boolean stillSearching = true;
124                Iterator<? extends Node> iter = network.getNodes().values().iterator();
125                while (iter.hasNext() && stillSearching) {
126                        Node startNode = iter.next();
127                        if (!visitedNodes.containsKey(startNode.getId())) {
128                                Map<Id<Node>, Node> cluster = this.findCluster(startNode, network);
129                                visitedNodes.putAll(cluster);
130                                if (cluster.size() > biggestCluster.size()) {
131                                        biggestCluster = cluster;
132                                        if (biggestCluster.size() >= (network.getNodes().size() - visitedNodes.size())) {
133                                                // stop searching here, because we cannot find a bigger cluster in the lasting nodes
134                                                stillSearching = false;
135                                        }
136                                }
137                        }
138                }
139                log.info("    The biggest cluster consists of " + biggestCluster.size() + " nodes.");
140                log.info("  done.");
141                return biggestCluster;
142        }
143
144        /** 
145         * Reducing the network so it only contains nodes included in the biggest Cluster.
146         * Loop over all nodes and check if they are in the cluster, if not, remove them from the network
147         */
148        public static void reduceToBiggestCluster(Network network, Map<Id<Node>, Node> biggestCluster) {
149                List<Node> allNodes2 = new ArrayList<>(network.getNodes().values());
150                for (Node node : allNodes2) {
151                        if (!biggestCluster.containsKey(node.getId())) {
152                                network.removeNode(node.getId());               // removeNode takes care of removing links too in the network
153                        }
154                }
155                log.info("  resulting network contains " + network.getNodes().size() + " nodes and " +
156                                network.getLinks().size() + " links.");
157                log.info("done.");
158        }
159        
160        @Override
161        public void run(final Network network) {
162                Map<Id<Node>, Node> biggestCluster = this.searchBiggestCluster(network);
163                reduceToBiggestCluster(network, biggestCluster);
164        }
165
166        private static DoubleFlagRole getDoubleFlag(final Node n, final Map<Node, DoubleFlagRole> nodeRoles) {
167                DoubleFlagRole r = nodeRoles.get(n);
168                if (null == r) {
169                        r = new DoubleFlagRole();
170                        nodeRoles.put(n, r);
171                }
172                return r;
173        }
174
175        static class DoubleFlagRole {
176                boolean forwardFlag = false;
177                boolean backwardFlag = false;
178        }
179
180}
