021package org.matsim.core.network.algorithms;
022
023import java.io.BufferedWriter;
024import java.io.IOException;
025
026import org.apache.log4j.Logger;
027import org.matsim.api.core.v01.Coord;
028import org.matsim.api.core.v01.network.Link;
029import org.matsim.api.core.v01.network.Network;
030import org.matsim.api.core.v01.network.Node;
031import org.matsim.core.api.internal.NetworkRunnable;
032import org.matsim.core.network.NetworkUtils;
033import org.matsim.core.utils.geometry.CoordUtils;
034import org.matsim.core.utils.io.IOUtils;
035import org.matsim.core.utils.misc.Time;
036
037public final class NetworkWriteAsTable implements NetworkRunnable {
038
039        //////////////////////////////////////////////////////////////////////
040        // member variables
041        //////////////////////////////////////////////////////////////////////
042
043        private final static Logger log = Logger.getLogger(NetworkWriteAsTable.class);
044
045        private final String outdir;
046        private final double offset;
047
048        //////////////////////////////////////////////////////////////////////
049        // constructors
050        //////////////////////////////////////////////////////////////////////
051
052        public NetworkWriteAsTable(final String outdir, double offset) {
053                this.offset = offset;
054                this.outdir = outdir;
055        }
056
057        public NetworkWriteAsTable(final String outdir) {
058                this(outdir,0.0);
059        }
060
061        public NetworkWriteAsTable() {
062                this("output");
063        }
064
065        //////////////////////////////////////////////////////////////////////
066        // final method
067        //////////////////////////////////////////////////////////////////////
068
069        //////////////////////////////////////////////////////////////////////
070        // private methods
071        //////////////////////////////////////////////////////////////////////
072
073        //////////////////////////////////////////////////////////////////////
074        // run methods
075        //////////////////////////////////////////////////////////////////////
076
077        @Override
078        public void run(Network network) {
079                double capperiod = network.getCapacityPeriod();
080                capperiod = capperiod / 3600;
081                log.info("capperiod = " + capperiod);
082
083                try {
084                        BufferedWriter out_n = IOUtils.getBufferedWriter(this.outdir + "/nodes.txt.gz");
085                        out_n.write("ID\tX\tY\n");
086                        out_n.flush();
087                        for (Node n : network.getNodes().values()) {
088                                out_n.write(n.getId() + "\t" + n.getCoord().getX() + "\t" + n.getCoord().getY() + "\n");
089                                out_n.flush();
090                        }
091                        out_n.flush();
092                        out_n.close();
093
094                        BufferedWriter out_l = IOUtils.getBufferedWriter(this.outdir + "/links.txt.gz");
095                        out_l.write("ID\tX1\tY1\tX2\tY2\tLENGHT\tFREESPEED\tCAPACITY\tPERMLANES\tMODES\n");
096                        out_l.flush();
097                        BufferedWriter out_et = IOUtils.getBufferedWriter(this.outdir + "/linksET.txt.gz");
098                        out_et.write("ID\tFROMID\tTOID\tLENGTH\tSPEED\tCAP\tLANES\tORIGID\tTYPE\tMODES\n");
099                        out_et.flush();
100                        for (Link l : network.getLinks().values()) {
101                                Node f = l.getFromNode();
102                                Node t = l.getToNode();
103
104                                final double y = -t.getCoord().getX() + f.getCoord().getX();
105                                Coord offsetVector = new Coord(t.getCoord().getY() - f.getCoord().getY(), y);
106                                offsetVector = CoordUtils.scalarMult(offset/CoordUtils.length(offsetVector),offsetVector);
107                                Coord fc = CoordUtils.plus(f.getCoord(),offsetVector);
108                                Coord tc = CoordUtils.plus(t.getCoord(),offsetVector);
109                                
110                                out_l.write(l.getId() + "\t" + fc.getX() + "\t" + fc.getY() + "\t");
111                                out_l.write(tc.getX() + "\t" + tc.getY() + "\t" + l.getLength() + "\t");
112                                out_l.write(l.getFreespeed()+"\t"
113                                                +(l.getCapacity()/capperiod)+"\t"
114                                                + NetworkUtils.getNumberOfLanesAsInt(Time.UNDEFINED_TIME, l)+"\t"
115                                                +l.getAllowedModes().toString()+"\n");
116                                out_l.flush();
117
118                                out_et.write(l.getId() + "\t" + l.getFromNode().getId() + "\t" + l.getToNode().getId() + "\t");
119                                out_et.write(Math.round(l.getLength()) + "\t" + Math.round(l.getFreespeed()*3.6) + "\t");
120                                out_et.write(Math.round(l.getCapacity()/capperiod) + "\t" + NetworkUtils.getNumberOfLanesAsInt(Time.UNDEFINED_TIME, l) + "\t");
121                                out_et.write(NetworkUtils.getOrigId( ((Link) l) ) + "\t" + NetworkUtils.getType(((Link) l)) + "\t"+l.getAllowedModes().toString()+"\n");
122                                out_et.write(fc.getX() + "\t" + fc.getY() + "\n");
123                                out_et.write(tc.getX() + "\t" + tc.getY() + "\n");
124                                out_et.write("END\n");
125                        }
126                        out_l.flush();
127                        out_l.close();
128                        out_et.write("END\n");
129                        out_et.flush();
130                        out_et.close();
131                } catch (IOException e) {
132                        throw new RuntimeException(e);
133                }
134        }
135}
