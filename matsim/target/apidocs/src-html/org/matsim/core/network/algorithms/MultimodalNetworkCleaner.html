038/**
039 * Variant of {@link org.matsim.core.network.algorithms.NetworkCleaner NetworkCleaner} that supports
040 * multi-modal networks. If the cleaner is run, it will make sure that the sub-network of all those
041 * links having at least one of the given transport modes is strongly connected. Other links are not
042 * modified. If a link does not belong to the biggest cluster, the to-be-cleaned modes are removed
043 * from the set of allowed modes for this link. If a link has no allowed mode anymore, it is removed
044 * from the network, along with nodes that lose all their in- and out-links by that way.
045 *
046 * @author mrieser
047 */
048public final class MultimodalNetworkCleaner {
049
050        private final static Logger log = Logger.getLogger(MultimodalNetworkCleaner.class);
051
052        private final Network network;
053        
054        private final Set<Id<Link>> removedLinks = new HashSet<>();
055        private final Set<Id<Link>> modifiedLinks = new HashSet<>();
056
057        public MultimodalNetworkCleaner(final Network network) {
058                this.network = network;
059        }
060
061        /**
062         * Removes nodes from the network that have no incoming or outgoing links attached to them.
063         */
064        public void removeNodesWithoutLinks() {
065                List<Node> toBeRemoved = new ArrayList<>();
066                for (Node node : this.network.getNodes().values()) {
067                        if ((node.getInLinks().size() == 0) && (node.getOutLinks().size() == 0)) {
068                                toBeRemoved.add(node);
069                        }
070                }
071                for (Node node : toBeRemoved) {
072                        this.network.removeNode(node.getId());
073                }
074        }
075
076        /**
077         * Modifies the network such that the subnetwork containing only links that have at least
078         * one of the specified transport modes in their set of allowed transport modes is strongly
079         * connected (=every link/node can be reached by every other link/node). If multiple modes
080         * are given, the algorithm does <em>not</em> guarantee that the resulting network is strongly
081         * connected for each of the modes individually! Nodes having links connected to them before
082         * cleaning, but none after cleaning, are removed from the network.
083         *
084         * @param modes
085         */
086        public void run(final Set<String> modes) {
087                run(modes, new HashSet<String>());
088        }
089
090        /**
091         * Modifies the network such that the subnetwork containing only links that have at least
092         * one of the specified transport modes (<code>cleaningModes</code> as well as
093         * <code>connectivityModes</code>) in their set of allowed transport modes is strongly
094         * connected (=every link/node can be reached by every other link/node). In contrast to
095         * {@link #run(Set)}, this method will only remove <code>cleaningModes</code> from links,
096         * but not <code>connectivityModes</code>. Thus, the resulting  network may still contain
097         * nodes that are sources or sinks for modes of <code>connectivityModes</code>, but not
098         * for modes of <code>cleaningModes</code> and <code>connectivityModes</code> combined. If
099         * multiple modes are given as <code>cleaningModes</code>, the algorithm does <em>not</em>
100         * guarantee that the resulting network is strongly connected for each of the modes
101         * individually! The subnetwork consisting of links having only modes from
102         * <code>cleaningModes</code> may not be strongly connected, only in combination with links
103         * having modes from <code>connectivityModes</code>, a strongly connected subnetwork emerges.
104         * Nodes having links connected to them before cleaning, but none after cleaning, are removed
105         * from the network.
106         *
107         * @param cleaningModes
108         * @param connectivityModes
109         */
110        public void run(final Set<String> cleaningModes, final Set<String> connectivityModes) {
111                final Set<String> combinedModes = new HashSet<>(cleaningModes);
112                combinedModes.addAll(connectivityModes);
113                final Map<Id<Link>, Link> visitedLinks = new TreeMap<>();
114                Map<Id<Link>, Link> biggestCluster = new TreeMap<>();
115
116                log.info("running " + this.getClass().getName() + " algorithm for modes " + Arrays.toString(cleaningModes.toArray())
117                                + " with connectivity modes " + Arrays.toString(connectivityModes.toArray()) + "...");
118
119                // search the biggest cluster of nodes in the network
120                log.info("  checking " + this.network.getNodes().size() + " nodes and " +
121                                this.network.getLinks().size() + " links for dead-ends...");
122                boolean stillSearching = true;
123                Iterator<? extends Link> iter = this.network.getLinks().values().iterator();
124                while (iter.hasNext() && stillSearching) {
125                        Link startLink = iter.next();
126                        if ((!visitedLinks.containsKey(startLink.getId())) && (intersectingSets(combinedModes, startLink.getAllowedModes()))) {
127                                Map<Id<Link>, Link> cluster = this.findCluster(startLink, combinedModes);
128                                visitedLinks.putAll(cluster);
129                                if (cluster.size() > biggestCluster.size()) {
130                                        biggestCluster = cluster;
131                                        if (biggestCluster.size() >= (this.network.getLinks().size() - visitedLinks.size())) {
132                                                // stop searching here, because we cannot find a bigger cluster in the lasting nodes
133                                                stillSearching = false;
134                                        }
135                                }
136                        }
137                }
138                log.info("    The biggest cluster consists of " + biggestCluster.size() + " links.");
139                log.info("  done.");
140
141                /* Remove the modes from all links not being part of the cluster. If a link has no allowed mode
142                 * anymore after this, remove the link from the network.
143                 */
144                List<Link> allLinks = new ArrayList<>(this.network.getLinks().values());
145                for (Link link : allLinks) {
146                        if (!biggestCluster.containsKey(link.getId())) {
147                                Set<String> reducedModes = new HashSet<>(link.getAllowedModes());
148                                reducedModes.removeAll(cleaningModes);
149                                link.setAllowedModes(reducedModes);
150                                if (reducedModes.isEmpty()) {
151                                        this.network.removeLink(link.getId());
152                                        if ((link.getFromNode().getInLinks().size() + link.getFromNode().getOutLinks().size()) == 0) {
153                                                this.network.removeNode(link.getFromNode().getId());
154                                        }
155                                        if ((link.getToNode().getInLinks().size() + link.getToNode().getOutLinks().size()) == 0) {
156                                                this.network.removeNode(link.getToNode().getId());
157                                        }
158                                        this.removedLinks.add(link.getId());
159                                }
160                                if(!removedLinks.contains(link.getId())) modifiedLinks.add(link.getId());
161                        }
162                }
163                log.info("  resulting network contains " + this.network.getNodes().size() + " nodes and " +
164                                this.network.getLinks().size() + " links.");
165                log.info("done.");
166        }
167
168        /**
169         * Finds the cluster of links <pre>startLink</pre> is part of. The cluster
170         * contains all links which can be reached starting at <code>startLink</code>
171         * and from where it is also possible to return again to <code>startLink</code>.
172         *
173         * @param startLink the link to start building the cluster
174         * @param modes the set of modes that are allowed to
175         * @return cluster of links <pre>startLink</pre> is part of
176         */
177        private Map<Id<Link>, Link> findCluster(final Link startLink, final Set<String> modes) {
178
179                final Map<Id<Link>, DoubleFlagRole> linkRoles = new HashMap<>(this.network.getLinks().size());
180
181                ArrayList<Node> pendingForward = new ArrayList<>();
182                ArrayList<Node> pendingBackward = new ArrayList<>();
183
184                TreeMap<Id<Link>, Link> clusterLinks = new TreeMap<>();
185
186                pendingForward.add(startLink.getToNode());
187                pendingBackward.add(startLink.getFromNode());
188
189                // step through the network in forward mode
190                while (pendingForward.size() > 0) {
191                        int idx = pendingForward.size() - 1;
192                        Node currNode = pendingForward.remove(idx); // get the last element to prevent object shifting in the array
193                        for (Link link : currNode.getOutLinks().values()) {
194                                if (intersectingSets(modes, link.getAllowedModes())) {
195                                        DoubleFlagRole r = getDoubleFlag(link, linkRoles);
196                                        if (!r.forwardFlag) {
197                                                r.forwardFlag = true;
198                                                pendingForward.add(link.getToNode());
199                                        }
200                                }
201                        }
202                }
203
204                // now step through the network in backward mode
205                while (pendingBackward.size() > 0) {
206                        int idx = pendingBackward.size()-1;
207                        Node currNode = pendingBackward.remove(idx); // get the last element to prevent object shifting in the array
208                        for (Link link : currNode.getInLinks().values()) {
209                                if (intersectingSets(modes, link.getAllowedModes())) {
210                                        DoubleFlagRole r = getDoubleFlag(link, linkRoles);
211                                        if (!r.backwardFlag) {
212                                                r.backwardFlag = true;
213                                                pendingBackward.add(link.getFromNode());
214                                                if (r.forwardFlag) {
215                                                        // the node can be reached forward and backward, add it to the cluster
216                                                        clusterLinks.put(link.getId(), link);
217                                                }
218                                        }
219                                }
220                        }
221                }
222
223                return clusterLinks;
224        }
225
226        /**
227         * @return the removedLinks
228         */
229        public final Set<Id<Link>> getRemovedLinkIds() {
230                return removedLinks;
231        }
232
233        /**
234         * @return the modifiedLinks
235         */
236        public final Set<Id<Link>> getModifiedLinkIds() {
237                return modifiedLinks;
238        }
239
240        /**
241         * An optimized method to find out if two sets have common elements.
242         * The basic approach would be to copy one set, and then use
243         * {@link Set#retainAll(java.util.Collection)} and see if the
244         * resulting set is not empty. But that way creates additional objects,
245         * which this method tries to avoid.
246         *
247         * @param <T> the type of objects in the two sets
248         * @param setA the first set
249         * @param setB the second set
250         * @return <code>true</code> if the intersection of two sets is not empty
251         */
252        private <T> boolean intersectingSets(final Set<T> setA, final Set<T> setB) {
253                for (T t : setA) {
254                        if (setB.contains(t)) {
255                                return true;
256                        }
257                }
258                return false;
259        }
260
261        private static DoubleFlagRole getDoubleFlag(final Link l, final Map<Id<Link>, DoubleFlagRole> linkRoles) {
262                DoubleFlagRole r = linkRoles.get(l.getId());
263                if (null == r) {
264                        r = new DoubleFlagRole();
265                        linkRoles.put(l.getId(), r);
266                }
267                return r;
268        }
269
270        static class DoubleFlagRole {
271                boolean forwardFlag = false;
272                boolean backwardFlag = false;
273        }
