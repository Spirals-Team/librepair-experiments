016/**
017 * Converts a network to an inverted network. Inverted nodes are situated at
018 * the end of the real link. Inverted link attributes are copied from toLink
019 * of the real network, thus every inverted link actually starts at the
020 * location of a real node.
021 * Each link Id of the real network corresponds to a node Id of the inverted network.
022 *
023 * @author aneumann
024 * @author dgrether
025 */
026public final class NetworkInverter {
027
028        final private static Logger log = Logger.getLogger(NetworkInverter.class);
029
030        private Network originalNetwork;
031
032        private Network invertedNetwork = null;
033
034        private Map<Id<Link>, List<TurnInfo>>  inLinkTurnInfoMap = null;
035
036        public NetworkInverter(Network originalNet, Map<Id<Link>, List<TurnInfo>>  inLinkTurnInfoMap) {
037                this.originalNetwork = originalNet;
038                this.inLinkTurnInfoMap = inLinkTurnInfoMap;
039        }
040
041        public Network getInvertedNetwork() {
042                if (this.invertedNetwork == null){
043                        invertNetwork();
044                }
045                return this.invertedNetwork;
046        }
047
048        private void invertNetwork(){
049                this.invertedNetwork = NetworkUtils.createNetwork();
050                int numberOfNodesGenerated = 0;
051                int numberOfLinksGenerated = 0;
052
053                for (Link link : this.originalNetwork.getLinks().values()) {
054                        NetworkUtils.createAndAddNode(this.invertedNetwork, Id.create(link.getId(), Node.class), link.getToNode().getCoord());
055                        numberOfNodesGenerated++;
056                }
057
058                for (Node node : this.originalNetwork.getNodes().values()) {
059                        for (Link inLink : node.getInLinks().values()) {
060                                for (Link outLink : node.getOutLinks().values()) {
061                                        List<TurnInfo> turnInfos = this.inLinkTurnInfoMap.get(inLink.getId());
062                                        TurnInfo ti = NetworkTurnInfoBuilder.getTurnInfoForOutlinkId(turnInfos, outLink.getId());
063                                        if (ti != null){
064                                                numberOfLinksGenerated = this.createInvertedLink(inLink, outLink, numberOfLinksGenerated, ti.getModes());
065                                        }
066                                }
067                        }
068                }
069
070                log.info("Generated " + numberOfNodesGenerated + " Nodes and " + numberOfLinksGenerated + " Links");
071
072                // Debug only
073                // NetworkWriter myNetworkWriter = new NetworkWriter(wrappedNetwork,
074                // "wrappedNetwork");
075                // myNetworkWriter.write();
076        }
077
078        private int createInvertedLink(Link inLink, Link outLink, int numberOfLinksGenerated, Set<String> modes){
079                Link link = NetworkUtils.createAndAddLink(this.invertedNetwork,Id.create(numberOfLinksGenerated + 1, Link.class), this.invertedNetwork.getNodes().get(Id.create(inLink.getId(), Node.class)), this.invertedNetwork.getNodes().get(Id.create(outLink.getId(), Node.class)), outLink.getLength(), outLink.getFreespeed(), outLink.getCapacity(), outLink.getNumberOfLanes() );
080                link.setAllowedModes(modes);
081//              log.error("created inverted link " + link.getId() + " from " + inLink.getId() + " to " + outLink.getId() + " with modes " + modes);
082                NetworkUtils.setType( ((Link) link), NetworkUtils.getType(((Link) outLink)));
083                return numberOfLinksGenerated + 1;
084        }
085
086        public List<Link> convertInvertedNodesToLinks(List<Node> nodes) {
087                List<Link> ret = new ArrayList<Link>(nodes.size());
088                for (Node n : nodes){
089                        ret.add(this.originalNetwork.getLinks().get(Id.create(n.getId(), Link.class)));
090                }
091                return ret;
092        }
093
094}
