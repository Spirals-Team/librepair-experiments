045/**
046 * Simplifies a given network, by merging links. All other criteria met, no 
047 * link should have a length less than a given threshold. This class is based
048 * on {@link NetworkSimplifier}. It goes through the network in two steps. 
049 * First it merges links if <i>both</i> are shorter then the threshold. It then
050 * goes through the network a <i>second</i> time and merges links that are
051 * shorter than the threshold with either of the links' mergeable neighbours.<br><br>
052 * 
053 * If no link threshold is given, an infinite threshold is assumed. This should
054 * behave the same as a 'clean' network.
055 *
056 * @author aneumann, jwjoubert
057 *
058 */
059public final class NetworkSimplifier {
060
061        private static final Logger log = Logger.getLogger(NetworkSimplifier.class);
062        private boolean mergeLinksWithDifferentAttributes = false;
063        private Collection<Integer> nodeTopoToMerge = Arrays.asList( NetworkCalcTopoType.PASS1WAY , NetworkCalcTopoType.PASS2WAY );
064
065        private Set<Id<Node>> nodesNotToMerge = new HashSet<Id<Node>>();
066
067        private final Map<Id<Link>,List<Node>> mergedLinksToIntermediateNodes = new HashMap<>();
068
069
070        /**
071         * Merges all qualifying links, ignoring length threshold.
072         * @param network
073         */
074        public void run(final Network network){
075                run(network, Double.POSITIVE_INFINITY, thresholdExceeded.EITHER);
076        }
077        
078        
079        /**
080         * Merges all qualifying links while ensuring no link is shorter than the
081         * given threshold.
082         * <br/>
083         * Comments:<ul>
084         *     <li>I would argue against setting the thresholdLength to anything different from POSITIVE_INFINITY, since
085         *     the result of the method depends on the sequence in which the algorithm goes through the nodes.  </li>
086         * </ul>
087         * @param network
088         * @param thresholdLength
089         */
090        @Deprecated
091        public void run(final Network network, double thresholdLength){
092                run(network, thresholdLength, thresholdExceeded.BOTH);
093                run(network, thresholdLength, thresholdExceeded.EITHER);
094        }
095
096        /**
097         * Specifies a set of nodes of which all outgoing and ingoing links should not be merged.
098         * Should probably not be used if nodes of type {@link NetworkCalcTopoType.INTERSECTION} are to be merged.
099         * tschlenther jun'17
100         * @param nodeIDs
101         */
102        public void setNodesNotToMerge(Set<Long> nodeIDs){
103                for(Long l : nodeIDs){
104                        this.nodesNotToMerge.add(Id.createNodeId(l));
105                }
106        }
107        
108        private void run(final Network network, double thresholdLength, thresholdExceeded type) {
109
110                if(this.nodeTopoToMerge.size() == 0){
111                        throw new RuntimeException("No types of node specified. Please use setNodesToMerge to specify which nodes should be merged");
112                }
113
114                log.info("running " + this.getClass().getName() + " algorithm...");
115
116                NetworkCalcTopoType nodeTopo = new NetworkCalcTopoType();
117                nodeTopo.run(network);
118
119                for (Node node : network.getNodes().values()) {
120                        
121                        if(this.nodeTopoToMerge.contains(Integer.valueOf(nodeTopo.getTopoType(node))) && (!this.nodesNotToMerge.contains(node.getId())) ){
122
123                                List<Link> iLinks = new ArrayList<Link> (node.getInLinks().values());
124
125                                for (Link iL : iLinks) {
126                                        Link inLink = (Link) iL;
127
128                                        List<Link> oLinks = new ArrayList<Link> (node.getOutLinks().values());
129
130                                        for (Link oL : oLinks) {
131                                                Link outLink = (Link) oL;
132
133                                                if(inLink != null && outLink != null){
134//                                                      if(!outLink.getToNode().equals(inLink.getFromNode())){
135                                                        if(  areLinksMergeable(inLink, outLink) ){
136                                                                if(this.mergeLinksWithDifferentAttributes){
137
138                                                                        // Only merge if threshold criteria is met.  
139                                                                        boolean criteria = false;
140                                                                        switch (type) {
141                                                                        case BOTH:
142                                                                                criteria = bothLinksAreShorterThanThreshold(inLink, outLink, thresholdLength);
143                                                                                break;
144                                                                        case EITHER:
145                                                                                criteria = eitherLinkIsShorterThanThreshold(inLink, outLink, thresholdLength);
146                                                                                break;
147                                                                        default:
148                                                                                break;
149                                                                        }
150                                                                        
151                                                                        // yyyy The approach here depends on the sequence in which this goes through the nodes:
152                                                                        // * in the "EITHER" situation, a long link may gobble up short neighboring links
153                                                                        // until it hits another long link doing the same.
154                                                                        // * In the "BOTH" situation, something like going through nodes randomly will often merge
155                                                                        // the neighboring links, while going through the nodes along some path will mean that it will
156                                                                        // gobble up until the threshold is met.
157                                                                        // I would strongly advise against setting thresholdLength to anything other than POSITIVE_INFINITY.
158                                                                        // kai, feb'18
159                                                                        
160                                                                        if(criteria){
161                                                                                // Try to merge both links by guessing the resulting links attributes
162                                                                                Link link = network.getFactory().createLink(
163                                                                                                Id.create(inLink.getId() + "-" + outLink.getId(), Link.class),
164                                                                                                inLink.getFromNode(),
165                                                                                                outLink.getToNode());
166                                                                                
167                                                                                // length can be summed up
168                                                                                link.setLength(inLink.getLength() + outLink.getLength());
169                                                                                
170                                                                                // freespeed depends on total length and time needed for inLink and outLink
171                                                                                link.setFreespeed(
172                                                                                                (inLink.getLength() + outLink.getLength()) /
173                                                                                                (NetworkUtils.getFreespeedTravelTime(inLink) + NetworkUtils.getFreespeedTravelTime(outLink))
174                                                                                                );
175                                                                                
176                                                                                // the capacity and the new links end is important, thus it will be set to the minimum
177                                                                                link.setCapacity(Math.min(inLink.getCapacity(), outLink.getCapacity()));
178                                                                                
179                                                                                // number of lanes can be derived from the storage capacity of both links
180                                                                                link.setNumberOfLanes((inLink.getLength() * inLink.getNumberOfLanes()
181                                                                                                + outLink.getLength() * outLink.getNumberOfLanes())
182                                                                                                / (inLink.getLength() + outLink.getLength())
183                                                                                                );
184                                                                                
185//                                                                      inLink.getOrigId() + "-" + outLink.getOrigId(),
186                                                                                network.addLink(link);
187                                                                                network.removeLink(inLink.getId());
188                                                                                network.removeLink(outLink.getId());
189                                                                                collectMergedLinkNodeInfo(inLink, outLink, link.getId());
190                                                                        }
191                                                                } else {
192
193                                                                        // Only merge links with same attributes
194                                                                        if(bothLinksHaveSameLinkStats(inLink, outLink)){
195                                                                                
196                                                                                // Only merge if threshold criteria is met.  
197                                                                                boolean isHavingShortLinks = false;
198                                                                                switch (type) {
199                                                                                case BOTH:
200                                                                                        isHavingShortLinks = bothLinksAreShorterThanThreshold(inLink, outLink, thresholdLength);
201                                                                                        break;
202                                                                                case EITHER:
203                                                                                        isHavingShortLinks = eitherLinkIsShorterThanThreshold(inLink, outLink, thresholdLength);
204                                                                                        break;
205                                                                                default:
206                                                                                        break;
207                                                                                }
208                                                                                                                                                                
209                                                                                if(isHavingShortLinks){
210                                                                                        Link newLink = NetworkUtils.createAndAddLink(((Network) network),Id.create(inLink.getId() + "-" + outLink.getId(), Link.class), inLink.getFromNode(), outLink.getToNode(), inLink.getLength() + outLink.getLength(), inLink.getFreespeed(), inLink.getCapacity(), inLink.getNumberOfLanes(), NetworkUtils.getOrigId( inLink ) + "-" + NetworkUtils.getOrigId( outLink ), null);
211                                                                                        
212                                                                                        newLink.setAllowedModes(inLink.getAllowedModes());
213                                                                                        
214                                                                                        network.removeLink(inLink.getId());
215                                                                                        network.removeLink(outLink.getId());
216                                                                                        collectMergedLinkNodeInfo(inLink, outLink, newLink.getId());
217                                                                                }
218                                                                        }
219                                                                }
220                                                        }
221                                                }
222                                        }
223                                }
224                        }
225                }
226
227                log.info("  resulting network contains " + network.getNodes().size() + " nodes and " +
228                                network.getLinks().size() + " links.");
229                log.info("done.");
230
231                // writes stats as a side effect
232                nodeTopo = new NetworkCalcTopoType();
233                nodeTopo.run(network);
234        }

236        private boolean areLinksMergeable(Link inLink, Link outLink) {
237                List<Node> fromNodes = this.mergedLinksToIntermediateNodes.get(inLink.getId());
238                if (fromNodes==null) fromNodes = new ArrayList<>();
239                fromNodes.add(inLink.getFromNode());
240
241                List<Node> toNodes = this.mergedLinksToIntermediateNodes.get(outLink.getId());
242                if(toNodes==null) toNodes = new ArrayList<>();
243                toNodes.add(outLink.getToNode());
244
245                for(Node n :fromNodes) {
246                        if (toNodes.contains(n)) return false;
247                }
248                return true;
249        }
250
251        private void collectMergedLinkNodeInfo(Link inLink, Link outLink, Id<Link> mergedLinkId) {
252                List<Node> nodes = new ArrayList<>();
253                if (this.mergedLinksToIntermediateNodes.containsKey(inLink.getId())) nodes.addAll(this.mergedLinksToIntermediateNodes
254                                .remove(inLink.getId()));
255                if (this.mergedLinksToIntermediateNodes.containsKey(outLink.getId())) nodes.addAll(this.mergedLinksToIntermediateNodes
256                                .remove(outLink.getId()));
257                nodes.add(inLink.getToNode());
258
259                this.mergedLinksToIntermediateNodes.put(mergedLinkId, nodes);
260        }

273        /**
274         *
275         * @param mergeLinksWithDifferentAttributes If set true, links will be merged despite their different attributes.
276         *  If set false, only links with the same attributes will be merged, thus preserving as much information as possible.
277         *  Default is set false.
278         */
279        public void setMergeLinkStats(boolean mergeLinksWithDifferentAttributes){
280                this.mergeLinksWithDifferentAttributes = mergeLinksWithDifferentAttributes;
281        }

285        /**
286         * Quick check to see whether <i>both</i> the links are shorter than the 
287         * given threshold.
288         * @param linkA
289         * @param linkB
290         * @param thresholdLength
291         * @return true if <i>both</i> links are shorter than the given threshold, 
292         * false otherwise. 
293         */
294        private boolean bothLinksAreShorterThanThreshold(Link linkA, Link linkB, double thresholdLength){
295                boolean hasTwoShortLinks = false;
296                if(linkA.getLength() < thresholdLength && linkB.getLength() < thresholdLength){
297                        hasTwoShortLinks = true;
298                }
299                return hasTwoShortLinks;
300        }
301        
302        /**
303         * Quick check to see whether <i>either</i> the links are shorter than the 
304         * given threshold.
305         * @param linkA
306         * @param linkB
307         * @param thresholdLength
308         * @return true if <i>either</i> links are shorter than the given threshold, 
309         * false otherwise. 
310         */
311        private boolean eitherLinkIsShorterThanThreshold(Link linkA, Link linkB, double thresholdLength){
312                boolean hasShortLink = false;
313                if(linkA.getLength() < thresholdLength || linkB.getLength() < thresholdLength){
314                        hasShortLink = true;
315                }
316                return hasShortLink;
317        }
318        
319        /**
320         * Compare link attributes. Return whether they are the same or not.
321         */
322        private boolean bothLinksHaveSameLinkStats(Link linkA, Link linkB){
323
324                boolean bothLinksHaveSameLinkStats = true;
325
326                if(!linkA.getAllowedModes().equals(linkB.getAllowedModes())){ bothLinksHaveSameLinkStats = false; }
327
328                if(linkA.getFreespeed() != linkB.getFreespeed()){ bothLinksHaveSameLinkStats = false; }
329
330                if(linkA.getCapacity() != linkB.getCapacity()){ bothLinksHaveSameLinkStats = false; }
331
332                if(linkA.getNumberOfLanes() != linkB.getNumberOfLanes()){ bothLinksHaveSameLinkStats = false; }
333
334                return bothLinksHaveSameLinkStats;
335        }

337        public static void main(String[] args) {
338                final String inNetworkFile = args[ 0 ];
339                final String outNetworkFile = args[ 1 ];
340
341                Set<Integer> nodeTypesToMerge = new TreeSet<Integer>();
342                nodeTypesToMerge.add(new Integer(4));
343                nodeTypesToMerge.add(new Integer(5));
344
345                Scenario scenario = ScenarioUtils.createScenario(ConfigUtils.createConfig());
346                final Network network = scenario.getNetwork();
347                new MatsimNetworkReader(scenario.getNetwork()).readFile( inNetworkFile );
348
349                NetworkSimplifier nsimply = new NetworkSimplifier();
350                nsimply.setNodesToMerge(nodeTypesToMerge);
351//              nsimply.setMergeLinkStats(true);
352                nsimply.run(network, Double.NEGATIVE_INFINITY);
353
354                new NetworkWriter(network).write( outNetworkFile );
355
356        }
