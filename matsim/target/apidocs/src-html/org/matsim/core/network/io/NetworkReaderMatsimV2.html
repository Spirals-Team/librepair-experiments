044/**
045 * A reader for network-files of MATSim according to <code>network_v1.dtd</code>.
046 *
047 * @author mrieser
048 */
049public final class NetworkReaderMatsimV2 extends MatsimXmlParser {
050
051        private final static String NETWORK = "network";
052        private final static String LINKS = "links";
053        private final static String NODE = "node";
054        private final static String LINK = "link";
055        private final static String ATTRIBUTES = "attributes";
056        private final static String ATTRIBUTE = "attribute";
057
058        private final Network network;
059
060        private final AttributesXmlReaderDelegate attributesDelegate = new AttributesXmlReaderDelegate();
061        private org.matsim.utils.objectattributes.attributable.Attributes currentAttributes = null;
062
063        // final or settable?
064        private final CoordinateTransformation transformation;
065
066        private final static Logger log = Logger.getLogger(NetworkReaderMatsimV2.class);
067
068        public NetworkReaderMatsimV2(Network network) {
069                this( new IdentityTransformation() , network );
070        }
071
072        public NetworkReaderMatsimV2(
073                        final CoordinateTransformation transformation,
074                        final Network network) {
075                this.transformation = transformation;
076                this.network = network;
077        }
078
079        @Override
080        public void startTag(final String name, final Attributes atts, final Stack<String> context) {
081                switch( name ) {
082                        case NODE:
083                                startNode(atts);
084                                break;
085                        case LINK:
086                                startLink(atts);
087                                break;
088                        case NETWORK:
089                                startNetwork(atts);
090                                break;
091                        case LINKS:
092                                startLinks(atts);
093                                break;
094                        case ATTRIBUTES:
095                        case ATTRIBUTE:
096                                attributesDelegate.startTag( name , atts , context , currentAttributes );
097                                break;
098                }
099        }
100
101        @Override
102        public void endTag(final String name, final String content, final Stack<String> context) {
103                switch( name ) {
104                        case ATTRIBUTES:
105                        case ATTRIBUTE:
106                                attributesDelegate.endTag(name, content, context);
107                                break;
108                }
109        }
110
111        private void startNetwork(final Attributes atts) {
112                if (atts.getValue("type") != null) {
113                        log.info("Attribute 'type' is deprecated. There's always only ONE network, where the links and nodes define, which " +
114                                        "transportation mode is allowed to use it (for the future)");
115                }
116                this.network.setName(atts.getValue("name"));
117                if (atts.getValue("capDivider") != null) {
118                        log.warn("capDivider defined. it will be used but should be gone eventually. " +
119                                        "-- This is a weird comment, since the matsim public api tells to put this into the network rather than" +
120                                        " into the ``links''.  kai, jun'11");
121                        String capperiod = atts.getValue("capDivider") + ":00:00";
122                        this.network.setCapacityPeriod(Time.parseTime(capperiod));
123                }
124
125                currentAttributes = network.getAttributes();
126        }
127
128        private void startLinks(final Attributes atts) {
129                double capacityPeriod = 3600.0; //the default of one hour
130                String capperiod = atts.getValue("capperiod");
131                if (capperiod != null) {
132                        capacityPeriod = Time.parseTime(capperiod);
133                }
134                else {
135                        log.warn("capperiod was not defined. Using default value of " + Time.writeTime(capacityPeriod) + ".");
136                }
137                this.network.setCapacityPeriod(capacityPeriod);
138
139                String effectivecellsize = atts.getValue("effectivecellsize");
140                if (effectivecellsize == null){
141                        this.network.setEffectiveCellSize(7.5); // we use a default cell size of 7.5 meters
142                } else {
143                        this.network.setEffectiveCellSize(Double.parseDouble(effectivecellsize));
144                }
145
146                String effectivelanewidth = atts.getValue("effectivelanewidth");
147                if (effectivelanewidth == null) {
148                        this.network.setEffectiveLaneWidth(3.75); // the default lane width is 3.75
149                } else {
150                        this.network.setEffectiveLaneWidth(Double.parseDouble(effectivelanewidth));
151                }
152
153                if ((atts.getValue("capPeriod") != null) || (atts.getValue("capDivider") != null) || (atts.getValue("capdivider") != null)) {
154                        log.warn("Found capPeriod, capDivider and/or capdivider in the links element.  They will be ignored, since they " +
155                                        "should be set in the network element. -- This is a weird warning, since setting them in the " +
156                                        "network element also produces a warning.");
157                        log.warn("At this point, it seems that, in network.xml, one sets capperiod in the `links' section, but in the " +
158                                        "matsim api, the corresponding entry belongs into the `network' object. kai, jun'11") ;
159                }
160        }
161
162        private void startNode(final Attributes atts) {
163                final Node node =
164                                this.network.getFactory().createNode(
165                                                Id.create(atts.getValue("id"), Node.class),
166                                                parseCoord(atts));
167                this.network.addNode(node);
168
169                NetworkUtils.setType(node,atts.getValue("type"));
170                // (did not have a null check when I found it.  kai, jul'16) 
171
172                if (atts.getValue(NetworkUtils.ORIGID) != null) {
173                        NetworkUtils.setOrigId( node, atts.getValue(NetworkUtils.ORIGID) ) ;
174                }
175
176                currentAttributes = node.getAttributes();
177        }
178
179        private Coord parseCoord(Attributes atts) {
180                final Coord c = atts.getValue( "z" ) == null ?
181                                new Coord(
182                                                Double.parseDouble(atts.getValue("x")),
183                                                Double.parseDouble(atts.getValue("y"))) :
184                                new Coord(
185                                                Double.parseDouble(atts.getValue("x")),
186                                                Double.parseDouble(atts.getValue("y")),
187                                                Double.parseDouble(atts.getValue("z")));
188                return transformation.transform( c );
189        }
190
191        private void startLink(final Attributes atts) {
192                final String fromNodeStr = atts.getValue("from");
193                Node fromNode = this.network.getNodes().get(Id.create(fromNodeStr, Node.class));
194                if ( fromNode==null ) {
195                        throw new RuntimeException("node id given by link cannot be dereferenced; node label=" + fromNodeStr ) ;
196                }
197                final String toNodeStr = atts.getValue("to");
198                Node toNode = this.network.getNodes().get(Id.create(toNodeStr, Node.class));
199                if ( toNode==null ) {
200                        throw new RuntimeException("node id given by link cannot be dereferenced; node label=" + toNodeStr ) ;
201                }
202                Link l = this.network.getFactory().createLink(Id.create(atts.getValue("id"), Link.class), fromNode, toNode);
203                l.setLength(Double.parseDouble(atts.getValue("length")));
204                l.setFreespeed(Double.parseDouble(atts.getValue("freespeed")));
205                l.setCapacity(Double.parseDouble(atts.getValue("capacity")));
206                l.setNumberOfLanes(Double.parseDouble(atts.getValue("permlanes")));
207                this.network.addLink(l);
208                {
209                        String value = atts.getValue(NetworkUtils.ORIGID);
210                        if ( value != null ) {
211                                NetworkUtils.setOrigId( (l), value ) ; // will now put it into the attributes. kai, dec'16
212                        }
213                }
214                {
215                        final String value = atts.getValue(NetworkUtils.TYPE);
216                        if ( value != null ) {
217                                NetworkUtils.setType( (l), value); // will now put it into the attributes. kai, dec'16
218                        }
219                }
220                
221                if (atts.getValue("modes") != null) {
222                        String[] strModes = StringUtils.explode(atts.getValue("modes"), ',');
223                        if ((strModes.length == 1) && strModes[0].isEmpty()) {
224                                l.setAllowedModes(new HashSet<String>());
225                        } else {
226                                Set<String> modes = new HashSet<>();
227                                for (String strMode : strModes) {
228                                        modes.add(strMode.trim().intern());
229                                }
230                                l.setAllowedModes(modes);
231                        }
232                }
233
234                currentAttributes = l.getAttributes();
235        }
236
237        public void putAttributeConverters(Map<Class<?>, AttributeConverter<?>> converters) {
238                this.attributesDelegate.putAttributeConverters( converters );
239        }
240}
