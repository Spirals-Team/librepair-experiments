039    protected final Scenario scenario;
040
041    @Inject
042    public NetworkTurnInfoBuilder(Scenario scenario)
043    {
044        this.scenario = scenario;
045    }

047    
048    /* (non-Javadoc)
049 * @see org.matsim.core.network.algorithms.NetworkTurnInfoBuilderI#createAllowedTurnInfos()
050 */
051@Override
052public Map<Id<Link>, List<TurnInfo>> createAllowedTurnInfos(){
053        Map<Id<Link>, List<TurnInfo>> allowedInLinkTurnInfoMap = new HashMap<>();
054    
055        createAndAddTurnInfo(TransportMode.car, allowedInLinkTurnInfoMap);
056    
057        if ( scenario.getConfig().network().getLaneDefinitionsFile()!=null || //
058                scenario.getConfig().qsim().isUseLanes()) {
059            Lanes ld = scenario.getLanes();
060            Map<Id<Link>, List<TurnInfo>> lanesTurnInfoMap = createTurnInfos(ld);
061            mergeTurnInfoMaps(allowedInLinkTurnInfoMap, lanesTurnInfoMap);
062        }
063        return allowedInLinkTurnInfoMap;
064    }
065
066    
067    private Map<Id<Link>, List<TurnInfo>> createTurnInfos(Lanes laneDefs) {
068        Map<Id<Link>, List<TurnInfo>> inLinkIdTurnInfoMap = new HashMap<>();
069        Set<Id<Link>> toLinkIds = new HashSet<>();
070        for (LanesToLinkAssignment l2l : laneDefs.getLanesToLinkAssignments().values()) {
071            toLinkIds.clear();
072            for (Lane lane : l2l.getLanes().values()) {
073                if (lane.getToLinkIds() != null
074                        && (lane.getToLaneIds() == null || lane.getToLaneIds().isEmpty())) { // make sure that it is a lane at the end of a link
075                    toLinkIds.addAll(lane.getToLinkIds());
076                }
077            }
078            if (!toLinkIds.isEmpty()) {
079                List<TurnInfo> turnInfoList = new ArrayList<TurnInfo>();
080                for (Id<Link> toLinkId : toLinkIds) {
081                    turnInfoList.add(new TurnInfo(l2l.getLinkId(), toLinkId));
082                }
083                inLinkIdTurnInfoMap.put(l2l.getLinkId(), turnInfoList);
084            }
085        }
086    
087        return inLinkIdTurnInfoMap;
088    }

092        /**
093         * Creates a List of TurnInfo objects for every existing link of the network. If the links have mode attributes set,
094         * those are considered in TurnInfo creation.
095         */
096        private void createAndAddTurnInfo(String mode, Map<Id<Link>, List<TurnInfo>> inLinkTurnInfoMap) {
097                TurnInfo turnInfo = null;
098                Set<String> modes = null;
099                List<TurnInfo> turnInfosForInLink = null;
100                for (Node node : scenario.getNetwork().getNodes().values()) {
101                        for (Link inLink : node.getInLinks().values()) {
102                                turnInfosForInLink = inLinkTurnInfoMap.get(inLink.getId());
103                                if (turnInfosForInLink == null) {
104                                        turnInfosForInLink = new ArrayList<TurnInfo>();
105                                        inLinkTurnInfoMap.put(inLink.getId(), turnInfosForInLink);
106                                }
107
108                                for (Link outLink : node.getOutLinks().values()) {
109                                        if (! inLink.getAllowedModes().isEmpty() && ! outLink.getAllowedModes().isEmpty()) {
110                                                if (inLink.getAllowedModes().contains(mode) && outLink.getAllowedModes().contains(mode)) {
111                                                        modes = new HashSet<String>();
112                                                        modes.add(mode);
113                                                        turnInfo = new TurnInfo(inLink.getId(), outLink.getId(), modes);
114                                                        turnInfosForInLink.add(turnInfo);
115                                                }
116                                        }
117                                        else { // we have no mode information at all
118                                                turnInfo = new TurnInfo(inLink.getId(), outLink.getId());
119                                                turnInfosForInLink.add(turnInfo);
120                                        }
121                                }
122                        }
123                }
124        }
125
126        /**
127         * Modifies the first Map containing the allowed turning moves: All turning moves of a fromLink for that the second
128         * Map contains (only in this case!) an entry are checked for differences concerning outLinks and modes. If an outLink
129         * or mode is not contained in the restriction, the corresponding TurnInfo or mode is removed or modified in the first
130         * map.
131         */
132        public final void mergeTurnInfoMaps(Map<Id<Link>, List<TurnInfo>> allowedInLinkTurnInfoMap,
133                        Map<Id<Link>, List<TurnInfo>> restrictingTurnInfoMap) {
134                for (Map.Entry<Id<Link>, List<TurnInfo>> e : allowedInLinkTurnInfoMap.entrySet()) {
135            Id<Link> inLinkId = e.getKey();
136            List<TurnInfo> restrictingTurnInfos = restrictingTurnInfoMap.get(inLinkId);
137
138            if (restrictingTurnInfos != null) { // there are restrictions for the inLink
139                                List<TurnInfo> allowedTurnInfos = new ArrayList<TurnInfo>(e.getValue());
140                                for (TurnInfo allowedForOutlink : allowedTurnInfos) {
141                                        TurnInfo restrictionForOutlink = getTurnInfoForOutlinkId(
142                                                        restrictingTurnInfos, allowedForOutlink.getToLinkId());
143                                        if (restrictionForOutlink == null) { // there is no turn at all allowed from the inLink to the outLink
144                                                allowedInLinkTurnInfoMap.get(inLinkId).remove(allowedForOutlink);
145                                        }
146                                        else { // turns are restricted to some modes or allowed without any mode information
147                                                if (restrictionForOutlink.getModes() != null && allowedForOutlink.getModes() != null){
148                                                        Set<String> commonModes = this.calculateCommonModes(
149                                                                        restrictionForOutlink, allowedForOutlink);
150                                                        Set<String> allowedModes = allowedForOutlink.getModes();
151                                                        for (String mode : allowedModes) {
152                                                                if (!commonModes.contains(mode)) {
153                                                                        allowedForOutlink.getModes().remove(mode);
154                                                                }
155                                                        }       
156                                                }
157                                        }
158                                }
159                        }
160                }
161        }
162
163        /**
164         * Linear search for the TurnInfo describing the turn to the outLinkId
165         */
166        static TurnInfo getTurnInfoForOutlinkId(List<TurnInfo> turnInfoList, Id<Link> outLinkId) {
167                for (TurnInfo ti : turnInfoList) {
168                        if (ti.getToLinkId().equals(outLinkId)) {
169                                return ti;
170                        }
171                }
172                return null;
173        }
174
175        private Set<String> calculateCommonModes(TurnInfo first, TurnInfo second) {
176                Set<String> modes = new HashSet<String>();
177                for (String mode : first.getModes()) {
178                        if (second.getModes().contains(mode)) {
179                                modes.add(mode);
180                        }
181                }
182                return modes;
183        }
