067        /**
068         * @return The bounding box of all the given nodes as <code>double[] = {minX, minY, maxX, maxY}</code>
069         */
070        public static double[] getBoundingBox(final Collection<? extends Node> nodes) {
071                double[] bBox = new double[4];
072                bBox[0] = Double.POSITIVE_INFINITY;
073                bBox[1] = Double.POSITIVE_INFINITY;
074                bBox[2] = Double.NEGATIVE_INFINITY;
075                bBox[3] = Double.NEGATIVE_INFINITY;
076
077                for (Node n : nodes) {
078                        if (n.getCoord().getX() < bBox[0]) {
079                                bBox[0] = n.getCoord().getX();
080                        }
081                        if (n.getCoord().getX() > bBox[2]) {
082                                bBox[2] = n.getCoord().getX();
083                        }
084                        if (n.getCoord().getY() > bBox[3]) {
085                                bBox[3] = n.getCoord().getY();
086                        }
087                        if (n.getCoord().getY() < bBox[1]) {
088                                bBox[1] = n.getCoord().getY();
089                        }
090                }
091                return bBox;
092        }

108        /**
109         * @param nodes list of node ids, separated by one or multiple whitespace (space, \t, \n)
110         * @return list containing the specified nodes.
111         * @throws IllegalArgumentException if a specified node is not found in the network
112         */
113        public static List<Node> getNodes(final Network network, final String nodes) {
114                if (nodes == null) {
115                        return new ArrayList<>(0);
116                }
117                String trimmed = nodes.trim();
118                if (trimmed.length() == 0) {
119                        return new ArrayList<>(0);
120                }
121                String[] parts = trimmed.split("[ \t\n]+");
122                final List<Node> nodesList = new ArrayList<>(parts.length);
123
124                for (String id : parts) {
125                        Node node = network.getNodes().get(Id.create(id, Node.class));
126                        if (node == null) {
127                                throw new IllegalArgumentException("no node with id " + id);
128                        }
129                        nodesList.add(node);
130                }
131                return nodesList;
132        }

232        public static boolean isMultimodal(final Network network) {
233                String mode = null;
234                boolean hasEmptyModes = false;
235                for (Link link : network.getLinks().values()) {
236                        Set<String> modes = link.getAllowedModes();
237                        if (modes.size() > 1) {
238                                return true; // it must be multimodal with more than 1 mode
239                        } else if (modes.size() == 1) {
240                                String m2 = modes.iterator().next();
241                                if (mode == null) {
242                                        if (hasEmptyModes) {
243                                                // i.e. we have a mode restriction on the current link but not mode restriction on some other link. => multi-modal
244                                                return true;
245                                        }
246                                        mode = m2;
247                                        // (memorize that we have seen a link with a mode restriction)
248                                } else {
249                                        if (!m2.equals(mode)) {
250                                                // i.e. we have a mode restriction on the current link, and some other mode restriction on some other link. => multi-modal
251                                                return true;
252                                        }
253                                        // position here can be reached with "mode!=null" and "hasEmptyModes==true".  Should return "true" here but does not.
254                                        // Works anyways, since (it seems to me) that that state (mode!=null, hasEmptyModes==true) can never be reached.
255                                        // ???  kai, feb'15
256                                }
257                        } else if (modes.size() == 0) {
258                                if (mode != null) {
259                                        // i.e. we have no mode restriction on the current link, but a mode restriction on some other link. => multi-modal:
260                                        return true;
261                                }
262                                hasEmptyModes = true;
263                                // (memorize that we have seen a link without mode restrictions)
264                        }
265                }
266                return false;
267                
268        }

313    /**
314     * Finds the (approx.) nearest link to a given point on the map,
315     * such that the point lies on the right side of the directed link,
316     * if such a link exists.
317         *
318     * It searches first for the nearest node, and then for the nearest link
319     * originating or ending at that node and fulfilling the above constraint.
320     * <p>
321     * <b>Special cases:</b> {@code nodes:o ; links:<-- ; coord:x}
322     * <i>No right entry link exists</i>
323     * <pre>
324         * {@code
325     * o<-1--o returning
326     * | . . ^ nearest left
327     * |2 . 4| entry link
328     * v .x. | (link.id=3)
329     * o--3->o<br>
330         * }
331     * </pre>
332     * <i>No right entry link exists but more than one nearest left entry link exist</i>
333         * <pre>
334         * {@code
335         * o<-1--o returning
336     * | . . ^ nearest left
337     * |2 x 4| entry link with the
338     * v . . | lowest link id
339     * o--3->o (link.id=1)
340         * }
341         * </pre>
342     * <i>More than one nearest right entry link exist</i>
343         * <pre>
344         * {@code
345     * o--1->o returning
346     * ^ . . | nearest right
347     * |2 x 4| entry link with the
348     * | . . v lowest link id
349     * o<-3--o (link.id=1)
350         *
351     * o<----7&8--x->o (link.id=7)
352         * }
353         * </pre>
354     *
355     * @param coord
356     *          the coordinate for which the closest link should be found
357     * @return the link found closest to <code>coord</code> and oriented such that the
358     * point lies on the right of the link.
359     */
360    // TODO [balmermi] there should be only one 'getNearestLink' method
361    // which returns either the nearest 'left' or 'right' entry link, based on a global
362    // config param.
363    public static Link getNearestRightEntryLink(Network network, final Coord coord) {
364        Link nearestRightLink = null;
365        Link nearestOverallLink = null;
366        Node nearestNode = NetworkUtils.getNearestNode((network),coord);
367
368        double[] coordVector = new double[2];
369        coordVector[0] = nearestNode.getCoord().getX() - coord.getX();
370        coordVector[1] = nearestNode.getCoord().getY() - coord.getY();
371
372        // now find nearest link from the nearest node
373        double shortestRightDistance = Double.MAX_VALUE; // reset the value
374        double shortestOverallDistance = Double.MAX_VALUE; // reset the value
375        List<Link> incidentLinks = new ArrayList<>(nearestNode.getInLinks().values());
376        incidentLinks.addAll(nearestNode.getOutLinks().values());
377        for (Link link : incidentLinks) {
378                double dist = CoordUtils.distancePointLinesegment(link.getFromNode().getCoord(), link.getToNode().getCoord(), coord);
379            if (dist <= shortestRightDistance) {
380                // Generate a vector representing the link
381                double[] linkVector = new double[2];
382                linkVector[0] = link.getToNode().getCoord().getX()
383                        - link.getFromNode().getCoord().getX();
384                linkVector[1] = link.getToNode().getCoord().getY()
385                        - link.getFromNode().getCoord().getY();
386
387                // Calculate the z component of cross product of coordVector and the link
388                double crossProductZ = coordVector[0]*linkVector[1] - coordVector[1]*linkVector[0];
389                // If coord lies to the right of the directed link, i.e. if the z component
390                // of the cross product is negative, set it as new nearest link
391                if (crossProductZ < 0) {
392                    if (dist < shortestRightDistance) {
393                        shortestRightDistance = dist;
394                        nearestRightLink = link;
395                    }
396                    else { // dist == shortestRightDistance
397                        if (link.getId().compareTo(nearestRightLink.getId()) < 0) {
398                            shortestRightDistance = dist;
399                            nearestRightLink = link;
400                        }
401                    }
402                }
403            }
404            if (dist < shortestOverallDistance) {
405                shortestOverallDistance = dist;
406                nearestOverallLink = link;
407            }
408            else if (dist == shortestOverallDistance) {
409                if (link.getId().compareTo(nearestOverallLink.getId()) < 0) {
410                    shortestOverallDistance = dist;
411                    nearestOverallLink = link;
412                }
413            }
414        }
415
416        // Return the nearest overall link if there is no nearest link
417        // such that the given coord is on the right side of it
418        if (nearestRightLink == null) {
419            return nearestOverallLink;
420        }
421        return nearestRightLink;
422    }

424    /**
425     * Finds the (approx.) nearest link to a given point on the map.
426     * It searches first for the nearest node, and then for the nearest link
427     * originating or ending at that node.
428     *
429     * @param coord
430     *          the coordinate for which the closest link should be found
431     * @return the link found closest to coord
432     * 
433     * @see {@link NetworkUtils#getNearestLinkExactly(Network, Coord)}
434     */
435    public static Link getNearestLink(Network network, final Coord coord) {
436        Link nearestLink = null;
437        Node nearestNode = NetworkUtils.getNearestNode((network),coord);
438        if ( nearestNode == null ) {
439            log.warn("[nearestNode not found.  Will probably crash eventually. Maybe run NetworkCleaner?  " +
440                                                         "Also may mean that network for mode is not defined.]" + network) ;
441            return null ;
442        }
443
444        if ( nearestNode.getInLinks().isEmpty() && nearestNode.getOutLinks().isEmpty() ) {
445            log.warn(network + "[found nearest node that has no incident links.  Will probably crash eventually ...  Maybe run NetworkCleaner?]" ) ;
446        }
447
448        // now find nearest link from the nearest node
449        // [balmermi] it checks now ALL incident links, not only the outgoing ones.
450        // TODO [balmermi] Now it finds the first of the typically two nearest links (same nodes, other direction)
451        // It would be nicer to find the nearest link on the "right" side of the coordinate.
452        // (For Great Britain it would be the "left" side. Could be a global config param...)
453        double shortestDistance = Double.MAX_VALUE;
454        for (Link link : getIncidentLinks(nearestNode).values()) {
455                double dist = CoordUtils.distancePointLinesegment(link.getFromNode().getCoord(), link.getToNode().getCoord(), coord);
456            if (dist < shortestDistance) {
457                shortestDistance = dist;
458                nearestLink = link;
459            }
460        }
461        if ( nearestLink == null ) {
462            log.warn(network + "[nearestLink not found.  Will probably crash eventually ...  Maybe run NetworkCleaner?]" ) ;
463        }
464        return nearestLink;
465    }

468         * Calculates the most 'left' outLink for a given inLink (oriented from north to south).
469         * That's the link a driver would refer to when turning left (no u-turn),
470         * even if there is only one link going to the right.
471         *
472         * @param inLink The inLink given
473         * @return outLink, or null if there is only one outLink back to the inLink's fromNode.

475        public static Link getLeftmostTurnExcludingU(Link inLink){
476
477                TreeMap<Double, Link> result = getOutLinksSortedClockwiseByAngle(inLink);
478
479                if (result.size() == 0){
480                        return null;
481                }
482                return result.get(result.firstKey());

485        /**
486         * Calculates the orientation of outgoing links for a given 
487         * incoming link beginning from the right if the inLink goes 
488         * north to south. The most 'left' outLink comes last. The link back to the 
489         * inLink's upstream Node is ignored.
490         * <br><br>
491         * Comments/questions:<ul>
492         * <li> What does the "north to south" part mean?  Can't we just sort outLinks right to left no matter where we come from?  kai, aug'16
493         * <li> In fact, a test (NetworkUtilsTest.getOutLinksSortedByAngleTest()) does not confirm the javadoc.  The "north to south"
494         * is irrelevant.  But instead, it sorts links <i> left to right </i> instead of right to left.
495         * <br>
496         * </ul>
497         * If someone can confirm this, please comment out all these remarks. kai, aug'16
498         * <br><br>
499         * @param inLink The inLink given
500         * @return Collection of outLinks, or an empty collection if there are only
501         *  outLinks back to the inLink's fromNode.
502         */
503        public static TreeMap<Double, Link> getOutLinksSortedClockwiseByAngle(Link inLink){
504                Coord coordInLink = getVector(inLink);
505                double thetaInLink = Math.atan2(coordInLink.getY(), coordInLink.getX());
506                TreeMap<Double, Link> outLinksByOrientation = new TreeMap<>();
507
508                for (Link outLink : inLink.getToNode().getOutLinks().values()) {
509                        if (!(outLink.getToNode().equals(inLink.getFromNode()))){
510                                Coord coordOutLink = getVector(outLink);
511                                double thetaOutLink = Math.atan2(coordOutLink.getY(), coordOutLink.getX());
512                                double thetaDiff = thetaOutLink - thetaInLink;
513                                if (thetaDiff < -Math.PI){
514                                        thetaDiff += 2 * Math.PI;
515                                } else if (thetaDiff > Math.PI){
516                                        thetaDiff -= 2 * Math.PI;
517                                }
518                                outLinksByOrientation.put(Double.valueOf(-thetaDiff), outLink);
519                        }
520                }
521                return outLinksByOrientation;
522        }

524        private static Coord getVector(Link link){
525                double x = link.getToNode().getCoord().getX() - link.getFromNode().getCoord().getX();
526                double y = link.getToNode().getCoord().getY() - link.getFromNode().getCoord().getY();
527                return new Coord(x, y);
528        }

683        public static Link createAndAddLink(Network network, final Id<Link> id, final Node fromNode, final Node toNode, final double length, final double freespeed,
684                                final double capacity, final double numLanes, final String origId, final String type) {
685                if (network.getNodes().get(fromNode.getId()) == null) {
686                        throw new IllegalArgumentException(network+"[from="+fromNode+" does not exist]");
687                }
688        
689                if (network.getNodes().get(toNode.getId()) == null) {
690                        throw new IllegalArgumentException(network+"[to="+toNode+" does not exist]");
691                }
692        
693                Link link = network.getFactory().createLink(id, fromNode, toNode) ;
694                link.setLength(length);
695                link.setFreespeed(freespeed);
696                link.setCapacity(capacity);
697                link.setNumberOfLanes(numLanes);
698                setType( link, type);
699                setOrigId( link, origId ) ;
700        
701                network.addLink( link ) ;
702        
703                return link;
704        }
