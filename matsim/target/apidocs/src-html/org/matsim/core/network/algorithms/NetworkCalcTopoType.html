021package org.matsim.core.network.algorithms;
022
023import java.util.HashMap;
024import java.util.IdentityHashMap;
025import java.util.Map;
026
027import org.matsim.api.core.v01.Id;
028import org.matsim.api.core.v01.network.Link;
029import org.matsim.api.core.v01.network.Network;
030import org.matsim.api.core.v01.network.Node;
031import org.matsim.core.api.internal.NetworkRunnable;
032import org.matsim.core.utils.misc.Counter;
033
034/** See "http://www.ivt.ethz.ch/vpl/publications/reports/ab283.pdf"
035 * for a description of node types. It's the graph matching paper.
036 *
037 * @author balmermi
038 **/
039public final class NetworkCalcTopoType implements NetworkRunnable {
040
041        public final static Integer EMPTY        = 0;
042        public final static Integer SOURCE       = 1;
043        public final static Integer SINK         = 2;
044        public final static Integer DEADEND      = 3;
045        public final static Integer PASS1WAY     = 4;
046        public final static Integer PASS2WAY     = 5;
047        public final static Integer START1WAY    = 6;
048        public final static Integer END1WAY      = 7;
049        public final static Integer INTERSECTION = 8;
050        // yy IMO, replace the above by an enum.  kai, dec'17
051
052        private final Map<Node, Integer> topoTypePerNode = new IdentityHashMap<>(100000);
053
054        @Override
055        public void run(final Network network) {
056                System.out.println("    running " + this.getClass().getName() + " algorithm...");
057                Counter ctr = new Counter("node #");
058
059                for (Node n : network.getNodes().values()) {
060                        ctr.incCounter();
061                        int nOfInLinks = n.getInLinks().size();
062                        int nOfOutLinks = n.getOutLinks().size();
063                        if ((nOfInLinks + nOfOutLinks) == 0) { setTopoType(n, EMPTY); }
064                        else if (nOfInLinks == 0) { setTopoType(n, SOURCE); }
065                        else if (nOfOutLinks == 0) {setTopoType(n, SINK); }
066                        else if (getNOfIncidentNodes(n) == 1) { setTopoType(n, DEADEND); }
067                        else if (getNOfIncidentNodes(n) == 2) {
068                                if ((nOfOutLinks == 1) && (nOfInLinks == 1)) { setTopoType(n, PASS1WAY); }
069                                else if ((nOfOutLinks == 2) && (nOfInLinks == 2)) { setTopoType(n, PASS2WAY); }
070                                else if ((nOfOutLinks == 2) && (nOfInLinks == 1)) { setTopoType(n, START1WAY); }
071                                else if ((nOfOutLinks == 1) && (nOfInLinks == 2)) { setTopoType(n, END1WAY); }
072                                // The following case is not covered by the paper, but quite common, e.g. parallel roads connecting the same nodes.
073                                else if ((nOfOutLinks >= 1) && (nOfInLinks >= 1)) {
074                                        // yyyy intellij says that this condition is always true when the code comes to here.  kai, dec'17
075                                        setTopoType(n, INTERSECTION);
076                                }
077                                else { throw new RuntimeException("Node=" + n.toString() + " cannot be assigned to a topo type!"); }
078                        }
079                        else { // more than two neighbors nodes and no sink or source
080                                setTopoType(n, INTERSECTION);
081                        }
082                }
083
084                int [] cnt = {0,0,0,0,0,0,0,0,0};
085                for (Node n : network.getNodes().values()) {
086                        cnt[getTopoType(n)]++;
087                }
088
089                System.out.println("      #nodes        = " + network.getNodes().size());
090                System.out.println("      #EMTPY        = " + cnt[EMPTY]);
091                System.out.println("      #SOURCE       = " + cnt[SOURCE]);
092                System.out.println("      #SINK         = " + cnt[SINK]);
093                System.out.println("      #DEADEND      = " + cnt[DEADEND]);
094                System.out.println("      #PASS1WAY     = " + cnt[PASS1WAY]);
095                System.out.println("      #PASS2WAY     = " + cnt[PASS2WAY]);
096                System.out.println("      #START1WAY    = " + cnt[START1WAY]);
097                System.out.println("      #END1WAY      = " + cnt[END1WAY]);
098                System.out.println("      #INTERSECTION = " + cnt[INTERSECTION]);
099
100                System.out.println("    done.");
101        }
102
103        private void setTopoType(final Node node, final Integer topoType) {
104                this.topoTypePerNode.put(node, topoType);
105        }
106
107        public int getTopoType(final Node node) {
108                Integer i = this.topoTypePerNode.get(node);
109                if (i == null) {
110                        return Integer.MIN_VALUE;
111                }
112                return i;
113        }
114
115        private int getNOfIncidentNodes(final Node node) {
116                HashMap<Id<Node>, Node> nodes = new HashMap<>();
117                for (Link link : node.getInLinks().values()) {
118                        nodes.put(link.getFromNode().getId(), link.getFromNode());
119                }
120                for (Link link : node.getOutLinks().values()) {
121                        nodes.put(link.getToNode().getId(), link.getToNode());
122                }
123                return nodes.size();
124        }
125}
