021package org.matsim.core.network.algorithms;
022
023import java.util.HashMap;
024import java.util.IdentityHashMap;
025import java.util.Map;
026
027import org.apache.log4j.Logger;
028import org.matsim.api.core.v01.Id;
029import org.matsim.api.core.v01.network.Link;
030import org.matsim.api.core.v01.network.Network;
031import org.matsim.api.core.v01.network.Node;
032import org.matsim.core.api.internal.NetworkRunnable;
033import org.matsim.core.utils.misc.Counter;
034
035/** See "http://www.ivt.ethz.ch/vpl/publications/reports/ab283.pdf"
036 * for a description of node types. It's the graph matching paper.
037 *
038 * @author balmermi
039 **/
040public final class NetworkCalcTopoType implements NetworkRunnable {
041        private static final Logger log = Logger.getLogger(NetworkCalcTopoType.class) ;
042
043        public final static Integer EMPTY        = 0;
044        public final static Integer SOURCE       = 1;
045        public final static Integer SINK         = 2;
046        public final static Integer DEADEND      = 3;
047        public final static Integer PASS1WAY     = 4;
048        public final static Integer PASS2WAY     = 5;
049        public final static Integer START1WAY    = 6;
050        public final static Integer END1WAY      = 7;
051        public final static Integer INTERSECTION = 8;
052        // yy IMO, replace the above by an enum.  kai, dec'17
053
054        private final Map<Node, Integer> topoTypePerNode = new IdentityHashMap<>(100000);
055
056        @Override
057        public void run(final Network network) {
058                log.info("    running " + this.getClass().getName() + " algorithm...");
059                Counter ctr = new Counter("node #");
060
061                for (Node n : network.getNodes().values()) {
062                        ctr.incCounter();
063                        int nOfInLinks = n.getInLinks().size();
064                        int nOfOutLinks = n.getOutLinks().size();
065                        if ((nOfInLinks + nOfOutLinks) == 0) { setTopoType(n, EMPTY); }
066                        else if (nOfInLinks == 0) { setTopoType(n, SOURCE); }
067                        else if (nOfOutLinks == 0) {setTopoType(n, SINK); }
068                        else if (getNOfIncidentNodes(n) == 1) { setTopoType(n, DEADEND); }
069                        else if (getNOfIncidentNodes(n) == 2) {
070                                if ((nOfOutLinks == 1) && (nOfInLinks == 1)) { setTopoType(n, PASS1WAY); }
071                                else if ((nOfOutLinks == 2) && (nOfInLinks == 2)) { setTopoType(n, PASS2WAY); }
072                                else if ((nOfOutLinks == 2) && (nOfInLinks == 1)) { setTopoType(n, START1WAY); }
073                                else if ((nOfOutLinks == 1) && (nOfInLinks == 2)) { setTopoType(n, END1WAY); }
074                                // The following case is not covered by the paper, but quite common, e.g. parallel roads connecting the same nodes.
075                                else if ((nOfOutLinks >= 1) && (nOfInLinks >= 1)) {
076                                        // yyyy intellij says that this condition is always true when the code comes to here.  kai, dec'17
077                                        setTopoType(n, INTERSECTION);
078                                }
079                                else { throw new RuntimeException("Node=" + n.toString() + " cannot be assigned to a topo type!"); }
080                        }
081                        else { // more than two neighbors nodes and no sink or source
082                                setTopoType(n, INTERSECTION);
083                        }
084                }
085
086                int [] cnt = {0,0,0,0,0,0,0,0,0};
087                for (Node n : network.getNodes().values()) {
088                        cnt[getTopoType(n)]++;
089                }
090
091                log.info("      #nodes        = " + network.getNodes().size());
092                log.info("      #EMTPY        = " + cnt[EMPTY]);
093                log.info("      #SOURCE       = " + cnt[SOURCE]);
094                log.info("      #SINK         = " + cnt[SINK]);
095                log.info("      #DEADEND      = " + cnt[DEADEND]);
096                log.info("      #PASS1WAY     = " + cnt[PASS1WAY]);
097                log.info("      #PASS2WAY     = " + cnt[PASS2WAY]);
098                log.info("      #START1WAY    = " + cnt[START1WAY]);
099                log.info("      #END1WAY      = " + cnt[END1WAY]);
100                log.info("      #INTERSECTION = " + cnt[INTERSECTION]);
101
102                log.info("    done.");
103        }
104
105        private void setTopoType(final Node node, final Integer topoType) {
106                this.topoTypePerNode.put(node, topoType);
107        }
108
109        public int getTopoType(final Node node) {
110                Integer i = this.topoTypePerNode.get(node);
111                if (i == null) {
112                        return Integer.MIN_VALUE;
113                }
114                return i;
115        }
116
117        private int getNOfIncidentNodes(final Node node) {
118                HashMap<Id<Node>, Node> nodes = new HashMap<>();
119                for (Link link : node.getInLinks().values()) {
120                        nodes.put(link.getFromNode().getId(), link.getFromNode());
121                }
122                for (Link link : node.getOutLinks().values()) {
123                        nodes.put(link.getToNode().getId(), link.getToNode());
124                }
125                return nodes.size();
126        }
127}
