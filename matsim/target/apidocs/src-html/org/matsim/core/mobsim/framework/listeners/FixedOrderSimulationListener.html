021package org.matsim.core.mobsim.framework.listeners;
022
023import org.matsim.core.mobsim.framework.events.MobsimAfterSimStepEvent;
024import org.matsim.core.mobsim.framework.events.MobsimBeforeCleanupEvent;
025import org.matsim.core.mobsim.framework.events.MobsimBeforeSimStepEvent;
026import org.matsim.core.mobsim.framework.events.MobsimInitializedEvent;
027
028import java.util.ArrayList;
029import java.util.List;
030
031/**
032 * To avoid problems with the order of some SimulationListeners, this
033 * class can be used execute them in a fixed order. Even something like 
034 * ParallelListenerHandling (which is not available yet) should not able 
035 * to break the order.
036 * 
037 * @author cdobler
038 */
039public class FixedOrderSimulationListener implements MobsimInitializedListener,
040        MobsimBeforeSimStepListener, MobsimAfterSimStepListener,
041        MobsimBeforeCleanupListener {
042
043        List<MobsimInitializedListener> simulationInitializedListener;
044        List<MobsimBeforeSimStepListener> simulationBeforeSimStepListener;
045        List<MobsimAfterSimStepListener> simulationAfterSimStepListener;
046        List<MobsimBeforeCleanupListener> simulationBeforeCleanupListener;
047
048        public FixedOrderSimulationListener() {
049                simulationInitializedListener = new ArrayList<MobsimInitializedListener>();
050                simulationBeforeSimStepListener = new ArrayList<MobsimBeforeSimStepListener>();
051                simulationAfterSimStepListener = new ArrayList<MobsimAfterSimStepListener>();
052                simulationBeforeCleanupListener = new ArrayList<MobsimBeforeCleanupListener>();
053        }
054
055        /**
056         * Adds the SimulationListener to all ListenerLists that it supports.
057         */
058        public void addSimulationListener(MobsimListener listener) {
059                if (listener instanceof MobsimInitializedListener) {
060                        addSimulationInitializedListener((MobsimInitializedListener) listener);
061                }
062                if (listener instanceof MobsimBeforeSimStepListener) {
063                        addSimulationBeforeSimStepListener((MobsimBeforeSimStepListener) listener);
064                }
065                if (listener instanceof MobsimAfterSimStepListener) {
066                        addSimulationAfterSimStepListener((MobsimAfterSimStepListener) listener);
067                }
068                if (listener instanceof MobsimBeforeCleanupListener) {
069                        addSimulationBeforeCleanupListener((MobsimBeforeCleanupListener) listener);
070                }
071        }
072        
073        /**
074         * Removes the SimulationListener from all ListenerLists that it supports.
075         */
076        public void removeSimulationListener(MobsimListener listener) {
077                if (listener instanceof MobsimInitializedListener) {
078                        removeSimulationInitializedListener((MobsimInitializedListener) listener);
079                }
080                if (listener instanceof MobsimBeforeSimStepListener) {
081                        removeSimulationBeforeSimStepListener((MobsimBeforeSimStepListener) listener);
082                }
083                if (listener instanceof MobsimAfterSimStepListener) {
084                        removeSimulationAfterSimStepListener((MobsimAfterSimStepListener) listener);
085                }
086                if (listener instanceof MobsimBeforeCleanupListener) {
087                        removeSimulationBeforeCleanupListener((MobsimBeforeCleanupListener) listener);
088                }
089        }
090        
091        public void addSimulationInitializedListener(MobsimInitializedListener listener) {
092                simulationInitializedListener.add(listener);
093        }
094
095        public void removeSimulationInitializedListener(MobsimInitializedListener listener) {
096                simulationInitializedListener.remove(listener);
097        }
098
099        @Override
100        public void notifyMobsimInitialized(MobsimInitializedEvent e) {
101                for(MobsimInitializedListener listener : simulationInitializedListener) {
102                        listener.notifyMobsimInitialized(e);
103                }
104        }
105
106        public void addSimulationBeforeSimStepListener(MobsimBeforeSimStepListener listener) {
107                simulationBeforeSimStepListener.add(listener);
108        }
109
110        public void removeSimulationBeforeSimStepListener(MobsimBeforeSimStepListener listener) {
111                simulationBeforeSimStepListener.remove(listener);
112        }
113
114        @Override
115        public void notifyMobsimBeforeSimStep(MobsimBeforeSimStepEvent e) {
116                for(MobsimBeforeSimStepListener listener : simulationBeforeSimStepListener) {
117                        listener.notifyMobsimBeforeSimStep(e);
118                }
119        }
120
121        public void addSimulationAfterSimStepListener(MobsimAfterSimStepListener listener) {
122                simulationAfterSimStepListener.add(listener);
123        }
124
125        public void removeSimulationAfterSimStepListener(MobsimAfterSimStepListener listener) {
126                simulationAfterSimStepListener.remove(listener);
127        }
128
129        @Override
130        public void notifyMobsimAfterSimStep(MobsimAfterSimStepEvent e) {
131                for(MobsimAfterSimStepListener listener : simulationAfterSimStepListener) {
132                        listener.notifyMobsimAfterSimStep(e);
133                }
134        }
135
136        public void addSimulationBeforeCleanupListener(MobsimBeforeCleanupListener listener) {
137                simulationBeforeCleanupListener.add(listener);
138        }
139
140        public void removeSimulationBeforeCleanupListener(MobsimBeforeCleanupListener listener) {
141                simulationBeforeCleanupListener.remove(listener);
142        }
143
144        @Override
145        public void notifyMobsimBeforeCleanup(MobsimBeforeCleanupEvent e) {
146                for(MobsimBeforeCleanupListener listener : simulationBeforeCleanupListener) {
147                        listener.notifyMobsimBeforeCleanup(e);
148                }
149        }
150
151}
