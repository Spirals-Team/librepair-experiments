043public final class PopulationAgentSource implements AgentSource {
044        private static final Logger log = Logger.getLogger( PopulationAgentSource.class );
045
046        private final Population population;
047        private final AgentFactory agentFactory;
048        private final QSim qsim;
049        private final Collection<String> mainModes;
050        private Map<Id<Vehicle>,Id<Link>> seenVehicleIds = new HashMap<>() ;
051
052        @Inject
053        public PopulationAgentSource(Population population, AgentFactory agentFactory, QSim qsim ) {
054                this.population = population;
055                this.agentFactory = agentFactory;
056                this.qsim = qsim;  
057                this.mainModes = qsim.getScenario().getConfig().qsim().getMainModes();
058        }
059
060        @Override
061        public void insertAgentsIntoMobsim() {
062                for (Person p : population.getPersons().values()) {
063                        MobsimAgent agent = this.agentFactory.createMobsimAgentFromPerson(p);
064                        qsim.insertAgentIntoMobsim(agent);
065                }
066                for (Person p : population.getPersons().values()) {
067                        insertVehicles(p);
068                }
069        }
070
071        private void insertVehicles(Person p) {
072                // this is called in every iteration.  So if a route without a vehicle id is found (e.g. after mode choice),
073                // then the id is generated here.  kai/amit, may'18
074                
075                Plan plan = p.getSelectedPlan();
076                Map<String,Id<Vehicle>> seenModes = new HashMap<>();
077                for (PlanElement planElement : plan.getPlanElements()) {
078                        if (planElement instanceof Leg) {
079                                Leg leg = (Leg) planElement;
080                                if (this.mainModes.contains(leg.getMode())) { // only simulated modes get vehicles
081                                        NetworkRoute route = (NetworkRoute) leg.getRoute();
082                                        Id<Vehicle> vehicleId = null ;
083                                        if (route != null) {
084                                                vehicleId = route.getVehicleId();
085                                        }
086                                        if (!seenModes.keySet().contains(leg.getMode())) { // create one vehicle per simulated mode, put it on the home location
087                                                // yyyy this is already getting rather messy; need to consider simplifications ...  kai/amit, sep'16
088                                                
089                                                if (vehicleId == null) {
090                                                        // if mode choice is allowed, it is possible that a new mode is assigned to an agent and then the route does not have the vehicle
091                                                        // but scenario must have that vehicle; however, in order to find the vehicle, we need to identify the vehicle id. Amit July'17
092                                                        vehicleId = createAutomaticVehicleId(p.getId(), leg.getMode());
093
094                                                        Gbl.assertNotNull(route);
095                                                        // yyyy most execution paths through this method hedge against route==null, but this one does not. I
096                                                        // haven't investigated why this is the case, and if it could be changed.  Would need to understand the
097                                                        // logic of those conditionals first.  kai, may'18
098
099                                                        route.setVehicleId(vehicleId);
100                                                }
101
102                                                // so here we have a vehicle id, now try to find or create a physical vehicle:
103                                                Vehicle vehicle = qsim.getScenario().getVehicles().getVehicles().get(vehicleId);
104                                                
105                                                // place the vehicle:
106                                                Id<Link> vehicleLinkId = findVehicleLink(p);
107                                                
108                                                // Checking if the vehicle has been seen before:
109                                                Id<Link> result = this.seenVehicleIds.get( vehicleId ) ;
110                                                if ( result != null ) {
111                                                        // if seen before, but placed on same link, then it is ok:
112                                                        log.info( "have seen vehicle with id " + vehicleId + " before; not placing it again." );
113                                                        if ( result != vehicleLinkId ) {
114                                                                throw new RuntimeException("vehicle placement error: vehicleId=" + vehicleId + 
115                                                                                "; previous placement link=" + vehicleLinkId + "; current placement link=" + result ) ; 
116                                                        }
117                                                } else {
118                                                        this.seenVehicleIds.put( vehicleId, vehicleLinkId ) ;
119                                                        qsim.createAndParkVehicleOnLink(vehicle, vehicleLinkId);
120                                                }
121                                                seenModes.put(leg.getMode(),vehicleId);
122                                        } else {
123                                                if (vehicleId==null && route!=null) {
124                                                        vehicleId = seenModes.get(leg.getMode());
125                                                        route.setVehicleId( vehicleId );
126                                                }
127                                        }
128                                }
129                        }
130                }
131        }

133        /**
134         *      A more careful way to decide where this agent should have its vehicles created
135         *  than to ask agent.getCurrentLinkId() after creation.
136         * @param leg TODO
137         */
138        private Id<Link> findVehicleLink(Person p ) {
139                /* Cases that come to mind:
140                 * (1) multiple persons share car located at home, but possibly brought to different place by someone else.  
141                 *      This is treated by the following algo.
142                 * (2) person starts day with non-car leg and has car parked somewhere else.  This is NOT treated by the following algo.
143                 *      It could be treated by placing the vehicle at the beginning of the first link where it is needed, but this would not
144                 *      be compatible with variant (1).
145                 */
146                for (PlanElement planElement : p.getSelectedPlan().getPlanElements()) {
147                        if (planElement instanceof Activity) {
148                                Activity activity = (Activity) planElement;
149                                ActivityFacilities facilities = this.qsim.getScenario().getActivityFacilities() ;
150                                Config config = this.qsim.getScenario().getConfig() ;
151                                final Id<Link> activityLinkId = PopulationUtils.computeLinkIdFromActivity(activity, facilities, config ) ;
152                                if (activityLinkId != null) {
153                                        return activityLinkId;
154                                }
155                        } else if (planElement instanceof Leg) {
156                                Leg leg = (Leg) planElement;
157                                if (leg.getRoute().getStartLinkId() != null) {
158                                        return leg.getRoute().getStartLinkId();
159                                }
160                        }
161                }
162                throw new RuntimeException("Don't know where to put a vehicle for this agent.");
163        }
164
165        private Id<Vehicle> createAutomaticVehicleId(Id<Person> personId, String mode) {
166                Id<Vehicle> vehicleId ;
167                if (qsim.getScenario().getConfig().qsim().getUsePersonIdForMissingVehicleId()) {
168                        switch (qsim.getScenario().getConfig().qsim().getVehiclesSource()) {
169                                case defaultVehicle:
170                                case fromVehiclesData:
171                                        vehicleId = Id.createVehicleId(personId);
172                                        break;
173                                case modeVehicleTypesFromVehiclesData:
174                                        if(! mode.equals(TransportMode.car)) {
175                                                String vehIdString = personId.toString() + "_" + mode ;
176                                                vehicleId = Id.create(vehIdString, Vehicle.class);
177                                        } else {
178                                                vehicleId = Id.createVehicleId(personId);
179                                        }
180                                        break;
181                                default:
182                                        throw new RuntimeException("not implemented") ;
183                        }
184                } else {
185                        throw new IllegalStateException("Found a network route without a vehicle id.");
186                }
187                return vehicleId;
188        }
