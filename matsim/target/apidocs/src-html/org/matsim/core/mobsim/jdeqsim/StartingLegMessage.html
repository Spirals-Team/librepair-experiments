020package org.matsim.core.mobsim.jdeqsim;
021
022import org.matsim.api.core.v01.TransportMode;
023import org.matsim.api.core.v01.events.ActivityEndEvent;
024import org.matsim.api.core.v01.events.Event;
025import org.matsim.api.core.v01.events.PersonDepartureEvent;
026
027/**
028 * The micro-simulation internal handler for starting a leg.
029 * 
030 * @author rashid_waraich
031 */
032public class StartingLegMessage extends EventMessage {
033
034        public StartingLegMessage(Scheduler scheduler, Vehicle vehicle) {
035                super(scheduler, vehicle);
036                priority = JDEQSimConfigGroup.PRIORITY_DEPARTUARE_MESSAGE;
037        }
038
039        @Override
040        public void handleMessage() {
041                // if current leg is in car mode, then enter request in first road
042                if (vehicle.getCurrentLeg().getMode().equals(TransportMode.car)) {
043
044                        // if empty leg, then end leg, else simulate leg
045                        if (vehicle.getCurrentLinkRoute().length == 0) {
046                                // move to first link in next leg and schedule an end leg
047                                // message
048                                // duration of leg = 0 (departure and arrival time is the same)
049                                scheduleEndLegMessage(getMessageArrivalTime());
050
051                        } else {
052                                // start the new leg
053                                Road road = Road.getRoad(vehicle.getCurrentLinkId());
054                                road.enterRequest(vehicle, getMessageArrivalTime());
055                        }
056
057                } else {
058                        scheduleEndLegMessage(getMessageArrivalTime() + vehicle.getCurrentLeg().getTravelTime());
059                }
060        }
061
062        private void scheduleEndLegMessage(double time) {
063                // move to first link in next leg and schedule an end leg message
064                vehicle.moveToFirstLinkInNextLeg();
065                Road road = Road.getRoad(vehicle.getCurrentLinkId());
066                vehicle.scheduleEndLegMessage(time, road);
067        }
068
069        @Override
070        public void processEvent() {
071                Event event;
072
073                // schedule ActEndEvent
074                event = new ActivityEndEvent(this.getMessageArrivalTime(), vehicle.getOwnerPerson().getId(), vehicle.getCurrentLinkId(), vehicle
075                                .getPreviousActivity().getFacilityId(), vehicle.getPreviousActivity().getType());
076                eventsManager.processEvent(event);
077
078                // schedule AgentDepartureEvent
079                event = new PersonDepartureEvent(this.getMessageArrivalTime(), vehicle.getOwnerPerson().getId(), vehicle.getCurrentLinkId(),
080                                vehicle.getCurrentLeg().getMode());
081
082                eventsManager.processEvent(event);
083
084        }
085
086}
