001package org.matsim.core.mobsim.qsim;
002
003
004import java.util.ArrayList;
005import java.util.Collection;
006
007import javax.inject.Inject;
008
009import org.matsim.core.config.Config;
010import org.matsim.core.mobsim.framework.Mobsim;
011import org.matsim.core.mobsim.qsim.changeeventsengine.NetworkChangeEventsPlugin;
012import org.matsim.core.mobsim.qsim.messagequeueengine.MessageQueuePlugin;
013import org.matsim.core.mobsim.qsim.pt.ComplexTransitStopHandlerFactory;
014import org.matsim.core.mobsim.qsim.pt.TransitEnginePlugin;
015import org.matsim.core.mobsim.qsim.pt.TransitStopHandlerFactory;
016import org.matsim.core.mobsim.qsim.qnetsimengine.DefaultQNetworkFactory;
017import org.matsim.core.mobsim.qsim.qnetsimengine.QLanesNetworkFactory;
018import org.matsim.core.mobsim.qsim.qnetsimengine.QNetsimEnginePlugin;
019import org.matsim.core.mobsim.qsim.qnetsimengine.QNetworkFactory;
020
021import com.google.inject.Provides;
022
023public class QSimModule extends com.google.inject.AbstractModule {
024        @Inject Config config ;
025        
026        @Override
027        protected void configure() {
028                bind(Mobsim.class).toProvider(QSimProvider.class);
029                if ( config.qsim().isUseLanes() ) { 
030                        bind(QNetworkFactory.class).to( QLanesNetworkFactory.class ) ;
031                } else {
032                        bind(QNetworkFactory.class).to( DefaultQNetworkFactory.class ) ;
033                }
034                if ( config.transit().isUseTransit() ) {
035                        bind( TransitStopHandlerFactory.class ).to( ComplexTransitStopHandlerFactory.class ) ;
036                }
037                // yy see MATSIM-756
038        }
039
040        @SuppressWarnings("static-method")
041        @Provides
042        Collection<AbstractQSimPlugin> provideQSimPlugins(Config config1) {
043                final Collection<AbstractQSimPlugin> plugins = new ArrayList<>();
044                plugins.add(new MessageQueuePlugin(config1));
045                plugins.add(new ActivityEnginePlugin(config1));
046                plugins.add(new QNetsimEnginePlugin(config1));
047                if (config1.network().isTimeVariantNetwork()) {
048                        plugins.add(new NetworkChangeEventsPlugin(config1));
049                }
050                if (config1.transit().isUseTransit()) {
051                        plugins.add(new TransitEnginePlugin(config1));
052                }
053                plugins.add(new TeleportationPlugin(config1));
054                plugins.add(new PopulationPlugin(config1));
055                return plugins;
056        }
057}
