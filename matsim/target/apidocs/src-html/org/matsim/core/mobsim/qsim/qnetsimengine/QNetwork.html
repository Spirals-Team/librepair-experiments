021package org.matsim.core.mobsim.qsim.qnetsimengine;
022
023import java.util.Collections;
024import java.util.LinkedHashMap;
025import java.util.Map;
026
027import org.matsim.api.core.v01.Id;
028import org.matsim.api.core.v01.network.Link;
029import org.matsim.api.core.v01.network.Network;
030import org.matsim.api.core.v01.network.Node;
031import org.matsim.core.mobsim.framework.MobsimTimer;
032import org.matsim.core.mobsim.qsim.interfaces.AgentCounter;
033import org.matsim.core.mobsim.qsim.interfaces.NetsimNetwork;
034import org.matsim.core.mobsim.qsim.interfaces.NetsimNode;
035import org.matsim.vis.snapshotwriters.VisLink;
036
037/**
038 *
039 * @author david
040 * @author mrieser
041 * @author dgrether
042 */
043
044public class QNetwork implements NetsimNetwork {
045
046        private final Map<Id<Link>, QLinkI> links;
047
048        private final Map<Id<Node>, QNodeI> nodes;
049
050        private final Network network;
051
052        private final QNetworkFactory queueNetworkFactory;
053        QNetsimEngine simEngine;
054
055        QNetwork(final Network network, final QNetworkFactory netsimNetworkFactory ) {
056                this.network = network;
057                this.queueNetworkFactory = netsimNetworkFactory;
058                this.links = new LinkedHashMap<>((int)(network.getLinks().size()*1.1), 0.95f);
059                this.nodes = new LinkedHashMap<>((int)(network.getLinks().size()*1.1), 0.95f);
060        }
061
062        public void initialize(QNetsimEngine simEngine1, AgentCounter agentCounter, MobsimTimer simTimer) {
063                this.simEngine = simEngine1;
064                this.queueNetworkFactory.initializeFactory( agentCounter, simTimer, simEngine1.ii );
065                for (Node n : network.getNodes().values()) {
066                        this.nodes.put(n.getId(), this.queueNetworkFactory.createNetsimNode(n));
067                }
068                for (Link l : network.getLinks().values()) {
069                        final QLinkI qlink = this.queueNetworkFactory.createNetsimLink(l, this.nodes.get(l.getToNode().getId()));
070                        this.links.put(l.getId(), qlink);
071                }
072                for (QNodeI n : this.nodes.values()) {
073                        n.init();
074                }
075        }
076        
077        @Override
078        public Network getNetwork() {
079                return this.network;
080        }
081
082        @Override
083        public Map<Id<Link>, QLinkI> getNetsimLinks() {
084                return Collections.unmodifiableMap(this.links);
085        }
086
087        @Override
088        public Map<Id<Link>, ? extends VisLink> getVisLinks() {
089                return Collections.unmodifiableMap(this.links);
090        }
091
092        @Override
093        public Map<Id<Node>, QNodeI> getNetsimNodes() {
094                return Collections.unmodifiableMap(this.nodes);
095        }
096
097        @Override
098        public QLinkI getNetsimLink(final Id<Link> id) {
099                return this.links.get(id);
100        }
101
102        @Override
103        public NetsimNode getNetsimNode(final Id<Node> id) {
104                return this.nodes.get(id);
105        }
106
107
108}
