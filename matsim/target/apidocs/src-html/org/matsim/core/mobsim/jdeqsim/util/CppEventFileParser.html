041        public static ArrayList<EventLog> parseFile(final String filePath) {
042                int counter = 0;
043                ArrayList<EventLog> rows = new ArrayList<EventLog>();
044                BufferedReader br = null;
045                try {
046                        br = IOUtils.getBufferedReader(filePath);
047                        String line = null;
048                        StringTokenizer tokenizer = null;
049                        line = br.readLine();
050                        while (line != null) {
051                                counter++;
052
053                                if (counter % 1000000 == 0) {
054                                        System.out.println("noOfLines Read:" + counter);
055                                }
056
057                                tokenizer = new StringTokenizer(line);
058                                String token = null;
059                                token = tokenizer.nextToken();
060                                double first = Double.parseDouble(token);
061                                token = tokenizer.nextToken();
062                                int second = Integer.parseInt(token);
063                                token = tokenizer.nextToken();
064                                int third = Integer.parseInt(token);
065                                token = tokenizer.nextToken();
066                                int fourth = Integer.parseInt(token);
067                                token = tokenizer.nextToken();
068                                int fifth = Integer.parseInt(token);
069                                token = tokenizer.nextToken();
070                                int sixth = Integer.parseInt(token);
071                                token = tokenizer.nextToken();
072                                String eventType = token;
073
074                                // there is one eventType called 'enter net' => it is split into
075                                // two tockens, so need to take that into account
076                                if (tokenizer.hasMoreTokens()) {
077                                        token = tokenizer.nextToken();
078                                        eventType += " " + token;
079                                }
080
081                                // change type label
082                                if (eventType.equalsIgnoreCase("starting")) {
083                                        eventType = JDEQSimConfigGroup.START_LEG;
084                                } else if (eventType.equalsIgnoreCase("end")) {
085                                        eventType = JDEQSimConfigGroup.END_LEG;
086                                } else if (eventType.equalsIgnoreCase("enter")) {
087                                        eventType = JDEQSimConfigGroup.ENTER_LINK;
088                                } else if (eventType.equalsIgnoreCase("leave")) {
089                                        eventType = JDEQSimConfigGroup.LEAVE_LINK;
090                                }
091
092                                // ignore 'enter net' events (which seem useless)
093                                if (!eventType.equalsIgnoreCase("enter net")) {
094                                        rows.add(new EventLog(first, second, third, fourth, fifth, sixth, eventType));
095                                }
096
097                                line = br.readLine();
098                        }
099                } catch (IOException ex) {
100                        System.out.println(ex);
101                } finally {
102                        try {
103                                if (br != null)
104                                        br.close();
105                        } catch (IOException e) {
106                                e.printStackTrace();
107                        }
108
109                }
110
111                return rows;
112        }

124        /**
125         * Compares events produced by java and by C++ simulation
126         */
127        public static boolean equals(final Event personEvent, final EventLog deqSimEvent) {
128
129                if (personEvent.getTime() != deqSimEvent.getTime()) {
130                        CppEventFileParser.printNotEqualEvents(personEvent, deqSimEvent);
131                        return false;
132                }
133
134                if (personEvent instanceof PersonDepartureEvent) {
135                        if (Integer.parseInt(((PersonDepartureEvent) personEvent).getLinkId().toString()) != deqSimEvent.getLinkId()) {
136                                CppEventFileParser.printNotEqualEvents(personEvent, deqSimEvent);
137                                return false;
138                        }
139                        if (Integer.parseInt(((PersonDepartureEvent) personEvent).getPersonId().toString()) != deqSimEvent.getVehicleId()) {
140                                CppEventFileParser.printNotEqualEvents(personEvent, deqSimEvent);
141                                return false;
142                        }
143                        if (!deqSimEvent.getType().equalsIgnoreCase(JDEQSimConfigGroup.START_LEG)) {
144                                CppEventFileParser.printNotEqualEvents(personEvent, deqSimEvent);
145                                return false;
146                        }
147                }
148
149                if (personEvent instanceof LinkEnterEvent) {
150                        if (Integer.parseInt(((LinkEnterEvent) personEvent).getLinkId().toString()) != deqSimEvent.getLinkId()) {
151                                CppEventFileParser.printNotEqualEvents(personEvent, deqSimEvent);
152                                return false;
153                        }
154                        if (Integer.parseInt(((LinkEnterEvent) personEvent).getVehicleId().toString()) != deqSimEvent.getVehicleId()) {
155                                CppEventFileParser.printNotEqualEvents(personEvent, deqSimEvent);
156                                return false;
157                        }
158                        if (!deqSimEvent.getType().equalsIgnoreCase(JDEQSimConfigGroup.ENTER_LINK)) {
159                                CppEventFileParser.printNotEqualEvents(personEvent, deqSimEvent);
160                                return false;
161                        }
162                }
163
164                if (personEvent instanceof LinkLeaveEvent) {
165                        if (Integer.parseInt(((LinkLeaveEvent) personEvent).getLinkId().toString()) != deqSimEvent.getLinkId()) {
166                                CppEventFileParser.printNotEqualEvents(personEvent, deqSimEvent);
167                                return false;
168                        }
169                        if (Integer.parseInt(((LinkLeaveEvent) personEvent).getVehicleId().toString()) != deqSimEvent.getVehicleId()) {
170                                CppEventFileParser.printNotEqualEvents(personEvent, deqSimEvent);
171                                return false;
172                        }
173                        if (!deqSimEvent.getType().equalsIgnoreCase(JDEQSimConfigGroup.LEAVE_LINK)) {
174                                CppEventFileParser.printNotEqualEvents(personEvent, deqSimEvent);
175                                return false;
176                        }
177                }
178
179                if (personEvent instanceof PersonArrivalEvent) {
180                        if (Integer.parseInt(((PersonArrivalEvent) personEvent).getLinkId().toString()) != deqSimEvent.getLinkId()) {
181                                CppEventFileParser.printNotEqualEvents(personEvent, deqSimEvent);
182                                return false;
183                        }
184                        if (Integer.parseInt(((PersonArrivalEvent) personEvent).getPersonId().toString()) != deqSimEvent.getVehicleId()) {
185                                CppEventFileParser.printNotEqualEvents(personEvent, deqSimEvent);
186                                return false;
187                        }
188
189                        if (!deqSimEvent.getType().equalsIgnoreCase(JDEQSimConfigGroup.END_LEG)) {
190                                CppEventFileParser.printNotEqualEvents(personEvent, deqSimEvent);
191                                return false;
192                        }
193                }
194
195                return true;
196        }
