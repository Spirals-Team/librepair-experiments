105        @Override
106        public final Id<Link> chooseNextLinkId() {
107                NetworkRoute netR = getCarRoute();
108                List<Id<Link>> linkIds = netR.getLinkIds();
109                if (this.nextLinkIndex < linkIds.size()) {
110                        return linkIds.get(this.nextLinkIndex);
111                }
112                if (this.nextLinkIndex == linkIds.size()) {
113                        if (linkIds.size() == 0 && netR.getStartLinkId().equals(netR.getEndLinkId())) {
114                                // unfortunate exception for the unlikely but legal and test-covered case where a whole transit line
115                                // takes place on a single link.
116                                // would not need to be here if a route were simply a list of linkIds to traverse, with 1 being an allowed length.
117                                // this used to return endLinkId, which would then require a transit-specific extra case for arriving vehicles
118                                // in QLinkImpl, which was even worse.
119                                return null;
120                        }
121                        return netR.getEndLinkId();
122                }
123                assertAllStopsServed();
124                return null;
125        }

157        @Override
158        public double handleTransitStop(final TransitStopFacility stop, final double now) {
159                // yy can't make this final because of tests.  kai, oct'12
160
161                assertExpectedStop(stop);
162                processEventVehicleArrives(stop, now);
163
164                TransitRoute route = this.getTransitRoute();
165                List<TransitRouteStop> stopsToCome = route.getStops().subList(stopIterator.nextIndex(), route.getStops().size());
166                /*
167                 * If there are passengers leaving or entering, the stop time must be not greater than 1.0 in order to let them (de-)board every second. 
168                 * If a stopTime greater than 1.0 is used, this method is not necessarily triggered by the qsim, so (de-)boarding will not happen. Dg, 10-2012
169                 */
170                double stopTime = this.accessEgress.calculateStopTimeAndTriggerBoarding(getTransitRoute(), getTransitLine(), this.vehicle, stop, stopsToCome, now);
171
172                if(stopTime == 0.0){
173                        stopTime = longerStopTimeIfWeAreAheadOfSchedule(now, stopTime);
174                }
175                if (stopTime == 0.0) {
176                        depart(now);
177                }
178                return stopTime;
179        }

223        private void processEventVehicleArrives(final TransitStopFacility stop,
224                        final double now) {
225                if (this.currentStop == null) {
226                        this.currentStop = this.nextStop;
227                        double delay = now - this.getDeparture().getDepartureTime();
228                        if (! ( Double.isNaN(this.currentStop.getArrivalOffset()) 
229                                        &&   Double.isInfinite(this.currentStop.getArrivalOffset())) ){
230                                delay = delay - this.currentStop.getArrivalOffset();
231                        }
232                        else if (! (Double.isNaN(this.currentStop.getDepartureOffset()) &&   Double.isInfinite(this.currentStop.getDepartureOffset()))) {
233                                delay =  delay - this.currentStop.getDepartureOffset();
234                        }
235                        else {
236                                log.warn("Could not calculate delay!");
237                        }
238                        eventsManager.processEvent(new VehicleArrivesAtFacilityEvent(now, this.vehicle.getVehicle().getId(), stop.getId(),
239                                        delay));
240                }
241        }

243        private void assertExpectedStop(final TransitStopFacility stop) {
244                if (stop != this.nextStop.getStopFacility()) {
245                        throw new RuntimeException("Expected different stop.");
246                }
247        }

249        protected double longerStopTimeIfWeAreAheadOfSchedule(final double now,
250                        final double stopTime) {
251                if ((this.nextStop.isAwaitDepartureTime()) && (this.nextStop.getDepartureOffset() != Time.UNDEFINED_TIME)) {
252                        double earliestDepTime = getActivityEndTime() + this.nextStop.getDepartureOffset();
253                        if (now + stopTime < earliestDepTime) {
254                                return earliestDepTime - now;
255                        }
256                }
257                return stopTime;
258        }
259
260        private boolean isBadDouble(double d){
261                return Double.isNaN(d) || Double.isInfinite(d);
262        }
263
264        private void depart(final double now) {
265                double delay = now - this.getDeparture().getDepartureTime();
266                if (this.isBadDouble(this.getDeparture().getDepartureTime())){ //this is the case if the next stop is null
267                        delay = 0;
268                }
269                if (! this.isBadDouble(this.currentStop.getDepartureOffset())){
270                        delay =  delay - this.currentStop.getDepartureOffset();
271                }
272                else if (! this.isBadDouble(this.currentStop.getArrivalOffset()) ){
273                        delay = delay - this.currentStop.getArrivalOffset();
274                }
275                else {
276                        log.warn("Could not calculate delay!");
277                }
278                eventsManager.processEvent(new VehicleDepartsAtFacilityEvent(now, this.vehicle.getVehicle().getId(),
279                                this.currentStop.getStopFacility().getId(),
280                                delay));
281                this.nextStop = (stopIterator.hasNext() ? stopIterator.next() : null);
282                if(this.nextStop == null) {
283                        assertVehicleIsEmpty();
284                }
285                this.currentStop = null;
286        }
287
288        private void assertAllStopsServed() {
289                if (this.nextStop != null) {
290                        RuntimeException e = new RuntimeException("Transit vehicle is not yet at last stop! vehicle-id = " 
291                                        + this.vehicle.getVehicle().getId() + "; next-stop = " + this.nextStop.getStopFacility().getId());
292                        log.error(e);
293                        throw e;
294                }
295        }
296
297        private void assertVehicleIsEmpty() {
298                if (this.vehicle.getPassengers().size() > 0) {
299                        RuntimeException e = new RuntimeException("Transit vehicle is at last stop but still contains passengers that did not leave the vehicle!");
300                        log.error("Transit vehicle must be empty after last stop! vehicle-id = " + this.vehicle.getVehicle().getId(), e);
301                        for (PassengerAgent agent : this.vehicle.getPassengers()) {
302                                if (agent instanceof PersonDriverAgentImpl) {
303                                        log.error("Agent is still in transit vehicle: agent-id = " + ((PersonDriverAgentImpl) agent).getPerson().getId());
304                                }
305                        }
306                        throw e;
307                }
308        }
