024import org.matsim.api.core.v01.network.Link;
025import org.matsim.api.core.v01.network.Network;
026import org.matsim.api.core.v01.network.Node;

029import org.matsim.core.mobsim.framework.MobsimTimer;
030import org.matsim.core.mobsim.qsim.interfaces.AgentCounter;
031import org.matsim.core.mobsim.qsim.qnetsimengine.QNetsimEngine.NetsimInternalInterface;
032import org.matsim.core.mobsim.qsim.qnetsimengine.linkspeedcalculator.DefaultLinkSpeedCalculator;
033import org.matsim.core.mobsim.qsim.qnetsimengine.linkspeedcalculator.LinkSpeedCalculator;
034import org.matsim.vis.snapshotwriters.SnapshotLinkWidthCalculator;
035
036
037/**
038 * 
039 * @author knagel
040 * 
041 * @see DefaultQNetworkFactory
042 */
043public final class ConfigurableQNetworkFactory extends QNetworkFactory {
044        private QSimConfigGroup qsimConfig ;
045        private EventsManager events ;
046        private Network network ;
047        private Scenario scenario ;
048        private NetsimEngineContext context;
049        private NetsimInternalInterface netsimEngine ;
050        private LinkSpeedCalculator linkSpeedCalculator = new DefaultLinkSpeedCalculator() ;
051        private TurnAcceptanceLogic turnAcceptanceLogic = new DefaultTurnAcceptanceLogic() ;
052
053        public ConfigurableQNetworkFactory( EventsManager events, Scenario scenario ) {
054                this.events = events;
055                this.scenario = scenario;
056                this.network = scenario.getNetwork() ;
057                this.qsimConfig = scenario.getConfig().qsim() ;
058        }
059        @Override
060        void initializeFactory( AgentCounter agentCounter, MobsimTimer mobsimTimer, NetsimInternalInterface netsimEngine1 ) {
061                this.netsimEngine = netsimEngine1;
062                double effectiveCellSize = network.getEffectiveCellSize() ;
063                SnapshotLinkWidthCalculator linkWidthCalculator = new SnapshotLinkWidthCalculator();
064                linkWidthCalculator.setLinkWidthForVis( qsimConfig.getLinkWidthForVis() );
065                if (! Double.isNaN(network.getEffectiveLaneWidth())){
066                        linkWidthCalculator.setLaneWidth( network.getEffectiveLaneWidth() );
067                }
068                AbstractAgentSnapshotInfoBuilder agentSnapshotInfoBuilder = QNetsimEngine.createAgentSnapshotInfoBuilder( scenario, linkWidthCalculator );
069                context = new NetsimEngineContext( events, effectiveCellSize, agentCounter, agentSnapshotInfoBuilder, qsimConfig, mobsimTimer, linkWidthCalculator );
070        }
071        @Override
072        QLinkI createNetsimLink(final Link link, final QNodeI toQueueNode) {
073                QueueWithBuffer.Builder laneFactory = new QueueWithBuffer.Builder(context) ;
074
075                QLinkImpl.Builder linkBuilder = new QLinkImpl.Builder(context, netsimEngine) ;
076                linkBuilder.setLaneFactory(laneFactory);
077                linkBuilder.setLinkSpeedCalculator( linkSpeedCalculator ) ;
078
079                return linkBuilder.build(link, toQueueNode) ;
080        }
081        @Override
082        QNodeI createNetsimNode(final Node node) {
083                QNodeImpl.Builder builder = new QNodeImpl.Builder( netsimEngine, context ) ;
084
085                builder.setTurnAcceptanceLogic( this.turnAcceptanceLogic ) ;
086
087                return builder.build( node ) ;
088        }
089        public final void setLinkSpeedCalculator(LinkSpeedCalculator linkSpeedCalculator) {
090                this.linkSpeedCalculator = linkSpeedCalculator;
091        }
092        public final void setTurnAcceptanceLogic( TurnAcceptanceLogic turnAcceptanceLogic ) {
093                this.turnAcceptanceLogic = turnAcceptanceLogic;
094        }
095}
