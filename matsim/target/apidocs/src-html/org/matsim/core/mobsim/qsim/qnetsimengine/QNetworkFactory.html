032/**
033 * The current design idea of this is as follows:<ul>
034 * <li> It is now possible to inject this.  See {@link QSimModule} for how the default is set up.   It can also be overridden in the usual way.
035 * <li> For a specific example, see {@link DefaultQNetworkFactory}. 
036 * </ul>
037 * 
038 * <p></p>
039 * 
040 * @author dgrether, nagel
041 * 
042 * @see DefaultQNetworkFactory
043 * @see ConfigurableQNetworkFactory
044 *
045 */
046public abstract class QNetworkFactory implements MatsimFactory {
047
048        /**
049         * I need info from the mobsim.  However, as long as the factory is injected, it cannot get to "the" mobsim; at best, it
050         * can get an instance of the mobsim which is, however, not the same mobsim it will be working with.  So "initializeFactory"
051         * is called by the mobsim to provide some info about itself (e.g. agentCounter).
052         * <p></p>
053         * This could be solved using custom scopes.  They do, however, for the time being only exist in the guice inject framework, not
054         * in javax.inject, and thus we do not want to use them (yet?).
055         */
056        abstract void initializeFactory( AgentCounter agentCounter, MobsimTimer mobsimTimer, NetsimInternalInterface simEngine1 ) ;
057
058        abstract QNodeI createNetsimNode(Node node);
059
060        abstract QLinkI createNetsimLink(Link link, QNodeI queueNode);
061
062}
