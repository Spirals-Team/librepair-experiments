001package org.matsim.core.mobsim.qsim.jdeqsimengine;
002
003import org.matsim.core.mobsim.framework.Steppable;
004import org.matsim.core.mobsim.jdeqsim.Message;
005import org.matsim.core.mobsim.jdeqsim.MessageQueue;
006import org.matsim.core.mobsim.jdeqsim.Scheduler;
007
008import javax.inject.Inject;
009
010public class SteppableScheduler extends Scheduler implements Steppable {
011
012    private Message lookahead;
013    private boolean finished = false;
014
015    @Inject
016    public SteppableScheduler(MessageQueue queue) {
017        super(queue);
018    }
019
020    @Override
021    public void doSimStep(double time) {
022        finished = false; // I don't think we can restart once the queue has run dry, but just in case.
023        if (lookahead != null && time < lookahead.getMessageArrivalTime()) {
024            return;
025        }
026        if (lookahead != null) {
027            lookahead.processEvent();
028            lookahead.handleMessage();
029            lookahead = null;
030        }
031        while (!queue.isEmpty()) {
032            Message m = queue.getNextMessage();
033            if (m != null && m.getMessageArrivalTime() <= time) {
034                m.processEvent();
035                m.handleMessage();
036            } else {
037                lookahead = m;
038                return;
039            }
040        }
041        finished = true; // queue has run dry.
042    }
043
044    public boolean isFinished() {
045        return finished;
046    }
047
048}
