042public class ActivityEngine implements MobsimEngine, ActivityHandler {
043        private static final Logger log = Logger.getLogger( ActivityEngine.class ) ;
044
045        private EventsManager eventsManager;
046        @Inject
047        public ActivityEngine(EventsManager eventsManager) {
048                this.eventsManager = eventsManager;
049        }
050
051        public ActivityEngine(EventsManager eventsManager, AgentCounter agentCounter) {
052                this.eventsManager = eventsManager;
053        }

055        /**
056         * Agents cannot be added directly to the activityEndsList since that would
057         * not be thread-safe when within-day replanning is used. There, an agent's
058         * activity end time can be modified. As a result, the agent is located at
059         * the wrong position in the activityEndsList until it is updated by using
060         * rescheduleActivityEnd(...). However, if another agent is added to the list
061         * in the mean time, it might be inserted at the wrong position.
062         * cdobler, apr'12
063         */
064        private static class AgentEntry {
065                public AgentEntry(MobsimAgent agent, double activityEndTime) {
066                        this.agent = agent;
067                        this.activityEndTime = activityEndTime;
068                }
069                final MobsimAgent agent;
070                final double activityEndTime;
071        }
072
073        private InternalInterface internalInterface;
074        
075        /**
076         * This list needs to be a "blocking" queue since this is needed for
077         * thread-safety in the parallel qsim. cdobler, oct'10
078         */
079        private final Queue<AgentEntry> activityEndsList = new PriorityBlockingQueue<>(500, new Comparator<AgentEntry>() {
080
081                @Override
082                public int compare(AgentEntry arg0, AgentEntry arg1) {
083                        int cmp = Double.compare(arg0.activityEndTime, arg1.activityEndTime);
084                        if (cmp == 0) {
085                                // Both depart at the same time -> let the one with the larger id be first (=smaller)
086                                //
087                                // yy We are not sure what the above comment line is supposed to say.  Presumably, it is supposed
088                                // to say that the agent with the larger ID should be "smaller" one in the comparison.
089                                // In practice, it seems
090                                // that something like "emob_9" is before "emob_8", and something like "emob_10" before "emob_1".
091                                // It is unclear why this convention is supposed to be helpful.
092                                // kai & dominik, jul'12
093                                //
094                                return arg1.agent.getId().compareTo(arg0.agent.getId());
095                        }
096                        return cmp;
097                }
098
099        });
100        
101        // See handleActivity for the reason for this.
102        private boolean beforeFirstSimStep = true;
103
104        @Override
105        public void onPrepareSim() {
106                // Nothing to do here
107        }
108
109        @Override
110        public void doSimStep(double time) {
111                beforeFirstSimStep = false;
112                while (activityEndsList.peek() != null) {
113                        if (activityEndsList.peek().activityEndTime <= time) {
114                                MobsimAgent agent = activityEndsList.poll().agent;
115                                unregisterAgentAtActivityLocation(agent);
116                                agent.endActivityAndComputeNextState(time);
117                                internalInterface.arrangeNextAgentState(agent);
118                        } else {
119                                return;
120                        }
121                }
122        }
123
124        @Override
125        public void afterSim() {
126                double now = this.internalInterface.getMobsim().getSimTimer().getTimeOfDay();
127                for (AgentEntry entry : activityEndsList) {
128                        if (entry.activityEndTime!=Double.POSITIVE_INFINITY && entry.activityEndTime!=Time.UNDEFINED_TIME) {
129                                // since we are at an activity, it is not plausible to assume that the agents know mode or destination
130                                // link id.  Thus generating the event with ``null'' in the corresponding entries.  kai, mar'12
131                                eventsManager.processEvent(new PersonStuckEvent(now, entry.agent.getId(), null, null));
132                        }
133                }
134                activityEndsList.clear();
135        }
136
137        @Override
138        public void setInternalInterface(InternalInterface internalInterface) {
139                this.internalInterface = internalInterface;
140        }
141
142        
143        /**
144         * 
145         * This method is called by QSim to pass in agents which then "live" in the activity layer until they are handed out again
146         * through the internalInterface.
147         * 
148         * It is called not before onPrepareSim() and not after afterSim(), but it may be called before, after, or from doSimStep(),
149         * and even from itself (i.e. it must be reentrant), since internalInterface.arrangeNextAgentState() may trigger
150         * the next Activity.
151         * 
152         */
153        @Override
154        public boolean handleActivity(MobsimAgent agent) {
155                if (agent.getActivityEndTime() == Double.POSITIVE_INFINITY) {
156                        // This is the last planned activity.
157                        // So the agent goes to sleep.
158                        internalInterface.getMobsim().getAgentCounter().decLiving();
159                } else if (agent.getActivityEndTime() <= internalInterface.getMobsim().getSimTimer().getTimeOfDay() && !beforeFirstSimStep) {
160                        // This activity is already over (planned for 0 duration)
161                        // So we proceed immediately.
162                        agent.endActivityAndComputeNextState(internalInterface.getMobsim().getSimTimer().getTimeOfDay());
163                        internalInterface.arrangeNextAgentState(agent) ;
164                } else {
165                        // The agent commences an activity on this link.
166                        final AgentEntry agentEntry = new AgentEntry(agent, agent.getActivityEndTime());
167                        activityEndsList.add(agentEntry);
168                        internalInterface.registerAdditionalAgentOnLink(agent);
169                }
170                // Why beforeFirstSimStep matters:
171                // - If this class has never had a doSimStep() when this method is called, this means that this Agent is having its
172                // "overnight", i.e. first Activity.
173                // - This means that this ActivityEngine should not just pass this MobsimAgent along if its activityEndTime has already ended,
174                // but queue it in with other such Agents and let them all leave on doSimStep(), because we expect those Agents to leave
175                // in the order specified by the activityQueue (no matter if it is a good order or not, see comment there). The order in which new Agents enter
176                // the simulation and are passed into this method is a different one, so this matters.
177                // - This is safe (Agents will not miss a second), simply because doSimStep for this time step has not yet happened.
178                // - It also means that e.g. OTFVis will probably display all Agents while they are in their first Activity before you press play.
179                // - On the other hand, agents whose first activity is also their last activity go right to sleep "inside" this engine.
180                return true;
181        }
182
183        /**
184         * For within-day replanning. Tells this engine that the activityEndTime the agent reports may have changed since
185         * the agent was added to this engine through handleActivity.
186         * May be merged with handleActivity, since this engine can know by itself if it was called the first time
187         * or not.
188         *
189         * @param agent The agent.
190         */
191        @Override
192        public void rescheduleActivityEnd(final MobsimAgent agent) {
193                if ( agent.getState()!=State.ACTIVITY ) {
194                        return ;
195                }
196                
197                
198                double newActivityEndTime = agent.getActivityEndTime();
199                AgentEntry oldEntry = removeAgentFromQueue(agent);
200
201                // The intention in the following is that an agent that is no longer alive has an activity end time of infinity.  The number of
202                // alive agents is only modified when an activity end time is changed between a finite time and infinite.  kai, jun'11
203                if (oldEntry == null) {
204                        if (newActivityEndTime == Double.POSITIVE_INFINITY) {
205                                // agent was de-activated and still should be de-activated - nothing to do here
206                        } else {
207                                // re-activate the agent
208                                activityEndsList.add(new AgentEntry(agent, newActivityEndTime));
209                                internalInterface.registerAdditionalAgentOnLink(agent);
210                                ((org.matsim.core.mobsim.qsim.AgentCounter) internalInterface.getMobsim().getAgentCounter()).incLiving();
211                        }
212                } else if (newActivityEndTime == Double.POSITIVE_INFINITY) {
213                        /*
214                         * After the re-planning the agent's current activity has changed to its last activity.
215                         * Therefore the agent is de-activated. cdobler, oct'11
216                         */
217                        unregisterAgentAtActivityLocation(agent);
218                        internalInterface.getMobsim().getAgentCounter().decLiving();
219                } else {
220                        /*
221                         *  The activity is just rescheduled during the day, so we keep the agent active. cdobler, oct'11
222                         */
223                        activityEndsList.add(new AgentEntry(agent, newActivityEndTime));
224                }
225        }
226
227        private AgentEntry removeAgentFromQueue(MobsimAgent agent) {
228                Iterator<AgentEntry> iterator = activityEndsList.iterator();
229                while (iterator.hasNext()) {
230                        AgentEntry entry = iterator.next();
231                        if (entry.agent == agent) {
232                                iterator.remove();
233                                return entry;
234                        }
235                }
236                return null;
237        }
238
239        private void unregisterAgentAtActivityLocation(final MobsimAgent agent) {
240                Id<Person> agentId = agent.getId();
241                Id<Link> linkId = agent.getCurrentLinkId();
242                if (linkId != null) { // may be bushwacking
243                        internalInterface.unregisterAdditionalAgentOnLink(agentId, linkId);
244                }
245        }
