041public final class NetworkChangeEventsEngine implements NetworkChangeEventsEngineI {
042        private static final Logger log = Logger.getLogger(NetworkChangeEventsEngine.class) ;
043        
044        private Queue<NetworkChangeEvent> networkChangeEventsQueue = null;
045        private Netsim mobsim;
046        
047        private NetworkChangeEventsEngine() {}
048        
049        public static NetworkChangeEventsEngineI createNetworkChangeEventsEngine() {
050                return new NetworkChangeEventsEngine();
051        }
052        
053        @Override
054        public void setInternalInterface( InternalInterface internalInterface ) {
055                this.mobsim = internalInterface.getMobsim();
056        }
057
058        @Override
059        public void afterSim() {
060                
061        }
062
063        @Override
064        public void onPrepareSim() {
065                Queue<NetworkChangeEvent> changeEvents = NetworkUtils.getNetworkChangeEvents(this.mobsim.getScenario().getNetwork());
066                if ((changeEvents != null) && (changeEvents.size() > 0)) {
067                        this.networkChangeEventsQueue = new PriorityQueue<>(changeEvents.size(), new NetworkChangeEvent.StartTimeComparator());
068                        this.networkChangeEventsQueue.addAll(changeEvents);
069                }
070
071                // one could replace the above lines by
072//              this.networkChangeEventsQueue = NetworkUtils.getNetworkChangeEvents(this.mobsim.getScenario().getNetwork()) ;
073                // This also passes at least the core tests.  However, the code below "consumes" the network change events queue.
074                // It seems that we should rather not do this.
075
076        }
077
078        @Override
079        public void doSimStep(double time) {
080                if ((this.networkChangeEventsQueue != null) && (this.networkChangeEventsQueue.size() > 0)) {
081                        handleNetworkChangeEvents(time);
082                }
083        }
084
085        private void handleNetworkChangeEvents(final double time) {
086                while ((this.networkChangeEventsQueue.size() > 0) && (this.networkChangeEventsQueue.peek().getStartTime() <= time)) {
087                        NetworkChangeEvent event = this.networkChangeEventsQueue.poll();
088                        handleNetworkChangeEvent(event);
089                }
090        }
091        
092        public final void addNetworkChangeEvent( NetworkChangeEvent event ) {
093                // used (and thus implicitly tested) by bdi-abm-integration project.  A separate core test would be good. kai, feb'18
094                
095                log.warn("add change event coming from external (i.e. not in network change events data structure):" + event);
096                this.networkChangeEventsQueue.add(event);
097                final Queue<NetworkChangeEvent> centralNetworkChangeEvents = NetworkUtils.getNetworkChangeEvents(this.mobsim.getScenario().getNetwork());
098                if ( !centralNetworkChangeEvents.contains( event ) ) {
099                        centralNetworkChangeEvents.add( event ) ;
100                        // need to add this here since otherwise speed lookup in mobsim does not work. And need to hedge against
101                        // code that may already have added it by itself.  kai, feb'18
102                }
103
104                handleNetworkChangeEvent(event);
105        }
106        
107        private void handleNetworkChangeEvent(NetworkChangeEvent event) {
108                for (Link link : event.getLinks()) {
109                        final NetsimLink netsimLink = this.mobsim.getNetsimNetwork().getNetsimLink(link.getId());
110                        if ( netsimLink instanceof TimeVariantLink) {
111                                ((TimeVariantLink) netsimLink).recalcTimeVariantAttributes();
112                        } else {
113                                throw new RuntimeException("link not time variant") ;
114                        }
115                }
116        }
