040/**
041 * The idea here is that there are the following levels:<ul>
042 * <li> Run-specific objects, such as {@link QSimConfigGroup}, {@link EventsManager}, etc.  These are in general guice-injected, but can
043 * also be configured by the constructor.  In the longer run, I would like to get rid of {@link Scenario}, but I haven't checked where
044 * it us truly needed yet.
045 * <li> Mobsim-specific objects, such as {@link AgentCounter} or {@link QNetsimEngine}.  Since the mobsim is re-created in every
046 * iteration, they cannot be injected via guice, at least not via the global inject mechanism that has the override facility.
047 * <li> The last level are link- oder node-specific objects such as {@link Link}  or {@QNode}.  They are arguments to the 
048 * creational methods.
049 * <li> The main underlying implementations are {@link QueueWithBuffer}, the factory of which is inserted into {@link QLinkImpl}.  This
050 * was the syntax that could subsume all other syntactic variants.  Builders are used to set defaults and to avoid overly long
051 * constructors.
052 * <li> {@link QLinkImpl} is essentially the container where vehicles are parked, agents perform activities, etc.  MATSim has some tendency to
053 * have them centralized in the QSim (see, e.g., {@link TransitStopAgentTracker}), but both for parallel computing and for visualization, 
054 * having agents on a decentralized location is helpful.  
055 * <li> Most functionality of {@link QLinkImpl} is actually in {@link AbstractQLink}, which
056 * can also be used as basic infrastructure by other qnetworks.  
057 * <li> {@link QueueWithBuffer} is an instance of {@link QLaneI} and can be replaced accordingly.  
058 * <li> One can also replace the {@link VehicleQ} that works inside {@link QueueWithBuffer}. 
059 * </ul>
060 * 
061 * @author dgrether, knagel
062 * 
063 * @see ConfigurableQNetworkFactory
064 */
065public final class DefaultQNetworkFactory extends QNetworkFactory {
066        private EventsManager events ;
067        private Scenario scenario ;
068        // (vis needs network and may need population attributes and config; in consequence, makes sense to have scenario here. kai, apr'16)
069        private NetsimEngineContext context;
070        private NetsimInternalInterface netsimEngine ;
071        @Inject
072        DefaultQNetworkFactory( EventsManager events, Scenario scenario ) {
073                this.events = events;
074                this.scenario = scenario;
075        }
076        @Override
077        void initializeFactory( AgentCounter agentCounter, MobsimTimer mobsimTimer, NetsimInternalInterface netsimEngine1 ) {
078                this.netsimEngine = netsimEngine1;
079                double effectiveCellSize = scenario.getNetwork().getEffectiveCellSize() ;
080
081                SnapshotLinkWidthCalculator linkWidthCalculator = new SnapshotLinkWidthCalculator();
082                linkWidthCalculator.setLinkWidthForVis( scenario.getConfig().qsim().getLinkWidthForVis() );
083                linkWidthCalculator.setLaneWidth( scenario.getNetwork().getEffectiveLaneWidth() );
084
085                AbstractAgentSnapshotInfoBuilder agentSnapshotInfoBuilder = QNetsimEngine.createAgentSnapshotInfoBuilder( scenario, linkWidthCalculator );
086
087                context = new NetsimEngineContext( events, effectiveCellSize, agentCounter, agentSnapshotInfoBuilder, scenario.getConfig().qsim(), 
088                                mobsimTimer, linkWidthCalculator );
089        }
090        @Override
091        QLinkI createNetsimLink(final Link link, final QNodeI toQueueNode) {
092                QLinkImpl.Builder linkBuilder = new QLinkImpl.Builder(context, netsimEngine) ;
093                return linkBuilder.build(link, toQueueNode) ;
094        }
095        @Override
096        QNodeI createNetsimNode(final Node node) {
097                QNodeImpl.Builder builder = new QNodeImpl.Builder( netsimEngine, context ) ;
098                return builder.build( node ) ;
099        }
100}
