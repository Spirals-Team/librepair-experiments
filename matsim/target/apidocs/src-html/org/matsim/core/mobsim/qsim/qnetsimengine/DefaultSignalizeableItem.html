020package org.matsim.core.mobsim.qsim.qnetsimengine;
021
022import org.matsim.api.core.v01.Id;
023import org.matsim.api.core.v01.network.Link;
024import org.matsim.core.mobsim.qsim.interfaces.SignalGroupState;
025import org.matsim.core.mobsim.qsim.interfaces.SignalizeableItem;
026
027import java.util.HashMap;
028import java.util.Map;
029import java.util.Set;
030
031
032/**
033 * @author dgrether
034 *
035 */
036public final class DefaultSignalizeableItem implements SignalizeableItem {
037
038        private Map<Id<Link>, SignalGroupState> toLinkIdSignalStates = null;
039        private SignalGroupState allToLinksState = SignalGroupState.GREEN;
040        private boolean linkGreen = true;
041        private Set<Id<Link>> outLinks;
042        
043        public DefaultSignalizeableItem(Set<Id<Link>> outLinks){
044                this.outLinks = outLinks;
045        }
046
047        @Override
048        public void setSignalStateAllTurningMoves(SignalGroupState state) {
049                this.allToLinksState = state;
050                this.linkGreen = checkGreen(state);
051        }
052
053        private void initToLinkIdSignalStates(){
054                this.allToLinksState = null;
055                this.toLinkIdSignalStates = new HashMap<>();
056                for (Id<Link> outLinkId : this.outLinks){
057                        this.toLinkIdSignalStates.put(outLinkId, SignalGroupState.GREEN);
058                }
059        }
060        
061        @Override
062        public void setSignalStateForTurningMove(SignalGroupState state, Id<Link> toLinkId) {
063                if (this.toLinkIdSignalStates == null){
064                        this.initToLinkIdSignalStates();
065                }
066                this.toLinkIdSignalStates.put(toLinkId, state);
067                if (checkGreen(state)){
068                        this.linkGreen = true;
069                }
070                else {
071                        boolean foundGreen = false;
072                        for (SignalGroupState sgs : this.toLinkIdSignalStates.values()){
073                                if (checkGreen(sgs)){
074                                        foundGreen = true;
075                                }
076                        }
077                        this.linkGreen = foundGreen;
078                }
079        }
080
081        private static boolean checkGreen(SignalGroupState state) {
082                return (state.equals(SignalGroupState.GREEN) || state.equals(SignalGroupState.YELLOW) || state.equals(SignalGroupState.OFF));
083        }
084
085        /**
086         * returns true, if at least on signal at the link/lane shows green
087         */
088        public boolean hasGreenForAllToLinks() {
089                return linkGreen;
090        }
091        
092        public boolean hasGreenForToLink(Id<Link> toLinkId){
093                if (this.allToLinksState != null) {
094                        return checkGreen(this.allToLinksState);
095                }
096                return checkGreen(this.toLinkIdSignalStates.get(toLinkId));
097        }
098        
099        @Override
100        public void setSignalized(boolean isSignalized) {
101                //nothing to do
102        }
103
104}
