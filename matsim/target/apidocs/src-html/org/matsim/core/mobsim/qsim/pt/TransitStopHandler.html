021package org.matsim.core.mobsim.qsim.pt;
022
023import java.util.List;
024
025import org.matsim.core.mobsim.qsim.interfaces.MobsimVehicle;
026import org.matsim.pt.transitSchedule.api.TransitStopFacility;
027
028/**
029 * Provides a replaceable behavior for implementing different loading and unloading procedures
030 * for transit vehicles.
031 *
032 * @author mrieser
033 */
034public interface TransitStopHandler {
035
036        /**
037         * <p>Handles the loading and unloading for a transit vehicle. Given a list of passengers that
038         * want to either board or exit the vehicle, the method can define a behavior for the
039         * process of entering and leaving passengers. The method has to specifically call
040         * {@link PassengerAccessEgress#handlePassengerEntering(PTPassengerAgent, double)} and
041         * {@link PassengerAccessEgress#handlePassengerLeaving(PTPassengerAgent, double)} to let
042         * passengers in and out the vehicle. This allows implementations of this interface to
043         * define different strategies, e.g. a very simple one where everybody enters and leaves at
044         * the same time, or one where first all passengers leave, and at a later time all passengers
045         * enter, etc.</p>
046         * <p>The method must return the time how long the vehicle will have to stay at least at the stop
047         * to let the passengers enter/exit. If the method returns a value > 0, the method will be called
048         * again at a later time for the same stop, as in between some additional people could have arrived
049         * that want to enter. Only if the method returns 0, it may continue on its route to the next stop.
050         * </p>
051         *
052         * @param stop the stop the vehicle is currently at
053         * @param now the current time
054         * @param leavingPassengers the list of passengers that want to exit
055         * @param enteringPassengers the list of passengers that want to board
056         * @param handler provides methods for actually removing/adding passengers to the vehicle
057         * @return the time (in seconds) how long the vehicle will stay at least at this stop.
058         */
059        public double handleTransitStop(final TransitStopFacility stop, final double now,
060                        final List<PTPassengerAgent> leavingPassengers, final List<PTPassengerAgent> enteringPassengers,
061                        final PassengerAccessEgress handler, MobsimVehicle vehicle);
062
063}
