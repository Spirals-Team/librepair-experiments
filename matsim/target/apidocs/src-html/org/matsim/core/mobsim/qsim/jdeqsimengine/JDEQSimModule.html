001package org.matsim.core.mobsim.qsim.jdeqsimengine;
002
003import org.matsim.core.config.ConfigUtils;
004import org.matsim.core.mobsim.jdeqsim.JDEQSimConfigGroup;
005import org.matsim.core.mobsim.jdeqsim.MessageQueue;
006import org.matsim.core.mobsim.qsim.QSim;
007
008public class JDEQSimModule {
009
010    private JDEQSimModule() {}
011
012    public static void configure(QSim qsim) {
013        SteppableScheduler scheduler = new SteppableScheduler(new MessageQueue());
014        JDEQSimEngine jdeqSimEngine = new JDEQSimEngine(ConfigUtils.addOrGetModule(qsim.getScenario().getConfig(), JDEQSimConfigGroup.NAME, JDEQSimConfigGroup.class), qsim.getScenario(), qsim.getEventsManager(), qsim.getAgentCounter(), scheduler);
015        qsim.addMobsimEngine(jdeqSimEngine);
016        qsim.addActivityHandler(jdeqSimEngine);
017    }
018
019}
