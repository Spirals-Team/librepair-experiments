023package org.matsim.core.mobsim.qsim;
024
025import java.util.Collection;
026
027import javax.inject.Inject;
028
029import org.apache.log4j.Logger;
030import org.matsim.core.mobsim.framework.AgentSource;
031import org.matsim.core.mobsim.framework.listeners.MobsimListener;
032import org.matsim.core.mobsim.qsim.interfaces.ActivityHandler;
033import org.matsim.core.mobsim.qsim.interfaces.DepartureHandler;
034import org.matsim.core.mobsim.qsim.interfaces.MobsimEngine;
035import org.matsim.core.mobsim.qsim.interfaces.Netsim;
036
037import com.google.inject.Injector;
038import com.google.inject.Module;
039import com.google.inject.Provider;
040
041public class QSimProvider implements Provider<QSim> {
042        private static final Logger log = Logger.getLogger( QSimProvider.class ) ;
043        
044        private Injector injector;
045        private Collection<AbstractQSimPlugin> plugins;
046
047        @Inject
048    QSimProvider(Injector injector, Collection<AbstractQSimPlugin> plugins) {
049        this.injector = injector;
050                this.plugins = plugins;
051    }
052
053    @Override
054    public QSim get() {
055            com.google.inject.AbstractModule module = new com.google.inject.AbstractModule() {
056                        @Override
057                        protected void configure() {
058                                for (AbstractQSimPlugin plugin : plugins) {
059                                        // install each plugin's modules:
060                                        for (Module module1 : plugin.modules()) {
061                                                install(module1);
062                                        }
063                                }
064                                bind(QSim.class).asEagerSingleton();
065                                bind(Netsim.class).to(QSim.class);
066                        }
067                };
068        Injector qSimLocalInjector = injector.createChildInjector(module);
069        org.matsim.core.controler.Injector.printInjector( qSimLocalInjector, log ) ;
070        QSim qSim = qSimLocalInjector.getInstance(QSim.class);
071//        qSim.setChildInjector( qSimLocalInjector ) ;
072        for (AbstractQSimPlugin plugin : plugins) {
073                        // add each plugin's mobsim engines:
074                        for (Class<? extends MobsimEngine> mobsimEngine : plugin.engines()) {
075                                qSim.addMobsimEngine(qSimLocalInjector.getInstance(mobsimEngine));
076                        }
077                        // add each plugin's activity handlers:
078                        for (Class<? extends ActivityHandler> activityHandler : plugin.activityHandlers()) {
079                                qSim.addActivityHandler(qSimLocalInjector.getInstance(activityHandler));
080                        }
081                        // add each plugin's departure handlers:
082                        for (Class<? extends DepartureHandler> mobsimEngine : plugin.departureHandlers()) {
083                                qSim.addDepartureHandler(qSimLocalInjector.getInstance(mobsimEngine));
084                        }
085                        // add each plugin's mobsim listeners:
086                        for (Class<? extends MobsimListener> mobsimListener : plugin.listeners()) {
087                                qSim.addQueueSimulationListeners(qSimLocalInjector.getInstance(mobsimListener));
088                        }
089                        // add each plugin's agent sources:
090                        for (Class<? extends AgentSource> agentSource : plugin.agentSources()) {
091                                qSim.addAgentSource(qSimLocalInjector.getInstance(agentSource));
092                        }
093                }
094        return qSim;
095    }
096
097}
