026/**
027 * Includes all agents that have transportation modes unknown to the
028 * NetsimEngine (often all != "car") or have two activities on the same link
029 */
030public final class DefaultTeleportationEngine implements TeleportationEngine {
031        private static final Logger log = Logger.getLogger( DefaultTeleportationEngine.class ) ;
032        
033        private final Queue<Tuple<Double, MobsimAgent>> teleportationList = new PriorityQueue<>(
034                        30, new Comparator<Tuple<Double, MobsimAgent>>() {
035
036                @Override
037                public int compare(Tuple<Double, MobsimAgent> o1, Tuple<Double, MobsimAgent> o2) {
038                        int ret = o1.getFirst().compareTo(o2.getFirst()); // first compare time information
039                        if (ret == 0) {
040                                ret = o2.getSecond().getId().compareTo(o1.getSecond().getId()); // if they're equal, compare the Ids: the one with the larger Id should be first
041                        }
042                        return ret;
043                }
044        });
045        private final LinkedHashMap<Id<Person>, TeleportationVisData> teleportationData = new LinkedHashMap<>();
046        private InternalInterface internalInterface;
047        private Scenario scenario;
048        private EventsManager eventsManager;
049        
050        private final boolean withTravelTimeCheck ;
051
052        @Inject
053        public DefaultTeleportationEngine(Scenario scenario, EventsManager eventsManager) {
054                this.scenario = scenario;
055                this.eventsManager = eventsManager;
056                
057                withTravelTimeCheck = scenario.getConfig().qsim().isUsingTravelTimeCheckInTeleportation() ;
058        }
059
060        @Override
061        public boolean handleDeparture(double now, MobsimAgent agent, Id<Link> linkId) {
062                if ( agent.getExpectedTravelTime()==null || agent.getExpectedTravelTime()==Time.UNDEFINED_TIME ) {
063                        Logger.getLogger( this.getClass() ).info( "mode: " + agent.getMode() );
064                        throw new RuntimeException("teleportation does not work when travel time is undefined.  There is also really no magic fix for this,"
065                                        + " since we cannot guess travel times for arbitrary modes and arbitrary landscapes.  kai/mz, apr'15 & feb'16") ;
066                }
067
068                Double travelTime = agent.getExpectedTravelTime() ;
069                if ( withTravelTimeCheck ) {
070                        Double speed = scenario.getConfig().plansCalcRoute().getTeleportedModeSpeeds().get( agent.getMode() ) ;
071                        Facility<?> dpfac = agent.getCurrentFacility() ;
072                        Facility<?> arfac = agent.getDestinationFacility() ;
073                        travelTime = DefaultTeleportationEngine.travelTimeCheck(travelTime, speed, dpfac, arfac);
074                }
075        
076                double arrivalTime = now + travelTime ;
077                this.teleportationList.add(new Tuple<>(arrivalTime, agent));
078                
079                // === below here is only visualization, no dynamics ===
080                Id<Person> agentId = agent.getId();
081                Link currLink = this.scenario .getNetwork().getLinks().get(linkId);
082                Link destLink = this.scenario .getNetwork().getLinks().get(agent.getDestinationLinkId());
083                Coord fromCoord = currLink.getToNode().getCoord();
084                Coord toCoord = destLink.getToNode().getCoord();
085                TeleportationVisData agentInfo = new TeleportationVisData(now, agentId, fromCoord, toCoord, travelTime);
086                this.teleportationData.put(agentId, agentInfo);
087                
088                return true;
089        }
090
091        @Override
092        public Collection<AgentSnapshotInfo> addAgentSnapshotInfo(Collection<AgentSnapshotInfo> snapshotList) {
093                double time = internalInterface.getMobsim().getSimTimer().getTimeOfDay();
094                for (TeleportationVisData teleportationVisData : teleportationData.values()) {
095                        teleportationVisData.updatePosition(time);
096                        snapshotList.add(teleportationVisData);
097                }
098                return snapshotList;
099        }
100
101        @Override
102        public void doSimStep(double time) {
103                handleTeleportationArrivals();
104        }
105
106        private void handleTeleportationArrivals() {
107                double now = internalInterface.getMobsim().getSimTimer().getTimeOfDay();
108                while (teleportationList.peek() != null) {
109                        Tuple<Double, MobsimAgent> entry = teleportationList.peek();
110                        if (entry.getFirst() <= now) {
111                                teleportationList.poll();
112                                MobsimAgent personAgent = entry.getSecond();
113                                personAgent.notifyArrivalOnLinkByNonNetworkMode(personAgent
114                                                .getDestinationLinkId());
115                                double distance = personAgent.getExpectedTravelDistance();
116                                this.eventsManager.processEvent(new TeleportationArrivalEvent(this.internalInterface.getMobsim().getSimTimer().getTimeOfDay(), personAgent.getId(), distance));
117                                personAgent.endLegAndComputeNextState(now);
118                                this.teleportationData.remove(personAgent.getId());
119                                internalInterface.arrangeNextAgentState(personAgent);
120                        } else {
121                                break;
122                        }
123                }
124        }
125
126        @Override
127        public void onPrepareSim() {
128
129        }
130
131        @Override
132        public void afterSim() {
133                double now = internalInterface.getMobsim().getSimTimer().getTimeOfDay();
134                for (Tuple<Double, MobsimAgent> entry : teleportationList) {
135                        MobsimAgent agent = entry.getSecond();
136                        eventsManager.processEvent(new PersonStuckEvent(now, agent.getId(), agent.getDestinationLinkId(), agent.getMode()));
137                }
138                teleportationList.clear();
139        }
140
141        @Override
142        public void setInternalInterface(InternalInterface internalInterface) {
143                this.internalInterface = internalInterface;
144        }
145
146        private static Double travelTimeCheck(Double travelTime, Double speed, Facility<?> dpfac, Facility<?> arfac) {
147                if ( speed==null ) {
148                        // if we don't have a bushwhacking speed, the only thing we can do is trust the router
149                        return travelTime ;
150                } 
151                
152                if ( dpfac == null || arfac == null ) {
153                        log.warn( "dpfac = " + dpfac ) ;
154                        log.warn( "arfac = " + arfac ) ;
155                        throw new RuntimeException("have bushwhacking mode but nothing that leads to coordinates; don't know what to do ...") ;
156                        // (means that the agent is not correctly implemented)
157                }
158                
159                if ( dpfac.getCoord()==null || arfac.getCoord()==null ) {
160                        // yy this is for example the case if activities are initialized at links, without coordinates.  Could use the link coordinate
161                        // instead, but somehow this does not feel any better. kai, feb'16
162                        
163                        return travelTime ;
164                }
165                        
166                final Coord dpCoord = dpfac.getCoord();
167                final Coord arCoord = arfac.getCoord();
168                                
169                double dist = NetworkUtils.getEuclideanDistance( dpCoord, arCoord ) ;
170                double travelTimeTmp = dist / speed ;
171                
172                if ( travelTimeTmp < travelTime ) {
173                        return travelTime ;
174                }
175                        
176                return travelTimeTmp ;
177        }
