021package org.matsim.core.mobsim.qsim.agents;
022
023import java.util.List;
024
025import org.apache.log4j.Logger;
026import org.matsim.api.core.v01.Id;
027import org.matsim.api.core.v01.network.Link;
028import org.matsim.api.core.v01.population.Person;
029import org.matsim.api.core.v01.population.Plan;
030import org.matsim.api.core.v01.population.PlanElement;
031import org.matsim.core.mobsim.framework.HasPerson;
032import org.matsim.core.mobsim.framework.PlanAgent;
033import org.matsim.core.mobsim.qsim.interfaces.MobsimVehicle;
034import org.matsim.core.mobsim.qsim.interfaces.Netsim;
035import org.matsim.core.mobsim.qsim.pt.MobsimDriverPassengerAgent;
036import org.matsim.core.mobsim.qsim.pt.TransitVehicle;
037import org.matsim.facilities.Facility;
038import org.matsim.pt.transitSchedule.api.TransitLine;
039import org.matsim.pt.transitSchedule.api.TransitRoute;
040import org.matsim.pt.transitSchedule.api.TransitRouteStop;
041import org.matsim.pt.transitSchedule.api.TransitStopFacility;
042import org.matsim.vehicles.Vehicle;
043
044/**
045 * @author mrieser
046 */
047public final class TransitAgent implements MobsimDriverPassengerAgent, PlanAgent, HasPerson {
048        @SuppressWarnings("unused")
049        private static final Logger log = Logger.getLogger(TransitAgent.class);
050
051        private BasicPlanAgentImpl basicAgentDelegate ;
052        private PlanBasedDriverAgentImpl driverAgentDelegate ;
053        private TransitAgentImpl transitAgentDelegate ;
054
055        public static TransitAgent createTransitAgent(Person p, Netsim simulation) {
056                TransitAgent agent = new TransitAgent(p, simulation);
057                return agent;
058        }
059
060        private TransitAgent(final Person p, final Netsim simulation) {
061                basicAgentDelegate = new BasicPlanAgentImpl( p.getSelectedPlan(), simulation.getScenario(), simulation.getEventsManager(), 
062                                simulation.getSimTimer() ) ;
063                driverAgentDelegate = new PlanBasedDriverAgentImpl( basicAgentDelegate ) ;
064                transitAgentDelegate = new TransitAgentImpl( basicAgentDelegate );
065        }
066
067        @Override
068        public final void endLegAndComputeNextState(double now) {
069                basicAgentDelegate.endLegAndComputeNextState(now);
070        }
071        @Override
072        public final void setStateToAbort(double now) {
073                basicAgentDelegate.setStateToAbort(now);
074        }
075        @Override
076        public final void notifyArrivalOnLinkByNonNetworkMode(Id<Link> linkId) {
077                basicAgentDelegate.notifyArrivalOnLinkByNonNetworkMode(linkId);
078        }
079        @Override
080        public final void endActivityAndComputeNextState(double now) {
081                basicAgentDelegate.endActivityAndComputeNextState(now);
082        }
083        @Override
084        public final Id<Vehicle> getPlannedVehicleId() {
085                return basicAgentDelegate.getPlannedVehicleId();
086        }
087        @Override
088        public final String getMode() {
089                return basicAgentDelegate.getMode();
090        }
091        @Override
092        public String toString() {
093                return basicAgentDelegate.toString();
094        }
095        @Override
096        public final Double getExpectedTravelTime() {
097                return basicAgentDelegate.getExpectedTravelTime();
098        }
099        @Override
100        public final Double getExpectedTravelDistance() {
101                return basicAgentDelegate.getExpectedTravelDistance();
102        }       
103        @Override
104        public final PlanElement getCurrentPlanElement() {
105                return basicAgentDelegate.getCurrentPlanElement();
106        }
107        @Override
108        public final PlanElement getNextPlanElement() {
109                return basicAgentDelegate.getNextPlanElement();
110        }
111        @Override
112        public final Plan getCurrentPlan() {
113                return basicAgentDelegate.getCurrentPlan();
114        }
115        @Override
116        public final Id<Person> getId() {
117                return basicAgentDelegate.getId();
118        }
119        @Override
120        public final Person getPerson() {
121                return basicAgentDelegate.getPerson();
122        }
123        @Override
124        public final MobsimVehicle getVehicle() {
125                return basicAgentDelegate.getVehicle();
126        }
127        @Override
128        public final void setVehicle(MobsimVehicle vehicle) {
129                basicAgentDelegate.setVehicle(vehicle);
130        }
131        @Override
132        public final Id<Link> getCurrentLinkId() {
133                return basicAgentDelegate.getCurrentLinkId();
134        }
135        @Override
136        public final Id<Link> getDestinationLinkId() {
137                return basicAgentDelegate.getDestinationLinkId();
138        }
139        @Override
140        public final double getActivityEndTime() {
141                return basicAgentDelegate.getActivityEndTime();
142        }
143        @Override
144        public final State getState() {
145                return basicAgentDelegate.getState();
146        }
147        @Override
148        public final void notifyMoveOverNode(Id<Link> newLinkId) {
149                driverAgentDelegate.notifyMoveOverNode(newLinkId);
150        }
151        @Override
152        public final Id<Link> chooseNextLinkId() {
153                return driverAgentDelegate.chooseNextLinkId();
154        }
155        @Override
156        public final boolean isWantingToArriveOnCurrentLink() {
157                return driverAgentDelegate.isWantingToArriveOnCurrentLink();
158        }
159        @Override
160        public final boolean getExitAtStop(TransitStopFacility stop) {
161                return transitAgentDelegate.getExitAtStop(stop);
162        }
163        @Override
164        public final boolean getEnterTransitRoute(TransitLine line, TransitRoute transitRoute, List<TransitRouteStop> stopsToCome,
165                        TransitVehicle transitVehicle) {
166                return transitAgentDelegate.getEnterTransitRoute(line, transitRoute, stopsToCome, transitVehicle);
167        }
168        @Override
169        public final double getWeight() {
170                return transitAgentDelegate.getWeight();
171        }
172        @Override
173        public final Id<TransitStopFacility> getDesiredAccessStopId() {
174                return transitAgentDelegate.getDesiredAccessStopId();
175        }
176        @Override
177        public final Id<TransitStopFacility> getDesiredDestinationStopId() {
178                return transitAgentDelegate.getDesiredDestinationStopId();
179        }
180
181        @Override
182        public final PlanElement getPreviousPlanElement() {
183                return this.basicAgentDelegate.getPreviousPlanElement();
184        }
185
186        @Override
187        public Facility<? extends Facility<?>> getCurrentFacility() {
188                return this.basicAgentDelegate.getCurrentFacility();
189        }
190
191        @Override
192        public Facility<? extends Facility<?>> getDestinationFacility() {
193                return this.basicAgentDelegate.getDestinationFacility();
194        }
195
196}
