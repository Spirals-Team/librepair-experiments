117        @Override
118        public boolean doSimStep() {
119                double now = context.getSimTimer().getTimeOfDay() ;
120                qlane.initBeforeSimStep();
121                
122                if ( context.qsimConfig.isInsertingWaitingVehiclesBeforeDrivingVehicles() ) {
123                        this.moveWaitToRoad();
124                        this.getTransitQLink().handleTransitVehiclesInStopQueue(now);
125                        qlane.doSimStep();
126                } else {
127                        this.getTransitQLink().handleTransitVehiclesInStopQueue(now);
128                        qlane.doSimStep();
129                        this.moveWaitToRoad();
130                }
131                this.setActive(this.checkForActivity());
132                return isActive();
133                // yy seems to me that for symmetry there should be something like
134                //                      netElementActivationRegistry.registerLinkAsActive(this);
135                // and may be a qlink.deactivateLink(...) around it (analogous to qlink.activateLink).  
136                // That is, do NOT pass the deactivation of the link rather implicitly via returning a false here.
137                // kai, mar'16
138        }

141        /**
142         * Move as many waiting cars to the link as it is possible
143         */
144        private void moveWaitToRoad() {
145                while (!getWaitingList().isEmpty()) {
146                        if (!qlane.isAcceptingFromWait(this.getWaitingList().peek())) {
147                                return;
148                        }
149            QVehicle veh = this.getWaitingList().poll();
150                        
151                        double now = context.getSimTimer().getTimeOfDay() ;
152                        context.getEventsManager().processEvent(
153                                        new VehicleEntersTrafficEvent(now, veh.getDriver().getId(), this.getLink().getId(), veh.getId(), veh.getDriver().getMode(), 1.0));
154
155                        if ( this.getTransitQLink().addTransitToStopQueue(now, veh, this.getLink().getId()) ) {
156                                continue ;
157                        }
158
159                        if ( veh.getDriver().isWantingToArriveOnCurrentLink() ) {
160                                // If the driver wants to stop (again) on this link, give them a special treatment.
161                                // addFromWait doesn't work here, because after that, they cannot stop anymore.
162                                qlane.addTransitSlightlyUpstreamOfStop(veh) ;
163                                continue;
164                        }
165
166                        qlane.addFromWait(veh);
167                }
168        }

175        @Override public void recalcTimeVariantAttributes() {
176                double now = context.getSimTimer().getTimeOfDay() ;
177                qlane.changeUnscaledFlowCapacityPerSecond( ((Link) this.getLink()).getFlowCapacityPerSec(now) );
178                qlane.changeEffectiveNumberOfLanes(this.getLink().getNumberOfLanes(now));
179
180                //              qlane.changeSpeedMetersPerSecond( getLink().getFreespeed(now) ) ; flowCap & nLanes are "push", freeSpeed is
181                //"pull".  This is, however, not completely honest w.r.t. freeSpeed consequences to the fdiag, but also wasn't correctly
182                //thought through/implemented when I found it.  kai, feb'18
183
184                qlane.recalcTimeVariantAttributes();
185        }

187        @Override
188        public QVehicle getVehicle(Id<Vehicle> vehicleId) {
189                QVehicle ret = super.getVehicle(vehicleId);
190                if (ret != null) {
191                        return ret;
192                }
193                for (QVehicle veh : this.getWaitingList()) {
194                        if (veh.getId().equals(vehicleId))
195                                return veh;
196                }
197                return this.qlane.getVehicle( vehicleId ) ;
198        }

231        private boolean checkForActivity() {
232                /*
233                 * Leave Link active as long as there are vehicles on the link (ignore
234                 * buffer because the buffer gets emptied by nodes and not links) and leave
235                 * link active until buffercap has accumulated (so a newly arriving vehicle
236                 * is not delayed).
237                 */
238                return qlane.isActive()  || !this.getWaitingList().isEmpty() || !this.getTransitQLink().getTransitVehicleStopQueue().isEmpty() ;
239        }

263        /**
264         * Inner class to encapsulate visualization methods
265         *
266         * @author dgrether
267         */
268        class VisDataImpl implements VisData {
269
270                private VisLaneModelBuilder visModelBuilder = null;
271                private VisLinkWLanes visLink = null;
272
273                private VisDataImpl() {
274                        double nodeOffset = context.qsimConfig.getNodeOffset(); 
275                        if (nodeOffset != 0.0) {
276                                nodeOffset = nodeOffset +2.0; // +2.0: eventually we need a bit space for the signal
277                                visModelBuilder = new VisLaneModelBuilder();
278                                CoordinateTransformation transformation = new IdentityTransformation();
279                                visLink = visModelBuilder.createVisLinkLanes(transformation, QLinkImpl.this, nodeOffset, null);
280                                visModelBuilder.recalculatePositions(visLink, context.linkWidthCalculator);
281                        }
282                }
283
284                @Override
285                public Collection<AgentSnapshotInfo> addAgentSnapshotInfo( Collection<AgentSnapshotInfo> positions) {
286//                      AbstractAgentSnapshotInfoBuilder snapshotInfoBuilder = qnetwork.simEngine.getAgentSnapshotInfoBuilder();
287
288                        QLaneI.VisData roadVisData = getAcceptingQLane().getVisData() ;
289                        if (visLink != null) {
290                                ((QueueWithBuffer.VisDataImpl)roadVisData).setVisInfo(visLink.getLinkStartCoord(), visLink.getLinkEndCoord()) ;
291                                // yyyy not so great but an elegant solution needs more thinking about visualizer structure. kai, jun'13
292                        }
293
294                        double now = context.getSimTimer().getTimeOfDay() ;
295                        positions = roadVisData.addAgentSnapshotInfo(positions,now) ;
296
297                        int cnt2 = 10 ; // a counter according to which non-moving items can be "spread out" in the visualization
298                        // initialize a bit away from the lane
299
300                        // treat vehicles from transit stops
301                        cnt2 = context.snapshotInfoBuilder.positionVehiclesFromTransitStop(positions, getLink(), getTransitQLink().getTransitVehicleStopQueue(), cnt2 );
302
303                        // treat vehicles from waiting list:
304                        cnt2 = context.snapshotInfoBuilder.positionVehiclesFromWaitingList(positions, QLinkImpl.this.getLink(), cnt2,
305                                        QLinkImpl.this.getWaitingList());
306
307                        cnt2 = context.snapshotInfoBuilder.positionAgentsInActivities(positions, QLinkImpl.this.getLink(),
308                                        QLinkImpl.this.getAdditionalAgentsOnLink(), cnt2);
309
310                        return positions;
311                }
312
313        }
