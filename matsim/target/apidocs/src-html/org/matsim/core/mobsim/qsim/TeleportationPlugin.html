001package org.matsim.core.mobsim.qsim;
002
003import com.google.inject.AbstractModule;
004import com.google.inject.Module;
005import org.matsim.core.config.Config;
006import org.matsim.core.mobsim.qsim.interfaces.MobsimEngine;
007
008import java.util.Collection;
009import java.util.Collections;
010
011public class TeleportationPlugin extends AbstractQSimPlugin {
012        public TeleportationPlugin(Config config) {
013                super(config);
014        }
015
016        @Override
017        public Collection<? extends Module> modules() {
018                return Collections.singletonList(new AbstractModule() {
019                        @Override
020                        protected void configure() {
021                                bind(DefaultTeleportationEngine.class).asEagerSingleton();
022                        }
023                });
024        }
025
026        @Override
027        public Collection<Class<? extends MobsimEngine>> engines() {
028                return Collections.singletonList( DefaultTeleportationEngine.class );
029        }
030}
