058        protected Link link;
059
060        // see method enterRequest for a detailed description of variable 'gap'
061        private LinkedList<Double> gap;
062
063        /**
064         * all roads, which are interested in entering the road, but wasn't allowed
065         * to do so yet
066         */
067        private LinkedList<Vehicle> interestedInEnteringRoad = new LinkedList<>();
068        private double timeOfLastEnteringVehicle = Double.MIN_VALUE;
069        protected double timeOfLastLeavingVehicle = Double.MIN_VALUE;
070
071        /**
072         * the inverseFlowCapacity is simply the inverse of the respective
073         * capacities meaning, and corresponds to the minimal time between two cars
074         * entering/leaving the road
075         */
076        private double inverseInFlowCapacity = 0;
077        protected double inverseOutFlowCapacity = 0;
078
079        /**
080         * this variable keeps track of the number of cars, which are not on the
081         * road, but which have been promised to enter the road (given a time in
082         * future, when they can enter the road)
083         */
084        protected int noOfCarsPromisedToEnterRoad = 0;
085
086        // maximum number of cars on the road at one time
087        private long maxNumberOfCarsOnRoad = 0;
088
089        // the time it takes for a gap to get to the back of the road
090        private double gapTravelTime = 0;
091
092        // the cars, which are currently on the road
093        protected LinkedList<Vehicle> carsOnTheRoad = new LinkedList<>();
094        /**
095         * for each of the cars in carsOnTheRoad, the earliest departure time from
096         * road is written here
097         */
098        protected LinkedList<Double> earliestDepartureTimeOfCar = new LinkedList<>();
099        /**
100         * when trying to enter a road, a deadlock prevention message is put into
101         * the queue this allows a car to enter the road, even if no space on it
102         */
103        private LinkedList<DeadlockPreventionMessage> deadlockPreventionMessages = new LinkedList<>();
104
105        public Road(Scheduler scheduler, Link link) {
106                super(scheduler);
107                this.link = link;
108
109                /*
110                 * calculate the maximum number of cars, which can be on the road at the
111                 * same time
112                 */
113                this.maxNumberOfCarsOnRoad = Math.round(link.getLength()
114                                * NetworkUtils.getNumberOfLanesAsInt(Time.UNDEFINED_TIME, link)
115                                * config.getStorageCapacityFactor() / config.getCarSize());
116
117                /**
118                 * it is assured here, that a road must have the space of at least one
119                 * car
120                 */
121                if (this.maxNumberOfCarsOnRoad == 0) {
122                        this.maxNumberOfCarsOnRoad = 1;
123                }
124
125                double maxInverseInFlowCapacity = 3600 / (config.getMinimumInFlowCapacity()
126                                * config.getFlowCapacityFactor() * NetworkUtils.getNumberOfLanesAsInt(Time.UNDEFINED_TIME, link));
127
128                this.inverseOutFlowCapacity = 1 / (((Link) link).getFlowCapacityPerSec() * config.getFlowCapacityFactor());
129
130                if (this.inverseOutFlowCapacity > maxInverseInFlowCapacity) {
131                        this.inverseInFlowCapacity = maxInverseInFlowCapacity;
132                } else {
133                        this.inverseInFlowCapacity = this.inverseOutFlowCapacity;
134                }
135
136                this.gapTravelTime = link.getLength() / config.getGapTravelSpeed();
137
138                // gap must be initialized to null because of the application logic
139                this.gap = null;
140        }
141
142        public void leaveRoad(Vehicle vehicle, double simTime) {
143                assert (this.carsOnTheRoad.getFirst() == vehicle);
144                assert (this.interestedInEnteringRoad.size()==this.deadlockPreventionMessages.size());
145
146                this.carsOnTheRoad.removeFirst();
147                this.earliestDepartureTimeOfCar.removeFirst();
148                this.timeOfLastLeavingVehicle = simTime;
149
150                /*
151                 * the next car waiting for entering the road should now be alloted a
152                 * time for entering the road
153                 */
154                if (this.interestedInEnteringRoad.size() > 0) {
155                        Vehicle nextVehicle = this.interestedInEnteringRoad.removeFirst();
156                        DeadlockPreventionMessage m = this.deadlockPreventionMessages.removeFirst();
157                        assert (m.vehicle == nextVehicle);
158                        this.scheduler.unschedule(m);
159
160                        double nextAvailableTimeForEnteringStreet = Math.max(this.timeOfLastEnteringVehicle
161                                        + this.inverseInFlowCapacity, simTime + this.gapTravelTime);
162
163                        this.noOfCarsPromisedToEnterRoad++;
164
165                        nextVehicle.scheduleEnterRoadMessage(nextAvailableTimeForEnteringStreet, this);
166                } else {
167                        if (this.gap != null) {
168
169                                /*
170                                 * as long as the road is not full once, there is no need to
171                                 * keep track of the gaps
172                                 */
173                                this.gap.add(simTime + this.gapTravelTime);
174
175                                /*
176                                 * if no one is interested in entering this road (precondition)
177                                 * and there are no cars on the road, then reset gap (this is
178                                 * required, for enterRequest to function properly)
179                                 */
180                                if (this.carsOnTheRoad.size() == 0) {
181                                        this.gap = null;
182                                }
183                        }
184                }
185
186                /*
187                 * tell the car behind the fist car (which is the first car now), when
188                 * it reaches the end of the read
189                 */
190                if (this.carsOnTheRoad.size() > 0) {
191                        Vehicle nextVehicle = this.carsOnTheRoad.getFirst();
192                        double nextAvailableTimeForLeavingStreet = Math.max(this.earliestDepartureTimeOfCar.getFirst(),
193                                        this.timeOfLastLeavingVehicle + this.inverseOutFlowCapacity);
194                        nextVehicle.scheduleEndRoadMessage(nextAvailableTimeForLeavingStreet, this);
195                }
196
197        }
198
199        public void enterRoad(Vehicle vehicle, double simTime) {
200                // calculate time, when the car reaches the end of the road
201                double nextAvailableTimeForLeavingStreet = simTime + this.link.getLength()
202                                / this.link.getFreespeed(simTime);
203
204                this.noOfCarsPromisedToEnterRoad--;
205                this.carsOnTheRoad.add(vehicle);
206
207                /*
208                 * needed to remove the following assertion because for deadlock
209                 * prevention there might be more cars on the road than its capacity
210                 * assert maxNumberOfCarsOnRoad >= carsOnTheRoad.size() : "There are
211                 * more cars on the road, than its capacity!";
212                 */
213                this.earliestDepartureTimeOfCar.add(nextAvailableTimeForLeavingStreet);
214
215                /*
216                 * if we are in the front of the queue, then we can just drive with free
217                 * speed to the front and have to have at least inverseFlowCapacity
218                 * time-distance to the previous car
219                 */
220                if (this.carsOnTheRoad.size() == 1) {
221                        nextAvailableTimeForLeavingStreet = Math.max(nextAvailableTimeForLeavingStreet,
222                                        this.timeOfLastLeavingVehicle + this.inverseOutFlowCapacity);
223                        vehicle.scheduleEndRoadMessage(nextAvailableTimeForLeavingStreet, this);
224//              } else { // empty else clause
225                        /*
226                         * this car is not the front car in the street queue when the cars
227                         * in front of the current car leave the street and this car becomes
228                         * the front car, it will be waken up.
229                         */
230                }
231
232        }
233
234        public void enterRequest(Vehicle vehicle, double simTime) {
235                assert (this.interestedInEnteringRoad.size()==this.deadlockPreventionMessages.size());
236                /*
237                 * assert maxNumberOfCarsOnRoad >= carsOnTheRoad.size() : "There are
238                 * more cars on the road, than its capacity!"; assert
239                 * maxNumberOfCarsOnRoad >= carsOnTheRoad.size() +
240                 * noOfCarsPromisedToEnterRoad : "You promised too many cars, that they
241                 * can enter the street!"; These asserts has been commented out for
242                 * deadlock prevention: If a car waits too long, it is allowed to enter
243                 * the road.
244                 */
245
246                // is there any space on the road (including promised entries?)
247                if (this.carsOnTheRoad.size() + this.noOfCarsPromisedToEnterRoad < this.maxNumberOfCarsOnRoad) {
248                        /*
249                         * - check, if the gap needs to be considered for entering the road -
250                         * we can find out, the time since when we have a free road for
251                         * entrance for sure:
252                         */
253
254                        // the gap queue will only be empty in the beginning
255                        double arrivalTimeOfGap = Double.MIN_VALUE;
256                        // if the road has been full recently then find out, when the next
257                        // gap arrives
258                        if ((this.gap != null) && (this.gap.size() > 0)) {
259                                arrivalTimeOfGap = this.gap.remove();
260                        }
261
262                        this.noOfCarsPromisedToEnterRoad++;
263                        double nextAvailableTimeForEnteringStreet = Math.max(Math.max(this.timeOfLastEnteringVehicle
264                                        + this.inverseInFlowCapacity, simTime), arrivalTimeOfGap);
265
266                        this.timeOfLastEnteringVehicle = nextAvailableTimeForEnteringStreet;
267                        vehicle.scheduleEnterRoadMessage(nextAvailableTimeForEnteringStreet, this);
268                } else {
269                        /*
270                         * - if the road was empty then create a new queue else empty the
271                         * old queue As long as the gap is null, the road is not full (and
272                         * there is no reason to keep track of the gaps => see leaveRoad)
273                         * But when the road gets full once, we need to start keeping track
274                         * of the gaps Once the road is empty again, gap is reset to null
275                         * (see leaveRoad).
276                         *
277                         * The gap variable in only needed for the situation, where the
278                         * street has been full recently, but the interestedInEnteringRoad
279                         * is empty and a new car arrives (or a few). So, if the street is
280                         * long, it takes time for the gap to come back.
281                         *
282                         * As long as interestedInEnteringRoad is not empty, newly generated
283                         * gaps get used by the new cars (see leaveRoad)
284                         */
285                        if (this.gap == null) {
286                                this.gap = new LinkedList<>();
287                        } else {
288                                this.gap.clear();
289                        }
290
291                        this.interestedInEnteringRoad.add(vehicle);
292
293                        /*
294                         * the first car interested in entering a road has to wait
295                         * 'stuckTime' the car behind has to wait an additional stuckTime
296                         * (this logic was adapted to adhere to the C++ implementation)
297                         */
298                        if (this.deadlockPreventionMessages.size() > 0) {
299                                this.deadlockPreventionMessages.add(vehicle.scheduleDeadlockPreventionMessage(
300                                                this.deadlockPreventionMessages.getLast().getMessageArrivalTime()
301                                                                + config.getSqueezeTime(), this));
302
303                        } else {
304                                this.deadlockPreventionMessages.add(vehicle.scheduleDeadlockPreventionMessage(simTime
305                                                + config.getSqueezeTime(), this));
306                        }
307
308                        assert (this.interestedInEnteringRoad.size()==this.deadlockPreventionMessages.size()) :this.interestedInEnteringRoad.size() + " - " + this.deadlockPreventionMessages.size();
309                }
310        }
311
312        public void giveBackPromisedSpaceToRoad() {
313                this.noOfCarsPromisedToEnterRoad--;
314        }
315
316        public void incrementPromisedToEnterRoad() {
317                this.noOfCarsPromisedToEnterRoad++;
318        }

320        public Link getLink() {
321                return this.link;
322        }

324        public void setTimeOfLastEnteringVehicle(double timeOfLastEnteringVehicle) {
325                this.timeOfLastEnteringVehicle = timeOfLastEnteringVehicle;
326        }

328        public void removeFirstDeadlockPreventionMessage(DeadlockPreventionMessage dpMessage) {
329
330                assert (this.deadlockPreventionMessages.getFirst() == dpMessage) : "Inconsitency in logic!!! => this should only be invoked from the handler of this message";
331                this.deadlockPreventionMessages.removeFirst();
332        }
333
334        public void removeFromInterestedInEnteringRoad() {
335                this.interestedInEnteringRoad.removeFirst();
336                assert (this.interestedInEnteringRoad.size()==this.deadlockPreventionMessages.size());
337        }
