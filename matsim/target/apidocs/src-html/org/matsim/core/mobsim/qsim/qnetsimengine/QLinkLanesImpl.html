053/**
054 * Please read the docu of QBufferItem, QLane, QLinkInternalI (arguably to be renamed into something
055 * like AbstractQLink) and QLinkImpl jointly. kai, nov'11
056 * 
057 * @author dstrippgen
058 * @author dgrether
059 * @author mrieser
060 * 
061 *         A QLinkLanes can consist of one or more QLanes, which may have the following layout
062 *         (Dashes stand for the borders of the QLink, equal signs (===) depict one lane, plus signs
063 *         (+) symbolize a decision point where one lane splits into several lanes) :
064 * 
065 *         <pre>
066 * ----------------
067 * ================
068 * ----------------
069 * </pre>
070 * 
071 *         <pre>
072 * ----------------
073 *          =======
074 * ========+
075 *          =======
076 * ----------------
077 * </pre>
078 * 
079 *         <pre>
080 * ----------------
081 *                                                              ========
082 * =======+========
083 *                                                              ========
084 * ----------------
085 * </pre>
086 * 
087 * 
088 *         The following layouts are not allowed:
089 * 
090 *         <pre>
091 * ----------------
092 * ================
093 * ================
094 * ----------------
095 * </pre>
096 * 
097 *         <pre>
098 * ----------------
099 * =======
100 *        +========
101 * =======
102 * ----------------
103 * </pre>
104 * 
105 *         Queue Model Link implementation with the following properties:
106 *         <ul>
107 *         <li>The queue behavior itself is simulated by one or more instances of QueueLane</li>
108 *         <li>Each QueueLink has at least one QueueLane. All QueueVehicles added to the QueueLink
109 *         are placed on this instance.</li>
110 *         <li>All QueueLane instances which are connected to the ToQueueNode are held in the
111 *         attribute toNodeQueueLanes</li>
112 *         <li>QueueLink is active as long as at least one of its QueueLanes is active.</li>
113 *         </ul>
114 */

188        private void initLaneQueues() {
189                Stack<QLaneI> stack = new Stack<>();
190                Map<Id<Lane>, QLaneI> queueByIdMap = new HashMap<>();
191                Map<Id<Lane>, Set<Id<Link>>> laneIdToLinksMap = new HashMap<>();
192                for (ModelLane lane : lanes) { // lanes is sorted downstream to upstream
193                        Id<Lane> laneId = lane.getLaneData().getId();
194                        // --
195                        // QLaneI queue = new QueueWithBuffer(this, new FIFOVehicleQ(), laneId,
196                        // lane.getLength(), noEffectiveLanes,
197                        // (lane.getLaneData().getCapacityVehiclesPerHour()/3600.0));
198
199                        QueueWithBuffer.Builder builder = new QueueWithBuffer.Builder( context ) ;
200                        builder.setVehicleQueue(new FIFOVehicleQ());
201                        builder.setLaneId(laneId);
202                        builder.setLength(lane.getLength());
203                        builder.setEffectiveNumberOfLanes(lane.getLaneData().getNumberOfRepresentedLanes());
204                        builder.setFlowCapacity_s(lane.getLaneData().getCapacityVehiclesPerHour() / 3600.);
205                        QLaneI queue = builder.createLane(this);
206                        // --
207                        queueByIdMap.put(laneId, queue);
208                        firstLaneQueue = queue;
209                        stack.push(queue);
210                        Set<Id<Link>> toLinkIds = new HashSet<>();
211
212                        if (lane.getToLanes() == null || lane.getToLanes().isEmpty()) { // lane is at the end of
213                                                                                                                                                        // link
214                                this.toNodeLaneQueues.add(queue);
215                                toLinkIds.addAll(lane.getLaneData().getToLinkIds());
216                                laneIdToLinksMap.put(laneId, toLinkIds);
217                        } else { // lane is within the link and has no connection to a node
218                                Map<Id<Link>, List<QLaneI>> toLinkIdDownstreamQueues = new LinkedHashMap<>();
219                                nextQueueToLinkCache.put(Id.create(((QueueWithBuffer) queue).getId(), Lane.class),
220                                                toLinkIdDownstreamQueues);
221                                for (ModelLane toLane : lane.getToLanes()) {
222                                        Set<Id<Link>> toLinks = laneIdToLinksMap.get(toLane.getLaneData().getId());
223                                        if (toLinks == null){ 
224                                                // nothing to do here
225                                                break;
226                                        }
227                                        if (!laneIdToLinksMap.containsKey(laneId)) {
228                                                laneIdToLinksMap.put(laneId, new HashSet<Id<Link>>());
229                                        }
230                                        laneIdToLinksMap.get(laneId).addAll(toLinks);
231                                        for (Id<Link> toLinkId : toLinks) {
232                                                List<QLaneI> downstreamQueues = toLinkIdDownstreamQueues.get(toLinkId);
233                                                if (downstreamQueues == null) {
234                                                        downstreamQueues = new ArrayList<>();
235                                                        toLinkIdDownstreamQueues.put(toLinkId, downstreamQueues);
236                                                }
237                                                downstreamQueues.add(queueByIdMap.get(toLane.getLaneData().getId()));
238                                        }
239                                }
240                        }
241//                      queue.changeSpeedMetersPerSecond( this.getLink().getFreespeed() );
242                        
243                }
244                // reverse the order in the linked map, i.e. upstream to downstream
245                while (!stack.isEmpty()) {
246                        QLaneI queue = stack.pop();
247                        this.laneQueues.put(((QueueWithBuffer) queue).getId(), queue);
248                }
249        }

264        @Override
265        public boolean doSimStep() {
266                double now = context.getSimTimer().getTimeOfDay() ;
267                
268                boolean lanesActive = false;
269                boolean movedWaitToRoad = false;
270                if ( context.qsimConfig.isInsertingWaitingVehiclesBeforeDrivingVehicles() ) {
271                    //TODO
272                    //Because moveBufferToNextLane() (called from moveLanes()) is kind of "moveInternalNodes()",
273                    //it should be executed before moveWaitToRoad() to keep the sequence fully consistent.
274                    //The sequence is broken only if isInsertingWaitingVehiclesBeforeDrivingVehicles==true.
275                    //Currently, the buffer of the accepting lane gets emptied after moveWaitToRoad(),
276                    //which gives preference to already driving vehicles 
277                    //michalm, jan'17
278                        this.moveWaitToRoad(now);
279                        this.getTransitQLink().handleTransitVehiclesInStopQueue(now);
280                        lanesActive = this.moveLanes();
281                } else {
282                        this.getTransitQLink().handleTransitVehiclesInStopQueue(now);
283                        lanesActive = this.moveLanes();
284                        movedWaitToRoad = this.moveWaitToRoad(now);
285                }
286                this.setActive(lanesActive || movedWaitToRoad || (!this.getWaitingList().isEmpty())
287                                || !this.getTransitQLink().getTransitVehicleStopQueue().isEmpty());
288                return this.isActive();
289        }

291        private boolean moveLanes() {
292                boolean activeLane = false;
293                for (QLaneI lane : this.laneQueues.values()) {
294                        // (go through all lanes)
295                        
296                        /* part A */
297                        if (!this.toNodeLaneQueues.contains(lane)) {
298                                // (so it HAS a link-internal next lane)
299                                
300                                // move vehicles from the lane buffer to the next lane
301                                this.moveBufferToNextLane(lane);
302                        } else {
303                                // move vehicles from the lane buffer to the link buffer, i.e. prepare moving them
304                                // to the next link
305//                              ((QueueWithBuffer) queue).moveQueueToBuffer();
306                                // yy just commented out the above line.  Can't say why it might needed; doSimStep also calls moveQueueToBuffer.
307                                // Tests run ok, but it may have capacity ramifications outside tests.  kai, mar'16
308                        }
309                        /* end of part A */
310                }
311                
312        for (QLaneI lane : this.laneQueues.values()) {
313            lane.initBeforeSimStep();
314        }
315        
316                for (QLaneI lane : this.laneQueues.values()) {
317                        // (go through all lanes)
318                
319                        /* part B */
320                        // move vehicles to the lane buffer if they have reached their earliest lane exit time
321                        lane.doSimStep();
322                        /* end of part B */
323                        
324                        activeLane = activeLane || lane.isActive();
325                        
326                        /*
327                         * Remark: The order of part A and B influences the travel time on lanes.
328                         * 
329                         * Before Jul'15 order B, A was used, such that travel time on lanes was practically
330                         * rounded down. This has the unintended effect that introducing lanes on a link may
331                         * decrease its travel time: Dividing the link into sufficient many lanes (each one
332                         * shorter than the number of meters that an agent may travel in 1 second) reduces the
333                         * link travel time to 1 second.
334                         * 
335                         * Order A, B produces the same behavior for lanes as for links. I.e. up to one
336                         * additional second may occur for every lane, because travel times are now practically
337                         * rounded up.
338                         * 
339                         * Theresa, Jul'15
340                         */
341                }
342                return activeLane;
343        }

345        private void moveBufferToNextLane(QLaneI qlane) {
346                QVehicle veh;
347                while (!qlane.isNotOfferingVehicle()) {
348                        veh = qlane.getFirstVehicle();
349                        Id<Link> toLinkId = veh.getDriver().chooseNextLinkId();
350                        QLaneI nextQueue = this.chooseNextLane(qlane, toLinkId);
351                        if (nextQueue != null) {
352                                if (nextQueue.isAcceptingFromUpstream()) {
353                                        ((QueueWithBuffer) qlane).popFirstVehicle();
354//                                      context .getEventsManager() .processEvent(
355//                                                                      new LaneLeaveEvent(now, veh.getId(), this.getLink()
356//                                                                                      .getId(), ((QueueWithBuffer) qlane).getId()));
357                                        nextQueue.addFromUpstream(veh);
358//                                      context .getEventsManager() .processEvent(
359//                                                                      new LaneEnterEvent(now, veh.getId(), this.getLink()
360//                                                                                      .getId(), ((QueueWithBuffer) nextQueue).getId()));
361                                } else {
362                                        break;
363                                }
364                        } else {
365                                StringBuilder b = new StringBuilder();
366                                b.append("Person Id: ").append(veh.getDriver().getId());
367                                b.append(" is on Lane Id ").append(((QueueWithBuffer) qlane).getId());
368                                b.append(" on Link Id ").append(this.getLink().getId());
369                                b.append(" and wants to drive to Link Id ").append(toLinkId);
370                                b.append(" but there is no Lane leading to that Link!");
371                                log.error(b.toString());
372                                throw new IllegalStateException(b.toString());
373                        }
374                }
375        }

377        private QLaneI chooseNextLane(QLaneI queue, Id<Link> toLinkId) {
378                List<QLaneI> toQueues = this.nextQueueToLinkCache.get(queue.getId()).get(toLinkId);
379                QLaneI retLane = toQueues.get(0);
380                if (toQueues.size() == 1) {
381                        return retLane;
382                }
383                // else chose lane by storage cap
384                for (int i = 1; i < toQueues.size(); i++) {
385                        QLaneI toQueue = toQueues.get(i);
386                        if (((QueueWithBuffer) toQueue).getLoadIndicator() < ((QueueWithBuffer) retLane).getLoadIndicator()) {
387                                retLane = toQueue;
388                        }
389                }
390                return retLane;
391        }

393        /**
394         * Move as many waiting cars to the link as it is possible
395         * 
396         * @param now
397         *            the current time
398         * @return true if at least one vehicle is moved to the buffer of this lane
399         */
400        private boolean moveWaitToRoad(final double now) {
401                boolean movedWaitToRoad = false;
402                while (!getWaitingList().isEmpty()) {
403                    if (!firstLaneQueue.isAcceptingFromWait(this.getWaitingList().peek())) {
404                        return movedWaitToRoad;
405                    }
406                    QVehicle veh = this.getWaitingList().poll();
407
408                        movedWaitToRoad = true;
409                        context .getEventsManager() .processEvent(
410                                                        new VehicleEntersTrafficEvent(now, veh.getDriver().getId(),
411                                                                        this.getLink().getId(), veh.getId(), veh.getDriver().getMode(), 1.0));
412
413                        if (this.getTransitQLink().addTransitToStopQueue(now, veh, this.getLink().getId())) {
414                                continue;
415                        }
416
417                        // if (veh.getDriver().chooseNextLinkId() == null) {
418                        if (veh.getDriver().isWantingToArriveOnCurrentLink()) {
419                                // If the driver wants to stop on this link, give them a special treatment.
420                                // addFromWait doesn't work here, because after that, they cannot stop anymore.
421                                this.firstLaneQueue.addTransitSlightlyUpstreamOfStop(veh);
422                                continue;
423                        }
424
425                        this.firstLaneQueue.addFromWait(veh);
426                }
427                return movedWaitToRoad;
428        }

453        @Override
454        public QVehicle getVehicle(Id<Vehicle> vehicleId) {
455                QVehicle ret = super.getVehicle(vehicleId);
456                if (ret != null) {
457                        return ret;
458                }
459                for (QVehicle veh : this.getWaitingList()) {
460                        if (veh.getId().equals(vehicleId))
461                                return veh;
462                }
463                for (QLaneI lane : this.laneQueues.values()) {
464                        ret = lane.getVehicle(vehicleId);
465                        if (ret != null) {
466                                return ret;
467                        }
468                }
469                return ret;
470        }

541                        if (nodeOffset != 0.0) {

545                                CoordinateTransformation transformation = new IdentityTransformation();
546                                visLink = visModelBuilder.createVisLinkLanes(transformation, QLinkLanesImpl.this, nodeOffset, lanes);
547                                visModelBuilder.recalculatePositions(visLink, context.linkWidthCalculator);
548                        }
549                }

551                @Override
552                public Collection<AgentSnapshotInfo> addAgentSnapshotInfo(
553                                final Collection<AgentSnapshotInfo> positions) {
554                        
555                        double now = context.getSimTimer().getTimeOfDay() ;
556
557
558                        if (visLink != null) {
559                                for (QLaneI ql : QLinkLanesImpl.this.laneQueues.values()) {
560                                        VisLane otfLane = visLink.getLaneData().get(
561                                                        ((QueueWithBuffer) ql).getId().toString());
562                                        ((QueueWithBuffer.VisDataImpl) ql.getVisData()).setVisInfo(
563                                                        otfLane.getStartCoord(), otfLane.getEndCoord());
564                                }
565                        }
566
567                        for (QLaneI road : QLinkLanesImpl.this.getQueueLanes().values()) {
568                                road.getVisData().addAgentSnapshotInfo(positions, now);
569                        }
570
571                        int cnt2 = 10;
572
573                        // treat vehicles from transit stops
574                        cnt2 = context.snapshotInfoBuilder.positionVehiclesFromTransitStop(positions, getLink(),
575                                        getTransitQLink().getTransitVehicleStopQueue(), cnt2);
576                        // treat vehicles from waiting list:
577                        context.snapshotInfoBuilder.positionVehiclesFromWaitingList(positions,
578                                        QLinkLanesImpl.this.getLink(), cnt2, QLinkLanesImpl.this.getWaitingList());
579                        cnt2 = QLinkLanesImpl.this.getWaitingList().size();
580                        context.snapshotInfoBuilder.positionAgentsInActivities(positions, QLinkLanesImpl.this.getLink(),
581                                        QLinkLanesImpl.this.getAdditionalAgentsOnLink(), cnt2);
582
583                        return positions;
584                }
585        }
