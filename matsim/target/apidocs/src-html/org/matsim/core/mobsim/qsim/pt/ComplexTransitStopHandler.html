020package org.matsim.core.mobsim.qsim.pt;
021
022import java.util.List;
023
024import org.apache.log4j.Logger;
025import org.matsim.core.mobsim.qsim.interfaces.MobsimVehicle;
026import org.matsim.pt.transitSchedule.api.TransitStopFacility;
027import org.matsim.vehicles.Vehicle;
028import org.matsim.vehicles.VehicleType;

030/**
031 * Continuous queue model like transit stop handler using either serial or parallel doors operation mode.
032 * 
033 * 
034 * @author aneumann
035 */
036public class ComplexTransitStopHandler implements TransitStopHandler {
037        
038        private final static Logger log = Logger.getLogger(ComplexTransitStopHandler.class);
039
040        private boolean doorsOpen = false;
041        private double passengersLeavingTimeFraction = 0.0;
042        private double passengersEnteringTimeFraction = 0.0;
043
044        private final double personEntersTime;
045        private final double personLeavesTime;
046        private final VehicleType.DoorOperationMode doorOperationMode;
047        
048        // TODO make it dynamic
049        private static final double openDoorsDuration = 1.0;
050        private static final double closeDoorsDuration = 1.0;
051
052        /*package*/ ComplexTransitStopHandler(Vehicle vehicle) {
053                this.personEntersTime = vehicle.getType().getAccessTime();
054                this.personLeavesTime = vehicle.getType().getEgressTime();
055                this.doorOperationMode = vehicle.getType().getDoorOperationMode();
056        }
057
058        @Override
059        public double handleTransitStop(TransitStopFacility stop, double now, List<PTPassengerAgent> leavingPassengers,
060                        List<PTPassengerAgent> enteringPassengers, PassengerAccessEgress handler, MobsimVehicle vehicle) {
061                
062                if(this.doorOperationMode == VehicleType.DoorOperationMode.parallel){                   
063                        return handleParallelStop(stop, now, leavingPassengers, enteringPassengers, handler, vehicle);                  
064                } else if (this.doorOperationMode == VehicleType.DoorOperationMode.serial){
065                        return handleSerialStop(stop, now, leavingPassengers, enteringPassengers, handler, vehicle);
066                } else {
067                        log.info("Unimplemented door operation mode " + this.doorOperationMode + " set. Using parralel mode as default.");
068                        return handleParallelStop(stop, now, leavingPassengers, enteringPassengers, handler, vehicle);
069                }               
070        }
071        
072        private double handleSerialStop(TransitStopFacility stop, double now, List<PTPassengerAgent> leavingPassengers, 
073                        List<PTPassengerAgent> enteringPassengers, PassengerAccessEgress handler, MobsimVehicle vehicle){
074                double stopTime = 0.0;
075
076                int cntEgress = leavingPassengers.size();
077                int cntAccess = enteringPassengers.size();
078
079                if (!this.doorsOpen) {
080                        // doors are closed
081
082                        if ((cntAccess > 0) || (cntEgress > 0)) {
083                                // case doors are shut, but passengers want to leave or enter
084                                // the veh
085                                this.doorsOpen = true;
086                                stopTime = openDoorsDuration; // Time to open doors
087                        } else {
088                                // case nobody wants to leave or enter the veh
089                                stopTime = 0.0;
090                        }
091
092                } else {
093                        // doors are already open
094
095                        if ((cntAccess > 0) || (cntEgress > 0)) {
096                                // somebody wants to leave or enter the veh
097                                
098                                if (cntEgress > 0) {
099
100                                        if (this.passengersLeavingTimeFraction < 1.0) {
101                                                // next passenger can leave the veh
102
103                                                while (this.passengersLeavingTimeFraction < 1.0) {
104                                                        if (leavingPassengers.size() == 0) {
105                                                                break;
106                                                        }
107
108                                                        if(handler.handlePassengerLeaving(leavingPassengers.get(0), vehicle, stop.getLinkId(), now)){
109                                                                leavingPassengers.remove(0);
110                                                                this.passengersLeavingTimeFraction += personLeavesTime;
111                                                        } else {
112                                                                break;
113                                                        }
114
115                                                }
116
117                                                this.passengersLeavingTimeFraction -= 1.0;
118                                                stopTime = 1.0;
119
120                                        } else {
121                                                // still time needed to allow next passenger to leave
122                                                this.passengersLeavingTimeFraction -= 1.0;
123                                                stopTime = 1.0;
124                                        }
125
126                                } else {
127                                        this.passengersLeavingTimeFraction -= 1.0;
128                                        this.passengersLeavingTimeFraction = Math.max(0, this.passengersLeavingTimeFraction);
129                                        
130                                        if (cntAccess > 0) {
131
132                                                if (this.passengersEnteringTimeFraction < 1.0) {
133
134                                                        // next passenger can enter the veh
135
136                                                        while (this.passengersEnteringTimeFraction < 1.0) {
137                                                                if (enteringPassengers.size() == 0) {
138                                                                        break;
139                                                                }
140
141                                                                if(handler.handlePassengerEntering(enteringPassengers.get(0), vehicle, stop.getId(), now)){
142                                                                        enteringPassengers.remove(0);
143                                                                        this.passengersEnteringTimeFraction += personEntersTime;
144                                                                } else {
145                                                                        break;
146                                                                }
147
148                                                        }
149
150                                                        this.passengersEnteringTimeFraction -= 1.0;
151                                                        stopTime = 1.0;
152
153                                                } else {
154                                                        // still time needed to allow next passenger to enter
155                                                        this.passengersEnteringTimeFraction -= 1.0;
156                                                        stopTime = 1.0;
157                                                }
158
159                                        } else {
160                                                this.passengersEnteringTimeFraction -= 1.0;
161                                                this.passengersEnteringTimeFraction = Math.max(0, this.passengersEnteringTimeFraction);
162                                        }
163                                }                               
164
165                        } else {
166
167                                // nobody left to handle
168
169                                if (this.passengersEnteringTimeFraction < 1.0 && this.passengersLeavingTimeFraction < 1.0) {
170                                        // every passenger entered or left the veh so close and
171                                        // leave
172
173                                        this.doorsOpen = false;
174                                        this.passengersEnteringTimeFraction = 0.0;
175                                        this.passengersLeavingTimeFraction = 0.0;
176                                        stopTime = closeDoorsDuration; // Time to shut the doors
177                                }
178
179                                // somebody is still leaving or entering the veh so wait again
180
181                                if (this.passengersEnteringTimeFraction >= 1) {
182                                        this.passengersEnteringTimeFraction -= 1.0;
183                                        stopTime = 1.0;
184                                }
185
186                                if (this.passengersLeavingTimeFraction >= 1) {
187                                        this.passengersLeavingTimeFraction -= 1.0;
188                                        stopTime = 1.0;
189                                }
190
191                        }
192
193                }
194
195                return stopTime;
196        }
