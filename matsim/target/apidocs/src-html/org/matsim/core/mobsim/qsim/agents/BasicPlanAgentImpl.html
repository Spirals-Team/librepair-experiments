063        public BasicPlanAgentImpl(Plan plan2, Scenario scenario, EventsManager events, MobsimTimer simTimer) {
064
065                this.plan = PopulationUtils.unmodifiablePlan(plan2) ;
066                // yy MZ suggests, and I agree, to always give the agent a full plan, and consume that plan as the agent goes.  kai, nov'14
067
068                this.scenario = scenario ;
069                this.events = events ;
070                this.simTimer = simTimer ;
071                List<PlanElement> planElements = this.getCurrentPlan().getPlanElements();
072                if (planElements.size() > 0) {
073                        Activity firstAct = (Activity) planElements.get(0);                             
074//                      this.setCurrentLinkId( firstAct.getLinkId() ) ;
075                        final Id<Link> linkId = PopulationUtils.computeLinkIdFromActivity(firstAct, scenario.getActivityFacilities(), scenario.getConfig() );
076                        Gbl.assertIf( linkId!=null );
077                        this.setCurrentLinkId( linkId );
078                        this.setState(MobsimAgent.State.ACTIVITY) ;
079                        calculateAndSetDepartureTime(firstAct);
080                }
081}

083        @Override
084        public final void endLegAndComputeNextState(final double now) {
085                this.getEvents().processEvent(new PersonArrivalEvent( now, this.getId(), this.getDestinationLinkId(), getCurrentLeg().getMode()));
086                if( (!(this.getCurrentLinkId() == null && this.getDestinationLinkId() == null)) 
087                                && !this.getCurrentLinkId().equals(this.getDestinationLinkId())) {
088                        log.error("The agent " + this.getPerson().getId() + " has destination link " + this.getDestinationLinkId()
089                                        + ", but arrived on link " + this.getCurrentLinkId() + ". Setting agent state to ABORT.");
090                        this.setState(MobsimAgent.State.ABORT) ;
091                } else {
092                        // note that when we are here we don't know if next is another leg, or an activity  Therefore, we go to a general method:
093                        advancePlan(now) ;
094                }
095                
096                this.currentLinkIndex = 0 ;
097        }

110        private void advancePlan(double now) {
111                this.currentPlanElementIndex++ ;
112        
113                // check if plan has run dry:
114                if ( this.getCurrentPlanElementIndex() >= this.getCurrentPlan().getPlanElements().size() ) {
115                        log.error("plan of agent with id = " + this.getId() + " has run empty.  Setting agent state to ABORT (but continuing the mobsim).") ;
116                        this.setState(MobsimAgent.State.ABORT) ;
117                        return;
118                }
119        
120                PlanElement pe = this.getCurrentPlanElement() ;
121                if (pe instanceof Activity) {
122                        Activity act = (Activity) pe;
123                        initializeActivity(act, now);
124                } else if (pe instanceof Leg) {
125                        Leg leg = (Leg) pe;
126                        initializeLeg(leg);
127                } else {
128                        throw new RuntimeException("Unknown PlanElement of type: " + pe.getClass().getName());
129                }
130        }
131
132        private void initializeLeg(Leg leg) {
133                this.setState(MobsimAgent.State.LEG) ;          
134                this.currentLinkIndex = 0 ;
135                if (leg.getRoute() == null) {
136                        log.error("The agent " + this.getPerson().getId() + " has no route in its leg.  Setting agent state to ABORT.");
137                        if ( noRouteWrnCnt < 1 ) {
138                                log.info( "(Route is needed inside Leg even if you want teleportation since Route carries the start/endLinkId info.)") ;
139                                noRouteWrnCnt++ ;
140                        }
141                        this.setState(MobsimAgent.State.ABORT) ;
142//                      throw new RuntimeException("no route in leg") ;
143                } 
144        }
145
146        private void initializeActivity(Activity act, double now) {
147                this.setState(MobsimAgent.State.ACTIVITY) ;
148                this.getEvents().processEvent( new ActivityStartEvent(now, this.getId(), this.getCurrentLinkId(), act.getFacilityId(), act.getType()));
149                calculateAndSetDepartureTime(act);
150                getModifiablePlan(); // this is necessary to make the plan modifiable, so that setting the start time (next line) is actually feasible. kai/mz, oct'16
151                ((Activity) getCurrentPlanElement()).setStartTime(now);
152        }

154        /**
155         * If this method is called to update a changed ActivityEndTime please
156         * ensure that the ActivityEndsList in the {@link QSim} is also updated.
157         */
158        private final void calculateAndSetDepartureTime(Activity act) {
159                PlansConfigGroup.ActivityDurationInterpretation activityDurationInterpretation = this.getScenario().getConfig().plans().getActivityDurationInterpretation();
160                double now = this.getSimTimer().getTimeOfDay() ;
161                double departure = ActivityDurationUtils.calculateDepartureTime(act, now, activityDurationInterpretation);
162        
163                if ( this.getCurrentPlanElementIndex() == this.getCurrentPlan().getPlanElements().size()-1 ) {
164                        if ( finalActHasDpTimeWrnCnt < 1 && departure!=Double.POSITIVE_INFINITY ) {
165                                log.error( "last activity of person driver agent id " + this.getId() + " has end time < infty; setting it to infty") ;
166                                log.error( Gbl.ONLYONCE ) ;
167                                finalActHasDpTimeWrnCnt++ ;
168                        }
169                        departure = Double.POSITIVE_INFINITY ;
170                }
171        
172                this.activityEndTime = departure ;
173        }

176        public final void endActivityAndComputeNextState(final double now) {
177                Activity act = (Activity) this.getCurrentPlanElement() ;
178                this.getEvents().processEvent( new ActivityEndEvent(now, this.getPerson().getId(), this.currentLinkId, act.getFacilityId(), act.getType()));
179        
180                // note that when we are here we don't know if next is another leg, or an activity  Therefore, we go to a general method:
181                advancePlan(now);
182        }

185                if ( this.getCurrentPlanElement() instanceof Activity ) {
186                        Activity act = (Activity) this.getCurrentPlanElement() ;
187                        this.calculateAndSetDepartureTime(act);
188                }
189        }

194        /**
195         * Returns a modifiable Plan for use by WithinDayAgentUtils in this package.
196         * This agent retains the copied plan and forgets the original one.  However, the original plan remains in the population file
197         * (and will be scored).  This is deliberate behavior!
198         */
199        @Override
200        public final Plan getModifiablePlan() {
201                // yy MZ suggests, and I agree, to always give the agent a full plan, and consume that plan as the agent goes.  kai, nov'14
202                if (firstTimeToGetModifiablePlan) {
203                        firstTimeToGetModifiablePlan = false ;
204                        Plan newPlan = PopulationUtils.createPlan(this.getCurrentPlan().getPerson());
205                        PopulationUtils.copyFromTo(this.getCurrentPlan(), newPlan);
206                        this.plan = newPlan;
207                }
208                return this.getCurrentPlan();
209        }
210        @Override
211        public final Id<Vehicle> getPlannedVehicleId() {
212                NetworkRoute route = (NetworkRoute) this.getCurrentLeg().getRoute(); // if casts fail: illegal state.
213                if (route.getVehicleId() != null) {
214                        return route.getVehicleId();
215                }
216                Gbl.assertIf( scenario.getConfig().qsim().getUsePersonIdForMissingVehicleId() );
217                return Id.create(this.getId(), Vehicle.class); // we still assume the vehicleId is the agentId if no vehicleId is given.
218        }
219        @Override
220        public final String getMode() {
221                if( this.getCurrentPlanElementIndex() >= this.getCurrentPlan().getPlanElements().size() ) {
222                        // just having run out of plan elements it not an argument for not being able to answer the "mode?" question,
223                        // thus we answer with "null".  This will likely result in an "abort". kai, nov'14
224                        return null ;
225                }
226                PlanElement currentPlanElement = this.getCurrentPlanElement();
227                if (!(currentPlanElement instanceof Leg)) {
228                        return null;
229                }
230                return ((Leg) currentPlanElement).getMode() ;
231        }
232        @Override
233        public final Double getExpectedTravelTime() {
234                PlanElement currentPlanElement = this.getCurrentPlanElement();
235                if (!(currentPlanElement instanceof Leg)) {
236                        return null;
237                }
238                final double travelTimeFromRoute = ((Leg) currentPlanElement).getRoute().getTravelTime();
239                if (  travelTimeFromRoute != Time.UNDEFINED_TIME ) {
240                        return travelTimeFromRoute ;
241                } else if ( ((Leg) currentPlanElement).getTravelTime() != Time.UNDEFINED_TIME ) {
242                        return ((Leg) currentPlanElement).getTravelTime()  ;
243                } else {
244                        return null ;
245                }
246        }

377        @Override
378        public Facility<? extends Facility<?>> getCurrentFacility() {
379                PlanElement pe = this.getCurrentPlanElement() ;
380                Activity activity ;
381                if ( pe instanceof Activity ) {
382                        activity = (Activity) pe;
383                } else if ( pe instanceof Leg ) {
384                        activity = this.getPreviousActivity() ;
385                } else {
386                        throw new RuntimeException("unexpected type of PlanElement") ;
387                }
388                ActivityFacility fac = this.scenario.getActivityFacilities().getFacilities().get( activity.getFacilityId() ) ;
389                if ( fac != null ) {
390                        System.out.println("facility=" + fac );
391                        return fac ;
392                } else {
393                        return new ActivityWrapperFacility( activity ) ; 
394                }
395        }

397        @Override
398        public Facility<? extends Facility<?>> getDestinationFacility() {
399                PlanElement pe = this.getCurrentPlanElement() ;
400                if ( pe instanceof Leg ) {
401                        Activity activity = this.getNextActivity() ;
402                        ActivityFacility fac = this.scenario.getActivityFacilities().getFacilities().get( activity.getFacilityId() ) ;
403                        if ( fac != null ) {
404                                System.out.println("facility=" + fac );
405                                return fac ;
406                        } else {
407                                return new ActivityWrapperFacility( activity ) ; 
408                        }
409                        // the above assumes alternating acts/legs.  I start having the suspicion that we should revoke our decision to give that up.
410                        // If not, one will have to use TripUtils to find the preceeding activity ... but things get more difficult.  Preferably, the
411                        // factility should then sit in the leg (since there it is used for routing).  kai, dec'15
412                } else if ( pe instanceof Activity ) {
413                        return null ;
414                }
415                throw new RuntimeException("unexpected type of PlanElement") ;
416        }
