066        public QVehicle(final Vehicle basicVehicle) {
067                this.id = basicVehicle.getId();
068                this.vehicle = basicVehicle;
069                this.passengers = new ArrayList<>();
070
071                VehicleCapacity capacity = basicVehicle.getType().getCapacity();
072                if (capacity == null) {
073                        this.passengerCapacity = 4;
074                        if (warnCount < 10) {
075                                log.warn("No VehicleCapacity (= maximum number of passengers) set in Vehicle. "
076                                                + "Using default value of 4.  This is only a problem if you need vehicles with different "
077                                                + "capacities, e.g. for minibuses.");
078                                warnCount++;
079                                if ( warnCount == 10 ) {
080                                        log.warn( Gbl.FUTURE_SUPPRESSED ) ;
081                                }
082                        }
083                } else {
084                        // do *not* subtract one for the driver! Most pt vehicles define the capacity without the driver.
085                        // for private cars, think about if we should subtract one from the capacity if the driver is set?
086                        // But if we do, change the number of seats of the default vehicle from 4 to 5.
087                        this.passengerCapacity = capacity.getSeats() +
088                                        (capacity.getStandingRoom() == null ? 0 : capacity.getStandingRoom());
089                }
090        }

140
141        @Override
142        public MobsimDriverAgent getDriver() {
143                if ( this.driver instanceof MobsimDriverAgent ) {
144                        return (MobsimDriverAgent) this.driver;
145                } else if ( this.driver==null ) {
146                        return null ;
147                } else {
148                        throw new RuntimeException( "error (downstream methods need to be made to accept DriverAgent)") ;
149                }
150        }
151
152        public void setDriver(final DriverAgent driver) {
153                if (driver != null) {
154                        if (this.driver != null && !this.driver.getId().equals(driver.getId())) {
155                                throw new RuntimeException( "A driver (" + this.driver.getId() +") " +
156                                                "is already set in vehicle " + this.getId() + ". " +
157                                                "Setting agent " + driver.getId().toString() + " is not possible!");
158                        }
159                }
160                // TODO: To make this check possible, we would need something like removeDriver().
161//              else {
162//                      throw new RuntimeException( "Driver to be set in vehicle " + this.getId() +
163//                                      " is null!");
164//              }
165
166                this.driver = driver;
167        }

169        @Override
170        public Id<Vehicle> getId() {
171                return this.id;
172        }
173
174    @Override
175    public double getSizeInEquivalents() {
176        return vehicle.getType().getPcuEquivalents();
177    }
178
179    public double getFlowCapacityConsumptionInEquivalents() {
180        return vehicle.getType().getPcuEquivalents() / vehicle.getType().getFlowEfficiencyFactor();
181    }
182
183        @Override
184        public Vehicle getVehicle() {
185                return this.vehicle;
186        }
187
188        @Override
189        public String toString() {
190                return "Vehicle Id " + getId() + ", driven by (personId) " + this.driver.getId()
191                                + ", on link " + this.currentLink.getId();
192        }
193
194        public double getMaximumVelocity() {
195                return vehicle.getType().getMaximumVelocity();
196        }
197
198        @Override
199        public Collection<? extends PassengerAgent> getPassengers() {
200                return Collections.unmodifiableCollection(this.passengers);
201        }
202
203        @Override
204        public boolean addPassenger(PassengerAgent passenger) {
205                if (this.passengers.size() < this.passengerCapacity) {
206                        return this.passengers.add(passenger);
207                }
208                return false;
209        }
210
211        @Override
212        public boolean removePassenger(PassengerAgent passenger) {
213                return this.passengers.remove(passenger);
214        }
215

217        public int getPassengerCapacity() {
218                return this.passengerCapacity;
219        }
