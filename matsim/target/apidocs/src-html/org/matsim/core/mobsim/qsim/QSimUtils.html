074        public static QSim createQSim( final Scenario scenario, final EventsManager eventsManager,
075                                                                   final Collection<AbstractModule> overrides, final Collection<AbstractQSimPlugin> plugins ) {
076                // yy I patched the following together from the two (original ones) above.  But I cannot say why in the first case it is
077                // sufficient to override the module, while in the second case the output from the AbstractModule.override(...) method needs
078                // to be taken.  Maybe it does not matter, maybe it does, it would be nice to be certain.  kai, feb'18
079                
080                final StandaloneQSimModule module = new StandaloneQSimModule(scenario, eventsManager);
081                for ( AbstractModule override : overrides ) {
082                        org.matsim.core.controler.AbstractModule.override(Collections.singleton(module), override) ;
083                }
084                final AbstractModule override = AbstractModule.override(Collections.singleton(module),
085                                new AbstractModule() {
086                                        @Override
087                                        public void install() {
088                                                bind(new TypeLiteral<Collection<AbstractQSimPlugin>>() {
089                                                }).toInstance(plugins);
090                                        }
091                                });
092                Injector injector = org.matsim.core.controler.Injector.createInjector(scenario.getConfig(), override);
093                return (QSim) injector.getInstance(Mobsim.class);
094        }
095
096        private static class StandaloneQSimModule extends org.matsim.core.controler.AbstractModule {
097                private final Scenario scenario;
098                private final EventsManager eventsManager;
099
100                public StandaloneQSimModule(Scenario scenario, EventsManager eventsManager) {
101                        this.scenario = scenario;
102                        this.eventsManager = eventsManager;
103                }
104
105                @Override
106                public void install() {
107                        install(new ScenarioByInstanceModule(scenario));
108                        bind(EventsManager.class).toInstance(eventsManager);
109                        install(new QSimModule());
110                }
111        }
