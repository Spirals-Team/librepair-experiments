021package org.matsim.core.mobsim.qsim.interfaces;
022
023import java.util.Collection;
024
025import org.matsim.api.core.v01.network.Link;
026import org.matsim.core.mobsim.framework.PassengerAgent;
027import org.matsim.vis.snapshotwriters.VisVehicle;
028
029/**
030 * @author nagel
031 *
032 */
033public interface MobsimVehicle extends VisVehicle{
034        
035        Link getCurrentLink();
036        
037        double getSizeInEquivalents();
038        
039        /**
040         * Adds a passenger to this vehicle.
041         *
042         * @return <tt>true</tt> when the agent was added as a passenger (as per the general contract of the Collection.add method).
043         */
044        boolean addPassenger(final PassengerAgent passenger);
045
046        /**
047         * Removes the passenger from this vehicle.
048         *
049         * @return <tt>true</tt> when the agent was removed as a passenger, <tt>false</tt> if the agent was not a passenger of this vehicle or could not be removed for other reasons
050         */
051        boolean removePassenger(final PassengerAgent passenger);
052
053        /**
054         * @return an immutable Collection of all passengers in this vehicle.  Should <i> not </i> include the driver
055         */
056        Collection<? extends PassengerAgent> getPassengers();
057
058        /**
059         * @return number of passengers this vehicle can transport
060         */
061        int getPassengerCapacity();
062
063}
