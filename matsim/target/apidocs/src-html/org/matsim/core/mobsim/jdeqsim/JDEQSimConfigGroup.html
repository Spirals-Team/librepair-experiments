020package org.matsim.core.mobsim.jdeqsim;
021
022import org.matsim.core.config.ReflectiveConfigGroup;
023import org.matsim.core.utils.misc.Time;
024
025/**
026 * The micro-simulation parameters.
027 *
028 * @author rashid_waraich
029 */
030public class JDEQSimConfigGroup extends ReflectiveConfigGroup {
031
032        public final static String NAME = "JDEQSim";
033
034        // CONSTANTS
035        public static final String START_LEG = "start leg";
036        public static final String END_LEG = "end leg";
037        public static final String ENTER_LINK = "enter link";
038        public static final String LEAVE_LINK = "leave link";
039        /**
040         *
041         * the priorities of the messages. a higher priority comes first in the
042         * message queue (when same time) usage: for example a person has a enter
043         * road message at the same time as leaving the previous road (need to keep
044         * the messages in right order) for events with same time stamp: <br>
045         * leave < arrival < departure < enter especially for testing this is
046         * important
047         *
048         */
049        public static final int PRIORITY_LEAVE_ROAD_MESSAGE = 200;
050        public static final int PRIORITY_ARRIVAL_MESSAGE = 150;
051        public static final int PRIORITY_DEPARTUARE_MESSAGE = 125;
052        public static final int PRIORITY_ENTER_ROAD_MESSAGE = 100;
053        public final static String SQUEEZE_TIME = "squeezeTime";
054        public final static String FLOW_CAPACITY_FACTOR = "flowCapacityFactor";
055        public final static String STORAGE_CAPACITY_FACTOR = "storageCapacityFactor";
056        public final static String MINIMUM_INFLOW_CAPACITY = "minimumInFlowCapacity";
057        public final static String CAR_SIZE = "carSize";
058        public final static String GAP_TRAVEL_SPEED = "gapTravelSpeed";
059        public final static String END_TIME = "endTime";
060
061        // INPUT
062        private double simulationEndTime = Double.MAX_VALUE; // in s
063        private double gapTravelSpeed = 15.0; // in m/s
064        private double flowCapacityFactor = 1.0;
065        private double storageCapacityFactor = 1.0;
066        private double carSize = 7.5; // in meter
067        // in [vehicles/hour] per lane, can be scaled with flow capacity factor
068        private double minimumInFlowCapacity = 1800;
069        /**
070         * stuckTime is used for deadlock prevention. when a car waits for more than
071         * 'stuckTime' for entering next road, it will enter the next. in seconds
072         */
073        private double squeezeTime = 1800;
074
075        public JDEQSimConfigGroup() {
076                super(NAME);
077        }
078
079        // should garbage collection of messages be activated
080        private static boolean GC_MESSAGES = false;
081
082        public static boolean isGC_MESSAGES() {
083                return GC_MESSAGES;
084        }
085
086        public static void setGC_MESSAGES(boolean gc_messages) {
087                GC_MESSAGES = gc_messages;
088        }
089
090        @StringGetter(END_TIME)
091        public String getSimulationEndTimeAsString() {
092                if (simulationEndTime != Double.MAX_VALUE) {
093                        return Time.writeTime(simulationEndTime);
094                }
095                else {
096                        return Time.writeTime(Time.UNDEFINED_TIME);
097                }
098        }
099
100        @StringSetter(END_TIME)
101        public void setSimulationEndTime(String simulationEndTime) {
102                double parsedTime = Time.parseTime(simulationEndTime);
103                if (!Time.isUndefinedTime(parsedTime)) {
104                        this.simulationEndTime = parsedTime;
105                }
106        }
107
108        public double getSimulationEndTime() {
109                return simulationEndTime;
110        }
111
112        @StringGetter(GAP_TRAVEL_SPEED)
113        public double getGapTravelSpeed() {
114                return gapTravelSpeed;
115        }
116
117        @StringSetter(GAP_TRAVEL_SPEED)
118        public void setGapTravelSpeed(double gapTravelSpeed) {
119                this.gapTravelSpeed = gapTravelSpeed;
120        }
121
122        @StringGetter(FLOW_CAPACITY_FACTOR)
123        public double getFlowCapacityFactor() {
124                return flowCapacityFactor;
125        }
126
127        @StringSetter(FLOW_CAPACITY_FACTOR)
128        public void setFlowCapacityFactor(double flowCapacityFactor) {
129                this.flowCapacityFactor = flowCapacityFactor;
130        }
131
132        @StringGetter(STORAGE_CAPACITY_FACTOR)
133        public double getStorageCapacityFactor() {
134                return storageCapacityFactor;
135        }
136
137        @StringSetter(STORAGE_CAPACITY_FACTOR)
138        public void setStorageCapacityFactor(double storageCapacityFactor) {
139                this.storageCapacityFactor = storageCapacityFactor;
140        }
141
142        @StringGetter(CAR_SIZE)
143        public double getCarSize() {
144                return carSize;
145        }
146
147        @StringSetter(CAR_SIZE)
148        public void setCarSize(double carSize) {
149                this.carSize = carSize;
150        }
151
152        @StringGetter(MINIMUM_INFLOW_CAPACITY)
153        public double getMinimumInFlowCapacity() {
154                return minimumInFlowCapacity;
155        }
156
157        @StringSetter(MINIMUM_INFLOW_CAPACITY)
158        public void setMinimumInFlowCapacity(double minimumInFlowCapacity) {
159                this.minimumInFlowCapacity = minimumInFlowCapacity;
160        }
161
162        @StringGetter(SQUEEZE_TIME)
163        public double getSqueezeTime() {
164                return squeezeTime;
165        }
166
167        @StringSetter(SQUEEZE_TIME)
168        public void setSqueezeTime(double squeezeTime) {
169                this.squeezeTime = squeezeTime;
170        }
171
172}
