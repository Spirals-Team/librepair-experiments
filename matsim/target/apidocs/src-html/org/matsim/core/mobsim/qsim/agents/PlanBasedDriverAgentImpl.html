052        @Override
053        public final void notifyMoveOverNode(Id<Link> newLinkId) {
054                if (expectedLinkWarnCount < 10 && !newLinkId.equals(this.cachedNextLinkId)) {
055                        log.warn("Agent did not end up on expected link. Ok for within-day replanning agent, otherwise not.  Continuing " +
056                                        "anyway ... This warning is suppressed after the first 10 warnings.") ;
057                        expectedLinkWarnCount++;
058                }
059                Gbl.assertNotNull(newLinkId);
060                this.basicPlanAgentDelegate.setCurrentLinkId( newLinkId ) ;
061                this.basicPlanAgentDelegate.incCurrentLinkIndex();
062                this.cachedNextLinkId = null; //reset cached nextLink
063        }

065        /**
066         * Returns the next link the vehicle will drive along.
067         *
068         * @return The next link the vehicle will drive on, or null if an error has happened.
069         */
070        @Override
071        public final Id<Link> chooseNextLinkId() {
072                // To note: there is something really stupid going on here: A vehicle that is at the end of its route and on the destination link will arrive.
073                // However, a vehicle that is on the destination link BUT NOT AT THE END OF ITS ROUTE will NOT arrive.  This makes the whole thing
074                // very extremely messy.  kai, nov'14 (This behavior is no longer there. kai, nov'14)
075
076                // Please, let's try, amidst all checking and caching, to have this method return the same thing
077                // if it is called several times in a row. Otherwise, you get Heisenbugs.
078                // I just fixed a situation where this method would give a warning about a bad route and return null
079                // the first time it is called, and happily return a link id when called the second time.  michaz 2013-08
080
081                // Agreed.  One should also not assume that anything here is the result of one consistent design process.  Rather, many people added
082                // and removed material as they needed it for their own studies.  Making the whole code more consistent would be highly
083                // desirable.  kai, nov'14
084
085                // (1) if there is a cached link id, use that one: 
086                if (this.cachedNextLinkId != null && !(this.cachedNextLinkId.equals(this.getCurrentLinkId())) ) {
087                        // cachedNextLinkId used to be set to null when a leg started.  Now the BasicPlanAgentImpl does not longer have access to cached
088                        // value.  kai, nov'14
089
090                        return this.cachedNextLinkId;
091                }
092
093                // (2) routes that are not network routes cannot be interpreted
094                if ( ! ( this.basicPlanAgentDelegate.getCurrentLeg().getRoute() instanceof NetworkRoute ) ) {
095                        return null ;
096                }
097
098                List<Id<Link>> routeLinkIds = ((NetworkRoute) this.basicPlanAgentDelegate.getCurrentLeg().getRoute()).getLinkIds();
099                
100                // (3) if route has run dry, we return the destination link (except for one special case, which however may not be necessary any more):
101                if (this.basicPlanAgentDelegate.getCurrentLinkIndex() >= routeLinkIds.size() ) {
102
103                        // special case:
104                        if (this.getCurrentLinkId().equals( this.getDestinationLinkId() )  
105                                        && this.basicPlanAgentDelegate.getCurrentLinkIndex() > routeLinkIds.size()) {
106                                // this can happen if the last link in a route is a loop link. Don't ask, it can happen in special transit simulation cases... mrieser/jan2014
107
108                                // the condition for arrival used to be "route has run dry AND destination link not attached to current link".  now with loop links,
109                                // this condition was not triggered.  So no wonder that for such cases a special condition was needed.  kai, nov'14
110                                
111                                // The special condition may not be necessary any more. kai, nov'14
112
113                                return null;
114                        }
115                        
116                        this.cachedNextLinkId = this.getDestinationLinkId();
117                        return this.cachedNextLinkId;
118
119                }
120
121                // (4) otherwise (normal case): return the next link of the plan (after caching it):
122                this.cachedNextLinkId = routeLinkIds.get(this.basicPlanAgentDelegate.getCurrentLinkIndex());
123                return this.cachedNextLinkId;
124                
125        }

175        /**
176         * Some data of the currently simulated Leg is cached to speed up
177         * the simulation. If the Leg changes (for example the Route or
178         * the Destination Link), those cached data has to be reseted.
179         *</p>
180         * If the Leg has not changed, calling this method should have no effect
181         * on the Results of the Simulation!
182         */
183        /* package */ public final void resetCaches() {
184
185                // moving this method not to WithinDay for the time being since it seems to make some sense to keep this where the internals are
186                // known best.  kai, oct'10
187                // Compromise: package-private here; making it public in the Withinday class.  kai, nov'10
188
189                this.cachedNextLinkId = null;
190                
191                if( this.basicPlanAgentDelegate.getCurrentPlanElement()==null ) {
192                        throw new RuntimeException("encountered unexpected null pointer" ) ;
193                }
194
195                if (this.basicPlanAgentDelegate.getCurrentPlanElement() instanceof Leg) {
196                        if (basicPlanAgentDelegate.getCurrentLeg().getRoute() == null) {
197                                log.error("The agent " + this.getId() + " has no route in its leg. Setting agent state to abort." );
198                                this.basicPlanAgentDelegate.setState(MobsimAgent.State.ABORT) ;
199                        }
200                } 
201//              else {                  
202//                      this.basicPlanAgentDelegate.calculateAndSetDepartureTime((Activity) this.basicPlanAgentDelegate.getCurrentPlanElement());
203//              }
204                this.basicPlanAgentDelegate.resetCaches(); 
205        }
