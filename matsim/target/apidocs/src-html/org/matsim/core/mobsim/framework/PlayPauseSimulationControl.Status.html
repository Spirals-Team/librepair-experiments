019package org.matsim.core.mobsim.framework;
020
021import java.util.concurrent.Phaser;
022import java.util.concurrent.Semaphore;
023
024import org.matsim.core.mobsim.framework.events.MobsimAfterSimStepEvent;
025import org.matsim.core.mobsim.framework.events.MobsimBeforeCleanupEvent;
026import org.matsim.core.mobsim.framework.events.MobsimBeforeSimStepEvent;
027import org.matsim.core.mobsim.framework.listeners.MobsimAfterSimStepListener;
028import org.matsim.core.mobsim.framework.listeners.MobsimBeforeCleanupListener;
029import org.matsim.core.mobsim.framework.listeners.MobsimBeforeSimStepListener;
030
031/**
032 * Extracted the play/pause functionality from otfvis to make it available for other purposes (specifically, RMITs 
033 * CombineSim approach).
034 * 
035 * @author nagel
036 */
037public class PlayPauseSimulationControl implements PlayPauseSimulationControlI {
038
039        public enum Status {
040                PAUSE, PLAY
041        }
042
043        private volatile Status status = Status.PLAY;
044
045        private final Semaphore access = new Semaphore(1, true);
046
047        private volatile double localTime = -1;
048
049        private final Phaser stepDone = new Phaser(1);
050
051        public PlayPauseSimulationControl(ObservableMobsim qSim) {
052                PlayPauseMobsimListener playPauseMobsimListener = new PlayPauseMobsimListener();
053                qSim.addQueueSimulationListeners(playPauseMobsimListener);
054        }
055
056        private class PlayPauseMobsimListener implements MobsimBeforeSimStepListener, MobsimAfterSimStepListener, MobsimBeforeCleanupListener {
057                @Override
058                public void notifyMobsimBeforeSimStep(MobsimBeforeSimStepEvent event) {
059                        try {
060                                access.acquire();
061                        } catch (InterruptedException e) {
062                                throw new RuntimeException(e);
063                        }
064                }
065                @Override
066                public void notifyMobsimAfterSimStep(MobsimAfterSimStepEvent event) {
067                        access.release();
068                        localTime = (int) event.getSimulationTime();
069                        // yy I am not so sure about the "int".  kai, nov'17
070                        stepDone.arriveAndAwaitAdvance();
071                }
072                @Override
073                public void notifyMobsimBeforeCleanup(MobsimBeforeCleanupEvent e) {
074                        localTime = Double.MAX_VALUE;
075                        stepDone.arriveAndDeregister();
076                }
077        }
078
079        @Override
080        public final void doStep(int time) {
081                if (status == Status.PLAY) {
082                        throw new IllegalStateException();
083                }
084                while (localTime < time) {
085                        stepDone.arriveAndAwaitAdvance();
086                }
087        }
088
089        @Override
090        public final void pause() {
091                if (status != Status.PAUSE) {
092                        stepDone.register();
093                        status = Status.PAUSE;
094                }
095        }
096
097        @Override
098        public final void play() {
099                if (status != Status.PLAY) {
100                        stepDone.arriveAndDeregister();
101                        status = Status.PLAY;
102                }
103        }
104
105        public final Semaphore getAccess() {
106                return access;
107        }
108
109        public final boolean isFinished() {
110                return ( localTime == Double.MAX_VALUE ) ;
111        }
112
113        public final double getLocalTime() {
114                return localTime;
115        }
116
117}
