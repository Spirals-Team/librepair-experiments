021package org.matsim.core.mobsim.jdeqsim;
022
023import org.apache.log4j.Logger;
024import org.matsim.api.core.v01.Id;
025import org.matsim.api.core.v01.Scenario;
026import org.matsim.api.core.v01.network.Link;
027import org.matsim.api.core.v01.population.Person;
028import org.matsim.core.api.experimental.events.EventsManager;
029import org.matsim.core.config.groups.PlansConfigGroup;
030import org.matsim.core.mobsim.framework.Mobsim;
031import org.matsim.core.mobsim.jdeqsim.util.Timer;
032
033import javax.inject.Inject;
034
035import java.util.HashMap;
036
037
038/**
039 * The starting point of the whole micro-simulation.
040 * @see <a href="http://www.matsim.org/docs/jdeqsim">http://www.matsim.org/docs/jdeqsim</a>
041 * @author rashid_waraich
042 */
043public class JDEQSimulation implements Mobsim {
044
045        private final static Logger log = Logger.getLogger(JDEQSimulation.class);
046
047        private final JDEQSimConfigGroup config;
048        protected Scenario scenario;
049        private final EventsManager events;
050
051        protected final PlansConfigGroup.ActivityDurationInterpretation activityDurationInterpretation;
052
053        @Inject
054        public JDEQSimulation(final JDEQSimConfigGroup config, final Scenario scenario, final EventsManager events) {
055                Road.setConfig(config);
056                Message.setEventsManager(events);
057                this.config = config;
058                this.scenario = scenario;
059                this.events = events;
060                this.activityDurationInterpretation = this.scenario.getConfig().plans().getActivityDurationInterpretation();
061        }
062
063        @Override
064        public void run() {
065                events.initProcessing();
066                Timer t = new Timer();
067                t.startTimer();
068
069                Scheduler scheduler = new Scheduler(new MessageQueue(), config.getSimulationEndTime());
070                Road.setAllRoads(new HashMap<Id<Link>, Road>());
071
072                // initialize network
073                Road road;
074                for (Link link : this.scenario.getNetwork().getLinks().values()) {
075                        road = new Road(scheduler, link);
076                        Road.getAllRoads().put(link.getId(), road);
077                }
078
079                for (Person person : this.scenario.getPopulation().getPersons().values()) {
080                        new Vehicle(scheduler, person, activityDurationInterpretation); // the vehicle registers itself to the scheduler
081                }
082
083                scheduler.startSimulation();
084
085                t.endTimer();
086                log.info("Time needed for one iteration (only JDEQSimulation part): " + t.getMeasuredTime() + "[ms]");
087                events.finishProcessing();
088        }
089}
