020package org.matsim.core.router;
021
022import java.util.List;
023
024import org.matsim.api.core.v01.population.Person;
025import org.matsim.api.core.v01.population.PlanElement;
026import org.matsim.core.router.util.TravelDisutility;
027import org.matsim.core.router.util.TravelTime;
028import org.matsim.facilities.Facility;
029
030/**
031 * Defines classes responsible for routing for a given
032 * (main)-mode.
033 * It does not modify the plan.  
034 * <p></p>
035 * See {@link tutorial.programming.example12PluggableTripRouter.RunPluggableTripRouterExample} for an example how to use this interface.
036 * 
037 * @author thibautd
038 */
039public interface RoutingModule {
040        /**
041         * Computes a route, as a sequence of plan elements. The plan elements can
042         * be only legs, or a sequence of legs and "dummy" activities. All activity
043         * types inserted as dummy activities must be notified by the
044         * {@link #getStageActivityTypes()} method.
045         * <br>
046         * <b>important:</b> if route computation relies on a shortest path algorithm
047         * using {@link TravelTime} and/or {@link TravelDisutility}
048         * estimators, this method is responsible for setting the person to the argument
049         * person in those estimators before running the shortest path algorithm.
050         *
051         * @param fromFacility a {@link Facility} representing the departure location
052         * @param toFacility a {@link Facility} representing the arrival location
053         * @param departureTime the departure time
054         * @param person the {@link Person} to route
055         * @return a list of {@link PlanElement}, in proper order, representing the trip.
056         */
057        public List<? extends PlanElement> calcRoute(
058                        Facility<?> fromFacility,
059                        Facility<?> toFacility,
060                        double departureTime,
061                        Person person);
062
063        /**
064         * Gives access to the activity types to consider as stages.
065         * </ul>
066         * @return a non-null {@link StageActivityTypes}. This should always return
067         * the same instance, or at least return instances which are equal for equals().
068         * Otherwise, replacement of modules in the {@link TripRouter} will not work.
069         */
070        public StageActivityTypes getStageActivityTypes();
071}
