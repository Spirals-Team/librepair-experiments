021package org.matsim.core.router.util;
022
023import java.util.Set;
024
025import org.matsim.api.core.v01.Coord;
026import org.matsim.api.core.v01.Id;
027import org.matsim.api.core.v01.network.Link;
028import org.matsim.api.core.v01.network.Node;
029import org.matsim.utils.objectattributes.attributable.Attributes;
030
031public abstract class AbstractRoutingNetworkLink implements RoutingNetworkLink {
032
033        private final Link link;
034        private final RoutingNetworkNode toNode;
035        private final RoutingNetworkNode fromNode;
036        
037        /*package*/ AbstractRoutingNetworkLink(Link link, RoutingNetworkNode fromNode, RoutingNetworkNode toNode) {
038                this.link = link;
039                this.fromNode = fromNode;
040                this.toNode = toNode;
041        }
042        @Override
043        public double getFlowCapacityPerSec() {
044                return this.link.getFlowCapacityPerSec() ;
045        }
046        @Override
047        public double getFlowCapacityPerSec( double time ) {
048                return this.link.getFlowCapacityPerSec(time) ;
049        }
050        
051        @Override
052        public Link getLink() {
053                return link;
054        }
055        
056        @Override
057        public Id<Link> getId() {
058                return link.getId();
059        }
060
061        @Override
062        public RoutingNetworkNode getFromNode() {
063                return fromNode;
064        }
065        
066        @Override
067        public RoutingNetworkNode getToNode() {
068                return toNode;
069        }
070
071        @Override
072        public Set<String> getAllowedModes() {
073                return link.getAllowedModes();
074        }
075
076        @Override
077        public double getCapacity() {
078                return link.getCapacity();
079        }
080
081        @Override
082        public double getCapacity(double time) {
083                return link.getCapacity(time);
084        }
085
086        @Override
087        public double getFreespeed() {
088                return link.getFreespeed();
089        }
090
091        @Override
092        public double getFreespeed(double time) {
093                return link.getFreespeed(time);
094        }
095
096        @Override
097        public double getLength() {
098                return link.getLength();
099        }
100
101        @Override
102        public double getNumberOfLanes() {
103                return link.getNumberOfLanes();
104        }
105
106        @Override
107        public double getNumberOfLanes(double time) {
108                return link.getNumberOfLanes(time);
109        }
110
111        @Override
112        public void setAllowedModes(Set<String> modes) {
113                throw new RuntimeException("Not supported operation!");
114        }
115
116        @Override
117        public void setCapacity(double capacity) {
118                throw new RuntimeException("Not supported operation!");
119        }
120
121        @Override
122        public void setFreespeed(double freespeed) {
123                throw new RuntimeException("Not supported operation!");
124        }
125
126        @Override
127        public boolean setFromNode(Node node) {
128                throw new RuntimeException("Not supported operation!");
129        }
130
131        @Override
132        public void setLength(double length) {
133                throw new RuntimeException("Not supported operation!");
134        }
135
136        @Override
137        public void setNumberOfLanes(double lanes) {
138                throw new RuntimeException("Not supported operation!");
139        }
140
141        @Override
142        public boolean setToNode(Node node) {
143                throw new RuntimeException("Not supported operation!");
144        }
145
146        @Override
147        public Coord getCoord() {
148                return this.link.getCoord();
149        }
150
151        @Override
152        public Attributes getAttributes() {
153                return link.getAttributes();
154        }
155}
