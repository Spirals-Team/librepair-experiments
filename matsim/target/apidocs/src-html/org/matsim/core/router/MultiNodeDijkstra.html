040/**
041 * <p>An extended implementation of the Dijkstra algorithm that supports multiple
042 * start- and/or end nodes.</p>
043 * 
044 * <p>To do so, ImaginaryNodes and IntialNodes are introduced which is required for
045 * backwards compatibility with the default Dijkstra implementation which expects
046 * only single nodes as arguments.</p>
047 * 
048 * <p>Therefore, ImaginaryNodes are introduced. They contain a collection of
049 * InitialNodes that represents the start or end nodes to be used by the algorithm.
050 * Each InitialNode represents a node from the network. In addition, initial time and
051 * costs can be defined, i.e. the time respectively costs to reach that node.<p>
052 * 
053 * <p>By default, only the cheapest path between is calculated, i.e. the routing algorithm
054 * will terminate before all end nodes have been reached. This behaviour can be changed
055 * by setting the searchAllEndNodes parameter to true.</p>
056 * 
057 * @see org.matsim.core.router.Dijkstra
058 * @see org.matsim.core.router.InitialNode
059 * @see org.matsim.core.router.ImaginaryNode
060 * 
061 * @author cdobler
062 */
063public class MultiNodeDijkstra extends Dijkstra implements MultiNodePathCalculator {
064        
065        private final static Logger log = Logger.getLogger(MultiNodeDijkstra.class);
066        
067        /*
068         * If this value is true, the algorithm tries to find routes to all end nodes.
069         * Otherwise only the one(s) with the lowest costs are found.
070         * 
071         * When enabling this option, select end nodes with care! If only one of them
072         * is in a not reachable part of the network, the algorithm will process the
073         * entire reachable part of the network before terminating.
074         */
075        private boolean searchAllEndNodes;
076        
077        MultiNodeDijkstra(Network network, TravelDisutility costFunction, TravelTime timeFunction, boolean searchAllEndNodes) {
078                super(network, costFunction, timeFunction);
079                this.searchAllEndNodes = searchAllEndNodes;
080        }

083                        final PreProcessDijkstra preProcessData, boolean searchAllEndNodes) {
084                super(network, costFunction, timeFunction, preProcessData);
085                this.searchAllEndNodes = searchAllEndNodes;
086        }

089                return new ImaginaryNode(nodes);

093                return new ImaginaryNode(nodes, coord);

096        /*
097         * We have to extend this method from the original Dijkstra. The given input nodes might be
098         * ImaginaryNodes which contain multiple start / end nodes for the routing process. Those
099         * nodes should be part of the routing network, therefore we perform the check for each of them.
100         * 
101         * Regular nodes are directly passed to the super class.
102         * 
103         * cdobler, jun'14
104         */
105        /*package*/ @Override
106        void checkNodeBelongToNetwork(Node node) {
107                if (node instanceof ImaginaryNode) {
108                        ImaginaryNode imaginaryNode = (ImaginaryNode) node;
109                        for (InitialNode initialNode : imaginaryNode.initialNodes) super.checkNodeBelongToNetwork(initialNode.node);
110                } else super.checkNodeBelongToNetwork(node);
111        }
112        
113        @Override
114        /*package*/ Node searchLogic(final Node fromNode, final Node toNode, final RouterPriorityQueue<Node> pendingNodes) {
115                
116                // If it is an imaginary node...
117                if (toNode instanceof ImaginaryNode) {
118                        
119                        Map<Id<Node>, InitialNode> endNodes = new HashMap<>();
120                        
121                        Collection<? extends InitialNode> initialNodes = ((ImaginaryNode) toNode).initialNodes;
122                        for (InitialNode initialNode : initialNodes) endNodes.put(initialNode.node.getId(), initialNode);
123
124                        // find out which one is the cheapest end node
125                        double minCost = Double.POSITIVE_INFINITY;
126                        Node minCostNode = null;
127                        
128                        // continue searching as long as unvisited end nodes are available
129                        boolean stillSearching = endNodes.size() > 0;
130                        
131                        while (stillSearching) {
132                                Node outNode = pendingNodes.poll();
133                                
134                                if (outNode == null) {
135                                        /*
136                                         * This is not necessarily a problem. Some of the out nodes might be reachable only
137                                         * by passing another out node. Those nodes will never become the cheapest out node.
138                                         * Therefore only print a warning if no path to any of the out nodes was found.
139                                         */
140                                        if (minCostNode == null) {
141                                                if ( log.isTraceEnabled() ) {
142                                                        log.trace("No route was found from node " + fromNode.getId() + " to any of the destination nodes was found.");
143                                                        // seems we have no more nodes left, but not yet reached all endNodes...
144                                                        StringBuffer sb = new StringBuffer("\tnot reached destionation nodes: ");
145                                                        for (InitialNode endNode : endNodes.values()) {
146                                                                sb.append(endNode.node.getId().toString());
147                                                                sb.append("; ");
148                                                        }
149                                                        log.trace(sb.toString());
150                                                }
151                                        }
152                                        
153                                        if (searchAllEndNodes && endNodes.size() > 0) {
154                                                for (InitialNode endNode : endNodes.values()) {
155                                                        log.trace("No route was found from node " + fromNode.getId() + " to destination node " + endNode.node.getId() + ".");
156                                                }
157                                        }
158                                        
159                                        endNodes.clear();
160                                        stillSearching = false;
161                                } else {
162                                        DijkstraNodeData data = getData(outNode);
163                                        InitialNode initData = endNodes.remove(outNode.getId());
164                                                                                
165                                        /*
166                                         * If the node is an end node.
167                                         * Note: 
168                                         * The node was head of the priority queue, i.e. the shortest path to the 
169                                         * node has been found. The algorithm will not re-visit the node on another
170                                         * route!
171                                         */
172                                        if (initData != null) {
173                                                double cost = data.getCost() + initData.initialCost;
174                                                if (cost < minCost) {
175                                                        minCost = cost;
176                                                        minCostNode = outNode;
177                                                }
178                                        }
179                                        
180                                        if (searchAllEndNodes) {
181                                                relaxNode(outNode, null, pendingNodes);
182                                                stillSearching = endNodes.size() > 0;
183                                        } else {
184                                                if (data.getCost() > minCost) {
185                                                        endNodes.clear(); // we can't get any better now
186                                                        stillSearching = false;
187                                                } else {
188                                                        relaxNode(outNode, null, pendingNodes);
189                                                }
190                                        }
191                                }
192                        }
193                        
194                        return minCostNode;
195                } 
196                // ... otherwise: default behaviour.
197                else return super.searchLogic(fromNode, toNode, pendingNodes);
198        }

200        /*
201         * initFromNode -> first Node in pendingNodes -> calcLeastCostPath will relax that node
202         * -> we relax that dummy node here
203         */
204        @Override
205        /*package*/ void initFromNode(final Node fromNode, final Node toNode, final double startTime,
206                        final RouterPriorityQueue<Node> pendingNodes) {
207                
208                // If it is an imaginary node, we relax it.
209                if (fromNode instanceof ImaginaryNode) {                        
210                        relaxImaginaryNode((ImaginaryNode) fromNode, pendingNodes, startTime);
211                }
212                // ... otherwise: default behaviour.
213                else super.initFromNode(fromNode, toNode, startTime, pendingNodes);
214        }
215        
216        protected void relaxImaginaryNode(final ImaginaryNode outNode, final RouterPriorityQueue<Node> pendingNodes,
217                        final double currTime) {
218                
219                double currCost = 0.0;  // should be 0
220
221                for (InitialNode initialNode : outNode.initialNodes) {
222                        double travelTime = initialNode.initialTime;
223                        double travelCost = initialNode.initialCost;
224                        DijkstraNodeData data = getData(initialNode.node);
225                        Link l = null;  // fromLink - use a dummy link here??
226                        visitNode(initialNode.node, data, pendingNodes, currTime + travelTime, currCost + travelCost, l);                       
227                }
228        }

230        @Override
231        protected Path constructPath(Node fromNode, Node toNode, double startTime, double arrivalTime) {
232                ArrayList<Node> nodes = new ArrayList<Node>();
233                ArrayList<Link> links = new ArrayList<Link>();
234
235                nodes.add(0, toNode);
236                Link tmpLink = getData(toNode).getPrevLink();
237                
238                // Only this part has been adapted. Could probably also be changed in the super class.
239                while (tmpLink != null) {
240                        links.add(0, tmpLink);
241                        nodes.add(0, tmpLink.getFromNode());
242                        tmpLink = getData(tmpLink.getFromNode()).getPrevLink();
243                }
244
245                // Ignore the initial time and cost of the start node!
246                DijkstraNodeData startNodeData = getData(nodes.get(0));
247                DijkstraNodeData toNodeData = getData(toNode);
248                Path path = new Path(nodes, links, toNodeData.getTime() - startNodeData.getTime(), toNodeData.getCost() - startNodeData.getCost());
249//              double travelTime = arrivalTime - startTime;
250//              Path path = new Path(nodes, links, travelTime, toNodeData.getCost());
251
252                return path;
253        }
254        
255        /**
256         * This method should only be called from outside after calcLeastCostPath(...) has
257         * been executed. After that, the paths between the multiple start and/or end nodes
258         * can be constructed using this method.
259         * 
260         * Is there a way to check whether this method is called as intended??
261         * cdobler, oct'13
262         */
263        public Path constructPath(Node fromNode, Node toNode, double startTime) {
264                if (toNode == null || fromNode == null) return null;
265                else {
266                        DijkstraNodeData toData = getData(toNode);
267                        if (!toData.isVisited(this.getIterationId())) return null;
268
269                        DijkstraNodeData fromData = getData(fromNode);
270                        if (!fromData.isVisited(this.getIterationId())) return null;
271
272                        double arrivalTime = toData.getTime();
273                        
274                        // now construct and return the path
275                        return constructPath(fromNode, toNode, startTime, arrivalTime);                 
276                }
277        }
