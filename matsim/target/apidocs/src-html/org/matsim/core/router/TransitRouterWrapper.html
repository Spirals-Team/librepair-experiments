040/**
041 * Wraps a {@link TransitRouter}.
042 *
043 * @author thibautd
044 */
045public class TransitRouterWrapper implements RoutingModule {
046        private static final StageActivityTypes CHECKER =
047                        new StageActivityTypesImpl(PtConstants.TRANSIT_ACTIVITY_TYPE);
048        private final TransitRouter router;
049        private final RoutingModule walkRouter;
050        private final TransitSchedule transitSchedule;
051        private final Network network;
052
053
054        public TransitRouterWrapper(
055                        final TransitRouter router,
056                        final TransitSchedule transitSchedule,
057                        Network network, final RoutingModule walkRouter) {
058                if (router == null) {
059                        throw new NullPointerException("The router object is null, but is required later.");
060                }
061                this.router = router;
062                this.transitSchedule = transitSchedule;
063                this.network = network;
064                if (walkRouter == null) {
065                        throw new NullPointerException("The walkRouter object is null, but is required later.");
066                }
067                this.walkRouter = walkRouter;
068        }
069
070        /**
071         * Just links to {@link TransitRouter#calcRoute(Facility, Facility, double, Person)}.
072         *
073         * @return the list of legs returned by the transit router.
074         */
075        @Override
076        public List<? extends PlanElement> calcRoute(
077                        final Facility<?> fromFacility,
078                        final Facility<?> toFacility,
079                        final double departureTime,
080                        final Person person) {
081                List<Leg> baseTrip = router.calcRoute(
082                                fromFacility,
083                                toFacility,
084                                departureTime,
085                                person);
086
087                // the previous approach was to return null when no trip was found and
088                // not to replace the trip if so.
089                // However, this makes the output of routing modules more tricky to handle.
090                // Thus, every module should return a valid trip. When available, the "main
091                // mode" flag should be put to the mode of the routing module.
092                return baseTrip != null ?
093                                fillWithActivities(baseTrip, fromFacility, toFacility, departureTime, person) :
094                                        walkRouter.calcRoute(fromFacility, toFacility, departureTime, person);
095        }
096
097        /**
098         * This treats the TransitRouter as a third-party interface, where missing fields
099         * must be filled in (distance, travel-time in routes).
100         */
101        private List<PlanElement> fillWithActivities(
102                        final List<Leg> baseTrip,
103                        final Facility fromFacility,
104                        final Facility toFacility, double departureTime, Person person) {
105                List<PlanElement> trip = new ArrayList<>();
106                Coord nextCoord = null;
107                int i = 0;
108                for (Leg leg : baseTrip) {
109                        if (i == 0) {
110                                // (access leg)
111                                Facility firstToFacility;
112                                if (baseTrip.size() > 1) { // at least one pt leg available
113                                        ExperimentalTransitRoute tRoute = (ExperimentalTransitRoute) baseTrip.get(1).getRoute();
114                                        firstToFacility = this.transitSchedule.getFacilities().get(tRoute.getAccessStopId());
115                                } else {
116                                        firstToFacility = toFacility;
117                                }
118                                // (*)
119                                Route route = createWalkRoute(fromFacility, departureTime, person, leg.getTravelTime(), firstToFacility);
120                                leg.setRoute(route);
121                        } else {
122                                if (leg.getRoute() instanceof ExperimentalTransitRoute) {
123                                        ExperimentalTransitRoute tRoute = (ExperimentalTransitRoute) leg.getRoute();
124                                        tRoute.setTravelTime(leg.getTravelTime());
125                                        tRoute.setDistance(RouteUtils.calcDistance(tRoute, transitSchedule, network));
126                                        Activity act = PopulationUtils.createActivityFromCoordAndLinkId(PtConstants.TRANSIT_ACTIVITY_TYPE, this.transitSchedule.getFacilities().get(tRoute.getAccessStopId()).getCoord(), tRoute.getStartLinkId());
127                                        act.setMaximumDuration(0.0);
128                                        trip.add(act);
129                                        nextCoord = this.transitSchedule.getFacilities().get(tRoute.getEgressStopId()).getCoord();
130                                } else { 
131                                        // it is not an instance of an ExperimentalTransitRoute so it must be a (transit) walk leg.
132
133                                        // walk legs don't have a coord, use the coord from the last egress point.  yyyy But I don't understand why in one case we take "nextCoord", while in the
134                                        // other case we retrieve the facility from the previous route.
135
136                                        if (i == baseTrip.size() - 1) {
137                                                // if this is the last leg, we don't believe the leg from the TransitRouter.  Why?
138
139                                                ExperimentalTransitRoute tRoute = (ExperimentalTransitRoute) baseTrip.get(baseTrip.size() - 2).getRoute();
140                                                Facility lastFromFacility = this.transitSchedule.getFacilities().get(tRoute.getEgressStopId());
141                                                
142                                                Route route = createWalkRoute(lastFromFacility, departureTime, person, leg.getTravelTime(), toFacility);
143                                                leg.setRoute(route);
144                                        }
145                                        Activity act = PopulationUtils.createActivityFromCoordAndLinkId(PtConstants.TRANSIT_ACTIVITY_TYPE, nextCoord, leg.getRoute().getStartLinkId());
146                                        act.setMaximumDuration(0.0);
147                                        trip.add(act);
148                                }
149                        }
150                        trip.add(leg);
151                        i++;
152                }
153                return trip;
154        }
155
156        private Route createWalkRoute(final Facility fromFacility, double departureTime, Person person, double travelTime, Facility firstToFacility) {
157                // yyyy I extracted this method to make a bit more transparent that it is used twice.  But I don't know why it is done in this way
158                // (take distance from newly computed walk leg, but take travelTime from elsewhere).  Possibly, the problem is that the TransitRouter 
159                // historically just does not compute the distances.  kai, may'17
160                
161                Route route = RouteUtils.createGenericRouteImpl(fromFacility.getLinkId(), firstToFacility.getLinkId());
162                final List<? extends PlanElement> walkRoute = walkRouter.calcRoute(fromFacility, firstToFacility, departureTime, person);
163                route.setDistance(((Leg) walkRoute.get(0)).getRoute().getDistance());
164                route.setTravelTime(travelTime);
165                return route;
166        }
167
168        @Override
169        public StageActivityTypes getStageActivityTypes() {
170                return CHECKER;
171        }
172}
