023package org.matsim.core.router;
024
025import com.google.inject.Key;
026import com.google.inject.name.Names;
027import org.matsim.api.core.v01.TransportMode;
028import org.matsim.core.config.groups.PlansCalcRouteConfigGroup;
029import org.matsim.core.controler.AbstractModule;
030import org.matsim.core.network.algorithms.NetworkTurnInfoBuilder;
031import org.matsim.core.network.algorithms.NetworkTurnInfoBuilderI;
032import org.matsim.pt.router.TransitRouterModule;
033
034
035public class TripRouterModule extends AbstractModule {
036
037    @Override
038    public void install() {
039            // yy The code below will install _one_ LeastCostPathCalculator, which will be Dijkstra or Landmarks or something.  It will be the
040            // same Landmarks instance for all modes ... although one could do better by doing the preprocessing separately for the different modes.
041            // kai/mm, jan'17
042            
043        bind(TripRouter.class); // not thread-safe, not a singleton
044        bind(MainModeIdentifier.class).to(MainModeIdentifierImpl.class);
045        install(new LeastCostPathCalculatorModule());
046        install(new TransitRouterModule());
047        bind(SingleModeNetworksCache.class).asEagerSingleton();
048        PlansCalcRouteConfigGroup routeConfigGroup = getConfig().plansCalcRoute();
049        for (String mode : routeConfigGroup.getTeleportedModeFreespeedFactors().keySet()) {
050            if (getConfig().transit().isUseTransit() && getConfig().transit().getTransitModes().contains(mode)) {
051                // default config contains "pt" as teleported mode, but if we have simulated transit, this is supposed to override it
052                // better solve this on the config level eventually.
053                continue;
054            }
055            addRoutingModuleBinding(mode).toProvider(new FreespeedFactorRouting(getConfig().plansCalcRoute().getModeRoutingParams().get(mode)));
056        }
057        for (String mode : routeConfigGroup.getTeleportedModeSpeeds().keySet()) {
058            addRoutingModuleBinding(mode).toProvider(new BeelineTeleportationRouting(getConfig().plansCalcRoute().getModeRoutingParams().get(mode)));
059        }
060        
061        boolean linkToLinkRouting = getConfig().controler().isLinkToLinkRoutingEnabled();
062        if (linkToLinkRouting) {
063            bind(NetworkTurnInfoBuilderI.class).to(NetworkTurnInfoBuilder.class) ;
064        }
065        for (String mode : routeConfigGroup.getNetworkModes()) {
066            addRoutingModuleBinding(mode).toProvider(linkToLinkRouting ? //
067                    new LinkToLinkRouting(mode) : new NetworkRoutingProvider(mode));
068        }
069        if (getConfig().transit().isUseTransit()) {
070            for (String mode : getConfig().transit().getTransitModes()) {
071                addRoutingModuleBinding(mode).toProvider(Transit.class);
072            }
073            addRoutingModuleBinding(TransportMode.transit_walk).to(Key.get(RoutingModule.class, Names.named(TransportMode.walk)));
074        }
075    }
076}
