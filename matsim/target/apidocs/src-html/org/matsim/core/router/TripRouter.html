120         * Sets the {@link RoutingModule} to use for the given (main) mode.
121         * @param mainMode the mode
122         * @param module the module to use with this mode
123         * @return the previously registered {@link RoutingModule} for this mode if any, null otherwise.

125        @Deprecated // use the Builder instead.  kai, oct'17
126        public RoutingModule setRoutingModule(
127                        final String mainMode,
128                        final RoutingModule module) {
129                RoutingModule old = routingModules.put( mainMode , module );
130
131                if (old != null) {
132                        final StageActivityTypes oldTypes = old.getStageActivityTypes();
133                        final boolean removed = checker.removeActivityTypes( oldTypes );
134                        if ( !removed ) {
135                                throw new RuntimeException( "could not remove "+oldTypes+" associated to "+old+". This may be due to a routing module creating a new instance at each call of getStageActivityTypes()" );
136                        }
137                }
138
139                final StageActivityTypes types = module.getStageActivityTypes();
140                if ( types == null ) {
141                        // we do not want to accept that, this would risk to mess up
142                        // with replacement, and it generally makes code messy.
143                        throw new RuntimeException( module+" returns null stage activity types. This is not a valid value. Return EmptyStageActivityTypes.INSTANCE instead." );
144                }
145                checker.addActivityTypes( types );
146
147                return old;
148        }

188        /**
189         * Routes a trip between the given O/D pair, with the given main mode.
190         *
191         * @param mainMode the main mode for the trip
192         * @param fromFacility a {@link Facility} representing the departure location
193         * @param toFacility a {@link Facility} representing the arrival location
194         * @param departureTime the departure time
195         * @param person the {@link Person} to route
196         * @return a list of {@link PlanElement}, in proper order, representing the trip.
197         *
198         * @throws UnknownModeException if no RoutingModule is registered for the
199         * given mode.
200         */
201        public synchronized List<? extends PlanElement> calcRoute(
202                        final String mainMode,
203                        final Facility fromFacility,
204                        final Facility toFacility,
205                        final double departureTime,
206                        final Person person) {
207                // I need this "synchronized" since I want mobsim agents to be able to call this during the mobsim.  So when the
208                // mobsim is multi-threaded, multiple agents might call this here at the same time.  kai, nov'17
209                
210                Gbl.assertNotNull( fromFacility );
211                Gbl.assertNotNull( toFacility );
212                
213                RoutingModule module = routingModules.get( mainMode );
214                
215                if (module != null) {
216                        final List<? extends PlanElement> trip =
217                                        module.calcRoute(
218                                                fromFacility,
219                                                toFacility,
220                                                departureTime,
221                                                person);
222
223                        if ( trip == null ) {
224                                throw new NullPointerException( "Routing module "+module+" returned a null Trip for main mode "+mainMode );
225                        }
226
227                        return trip;
228                }
229
230                throw new UnknownModeException( "unregistered main mode |"+mainMode+"|: does not pertain to "+routingModules.keySet() );
231        }

292        /**
293         * Inserts a trip between two activities in the sequence of plan elements
294         * returned by the {@link Plan#getPlanElements()} method of a plan. Note
295         * that the plan will be modified only if the returned list is the internal
296         * reference!
297         * <p></p>
298         * Note that this methods returns a unique solution because it expects the activity object references as arguments, which are unique.
299         *
300         * @param plan the plan to modify
301         * @param origin the activity to use as origin. It must be a member of the list of plan elements.
302         * @param trip the trip to insert
303         * @param destination the destination activity. It must be a member of the list.
304         * @return the "old trip": the sequence of plan elements originally existing between the origin and the destination
305         */
306        public static List<PlanElement> insertTrip(
307                        final Plan plan,
308                        final Activity origin,
309                        final List<? extends PlanElement> trip,
310                        final Activity destination) {
311                return insertTrip(
312                                plan.getPlanElements(),
313                                origin,
314                                trip,
315                                destination);
316        }

318        /**
319         * Inserts a trip between two activities in a sequence of plan elements.
320         * <p></p>
321         * Note that this methods returns a unique solution because it expects the activity object references as arguments, which are unique.
322         * @param plan the sequence of plan elements to modify
323         * @param origin the activity to use as origin. It must be a member of the list of plan elements.
324         * @param trip the trip to insert
325         * @param destination the destination activity. It must be a member of the list.
326         * @return the "old trip": the sequence of plan elements originally existing between the origin and the destination
327         */
328        public static List<PlanElement> insertTrip(
329                        final List<PlanElement> plan,
330                        final Activity origin,
331                        final List<? extends PlanElement> trip,
332                        final Activity destination) {
333                int indexOfOrigin = -1;
334                int indexOfDestination = -1;
335
336                // search the trip
337                int currentIndex = 0;
338                for (PlanElement pe : plan) {
339                        if (pe == origin) {
340                                indexOfOrigin = currentIndex;
341                        }
342                        if (pe == destination) {
343                                indexOfDestination = currentIndex;
344                                if (indexOfDestination < indexOfOrigin ) {
345                                        throw new RuntimeException(
346                                                        "destination "+destination+" found before origin "+
347                                                        origin+" in "+plan );
348                                }
349                                break;
350                        }
351                        currentIndex++;
352                }
353
354                // check validity
355                if (indexOfOrigin < 0) {
356                        throw new RuntimeException( "could not find origin "+origin+" in "+plan ); 
357                }
358                if (indexOfDestination < 0) {
359                        throw new RuntimeException( "could not find destination "+destination+" in "+plan ); 
360                }
361
362                // replace the trip and return the former one
363                List<PlanElement> seq = plan.subList( indexOfOrigin + 1 , indexOfDestination );
364                List<PlanElement> oldTrip = new ArrayList<>( seq );
365                seq.clear();
366                assert trip != null;
367                seq.addAll( trip );
368
369                return oldTrip;
370        }
