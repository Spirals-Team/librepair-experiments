021package org.matsim.core.router.util;
022
023import org.matsim.api.core.v01.network.Link;
024import org.matsim.api.core.v01.population.Person;
025import org.matsim.vehicles.Vehicle;
026
027/**
028 * A simple interface to retrieve the disutility to travel on links.
029 * <p></p>
030 * Design comments:<ul>
031 * <li>If I understand this correctly, implementations of this interface need to be thread-safe, since the same code may be called from different threads
032 * simultaneously.  If I understand this correctly, this can only be achieved by never writing to fields in any methods besides the constructor.  kai, jan'13
033 * </ul>
034 * <p></p>
035 * See {@link tutorial.programming.randomizingRouter.RunRandomizingRouterExample} for an example.
036 *
037 * @author mrieser
038 * @see {@link org.matsim.core.router.costcalculators.TravelDisutilityFactory}
039 */
040public interface TravelDisutility {
041
042        /**
043         * Returns the disutility to travel on the specified link at the specified time.
044         *
045         * @param link The link for which the travel disutility is calculated.
046         * @param time The departure time (in seconds since 00:00) at the beginning of the link for which the disutility is calculated.
047         * @param person The person that wants to travel along the link. Note that this parameter can be <code>null</code>!
048         * @param vehicle The vehicle with which the person wants to travel along the link. Note that this parameter can be <code>null</code>!
049         * @return The disutility to travel over the link <code>link</code>, departing at time <code>time</code>.
050         */
051        public double getLinkTravelDisutility(final Link link, final double time, final Person person, final Vehicle vehicle);
052        // There was some thought of replacing the "link" argument (resulting in a link travel time lookup) by a "linkTtime" argument (since
053        // the link travel time is known in the router anyways. 
054        // I have at least one argument why this will not carry through in an easy way: We also need the link argument for toll, and other
055        // people may want to look up other link attributes and include them into their router, such as slope or road type.
056        // kai, oct'13
057        
058        /**
059         * @param link the link for which the minimal travel disutility over all time slots is calculated
060         * @return Minimal costs to travel over the link <pre>link</pre>, departing at time <pre>time</pre>
061         */
062        public double getLinkMinimumTravelDisutility(final Link link);
063
064}
