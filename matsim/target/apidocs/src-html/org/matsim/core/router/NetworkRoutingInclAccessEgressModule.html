049/**
050 * This wraps a "computer science" {@link LeastCostPathCalculator}, which routes from a node to another node, into something that
051 * routes from a {@link Facility} to another {@link Facility}, as we need in MATSim.
052 * 
053 * @author thibautd, nagel
054 */
055public final class NetworkRoutingInclAccessEgressModule implements RoutingModule {
056        private static final Logger log = Logger.getLogger( NetworkRoutingInclAccessEgressModule.class );
057
058        private final class AccessEgressStageActivityTypes implements StageActivityTypes {
059                @Override public boolean isStageActivity(String activityType) {
060                        if ( NetworkRoutingInclAccessEgressModule.this.stageActivityType.equals( activityType ) ) {
061                                return true ;
062                        } else {
063                                return false ;
064                        }
065                }
066                @Override public boolean equals( Object obj ) {
067                        if ( !(obj instanceof AccessEgressStageActivityTypes) ) {
068                                return false ;
069                        }
070                        AccessEgressStageActivityTypes other = (AccessEgressStageActivityTypes) obj ;
071                        return other.isStageActivity(NetworkRoutingInclAccessEgressModule.this.stageActivityType) ;
072                }
073                @Override public int hashCode() {
074                        return NetworkRoutingInclAccessEgressModule.this.stageActivityType.hashCode() ;
075                }
076        }
077

079        private final String mode;
080        private final PopulationFactory populationFactory;
081
082        private final Network network;
083        private final LeastCostPathCalculator routeAlgo;
084        private String stageActivityType;
085
086        public NetworkRoutingInclAccessEgressModule(
087                        final String mode,
088                        final PopulationFactory populationFactory,
089                        final Network network,
090                        final LeastCostPathCalculator routeAlgo,
091                        PlansCalcRouteConfigGroup calcRouteConfig) {
092                Gbl.assertNotNull(network);
093                Gbl.assertIf( network.getLinks().size()>0 ) ; // otherwise network for mode probably not defined
094                this.network = network;
095                this.routeAlgo = routeAlgo;
096                this.mode = mode;
097                this.populationFactory = populationFactory;
098                this.stageActivityType = this.mode + " interaction";
099                if ( !calcRouteConfig.isInsertingAccessEgressWalk() ) {
100                        throw new RuntimeException("trying to use access/egress but not switched on in config.  "
101                                        + "currently not supported; there are too many other problems") ;
102                }
103        }

105        @Override
106        public synchronized List<? extends PlanElement> calcRoute(
107                        final Facility fromFacility,
108                        final Facility toFacility,
109                        final double departureTime,
110                        final Person person) {
111                // I need this "synchronized" since I want mobsim agents to be able to call this during the mobsim.  So when the
112                // mobsim is multi-threaded, multiple agents might call this here at the same time.  kai, nov'17
113                
114                Gbl.assertNotNull(fromFacility);
115                Gbl.assertNotNull(toFacility);
116
117                Link accessActLink = decideOnLink(fromFacility);
118
119                Link egressActLink = decideOnLink(toFacility);
120                
121                double now = departureTime ;
122
123                List<PlanElement> result = new ArrayList<>() ;
124
125                // === access:
126                {
127                        if ( fromFacility.getCoord() != null ) { // otherwise the trip starts directly on the link; no need to bushwhack
128
129                                Coord accessActCoord  = accessActLink.getToNode().getCoord() ;
130                                // yyyy think about better solution: this may generate long walks along the link.
131                                // (e.g. orthogonal projection)
132                                Gbl.assertNotNull(accessActCoord);
133
134                                Leg accessLeg = this.populationFactory.createLeg( TransportMode.access_walk ) ;
135                                accessLeg.setDepartureTime( now );
136                                now += routeBushwhackingLeg(person, accessLeg, fromFacility.getCoord(), accessActCoord, now, accessActLink.getId(), 
137                                                accessActLink.getId(), this.populationFactory ) ;
138                                // yyyy might be possible to set the link ids to null. kai & dominik, may'16
139                                
140                                result.add( accessLeg ) ;
141//                              log.warn( accessLeg );
142
143                                final Activity interactionActivity = createInteractionActivity(accessActCoord, accessActLink.getId() );
144                                result.add( interactionActivity ) ;
145//                              log.warn( interactionActivity );
146                        }
147                }
148
149                // === compute the network leg:
150                {
151                        Leg newLeg = this.populationFactory.createLeg( this.mode );
152                        newLeg.setDepartureTime( now );
153                        now += routeLeg( person, newLeg, accessActLink, egressActLink, now );
154
155                        result.add( newLeg ) ;
156//                      log.warn( newLeg );
157                }
158
159                // === egress:
160                {
161                        if ( toFacility.getCoord() != null ) { // otherwise the trip ends directly on the link; no need to bushwhack
162
163                                Coord egressActCoord = egressActLink.getToNode().getCoord() ;
164                                Gbl.assertNotNull( egressActCoord );
165
166                                final Activity interactionActivity = createInteractionActivity( egressActCoord, egressActLink.getId() );
167                                result.add( interactionActivity ) ;
168//                              log.warn( interactionActivity );
169
170                                Leg egressLeg = this.populationFactory.createLeg( TransportMode.egress_walk ) ;
171                                egressLeg.setDepartureTime( now );
172                                now += routeBushwhackingLeg(person, egressLeg, egressActCoord, toFacility.getCoord(), now, egressActLink.getId(), 
173                                                egressActLink.getId(), this.populationFactory ) ;
174                                result.add( egressLeg ) ;
175//                              log.warn( egressLeg );
176                        }
177//                      log.warn( "===" );
178                }
179
180                return result ;
181        }

183        private Link decideOnLink(final Facility fromFacility) {
184                Link accessActLink = null ;
185                if ( fromFacility.getLinkId()!=null ) {
186                        accessActLink = this.network.getLinks().get( fromFacility.getLinkId() );
187                        // i.e. if street address is in mode-specific subnetwork, I just use that, and do not search for another (possibly closer)
188                        // other link.
189                        
190                } 
191                
192                if ( accessActLink==null ) {
193                        // this is the case where the postal address link is NOT in the subnetwork, i.e. does NOT serve the desired mode,
194                        // OR the facility does not have a street address link in the first place.
195
196                        if( fromFacility.getCoord()==null ) {
197                                throw new RuntimeException("access/egress bushwhacking leg not possible when neither facility link id nor facility coordinate given") ;
198                        }
199                        
200                        accessActLink = NetworkUtils.getNearestLink(this.network, fromFacility.getCoord()) ;
201                        if ( accessActLink == null ) {
202                                int ii = 0 ;
203                                for ( Link link : this.network.getLinks().values() ) {
204                                        if ( ii==10 ) {
205                                                break ;
206                                        }
207                                        ii++ ;
208                                        log.warn( link );
209                                }
210                        }
211                        Gbl.assertNotNull(accessActLink);
212                }
213                return accessActLink;
214        }
215
216        private Activity createInteractionActivity(final Coord interactionCoord, final Id<Link> interactionLink) {
217                Activity act = PopulationUtils.createActivityFromCoordAndLinkId(this.stageActivityType, interactionCoord, interactionLink);
218                act.setMaximumDuration(0.0);
219                return act;
220        }

223        private static double routeBushwhackingLeg(Person person, Leg leg, Coord fromCoord, Coord toCoord, double depTime, 
224                        Id<Link> dpLinkId, Id<Link> arLinkId, PopulationFactory pf) {
225                // I don't think that it makes sense to use a RoutingModule for this, since that again makes assumptions about how to
226                // map facilities, and if you follow through to the teleportation routers one even finds activity wrappers, which is yet another
227                // complication which I certainly don't want here.  kai, dec'15
228
229                // dpLinkId, arLinkId need to be in Route for lots of code to function.   So I am essentially putting in the "street address"
230                // for completeness. Note that if we are walking to a parked car, this can be different from the car link id!!  kai, dec'15
231
232                // make simple assumption about distance and walking speed
233                double dist = CoordUtils.calcEuclideanDistance(fromCoord,toCoord);
234
235                // create an empty route, but with realistic travel time
236                Route route =pf.getRouteFactories().createRoute(Route.class, dpLinkId, arLinkId ); 
237
238                double beelineDistanceFactor = 1.3 ;
239                double networkTravelSpeed = 2.0 ;
240                // yyyyyy take this from config!
241
242                double estimatedNetworkDistance = dist * beelineDistanceFactor;
243                int travTime = (int) (estimatedNetworkDistance / networkTravelSpeed);
244                route.setTravelTime(travTime);
245                route.setDistance(estimatedNetworkDistance);
246                leg.setRoute(route);
247                leg.setDepartureTime(depTime);
248                leg.setTravelTime(travTime);
249                return travTime;
250        }

264        /*package (Tests)*/ double routeLeg(Person person, Leg leg, Link fromLink, Link toLink, double depTime) {
265                double travTime = 0;
266
267                Node startNode = fromLink.getToNode();  // start at the end of the "current" link
268                Node endNode = toLink.getFromNode(); // the target is the start of the link
269
270                if (toLink != fromLink) { // (a "true" route)
271
272                        Path path = this.routeAlgo.calcLeastCostPath(startNode, endNode, depTime, person, null);
273                        if (path == null) throw new RuntimeException("No route found from node " + startNode.getId() + " to node " + endNode.getId() + ".");
274
275                        NetworkRoute route = this.populationFactory.getRouteFactories().createRoute(NetworkRoute.class, fromLink.getId(), toLink.getId());
276                        route.setLinkIds(fromLink.getId(), NetworkUtils.getLinkIds(path.links), toLink.getId());
277                        route.setTravelTime((int) path.travelTime);
278                        route.setTravelCost(path.travelCost);
279                        route.setDistance(RouteUtils.calcDistance(route, 1.0,1.0,this.network));
280                        leg.setRoute(route);
281                        travTime = (int) path.travelTime;
282
283                } else {
284                        // create an empty route == staying on place if toLink == endLink
285                        // note that we still do a route: someone may drive from one location to another on the link. kai, dec'15
286                        NetworkRoute route = this.populationFactory.getRouteFactories().createRoute(NetworkRoute.class, fromLink.getId(), toLink.getId());
287                        route.setTravelTime(0);
288                        route.setDistance(0.0);
289                        leg.setRoute(route);
290                        travTime = 0;
291                }
292
293                leg.setDepartureTime(depTime);
294                leg.setTravelTime(travTime);
295
296                return travTime;
297        }
