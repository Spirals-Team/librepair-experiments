021import java.util.Arrays;
022import java.util.List;
023
024import org.matsim.api.core.v01.population.Activity;
025import org.matsim.api.core.v01.population.Leg;
026import org.matsim.api.core.v01.population.Person;
027import org.matsim.api.core.v01.population.PlanElement;
028import org.matsim.api.core.v01.population.PopulationFactory;
029import org.matsim.api.core.v01.population.Route;
030import org.matsim.core.utils.geometry.CoordUtils;
031import org.matsim.facilities.Facility;
032
033
034
035/**
036 * @author thibautd
037 */
038public class TeleportationRoutingModule implements RoutingModule {
039
040        private final String mode;
041        private final PopulationFactory populationFactory;
042
043        private final double beelineDistanceFactor;
044        private final double networkTravelSpeed;
045
046         public TeleportationRoutingModule(
047                        final String mode,
048                        final PopulationFactory populationFactory,
049                        final double networkTravelSpeed,
050                        final double beelineDistanceFactor) {
051                this.networkTravelSpeed = networkTravelSpeed;
052                this.beelineDistanceFactor = beelineDistanceFactor;
053                this.mode = mode;
054                this.populationFactory = populationFactory;
055        }
056
057        @Override
058        public List<? extends PlanElement> calcRoute(
059                        final Facility fromFacility,
060                        final Facility toFacility,
061                        final double departureTime,
062                        final Person person) {
063                Leg newLeg = this.populationFactory.createLeg( this.mode );
064                newLeg.setDepartureTime( departureTime );
065
066                double travTime = routeLeg(
067                                person,
068                                newLeg,
069                                new FacilityWrapperActivity( fromFacility ),
070                                new FacilityWrapperActivity( toFacility ),
071                                departureTime);
072
073                // otherwise, information may be lost
074                newLeg.setTravelTime( travTime );
075
076                return Arrays.asList(newLeg);
077        }
078
079        @Override
080        public StageActivityTypes getStageActivityTypes() {
081                return EmptyStageActivityTypes.INSTANCE;
082        }
083
084        @Override
085        public String toString() {
086                return "[TeleportationRoutingModule: mode="+this.mode+"]";
087        }
088
089
090        /* package */ double routeLeg(Person person, Leg leg, Activity fromAct, Activity toAct, double depTime) {
091                // make simple assumption about distance and walking speed
092                double dist = CoordUtils.calcEuclideanDistance(fromAct.getCoord(), toAct.getCoord());
093                // create an empty route, but with realistic travel time
094                Route route = this.populationFactory.getRouteFactories().createRoute(Route.class, fromAct.getLinkId(), toAct.getLinkId());
095                double estimatedNetworkDistance = dist * this.beelineDistanceFactor;
096                int travTime = (int) (estimatedNetworkDistance / this.networkTravelSpeed);
097                route.setTravelTime(travTime);
098                route.setDistance(estimatedNetworkDistance);
099                leg.setRoute(route);
100                leg.setDepartureTime(depTime);
101                leg.setTravelTime(travTime);
102                Leg r = (leg);
103                r.setTravelTime( depTime + travTime - r.getDepartureTime() ); // yy something needs to be done once there are alternative implementations of the interface.  kai, apr'10
104                return travTime;
105        }
106
107}
