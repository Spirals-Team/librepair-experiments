020package org.matsim.core.router;
021
022import java.util.ArrayList;
023import java.util.Collection;
024
025/**
026 * Implementation of a {@link StageActivityTypes} meant to aggregate the
027 * activity checkers for all modes.
028 *
029 * @author thibautd
030 */
031public class CompositeStageActivityTypes implements StageActivityTypes {
032        // set or list?
033        private final Collection<StageActivityTypes> checkers = new ArrayList<StageActivityTypes>();
034
035        // not necessary (empty varargs valid), but better for javadocs
036        /**
037         * Creates an empty instance
038         */
039        public CompositeStageActivityTypes() {}
040        
041        /**
042         * Creates an instance filled with the components passed as parameter
043         */
044        public CompositeStageActivityTypes(
045                        final StageActivityTypes... components) {
046                for ( StageActivityTypes c : components ) {
047                        addActivityTypes( c );
048                }
049        }
050
051        /**
052         * Adds a {@link StageActivityTypes} to the delegates
053         * @param checker an instance which will be used to check whether an activity
054         * type corresponds to a stage activity
055         */
056        public void addActivityTypes(final StageActivityTypes checker) {
057                if (checker instanceof EmptyStageActivityTypes) {
058                        // avoid having to loop through lots of those,
059                        // which would be useless.
060                        return;
061                }
062                checkers.add( checker );
063        }
064
065        /**
066         * Removes the first registered {@link StageActivityTypes} instance
067         * considered equal to the parameter instance
068         * @param checker the instance to use to identify the instance to remove
069         * @return true if an instance was actually removed, false otherwise.
070         * Note that this will always return true for {@link EmptyStageActivityTypes}
071         * instances.
072         */
073        public boolean removeActivityTypes(final StageActivityTypes checker) {
074                if (checker instanceof EmptyStageActivityTypes) {
075                        // those checkers are not added: consider them removed.
076                        return true;
077                }
078                return checkers.remove( checker );
079        }
080
081        /**
082         * @return true if at least one of the registered checkers considers this activity
083         * type as a stage type.
084         */
085        @Override
086        public boolean isStageActivity(final String activityType) {
087                for (StageActivityTypes checker : checkers) {
088                        if ( checker.isStageActivity( activityType ) ) {
089                                return true;
090                        }
091                }
092
093                return false;
094        }
095}
