036import org.matsim.core.router.util.RoutingNetwork;

040import org.matsim.core.router.util.TravelTime;
041
042/**
043 * @author cdobler
044 */
045@Singleton
046public class FastAStarLandmarksFactory implements LeastCostPathCalculatorFactory {
047        
048        private final RoutingNetworkFactory routingNetworkFactory;
049        private final Map<Network, RoutingNetwork> routingNetworks = new HashMap<>();
050        private final Map<Network, PreProcessLandmarks> preProcessData = new HashMap<>();
051        
052        @Inject GlobalConfigGroup globalConfig ;
053
054        @Inject
055        public FastAStarLandmarksFactory() {
056                this(FastRouterType.ARRAY);
057        }
058
059        private FastAStarLandmarksFactory(final FastRouterType fastRouterType) {
060                switch (fastRouterType) {
061                case ARRAY:
062                        this.routingNetworkFactory = new ArrayRoutingNetworkFactory();
063                        break;
064                case POINTER:
065                        throw new RuntimeException("PointerRoutingNetworks are no longer supported. Use ArrayRoutingNetworks instead. Aborting!");
066                default:
067                        throw new RuntimeException("Undefined FastRouterType: " + fastRouterType);
068                }
069        }
070
071        @Override
072        public synchronized LeastCostPathCalculator createPathCalculator(final Network network, final TravelDisutility travelCosts, final TravelTime travelTimes) {
073                RoutingNetwork routingNetwork = this.routingNetworks.get(network);
074                PreProcessLandmarks preProcessLandmarks = this.preProcessData.get(network);
075                
076                if (routingNetwork == null) {
077                        routingNetwork = this.routingNetworkFactory.createRoutingNetwork(network);
078                        
079                        if (preProcessLandmarks == null) {
080                                preProcessLandmarks = new PreProcessLandmarks(travelCosts);
081                                if ( globalConfig==null ) {
082                                        preProcessLandmarks.setNumberOfThreads(8);
083                                        // (if used without injection.  not so beautiful. kai, nov'17)
084                                } else {
085                                        preProcessLandmarks.setNumberOfThreads(globalConfig.getNumberOfThreads());
086                                }
087                                preProcessLandmarks.run(network);
088                                this.preProcessData.put(network, preProcessLandmarks);
089                                
090                                for (RoutingNetworkNode node : routingNetwork.getNodes().values()) {
091                                        node.setDeadEndData(preProcessLandmarks.getNodeData(node.getNode()));
092                                }
093                        }                               
094                        
095                        this.routingNetworks.put(network, routingNetwork);
096                }
097                FastRouterDelegateFactory fastRouterFactory = new ArrayFastRouterDelegateFactory();
098                
099                final double overdoFactor = 1.0;
100                return new FastAStarLandmarks(routingNetwork, preProcessLandmarks, travelCosts, travelTimes, overdoFactor, fastRouterFactory);
101        }
102}
