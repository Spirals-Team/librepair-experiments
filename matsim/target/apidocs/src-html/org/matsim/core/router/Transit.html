001package org.matsim.core.router;
002
003import com.google.inject.name.Named;
004import org.matsim.api.core.v01.Scenario;
005import org.matsim.api.core.v01.TransportMode;
006import org.matsim.pt.router.TransitRouter;
007
008import javax.inject.Inject;
009import javax.inject.Provider;
010
011public class Transit implements Provider<RoutingModule> {
012
013        private final TransitRouter transitRouter;
014
015        private final Scenario scenario;
016
017        private final RoutingModule transitWalkRouter;
018
019        @Inject
020    Transit(TransitRouter transitRouter, Scenario scenario, @Named(TransportMode.transit_walk) RoutingModule transitWalkRouter) {
021                this.transitRouter = transitRouter;
022                this.scenario = scenario;
023                this.transitWalkRouter = transitWalkRouter;
024        }
025
026        @Override
027        public RoutingModule get() {
028                return new TransitRouterWrapper(transitRouter,
029                                        scenario.getTransitSchedule(),
030                                        scenario.getNetwork(),
031                                        transitWalkRouter);
032        }
033}
