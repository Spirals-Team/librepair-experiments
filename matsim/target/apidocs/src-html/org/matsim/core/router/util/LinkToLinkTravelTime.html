020package org.matsim.core.router.util;
021
022import org.matsim.api.core.v01.network.Link;
023
024
025
026/**
027 * Interface for travel time queries that include the link travel time and the
028 * time needed for the turning move.
029 * @author dgrether
030 *
031 */
032public interface LinkToLinkTravelTime {
033
034        /**
035         * Returns the travel time at the specified time for the given fromLink inclusive the 
036         * turning move time to enter the given toLink.
037         *
038         * @param fromLink The link for which the travel time is calculated.
039         * @param toLink The link that is entered from the fromLink the turning
040         * move time is calculated for the fromLink toLink relationship.
041         * @param time The departure time (in seconds since 00:00) at the beginning
042         *              of the fromLink for which the travel time is calculated.
043         * @return The time (in seconds) needed to travel over the fromLink
044         *              <code>fromLink</code> and enter the toLink <code>toLink</code>, 
045         * departing at time <code>time</code>.
046         */
047        public double getLinkToLinkTravelTime(Link fromLink, Link toLink, double time);
048
049}
