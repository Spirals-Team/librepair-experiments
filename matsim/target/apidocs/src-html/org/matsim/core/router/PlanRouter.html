037/**
038 * {@link PlanAlgorithm} responsible for routing all trips of a plan.
039 * Activity times are not updated, even if the previous trip arrival time
040 * is after the activity end time.
041 *
042 * @author thibautd
043 */
044public class PlanRouter implements PlanAlgorithm, PersonAlgorithm {
045        private final TripRouter routingHandler;
046        private final ActivityFacilities facilities;
047
048        /**
049         * Initialises an instance.
050         * @param routingHandler the {@link TripRouter} to use to route individual trips
051         * @param facilities the {@link ActivityFacilities} to which activities are refering.
052         * May be <tt>null</tt>: in this case, the router will be given facilities wrapping the
053         * origin and destination activity.
054         */
055        public PlanRouter(
056                        final TripRouter routingHandler,
057                        final ActivityFacilities facilities) {
058                this.routingHandler = routingHandler;
059                this.facilities = facilities;
060        }
061
062        /**
063         * Short for initialising without facilities.
064         */
065        public PlanRouter(
066                        final TripRouter routingHandler) {
067                this( routingHandler , null );
068        }
069
070        /**
071         * Gives access to the {@link TripRouter} used
072         * to compute routes.
073         *
074         * @return the internal TripRouter instance.
075         */
076        @Deprecated // get TripRouter out of injection instead. kai, feb'16
077        public TripRouter getTripRouter() {
078                return routingHandler;
079        }
080
081        @Override
082        public void run(final Plan plan) {
083                final List<Trip> trips = TripStructureUtils.getTrips( plan , routingHandler.getStageActivityTypes() );
084
085                for (Trip oldTrip : trips) {
086                        final List<? extends PlanElement> newTrip =
087                                        routingHandler.calcRoute(
088                                                        routingHandler.getMainModeIdentifier().identifyMainMode( oldTrip.getTripElements() ),
089                                                        toFacility( oldTrip.getOriginActivity() ),
090                                                        toFacility( oldTrip.getDestinationActivity() ),
091                                                        calcEndOfActivity( oldTrip.getOriginActivity() , plan, routingHandler.getConfig() ),
092                                                        plan.getPerson() );
093                        putVehicleFromOldTripIntoNewTripIfMeaningful(oldTrip, newTrip);
094                        TripRouter.insertTrip(
095                                        plan, 
096                                        oldTrip.getOriginActivity(),
097                                        newTrip,
098                                        oldTrip.getDestinationActivity());
099                }
100        }
101
102        /**
103         * If the old trip had vehicles set in its network routes, and it used a single vehicle,
104         * and if the new trip does not come with vehicles set in its network routes,
105         * then put the vehicle of the old trip into the network routes of the new trip.
106         * @param oldTrip The old trip
107         * @param newTrip The new trip
108         */
109        private static void putVehicleFromOldTripIntoNewTripIfMeaningful(Trip oldTrip, List<? extends PlanElement> newTrip) {
110                Id<Vehicle> oldVehicleId = getUniqueVehicleId(oldTrip);
111                if (oldVehicleId != null) {
112                        for (Leg leg : TripStructureUtils.getLegs(newTrip)) {
113                                if (leg.getRoute() instanceof NetworkRoute) {
114                                        if (((NetworkRoute) leg.getRoute()).getVehicleId() == null) {
115                                                ((NetworkRoute) leg.getRoute()).setVehicleId(oldVehicleId);
116                                        }
117                                }
118                        }
119                }
120        }
121
122        private static Id<Vehicle> getUniqueVehicleId(Trip trip) {
123                Id<Vehicle> vehicleId = null;
124                for (Leg leg : trip.getLegsOnly()) {
125                        if (leg.getRoute() instanceof NetworkRoute) {
126                                if (vehicleId != null && (!vehicleId.equals(((NetworkRoute) leg.getRoute()).getVehicleId()))) {
127                                        return null; // The trip uses several vehicles.
128                                }
129                                vehicleId = ((NetworkRoute) leg.getRoute()).getVehicleId();
130                        }
131                }
132                return vehicleId;
133        }
134
135        @Override
136        public void run(final Person person) {
137                for (Plan plan : person.getPlans()) {
138                        run( plan );
139                }
140        }

145        private Facility toFacility(final Activity act) {
146                if (  (act.getLinkId() == null && act.getCoord() == null)  // yyyy this used to be || instead of && --???  kai, jun'16
147                                && facilities != null
148                                && !facilities.getFacilities().isEmpty()) {
149                        // use facilities only if the activity does not provide the required fields.
150                        // yyyyyy Seems to me that the Access/EgressRoutingModule only needs either link or coord to start from.  So we only go
151                        // to facilities if neither is provided.  --  This may, however, be at odds of how it is done in the AccessEgressRoutingModule, so we 
152                        // need to conceptually sort this out!!  kai, jun'16
153                        return facilities.getFacilities().get( act.getFacilityId() );
154                }
155                return new ActivityWrapperFacility( act );
156        }

158        public static double calcEndOfActivity(
159                        final Activity activity,
160                        final Plan plan,
161                        final Config config ) {
162                
163                if (activity.getEndTime() != Time.UNDEFINED_TIME) return activity.getEndTime();
164
165                // no sufficient information in the activity...
166                // do it the long way.
167                // XXX This is inefficient! Using a cache for each plan may be an option
168                // (knowing that plan elements are iterated in proper sequence,
169                // no need to re-examine the parts of the plan already known)
170                double now = 0;
171
172                for (PlanElement pe : plan.getPlanElements()) {
173                        now = updateNow( now , pe, config );
174                        if (pe == activity) return now;
175                }
176
177                throw new RuntimeException( "activity "+activity+" not found in "+plan.getPlanElements() );
178        }
