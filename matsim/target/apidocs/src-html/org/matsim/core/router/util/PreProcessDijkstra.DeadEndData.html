021package org.matsim.core.router.util;
022
023import java.util.ArrayList;
024import java.util.Iterator;
025import java.util.Map;
026import java.util.TreeMap;
027import java.util.concurrent.ConcurrentHashMap;
028
029import org.apache.log4j.Logger;
030import org.matsim.api.core.v01.Id;
031import org.matsim.api.core.v01.network.Link;
032import org.matsim.api.core.v01.network.Network;
033import org.matsim.api.core.v01.network.Node;
034
035/**
036 * Pre-processes a given network, gathering information which
037 * can be used by a Dijkstra when computing least-cost paths
038 * between a start and an end node. Specifically, marks the nodes
039 * that are in dead-end, i.e. that are connected to the rest of
040 * the network by a single 'entry' node only.
041 *
042 * @author lnicolas
043 */
044public class PreProcessDijkstra {
045
046        private static final Logger log = Logger.getLogger(PreProcessDijkstra.class);
047
048        private boolean containsData = false;
049
050        protected Map<Node, DeadEndData> nodeData = null;
051
052        public void run(final Network network) {
053                markDeadEnds(network);
054                this.containsData = true;
055        }
056
057        /**
058         * Marks nodes that are in dead ends (i.e. nodes that are connected through a single node to the rest of the network), starting at all nodes with degree 2.
059         * @param network The network on which to process.
060         */
061        private void markDeadEnds(final Network network) {
062                long now = System.currentTimeMillis();
063
064                /*
065                 * We use a concurrentHashMap because if FastRouters are used for the re-routing,
066                 * their parallel initialization in multiple threads may result in concurrent
067                 * calls to getNodeData(...).
068                 */
069                this.nodeData = new ConcurrentHashMap<Node, DeadEndData>(network.getNodes().size());
070
071                DeadEndData deadEndData;
072                for (Node node : network.getNodes().values()) {
073                        deadEndData = getNodeData(node);
074
075                        Map<Id<Node>, Node> incidentNodes = getIncidentNodes(node);
076                        if (incidentNodes.size() == 1) {
077                                ArrayList<Node> deadEndNodes = new ArrayList<Node>();
078
079                                while (deadEndData.getInDeadEndCount() == incidentNodes.size() - 1) {
080
081                                        deadEndNodes.add(node);
082                                        deadEndNodes.addAll(deadEndData.getDeadEndNodes());
083                                        deadEndData.getDeadEndNodes().clear();
084
085                                        // Just set a dummy DeadEndEntryNode, such that is
086                                        // isn't null. We use this as a flag to determine
087                                        // whether we already processed this node.
088                                        deadEndData.setDeadEndEntryNode(node);
089
090                                        Iterator<? extends Node> it = incidentNodes.values().iterator();
091                                        while (deadEndData.getDeadEndEntryNode() != null && it.hasNext()) {
092                                                node = it.next();
093                                                deadEndData = getNodeData(node);
094                                        }
095                                        if (deadEndData.getDeadEndEntryNode() == null) {
096                                                deadEndData.incrementInDeadEndCount();
097                                                incidentNodes = getIncidentNodes(node);
098                                        } else {
099                                                log.error("All " + incidentNodes.size() + " incident nodes of node " + node.getId() + " are dead ends!");
100                                                return;
101                                        }
102                                }
103                                deadEndData.getDeadEndNodes().addAll(deadEndNodes);
104                        }
105                }
106
107                // Now set the proper deadEndEntryNode for each node
108                int deadEndNodeCount = 0;
109                for (Node node : network.getNodes().values()) {
110                        deadEndData = getNodeData(node);
111                        for (Node n : deadEndData.getDeadEndNodes()) {
112                                DeadEndData r = getNodeData(n);
113                                r.setDeadEndEntryNode(node);
114                                deadEndNodeCount++;
115                        }
116                        deadEndData.getDeadEndNodes().clear();
117                }
118                log.info("nodes in dead ends: " + deadEndNodeCount
119                                + " (total nodes: " + network.getNodes().size() + "). Done in "
120                                + (System.currentTimeMillis() - now) + " ms");
121        }
122
123        private static Map<Id<Node>, Node> getIncidentNodes(Node node) {
124                Map<Id<Node>, Node> nodes = new TreeMap<>();
125                for (Link link : node.getInLinks().values()) {
126                        nodes.put(link.getFromNode().getId(), link.getFromNode());
127                }
128                for (Link link : node.getOutLinks().values()) {
129                        nodes.put(link.getToNode().getId(), link.getToNode());
130                }
131                return nodes;
132        }
133
134        /**
135         * Returns the role for the given Node. Creates a new Role if none exists yet.
136         * @param n The Node for which to create a role.
137         * @return The role for the given Node
138         */
139        public DeadEndData getNodeData(final Node n) {
140                DeadEndData r = this.nodeData.get(n);
141                if (null == r) {
142                        r = new DeadEndData();
143                        this.nodeData.put(n, r);
144                }
145                return r;
146        }
147
148        /**
149         * Contains information whether the associated node is in a dead end.
150         * @author lnicolas
151         */
152        public static class DeadEndData {
153                private Node deadEndEntryNode = null;
154
155                private int inDeadEndCount = 0;
156
157                private ArrayList<Node> deadEndNodes = new ArrayList<Node>(2);
158
159                ArrayList<Node> getDeadEndNodes() {
160                        return this.deadEndNodes;
161                }
162
163                /**
164                 * @return the inDeadEndCount
165                 */
166                /*package*/ int getInDeadEndCount() {
167                        return this.inDeadEndCount;
168                }
169
170                /**
171                 * @param inDeadEndCount
172                 *            the inDeadEndCount to set
173                 */
174                /*package*/ void incrementInDeadEndCount() {
175                        this.inDeadEndCount++;
176                }
177
178
179                /**
180                 * @return The node that connects the associated node to the rest
181                 * of the network (i.e. without the deadEndEntryNode, the associated
182                 * node would not be connected to the network) and null otherwise
183                 * (i.e. if the associated node is not in a dead end).
184                 */
185                public Node getDeadEndEntryNode() {
186                        return this.deadEndEntryNode;
187                }
188
189                /*package*/ void setDeadEndEntryNode(final Node node) {
190                        this.deadEndEntryNode = node;
191                }
192        }
193
194        public boolean containsData() {
195                return this.containsData;
196        }
197}
