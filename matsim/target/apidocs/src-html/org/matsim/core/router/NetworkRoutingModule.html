039/**
040 * This wraps a "computer science" {@link LeastCostPathCalculator}, which routes from a node to another node, into something that
041 * routes from a {@link Facility} to another {@link Facility}, as we need in MATSim.
042 * 
043 * @author thibautd
044 */
045public final class NetworkRoutingModule implements RoutingModule {
046        // I think it makes sense to NOT add the bushwhacking mode directly into here ...
047        // ... since it makes sense be able to to route from facility.getLinkId() to facility.getLinkId(). kai, dec'15
048
049        private final String mode;
050        private final PopulationFactory populationFactory;
051
052        private final Network network;
053        private final LeastCostPathCalculator routeAlgo;
054
055
056         public NetworkRoutingModule(
057                        final String mode,
058                        final PopulationFactory populationFactory,
059                        final Network network,
060                        final LeastCostPathCalculator routeAlgo) {
061                 Gbl.assertNotNull(network);
062//               Gbl.assertIf( network.getLinks().size()>0 ) ; // otherwise network for mode probably not defined
063                 // makes many tests fail.  
064                 this.network = network;
065                 this.routeAlgo = routeAlgo;
066                 this.mode = mode;
067                 this.populationFactory = populationFactory;
068        }
069
070        @Override
071        public List<? extends PlanElement> calcRoute(final Facility<?> fromFacility, final Facility<?> toFacility, final double departureTime,
072                        final Person person) {          
073                Leg newLeg = this.populationFactory.createLeg( this.mode );
074                
075                Gbl.assertNotNull(fromFacility);
076                Gbl.assertNotNull(toFacility);
077
078                Link fromLink = this.network.getLinks().get(fromFacility.getLinkId());
079                if ( fromLink==null ) {
080                        Gbl.assertNotNull( fromFacility.getCoord() ) ;
081                        fromLink = NetworkUtils.getNearestLink( network, fromFacility.getCoord()) ;
082                }
083                Link toLink = this.network.getLinks().get(toFacility.getLinkId());
084                if ( toLink==null ) {
085                        Gbl.assertNotNull( toFacility.getCoord() ) ;
086                        toLink = NetworkUtils.getNearestLink(network, toFacility.getCoord());
087                }
088                Gbl.assertNotNull(fromLink);
089                Gbl.assertNotNull(toLink);
090                
091                if (toLink != fromLink) {
092                        // (a "true" route)
093                        Node startNode = fromLink.getToNode(); // start at the end of the "current" link
094                        Node endNode = toLink.getFromNode(); // the target is the start of the link
095                        Path path = this.routeAlgo.calcLeastCostPath(startNode, endNode, departureTime, person, null);
096                        if (path == null)
097                                throw new RuntimeException("No route found from node " + startNode.getId() + " to node " + endNode.getId() + ".");
098                        NetworkRoute route = this.populationFactory.getRouteFactories().createRoute(NetworkRoute.class, fromLink.getId(), toLink.getId());
099                        route.setLinkIds(fromLink.getId(), NetworkUtils.getLinkIds(path.links), toLink.getId());
100                        route.setTravelTime(path.travelTime);
101                        route.setTravelCost(path.travelCost);
102                        route.setDistance(RouteUtils.calcDistance(route, 1.0, 1.0, this.network));
103                        newLeg.setRoute(route);
104                        newLeg.setTravelTime(path.travelTime);
105                } else {
106                        // create an empty route == staying on place if toLink == endLink
107                        // note that we still do a route: someone may drive from one location to another on the link. kai, dec'15
108                        NetworkRoute route = this.populationFactory.getRouteFactories().createRoute(NetworkRoute.class, fromLink.getId(), toLink.getId());
109                        route.setTravelTime(0);
110                        route.setDistance(0.0);
111                        newLeg.setRoute(route);
112                        newLeg.setTravelTime(0);
113                }
114                newLeg.setDepartureTime(departureTime);
115
116                return Arrays.asList( newLeg );
117        }
118
119        @Override
120        public StageActivityTypes getStageActivityTypes() {
121                return EmptyStageActivityTypes.INSTANCE;
122        }
123
124        @Override
125        public String toString() {
126                return "[NetworkRoutingModule: mode="+this.mode+"]";
127        }
128
129}
