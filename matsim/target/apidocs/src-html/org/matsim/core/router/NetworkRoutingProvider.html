001package org.matsim.core.router;
002
003import org.apache.log4j.Level;
004import org.apache.log4j.Logger;
005import org.matsim.api.core.v01.network.Network;
006import org.matsim.api.core.v01.population.PopulationFactory;
007import org.matsim.core.config.groups.PlansCalcRouteConfigGroup;
008import org.matsim.core.network.NetworkUtils;
009import org.matsim.core.network.algorithms.TransportModeNetworkFilter;
010import org.matsim.core.router.costcalculators.TravelDisutilityFactory;
011import org.matsim.core.router.util.LeastCostPathCalculator;
012import org.matsim.core.router.util.LeastCostPathCalculatorFactory;
013import org.matsim.core.router.util.TravelTime;
014
015import javax.inject.Inject;
016import javax.inject.Provider;
017import java.util.HashSet;
018import java.util.Map;
019import java.util.Set;
020
021public class NetworkRoutingProvider implements Provider<RoutingModule> {
022        private static final Logger log = Logger.getLogger( NetworkRoutingProvider.class ) ;
023        
024        private final String routingMode;
025        @Inject
026    Map<String, TravelTime> travelTimes;
027
028        @Inject
029        Map<String, TravelDisutilityFactory> travelDisutilityFactories;
030
031        @Inject
032        SingleModeNetworksCache singleModeNetworksCache;
033
034        @Inject
035        PlansCalcRouteConfigGroup plansCalcRouteConfigGroup;
036
037        @Inject
038    Network network;
039
040        @Inject
041    PopulationFactory populationFactory;
042
043        @Inject
044    LeastCostPathCalculatorFactory leastCostPathCalculatorFactory;
045        
046        /**
047         * This is the older (and still more standard) constructor, where the routingMode and the resulting mode were the
048         * same.
049         *
050         * @param mode
051         */
052        public NetworkRoutingProvider(String mode) {
053                this( mode, mode ) ;
054        }
055        
056        /**
057         * The effect of this constructor is a router configured for "routingMode" will be used for routing, but the route
058         * will then have the mode "mode".   So one can, for example, have an uncongested and a congested within-day router,
059         * for travellers who first might be unaware, but then switch on some help, and the both produce a route of type "car".
060         *
061         * @param mode
062         * @param routingMode
063         */
064        public NetworkRoutingProvider(String mode, String routingMode ) {
065//              log.setLevel(Level.DEBUG);
066                
067                this.mode = mode;
068                this.routingMode = routingMode ;
069        }
070
071        private final String mode;
072
073        @Override
074        public RoutingModule get() {
075                log.debug( "requesting network routing module with routingMode="
076                                                   + routingMode + ";\tmode=" + mode) ;
077                
078                // the network refers to the (transport)mode:
079                Network filteredNetwork = null;
080
081                // Ensure this is not performed concurrently by multiple threads!
082                synchronized (this.singleModeNetworksCache.getSingleModeNetworksCache()) {
083                        filteredNetwork = this.singleModeNetworksCache.getSingleModeNetworksCache().get(mode);
084                        if (filteredNetwork == null) {
085                                TransportModeNetworkFilter filter = new TransportModeNetworkFilter(network);
086                                Set<String> modes = new HashSet<>();
087                                modes.add(mode);
088                                filteredNetwork = NetworkUtils.createNetwork();
089                                filter.filter(filteredNetwork, modes);
090                                this.singleModeNetworksCache.getSingleModeNetworksCache().put(mode, filteredNetwork);
091                        }
092                }
093
094                // the travel time & disutility refer to the routing mode:
095                TravelDisutilityFactory travelDisutilityFactory = this.travelDisutilityFactories.get(routingMode);
096                if (travelDisutilityFactory == null) {
097                        throw new RuntimeException("No TravelDisutilityFactory bound for mode "+routingMode+".");
098                }
099                TravelTime travelTime = travelTimes.get(routingMode);
100                if (travelTime == null) {
101                        throw new RuntimeException("No TravelTime bound for mode "+routingMode+".");
102                }
103                LeastCostPathCalculator routeAlgo =
104                                leastCostPathCalculatorFactory.createPathCalculator(
105                                                filteredNetwork,
106                                                travelDisutilityFactory.createTravelDisutility(travelTime),
107                                                travelTime);
108
109                // the following again refers to the (transport)mode, since it will determine the mode of the leg on the network:
110                if ( plansCalcRouteConfigGroup.isInsertingAccessEgressWalk() ) {
111                        return DefaultRoutingModules.createAccessEgressNetworkRouter(mode, populationFactory, filteredNetwork, routeAlgo,
112                                        plansCalcRouteConfigGroup) ;
113                } else {
114                        return DefaultRoutingModules.createPureNetworkRouter(mode, populationFactory, filteredNetwork, routeAlgo);
115                }
116        }
117}
