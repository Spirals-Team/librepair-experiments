021package org.matsim.core.router;
022
023import org.matsim.api.core.v01.network.Network;
024import org.matsim.core.router.util.ArrayRoutingNetworkFactory;
025import org.matsim.core.router.util.LeastCostPathCalculator;
026import org.matsim.core.router.util.LeastCostPathCalculatorFactory;
027import org.matsim.core.router.util.PreProcessEuclidean;
028import org.matsim.core.router.util.RoutingNetwork;
029import org.matsim.core.router.util.RoutingNetworkFactory;
030import org.matsim.core.router.util.RoutingNetworkNode;
031import org.matsim.core.router.util.TravelDisutility;
032import org.matsim.core.router.util.TravelTime;
033
034import java.util.HashMap;
035import java.util.Map;
036
037/**
038 * @author cdobler
039 */
040public class FastAStarEuclideanFactory implements LeastCostPathCalculatorFactory {
041
042        private final RoutingNetworkFactory routingNetworkFactory;
043        private final Map<Network, RoutingNetwork> routingNetworks = new HashMap<>();
044        private final Map<Network, PreProcessEuclidean> preProcessData = new HashMap<>();
045        private final double overdoFactor;
046
047        public FastAStarEuclideanFactory() {
048                this(1);                
049        }
050        
051        public FastAStarEuclideanFactory(double overdoFactor) {
052                this(FastRouterType.ARRAY, overdoFactor);
053        }
054
055        private FastAStarEuclideanFactory(final FastRouterType fastRouterType, double overdoFactor) {
056                this.overdoFactor = overdoFactor;
057                switch (fastRouterType) {
058                case ARRAY:
059                        this.routingNetworkFactory = new ArrayRoutingNetworkFactory();
060                        break;
061                case POINTER:
062                        throw new RuntimeException("PointerRoutingNetworks are no longer supported. Use ArrayRoutingNetworks instead. Aborting!");
063                default:
064                        throw new RuntimeException("Undefined FastRouterType: " + fastRouterType);
065                }
066        }
067
068        @Override
069        public synchronized LeastCostPathCalculator createPathCalculator(final Network network, final TravelDisutility travelCosts, final TravelTime travelTimes) {
070                RoutingNetwork routingNetwork = this.routingNetworks.get(network);
071                PreProcessEuclidean preProcessEuclidean = this.preProcessData.get(network);
072                
073                if (routingNetwork == null) {
074                        routingNetwork = this.routingNetworkFactory.createRoutingNetwork(network);
075                        
076                        if (preProcessEuclidean == null) {
077                                preProcessEuclidean = new PreProcessEuclidean(travelCosts);
078                                preProcessEuclidean.run(network);
079                                this.preProcessData.put(network, preProcessEuclidean);
080
081                                for (RoutingNetworkNode node : routingNetwork.getNodes().values()) {
082                                        node.setDeadEndData(preProcessEuclidean.getNodeData(node.getNode()));
083                                }
084                        }                               
085                        
086                        this.routingNetworks.put(network, routingNetwork);
087                }
088                FastRouterDelegateFactory fastRouterFactory = new ArrayFastRouterDelegateFactory();
089                
090                return new FastAStarEuclidean(routingNetwork, preProcessEuclidean, travelCosts, travelTimes, overdoFactor, fastRouterFactory);
091        }
092}
