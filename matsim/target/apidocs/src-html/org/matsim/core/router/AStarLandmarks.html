038/**
039 * Implements the A* router algorithm for a given NetworkLayer
040 * using some so called landmark nodes that are used to get a
041 * better estimation of the remaining travel cost. This
042 * way we can accelerate the routing speed, and the routes we find
043 * are still guaranteed to be the best ones with respect to the travel cost.
044 *
045 * <p>For every router, there exists a class which computes some
046 * preprocessing data and is passed to the router class
047 * constructor in order to accelerate the routing procedure.
048 * The one used for AStarLandmarks is org.matsim.demandmodeling.router.util.PreProcessLandmarks.<br>
049 *
050 * AStarLandmarks is about double as fast as AStarEuclidean. PreProcessLandmarks.run() takes
051 * about 1 minute on 400'000 nodes on a AMD Opteron processor 275 with 2.2GHz and
052 * requires 2*X double values per node, where X is the number of landmarks. Currently, it is
053 * set to 16 (but can be set to another value, as 12 or 8 for example), so with 400'000
054 * nodes we would need 2*8*16*400'000 bytes = about 100MB of additional memory.<br>
055 * Conditions: The same as for AStarEuclidean.<br>
056 * Code example:<br>
057 * <code><br> TravelMinCost costFunction = ...<br>
058 * PreProcessLandmarks preProcessData = new PreProcessLandmarks(costFunction);<br>
059 * preProcessData.run(network);<br>...<br>
060 * LeastCostPathCalculator routingAlgo = new AStarLandmarks(network, preProcessData);
061 * <br>...</code></p>
062 * @see org.matsim.core.router.AStarEuclidean
063 * @see org.matsim.core.router.util.PreProcessLandmarks
064 * @see org.matsim.core.router.Dijkstra
065 * @author lnicolas
066 */
067public class AStarLandmarks extends AStarEuclidean {
068
069        protected int[] activeLandmarkIndexes;
070
071        protected final Node[] landmarks;
072
073        /*package*/ static final int controlInterval = 40;
074        /*package*/ int controlCounter = 0;
075
076        /**
077         * Default constructor; sets the overdo factor to 1.
078         * @param network Where we do the routing.
079         * @param preProcessData The pre-process data (containing the landmarks etc.).
080         * @param timeFunction Calculates the travel time on links.
081         * @param costFunction Calculates the travel cost on links.
082         */
083        AStarLandmarks(final Network network, final PreProcessLandmarks preProcessData,
084                        final TravelDisutility costFunction, final TravelTime timeFunction) {
085                this(network, preProcessData, costFunction, timeFunction, 1);
086        }
087
088        /**
089         * Default constructor; sets the overdo factor to 1.
090         * @param network Where we do the routing.
091         * @param preProcessData The pre-process data (containing the landmarks etc.).
092         * @param timeFunction Calculates the travel time on links.
093         */
094        AStarLandmarks(final Network network, final PreProcessLandmarks preProcessData, final TravelTime timeFunction) {
095                this(network, preProcessData, preProcessData.getCostFunction(), timeFunction, 1);
096        }
097
098        /**
099         * @param network Where we do the routing.
100         * @param preProcessData The pre-process data (containing the landmarks etc.).
101         * @param costFunction
102         * @param timeFunction Calculates the travel time on links.
103         * @param overdoFactor The factor which is multiplied with the output of the A* heuristic function. 
104         * The higher the overdo factor the greedier the router, i.e. it visits less nodes during routing and 
105         * is thus faster, but for an overdo factor > 1, it is not guaranteed that the router returns the 
106         * least-cost paths. Rather it tends to return distance-minimal paths.
107         * @see #AStarLandmarks(Network, PreProcessLandmarks, TravelTime)
108         */
109        AStarLandmarks(final Network network, final PreProcessLandmarks preProcessData,
110                        final TravelDisutility costFunction, final TravelTime timeFunction, final double overdoFactor) {
111                super(network, preProcessData, costFunction, timeFunction, overdoFactor);
112
113                this.landmarks = preProcessData.getLandmarks();
114        }
115
116        @Override
117        public Path calcLeastCostPath(final Node fromNode, final Node toNode, final double startTime, final Person person, final Vehicle vehicle) {
118                this.controlCounter = 0;        // reset counter for each calculated path!
119                
120                if (this.landmarks.length >= 2) {
121                        initializeActiveLandmarks(fromNode, toNode, 2);
122                } else {
123                        initializeActiveLandmarks(fromNode, toNode, this.landmarks.length);
124                }
125                return super.calcLeastCostPath(fromNode, toNode, startTime, person, vehicle);
126        }
127
128        @Override
129        protected void relaxNode(final Node outNode, final Node toNode, final RouterPriorityQueue<Node> pendingNodes) {
130                this.controlCounter++;
131                if (this.controlCounter == controlInterval) {
132                        int newLandmarkIndex = checkToAddLandmark(outNode, toNode);
133                        if (newLandmarkIndex > 0) {
134                                updatePendingNodes(newLandmarkIndex, toNode, pendingNodes);
135                        }
136                        this.controlCounter = 0;
137                }
138                super.relaxNode(outNode, toNode, pendingNodes);
139        }
140
141        /**
142         * Inspect all landmarks and determines the actLandmarkCount best ones that will be used for routing (so-called active landmarks).
143         * 
144         * @param fromNode The node for which we estimate the travel time to the toNode in order to rank the landmarks.
145         * @param toNode The node to which we estimate the travel time from the fromNode in order to rank the landmarks.
146         * @param actLandmarkCount The number of active landmarks landmarks to set.
147         */
148        /*package*/ void initializeActiveLandmarks(final Node fromNode, final Node toNode, final int actLandmarkCount) {
149                final PreProcessLandmarks.LandmarksData fromData = getPreProcessData(fromNode);
150                final PreProcessLandmarks.LandmarksData toData = getPreProcessData(toNode);
151
152                // Sort the landmarks according to the accuracy of their distance estimation they yield.
153                double[] estTravelTimes = new double[actLandmarkCount];
154                this.activeLandmarkIndexes = new int[actLandmarkCount];
155                for (int i = 0; i < estTravelTimes.length; i++) {
156                        estTravelTimes[i] = Time.UNDEFINED_TIME;
157                }
158                double tmpTravTime;
159                for (int i = 0; i < this.landmarks.length; i++) {
160                        tmpTravTime = estimateRemainingTravelCost(fromData, toData, i);
161                        for (int j = 0; j < estTravelTimes.length; j++) {
162                                if (tmpTravTime > estTravelTimes[j]) {
163                                        for (int k = estTravelTimes.length - 1; k > j; k--) {
164                                                estTravelTimes[k] = estTravelTimes[k - 1];
165                                                this.activeLandmarkIndexes[k] = this.activeLandmarkIndexes[k - 1];
166                                        }
167                                        estTravelTimes[j] = tmpTravTime;
168                                        this.activeLandmarkIndexes[j] = i;
169                                        break;
170                                }
171                        }
172                }
173        }
174
175        @Override
176        protected PreProcessLandmarks.LandmarksData getPreProcessData(final Node n) {
177                return (PreProcessLandmarks.LandmarksData) super.getPreProcessData(n);
178        }
179
180        /**
181         * Estimates the remaining travel cost from fromNode to toNode using the landmarks on the network.
182         * 
183         * @param fromNode The first node.
184         * @param toNode The second node.
185         * @return The travel cost when traveling between the two given nodes.
186         */
187        @Override
188        protected double estimateRemainingTravelCost(final Node fromNode, final Node toNode) {
189
190                PreProcessLandmarks.LandmarksData fromRole = getPreProcessData(fromNode);
191                PreProcessLandmarks.LandmarksData toRole = getPreProcessData(toNode);
192                double tmpTravCost;
193                double travCost = 0;
194                for (int i = 0, n = this.activeLandmarkIndexes.length; i < n; i++) {
195                        tmpTravCost = estimateRemainingTravelCost(fromRole, toRole, this.activeLandmarkIndexes[i]);
196                        if (tmpTravCost > travCost) {
197                                travCost = tmpTravCost;
198                        }
199                }
200                tmpTravCost = super.estimateRemainingTravelCost(fromNode, toNode);
201                if (travCost > tmpTravCost) {
202                        return travCost;
203                }
204                /* else */
205                return tmpTravCost;
206        }
207
208        /**
209         * If a landmark has been added to the set of the active landmarks, this function re-evaluates the estimated 
210         * remaining travel time based on the new set of active landmarks of the nodes contained in pendingNodes. 
211         * If this estimation improved, the node's position in the pendingNodes queue is updated.
212         * 
213         * @param newLandmarkIndex The index of the new active landmark.
214         * @param toNode The target node of the route to be calculated.
215         * @param pendingNodes The nodes visited so far.
216         */
217        /*package*/ void updatePendingNodes(final int newLandmarkIndex, final Node toNode, final RouterPriorityQueue<Node> pendingNodes) {
218                Iterator<Node> it = pendingNodes.iterator();
219                PreProcessLandmarks.LandmarksData toRole = getPreProcessData(toNode);
220                List<Double> newEstRemTravCosts = new ArrayList<>();
221                List<Node> nodesToBeUpdated = new ArrayList<>();
222                while (it.hasNext()) {
223                        Node node = it.next();
224                        AStarNodeData data = getData(node);
225                        PreProcessLandmarks.LandmarksData ppRole = getPreProcessData(node);
226                        double estRemTravCost = data.getExpectedRemainingCost();
227                        double newEstRemTravCost = estimateRemainingTravelCost(ppRole, toRole, newLandmarkIndex);
228                        if (newEstRemTravCost > estRemTravCost) {
229                                nodesToBeUpdated.add(node);
230                                newEstRemTravCosts.add(newEstRemTravCost);
231                        }
232                }
233                for (Node node : nodesToBeUpdated) {
234                        pendingNodes.remove(node);
235                }
236                for (int i = 0; i < nodesToBeUpdated.size(); i++) {
237                        Node node = nodesToBeUpdated.get(i);
238                        AStarNodeData data = getData(node);
239                        data.setExpectedRemainingCost(newEstRemTravCosts.get(i));
240                        pendingNodes.add(node, getPriority(data));
241                }
242        }
243
244        /**
245         * Checks whether there is a landmark in the set of the non-active landmarks that yields a better estimation than 
246         * the best active landmark. If there is, this landmark is added to the set of active landmark and its index is returned.
247         * 
248         * @param fromNode The node for which we estimate the remaining travel time to the toNode.
249         * @param toNode The target node. 
250         * @return The index of the landmark that has been added to the set of active landmarks, or -1 if no landmark was added.
251         */
252        /*package*/ int checkToAddLandmark(final Node fromNode, final Node toNode) {
253                double bestTravCostEst = estimateRemainingTravelCost(fromNode, toNode);
254                PreProcessLandmarks.LandmarksData fromRole = getPreProcessData(fromNode);
255                PreProcessLandmarks.LandmarksData toRole = getPreProcessData(toNode);
256                int bestIndex = -1;
257                for (int i = 0; i < this.landmarks.length; i++) {
258                        double tmpTravTime = estimateRemainingTravelCost(fromRole, toRole, i);
259                        if (tmpTravTime > bestTravCostEst) {
260                                bestIndex = i;
261                                bestTravCostEst = tmpTravTime;
262                        }
263                }
264                if (bestIndex != -1) {
265                        int[] newActiveLandmarks = new int[this.activeLandmarkIndexes.length + 1];
266                        System.arraycopy(this.activeLandmarkIndexes, 0, newActiveLandmarks, 0, this.activeLandmarkIndexes.length);
267                        newActiveLandmarks[this.activeLandmarkIndexes.length] = bestIndex;
268                        this.activeLandmarkIndexes = newActiveLandmarks;
269                }
270                return bestIndex;
271        }
272
273        /**
274         * Estimates the remaining travel cost from fromNode to toNode using the landmark given by index.
275         * 
276         * @param fromRole The first node/role.
277         * @param toRole The second node/role.
278         * @param index The index of the landmarks that should be used for
279         * the estimation of the travel cost.
280         * @return The travel cost when traveling between the two given nodes.
281         */
282        protected double estimateRemainingTravelCost(final PreProcessLandmarks.LandmarksData fromRole,
283                        final PreProcessLandmarks.LandmarksData toRole, final int index) {
284                double tmpTravTime;
285                final double fromMinLandmarkTravelTime = fromRole.getMinLandmarkTravelTime(index);
286                final double toMaxLandmarkTravelTime = toRole.getMaxLandmarkTravelTime(index);
287                tmpTravTime = fromMinLandmarkTravelTime - toMaxLandmarkTravelTime;
288                if (tmpTravTime < 0) {
289                        tmpTravTime = toRole.getMinLandmarkTravelTime(index) - fromRole.getMaxLandmarkTravelTime(index);
290                        if (tmpTravTime <= 0) {
291                                return 0;
292                        }
293                }
294                return tmpTravTime * this.overdoFactor;
295        }
