022

024import org.matsim.api.core.v01.network.Link;
025import org.matsim.api.core.v01.network.Network;
026import org.matsim.api.core.v01.network.Node;

031/**
032 * Implements the <a href="http://en.wikipedia.org/wiki/A%2A">A* router algorithm</a>
033 * for a given NetworkLayer by using the euclidean distance divided by the
034 * maximal free speed per length unit as the heuristic estimate during routing.
035 *
036 * AStarEuclidean about 3 times faster than Dijkstra.<br />
037 *
038 * <p>For every router, there exists a class which computes some
039 * preprocessing data and is passed to the router class
040 * constructor in order to accelerate the routing procedure.
041 * The one used for AStarEuclidean is org.matsim.demandmodeling.router.util.PreProcessEuclidean.<br />
042 *
043 * Network conditions:<br />
044 * <ul>
045 * <li>The same as for Dijkstra: The link cost must be non-negative,
046 * otherwise Dijkstra does not work.</li>
047 * <li>The length stored in the links must be greater or equal to the euclidean distance of the
048 * link's start and end node, otherwise the algorithm is not guaranteed to deliver least-cost paths.
049 * In this case PreProcessEuclidean gives out a warning message.</li>
050 * <li>The CostCalculator which calculates the cost for each link must implement the TravelMinCost
051 * interface, i.e. it must implement the function getLinkMinimumTravelCost(Link link).
052 * The TravelTimeCalculator class does implement it.</li></p>
053 * <p><code> PreProcessEuclidean.run() </code>is very fast and needs (almost) no additional
054 * memory.<code><br />
055 * </code> Code Example:<code><br />
056 * TravelMinCost costFunction = ...<br />
057 * PreProcessEuclidean preProcessData = new PreProcessEuclidean(costFunction);<br />
058 * preProcessData.run(network);<br />...<br />
059 * LeastCostPathCalculator routingAlgo = new AStarEuclidean(network, preProcessData);<br />
060 * ...</code></p>
061 * <p>A note about the so-called overdo factor: You can drastically accelerate the routing of
062 * AStarEuclidean by providing an overdo factor &gt; 1 (e.g. 1.5, 2 or 3). In this case,
063 * AStarEuclidean does not calculate least-cost paths anymore but tends to deliver distance-minimal
064 * paths. The greater the overdo factor, the faster the algorithm but the more the calculated routes
065 * diverge from the least-cost ones.<br />
066 * A typical invocation then looks like this:<br>
067 * <code>LeastCostPathCalculator routingAlgo = new AStarEuclidean(network, preProcessData, 2);</code>
068 * <br />
069 * @see org.matsim.core.router.util.PreProcessEuclidean
070 * @see org.matsim.core.router.Dijkstra
071 * @see org.matsim.core.router.AStarLandmarks
072 * @author lnicolas
073 */
074public class AStarEuclidean extends Dijkstra {
075        private static final Logger log = Logger.getLogger( AStarEuclidean.class ) ;
076
077        protected final double overdoFactor;
078
079        private double minTravelCostPerLength;
080
081        /**
082         * Default constructor; sets the overdo factor to 1.
083         * @param network
084         * @param preProcessData
085         * @param timeFunction
086         */
087        AStarEuclidean(final Network network, final PreProcessEuclidean preProcessData,
088                        final TravelTime timeFunction) {
089                this(network, preProcessData, timeFunction, 1);
090        }
091
092        /**
093         * @param network Where we do the routing.
094         * @param preProcessData The pre-process data (containing the landmarks etc.).
095         * @param timeFunction Determines the travel time on each link.
096         * @param overdoFactor The factor which is multiplied with the output of the A* heuristic function. The higher 
097         * the overdo factor the greedier the router, i.e. it visits less nodes during routing and is thus faster, but 
098         * for an overdo factor > 1, it is not guaranteed that the router returns the least-cost paths. Rather it tends 
099         * to return distance-minimal paths.
100         */
101        AStarEuclidean(final Network network, final PreProcessEuclidean preProcessData,
102                        final TravelTime timeFunction, final double overdoFactor) {
103                this(network, preProcessData, preProcessData.getCostFunction(),
104                                timeFunction, overdoFactor);
105        }
106
107        /**
108         * @param network Where we do the routing.
109         * @param preProcessData The pre-process data (containing the landmarks etc.).
110         * @param timeFunction Determines the travel time on each link.
111         * @param costFunction Calculates the travel cost on links.
112         * @param overdoFactor The factor which is multiplied with the output of the A* heuristic function. The higher 
113         * the overdo factor the greedier the router, i.e. it visits less nodes during routing and is thus faster, but 
114         * for an overdo factor > 1, it is not guaranteed that the router returns the least-cost paths. Rather it tends 
115         * to return distance-minimal paths.
116         */
117        AStarEuclidean(final Network network,
118                        final PreProcessEuclidean preProcessData,
119                        final TravelDisutility costFunction, final TravelTime timeFunction, final double overdoFactor) {
120                super(network, costFunction, timeFunction, preProcessData);
121
122                setMinTravelCostPerLength(preProcessData.getMinTravelCostPerLength());
123
124                this.overdoFactor = overdoFactor;
125        }
126
127        /**
128         * Initializes the first node of a route.
129         *
130         * @param fromNode
131         *            The Node to be initialized.
132         * @param toNode
133         *            The Node at which the route should end.
134         * @param pendingNodes
135         *            The pending nodes so far.
136         * @param toNode
137         *            The Node at which the route should end.
138         * @param startTime
139         *            The time we start routing.
140         */
141        @Override
142        protected void initFromNode(final Node fromNode, final Node toNode, final double startTime, final RouterPriorityQueue<Node> pendingNodes) {
143                AStarNodeData data = getData(fromNode);
144                visitNode(fromNode, data, pendingNodes, startTime, 0, null);
145                data.setExpectedRemainingCost(estimateRemainingTravelCost(fromNode, toNode));
146        }
147
148        @Override
149        protected boolean addToPendingNodes(final Link l, final Node n, final RouterPriorityQueue<Node> pendingNodes,
150                        final double currTime, final double currCost, final Node toNode) {
151
152                final double travelTime = this.timeFunction.getLinkTravelTime(l, currTime, this.person, this.vehicle);
153                final double travelCost = this.costFunction.getLinkTravelDisutility(l, currTime, this.person, this.vehicle);            
154                final AStarNodeData data = getData(n);
155                if (!data.isVisited(getIterationId())) {
156                        double remainingTravelCost = estimateRemainingTravelCost(n, toNode);
157                        visitNode(n, data, pendingNodes, currTime + travelTime, currCost + travelCost, remainingTravelCost, l);
158                        return true;
159                }
160                
161                final double nCost = data.getCost();
162                final double totalCost = currCost + travelCost;
163                if (totalCost < nCost) {
164                        revisitNode(n, data, pendingNodes, currTime + travelTime, totalCost, l);
165                        return true;
166                } else if (totalCost == nCost) {
167                        // Special case: a node can be reached from two links with exactly the same costs.
168                        // Decide based on the linkId which one to take... just have to common criteria to be deterministic.
169                        
170                        if ( totalCost==0. ) {
171                                log.warn( "finding totalCost=" + totalCost + "; this will often (or always?) lead to a null " +
172                                                                  "pointer exception later.  In my own case, it was related to a network " +
173                                                                  "having freespeed infinity at places.  kai, jan'18") ;
174                        }
175                        
176                        if (data.getPrevLink().getId().compareTo(l.getId()) > 0) {
177                                revisitNode(n, data, pendingNodes, currTime + travelTime, totalCost, l);
178                                return true;
179                        }
180                }
181                return false;
182        }
183
184        /**
185         * Inserts the given Node n into the pendingNodes queue and updates its time and cost information.
186         * 
187         * @param n The Node that is revisited.
188         * @param data The data for node.
189         * @param pendingNodes The nodes visited and not processed yet.
190         * @param time The time of the visit of n.
191         * @param cost The accumulated cost at the time of the visit of n.
192         * @param expectedRemainingCost The expected remaining travel cost when
193         * traveling from n to the target node of the route.
194         * @param outLink The link from which we came visiting n.
195         */
196        private void visitNode(final Node n, final AStarNodeData data, final RouterPriorityQueue<Node> pendingNodes, 
197                        final double time, final double cost, final double expectedRemainingCost, final Link outLink) {
198                data.setExpectedRemainingCost(expectedRemainingCost);
199                super.visitNode(n, data, pendingNodes, time, cost, outLink);
200        }
201        
202        /**
203         * @return The overdo factor used.
204         */
205        public double getOverdoFactor() {
206                return this.overdoFactor;
207        }
208
209        /**
210         * Estimates the remaining travel cost from fromNode to toNode using the euclidean distance between them.
211         * 
212         * @param fromNode The first node.
213         * @param toNode The second node.
214         * @return The travel cost when traveling between the two given nodes.
215         */
216        protected double estimateRemainingTravelCost(final Node fromNode, final Node toNode) {
217                double dist = CoordUtils.calcEuclideanDistance(fromNode.getCoord(), toNode.getCoord()) * getMinTravelCostPerLength();
218                return dist * this.overdoFactor;
219        }
220
221        /**
222         * Returns the data for the given Node. Creates a new AStarNodeData if none exists yet.
223         *
224         * @param n The node for which to return the data for..
225         * @return The data to the given Node
226         */
227        @Override
228        protected AStarNodeData getData(final Node n) {
229                return (AStarNodeData) super.getData(n);
230        }
231
232        @Override
233        protected AStarNodeData createNodeData() {
234                return new AStarNodeData();
235        }
236        
237        /**
238         * Sets minTravelCostPerLength to the given value.
239         * 
240         * @param minTravelCostPerLength
241         *            the minTravelCostPerLength to set
242         */
243        /*package*/ void setMinTravelCostPerLength(final double minTravelCostPerLength) {
244                this.minTravelCostPerLength = minTravelCostPerLength;
245        }
246
247        /**
248         * Returns the minimal travel cost per length unit on a link in the network.
249         * 
250         * @return the minimal travel cost per length unit on a link in the network.
251         */
252        public final double getMinTravelCostPerLength() {
253                return this.minTravelCostPerLength;
254        }
255
256        /**
257         * The value used to sort the pending nodes during routing. This implementation compares the total 
258         * estimated remaining travel cost to sort the nodes in the pending nodes queue during routing.
259         */
260        @Override
261        protected double getPriority(final DijkstraNodeData data) {
262                return ((AStarNodeData) data).getExpectedCost();
263        }
