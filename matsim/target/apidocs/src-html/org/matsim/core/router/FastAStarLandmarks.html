021package org.matsim.core.router;
022
023import org.matsim.api.core.v01.network.Node;
024import org.matsim.api.core.v01.population.Person;
025import org.matsim.core.router.priorityqueue.BinaryMinHeap;
026import org.matsim.core.router.util.AStarNodeData;
027import org.matsim.core.router.util.AStarNodeDataFactory;
028import org.matsim.core.router.util.ArrayRoutingNetwork;
029import org.matsim.core.router.util.ArrayRoutingNetworkNode;
030import org.matsim.core.router.util.PreProcessLandmarks;
031import org.matsim.core.router.util.RoutingNetwork;
032import org.matsim.core.router.util.RoutingNetworkNode;
033import org.matsim.core.router.util.TravelDisutility;
034import org.matsim.core.router.util.TravelTime;
035import org.matsim.core.utils.collections.RouterPriorityQueue;
036import org.matsim.vehicles.Vehicle;

038/**
039 * <p>
040 * Performance optimized version of the Dijkstra {@link org.matsim.core.router.FastAStarLandmarks} 
041 * least cost path router which uses its own network to route within.
042 * </p>
043 * 
044 * @see org.matsim.core.router.FastAStarLandmarks
045 * @see org.matsim.core.router.util.RoutingNetwork
046 * @author cdobler
047 */
048public class FastAStarLandmarks extends AStarLandmarks {
049
050        private final RoutingNetwork routingNetwork;
051        private final FastRouterDelegate fastRouter;
052        private BinaryMinHeap<ArrayRoutingNetworkNode> heap = null;
053        private int maxSize = -1;
054
055        FastAStarLandmarks(final RoutingNetwork routingNetwork, final PreProcessLandmarks preProcessData,
056                        final TravelDisutility costFunction, final TravelTime timeFunction, final double overdoFactor,
057                        final FastRouterDelegateFactory fastRouterFactory) {
058                super(routingNetwork, preProcessData, costFunction, timeFunction, overdoFactor);
059
060                this.routingNetwork = routingNetwork;
061                this.fastRouter = fastRouterFactory.createFastRouterDelegate(this, new AStarNodeDataFactory(), routingNetwork);
062                                
063                this.nodeData.clear();
064        }
065
066        /*
067         * Replace the references to the from and to nodes with their corresponding
068         * nodes in the routing network.
069         */
070        @Override
071        public Path calcLeastCostPath(final Node fromNode, final Node toNode, final double startTime, final Person person, final Vehicle vehicle) {
072                
073                this.fastRouter.initialize();
074                this.routingNetwork.initialize();
075                
076                RoutingNetworkNode routingNetworkFromNode = routingNetwork.getNodes().get(fromNode.getId());
077                RoutingNetworkNode routingNetworkToNode = routingNetwork.getNodes().get(toNode.getId());
078                
079                return super.calcLeastCostPath(routingNetworkFromNode, routingNetworkToNode, startTime, person, vehicle);
080        }
081        
082        @Override
083        /*package*/ RouterPriorityQueue<? extends Node> createRouterPriorityQueue() {
084                /*
085                 * Re-use existing BinaryMinHeap instead of creating a new one. For large networks (> 10^6 nodes and links) this reduced
086                 * the computation time by 40%! cdobler, oct'15
087                 */
088                if (this.routingNetwork instanceof ArrayRoutingNetwork) {
089                        int size = this.routingNetwork.getNodes().size();
090                        if (this.heap == null || this.maxSize != size) {
091                                this.maxSize = size;
092                                this.heap = new BinaryMinHeap<>(maxSize);
093                                return this.heap;
094                        } else {
095                                this.heap.reset();
096                                return this.heap;
097                        }
098                } else {
099                        return super.createRouterPriorityQueue();
100                }
101        }
102        
103        /*
104         * Constructs the path and replaces the nodes and links from the routing network
105         * with their corresponding nodes and links from the network.
106         */
107        @Override
108        protected Path constructPath(Node fromNode, Node toNode, double startTime, double arrivalTime) {
109                return this.fastRouter.constructPath(fromNode, toNode, startTime, arrivalTime);
110        }
111        
112        /*
113         * For performance reasons the outgoing links of a node are stored in
114         * the routing network in an array instead of a map. Therefore we have
115         * to iterate over an array instead of over a map. 
116         */
117        @Override
118        protected void relaxNode(final Node outNode, final Node toNode, final RouterPriorityQueue<Node> pendingNodes) {
119                this.controlCounter++;
120                if (this.controlCounter == controlInterval) {
121                        int newLandmarkIndex = checkToAddLandmark(outNode, toNode);
122                        if (newLandmarkIndex > 0) {
123                                updatePendingNodes(newLandmarkIndex, toNode, pendingNodes);
124                        }
125                        this.controlCounter = 0;
126                }
127                
128                this.fastRouter.relaxNode(outNode, toNode, pendingNodes);
129        }
130        
131        /*
132         * The DijkstraNodeData is taken from the RoutingNetworkNode and not from a map.
133         */
134        @Override
135        protected AStarNodeData getData(final Node n) {
136                return (AStarNodeData) this.fastRouter.getData(n);
137        }
138
139        /*
140         * The LandmarksData is taken from the RoutingNetworkNode and not from a map.
141         */
142        @Override
143        protected PreProcessLandmarks.LandmarksData getPreProcessData(final Node n) {
144                return (PreProcessLandmarks.LandmarksData) this.fastRouter.getPreProcessData(n);
145        }
146}
