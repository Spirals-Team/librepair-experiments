039/**
040 * Performs several checks that persons are ready for a mobility simulation.
041 * It is intended to run only once after the initial plans are read from file,
042 * as we expect that no "damage" happens to the plans during the iterations.
043 * <br>
044 * Currently, this only checks that in all plans the act's have a link assigned
045 * and that all plans have valid routes, calculating missing links and
046 * routes if required. Additionally, it will output a warning to the
047 * log if a person has no plans at all.
048 *
049 * @author mrieser
050 */
051public final class PersonPrepareForSimMultimodal extends AbstractPersonAlgorithm {
052
053        private final PlanAlgorithm router;
054        private final XY2Links xy2links;
055        private final Network network;
056
057        private static final Logger log = Logger.getLogger(PersonPrepareForSimMultimodal.class);
058
059        /*
060         * To be used by the controller which creates multiple instances of this class which would
061         * create multiple copies of a car-only-network. Instead, we can create that network once in
062         * the Controller and re-use it for each new instance. cdobler, sep'15
063         */
064        public PersonPrepareForSimMultimodal(final PlanAlgorithm router, final Scenario scenario, final Network carNetwork) {
065                super();
066                this.router = router;
067                this.network = scenario.getNetwork();
068                if (NetworkUtils.isMultimodal(carNetwork)) {
069                        throw new RuntimeException("Expected carOnlyNetwork not to be multi-modal. Aborting!");
070                }
071                this.xy2links = new XY2Links(carNetwork, scenario.getActivityFacilities());
072        }

090        @Override
091        public void run(final Person person) {
092                // first make sure we have a selected plan
093                Plan selectedPlan = person.getSelectedPlan();
094                if (selectedPlan == null) {
095                        // the only way no plan can be selected should be when the person has no plans at all
096                        log.warn("Person " + person.getId() + " has no plans!");
097                        return;
098                }
099
100                // make sure all the plans have valid act-locations and valid routes
101                for (Plan plan : person.getPlans()) {
102                        boolean needsXY2Links = false;
103                        boolean needsReRoute = false;
104                        for (PlanElement pe : plan.getPlanElements()) {
105                                if (pe instanceof Activity) {
106                                        Activity act = (Activity) pe;
107                                        if (act.getLinkId() == null) { // = no street address
108                                                needsXY2Links = true;
109                                                needsReRoute = true;
110                                                break;
111                                        }
112                                } else if (pe instanceof Leg) {
113                                        Leg leg = (Leg) pe;
114                                        if (leg.getRoute() == null) {
115                                                needsReRoute = true;
116                                        }
117                                        else if (Double.isNaN(leg.getRoute().getDistance())){
118                                                Double dist = null;
119                                                if (leg.getRoute() instanceof NetworkRoute){
120                                                        // (presumably, this is a case where a network route exists, but the distance was not included. kai, nov'16)
121                                                        
122                                                        /* So far, 1.0 is always used as relative position on start and end link. 
123                                                         * This means that the end link is considered in route distance and the start link not.
124                                                         * tt feb'16
125                                                         */
126                                                        double relativePositionStartLink = 1.0;
127                                                        double relativePositionEndLink  = 1.0;
128
129                                                        dist = RouteUtils.calcDistance((NetworkRoute) leg.getRoute(), relativePositionStartLink, relativePositionEndLink, this.network);
130                                                }
131                                                if (dist != null){
132                                                        leg.getRoute().setDistance(dist);
133                                                }
134                                        }
135                                }
136                        }
137                        if (needsXY2Links) {
138                                this.xy2links.run(plan);
139                        }
140                        if (needsReRoute) {
141                                this.router.run(plan);
142                        }
143                }
144
145        }
