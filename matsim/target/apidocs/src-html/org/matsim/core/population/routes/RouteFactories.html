020package org.matsim.core.population.routes;
021
022import java.util.HashMap;
023import java.util.Map;
024
025import org.matsim.api.core.v01.Id;
026import org.matsim.api.core.v01.network.Link;
027import org.matsim.api.core.v01.population.Route;
028import org.matsim.pt.routes.ExperimentalTransitRoute;
029import org.matsim.pt.routes.ExperimentalTransitRouteFactory;
030
031/**
032 * @author mrieser / senozon
033 */
034public final class RouteFactories {
035        private final Map<Class<? extends Route>, RouteFactory> routeFactories = new HashMap<>();
036        private RouteFactory defaultFactory = new GenericRouteFactory();
037        private final Map<String, Class<? extends Route>> type2class = new HashMap<>();
038
039        public RouteFactories() {
040                this.setRouteFactory(NetworkRoute.class, new LinkNetworkRouteFactory());
041                this.setRouteFactory(ExperimentalTransitRoute.class, new ExperimentalTransitRouteFactory());
042        }
043        
044        /**
045         * @param routeClass the requested class of the route
046         * @param startLinkId the link where the route starts
047         * @param endLinkId the link where the route ends
048         * @return a new Route of the specified route type
049         *
050         * @see #setRouteFactory(Class, RouteFactory)
051         */
052        @SuppressWarnings("unchecked")
053        public <R extends Route> R createRoute(final Class<R> routeClass, final Id<Link> startLinkId, final Id<Link> endLinkId) {
054                RouteFactory factory = this.routeFactories.get(routeClass);
055                if (factory == null) {
056                        factory = this.defaultFactory;
057                }
058                return (R) factory.createRoute(startLinkId, endLinkId);
059        }
060
061        /**
062         * Registers a {@link RouteFactory} for the specified route type. If <code>factory</code> is <code>null</code>,
063         * the existing entry for this <code>routeClass</code> will be deleted. If <code>routeClass</code> is <code>null</code>,
064         * then the default factory is set that is used if no specific RouteFactory for a routeType is set.
065         *
066         * @param routeClass
067         * @param factory
068         */
069        public void setRouteFactory(final Class<? extends Route> routeClass, final RouteFactory factory) {
070                if (routeClass == null) {
071                        this.defaultFactory = factory;
072                } else {
073                        if (factory == null) {
074                                this.routeFactories.remove(routeClass);
075                        } else {
076                                this.routeFactories.put(routeClass, factory);
077                                this.type2class.put(factory.getCreatedRouteType(), routeClass);
078                        }
079                }
080        }
081
082        public Class<? extends Route> getRouteClassForType(String routeType) {
083                Class<? extends Route> routeClass = this.type2class.get(routeType);
084                if (routeClass == null) {
085                        routeClass = Route.class; // will result in a generic route
086                }
087                return routeClass;
088        }
089
090}
