021package org.matsim.core.population.algorithms;
022
023import java.util.List;
024import java.util.Random;
025
026import org.matsim.api.core.v01.population.Activity;
027import org.matsim.api.core.v01.population.Activity;
028import org.matsim.api.core.v01.population.Leg;
029import org.matsim.api.core.v01.population.Leg;
030import org.matsim.api.core.v01.population.Plan;
031import org.matsim.api.core.v01.population.PlanElement;
032import org.matsim.core.utils.misc.Time;
033
034/**
035 * Mutates the duration of activities randomly within a specified range.
036 * For the first act, the end-time is mutated. For the last act, duration and end-time are set to UNDEFINED.
037 * For all other acts, the duration is mutated, and the end-time is updated accordingly.
038 * If an activity has no duration set (UNDEFINED_TIME), a random time between 0 and 24h will be chosen.
039 * Departure and arrival times of legs are updated according to the activity durations, but the
040 * leg travel time is not modified (e.g. updated according to new expected departure time).
041 *
042 * @author mrieser
043 */
044public final class PlanMutateTimeAllocation implements PlanAlgorithm {
045
046        private final double mutationRange;
047        private final Random random;
048
049        public PlanMutateTimeAllocation(final double mutationRange, final Random random) {
050                this.mutationRange = mutationRange;
051                this.random = random;
052        }
053
054        @Override
055        public void run(final Plan plan) {
056                mutatePlan(plan);
057        }
058
059        private void mutatePlan(final Plan plan) {
060
061                List<PlanElement> planElements = plan.getPlanElements();
062                int max = planElements.size();
063
064                double now = 0;
065
066                // apply mutation to all activities except the last home activity
067                for (int i = 0; i < max; i++) {
068
069                        PlanElement pe = planElements.get(i);
070                        
071                        if (pe instanceof Activity) {
072                                Activity act = (Activity) pe;
073
074                                // handle first activity
075                                if (i == 0) {
076                                        // set start to midnight
077                                        act.setStartTime(now);
078                                        // mutate the end time of the first activity
079                                        act.setEndTime(mutateTime(act.getEndTime()));
080                                        // calculate resulting duration
081                                        act.setMaximumDuration(act.getEndTime() - act.getStartTime());
082                                        // move now pointer
083                                        now += act.getEndTime();
084
085                                // handle middle activities
086                                } else if (i < (max - 1)) {
087
088                                        // assume that there will be no delay between arrival time and activity start time
089                                        act.setStartTime(now);
090                                                if (act.getMaximumDuration() != Time.UNDEFINED_TIME) {
091                                                        // mutate the durations of all 'middle' activities
092                                                        act.setMaximumDuration(mutateTime(act.getMaximumDuration()));
093                                                        now += act.getMaximumDuration();
094                                                        // set end time accordingly
095                                                        act.setEndTime(now);
096                                                } else {
097                                                        double newEndTime = mutateTime(act.getEndTime());
098                                                        if (newEndTime < now) {
099                                                                newEndTime = now;
100                                                        }
101                                                        act.setEndTime(newEndTime);
102                                                        now = newEndTime;
103                                                }
104                                // handle last activity
105                                } else {
106
107                                        // assume that there will be no delay between arrival time and activity start time
108                                        act.setStartTime(now);
109                                        // invalidate duration and end time because the plan will be interpreted 24 hour wrap-around
110                                        act.setMaximumDuration(Time.UNDEFINED_TIME);
111                                        act.setEndTime(Time.UNDEFINED_TIME);
112
113                                }
114
115                        } else if (pe instanceof Leg) {
116
117                                Leg leg = (Leg) pe;
118
119                                // assume that there will be no delay between end time of previous activity and departure time
120                                leg.setDepartureTime(now);
121                                // let duration untouched. if defined add it to now
122                                if (leg.getTravelTime() != Time.UNDEFINED_TIME) {
123                                        now += leg.getTravelTime();
124                                }
125                                final double arrTime = now;
126                                // set planned arrival time accordingly
127                                leg.setTravelTime( arrTime - leg.getDepartureTime() );
128
129                        }
130                }
131        }
132
133        private double mutateTime(final double time) {
134                double t = time;
135                if (t != Time.UNDEFINED_TIME) {
136                        t = t + (int)((this.random.nextDouble() * 2.0 - 1.0) * this.mutationRange);
137                        if (t < 0) t = 0;
138                        if (t > 24*3600) t = 24*3600;
139                } else {
140                        t = this.random.nextInt(24*3600);
141                }
142                return t;
143        }
144
145}
