020package org.matsim.core.population.algorithms;
021
022import java.util.List;
023
024import org.matsim.api.core.v01.population.Plan;
025import org.matsim.api.core.v01.population.PlanElement;
026import org.matsim.core.population.PopulationUtils;
027import org.matsim.core.router.MainModeIdentifier;
028import org.matsim.core.router.StageActivityTypes;
029import org.matsim.core.router.TripRouter;
030import org.matsim.core.router.TripStructureUtils;
031import org.matsim.core.router.TripStructureUtils.Trip;
032
033/**
034 * Simplifies a plan to its structure, so that plan algorithms
035 * working with strict leg/act alternance can work on it.
036 * Routing will be necessary before execution.
037 *
038 * @author thibautd
039 */
040public final class TripsToLegsAlgorithm implements PlanAlgorithm {
041        private final StageActivityTypes blackList;
042        private final MainModeIdentifier mainModeIdentifier;
043
044        public TripsToLegsAlgorithm(final TripRouter router) {
045                this(
046                                router.getStageActivityTypes(),
047                                router.getMainModeIdentifier() );
048        }
049
050
051        public TripsToLegsAlgorithm(
052                        final StageActivityTypes blackList,
053                        final MainModeIdentifier mainModeIdentifier) {
054                this.blackList = blackList;
055                this.mainModeIdentifier = mainModeIdentifier;
056        }
057
058        @Override
059        public void run(final Plan plan) {
060                final List<PlanElement> planElements = plan.getPlanElements();
061                final List<Trip> trips = TripStructureUtils.getTrips( plan , blackList );
062
063                for ( Trip trip : trips ) {
064                        final List<PlanElement> fullTrip =
065                                planElements.subList(
066                                                planElements.indexOf( trip.getOriginActivity() ) + 1,
067                                                planElements.indexOf( trip.getDestinationActivity() ));
068                        final String mode = mainModeIdentifier.identifyMainMode( fullTrip );
069                        fullTrip.clear();
070                        fullTrip.add( PopulationUtils.createLeg(mode) );
071                        if ( fullTrip.size() != 1 ) throw new RuntimeException( fullTrip.toString() );
072                }
073        }
074}
