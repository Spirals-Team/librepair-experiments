155                } ;
156                try {
157                        matsimWriter.openHere(filename);
158                        this.handler.writeHeaderAndStartElement(matsimWriter.getWriter());
159                        this.handler.startPlans(fakepop, matsimWriter.getWriter());
160                        this.handler.writeSeparator(matsimWriter.getWriter());
161                } catch (IOException e) {
162                        throw new UncheckedIOException(e);

178        public final void writePerson(final Person person) {
179                try {
180                        if ((this.write_person_fraction < 1.0) && (MatsimRandom.getRandom().nextDouble() >= this.write_person_fraction)) {
181                                return;
182                        }
183                        this.handler.writePerson(person, matsimWriter.getWriter());
184                        counter.incCounter();
185                } catch (IOException e) {
186                        throw new UncheckedIOException(e);
187                }
188        }
189
190        public final void writeEndPlans() {
191                try {
192                        this.handler.endPlans(matsimWriter.getWriter());
193                } catch (IOException e) {
194                        throw new UncheckedIOException(e);
195                }
196                matsimWriter.closeHere();
197        }
