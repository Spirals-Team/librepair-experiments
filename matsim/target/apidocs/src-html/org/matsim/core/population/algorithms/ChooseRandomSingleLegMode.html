021package org.matsim.core.population.algorithms;
022
023import java.util.ArrayList;
024import java.util.Random;
025import org.matsim.api.core.v01.TransportMode;
026import org.matsim.api.core.v01.population.Leg;
027import org.matsim.api.core.v01.population.Plan;
028import org.matsim.api.core.v01.population.PlanElement;
029import org.matsim.api.core.v01.population.Route;
030import org.matsim.core.gbl.MatsimRandom;
031import org.matsim.core.population.PersonUtils;
032import org.matsim.core.population.routes.NetworkRoute;
033
034/**
035 * Changes the transportation mode of one leg in a plan to a randomly chosen
036 * mode, given a list of possible modes. Insures that the newly chosen mode
037 * is different from the existing mode (if possible).
038 *
039 * @author mrieser
040 */
041public final class ChooseRandomSingleLegMode implements PlanAlgorithm {
042
043        private final String[] possibleModes;
044        private boolean ignoreCarAvailability = true;
045
046        private final Random rng;
047
048        /**
049         * @param possibleModes
050         * @param rng The random number generator used to draw random numbers to select another mode.
051         * @see TransportMode
052         * @see MatsimRandom
053         */
054        public ChooseRandomSingleLegMode(final String[] possibleModes, final Random rng) {
055                this.possibleModes = possibleModes.clone();
056                this.rng = rng;
057        }
058
059        public void setIgnoreCarAvailability(final boolean ignoreCarAvailability) {
060                this.ignoreCarAvailability = ignoreCarAvailability;
061        }
062
063        @Override
064        public void run(final Plan plan) {
065                boolean forbidCar = false;
066                if (!this.ignoreCarAvailability) {
067                        String carAvail = PersonUtils.getCarAvail(plan.getPerson());
068                        if ("never".equals(carAvail)) {
069                                forbidCar = true;
070                        }
071                }
072
073                ArrayList<Leg> legs = new ArrayList<Leg>();
074                int cnt = 0;
075                for (PlanElement pe : plan.getPlanElements()) {
076                        if (pe instanceof Leg) {
077                                legs.add((Leg) pe);
078                                cnt++;
079                        }
080                }
081                if (cnt == 0) {
082                        return;
083                }
084                int rndIdx = this.rng.nextInt(cnt);
085                setRandomLegMode(legs.get(rndIdx), forbidCar);
086        }
087
088        private void setRandomLegMode(final Leg leg, final boolean forbidCar) {
089                leg.setMode(chooseModeOtherThan(leg.getMode(), forbidCar));
090                Route route = leg.getRoute() ;
091                if ( route != null && route instanceof NetworkRoute) {
092                        ((NetworkRoute)route).setVehicleId(null);
093                }
094        }
095
096        private String chooseModeOtherThan(final String currentMode, final boolean forbidCar) {
097                String newMode;
098                while (true) {
099                        int newModeIdx = this.rng.nextInt(this.possibleModes.length - 1);
100                        for (int i = 0; i <= newModeIdx; i++) {
101                                if (this.possibleModes[i].equals(currentMode)) {
102                                        /* if the new Mode is after the currentMode in the list of possible
103                                         * modes, go one further, as we have to ignore the current mode in
104                                         * the list of possible modes. */
105                                        newModeIdx++;
106                                        break;
107                                }
108                        }
109                        newMode = this.possibleModes[newModeIdx];
110                        if (!(forbidCar && TransportMode.car.equals(newMode))) {
111                                break;
112                        } else {
113                                if (this.possibleModes.length == 2) {
114                                        newMode = currentMode; // there is no other mode available
115                                        break;
116                                }
117                        }
118                }
119                return newMode;
120        }
121
122}
