021package org.matsim.core.population.algorithms;
022
023import java.util.ArrayList;
024import java.util.Arrays;
025import java.util.List;
026import java.util.Random;
027import org.matsim.api.core.v01.TransportMode;
028import org.matsim.api.core.v01.population.Leg;
029import org.matsim.api.core.v01.population.Plan;
030import org.matsim.api.core.v01.population.PlanElement;
031import org.matsim.api.core.v01.population.Route;
032import org.matsim.core.gbl.MatsimRandom;
033import org.matsim.core.population.PersonUtils;
034import org.matsim.core.population.routes.NetworkRoute;
035
036/**
037 * Changes the transportation mode of one leg in a plan to a randomly chosen
038 * mode, given a list of possible modes. Insures that the newly chosen mode
039 * is different from the existing mode (if possible).
040 *
041 * @author mrieser
042 */
043public final class ChooseRandomSingleLegMode implements PlanAlgorithm {
044
045        private final String[] possibleModes;
046        private boolean ignoreCarAvailability = true;
047        private boolean allowSwitchFromListedModesOnly;
048        private final Random rng;
049        private final List<String> possibleFromModes = new ArrayList<>();
050
051
052        /**
053         * @param possibleModes possible transportmodes to chose from
054         * @param rng The random number generator used to draw random numbers to select another mode.
055         * @see TransportMode
056         * @see MatsimRandom
057         */
058        public ChooseRandomSingleLegMode(final String[] possibleModes, final Random rng, boolean allowSwitchFromListedModesOnly) {
059                this.possibleModes = possibleModes.clone();
060                this.allowSwitchFromListedModesOnly=allowSwitchFromListedModesOnly;
061                if (allowSwitchFromListedModesOnly){
062                        this.possibleFromModes.addAll(Arrays.asList(possibleModes));
063                        }
064                this.rng = rng;
065        }
066
067        public void setIgnoreCarAvailability(final boolean ignoreCarAvailability) {
068                this.ignoreCarAvailability = ignoreCarAvailability;
069        }
070
071        @Override
072        public void run(final Plan plan) {
073                boolean forbidCar = false;
074                if (!this.ignoreCarAvailability) {
075                        String carAvail = PersonUtils.getCarAvail(plan.getPerson());
076                        if ("never".equals(carAvail)) {
077                                forbidCar = true;
078                        }
079                }
080
081                ArrayList<Leg> legs = new ArrayList<>();
082                int cnt = 0;
083                for (PlanElement pe : plan.getPlanElements()) {
084                        if (pe instanceof Leg) {
085                                if (allowSwitchFromListedModesOnly){
086                                        if (this.possibleFromModes.contains(((Leg) pe).getMode())) {
087                                                legs.add((Leg) pe);
088                                                cnt++;
089                                        }
090                                }
091                                else {
092                                legs.add((Leg) pe);
093                                cnt++;
094                                }
095                        }
096                }
097                if (cnt == 0) {
098                        return;
099                }
100                int rndIdx = this.rng.nextInt(cnt);
101                setRandomLegMode(legs.get(rndIdx), forbidCar);
102        }
103
104        private void setRandomLegMode(final Leg leg, final boolean forbidCar) {
105                leg.setMode(chooseModeOtherThan(leg.getMode(), forbidCar));
106                Route route = leg.getRoute() ;
107                if ( route != null && route instanceof NetworkRoute) {
108                        ((NetworkRoute)route).setVehicleId(null);
109                }
110        }
111
112        private String chooseModeOtherThan(final String currentMode, final boolean forbidCar) {
113                String newMode;
114                while (true) {
115                        int newModeIdx = this.rng.nextInt(this.possibleModes.length - 1);
116                        for (int i = 0; i <= newModeIdx; i++) {
117                                if (this.possibleModes[i].equals(currentMode)) {
118                                        /* if the new Mode is after the currentMode in the list of possible
119                                         * modes, go one further, as we have to ignore the current mode in
120                                         * the list of possible modes. */
121                                        newModeIdx++;
122                                        break;
123                                }
124                        }
125                        newMode = this.possibleModes[newModeIdx];
126                        if (!(forbidCar && TransportMode.car.equals(newMode))) {
127                                break;
128                        } else {
129                                if (this.possibleModes.length == 2) {
130                                        newMode = currentMode; // there is no other mode available
131                                        break;
132                                }
133                        }
134                }
135                return newMode;
136        }
137
138}
