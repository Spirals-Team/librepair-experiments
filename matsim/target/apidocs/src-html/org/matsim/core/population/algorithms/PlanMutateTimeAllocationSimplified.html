021package org.matsim.core.population.algorithms;
022
023import java.util.Random;
024
025import org.matsim.api.core.v01.population.Activity;
026import org.matsim.api.core.v01.population.Plan;
027import org.matsim.core.router.EmptyStageActivityTypes;
028import org.matsim.core.router.StageActivityTypes;
029import org.matsim.core.router.TripStructureUtils;
030import org.matsim.core.utils.misc.Time;
031
032/**
033 * Mutates the duration of activities randomly within a specified range.
034 * <br>
035 * Other than the "full" version, this version just mutates activity end time and activity duration if they are defined,
036 * without thinking any further.  If this produces invalid plans, they will eventually be removed through the selection
037 * process.  kai, jun'12
038 *
039 * @author knagel
040 */
041public final class PlanMutateTimeAllocationSimplified implements PlanAlgorithm {
042
043        private final StageActivityTypes blackList;
044        private final double mutationRange;
045        private final Random random;
046        private final boolean affectingDuration;
047
048        /**
049         * Initializes an instance mutating all activities in a plan
050         * @param mutationRange
051         * @param affectingDuration TODO
052         * @param random
053         */
054        public PlanMutateTimeAllocationSimplified(final double mutationRange, boolean affectingDuration, final Random random) {
055                this( EmptyStageActivityTypes.INSTANCE , mutationRange , affectingDuration, random );
056        }
057        /**
058         * Initializes an instance mutating all non-stage activities in a plan
059         * @param mutationRange
060         * @param affectingDuration TODO
061         * @param random
062         */
063        public PlanMutateTimeAllocationSimplified(final StageActivityTypes blackList, final double mutationRange, boolean affectingDuration, final Random random) {
064                this.blackList = blackList;
065                this.mutationRange = mutationRange;
066                this.affectingDuration = affectingDuration;
067                this.random = random;
068        }
069
070        @Override
071        public void run(final Plan plan) {
072                for ( Activity act : TripStructureUtils.getActivities( plan , blackList ) ) {
073                        // this is deliberately simplistic.  Cleanup up of the time information should be done somewhere else.
074                        if (act.getEndTime() != Time.UNDEFINED_TIME) {
075                                act.setEndTime(mutateTime(act.getEndTime()));
076                        }
077                        if ( affectingDuration ) {
078                                if (act.getMaximumDuration() != Time.UNDEFINED_TIME) {
079                                        act.setMaximumDuration(mutateTime(act.getMaximumDuration()));
080                                }
081                        }
082                }
083                // the legs are not doing anything. kai, jun'12
084        }
085
086        private double mutateTime(final double time) {
087                double t = time;
088                t = t + (int)((this.random.nextDouble() * 2.0 - 1.0) * this.mutationRange);
089
090                if (t < 0) {
091                        t = 0;
092                }
093                // note that this also affects duration
094                
095                return t;
096        }
097
098}
