021package org.matsim.core.population.algorithms;
022
023import java.util.List;
024
025import org.matsim.api.core.v01.Scenario;
026import org.matsim.api.core.v01.network.Link;
027import org.matsim.api.core.v01.network.Network;
028import org.matsim.api.core.v01.population.Activity;
029import org.matsim.api.core.v01.population.Person;
030import org.matsim.api.core.v01.population.Plan;
031import org.matsim.api.core.v01.population.PlanElement;
032import org.matsim.core.network.NetworkUtils;
033import org.matsim.facilities.ActivityFacilities;
034import org.matsim.facilities.ActivityFacility;
035
036/**
037 * Assigns each activity in a plan a link where the activity takes place
038 * based on the coordinates given for the activity.
039 *
040 * @author mrieser
041 */
042public final class XY2Links extends AbstractPersonAlgorithm implements PlanAlgorithm {
043
044        private final Network network;
045        private final ActivityFacilities facilities;
046
047        /**
048         * When running XY2Links with given facilities, the linkIds of activities are
049         * taken from the facilities where they are performed. This ensures that the
050         * activity is performed at the same link where the facility is located. 
051         */
052        public XY2Links(final Network network, final ActivityFacilities facilities) {
053                super();
054                this.network = network;
055                this.facilities = facilities;
056        }
057
058        public XY2Links(final Scenario scenario) {
059                this(scenario.getNetwork(), scenario.getActivityFacilities());
060        }
061        
062        /** Assigns links to each activity in all plans of the person. */
063        @Override
064        public void run(final Person person) {
065                for (Plan plan : person.getPlans()) {
066                        processPlan(plan);
067                }
068        }
069
070        /** Assigns links to each activity in the plan. */
071        @Override
072        public void run(final Plan plan) {
073                processPlan(plan);
074        }
075
076        private void processPlan(final Plan plan) {
077                List<PlanElement> planElements = plan.getPlanElements();
078                for (PlanElement planElement : planElements) {
079                        if (planElement instanceof Activity) {
080                                Activity act = (Activity) planElement;
081                                
082                                if ( facilities != null ) {
083                                        // since the facilities in Scenario are now permanently enabled, this can only happen when called through the
084                                        // more specific constructor. kai, feb'16
085                                        
086                                        if (act.getFacilityId() != null) {
087                                                ActivityFacility facility = facilities.getFacilities().get(act.getFacilityId());
088
089                                                if (facility != null) {
090                                                        act.setLinkId(facility.getLinkId());
091                                                        // yy facility.getLinkId may be null, in particular since linkId is not even part of the facilities DTD. kai, feb'16
092
093                                                        if (act.getLinkId() == null && act.getCoord()==null){
094                                                                // for FacilitiesSource.onePerActivityLocationInPlansFile, one can opt to keep coords in facility only. Amit Jan'18
095                                                                act.setCoord(facility.getCoord());
096                                                        }
097                                                }
098                                        }
099                                }
100                                
101                                if ( act.getLinkId() != null ) {
102                                        // there may be activities in a plan that have a link and others that have a coordinate.  
103                                        // Those that have a link do not need a new link.  In addition, they may not even have a 
104                                        // coordinate.  kai/dominik, nov'11
105                                        continue ;
106                                }
107
108                                // If the linkId is still null get nearest link from the network
109//                              Link link = this.network.getNearestLinkExactly(act.getCoord());
110                                Link link = NetworkUtils.getNearestLink(this.network, act.getCoord());
111                                // getNearestLinkExactly not necessarily better than getNearestLink.  E.g.
112                                // n--n-----------------------------n
113                                // A home location slightly to the right of the middle node will take:
114                                // * the left link with getNearestLink
115                                // * the right link with getNearestLinkExactly
116                                // kai/dominik, jan'13
117                                /* ownPrepareForSimExample in matsim tutorials gives an example how to use
118                                 * getNearestLinkExactly anyway. tt feb'2016
119                                 */
120                                
121                                if (null == link) {
122                                        throw new RuntimeException("For person id="+plan.getPerson().getId()+": getNearestLink returned Null! act="+act);
123                                }
124                                act.setLinkId(link.getId());                            
125                        }
126                }
127        }
128}
