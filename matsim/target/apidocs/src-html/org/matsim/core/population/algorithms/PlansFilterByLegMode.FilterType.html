021package org.matsim.core.population.algorithms;
022
023import java.util.Iterator;
024import java.util.TreeSet;
025
026import org.apache.log4j.Logger;
027import org.matsim.api.core.v01.Id;
028import org.matsim.api.core.v01.TransportMode;
029import org.matsim.api.core.v01.population.Leg;
030import org.matsim.api.core.v01.population.Person;
031import org.matsim.api.core.v01.population.Plan;
032import org.matsim.api.core.v01.population.Population;
033import org.matsim.core.gbl.Gbl;
034import org.matsim.core.population.PersonUtils;
035
036/**
037 * This algorithm filters out all persons having plans with legs with a certain leg mode.
038 *
039 * Plans which do not fulfill the filter criteria are removed from a person, Persons with
040 * no plans are removed from the population.
041 */
042public final class PlansFilterByLegMode {
043        private static final Logger log = Logger.getLogger(PlansFilterByLegMode.class);
044
045        private String legMode;
046
047        // optimization: instead of doing a String.equals() every time, we do it once and store the result
048        private boolean legModeIsCar;
049
050        //////////////////////////////////////////////////////////////////////
051        // constructors
052        //////////////////////////////////////////////////////////////////////
053
054        public FilterType filterType ;
055        public enum FilterType { keepAllPlansWithMode, removeAllPlansWithMode, keepPlansWithOnlyThisMode }
056
057        public PlansFilterByLegMode(final String legMode, final FilterType filterType) {
058                super() ;
059                this.legMode = legMode ;
060                this.filterType = filterType ;
061                this.legModeIsCar = legMode.equals(TransportMode.car);
062        }
063
064        //////////////////////////////////////////////////////////////////////
065        // run methods
066        //////////////////////////////////////////////////////////////////////
067
068        private static int cnt = 0 ;
069        public void run(Population plans) {
070                int planCount = 0;
071                System.out.println("    running " + this.getClass().getName() + " algorithm...");
072                String never = "never";
073
074                TreeSet<Id<Person>> pid_set = new TreeSet<>();  // ids of persons to remove
075                Iterator<Id<Person>> pid_it = plans.getPersons().keySet().iterator();
076                while (pid_it.hasNext()) {
077                        Id<Person> personId = pid_it.next();
078                        Person person = plans.getPersons().get(personId);
079
080                        for (int i=person.getPlans().size()-1; i>=0; i--) {
081                                Plan plan = person.getPlans().get(i);
082                                boolean hasSearchedLegMode = false;
083                                boolean hasOtherLegMode = false;
084
085                                for (int j=1; j<plan.getPlanElements().size(); j+=2) {
086                                        Leg leg = (Leg)plan.getPlanElements().get(j);
087                                        if (leg.getMode().equals(this.legMode)) {
088                                                hasSearchedLegMode = true;
089                                        } else {
090                                                hasOtherLegMode = true;
091                                        }
092                                }
093                                if (this.legModeIsCar && never.equals(PersonUtils.getCarAvail(person))) {
094                                        // person cannot drive car if she has no car. this means, the person was given a lift by someone else
095                                        // --> do not include this person, as we're only interested in the driver
096                                        if ( cnt < 1 ) {
097                                                cnt ++ ;
098                                                log.warn("This method assumes that mode=car without car availability means `in car as passenger'.  Note that this is different "
099                                                                + "from what is often done in transport planning, where `in car as passenger' is a separate mode.");
100                                                log.warn(Gbl.ONLYONCE) ;
101                                        }
102                                        hasSearchedLegMode = false;
103                                }
104                                if ( filterType==FilterType.keepAllPlansWithMode || filterType==FilterType.keepPlansWithOnlyThisMode ) {
105                                        if ((!hasSearchedLegMode) || (hasOtherLegMode && this.filterType==FilterType.keepPlansWithOnlyThisMode )) {
106                                                person.getPlans().remove(i);
107                                                i--;    //otherwise, we would skip one plan
108                                                planCount++;
109                                        }
110                                } else if ( filterType==FilterType.removeAllPlansWithMode ) {
111                                        if ( hasSearchedLegMode ) {
112                                                person.getPlans().remove(i);
113                                                i--;    //otherwise, we would skip one plan
114                                                planCount++;
115                                        }
116                                } else {
117                                        throw new RuntimeException("should not happen;");
118                                }
119                        }
120                        if (person.getPlans().isEmpty()) {
121                                // the person has no plans left. remove the person afterwards (so we do not disrupt the Iterator)
122                                pid_set.add(personId);
123                        }
124                }
125
126                // okay, now remove in a 2nd step all persons we do no longer need
127                pid_it = pid_set.iterator();
128                while (pid_it.hasNext()) {
129                        Id<Person> pid = pid_it.next();
130                        plans.getPersons().remove(pid);
131                }
132
133                System.out.println("    done.");
134                System.out.println("Number of plans removed:   " + planCount);
135                System.out.println("Number of persons removed: " + pid_set.size());
136        }
137}
