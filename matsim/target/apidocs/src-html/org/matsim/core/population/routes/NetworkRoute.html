021package org.matsim.core.population.routes;
022
023import java.util.List;
024
025import org.matsim.api.core.v01.Id;
026import org.matsim.api.core.v01.network.Link;
027import org.matsim.api.core.v01.population.Route;
028import org.matsim.vehicles.Vehicle;
029
030
031/**
032 * A route that describes a path in a (road) network. Thus, the route is described usually
033 * as a series of links or nodes.
034 *
035 * @author mrieser
036 */
037public interface NetworkRoute extends Route {
038
039        public void setLinkIds(final Id<Link> startLinkId, final List<Id<Link>> linkIds, final Id<Link> endLinkId);
040
041        /**Design thoughts:<ul>
042         * <li> yyyy It this general cost or monetary cost?  kai/benjamin, jun'11
043         * </ul>
044         */
045        public void setTravelCost(final double travelCost);
046
047        /**Design thoughts:<ul>
048         * <li> yyyy It this general cost or monetary cost?  kai/benjamin, jun'11
049         * </ul>
050         */
051        public double getTravelCost();
052
053        /**
054         * Returns the list of link ids that build the route. The links where the route
055         * starts and ends (the links where the activities are on) are <b>not</b>
056         * included in the list (note that they are in getStartLinkId() and getEndLinkId() of the Route super-interface).
057         * @return a list containing the link ids the agents plans to travel along
058         */
059        public List<Id<Link>> getLinkIds();
060
061        /**
062         * This method returns a new Route object with the subroute of this, using fromLinkId as the
063         * subroute's startLink, toLinkId as the subroute's endLink, and the links in between fromLinkId
064         * and toLinkIds set as the route's link
065         * @param fromLinkId
066         * @param toLinkId
067         * @return subroute of this route starting at fromLinkId and ending at toLinkId
068         */
069        public NetworkRoute getSubRoute(final Id<Link> fromLinkId, final Id<Link> toLinkId);
070        // yyyy my intuition is that this should be removed from the API (and replaced by a static method in RouteUtils). kai, oct'17
071
072        /**
073         * Sets the id of the vehicle that should be used to drive along this route.
074         *
075         * @param vehicleId
076         */
077        public abstract void setVehicleId(final Id<Vehicle> vehicleId);
078
079        /**
080         * @return the id of the vehicle that should be used to drive along this route.
081         */
082        public abstract Id<Vehicle> getVehicleId();
083        // Does it really make sense to couple the vehicle to the route?  I would have coupled it to the leg.  kai, aug'10
084        // Well, I guess now it belongs to the route. :-)  kai, aug'10
085
086        @Override
087        public NetworkRoute clone();
088        // to get the correct interface type.  kai, sep'17
089
090}
