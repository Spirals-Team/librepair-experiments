021package org.matsim.core.population.algorithms;
022
023import java.lang.Thread.UncaughtExceptionHandler;
024import java.util.LinkedList;
025import java.util.List;
026import java.util.concurrent.atomic.AtomicBoolean;
027
028import org.apache.log4j.Logger;
029import org.matsim.api.core.v01.population.Person;
030import org.matsim.api.core.v01.population.Population;
031import org.matsim.core.utils.misc.Counter;
032
033/**
034 * An abstract/static helper class for running {@link AbstractPersonAlgorithm}s in parallel using threads.
035 *
036 * @author mrieser
037 */
038public final class ParallelPersonAlgorithmUtils {
039        private ParallelPersonAlgorithmUtils(){} // do not instantiate
040
041        private final static Logger log = Logger.getLogger(ParallelPersonAlgorithmUtils.class);
042        
043        public interface PersonAlgorithmProvider {
044                public PersonAlgorithm getPersonAlgorithm();
045        }
046
047        /**
048         * Handles each person of the given <code>population</code> with the specified <code>algorithm</code>,
049         * using up to <code>numberOfThreads</code> threads to speed things up. Use this method only if the given
050         * algorithm is thread-safe! Otherwise, use {@link #run(Population, int, PersonAlgorithmProvider)}.
051         *
052         * @param population
053         * @param numberOfThreads
054         * @param algorithm
055         */
056        public static void run(final Population population, final int numberOfThreads, final PersonAlgorithm algorithm) {
057                run(population, numberOfThreads, new PersonAlgorithmProvider() {
058                        @Override public PersonAlgorithm getPersonAlgorithm() {
059                                return algorithm;
060                        }
061                });
062        }
063
064        /**
065         * Handles each person of the given <code>population</code> with a AbstractPersonAlgorithm provided by <code>algoProvider</code>,
066         * using up to <code>numberOfThreads</code> threads to speed things up. This method will request a new instance of the
067         * AbstractPersonAlgorithm for each thread it allocates, thus enabling the parallel use of non-thread-safe algorithms.
068         * For thread-safe algorithms, {@link #run(Population, int, AbstractPersonAlgorithm)} may be an easier method to use.
069         *
070         * @param population
071         * @param numberOfThreads
072         * @param algoProvider
073         */
074        public static void run(final Population population, final int numberOfThreads, final PersonAlgorithmProvider algoProvider) {
075                int numOfThreads = Math.max(numberOfThreads, 1); // it should be at least 1 here; we allow 0 in other places for "no threads"
076                PersonAlgoThread[] algoThreads = new PersonAlgoThread[numOfThreads];
077                Thread[] threads = new Thread[numOfThreads];
078                String name = null;
079                Counter counter = null;
080                
081                final AtomicBoolean hadException = new AtomicBoolean(false);
082                final ExceptionHandler uncaughtExceptionHandler = new ExceptionHandler(hadException);
083
084                // setup threads
085                for (int i = 0; i < numOfThreads; i++) {
086                        PersonAlgorithm algo = algoProvider.getPersonAlgorithm();
087                        if (i == 0) {
088                                name = algo.getClass().getSimpleName();
089                                counter = new Counter("[" + name + "] handled person # ");
090                        }
091                        PersonAlgoThread algothread = new PersonAlgoThread(algo, counter);
092                        Thread thread = new Thread(algothread, name + "." + i);
093                        thread.setUncaughtExceptionHandler(uncaughtExceptionHandler);
094                        threads[i] = thread;
095                        algoThreads[i] = algothread;
096                }
097
098                // distribute workload between threads, as long as threads are not yet started, so we don't need synchronized data structures
099                int i = 0;
100                for (Person person : population.getPersons().values()) {
101                        algoThreads[i % numOfThreads].handlePerson(person);
102                        i++;
103                }
104
105                // start the threads
106                for (Thread thread : threads) {
107                        thread.start();
108                }
109
110                // wait for the threads to finish
111                try {
112                        for (Thread thread : threads) {
113                                thread.join();
114                        }
115                        counter.printCounter();
116                } catch (InterruptedException e) {
117                        throw new RuntimeException(e);
118                }
119                if (hadException.get()) {
120                        throw new RuntimeException("Exception while processing persons. Cannot guarantee that all persons have been fully processed.");
121                }
122        }
123
124        /**
125         * The thread class that really handles the persons.
126         */
127        private static class PersonAlgoThread implements Runnable {
128
129                private final PersonAlgorithm personAlgo;
130                private final List<Person> persons = new LinkedList<Person>();
131                private final Counter counter;
132
133                public PersonAlgoThread(final PersonAlgorithm algo, final Counter counter) {
134                        this.personAlgo = algo;
135                        this.counter = counter;
136                }
137
138                public void handlePerson(final Person person) {
139                        this.persons.add(person);
140                }
141
142                @Override
143                public void run() {
144                        for (Person person : this.persons) {
145                                this.personAlgo.run(person);
146                                counter.incCounter();
147                        }
148                }
149        }
150        
151        /**
152         * @author mrieser
153         */
154        private static class ExceptionHandler implements UncaughtExceptionHandler {
155
156                private final AtomicBoolean hadException;
157
158                public ExceptionHandler(final AtomicBoolean hadException) {
159                        this.hadException = hadException;
160                }
161
162                @Override
163                public void uncaughtException(Thread t, Throwable e) {
164                        log.error("Thread " + t.getName() + " died with exception while handling events.", e);
165                        this.hadException.set(true);
166                }
167
168        }
169
