049        /**
050         * Returns all nodes the route passes between the start- and the end-link of the route.
051         *
052         * @param route
053         * @param network
054         * @return
055         */
056        public static List<Node> getNodes(final NetworkRoute route, final Network network) {
057                List<Id<Link>> linkIds = route.getLinkIds();
058                List<Node> nodes = new ArrayList<>(linkIds.size() + 1);
059                if ((linkIds.size() > 0)) {
060                        nodes.add(network.getLinks().get(linkIds.get(0)).getFromNode());
061                        for (Id<Link> linkId : linkIds) {
062                                Link link = network.getLinks().get(linkId);
063                                nodes.add(link.getToNode());
064                        }
065                } else if (!route.getStartLinkId().equals(route.getEndLinkId())) {
066                        nodes.add(network.getLinks().get(route.getStartLinkId()).getToNode());
067                }
068                return nodes;
069        }
070
071        public static List<Link> getLinksFromNodes(final List<Node> nodes) {
072                ArrayList<Link> links = new ArrayList<>(nodes.size());
073                Node prevNode = null;
074                for (Node node : nodes) {
075                        if (prevNode != null) {
076                                Link foundLink = findLink(prevNode, node);
077                                if (foundLink != null) {
078                                        links.add(foundLink);
079                                }
080                        }
081                        prevNode = node;
082                }
083                links.trimToSize();
084                return links;
085        }
086
087        public static List<Link> getLinksFromNodeIds(final Network network, final List<Id<Node>> nodeIds) {
088                ArrayList<Link> links = new ArrayList<>(nodeIds.size());
089                Node prevNode = null;
090                for (Id<Node> nodeId : nodeIds) {
091                        Node node = network.getNodes().get(nodeId);
092                        if (prevNode != null) {
093                                Link foundLink = findLink(prevNode, node);
094                                if (foundLink != null) {
095                                        links.add(foundLink);
096                                }
097                        }
098                        prevNode = node;
099                }
100                links.trimToSize();
101                return links;
102        }
103
104        private static Link findLink(Node prevNode, Node node) {
105                for (Link link : prevNode.getOutLinks().values()) {
106                        if (link.getToNode().equals(node)) {
107                                return link;
108                        }
109                }
110                return null;
111        }

113        public static NetworkRoute getSubRoute(final NetworkRoute route, final Node fromNode, final Node toNode, final Network network) {
114                Id<Link> fromLinkId = null;
115                Id<Link> toLinkId = null;
116
117                List<Id<Link>> linkIds = new ArrayList<>(route.getLinkIds().size() + 2);
118                linkIds.add(route.getStartLinkId());
119                linkIds.addAll(route.getLinkIds());
120                linkIds.add(route.getEndLinkId());
121                for (Id<Link> linkId : linkIds) {
122                        Link link = network.getLinks().get(linkId);
123                        if (link.getToNode() == fromNode) {
124                                fromLinkId = link.getId();
125                        }
126                        if (link.getFromNode() == toNode) {
127                                toLinkId = link.getId();
128                                break; // we found the toLinkId, so we can stop searching, fromLinkId should be before after all
129                        }
130                }
131
132                return route.getSubRoute(fromLinkId, toLinkId);
133        }

135        /**
136         * Calculates the distance of the complete route, <b>excluding</b> the distance traveled
137         * on the start- and end-link of the route.
138         *
139         * @param route
140         * @param network
141         * @return
142         */
143        public static double calcDistanceExcludingStartEndLink(final NetworkRoute route, final Network network) {
144                double dist = 0;
145                for (Id<Link> linkId : route.getLinkIds()) {
146                        dist += network.getLinks().get(linkId).getLength();
147                }
148                return dist;
149        }

152        /**
153         * Calculates the distance of the complete route, <b>including</b> the distance traveled
154         * on the start- and end-link of the route.
155         * 
156         * @param networkRoute
157         * @param relPosOnDepartureLink relative position on the departure link where vehicles start traveling
158         * @param relPosOnArrivalLink relative position on the arrival link where vehicles stop traveling
159         * @param network
160         * @return
161         */
162        public static double calcDistance(final NetworkRoute networkRoute, final double relPosOnDepartureLink, final double relPosOnArrivalLink, final Network network) {
163                // sum distance of all link besides departure and arrival link
164                double routeDistance = calcDistanceExcludingStartEndLink(networkRoute, network);
165                // add relative distance of departure link
166                routeDistance += network.getLinks().get(networkRoute.getStartLinkId()).getLength() * (1.0 - relPosOnDepartureLink);
167                if (!networkRoute.getStartLinkId().equals(networkRoute.getEndLinkId())){
168                        // add relative distance of arrival link
169                        routeDistance += network.getLinks().get(networkRoute.getEndLinkId()).getLength() * relPosOnArrivalLink;
170                } else { // i.e. departure = arrival link
171                        // subtract relative distance of arrival link that is not traveled
172                        routeDistance -= network.getLinks().get(networkRoute.getEndLinkId()).getLength() * (1.0 - relPosOnArrivalLink);
173                }
174                return routeDistance;
175        }

177        public static double calcDistance( final LeastCostPathCalculator.Path path ) {
178                double length = 0. ;
179                for ( Link link : path.links ) {
180                        length += link.getLength() ;
181                }
182                return length ;
183        }

185        public static NetworkRoute createNetworkRoute(List<Id<Link>> routeLinkIds, final Network network) {
186                Id<Link> startLinkId = routeLinkIds.get(0);
187                List<Id<Link>> linksBetween = (routeLinkIds.size() > 2) ? routeLinkIds.subList(1, routeLinkIds.size() - 1) : new ArrayList<>(0);
188                Id<Link> endLinkId = routeLinkIds.get(routeLinkIds.size() - 1);
189                NetworkRoute route = createLinkNetworkRouteImpl(startLinkId, endLinkId);
190                route.setLinkIds(startLinkId, linksBetween, endLinkId);
191                return route;
192        }

194        public static double calcDistance(ExperimentalTransitRoute route, TransitSchedule ts, Network network) {
195                
196                Id<TransitLine> lineId = route.getLineId();
197                Id<TransitRoute> routeId = route.getRouteId();
198                Id<TransitStopFacility> enterStopId = route.getAccessStopId();
199                Id<TransitStopFacility> exitStopId = route.getEgressStopId();
200        
201                TransitLine line = ts.getTransitLines().get(lineId);
202                TransitRoute tr = line.getRoutes().get(routeId);
203        
204                Id<Link> enterLinkId = ts.getFacilities().get(enterStopId).getLinkId();
205                Id<Link> exitLinkId = ts.getFacilities().get(exitStopId).getLinkId();
206        
207                NetworkRoute nr = tr.getRoute();
208                double dist = 0;
209                boolean count = false;
210                if (enterLinkId.equals(nr.getStartLinkId())) {
211                        count = true;
212                }
213                for (Id<Link> linkId : nr.getLinkIds()) {
214                        if (count) {
215                                Link l = network.getLinks().get(linkId);
216                                if ( l==null ) {
217                                        log.error( "link is null; linkId=" + linkId + "; network=" + network ) ;
218                                }
219                                dist += l.getLength();
220                        }
221                        if (enterLinkId.equals(linkId)) {
222                                count = true;
223                        }
224                        if (exitLinkId.equals(linkId)) {
225                                count = false;
226                                break;
227                        }
228                }
229                if (count) {
230                        Link l = network.getLinks().get(nr.getEndLinkId());
231                        dist += l.getLength();
232                }
233                return dist;
234        }

236        /**
237         * How much of route is "covered" by the links of route2.  Based on Ramming.  Note that this is not symmetric,
238         * i.e. route1 can be fully covered by route2, but not the other way around.  kai, nov'13
239         * 
240         * @param route1
241         * @param route2
242         * @return a number between 0 (no coverage) and 1 (route2 fully covers route1)
243         */
244        public static double calculateCoverage(NetworkRoute route1, NetworkRoute route2, Network network ) {
245                double routeLength = 0. ;
246                double coveredLength = 0. ;
247                for ( Id<Link> id : route1.getLinkIds() ) {
248                        routeLength += network.getLinks().get( id ).getLength() ;
249                        if ( route2.getLinkIds().contains(id) ) {
250                                coveredLength += network.getLinks().get( id ).getLength() ;
251                        }
252                }
253                if ( routeLength > 0. ) {
254                        return coveredLength/routeLength ;
255                } else {
256                        return 1. ; // route has zero length = fully covered by any other route.  (but they are not similar!?!?!?)
257                }
258        }
