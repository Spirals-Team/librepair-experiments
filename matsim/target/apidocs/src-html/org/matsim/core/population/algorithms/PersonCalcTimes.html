021package org.matsim.core.population.algorithms;
022
023import java.util.List;
024
025import org.matsim.api.core.v01.population.Activity;
026import org.matsim.api.core.v01.population.Leg;
027import org.matsim.api.core.v01.population.Person;
028import org.matsim.api.core.v01.population.Plan;
029import org.matsim.api.core.v01.population.PlanElement;
030import org.matsim.core.utils.misc.Time;
031
032/**
033 * Calculates all time informations in all plans of a person<br>
034 * The algorithm does the following steps:
035 * <ul>
036 * <li>set start-time of first act to 00:00</li>
037 * <li>use end-time of first act as "current" time</li>
038 * <li>sets start-time, duration, end-time of acts and departure-time,
039 * travel-time, arrival-time of legs by continually adding duration of acts
040 * and travel-time of legs to the current time.</li>
041 * <li>the last act is extended until 24:00 if it ends before this time.
042 * Otherwise, a duration of 0 is assumed for the last act.</li>
043 * </ul>
044 * If a leg has no travel-time set, 0 will be used instead.
045 */
046public final class PersonCalcTimes extends AbstractPersonAlgorithm {
047
048        //////////////////////////////////////////////////////////////////////
049        // constructors
050        //////////////////////////////////////////////////////////////////////
051
052        public PersonCalcTimes() {
053                super();
054        }
055
056        //////////////////////////////////////////////////////////////////////
057        // run methods
058        //////////////////////////////////////////////////////////////////////
059
060  @Override
061        public void run(Person person) {
062                List<? extends Plan> plans = person.getPlans();
063                for (int i=0; i<plans.size(); i++) {
064                        Plan plan = plans.get(i);
065
066                        Activity act = null;
067                        Leg leg = null;
068                        int cnt = 0;
069                        int max = plan.getPlanElements().size();
070                        for (PlanElement pe : plan.getPlanElements()) {
071                                cnt++;
072                                if (pe instanceof Activity) {
073                                        act = (Activity) pe;
074
075                                        if (cnt == 1) {
076                                                // first activity
077                                                act.setStartTime(0);
078                                                act.setMaximumDuration(act.getEndTime());
079                                        } else if (cnt == max) {
080                                                double time = leg.getDepartureTime() + leg.getTravelTime();
081                                                act.setStartTime(time);
082                                                if (time < 24*3600) {
083                                                        time = 24*3600;
084                                                }
085                                                act.setEndTime(time);
086                                                act.setMaximumDuration(time - act.getStartTime());
087                                        }
088                                        else {
089                                                act.setStartTime(leg.getDepartureTime() + leg.getTravelTime());
090                                                act.setEndTime(act.getStartTime()+act.getMaximumDuration());
091                                        }
092                                }
093                                if (pe instanceof Leg) {
094                                        leg = (Leg) pe;
095
096                                        leg.setDepartureTime(act.getEndTime());
097                                        double ttime = leg.getTravelTime();
098                                        if (ttime == Time.UNDEFINED_TIME) {
099                                                ttime = 0;
100                                        }
101                                        leg.setTravelTime( leg.getDepartureTime() + ttime - leg.getDepartureTime() );
102                                }
103                        }
104                }
105  }
106}
