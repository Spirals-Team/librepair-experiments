031        public static void removeUnselectedPlans(Person person) {
032                for (Iterator<? extends Plan> iter = person.getPlans().iterator(); iter.hasNext(); ) {
033                        Plan plan = iter.next();
034                        if (!PersonUtils.isSelected(plan)) {
035                                iter.remove();
036                        }
037                }
038        }

133        @Deprecated // yyyy is there a way to use person.getAttributes instead??  kai, nov'16
134        public static void addTravelcard(Person person, final String type) {
135                if (getTravelcards(person) == null) {
136                        person.getCustomAttributes().put(TRAVELCARDS, new TreeSet<String>());
137                }
138                if (getTravelcards(person).contains(type)) {
139                        log.info(person + "[type=" + type + " already exists]");
140                } else {
141                        getTravelcards(person).add(type.intern());
142                }
143        }
