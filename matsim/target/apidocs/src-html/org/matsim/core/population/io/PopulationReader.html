021package org.matsim.core.population.io;
022
023import java.util.HashMap;
024import java.util.Map;
025import java.util.Stack;
026
027import org.apache.log4j.Logger;
028import org.matsim.api.core.v01.Scenario;
029import org.matsim.core.population.io.StreamingPopulationReader.StreamingPopulation;
030import org.matsim.core.utils.geometry.CoordinateTransformation;
031import org.matsim.core.utils.geometry.transformations.IdentityTransformation;
032import org.matsim.core.utils.io.MatsimXmlParser;
033import org.matsim.utils.objectattributes.AttributeConverter;
034import org.xml.sax.Attributes;
035
036/**
037 * A population reader that reads the MATSim format. This reader recognizes the format of the plans-file and uses
038 * the correct reader for the specific plans-version, without manual setting.
039 *
040 * @author mrieser
041 */
042public final class PopulationReader extends MatsimXmlParser {
043
044        private final static String PLANS    = "plans.dtd"; // a special, inofficial case, handle it like plans_v0
045        private final static String PLANS_V0 = "plans_v0.dtd";
046        private final static String PLANS_V1 = "plans_v1.dtd";
047        private final static String PLANS_V4 = "plans_v4.dtd";
048        private final static String POPULATION_V5 = "population_v5.dtd";
049        private final static String POPULATION_V6 = "population_v6.dtd";
050
051        private final CoordinateTransformation coordinateTransformation;
052
053        private MatsimXmlParser delegate = null;
054        private final Scenario scenario;
055
056        private Map<Class<?>, AttributeConverter<?>> attributeConverters = new HashMap<>();
057
058        private static final Logger log = Logger.getLogger(PopulationReader.class);
059
060        public PopulationReader(final Scenario scenario) {
061                this( new IdentityTransformation() , scenario );
062        }
063
064        public PopulationReader(
065                        final CoordinateTransformation coordinateTransformation,
066                        final Scenario scenario ) {
067                this( coordinateTransformation, scenario, false ) ;
068        }
069        
070        /*deliberately package*/ PopulationReader(
071                                final CoordinateTransformation coordinateTransformation,
072                                final Scenario scenario, boolean streaming ) {
073                if ( !streaming && scenario.getPopulation() instanceof StreamingPopulation ) {
074                        throw new RuntimeException("MatsimPopulationReader called directly with an instance of StreamingPopulation "
075                                        + "in scenario.  Call via StreamingPopulationReader or ask for help.  kai, jul'16") ;
076                }
077                this.coordinateTransformation = coordinateTransformation;
078                this.scenario = scenario;
079        }
080
081        public void putAttributeConverter( final Class<?> clazz , AttributeConverter<?> converter ) {
082                attributeConverters.put( clazz , converter );
083        }
084
085        public void putAttributeConverters( final Map<Class<?>, AttributeConverter<?>> converters ) {
086                attributeConverters.putAll( converters );
087        }
088
089        @Override
090        public void startTag(final String name, final Attributes atts, final Stack<String> context) {
091                this.delegate.startTag(name, atts, context);
092        }
093
094        @Override
095        public void endTag(final String name, final String content, final Stack<String> context) {
096                this.delegate.endTag(name, content, context);
097        }
098
099        @Override
100        protected void setDoctype(final String doctype) {
101                super.setDoctype(doctype);
102                switch ( doctype ) {
103                        case POPULATION_V6:
104                                this.delegate =
105                                                new PopulationReaderMatsimV6(
106                                                                coordinateTransformation,
107                                                                this.scenario);
108                                ((PopulationReaderMatsimV6) delegate).putAttributeConverters( attributeConverters );
109                                log.info("using population_v6-reader.");
110                                break;
111                        case POPULATION_V5:
112                                this.delegate =
113                                                new PopulationReaderMatsimV5(
114                                                                coordinateTransformation,
115                                                                this.scenario);
116                                log.info("using population_v5-reader.");
117                                break;
118                        case PLANS_V4:
119                                // Replaced non-parallel reader with parallel implementation. cdobler, mar'12.
120                                this.delegate =
121                                                new ParallelPopulationReaderMatsimV4(
122                                                                coordinateTransformation,
123                                                                this.scenario);
124                                log.info("using plans_v4-reader.");
125                                break;
126                        case PLANS_V1:
127                                this.delegate =
128                                                new PopulationReaderMatsimV1(
129                                                                coordinateTransformation,
130                                                                this.scenario);
131                                log.info("using plans_v1-reader.");
132                                break;
133                        case PLANS_V0:
134                        case PLANS:
135                                this.delegate =
136                                                new PopulationReaderMatsimV0(
137                                                                coordinateTransformation,
138                                                                this.scenario);
139                                log.info("using plans_v0-reader.");
140                                break;
141                        default:
142                                throw new IllegalArgumentException("No population reader available for doctype \"" + doctype + "\".");
143                }
144        }
145
146}
