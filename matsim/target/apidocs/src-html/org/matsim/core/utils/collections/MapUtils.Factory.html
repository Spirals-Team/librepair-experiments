030/**
031 * Utils class for the common pattern of creating an entry in a map if nothing is associated to the requested key.
032 * This is typically useful for weights or counters (Double or Integer), or one-to-many mappings (Lists or Sets).
033 *
034 * @author thibautd
035 */
036public final class MapUtils {
037        private MapUtils() {}
038
039        /**
040         * Gets the collection associated with the key in this map, or create an empty list if no mapping exists yet.
041         *
042         * @param key the key of the mapping
043         * @param map the map in which to search
044         * @param <K> type of the key
045         * @param <V> parameter of the collection type
046         * @return the collection (evt. newly) associated with the key
047         */
048        public static <K,V> Collection<V> getCollection(
049                        final K key,
050                        final Map<K, Collection<V>> map) {
051                Collection<V> coll = map.get( key );
052
053                if ( coll == null ) {
054                        coll = new ArrayList<V>();
055                        map.put( key , coll );
056                }
057
058                return coll;
059        }
060
061        /**
062         * Gets the collection associated with the key in this map, or create an empty list if no mapping exists yet.
063         *
064         * @param key the key of the mapping
065         * @param map the map in which to search
066         * @param <K> type of the key
067         * @param <V> parameter of the List type
068         * @return the List (evt. newly) associated with the key
069         */
070        public static <K,V> List<V> getList(
071                        final K key,
072                        final Map<K, List<V>> map) {
073                List<V> coll = map.get( key );
074
075                if ( coll == null ) {
076                        coll = new ArrayList<V>();
077                        map.put( key , coll );
078                }
079
080                return coll;
081        }

083        /**
084         * Gets the set associated with the key in this map, or create an empty set if no mapping exists yet.
085         *
086         * @param key the key of the mapping
087         * @param map the map in which to search
088         * @param <K> type of the key
089         * @param <V> parameter of the set type
090         * @return the set (evt. newly) associated with the key
091         */
092        public static <K,V> Set<V> getSet(
093                        final K key,
094                        final Map<K, Set<V>> map) {
095                Set<V> coll = map.get( key );
096
097                if ( coll == null ) {
098                        coll = new HashSet<V>();
099                        map.put( key , coll );
100                }
101
102                return coll;
103        }
104
105        /**
106         * Gets the map associated with the key in this map, or create an empty map if no mapping exists yet.
107         *
108         * @param key the key of the mapping
109         * @param map the map in which to search
110         * @param <K> type of the in the primary map
111         * @param <C> type of the key in the secondary map
112         * @param <V> type of the values in the secondary map
113         * @return the Map (evt. newly) associated with the key
114         */
115        public static <K,C,V> Map<C,V> getMap(
116                        final K key,
117                        final Map<K, Map<C,V>> map) {
118                Map<C,V> coll = map.get( key );
119
120                if ( coll == null ) {
121                        coll = new HashMap<C,V>();
122                        map.put( key , coll );
123                }
124
125                return coll;
126        }
127
128        /**
129         * Gets the object associated with the key in this map, or create one if no mapping exists yet.
130         *
131         * @param key the key of the mapping
132         * @param map the map in which to search
133         * @param <K> type of the key
134         * @param <T> parameter of the values type
135         * @param fact the factory to use to create a default object if none is found
136         * @return the Object (evt. newly) associated with the key
137         */
138        public static <K,T> T getArbitraryObject(
139                        final K key,
140                        final Map<K, T> map,
141                        final Factory<T> fact) {
142                T coll = map.get( key );
143
144                if ( coll == null ) {
145                        coll = fact.create();
146                        map.put( key , coll );
147                }
148
149                return coll;
150        }
151
152        /**
153         * Gets the double associated with the key in this map, or create a new one if no mapping exists yet.
154         *
155         * @param key the key of the mapping
156         * @param map the map in which to search
157         * @param initialValue the value to which new entries should be initialized
158         * @param <K> type of the key
159         * @return the value (evt. newly) associated with the key
160         */
161        public static <K> Double getDouble(
162                        final K key,
163                        final Map<K, Double> map,
164                        final double initialValue) {
165                Double d = map.get( key );
166
167                if ( d == null ) {
168                        d = initialValue;
169                        map.put( key , d );
170                }
171
172                return d;
173        }
174
175        /**
176         * Add a given value to the Double associated with the key in this map, or initialize a new one if no mapping exists
177         * yet.
178         *
179         * @param key the key of the mapping
180         * @param map the map in which to search
181         * @param initialValue the value to which new entries should be initialized
182         * @param toAdd the value to add to the existing mapped values
183         * @param <K> type of the key
184         * @return the collection (evt. newly) associated with the key
185         */
186        public static <K> double addToDouble(
187                        final K key,
188                        final Map<K, Double> map,
189                        final double initialValue,
190                        final double toAdd) {
191                final double newValue =
192                        getDouble( key , map , initialValue ) +
193                        toAdd;
194                map.put( key , newValue );
195                return newValue;
196        }

222        /**
223         * Add a given value to the Integer associated with the key in this map, or initialize a new one if no mapping exists
224         * yet.
225         *
226         * @param key the key of the mapping
227         * @param map the map in which to search
228         * @param initialValue the value to which new entries should be initialized
229         * @param toAdd the value to add to the existing mapped values
230         * @param <K> type of the key
231         * @return the collection (evt. newly) associated with the key
232         */
233        public static <K> double addToInteger(
234                        final K key,
235                        final Map<K, Integer> map,
236                        final int initialValue,
237                        final int toAdd) {
238                final int newValue =
239                        getInteger( key , map , initialValue ) +
240                        toAdd;
241                map.put( key , newValue );
242                return newValue;
243        }
244
245        /**
246         * Fill in a collection with the values associated with the requested keys
247         *
248         * @param keys the keys to search for. If a key appears several times, its value will be added to the collection the
249         *             corresponding number of times.
250         * @param map the map in which to search
251         * @param <K> type of the key
252         * @param <V> type of the values
253         * @return a new collection, filled in with the values associated with the keys provided, in iteration order.
254         */
255        public static <K,V> Collection<V> get(
256                        final Iterable<K> keys,
257                        final Map<K, V> map ) {
258                final List<V> coll = new ArrayList<V>();
259
260                for ( K k : keys ) coll.add( map.get( k ) );
261
262                return coll;
263        }
264
265        public interface Factory<T> {
266                T create();
267        }
268
269        /**
270         * Helper class, to use as a factory if objects should be instanciated using their parameter-less constructor.
271         *
272         * @param <T>
273         */
274        public static class DefaultFactory<T> implements Factory<T> {
275                private final Class<? extends T> theClass;
276
277                public DefaultFactory( final Class<? extends T> theClass ) {
278                        this.theClass = theClass;
279                }
280
281                /**
282                 * @throws RuntimeException if the constructor does not exist or throws an exception
283                 * @return a new instance
284                 */
285                @Override
286                public T create() {
287                        try {
288                                return theClass.getConstructor().newInstance();
289                        } catch (Exception e) {
290                                throw new RuntimeException( e );
291                        }
292                }
293        }
