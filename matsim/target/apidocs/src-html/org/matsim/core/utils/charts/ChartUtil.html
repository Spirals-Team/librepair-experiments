021package org.matsim.core.utils.charts;
022
023import java.awt.Color;
024import java.awt.Image;
025import java.io.File;
026import java.io.IOException;
027
028import org.jfree.chart.ChartUtilities;
029import org.jfree.chart.JFreeChart;
030import org.jfree.chart.title.ImageTitle;
031import org.jfree.chart.title.Title;
032import org.jfree.ui.HorizontalAlignment;
033import org.jfree.ui.RectangleEdge;
034import org.jfree.ui.VerticalAlignment;
035import org.matsim.core.gbl.MatsimResource;
036
037/**
038 * An abstract class to simplify the creation of JFreeCharts.
039 *
040 * @author mrieser
041 * @author ychen
042 */
043public abstract class ChartUtil {
044
045        protected final String chartTitle;
046        protected final String xAxisLabel;
047        protected final String yAxisLabel;
048        protected JFreeChart chart = null;
049
050        public ChartUtil(final String title, final String xAxisLabel, final String yAxisLabel) {
051                this.chartTitle = title;
052                this.xAxisLabel = xAxisLabel;
053                this.yAxisLabel = yAxisLabel;
054        }
055
056        public abstract JFreeChart getChart();
057
058        /**
059         * Stores the chart as PNG image file.
060         *
061         * @param filename The filename of the resulting PNG file.
062         * @param width The width of the chart.
063         * @param height The height of the chart.
064         */
065        public void saveAsPng(final String filename, final int width, final int height) {
066                try {
067                        ChartUtilities.saveChartAsPNG(new File(filename), getChart(), width, height, null, true, 9);
068                } catch (IOException e) {
069                        e.printStackTrace();
070                } catch ( Exception e ) {
071                        e.printStackTrace() ;
072                        // I just had an out of bounds error inside the method; don't know what that means but does not feel like a reason 
073                        // to not continue.  kai, apr'30
074                }
075        }
076
077        /**
078         * Adds the MATSim Logo in the lower right corner of the chart.
079         */
080        public void addMatsimLogo() {
081                try {
082                        Image image = MatsimResource.getAsImage("matsim_logo_transparent_small.png");
083                        Title subtitle = new ImageTitle(image, RectangleEdge.BOTTOM, HorizontalAlignment.RIGHT, VerticalAlignment.BOTTOM);
084                        this.chart.addSubtitle(subtitle);
085                } catch ( Exception e ) {
086                        e.printStackTrace() ;
087                        // I just had a resource-not-found error inside the method; don't know what that means but does not feel like a reason 
088                        // to not continue.  kai, apr'30
089                }
090        }
091
092        /**
093         * Adds default formatting options for the charts, like a white background etc.
094         * Requires the member {@link #chart} to be set by the overriding class!
095         */
096        protected void addDefaultFormatting() {
097                this.chart.setBackgroundPaint(new Color(1.0f, 1.0f, 1.0f, 1.0f));
098                this.chart.getLegend().setBorder(0.0, 0.0, 0.0, 0.0);
099        }
100
101}
