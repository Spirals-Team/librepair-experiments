040/**
041 * An abstract XML-Parser which can be easily extended for reading custom XML-formats. This class handles all the low level
042 * functionality required to parse xml-files. Extending classes have only to implement {@link #startTag} and {@link #endTag}
043 * to implement a custom parser.<br>
044 * The parser implements a custom <code>EntityResolver</code> to look for DTDs in the MATSim world.
045 * <p></p>
046 * Notes:<ul>
047 * <li> If implementing classes want to override the final methods, the will have to resort to delegation.   
048 * </ul>
049 *
050 * @author mrieser
051 */

136        /**
137         * Parses the specified file. The file can be gzip-compressed and is decompressed on-the-fly while parsing. A gzip-compressed

149                parse(new InputSource(IOUtils.getBufferedReader(filename)));

152        public final void parse(final URL url) throws UncheckedIOException {
153                Gbl.assertNotNull(url);
154                this.theSource = url.toString();
155                log.info("starting to parse xml from url " + this.theSource + " ...");
156                System.out.flush();
157                if (url.getFile().endsWith(".gz")) {
158                        try {
159                                parse(new InputSource(new GZIPInputStream(url.openStream())));
160                        } catch (IOException e) {
161                                throw new RuntimeException(e);
162                        }
163                } else {
164                        parse(new InputSource(url.toExternalForm()));
165                }
166        }

168        public final void parse(final InputStream stream) throws UncheckedIOException {
169                this.theSource = "stream";
170                parse(new InputSource(stream));
171        }

173        public final void parse(final InputSource input) throws UncheckedIOException {
174                try {
175                        SAXParserFactory factory = SAXParserFactory.newInstance();
176                        factory.setValidating(this.isValidating);
177                        factory.setNamespaceAware(this.isNamespaceAware);
178                        if (this.isValidating) {
179                                // enable optional support for XML Schemas
180                                factory.setFeature("http://apache.org/xml/features/validation/schema", true);
181                                SAXParser parser = factory.newSAXParser();
182                                XMLReader reader = parser.getXMLReader();
183                                reader.setContentHandler(this);
184//                              reader.setErrorHandler(getErrorHandler());      // (**)
185//                              reader.setEntityResolver(getEntityResolver()); // (**)
186                                reader.setErrorHandler(this);
187                                reader.setEntityResolver(this);
188                                reader.parse(input);
189                        } else {
190                                SAXParser parser = factory.newSAXParser();
191                                parser.parse(input, this);
192                        }
193                } catch (SAXException | ParserConfigurationException | IOException e) {
194                        throw new UncheckedIOException(e);
195                }
196        }

221        @Override
222        public final InputSource resolveEntity(final String publicId, final String systemId) {
223                // ConfigReader* did override this.  Not sure if it did that for good reaons.  kai, jul'16
224                
225                // extract the last part of the systemId
226                int index = systemId.replace('\\', '/').lastIndexOf('/');
227                String shortSystemId = systemId.substring(index + 1);
228
229                if (this.doctype == null) {
230                        // this is the first systemId we have to resolve, assume it's the doctype
231                        // I haven't found any other way to determine the doctype of the currently read file
232                        setDoctype(shortSystemId);
233                }
234
235                InputSource source;
236                if (this.preferLocalDtds) {
237                        source = findDtdInLocalFilesystem(shortSystemId);
238                        if (source == null) {
239                                source = findDtdInClasspath(shortSystemId);
240                        }
241                        if (source == null) {
242                                source = findDtdInDefaultLocation(shortSystemId);
243                        }
244                        if (source == null) {
245                                source = findDtdInRemoteLocation(systemId);
246                        }
247                } else {
248                        source = findDtdInRemoteLocation(systemId);
249                        if (source == null) {
250                                source = findDtdInLocalFilesystem(shortSystemId);
251                        }
252                        if (source == null) {
253                                source = findDtdInClasspath(shortSystemId);
254                        }
255                        if (source == null) {
256                                source = findDtdInDefaultLocation(shortSystemId);
257                        }
258                }
259
260                if (source == null) {
261                        // We could neither get the remote nor the local version of the dtd, show a warning
262                        log.warn("Could neither get the DTD from the web nor a local one. " + systemId);
263                } else {
264            source.setSystemId(systemId);
265        }
266                return source;
267    }

269        private static InputSource findDtdInRemoteLocation(final String fullSystemId) {
270                log.info("Trying to load " + fullSystemId + ". In some cases (e.g. network interface up but no connection), this may take a bit.");
271                try {
272                        URL url = new URL(fullSystemId);
273                        URLConnection urlConn = url.openConnection();
274                        urlConn.setConnectTimeout(5000);
275                        urlConn.setReadTimeout(5000);
276                        urlConn.setAllowUserInteraction(false);         
277                        
278                        InputStream is = urlConn.getInputStream();
279                        /* If there was no exception until here, than the path is valid.
280                         * Return the opened stream as a source. If we would return null, then the SAX-Parser
281                         * would have to fetch the same file again, requiring two accesses to the webserver */
282                        return new InputSource(is);
283                } catch (IOException e) {
284                        // There was a problem getting the (remote) file, just show the error as information for the user
285                        log.info(e.toString() + ". May not be fatal, will try to load it locally.");
286                }
287                return null;
288        }
289        
290        private InputSource findDtdInLocalFilesystem(final String shortSystemId) {
291                if (this.localDtdBase != null) {
292                        String localFileName = this.localDtdBase + "/" + shortSystemId;
293                        File dtdFile = new File(localFileName);
294//                      log.debug("dtdfile: " + dtdFile.getAbsolutePath());
295                        if (dtdFile.exists() && dtdFile.isFile() && dtdFile.canRead()) {
296                                log.info("Using the local DTD " + localFileName + " with absolute path " + dtdFile.getAbsolutePath() );
297                                return new InputSource(dtdFile.getAbsolutePath());
298                        }
299                }
300                return null;
301        }
302        
303        private InputSource findDtdInClasspath(final String shortSystemId) {
304                // still no success, try to load it with the ClassLoader, in case we're stuck in a jar...
305                InputStream stream = this.getClass().getResourceAsStream("/dtd/" + shortSystemId);
306                if (stream != null) {
307                        log.info("Using local DTD from classpath:dtd/" + shortSystemId);
308                        return new InputSource(stream);
309                }
310                return null;
311        }
312        
313        private static InputSource findDtdInDefaultLocation(final String shortSystemId) {
314                log.info("Trying to access local dtd folder at standard location ./dtd...");
315                File dtdFile = new File("./dtd/" + shortSystemId);
316                if (dtdFile.exists() && dtdFile.isFile() && dtdFile.canRead()) {
317                        log.info("Using the local DTD " + dtdFile.getAbsolutePath());
318                        return new InputSource(dtdFile.getAbsolutePath());
319                }
320                return null;
321        }

326        public void characters(final char[] ch, final int start, final int length) throws SAXException {
327                // has to be non-final since otherwise the events parser does not work.  Probably ok (this here is just a default implementation). kai, jul'16
328                
329                StringBuffer buffer = this.buffers.peek();
330                if (buffer != null) {
331                        buffer.append(ch, start, length);
332                }
333        }

335        @Override
336        public final void startElement(final String uri, final String localName, final String qName, Attributes atts) throws SAXException {
337                // I have not good intuition if making this one non-final might be ok.  kai, jul'16
338
339                String tag = (uri.length() == 0) ? qName : localName;
340                this.buffers.push(new StringBuffer());
341                this.startTag(tag, atts, this.theContext);
342                this.theContext.push(tag);
343        }
344
345        @Override
346        public final void endElement(final String uri, final String localName, final String qName) throws SAXException {
347                // I have not good intuition if making this one non-final might be ok.  kai, jul'16
348                
349                String tag = (uri.length() == 0) ? qName : localName;
350                this.theContext.pop();
351                StringBuffer buffer = this.buffers.pop();
352                this.endTag(tag, buffer.toString(), this.theContext);
353        }

357        @Override
358        public final void error(final SAXParseException ex) throws SAXException {
359                if (this.theContext.isEmpty()) {
360                        System.err.println("Missing DOCTYPE.");
361                }
362                System.err.println("XML-ERROR: " + getInputSource(ex) + ", line " + ex.getLineNumber() + ", column " + ex.getColumnNumber() + ":");
363                System.err.println(ex.toString());
364                throw ex;
365        }
