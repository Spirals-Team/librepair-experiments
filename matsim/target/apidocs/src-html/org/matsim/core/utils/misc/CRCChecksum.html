021package org.matsim.core.utils.misc;
022
023import java.io.BufferedInputStream;
024import java.io.File;
025import java.io.FileInputStream;
026import java.io.IOException;
027import java.io.InputStream;
028import java.util.zip.CRC32;
029import java.util.zip.CheckedInputStream;
030import java.util.zip.GZIPInputStream;
031
032import org.apache.log4j.Logger;
033
034public class CRCChecksum {
035        private static final Logger log = Logger.getLogger( CRCChecksum.class );
036
037        private static long getCRCFromStream(final InputStream in) {
038                long check = 0;
039                if (in == null) return 0;
040                CRC32 crc = new CRC32();
041                try (CheckedInputStream cis = new CheckedInputStream(in, crc)) {
042                        byte[] buffer = new byte[4096];
043                        while (cis.read(buffer) != -1) {
044                                /* Read until the end of the stream is reached. */
045                        }
046                        check = crc.getValue();
047                        cis.close();
048                }
049                catch (IOException e) {
050                        throw new RuntimeException(e);
051                }
052                return check;
053        }
054
055        /**
056         * Calculates the checksum of the content of the given file. If the filename ends in ".gz",
057         * the file is assumed to be gzipped and the checksum over the <em>uncompressed</em> content 
058         * will be calculated. If a file is not found at its expected place, it is searched via the class loader.
059         * <p></p>
060         * Comments:<ul>
061         * <li> Some version of this method, possibly the variant with the class loader, does some caching: If I replace
062         * the original file in a test case, I need to restart eclipse before it works correctly.  ???  kai, feb'14
063         * </ul>
064         * 
065         * @param filename
066         * @return CRC32-Checksum of the file's content.
067         */
068        public static long getCRCFromFile(final String filename) {
069                log.info("filename=" + filename ) ;
070                if (new File(filename).exists()) {
071                        log.info( "file exists");
072                        if (filename.endsWith(".gz")) {
073                                log.info( "file ends in gz");
074                                try ( InputStream in = new GZIPInputStream(new BufferedInputStream(new FileInputStream(filename))) ) {
075                                        long result = getCRCFromStream(in);
076                                        in.close();
077                                        return result ;
078                                } catch (IOException e) {
079                                        throw new RuntimeException(e);
080                                }
081                        } else { // not with gz
082                                log.info( "file does not end in gz");
083                                try ( InputStream in = new BufferedInputStream(new FileInputStream(filename)) ) {
084                                        long result = getCRCFromStream(in);
085                                        in.close();
086                                        return result ;
087                                } catch (IOException e) {
088                                        throw new RuntimeException(e);
089                                }
090                        }
091                } else { // does not exist
092                        log.info( "file does not exist; search via class loader");
093                        // (the logic is: if the file cannot be found directly on the file system, this is some method to search
094                        // in generic locations (which ones?).  kai, feb'14)
095                        if (filename.endsWith(".gz")) {
096                                log.info( "file ends in gz");
097                                try ( InputStream stream = CRCChecksum.class.getClassLoader().getResourceAsStream(filename) ;
098                                        InputStream in = new GZIPInputStream(new BufferedInputStream(stream)); ) {
099                                        long result = getCRCFromStream(in);
100                                        in.close();
101                                        return result ;
102                                } catch (IOException e) {
103                                        throw new RuntimeException(e);
104                                }
105                        } else { // not work gz
106                                log.info( "file does not end in gz");
107                                try ( InputStream stream = CRCChecksum.class.getClassLoader().getResourceAsStream(filename) ;
108                                        InputStream in = new BufferedInputStream(stream); ) {
109                                        long result = getCRCFromStream(in);
110                                        in.close();
111                                        return result ;
112                                } catch (IOException e) {
113                                        throw new RuntimeException(e);
114                                }
115                        }
116                }
117        }
118}
