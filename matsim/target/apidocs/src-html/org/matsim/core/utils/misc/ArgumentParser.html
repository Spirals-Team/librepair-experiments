021package org.matsim.core.utils.misc;
022
023import java.util.ArrayList;
024import java.util.Iterator;
025import java.util.List;
026
027/**
028 * <p>This class analyses arguments passed on program start-up and provides an
029 * iterator for easy access to the single options and arguments.
030 * The parser understands normal arguments (like filenames) and options
031 * (arguments beginning with - or --) and supports value-assignments of options
032 * (e.g. "-rate 10" or "-rate=10").</p>
033 *
034 * <p>Additionally, the parser can be instructed to specifically expect short
035 * options. In that case, options starting with one hyphen can only be one
036 * character long. If more than one character follow a single hyphen, each
037 * character will be interpreted as an option. The iterator returns every
038 * option on its own in that case.<br>
039 *
040 * Example: <code>tar -xzvf file</code><br />
041 * If shortOptions are enabled, the above argument list would be translated into
042 * <code>tar -x -z -v -f file</code>, whereas if shortOptions are disabled, a
043 * single option named "xzvf" should exist (like in <code>java -jar
044 * file.jar</code>).</p>
045 *
046 * <p>Value assignments to options can be delimitted by a space or the equal-sign.
047 * In both cases, option and value are separated into two entities and returned
048 * separately by the iterator.<br />
049 *
050 * Example: <code>myapp -start 5</code> or <code>myapp -start=5</code> both
051 * work.</p>
052 *
053 *
054 * @author mrieser
055 */
056public class ArgumentParser implements Iterable<String> {
057
058        private List<String> args = null;
059        private boolean enableShortOptions = true;
060
061        /**
062         * Constructs a new <code>ArgumentParser</code> with support for shortOptions
063         * disabled.
064         *
065         * @param args The arguments to parse.
066         */
067        public ArgumentParser(final String[] args) {
068                this(args, false);
069        }
070
071        /**
072         * Constructs a new <code>ArgumentParser</code> with the specified behaviour
073         * regarding shortOptions.
074         *
075         * @param args The arguments to parse.
076         * @param enableShortOptions whether options with a single hyphen should be
077         *              treated as short options (only one character long) or not.
078         */
079        public ArgumentParser(final String[] args, final boolean enableShortOptions) {
080                this.args = new ArrayList<String>();
081                this.enableShortOptions = enableShortOptions;
082                parse(args);
083        }
084
085
086        /**
087         * Returns an iterator over all single options and arguments.
088         */
089        @Override
090        public Iterator<String> iterator() {
091                return this.args.iterator();
092        }
093
094
095        private void parse(final String[] args) {
096                if (this.enableShortOptions) {
097                        for (String arg : args) {
098                                if (arg.startsWith("--")) {
099                                        parseLongOption(arg.substring(2));
100                                } else if (arg.startsWith("-")) {
101                                        parseShortOption(arg.substring(1));
102                                } else {
103                                        this.args.add(arg);
104                                }
105                        }
106                } else {
107                        for (String arg : args) {
108                                if (arg.startsWith("-")) {
109                                        parseOption(arg);
110                                } else {
111                                        this.args.add(arg);
112                                }
113                        }
114                }
115        }
116
117        private void parseShortOption(final String arg) {
118                if (arg.length() == 0) {
119                        this.args.add("-");
120                        return;
121                }
122                for (int i = 0; i < arg.length(); i++) {
123                        char ch = arg.charAt(i);
124                        if (ch == '=') {
125                                if ((i == 0) && (arg.length() == 1)) {
126                                        // '=' is the only char
127                                        this.args.add("-=");
128                                        return;
129                                } else if (i == 0) {
130                                        // arg is a string in the form of '-=*', for me this makes only sense as a string-param, so just add it
131                                        this.args.add('-' + arg);
132                                        return;
133                                } else {
134                                        this.args.add(arg.substring(i+1));
135                                        return;
136                                }
137                        }
138                        // else...
139                        this.args.add("-" + ch);
140                }
141        }
142
143        private void parseLongOption(final String arg) {
144                StringBuilder argname = new StringBuilder("--");
145                for (int i = 0; i < arg.length(); i++) {
146                        char ch = arg.charAt(i);
147                        if (ch == '=') {
148                                if ((i == 0) && (arg.length() == 1)) {
149                                        // '=' is the only char
150                                        this.args.add("--=");
151                                        return;
152                                } else if (i == 0) {
153                                        // arg is a string in the form of '--=*', interpret it as a string argument
154                                        this.args.add("--" + arg);
155                                        return;
156                                } else {
157                                        this.args.add(argname.toString());
158                                        this.args.add(arg.substring(i + 1));
159                                        return;
160                                }
161                        }
162                        // else...
163                        argname.append(ch);
164                }
165                this.args.add(argname.toString());
166        }
167
168        private void parseOption(final String arg) {
169                StringBuilder argname = new StringBuilder();
170                argname.append('-');  // every option starts with '-'
171                for (int i = 1; i < arg.length(); i++) { // options always start with '-', so we can start at char 1
172                        char ch = arg.charAt(i);
173                        if (ch == '=') {
174                                if (argname.toString().equals("-")) {
175                                        // arg is a string in the form of '-=*', interpret it as a strig argument
176                                        this.args.add(arg);
177                                        return;
178                                }
179                                if (argname.toString().equals("--")) {
180                                        // arg is a string in the form of '--=*', interpret it as a strig argument
181                                        this.args.add(arg);
182                                        return;
183                                }
184                                // it seems there are already some real characters before '=' that make a correct argument
185                                this.args.add(argname.toString());
186                                this.args.add(arg.substring(i + 1));
187                                return;
188                        }
189                        // else...
190                        argname.append(ch);
191                }
192                this.args.add(argname.toString());
193        }
194
195}
