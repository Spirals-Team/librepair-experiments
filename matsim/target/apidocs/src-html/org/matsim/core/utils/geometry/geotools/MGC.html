104        /**
105         * Converts a MATSim {@link org.matsim.api.core.v01.Coord} into a Geotools <code>Coordinate</code>
106         * @param coord MATSim coordinate
107         * @return Geotools coordinate
108         */
109        public static Coordinate coord2Coordinate(final Coord coord) {
110                return new Coordinate(coord.getX(), coord.getY());
111        }

114         * Converts a Geotools <code>Coordinate</code> into a MATSim {@link org.matsim.api.core.v01.Coord}
115         * @param coord MATSim coordinate
116         * @return Geotools coordinate
117         */
118        public static Coord coordinate2Coord(final Coordinate coord) {
119                return new Coord(coord.x, coord.y);

158        /**
159         * Generates a Geotools <code>CoordinateReferenceSystem</code> from a coordinate system <code>String</code>. The coordinate system
160         * can either be specified as shortened names, as defined in {@link TransformationFactory}, as EPSG Code or as
161         * Well-Known-Text (WKT) as supported by the GeoTools.
162         * @param wktOrAuthorityCodeOrShorthandName
163         * @return crs
164         */
165        public static CoordinateReferenceSystem getCRS(final String wktOrAuthorityCodeOrShorthandName) {
166                String wktOrAuthorityCode = COORDINATE_REFERENCE_SYSTEMS.get(wktOrAuthorityCodeOrShorthandName);
167                if (wktOrAuthorityCode == null) {
168                        wktOrAuthorityCode = wktOrAuthorityCodeOrShorthandName;
169                }
170                CoordinateReferenceSystem crs;
171                try {
172                        crs = CRS.parseWKT(wktOrAuthorityCode);
173                } catch (FactoryException fe) {
174                        try {
175                                log.warn("Assuming that coordinates are in longitude first notation, i.e. (longitude, latitude).");
176                                crs = CRS.decode(wktOrAuthorityCode, true);
177                        } catch (FactoryException e) {
178                                throw new IllegalArgumentException(e);
179                        }
180                }
181                return crs;
182        }

184        /**
185         * Guesses the  Universal Transverse Mercator (UTM) zone for a given WGS 84 coordinate, returns the corresponding 
186         * EPSG code  
187         * @param lon the longitude of the WGS 84 coordinate
188         * @param lat the latitude of the WGS 84 coordinate
189         * @return EPSG code
190         */
191        public static String getUTMEPSGCodeForWGS84Coordinate(final double lon, final double lat) {
192                int utmZone = (int) (Math.ceil((180+lon) / 6)+0.5);
193                String epsgCode = null;
194                if (lat > 0 ) { //northern hemisphere 
195                  epsgCode = "EPSG:326" + utmZone;
196                } else { //southern hemisphere
197                  epsgCode = "EPSG:327" + utmZone;
198                }
199                return epsgCode;
200        }
