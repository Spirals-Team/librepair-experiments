021package org.matsim.core.utils.charts;
022
023import org.jfree.chart.ChartFactory;
024import org.jfree.chart.JFreeChart;
025import org.jfree.chart.axis.LogarithmicAxis;
026import org.jfree.chart.plot.PlotOrientation;
027import org.jfree.chart.plot.XYPlot;
028import org.jfree.data.xy.XYSeries;
029import org.jfree.data.xy.XYSeriesCollection;
030
031/**
032 * Creates a new XY-ScatterChart.
033 * 
034 * @author mrieser
035 */
036public class XYScatterChart extends ChartUtil {
037
038        private final XYSeriesCollection dataset;
039        private final boolean isLogarithmicAxis;
040
041        public XYScatterChart(final String title, final String xAxisLabel,
042                        final String yAxisLabel) {
043                this(title, xAxisLabel, yAxisLabel, false);
044        }
045
046        public XYScatterChart(final String title, final String xAxisLabel,
047                        final String yAxisLabel, boolean isLogarithmicAxis) {
048                super(title, xAxisLabel, yAxisLabel);
049                this.isLogarithmicAxis = isLogarithmicAxis;
050                this.dataset = new XYSeriesCollection();
051                this.chart = createChart(title, xAxisLabel, yAxisLabel, this.dataset);
052                addDefaultFormatting();
053        }
054
055        @Override
056        public JFreeChart getChart() {
057                return this.chart;
058        }
059
060        private JFreeChart createChart(final String title,
061                        final String categoryAxisLabel, final String valueAxisLabel,
062                        final XYSeriesCollection dataset) {
063                JFreeChart c = ChartFactory.createScatterPlot(title, categoryAxisLabel,
064                                valueAxisLabel, dataset, PlotOrientation.VERTICAL, true, // legend?
065                                false, // tooltips?
066                                false // URLs?
067                                );
068                if (this.isLogarithmicAxis) {
069                        XYPlot p = (XYPlot) c.getPlot();
070                        LogarithmicAxis axis_x = new LogarithmicAxis(this.xAxisLabel);
071                        LogarithmicAxis axis_y = new LogarithmicAxis(this.yAxisLabel);
072                        axis_x.setAllowNegativesFlag(false);
073                        axis_y.setAllowNegativesFlag(false);
074                        p.setDomainAxis(axis_x);
075                        p.setRangeAxis(axis_y);
076                }
077                return c;
078        }
079
080        /**
081         * Adds a new data series to the chart with the specified title.
082         * <code>xs<code> and <code>ys</code> should have the same length. If not,
083         * only as many items are shown as the shorter array contains.
084         * 
085         * @param title
086         * @param xs
087         *            The x values.
088         * @param ys
089         *            The y values.
090         */
091        public void addSeries(final String title, final double[] xs,
092                        final double[] ys) {
093                XYSeries series = new XYSeries(title, false, true);
094                for (int i = 0, n = Math.min(xs.length, ys.length); i < n; i++) {
095                        series.add(xs[i], ys[i]);
096                }
097                this.dataset.addSeries(series);
098        }
099
100}
