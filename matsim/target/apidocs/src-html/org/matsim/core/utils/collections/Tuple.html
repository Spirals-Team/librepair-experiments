021package org.matsim.core.utils.collections;
022
023import java.io.Serializable;
024
025/**
026 * A Tuple stores two values (a "pair") and respects their order.
027 * This generic class implements a commonly used data structure which is not present in
028 * the current collection framework. Although it could be simulated with a List containing
029 * two Objects, this implementation offers type safety and maximizes convenience for programmers.
030 *
031 * @author dgrether
032 *
033 * @param <A>
034 * @param <B>
035 */
036public class Tuple<A, B> implements Serializable{
037        private static final long serialVersionUID = 1L;
038        /**
039         * First entry of the tuple
040         */
041        private final A first;
042        /**
043         * Second entry of the tuple
044         */
045        private final B second;
046        /**
047         * Creates a new tuple with the two entries.
048         * @param first
049         * @param second
050         */
051        public Tuple(final A first, final B second) {
052                this.first = first;
053                this.second = second;
054        }
055
056        public A getFirst() {
057                return this.first;
058        }
059
060        public B getSecond() {
061                return this.second;
062        }
063
064        /**
065         * @see java.lang.Object#equals(java.lang.Object)
066         */
067        @SuppressWarnings("unchecked")
068        @Override
069        public boolean equals(final Object other) {
070                if (!(other instanceof Tuple)) return false;
071                Tuple o = (Tuple) other;
072                if (this.first != null && this.second != null && o.first != null && o.second != null) {
073                        return (this.first.equals(o.first) && this.second.equals(o.second));
074                }
075                boolean firstEquals = (this.first == null) && (o.first == null);
076                boolean secondEquals = (this.second == null) && (o.second == null);
077                if (!firstEquals && this.first != null && o.first != null) {
078                        firstEquals = this.first.equals(o.first);
079                }
080                if (!secondEquals && this.second != null && o.second != null) {
081                        secondEquals = this.second.equals(o.second);
082                }
083                return firstEquals && secondEquals;
084        }
085
086        /**
087         * @see java.lang.Object#hashCode()
088         */
089        @Override
090        public int hashCode() {
091                return (this.first == null ? 0 : this.first.hashCode()) +
092                                (this.second == null ? 0 : this.second.hashCode());
093        }
094
095        /**
096         * @see java.lang.Object#toString()
097         */
098        @Override
099        public String toString() {
100                StringBuilder buffer = new StringBuilder(50);
101                buffer.append("[Tuple: [First: " );
102                buffer.append(this.first.toString());
103                buffer.append("], [Second: ");
104                buffer.append(this.second.toString());
105                buffer.append("]]");
106                return buffer.toString();
107        }
108
109}
