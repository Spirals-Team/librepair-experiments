021package org.matsim.core.utils.geometry.transformations;
022
023import org.matsim.api.core.v01.Coord;
024import org.matsim.core.utils.geometry.CoordinateTransformation;
025
026/**
027 * Transforms coordinates from WGS84 to the Swiss-Grid coordinate system.
028 * 
029 * @author meisterk
030 * @author mrieser
031 * 
032 * @see <a href="http://de.wikipedia.org/wiki/WGS84">de.wikipedia.org/wiki/WGS84</a>
033 * @see <a href="http://www.swisstopo.ch/pub/down/basics/geo/system/ch1903_wgs84_de.pdf">Swisstopo Umrechnungen (PDF)</a>
034 */
035public class WGS84toCH1903LV03 implements CoordinateTransformation {
036
037        @Override
038        public Coord transform(Coord coord) {
039
040                double lonNorm = (coord.getX() * 3600 - 26782.5) / 10000;
041                double latNorm = (coord.getY() * 3600 - 169028.66) / 10000;
042                
043                double CH1903X = 
044                        200147.07 +
045                        308807.95 * latNorm +
046                        3745.25 * Math.pow(lonNorm, 2) +
047                        76.63 * Math.pow(latNorm, 2) -
048                        194.56 * Math.pow(lonNorm, 2) * latNorm +
049                        119.79 * Math.pow(latNorm, 3);
050                
051                double CH1903Y = 
052                        600072.37 +
053                        211455.93 * lonNorm -
054                        10938.51 * lonNorm * latNorm -
055                        0.36 * lonNorm * Math.pow(latNorm, 2) -
056                        44.54 * Math.pow(lonNorm, 3);
057                
058                /* Important Note: in the Swiss Grid, y describes easting and x describes 
059                 * northing, contrary to the usual naming conventions!           */
060                double elevation;
061                try{
062                        elevation = coord.getZ();
063                        return new Coord(CH1903Y, CH1903X, elevation);
064                } catch (Exception e){
065                        return new Coord(CH1903Y, CH1903X);
066                }
067        }
068
069}
