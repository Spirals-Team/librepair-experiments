051/**
052 * Reads in an OSM-File, exported from <a href="http://openstreetmap.org/" target="_blank">OpenStreetMap</a>,
053 * and extracts information about roads to generate a MATSim-Network.
054 *
055 * OSM-Files can be obtained:
056 * <ul>
057 * <li>by <a href="http://openstreetmap.org/export" target="_blank">exporting</a> the data directly from OpenStreetMap.
058 * This works only for smaller regions.</li>
059 * <li>by <a href="http://wiki.openstreetmap.org/wiki/Getting_Data" target="_blank">downloading</a> the requested
060 * data from OpenStreetMap.
061 * <li>by extracting the corresponding data from <a href="http://planet.openstreetmap.org/" target="_blank">Planet.osm</a>. Planet.osm contains
062 * the <em>complete</em> data from OpenStreetMap and is huge! Thus, you must extract a subset of data to be able
063 * to process it.  For some countries, there are
064 * <a href="http://wiki.openstreetmap.org/wiki/Planet.osm#Extracts" target="_blank">Extracts</a> available containing
065 * only the data of a single country.</li>
066 * </ul>
067 *
068 * OpenStreetMap only contains limited information regarding traffic flow in the streets. The most valuable attribute
069 * of OSM data is a <a href="http://wiki.openstreetmap.org/wiki/Map_Features#Highway" target="_blank">categorization</a>
070 * of "ways". This reader allows to set {@link #setHighwayDefaults(int, String, double, double, double, double) defaults} how
071 * those categories should be interpreted to create a network with suitable attributes for traffic simulation.
072 * For the most common highway-types, some basic defaults can be loaded automatically (see code), but they can be
073 * overwritten if desired. If the optional attributes <code>lanes</code> and <code>oneway</code> are set in the
074 * osm data, they overwrite the default values. Using {@link #setHierarchyLayer(double, double, double, double, int) hierarchy layers},
075 * multiple overlapping areas can be specified, each with a different denseness, e.g. one only containing motorways,
076 * a second one containing every link down to footways.
077 *
078 * @author mrieser, aneumann
079 */
080public class OsmNetworkReader implements MatsimSomeReader {

190        /**
191         * Either osmFilename or stream must be <code>null</code>, but not both.
192         *
193         * @param osmFilename
194         * @param stream
195         * @throws UncheckedIOException
196         */
197        private void parse(final String osmFilename, final InputStream stream) throws UncheckedIOException {
198                if(this.hierarchyLayers.isEmpty()){
199                        log.warn("No hierarchy layer specified. Will convert every highway specified by setHighwayDefaults.");
200                }
201
202                OsmXmlParser parser = null;
203                if (this.slowButLowMemory) {
204                        log.info("parsing osm file first time: identifying nodes used by ways");
205                        parser = new OsmXmlParser(this.nodes, this.ways, this.transform);
206                        parser.enableOptimization(1);
207                        if (stream != null) {
208                                parser.parse(new InputSource(stream));
209                        } else {
210                                parser.readFile(osmFilename);
211                        }
212                        log.info("parsing osm file second time: loading required nodes and ways");
213                        parser.enableOptimization(2);
214                        if (stream != null) {
215                                parser.parse(new InputSource(stream));
216                        } else {
217                                parser.readFile(osmFilename);
218                        }
219                        log.info("done loading data");
220                } else {
221                        parser = new OsmXmlParser(this.nodes, this.ways, this.transform);
222                        if (stream != null) {
223                                parser.parse(new InputSource(stream));
224                        } else {
225                                parser.readFile(osmFilename);
226                        }
227                        log.info("done loading data");
228                }
229                convert();
230                log.info("= conversion statistics: ==========================");
231                log.info("osm: # nodes read:       " + parser.nodeCounter.getCounter());
232                log.info("osm: # ways read:        " + parser.wayCounter.getCounter());
233                log.info("MATSim: # nodes created: " + this.network.getNodes().size());
234                log.info("MATSim: # links created: " + this.network.getLinks().size());
235
236                if (this.unknownHighways.size() > 0) {
237                        log.info("The following highway-types had no defaults set and were thus NOT converted:");
238                        for (String highwayType : this.unknownHighways) {
239                                log.info("- \"" + highwayType + "\"");
240                        }
241                }
242                log.info("= end of conversion statistics ====================");
243        }

277        /**
278         * Sets whether the detailed geometry of the roads should be retained in the conversion or not.
279         * Keeping the detailed paths results in a much higher number of nodes and links in the resulting MATSim network.
280         * Not keeping the detailed paths removes all nodes where only one road passes through, thus only real intersections
281         * or branchings are kept as nodes. This reduces the number of nodes and links in the network, but can in some rare
282         * cases generate extremely long links (e.g. for motorways with only a few ramps every few kilometers).
283         *
284         * Defaults to <code>false</code>.
285         *
286         * @param keepPaths <code>true</code> to keep all details of the OSM roads
287         */
288        public void setKeepPaths(final boolean keepPaths) {
289                this.keepPaths = keepPaths;

350        private void convert() {
351                this.network.setCapacityPeriod(3600);
352
353                log.info("Remove ways that have at least one node that was not read previously ...");
354                // yy I _think_ this is what it does.  kai, may'16
355                Iterator<Entry<Long, OsmWay>> it = this.ways.entrySet().iterator();
356                int counter = 0;
357                while (it.hasNext()) {
358                        Entry<Long, OsmWay> entry = it.next();
359                        for (Long nodeId : entry.getValue().nodes) {
360                                if (this.nodes.get(nodeId) == null) {
361                                        it.remove();
362                                        break;
363                                }
364                        }
365                }
366                log.info("... done removing " + counter + "ways that have at least one node that was not read previously.");
367
368                log.info("Mark OSM nodes that shoud be kept ...");
369                for (OsmWay way : this.ways.values()) {
370                        String highway = way.tags.get(TAG_HIGHWAY);
371                        if ((highway != null) && (this.highwayDefaults.containsKey(highway))) {
372                                // check to which level a way belongs
373                                way.hierarchy = this.highwayDefaults.get(highway).hierarchy;
374
375                                // first and last are counted twice, so they are kept in all cases
376                                this.nodes.get(way.nodes.get(0)).ways++;
377                                this.nodes.get(way.nodes.get(way.nodes.size()-1)).ways++;
378
379                                for (Long nodeId : way.nodes) {
380                                        OsmNode node = this.nodes.get(nodeId);
381                                        if (this.hierarchyLayers.isEmpty()) {
382                                                node.used = true;
383                                                node.ways++;
384                                        } else {
385                                                for (OsmFilter osmFilter : this.hierarchyLayers) {
386                                                        if(osmFilter.coordInFilter(node.coord, way.hierarchy)){
387                                                                node.used = true;
388                                                                node.ways++;
389                                                                break;
390                                                        }
391                                                }
392                                        }
393                                }
394                        }
395                }
396                log.info("... done marking OSM nodes that shoud be kept.");
397
398                if (!this.keepPaths) {
399
400                        log.info("Mark nodes as unused where only one way leads through ...") ;
401                        for (OsmNode node : this.nodes.values()) {
402                                if (node.ways == 1) {
403                                        node.used = false;
404                                }
405                        }
406                        log.info("... done marking nodes as unused where only one way leads through.") ;
407
408                        log.info("Verify we did not mark nodes as unused that build a loop ...") ;
409                        for (OsmWay way : this.ways.values()) {
410                                String highway = way.tags.get(TAG_HIGHWAY);
411                                if ((highway != null) && (this.highwayDefaults.containsKey(highway))) {
412                                        int prevRealNodeIndex = 0;
413                                        OsmNode prevRealNode = this.nodes.get(way.nodes.get(prevRealNodeIndex));
414
415                                        for (int i = 1; i < way.nodes.size(); i++) {
416                                                OsmNode node = this.nodes.get(way.nodes.get(i));
417                                                if (node.used) {
418                                                        if (prevRealNode == node) {
419                                                                /* We detected a loop between two "real" nodes.
420                                                                 * Set some nodes between the start/end-loop-node to "used" again.
421                                                                 * But don't set all of them to "used", as we still want to do some network-thinning.
422                                                                 * I decided to use sqrt(.)-many nodes in between...
423                                                                 */
424                                                                double increment = Math.sqrt(i - prevRealNodeIndex);
425                                                                double nextNodeToKeep = prevRealNodeIndex + increment;
426                                                                for (double j = nextNodeToKeep; j < i; j += increment) {
427                                                                        int index = (int) Math.floor(j);
428                                                                        OsmNode intermediaryNode = this.nodes.get(way.nodes.get(index));
429                                                                        intermediaryNode.used = true;
430                                                                }
431                                                        }
432                                                        prevRealNodeIndex = i;
433                                                        prevRealNode = node;
434                                                }
435                                        }
436                                }
437                        }
438                        log.info("... done verifying that we did not mark nodes as unused that build a loop.") ;
439
440                }
441
442                //check here whether node is needed for counts
443                //tschlenther jun'17
444                if(nodeIDsToKeep != null){
445                        int cnt = 0;
446                        log.info("...assure that all nodes that are definitely to be kept are marked as used");
447                        for(Long nodeToBeKept : this.nodeIDsToKeep){
448                                OsmNode node = this.nodes.get(nodeToBeKept);
449                                if(node==null){
450                                        log.warn("cannot find node " + nodeToBeKept + ". maybe it was not read in or got deleted..");
451                                }
452                                else{
453                                        node.used = true;
454                                        cnt ++;
455                                }
456                        }
457                        log.info("..found " + cnt + " out of " + nodeIDsToKeep.size() + " nodes to keep and marked them as used..");
458                }
459                
460                log.info("Create the required nodes ...") ;
461                for (OsmNode node : this.nodes.values()) {
462                        if (node.used) {
463                                Node nn = this.network.getFactory().createNode(Id.create(node.id, Node.class), node.coord);
464                                setOrModifyNodeAttributes(nn, node);
465                                this.network.addNode(nn);
466                        }
467                }
468                log.info("... done creating the required nodes.");
469
470                log.info( "Create the links ...") ;
471                this.id = 1;
472                for (OsmWay way : this.ways.values()) {
473                        String highway = way.tags.get(TAG_HIGHWAY);
474                        if (highway != null) {
475                                OsmNode fromNode = this.nodes.get(way.nodes.get(0));
476                                double length = 0.0;
477                                OsmNode lastToNode = fromNode;
478                                if (fromNode.used) {
479                                        for (int i = 1, n = way.nodes.size(); i < n; i++) {
480                                                OsmNode toNode = this.nodes.get(way.nodes.get(i));
481                                                if (toNode != lastToNode) {
482                                                        length += CoordUtils.calcEuclideanDistance(lastToNode.coord, toNode.coord);
483                                                        if (toNode.used) {
484
485                                                                if(this.hierarchyLayers.isEmpty()) {
486                                                                        createLink(this.network, way, fromNode, toNode, length);
487                                                                } else {
488                                                                        for (OsmFilter osmFilter : this.hierarchyLayers) {
489                                                                                if(osmFilter.coordInFilter(fromNode.coord, way.hierarchy)){
490                                                                                        createLink(this.network, way, fromNode, toNode, length);
491                                                                                        break;
492                                                                                }
493                                                                                if(osmFilter.coordInFilter(toNode.coord, way.hierarchy)){
494                                                                                        createLink(this.network, way, fromNode, toNode, length);
495                                                                                        break;
496                                                                                }
497                                                                        }
498                                                                }
499                                                                fromNode = toNode;
500                                                                length = 0.0;
501                                                        }
502                                                        lastToNode = toNode;
503                                                }
504                                        }
505                                }
506                        }
507                }
508                log.info("... done creating the links.");
509
510                // free up memory
511                this.nodes.clear();
512                this.ways.clear();
513        }

515        private void createLink(final Network network, final OsmWay way, final OsmNode fromNode, final OsmNode toNode, 
516                        final double length) {
517                String highway = way.tags.get(TAG_HIGHWAY);
518
519        if ("no".equals(way.tags.get(TAG_ACCESS))) {
520             return;
521        }
522                
523                // load defaults
524                OsmHighwayDefaults defaults = this.highwayDefaults.get(highway);
525                if (defaults == null) {
526                        this.unknownHighways.add(highway);
527                        return;
528                }
529
530                double nofLanesForward = defaults.lanesPerDirection;
531                double nofLanesBackward = defaults.lanesPerDirection;
532                double laneCapacity = defaults.laneCapacity;
533                double freespeed = defaults.freespeed;
534                double freespeedFactor = defaults.freespeedFactor;
535                
536                boolean oneway = isOneway(way);
537                boolean onewayReverse = isOnewayReverse(way);
538
539        // In case trunks, primary and secondary roads are marked as oneway,
540        // the default number of lanes should be two instead of one.
541        if(highway.equalsIgnoreCase("trunk") || highway.equalsIgnoreCase("primary") || highway.equalsIgnoreCase("secondary")){
542                        if (oneway && nofLanesForward == 1.0) {
543                                nofLanesForward = 2.0;
544                        } else if(onewayReverse && nofLanesBackward == 1.0){
545                                nofLanesBackward = 2.0;
546                        }
547                }
548
549                String maxspeedTag = way.tags.get(TAG_MAXSPEED);
550                if (maxspeedTag != null) {
551                        try {
552                                if(maxspeedTag.endsWith("mph")) {
553                                        freespeed = Double.parseDouble(maxspeedTag.replace("mph", "").trim()) * 1.609344 / 3.6; // convert mph to m/s
554                                } else {
555                                        freespeed = Double.parseDouble(maxspeedTag) / 3.6; // convert km/h to m/s
556                                }
557                        } catch (NumberFormatException e) {
558                                if (!this.unknownMaxspeedTags.contains(maxspeedTag)) {
559                                        this.unknownMaxspeedTags.add(maxspeedTag);
560                                        log.warn("Could not parse maxspeed tag:" + e.getMessage() + ". Ignoring it.");
561                                }
562                        }
563                }
564
565                // check tag "lanes"
566                String lanesTag = way.tags.get(TAG_LANES);
567                if (lanesTag != null) {
568                        try {
569                                double totalNofLanes = Double.parseDouble(lanesTag);
570                                if (totalNofLanes > 0) {
571                                        // If the tag "lanes:forward" or "lanes:backward" is set, use them.
572                                        // If not, give either direction half of the total number of lanes.
573                                        //fzwick,nkuehnel, apr'17
574                                        String lanesForwardTag = way.tags.get(TAG_LANES_FORWARD);
575                                        if (lanesForwardTag != null) {
576                                                double parsedForwardLanes = Double.parseDouble(lanesForwardTag);
577                                                if(parsedForwardLanes > 0) {
578                                                        nofLanesForward = parsedForwardLanes;
579                                                }
580                                        } else {
581                                                //By default, the OSM lanes tag specifies the total number of lanes in both directions.
582                                                //So if the road is not oneway (onewayReverse), let's distribute them between both directions
583                                                //michalm, jan'16
584                                                if(oneway) {
585                                                        nofLanesForward = totalNofLanes;
586                                                } else {
587                                                        nofLanesForward = totalNofLanes / 2.;
588                                                }
589                                        }
590
591                                        String lanesBackwardTag = way.tags.get(TAG_LANES_BACKWARD);
592                                        if (lanesBackwardTag != null) {
593                                                double parsedBackwardLanes = Double.parseDouble(lanesBackwardTag);
594                                                if(parsedBackwardLanes > 0) {
595                                                        nofLanesBackward = parsedBackwardLanes;
596                                                }
597                                        } else {
598                                                if(onewayReverse) {
599                                                        nofLanesBackward = totalNofLanes;
600                                                } else {
601                                                        nofLanesBackward = totalNofLanes / 2.;
602                                                }
603                                        }
604                                }
605                        } catch (Exception e) {
606                                if (!this.unknownLanesTags.contains(lanesTag)) {
607                                        this.unknownLanesTags.add(lanesTag);
608                                        log.warn("Could not parse lanes tag:" + e.getMessage() + ". Ignoring it.");
609                                }
610                        }
611                }
612                double capacityForward = nofLanesForward * laneCapacity;
613                double capacityBackward = nofLanesBackward * laneCapacity;
614
615                if (this.scaleMaxSpeed) {
616                        freespeed = freespeed * freespeedFactor;
617                }
618
619                // only create link, if both nodes were found, node could be null, since nodes outside a layer were dropped
620                Id<Node> fromId = Id.create(fromNode.id, Node.class);
621                Id<Node> toId = Id.create(toNode.id, Node.class);
622                if(network.getNodes().get(fromId) != null && network.getNodes().get(toId) != null){
623                        String origId = Long.toString(way.id);
624
625                        // Forward direction (in relation to the direction of the OSM way object)
626                        if (forwardDirectionExists(way)) {
627                                Link l = network.getFactory().createLink(Id.create(this.id, Link.class), network.getNodes().get(fromId), network.getNodes().get(toId));
628                                l.setLength(length);
629                                l.setFreespeed(freespeed);
630                                l.setCapacity(capacityForward);
631                                l.setNumberOfLanes(nofLanesForward);
632                                NetworkUtils.setOrigId(l, origId);
633                                NetworkUtils.setType(l, highway);
634                                setOrModifyLinkAttributes(l, way, true);
635                                network.addLink(l);
636                                this.id++;
637                        }
638                        // Backward/reverse direction (in relation to the direction of the OSM way object)
639                        if (reverseDirectionExists(way)) {
640                                Link l = network.getFactory().createLink(Id.create(this.id, Link.class), network.getNodes().get(toId), network.getNodes().get(fromId));
641                                l.setLength(length);
642                                l.setFreespeed(freespeed);
643                                l.setCapacity(capacityBackward);
644                                l.setNumberOfLanes(nofLanesBackward);
645                                NetworkUtils.setOrigId(l, origId);
646                                NetworkUtils.setType(l, highway);
647                                setOrModifyLinkAttributes(l, way, false);
648                                network.addLink(l);
649                                this.id++;
650                        }
651                }
652        }

654        protected boolean isOneway(OsmWay way) {
655                String onewayTag = way.tags.get(TAG_ONEWAY);
656                if (onewayTag != null) {
657                        if ("yes".equals(onewayTag) || "true".equals(onewayTag) || "1".equals(onewayTag)) {
658                                return true;
659                        } else if ("-1".equals(onewayTag) || "reverse".equals(onewayTag)) {
660                                return false;
661                        } else if ("no".equals(onewayTag) || "false".equals(onewayTag) || "0".equals(onewayTag)) {
662                                return false; // may be used to overwrite defaults
663                        }
664                        else {
665                                log.warn("Could not interpret oneway tag:" + onewayTag + ". Ignoring it.");
666                        }
667                }
668                // If no oneway tag was found, see if it is a link in a roundabout
669                if ("roundabout".equals(way.tags.get(TAG_JUNCTION))) {
670                        // if "junction" is not set in tags, get() returns null and equals() evaluates to false
671                        return true;
672                }
673                // If no direction-specific tag was found, use default value
674                OsmHighwayDefaults defaults = this.highwayDefaults.get(way.tags.get(TAG_HIGHWAY));
675                return defaults.oneway;
676        }

678        protected boolean isOnewayReverse(OsmWay way) {
679                String onewayTag = way.tags.get(TAG_ONEWAY);
680                if (onewayTag != null) {
681                        if ("yes".equals(onewayTag) || "true".equals(onewayTag) || "1".equals(onewayTag)) {
682                                return false;
683                        } else if ("-1".equals(onewayTag) || "reverse".equals(onewayTag)) {
684                                return true;
685                        } else if ("no".equals(onewayTag) || "false".equals(onewayTag) || "0".equals(onewayTag)) {
686                                return false; // may be used to overwrite defaults
687                        }
688                        else {
689                                log.warn("Could not interpret oneway tag:" + onewayTag + ". Ignoring it.");
690                        }
691                }
692                // Was set like this before; rather get this from defaults like for (forward) oneway? tt, aug'17
693                return false;
694        }

723                boolean coordInFilter( final Coord coord, final int hierarchyLevel ) ;
724        }

726        private static class OsmFilterImpl implements OsmFilter {
727                private final Coord coordNW;
728                private final Coord coordSE;
729                private final int hierarchy;
730
731                OsmFilterImpl(final Coord coordNW, final Coord coordSE, final int hierarchy) {
732                        this.coordNW = coordNW;
733                        this.coordSE = coordSE;
734                        this.hierarchy = hierarchy;
735                }
736
737                @Override
738                public boolean coordInFilter(final Coord coord, final int hierarchyLevel){
739                        if(this.hierarchy < hierarchyLevel){
740                                return false;
741                        }
742
743                        return ((this.coordNW.getX() < coord.getX() && coord.getX() < this.coordSE.getX()) &&
744                                (this.coordNW.getY() > coord.getY() && coord.getY() > this.coordSE.getY()));
745                }
746        }

748        private static class GeographicallyNonrestrictingOsmFilterImpl implements OsmFilter {
749                private final int hierarchy;
750                GeographicallyNonrestrictingOsmFilterImpl(final int hierarchy) {
751                        this.hierarchy = hierarchy;
752                }
753
754                @Override
755                public boolean coordInFilter(final Coord coord, final int hierarchyLevel){
756                        if(this.hierarchy < hierarchyLevel){
757                                return false;
758                        }
759                        return true ;
760                }

805        private class OsmXmlParser extends MatsimXmlParser {
806                private OsmWay currentWay = null;
807                private final Map<Long, OsmNode> nodes;
808                private final Map<Long, OsmWay> ways;
809                /*package*/ final Counter nodeCounter = new Counter("node ");
810                /*package*/ final Counter wayCounter = new Counter("way ");
811                private final CoordinateTransformation transform;
812                private boolean loadNodes = true;
813                private boolean loadWays = true;
814                private boolean mergeNodes = false;
815                private boolean collectNodes = false;
816
817                public OsmXmlParser(final Map<Long, OsmNode> nodes, final Map<Long, OsmWay> ways, final CoordinateTransformation transform) {
818                        super();
819                        this.nodes = nodes;
820                        this.ways = ways;
821                        this.transform = transform;
822                        this.setValidating(false);
823                }
824
825                public void enableOptimization(final int step) {
826                        this.loadNodes = false;
827                        this.loadWays = false;
828                        this.collectNodes = false;
829                        this.mergeNodes = false;
830                        if (step == 1) {
831                                this.collectNodes = true;
832                        } else if (step == 2) {
833                                this.mergeNodes = true;
834                                this.loadWays = true;
835                        }
836                }
837
838                @Override
839                public void startTag(final String name, final Attributes atts, final Stack<String> context) {
840                        if ("node".equals(name)) {
841                                if (this.loadNodes) {
842                                        Long id = Long.valueOf(atts.getValue("id"));
843                                        double lat = Double.parseDouble(atts.getValue("lat"));
844                                        double lon = Double.parseDouble(atts.getValue("lon"));
845                                        this.nodes.put(id, new OsmNode(id, this.transform.transform(new Coord(lon, lat))));
846                                        this.nodeCounter.incCounter();
847                                } else if (this.mergeNodes) {
848                                        OsmNode node = this.nodes.get(Long.valueOf(atts.getValue("id")));
849                                        if (node != null) {
850                                                double lat = Double.parseDouble(atts.getValue("lat"));
851                                                double lon = Double.parseDouble(atts.getValue("lon"));
852                                                Coord c = this.transform.transform(new Coord(lon, lat));
853//                                              node.coord.setXY(c.getX(), c.getY());
854                                                node.coord = c ;
855                                                this.nodeCounter.incCounter();
856                                        }
857                                }
858                        } else if ("way".equals(name)) {
859                                this.currentWay = new OsmWay(Long.parseLong(atts.getValue("id")));
860                        } else if ("nd".equals(name)) {
861                                if (this.currentWay != null) {
862                                        this.currentWay.nodes.add(Long.parseLong(atts.getValue("ref")));
863                                }
864                        } else if ("tag".equals(name)) {
865                                if (this.currentWay != null) {
866                                        String key = StringCache.get(atts.getValue("k"));
867                                        for (String tag : allTags) {
868                                                if (tag.equals(key)) {
869                                                        this.currentWay.tags.put(key, StringCache.get(atts.getValue("v")));
870                                                        break;
871                                                }
872                                        }
873                                }
874                        }
875                }
876
877                @Override
878                public void endTag(final String name, final String content, final Stack<String> context) {
879                        if ("way".equals(name)) {
880                                if (!this.currentWay.nodes.isEmpty()) {
881                                        boolean used = false;
882                                        OsmHighwayDefaults osmHighwayDefaults = OsmNetworkReader.this.highwayDefaults.get(this.currentWay.tags.get(TAG_HIGHWAY));
883                                        if (osmHighwayDefaults != null) {
884                                                int hierarchy = osmHighwayDefaults.hierarchy;
885                                                this.currentWay.hierarchy = hierarchy;
886                                                if (OsmNetworkReader.this.hierarchyLayers.isEmpty()) {
887                                                        used = true;
888                                                }
889                                                if (this.collectNodes) {
890                                                        used = true;
891                                                } else {
892                                                        for (OsmFilter osmFilter : OsmNetworkReader.this.hierarchyLayers) {
893                                                                for (Long nodeId : this.currentWay.nodes) {
894                                                                        OsmNode node = this.nodes.get(nodeId);
895                                                                        if(node != null && osmFilter.coordInFilter(node.coord, this.currentWay.hierarchy)){
896                                                                                used = true;
897                                                                                break;
898                                                                        }
899                                                                }
900                                                                if (used) {
901                                                                        break;
902                                                                }
903                                                        }
904                                                }
905                                        }
906                                        if (used) {
907                                                if (this.collectNodes) {
908                                                        for (long id : this.currentWay.nodes) {
909                                                                this.nodes.put(id, new OsmNode(id, new Coord(0, 0)));
910                                                        }
911                                                } else if (this.loadWays) {
912                                                        this.ways.put(this.currentWay.id, this.currentWay);
913                                                        this.wayCounter.incCounter();
914                                                }
915                                        }
916                                }
917                                this.currentWay = null;
918                        }
919                }
920
921        }

923        private static class StringCache {
924                private static ConcurrentHashMap<String, String> cache = new ConcurrentHashMap<String, String>(10000);
925                
926                /**
927                 * Returns the cached version of the given String. If the strings was
928                 * not yet in the cache, it is added and returned as well.
929                 *
930                 * @param string
931                 * @return cached version of string
932                 */
933                public static String get(final String string) {
934                        String s = cache.putIfAbsent(string, string);
935                        if (s == null) {
936                                return string;
937                        }
938                        return s;
939                }
