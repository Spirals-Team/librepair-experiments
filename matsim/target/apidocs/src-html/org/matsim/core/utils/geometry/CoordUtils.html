043        public static Coord plus ( Coord coord1, Coord coord2 ) {
044                if( !coord1.hasZ() && !coord2.hasZ() ){
045                        /* Both are 2D coordinates. */
046                        double xx = coord1.getX() + coord2.getX();
047                        double yy = coord1.getY() + coord2.getY();
048                        return new Coord(xx, yy);                       
049                } else if( coord1.hasZ() && coord2.hasZ() ){
050                        /* Both are 3D coordinates. */
051                        double xx = coord1.getX() + coord2.getX();
052                        double yy = coord1.getY() + coord2.getY();
053                        double zz = coord1.getZ() + coord2.getZ();
054                        return new Coord(xx, yy, zz);                   
055                } else{
056                        throw new RuntimeException("Cannot 'plus' coordinates if one has elevation (z) and the other not; coord1.hasZ=" + coord1.hasZ()
057                        + "; coord2.hasZ=" + coord2.hasZ() );
058                }
059        }
060        
061        public static Coord minus ( Coord coord1, Coord coord2 ) {
062                if( !coord1.hasZ() && !coord2.hasZ() ){
063                        /* Both are 2D coordinates. */
064                        double xx = coord1.getX() - coord2.getX();
065                        double yy = coord1.getY() - coord2.getY();
066                        return new Coord(xx, yy);                       
067                } else if( coord1.hasZ() && coord2.hasZ() ){
068                        /* Both are 3D coordinates. */
069                        double xx = coord1.getX() - coord2.getX();
070                        double yy = coord1.getY() - coord2.getY();
071                        double zz = coord1.getZ() - coord2.getZ();
072                        return new Coord(xx, yy, zz);                   
073                } else{
074                        throw new RuntimeException("Cannot 'minus' coordinates if one has elevation (z) and the other not.");
075                }
076        }
077        
078        public static Coord scalarMult( double alpha, Coord coord ) {
079                if(!coord.hasZ()){
080                        /* 2D coordinate. */
081                        double xx = alpha * coord.getX();
082                        double yy = alpha * coord.getY();
083                        return new Coord(xx, yy);                       
084                } else {
085                        /* 3D coordinate. */
086                        double xx = alpha * coord.getX();
087                        double yy = alpha * coord.getY();
088                        double zz = alpha * coord.getZ();
089                        return new Coord(xx, yy, zz);                   
090                } 
091        }

094        public static Coord getCenter( Coord coord1, Coord coord2 ) {
095                if( !coord1.hasZ() && !coord2.hasZ() ){
096                        /* Both are 2D coordinates. */
097                        double xx = 0.5*( coord1.getX() + coord2.getX() ) ;
098                        double yy = 0.5*( coord1.getY() + coord2.getY() ) ;
099                        return new Coord(xx, yy);                       
100                } else if( coord1.hasZ() && coord2.hasZ() ){
101                        /* Both are 3D coordinates. */
102                        double xx = 0.5*( coord1.getX() + coord2.getX() ) ;
103                        double yy = 0.5*( coord1.getY() + coord2.getY() ) ;
104                        double zz = 0.5*( coord1.getZ() + coord2.getZ() ) ;
105                        return new Coord(xx, yy, zz);                   
106                } else{
107                        throw new RuntimeException("Cannot get the center for coordinates if one has elevation (z) and the other not.");
108                }
109        }
110        
111        public static double length( Coord coord ) {
112                if(!coord.hasZ()){
113                        return Math.sqrt( 
114                                        coord.getX()*coord.getX() + 
115                                        coord.getY()*coord.getY() ) ;
116                } else{
117                        return Math.sqrt( 
118                                        coord.getX()*coord.getX() + 
119                                        coord.getY()*coord.getY() +
120                                        coord.getZ()*coord.getZ()) ;
121                }
122        }

130        public static Coord rotateToRight( Coord coord ) {
131                if( !coord.hasZ() ){
132                        /* 2D coordinate */
133                        final double y = -coord.getX();
134                        return new Coord(coord.getY(), y);
135                } else{
136                        /* 3D coordinate */
137                        final double y = -coord.getX();
138                        return new Coord(coord.getY(), y, coord.getZ());                        
139                }
140        }

143        public static Coord getCenterWOffset( Coord coord1, Coord coord2 ) {
144                if( !coord1.hasZ() && !coord2.hasZ() ){
145                        /* Both are 2D coordinates. */
146                        Coord fromTo = minus( coord2, coord1 ) ;
147                        Coord offset = scalarMult( 0.1 , rotateToRight( fromTo ) ) ;
148                        Coord centerWOffset = plus( getCenter( coord1, coord2 ) , offset ) ;
149                        return centerWOffset ;
150                } else if( coord1.hasZ() && coord2.hasZ() ){
151                        /* TODO Both are 3D coordinates. */
152                        throw new RuntimeException("3D version not implemented.");
153                } else{
154                        throw new RuntimeException("Cannot get the center for coordinates if one has elevation (z) and the other not.");
155                }
156        }

158        public static double calcEuclideanDistance(Coord coord, Coord other) {
159                /* Depending on the coordinate system that is used, determining the 
160                 * distance based on the euclidean distance will lead to wrong results. 
161                 * However, if the distance is not to large (<1km) this will be a usable 
162                 * distance estimation. Another comfortable way to calculate correct 
163                 * distances would be, to use the distance functions provided by 
164                 * geotools lib. May be we need to discuss what part of GIS functionality 
165                 * we should implement by our own and for what part we could use an 
166                 * existing GIS like geotools. We need to discuss this in terms of code 
167                 * robustness, performance and so on ... [gl] */
168                if( !coord.hasZ() && !other.hasZ() ){
169                        /* Both are 2D coordinates. */
170                        double xDiff = other.getX()-coord.getX();
171                        double yDiff = other.getY()-coord.getY();
172                        return Math.sqrt((xDiff*xDiff) + (yDiff*yDiff));
173                } else if( coord.hasZ() && other.hasZ() ){
174                        /* Both are 3D coordinates. */
175                        double xDiff = other.getX()-coord.getX();
176                        double yDiff = other.getY()-coord.getY();
177                        double zDiff = other.getZ()-coord.getZ();
178                        return Math.sqrt((xDiff*xDiff) + (yDiff*yDiff) + (zDiff*zDiff));
179                } else{
180                        LOG.warn("Mixed use of elevation in coordinates: " + coord.toString() + 
181                                        "; " + other.toString());
182                        LOG.warn("Returning projected coordinate distance (using x and y components only)");
183                        return calcProjectedEuclideanDistance(coord, other);
184                }
185        }

188        /**
189         * Method to deal with distance calculation when only the x and y-components
190         * of the coordinates are used. The elevation (z component) is ignored,
191         * whether it is available or not. 
192         * (xy-plane)
193         * @param coord
194         * @param other
195         * @return
196         */
197        public static double calcProjectedEuclideanDistance(Coord coord, Coord other) {
198                double xDiff = other.getX()-coord.getX();
199                double yDiff = other.getY()-coord.getY();
200                return Math.sqrt((xDiff*xDiff) + (yDiff*yDiff));
201        }

204        /**
205         * Method should only be used in within this class, and only by 
206         * {@link #distancePointLinesegment(Coord, Coord, Coord)}. 
207         * @param coord1
208         * @param coord2
209         * @return
210         */
211        private static double dotProduct( Coord coord1, Coord coord2 ) {
212                if( !coord1.hasZ() && !coord2.hasZ() ){
213                        /* Both are 2D coordinates. */
214                        return  coord1.getX()*coord2.getX() + 
215                                        coord1.getY()*coord2.getY();
216                } else if( coord1.hasZ() && coord2.hasZ() ){
217                        /* Both are 3D coordinates. */
218                        return  coord1.getX()*coord2.getX() + 
219                                        coord1.getY()*coord2.getY() +
220                                        coord1.getZ()*coord2.getZ();
221                } else{
222                        throw new RuntimeException("Cannot get the dot-product of coordinates if one has elevation (z) and the other not.");
223                }
224        }

227        /**
228         * Calculates the shortest distance of a point to a line segment. The line segment
229         * is given by two points, <code>lineFrom</code> and <code>lineTo</code>. Note that
230         * the line segment has finite length, and thus the shortest distance cannot
231         * always be the distance on the tangent to the line through <code>point</code>.
232         * 
233         * <br><br>
234         * The 3D version was adapted from the C++ implementation of 
235         * <a href="http://geomalgorithms.com/a02-_lines.html">Dan Sunday</a>. 
236         *
237         * @param lineFrom The start point of the line segment
238         * @param lineTo The end point of the line segment
239         * @param point The point whose distance to the line segment should be calculated
240         * @return the distance of <code>point</code> to the line segment given by the two
241         *    end points of the line segment, <code>lineFrom</code> and <code>lineTo</code>
242         *
243         * @author mrieser, jwjoubert
244         */
245        public static double distancePointLinesegment(final Coord lineFrom, final Coord lineTo, final Coord point) {
246                if( !lineFrom.hasZ() && !lineTo.hasZ() && !point.hasZ() ){
247                        /* All coordinates are 2D and in the XY plane. */
248
249                        /* The shortest distance is where the tangent of the line goes 
250                         * through "point". The dot product (point - P) dot (lineTo - lineFrom) 
251                         * must be 0, when P is a point on the line. P can be substituted 
252                         * with lineFrom + u*(lineTo - lineFrom). Thus it must be:
253                         *    (point - lineFrom - u*(lineTo - lineFrom)) dot (lineTo - lineFrom) == 0
254                         * From this follows:
255                         *        (point.x - lineFrom.x)(lineTo.x - lineFrom.x) + (point.y - lineFrom.y)(lineTo.y - lineFrom.y)
256                         *    u = ---------------------------------------------------------------------------------------------
257                         *       (lineTo.x - lineFrom.x)(lineTo.x - lineFrom.x) + (lineTo.y - lineFrom.y)(lineTo.y - lineFrom.y)
258                         *
259                         * Substituting this gives:
260                         *   x = lineFrom.x + u*(lineFrom.x - lineTo.x) , y = lineFrom.y + u*(lineFrom.y - lineTo.y)
261                         *
262                         * The shortest distance is now the distance between "point" and 
263                         * (x | y)
264                         */
265                        double lineDX = lineTo.getX() - lineFrom.getX();
266                        double lineDY = lineTo.getY() - lineFrom.getY();
267                        
268                        if ((lineDX == 0.0) && (lineDY == 0.0)) {
269                                // the line segment is a point without dimension
270                                return calcEuclideanDistance(lineFrom, point);
271                        }
272                        
273                        double u = ((point.getX() - lineFrom.getX())*lineDX + (point.getY() - lineFrom.getY())*lineDY) /
274                                        (lineDX*lineDX + lineDY*lineDY);
275                        
276                        if (u <= 0) {
277                                // (x | y) is not on the line segment, but before lineFrom
278                                return calcEuclideanDistance(lineFrom, point);
279                        }
280                        if (u >= 1) {
281                                // (x | y) is not on the line segment, but after lineTo
282                                return calcEuclideanDistance(lineTo, point);
283                        }
284                        
285                        return calcEuclideanDistance(new Coord(lineFrom.getX() + u * lineDX, lineFrom.getY() + u * lineDY), point);
286                } else if( lineFrom.hasZ() && lineTo.hasZ() && point.hasZ() ){
287                        /* All coordinates are 3D. */
288                        double lineDX = lineTo.getX() - lineFrom.getX();
289                        double lineDY = lineTo.getY() - lineFrom.getY();
290                        double lineDZ = lineTo.getZ() - lineFrom.getZ();
291                        
292                        if((lineDX == 0.0) && (lineDY == 0.0) && (lineDZ == 0.0)){
293                                return calcEuclideanDistance(lineFrom, point);
294                        }
295                        
296                        Coord v = minus(lineTo, lineFrom);
297                        Coord w = minus(point, lineFrom);
298                        
299                        double c1 = dotProduct(w, v);
300                        if(c1 <= 0.0){
301                                Coord m = minus(point, lineFrom);
302                                return Math.sqrt(dotProduct(m, m));
303                        }
304                        
305                        double c2 = dotProduct(v, v);
306                        if(c2 <= c1){
307                                Coord m = minus(point, lineTo);
308                                return Math.sqrt(dotProduct(m, m));
309                        }
310                        
311                        double b = c1 / c2;
312                        Coord p = plus(lineFrom, scalarMult(b, v));
313                        Coord m = minus(point, p);
314                        return Math.sqrt(dotProduct(m, m));
315                } else{
316                        throw new RuntimeException("All given coordinates must either be 2D, or 3D. A mix is not allowed.");
317                }
318        }

321        /**
322         * Calculates the coordinate of the intersection point of the orthogonal projection
323         * of a given point on a line segment with that line segment. The line segment
324         * is given by two points, <code>lineFrom</code> and <code>lineTo</code>. If the
325         * projection point does not lie *on* the line segment (but only somewhere on
326         * the extension of the line segment, i.e. the infinite line), the end point of
327         * the line segment which is closest to the given point is returned.
328         *
329         * <br><br>
330         * The 3D version was adapted from the documentation of 
331         * <a href="http://www.geometrictools.com/Documentation/DistancePointLine.pdf">
332         * David Eberly/a>. 
333         *
334         * @param lineFrom The start point of the line segment
335         * @param lineTo The end point of the line segment
336         * @param point The point whose distance to the line segment should be calculated
337         * @return the <code>coordinate</code> of the intersection point of the orthogonal
338         * projection of a given point on a line segment with that line segment
339         *
340         * @author dziemke, jwjoubert
341         */
342        public static Coord orthogonalProjectionOnLineSegment(final Coord lineFrom, final Coord lineTo, final Coord point) {
343                if( !lineFrom.hasZ() && !lineTo.hasZ() && !point.hasZ() ){
344                        /* All coordinates are 2D. */
345
346                        /* Concerning explanation and usage of the dot product for these calculation, please
347                         * read comments of "distancePointLinesegment".
348                         */
349                        double lineDX = lineTo.getX() - lineFrom.getX();
350                        double lineDY = lineTo.getY() - lineFrom.getY();
351                        
352                        if ((lineDX == 0.0) && (lineDY == 0.0)) {
353                                // the line segment is a point without dimension
354                                return lineFrom;
355                        }
356                        
357                        double u = ((point.getX() - lineFrom.getX())*lineDX + (point.getY() - lineFrom.getY())*lineDY) /
358                                        (lineDX*lineDX + lineDY*lineDY);
359                        
360                        if (u <= 0) {
361                                // (x | y) is not on the line segment, but before lineFrom
362                                return lineFrom;
363                        }
364                        if (u >= 1) {
365                                // (x | y) is not on the line segment, but after lineTo
366                                return lineTo;
367                        }
368                        return new Coord(lineFrom.getX() + u * lineDX, lineFrom.getY() + u * lineDY);
369                } else if(lineFrom.hasZ() && lineTo.hasZ() && point.hasZ() ){
370                        /* All coordinates are 3D. */
371                        Coord direction = minus(lineTo, lineFrom);
372                        
373                        double t0 = dotProduct(direction, minus(point, lineFrom)) / dotProduct(direction, direction);
374                        Coord q = plus(lineFrom, scalarMult(t0, direction));
375                        return q;
376                } else{
377                        throw new RuntimeException("All given coordinates must either be 2D, or 3D. A mix is not allowed.");
378                }
379        }
