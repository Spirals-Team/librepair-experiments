021package org.matsim.core.utils.misc;
022
023/**
024 * A helper class with utility functions for <code>String</code>s.
025 *
026 * @author mrieser
027 */
028public class StringUtils {
029
030        /**
031         * Splits a string at the specified delimiter into multiple substrings. This
032         * method is similar to <code>String.split()</code>, but does only take a
033         * single char as a delimiter and not a regular expression, making this
034         * method a lot faster than <code>String.split()</code> if no regular
035         * expressions are required.
036         *
037         * The <code>limit</code> parameter controls how many parts will be returned
038         * at most. If the limit is zero, no limit is used and the string is splitted
039         * as many times as possible. Negative values for <code>limit</code> are not
040         * supported. If there are more delimiters than <code>limit</code>, only the
041         * first <em>limit-1</em> ones will be used for splitting the string, the
042         * additional delimiters are contained in the last part returned. Trailing
043         * empty strings are not included in the resulting array.
044         *
045         * @param str The string to be splitted.
046         * @param delimiter A single character where <code>str</code> should be splitted.
047         * @param limit The maximum number of parts returned.
048         * @return Returns an array of strings, each being a substring of
049         *              <code>str</code>, that were connected by <code>delimiter</code> in the
050         *              original string <code>str</code>.
051         *
052         * @author mrieser
053         */
054        public static String[] explode(final String str, final char delimiter, final int limit) {
055                int count = 0;
056                int len = str.length();
057                int maxPos = 0; // the position of the last non-delimiter char in the string
058                int countAtMaxPos = 0;
059                int upperLimit = limit-1;
060                // count how often the delimiter occurs in the string
061                for (int pos = 0; pos < len; pos++) {
062                        if ((str.charAt(pos) == delimiter) && (limit == 0 || count < upperLimit)) {
063                                count++;
064                        } else {
065                                maxPos = pos;
066                                countAtMaxPos = count;
067                        }
068                }
069
070                if (count == 0) {
071                        String[] parts = new String[1];
072                        parts[0] = str;
073                        return parts;
074                }
075
076                // create a big enough array to hold the result
077                String[] parts = new String[countAtMaxPos + 1];
078                int startPos = 0;
079                count = 0;
080                // loop once through the string and always take the substrings between the delimiters
081                for (int pos = 0; pos < maxPos; pos++) {
082                        if ((str.charAt(pos) == delimiter) && (limit == 0 || count < upperLimit)) {
083                                parts[count] = str.substring(startPos, pos);
084                                startPos = pos + 1;
085                                count++;
086                        }
087                }
088                /* Add the final part ended by the string, not by a delimiter, to our list.
089                 * There is always a final part because of the way maxPos and countAtMaxPos
090                 * are derived: maxPos points to a non-delimiter. This means there is always
091                 * at least one character between the last delimiter and maxPos, and that's
092                 * the part we are still missing... so add it! */
093                parts[count] = str.substring(startPos, maxPos+1);
094
095                return parts;
096        }
097
098        /**
099         * Splits a string at the specified delimiter into multiple substrings. This
100         * method is similar to <code>String.split()</code>, but does only take a
101         * single char as a delimiter and not a regular expression, making this
102         * method a lot faster than <code>String.split()</code> if no regular
103         * expressions are required.
104         *
105         * @param str The string to be splitted.
106         * @param delimiter A single character where <code>str</code> should be splitted.
107         * @return Returns an array of strings, each being a substring of
108         *              <code>str</code>, that were connected by <code>delimiter</code> in the
109         *              original string <code>str</code>.
110         * @author mrieser
111         */
112        public static String[] explode(final String str, final char delimiter) {
113                return explode(str, delimiter, 0);
114        }
115}
