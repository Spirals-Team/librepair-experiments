021package org.matsim.core.utils.geometry.transformations;
022
023import org.matsim.api.core.v01.Coord;
024import org.matsim.core.utils.geometry.CoordinateTransformation;
025
026/**
027 * Transforms coordinates from Gauss Krueger 4 to WGS84
028 *
029 * @author mrieser
030 *
031 * @see <a href="http://de.wikipedia.org/wiki/WGS84">de.wikipedia.org/wiki/WGS84</a>
032 * @see <a href="http://remotesensing.org/geotiff/proj_list/transverse_mercator.html">remotesensing.org, transverse mercator</a>
033 * @see <a href="www.epsg.org/guides/docs/G7-2.pdf">OGP Surveying and Positioning Guidance Note number 7, part 2</a>
034 */
035public class GK4toWGS84 implements CoordinateTransformation {
036
037        /*
038         * GK4 is a specially parameterized version of the Transverse Mercator coordinate system.
039         * Thus, the transformation from Transverse Mercator to WGS84 can be used to convert
040         * GK4 to WGS84.
041         *
042         * see http://www.epsg.org/guides/docs/G7-2.pdf, chapter on Transverse Mercator,
043         * for the conversion.
044         */
045
046        // ellipsoid: bessel 1841
047        private static final double a = 6377397.155;
048        // private static final double f = 1/299.15281; // flattening of bessel 1841, not used here
049        private static final double e = 0.081696831;
050
051        private static final double falseEasting = 4500110.0;
052        private static final double falseNorthing = 116.0;
053        private static final double k0 = 1.0;
054        // private static final double projectionLatitude = 0.0;
055        private static final double projectionLongitude = 12.0 * Math.PI / 180.0;
056
057        private static final double e2 = e*e;
058        private static final double e_2 = e2 / (1 - e2);
059        private static final double e1 = (1 - Math.pow(1 - e2, 0.5)) / (1 + Math.pow(1 - e2, 0.5));
060        private static final double M0 = 0.0;
061        private static final double MU1_DIVISOR = a * (1 - e2*(1.0/4.0 - e2*(3.0/64.0) - e2*(5.0/256.0)));
062
063        private static final double[] PHI1_DIVIDENS = {
064                (3 * e1 / 2 - 27 * Math.pow(e1, 3) / 32),
065                (21 * Math.pow(e1, 2) / 16 - 55 * Math.pow(e1, 4) / 32),
066                (151 * Math.pow(e1, 3) / 96),
067                (1097 * Math.pow(e1, 4) / 512)
068        };
069
070        @Override
071        public Coord transform(final Coord coord) {
072
073                double easting = coord.getX();
074                double northing = coord.getY();
075
076                double M1 = M0 + (northing - falseNorthing) / k0;
077                double mu1 = M1 / MU1_DIVISOR;
078
079                double phi1 = mu1
080                                + PHI1_DIVIDENS[0] * Math.sin(2 * mu1)
081                                + PHI1_DIVIDENS[1] * Math.sin(4 * mu1)
082                                + PHI1_DIVIDENS[2] * Math.sin(6 * mu1)
083                                + PHI1_DIVIDENS[3] * Math.sin(8 * mu1);
084
085                double cosphi1 = Math.cos(phi1);
086                double sinphi1 = Math.sin(phi1);
087                double tanphi1 = Math.tan(phi1);
088
089                double C1 = e_2 * cosphi1 * cosphi1;
090                double T1 = Math.pow(tanphi1, 2);
091                double rho1 = a * (1 - e2) / Math.pow((1 - e2 * sinphi1 * sinphi1), 1.5);
092                double nu1 = a / Math.pow((1 - e2 * sinphi1 * sinphi1), 0.5);
093                double D = (easting - falseEasting) / (nu1 * k0);
094
095                double latitude = phi1
096                                - (nu1 * tanphi1 / rho1) * (Math.pow(D, 2) / 2
097                                - ( 5 +  3*T1 +  10 * C1 -  4*C1*C1 - 9 * e_2) * Math.pow(D, 4) / 24
098                                + (61 + 90*T1 + 298 * C1 + 45*T1*T1 - 252 * e_2 - 3 * C1*C1) * Math.pow(D, 6) / 720);
099
100                double longitude = projectionLongitude + (D
101                                - (1 + 2 * T1 + C1) * Math.pow(D, 3) / 6
102                                + (5 - 2 * C1 + 28 * T1 - 3 * C1*C1 + 8 * e_2 + 24 * T1*T1) * Math.pow(D, 5) / 120
103                                ) / cosphi1;
104
105                latitude = latitude * 180.0 / Math.PI;
106                longitude = longitude * 180.0 / Math.PI;
107
108                double elevation;
109                try{
110                        elevation = coord.getZ();
111                        return new Coord(longitude, latitude, elevation);
112                } catch (Exception e){
113                        return new Coord(longitude, latitude);
114                }
115        }
116
117}
