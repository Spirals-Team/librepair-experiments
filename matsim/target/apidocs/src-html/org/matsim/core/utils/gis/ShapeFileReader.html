051public class ShapeFileReader implements MatsimSomeReader {
052        private static final Logger log = Logger.getLogger(ShapeFileReader.class);
053
054        private SimpleFeatureSource featureSource = null;
055
056        private ReferencedEnvelope bounds = null;
057
058        private DataStore dataStore = null;
059
060        private SimpleFeatureCollection featureCollection = null;
061
062        private SimpleFeatureType schema = null;
063
064        private Collection<SimpleFeature> featureSet = null;
065
066        private CoordinateReferenceSystem crs;
067
068        public static Collection<SimpleFeature> getAllFeatures(final String filename) {
069                try {
070                        File dataFile = new File(filename);
071                        log.info( "will try to read from " + dataFile.getAbsolutePath() ) ;
072                        Gbl.assertIf( dataFile.exists() );
073                        FileDataStore store = FileDataStoreFinder.getDataStore(dataFile);
074                        SimpleFeatureSource featureSource = store.getFeatureSource();
075
076                        SimpleFeatureIterator it = featureSource.getFeatures().features();
077                        List<SimpleFeature> featureSet = new ArrayList<SimpleFeature>();
078                        while (it.hasNext()) {
079                                SimpleFeature ft = it.next();
080                                featureSet.add(ft);
081                        }
082                        it.close();
083                        store.dispose();
084                        return featureSet;
085                } catch (IOException e) {
086                        throw new UncheckedIOException(e);
087                }
088        }
089
090        /**
091         * Reads all Features in the file into the returned Set and initializes the instance of this class.
092         */
093        public Collection<SimpleFeature> readFileAndInitialize(final String filename) throws UncheckedIOException {
094                try {
095                        this.featureSource = ShapeFileReader.readDataFile(filename);
096                        this.init();
097                        SimpleFeature ft = null;
098                        SimpleFeatureIterator it = this.featureSource.getFeatures().features();
099                        this.featureSet = new ArrayList<SimpleFeature>();
100                        log.info("features to read #" + this.featureSource.getFeatures().size());
101                        Counter cnt = new Counter("features read #");
102                        while (it.hasNext()) {
103                                ft = it.next();
104                                this.featureSet.add(ft);
105                                cnt.incCounter();
106                        }
107                        cnt.printCounter();
108                        it.close();
109                        return this.featureSet;
110                } catch (IOException e) {
111                        throw new UncheckedIOException(e);
112                }
113        }
114
115        /**
116         * <em>VERY IMPORTANT NOTE</em><br>
117         * <p></p>
118         * There are many ways to use that class in a wrong way. The safe way is the following:
119         * <p></p>
120         * <pre> ShapeFileReader shapeFileReader = new ShapeFileReader();
121         * shapeFileReader.readFileAndInitialize(zonesShapeFile); </pre>
122         * <p></p>
123         * Then, get the features by
124         * <p></p>
125         * <pre> Set<{@link Feature}> features = shapeFileReader.getFeatureSet(); </pre>
126         * <p></p>
127         * If you need metadata you can use
128         * <p></p>
129         * <pre> FeatureSource fs = shapeFileReader.getFeatureSource(); </pre>
130         * <p></p>
131         * to get access to the feature source.<br>
132         * <em>BUT NEVER CALL <code>fs.getFeatures();</code> !!! It can happen that you will read from disk again!!! </em>
133         * <p></p>
134         * <p>
135         * Actually, the whole class must be fixed. But since it is anyway necessary to move to a more recent version of the geotools only this javadoc is added instead.
136         * </p>
137         * <p></p>
138         * <p>
139         * The following old doc is kept here:
140         * </p>
141         * <p></p>
142         * Provides access to a shape file and returns a <code>FeatureSource</code> containing all features.
143         * Take care access means on disk access, i.e. the FeatureSource is only a pointer to the information
144         * stored in the file. This can be horribly slow if invoked many times and throw exceptions if two many read
145         * operations to the same file are performed. In those cases it is recommended to use the method readDataFileToMemory
146         * of this class.
147         *
148         * @param filename File name of a shape file (ending in <code>*.shp</code>)
149         * @return FeatureSource containing all features.
150         * @throws UncheckedIOException if the file cannot be found or another error happens during reading
151         */
152        public static SimpleFeatureSource readDataFile(final String filename) throws UncheckedIOException {
153                try {
154                        log.warn("Unsafe method! store.dispose() is not called from within this method");
155                        File dataFile = new File(filename);
156                        FileDataStore store = FileDataStoreFinder.getDataStore(dataFile);
157                        return store.getFeatureSource();
158                } catch (IOException e) {
159                        throw new UncheckedIOException(e);
160                }
161        }
162
163        private void init() {
164                try {
165                        this.bounds = this.featureSource.getBounds();
166                        this.dataStore = (DataStore) this.featureSource.getDataStore();
167                        this.featureCollection = this.featureSource.getFeatures();
168                        this.schema = this.featureSource.getSchema();
169                        this.crs = this.featureSource.getSchema().getCoordinateReferenceSystem();
170                } catch (IOException e) {
171                        throw new UncheckedIOException(e);
172                }
173        }
174
175        public SimpleFeatureSource getFeatureSource() {
176                return featureSource;
177        }
178
179        public ReferencedEnvelope getBounds() {
180                return bounds;
181        }
182
183        public DataStore getDataStore() {
184                return dataStore;
185        }
186
187        public SimpleFeatureCollection getFeatureCollection() {
188                return featureCollection;
189        }
190
191        public SimpleFeatureType getSchema() {
192                return schema;
193        }
194
195        public Collection<SimpleFeature> getFeatureSet() {
196                return featureSet;
197        }
198        
199        public CoordinateReferenceSystem getCoordinateSystem(){
200                return this.crs;
201        }
