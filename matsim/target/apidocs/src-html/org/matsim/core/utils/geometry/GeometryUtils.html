004package org.matsim.core.utils.geometry;

006import java.util.ArrayList;
007import java.util.LinkedHashMap;
008import java.util.List;
009import java.util.Map;
010import java.util.Map.Entry;
011
012import com.vividsolutions.jts.geom.*;
013import org.geotools.geometry.jts.GeometryBuilder;
014import org.matsim.api.core.v01.Coord;
015import org.matsim.api.core.v01.network.Link;
016import org.matsim.api.core.v01.network.Network;
017
018/**
019 * @author kainagel
020 *
021 */
022public class GeometryUtils {
023        private GeometryUtils() {} // do not instantiate
024
025        /**
026         * Method to find all links in {@link Network} that intersect a given {@link Link}.  Convenience method that
027         * only uses MATSim objects.
028         * 
029         * @param link
030         * @param network
031         * @return
032         */
033        public static List<Link> findIntersectingLinks( Link link, final Network network ) {
034                LineString segment = GeometryUtils.createGeotoolsLineString(link) ;
035                return GeometryUtils.findIntersectingLinks(segment, network) ;
036        }
037
038        /**
039         * Method to find all links in {@link Network} that intersect a given {@link LineString}.
040         * 
041         * @param lineString
042         * @param network
043         * @return
044         * 
045         * @see {@link GeometryUtilsTest#testIntersectingLinks()}
046         */
047        public static List<Link> findIntersectingLinks(LineString lineString, final Network network) {
048                // yy One could probably improve this method by using the (already existing) link quadtree to look only at
049                // those links that are in the bounding box.  kai, oct'17
050                
051                // convert matsim links into geotools line strings:
052                Map<Link,LineString> segments = new LinkedHashMap<>() ;
053                for ( Link link : network.getLinks().values() ) {
054                        LineString theSegment = GeometryUtils.createGeotoolsLineString(link);
055                        segments.put( link, theSegment ) ;
056                }
057        
058                // find the intersecting segments:
059                List<Link> resultList = new ArrayList<>();
060                for ( Entry<Link, LineString> entry : segments.entrySet() ) {
061                        if( lineString.intersects( entry.getValue() ) ){
062                                resultList.add( entry.getKey() ) ;
063                        }
064                }
065                return resultList;
066        }
067
068        /**
069         * Helper method for type conversion.
070         * 
071         * @param link
072         * @return
073         */
074        public static LineString createGeotoolsLineString(Link link) {
075                Coordinate fromCoord = CoordUtils.createGeotoolsCoordinate( link.getFromNode().getCoord() ) ;
076                Coordinate toCoord = CoordUtils.createGeotoolsCoordinate( link.getToNode().getCoord() ) ;
077                LineString theSegment = new GeometryFactory().createLineString(new Coordinate[]{ fromCoord, toCoord });
078                return theSegment;
079        }
080        
081        public static Point createGeotoolsPoint(Coord coord ) {
082                Coordinate coordinate = CoordUtils.createGeotoolsCoordinate(coord) ;
083                Point point = new GeometryFactory().createPoint( coordinate ) ;
084                return point ;
085        }
086        
087        public static Polygon createGeotoolsPolygon(List<Coord> coords ) {
088                // better way to do this is welcome.  kai, dec'17
089                double [] flatArray = new double[coords.size()*2] ;
090                int ii=0 ;
091                for ( Coord coord : coords ) {
092                        flatArray[ii] = coord.getX() ;
093                        ii++ ;
094                        flatArray[ii] = coord.getY() ;
095                        ii++ ;
096                }
097                return new GeometryBuilder().polygon( flatArray ) ;
098        }
099        
100}
