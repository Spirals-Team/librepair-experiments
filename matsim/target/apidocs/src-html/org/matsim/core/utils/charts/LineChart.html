021package org.matsim.core.utils.charts;
022
023import org.jfree.chart.ChartFactory;
024import org.jfree.chart.JFreeChart;
025import org.jfree.chart.plot.PlotOrientation;
026import org.jfree.data.category.CategoryDataset;
027import org.jfree.data.category.DefaultCategoryDataset;
028
029/**
030 * Creates two-dimensional line charts with support for multiple series in one chart.
031 *
032 * @author mrieser
033 * @author ychen
034 */
035public class LineChart extends ChartUtil {
036
037        private final String[] categories;
038        private final DefaultCategoryDataset dataset;
039
040        /**
041         * Creates a new BarChart with default category-labels (numbered from 1 to the number of categories).
042         * The same as {@link #LineChart(String, String, String, String[]) BarChart(title, xAxisLabel, yAxisLabel, null)}.
043         *
044         * @param title
045         * @param xAxisLabel
046         * @param yAxisLabel
047         */
048        public LineChart(final String title, final String xAxisLabel, final String yAxisLabel) {
049                this(title, xAxisLabel, yAxisLabel, new String[]{});
050        }
051
052        /**
053         * Creates a new BarChart with the specified category-labels.
054         *
055         * @param title
056         * @param xAxisLabel
057         * @param yAxisLabel
058         * @param categories
059         */
060        public LineChart(final String title, final String xAxisLabel, final String yAxisLabel, final String[] categories) {
061                super(title, xAxisLabel, yAxisLabel);
062                this.dataset = new DefaultCategoryDataset();
063                this.chart = createChart(title, xAxisLabel, yAxisLabel, this.dataset);
064                this.categories = categories.clone();
065                addDefaultFormatting();
066        }
067
068        @Override
069        public JFreeChart getChart() {
070                return this.chart;
071        }
072
073        private JFreeChart createChart(final String title, final String categoryAxisLabel,
074                        final String valueAxisLabel, final CategoryDataset dataset) {
075                return ChartFactory.createLineChart(title, categoryAxisLabel, valueAxisLabel,
076                                dataset, PlotOrientation.VERTICAL, true, // legend?
077                                false, // tooltips?
078                                false // URLs?
079                                );
080        }
081
082        /**
083         * Adds a new data series to the chart with the specified title.
084         *
085         * @param title
086         * @param values
087         */
088        public void addSeries(final String title, final double[] values) {
089                int cnt = 1;
090                for (double value : values) {
091                        String category = (cnt > this.categories.length ? Integer.toString(cnt) : this.categories[cnt-1]);
092                        this.dataset.addValue(value, title, category);
093                        cnt++;
094                }
095        }
096
097}
