026/**
027 * An implementation of a QuadTree to store data assigned to geometric points.
028 * The expected bounds of all added points must be given to the constructor for
029 * working properly. Correct functioning of the QuadTree with elements being
030 * added outside of the given bounds cannot be guaranteed.<br />
031 * At one location, several different objects can be put. An object can be put
032 * to the QuadTree at different locations. But an object cannot be put more than
033 * once at the same location.
034 *
035 * @author mrieser
036 * @param <T> The type of data to be stored in the QuadTree.
037 */
038public class QuadTree<T> implements Serializable {
039
040        private static final long serialVersionUID = 1L;
041
042        /** The top node or root of the tree */
043        protected Node<T> top = null;
044
045        /** The number of entries in the tree */
046        private int size = 0;
047
048        /** The number of structural modifications to the tree. */
049        private transient int modCount = 0;
050
051        /**
052         * A cache to store all values of the QuadTree so it does not have to be
053         * computed for every call to {@link #values()}. This is similar to
054         * TreeMap.java and AbstractMap.java
055         */
056        transient volatile Collection<T> values = null;
057
058        private void incrementSize() { this.modCount++; this.size++; this.values = null; }
059        private void decrementSize() { this.modCount++; this.size--; this.values = null; }
060
061        /**
062         * Creates an empty QuadTree with the bounds minX/minY -- maxX/maxY. For
063         * optimal performance, all points should be evenly distributed within this
064         * rectangle.
065         *
066         * @param minX The smallest x coordinate (easting, longitude) expected
067         * @param minY The smallest y coordinate (northing, latitude) expected
068         * @param maxX The largest x coordinate (easting, longitude) expected
069         * @param maxY The largest y coordinate (northing, latitude) expected
070         */
071        public QuadTree(final double minX, final double minY, final double maxX, final double maxY) {
072                this.top = new Node<T>(minX, minY, maxX, maxY);
073        }

075        /**
076         * Associates the specified value with the specified coordinates in this
077         * QuadTree.
078         *
079         * @param x x-coordinate where the specified value is to be associated.
080         * @param y y-coordinate where the specified value is to be associated.
081         * @param value value to be associated with the specified coordinates.
082         *
083         * @return true if insertion was successful and the data structure changed,
084         *         false otherwise.
085         */
086        public boolean put(final double x, final double y, final T value) {
087                if (this.top.put(x, y, value)) {
088                        incrementSize();
089                        return true;
090                }
091                return false;
092        }

094        /**
095         * Removes the specified object from the specified location.
096         *
097         * @param x x-coordinate from which the specified value should be removed
098         * @param y y-coordinate from which the specified value should be removed
099         * @param value the value to be removed from the specified coordinates
100         *
101         * @return true if the specified value was found at the specified coordinates
102         *         and was successfully removed (data structure changed), false
103         *         otherwise.
104         */
105        public boolean remove(final double x, final double y, final T value) {
106                if (this.top.remove(x, y, value)) {
107                        decrementSize();
108                        return true;
109                }
110                return false;
111        }

113        /** Clear the QuadTree. */
114        public void clear() {
115                this.top.clear();
116                this.size = 0;
117                this.modCount++;
118        }

145         *
146         * Note by JI (sept '15): This method can be significant faster than calling {@link #getDisk(double, double, double)}

160        /**
161         * Gets all objects within an elliptical region.
162         *
163         * @param x1 first focus, longitude
164         * @param y1 first focus, latitude
165         * @param x2 second focus, longitude
166         * @param y2 second focus, latitude
167         * @param distance the maximal sum of the distances between an object and the two foci
168         * @return the objects found in the elliptical region
169         * @throws IllegalArgumentException if the distance is shorter than the distance between the foci
170         */
171        public Collection<T> getElliptical(
172                        final double x1,
173                        final double y1,
174                        final double x2,
175                        final double y2,
176                        final double distance) {
177                if ( Math.pow( distance , 2 ) < Math.pow( (x1 - x2), 2 ) + Math.pow( (y1 - y2) , 2 ) ) {
178                        throw new IllegalArgumentException( "wrong ellipse specification: distance must be greater than distance between foci."
179                                        +" x1="+x1
180                                        +" y1="+y1
181                                        +" x2="+x2
182                                        +" y2="+y2
183                                        +" distance="+distance );
184                }
185                return this.top.getElliptical(x1, y1, x2, y2, distance, new ArrayList<T>());
186        }

273        /**
274         * Returns a collection view of the values contained in this map.  The
275         * collection's iterator will return the values in the order that their
276         * corresponding keys appear in the tree.  The collection is backed by
277         * this <tt>QuadMap</tt> instance, so changes to this map are reflected in
278         * the collection.
279         *
280         * @return a collection view of the values contained in this map.
281         */
282        public Collection<T> values() {
283                if (this.values == null) {
284                        this.values = new AbstractCollection<T>() {
285                                @Override
286                                public Iterator<T> iterator() {
287                                        Iterator<T> iterator = new Iterator<T>() {
288                                                private final int expectedModCount = QuadTree.this.modCount;
289                                                private Leaf<T> currentLeaf = firstLeaf();
290                                                private int nextIndex = 0;
291                                                private T next = first();
292
293                                                private T first() {
294                                                        if (this.currentLeaf == null) {
295                                                                return null;
296                                                        }
297                                                        this.nextIndex = 0;
298                                                        loadNext();
299                                                        return this.next;
300                                                }
301
302                                                @Override
303                                                public boolean hasNext() {
304                                                        return this.next != null;
305                                                }
306
307                                                @Override
308                                                public T next() {
309                                                        if (this.next == null) {
310                                                                return null;
311                                                        }
312                                                        if (QuadTree.this.modCount != this.expectedModCount) {
313                                                                throw new ConcurrentModificationException();
314                                                        }
315                                                        T current = this.next;
316                                                        loadNext();
317                                                        return current;
318                                                }
319
320                                                private void loadNext() {
321                                                        boolean searching = true;
322                                                        while (searching) {
323                                                                if (this.nextIndex < this.currentLeaf.values.size()) {
324                                                                        this.nextIndex++;
325                                                                        this.next = this.currentLeaf.values.get(this.nextIndex - 1);
326                                                                        searching = false;
327                                                                } else {
328                                                                        this.currentLeaf = nextLeaf(this.currentLeaf);
329                                                                        if (this.currentLeaf == null) {
330                                                                                this.next = null;
331                                                                                searching = false;
332                                                                        } else {
333                                                                                this.nextIndex = 0;
334                                                                        }
335                                                                }
336                                                        }
337                                                }
338
339                                                @Override
340                                                public void remove() {
341                                                        throw new UnsupportedOperationException();
342                                                }
343
344                                        };
345                                        return iterator;
346                                }
347
348                                @Override
349                                public int size() {
350                                        return QuadTree.this.size;
351                                }
352                        };
353                }
354                return this.values;
355        }
356

358                return this.top.firstLeaf();
359        }
360
361        private Leaf<T> nextLeaf(final Leaf<T> currentLeaf) {
362                return this.top.nextLeaf(currentLeaf);
363        }

393        public static class Rect implements Serializable {
394                private static final long serialVersionUID = -837712701959689133L;
395                public final double minX;
396                public final double minY;
397                public final double maxX;
398                public final double maxY;
399                public final double centerX;
400                public final double centerY;
401
402                public Rect(final double minX, final double minY, final double maxX, final double maxY) {
403                        this.minX = Math.min(minX, maxX);
404                        this.minY = Math.min(minY, maxY);
405                        this.maxX = Math.max(minX, maxX);
406                        this.maxY = Math.max(minY, maxY);
407                        this.centerX = (minX + maxX) / 2;
408                        this.centerY = (minY + maxY) / 2;
409                }
410
411                /**
412                 * Calculates the (minimum) distance of a given point to the border of the
413                 * rectangle. If the point lies within the rectangle, the distance
414                 * is zero.
415                 *
416                 * @param x left-right location
417                 * @param y up-down location
418                 * @return distance to border, 0 if inside rectangle or on border
419                 */
420                public double calcDistance(final double x, final double y) {
421                        double distanceX;
422                        double distanceY;
423
424                        if (this.minX <= x && x <= this.maxX) {
425                                distanceX = 0;
426                        } else {
427                                distanceX = Math.min(Math.abs(this.minX - x), Math.abs(this.maxX - x));
428                        }
429                        if (this.minY <= y && y <= this.maxY) {
430                                distanceY = 0;
431                        } else {
432                                distanceY = Math.min(Math.abs(this.minY - y), Math.abs(this.maxY - y));
433                        }
434
435                        return Math.sqrt(distanceX * distanceX + distanceY * distanceY);
436                }
437
438                /**
439                 * Calculates the distance from the given point to the furthest corner of the rectangle
440                 * 
441                 * @param x left-right location
442                 * @param y up-down location
443                 * @return distance to furthest corner of the rectangle
444                 */
445                public double calcMaxDistance(final double x, final double y) {
446                        double distanceX = Math.max(Math.abs(this.minX - x), Math.abs(this.maxX - x));
447                        double distanceY = Math.max(Math.abs(this.minY - y), Math.abs(this.maxY - y));
448
449                        return Math.sqrt(distanceX * distanceX + distanceY * distanceY);
450                }
451
452
453                /**
454                 * Tests if a specified coordinate is inside the boundary of this <code>Rect</code>.
455                 * @param x the x-coordinate to test
456                 * @param y the y-coordinate to test
457                 * @return <code>true</code> if the specified coordinates are
458                 * inside the boundary of this <code>Rect</code>;
459                 * <code>false</code> otherwise.
460                 */
461                public boolean contains(final double x, final double y) {
462                        return (x >= this.minX &&
463                                        y >= this.minY &&
464                                        x < this.maxX &&
465                                        y < this.maxY);
466                }
467
468                public boolean containsOrEquals(final double x, final double y) {
469                        return (x >= this.minX &&
470                                        y >= this.minY &&
471                                        x <= this.maxX &&
472                                        y <= this.maxY);
473                }
474
475                /**
476                 * Tests if a specified rect is inside or on the boundary of this <code>Rect</code>.
477                 * @param rect the rect to test
478                 * @return <code>true</code> if the specified rect is
479                 * inside or on the boundary of this <code>Rect</code>;
480                 * <code>false</code> otherwise.
481                 */
482                public boolean containsOrEquals(final Rect rect) {
483                        return (rect.minX >= this.minX &&
484                                        rect.minY >= this.minY &&
485                                        rect.maxX <= this.maxX &&
486                                        rect.maxY <= this.maxY);
487                }
488
489                /**
490                 * Tests if this <code>Rect</code> intersects another <code>Rect</code>.
491                 * Intersection is either interior or border of rect.
492                 * @param other The rectangle that should be tested for intersection.
493                 * @return <code>true</code> if this <code>Rect</code>
494                 * intersects the interior of the other <code>Rect</code>; <code>false</code> otherwise.
495                 */
496                public boolean intersects(final Rect other) {
497                        if ((this.maxX-this.minX) < 0 || (this.maxY-this.minY) < 0) {
498                                return false;
499                        }
500                        return (other.maxX >= this.minX &&
501                                        other.maxY >= this.minY &&
502                                        other.minX <= this.maxX &&
503                                        other.minY <= this.maxY);
504                }
505
506                /**
507                 * Computes the intersection of this <code>Rect</code> with the
508                 * specified <code>Rect</code>. Returns a new <code>Rect</code>
509                 * that represents the intersection of the two rectangles.
510                 * If the two rectangles do not intersect, the result will be
511                 * null.
512                 *
513                 * @param     r   the specified <code>Rectangle</code>
514                 * @return    the largest <code>Rectangle</code> contained in both the
515                 *            specified <code>Rectangle</code> and in
516                 *                this <code>Rectangle</code>; or if the rectangles
517                 *            do not intersect, an empty rectangle.
518                 */
519                public Rect intersection(final Rect r) {
520                        double tx1 = this.minX;
521                        double ty1 = this.minY;
522                        double tx2 = this.maxX;
523                        double ty2 = this.maxY;
524                        if (this.minX < r.minX) tx1 = r.minX;
525                        if (this.minY < r.minY) ty1 = r.minY;
526                        if (tx2 > r.maxX) tx2 = r.maxX;
527                        if (ty2 > r.maxY) ty2 = r.maxY;
528                        // did they intersect at all?
529                        if(tx2-tx1 <=0.f || ty2-ty1 <= 0.f) return null;
530
531                        return new Rect(tx1, ty1, tx2, ty2);
532                }
533
534                /**
535                 * Adds a <code>Rect</code> to this <code>Rect</code>.
536                 * The resulting <code>Rect</code> is the union of the two
537                 * rectangles (i.e. the minimum rectangle that contains the two original rectangles)
538                 * @param  r the specified <code>Rect</code>
539                 */
540                public Rect union(final Rect r) {
541                        return new Rect( Math.min(this.minX, r.minX),
542                                        Math.min(this.minY, r.minY),
543                                        Math.max(this.maxX, r.maxX),
544                                        Math.max(this.maxY, r.maxY));
545                }
546
547                /**
548                 * Increases the size of the rectangle by scaleX and scaleY.
549                 * 
550                 * (and "increase by" means:
551                 *   1.0: increase it by 100% (scale it by 2.0)
552                 *   -0.5: decrease it by 50% (scale it by 0.5)
553                 *   0.0: do nothing)   michaz
554                 * 
555                 */
556                
557                public Rect scale(double scaleX, double scaleY) {
558                        scaleY *= this.centerY - this.minY;
559                        scaleX *= this.centerX - this.minX;
560                        return new Rect(this.minX - scaleX, this.minY-scaleY, this.maxX + scaleX, this.maxY + scaleY);
561                }
562                
563                @Override
564                public String toString() {
565                        return "topLeft: ("+minX+","+minY+") bottomRight: ("+maxX+","+maxY+")";
566                }
567                
568        }

570        protected static class Leaf<T> implements Serializable {
571                private static final long serialVersionUID = -6527830222532634476L;
572                final public double x;
573                final public double y;
574                final public ArrayList<T> values;
575
576                public Leaf(final double x, final double y, final T value) {
577                        this.x = x;
578                        this.y = y;
579                        this.values = new ArrayList<T>(1);
580                        this.values.add(value);
581                }
582        }

584        protected static class Node<T> implements Serializable {
585                private static final long serialVersionUID = 8151154226742383421L;
586
587                private Leaf<T> leaf = null;
588
589                private boolean hasChilds = false;
590                private Node<T> northwest = null;
591                private Node<T> northeast = null;
592                private Node<T> southeast = null;
593                private Node<T> southwest = null;
594                private final Rect bounds;
595
596                public Node(final double minX, final double minY, final double maxX, final double maxY) {
597                        this.bounds = new Rect(minX, minY, maxX, maxY);
598                }
599
600                public boolean put(final Leaf<T> leaf) {
601                        if ( !this.bounds.containsOrEquals( leaf.x , leaf.y ) ) {
602                                throw new IllegalArgumentException( "cannot add a point at x="+leaf.x+", y="+leaf.y+" with bounds "+bounds );
603                        }
604                        if (this.hasChilds) return getChild(leaf.x, leaf.y).put(leaf);
605                        if (this.leaf == null) {
606                                this.leaf = leaf;
607                                return true;
608                        }
609                        if (this.leaf.x == leaf.x && this.leaf.y == leaf.y) {
610                                boolean changed = false;
611                                for (T value : leaf.values) {
612                                        if (!this.leaf.values.contains(value)) {
613                                                changed = this.leaf.values.add(value) || changed;
614                                        }
615                                }
616                                return changed;
617                        }
618                        this.split();
619                        return getChild(leaf.x, leaf.y).put(leaf);
620                }
621
622                public boolean put(final double x, final double y, final T value) {
623                        return put(new Leaf<T>(x, y, value));
624                }
625
626                public boolean remove(final double x, final double y, final T value) {
627                        if (this.hasChilds) return getChild(x, y).remove(x, y, value);
628                        if (this.leaf != null && this.leaf.x == x && this.leaf.y == y) {
629                                if (this.leaf.values.remove(value)) {
630                                        if (this.leaf.values.size() == 0) {
631                                                this.leaf = null;
632                                        }
633                                        return true;
634                                }
635                        }
636                        return false;
637                }
638
639                public void clear() {
640                        // we could as well just set everything to null and let the
641                        // garbage collection do its job.
642                        if (this.hasChilds) {
643                                this.northwest.clear();
644                                this.northeast.clear();
645                                this.southeast.clear();
646                                this.southwest.clear();
647                                this.northwest = null;
648                                this.northeast = null;
649                                this.southeast = null;
650                                this.southwest = null;
651                                this.hasChilds = false;
652                        } else {
653                                if (this.leaf != null) {
654                                        this.leaf.values.clear();
655                                        this.leaf = null;
656                                }
657                        }
658                }
659
660                /* default */ T get(final double x, final double y, final MutableDouble bestDistance) {
661                        if (this.hasChilds) {
662                                T closest = null;
663                                Node<T> bestChild = this.getChild(x, y);
664                                if (bestChild != null) {
665                                        closest = bestChild.get(x, y, bestDistance);
666                                }
667                                if (bestChild != this.northwest && this.northwest.bounds.calcDistance(x, y) < bestDistance.value) {
668                                        T value = this.northwest.get(x, y, bestDistance);
669                                        if (value != null) { closest = value; }
670                                }
671                                if (bestChild != this.northeast && this.northeast.bounds.calcDistance(x, y) < bestDistance.value) {
672                                        T value = this.northeast.get(x, y, bestDistance);
673                                        if (value != null) { closest = value; }
674                                }
675                                if (bestChild != this.southeast && this.southeast.bounds.calcDistance(x, y) < bestDistance.value) {
676                                        T value = this.southeast.get(x, y, bestDistance);
677                                        if (value != null) { closest = value; }
678                                }
679                                if (bestChild != this.southwest && this.southwest.bounds.calcDistance(x, y) < bestDistance.value) {
680                                        T value = this.southwest.get(x, y, bestDistance);
681                                        if (value != null) { closest = value; }
682                                }
683                                return closest;
684                        }
685                        // no more childs, so we must contain the closest object
686                        if (this.leaf != null && this.leaf.values.size() > 0) {
687                                T value = this.leaf.values.get(0);
688                                double distance = Math.sqrt(
689                                                (this.leaf.x - x) * (this.leaf.x - x)
690                                                + (this.leaf.y - y) * (this.leaf.y - y));
691                                if (distance < bestDistance.value) {
692                                        bestDistance.value = distance;
693                                        return value;
694                                }
695                        }
696                        return null;
697                }

699                /* default */ Collection<T> getElliptical(
700                                final double x1,
701                                final double y1,
702                                final double x2,
703                                final double y2,
704                                final double maxDistance,
705                                final Collection<T> values) {
706                        assert Math.pow( maxDistance , 2 ) >= Math.pow( (x1 - x2), 2 ) + Math.pow( (y1 - y2) , 2 );
707                        if (this.hasChilds) {
708                                // note: this could probably be improved. The idea here
709                                // is NOT to dive in quadrants which we know do not contain points
710                                // in the ellipse. 
711                                // This is done, but we will also dive in some quadrants not intersecting
712                                // the ellipse, which is just a loss of time (the sum of the minimum distances
713                                // to an area is always lower than the munimum of the sum of the distances,
714                                // and the difference can be substantial. Just not sure how efficiently
715                                // one can estimate the minimum of the sum of the distances).
716
717                                // this is a trick to avoid computing distance of quadrant to second focus
718                                // if the quandrant is already too far from first focus.
719                                // Some tests showed a huge improvement with this, when a large part
720                                // of the space has to be pruned (which is typically the case in
721                                // real applications). By "huge improvement", I mean several times
722                                // faster (how much depends on the particular instance).
723                                final double nw1 = this.northwest.bounds.calcDistance(x1, y1);
724                                if ( nw1 <= maxDistance && nw1 + this.northwest.bounds.calcDistance(x2, y2) <= maxDistance) {
725                                        this.northwest.getElliptical(x1, y1, x2, y2, maxDistance, values);
726                                }
727                                final double ne1 = this.northeast.bounds.calcDistance(x1, y1);
728                                if (ne1 <= maxDistance && ne1 + this.northeast.bounds.calcDistance(x2, y2) <= maxDistance) {
729                                        this.northeast.getElliptical(x1, y1, x2, y2, maxDistance, values);
730                                }
731                                final double se1 = this.southeast.bounds.calcDistance(x1, y1);
732                                if (se1 <= maxDistance && se1 + this.southeast.bounds.calcDistance(x2, y2) <= maxDistance) {
733                                        this.southeast.getElliptical(x1, y1, x2, y2, maxDistance, values);
734                                }
735                                final double sw1 = this.southwest.bounds.calcDistance(x1, y1);
736                                if (sw1 <= maxDistance && sw1 + this.southwest.bounds.calcDistance(x2, y2) <= maxDistance) {
737                                        this.southwest.getElliptical(x1, y1, x2, y2, maxDistance, values);
738                                }
739                                return values;
740                        }
741                        // no more childs, so we must contain the closest object
742                        if (this.leaf != null && this.leaf.values.size() > 0) {
743                                final double distance1 = Math.sqrt(
744                                                (this.leaf.x - x1) * (this.leaf.x - x1)
745                                                + (this.leaf.y - y1) * (this.leaf.y - y1));
746                                // same trick as above, though it should not be useul in the vast
747                                // majority of cases
748                                if (distance1 <= maxDistance ) {
749                                        final double distance2 = Math.sqrt(
750                                                        (this.leaf.x - x2) * (this.leaf.x - x2)
751                                                        + (this.leaf.y - y2) * (this.leaf.y - y2));
752                                        if ( distance1 + distance2 <= maxDistance) {
753                                                values.addAll(this.leaf.values);
754                                        }
755                                }
756                        }
757                        return values;
758                }

760                /* default */ Collection<T> get(final double x, final double y, final double maxDistance, final Collection<T> values) {
761                        if (this.hasChilds) {
762                                if (this.northwest.bounds.calcDistance(x, y) <= maxDistance) {
763                                        this.northwest.get(x, y, maxDistance, values);
764                                }
765                                if (this.northeast.bounds.calcDistance(x, y) <= maxDistance) {
766                                        this.northeast.get(x, y, maxDistance, values);
767                                }
768                                if (this.southeast.bounds.calcDistance(x, y) <= maxDistance) {
769                                        this.southeast.get(x, y, maxDistance, values);
770                                }
771                                if (this.southwest.bounds.calcDistance(x, y) <= maxDistance) {
772                                        this.southwest.get(x, y, maxDistance, values);
773                                }
774                                return values;
775                        }
776                        // no more childs, so we must contain the closest object
777                        if (this.leaf != null && this.leaf.values.size() > 0) {
778                                double distance = Math.sqrt(
779                                                (this.leaf.x - x) * (this.leaf.x - x)
780                                                + (this.leaf.y - y) * (this.leaf.y - y));
781                                if (distance <= maxDistance) {
782                                        values.addAll(this.leaf.values);
783                                }
784                        }
785                        return values;
786                }

788                /* default */ Collection<T> get(final double x, final double y, final double r_min, final double r_max,
789                                                                                Collection<T> values) {
790                        if (this.hasChilds) {
791                                stepInto(this.northwest, x, y, r_min, r_max, values);
792                                stepInto(this.northeast, x, y, r_min, r_max, values);
793                                stepInto(this.southeast, x, y, r_min, r_max, values);
794                                stepInto(this.southwest, x, y, r_min, r_max, values);
795                                return values;
796                        }
797                        // no more childs, so we must contain the closest object
798                        if (this.leaf != null && this.leaf.values.size() > 0) {
799                                double distance = Math.sqrt(
800                                                (this.leaf.x - x) * (this.leaf.x - x)
801                                                                + (this.leaf.y - y) * (this.leaf.y - y));
802                                if (distance <= r_max && distance >= r_min) {
803                                        values.addAll(this.leaf.values);
804                                }
805                        }
806                        return values;
807                }

809                private void stepInto(Node node, double x, double y, double r_min, double r_max, Collection<T> values) {
810                        double minDistance = node.bounds.calcDistance(x, y);
811                        double maxDistance = node.bounds.calcMaxDistance(x, y);
812
813                        if(minDistance <= r_max && maxDistance >= r_min) {
814                                node.get(x, y, r_min, r_max, values);
815                        }
816                }

818                /* default */ Collection<T> get(final Rect bounds, final Collection<T> values) {
819                        if (this.hasChilds) {
820                                if (this.northwest.bounds.intersects(bounds)) {
821                                        this.northwest.get(bounds, values);
822                                }
823                                if (this.northeast.bounds.intersects(bounds)) {
824                                        this.northeast.get(bounds, values);
825                                }
826                                if (this.southeast.bounds.intersects(bounds)) {
827                                        this.southeast.get(bounds, values);
828                                }
829                                if (this.southwest.bounds.intersects(bounds)) {
830                                        this.southwest.get(bounds, values);
831                                }
832                                return values;
833                        }
834                        // no more childs, so we must contain the closest object
835                        if (this.leaf != null && this.leaf.values.size() > 0 && bounds.containsOrEquals(this.leaf.x, this.leaf.y)) {
836                                values.addAll(this.leaf.values);
837                        }
838                        return values;
839                }

841                /* default */ int execute(final Rect globalBounds, final Executor<T> executor) {
842                        int count = 0;
843                        if (this.hasChilds) {
844                                if (this.northwest.bounds.intersects(globalBounds)) {
845                                        count += this.northwest.execute(globalBounds, executor);
846                                }
847                                if (this.northeast.bounds.intersects(globalBounds)) {
848                                        count += this.northeast.execute(globalBounds, executor);
849                                }
850                                if (this.southeast.bounds.intersects(globalBounds)) {
851                                        count += this.southeast.execute(globalBounds, executor);
852                                }
853                                if (this.southwest.bounds.intersects(globalBounds)) {
854                                        count += this.southwest.execute(globalBounds, executor);
855                                }
856                                return count;
857                        }
858                        // no more childs, so we must contain the closest object
859                        if (this.leaf != null && this.leaf.values.size() > 0 && globalBounds.contains(this.leaf.x, this.leaf.y)) {
860                                count += this.leaf.values.size();
861                                for (T object : this.leaf.values) executor.execute(this.leaf.x, this.leaf.y, object);
862                        }
863                        return count;
864                }

866                private void split() {
867                        this.northwest = new Node<T>(this.bounds.minX, this.bounds.centerY, this.bounds.centerX, this.bounds.maxY);
868                        this.northeast = new Node<T>(this.bounds.centerX, this.bounds.centerY, this.bounds.maxX, this.bounds.maxY);
869                        this.southeast = new Node<T>(this.bounds.centerX, this.bounds.minY, this.bounds.maxX, this.bounds.centerY);
870                        this.southwest = new Node<T>(this.bounds.minX, this.bounds.minY, this.bounds.centerX, this.bounds.centerY);
871                        this.hasChilds = true;
872                        if (this.leaf != null) {
873                                getChild(this.leaf.x, this.leaf.y).put(this.leaf);
874                                this.leaf = null;
875                        }
876                }

878                private Node<T> getChild(final double x, final double y) {
879                        if (this.hasChilds) {
880                                if (x < this.bounds.centerX) {
881                                        if (y < this.bounds.centerY)
882                                                return this.southwest;
883                                        return this.northwest;
884                                }
885                                if (y < this.bounds.centerY)
886                                        return this.southeast;
887                                return this.northeast;
888                        }
889                        return null;
890                }

892                /* default */ Leaf<T> firstLeaf() {
893                        if (this.hasChilds) {
894                                Leaf<T> leaf = this.southwest.firstLeaf();
895                                if (leaf == null) { leaf = this.northwest.firstLeaf(); }
896                                if (leaf == null) { leaf = this.southeast.firstLeaf(); }
897                                if (leaf == null) { leaf = this.northeast.firstLeaf(); }
898                                return leaf;
899                        }
900                        return this.leaf;
901                }

903                /* default */ boolean nextLeaf(final Leaf<T> currentLeaf, final MutableLeaf<T> nextLeaf) {
904                        if (this.hasChilds) {
905                                boolean found = false;
906                                if (currentLeaf.x <= this.bounds.centerX && currentLeaf.y <= this.bounds.centerY) {
907                                        found = this.southwest.nextLeaf(currentLeaf, nextLeaf);
908                                        if (found) {
909                                                if (nextLeaf.value == null) { nextLeaf.value = this.northwest.firstLeaf(); }
910                                                if (nextLeaf.value == null) { nextLeaf.value = this.southeast.firstLeaf(); }
911                                                if (nextLeaf.value == null) { nextLeaf.value = this.northeast.firstLeaf(); }
912                                                return true;
913                                        }
914                                }
915                                if (currentLeaf.x <= this.bounds.centerX && currentLeaf.y >= this.bounds.centerY) {
916                                        found = this.northwest.nextLeaf(currentLeaf, nextLeaf);
917                                        if (found) {
918                                                if (nextLeaf.value == null) { nextLeaf.value = this.southeast.firstLeaf(); }
919                                                if (nextLeaf.value == null) { nextLeaf.value = this.northeast.firstLeaf(); }
920                                                return true;
921                                        }
922                                }
923                                if (currentLeaf.x >= this.bounds.centerX && currentLeaf.y <= this.bounds.centerY) {
924                                        found = this.southeast.nextLeaf(currentLeaf, nextLeaf);
925                                        if (found) {
926                                                if (nextLeaf.value == null) { nextLeaf.value = this.northeast.firstLeaf(); }
927                                                return true;
928                                        }
929                                }
930                                if (currentLeaf.x >= this.bounds.centerX && currentLeaf.y >= this.bounds.centerY) {
931                                        return this.northeast.nextLeaf(currentLeaf, nextLeaf);
932                                }
933                                return false;
934                        }
935                        return currentLeaf == this.leaf;
936                }
