034        /** 
035         * Never change this to NaN, as a compare of any valid time
036         * to this should result to "greater" for some algorithms to work
037         * still we found the name "UNDEFINED" more suitable than TIME_MIN_VALUE
038         * <br><b><i>Note:</i></b> do not interpret the "UNDEFINED" as "time does 
039         * not matter", as this has implications for, example, routing. If start 
040         * time is given as {@link Time#UNDEFINED_TIME} then {@link Path#travelTime}
041         * will return {@link Double#NaN}, even though the {@link TravelTime#getLinkTravelTime}
042         * is independent of the start time. */
043        @Deprecated // rather use Time.isUndefinedTime( time ), since that opens up the path to a later change
044        // of the convention.  kai, nov'17
045        public final static double UNDEFINED_TIME = Double.NEGATIVE_INFINITY;

047         * The end of a day in MATSim in seconds

049        public final static double MIDNIGHT = 24 * 3600.0;

051        public static final String TIMEFORMAT_HHMM = "HH:mm";
052        public static final String TIMEFORMAT_HHMMSS = "HH:mm:ss";

057        private static String defaultTimeFormat = TIMEFORMAT_HHMMSS;

061        public static boolean isUndefinedTime( final double time ) {

074        static {
075                timeElements = new String[60];
076                for (int i = 0; i < 10; i++) {
077                        timeElements[i] = "0" + i;
078                }
079                for (int i = 10; i < 60; i++) {
080                        timeElements[i] = Integer.toString(i);
081                }
082        }

106        /**
107         * Converts the given time in seconds after midnight into a textual representation
108         *
109         * @param seconds The time to convert, measured in seconds after midnight.
110         * @param timeformat
111         * @param separator
112         * @return A textual representation of the passed time.
113         */
114        public static final String writeTime(final double seconds, final String timeformat, final char separator) {
115                if (TIMEFORMAT_SSSS.equals(timeformat)) {
116                        return Long.toString((long)seconds);
117                }
118                if (seconds < 0) {
119                        if (seconds == UNDEFINED_TIME) return "undefined";
120                        return "-" + writeTime(Math.abs(seconds), timeformat, separator);
121                }
122                double s = seconds;
123                long h = (long)(s / 3600);
124                s = s % 3600;
125                int m = (int)(s / 60);
126                s = s % 60;
127
128                StringBuilder str = new StringBuilder(10);
129
130                if (h < timeElements.length) {
131                        str.append(timeElements[(int) h]);
132                } else {
133                        str.append(Long.toString(h));
134                }
135
136                str.append(separator);
137                str.append(timeElements[m]);
138
139                if (TIMEFORMAT_HHMM.equals(timeformat)) {
140                        return str.toString();
141                }
142                if (TIMEFORMAT_HHMMSS.equals(timeformat)) {
143                        str.append(separator);
144                        str.append(timeElements[(int)s]);
145                        return str.toString();
146                }
147                if ( TIMEFORMAT_HHMMSSDOTSS.equals(timeformat)) {
148                        str.append(separator);
149
150                        if ( s < 10. ) {
151                                str.append("0") ;
152                        }
153                        str.append(s);
154                        return str.toString();
155                }
156
157                throw new IllegalArgumentException("The time format (" + timeformat + ") is not known.");
158        }

160        /**
161         * Parses the given string for a textual representation for time and returns
162         * the time value in seconds past midnight. It is the same as {@link #parseTime(String, char)}
163         * with the separator set to ':'.
164         *
165         * @param time the string describing a time to parse.
166         *
167         * @return the parsed time as seconds after midnight.
168         *
169         * @throws IllegalArgumentException when the string cannot be interpreted as a valid time.
170         */
171        public static final double parseTime(final String time) {
172                return parseTime(time, ':');
173        }
174
175        /**
176         * Parses the given string for a textual representation for time and returns
177         * the time value in seconds past midnight. The following formats are recognized:
178         * HH:mm:ss, HH:mm, ssss.
179         *
180         * @param time the string describing a time to parse.
181         * @param separator the character used between hours and minutes, and minutes and seconds.
182         *
183         * @return the parsed time as seconds after midnight.
184         *
185         * @throws IllegalArgumentException when the string cannot be interpreted as a valid time.
186         */
187        public static final double parseTime(final String time, final char separator) {
188                if (time == null || time.length() == 0 || time.equals("undefined")) {
189                        return Time.UNDEFINED_TIME;
190                }
191                boolean isNegative = (time.charAt(0) == '-');
192                String[] strings = (isNegative
193                                ? StringUtils.explode(time.substring(1), separator)
194                                                : StringUtils.explode(time, separator));
195                double seconds = 0;
196                if (strings.length == 1) {
197                        seconds = Math.abs(Double.parseDouble(strings[0]));
198                } else if (strings.length == 2) {
199                        long h = Long.parseLong(strings[0]);
200                        int m = Integer.parseInt(strings[1]);
201
202                        if ((m < 0) || (m > 59)) {
203                                throw new IllegalArgumentException("minutes are out of range in " + time);
204                        }
205
206                        seconds = Math.abs(h) * 3600 + m * 60;
207                } else if (strings.length == 3) {
208                        long h = Long.parseLong(strings[0]);
209                        int m = Integer.parseInt(strings[1]);
210                        double s = Double.parseDouble(strings[2]);
211
212                        if ((m < 0) || (m > 59)) {
213                                throw new IllegalArgumentException("minutes are out of range in " + time);
214                        }
215                        if ((s < 0) || (s >= 60)) {
216                                throw new IllegalArgumentException("seconds are out of range in " + time);
217                        }
218
219                        seconds = Math.abs(h) * 3600 + m * 60 + s;
220                } else {
221                        throw new IllegalArgumentException("time format is not valid in " + time);
222                }
223
224                if (isNegative) {
225                        seconds = -seconds;
226                }
227                return seconds;
228        }
229
230        /**
231         * Converts a number like 1634 to the time value of 16:34:00.
232         * 
233         * @param hhmm the time-representing number to convert.
234         * @return the time as seconds after midnight.
235         */
236        public static double convertHHMMInteger(int hhmm) {
237                int h = hhmm / 100;
238                int m = hhmm - (h * 100);
239                double seconds = Math.abs(h) * 3600 + m * 60;
240                return seconds;
241        }
