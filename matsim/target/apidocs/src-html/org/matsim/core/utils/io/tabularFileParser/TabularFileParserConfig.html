021package org.matsim.core.utils.io.tabularFileParser;
022
023import java.nio.charset.Charset;
024
025/**
026 * Configuration for a <code>TabularFileParser</code>.
027 *
028 * @author gunnar
029 *
030 */
031public class TabularFileParserConfig {
032
033    // -------------------- CLASS VARIABLES --------------------
034
035    private static final String ANYTHING = ".*";
036
037    private static final String ANY_SPACE = "\\s*";
038
039    // -------------------- INSTANCE VARIABLES --------------------
040
041    private String file = null;
042
043    private String startRegex = null;
044
045    private String endRegex = null;
046
047    private String commentRegex = null;
048
049    private String delimiterRegex = null;
050    
051    private Charset charset = Charset.forName("UTF8");
052
053    // -------------------- CONSTRUCTION --------------------
054
055    /**
056     * Empty default constructor.
057     */
058    public TabularFileParserConfig() {
059    }
060
061    // -------------------- SETTERS --------------------
062
063    /**
064     * Sets the file to be parsed.
065     *
066     * @param file
067     *            the file to be parsed
068     */
069    public void setFileName(String file) {
070        this.file = file;
071    }
072    
073    /**
074     * Sets the charset for the file. Defaults to UTF-8
075     * 
076     * @param charset the charset used to reade the file
077     */
078    public void setCharset(Charset charset) {
079        this.charset = charset;
080    }
081
082    // ---------- DIRECT SETTING OF REGULAR EXPRESSIONS ----------
083
084    /**
085     * Sets the regular expression that identifies the the first line of the
086     * file section to be parsed.
087     *
088     * @param regex
089     *            the regular expression that identifies the the first line of
090     *            the file section to be parsed
091     */
092    public void setStartRegex(String regex) {
093        this.startRegex = regex;
094    }
095
096    /**
097     * Sets the regular expression that identifies the first line <em>after</em>
098     * the file section to be parsed.
099     *
100     * @param regex
101     *            the regular expression that identifies the first line
102     *            <em>after</em> the file section to be parsed
103     */
104    public void setEndRegex(String regex) {
105        this.endRegex = regex;
106    }
107
108    /**
109     * Sets the regular expression that identifies lines to be ignored during
110     * parsing.
111     *
112     * @param regex
113     *            the regular expression that identifies lines to be ignored
114     *            during parsing
115     */
116    public void setCommentRegex(String regex) {
117        this.commentRegex = regex;
118    }
119
120    /**
121     * Sets the regular expression that identifies splitting locations in a
122     * parsed line.
123     *
124     * @param regex
125     *            the regular expression that identifies splitting locations in
126     *            a parsed line
127     */
128    public void setDelimiterRegex(String regex) {
129        this.delimiterRegex = regex;
130    }
131
132    // ---------- CREATION OF REGULAR EXPRESSIONS FROM TAGS ----------
133
134    /**
135     * Parsing starts with the first occurrence of <code>tag</code> at the
136     * beginning of a line.
137     *
138     * @param tag
139     *            the first line to be parsed begins with this
140     *            <code>String</code>
141     */
142    public void setStartTag(String tag) {
143        if (tag != null)
144            startRegex = quote(tag) + ANYTHING;
145    }
146
147    /**
148     * Parsing ends with the first occurence of <code>tag</code> at the
149     * beginning of a line.
150     *
151     * @param tag
152     *            the line before which parsing stops begins with this
153     *            <code>String</code>
154     */
155    public void setEndTag(String tag) {
156        if (tag != null)
157            endRegex = quote(tag) + ANYTHING;
158    }
159
160    /**
161     * All lines that begin with an element in the <code>tags</code> array are
162     * ignored
163     *
164     * @param tags
165     *            an array of line beginnings that are to be ignored
166     */
167    public void setCommentTags(String[] tags) {
168        commentRegex = alternativeExpr(tags) + ANYTHING;
169    }
170
171    /**
172     * A line is split into seperate rows wherever the parser encounters an
173     * element of the <code>tags</code> array, which is be preceeded and
174     * succeeded by zero or more whitespaces.
175     *
176     * @param tags
177     *            an array of <code>String</code>s denoting column delimiters
178     */
179    public void setDelimiterTags(String[] tags) {
180        if (tags == null || tags.length == 0)
181            delimiterRegex = null;
182        else
183            delimiterRegex = ANY_SPACE + alternativeExpr(tags) + ANY_SPACE;
184    }
185
186    // ---------- GENERATION OF REGULAR EXPRESSIONS ----------
187
188    private String alternativeExpr(String[] alternatives) {
189        StringBuilder result = new StringBuilder();
190
191        if (alternatives != null)
192            for (int i = 0; i < alternatives.length; i++) {
193                result.append(quote(alternatives[i]));
194                if (i < alternatives.length - 1)
195                    result.append('|');
196            }
197        return result.toString();
198    }
199
200    private String quote(String expr) {
201        return "\\Q" + expr + "\\E";
202    }
203
204    // -------------------- DATA ACCESS --------------------
205
206    public String getFile() {
207        return file;
208    }
209
210    public String getStartRegex() {
211        return startRegex;
212    }
213
214    public String getEndRegex() {
215        return endRegex;
216    }
217
218    public String getCommentRegex() {
219        return commentRegex;
220    }
221
222    public String getDelimiterRegex() {
223        return delimiterRegex;
224    }
225    
226    public Charset getCharset() {
227        return this.charset;
228    }
229
230    // MISC
231
232    @Override
233                public String toString() {
234        StringBuilder result = new StringBuilder(100);
235
236        result.append("TabularFileParserConfig:\n\tfile=");
237        result.append(file);
238        result.append("\n\tstartRegex=");
239        result.append(startRegex);
240        result.append("\n\tendRegex=");
241        result.append(endRegex);
242        result.append("\n\tcommentRegex=");
243        result.append(commentRegex);
244        result.append("\n\tdelimiterRegex=");
245        result.append(delimiterRegex);
246
247        return result.toString();
248    }
249
250}
