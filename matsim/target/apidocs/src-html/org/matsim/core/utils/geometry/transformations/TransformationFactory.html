021package org.matsim.core.utils.geometry.transformations;
022
023import org.matsim.core.utils.geometry.CoordinateTransformation;
024
025/**
026 * A factory to instantiate a specific coordinate transformation.
027 *
028 * @author mrieser
029 *
030 */
031public abstract class TransformationFactory {
032
033        public final static String WGS84 = "WGS84";
034        public final static String ATLANTIS = "Atlantis";
035        public final static String CH1903_LV03 = "CH1903_LV03"; // switzerland
036        public final static String CH1903_LV03_Plus = "CH1903_LV03_Plus"; // switzerland new
037        public final static String GK4 = "GK4"; // berlin/germany, own implementation
038        public final static String WGS84_UTM47S = "WGS84_UTM47S"; // indonesia
039        public final static String WGS84_UTM48N = "WGS84_UTM48N"; // Singapore
040        public final static String WGS84_UTM35S = "WGS84_UTM35S"; // South Africa (Gauteng)
041        public final static String WGS84_UTM36S = "WGS84_UTM36S"; // South Africa (eThekwini, Kwazulu-Natal)
042        public final static String WGS84_Albers = "WGS84_Albers"; // South Africa (Africa Albers Equal Conic)
043        public final static String WGS84_SA_Albers = "WGS84_SA_Albers"; // South Africa (Adapted version of Africa Albers Equal) 
044        public final static String HARTEBEESTHOEK94_LO19 = "SA_Lo19"; // South Africa adaption of Transverse Mercator.
045        public final static String HARTEBEESTHOEK94_LO29 = "SA_Lo29"; // South Africa adaption of Transverse Mercator.
046        public final static String HARTEBEESTHOEK94_LO31 = "SA_Lo31"; // South Africa adaption of Transverse Mercator.
047        public final static String WGS84_UTM33N = "WGS84_UTM33N"; // Berlin
048        public final static String DHDN_GK4 = "DHDN_GK4"; // berlin/germany, for GeoTools
049        public final static String WGS84_UTM29N = "WGS84_UTM29N"; // coimbra/portugal
050    public final static String WGS84_UTM31N = "WGS84_UTM31N"; // Barcelona/Spain
051        public final static String CH1903_LV03_GT = "CH1903_LV03_GT"; //use geotools also for swiss coordinate system
052        public final static String CH1903_LV03_Plus_GT = "CH1903_LV03_Plus_GT"; //use geotools also for swiss coordinate system
053        public final static String WGS84_SVY21 = "WGS84_SVY21"; //Singapore2
054        public final static String NAD83_UTM17N = "NAD83_UTM17N"; //Toronto, Canada
055        public static final String WGS84_TM = "WGS84_TM"; //Singapore3
056        public static final String PCS_ITRF2000_TM_UOS = "PCS_ITRF2000_TM_UOS"; // South Korea - but used by University of Seoul - probably a wrong one...
057        public static final String DHDN_SoldnerBerlin = "DHDN_SoldnerBerlin"; // Berlin
058        
059        /**
060         * Returns a coordinate transformation to transform coordinates from one
061         * coordinate system to another one.
062         *
063         * @param fromSystem The source coordinate system.
064         * @param toSystem The destination coordinate system.
065         * @return Coordinate Transformation
066         */
067        public static CoordinateTransformation getCoordinateTransformation(final String fromSystem, final String toSystem) {
068                if (fromSystem.equals(toSystem)) return new IdentityTransformation();
069                if (WGS84.equals(fromSystem)) {
070                        if (CH1903_LV03.equals(toSystem)) return new WGS84toCH1903LV03();
071                        if (CH1903_LV03_Plus.equals(toSystem)) return new WGS84toCH1903LV03Plus();
072                        if (ATLANTIS.equals(toSystem)) return new WGS84toAtlantis();
073                }
074                if (WGS84.equals(toSystem)) {
075                        if (CH1903_LV03.equals(fromSystem)) return new CH1903LV03toWGS84();
076                        if (CH1903_LV03_Plus.equals(fromSystem)) return new CH1903LV03PlustoWGS84();
077                        if (GK4.equals(fromSystem)) return new GK4toWGS84();
078                        if (ATLANTIS.equals(fromSystem)) return new AtlantisToWGS84();
079                }
080                return new GeotoolsTransformation(fromSystem, toSystem);
081        }
082}
