021package org.matsim.core.utils.io;
022
023import java.io.BufferedWriter;
024import java.io.IOException;
025import java.io.OutputStream;
026import java.io.OutputStreamWriter;
027import java.util.zip.GZIPOutputStream;
028
029
030/**
031 * A simple, abstract helper class to open files for writing with support for
032 * gzip-compression.
033 *
034 * @author mrieser
035 */
036public abstract class AbstractMatsimWriter {
037
038        /** The Unix newline character. */
039        protected static final String NL = "\n";
040
041        /** The writer output can be written to. */
042        protected BufferedWriter writer = null;
043
044        /** Whether or not the output is gzip-compressed. If <code>null</code>, the
045         * usage of compression is decided by the filename (whether it ends with .gz
046         * or not). */
047        protected Boolean useCompression = null;
048
049        /**
050         * Sets whether the file should be gzip-compressed or not. Must be set before
051         * the file is opened for writing. If not set explicitly, the usage of
052         * compression is defined by the ending of the filename.
053         *
054         * @param useCompression1
055         */
056        public final void useCompression(final boolean useCompression1) {
057                this.useCompression = Boolean.valueOf(useCompression1);
058        }
059
060        /**
061         * Opens the specified file for writing.
062         *
063         * @param filename
064         * @throws UncheckedIOException
065         */
066        protected final void openFile(final String filename) throws UncheckedIOException {
067                assertNotAlreadyOpen();
068                if (this.useCompression == null) {
069                        this.writer = IOUtils.getBufferedWriter(filename);
070                } else {
071                        this.writer = IOUtils.getBufferedWriter(filename, this.useCompression.booleanValue());
072                }
073        }
074
075        /**
076         * Uses the specified OutputStream for writing.
077         * 
078         */
079        protected final void openOutputStream(OutputStream outputStream) {
080                assertNotAlreadyOpen();
081                try {
082                        if (this.useCompression == null || this.useCompression.booleanValue()) {
083                                this.writer = new BufferedWriter(new OutputStreamWriter(outputStream));
084                        } else {
085                                this.writer = new BufferedWriter(new OutputStreamWriter(new GZIPOutputStream(outputStream)));
086                        }
087                } catch (IOException e) {
088                        throw new UncheckedIOException(e);
089                }
090        }
091
092        private void assertNotAlreadyOpen() {
093                if (this.writer != null) {
094                        throw new RuntimeException("File already open.");
095                }
096        }
097
098        /**
099         * Closes the file if it is still open.
100         *
101         * @throws UncheckedIOException
102         */
103        protected final void close() throws UncheckedIOException {
104                if (this.writer != null) {
105                        try {
106                                this.writer.flush();
107                                this.writer.close();
108                        } catch (IOException e) {
109                                throw new UncheckedIOException(e);
110                        } finally {
111                                this.writer = null;
112                        }
113                }
114        }
115
116}
