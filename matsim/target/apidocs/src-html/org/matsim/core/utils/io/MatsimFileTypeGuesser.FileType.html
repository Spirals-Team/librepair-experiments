024import java.util.Locale;

026import javax.xml.parsers.ParserConfigurationException;

037/**
038 * Tries to figure out the file format of the given file.
039 *
040 * @author mrieser
041 */
042public class MatsimFileTypeGuesser extends DefaultHandler {
043
044        private static final Logger log = Logger.getLogger(MatsimFileTypeGuesser.class);
045        /**
046         * This enum only informs about the correct container, not about the version of the input file.
047         */
048        public enum FileType {Config, Network, Facilities, Population, World,
049                Counts, Events, Households, TransimsVehicle, OTFVis, SignalSystems, LaneDefinitions, SignalGroups, SignalControl, AmberTimes,
050                TransitSchedule, Vehicles, ObjectAttributes}
051
052        public static final String SYSTEMIDNOTFOUNDMESSAGE = "System Id of xml document couldn't be detected. " +
053        "Make sure that you try to read a xml document with a valid header. " +
054        "If your header seems to be ok, make shure that there is no / at the " +
055        "end of the first part of the tuple used as value for xsi:schemaLocation.";
056
057
058        private FileType fileType = null;
059        private String xmlPublicId = null;
060        private String xmlSystemId = null;
061
062        public MatsimFileTypeGuesser(final String fileName) throws UncheckedIOException {
063                String name = fileName.toLowerCase(Locale.ROOT);
064                if (name.endsWith(".xml.gz") || name.endsWith(".xml")) {
065                        guessFileTypeXml(fileName);
066                        // I think the following would also be useful for the API, but with which name?
067                        String shortSystemId = null;
068                        if (this.xmlSystemId != null) {
069                                 shortSystemId = this.xmlSystemId.substring(this.xmlSystemId.replace('\\', '/').lastIndexOf('/') + 1);
070                        }
071                        if (shortSystemId != null) {
072                                if (shortSystemId.startsWith("network_")) {
073                                        this.fileType = FileType.Network;
074                                } else if (shortSystemId.startsWith("world_")) {
075                                        this.fileType = FileType.World;
076                                } else if (shortSystemId.startsWith("plans_")) {
077                                        this.fileType = FileType.Population;
078                                } else if (shortSystemId.startsWith("population_")) {
079                                        this.fileType = FileType.Population;
080                                } else if (shortSystemId.startsWith("facilities_")) {
081                                        this.fileType = FileType.Facilities;
082                                } else if (shortSystemId.startsWith("config_")) {
083                                        this.fileType = FileType.Config;
084                                } else if (shortSystemId.startsWith("counts_")) {
085                                        this.fileType = FileType.Counts;
086                                } else if (shortSystemId.startsWith("vehicleDefinitions_")) {
087                                        this.fileType = FileType.Vehicles;
088                                } else if (shortSystemId.startsWith("transitSchedule_")) {
089                                        this.fileType = FileType.TransitSchedule;
090                                } else if (shortSystemId.startsWith("objectattributes_")) {
091                                        this.fileType = FileType.ObjectAttributes;
092                                }
093                        }
094
095                } else if (name.endsWith(".txt.gz") || name.endsWith(".txt")) {
096                        this.fileType = FileType.Events;
097                } else if (name.endsWith(".mvi.gz") || name.endsWith(".mvi")) {
098                        this.fileType = FileType.OTFVis;
099                } else if (name.endsWith(".veh.gz") || name.endsWith(".veh")) {
100                        this.fileType = FileType.TransimsVehicle;
101                }
102        }
103
104        public FileType getGuessedFileType() {
105                return this.fileType;
106        }
107
108        /**
109         * @return if the file is an XML file, this returns the public-id of the declared type,
110         *  <code>null</code> if no type is declared or if it is not an XML file.
111         */
112        public String getPublicId() {
113                return this.xmlPublicId;
114        }
115
116        /**
117         * @return if the file is an XML file, this returns the system-id of the declared type,
118         *  <code>null</code> if no type is declared or if it is not an XML file.
119         */
120        public String getSystemId() {
121                return this.xmlSystemId;
122        }
123
124        private void guessFileTypeXml(final String fileName) throws UncheckedIOException {
125                SAXParserFactory factory = SAXParserFactory.newInstance();
126                factory.setValidating(false);
127                factory.setNamespaceAware(true);
128                try {
129                        XmlHandler handler = new XmlHandler();
130                        InputSource input = new InputSource(IOUtils.getBufferedReader(fileName));
131                        factory.setFeature("http://apache.org/xml/features/validation/schema", true);
132                        SAXParser parser = factory.newSAXParser();
133                        XMLReader reader = parser.getXMLReader();
134                        reader.setContentHandler(handler);
135                        reader.setErrorHandler(handler);
136                        reader.setEntityResolver(handler);
137                        reader.parse(input);
138                } catch (IOException e) {
139                        throw new UncheckedIOException(e);
140                } catch (SAXException e) {
141                        throw new UncheckedIOException(e);
142                } catch (ParserConfigurationException e) {
143                        throw new UncheckedIOException(e);
144                } catch (XMLTypeDetectionException e) {
145                        this.xmlPublicId = e.publicId;
146                        this.xmlSystemId = e.systemId;
147                        log.debug("Detected public id: " + this.xmlPublicId);
148                        log.debug("Detected system Id: " + this.xmlSystemId);
149                        if (e.rootTag != null) {
150                                log.debug("Detected root tag: " +  e.rootTag);
151                                if      ("events".equals(e.rootTag)) {
152                                        this.fileType = FileType.Events;
153                                } else if ("signalGroups".equals(e.rootTag)) {
154                                        this.fileType = FileType.SignalGroups;
155                                } else if ("signalSystems".equals(e.rootTag)) {
156                                        this.fileType = FileType.SignalSystems;
157                                } else if ("signalControl".equals(e.rootTag)) {
158                                        this.fileType = FileType.SignalControl;
159                                } else if ("laneDefinitions".equals(e.rootTag)) {
160                                        this.fileType = FileType.LaneDefinitions;
161                                }       else if ("counts".equals(e.rootTag)) {
162                                        this.fileType = FileType.Counts;
163                                } else if ("transitSchedule".equals(e.rootTag)) {
164                                        this.fileType = FileType.TransitSchedule;
165                                } else if ("objectAttributes".equals(e.rootTag)) {
166                                        this.fileType = FileType.ObjectAttributes;
167                                } else {
168                                        log.warn("got unexpected rootTag: " + e.rootTag);
169                                }
170                        }
171                }
172        }
173
174        private final static class XmlHandler extends DefaultHandler {
175
176                private XMLTypeDetectionException exception;
177
178                private boolean detectedFirstEntity = false;
179
180                public XmlHandler() {
181                        // public constructor for private inner class
182                }
183
184                @Override
185                public InputSource resolveEntity(final String publicId, final String systemId) {
186                        /**
187                         * As the xml schema of interest may be derived from other schema instances we
188                         * are only interested in the first entity resolved.
189                         */
190                        if (! this.detectedFirstEntity){
191                                this.exception  = new XMLTypeDetectionException(publicId, systemId);
192                                this.detectedFirstEntity = true;
193                        }
194                        if (systemId.endsWith(".dtd")){
195                                throw this.exception;
196                        }
197                        return null;
198                }
199
200                @Override
201                public void startElement(final String uri, final String localName, final String qName, final Attributes atts) throws SAXException {
202                        String tag = (uri.length() == 0) ? qName : localName;
203                        if (this.exception == null) {
204                                this.exception = new XMLTypeDetectionException(null, null);
205                        }
206                        this.exception.rootTag = tag;
207                        throw this.exception;
208                }
209        }
210
211        /**
212         * Used to return the declared type encountered in the XML file, the root tag
213         * and to stop parsing the file
214         * @author dgrether
215         *
216         */
217        private final static class XMLTypeDetectionException extends RuntimeException {
218
219                private static final long serialVersionUID = 1L;
220
221                public final String publicId;
222                public final String systemId;
223                public String rootTag;
224
225                public XMLTypeDetectionException(final String publicId, final String systemId){
226                        this.publicId = publicId;
227                        this.systemId = systemId;
228                }
229
230                @Override
231                public synchronized Throwable fillInStackTrace() {
232                        return this; // optimization, as we're never interested in that stack trace
233                }
234
235        }
