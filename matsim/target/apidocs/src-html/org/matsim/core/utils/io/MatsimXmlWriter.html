021package org.matsim.core.utils.io;
022
023import java.io.IOException;
024import java.util.List;
025
026import org.matsim.core.utils.collections.Tuple;
027import org.matsim.core.utils.misc.Time;
028
029/**
030 * A simple abstract class to write XML files.
031 *
032 * @author mrieser
033 */
034public abstract class MatsimXmlWriter extends AbstractMatsimWriter {
035
036        /** The default location where dtds are stored on the internet. */
037        public static final String DEFAULT_DTD_LOCATION = "http://www.matsim.org/files/dtd/";
038        /**
039         * The namespace used in the matsim xml formats
040         */
041        public static final String MATSIM_NAMESPACE = "http://www.matsim.org/files/dtd";
042        /**
043         * Constant for the xml namespace attribute
044         */
045        public static final String XMLNS = "xmlns";
046        /**
047         * Default location of the namespace of xml schema
048         */
049        public static final String DEFAULTSCHEMANAMESPACELOCATION = "http://www.w3.org/2001/XMLSchema-instance";
050
051        /**
052         * Stores the current level of indentation
053         */
054        private int indentationLevel = 0;
055        /**
056         * The String used for indentation
057         */
058        private String indentationString = "\t";
059        /**
060         * Set this flag to true for pretty print xml output, set it to false if
061         * no indentation, whitespaces and newlines shall be printed.
062         */
063        private boolean doPrettyPrint = true;
064        private boolean noWhitespaces = false;
065
066        /**
067         * Set the pretty print flag of the XMLWriter, see comment of flag.
068         * @param doPrettyPrint
069         */
070        public final void setPrettyPrint(boolean doPrettyPrint){
071                this.doPrettyPrint = doPrettyPrint;
072        }
073        /**
074         * Set the String used for indentation in the pretty print xml mode.
075         * @param indentationString
076         */
077        public final void setIndentationString(String indentationString) {
078                this.indentationString = indentationString;
079        }
080
081        /**
082         * Set the indentation level of the writer for
083         * pretty print option
084         * @param level
085         */
086        protected final void setIndentationLevel(int level){
087                this.indentationLevel = level;
088        }
089
090//      protected int getIndentationLevel(){
091//              return this.indentationLevel;
092//      }
093
094        /**
095         * Writes the standard xml 1.0 header to the output.
096         *
097         * @throws UncheckedIOException
098         */
099        protected final void writeXmlHead() throws UncheckedIOException {
100                try {
101                        this.writer.write("<?xml version=\"1.0\" encoding=\"UTF-8\"?>");
102                        if (doPrettyPrint) {
103                                this.writer.write(NL);
104                        }
105                } catch (IOException e) {
106                        throw new UncheckedIOException(e);
107                }
108        }
109
110        /**
111         * Writes the doctype declaration to the output.
112         *
113         * @param rootTag The root tag of the written XML document.
114         * @param dtdUrl The location of the document type definition of this XML document.
115         * @throws UncheckedIOException
116         */
117        protected final void writeDoctype(String rootTag, String dtdUrl) throws UncheckedIOException {
118                try {
119                        this.writer.write("<!DOCTYPE " + rootTag + " SYSTEM \"" + dtdUrl + "\">\n");
120                } catch (IOException e) {
121                        throw new UncheckedIOException(e);
122                }
123        }
124
125        /**
126         * Convenience method to create XML Attributes written by startTag()
127         */
128        protected static Tuple<String, String> createTuple(String one, String two){
129                return new Tuple<>(one, two);
130        }
131        /**
132         * Convenience method to create XML Attributes written by startTag()
133         */
134        protected static Tuple<String, String> createTuple(String one, int two) {
135                return MatsimXmlWriter.createTuple(one, Integer.toString(two));
136        }
137
138        /**
139         * Convenience method to create XML Attributes written by startTag()
140         */
141        protected static Tuple<String, String> createTuple(String one, double two) {
142                return MatsimXmlWriter.createTuple(one, Double.toString(two));
143        }
144
145        /**
146         * Convenience method to create XML Attributes written by startTag()
147         */
148        protected static Tuple<String, String> createTuple(String one, boolean two) {
149                return MatsimXmlWriter.createTuple(one, Boolean.toString(two));
150        }
151
152        /**
153         * Convenience method to create XML Attributes written by startTag()
154         */
155        protected static Tuple<String, String> createTimeTuple(String one, double sec) {
156                return MatsimXmlWriter.createTuple(one, Time.writeTime(sec));
157        }
158
159
160        private void indent() throws IOException{
161                for (int i = 0; i < this.indentationLevel; i++) {
162                        this.writer.write(this.indentationString);
163                }
164        }
165        /**
166         * Writes a start tag with all attributes on the writer
167         * @param tagname
168         * @param attributes
169         * @throws UncheckedIOException
170         */
171        protected final void writeStartTag(String tagname, List<Tuple<String, String>> attributes) throws UncheckedIOException{
172                this.writeStartTag(tagname, attributes, false);
173        }
174
175        protected final void writeStartTag(String tagname, List<Tuple<String, String>> attributes, boolean closeElement) throws UncheckedIOException {
176                try {
177                if (doPrettyPrint) {
178                        this.writer.write(NL);
179                        indent();
180                        this.indentationLevel++;
181                }
182                        this.writer.write("<" + tagname);
183                        if (attributes != null) {
184                                for (Tuple<String, String> t : attributes){
185                                        this.writer.write(" " + t.getFirst() + "=\"" + encodeAttributeValue(t.getSecond()) + "\"");
186                                }
187                        }
188                        if (closeElement) {
189                                this.writer.write("/>");
190                                if (doPrettyPrint) {
191                                        this.indentationLevel--;
192                                }
193                        }
194                        else {
195                                this.writer.write(">");
196                        }
197                } catch (IOException e) {
198                        throw new UncheckedIOException(e);
199                }
200        }
201
202
203        protected final void writeContent(String content, boolean allowWhitespaces) throws UncheckedIOException{
204                try {
205                        if (doPrettyPrint) {
206                                this.noWhitespaces = !allowWhitespaces;
207                                if (!this.noWhitespaces) {
208                                        this.writer.write(NL);
209                                        this.indentationLevel++;
210                                        indent();
211                                }
212                        }
213                        writer.write(encodeContent(content));
214                        if (!this.noWhitespaces) {
215                                this.indentationLevel--;
216                        }
217                } catch (IOException e) {
218                        throw new UncheckedIOException(e);
219                }
220        }
221
222        /**
223         * Writes a XML end tag with the given name to the given writer instance
224         * @param tagname
225         * @throws UncheckedIOException
226         */
227        protected final void writeEndTag(String tagname) throws UncheckedIOException {
228                try {
229                        if (doPrettyPrint) {
230                                this.indentationLevel--;
231                                if (!this.noWhitespaces) {
232                                        this.writer.write(NL);
233                                        indent();
234                                }
235                                else {
236                                        this.noWhitespaces = false;
237                                }
238                        }
239                        this.writer.write("</" + tagname + ">");
240                } catch (IOException e) {
241                        throw new UncheckedIOException(e);
242                }
243        }
244
245        /**
246         * Encodes the given string in such a way that it no longer contains
247         * characters that have a special meaning in xml.
248         * 
249         * @see <a href="http://www.w3.org/International/questions/qa-escapes#use">http://www.w3.org/International/questions/qa-escapes#use</a>
250         * @param attributeValue
251         * @return String with some characters replaced by their xml-encoding.
252         */
253        protected static String encodeAttributeValue(final String attributeValue) {
254                if (attributeValue.contains("&") || attributeValue.contains("\"") || attributeValue.contains("<") || attributeValue.contains(">")) {
255                        return attributeValue.replace("&", "&amp;").replace("\"", "&quot;").replace("<", "&lt;").replace(">", "&gt;");
256                }
257                return attributeValue;
258        }
259
260        protected static String encodeContent(final String content) {
261                if (content.contains("&") || content.contains("<") || content.contains(">")) {
262                        return content.replace("&", "&amp;").replace("<", "&lt;").replace(">", "&gt;");
263                }
264                return content;
265        }
