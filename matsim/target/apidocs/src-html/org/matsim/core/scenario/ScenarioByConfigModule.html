001package org.matsim.core.scenario;
002
003
004import com.google.inject.Provides;
005import com.google.inject.Singleton;
006import org.matsim.api.core.v01.Scenario;
007import org.matsim.core.controler.AbstractModule;
008
009public final class ScenarioByConfigModule extends AbstractModule {
010        @Override
011        public void install() {
012                bind( ScenarioLoaderImpl.class ).toInstance( new ScenarioLoaderImpl( getConfig() ) );
013                install( new ScenarioByInstanceModule( null ) );
014        }
015
016        @Provides
017        @Singleton
018        private Scenario createScenario( final ScenarioLoaderImpl loader ) {
019                return loader.loadScenario();
020        }
021}
