045/**
046 * @author dgrether
047 * @author mrieser
048 */
049public final class MutableScenario implements Scenario, Lockable {
050        @SuppressWarnings("unused")
051        private static final Logger log = Logger.getLogger(MutableScenario.class);
052
053        private boolean locked = false ;
054
055        private final Map<String, Object> elements = new HashMap<>();
056
057        //mandatory attributes
058        private final Config config;
059        private Network network;
060        private Population population;
061        private ActivityFacilities facilities;
062
063        //non-mandatory attributes
064        private TransitSchedule transitSchedule = null;
065        private Lanes lanes = null;
066        private Households households;
067        private Vehicles transitVehicles;
068
069        private Vehicles vehicles ;
070
071        MutableScenario(Config config) {
072                this.config = config;
073                this.network = NetworkUtils.createNetwork(this.config);
074                this.population = PopulationUtils.createPopulation(this.config, this.network);
075                this.facilities = new ActivityFacilitiesImpl();
076                this.households = new HouseholdsImpl();
077                this.lanes = LanesUtils.createLanesContainer();
078                this.vehicles = VehicleUtils.createVehiclesContainer();
079                this.transitVehicles = VehicleUtils.createVehiclesContainer();
080                this.transitSchedule = new TransitScheduleFactoryImpl().createTransitSchedule();
081                
082                this.config.network().setLocked();
083        }
084
085        @Override
086        public final ActivityFacilities getActivityFacilities() {
087                return this.facilities;
088        }
089
090        @Override
091        public final Network getNetwork() {
092                return this.network;
093        }
094
095        @Override
096        public final Population getPopulation() {
097                return this.population;
098        }
099
100        @Override
101        public final Config getConfig() {
102                // yy should throw an exception if null. kai, based on https://matsim.atlassian.net/browse/MATSIM-301 , may'15
103                return this.config;
104        }
105
106        public final void setNetwork(Network network) {
107                testForLocked();
108                this.network = network;
109        }
110
111        public final void setPopulation(Population population) {
112                testForLocked();
113                this.population = population;
114        }
115
116        // NOTE: Thibaut is not so happy about just returning null, since someone may code something like
117        // Vehicles vehicles = scenario.getVehicles() ; // returning null
118        // ... // lots of code
119        // ... vehicles.getVehicles() ... // throwing a null pointer exception
120        // That is, in such a situation the null pointer exception is thrown much later than when the null pointer was obtained.
121        // He rather wants to have this fail when it is called.
122        // I (kn) am a bit sceptic if it makes sense to establish such a convention when most other people use "if ... == null" use as the
123        // official dialect to check if something is there (since this would cause an exception with Thibaut's approach). kai, feb'15
124
125        @Override
126        public final Households getHouseholds() {
127                return this.households;
128        }
129
130        @Override
131        public Lanes getLanes() {
132                return this.lanes;
133        }
134
135        @Override
136        public final Vehicles getTransitVehicles() {
137                return this.transitVehicles;
138        }
139
140        @Override
141        final public Vehicles getVehicles() {
142                return this.vehicles;
143        }
144
145
146
147        @Override
148        public final TransitSchedule getTransitSchedule() {
149                return this.transitSchedule;
150        }
151
152        @Override
153        public final void addScenarioElement(
154                        final String name,
155                        final Object o) {
156                // Once the "removal" operation is locked, you cannot add under the same name. kai, sep'14
157                if ( o == null ) throw new NullPointerException( name );
158                final Object former = elements.put( name , o );
159                if ( former != null ) {
160                        throw new IllegalStateException( former+" is already associated with name "+name+" when adding "+o );
161                }
162        }
163
164        public final Object removeScenarioElement(final String name) {
165                testForLocked();
166                return elements.remove( name );
167        }
168
169        @Override
170        public final Object getScenarioElement(final String name) {
171                // yy should throw an exception if null. kai, based on https://matsim.atlassian.net/browse/MATSIM-301 , may'15
172                return elements.get( name );
173        }
174
175        @Override
176        public final void setLocked() {
177                this.locked = true ;
178        }
179
180        private void testForLocked() {
181                if ( locked ) {
182                        throw new RuntimeException( "Scenario is locked; too late to do this.  See comments in code.") ;
183                        /* The decision is roughly as follows:
184                         * - It is ok to set network, population, etc. in the Scenario during initial demand generation.
185                         * - It is NOT ok to do this once the services is running.
186                         * - But we do not want to make a defensive copy of the whole thing at services startup.
187                         * - We also want to be able to plug alternative Scenario implementations into the services.
188                         * - But then the services only gets the "Scenario" not the "MutableScenario", so it does not have to worry about setNetwork and the like
189                         * since it does not exist in the published interface.
190                         * kai, sep'14
191                         */
192                }
193        }
194
195        public final void setActivityFacilities( ActivityFacilities facilities ) {
196                testForLocked() ;
197                this.facilities = facilities ;
198        }
199        public final void setHouseholds( Households households ) {
200                testForLocked() ;
201                this.households = households ;
202        }
203        public final void setTransitSchedule( TransitSchedule schedule ) {
204                testForLocked();
205                this.transitSchedule = schedule ;
206        }
207        public final void setTransitVehicles( Vehicles vehicles ) {
208                testForLocked();
209                this.transitVehicles = vehicles ;
210        }
211        public final void setLanes( Lanes lanes ) {
212                testForLocked();
213                this.lanes = lanes ;
214        }
