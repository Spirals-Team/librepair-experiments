021package org.matsim.core.replanning.modules;
022
023import org.matsim.core.config.groups.ChangeModeConfigGroup;
024import org.matsim.core.config.groups.GlobalConfigGroup;
025import org.matsim.core.gbl.MatsimRandom;
026import org.matsim.core.population.algorithms.ChooseRandomLegMode;
027import org.matsim.core.population.algorithms.PlanAlgorithm;
028
029/**
030 * Changes the transportation mode of all legs in a plan to a randomly chosen
031 * different mode (but the same mode for all legs in that plan) given a list
032 * of possible modes.
033 *
034 * By default, the module chooses between "car" and "pt". If other modes should
035 * be used, it can be done so in the configuration. Also, this module is able to (optionally)
036 * respect car-availability:
037 * <pre>
038 * &lt;module name="changeLegMode">
039 *   &lt!-- provide a comma-separated list of leg modes -->
040 *   &lt;param name="modes" value="car,walk,bike" />
041 *   &lt;param name="ignoreCarAvailability" value="false" />
042 * &lt;/module>
043 * </pre>
044 *
045 * @see ChooseRandomLegMode
046 *
047 * @author mrieser
048 */
049public class ChangeLegMode extends AbstractMultithreadedModule {
050
051        // (I made the above static final variables public so they can be used in scripts-in-java. kai, jun'15)
052
053        private String[] availableModes;
054        private boolean ignoreCarAvailability;
055        private boolean allowSwitchFromListedModesOnly;
056
057
058        public ChangeLegMode(final GlobalConfigGroup globalConfigGroup, ChangeModeConfigGroup changeLegModeConfigGroup) {
059                super(globalConfigGroup.getNumberOfThreads());
060                this.availableModes = changeLegModeConfigGroup.getModes();
061                this.ignoreCarAvailability = changeLegModeConfigGroup.getIgnoreCarAvailability();
062                if (changeLegModeConfigGroup.getBehavior().equals(ChangeModeConfigGroup.Behavior.fromSpecifiedModesToSpecifiedModes)) {
063                        this.allowSwitchFromListedModesOnly = true;
064                } else this.allowSwitchFromListedModesOnly=false;
065        }
066        
067        public ChangeLegMode(final int nOfThreads, final String[] modes, final boolean ignoreCarAvailabilty, boolean allowSwitchFromListedModesOnly) {
068                super(nOfThreads);
069                this.availableModes = modes.clone();
070                this.ignoreCarAvailability = ignoreCarAvailabilty;
071        }
072
073        @Override
074        public PlanAlgorithm getPlanAlgoInstance() {
075                ChooseRandomLegMode algo = new ChooseRandomLegMode(this.availableModes, MatsimRandom.getLocalInstance(), false);
076                algo.setIgnoreCarAvailability(this.ignoreCarAvailability);
077                return algo;
078        }
079
080}
