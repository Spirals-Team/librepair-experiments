041/**
042 * select an existing Plan according to the Path Size Logit (e.g. Frejinger, E. and Bierlaire, M.: Capturing Correlation
043 * with subnetworks in route choice models, Transportation Research Part B (2006), doi:10.1016/j.trb.2006.06.003.)
044 * If there are unscored plans one of it will be chosen randomly (optimistic strategy).
045 *
046 * @author laemmel
047 */
048public final class PathSizeLogitSelector extends AbstractPlanSelector {
049
050        private final double pathSizeLogitExponent;
051        private final double logitScaleFactor;
052        private Network network;
053
054        public PathSizeLogitSelector( final double pathSizeLogitExponent, final double logitScaleFactor, final Network network ) {
055                this.pathSizeLogitExponent = pathSizeLogitExponent ;
056                this.logitScaleFactor = logitScaleFactor ;
057                this.network = network ;
058        }
059
060        public PathSizeLogitSelector(final PlanCalcScoreConfigGroup config, final Network network) {
061                this( config.getPathSizeLogitBeta(), config.getBrainExpBeta(), network ) ;
062        }
063
064        //updates the path size logit weights
065        @Override
066        protected Map<Plan,Double> calcWeights(final List<? extends Plan> plans ) {
067                // ("plans" is the list of plans of a single person)
068                
069                Map<Plan,Double> weights = new HashMap<Plan,Double>() ;
070
071                double maxScore = Double.NEGATIVE_INFINITY;
072
073                HashMap<Id<Link>, ArrayList<Double>> linksInTime = new HashMap<>();
074                // (a data structure that memorizes possible leg start times for link utilization (??))
075                
076                HashMap<Integer,Double> planLength = new HashMap<Integer, Double>();
077                // (a data structure that memorizes the total travel distance of each plan)
078                // (yyyy is it obvious that no two plans can have the same hash code?  what happens if they do?  kai, oct'12)
079                // (why not just <Plan,Double>?????)
080                
081                //this gets the choice sets C_n
082                //TODO [GL] since the lack of information in Route(),
083                //the very first and the very last link of a path will be ignored - gl
084                //dg, 09-2013: as first and last link are equal for all routes between to activities this is no major issue
085
086                for (Plan plan : plans) {
087
088                        if (plan.getScore() > maxScore) maxScore = plan.getScore();
089
090                        double pathSize = 0;
091                        double currentEndTime = 0.0;
092                        for (PlanElement pe : plan.getPlanElements()) {
093                                if (pe instanceof Leg) {
094                                        Leg leg = (Leg) pe;
095                                        currentEndTime = leg.getDepartureTime();
096
097                                        NetworkRoute r = (NetworkRoute) leg.getRoute();
098                                        // (yyyy this will fail when the route is not a network route.  kai, oct'12)
099
100                                        pathSize += RouteUtils.calcDistanceExcludingStartEndLink(r, network);
101                                        // (i.e. pathSize will be the sum over all routes of the plan)
102                                        
103                                        for (Id<Link> linkId : r.getLinkIds()){
104                                                ArrayList<Double> lit = linksInTime.get(linkId);
105                                                if (lit == null){
106                                                        lit = new ArrayList<Double>();
107                                                        linksInTime.put(linkId, lit);
108                                                }
109                                                lit.add(currentEndTime);
110                                        }
111                                }
112                        }
113                        planLength.put(plan.hashCode(), pathSize);
114                }
115
116                for (Plan plan : plans) {
117
118                        double tmp = 0;
119                        for (PlanElement pe : plan.getPlanElements()) {
120                                if (pe instanceof Leg) {
121                                        Leg leg = (Leg) pe;
122                                        double currentTime = leg.getDepartureTime();
123                                        NetworkRoute route = (NetworkRoute) leg.getRoute();
124                                        for (Id<Link> linkId : route.getLinkIds()){
125                                                double denominator = 0;
126                                                for (double dbl : linksInTime.get(linkId)){
127                                                        //TODO this is just for testing (those legs where the departure time differs more then 3600 seconds will not compared to each other) - need a
128                                                        //little bit to brood on it - gl
129                                                        // An alternative might be to use a kernal, e.g. a Gaussian.  Something like
130                                                        // denominator += exp( (dbl-currentTime)^2 / sigma^2 ) .  kai, oct'12
131                                                        if (Math.abs(dbl - currentTime) <= 3600)
132                                                                denominator++;
133                                                }
134                                                // (the meaning seems to be: for each link that the plan uses, it checks how many other times the
135                                                // same link is used by a leg that has roughly the same departure time (*))
136
137                                                Link link = network.getLinks().get(linkId);
138                                                tmp += link.getLength() / denominator;
139                                                // (for a plan, the weight of a link is divided by the number of times it is used)
140                                        }
141                                }
142                        }
143                        // tmp is now a number that contains the ``reduced'' travel distance of the plan.  Divide it by the full travel distance
144                        // of the plan, and take to the power of this.beta:
145                        double PSi = Math.pow(tmp/planLength.get(plan.hashCode()), this.pathSizeLogitExponent);
146                        
147                        double weight;
148                        if (Double.isInfinite(maxScore)) {
149                                // (isInfinite(x) also returns true when x==-Infinity) 
150                                
151                                // likely that wc.maxScore == -Infinity, and thus plan.getScoreAsPrimitiveType() also == -Infinity, handle it like any other case where getScore() == maxScore
152                                // I do not understand the line above.  kai, oct'12
153                                
154                                weight = PSi;
155                                // (yy I do not understand.  Presumably, wc.maxScore may be -Infinity, in which case ALL plan scores are -Infinity 
156                                // (or NaN or null or something similar).  In this case, plans are simply weighted by their PSi, so that 
157                                // overlapping plans get less weight than very different plans. kai, oct'12)
158                        } else {
159                                weight = Math.exp(this.logitScaleFactor * (plan.getScore() - maxScore))*PSi;
160                                // (this is essentially $PSi * exp( tau * score )$, the "-wc.maxScore" is (presumably) the computational trick
161                                // to avoid overflow)
162                        }
163
164                        if (weight <= 0.0) weight = 0;
165                        // (yy how can weight become negative?? kai, oct'12) 
166
167                        // the weight is memorized; the sum of all weights in computed.  Choice will be based on those weights
168                        weights.put( plan, weight) ;
169                }
170                
171                return weights ;
172        }
