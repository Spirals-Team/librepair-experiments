021package org.matsim.core.replanning.selectors;
022
023import java.util.Map;
024import java.util.concurrent.ConcurrentHashMap;
025
026import org.matsim.api.core.v01.population.HasPlansAndId;
027import org.matsim.api.core.v01.population.Person;
028import org.matsim.api.core.v01.population.Plan;
029
030/**
031 * <p>Selects the worst plan of a person (most likely for removal), but respects
032 * the set plan types in a way the no plan is selected that is the last one of
033 * its type.</p> 
034 * <p>(I would say that it can select the last of its type if it finds nothing else.  However, this algo should only
035 * be used if an agent has more plans than maxPlansPerAgent, so make sure that that parameter is set large enough for
036 * your purposes.  kai, oct'09)</p>
037 * <p>Plans without a score are seen as worst and selected accordingly.</p>
038 *
039 * @author mrieser
040 */
041public class WorstPlanForRemovalSelector implements PlanSelector<Plan, Person> {
042
043        private static final String UNDEFINED_TYPE = "undefined";
044
045        @Override
046        public Plan selectPlan(HasPlansAndId<Plan, Person> person) {
047
048                // hashmap that returns "Integer" count for given plans type:
049                Map<String, Integer> typeCounts = new ConcurrentHashMap<String, Integer>();
050
051                // count how many plans per type an agent has:
052                for (Plan plan : person.getPlans()) {
053                        String type = plan.getType();
054                        if ( type==null ) {
055                                type = UNDEFINED_TYPE ;
056                        }
057                        Integer cnt = typeCounts.get(type);
058                        if (cnt == null) {
059                                typeCounts.put(type, Integer.valueOf(1));
060                        } else {
061                                typeCounts.put(type, Integer.valueOf(cnt.intValue() + 1));
062                        }
063                }
064
065                Plan worst = null;
066                double worstScore = Double.POSITIVE_INFINITY;
067                for (Plan plan : person.getPlans()) {
068
069                        String type = plan.getType();
070                        if ( type==null ) {
071                                type = UNDEFINED_TYPE;
072                        }
073                        if (typeCounts.get(type).intValue() > 1) {
074                                // (if we have more than one plan of the same type:)
075
076                                // if this plan has no score yet:
077                                if (plan.getScore() == null || plan.getScore().isNaN() ) {
078                                        // say that the plan without score now is the "worst":
079                                        worst = plan;
080
081                                        // make sure that this one remains the selected plan:
082                                        worstScore = Double.NEGATIVE_INFINITY;
083
084                                // otherwise do the usual logic to find the plan with the minimum score:
085                                } else if (plan.getScore().doubleValue() < worstScore) {
086                                        worst = plan;
087                                        worstScore = plan.getScore().doubleValue();
088                                }
089                        }
090                        // (otherwise we just keep "worst=null")
091                        
092                }
093
094                if (worst == null) {
095                        // there is exactly one plan, or we have of each plan-type exactly one.
096                        // select the one with worst score globally, or the first one with score=null
097                        for (Plan plan : person.getPlans()) {
098                                if (plan.getScore() == null || plan.getScore().isNaN() ) {
099                                        return plan;
100                                }
101                                if (plan.getScore().doubleValue() < worstScore) {
102                                        worst = plan;
103                                        worstScore = plan.getScore().doubleValue();
104                                }
105                        }
106                }
107                return worst;
108        }
109
110}
