021package org.matsim.core.replanning.modules;
022
023import org.apache.log4j.Logger;
024import org.matsim.api.core.v01.population.Plan;
025import org.matsim.api.core.v01.replanning.PlanStrategyModule;
026import org.matsim.core.config.groups.GlobalConfigGroup;
027import org.matsim.core.population.algorithms.PlanAlgorithm;
028import org.matsim.core.replanning.ReplanningContext;
029import org.matsim.core.utils.misc.Counter;
030
031import java.lang.Thread.UncaughtExceptionHandler;
032import java.util.LinkedList;
033import java.util.List;
034import java.util.concurrent.atomic.AtomicReference;
035
036/**
037 * An abstract strategy module for running multiple plan algorithms in parallel.
038 * Can be used to easily generate multi-threaded strategy modules, as long as the
039 * real functionality of the strategy module is handled by a PlanAlgorithm.
040 * Just overwrite getPlanAlgoInstance() to return an instance of your plan
041 * algorithm.
042 * <p></p>
043 * <code>initThreads()</code> creates the threads, but does not yet start them.
044 * <p></p>
045 * <code>handlePlan(Plan)</code> distributes the plans equally to all threads.
046 * <p></p>
047 * <code>finishReplanning()</code> finally starts the threads and waits for all threads to be finished.
048 * <p></p>
049 * While this approach does not lead to optimal performance gains ("slow threads" vs.
050 * "fast threads"), it helps building reproducible runs.  Additionally, as the threads are only
051 * started after all to-be-handled plans are added, we can use unsynchronized data structures.
052 * <p></p>
053 * Design comments/questions:<ul>
054 * <li> As a consequence of the design, the instances that getPlanAlgoInstance() returns, need to be thread-safe.  kai, dec'12
055 * For an example with discussions, see {@link tutorial.programming.multiThreadedPlanStrategy.RunWithMultithreadedModule}
056 * </ul>
057 *
058 * @author mrieser
059 */
060abstract public class AbstractMultithreadedModule implements PlanStrategyModule {
061        private final int numOfThreads;
062
063        private PlanAlgoThread[] algothreads = null;
064        private Thread[] threads = null;
065        private PlanAlgorithm directAlgo = null;
066        private String name = null;
067
068        private int count = 0;
069
070        private final AtomicReference<Throwable> hadException = new AtomicReference<>(null);
071        private final ExceptionHandler exceptionHandler = new ExceptionHandler(this.hadException);
072
073        private ReplanningContext replanningContext;
074
075        static final private Logger log = Logger.getLogger(AbstractMultithreadedModule.class);
076
077        /**
078         * Design comments:<ul>
079         * <li> The way I understand this, the instances that this method returns need to be thread-safe (i.e. independent from each other).  They can,
080         * for example, not rely on the same instance of the router.  kai, dec'12
081         * </ul>
082         */
083        abstract public PlanAlgorithm getPlanAlgoInstance();
084
085        public AbstractMultithreadedModule(GlobalConfigGroup globalConfigGroup) {
086                this.numOfThreads = globalConfigGroup.getNumberOfThreads();
087        }
088
089        public AbstractMultithreadedModule(final int numOfThreads) {
090                this.numOfThreads = numOfThreads;
091        }
092        
093        protected void beforePrepareReplanningHook(@SuppressWarnings("unused") ReplanningContext replanningContextTmp) {
094                // left empty for inheritance
095        }
096
097        protected void afterPrepareReplanningHook(@SuppressWarnings("unused") ReplanningContext replanningContextTmp) {
098                // left empty for inheritance
099        }
100
101        @Override
102        public final void prepareReplanning(ReplanningContext replanningContextTmp) {
103                this.beforePrepareReplanningHook(replanningContextTmp);
104                this.replanningContext = replanningContextTmp;
105                if (this.numOfThreads == 0) {
106                        // it seems, no threads are desired :(
107                        this.directAlgo = getPlanAlgoInstance();
108                } else {
109                        initThreads();
110                }
111                this.afterPrepareReplanningHook(replanningContextTmp);
112        }
113
114        protected final ReplanningContext getReplanningContext() {
115                return replanningContext;
116        }
117
118        @Override
119        public final void handlePlan(final Plan plan) {
120                if (this.directAlgo == null) {
121                        this.algothreads[this.count % this.numOfThreads].addPlanToThread(plan);
122                        this.count++;
123                } else {
124                        this.directAlgo.run(plan);
125                }
126        }
127
128        protected void beforeFinishReplanningHook() {
129                // left empty for inheritance
130        }
131        protected void afterFinishReplanningHook() {
132                // left empty for inheritance
133        }
134        
135        @Override
136        public final void finishReplanning() {
137                this.beforeFinishReplanningHook();
138                
139                if (this.directAlgo == null) {
140                        // only try to start threads if we did not directly work on all the plans
141                        log.info("[" + this.name + "] starting " + this.threads.length + " threads, handling " + this.count + " plans");
142
143                        // start threads
144                        for (Thread thread : this.threads) {
145                                thread.start();
146                        }
147
148                        // wait until each thread is finished
149                        try {
150                                for (Thread thread : this.threads) {
151                                        thread.join();
152                                }
153                        } catch (InterruptedException e) {
154                                throw new RuntimeException(e);
155                        }
156                        log.info("[" + this.name + "] all " + this.threads.length + " threads finished.");
157                        Throwable throwable = this.hadException.get();
158                        if (throwable != null) {
159                                throw new RuntimeException("Some threads crashed, thus not all plans may have been handled.", throwable);
160                        }
161                }
162                // reset
163                this.algothreads = null;
164                this.threads = null;
165                this.replanningContext = null;
166                this.count = 0;
167                
168                this.afterFinishReplanningHook();
169        }
170
171        private void initThreads() {
172                if (this.threads != null) {
173                        throw new RuntimeException("threads are already initialized");
174                }
175
176                this.hadException.set(null);
177                this.threads = new Thread[this.numOfThreads];
178                this.algothreads = new PlanAlgoThread[this.numOfThreads];
179
180                Counter counter = null;
181                // setup threads
182                for (int i = 0; i < this.numOfThreads; i++) {
183                        PlanAlgorithm algo = getPlanAlgoInstance();
184                        if (i == 0) {
185                                this.name = algo.getClass().getSimpleName();
186                                counter = new Counter("[" + this.name + "] handled plan # ");
187                        }
188                        PlanAlgoThread algothread = new PlanAlgoThread(algo, counter);
189                        Thread thread = new Thread(algothread, this.name + "." + i);
190                        thread.setUncaughtExceptionHandler(this.exceptionHandler);
191                        this.threads[i] = thread;
192                        this.algothreads[i] = algothread;
193                }
194        }
195
196        /* package (for a test) */ final int getNumOfThreads() {
197                return numOfThreads;
198        }
199
200        private final static class ExceptionHandler implements UncaughtExceptionHandler {
201
202                private final AtomicReference<Throwable> hadException;
203
204                public ExceptionHandler(final AtomicReference<Throwable> hadException) {
205                        this.hadException = hadException;
206                }
207
208                @Override
209                public void uncaughtException(Thread t, Throwable e) {
210                        log.error("Thread " + t.getName() + " died with exception. Will stop after all threads finished.", e);
211                        this.hadException.set(e);
212                }
213
214        }
215
216        private final static class PlanAlgoThread implements Runnable {
217
218                private final PlanAlgorithm planAlgo;
219                private final List<Plan> plans = new LinkedList<>();
220                private final Counter counter;
221
222                public PlanAlgoThread(final PlanAlgorithm algo, final Counter counter) {
223                        this.planAlgo = algo;
224                        this.counter = counter;
225                }
226
227                public void addPlanToThread(final Plan plan) {
228                        this.plans.add(plan);
229                }
230
231                @Override
232                public void run() {
233                        for (Plan plan : this.plans) {
234                                this.planAlgo.run(plan);
235                                this.counter.incCounter();
236                        }
237                }
238        }
239}
