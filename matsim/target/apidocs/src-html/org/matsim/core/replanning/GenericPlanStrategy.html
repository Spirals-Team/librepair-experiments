019package org.matsim.core.replanning;
020
021import org.matsim.api.core.v01.population.BasicPlan;
022import org.matsim.api.core.v01.population.HasPlansAndId;
023
024/**
025 * Notes:<ul>
026 * <li> If an implementation of this interface is "innovative", i.e. modifies plans, then it should first copy that plan, add the new plan to the choice set,
027 * and then modify that new copy.  Otherwise, the evolutionary functionality of MATSim will probably be destroyed. kai, jan'15
028 * </ul>
029 * 
030 * @author (of the generic version) Kai Nagel
031 *
032 */
033public interface GenericPlanStrategy<T extends BasicPlan, I> {
034        /**
035         * Adds a person to this strategy to be handled. It is not required that
036         * the person is immediately handled during this method-call (e.g. when using
037         * multi-threaded strategy-modules).  This method ensures that an unscored
038         * plan is selected if the person has such a plan ("optimistic behavior").
039         *
040         * @param person
041         * @see #finish()
042         */
043        public void run(final HasPlansAndId<T, I> person);
044
045        /**
046         * Tells this strategy to initialize its modules. Called before a bunch of
047         * person are handed to this strategy.
048         * @param replanningContext TODO
049         */
050        public void init(ReplanningContext replanningContext);
051
052        /**
053         * Indicates that no additional persons will be handed to this module and
054         * waits until this strategy has finished handling all persons.
055         * @see #run(HasPlansAndId)
056         */
057        public void finish();
058
059
060}
