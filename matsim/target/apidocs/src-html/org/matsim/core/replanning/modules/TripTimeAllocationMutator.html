021package org.matsim.core.replanning.modules;
022
023import com.google.inject.Inject;
024import org.matsim.core.config.Config;
025import org.matsim.core.config.groups.PlansConfigGroup;
026import org.matsim.core.config.groups.PlansConfigGroup.ActivityDurationInterpretation;
027import org.matsim.core.gbl.MatsimRandom;
028import org.matsim.core.population.algorithms.PlanAlgorithm;
029import org.matsim.core.population.algorithms.TripPlanMutateTimeAllocation;
030import org.matsim.core.router.TripRouter;
031
032import javax.inject.Provider;
033
034/**
035 * Copy/Paste of TimeAllocationMutator, that calls TransitPlanMutateTimeAllocation instead
036 * of PlanMutateTimeAllocation.
037 *
038 * @author mrieser
039 */
040public class TripTimeAllocationMutator extends AbstractMultithreadedModule {
041
042        private final Provider<TripRouter> tripRouterProvider;
043
044        public final static String CONFIG_GROUP = "TimeAllocationMutator";
045        public final static String CONFIG_MUTATION_RANGE = "mutationRange";
046
047        private double mutationRange = 1800.0;
048        private boolean useActivityDurations = true;
049        private final boolean affectingDuration;
050
051        /**
052         * Creates a new TimeAllocationMutator with a mutation range as defined in
053         * the configuration (module "TimeAllocationMutator", param "mutationRange").
054         */
055        public TripTimeAllocationMutator(Config config, Provider<TripRouter> tripRouterProvider) {
056                super(config.global());
057                this.tripRouterProvider = tripRouterProvider;
058                this.mutationRange = config.timeAllocationMutator().getMutationRange() ;
059                this.affectingDuration = config.timeAllocationMutator().isAffectingDuration() ;
060                PlansConfigGroup.ActivityDurationInterpretation actDurInterpr = ( config.plans().getActivityDurationInterpretation() ) ;
061                if ( actDurInterpr == PlansConfigGroup.ActivityDurationInterpretation.minOfDurationAndEndTime ) {
062                        useActivityDurations = true ;
063                } else if ( actDurInterpr == PlansConfigGroup.ActivityDurationInterpretation.endTimeOnly ) {
064                        useActivityDurations = false ;
065                } else if ( actDurInterpr == PlansConfigGroup.ActivityDurationInterpretation.tryEndTimeThenDuration ) {
066                        throw new UnsupportedOperationException( "need to clarify the correct setting here.  Probably not a big deal, but not done yet.  kai, aug'10") ;
067                } else {
068                        throw new IllegalStateException( "beahvior not defined for this configuration setting") ;
069                }
070        }
071
072        public TripTimeAllocationMutator(Config config, Provider<TripRouter> tripRouterProvider, final double mutationRange, boolean affectingDuration) {
073                super(config.global());
074                this.tripRouterProvider = tripRouterProvider;
075                this.mutationRange = mutationRange;
076                this.affectingDuration = affectingDuration;
077        }
078
079        @Override
080        public PlanAlgorithm getPlanAlgoInstance() {
081                TripPlanMutateTimeAllocation pmta =
082                        new TripPlanMutateTimeAllocation(
083                                        tripRouterProvider.get().getStageActivityTypes(),
084                                        this.mutationRange,
085                                        affectingDuration, MatsimRandom.getLocalInstance());
086                pmta.setUseActivityDurations(this.useActivityDurations);
087                return pmta;
088        }
089
090}
