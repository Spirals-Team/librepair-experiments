047/**
048 * Basic wrapper for any call to external "plans-to-plans" modules. As basic handling of
049 * such modules is alike for every module:
050 * 1.) Write a plans header
051 * 2.) dump every person with the selected plan only
052 * 3.) close plans file and write a config file based on a config template file
053 * 4.) Execute the external program with this config file
054 * 5.) Mutate original plans according to re-read plans
055 *
056 * @author dstrippgen
057 * @author mrieser
058 * @author michaz
059 */
060public class ExternalModule implements PlanStrategyModule {
061
062    interface ExeRunnerDelegate {
063        public boolean invoke();
064    }
065
066        private static final String SCENARIO = "scenario";
067        private static final String SCENARIO_INPUT_PLANS_FILENAME = "inputPlansFilename";
068        private static final String SCENARIO_WORKING_PLANS_FILENAME = "workingPlansFilename";
069        private static final String SCENARIO_WORKING_EVENTS_TXT_FILENAME = "workingEventsTxtFilename";
070        private static final String SCENARIO_NETWORK_FILENAME = "networkFilename";
071
072        private static final String ExternalInFileName = "plans.in.xml";
073        private static final String ExternalOutFileName = "plans.out.xml";
074        private static final String ExternalConfigFileName = "config.xml";
075
076        private final Scenario scenario;
077    private final ExeRunnerDelegate exeRunnerDelegate;
078    private String modulePrefix = "";
079        private String outFileRoot = "";
080
081        private final OutputDirectoryHierarchy controler;
082        
083        private int currentIteration = -1;
084        
085        private Population exportPopulation;
086        
087        private Map<Id<Person>, Plan> plansToMutate = new HashMap<>();
088
089        public ExternalModule(final String exePath, final String moduleId, final OutputDirectoryHierarchy controler, final Scenario scenario) {
090        this.modulePrefix = moduleId + "_";
091                this.controler = controler;
092                this.outFileRoot = controler.getTempPath() + "/";
093                this.scenario = scenario;
094        this.exeRunnerDelegate = new ExeRunnerDelegate() {
095            @Override
096            public boolean invoke() {
097                String cmd = exePath + " " + outFileRoot + modulePrefix + ExternalConfigFileName;
098                String logfilename = controler.getIterationFilename(currentIteration, modulePrefix + "stdout.log");
099                return (ExeRunner.run(cmd, logfilename, 3600) == 0);
100            }
101        };
102        }
103
104    ExternalModule(ExeRunnerDelegate exeRunnerDelegate, final String moduleId, final OutputDirectoryHierarchy controler, final Scenario scenario) {
105        this.modulePrefix = moduleId + "_";
106        this.controler = controler;
107        this.outFileRoot = controler.getTempPath() + "/";
108        this.scenario = scenario;
109        this.exeRunnerDelegate = exeRunnerDelegate;
110    }
111
112        @Override
113        public void prepareReplanning(ReplanningContext replanningContext) {
114                this.currentIteration = replanningContext.getIteration();
115                this.exportPopulation = ScenarioUtils.createScenario(ConfigUtils.createConfig()).getPopulation();
116        }
117
118        @Override
119        public void handlePlan(final Plan plan) {
120                // Creating a dummy population which only contains the plans which are passed here.
121                // I need to copy the plans because I am not supposed to add a plan to a different Person.
122                // I also need to memorize the plans which are passed here, because I am supposed to mutate them.
123                
124                final Person personWithOnlySelectedPlan = this.exportPopulation.getFactory().createPerson(plan.getPerson().getId());
125                final Plan planForNewPerson = PopulationUtils.createPlan(personWithOnlySelectedPlan);
126                PopulationUtils.copyFromTo(plan, planForNewPerson);
127                personWithOnlySelectedPlan.addPlan(planForNewPerson);
128                this.exportPopulation.addPerson(personWithOnlySelectedPlan);
129                this.plansToMutate.put(plan.getPerson().getId(), plan);
130        }
131
132        @Override
133        public void finishReplanning() {
134                exportPopulation();
135        prepareExternalExeConfig();
136        boolean successful = this.exeRunnerDelegate.invoke();
137        if (successful) {
138                        importPopulationAndMutatePlans();
139                } else {
140                        throw new RuntimeException("External Replanning exited with error.");
141                }
142        }
143
144    private void exportPopulation() {
145                String filename = this.outFileRoot + this.modulePrefix + ExternalInFileName;
146                PopulationWriter plansWriter = new PopulationWriter(exportPopulation, scenario.getNetwork());
147                plansWriter.write(filename);
148        }
149
150    private void prepareExternalExeConfig() {
151                Config extConfig;
152                String configFileName = this.scenario.getConfig().strategy().getExternalExeConfigTemplate();
153                if (configFileName == null) {
154                        extConfig = new Config();
155                } else {
156                        extConfig = new Config();
157                        ConfigReader reader = new ConfigReader(extConfig);
158                        reader.readFile(configFileName);
159                }
160                // Change scenario config according to given output- and input-filenames: events, plans, network
161                extConfig.setParam(SCENARIO, SCENARIO_INPUT_PLANS_FILENAME, this.outFileRoot + "/" + this.modulePrefix + ExternalInFileName);
162                extConfig.setParam(SCENARIO, SCENARIO_WORKING_PLANS_FILENAME, this.outFileRoot + "/" + this.modulePrefix + ExternalOutFileName);
163                extConfig.setParam(SCENARIO, SCENARIO_WORKING_EVENTS_TXT_FILENAME, this.controler.getIterationFilename(this.currentIteration - 1, "events.txt"));
164                String networkFilename = this.scenario.getConfig().findParam("network", "inputNetworkFile");
165                extConfig.setParam(SCENARIO, SCENARIO_NETWORK_FILENAME, networkFilename);
166                new ConfigWriter(extConfig).write(this.outFileRoot + this.modulePrefix + ExternalConfigFileName);
167        }
168
169        private void importPopulationAndMutatePlans() {
170                Scenario dummyScenario = ScenarioUtils.createScenario(ConfigUtils.createConfig());
171                MatsimReader plansReader = new PopulationReader(dummyScenario);
172                Population plans = dummyScenario.getPopulation();
173                plansReader.readFile(this.outFileRoot + this.modulePrefix + ExternalOutFileName);
174                new UpdatePlansAlgo().run(plans);
175        }
176
177        private class UpdatePlansAlgo extends AbstractPersonAlgorithm {
178                @Override
179                public void run(final Person dummyPerson) {
180                        Plan newPlan = dummyPerson.getPlans().get(0);
181                        Plan planToMutate = plansToMutate.get(dummyPerson.getId());
182            planToMutate.getPlanElements().clear();
183                        PopulationUtils.copyFromTo(newPlan, planToMutate);
184                }
185        }
