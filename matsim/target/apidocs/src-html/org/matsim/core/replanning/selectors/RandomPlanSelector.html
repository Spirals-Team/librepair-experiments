021package org.matsim.core.replanning.selectors;
022
023import org.matsim.api.core.v01.population.BasicPlan;
024import org.matsim.api.core.v01.population.HasPlansAndId;
025import org.matsim.core.gbl.MatsimRandom;
026
027
028/**
029 * Select randomly one of the existing plans of the person.
030 *
031 * @author mrieser
032 */
033public class RandomPlanSelector<T extends BasicPlan, I> implements PlanSelector<T, I> {
034
035        /**
036         * Choose a random plan from the person and return it.
037         * @return The newly selected plan for this person; <code>null</code> if the person has no plans.
038         */
039        @Override
040        public T selectPlan(final HasPlansAndId<T, I> person) {
041                // this used to use person.getRandomPlan(), but I inlined the function here in order to get rid of the function of the data class.
042                // kai, nov'13
043                if (person.getPlans().size() == 0) {
044                        return null;
045                }
046                
047                int index = (int)(MatsimRandom.getRandom().nextDouble()*person.getPlans().size());
048                // yyyy As far as I can tell, this produces race conditions when running multi-threaded.  I.e. when running the same
049                // setup twice, this function may return different results per thread or per person.  kai, jun'14
050                
051                return person.getPlans().get(index);
052        }
053}
