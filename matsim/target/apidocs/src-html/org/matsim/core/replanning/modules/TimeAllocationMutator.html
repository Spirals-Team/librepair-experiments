021package org.matsim.core.replanning.modules;

023import java.util.Collection;
024import java.util.HashMap;
025import java.util.Map;
026
027import javax.inject.Provider;
028
029import org.apache.log4j.Logger;
030import org.matsim.api.core.v01.population.Population;
031import org.matsim.api.core.v01.replanning.PlanStrategyModule;

037import org.matsim.core.config.groups.TimeAllocationMutatorConfigGroup.TimeAllocationMutatorSubpopulationSettings;
038import org.matsim.core.gbl.MatsimRandom;
039import org.matsim.core.population.algorithms.PlanAlgorithm;

045/**
046 * Wraps the {@link org.matsim.core.population.algorithms.PlanMutateTimeAllocation}-
047 * PlanAlgorithm into a {@link PlanStrategyModule} so it can be used for plans
048 * replanning. Supports multiple threads.
049 *
050 * @author mrieser
051 * @see org.matsim.core.population.algorithms.PlanMutateTimeAllocation
052 */
053public class TimeAllocationMutator extends AbstractMultithreadedModule {
054
055        private static final Logger log = Logger.getLogger(TimeAllocationMutator.class);
056        
057        private final Provider<TripRouter> tripRouterProvider;
058        private final double mutationRange;
059        private final boolean affectingDuration;
060        private final String subpopulationAttribute;
061        private final ObjectAttributes personAttributes;
062        private final Map<String, Double> subpopulationMutationRanges;
063        private final Map<String, Boolean> subpopulationAffectingDuration;
064        private final PlansConfigGroup.ActivityDurationInterpretation activityDurationInterpretation;
065
066        /**
067         * Creates a new TimeAllocationMutator with a mutation range as defined in
068         * the configuration (module "TimeAllocationMutator", param "mutationRange").
069         */
070        @Deprecated
071        public TimeAllocationMutator(Config config, Provider<TripRouter> tripRouterProvider, final double mutationRange, boolean affectingDuration) {
072                super(config.global());
073                this.tripRouterProvider = tripRouterProvider;
074                this.affectingDuration = affectingDuration;
075                this.mutationRange = mutationRange;
076                this.activityDurationInterpretation = (config.plans().getActivityDurationInterpretation());
077                this.personAttributes = null;
078                this.subpopulationAttribute = null;
079                this.subpopulationMutationRanges = null;
080                this.subpopulationAffectingDuration = null;
081                log.warn("deprecated constructor was used - individual time allocation mutator settings for subpopulations is not supported!");
082        }
083        
084        public TimeAllocationMutator(Provider<TripRouter> tripRouterProvider, PlansConfigGroup plansConfigGroup, TimeAllocationMutatorConfigGroup timeAllocationMutatorConfigGroup, GlobalConfigGroup globalConfigGroup) {
085                this(tripRouterProvider, plansConfigGroup, timeAllocationMutatorConfigGroup, globalConfigGroup, null);
086        }
087        
088        public TimeAllocationMutator(Provider<TripRouter> tripRouterProvider, PlansConfigGroup plansConfigGroup, TimeAllocationMutatorConfigGroup timeAllocationMutatorConfigGroup, GlobalConfigGroup globalConfigGroup,
089                        final Population population) {
090                super(globalConfigGroup);
091                this.tripRouterProvider = tripRouterProvider;
092                this.activityDurationInterpretation = plansConfigGroup.getActivityDurationInterpretation();
093                this.mutationRange = timeAllocationMutatorConfigGroup.getMutationRange();
094                this.affectingDuration = timeAllocationMutatorConfigGroup.isAffectingDuration();
095                
096                // in case we have subpopulations and individual settings for them
097                if (plansConfigGroup.getSubpopulationAttributeName() != null && timeAllocationMutatorConfigGroup.isUseIndividualSettingsForSubpopulations() && population != null) {
098                        this.subpopulationAttribute = plansConfigGroup.getSubpopulationAttributeName();
099                        this.subpopulationMutationRanges = new HashMap<>();
100                        this.subpopulationAffectingDuration = new HashMap<>();
101                        this.personAttributes = population.getPersonAttributes();
102                        
103                        Collection<? extends ConfigGroup> settings = timeAllocationMutatorConfigGroup.getParameterSets(TimeAllocationMutatorSubpopulationSettings.SET_NAME);
104                        for (ConfigGroup group : settings) {
105                                TimeAllocationMutatorSubpopulationSettings subpopulationSettings = (TimeAllocationMutatorSubpopulationSettings) group;
106                                String subpopulation = subpopulationSettings.getSubpopulation();
107                                this.subpopulationMutationRanges.put(subpopulation, subpopulationSettings.getMutationRange());
108                                this.subpopulationAffectingDuration.put(subpopulation, subpopulationSettings.isAffectingDuration());
109                                log.info("Found individual time mutator settings for subpopulation: " + subpopulation);
110                        }
111                } else {
112                        this.personAttributes = null;
113                        this.subpopulationAttribute = null;
114                        this.subpopulationMutationRanges = null;
115                        this.subpopulationAffectingDuration = null;
116                }
117        }
118
119        @Override
120        public PlanAlgorithm getPlanAlgoInstance() {
121                PlanAlgorithm pmta;
122                switch (this.activityDurationInterpretation) {
123                case minOfDurationAndEndTime:
124                        pmta = new TripPlanMutateTimeAllocation(this.tripRouterProvider.get().getStageActivityTypes(), this.mutationRange, this.affectingDuration, MatsimRandom.getLocalInstance(),
125                                        this.subpopulationAttribute, this.personAttributes, this.subpopulationMutationRanges, this.subpopulationAffectingDuration);
126                        break;
127                default:
128                        pmta = new PlanMutateTimeAllocationSimplified(
129                                        this.tripRouterProvider.get().getStageActivityTypes(), this.mutationRange, this.affectingDuration, MatsimRandom.getLocalInstance());
130                }
131                return pmta;
132        }
133}
