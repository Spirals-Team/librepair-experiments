050        private final GenericStrategyManager<Plan, Person> delegate;

052        @Inject
053        StrategyManager(StrategyConfigGroup strategyConfigGroup, PlansConfigGroup plansConfigGroup, ControlerConfigGroup controlerConfigGroup, Map<StrategyConfigGroup.StrategySettings, PlanStrategy> planStrategies) {
054                this();
055                setMaxPlansPerAgent(strategyConfigGroup.getMaxAgentPlanMemorySize());
056
057                int globalInnovationDisableAfter = (int) ((controlerConfigGroup.getLastIteration() - controlerConfigGroup.getFirstIteration())
058                                * strategyConfigGroup.getFractionOfIterationsToDisableInnovation() + controlerConfigGroup.getFirstIteration());
059                log.info("global innovation switch off after iteration: " + globalInnovationDisableAfter);
060
061                setSubpopulationAttributeName(plansConfigGroup.getSubpopulationAttributeName());
062                for (Map.Entry<StrategyConfigGroup.StrategySettings, PlanStrategy> entry : planStrategies.entrySet()) {
063                        PlanStrategy strategy = entry.getValue();
064                        StrategyConfigGroup.StrategySettings settings = entry.getKey();
065                        addStrategy(strategy, settings.getSubpopulation(), settings.getWeight());
066
067                        // now check if this modules should be disabled after some iterations
068                        int maxIter = settings.getDisableAfter();
069                        if ( maxIter > globalInnovationDisableAfter || maxIter==-1 ) {
070                                if (!PlanStrategies.isOnlySelector(strategy)) {
071                                        maxIter = globalInnovationDisableAfter ;
072                                }
073                        }
074
075                        if (maxIter >= 0) {
076                                if (maxIter >= controlerConfigGroup.getFirstIteration()) {
077                                        addChangeRequest(maxIter + 1, strategy, settings.getSubpopulation(), 0.0);
078                                } else {
079                                        /* The services starts at a later iteration than this change request is scheduled for.
080                                         * make the change right now.                                    */
081                                        changeWeightOfStrategy(strategy, settings.getSubpopulation(), 0.0);
082                                }
083                        }
084                }
085        }

160        /**
161         * Randomly chooses for each person of the population a strategy and uses that
162         * strategy on the person, after adapting the strategies to any pending change
163         * requests for the specified iteration.
164         *
165         * @param iteration the current iteration we're handling
166         */
167        public final void run(final Population population, final int iteration, final ReplanningContext replanningContext) {
168                // (this is not directly delegated since the run method of this StrategyManager includes two "hooks").
169                delegate.handleChangeRequests(iteration);
170                run(population, replanningContext);
171        }

178                // left empty for inheritance

181        /**
182         * Randomly chooses for each person of the population a strategy and uses that
183         * strategy on the person.
184         *
185         */
186        public final void run(final Population population, final ReplanningContext replanningContext) {
187                beforePopulationRunHook(population, replanningContext);
188                delegate.run(population.getPersons().values(), population.getPersonAttributes(), replanningContext);
189                afterRunHook(population);
190        }

193         * @param population  
194         */
195        protected void afterRunHook(Population population) {
196                // left empty for inheritance
197        }

247        /**
248         * Sets a plan selector to be used for choosing plans for removal, if they
249         * have more plans than the specified maximum. This defaults to
250         * {@link WorstPlanForRemovalSelector}.
251         * <p></p>
252         * Thoughts about using the logit-type selectors with negative logit model scale parameter:<ul>
253         * <li> Look at one agent.
254         * <li> Assume she has the choice between <i>n</i> different plans.
255         * <li> (Continuous) fraction <i>f(i)</i> of plan <i>i</i> develops as (master equation)
256         * <blockquote><i>
257         * df(i)/dt = - p(i) * f(i) + 1/n
258         * </i></blockquote>
259         * where <i>p(i)</i> is from the choice model.
260         * <li> Steady state solution (<i>df/dt=0</i>) <i> f(i) = 1/n * 1/p(i) </i>.
261         * <li> If <i> p(i) = e<sup>-b*U(i)</sup></i>, then <i> f(i) = e<sup>b*U(i)</sup> / n </i>.  Or in words:
262         * <i><b> If you use a logit model with a minus in front of the beta for plans removal, the resulting steady state distribution is
263         * the same logit model with normal beta.</b></i>
264         *
265         * </ul>
266         * The implication seems to be: divide the user-configured beta by two, use one half for choice and the other for plans removal.
267         * <p></p>
268         * The path size version still needs to be tested (both for choice and for plans removal).
269         *
270         *
271         * @see #setMaxPlansPerAgent(int)
272         */
273        @Inject
274        public final void setPlanSelectorForRemoval(final PlanSelector<Plan, Person> planSelector) {
275                delegate.setPlanSelectorForRemoval(planSelector);
276        }
