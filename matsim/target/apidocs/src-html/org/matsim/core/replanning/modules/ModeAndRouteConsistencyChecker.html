001package org.matsim.core.replanning.modules;
002
003import org.apache.log4j.Logger;
004import org.matsim.api.core.v01.TransportMode;
005import org.matsim.api.core.v01.population.Leg;
006import org.matsim.api.core.v01.population.Plan;
007import org.matsim.api.core.v01.replanning.PlanStrategyModule;
008import org.matsim.core.population.routes.NetworkRoute;
009import org.matsim.core.replanning.ReplanningContext;
010import org.matsim.core.router.TripStructureUtils;
011
012public class ModeAndRouteConsistencyChecker implements PlanStrategyModule {
013        @Override public void prepareReplanning(ReplanningContext replanningContext) { }
014        
015        @Override public void handlePlan(Plan plan) {
016                for (Leg leg : TripStructureUtils.getLegs(plan)) {
017                        if (leg.getRoute() instanceof NetworkRoute) {
018                                switch ( leg.getMode() ) {
019                                        case TransportMode.car:
020                                        case TransportMode.bike:
021                                        case TransportMode.walk:
022                                                break;
023                                        default:
024                                                Logger.getLogger(this.getClass()).warn( "route is of type=" + leg.getRoute().getClass() ) ;
025                                                Logger.getLogger(this.getClass()).warn( "mode=" + leg.getMode() ) ;
026                                                throw new RuntimeException("inconsistent");
027                                }
028                        }
029                }
030        }
031        
032        @Override public void finishReplanning() { }
033}
