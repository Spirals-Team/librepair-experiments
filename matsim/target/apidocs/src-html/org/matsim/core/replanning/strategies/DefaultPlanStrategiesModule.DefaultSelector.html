042public class DefaultPlanStrategiesModule extends AbstractModule {
043    private static final Logger log = Logger.getLogger( DefaultPlanStrategiesModule.class );
044
045    public enum DefaultPlansRemover { WorstPlanSelector, SelectRandom, SelectExpBetaForRemoval, ChangeExpBetaForRemoval,
046                PathSizeLogitSelectorForRemoval }
047
048    @Override
049    public void install() {
050        if (getConfig().strategy().getPlanSelectorForRemoval().equals(DefaultPlansRemover.WorstPlanSelector.toString())) {
051            bindPlanSelectorForRemoval().to(WorstPlanForRemovalSelector.class);
052        }
053        if (getConfig().strategy().getPlanSelectorForRemoval().equals(DefaultPlansRemover.SelectRandom.toString())) {
054            bindPlanSelectorForRemoval().to(new TypeLiteral<RandomPlanSelector<Plan, Person>>(){});
055        }
056        if (getConfig().strategy().getPlanSelectorForRemoval().equals(DefaultPlansRemover.SelectExpBetaForRemoval.toString())) {
057            bindPlanSelectorForRemoval().toProvider(ExpBetaPlanSelectorForRemoval.class);
058        }
059        if (getConfig().strategy().getPlanSelectorForRemoval().equals(DefaultPlansRemover.ChangeExpBetaForRemoval.toString())) {
060            bindPlanSelectorForRemoval().toProvider(ExpBetaPlanChangerForRemoval.class);
061        }
062        if (getConfig().strategy().getPlanSelectorForRemoval().equals(DefaultPlansRemover.PathSizeLogitSelectorForRemoval.toString())) {
063            bindPlanSelectorForRemoval().toProvider(PathSizeLogitSelectorForRemoval.class);
064        }
065
066        // We only bind those stategies that the StrategyManager is actually going to use,
067        // according to the Config.
068        // If you bind your own strategy from your own module, and then don't reference it from the Config,
069        // that's fine: The StrategyManager will still only add those strategies to itself which are configured.
070        // But we don't want to clutter the container here.
071        Set<String> usedStrategyNames = new HashSet<>();
072        for (StrategyConfigGroup.StrategySettings settings : getConfig().strategy().getStrategySettings()) {
073            usedStrategyNames.add(settings.getStrategyName());
074        }
075
076        // strategy packages that only select:
077        if (usedStrategyNames.contains(DefaultSelector.KeepLastSelected.toString())) {
078            addPlanStrategyBinding(DefaultSelector.KeepLastSelected.toString()).toProvider(KeepLastSelected.class);
079        }
080        if (usedStrategyNames.contains(DefaultSelector.BestScore.toString())) {
081            addPlanStrategyBinding(DefaultSelector.BestScore.toString()).toProvider(SelectBest.class);
082        }
083        if (usedStrategyNames.contains(DefaultSelector.SelectExpBeta.toString())) {
084            addPlanStrategyBinding(DefaultSelector.SelectExpBeta.toString()).toProvider(SelectExpBeta.class);
085        }
086        if (usedStrategyNames.contains(DefaultSelector.ChangeExpBeta.toString())) {
087            addPlanStrategyBinding(DefaultSelector.ChangeExpBeta.toString()).toProvider(ChangeExpBeta.class);
088        }
089        if (usedStrategyNames.contains(DefaultSelector.SelectRandom.toString())) {
090            addPlanStrategyBinding(DefaultSelector.SelectRandom.toString()).toProvider(SelectRandom.class);
091        }
092        if (usedStrategyNames.contains(DefaultSelector.SelectPathSizeLogit.toString())) {
093            addPlanStrategyBinding(DefaultSelector.SelectPathSizeLogit.toString()).toProvider(SelectPathSizeLogit.class);
094        }
095
096        // strategy packages that select, copy, and modify.  (The copying is done implicitly as soon as "addStrategyModule" is called
097        // at least once).
098
099        if (usedStrategyNames.contains(DefaultStrategy.ReRoute.toString())) {
100            addPlanStrategyBinding(DefaultStrategy.ReRoute.toString()).toProvider(ReRoute.class);
101        }
102        if (usedStrategyNames.contains(DefaultStrategy.TimeAllocationMutator.toString())) {
103            addPlanStrategyBinding(DefaultStrategy.TimeAllocationMutator.toString()).toProvider(TimeAllocationMutator.class);
104        }
105        if (usedStrategyNames.contains(DefaultStrategy.TimeAllocationMutator_ReRoute.toString())) {
106            addPlanStrategyBinding(DefaultStrategy.TimeAllocationMutator_ReRoute.toString()).toProvider(TimeAllocationMutatorReRoute.class);
107        }
108        if (usedStrategyNames.contains(DefaultStrategy.SubtourModeChoice.toString())) {
109            addPlanStrategyBinding(DefaultStrategy.SubtourModeChoice.toString()).toProvider(SubtourModeChoice.class);
110        }
111        if (usedStrategyNames.contains(DefaultStrategy.ChangeTripMode.toString())) {
112            addPlanStrategyBinding(DefaultStrategy.ChangeTripMode.toString()).toProvider(ChangeTripMode.class);
113        }
114        if (usedStrategyNames.contains(DefaultStrategy.ChangeSingleTripMode.toString())) {
115            addPlanStrategyBinding(DefaultStrategy.ChangeSingleTripMode.toString()).toProvider(ChangeSingleTripMode.class);
116        }
117
118        // td, 15 feb 16: removed the "Leg" versions of strategies. Notify the users that they should switch to the
119        // "Trip" versions. Should be left in 0.8.XXX releases, and then deleted, along with their name in the enum.
120        if ( usedStrategyNames.contains( DefaultStrategy.ChangeLegMode.toString() ) ) {
121            log.error( DefaultStrategy.ChangeLegMode+" replanning strategy does not exist anymore. Please use "+DefaultStrategy.ChangeTripMode+" instead." );
122        }
123        if ( usedStrategyNames.contains( DefaultStrategy.ChangeSingleLegMode.toString() ) ) {
124            log.error( DefaultStrategy.ChangeSingleLegMode+" replanning strategy does not exist anymore. Please use "+DefaultStrategy.ChangeSingleTripMode+" instead." );
125        }
126        if ( usedStrategyNames.contains( DefaultStrategy.TripSubtourModeChoice.toString() ) ) {
127            log.error( DefaultStrategy.TripSubtourModeChoice+" replanning strategy does not exist anymore. Please use "+DefaultStrategy.SubtourModeChoice+" instead." );
128        }
129    }

131    public interface DefaultSelector {
132            String KeepLastSelected="KeepLastSelected" ;
133            String BestScore="BestScore";
134            String ChangeExpBeta="ChangeExpBeta";
135            String SelectExpBeta="SelectExpBeta";
136            String SelectRandom="SelectRandom";
137            String SelectPathSizeLogit="SelectPathSizeLogit" ;
138    }
139    
140
141    public interface DefaultStrategy {
142        String ReRoute="ReRoute";
143        String TimeAllocationMutator="TimeAllocationMutator";
144        @Deprecated String ChangeLegMode="ChangeLegMode";
145        String TimeAllocationMutator_ReRoute="TimeAllocationMutator_ReRoute" ;
146        @Deprecated String ChangeSingleLegMode = "ChangeSingleLegMode" ;
147        String ChangeSingleTripMode="ChangeSingleTripMode" ;
148        String SubtourModeChoice = "SubtourModeChoice" ;
149        String ChangeTripMode = "ChangeTripMode" ;
150        @Deprecated String TripSubtourModeChoice = "TripSubtourModeChoice" ;
151    }

156    private static class ExpBetaPlanSelectorForRemoval implements Provider<ExpBetaPlanSelector<Plan, Person>> {
157
158        @Inject private PlanCalcScoreConfigGroup config;
159
160        @Override
161        public ExpBetaPlanSelector<Plan, Person> get() {
162            return new ExpBetaPlanSelector<>( - config.getBrainExpBeta());
163        }
164    }
165
166    private static class ExpBetaPlanChangerForRemoval implements Provider<ExpBetaPlanChanger<Plan, Person>> {
167
168        @Inject private PlanCalcScoreConfigGroup config;
169
170        @Override
171        public ExpBetaPlanChanger<Plan, Person> get() {
172            return new ExpBetaPlanChanger<>( - config.getBrainExpBeta());
173        }
174    }
175
176    private static class PathSizeLogitSelectorForRemoval implements Provider<PathSizeLogitSelector> {
177
178        @Inject PlanCalcScoreConfigGroup config;
179        @Inject Network network;
180
181        @Override
182        public PathSizeLogitSelector get() {
183            return new PathSizeLogitSelector(config.getPathSizeLogitBeta(), -config.getBrainExpBeta(),
184                    network);
185        }
186    }
