023package org.matsim.core.replanning;
024
025import com.google.inject.Key;
026import com.google.inject.Singleton;
027import com.google.inject.multibindings.MapBinder;
028import com.google.inject.name.Names;
029import org.matsim.api.core.v01.Scenario;
030import org.matsim.api.core.v01.population.Person;
031import org.matsim.api.core.v01.population.Plan;
032import org.matsim.core.config.groups.StrategyConfigGroup;
033import org.matsim.core.controler.AbstractModule;
034import org.matsim.core.controler.OutputDirectoryHierarchy;
035import org.matsim.core.replanning.modules.ExternalModule;
036import org.matsim.core.replanning.selectors.RandomPlanSelector;
037import org.matsim.core.replanning.strategies.DefaultPlanStrategiesModule;
038
039import javax.inject.Inject;
040import javax.inject.Provider;
041
042public class StrategyManagerModule extends AbstractModule {
043        @Override
044        public void install() {
045                int externalCounter = 0;
046                
047                install(new DefaultPlanStrategiesModule());
048                // (does commands of type "bind(PlanStrategy.class).annotatedWith(Names.named(strategyName))", i.e.
049                // plan strategies can be looked up under their names (*))
050                
051                bind(StrategyManager.class).in(Singleton.class);
052                bind(ReplanningContext.class).to(ReplanningContextImpl.class).asEagerSingleton();
053                
054                MapBinder<StrategyConfigGroup.StrategySettings, PlanStrategy> planStrategyMapBinder = MapBinder.newMapBinder(binder(), StrategyConfigGroup.StrategySettings.class, PlanStrategy.class);
055                // (this will bind a Map that has StrategySettings as key, and PlanStrategy as value.  Not sure why StrategySettings as key, and not just the name, but possibly this is mean to allow adding
056                // the same strategy multiple times, with possibly different settings.)
057                
058                for (StrategyConfigGroup.StrategySettings settings : getConfig().strategy().getStrategySettings()) {
059                        String name = settings.getStrategyName() ;
060                        if (name.equals("ExternalModule")) {
061                                // plan strategy is some external executable:
062                                externalCounter++;
063                                planStrategyMapBinder.addBinding(settings).toProvider(new ExternalModuleProvider(externalCounter, settings.getExePath()));
064                        } else if (name.contains(".")) {
065                                // plan strategy is in Java, but it is found via the class loader:
066                                if (name.startsWith("org.matsim.") && !name.startsWith("org.matsim.contrib.")) {
067                                        // org.matsim strategies are not to be loaded via the class loader:
068                                        throw new RuntimeException("Strategies in the org.matsim package must not be loaded by name!");
069                                } else {
070                                        try {
071                                                Class klass = Class.forName(name);
072                                                if (PlanStrategy.class.isAssignableFrom(klass)) {
073                                                        planStrategyMapBinder.addBinding(settings).to(klass);
074                                                } else if (Provider.class.isAssignableFrom(klass)) {
075                                                        planStrategyMapBinder.addBinding(settings).toProvider(klass);
076                                                } else {
077                                                        throw new RuntimeException("You specified a class name as a strategy, but it is neither a PlanStrategy nor a Provider.");
078                                                }
079                                        } catch (ClassNotFoundException e) {
080                                                throw new RuntimeException("You specified something which looks like a class name as a strategy, but the class could not be found.", e);
081                                        }
082                                }
083                        } else {
084                                // this is the normal case: plan strategy comes from within matsim
085                                planStrategyMapBinder.addBinding(settings).to(Key.get(PlanStrategy.class, Names.named(settings.getStrategyName())));
086                                // (settings is the key ... ok.  The Key.get(...) returns the PlanStrategy that was registered under its name at (*) above.)
087                        }
088                }
089        }
090        
091        /**
092         * If plan strategy comes from some external executable.  E.g. some external router that is not in Java.
093         */
094        private static class ExternalModuleProvider implements Provider<PlanStrategy> {
095
096                @Inject
097                private OutputDirectoryHierarchy controlerIO;
098
099                @Inject
100                private Scenario scenario;
101
102                private int externalCounter;
103                private String exePath;
104
105                public ExternalModuleProvider(int externalCounter, String exePath) {
106                        this.externalCounter = externalCounter;
107                        this.exePath = exePath;
108                }
109
110                @Override
111                public PlanStrategy get() {
112                        PlanStrategyImpl.Builder builder = new PlanStrategyImpl.Builder(new RandomPlanSelector<Plan, Person>());
113                        builder.addStrategyModule(new ExternalModule(exePath, "ext" + externalCounter, controlerIO, scenario));
114                        return builder.build();
115                }
116        }
117}
