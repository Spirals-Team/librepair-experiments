021package org.matsim.core.replanning.selectors;
022
023import java.util.LinkedHashMap;
024import java.util.Map;
025
026import org.apache.log4j.Logger;
027import org.matsim.api.core.v01.population.BasicPlan;
028import org.matsim.api.core.v01.population.HasPlansAndId;
029import org.matsim.core.config.groups.PlanCalcScoreConfigGroup;
030import org.matsim.core.gbl.MatsimRandom;
031
032/**
033 * Selects one of the existing plans of the person based on the
034 * weight = exp(beta*score).
035 *
036 * @author mrieser
037 */
038public class ExpBetaPlanSelector<T extends BasicPlan, I> implements PlanSelector<T, I> {
039
040        protected static final double MIN_WEIGHT = Double.MIN_VALUE;
041        protected final double beta;
042        
043        public ExpBetaPlanSelector( final double logitScaleFactor ) {
044                this.beta = logitScaleFactor ;
045        }
046
047        public ExpBetaPlanSelector(PlanCalcScoreConfigGroup charyparNagelScoringConfigGroup) {
048                this( charyparNagelScoringConfigGroup.getBrainExpBeta() ) ;
049        }
050
051        /**
052         * @return a random plan from the person, random but according to its weight.
053         */
054        @Override
055        public T selectPlan(final HasPlansAndId<T, I> person) {
056
057                // get the weights of all plans
058                Map<T, Double> weights = this.calcWeights(person);
059
060                double sumWeights = 0.0;
061                for (Double weight : weights.values()) {
062                        sumWeights += weight;
063                }
064
065                // choose a random number over interval [0, sumWeights[
066                double selnum = sumWeights * MatsimRandom.getRandom().nextDouble();
067                for (T plan : person.getPlans()) {
068                        selnum -= weights.get(plan);
069                        if (selnum <= 0.0) {
070                                return plan;
071                        }
072                }
073
074                // hmm, no plan returned... either the person has no plans, or the plan(s) have no score.
075                if (person.getPlans().size() > 0) {
076                        return person.getPlans().get(0);
077                }
078
079                // this case should never happen, except a person has no plans at all.
080                return null;
081        }
082
083        /**
084         * Calculates the weight of a single plan.
085         *
086         * @return the weight of the plan
087         */
088        protected double calcPlanWeight(final T plan, final double maxScore) {
089                // NOTE: The deduction of "maxScore" from all scores is a numerical trick.  It ensures that the values of exp(...)
090                // are in some normal range, instead of close to numerical infinity.  The latter leads to numerically instable
091                // results (this is not fiction; we had that some time ago). kai, aug'12
092
093                if (plan.getScore() == null) {
094                        return Double.NaN;
095                }
096                double weight = Math.exp(this.beta * (plan.getScore() - maxScore));
097                if (weight < MIN_WEIGHT) weight = MIN_WEIGHT;
098                return weight;
099        }
100
101        /**
102         * Builds the weights of all plans.
103         *
104         * @return a map containing the weights of all plans
105         */
106        Map<T, Double> calcWeights(final HasPlansAndId<T, ?> person) {
107
108                // - first find the max. score of all plans of this person
109                double maxScore = Double.NEGATIVE_INFINITY;
110                for (T plan1 : person.getPlans()) {
111                        if ( (plan1.getScore() != null) && plan1.getScore().isNaN() ) {
112                                Logger.getLogger(this.getClass()).error("encountering getScore().isNaN().  This class is not well behaved in this situation.  Continuing anyway ...") ;
113                        }
114                        if ((plan1.getScore() != null) && (plan1.getScore() > maxScore)) {
115                                maxScore = plan1.getScore();
116                        }
117                }
118
119                Map<T, Double> weights = new LinkedHashMap<T, Double>(person.getPlans().size());
120
121                for (T plan : person.getPlans()) {
122                        weights.put(plan, this.calcPlanWeight(plan, maxScore));
123                        // see note in calcPlanWeight!
124                }
125
126                return weights;
127        }
128
129    /**
130     * @return the probability that this expBetaPlanSelector will select this plan for this person.
131     */
132        public static <T extends BasicPlan, I> double getSelectionProbability(ExpBetaPlanSelector<T, I> expBetaPlanSelector, HasPlansAndId<T, ?> person, final T plan) {
133                Map<T, Double> weights = expBetaPlanSelector.calcWeights(person);
134                double thisWeight = weights.get(plan);
135
136                double sumWeights = 0.0;
137                for (Double weight : weights.values()) {
138                        sumWeights += weight;
139                }
140
141                return (thisWeight / sumWeights);
142        }
143
144}
