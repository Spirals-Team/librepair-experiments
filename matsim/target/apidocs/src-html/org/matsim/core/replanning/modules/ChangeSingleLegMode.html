021package org.matsim.core.replanning.modules;
022
023import org.matsim.api.core.v01.TransportMode;
024import org.matsim.core.config.groups.ChangeModeConfigGroup;
025import org.matsim.core.config.groups.GlobalConfigGroup;
026import org.matsim.core.gbl.MatsimRandom;
027import org.matsim.core.population.algorithms.ChooseRandomSingleLegMode;
028import org.matsim.core.population.algorithms.PlanAlgorithm;
029
030/**
031 * Changes the transportation mode of one leg in a plan to a randomly chosen
032 * mode, given a list of possible modes. Insures that the newly chosen mode
033 * is different from the existing mode (if possible)
034 *
035 * By default, the module chooses between "car" and "pt". If other modes should
036 * be used, it can be done so in the configuration. Also, this module is able to (optionally)
037 * respect car-availability:
038 * <pre>
039 * &lt;module name="changeMode">
040 *   &lt!-- provide a comma-separated list of leg modes -->
041 *   &lt;param name="modes" value="car,walk,bike" />
042 *   &lt;param name="ignoreCarAvailability" value="false" />
043 * &lt;/module>
044 * </pre>
045 *
046 * @see ChooseRandomSingleLegMode
047 *
048 * @author mrieser
049 */
050public class ChangeSingleLegMode extends AbstractMultithreadedModule {
051
052        private String[] availableModes = new String[] { TransportMode.car, TransportMode.pt };
053        private boolean ignoreCarAvailability = true;
054
055        public ChangeSingleLegMode(final GlobalConfigGroup globalConfigGroup, ChangeModeConfigGroup changeLegModeConfigGroup) {
056                super(globalConfigGroup.getNumberOfThreads());
057                this.availableModes = changeLegModeConfigGroup.getModes();
058                this.ignoreCarAvailability = changeLegModeConfigGroup.getIgnoreCarAvailability();
059        }
060        
061        public ChangeSingleLegMode(final int nOfThreads, final String[] modes, final boolean ignoreCarAvailabilty) {
062                super(nOfThreads);
063                this.availableModes = modes.clone();
064                this.ignoreCarAvailability = ignoreCarAvailabilty;
065        }
066
067        @Override
068        public PlanAlgorithm getPlanAlgoInstance() {
069                ChooseRandomSingleLegMode algo = new ChooseRandomSingleLegMode(this.availableModes, MatsimRandom.getLocalInstance());
070                algo.setIgnoreCarAvailability(this.ignoreCarAvailability);
071                return algo;
072        }
073
074}
