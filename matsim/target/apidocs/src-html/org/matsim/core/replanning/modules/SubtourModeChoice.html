021package org.matsim.core.replanning.modules;
022
023
024import javax.inject.Provider;
025
026import org.matsim.core.config.groups.GlobalConfigGroup;
027import org.matsim.core.config.groups.SubtourModeChoiceConfigGroup;
028import org.matsim.core.gbl.MatsimRandom;
029import org.matsim.core.population.algorithms.ChooseRandomLegModeForSubtour;
030import org.matsim.core.population.algorithms.PermissibleModesCalculator;
031import org.matsim.core.population.algorithms.PermissibleModesCalculatorImpl;
032import org.matsim.core.population.algorithms.PlanAlgorithm;
033import org.matsim.core.router.TripRouter;
034
035/**
036 * Changes the transportation mode of all legs of one randomly chosen subtour in a plan to a randomly chosen
037 * different mode given a list of possible modes.
038 *
039 * A subtour is a consecutive subset of a plan which starts and ends at the same link.
040 * 
041 * Certain modes are considered only if the choice would not require some resource to appear
042 * out of thin air. For example, you can only drive your car back from work if you have previously parked it
043 * there. These are called chain-based modes.
044 * 
045 * The assumption is that each chain-based mode requires one resource (car, bike, ...) and that this
046 * resource is initially positioned at home. Home is the location of the first activity in the plan.
047 * 
048 * If the plan initially violates this constraint, this module may (!) repair it. 
049 * 
050 * @author michaz
051 * 
052 */
053public class SubtourModeChoice extends AbstractMultithreadedModule {
054
055        private final Provider<TripRouter> tripRouterProvider;
056
057        private PermissibleModesCalculator permissibleModesCalculator;
058        
059        private final String[] chainBasedModes;
060        private final String[] modes;
061        
062        public SubtourModeChoice(Provider<TripRouter> tripRouterProvider, GlobalConfigGroup globalConfigGroup, SubtourModeChoiceConfigGroup subtourModeChoiceConfigGroup) {
063                this(globalConfigGroup.getNumberOfThreads(),
064                                subtourModeChoiceConfigGroup.getModes(),
065                                subtourModeChoiceConfigGroup.getChainBasedModes(),
066                                subtourModeChoiceConfigGroup.considerCarAvailability(), tripRouterProvider);
067        }
068
069        public SubtourModeChoice(
070                        final int numberOfThreads,
071                        final String[] modes,
072                        final String[] chainBasedModes,
073                        final boolean considerCarAvailability, Provider<TripRouter> tripRouterProvider) {
074                super(numberOfThreads);
075                this.tripRouterProvider = tripRouterProvider;
076                this.modes = modes.clone();
077                this.chainBasedModes = chainBasedModes.clone();
078                this.permissibleModesCalculator =
079                        new PermissibleModesCalculatorImpl(
080                                        this.modes,
081                                        considerCarAvailability);
082        }
083        
084        protected String[] getModes() {
085                return modes.clone();
086        }
087
088        @Override
089        public PlanAlgorithm getPlanAlgoInstance() {
090                final TripRouter tripRouter = tripRouterProvider.get();
091                final ChooseRandomLegModeForSubtour chooseRandomLegMode =
092                                new ChooseRandomLegModeForSubtour(
093                                                tripRouter.getStageActivityTypes(),
094                                                tripRouter.getMainModeIdentifier(),
095                                                this.permissibleModesCalculator,
096                                                this.modes,
097                                                this.chainBasedModes,
098                                                MatsimRandom.getLocalInstance());
099                chooseRandomLegMode.setAnchorSubtoursAtFacilitiesInsteadOfLinks( false );
100                return chooseRandomLegMode;
101        }
102
103        /**
104         * Decides if a person may use a certain mode of transport. Can be used for car ownership.
105         * 
106         */
107        public void setPermissibleModesCalculator(PermissibleModesCalculator permissibleModesCalculator) {
108                this.permissibleModesCalculator = permissibleModesCalculator;
109        }
110
111}
