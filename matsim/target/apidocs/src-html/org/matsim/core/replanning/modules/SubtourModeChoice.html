021package org.matsim.core.replanning.modules;
022
023
024import javax.inject.Provider;
025
026import org.matsim.core.config.groups.GlobalConfigGroup;
027import org.matsim.core.config.groups.SubtourModeChoiceConfigGroup;
028import org.matsim.core.gbl.MatsimRandom;
029import org.matsim.core.population.algorithms.ChooseRandomLegModeForSubtour;
030import org.matsim.core.population.algorithms.PermissibleModesCalculator;
031import org.matsim.core.population.algorithms.PermissibleModesCalculatorImpl;
032import org.matsim.core.population.algorithms.PlanAlgorithm;
033import org.matsim.core.router.TripRouter;
034
035/**
036 * Changes the transportation mode of all legs of one randomly chosen subtour in a plan to a randomly chosen
037 * different mode given a list of possible modes.
038 *
039 * A subtour is a consecutive subset of a plan which starts and ends at the same link.
040 * 
041 * Certain modes are considered only if the choice would not require some resource to appear
042 * out of thin air. For example, you can only drive your car back from work if you have previously parked it
043 * there. These are called chain-based modes.
044 * 
045 * The assumption is that each chain-based mode requires one resource (car, bike, ...) and that this
046 * resource is initially positioned at home. Home is the location of the first activity in the plan.
047 * 
048 * If the plan initially violates this constraint, this module may (!) repair it. 
049 * 
050 * @author michaz
051 * 
052 */
053public class SubtourModeChoice extends AbstractMultithreadedModule {
054        
055        private final double probaForChangeSingleTripMode ;
056        
057        public enum Behavior { fromAllModesToSpecifiedModes, fromSpecifiedModesToSpecifiedModes }
058        private Behavior behavior = Behavior.fromSpecifiedModesToSpecifiedModes ;
059
060        private final Provider<TripRouter> tripRouterProvider;
061
062        private PermissibleModesCalculator permissibleModesCalculator;
063        
064        private final String[] chainBasedModes;
065        private final String[] modes;
066        
067        public SubtourModeChoice(Provider<TripRouter> tripRouterProvider, GlobalConfigGroup globalConfigGroup,
068                                                         SubtourModeChoiceConfigGroup subtourModeChoiceConfigGroup) {
069                this(globalConfigGroup.getNumberOfThreads(),
070                                subtourModeChoiceConfigGroup.getModes(),
071                                subtourModeChoiceConfigGroup.getChainBasedModes(),
072                                subtourModeChoiceConfigGroup.considerCarAvailability(),
073                                subtourModeChoiceConfigGroup.getProbaForRandomSingleTripMode(),
074                                tripRouterProvider
075                );
076                this.setBehavior( subtourModeChoiceConfigGroup.getBehavior() );
077        }
078
079        public SubtourModeChoice(
080                        final int numberOfThreads,
081                        final String[] modes,
082                        final String[] chainBasedModes,
083                        final boolean considerCarAvailability,
084                        double probaForChangeSingleTripMode,
085                        Provider<TripRouter> tripRouterProvider) {
086                super(numberOfThreads);
087                this.probaForChangeSingleTripMode = probaForChangeSingleTripMode;
088                this.tripRouterProvider = tripRouterProvider;
089                this.modes = modes.clone();
090                this.chainBasedModes = chainBasedModes.clone();
091                this.permissibleModesCalculator =
092                        new PermissibleModesCalculatorImpl(
093                                        this.modes,
094                                        considerCarAvailability);
095        }
096        
097        @Deprecated // only use when backwards compatibility is needed. kai, may'18
098        public final void setBehavior ( Behavior behavior ) {
099                this.behavior = behavior ;
100        }
101        
102        protected String[] getModes() {
103                return modes.clone();
104        }
105
106        @Override
107        public PlanAlgorithm getPlanAlgoInstance() {
108                final TripRouter tripRouter = tripRouterProvider.get();
109                
110                final ChooseRandomLegModeForSubtour chooseRandomLegMode =
111                                new ChooseRandomLegModeForSubtour(
112                                                tripRouter.getStageActivityTypes(),
113                                                tripRouter.getMainModeIdentifier(),
114                                                this.permissibleModesCalculator,
115                                                this.modes,
116                                                this.chainBasedModes,
117                                                MatsimRandom.getLocalInstance(), behavior, probaForChangeSingleTripMode);
118                chooseRandomLegMode.setAnchorSubtoursAtFacilitiesInsteadOfLinks( false );
119                return chooseRandomLegMode;
120        }
121
122        /**
123         * Decides if a person may use a certain mode of transport. Can be used for car ownership.
124         * 
125         */
126        public void setPermissibleModesCalculator(PermissibleModesCalculator permissibleModesCalculator) {
127                this.permissibleModesCalculator = permissibleModesCalculator;
128        }
129
130}
