019package org.matsim.core.replanning.selectors;
020
021import java.util.List;
022import java.util.Map;
023
024import org.matsim.api.core.v01.population.HasPlansAndId;
025import org.matsim.api.core.v01.population.Person;
026import org.matsim.api.core.v01.population.Plan;
027import org.matsim.core.gbl.MatsimRandom;
028
029/**
030 * Standardizing plans selection when weights are given.
031 * 
032 * @author nagel
033 *
034 */
035public abstract class AbstractPlanSelector implements PlanSelector<Plan, Person> {
036
037        @Override
038        public final Plan selectPlan(HasPlansAndId<Plan, Person> person) {
039                // First check if there are any unscored plans
040                Plan selectedPlan = new RandomUnscoredPlanSelector<Plan, Person>().selectPlan(person);
041                if (selectedPlan != null) return selectedPlan;
042                // Okay, no unscored plans...
043
044                // Build the weights of all plans
045
046                // - now calculate the weights
047                Map<Plan,Double> wc = calcWeights(person.getPlans() );
048                double sumWeights = 0. ;
049                for ( Double score : wc.values() ) {
050                        sumWeights += score ;
051                }
052
053                // choose a random number over interval [0,sumWeights[
054                double selnum = sumWeights*MatsimRandom.getRandom().nextDouble();
055                for (Plan plan : person.getPlans()) {
056                        selnum -= wc.get(plan);
057                        if (selnum <= 0.0) {
058                                return plan;
059                        }
060                }
061
062                // this case should never happen, except a person has no plans at all.
063                return null;
064
065        }
066        
067        abstract protected Map<Plan,Double> calcWeights( List<? extends Plan> plans ) ;
068        
069}
