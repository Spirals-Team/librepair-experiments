020package org.matsim.core.replanning.modules;
021
022import org.matsim.core.config.groups.GlobalConfigGroup;
023import org.matsim.core.population.algorithms.PlanAlgorithm;
024import org.matsim.core.population.algorithms.TripsToLegsAlgorithm;
025import org.matsim.core.router.CompositeStageActivityTypes;
026import org.matsim.core.router.StageActivityTypes;
027import org.matsim.core.router.TripRouter;
028
029import javax.inject.Provider;
030
031/**
032 * Removes trips and replaces them by legs.
033 * The aim is to simplify the plan before passing it to plan algorithms
034 * unable to handle multi-planElement trips.
035 * The plan must be re-routed before execution!
036 * @author thibautd
037 */
038public class TripsToLegsModule extends AbstractMultithreadedModule {
039
040        private final StageActivityTypes additionalBlackList;
041        private final Provider<TripRouter> tripRouterProvider;
042
043        /**
044         * Initializes an instance using the stage activity types from the controler
045         */
046        public TripsToLegsModule(Provider<TripRouter> tripRouterProvider, GlobalConfigGroup globalConfigGroup) {
047                this(null, tripRouterProvider, globalConfigGroup);
048        }
049
050        /**
051         * Initializes an instance, allowing to specify additional activity types to
052         * consider as stage activities.
053         * @param additionalBlackList a {@link StageActivityTypes} instance identifying
054         * @param tripRouterProvider
055         * @param globalConfigGroup
056         */
057        public TripsToLegsModule(final StageActivityTypes additionalBlackList, Provider<TripRouter> tripRouterProvider, GlobalConfigGroup globalConfigGroup) {
058                super(globalConfigGroup);
059                this.tripRouterProvider = tripRouterProvider;
060                this.additionalBlackList = additionalBlackList;
061        }
062
063        @Override
064        public PlanAlgorithm getPlanAlgoInstance() {
065                TripRouter router = tripRouterProvider.get();
066                StageActivityTypes blackListToUse = router.getStageActivityTypes();
067
068                if (additionalBlackList != null) {
069                        CompositeStageActivityTypes composite = new CompositeStageActivityTypes();
070                        composite.addActivityTypes( blackListToUse );
071                        composite.addActivityTypes( additionalBlackList );
072                        blackListToUse = composite;
073                }
074
075                return new TripsToLegsAlgorithm( 
076                                blackListToUse,
077                                router.getMainModeIdentifier() );
078        }
079}
