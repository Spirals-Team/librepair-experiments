060        static class StrategyWeights<T extends BasicPlan, I> {
061                final List<GenericPlanStrategy<T, I>> strategies = new ArrayList<>();
062                final List<GenericPlanStrategy<T, I>> unmodifiableStrategies = Collections.unmodifiableList( strategies );
063                final List<Double> weights = new ArrayList<>();
064                final List<Double> unmodifiableWeights = Collections.unmodifiableList(weights);
065                double totalWeights = 0.0;
066                final Map<Integer, Map<GenericPlanStrategy<T, I>, Double>> changeRequests = new TreeMap<>();
067        }

085        /**
086         * Adds a strategy to this manager with the specified weight. This weight
087         * compared to the sum of weights of all strategies in this manager defines
088         * the probability this strategy will be used for an agent.
089         *
090         */
091        public final void addStrategy(
092                        final GenericPlanStrategy<T, I> strategy,
093                        final String subpopulation,
094                        final double weight) {
095                final StrategyWeights<T, I> weights = getStrategyWeights( subpopulation );
096                if ( weights.strategies.contains( strategy ) ) {
097                        log.error( "Strategy "+strategy+" is already defined for subpopulation "+subpopulation  );
098                        log.error( "This can lead to undefined behavior. Please only specify each strategy once" );
099                        throw new IllegalStateException( "Strategy "+strategy+" is already defined for subpopulation "+subpopulation  );
100                }
101                weights.strategies.add(strategy);
102                weights.weights.add(weight);
103                weights.totalWeights += weight;
104        }
105
106        /**
107         * removes the specified strategy from this manager for the specified subpopulation
108         *
109         * @param strategy the strategy to be removed
110         * @param subpopulation the subpopulation for which the strategy must be removed
111         * @return true if the strategy was successfully removed from this manager,
112         *              false if the strategy was not part of this manager and could thus not be removed.
113         */
114        final boolean removeStrategy(
115                        final GenericPlanStrategy<T, I> strategy,
116                        final String subpopulation)
117        {
118                final StrategyWeights<T, I> weights = getStrategyWeights( subpopulation );
119                int idx = weights.strategies.indexOf(strategy);
120                if (idx != -1) {
121                        weights.strategies.remove(idx);
122                        double weight = weights.weights.remove(idx);
123                        weights.totalWeights -= weight;
124                        return true;
125                }
126                return false;
127        }

131        private StrategyWeights<T, I> getStrategyWeights(final String subpop) {
132                StrategyWeights<T, I> weights = weightsPerSubpopulation.get(subpop);
133
134                if ( weights == null ) {
135                        weights = new StrategyWeights<>();
136                        weightsPerSubpopulation.put(subpop, weights);
137                }
138
139                return weights;
140        }
141
142        /**
143         * changes the weight of the specified strategy
144         *
145         * @return true if the strategy is part of this manager and the weight could
146         *              be changed successfully, false otherwise.
147         */
148        final boolean changeWeightOfStrategy(
149                        final GenericPlanStrategy<T, I> strategy,
150                        final String subpopulation,
151                        final double newWeight) {
152                final StrategyWeights<T, I> weights = getStrategyWeights(subpopulation);
153                int idx = weights.strategies.indexOf(strategy);
154                if (idx != -1) {
155                        double oldWeight = weights.weights.set(idx, newWeight);
156                        weights.totalWeights += (newWeight - oldWeight);
157                        Logger.getLogger(this.getClass()).info( strategy.toString() + ": oldWeight=" + oldWeight + " newWeight=" + newWeight );
158                        return true;
159                }
160                return false;
161        }

163        /**
164         * Randomly chooses for each person of the population a strategy and uses that
165         * strategy on the person, after adapting the strategies to any pending change
166         * requests for the specified iteration.
167         *
168         * @param iteration the current iteration we're handling
169         */
170        public final void run(
171                        final Iterable<? extends HasPlansAndId<T, I>> persons,
172                                        ObjectAttributes subpopLookup,
173                                        final int iteration,
174                                        final ReplanningContext replanningContext ) {
175                handleChangeRequests(iteration);
176                run(persons, subpopLookup, replanningContext);
177        }
178
179        /**
180         * Randomly chooses for each person of the population a strategy and uses that
181         * strategy on the person.
182         *
183         */
184        final void run(
185                        final Iterable<? extends HasPlansAndId<T, I>> persons,
186                                        ObjectAttributes subPopLookup,
187                                        final ReplanningContext replanningContext) {
188
189                // initialize all strategies
190                for (GenericPlanStrategy<T, I> strategy : distinctStrategies()) {
191                        strategy.init(replanningContext);
192                }
193
194                // then go through the population and ...
195                for (HasPlansAndId<T, I> person : persons ) {
196
197                        // ... reduce the number of plans to the allowed maximum (in evol comp lang this is "selection")
198                        if ((this.maxPlansPerAgent > 0) && (person.getPlans().size() > this.maxPlansPerAgent)) {
199                                removePlans( person, this.maxPlansPerAgent);
200                        }
201
202                        // ... choose the strategy to be used for this person (in evol comp lang this would be the choice of the mutation operator)
203                        String subpopName = null;
204                        if (this.subpopulationAttributeName != null) {
205                                subpopName = (String) subPopLookup.getAttribute(person.getId().toString(), this.subpopulationAttributeName);
206                        }
207                        GenericPlanStrategy<T, I> strategy = this.chooseStrategy(person, subpopName);
208
209                        if (strategy==null) {
210                                throw new RuntimeException("No strategy found! Have you defined at least one replanning strategy per subpopulation?");
211                        }
212                        
213                        // ... and run the strategy:
214                        strategy.run(person);
215                }
216
217                // finally make sure all strategies have finished there work
218                for (GenericPlanStrategy<T, I> strategy : distinctStrategies()) {
219                        strategy.finish();
220                }
221
222        }
223
224        private Collection<GenericPlanStrategy<T, I>> distinctStrategies() {
225                // Leaving out duplicate strategies in different subpopulations
226                Collection<GenericPlanStrategy<T, I>> strategies = new LinkedHashSet<>();
227                for (StrategyWeights<T, I> weights : weightsPerSubpopulation.values()) {
228                        strategies.addAll(weights.strategies);
229                }
230                return strategies;
231        }
232
233        private void removePlans(final HasPlansAndId<T, I> person, final int maxNumberOfPlans) {
234                while (person.getPlans().size() > maxNumberOfPlans) {
235                        T plan = this.removalPlanSelector.selectPlan(person);
236                        person.removePlan(plan);
237                        if (plan == person.getSelectedPlan()) {
238                                final T newPlanToSelect = new RandomPlanSelector<T, I>().selectPlan(person) ;
239                                if ( newPlanToSelect == null ) {
240                                        throw new IllegalStateException( "could not find a plan to select for person "+person );
241                                }
242                                person.setSelectedPlan( newPlanToSelect );
243                        }
244                }
245        }

247        /**
248         * modifies the loaded strategies according to pending change requests for this iteration.
249         *
250         */
251        final void handleChangeRequests(final int iteration) {
252                for ( int ii = 0 ; ii <= iteration ; ii++ ) {
253                        // (playing back history for those installations which recreate the strategy manager in every iteration)
254                        for ( Map.Entry<String, StrategyWeights<T, I>> wentry : weightsPerSubpopulation.entrySet() ) {
255                                final String subpop = wentry.getKey();
256                                final StrategyWeights<T, I> weights = wentry.getValue();
257                                Map<GenericPlanStrategy<T, I>, Double> changes = weights.changeRequests.remove(ii);
258                                if (changes != null) {
259                                        for (Map.Entry<GenericPlanStrategy<T, I>, Double> entry : changes.entrySet()) {
260                                                changeWeightOfStrategy( entry.getKey(), subpop, entry.getValue());
261                                        }
262                                }
263                        }
264                }
265        }

267        /**
268         * chooses a (weight-influenced) random strategy
269         *
270         * @return the chosen strategy
271         */
272        /* deliberately package */ GenericPlanStrategy<T, I> chooseStrategy(HasPlansAndId<T, I> person, final String subpopulation) {
273                // yyyyyy I can see that this would need to be replaceable, but need to find some other way than inheritance.  kai, mar'18
274                
275                final StrategyWeights<T, I> weights = getStrategyWeights(subpopulation);
276
277                double rnd = MatsimRandom.getRandom().nextDouble() * weights.totalWeights;
278
279                double sum = 0.0;
280                for (int i = 0, max = weights.weights.size(); i < max; i++) {
281                        sum += weights.weights.get(i);
282                        if (rnd <= sum) {
283                                return weights.strategies.get(i);
284                        }
285                }
286                return null;
287        }

300        /**
301         * Schedules a {@link #changeWeightOfStrategy(GenericPlanStrategy, String, double)} command for a later iteration. The
302         * change will take place before the strategies are applied.
303         *
304         */
305        public final void addChangeRequest(
306                        final int iteration,
307                        final GenericPlanStrategy<T, I> strategy,
308                        final String subpopulation,
309                        final double newWeight) {
310                final StrategyWeights<T, I> weights = getStrategyWeights( subpopulation );
311                Integer iter = iteration;
312                Map<GenericPlanStrategy<T, I>, Double> iterationRequests = weights.changeRequests.get(iter);
313                if (iterationRequests == null) {
314                        iterationRequests = new HashMap<>(3);
315                        weights.changeRequests.put(iter, iterationRequests);
316                }
317                iterationRequests.put(strategy, newWeight);
318                Logger.getLogger(this.getClass()).info( "added change request: "
319                                + " iteration=" + iter + " newWeight=" + newWeight + " strategy=" + strategy.toString() );
320        }

322        /**
323         * Sets a plan selector to be used for choosing plans for removal, if they
324         * have more plans than the specified maximum. This defaults to
325         * {@link WorstPlanForRemovalSelector}.
326         * <p></p>
327         * Thoughts about using the logit-type selectors with negative logit model scale parameter:<ul>
328         * <li> Look at one agent.
329         * <li> Assume she has the choice between <i>n</i> different plans (although fewer of them are in the MATSim choice set).
330         * <li> (Continuous) fraction <i>f(i)</i> of plan <i>i</i> develops as (master equation)
331         * <blockquote><i>
332         * df(i)/dt = - p(i) * f(i) + 1/n
333         * </i></blockquote>
334         * where <i>p(i)</i> is from the removal selector, and <i>1/n</i> makes the assumption that each possible plan is (re-)inserted with equal probability
335         * by the innovative modules.
336         * <li> Steady state solution (<i>df/dt=0</i>) <i> f(i) = 1/n * 1/p(i) </i>.
337         * <li> If <i> p(i) = e<sup>-b*U(i)</sup></i>, then <i> f(i) = e<sup>b*U(i)</sup> / n </i>.  Or in words:
338         * <i><b> If you use a logit model with a minus in front of the beta for plans removal, the resulting steady state distribution is
339         * the same logit model with normal beta.</b></i>
340         *
341         * </ul>
342         * The implication seems to be: divide the user-configured beta by two, use one half for choice and the other for plans removal.
343         * <p></p>
344         * The path size version still needs to be tested (both for choice and for plans removal).
345         *
346         *
347         * @see #setMaxPlansPerAgent(int)
348         */
349        public final void setPlanSelectorForRemoval(final PlanSelector<T, I> planSelector) {
350                Logger.getLogger(this.getClass()).info("setting PlanSelectorForRemoval to " + planSelector.getClass() ) ;
351                this.removalPlanSelector = planSelector;
352        }
