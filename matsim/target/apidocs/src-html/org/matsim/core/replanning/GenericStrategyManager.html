059        static class StrategyWeights<T extends BasicPlan, I> {
060                final List<GenericPlanStrategy<T, I>> strategies = new ArrayList<>();
061                final List<GenericPlanStrategy<T, I>> unmodifiableStrategies = Collections.unmodifiableList( strategies );
062                final List<Double> weights = new ArrayList<>();
063                final List<Double> unmodifiableWeights = Collections.unmodifiableList(weights);
064                double totalWeights = 0.0;
065                final Map<Integer, Map<GenericPlanStrategy<T, I>, Double>> changeRequests = new TreeMap<>();
066        }

087        /**
088         * Adds a strategy to this manager with the specified weight. This weight
089         * compared to the sum of weights of all strategies in this manager defines
090         * the probability this strategy will be used for an agent.
091         *
092         */
093        public final void addStrategy(
094                        final GenericPlanStrategy<PL, AG> strategy,
095                        final String subpopulation,
096                        final double weight) {
097                final StrategyWeights<PL, AG> weights = getStrategyWeights( subpopulation );
098                if ( weights.strategies.contains( strategy ) ) {
099                        log.error( "Strategy "+strategy+" is already defined for subpopulation "+subpopulation  );
100                        log.error( "This can lead to undefined behavior. Please only specify each strategy once" );
101                        throw new IllegalStateException( "Strategy "+strategy+" is already defined for subpopulation "+subpopulation  );
102                }
103                weights.strategies.add(strategy);
104                weights.weights.add(weight);
105                weights.totalWeights += weight;
106        }
107
108        /**
109         * removes the specified strategy from this manager for the specified subpopulation
110         *
111         * @param strategy the strategy to be removed
112         * @param subpopulation the subpopulation for which the strategy must be removed
113         * @return true if the strategy was successfully removed from this manager,
114         *              false if the strategy was not part of this manager and could thus not be removed.
115         */
116        final boolean removeStrategy(
117                        final GenericPlanStrategy<PL, AG> strategy,
118                        final String subpopulation)
119        {
120                final StrategyWeights<PL, AG> weights = getStrategyWeights( subpopulation );
121                int idx = weights.strategies.indexOf(strategy);
122                if (idx != -1) {
123                        weights.strategies.remove(idx);
124                        double weight = weights.weights.remove(idx);
125                        weights.totalWeights -= weight;
126                        return true;
127                }
128                return false;
129        }

133        private StrategyWeights<PL, AG> getStrategyWeights(final String subpop) {
134                StrategyWeights<PL, AG> weights = weightsPerSubpopulation.get(subpop);
135
136                if ( weights == null ) {
137                        weights = new StrategyWeights<>();
138                        weightsPerSubpopulation.put(subpop, weights);
139                }
140
141                return weights;
142        }
143
144        /**
145         * changes the weight of the specified strategy
146         *
147         * @return true if the strategy is part of this manager and the weight could
148         *              be changed successfully, false otherwise.
149         */
150        final boolean changeWeightOfStrategy(
151                        final GenericPlanStrategy<PL, AG> strategy,
152                        final String subpopulation,
153                        final double newWeight) {
154                final StrategyWeights<PL, AG> weights = getStrategyWeights(subpopulation);
155                int idx = weights.strategies.indexOf(strategy);
156                if (idx != -1) {
157                        double oldWeight = weights.weights.set(idx, newWeight);
158                        weights.totalWeights += (newWeight - oldWeight);
159                        Logger.getLogger(this.getClass()).info( strategy.toString() + ": oldWeight=" + oldWeight + " newWeight=" + newWeight );
160                        return true;
161                }
162                return false;
163        }

165        /**
166         * Randomly chooses for each person of the population a strategy and uses that
167         * strategy on the person, after adapting the strategies to any pending change
168         * requests for the specified iteration.
169         *
170         * @param iteration the current iteration we're handling
171         */
172        public final void run(
173                        final Iterable<? extends HasPlansAndId<PL, AG>> persons,
174                                        ObjectAttributes subpopLookup,
175                                        final int iteration,
176                                        final ReplanningContext replanningContext ) {
177                handleChangeRequests(iteration);
178                run(persons, subpopLookup, replanningContext);
179        }
180
181        /**
182         * Randomly chooses for each person of the population a strategy and uses that
183         * strategy on the person.
184         *
185         */
186        final void run(
187                        final Iterable<? extends HasPlansAndId<PL, AG>> persons,
188                                        ObjectAttributes subPopLookup,
189                                        final ReplanningContext replanningContext) {
190
191                // initialize all strategies
192                for (GenericPlanStrategy<PL, AG> strategy : distinctStrategies()) {
193                        strategy.init(replanningContext);
194                }
195
196                // then go through the population and ...
197                for (HasPlansAndId<PL, AG> person : persons ) {
198
199                        // ... reduce the number of plans to the allowed maximum (in evol comp lang this is "selection")
200                        if ((this.maxPlansPerAgent > 0) && (person.getPlans().size() > this.maxPlansPerAgent)) {
201                                removePlans( person, this.maxPlansPerAgent);
202                        }
203
204                        // ... choose the strategy to be used for this person (in evol comp lang this would be the choice of the mutation operator)
205                        String subpopName = null;
206                        if (this.subpopulationAttributeName != null) {
207                                subpopName = (String) subPopLookup.getAttribute(person.getId().toString(), this.subpopulationAttributeName);
208                        }
209                        GenericPlanStrategy<PL, AG> strategy = this.chooseStrategy(person, subpopName);
210
211                        if (strategy==null) {
212                                throw new RuntimeException("No strategy found! Have you defined at least one replanning strategy per subpopulation?");
213                        }
214                        
215                        // ... and run the strategy:
216                        strategy.run(person);
217                }
218
219                // finally make sure all strategies have finished there work
220                for (GenericPlanStrategy<PL, AG> strategy : distinctStrategies()) {
221                        strategy.finish();
222                }
223
224        }
225
226        private Collection<GenericPlanStrategy<PL, AG>> distinctStrategies() {
227                // Leaving out duplicate strategies in different subpopulations
228                Collection<GenericPlanStrategy<PL, AG>> strategies = new LinkedHashSet<>();
229                for (StrategyWeights<PL, AG> weights : weightsPerSubpopulation.values()) {
230                        strategies.addAll(weights.strategies);
231                }
232                return strategies;
233        }
234
235        private void removePlans(final HasPlansAndId<PL, AG> person, final int maxNumberOfPlans) {
236                while (person.getPlans().size() > maxNumberOfPlans) {
237                        PL plan = this.removalPlanSelector.selectPlan(person);
238                        person.removePlan(plan);
239                        if (plan == person.getSelectedPlan()) {
240                                final PL newPlanToSelect = new RandomPlanSelector<PL, AG>().selectPlan(person) ;
241                                if ( newPlanToSelect == null ) {
242                                        throw new IllegalStateException( "could not find a plan to select for person "+person );
243                                }
244                                person.setSelectedPlan( newPlanToSelect );
245                        }
246                }
247        }
248
249        /**
250         * modifies the loaded strategies according to pending change requests for this iteration.
251         *
252         */
253        final void handleChangeRequests(final int iteration) {
254                for ( int ii = 0 ; ii <= iteration ; ii++ ) {
255                        // (playing back history for those installations which recreate the strategy manager in every iteration)
256                        for ( Map.Entry<String, StrategyWeights<PL, AG>> wentry : weightsPerSubpopulation.entrySet() ) {
257                                final String subpop = wentry.getKey();
258                                final StrategyWeights<PL, AG> weights = wentry.getValue();
259                                Map<GenericPlanStrategy<PL, AG>, Double> changes = weights.changeRequests.remove(ii);
260                                if (changes != null) {
261                                        for (Map.Entry<GenericPlanStrategy<PL, AG>, Double> entry : changes.entrySet()) {
262                                                changeWeightOfStrategy( entry.getKey(), subpop, entry.getValue());
263                                        }
264                                }
265                        }
266                }
267        }

269        private interface StrategyChooser <T extends BasicPlan, I extends HasPlansAndId<? extends BasicPlan, I>> {
270                GenericPlanStrategy<T,I> chooseStrategy( HasPlansAndId<T,I> person, final String subpopulation ) ;
271        }
272        private class MyStrategyChooser implements StrategyChooser<PL, AG> {
273                @Override
274                public GenericPlanStrategy<PL, AG> chooseStrategy(HasPlansAndId<PL, AG> person, String subpopulation) {
275                        StrategyWeights<PL, AG> weights = GenericStrategyManager.this.getStrategyWeights(subpopulation);
276                        
277                        double rnd = MatsimRandom.getRandom().nextDouble() * weights.totalWeights;
278                        
279                        double sum = 0.0;
280                        for (int i = 0, max = weights.weights.size(); i < max; i++) {
281                                sum += weights.weights.get(i);
282                                if (rnd <= sum) {
283                                        return weights.strategies.get(i);
284                                }
285                        }
286                        return null;
287                }
288        }
289        private MyStrategyChooser myStrategyChooser = new MyStrategyChooser() ;

313        /**
314         * Schedules a {@link #changeWeightOfStrategy(GenericPlanStrategy, String, double)} command for a later iteration. The
315         * change will take place before the strategies are applied.
316         *
317         */
318        public final void addChangeRequest(
319                        final int iteration,
320                        final GenericPlanStrategy<PL, AG> strategy,
321                        final String subpopulation,
322                        final double newWeight) {
323                final StrategyWeights<PL, AG> weights = getStrategyWeights( subpopulation );
324                Integer iter = iteration;
325                Map<GenericPlanStrategy<PL, AG>, Double> iterationRequests = weights.changeRequests.get(iter);
326                if (iterationRequests == null) {
327                        iterationRequests = new HashMap<>(3);
328                        weights.changeRequests.put(iter, iterationRequests);
329                }
330                iterationRequests.put(strategy, newWeight);
331                Logger.getLogger(this.getClass()).info( "added change request: "
332                                + " iteration=" + iter + " newWeight=" + newWeight + " strategy=" + strategy.toString() );
333        }

335        /**
336         * Sets a plan selector to be used for choosing plans for removal, if they
337         * have more plans than the specified maximum. This defaults to
338         * {@link WorstPlanForRemovalSelector}.
339         * <p></p>
340         * Thoughts about using the logit-type selectors with negative logit model scale parameter:<ul>
341         * <li> Look at one agent.
342         * <li> Assume she has the choice between <i>n</i> different plans (although fewer of them are in the MATSim choice set).
343         * <li> (Continuous) fraction <i>f(i)</i> of plan <i>i</i> develops as (master equation)
344         * <blockquote><i>
345         * df(i)/dt = - p(i) * f(i) + 1/n
346         * </i></blockquote>
347         * where <i>p(i)</i> is from the removal selector, and <i>1/n</i> makes the assumption that each possible plan is (re-)inserted with equal probability
348         * by the innovative modules.
349         * <li> Steady state solution (<i>df/dt=0</i>) <i> f(i) = 1/n * 1/p(i) </i>.
350         * <li> If <i> p(i) = e<sup>-b*U(i)</sup></i>, then <i> f(i) = e<sup>b*U(i)</sup> / n </i>.  Or in words:
351         * <i><b> If you use a logit model with a minus in front of the beta for plans removal, the resulting steady state distribution is
352         * the same logit model with normal beta.</b></i>
353         *
354         * </ul>
355         * The implication seems to be: divide the user-configured beta by two, use one half for choice and the other for plans removal.
356         * <p></p>
357         * The path size version still needs to be tested (both for choice and for plans removal).
358         *
359         *
360         * @see #setMaxPlansPerAgent(int)
361         */
362        public final void setPlanSelectorForRemoval(final PlanSelector<PL, AG> planSelector) {
363                Logger.getLogger(this.getClass()).info("setting PlanSelectorForRemoval to " + planSelector.getClass() ) ;
364                this.removalPlanSelector = planSelector;
365        }
