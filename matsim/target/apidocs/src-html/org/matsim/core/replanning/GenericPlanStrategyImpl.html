019package org.matsim.core.replanning;
020
021import org.apache.log4j.Logger;
022import org.matsim.api.core.v01.population.BasicPlan;
023import org.matsim.api.core.v01.population.HasPlansAndId;
024import org.matsim.core.replanning.modules.GenericPlanStrategyModule;
025import org.matsim.core.replanning.selectors.PlanSelector;
026import org.matsim.core.replanning.selectors.RandomUnscoredPlanSelector;
027
028import java.util.ArrayList;
029
030/**
031 * @author nagel
032 *
033 */
034public class GenericPlanStrategyImpl<T extends BasicPlan, I> implements GenericPlanStrategy<T, I> {
035
036        private PlanSelector<T, I> planSelector = null;
037        private GenericPlanStrategyModule<T> firstModule = null;
038        private final ArrayList<GenericPlanStrategyModule<T>> modules = new ArrayList<>();
039        private final ArrayList<T> plans = new ArrayList<>();
040        private long counter = 0;
041        private ReplanningContext replanningContext;
042        private final static Logger log = Logger.getLogger(PlanStrategyImpl.class);
043
044        /**
045         * Creates a new strategy using the specified planSelector.
046         *
047         */
048        public GenericPlanStrategyImpl(final PlanSelector<T, I> planSelector) {
049                this.planSelector = planSelector;
050        }
051
052        public void addStrategyModule(final GenericPlanStrategyModule<T> module) {
053                if (this.firstModule == null) {
054                        this.firstModule = module;
055                } else {
056                        this.modules.add(module);
057                }
058        }
059        
060        public int getNumberOfStrategyModules() {
061                if (this.firstModule == null) {
062                        return 0;
063                }
064                return this.modules.size() + 1; // we also have to count "firstModule", thus +1
065        }
066        
067        @Override
068        public void run(final HasPlansAndId<T, I> person) {
069                this.counter++;
070                
071                // if there is at least one unscored plan, find that one:
072                T plan = new RandomUnscoredPlanSelector<T, I>().selectPlan(person) ;
073                
074                // otherwise, find one according to selector (often defined in PlanStrategy ctor):
075                if (plan == null) {
076                        plan = this.planSelector.selectPlan(person);
077                }
078                
079                // "select" that plan:
080                if ( plan != null ) {
081                        person.setSelectedPlan(plan);
082                }
083                else {
084                        log.error( planSelector+" returned no plan: not changing selected plan for person "+person );
085                }
086
087                // if there is a "module" (i.e. "innovation"):
088                if (this.firstModule != null) {
089                        
090                        // set the working plan to a copy of the selected plan:
091                        plan = person.createCopyOfSelectedPlanAndMakeSelected();
092                        
093                        // add new plan to container that contains the plans that are handled by this PlanStrategy:
094                        this.plans.add(plan);
095
096                        // start working on this new plan:
097                        this.firstModule.handlePlan(plan);
098                }
099
100        }
101
102        @Override
103        public void init(ReplanningContext replanningContext0) {
104                this.replanningContext = replanningContext0;
105                if (this.firstModule != null) {
106                        this.firstModule.prepareReplanning(replanningContext0);
107                }
108        }
109
110        @Override
111        public void finish() {
112                if (this.firstModule != null) {
113                        // finish the first module
114                                this.firstModule.finishReplanning();
115                        // now work through the others
116                        for (GenericPlanStrategyModule<T> module : this.modules) {
117                                module.prepareReplanning(replanningContext);
118                                for (T plan : this.plans) {
119                                        module.handlePlan(plan);
120                                }
121                                module.finishReplanning();
122                        }
123                }
124                this.plans.clear();
125                log.info("Plan-Strategy finished, " + this.counter + " plans handled. Strategy: " + this.toString());
126                this.counter = 0;
127        }
128
129        @Override
130        public String toString() {
131                StringBuilder name = new StringBuilder(20);
132                name.append(this.planSelector.getClass().getSimpleName());
133                if (this.firstModule != null) {
134                        name.append('_');
135                        name.append(this.firstModule.getClass().getSimpleName());
136                        for (Object module : this.modules) {
137                                name.append('_');
138                                name.append(module.getClass().getSimpleName());
139                        }
140                }
141                return name.toString();
142        }
143
144        public PlanSelector<T, I> getPlanSelector() {
145                return planSelector;
146        }
147        
148
149}
