021package org.matsim.core.replanning.selectors;
022
023import org.apache.log4j.Logger;
024import org.matsim.api.core.v01.population.BasicPlan;
025import org.matsim.api.core.v01.population.HasPlansAndId;
026import org.matsim.core.gbl.MatsimRandom;
027
028/**
029 * Changes to another plan if that plan is better.  Probability to change depends on score difference.
030 *
031 * @author kn based on mrieser
032 */
033public final class ExpBetaPlanChanger<T extends BasicPlan, I> implements PlanSelector<T, I> {
034        private static final Logger log = Logger.getLogger(ExpBetaPlanChanger.class);
035
036        private final double beta;
037        static boolean betaWrnFlag = true ;
038        static boolean scoreWrnFlag = true ;
039
040        public ExpBetaPlanChanger(double beta) {
041                this.beta = beta;
042        }
043
044        /**
045         * Changes to another plan with a probability proportional to exp( Delta scores ).
046         * Need to think through if this goes to Nash Equilibrium or to SUE !!!
047         */
048        @Override
049        public T selectPlan(final HasPlansAndId<T, I> person) {
050                // current plan and random plan:
051                T currentPlan = person.getSelectedPlan();
052                T otherPlan = new RandomPlanSelector<T, I>().selectPlan(person);
053
054                if (currentPlan == null) {
055                        // this case should only happen when the agent has no plans at all
056                        return null;
057                }
058
059                if ((currentPlan.getScore() == null) || (otherPlan.getScore() == null)) {
060                        /* With the previous behavior, Double.NaN was returned if no score was available.
061                         * This resulted in weight=NaN below as well, and then ultimately in returning
062                         * the currentPlan---what we're doing right now as well.
063                         */
064                        if ( currentPlan.getScore()!=null && otherPlan.getScore()==null ) {
065                                if ( scoreWrnFlag ) {
066                                        log.error( "yyyyyy not switching to other plan although it needs to be explored.  "
067                                                        + "Possibly a serious bug; ask kai if you encounter this.  kai, sep'10" ) ;
068                                        scoreWrnFlag = false ;
069                                }
070                        }
071                        return currentPlan;
072                }
073                
074                // defending against NaN (which should not happen, but happens):
075                if ( currentPlan.getScore().isNaN() ) {
076                        return otherPlan ;
077                }
078                if ( otherPlan.getScore().isNaN() ) {
079                        return currentPlan ;
080                }
081                
082                double currentScore = currentPlan.getScore().doubleValue();
083                double otherScore = otherPlan.getScore().doubleValue();
084
085                if ( betaWrnFlag ) {
086                        log.warn("Would make sense to revise this once more.  See comments in code.  kai, nov08") ;
087                        /*** Gunnar says, rightly I think, that what is below hits the "0.01*weight > 1" threshold fairly quickly.
088                         *   An alternative might be to divide by exp(0.5*beta*oS)+exp(0.5*beta*cS), or the max of these two numbers.  But:
089                         *   (1) someone would need to go through the theory to make sure that we remain within what we have said before
090                         *       (convergence to logit and proba of jump between equal options = 0.01
091                         *   (2) someone would need to test if the "traffic" results are similar
092                         */
093                        betaWrnFlag = false ;
094                }
095                double weight = Math.exp( 0.5 * this.beta * (otherScore - currentScore) );
096                // (so far, this is >1 if otherScore>currentScore, and <=1 otherwise)
097                // (beta is the slope (strength) of the operation: large beta means strong reaction)
098
099                if (MatsimRandom.getRandom().nextDouble() < 0.01*weight ) { // as of now, 0.01 is hardcoded (proba to change when both
100                                                                   // scores are the same)
101                        return otherPlan;
102                }
103                return currentPlan;
104        }
105
106}
