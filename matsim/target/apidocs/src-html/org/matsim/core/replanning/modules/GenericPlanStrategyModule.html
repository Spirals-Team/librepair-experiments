001package org.matsim.core.replanning.modules;
002
003import org.matsim.api.core.v01.population.BasicPlan;
004import org.matsim.core.replanning.ReplanningContext;
005
006public interface GenericPlanStrategyModule<T extends BasicPlan> {
007
008        /**
009         * Initializes this module before handling plans. Modules using an external
010         * routine could e.g. open a file here to write the plans out and pass them to
011         * the external routines. Multi-threaded modules could initialize and start
012         * their threads in this method.
013         * @param replanningContext TODO
014         */
015        public void prepareReplanning(ReplanningContext replanningContext);
016        
017        /**
018         * Tells this module to handle the specified plan. It is not required that
019         * the plan must immediately be handled, e.g. modules calling external 
020         * routines could just collect the plans here and start the external routine
021         * in {@link #finishReplanning()}, or multi-threaded modules could just add the
022         * plan to a synchronized queue for the threads.
023         *
024         * @param plan
025         * @see #finishReplanning()
026         */
027        public void handlePlan(T plan);
028        
029        /**
030         * Indicates that no additional plans will be handed to this module and waits
031         * until this module has finished handling all plans. Modules calling external
032         * routines can call those here, or multi-threaded modules can wait here until
033         * all threads are finished with their work.
034         * 
035         * @see #handlePlan(T)
036         */
037        public void finishReplanning();
038
039}
