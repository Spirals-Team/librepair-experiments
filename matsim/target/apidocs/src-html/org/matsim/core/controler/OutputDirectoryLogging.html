021package org.matsim.core.controler;
022
023import org.apache.log4j.Appender;
024import org.apache.log4j.FileAppender;
025import org.apache.log4j.Level;
026import org.apache.log4j.Logger;
027import org.apache.log4j.spi.LoggingEvent;
028import org.matsim.core.gbl.Gbl;
029import org.matsim.core.utils.io.CollectLogMessagesAppender;
030import org.matsim.core.utils.io.IOUtils;
031
032import java.io.IOException;
033
034/**
035 * 
036 * Put MATSim logs into the output directory.
037 * This is static, like Loggers.
038 * Call initLogging with either an OutputDirectoryHierarchy or simply with a pathname.
039 * Don't forget to call closeOutputDirLogging before quitting the JVM.
040 * Call catchLogEntries() before either of the initLogging methods to memorize log entries between that call and the init
041 * call and put them into the log file. Useful if creating the log directory only happens after some set-up which
042 * allready produces log messages.
043 * 
044 * @author dgrether, michaz
045 *
046 */
047public final class OutputDirectoryLogging {
048        private OutputDirectoryLogging(){} // do not instantiate
049
050        public static final String LOGFILE = "logfile.log";
051
052        public static final String WARNLOGFILE = "logfileWarningsErrors.log";
053
054        private static Logger log = Logger.getLogger(OutputDirectoryLogging.class);
055
056        /**
057         * This variable is used to store the log4j output before it can be written
058         * to a file. This is needed to set the output directory before logging.
059         */
060        private static CollectLogMessagesAppender collectLogMessagesAppender = null;
061
062        public static void catchLogEntries() {
063                if ( collectLogMessagesAppender != null ) {
064                        // create a new instance only if there is not one yet, to allow
065                        // collecting log messages issued _before_ controller construction.
066                        // Otherwise, all log messages before the last call to this method
067                        // are lost.
068                        // td, mar. 2013
069                        return;
070                }
071
072                collectLogMessagesAppender = new CollectLogMessagesAppender();
073                Logger.getRootLogger().addAppender(collectLogMessagesAppender);
074        }
075
076        /**
077         * Initializes log4j to write log output to files in output directory.
078         * @param outputDirectoryHierarchy TODO
079         */
080        public final static void initLogging(OutputDirectoryHierarchy outputDirectoryHierarchy) {
081                if (collectLogMessagesAppender != null) {
082                        Logger.getRootLogger().removeAppender(collectLogMessagesAppender);
083                }
084                try {
085                        String outputFilename = outputDirectoryHierarchy.getOutputFilename(LOGFILE);
086                        String warnLogfileName = outputDirectoryHierarchy.getOutputFilename(WARNLOGFILE);
087                        initLogging(outputFilename, warnLogfileName);
088                } catch (IOException e) {
089                        log.error("Cannot create logfiles: " + e.getMessage());
090                        e.printStackTrace();
091                }
092        }
093
094        /**
095         * Call this method to create 2 log4j logfiles in the output directory specified as parameter.
096         * The first logfile contains all messages the second only those above log Level.WARN (Priority.WARN).
097         * After the end of the programm run it is strongly recommended to close the file logger by calling
098         * the method closeOutputDirLogging().
099         *
100         * @param outputDirectory the outputdirectory to create the files, whithout seperator at the end.
101         * @throws IOException
102         * @see IOUtils#closeOutputDirLogging()
103         * @author dgrether
104         */
105        public static void initLoggingWithOutputDirectory(final String outputDirectory) throws IOException {
106                if (collectLogMessagesAppender != null) {
107                        Logger.getRootLogger().removeAppender(collectLogMessagesAppender);
108                }
109                String logfilename = outputDirectory + System.getProperty("file.separator") + LOGFILE;
110                String warnlogfilename = outputDirectory + System.getProperty("file.separator") + WARNLOGFILE;
111                initLogging(logfilename, warnlogfilename);
112        }
113
114        private static void initLogging(String outputFilename, String warnLogfileName) throws IOException {
115                Logger root = Logger.getRootLogger();
116                final boolean appendToExistingFile = false; 
117                FileAppender appender = new FileAppender(Controler.DEFAULTLOG4JLAYOUT, outputFilename, appendToExistingFile);
118                appender.setName(LOGFILE);
119                root.addAppender(appender);
120                FileAppender warnErrorAppender = new FileAppender(Controler.DEFAULTLOG4JLAYOUT, warnLogfileName, appendToExistingFile);
121                warnErrorAppender.setName(WARNLOGFILE);
122                warnErrorAppender.setThreshold(Level.WARN);
123                root.addAppender(warnErrorAppender);
124                if (collectLogMessagesAppender != null) {
125                        for (LoggingEvent e : collectLogMessagesAppender.getLogEvents()) {
126                                appender.append(e);
127                                if (e.getLevel().isGreaterOrEqual(Level.WARN)) {
128                                        warnErrorAppender.append(e);
129                                }
130                        }
131                        collectLogMessagesAppender.close();
132                        collectLogMessagesAppender = null;
133                }
134                Gbl.printSystemInfo();
135                Gbl.printBuildInfo();
136        }
137
138        /**
139         * Call this method to close the log file streams opened by a call of initOutputDirLogging().
140         * This avoids problems concerning open streams after the termination of the program.
141         * @see IOUtils#initLoggingWithOutputDirectory(String)
142         */
143        public static void closeOutputDirLogging() {
144                //might also be sent to the warn logstream but then you end up with a warning even if everything is alright
145                String endLoggingInfo = "closing the logfile, i.e. messages sent to the logger after this message are not written to the logfile.";
146                log.info(endLoggingInfo);
147                Logger root = Logger.getRootLogger();
148                Appender app = root.getAppender(LOGFILE);
149                if (app != null) {
150                        root.removeAppender(app);
151                        app.close();
152                }
153                app = root.getAppender(WARNLOGFILE);
154                if (app != null) {
155                        root.removeAppender(app);
156                        app.close();
157                }
158        }
