067        @Override
068        public void run() {
069                {
070                        log.warn("===") ;
071                        for ( Link link : network.getLinks().values() ) {
072                                if ( !link.getAllowedModes().contains( TransportMode.car ) ) {
073                                        log.warn("link that does not allow car: " + link.toString() ) ;
074                                }
075                        }
076                        log.warn("---") ;
077                }
078                
079                /*
080                 * Create single-mode network here and hand it over to PersonPrepareForSim. Otherwise, each instance would create its
081                 * own single-mode network. However, this assumes that the main mode is car - which PersonPrepareForSim also does. Should
082                 * be probably adapted in a way that other main modes are possible as well. cdobler, oct'15.
083                 */
084                final Network carNetwork ; // for postal address
085                if (NetworkUtils.isMultimodal(network)) {
086                        log.info("Network seems to be multimodal. Create car-only network which is handed over to PersonPrepareForSim.");
087                        TransportModeNetworkFilter filter = new TransportModeNetworkFilter(network);
088                        carNetwork  = NetworkUtils.createNetwork();
089                        HashSet<String> modes = new HashSet<>();
090                        modes.add(TransportMode.car);
091                        filter.filter(carNetwork , modes);
092                } else {
093                        carNetwork  = network;
094                }
095
096                {
097                        log.warn("---") ;
098                        int ii = 0 ;
099                        for ( Link link : carNetwork.getLinks().values() ) {
100                                if ( ii < 10 ) {
101                                        ii++ ;
102                                        log.warn( link.getId() + "; " + link.getAllowedModes() );
103                                }
104                        }
105                        log.warn("===") ;
106                }
107        
108                // make sure all routes are calculated.
109                ParallelPersonAlgorithmUtils.run(population, globalConfigGroup.getNumberOfThreads(),
110                                new ParallelPersonAlgorithmUtils.PersonAlgorithmProvider() {
111                                        @Override
112                                        public AbstractPersonAlgorithm getPersonAlgorithm() {
113                                                final PlanRouter planRouter = new PlanRouter(tripRouterProvider.get(), activityFacilities);
114                                                return new PersonPrepareForSimMultimodal(planRouter, scenario, carNetwork );
115                                        }
116                                });
117
118                // though the vehicles should be created before creating a route, however,
119                // as of now, it is not clear how to provide (store) vehicle id to the route afterwards. Amit may'17
120
121                Map<String, VehicleType> modeVehicleTypes = getMode2VehicleType();
122                for(Person person : scenario.getPopulation().getPersons().values()) {
123                        for (Plan plan : person.getPlans()) { // go through with all plans ( when it was in population agent source, then going through only with selected plan was sufficient.) Amit May'17
124                                Map<String, Id<Vehicle>> seenModes = new HashMap<>();
125                                for (PlanElement planElement : plan.getPlanElements()) {
126                                        if (planElement instanceof Leg) {
127                                                Leg leg = (Leg) planElement;
128                                                if (qSimConfigGroup.getMainModes().contains(leg.getMode())) {// only simulated modes get vehicles
129                                                        NetworkRoute route = (NetworkRoute) leg.getRoute();
130                                                        Id<Vehicle> vehicleId = null;
131                                                        if (route != null) {
132                                                                vehicleId = route.getVehicleId(); // may be null!
133                                                        } else {
134                                                                throw new RuntimeException("Route not found.");
135                                                        }
136
137                                                        if (!seenModes.keySet().contains(leg.getMode())) { // create one vehicle per simulated mode, put it on the home location
138
139                                                                if (vehicleId == null) {
140                                                                        vehicleId = createAutomaticVehicleId(person.getId(), leg.getMode(), route);
141                                                                }
142
143                                                                // so here we have a vehicle id, now try to find or create a physical vehicle:
144                                                                createAndAddVehicleIfNotPresent( vehicleId, modeVehicleTypes.get(leg.getMode()));
145                                                                seenModes.put(leg.getMode(), vehicleId);
146                                                        } else {
147                                                                if (vehicleId == null) {
148                                                                        vehicleId = seenModes.get(leg.getMode());
149                                                                        route.setVehicleId(vehicleId);
150                                                                }
151                                                        }
152
153                                                }
154                                        }
155                                }
156                        }
157                }
158
159                // create vehicles and add to scenario if using mode choice. Amit July'17
160                createVehiclesInAdvance(modeVehicleTypes);
161
162                if (scenario instanceof Lockable) {
163                        ((Lockable)scenario).setLocked();
164                        // see comment in ScenarioImpl. kai, sep'14
165                }
166
167                if (population instanceof Lockable) {
168                        ((Lockable) population).setLocked();
169                }
170                
171                if ( network instanceof Lockable ) {
172                        ((Lockable) network).setLocked();
173                }
174                
175                // (yyyy means that if someone replaces prepareForSim and does not add the above lines, the containers are not locked.  kai, nov'16)
176
177        }

179        private void createVehiclesInAdvance(final Map<String, VehicleType> modeVehicleTypes) {
180                boolean isModeChoicePresent = false;
181                Collection<StrategyConfigGroup.StrategySettings> strategySettings = scenario.getConfig().strategy().getStrategySettings();
182                for (StrategyConfigGroup.StrategySettings strategySetting : strategySettings) {
183                        String name = strategySetting.getStrategyName();
184                        if ( name.equals(DefaultPlanStrategiesModule.DefaultStrategy.ChangeSingleTripMode.toString())
185                                        || name.equals(DefaultPlanStrategiesModule.DefaultStrategy.ChangeTripMode.toString())
186                                        ) {
187                                isModeChoicePresent = true;
188                        } else if (name.equals(DefaultPlanStrategiesModule.DefaultStrategy.SubtourModeChoice.toString())) {
189                                isModeChoicePresent = true;
190                                log.warn("Creating one vehicle corresponding to each network mode for every agent and parking it to the departure link. \n" +
191                                                "If this is undesirable, then write a new PrepareForSim +" +
192                                                "or, somehow get vehicles generation in your plan strategy.");
193                        }
194                }
195
196                if (isModeChoicePresent) {
197                        Collection<String> networkModes = scenario.getConfig().plansCalcRoute().getNetworkModes();
198                        for (Id<Person> personId : scenario.getPopulation().getPersons().keySet()) {
199                                for (String mode : networkModes) {
200                                        Id<Vehicle> vehicleId = createAutomaticVehicleId(personId, mode, null);
201                                        createAndAddVehicleIfNotPresent(vehicleId, modeVehicleTypes.get(mode));
202                                }
203                        }
204                }
205        }

207        private  Map<String, VehicleType> getMode2VehicleType(){
208                Map<String, VehicleType> modeVehicleTypes = new HashMap<>();
209                switch ( this.qSimConfigGroup.getVehiclesSource() ) {
210                        case defaultVehicle:
211                                for (String mode : this.qSimConfigGroup.getMainModes()) {
212                                        // initialize each mode with default vehicle type:
213                                        VehicleType defaultVehicleType = VehicleUtils.getDefaultVehicleType();
214                                        modeVehicleTypes.put(mode, defaultVehicleType);
215                                                if( scenario.getVehicles().getVehicleTypes().get(defaultVehicleType.getId())==null) {
216                                                scenario.getVehicles().addVehicleType(defaultVehicleType); // adding default vehicle type to vehicles container
217                                        }
218                                }
219                                break;
220                        case modeVehicleTypesFromVehiclesData:
221                                for (String mode : qSimConfigGroup.getMainModes()) {
222                                        VehicleType vehicleType = scenario.getVehicles().getVehicleTypes().get( Id.create(mode, VehicleType.class) ) ;
223                                        Gbl.assertNotNull(vehicleType);
224                                        modeVehicleTypes.put(mode, vehicleType );
225                                }
226                                break;
227                        case fromVehiclesData:
228                                // don't do anything
229                                break;
230                        default:
231                                throw new RuntimeException("not implemented yet.");
232                }
233                return modeVehicleTypes;
234        }

236        private Vehicle createAndAddVehicleIfNotPresent(Id<Vehicle> vehicleId, VehicleType vehicleType) {
237                // try to get vehicle from the vehicles container:
238                Vehicle vehicle = scenario.getVehicles().getVehicles().get(vehicleId);
239
240                if ( vehicle==null ) {
241                        // if it was not found, next step depends on config:
242                        switch ( qSimConfigGroup.getVehiclesSource() ) {
243                                case defaultVehicle:
244                                        vehicle = VehicleUtils.getFactory().createVehicle(vehicleId, vehicleType);
245                                        scenario.getVehicles().addVehicle(vehicle);
246                                        break;
247                                case modeVehicleTypesFromVehiclesData:
248                                        vehicle = VehicleUtils.getFactory().createVehicle(vehicleId, vehicleType);
249                                        scenario.getVehicles().addVehicle(vehicle);
250                                        break;
251                                case fromVehiclesData:
252                                        // otherwise complain:
253                                        throw new IllegalStateException("Expecting a vehicle id which is missing in the vehicles database: " + vehicleId);
254                                default:
255                                        // also complain when someone added another config option here:
256                                        throw new RuntimeException("not implemented") ;
257                        }
258                }
259                return vehicle;
260        }

262        private Id<Vehicle> createAutomaticVehicleId(Id<Person> personId, String mode, NetworkRoute route) {
263                Id<Vehicle> vehicleId ;
264                if (qSimConfigGroup.getUsePersonIdForMissingVehicleId()) {
265
266                        switch (qSimConfigGroup.getVehiclesSource()) {
267                                case defaultVehicle:
268                                case fromVehiclesData:
269                                        vehicleId = Id.createVehicleId(personId);
270                                        break;
271                                case modeVehicleTypesFromVehiclesData:
272                                        if(! mode.equals(TransportMode.car)) {
273                                                String vehIdString = personId.toString() + "_" + mode ;
274                                                vehicleId = Id.create(vehIdString, Vehicle.class);
275                                        } else {
276                                                vehicleId = Id.createVehicleId(personId);
277                                        }
278                                        break;
279                                default:
280                                        throw new RuntimeException("not implemented") ;
281                        }
282                        if(route!=null) route.setVehicleId(vehicleId);
283                } else {
284                        throw new IllegalStateException("Found a network route without a vehicle id.");
285                }
286                return vehicleId;
287        }
