019package org.matsim.core.controler;

021import java.io.PrintWriter;
022import java.io.StringWriter;
023
024import org.apache.log4j.Logger;
025import org.matsim.core.config.Config;
026import org.matsim.core.config.ConfigWriter;
027import org.matsim.core.gbl.Gbl;
028
029/**
030 * @author nagel
031 *
032 */
033public final class ControlerUtils {
034        private static final Logger log = Logger.getLogger( ControlerUtils.class ) ;
035        
036        private ControlerUtils() {} // namespace for static methods only should not be instantiated
037
038        /**
039         * Design decisions:
040         * <ul>
041         * <li>I extracted this method since it is now called <i>twice</i>: once
042         * directly after reading, and once before the iterations start. The second
043         * call seems more important, but I wanted to leave the first one there in
044         * case the program fails before that config dump. Might be put into the
045         * "unexpected shutdown hook" instead. kai, dec'10
046         *
047         * Removed the first call for now, because I am now also checking for
048         * consistency with loaded controler modules. If still desired, we can
049         * put it in the shutdown hook.. michaz aug'14
050         *
051         * </ul>
052         *
053         * @param config  TODO
054         * @param message the message that is written just before the config dump
055         */
056        public static final void checkConfigConsistencyAndWriteToLog(Config config,
057                                                                        final String message) {
058            log.info(message);
059            String newline = System.getProperty("line.separator");// use native line endings for logfile
060            StringWriter writer = new StringWriter();
061            new ConfigWriter(config).writeStream(new PrintWriter(writer), newline);
062            log.info(newline + newline + writer.getBuffer().toString());
063            log.info("Complete config dump done.");
064            log.info("Checking consistency of config...");
065            config.checkConsistency();
066            log.info("Checking consistency of config done.");
067        }
068
069        @Deprecated // use      OutputDirectoryLogging.catchLogEntries() directly.  kai, mar'18
070        public static final void initializeOutputLogging() {
071                OutputDirectoryLogging.catchLogEntries();
072        }
073
074        
075}
