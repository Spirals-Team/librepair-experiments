053public final class Injector {
054        private Injector(){} // namespace only, do not instantiate
055
056        private static Logger logger = Logger.getLogger(Injector.class);
057
058        public static com.google.inject.Injector createInjector(final Config config, Module... modules) {
059                com.google.inject.Injector bootstrapInjector = Guice.createInjector(new Module() {
060                        @Override
061                        public void configure(Binder binder) {
062                                binder.requireExplicitBindings(); // For now, we are conservative and disable this kind of magic.
063                                binder.install(new ExplodedConfigModule(config));
064                        }
065                });
066                // A MATSim module needs the config at configuration time in order to decide what
067                // features to provide. So we create a bootstrapInjector which already has the config
068                // and provides it to the MATSim modules.
069                List<com.google.inject.Module> guiceModules = new ArrayList<>();
070                for (Module module : modules) {
071                        bootstrapInjector.injectMembers(module);
072                        guiceModules.add(module);
073                }
074                com.google.inject.Injector realInjector = bootstrapInjector.createChildInjector(insertMapBindings(guiceModules));
075                printInjector(realInjector, logger);
076                return realInjector;
077        }
078
079        public static void printInjector(com.google.inject.Injector injector, Logger log) {
080                Level level = Level.INFO ;
081                log.log(level,"=== printInjector start ===") ;
082                for (Map.Entry<Key<?>, Binding<?>> entry : injector.getBindings().entrySet()) {
083                        if ( entry.getKey().toString().contains("type=org.matsim") ) {
084                                Annotation annotation = entry.getKey().getAnnotation();
085                                log.log( level, entry.getKey().getTypeLiteral() + " " + (annotation != null ? annotation.toString() : ""));
086                                log.log(level, "  --> provider: " + entry.getValue().getProvider());
087                                log.log(level, "  --> source: " + entry.getValue().getSource() );
088                                if ( entry.getValue() instanceof BindingImpl ) {
089                                        log.log( level, "  --> scope: " + ((BindingImpl<?>)entry.getValue()).getScoping() ) ;
090                                }
091                                if ( entry.getValue() instanceof LinkedKeyBinding) {
092                                        log.log( level, "  --> target: " + ((LinkedKeyBinding) entry.getValue()).getLinkedKey() ) ;
093                                }
094                                log.log(level, "  ==full==> " + entry.getValue() );
095                                // yy could probably format the above in a better way. kai, may'16
096                                log.log(level,  "" );
097                        }
098                }
099                log.log(level,"=== printInjector end ===") ;
100        }
101
102        private static Module insertMapBindings(List<Module> guiceModules) {
103                com.google.inject.AbstractModule routingModuleBindings = createMapBindingsForType(guiceModules, RoutingModule.class);
104                com.google.inject.AbstractModule travelTimeBindings = createMapBindingsForType(guiceModules, TravelTime.class);
105                com.google.inject.AbstractModule travelDisutilityFactoryBindings = createMapBindingsForType(guiceModules, TravelDisutilityFactory.class);
106                return Modules.combine(Modules.combine(guiceModules), routingModuleBindings, travelTimeBindings, travelDisutilityFactoryBindings);
107        }
108
109        private static <T> com.google.inject.AbstractModule createMapBindingsForType(List<Module> guiceModules, final Class<T> aClass) {
110                final Set<String> modes = new HashSet<>();
111                for (Element element : Elements.getElements(guiceModules)) {
112                        element.acceptVisitor(new DefaultElementVisitor<Object>() {
113                                @Override
114                                public <T> Object visit(Binding<T> binding) {
115                                        if (binding.getKey().getTypeLiteral().getRawType().equals(aClass)) {
116                                                if (binding.getKey().getAnnotation() instanceof com.google.inject.name.Named) {
117                                                        modes.add(((com.google.inject.name.Named) binding.getKey().getAnnotation()).value());
118                                                }
119                                        }
120                                        return null;
121                                }
122                        });
123                }
124                return new com.google.inject.AbstractModule() {
125                        @Override
126                        protected void configure() {
127                                MapBinder<String, T> routingModuleMultibinder = MapBinder.newMapBinder(binder(), String.class, aClass);
128                                for (String mode : modes) {
129                                        routingModuleMultibinder.addBinding(mode).to(Key.get(aClass, Names.named(mode)));
130                                }
131                        }
132                };
133        }
