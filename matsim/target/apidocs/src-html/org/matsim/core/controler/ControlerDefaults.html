019package org.matsim.core.controler;
020
021import org.matsim.api.core.v01.Scenario;
022import org.matsim.api.core.v01.TransportMode;
023import org.matsim.core.router.costcalculators.RandomizingTimeDistanceTravelDisutilityFactory;
024import org.matsim.core.router.costcalculators.TravelDisutilityFactory;
025import org.matsim.core.scoring.ScoringFunctionFactory;
026import org.matsim.core.scoring.functions.CharyparNagelScoringFunctionFactory;
027
028/**
029 * Intention of this class is to have the controler defaults clearly marked and visible.
030 * <p></p>
031 * The initial use case of this is that we want to deprecated the pattern of taking out a factory from the controler, wrapping something 
032 * around it, and putting it back in, since such things may depend on the calling sequence and may thus be unstable.  Users
033 * should then rather use the default factory (provided here) and wrap everything around it in a sequence they control themselves. 
034 * <p></p>
035 * I just renamed this from ControlerUtils to ControlerDefaults since XxxUtils is for us, in many case, the outmost user interface,
036 * and the material here IMO is not "outermost".  kai, nov'13
037 * 
038 * @author nagel
039 *
040 */
041public final class ControlerDefaults {
042        
043        private ControlerDefaults(){} // should not be instantiated
044
045        public static ScoringFunctionFactory createDefaultScoringFunctionFactory(Scenario scenario) {
046                return new CharyparNagelScoringFunctionFactory( scenario );
047        }
048
049        public static TravelDisutilityFactory createDefaultTravelDisutilityFactory(Scenario scenario) {
050                final RandomizingTimeDistanceTravelDisutilityFactory builder = new RandomizingTimeDistanceTravelDisutilityFactory( TransportMode.car, scenario.getConfig().planCalcScore() );
051                builder.setSigma(0.); // tendency to set this to 3. right away (i.e. through PlansCalcRouteConfigGroup default). kai/bk, mar'15
052                return builder;
053        }
054
055}
