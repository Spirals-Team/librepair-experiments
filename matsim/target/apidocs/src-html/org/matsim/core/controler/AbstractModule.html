060/**
061 * "Designed for inheritance."
062 * Extend this class, overwrite configure, and use the methods of this class to
063 * install your module.
064 * <p></p>
065 * See comments in subclasses.
066 * <p></p>
067 * AbstractModule is a custom version of com.google.inject.Module, with the following differences:<ul>
068 * <li> AbstractModule is a bit more restrictive.
069 * <li> AbstractModule already has the config (as getConfig() ), which is helpful since the config contains material that
070 * has to be used to configure the services.
071 * </ul>
072 *
073 * @author michaz
074 */
075public abstract class AbstractModule implements Module {
076
077        private Binder binder;
078        private Multibinder<EventHandler> eventHandlerMultibinder;
079        private Multibinder<ControlerListener> controlerListenerMultibinder;
080        private Multibinder<MobsimListener> mobsimListenerMultibinder;
081        private Multibinder<SnapshotWriter> snapshotWriterMultibinder;
082        private MapBinder<Class<?>, AttributeConverter<?>> attributeConverterMapBinder;
083
084        @Inject
085        com.google.inject.Injector bootstrapInjector;
086        private Config config;
087
088        public AbstractModule() {
089                // config will be injected later
090        }
091
092        public AbstractModule(Config config) {
093                this.config = config;
094        }
095
096        @Override
097        public final void configure(Binder binder) {
098                if (this.config == null) {
099                        this.config = bootstrapInjector.getInstance(Config.class);
100                }
101                // Guice error messages should give the code location of the error in the user's module,
102                // not in this class.
103                this.binder = binder.skipSources(AbstractModule.class);
104                this.mobsimListenerMultibinder = Multibinder.newSetBinder(this.binder, MobsimListener.class);
105                this.snapshotWriterMultibinder = Multibinder.newSetBinder(this.binder, SnapshotWriter.class);
106                this.eventHandlerMultibinder = Multibinder.newSetBinder(this.binder, EventHandler.class);
107                this.controlerListenerMultibinder = Multibinder.newSetBinder(this.binder, ControlerListener.class);
108                this.attributeConverterMapBinder =
109                                MapBinder.newMapBinder(
110                                                this.binder,
111                                                new TypeLiteral<Class<?>>(){},
112                                                new TypeLiteral<AttributeConverter<?>>() {} );
113                this.install();
114        }
115
116        public abstract void install();
117
118        protected final Config getConfig() {
119                return config;
120        }
121
122        protected final void install(Module module) {
123                bootstrapInjector.injectMembers(module);
124                binder.install(module);
125        }
126
127        protected final LinkedBindingBuilder<EventHandler> addEventHandlerBinding() {
128                return eventHandlerMultibinder.addBinding();
129        }
130
131        /**
132         * See {@link tutorial.programming.example07ControlerListener.RunControlerListenerExample} for an example.
133         * 
134         * @see ControlerListener
135         */
136        protected final LinkedBindingBuilder<ControlerListener> addControlerListenerBinding() {
137                return controlerListenerMultibinder.addBinding();
138        }
139
140        /**
141         * See {@link tutorial.programming.planStrategyForRemoval.RunPlanSelectorForRemovalExample} for an example.
142         * 
143         * @see StrategyManagerModule
144         */
145        protected final com.google.inject.binder.LinkedBindingBuilder<PlanSelector<Plan, Person>> bindPlanSelectorForRemoval() {
146                return bind(new TypeLiteral<PlanSelector<Plan, Person>>(){});
147        }
148
149        protected final com.google.inject.binder.LinkedBindingBuilder<PlanStrategy> addPlanStrategyBinding(String selectorName) {
150                return binder().bind(PlanStrategy.class).annotatedWith(Names.named(selectorName));
151        }
152
153        protected final com.google.inject.binder.LinkedBindingBuilder<Mobsim> bindMobsim() {
154                return bind(Mobsim.class);
155        }
156
157        protected final com.google.inject.binder.LinkedBindingBuilder<ScoringFunctionFactory> bindScoringFunctionFactory() {
158                return bind(ScoringFunctionFactory.class);
159        }
160
161        protected final com.google.inject.binder.LinkedBindingBuilder<MobsimListener> addMobsimListenerBinding() {
162                return mobsimListenerMultibinder.addBinding();
163        }
164
165        protected final com.google.inject.binder.LinkedBindingBuilder<SnapshotWriter> addSnapshotWriterBinding() {
166                return snapshotWriterMultibinder.addBinding();
167        }
168
169        protected final LinkedBindingBuilder<AttributeConverter<?>> addAttributeConverterBinding(final Class<?> clazz ) {
170                return attributeConverterMapBinder.addBinding( clazz );
171        }
172
173        protected final com.google.inject.binder.LinkedBindingBuilder<TravelDisutilityFactory> bindCarTravelDisutilityFactory() {
174                return bind(carTravelDisutilityFactoryKey());
175        }
176
177        @SuppressWarnings("static-method")
178        protected final Key<TravelDisutilityFactory> carTravelDisutilityFactoryKey() {
179                return Key.get(TravelDisutilityFactory.class, Names.named(TransportMode.car));
180        }
181
182        protected final com.google.inject.binder.LinkedBindingBuilder<TravelDisutilityFactory> addTravelDisutilityFactoryBinding(String mode) {
183                return binder().bind(TravelDisutilityFactory.class).annotatedWith(Names.named(mode));
184        }
185
186        protected final com.google.inject.binder.LinkedBindingBuilder<LeastCostPathCalculatorFactory> bindLeastCostPathCalculatorFactory() {
187                return bind(LeastCostPathCalculatorFactory.class);
188        }
189
190        protected final com.google.inject.binder.LinkedBindingBuilder<TravelTime> addTravelTimeBinding(String mode) {
191                return binder().bind(TravelTime.class).annotatedWith(Names.named(mode));
192        }
193
194        protected final LinkedBindingBuilder<RoutingModule> addRoutingModuleBinding(String mode) {
195                return binder().bind(RoutingModule.class).annotatedWith(Names.named(mode));
196        }
197
198        protected final com.google.inject.binder.LinkedBindingBuilder<EventsManager> bindEventsManager() {
199                return binder().bind(EventsManager.class);
200        }
201
202        protected final LinkedBindingBuilder<TravelTime> bindNetworkTravelTime() {
203                return bind(networkTravelTime());
204        }
205
206        @SuppressWarnings("static-method")
207        protected final Key<TravelTime> networkTravelTime() {
208                return Key.get(TravelTime.class, Names.named(TransportMode.car));
209        }
210
211        protected <T> AnnotatedBindingBuilder<T> bind(Class<T> aClass) {
212                return binder.bind(aClass);
213        }
214
215        protected <T> AnnotatedBindingBuilder<T> bind(TypeLiteral<T> typeLiteral) {
216                return binder.bind(typeLiteral);
217        }
218
219        protected <T> LinkedBindingBuilder<T> bind(Key<T> key) {
220                return binder.bind(key);
221        }
222
223        protected final Binder binder() {
224                return binder;
225        }
226
227        protected final <T> javax.inject.Provider<T> getProvider(TypeLiteral<T> typeLiteral) {
228                return binder.getProvider(Key.get(typeLiteral));
229        }
230
231        public static AbstractModule override(final Iterable<? extends AbstractModule> modules, final AbstractModule abstractModule) {
232                return new AbstractModule() {
233                        @Override
234                        public void install() {
235                                final List<com.google.inject.Module> guiceModules = new ArrayList<>();
236                                for (AbstractModule module : modules) {
237                                        bootstrapInjector.injectMembers(module);
238                                        guiceModules.add(module);
239                                }
240                                bootstrapInjector.injectMembers(abstractModule);
241                                binder().install(Modules.override(guiceModules).with(abstractModule));
242                        }
243                };
244        }
245
246        public static AbstractModule emptyModule() {
247                return new AbstractModule() {
248                        @Override
249                        public void install() {}
250                };
251        }
