055/**
056 * Calculates actual travel times on link from events and optionally also the link-to-link 
057 * travel times, e.g. if signaled nodes are used and thus turns in different directions
058 * at a node may take a different amount of time.
059 * <br>
060 * Travel times on links are collected and averaged in bins/slots with a specified size
061 * (<code>binSize</code>, in seconds, default 900 seconds = 15 minutes). The data for the travel times per link
062 * is stored in {@link TravelTimeData}-objects. If a short binSize is used, it is useful to
063 * use {@link TravelTimeDataHashMap} (see {@link #setTravelTimeDataFactory(TravelTimeDataFactory)}
064 * as that one does not use any memory to time bins where no traffic occurred. By default,
065 * {@link TravelTimeDataArray} is used.
066 * 
067 * @author dgrether
068 * @author mrieser
069 */
070public class TravelTimeCalculator implements LinkEnterEventHandler, LinkLeaveEventHandler, 
071        VehicleEntersTrafficEventHandler, VehicleLeavesTrafficEventHandler, VehicleArrivesAtFacilityEventHandler, 
072        VehicleAbortsEventHandler {

110        static TravelTimeCalculator configure(TravelTimeCalculator calculator, TravelTimeCalculatorConfigGroup config, Network network) {
111                // Customize micro-behavior of the TravelTimeCalculator based on config. Should not be necessary for most use cases.
112                switch ( config.getTravelTimeCalculatorType() ) {
113                        case TravelTimeCalculatorArray:
114                                calculator.setTravelTimeDataFactory(new TravelTimeDataArrayFactory(network, calculator.numSlots));
115                                break;
116                        case TravelTimeCalculatorHashMap:
117                                calculator.setTravelTimeDataFactory(new TravelTimeDataHashMapFactory(network));
118                                break;
119                        default:
120                                throw new RuntimeException(config.getTravelTimeCalculatorType() + " is unknown!");
121                }
122
123                AbstractTravelTimeAggregator travelTimeAggregator;
124                switch( config.getTravelTimeAggregatorType() ) {
125                        case "optimistic":
126                                travelTimeAggregator = new OptimisticTravelTimeAggregator(calculator.numSlots, calculator.timeSlice);
127                                break ;
128                        case "experimental_LastMile":
129                                travelTimeAggregator = new PessimisticTravelTimeAggregator(calculator.numSlots, calculator.timeSlice);
130                                break ;
131                        default:
132                                throw new RuntimeException(config.getTravelTimeAggregatorType() + " is unknown!");
133                }
134                calculator.setTravelTimeAggregator(travelTimeAggregator);
135
136                TravelTimeGetter travelTimeGetter;
137                switch( config.getTravelTimeGetterType() ) {
138                        case "average":
139                                travelTimeGetter = new AveragingTravelTimeGetter();
140                                break ;
141                        case "linearinterpolation":
142                                travelTimeGetter = new LinearInterpolatingTravelTimeGetter(calculator.numSlots, calculator.timeSlice);
143                                break ;
144                        default:
145                                throw new RuntimeException(config.getTravelTimeGetterType() + " is unknown!");
146                }
147                travelTimeAggregator.connectTravelTimeGetter(travelTimeGetter);
148                return calculator;
149        }

151        @Inject
152        TravelTimeCalculator(TravelTimeCalculatorConfigGroup ttconfigGroup, EventsManager eventsManager, Network network) {
153                // this injected constructor is not used when getSeparateModes is true
154                this(network, ttconfigGroup.getTraveltimeBinSize(), ttconfigGroup.getMaxTime(), ttconfigGroup.isCalculateLinkTravelTimes(), ttconfigGroup.isCalculateLinkToLinkTravelTimes(), ttconfigGroup.isFilterModes(), CollectionUtils.stringToSet(ttconfigGroup.getAnalyzedModes()));
155                eventsManager.addHandler(this);
156                configure(this, ttconfigGroup, network);
157        }

163        public TravelTimeCalculator(final Network network, final int timeslice, final int maxTime, TravelTimeCalculatorConfigGroup ttconfigGroup) {
164                this(network, timeslice, maxTime, ttconfigGroup.isCalculateLinkTravelTimes(), ttconfigGroup.isCalculateLinkToLinkTravelTimes(), ttconfigGroup.isFilterModes(), CollectionUtils.stringToSet(ttconfigGroup.getAnalyzedModes()));

167        TravelTimeCalculator(final Network network, final int timeslice, final int maxTime,
168                                                                boolean calculateLinkTravelTimes, boolean calculateLinkToLinkTravelTimes, boolean filterModes, Set<String> analyzedModes) {
169                this.calculateLinkTravelTimes = calculateLinkTravelTimes;
170                this.calculateLinkToLinkTravelTimes = calculateLinkToLinkTravelTimes;
171                this.filterAnalyzedModes = filterModes;
172                this.analyzedModes = analyzedModes;
173                this.timeSlice = timeslice;
174                this.numSlots = TimeBinUtils.getTimeBinCount(maxTime, timeslice);
175                this.aggregator = new OptimisticTravelTimeAggregator(this.numSlots, this.timeSlice);
176                this.ttDataFactory = new TravelTimeDataArrayFactory(network, this.numSlots);
177                if (this.calculateLinkTravelTimes){
178                        this.linkData = new ConcurrentHashMap<>((int) (network.getLinks().size() * 1.4));
179
180                        /*
181                         * So far, link data objects were stored in a HashMap. This lookup strategy is used
182                         * by a MapBasedDataContainerProvider.
183                         * When ArrayRoutingNetworks are used (as the FastRouter implementations do), the
184                         * getArrayIndex() methods from the RoutingLinks can be used to lookup the link
185                         * data objects in an array. This approach is implemented by the ArrayBasedDataContainerProvider.
186                         * Using a ArrayBasedDataContainerProvider instead of a MapBasedDataContainerProvider
187                         * increases the routing performance by 20-30%.
188                         * cdobler, oct'13
189                         */
190                        //              this.dataContainerProvider = new MapBasedDataContainerProvider(linkData, ttDataFactory);
191                        this.dataContainerProvider = new ArrayBasedDataContainerProvider(linkData, ttDataFactory, network);
192                } else this.dataContainerProvider = null;
193                if (this.calculateLinkToLinkTravelTimes){
194                        // assume that every link has 2 outgoing links as default
195                        this.linkToLinkData = new ConcurrentHashMap<>((int) (network.getLinks().size() * 1.4 * 2));
196                }
197                this.linkEnterEvents = new ConcurrentHashMap<>();
198
199                // if we just look at one mode, we need to ignore all vehicles with a different mode. However, the info re the mode is only in
200                // the vehicleEntersTraffic event.  So we need to memorize the ignored vehicles from there ...
201                this.vehiclesToIgnore = new HashSet<>();
202
203
204                this.reset(0);
205
206        }

208        @Override
209        public void handleEvent(final LinkEnterEvent e) {
210                /* if only some modes are analyzed, we check whether the vehicles
211                 * performs a trip with one of those modes. if not, we skip the event. */
212                if (filterAnalyzedModes && vehiclesToIgnore.contains(e.getVehicleId())) return;
213
214                LinkEnterEvent oldEvent = this.linkEnterEvents.remove(e.getVehicleId());
215                if ((oldEvent != null) && this.calculateLinkToLinkTravelTimes) {
216                        Tuple<Id<Link>, Id<Link>> fromToLink = new Tuple<>(oldEvent.getLinkId(), e.getLinkId());
217                        DataContainer data = getLinkToLinkTravelTimeData(fromToLink, true);
218                        this.aggregator.addTravelTime(data.ttData, oldEvent.getTime(), e.getTime());
219                        data.needsConsolidation = true;
220                }
221                this.linkEnterEvents.put(e.getVehicleId(), e);
222        }
223
224        @Override
225        public void handleEvent(final LinkLeaveEvent e) {
226                if (this.calculateLinkTravelTimes) {
227                        LinkEnterEvent oldEvent = this.linkEnterEvents.get(e.getVehicleId());
228                        if (oldEvent != null) {
229                                DataContainer data = this.dataContainerProvider.getTravelTimeData(e.getLinkId(), true);
230                                this.aggregator.addTravelTime(data.ttData, oldEvent.getTime(), e.getTime());
231                                data.needsConsolidation = true;
232                        }
233                }
234        }
235
236        @Override
237        public void handleEvent(VehicleEntersTrafficEvent event) {
238                /* if filtering transport modes is enabled and the vehicles
239                 * starts a leg on a non analyzed transport mode, add the vehicle 
240                 * to the filtered vehicles set. */
241                if (filterAnalyzedModes && !analyzedModes.contains(event.getNetworkMode())) { 
242                        this.vehiclesToIgnore.add(event.getVehicleId());
243                }
244        }
245
246        @Override
247        public void handleEvent(final VehicleLeavesTrafficEvent event) {
248                /* remove EnterEvents from list when a vehicle arrives.
249                 * otherwise, the activity duration would counted as travel time, when the
250                 * vehicle departs again and leaves the link! */
251                this.linkEnterEvents.remove(event.getVehicleId());
252
253                // try to remove vehicles from set with filtered vehicles
254                if (filterAnalyzedModes) this.vehiclesToIgnore.remove(event.getVehicleId());
255        }
256
257        @Override
258        public void handleEvent(VehicleArrivesAtFacilityEvent event) {
259                /* remove EnterEvents from list when a bus stops on a link.
260                 * otherwise, the stop time would counted as travel time, when the
261                 * bus departs again and leaves the link! */
262                this.linkEnterEvents.remove(event.getVehicleId());
263        }
264
265        @Override
266        public void handleEvent(VehicleAbortsEvent event) {
267                LinkEnterEvent e = this.linkEnterEvents.remove(event.getVehicleId());
268                if (e != null) {
269                        DataContainer data = this.dataContainerProvider.getTravelTimeData(e.getLinkId(), true);
270                        data.needsConsolidation = true;
271                        this.aggregator.addStuckEventTravelTime(data.ttData, e.getTime(), event.getTime());
272                        if (this.calculateLinkToLinkTravelTimes 
273                                        && event.getTime() < qsimConfig.getEndTime() 
274                                        // (we think that this only makes problems when the abort is not just because of mobsim end time. kai & theresa, jan'17) 
275                                        ){
276                                log.error(ERROR_STUCK_AND_LINKTOLINK);
277                                throw new IllegalStateException(ERROR_STUCK_AND_LINKTOLINK);
278                        }
279                }
280
281                // try to remove vehicle from set with filtered vehicles
282                if (filterAnalyzedModes) this.vehiclesToIgnore.remove(event.getVehicleId());
283        }

285        private DataContainer getLinkToLinkTravelTimeData(Tuple<Id<Link>, Id<Link>> fromLinkToLink, final boolean createIfMissing) {
286                DataContainer data = this.linkToLinkData.get(fromLinkToLink);
287                if ((null == data) && createIfMissing) {
288                        data = new DataContainer(this.ttDataFactory.createTravelTimeData(fromLinkToLink.getFirst()));
289                        this.linkToLinkData.put(fromLinkToLink, data);
290                }
291                return data;
292        }

341        public double getLinkToLinkTravelTime(final Id<Link> fromLinkId, final Id<Link> toLinkId, double time) {
342                if (!this.calculateLinkToLinkTravelTimes) {
343                        throw new IllegalStateException("No link to link travel time is available " +
344                                        "if calculation is switched off by config option!");
345                }
346                DataContainer data = this.getLinkToLinkTravelTimeData(new Tuple<>(fromLinkId, toLinkId), true);
347                if (data.needsConsolidation) {
348                        consolidateData(data);
349                }
350                return this.aggregator.getTravelTime(data.ttData, time);
351        }

353        @Override
354        public void reset(int iteration) {
355                if (this.calculateLinkTravelTimes) {
356                        for (DataContainer data : this.linkData.values()){
357                                data.ttData.resetTravelTimes();
358                                data.needsConsolidation = false;
359                        }
360                }
361                if (this.calculateLinkToLinkTravelTimes){
362                        for (DataContainer data : this.linkToLinkData.values()){
363                                data.ttData.resetTravelTimes();
364                                data.needsConsolidation = false;
365                        }
366                }
367                this.linkEnterEvents.clear();
368                this.vehiclesToIgnore.clear();
369        }

379        /**
380         * Makes sure that the travel times "make sense".
381         * <p></p>
382         * Imagine short bin sizes (e.g. 5min), small links (e.g. 300 veh/hour)
383         * and small sample sizes (e.g. 2%). This would mean that effectively
384         * in the simulation only 6 vehicles can pass the link in one hour,
385         * one every 10min. So, the travel time in one time slot could be 
386         * >= 10min if two cars enter the link at the same time. If no car
387         * enters in the next time bin, the travel time in that time bin should
388         * still be >=5 minutes (10min - binSize), and not freespeedTraveltime,
389         * because actually every car entering the link in this bin will be behind
390         * the car entered before, which still needs >=5min until it can leave.
391         * <p></p>
392         * This method ensures that the travel time in a time bin
393         * cannot be smaller than the travel time in the bin before minus the
394         * bin size.
395         * 
396         */
397        private void consolidateData(final DataContainer data) {
398                synchronized(data) {
399                        if (data.needsConsolidation) {
400                                TravelTimeData r = data.ttData;
401
402                                // initialize prevTravelTime with ttime from time bin 0 and time 0.  (The interface comment already states that
403                                // having both as argument does not make sense.)
404                                double prevTravelTime = r.getTravelTime(0, 0.0);
405                                // changed (1, 0.0) to (0, 0.0) since Michal has convinced me (by a test) that using "1" is wrong
406                                // because you get the wrong result for time slot number 1.  This change does not affect the existing
407                                // unit tests.  kai, oct'11
408
409                                // go from time slot 1 forward in time:
410                                for (int i = 1; i < this.numSlots; i++) {
411
412                                        // once more the getter is weird since it needs both the time slot and the time:
413                                        double travelTime = r.getTravelTime(i, i * this.timeSlice);
414
415                                        // if the travel time in the previous time slice was X, then now it is X-S, where S is the time slice:
416                                        double minTravelTime = prevTravelTime - this.timeSlice;
417
418                                        // if the travel time that has been measured so far is less than that minimum travel time, then do something:
419                                        if (travelTime < minTravelTime) {
420
421                                                r.setTravelTime(i, minTravelTime);
422                                                // (set the travel time to the smallest possible travel time that makes sense according to the argument above)
423
424                                        } 
425                                        prevTravelTime = r.getTravelTime(i, i * this.timeSlice ) ;
426                                }
427                                data.needsConsolidation = false;
428                        }
429                }
430        }
