021package org.matsim.core.trafficmonitoring;
022
023import java.util.HashMap;
024
025import org.apache.log4j.Logger;
026import org.matsim.api.core.v01.Id;
027import org.matsim.api.core.v01.events.LinkLeaveEvent;
028import org.matsim.api.core.v01.events.PersonDepartureEvent;
029import org.matsim.api.core.v01.events.VehicleLeavesTrafficEvent;
030import org.matsim.api.core.v01.events.VehicleEntersTrafficEvent;
031import org.matsim.api.core.v01.events.handler.LinkLeaveEventHandler;
032import org.matsim.api.core.v01.events.handler.PersonDepartureEventHandler;
033import org.matsim.api.core.v01.events.handler.VehicleLeavesTrafficEventHandler;
034import org.matsim.api.core.v01.events.handler.VehicleEntersTrafficEventHandler;
035import org.matsim.api.core.v01.network.Link;
036import org.matsim.api.core.v01.network.Network;
037import org.matsim.core.events.algorithms.Vehicle2DriverEventHandler;
038
039/**
040 * Computes average departure delay on a link in a given time slot.
041 *
042 * @author meisterk
043 */
044public class DepartureDelayAverageCalculator implements PersonDepartureEventHandler, LinkLeaveEventHandler, VehicleEntersTrafficEventHandler, VehicleLeavesTrafficEventHandler {
045
046        private Network network;
047        private int timeBinSize;
048        private HashMap<DepartureEvent, Double> departureEventsTimes = new HashMap<DepartureEvent, Double>();
049        private final HashMap<Id<Link>, DepartureDelayData> linkData;
050        
051        private Vehicle2DriverEventHandler delegate = new Vehicle2DriverEventHandler();
052        
053        private static final Logger log = Logger.getLogger(DepartureDelayAverageCalculator.class);
054
055        //////////////////////////////////////////////////////////////////////
056        // Constructor
057        //////////////////////////////////////////////////////////////////////
058
059        public DepartureDelayAverageCalculator(Network network, int timeBinSize) {
060                super();
061                this.network = network;
062                this.timeBinSize = timeBinSize;
063                this.linkData = new HashMap<>(this.network.getLinks().size());
064                this.resetDepartureDelays();
065        }
066
067        /**
068         * get the departure delay estimation for a given departure time HERE
069         *
070         * @param link
071         * @param departureTime
072         * @return departure delay estimation
073         */
074        public double getLinkDepartureDelay(Id<Link> linkId, double departureTime) {
075                DepartureDelayData ddd = this.getDepartureDelayRole(linkId);
076                if (ddd == null) {
077                        return 0.0;
078                }
079                return ddd.getDepartureDelay(departureTime);
080        }
081
082        //////////////////////////////////////////////////////////////////////
083        // Implementation of link role
084        //////////////////////////////////////////////////////////////////////
085
086        /*package*/ class DepartureDelayData {
087                private double[] timeSum = null;
088                private int[] timeCnt = null;
089
090                private DepartureDelayData() {
091                        this.resetDepartureDelays();
092                }
093
094                private int getTimeSlotIndex(double time) {
095                        int slice = (int)(time/DepartureDelayAverageCalculator.this.timeBinSize);
096                        if (slice >= timeSum.length) {
097                                slice = timeSum.length - 1;
098                        }
099                        return slice;
100                }
101
102                public void addDepartureDelay(double departureTime, double departureDelay) {
103                        int index = getTimeSlotIndex(departureTime);
104                        this.timeSum[index] += departureDelay;
105                        this.timeCnt[index]++;
106                }
107
108                public double getDepartureDelay(double time) {
109                        double departureDelay = 0.0;
110
111                        try {
112                                int index = getTimeSlotIndex(time);
113                                double sum = 0.0;
114                                sum = this.timeSum[index];
115                                if (sum > 0.0) {
116                                        int cnt = this.timeCnt[index];
117                                        if (cnt > 0) {
118                                                departureDelay = sum / cnt;
119                                        }
120                                }
121                        } catch (ArrayIndexOutOfBoundsException e) {
122                                log.warn("A departure delay for an invalid value of time was requested. Returning departureDelay = 0.0. time = " + Double.toString(time));
123                        }
124
125                        return departureDelay;
126
127                }
128
129                public void resetDepartureDelays() {
130                        int nofSlots = ((27*3600)/DepartureDelayAverageCalculator.this.timeBinSize);    // default number of slots
131                        this.timeSum = new double[nofSlots];
132                        this.timeCnt = new int[nofSlots];
133                }
134
135        }
136
137        private DepartureDelayData getDepartureDelayRole(Id<Link> linkId) {
138                return this.linkData.get(linkId);
139        }
140
141        @Override
142        public void handleEvent(PersonDepartureEvent event) {
143                DepartureEvent depEvent = new DepartureEvent(event.getPersonId());
144                this.departureEventsTimes.put(depEvent, event.getTime());
145        }
146
147        @Override
148        public void handleEvent(LinkLeaveEvent event) {
149                DepartureEvent removeMe = new DepartureEvent(delegate.getDriverOfVehicle(event.getVehicleId()));
150                Double departureTime = departureEventsTimes.remove(removeMe);
151                if (departureTime != null) {
152                        double departureDelay = event.getTime() - departureTime.intValue();
153                        if (departureDelay < 0) {
154                                throw new RuntimeException("departureDelay cannot be < 0.");
155                        }
156                        Id<Link> linkId = event.getLinkId();
157                        DepartureDelayData ddd = this.getDepartureDelayRole(linkId);
158                        if (ddd == null) {
159                                ddd = new DepartureDelayData();
160                                this.linkData.put(linkId, ddd);
161                        }
162                        ddd.addDepartureDelay(departureTime, departureDelay);
163                }
164        }
165
166        public void resetDepartureDelays() {
167                this.linkData.clear();
168                this.departureEventsTimes.clear();
169        }
170
171        @Override
172        public void reset(int iteration) {
173                resetDepartureDelays();
174                delegate.reset(iteration);
175        }
176
177        @Override
178        public String toString() {
179                return this.getClass().getSimpleName();
180        }
181
182        @Override
183        public void handleEvent(VehicleLeavesTrafficEvent event) {
184                delegate.handleEvent(event);
185        }
186
187        @Override
188        public void handleEvent(VehicleEntersTrafficEvent event) {
189                delegate.handleEvent(event);
190        }
191
192}
