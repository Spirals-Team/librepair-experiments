021package org.matsim.core.trafficmonitoring;
022
023import org.apache.log4j.Level;
024import org.apache.log4j.Logger;
025import org.matsim.api.core.v01.Id;
026import org.matsim.api.core.v01.network.Link;
027import org.matsim.api.core.v01.population.Person;
028import org.matsim.core.router.util.TravelTime;
029import org.matsim.vehicles.Vehicle;
030
031import javax.inject.Inject;
032
033/**
034 * Calculates and returns the FreeSpeedTravelTime on a link at the given time.
035 * @author cdobler
036 */
037public class FreeSpeedTravelTime implements TravelTime {
038        private static final Logger log = Logger.getLogger(FreeSpeedTravelTime.class ) ;
039        
040        @Inject
041        public FreeSpeedTravelTime() {
042        }
043        
044        @Override
045        public double getLinkTravelTime(Link link, double time, Person person, Vehicle vehicle) {
046                return link.getLength() / link.getFreespeed(time);
047        }
048
049}
