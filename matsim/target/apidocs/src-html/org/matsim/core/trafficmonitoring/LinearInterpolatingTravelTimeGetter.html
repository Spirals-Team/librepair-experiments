021package org.matsim.core.trafficmonitoring;
022
023public class LinearInterpolatingTravelTimeGetter implements TravelTimeGetter {
024
025        private AbstractTravelTimeAggregator travelTimeAggregator = null;
026        private final int numSlots;
027        private final int travelTimeBinSize;
028        private final double halfBinSize;
029        
030        public LinearInterpolatingTravelTimeGetter(int numSlots, int travelTimeBinSize) {
031                this.numSlots = numSlots;
032                this.travelTimeBinSize = travelTimeBinSize;
033                this.halfBinSize = ((double) travelTimeBinSize) / 2;
034        }
035        
036        @Override
037        public void setTravelTimeAggregator(AbstractTravelTimeAggregator travelTimeAggregator) {
038                this.travelTimeAggregator = travelTimeAggregator;               
039        }
040        
041        @Override
042        public double getTravelTime(TravelTimeData travelTimeData, double time) {
043                final int timeSlot = travelTimeAggregator.getTimeSlotIndex(time);
044                
045                // if time is in the first half of the first slot we do not interpolate
046                if (time <= halfBinSize) return travelTimeData.getTravelTime(timeSlot, time);
047                
048                // if time is in the second half of the last slot we do not interpolate
049                else if (time >= numSlots * travelTimeBinSize - halfBinSize) return travelTimeData.getTravelTime(timeSlot, time);
050                
051                
052                // time is inbetween, therefore we interpolate
053                int firstSlot; 
054                int secondSlot;
055                
056                // if time lies in the first half of the time slot
057                if (timeSlot * travelTimeBinSize + halfBinSize > time) {
058                        firstSlot = timeSlot - 1;
059                        secondSlot = timeSlot;
060                } else {
061                        firstSlot = timeSlot;
062                        secondSlot = timeSlot + 1;
063                }
064
065                // calculate travel times for both time slots
066                double firstTravelTime = travelTimeData.getTravelTime(firstSlot, time); 
067                double secondTravelTime = travelTimeData.getTravelTime(secondSlot, time);
068                
069                // interpolate travel time
070                double dx = time - (firstSlot * travelTimeBinSize + halfBinSize); 
071//              double dy = (secondTravelTime - firstTravelTime) * (travelTimeBinSize - dx) / travelTimeBinSize;
072                /*
073                 * The line above was a bug: we need to divide the change in travel time (i.e. second minus first travel time) by the change in time (i.e. the time bin size) 
074                 * to get the gradient of the line between the midpoint of the first interval and the midpoint of the second interval. 
075                 * Then we have to multiply it by dx to get dy. The last part was wrong (it was multiplied by time bin size minus dx).
076                 * The test (TravelTimeCalculatorTest) only tests one value in each time bin -- namely the starting time of each time bin, which is unfortunately 
077                 * always the midpoint on the interpolated lines, i.e. the only time per time bin where the bug has no influence... theresa, sep'17
078                 */
079                double dy = (secondTravelTime - firstTravelTime) * dx / travelTimeBinSize;
080                
081                return firstTravelTime + dy;
082        }
083
084}
