023package org.matsim.core.trafficmonitoring;
024
025import com.google.inject.Injector;
026import com.google.inject.Key;
027import com.google.inject.Singleton;
028import com.google.inject.name.Names;
029
030import org.matsim.api.core.v01.network.Network;
031import org.matsim.core.api.experimental.events.EventsManager;
032import org.matsim.core.config.groups.TravelTimeCalculatorConfigGroup;
033import org.matsim.core.controler.AbstractModule;
034import org.matsim.core.router.util.LinkToLinkTravelTime;
035import org.matsim.core.router.util.TravelTime;
036import org.matsim.core.utils.collections.CollectionUtils;
037
038import javax.inject.Inject;
039import javax.inject.Provider;
040
041
042/**
043 * The first module.
044 * Mind that it is retrofitted and I didn't touch the classes themselves.
045 * For new code, we wouldn't need the extra layer of Providers.
046 *
047 * @author michaz
048 */
049public class TravelTimeCalculatorModule extends AbstractModule {
050
051        @Override
052        public void install() {
053                if (getConfig().travelTimeCalculator().getSeparateModes()) {
054                        if (getConfig().travelTimeCalculator().isCalculateLinkToLinkTravelTimes()) {
055                                throw new RuntimeException("separate modes together with link2link routing currently not implemented. doesn't look difficult, "
056                                                + "but I cannot say if it would be picked up correctly by downstream modules.  kai, nov'16") ;
057                        }                       
058                        // go through all modes:
059                        for (final String mode : CollectionUtils.stringToSet(getConfig().travelTimeCalculator().getAnalyzedModes())) {
060                                
061                                // generate and bind the observer:
062                                bind(TravelTimeCalculator.class).annotatedWith(Names.named(mode)).toProvider(new SingleModeTravelTimeCalculatorProvider(mode)).in(Singleton.class);
063
064                                // bind the observer to travel time provider (for router):
065                                addTravelTimeBinding(mode).toProvider(new Provider<TravelTime>() {
066                                        @Inject Injector injector;
067                                        @Override public TravelTime get() {
068                                                return injector.getInstance(Key.get(TravelTimeCalculator.class, Names.named(mode))).getLinkTravelTimes();
069                                        }
070                                });
071                        }
072                } else {
073                        // (all analyzed modes are measured together, and the same result is returned to each mode)
074                        
075                        // bind the TravelTimeCalculator, which is the observer and aggregator:
076                        bind(TravelTimeCalculator.class).in(Singleton.class);
077                        
078                        // bind the TravelTime objects.  In this case, this just passes on the same information from TravelTimeCalculator to each individual mode:
079                        if (getConfig().travelTimeCalculator().isCalculateLinkTravelTimes()) {
080                                for (String mode : CollectionUtils.stringToSet(getConfig().travelTimeCalculator().getAnalyzedModes())) {
081                                        addTravelTimeBinding(mode).toProvider(ObservedLinkTravelTimes.class);
082                                }
083                        }
084                        if (getConfig().travelTimeCalculator().isCalculateLinkToLinkTravelTimes()) {
085                                bind(LinkToLinkTravelTime.class).toProvider(ObservedLinkToLinkTravelTimes.class);
086                        }
087                }
088        }
089
090        private static class SingleModeTravelTimeCalculatorProvider implements Provider<TravelTimeCalculator> {
091
092                @Inject TravelTimeCalculatorConfigGroup config;
093                @Inject EventsManager eventsManager;
094                @Inject Network network;
095
096                private String mode;
097
098                SingleModeTravelTimeCalculatorProvider(String mode) {
099                        this.mode = mode;
100                }
101
102                @Override
103                public TravelTimeCalculator get() {
104                        TravelTimeCalculator calculator = new TravelTimeCalculator(network, config.getTraveltimeBinSize(), config.getMaxTime(), 
105                                        config.isCalculateLinkTravelTimes(), config.isCalculateLinkToLinkTravelTimes(), true, CollectionUtils.stringToSet(mode));
106                        eventsManager.addHandler(calculator);
107                        return TravelTimeCalculator.configure(calculator, config, network);
108                }
109        }
110
111}
