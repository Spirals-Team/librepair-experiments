021package org.matsim.core.trafficmonitoring;
022
023import java.util.Map;
024
025import org.matsim.api.core.v01.Id;
026import org.matsim.api.core.v01.network.Link;
027import org.matsim.core.trafficmonitoring.TravelTimeCalculator.DataContainer;
028
029public class MapBasedDataContainerProvider implements DataContainerProvider {
030
031        private final Map<Id<Link>, DataContainer> linkData;
032        private final TravelTimeDataFactory ttDataFactory;
033        
034        public MapBasedDataContainerProvider(Map<Id<Link>, DataContainer> linkData, TravelTimeDataFactory ttDataFactory) {
035                this.linkData = linkData;
036                this.ttDataFactory = ttDataFactory;
037        }
038        
039        @Override 
040        public DataContainer getTravelTimeData(final Id<Link> linkId, final boolean createIfMissing) {
041                DataContainer data = this.linkData.get(linkId);
042                if ((null == data) && createIfMissing) {
043                        data = new DataContainer(this.ttDataFactory.createTravelTimeData(linkId));
044                        this.linkData.put(linkId, data);
045                }
046                return data;
047        }
048                
049        @Override
050        public DataContainer getTravelTimeData(Link link, boolean createIfMissing) {
051                return this.getTravelTimeData(link.getId(), createIfMissing);
052        }
053        
054}
