082        public static final class Builder {
083                private final Map<String, ActivityUtilityParameters.Builder> utilParams;
084                private final Map<String, ModeUtilityParameters.Builder> modeParams;
085
086                private double marginalUtilityOfWaiting_s;
087                private double marginalUtilityOfLateArrival_s;
088                private double marginalUtilityOfEarlyDeparture_s;
089                private double marginalUtilityOfWaitingPt_s;
090                private double marginalUtilityOfPerforming_s;
091                private double utilityOfLineSwitch;
092                private double marginalUtilityOfMoney;
093                private double abortedPlanScore;
094                private boolean scoreActs;
095                private boolean usingOldScoringBelowZeroUtilityDuration;
096                private double simulationPeriodInDays = 1.0;
097
098                public Builder(
099                                final Scenario scenario,
100                                final Id<Person> person ) {
101                        this(
102                                        scenario.getConfig().planCalcScore(),
103                                        scenario.getConfig().planCalcScore().getScoringParameters(
104                                                        (String)
105                                                                        scenario.getPopulation().getPersonAttributes().getAttribute(
106                                                                                        person.toString(),
107                                                                                        scenario.getConfig().plans().getSubpopulationAttributeName() ) ),
108                                        scenario.getConfig().scenario() );
109                }
110
111                public Builder(
112                                final PlanCalcScoreConfigGroup configGroup,
113                                final PlanCalcScoreConfigGroup.ScoringParameterSet scoringParameterSet,
114                                final ScenarioConfigGroup scenarioConfig) {
115                        this.simulationPeriodInDays = scenarioConfig.getSimulationPeriodInDays();
116                        
117                        this.usingOldScoringBelowZeroUtilityDuration = configGroup.isUsingOldScoringBelowZeroUtilityDuration() ;
118
119                        marginalUtilityOfWaiting_s = scoringParameterSet.getMarginalUtlOfWaiting_utils_hr() / 3600.0;
120                        marginalUtilityOfLateArrival_s = scoringParameterSet.getLateArrival_utils_hr() / 3600.0;
121                        marginalUtilityOfEarlyDeparture_s = scoringParameterSet.getEarlyDeparture_utils_hr() / 3600.0;
122                        marginalUtilityOfWaitingPt_s = scoringParameterSet.getMarginalUtlOfWaitingPt_utils_hr() / 3600.0 ;
123                        marginalUtilityOfPerforming_s = scoringParameterSet.getPerforming_utils_hr() / 3600.0;
124                        utilityOfLineSwitch = scoringParameterSet.getUtilityOfLineSwitch() ;
125                        marginalUtilityOfMoney = scoringParameterSet.getMarginalUtilityOfMoney() ;
126                        scoreActs = marginalUtilityOfPerforming_s != 0 || marginalUtilityOfWaiting_s != 0 ||
127                                        marginalUtilityOfLateArrival_s != 0 || marginalUtilityOfEarlyDeparture_s != 0;
128
129                        utilParams = new TreeMap<>() ;
130                        for (ActivityParams params : scoringParameterSet.getActivityParams()) {
131                                ActivityUtilityParameters.Builder factory = new ActivityUtilityParameters.Builder(params) ;
132                                utilParams.put(params.getActivityType(), factory ) ;
133                        }
134
135                        modeParams = new TreeMap<>() ;
136                        Map<String, PlanCalcScoreConfigGroup.ModeParams> modes = scoringParameterSet.getModes();
137                        double worstMarginalUtilityOfTraveling_s = 0.0;
138                        for (Map.Entry<String, PlanCalcScoreConfigGroup.ModeParams> mode : modes.entrySet()) {
139                                String modeName = mode.getKey();
140                                ModeParams params = mode.getValue();
141                                worstMarginalUtilityOfTraveling_s = Math.min(worstMarginalUtilityOfTraveling_s, params.getMarginalUtilityOfTraveling() / 3600. );
142                                modeParams.put(modeName, new ModeUtilityParameters.Builder( params ) );
143                        }
144
145                        abortedPlanScore = Math.min(
146                                        Math.min(marginalUtilityOfLateArrival_s, marginalUtilityOfEarlyDeparture_s),
147                                        Math.min(worstMarginalUtilityOfTraveling_s-marginalUtilityOfPerforming_s, marginalUtilityOfWaiting_s-marginalUtilityOfPerforming_s)
148                                        ) * 3600.0 * 24.0; // SCENARIO_DURATION
149                        // TODO 24 has to be replaced by a variable like scenario_dur (see also other places below)
150                        // This rather complicated definition has to do with the fact that exp(some_large_number) relatively quickly becomes Inf.
151                        // In consequence, the abortedPlanScore needs to be more strongly negative than anything else, but not much more.
152                        // kai, feb'12
153                }

223                public ScoringParameters build() {
224                        final Map<String, ModeUtilityParameters> modes = new TreeMap<>();
225                        for ( Map.Entry<String, ModeUtilityParameters.Builder> e : modeParams.entrySet() ) {
226                                modes.put( e.getKey() , e.getValue().build() );
227                        }
228
229                        final Map<String, ActivityUtilityParameters> acts = new TreeMap<>();
230                        for ( Map.Entry<String, ActivityUtilityParameters.Builder> e : utilParams.entrySet() ) {
231                                acts.put( e.getKey() , e.getValue().build() );
232                        }
233
234                        return new ScoringParameters(
235                                        acts,
236                                        modes,
237                                        marginalUtilityOfWaiting_s,
238                                        marginalUtilityOfLateArrival_s,
239                                        marginalUtilityOfEarlyDeparture_s,
240                                        marginalUtilityOfWaitingPt_s,
241                                        marginalUtilityOfPerforming_s,
242                                        utilityOfLineSwitch,
243                                        marginalUtilityOfMoney,
244                                        abortedPlanScore,
245                                        scoreActs,
246                                        usingOldScoringBelowZeroUtilityDuration,
247                                        this.simulationPeriodInDays);
248                }
