001package org.matsim.core.scoring;
002
003import org.apache.log4j.Logger;
004import org.matsim.api.core.v01.events.Event;
005import org.matsim.api.core.v01.population.Activity;
006import org.matsim.api.core.v01.population.Leg;
007import org.matsim.core.utils.misc.Time;
008
009import java.util.ArrayList;
010
011public final class SumScoringFunction implements ScoringFunction {
012
013        public interface BasicScoring {
014                public void finish();
015                public double getScore();
016        }
017
018        public interface ActivityScoring extends BasicScoring {
019                void handleFirstActivity(final Activity act);
020                void handleActivity(final Activity act);
021                void handleLastActivity(final Activity act);
022        }
023
024        public interface LegScoring extends BasicScoring {
025                void handleLeg(final Leg leg);
026        }
027
028        public interface MoneyScoring extends BasicScoring {
029                void addMoney(final double amount);
030        }
031
032        public interface AgentStuckScoring extends BasicScoring {
033                void agentStuck(final double time);
034        }
035
036        /**
037         * NOTE: Despite its somewhat misleading name, only Events that at the same time implement HasPersonId are passed 
038         * through this interface.  This excludes, in particular, LinkEnterEvent and LinkLeaveEvent.  This was done for performance reasons,
039         * since passing those events to all handlers imposes a significant additional burden.  See comments in 
040         * and implementation of the handleEvent 
041         * method in {@link EventsToScore}. (yyyyyy Those comments, and possibly also the events filtering, have gone
042         * from EventsToScore in commit e718809884cac6a86bdc35ea2a03c10195d37c69 .  I don't know if the performance consequences
043         * were tested.) 
044         * 
045         * @author nagel
046         */
047        public interface ArbitraryEventScoring extends BasicScoring {
048                void handleEvent( final Event event ) ;
049        }
050
051        private static Logger log = Logger.getLogger(SumScoringFunction.class);
052
053        private ArrayList<BasicScoring> basicScoringFunctions = new ArrayList<BasicScoring>();
054        private ArrayList<ActivityScoring> activityScoringFunctions = new ArrayList<ActivityScoring>();
055        private ArrayList<MoneyScoring> moneyScoringFunctions = new ArrayList<MoneyScoring>();
056        private ArrayList<LegScoring> legScoringFunctions = new ArrayList<LegScoring>();
057        private ArrayList<AgentStuckScoring> agentStuckScoringFunctions = new ArrayList<AgentStuckScoring>();
058        private ArrayList<ArbitraryEventScoring> arbtraryEventScoringFunctions = new ArrayList<ArbitraryEventScoring>() ;
059
060        @Override
061        public final void handleActivity(Activity activity) {
062                double startTime = activity.getStartTime();
063                double endTime = activity.getEndTime();
064                if (startTime == Time.UNDEFINED_TIME && endTime != Time.UNDEFINED_TIME) {
065                        for (ActivityScoring activityScoringFunction : activityScoringFunctions) {
066                                activityScoringFunction.handleFirstActivity(activity);
067                        }
068                } else if (startTime != Time.UNDEFINED_TIME && endTime != Time.UNDEFINED_TIME) {
069                        for (ActivityScoring activityScoringFunction : activityScoringFunctions) {
070                                activityScoringFunction.handleActivity(activity);
071                        }
072                } else if (startTime != Time.UNDEFINED_TIME && endTime == Time.UNDEFINED_TIME) {
073                        for (ActivityScoring activityScoringFunction : activityScoringFunctions) {
074                                activityScoringFunction.handleLastActivity(activity);
075                        }
076                } else {
077                        throw new RuntimeException("Trying to score an activity without start or end time. Should not happen.");        
078                }
079        }
080
081        @Override
082        public final void handleLeg(Leg leg) {
083                for (LegScoring legScoringFunction : legScoringFunctions) {
084                        legScoringFunction.handleLeg(leg);
085                }
086        }
087
088        @Override
089        public void addMoney(double amount) {
090                for (MoneyScoring moneyScoringFunction : moneyScoringFunctions) {
091                        moneyScoringFunction.addMoney(amount);
092                }
093        }
094
095        @Override
096        public void agentStuck(double time) {
097                for (AgentStuckScoring agentStuckScoringFunction : agentStuckScoringFunctions) {
098                        agentStuckScoringFunction.agentStuck(time);
099                }
100        }
101
102        @Override
103        public void handleEvent(Event event) {
104                for ( ArbitraryEventScoring eventScoringFunction : this.arbtraryEventScoringFunctions ) {
105                        eventScoringFunction.handleEvent(event) ;
106                }
107        }
108
109        @Override
110        public void finish() {
111                for (BasicScoring basicScoringFunction : basicScoringFunctions) {
112                        basicScoringFunction.finish();
113                }
114        }
115
116        /**
117         * Add the score of all functions.
118         */
119        @Override
120        public double getScore() {
121                double score = 0.0;
122                for (BasicScoring basicScoringFunction : basicScoringFunctions) {
123                        double contribution = basicScoringFunction.getScore();
124                        if (log.isTraceEnabled()) {
125                                log.trace("Contribution of scoring function: " + basicScoringFunction.getClass().getName() + " is: " + contribution);
126                        }
127                        if ( Double.isNaN( contribution ) ) {
128                                // I consider this dangerous enough to justify a crash. If somebody has strong arguments for NaN scores,
129                                // one might change this to "log.error(...)". td june 15
130                                throw new RuntimeException( "Contribution of scoring function: " + basicScoringFunction.getClass().getName() + " is NaN! Behavior with NaN scores is undefined." );
131                        }
132                        score += contribution;
133                }
134                return score;
135        }
136
137        public void addScoringFunction(BasicScoring scoringFunction) {
138                basicScoringFunctions.add(scoringFunction);
139
140                if (scoringFunction instanceof ActivityScoring) {
141                        activityScoringFunctions.add((ActivityScoring) scoringFunction);
142                }
143
144                if (scoringFunction instanceof AgentStuckScoring) {
145                        agentStuckScoringFunctions.add((AgentStuckScoring) scoringFunction);
146                }
147
148                if (scoringFunction instanceof LegScoring) {
149                        legScoringFunctions.add((LegScoring) scoringFunction);
150                }
151
152                if (scoringFunction instanceof MoneyScoring) {
153                        moneyScoringFunctions.add((MoneyScoring) scoringFunction);
154                }
155
156                if (scoringFunction instanceof ArbitraryEventScoring ) {
157                        this.arbtraryEventScoringFunctions.add((ArbitraryEventScoring) scoringFunction) ;
158                }
159
160        }
161
162
163}
