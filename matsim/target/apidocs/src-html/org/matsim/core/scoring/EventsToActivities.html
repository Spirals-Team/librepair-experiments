021 package org.matsim.core.scoring;
022
023import java.util.ArrayList;
024import java.util.HashMap;
025import java.util.List;
026import java.util.Map;
027
028import org.matsim.api.core.v01.Id;
029import org.matsim.api.core.v01.events.ActivityEndEvent;
030import org.matsim.api.core.v01.events.ActivityStartEvent;
031import org.matsim.api.core.v01.events.handler.ActivityEndEventHandler;
032import org.matsim.api.core.v01.events.handler.ActivityStartEventHandler;
033import org.matsim.api.core.v01.population.Activity;
034import org.matsim.api.core.v01.population.Person;
035import org.matsim.core.api.experimental.events.EventsManager;
036import org.matsim.core.controler.ControlerListenerManager;
037import org.matsim.core.controler.events.AfterMobsimEvent;
038import org.matsim.core.controler.listener.AfterMobsimListener;
039import org.matsim.core.population.PopulationUtils;
040
041import javax.inject.Inject;
042
043/**
044 * 
045 * Converts a stream of Events into a stream of Activities. Passes Activities to a single ActivityHandler which must be registered with this class.
046 * Mainly intended for scoring, but can be used for any kind of Activity related statistics. Essentially, it allows you to read
047 * Activities from the simulation like you would read Activities from Plans, except that the Plan does not even need to exist.
048 * <p></p>
049 * Note that the instances of Activity passed to the LegHandler will never be identical to those in the Scenario! Even
050 * in a "no-op" simulation which only reproduces the Plan, new instances will be created. So if you attach your own data
051 * to the Activities in the Scenario, that's your own lookout.
052 * 
053 * @author michaz
054 *
055 */
056public final class EventsToActivities implements ActivityStartEventHandler, ActivityEndEventHandler {
057        
058        public interface ActivityHandler {
059            void handleActivity(PersonExperiencedActivity activity);
060        }
061
062    private Map<Id<Person>, Activity> activities = new HashMap<>();
063    private List<ActivityHandler> activityHandlers = new ArrayList<>();
064
065    public EventsToActivities() {
066
067    }
068
069    @Inject
070    EventsToActivities(ControlerListenerManager controlerListenerManager, EventsManager eventsManager) {
071        controlerListenerManager.addControlerListener(new AfterMobsimListener() {
072            @Override
073            public void notifyAfterMobsim(AfterMobsimEvent event) {
074                finish();
075            }
076        });
077        eventsManager.addHandler(this);
078    }
079
080    @Override
081    public void handleEvent(ActivityEndEvent event) {
082        Activity activity = activities.get(event.getPersonId());
083        if (activity == null) {
084            Activity firstActivity = PopulationUtils.createActivityFromLinkId(event.getActType(), event.getLinkId());
085            firstActivity.setFacilityId(event.getFacilityId());
086            activity = firstActivity;
087        }
088        activity.setEndTime(event.getTime());
089        for (ActivityHandler activityHandler : activityHandlers) {
090            activityHandler.handleActivity(new PersonExperiencedActivity(event.getPersonId(), activity));
091        }
092        activities.remove(event.getPersonId());
093    }
094
095    @Override
096    public void handleEvent(ActivityStartEvent event) {
097        Activity activity = PopulationUtils.createActivityFromLinkId(event.getActType(), event.getLinkId());
098        activity.setFacilityId(event.getFacilityId());
099        activity.setStartTime(event.getTime());
100        activities.put(event.getPersonId(), activity);
101    }
102
103    @Override
104    public void reset(int iteration) {
105        activities.clear();
106    }
107
108    public void addActivityHandler(ActivityHandler activityHandler) {
109        this.activityHandlers.add(activityHandler);
110    }
111
112    public void finish() {
113        for (Map.Entry<Id<Person>, Activity> entry : activities.entrySet()) {
114            for (ActivityHandler activityHandler : activityHandlers) {
115                activityHandler.handleActivity(new PersonExperiencedActivity(entry.getKey(), entry.getValue()));
116            }
117        }
118    }
119
120}
