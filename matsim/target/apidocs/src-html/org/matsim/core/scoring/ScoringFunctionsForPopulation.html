059/**
060 * This class helps EventsToScore by keeping ScoringFunctions for the entire Population - one per Person -, and dispatching Activities
061 * and Legs to the ScoringFunctions. It also gives out the ScoringFunctions, so they can be given other events by EventsToScore.
062 * It is not independently useful. Please do not make public.
063 * 
064 * @author michaz
065 *
066 */
067public final class ScoringFunctionsForPopulation implements BasicEventHandler, EventsToLegs.LegHandler, EventsToActivities.ActivityHandler {
068        // yyyyyy there is currently only one place outside package where this is used, and I think it
069        // can be changed there.  kai, sep'17
070        
071        @SuppressWarnings("unused")
072        private final static Logger log = Logger.getLogger(ScoringFunctionsForPopulation.class);
073        private final Population population;
074        private final ScoringFunctionFactory scoringFunctionFactory;
075
076        /*
077         * Replaced TreeMaps with (Linked)HashMaps since they should perform much better. For 'partialScores'
078         * a LinkedHashMap is used to ensure that agents are written in a deterministic order to the output files.
079         *
080         * Replaced List with TDoubleCollection (TDoubleArrayList) in the partialScores map. This collection allows
081         * storing primitive objects, i.e. its double entries don't have to be wrapped into Double objects which
082         * should be faster and reduce the memory overhead.
083         *
084         * cdobler, nov'15
085         */
086        private final Map<Id<Person>, ScoringFunction> agentScorers = new HashMap<>();
087        private final Map<Id<Person>, TDoubleCollection> partialScores = new LinkedHashMap<>();
088        private final AtomicReference<Throwable> exception = new AtomicReference<>();
089        
090        /**
091         * For something like the bicycle scoring, we need to know individual links at the level of the scoring function.  This is a first sketch how this could be implemented.
092         * kai, mar'17
093         */
094        private boolean passLinkEventsToPerson = false;
095        
096        private Vehicle2DriverEventHandler delegate = new Vehicle2DriverEventHandler();
097
098        @Inject
099        ScoringFunctionsForPopulation(ControlerListenerManager controlerListenerManager, EventsManager eventsManager, EventsToActivities eventsToActivities, EventsToLegs eventsToLegs,
100                                                                  Population population, ScoringFunctionFactory scoringFunctionFactory) {
101                controlerListenerManager.addControlerListener(new IterationStartsListener() {
102                        @Override
103                        public void notifyIterationStarts(IterationStartsEvent event) {
104                                init();
105                        }
106                });
107                this.population = population;
108                this.scoringFunctionFactory = scoringFunctionFactory;
109                eventsManager.addHandler(this);
110                eventsToActivities.addActivityHandler(this);
111                eventsToLegs.addLegHandler(this);
112                if ( passLinkEventsToPerson ) {
113                        eventsManager.addHandler(delegate);
114                }
115        }
116
117        private void init() {
118                for (Person person : population.getPersons().values()) {
119                        ScoringFunction data = scoringFunctionFactory.createNewScoringFunction(person);
120                        this.agentScorers.put(person.getId(), data);
121                        this.partialScores.put(person.getId(), new TDoubleArrayList());
122                }
123        }
124
125        @Override
126        synchronized public void handleEvent(Event o) {
127                // this is for the stuff that is directly based on events.
128                // note that this passes on _all_ person events, even those which are aggregated into legs and activities.
129                // for the time being, not all PersonEvents may "implement HasPersonId".
130                // link enter/leave events are NOT passed on, for performance reasons.
131                // kai/dominik, dec'12
132                if (o instanceof HasPersonId) {
133                        ScoringFunction scoringFunction = getScoringFunctionForAgent(((HasPersonId) o).getPersonId());
134                        if (scoringFunction != null) {
135                                if (o instanceof PersonStuckEvent) {
136                                        scoringFunction.agentStuck(o.getTime());
137                                } else if (o instanceof PersonMoneyEvent) {
138                                        scoringFunction.addMoney(((PersonMoneyEvent) o).getAmount());
139                                        // yy looking at this, I am a bit skeptic if it truly makes sense to not pass this additionally into the general events handling function below.
140                                        // A use case might be different utilities of money by money transaction type (e.g. toll, fare, reimbursement, ...).  kai, mar'17
141                                } 
142//                              else {
143                                        scoringFunction.handleEvent(o);
144                                        // passing this on in any case, see comment above.  kai, mar'17
145//                              }
146                        }
147                }
148                if ( passLinkEventsToPerson ) {
149                        // Establish and end connection between driver and vehicle
150                        if (o instanceof VehicleEntersTrafficEvent) {
151                                delegate.handleEvent((VehicleEntersTrafficEvent) o);
152                        }
153                        if (o instanceof VehicleLeavesTrafficEvent) {
154                                delegate.handleEvent((VehicleLeavesTrafficEvent) o);
155                        }
156                        // Pass LinkEnterEvent to person scoring, required e.g. for bicycle where link attributes are observed in scoring
157                        if ( o instanceof LinkEnterEvent ) {
158                                Id<Vehicle> vehicleId = ((LinkEnterEvent)o).getVehicleId() ;
159                                Id<Person> driverId = delegate.getDriverOfVehicle(vehicleId) ;
160                                ScoringFunction scoringFunction = getScoringFunctionForAgent( driverId );
161                                // (this will NOT do the scoring function lookup twice since LinkEnterEvent is not an instance of HasPersonId.  kai, mar'17)
162                                if (scoringFunction != null) {
163                                        scoringFunction.handleEvent(o) ;
164                                }
165                        }
166                        /*
167                         * (This shouldn't really be more expensive than passing the link events to the router: here, we have a map lookup
168                         * for agentId, there we have a map lookup for linkId. Should be somewhat similar in terms of average
169                         * computational complexity. In BetaTravelTest, 194sec w/ "false", 193sec w/ "true". However, the experienced
170                         * plans service in fact does the same thing, so we should be able to get away without having to do this twice.
171                         * kai, mar'17)
172                         */
173                }
174        }
175
176        @Override
177        synchronized public void handleLeg(PersonExperiencedLeg o) {
178                Id<Person> agentId = o.getAgentId();
179                Leg leg = o.getLeg();
180                ScoringFunction scoringFunction = ScoringFunctionsForPopulation.this.getScoringFunctionForAgent(agentId);
181                if (scoringFunction != null) {
182                        scoringFunction.handleLeg(leg);
183                        TDoubleCollection partialScoresForAgent = partialScores.get(agentId);
184                        partialScoresForAgent.add(scoringFunction.getScore());
185                }
186        }
187
188        @Override
189        synchronized public void handleActivity(PersonExperiencedActivity o) {
190                Id<Person> agentId = o.getAgentId();
191                Activity activity = o.getActivity();
192                ScoringFunction scoringFunction = ScoringFunctionsForPopulation.this.getScoringFunctionForAgent(agentId);
193                if (scoringFunction != null) {
194                        scoringFunction.handleActivity(activity);
195                        TDoubleCollection partialScoresForAgent = partialScores.get(agentId);
196                        partialScoresForAgent.add(scoringFunction.getScore());
197                }
198        }
199
200        /**
201         * Returns the scoring function for the specified agent. If the agent
202         * already has a scoring function, that one is returned. If the agent does
203         * not yet have a scoring function, a new one is created and assigned to the
204         * agent and returned.
205         *
206         * @param agentId
207         *            The id of the agent the scoring function is requested for.
208         * @return The scoring function for the specified agent.
209         */
210        public ScoringFunction getScoringFunctionForAgent(final Id<Person> agentId) {
211                return this.agentScorers.get(agentId);
212        }
213
214        public void finishScoringFunctions() {
215                // Rethrow an exception in a scoring function (user code) if there was one.
216                Throwable throwable = exception.get();
217                if (throwable != null) {
218                        if (throwable instanceof RuntimeException) {
219                                throw ((RuntimeException) throwable);
220                        } else {
221                                throw new RuntimeException(throwable);
222                        }
223                }
224                for (ScoringFunction sf : this.agentScorers.values()) {
225                        sf.finish();
226                }
227                for (Entry<Id<Person>, TDoubleCollection> entry : this.partialScores.entrySet()) {
228                        entry.getValue().add(this.getScoringFunctionForAgent(entry.getKey()).getScore());
229                }
230        }
231
232        public void writePartialScores(String iterationFilename) {
233                try ( BufferedWriter out = IOUtils.getBufferedWriter(iterationFilename) ) {
234                        for (Entry<Id<Person>, TDoubleCollection> entry : this.partialScores.entrySet()) {
235                                out.write(entry.getKey().toString());
236                                TDoubleIterator iterator = entry.getValue().iterator();
237                                while (iterator.hasNext()) {
238                                        out.write('\t' + String.valueOf(iterator.next()));
239                                }
240                                out.newLine();
241                        }
242                        out.close();
243                } catch (IOException e) {
244                        throw new RuntimeException(e);
245                }
246        }
247
248        @Override
249        public void reset(int iteration) {
250
251        }
252
253        public boolean isPassLinkEventsToPerson() {
254                return passLinkEventsToPerson;
255        }
256
257        public void setPassLinkEventsToPerson(boolean passLinkEventsToPerson) {
258                this.passLinkEventsToPerson = passLinkEventsToPerson;
259        }
