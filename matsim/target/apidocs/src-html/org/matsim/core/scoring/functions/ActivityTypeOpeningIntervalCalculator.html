001package org.matsim.core.scoring.functions;
002
003import org.matsim.api.core.v01.population.Activity;
004
005public class ActivityTypeOpeningIntervalCalculator implements OpeningIntervalCalculator {
006        private final ScoringParameters params;
007
008        public ActivityTypeOpeningIntervalCalculator(ScoringParameters params) {
009                this.params = params;
010        }
011
012        @Override
013        public double[] getOpeningInterval(final Activity act) {
014
015                ActivityUtilityParameters actParams = this.params.utilParams.get(act.getType());
016                if (actParams == null) {
017                        throw new IllegalArgumentException("acttype \"" + act.getType() + "\" is not known in utility parameters " +
018                                        "(module name=\"planCalcScore\" in the config file).");
019                }
020
021                double openingTime = actParams.getOpeningTime();
022                double closingTime = actParams.getClosingTime();
023
024                //Â openInterval has two values
025                // openInterval[0] will be the opening time
026                // openInterval[1] will be the closing time
027
028                return new double[]{openingTime, closingTime};
029        }
030}
