021package org.matsim.core.scoring;
022
023import org.matsim.api.core.v01.events.Event;
024import org.matsim.api.core.v01.events.PersonMoneyEvent;
025import org.matsim.api.core.v01.population.Activity;
026import org.matsim.api.core.v01.population.Leg;
027
028/**
029 * A scoring function calculates the score for one plan of an agent.  The score
030 * usually depends on how much time an agent is traveling and how much time an
031 * agent spends at an activity.  Thus the scoring function gets informed when
032 * activities start and end as well as when legs start and end.<br>
033 * Note that one ScoringFunction calculates the score for exactly one agent.
034 * Thus every agents must have its own instance of a scoring function!
035 * <p></p>
036 * See {@link tutorial.programming.example16customscoring.RunCustomScoringExample} for an example.
037 *
038 * @author mrieser
039 */
040public interface ScoringFunction {
041
042        /**
043         * Tells the scoring function about an Activity. The Activity which
044         * the agent is in when the simulation starts will have a startTime
045         * of Time.UNDEFINED_TIME. The Activity which the agent is in when
046         * the simulation ends will have an endTime of Time.UNDEFINED_TIME.
047         * It is up to the implementation what to make of this, 
048         * especially to "wrap" it "around".
049         * @param activity
050         */
051    public void handleActivity(Activity activity);
052
053    /**
054     * Tells the scoring function about a Leg. Will contain complete route
055     * information for network routes (as you would expect in a Plan), but
056     * only a GenericRoute for everything else, especially transit.
057     */
058    public void handleLeg(Leg leg);
059
060        /**
061         * Tells the scoring function that the agent got stuck in the simulation and
062         * is removed from the simulation. This should usually lead to a high penalty
063         * in the score, as the agent was not able to perform its plan as wanted.
064         * An agent can get stuck while performing an activity or while driving.
065         *
066         * @param time The time at which the agent got stuck and was removed from the
067         * simulation.
068         */
069        public void agentStuck(final double time);
070
071        /**
072         * Adds the specified amount of utility to the agent's score. This is mostly
073         * used for handling {@link PersonMoneyEvent}s, allowing other parts of the
074         * code to influence an agent's score.
075         *
076         * @param amount amount to be added to the agent's score
077         */
078        public void addMoney(final double amount);
079
080        /**
081         * Tells the scoring function that no more information will be given to it
082         * and that the final score should be calculated.  But the score must <b>not</b>
083         * be written to the plan!
084         */
085        public void finish();
086
087        /**
088         * Returns the score for this plan.
089
090         * @return the score
091         */
092        public double getScore();
093
094        public void handleEvent( Event event ) ;
095
096}
