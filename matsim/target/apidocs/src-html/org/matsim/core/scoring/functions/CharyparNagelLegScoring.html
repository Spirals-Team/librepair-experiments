036/**
037 * This is a re-implementation of the original CharyparNagel function, based on a
038 * modular approach.
039 * @see <a href="http://www.matsim.org/node/263">http://www.matsim.org/node/263</a>
040 * @author rashid_waraich
041 */
042public class CharyparNagelLegScoring implements org.matsim.core.scoring.SumScoringFunction.LegScoring, org.matsim.core.scoring.SumScoringFunction.ArbitraryEventScoring {
043        // yyyy URL in above javadoc is broken.  kai, feb'17
044
045        protected double score;
046
047        /** The parameters used for scoring */
048        protected final ScoringParameters params;
049        protected Network network;
050        private boolean nextEnterVehicleIsFirstOfTrip = true ;
051        private boolean nextStartPtLegIsFirstOfTrip = true ;
052        private boolean currentLegIsPtLeg = false;
053        private double lastActivityEndTime = Time.UNDEFINED_TIME ;
054        
055        public CharyparNagelLegScoring(final ScoringParameters params, Network network) {
056                this.params = params;
057                this.network = network;
058                this.nextEnterVehicleIsFirstOfTrip = true ;
059                this.nextStartPtLegIsFirstOfTrip = true ;
060                this.currentLegIsPtLeg = false;
061        }
062
063        @Override
064        public void finish() {
065
066        }
067
068        @Override
069        public double getScore() {
070                return this.score;
071        }
072
073        private static int ccc=0 ;
074        
075        protected double calcLegScore(final double departureTime, final double arrivalTime, final Leg leg) {
076                double tmpScore = 0.0;
077                double travelTime = arrivalTime - departureTime; // travel time in seconds      
078                ModeUtilityParameters modeParams = this.params.modeParams.get(leg.getMode());
079                if (modeParams == null) {
080                        if (leg.getMode().equals(TransportMode.transit_walk) || leg.getMode().equals(TransportMode.access_walk) 
081                                        || leg.getMode().equals(TransportMode.egress_walk) ) {
082                                modeParams = this.params.modeParams.get(TransportMode.walk);
083                        } else {
084//                              modeParams = this.params.modeParams.get(TransportMode.other);
085                                throw new RuntimeException("just encountered mode for which no scoring parameters are defined: " + leg.getMode().toString() ) ;
086                        }
087                }
088                tmpScore += travelTime * modeParams.marginalUtilityOfTraveling_s;
089                if (modeParams.marginalUtilityOfDistance_m != 0.0
090                                || modeParams.monetaryDistanceCostRate != 0.0) {
091                        Route route = leg.getRoute();
092                        double dist = route.getDistance(); // distance in meters
093                        if ( Double.isNaN(dist) ) {
094                                if ( ccc<10 ) {
095                                        ccc++ ;
096                                        Logger.getLogger(this.getClass()).warn("distance is NaN. Will make score of this plan NaN. Possible reason: Simulation does not report " +
097                                                        "a distance for this trip. Possible reason for that: mode is teleported and router does not " +
098                                                        "write distance into plan.  Needs to be fixed or these plans will die out.") ;
099                                        if ( ccc==10 ) {
100                                                Logger.getLogger(this.getClass()).warn(Gbl.FUTURE_SUPPRESSED) ;
101                                        }
102                                }
103                        }
104                        tmpScore += modeParams.marginalUtilityOfDistance_m * dist;
105                        tmpScore += modeParams.monetaryDistanceCostRate * this.params.marginalUtilityOfMoney * dist;
106                }
107                tmpScore += modeParams.constant;
108                // (yyyy once we have multiple legs without "real" activities in between, this will produce wrong results.  kai, dec'12)
109                // (yy NOTE: the constant is added for _every_ pt leg.  This is not how such models are estimated.  kai, nov'12)
110                return tmpScore;
111        }
112        
113        @Override
114        public void handleEvent(Event event) {
115                if ( event instanceof ActivityEndEvent ) {
116                        // When there is a "real" activity, flags are reset:
117                        if ( !PtConstants.TRANSIT_ACTIVITY_TYPE.equals( ((ActivityEndEvent)event).getActType()) ) {
118                                this.nextEnterVehicleIsFirstOfTrip  = true ;
119                                this.nextStartPtLegIsFirstOfTrip = true ;
120                        }
121                        this.lastActivityEndTime = event.getTime() ;
122                }
123
124                if ( event instanceof PersonEntersVehicleEvent && currentLegIsPtLeg ) {
125                        if ( !this.nextEnterVehicleIsFirstOfTrip ) {
126                                // all vehicle entering after the first triggers the disutility of line switch:
127                                this.score  += params.utilityOfLineSwitch ;
128                        }
129                        this.nextEnterVehicleIsFirstOfTrip = false ;
130                        // add score of waiting, _minus_ score of travelling (since it is added in the legscoring above):
131                        this.score += (event.getTime() - this.lastActivityEndTime) * (this.params.marginalUtilityOfWaitingPt_s - this.params.modeParams.get(TransportMode.pt).marginalUtilityOfTraveling_s) ;
132                }
133
134                if ( event instanceof PersonDepartureEvent ) {
135                        this.currentLegIsPtLeg = TransportMode.pt.equals( ((PersonDepartureEvent)event).getLegMode() );
136                        if ( currentLegIsPtLeg ) {
137                                if ( !this.nextStartPtLegIsFirstOfTrip ) {
138                                        this.score -= params.modeParams.get(TransportMode.pt).constant ;
139                                        // (yyyy deducting this again, since is it wrongly added above.  should be consolidated; this is so the code
140                                        // modification is minimally invasive.  kai, dec'12)
141                                }
142                                this.nextStartPtLegIsFirstOfTrip = false ;
143                        }
144                }
145        }
146
147        @Override
148        public void handleLeg(Leg leg) {
149                double legScore = calcLegScore(leg.getDepartureTime(), leg.getDepartureTime() + leg.getTravelTime(), leg);
150                this.score += legScore;
151        }
