035        protected double score;
036        private double currentActivityStartTime;
037        private double firstActivityEndTime;

039        private static final double INITIAL_LAST_TIME = 0.0;
040        private static final double INITIAL_FIRST_ACT_END_TIME = Time.UNDEFINED_TIME;
041        private static final double INITIAL_SCORE = 0.0;
042
043        private static int firstLastActWarning = 0;
044        private static short firstLastActOpeningTimesWarning = 0;

046        private final ScoringParameters params;
047        private final OpeningIntervalCalculator openingIntervalCalculator;

053        public CharyparNagelActivityScoring(final ScoringParameters params) {
054                this(params, new ActivityTypeOpeningIntervalCalculator(params));

057        public CharyparNagelActivityScoring(final ScoringParameters params, final OpeningIntervalCalculator openingIntervalCalculator) {
058                this.params = params;
059                this.currentActivityStartTime = INITIAL_LAST_TIME;
060                this.firstActivityEndTime = INITIAL_FIRST_ACT_END_TIME;
061                this.score = INITIAL_SCORE;
062
063                firstLastActWarning = 0 ;
064                firstLastActOpeningTimesWarning = 0 ;
065                this.openingIntervalCalculator = openingIntervalCalculator;
066        }

068        @Override
069        public void finish() {
070                if (this.firstActivity != null) {
071                        handleMorningActivity();
072                }
073                // Else, no activity has started so far.
074                // This probably means that the plan contains at most one activity.
075                // We cannot handle that correctly, because we do not know what it is.
076        }

078        @Override
079        public double getScore() {
080                return this.score;
081        }

083        protected double calcActScore(final double arrivalTime, final double departureTime, final Activity act) {
084
085                ActivityUtilityParameters actParams = this.params.utilParams.get(act.getType());
086                if (actParams == null) {
087                        throw new IllegalArgumentException("acttype \"" + act.getType() + "\" is not known in utility parameters " +
088                                        "(module name=\"planCalcScore\" in the config file).");
089                }
090
091                double tmpScore = 0.0;
092
093                if (actParams.isScoreAtAll()) {
094                        /* Calculate the times the agent actually performs the
095                         * activity.  The facility must be open for the agent to
096                         * perform the activity.  If it's closed, but the agent is
097                         * there, the agent must wait instead of performing the
098                         * activity (until it opens).
099                         *
100                         *                                             Interval during which
101                         * Relationship between times:                 activity is performed:
102                         *
103                         *      O________C A~~D  ( 0 <= C <= A <= D )   D...D (not performed)
104                         * A~~D O________C       ( A <= D <= O <= C )   D...D (not performed)
105                         *      O__A+++++C~~D    ( O <= A <= C <= D )   A...C
106                         *      O__A++D__C       ( O <= A <= D <= C )   A...D
107                         *   A~~O++++++++C~~D    ( A <= O <= C <= D )   O...C
108                         *   A~~O+++++D__C       ( A <= O <= D <= C )   O...D
109                         *
110                         * Legend:
111                         *  A = arrivalTime    (when agent gets to the facility)
112                         *  D = departureTime  (when agent leaves the facility)
113                         *  O = openingTime    (when facility opens)
114                         *  C = closingTime    (when facility closes)
115                         *  + = agent performs activity
116                         *  ~ = agent waits (agent at facility, but not performing activity)
117                         *  _ = facility open, but agent not there
118                         *
119                         * assume O <= C
120                         * assume A <= D
121                         */
122
123                        double[] openingInterval = openingIntervalCalculator.getOpeningInterval(act);
124                        double openingTime = openingInterval[0];
125                        double closingTime = openingInterval[1];
126
127                        double activityStart = arrivalTime;
128                        double activityEnd = departureTime;
129
130                        if ((openingTime >=  0) && (arrivalTime < openingTime)) {
131                                activityStart = openingTime;
132                        }
133                        if ((closingTime >= 0) && (closingTime < departureTime)) {
134                                activityEnd = closingTime;
135                        }
136                        if ((openingTime >= 0) && (closingTime >= 0)
137                                        && ((openingTime > departureTime) || (closingTime < arrivalTime))) {
138                                // agent could not perform action
139                                activityStart = departureTime;
140                                activityEnd = departureTime;
141                        }
142                        double duration = activityEnd - activityStart;
143
144                        // disutility if too early
145                        if (arrivalTime < activityStart) {
146                                // agent arrives to early, has to wait
147                                tmpScore += this.params.marginalUtilityOfWaiting_s * (activityStart - arrivalTime);
148                        }
149
150                        // disutility if too late
151
152                        double latestStartTime = actParams.getLatestStartTime();
153                        if ((latestStartTime >= 0) && (activityStart > latestStartTime)) {
154                                tmpScore += this.params.marginalUtilityOfLateArrival_s * (activityStart - latestStartTime);
155                        }
156
157                        // utility of performing an action, duration is >= 1, thus log is no problem
158                        double typicalDuration = actParams.getTypicalDuration();
159
160                        if ( this.params.usingOldScoringBelowZeroUtilityDuration ) {
161                                if (duration > 0) {
162                                        double utilPerf = this.params.marginalUtilityOfPerforming_s * typicalDuration
163                                                        * Math.log((duration / 3600.0) / actParams.getZeroUtilityDuration_h());
164                                        double utilWait = this.params.marginalUtilityOfWaiting_s * duration;
165                                        tmpScore += Math.max(0, Math.max(utilPerf, utilWait));
166                                } else {
167                                        tmpScore += 2*this.params.marginalUtilityOfLateArrival_s*Math.abs(duration);
168                                }
169                        } else {
170                                if ( duration >= 3600.*actParams.getZeroUtilityDuration_h() ) {
171                                        double utilPerf = this.params.marginalUtilityOfPerforming_s * typicalDuration
172                                                        * Math.log((duration / 3600.0) / actParams.getZeroUtilityDuration_h());
173                                        // also removing the "wait" alternative scoring.
174                                        tmpScore += utilPerf ;
175                                } else {
176//                                      if ( wrnCnt < 1 ) {
177//                                              wrnCnt++ ;
178//                                              log.warn("encountering duration < zeroUtilityDuration; the logic for this was changed around mid-nov 2013.") ;
179//                                              log.warn( "your final score thus will be different from earlier runs; set usingOldScoringBelowZeroUtilityDuration to true if you "
180//                                                              + "absolutely need the old version.  See https://matsim.atlassian.net/browse/MATSIM-191." );
181//                                              log.warn( Gbl.ONLYONCE ) ;
182//                                      }
183                                        
184                                        // below zeroUtilityDuration, we linearly extend the slope ...:
185                                        double slopeAtZeroUtility = this.params.marginalUtilityOfPerforming_s * typicalDuration / ( 3600.*actParams.getZeroUtilityDuration_h() ) ;
186                                        if ( slopeAtZeroUtility < 0. ) {
187                                                // (beta_perf might be = 0)
188                                                System.err.println("beta_perf: " + this.params.marginalUtilityOfPerforming_s);
189                                                System.err.println("typicalDuration: " + typicalDuration );
190                                                System.err.println( "zero utl duration: " + actParams.getZeroUtilityDuration_h() );
191                                                throw new RuntimeException( "slope at zero utility < 0.; this should not happen ...");
192                                        }
193                                        double durationUnderrun = actParams.getZeroUtilityDuration_h()*3600. - duration ;
194                                        if ( durationUnderrun < 0. ) {
195                                                throw new RuntimeException( "durationUnderrun < 0; this should not happen ...") ;
196                                        }
197                                        tmpScore -= slopeAtZeroUtility * durationUnderrun ;
198                                }
199                                
200                        }
201
202                        // disutility if stopping too early
203                        double earliestEndTime = actParams.getEarliestEndTime();
204                        if ((earliestEndTime >= 0) && (activityEnd < earliestEndTime)) {
205                                tmpScore += this.params.marginalUtilityOfEarlyDeparture_s * (earliestEndTime - activityEnd);
206                        }
207
208                        // disutility if going to away to late
209                        if (activityEnd < departureTime) {
210                                tmpScore += this.params.marginalUtilityOfWaiting_s * (departureTime - activityEnd);
211                        }
212
213                        // disutility if duration was too short
214                        double minimalDuration = actParams.getMinimalDuration();
215                        if ((minimalDuration >= 0) && (duration < minimalDuration)) {
216                                tmpScore += this.params.marginalUtilityOfEarlyDeparture_s * (minimalDuration - duration);
217                        }
218                }
219                return tmpScore;
220        }

222        private void handleOvernightActivity(Activity lastActivity) {
223                assert firstActivity != null;
224                assert lastActivity != null;
225
226
227                if (lastActivity.getType().equals(this.firstActivity.getType()) || this.firstActivity.getType().equals("not specified") ) {
228                        // yyyy find better way to encode "not specified".  It is quite common for travel surveys that the type of the
229                        // first activity is not encoded at all, and then we can as well assume that it is the same as that of the last.  kai, sep'16
230                        
231                        // the first Act and the last Act have the same type:
232                        if (firstLastActOpeningTimesWarning <= 10) {
233                                double[] openInterval = openingIntervalCalculator.getOpeningInterval(lastActivity);
234                                if (openInterval[0] >= 0 || openInterval[1] >= 0){
235                                        log.warn("There are opening or closing times defined for the first and last activity. The correctness of the scoring function can thus not be guaranteed.");
236                                        log.warn("first activity: " + firstActivity ) ;
237                                        log.warn("last activity: " + lastActivity ) ;
238                                        if (firstLastActOpeningTimesWarning == 10) {
239                                                log.warn("Additional warnings of this type are suppressed.");
240                                        }
241                                        firstLastActOpeningTimesWarning++;
242                                }
243                        }
244                        
245                        double calcActScore = calcActScore(this.currentActivityStartTime, this.firstActivityEndTime + 24*3600, lastActivity);
246                        this.score += calcActScore; // SCENARIO_DURATION
247                } else {
248                        // the first Act and the last Act have NOT the same type:
249                        if (this.params.scoreActs) {
250                                int last=0 ;
251                                if (firstLastActWarning <= last) {
252                                        log.warn("The first and the last activity do not have the same type. " ) ;
253                                        log.warn( "Will score the first activity from midnight to its end, and the last activity from its start to midnight.") ;
254                                        log.warn("Because of the nonlinear function, this is not the same as scoring from start to end.");
255                                        log.warn("first activity: " + firstActivity ) ;
256                                        log.warn("last activity: " + lastActivity ) ;
257                                        log.warn("This may also happen when plans are not completed when the simulation ends.") ;
258                                        if (firstLastActWarning == last) {
259                                                log.warn("Additional warnings of this type are suppressed.");
260                                        }
261                                        firstLastActWarning++;
262                                }
263
264                                // score first activity
265                                this.score += calcActScore(0.0, this.firstActivityEndTime, firstActivity);
266                                // score last activity
267                                this.score += calcActScore(this.currentActivityStartTime, this.params.simulationPeriodInDays * 24*3600, lastActivity);
268                        }
269                }
270        }
271
272        private void handleMorningActivity() {
273                assert firstActivity != null;
274                // score first activity
275                this.score += calcActScore(0.0, this.firstActivityEndTime, firstActivity);
276        }
277
278        @Override
279        public void handleFirstActivity(Activity act) {
280                assert act != null;
281                this.firstActivityEndTime = act.getEndTime();
282                this.firstActivity = act;
283        }
284
285        @Override
286        public void handleActivity(Activity act) {
287                this.score += calcActScore(act.getStartTime(), act.getEndTime(), act);
288        }
289
290        @Override
291        public void handleLastActivity(Activity act) {
292                this.currentActivityStartTime = act.getStartTime();
293                this.handleOvernightActivity(act);
294                this.firstActivity = null;
295        }
