072/**
073 * 
074 * Converts a stream of Events into a stream of Legs. Passes Legs to a single LegHandler which must be registered with this class.
075 * Mainly intended for scoring, but can be used for any kind of Leg related statistics. Essentially, it allows you to read
076 * Legs from the simulation like you would read Legs from Plans, except that the Plan does not even need to exist.
077 * 
078 * Note that the instances of Leg passed to the LegHandler will never be identical to those in the Scenario! Even
079 * in a "no-op" simulation which only reproduces the Plan, new instances will be created. So if you attach your own data
080 * to the Legs in the Scenario, that's your own lookout.
081 * 
082 * @author michaz
083 *
084 */
085public final class EventsToLegs implements PersonDepartureEventHandler, PersonArrivalEventHandler, LinkLeaveEventHandler, LinkEnterEventHandler, 
086TeleportationArrivalEventHandler, TransitDriverStartsEventHandler, PersonEntersVehicleEventHandler, VehicleArrivesAtFacilityEventHandler, VehicleEntersTrafficEventHandler, VehicleLeavesTrafficEventHandler {
087
088        private Vehicle2DriverEventHandler delegate = new Vehicle2DriverEventHandler();
089
090        private class PendingTransitTravel {
091
092                final Id<Vehicle> vehicleId;
093                final Id<TransitStopFacility> accessStop;
094
095                public PendingTransitTravel(Id<Vehicle> vehicleId, Id<TransitStopFacility> accessStop) {
096                        this.vehicleId = vehicleId;
097                        this.accessStop = accessStop;
098                }
099
100        }
101
102        private class LineAndRoute {
103
104                final Id<TransitLine> transitLineId;
105                final Id<TransitRoute> transitRouteId;
106                final Id<Person> driverId;
107                Id<TransitStopFacility> lastFacilityId;
108
109                LineAndRoute(Id<TransitLine> transitLineId, Id<TransitRoute> transitRouteId, Id<Person> driverId) {
110                        this.transitLineId = transitLineId;
111                        this.transitRouteId = transitRouteId;
112                        this.driverId = driverId;
113                }
114
115                @Override
116                public String toString() {
117                        return "[" + super.toString() + 
118                                        " transitLineId=" + transitLineId +
119                                        " transitRouteId=" + transitRouteId +
120                                        " driverId=" + driverId +
121                                        " lastFacilityId=" + lastFacilityId + "]" ;
122                }
123
124        }

126        public interface LegHandler {
127                void handleLeg(PersonExperiencedLeg leg);
128        }

130        private Network network;
131        private TransitSchedule transitSchedule = null;
132
133        @Inject(optional=true)
134        public void setTransitSchedule(TransitSchedule transitSchedule) {
135                this.transitSchedule = transitSchedule;
136        }
137        private Map<Id<Person>, Leg> legs = new HashMap<>();
138        private Map<Id<Person>, List<Id<Link>>> experiencedRoutes = new HashMap<>();
139        private Map<Id<Person>, Double> relPosOnDepartureLinkPerPerson = new HashMap<>();
140        private Map<Id<Person>, Double> relPosOnArrivalLinkPerPerson = new HashMap<>();
141        private Map<Id<Person>, TeleportationArrivalEvent> routelessTravels = new HashMap<>();
142        private Map<Id<Person>, PendingTransitTravel> transitTravels = new HashMap<>();
143        private Map<Id<Vehicle>, LineAndRoute> transitVehicle2currentRoute = new HashMap<>();
144        private List<LegHandler> legHandlers = new ArrayList<>();

146
147        @Inject
148        EventsToLegs(Network network, EventsManager eventsManager) {
149                this.network = network;
150                eventsManager.addHandler(this);
151        }

155        public EventsToLegs(Scenario scenario) {
156                this.network = scenario.getNetwork();
157                if (scenario.getConfig().transit().isUseTransit()) {
158                        this.transitSchedule = scenario.getTransitSchedule();
159                }
160        }

162        @Override
163        public void reset(int iteration) {
164                legs.clear();
165                experiencedRoutes.clear();
166                transitTravels.clear();
167                routelessTravels.clear();
168                transitVehicle2currentRoute.clear();
169
170                delegate.reset(iteration);
171        }

175        @Override
176        public void handleEvent(PersonDepartureEvent event) {
177                Leg leg = PopulationUtils.createLeg(event.getLegMode());
178                leg.setDepartureTime(event.getTime());
179                legs.put(event.getPersonId(), leg);
180
181                List<Id<Link>> route = new ArrayList<>();
182                route.add(event.getLinkId());
183                experiencedRoutes.put(event.getPersonId(), route);
184        }
185
186        @Override
187        public void handleEvent(PersonEntersVehicleEvent event) {
188                LineAndRoute lineAndRoute = transitVehicle2currentRoute.get(event.getVehicleId());
189                if (lineAndRoute != null
190                                && !event.getPersonId().equals(lineAndRoute.driverId)) { // transit drivers are not considered to travel by transit
191                        transitTravels.put(event.getPersonId(), new PendingTransitTravel(event.getVehicleId(), lineAndRoute.lastFacilityId));
192                }
193        }
194
195        @Override
196        public void handleEvent(LinkLeaveEvent event) {
197
198        }
199
200        @Override
201        public void handleEvent(LinkEnterEvent event) {
202                Id<Person> driverOfVehicle = delegate.getDriverOfVehicle(event.getVehicleId());
203                List<Id<Link>> route = experiencedRoutes.get(driverOfVehicle);
204                route.add(event.getLinkId());
205        }
206
207        @Override
208        public void handleEvent(TeleportationArrivalEvent travelEvent) {
209                routelessTravels.put(travelEvent.getPersonId(), travelEvent);
210        }
211
212        @Override
213        public void handleEvent(VehicleArrivesAtFacilityEvent event) {
214                LineAndRoute lineAndRoute = transitVehicle2currentRoute.get(event.getVehicleId());
215                if (lineAndRoute != null) {
216                        lineAndRoute.lastFacilityId = event.getFacilityId();
217                }
218        }
219
220        @Override
221        public void handleEvent(PersonArrivalEvent event) {
222                Leg leg = legs.get(event.getPersonId());
223                leg.setTravelTime( event.getTime() - leg.getDepartureTime() );
224                double travelTime = leg.getDepartureTime() + leg.getTravelTime() - leg.getDepartureTime();
225                leg.setTravelTime(travelTime);
226                List<Id<Link>> experiencedRoute = experiencedRoutes.get(event.getPersonId());
227                assert experiencedRoute.size() >= 1  ;
228                PendingTransitTravel pendingTransitTravel;
229                if (experiencedRoute.size() > 1) { // different links processed
230                        NetworkRoute networkRoute = RouteUtils.createNetworkRoute(experiencedRoute, null);
231                        networkRoute.setTravelTime(travelTime);
232
233                        /* use the relative position of vehicle enter/leave traffic events on first/last links
234                         * to calculate the correct route distance including the first/last link. 
235                         * (see MATSIM-227) tt feb'16
236                         */
237                        double relPosOnDepartureLink = relPosOnDepartureLinkPerPerson.get(event.getPersonId());
238                        Double relPosOnArrivalLink = relPosOnArrivalLinkPerPerson.get(event.getPersonId());
239                        Gbl.assertNotNull( relPosOnArrivalLink );
240                        networkRoute.setDistance(RouteUtils.calcDistance(networkRoute, relPosOnDepartureLink, 
241                                        relPosOnArrivalLink, network));
242
243                        leg.setRoute(networkRoute);
244                } else if ((pendingTransitTravel = transitTravels.remove(event.getPersonId())) != null) {
245                        // i.e. experiencedRoute.size()==1 && pending transit travel (= person has entered a vehicle)
246
247                        final LineAndRoute lineAndRoute = transitVehicle2currentRoute.get(pendingTransitTravel.vehicleId);
248                        assert lineAndRoute!=null ;
249
250                        final TransitStopFacility accessFacility = transitSchedule.getFacilities().get(pendingTransitTravel.accessStop);
251                        assert accessFacility!=null ;
252
253                        final TransitLine line = transitSchedule.getTransitLines().get(lineAndRoute.transitLineId);
254                        assert line!=null ;
255
256                        final TransitRoute route = line.getRoutes().get(lineAndRoute.transitRouteId);
257                        assert route!=null ;
258
259                        final Id<TransitStopFacility> lastFacilityId = lineAndRoute.lastFacilityId;
260                        if ( lastFacilityId==null ) {
261                                Logger.getLogger(this.getClass()).warn("breakpoint");
262                        }
263                        assert lastFacilityId!=null ;
264
265                        final TransitStopFacility egressFacility = transitSchedule.getFacilities().get(lastFacilityId);
266                        assert egressFacility!=null ;
267
268                        ExperimentalTransitRoute experimentalTransitRoute = new ExperimentalTransitRoute(
269                                        accessFacility,
270                                        line, 
271                                        route,
272                                        egressFacility);
273                        experimentalTransitRoute.setTravelTime(travelTime);
274                        experimentalTransitRoute.setDistance(RouteUtils.calcDistance(experimentalTransitRoute, transitSchedule, network));
275                        leg.setRoute(experimentalTransitRoute);
276                } else {
277                        // i.e. experiencedRoute.size()==1 and no pendingTransitTravel
278
279                        TeleportationArrivalEvent travelEvent = routelessTravels.remove(event.getPersonId());
280                        Route genericRoute = RouteUtils.createGenericRouteImpl(experiencedRoute.get(0), event.getLinkId());
281                        genericRoute.setTravelTime(travelTime);
282                        if (travelEvent != null) {
283                                genericRoute.setDistance(travelEvent.getDistance());
284                        } else {
285                                genericRoute.setDistance(0.0);
286                        }
287                        leg.setRoute(genericRoute);
288                }
289                for (LegHandler legHandler : legHandlers) {
290                        legHandler.handleLeg(new PersonExperiencedLeg(event.getPersonId(), leg));
291                }
292        }
293
294        @Override
295        public void handleEvent(TransitDriverStartsEvent event) {
296                LineAndRoute lineAndRoute = new LineAndRoute(event.getTransitLineId(), event.getTransitRouteId(), event.getDriverId());
297                transitVehicle2currentRoute.put(event.getVehicleId(), lineAndRoute);
298        }
299
300        public void addLegHandler(LegHandler legHandler) {
301                this.legHandlers.add(legHandler);
302        }
303
304        @Override
305        public void handleEvent(VehicleEntersTrafficEvent event) {
306                delegate.handleEvent(event);
307
308                // remember the relative position on the link
309                relPosOnDepartureLinkPerPerson.put(event.getPersonId(), event.getRelativePositionOnLink());
310        }
311
312        @Override
313        public void handleEvent(VehicleLeavesTrafficEvent event) {
314                delegate.handleEvent(event);
315
316                // remember the relative position on the link
317                relPosOnArrivalLinkPerPerson.put(event.getPersonId(), event.getRelativePositionOnLink());
318        }
