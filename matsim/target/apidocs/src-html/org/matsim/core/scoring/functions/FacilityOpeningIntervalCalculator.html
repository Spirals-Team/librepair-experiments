021package org.matsim.core.scoring.functions;
022
023import java.util.Iterator;
024import java.util.Set;
025
026import org.matsim.api.core.v01.population.Activity;
027import org.matsim.core.utils.misc.Time;
028import org.matsim.facilities.ActivityFacilities;
029import org.matsim.facilities.ActivityFacility;
030import org.matsim.facilities.OpeningTime;
031
032/**
033 * Same as CharyparNagelScoringFunction, but retrieves opening time information
034 * from facility object of an activity instead of the config file.
035 *
036 * @author meisterk
037 *
038 */
039public final class FacilityOpeningIntervalCalculator implements OpeningIntervalCalculator {
040
041        private final ActivityFacilities facilities;
042
043        public FacilityOpeningIntervalCalculator(final ActivityFacilities facilities) {
044                this.facilities = facilities;
045        }
046
047        @Override
048        public double[] getOpeningInterval(Activity act) {
049
050                //Â openInterval has two values
051                // openInterval[0] will be the opening time
052                // openInterval[1] will be the closing time
053                double[] openInterval = new double[]{Time.UNDEFINED_TIME, Time.UNDEFINED_TIME};
054
055                boolean foundAct = false;
056
057                ActivityFacility facility = this.facilities.getFacilities().get(act.getFacilityId());
058                Iterator<String> facilityActTypeIterator = facility.getActivityOptions().keySet().iterator();
059                String facilityActType = null;
060                Set<OpeningTime> opentimes = null;
061
062                while (facilityActTypeIterator.hasNext() && !foundAct) {
063
064                        facilityActType = facilityActTypeIterator.next();
065                        if (act.getType().equals(facilityActType)) {
066                                foundAct = true;
067
068                                opentimes = facility.getActivityOptions().get(facilityActType).getOpeningTimes();
069                                if (opentimes != null && !opentimes.isEmpty()) {
070                                        // ignoring lunch breaks with the following procedure:
071                                        // if there is only one open time interval, use it
072                                        // if there are two or more, use the earliest start time and the latest end time
073                                        openInterval[0] = Double.MAX_VALUE;
074                                        openInterval[1] = Double.MIN_VALUE;
075
076                                        for (OpeningTime opentime : opentimes) {
077
078                                                openInterval[0] = Math.min(openInterval[0], opentime.getStartTime());
079                                                openInterval[1] = Math.max(openInterval[1], opentime.getEndTime());
080                                        }
081
082                                }
083
084                        }
085
086                }
087
088                if (!foundAct) {
089                        throw new RuntimeException("No suitable facility activity type found. Aborting...");
090                }
091
092                return openInterval;
093
094        }
095
096
097
098}
