027/**
028 * Class that converts the config parameters into parameters that are used by the scoring.
029 * </p>
030 * Design thoughts:<ul>
031 * <li> Class name feels a bit like a misnomer to me since the class seems to make sense only for the
032 * so-called Charypar-Nagel scoring, especially with the zeroUtilityDuration in the ctor.
033 *   Yet maybe one might be able to rework the approach so that the most
034 * important params (such as typical duration, slope at typical duration, ...) become universal. kai, nov'12
035 * </ul>
036 * 
037 * @author nagel
038 *
039 */
040public final class ActivityUtilityParameters implements MatsimParameters {
041        
042        public static interface ZeroUtilityComputation {
043                double computeZeroUtilityDuration_s( final double priority, final double typicalDuration_s ) ;
044        }
045        public static final class SameAbsoluteScore implements ZeroUtilityComputation {
046                @Override
047                public double computeZeroUtilityDuration_s(double priority, double typicalDuration_s) {
048                        final double priority1 = priority;
049                        final double typicalDuration_s1 = typicalDuration_s;
050                        final double zeroUtilityDuration = typicalDuration_s1 * Math.exp( -10.0 / (typicalDuration_s1 / 3600.0) / priority1 );
051                        // ( the 3600s are in there because the original formulation was in "hours".  So the values in seconds are first
052                        // translated into hours.  kai, sep'12 )
053                        
054                        return zeroUtilityDuration;
055                }
056        }
057        public static final class SameRelativeScore implements ZeroUtilityComputation {
058                @Override
059                public double computeZeroUtilityDuration_s(double priority, double typicalDuration_s) {
060                        final double priority1 = priority;
061                        final double typicalDuration_s1 = typicalDuration_s;
062                        final double zeroUtilityDuration = typicalDuration_s1 * Math.exp( -1.0 / priority1 );
063                        
064                        return zeroUtilityDuration;
065                }
066        }

070         * This is now deliberately an unmodifiable object which can only instantiated by a builder.  If you want/need to modify

076                private String type;
077                private double priority = 1. ;

079                private double closingTime;

081                private double latestStartTime;

087                /**
088                 * empty constructor; deliberately permitted
089                 */
090                public Builder() {
091                }
092
093                /**
094                 * Convenience constructor for main use case
095                 */
096                public Builder(ActivityParams ppp ) {
097                        this.type = ppp.getActivityType() ;
098                        this.priority = ppp.getPriority() ;
099                        this.typicalDuration_s = ppp.getTypicalDuration() ;
100                        this.closingTime = ppp.getClosingTime() ;
101                        this.earliestEndTime = ppp.getEarliestEndTime() ;
102                        this.latestStartTime = ppp.getLatestStartTime() ;
103                        this.minimalDuration = ppp.getMinimalDuration() ;
104                        this.openingTime = ppp.getOpeningTime() ;
105                        this.scoreAtAll = ppp.isScoringThisActivityAtAll() ;
106                        switch( ppp.getTypicalDurationScoreComputation() ) {
107                        case relative:
108                                this.zeroUtilityComputation = new SameRelativeScore() ;
109                                break;
110                        case uniform:
111                                this.zeroUtilityComputation = new SameAbsoluteScore() ;
112                                break;
113                        default:
114                                throw new RuntimeException("not defined");
115                        }
116                        // seems to be somewhat overkill to set a computation method that is only used in the builder ... but the builder has a method to
117                        // (re)set the 
118                }

166                public ActivityUtilityParameters build() {
167                        ActivityUtilityParameters params = new ActivityUtilityParameters(this.type) ;
168                        params.setScoreAtAll(this.scoreAtAll) ;
169                        params.setTypicalDuration( this.typicalDuration_s) ;
170                        params.setZeroUtilityDuration_s( this.zeroUtilityComputation.computeZeroUtilityDuration_s(priority, typicalDuration_s)) ;
171                        params.setClosingTime(this.closingTime) ;
172                        params.setEarliestEndTime(this.earliestEndTime) ;
173                        params.setLatestStartTime(this.latestStartTime) ;
174                        params.setMinimalDuration(this.minimalDuration) ;
175                        params.setOpeningTime(this.openingTime) ;
176                        params.checkConsistency();
177                        return params ;
178                }
179
180                public final Builder setZeroUtilityComputation(ZeroUtilityComputation zeroUtilityComputation) {
181                        this.zeroUtilityComputation = zeroUtilityComputation;
182                        return this;
183                }

191         * (from Dave's paper, ga-acts-iatbr03.tex, though he called it t_0)

193         */
194        private double zeroUtilityDuration_h; // in hours!
195        private double minimalDuration = -1;
196        private double openingTime = -1;

199        private double earliestEndTime = -1;

202        // use factory.  nov'12
203        /*package!*/ ActivityUtilityParameters(final String type) {
204                this.type = type;       
205        }
206
207        /*package!*/ final void checkConsistency() {
208                //if typical duration is <=48 seconds (and priority=1) then zeroUtilityDuration becomes 0.0 because of the double precision. This means it is not possible
209                // to have activities with a typical duration <=48 seconds (GL/June2011)
210                if (this.scoreAtAll && this.zeroUtilityDuration_h == 0.0) {
211                        throw new RuntimeException("zeroUtilityDuration of type " + type + " must be greater than 0.0. Did you forget to specify the typicalDuration?");
212                }
213        }
214        
215        /*package!*/ final void setScoreAtAll(boolean scoreAtAll) {
216                this.scoreAtAll = scoreAtAll;
217        }
218

220                this.typicalDuration_s = typicalDuration_s;
221        }
222
223        /*package!*/ final void setZeroUtilityDuration_s(final double val) {
224
225
226                this.zeroUtilityDuration_h = val / 3600. ;
227
228                // example: pt interaction activity with typical duration = 120sec.
229                // 120/3600 * exp( -10 / (120 / 3600) ) =  1.7 x 10^(-132)  (!!!!!!!!!!)
230                // In consequence, even a pt interaction of one seconds causes a fairly large utility.
231
232        }
