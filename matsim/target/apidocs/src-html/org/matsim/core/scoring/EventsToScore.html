040/**
041 * Calculates the score of the selected plans of a given scenario
042 * based on events. The final scores are written to the selected plans of each person in the
043 * scenario when you call finish on this instance if you created this instance with the corresponding
044 * factory method.
045 *
046 * The Controler does not use this class, but rather uses its delegates directly.
047 * Create your own instance if you want to compute scores from an Event file, for example.
048 *
049 * @author mrieser, michaz
050 */
051public final class EventsToScore implements BasicEventHandler {
052
053        private final NewScoreAssigner newScoreAssigner;
054        private final ControlerListenerManagerImpl controlerListenerManager;
055        private ScoringFunctionsForPopulation scoringFunctionsForPopulation;
056        private final Population population;
057
058        private boolean finished = false;
059
060        private int iteration = -1 ;
061
062        @Inject
063        private EventsToScore(ControlerListenerManagerImpl controlerListenerManager, EventsManager eventsManager, ScoringFunctionsForPopulation scoringFunctionsForPopulation, final Scenario scenario, NewScoreAssigner newScoreAssigner) {
064                this.controlerListenerManager = controlerListenerManager;
065                this.scoringFunctionsForPopulation = scoringFunctionsForPopulation;
066                this.population = scenario.getPopulation();
067                this.newScoreAssigner = newScoreAssigner;
068                eventsManager.addHandler(this);
069        }
070
071        public static EventsToScore createWithScoreUpdating(final Scenario scenario, final ScoringFunctionFactory scoringFunctionFactory, final EventsManager eventsManager) {
072                com.google.inject.Injector injector = Injector.createInjector(scenario.getConfig(),
073                                new ScenarioByInstanceModule(scenario),
074                                new ExperiencedPlansModule(),
075                                new AbstractModule() {
076                                        @Override
077                                        public void install() {
078                                                bind(ScoringFunctionsForPopulation.class).asEagerSingleton();
079                                                bind(ScoringFunctionFactory.class).toInstance(scoringFunctionFactory);
080                                                bind(NewScoreAssigner.class).to(NewScoreAssignerImpl.class).asEagerSingleton();
081                                                bind(EventsToScore.class).asEagerSingleton();
082                                                bind(ControlerListenerManagerImpl.class).asEagerSingleton();
083                                                bind(ControlerListenerManager.class).to(ControlerListenerManagerImpl.class);
084                                                bind(EventsManager.class).toInstance(eventsManager);
085                                        }
086                                });
087                return injector.getInstance(EventsToScore.class);
088        }
089
090        public static EventsToScore createWithoutScoreUpdating(Scenario scenario, final ScoringFunctionFactory scoringFunctionFactory, final EventsManager eventsManager) {
091                com.google.inject.Injector injector = Injector.createInjector(scenario.getConfig(),
092                                new ScenarioByInstanceModule(scenario),
093                                new ExperiencedPlansModule(),
094                                new AbstractModule() {
095                                        @Override
096                                        public void install() {
097                                                bind(ScoringFunctionsForPopulation.class).asEagerSingleton();
098                                                bind(ScoringFunctionFactory.class).toInstance(scoringFunctionFactory);
099                                                bind(NewScoreAssigner.class).to(NoopNewScoreAssignerImpl.class).asEagerSingleton();
100                                                bind(EventsToScore.class).asEagerSingleton();
101                                                bind(ControlerListenerManagerImpl.class).asEagerSingleton();
102                                                bind(ControlerListenerManager.class).to(ControlerListenerManagerImpl.class);
103                                                bind(EventsManager.class).toInstance(eventsManager);
104                                        }
105                                });
106                return injector.getInstance(EventsToScore.class);
107        }
108
109        public void beginIteration(int iteration) {
110                this.iteration = iteration;
111                this.controlerListenerManager.fireControlerIterationStartsEvent(iteration);
112        }
113
114        /**
115         * Finishes the calculation of the plans' scores and assigns the new scores
116         * to the plans if desired.
117         */
118        public void finish() {
119                if (iteration == -1) {
120                        throw new RuntimeException("Please initialize me before the iteration starts.");
121                }
122                controlerListenerManager.fireControlerAfterMobsimEvent(iteration);
123                scoringFunctionsForPopulation.finishScoringFunctions();
124                newScoreAssigner.assignNewScores(this.iteration, scoringFunctionsForPopulation, population);
125                finished = true;
126        }
127
128        /**
129         * Returns the score of a single agent. This method only returns useful
130         * values if the method {@link #finish() } was called before. description
131         *
132         * @param agentId
133         *            The id of the agent the score is requested for.
134         * @return The score of the specified agent.
135         */
136        public Double getAgentScore(final Id<Person> agentId) {
137                if (!finished) {
138                        throw new IllegalStateException("Must call finish first.");
139                }
140                ScoringFunction scoringFunction = scoringFunctionsForPopulation.getScoringFunctionForAgent(agentId);
141                if (scoringFunction == null)
142                        return null;
143                return scoringFunction.getScore();
144        }

157        }

164        }
