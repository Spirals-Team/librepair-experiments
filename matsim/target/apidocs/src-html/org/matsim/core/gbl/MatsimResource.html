021package org.matsim.core.gbl;
022
023import java.awt.Image;
024import java.io.File;
025import java.io.FileInputStream;
026import java.io.FileNotFoundException;
027import java.io.IOException;
028import java.io.InputStream;
029import java.net.MalformedURLException;
030import java.net.URL;
031
032import javax.imageio.ImageIO;
033
034import org.apache.log4j.Logger;
035
036/**
037 * Helper class to load data from files in the resource directory
038 * (<code>./res/</code>). Because the resource directory may be included into
039 * jar-files in releases, one must not directly access them with the hard-coded
040 * path (e.g. <code>new File("res/foo.bar");</code>), as the file would not be
041 * found if it is inside a jar-file. Instead, use the methods in this class to
042 * access files in the resource directory. The methods in this class ensure
043 * that the files can be loaded no matter whether they are inside a jar-file or
044 * not.<br>
045 *
046 * All filenames must be given relative to the resource directory. This means,
047 * that the name of the resource-directory must not be part of the filenames
048 * passed to the methods. E.g., to access the file <code>res/foo/bar.txt</code>,
049 * where <code>res/</code> is the resource directory, one must only pass
050 * <code>foo/bar.txt</code> as filename.
051 *
052 * @author mrieser
053 */
054public abstract class MatsimResource {
055
056        /** The path where resources are located in jar-files. */
057        private static final String RES_PATH_JARFILE = "/res/";
058
059        /** The path where resources are located in a local file system. */
060        private static final String RES_PATH_LOCAL = "./res/";  //NOPMD // this line should be ignored for PMD analysis
061        private static final String RES_PATH_LOCAL2 = "./src/main/resources/res/";  //NOPMD // this line should be ignored for PMD analysis
062
063        private static final Logger log = Logger.getLogger(MatsimResource.class);
064
065        /**
066         * @param filename relative path from within the resource directory to a file to be loaded
067         * @return an URL pointing to the requested resource file, or <code>null</code> if no such file exists.
068         */
069        public final static URL getAsURL(final String filename) {
070                // look for the file locally
071                {
072                File file = new File(RES_PATH_LOCAL + filename);
073                if (file.exists()) {
074                        try {
075                                return file.toURI().toURL();
076                        } catch (MalformedURLException e) {
077                                log.warn("Found resource-file, but could not return URL for it.", e);                           // just continue, maybe we have more luck in the classpath
078                        }
079                }
080                }
081                {
082                File file = new File(RES_PATH_LOCAL2 + filename);
083                if (file.exists()) {
084                        try {
085                                return file.toURI().toURL();
086                        } catch (MalformedURLException e) {
087                                log.warn("Found resource-file, but could not return URL for it.", e);                           // just continue, maybe we have more luck in the classpath
088                        }
089                }
090                }
091                // maybe we find the file in the classpath, possibly inside a jar-file
092                URL url = MatsimResource.class.getResource(RES_PATH_JARFILE + filename);
093                if (url == null) {
094                        log.warn("Resource '" + filename + "' not found!");
095                }
096                return url;
097        }
098
099        /**
100         * @param filename relative path from within the resource directory to a file to be loaded
101         * @return a Stream to the requested resource file, or <code>null</code> if no such file exists.
102         */
103        public final static InputStream getAsInputStream(final String filename) {
104                // look for the file locally
105                try {
106                        return new FileInputStream(RES_PATH_LOCAL + filename);
107                } catch (FileNotFoundException e) {
108                        log.info("Resource '" + filename + "' not found locally. May not be fatal.");
109                        // just continue, maybe we have more luck in the classpath
110                }
111                // maybe we find the file in the classpath, possibly inside a jar-file
112                InputStream stream = MatsimResource.class.getResourceAsStream(RES_PATH_JARFILE + filename);
113                if (stream == null) {
114                        log.warn("Resource '" + filename + "' not found!");
115                }
116                return stream;
117        }
118
119        /**
120         * @param filename relative path from within the resource directory to a file to be loaded
121         * @return a Stream to the requested resource file, or <code>null</code> if no such file exists.
122         */
123        public final static Image getAsImage(final String filename) {
124                final URL url = getAsURL(filename);
125                if (url == null) {
126                        return null;
127                }
128                try {
129                        return ImageIO.read(url);
130                } catch (IOException e) {
131                        log.error("Could not load requested image", e);
132                        return null;
133                }
134        }
135}
