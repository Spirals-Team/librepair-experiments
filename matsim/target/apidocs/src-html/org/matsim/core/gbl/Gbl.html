050        public static final void printMemoryUsage() {
051                long totalMem = Runtime.getRuntime().totalMemory();
052                long freeMem = Runtime.getRuntime().freeMemory();
053                long usedMem = totalMem - freeMem;
054                log.info("used RAM: " + usedMem + "B = " + (usedMem/1024) + "kB = " + (usedMem/1024/1024) + "MB" +
055                                "  free: " + freeMem + "B = " + (freeMem/1024/1024) + "MB  total: " + totalMem + "B = " + (totalMem/1024/1024) + "MB");
056        }

074        /** 
075         * Prints some information about the current build/revision of this code.
076         * Currently, this will only work with the Nightly-Build-Jars.
077         */
078        public static final String getBuildInfoString(String component, String resourceFilename) {
079                String revision = null;
080                String date = null;
081                URL url = Gbl.class.getResource(resourceFilename);
082                if (url != null) {
083                        try (BufferedReader reader = new BufferedReader(new InputStreamReader(url.openStream()))) {
084                                revision = reader.readLine();
085                                date = reader.readLine();
086                        } catch (IOException e) {
087                                e.printStackTrace();
088                        }
089                        if (revision == null) {
090                                return component + "-Build: unknown";
091                        } else {
092                                return component + "-Build: " + revision + " (" + date + ")";
093                        }
094                } else {
095                        return component + "-Build: unknown";
096                }
097        }

112        private static long measurementStartTime = Long.MAX_VALUE;
113
114        private static final String printTime() {
115                if (Gbl.measurementStartTime == Long.MAX_VALUE) {
116                        log.error("Did not start measurements.");
117                        return "";
118                }
119                return printTimeDiff(System.currentTimeMillis(), Gbl.measurementStartTime);
120        }
121
122        private static final String printTimeDiff(final long later, final long earlier) {
123                long elapsedTimeMillis = later - earlier;
124                float elapsedTimeSec = elapsedTimeMillis/1000F;
125                float elapsedTimeMin = elapsedTimeMillis/(60*1000F);
126                float elapsedTimeHour = elapsedTimeMillis/(60*60*1000F);
127                float elapsedTimeDay = elapsedTimeMillis/(24*60*60*1000F);
128
129                return elapsedTimeMillis + " msecs; " +
130                        elapsedTimeSec    + " secs; " +
131                        elapsedTimeMin    + " mins; " +
132                        elapsedTimeHour   + " hours; " +
133                        elapsedTimeDay    + " days ###";
134        }

149        //////////////////////////////////////////////////////////////////////
150        // thread performance
151        //////////////////////////////////////////////////////////////////////
152
153        private final static ThreadMXBean tbe = ManagementFactory.getThreadMXBean();
154
155        /**
156         * Tries to enable CPU time measurement for threads. Not all JVMs support this feature.
157         *
158         * @return <code>true</code> if the JVM supports time measurement for threads and the feature
159         * could be enabled, <code>false</code> otherwise.
160         */
161        public static final boolean enableThreadCpuTimeMeasurement() {
162                if (tbe.isThreadCpuTimeSupported()) {
163                        tbe.setThreadCpuTimeEnabled(true);
164                        return true;
165                }
166                return false;
167        }
168
169        /**
170         * @param thread
171         * @return cpu time for the given thread in seconds, <code>-1</code> if cpu time is not measured.
172         */
173        public static final double getThreadCpuTime(final Thread thread) {
174                if (tbe.isThreadCpuTimeEnabled()) {
175                        return tbe.getThreadCpuTime(thread.getId()) / 1.0e9;
176                }
177                return -1;
178        }
179
180        /**
181         * Prints the cpu time for the given thread, i.e. the time the thread was effectively active on the CPU.
182         *
183         * @param thread
184         */
185        public static final void printThreadCpuTime(final Thread thread) {
186                if (tbe.isThreadCpuTimeEnabled()) {
187                        log.info("Thread performance: Thread=" + thread.getName() + "  cpu-time=" + getThreadCpuTime(thread) + "sec");
188                }
189        }
190
191        /**
192         * Prints the cpu time for the current thread, i.e. the time the current thread was effectively active on the CPU.
193         */
194        public static final void printCurrentThreadCpuTime() {
195                printThreadCpuTime(Thread.currentThread());
196        }
