043        private boolean removeMalformed = false;
044        
045        public LanesConsistencyChecker(Network net, Lanes laneDefs) {
046                this.network = net;
047                this.lanes = laneDefs;
048        }
049        
050        public void checkConsistency() {
051                log.info("checking consistency...");
052                List<Id<Link>> linksWithMalformedLanes = new LinkedList<>();
053                for (LanesToLinkAssignment l2l : this.lanes.getLanesToLinkAssignments().values()){
054                        if (!isLaneOnLinkConsistent(l2l)){
055                                linksWithMalformedLanes.add(l2l.getLinkId());
056                        }
057                }               
058                
059                if (this.removeMalformed){
060                        for (Id<Link> linkId : linksWithMalformedLanes) {
061                                this.lanes.getLanesToLinkAssignments().remove(linkId);
062                                log.info("remove lanes on link " + linkId);
063                        }
064                }
065                log.info("checked consistency. Lanes on " + linksWithMalformedLanes.size() + " links have been removed.");
066        }
067

070                if (!this.network.getLinks().containsKey(l2l.getLinkId())) {
071                        log.error("No link found for lanesToLinkAssignment on link Id(linkIdRef): "  + l2l.getLinkId());

073                }
074                //check length
075                else {
076                        Link link = this.network.getLinks().get(l2l.getLinkId());
