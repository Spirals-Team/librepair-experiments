001package org.matsim.lanes.data;
002
003import org.matsim.api.core.v01.Id;
004import org.matsim.api.core.v01.network.Link;
005import org.matsim.utils.objectattributes.attributable.Attributable;
006
007import java.util.List;
008
009/**
010 * Models a lane on a link.
011 * 
012 * The position is specified via the distance the lane starts measured from the
013 * downstream node of the link and the ids of the downstream lanes. If the
014 * distance to the downstream node equals the link length, this implies that
015 * this is the most upstream (first) lane on a link. More than one first lane is
016 * permitted. If the first or any other lane shall model more than one lane that
017 * exists in the reality to be modeled, increase the number of represented lanes
018 * attribute.
019 * 
020 * The alignment is used to visualize parallel lanes in the correct order. The
021 * higher the alignment the more right the lane is visualized on a link. Lanes
022 * leading straight on are assigned with an alignment of zero, lanes leading
023 * left with an negative and lanes leading right with an positive alignment.
024 * 
025 * For an easy modeling use the v11 lane model and the conversion class in the
026 * package.
027 * 
028 * @author dgrether
029 * 
030 */
031public interface Lane extends Attributable {
032
033        public Id<Lane> getId();
034
035        public double getNumberOfRepresentedLanes();
036
037        public void setNumberOfRepresentedLanes(double number);
038
039        public void setStartsAtMeterFromLinkEnd(double meter);
040
041        public double getStartsAtMeterFromLinkEnd();
042
043        public double getCapacityVehiclesPerHour();
044
045        public void setCapacityVehiclesPerHour(double capacity);
046
047        public void addToLinkId(Id<Link> id);
048
049        public void addToLaneId(Id<Lane> id);
050        /**
051         *
052         * @return List may be null if nothing is set
053         */
054        public List<Id<Link>> getToLinkIds();
055        /**
056         *
057         * @return List may be null if nothing is set
058         */
059        public List<Id<Lane>> getToLaneIds();
060
061        public void setAlignment(int alignment);
062
063        public int getAlignment();
064
065}
