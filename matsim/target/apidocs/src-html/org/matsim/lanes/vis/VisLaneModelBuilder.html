044        public void recalculatePositions(VisLinkWLanes linkData, SnapshotLinkWidthCalculator linkWidthCalculator) {
045//              log.error("recalculatePositions...");
046                double linkWidth = linkWidthCalculator.calculateLinkWidth(linkData.getNumberOfLanes()) ;
047                linkData.setLinkWidth(linkWidth);
048                Point2D.Double linkStartCenter = this.calculatePointOnLink(linkData, 0.0, 0.5);
049                linkData.setLinkStartCenterPoint(linkStartCenter);
050                if (linkData.getLaneData() == null || linkData.getLaneData().isEmpty()){
051                        //Calculate end point center
052                        double x = linkData.getLinkEnd().x + (0.5 * linkWidth * linkData.getLinkOrthogonalVector().x);
053                        double y = linkData.getLinkEnd().y + (0.5 * linkWidth * linkData.getLinkOrthogonalVector().y);
054                        linkData.setLinkEndCenterPoint(new Point2D.Double(x, y));
055//                      log.error("link " + linkData.getLinkId() + " without lanes starts at " + linkData.getLinkStartCenterPoint() + " and ends at " + x + " " + y + " but " + linkData.getLinkEnd());
056                }
057                else {
058                        double numberOfLinkParts = (2 * linkData.getMaximalAlignment()) + 2;
059//                      log.error("link with lanes " + linkData.getLinkId() + " starts at " + linkData.getLinkStart() + " and ends at " + linkData.getLinkEnd());
060                        for (VisLane lane : linkData.getLaneData().values()){
061                                double horizontalFraction = 0.5 - (lane.getAlignment() / numberOfLinkParts);
062                                Point2D.Double laneStart = calculatePointOnLink(linkData, lane.getStartPosition(), horizontalFraction);
063                                Point2D.Double laneEnd = calculatePointOnLink(linkData, lane.getEndPosition(), horizontalFraction);
064                                lane.setStartEndPoint(laneStart, laneEnd);
065//                              log.error("lane " + lane.getId() + " starts at " + laneStart + " ends at " + laneEnd);
066                                if (lane.getNumberOfLanes() >= 2.0){
067                                        double noLanesFloor = Math.floor(lane.getNumberOfLanes());
068                                        double laneOffset = - noLanesFloor / 2 * linkWidthCalculator.getLaneWidth();
069                                        if (noLanesFloor % 2 == 0){
070                                                laneOffset = laneOffset + (linkWidthCalculator.getLaneWidth()/2);
071                                        }
072                                        for (int i = 1; i <= noLanesFloor; i++){
073                                                Point2D.Double drivingLaneStart = this.calcPoint(laneStart, linkData.getLinkOrthogonalVector(), laneOffset);
074                                                Point2D.Double drivingLaneEnd = this.calcPoint(laneEnd, linkData.getLinkOrthogonalVector(), laneOffset);
075                                                lane.addDrivingLane(i, drivingLaneStart, drivingLaneEnd);
076                                                laneOffset = laneOffset + linkWidthCalculator.getLaneWidth();
077                                        }
078                                }
079                        }
080                }
081        }
082        
083        private Point2D.Double calculatePointOnLink(final VisLinkWLanes laneLinkData, final double position, final double horizontalFraction) {
084                Point2D.Double lenghtPoint = this.calcPoint(laneLinkData.getLinkStart(), laneLinkData.getNormalizedLinkVector(), position);
085                return this.calcPoint(lenghtPoint, laneLinkData.getLinkOrthogonalVector(), horizontalFraction * laneLinkData.getLinkWidth());
086        }
087        
088        public Point2D.Double calcPoint(Point2D.Double start, Point2D.Double vector, double distance){
089                double x = start.getX() + (distance * vector.x);
090                double y = start.getY() + (distance * vector.y);
091                return new Point2D.Double(x, y);
092        }

094        private VisLane createVisLane(ModelLane qlane, double linkLength, double linkScale, double linkLengthCorrectionFactor) {
095                String id = qlane.getLaneData().getId().toString();
096                double startPosition = (linkLength -  qlane.getLaneData().getStartsAtMeterFromLinkEnd()) * linkScale * linkLengthCorrectionFactor;
097                double endPosition = startPosition + (qlane.getLength() *  linkScale * linkLengthCorrectionFactor);
098//              log.error("lane " + qlane.getId() + " starts at: " + startPosition + " and ends at : " +endPosition);
099                int alignment = qlane.getLaneData().getAlignment();
100                VisLane lane = new VisLane(id);
101                lane.setStartPosition(startPosition);
102                lane.setEndPosition(endPosition);
103                lane.setAlignment(alignment);
104                lane.setNumberOfLanes(qlane.getLaneData().getNumberOfRepresentedLanes());
105                return lane;
106        }

110        public void connect(Map<String, VisLinkWLanes> otfNetwork){
111                for (VisLinkWLanes otfLink : otfNetwork.values()){
112                        if (otfLink.getLaneData() == null || otfLink.getLaneData().isEmpty()){
113                                if (otfLink.getToLinkIds() != null){
114                                        for (String toLinkId : otfLink.getToLinkIds()){
115                                                VisLinkWLanes toLink = otfNetwork.get(toLinkId);
116                                                otfLink.addToLink(toLink);
117                                        }
118                                }
119                        }
120                        else {
121                                for (VisLane otfLane : otfLink.getLaneData().values()){
122                                        if (otfLane.getToLinkIds() != null) {
123                                                for (String toLinkId : otfLane.getToLinkIds()){
124                                                        VisLinkWLanes toLink = otfNetwork.get(toLinkId);
125                                                        otfLane.addToLink(toLink);
126                                                }
127                                        }
128                                }
129                        }
130                }
131        }

133        public VisLinkWLanes createVisLinkLanes(CoordinateTransformation transform, VisLink link, double nodeOffsetMeter, List<ModelLane> lanes) {
134//              log.error("");
135//              log.error("link " + link.getLink().getId() + " ... ");
136//              log.debug("  fromNode: " + link.getLink().getFromNode().getId() + " coord: " + link.getLink().getFromNode().getCoord());
137//              log.debug("  toNode: " + link.getLink().getToNode().getId() + " coord: " + link.getLink().getToNode().getCoord());
138                Coord linkStartCoord = transform.transform(link.getLink().getFromNode().getCoord());
139                Coord linkEndCoord = transform.transform(link.getLink().getToNode().getCoord());
140                Point2D.Double linkStart = new Point2D.Double(linkStartCoord.getX(), linkStartCoord.getY());
141                Point2D.Double linkEnd =  new Point2D.Double(linkEndCoord.getX(), linkEndCoord.getY());
142                
143                //calculate length and normal
144                Point2D.Double deltaLink = new Point2D.Double(linkEnd.x - linkStart.x, linkEnd.y - linkStart.y);
145                double euclideanLinkLength = this.calculateEuclideanLinkLength(deltaLink);
146                //calculate the correction factor if real link length is different than euclidean distance
147                double linkLengthCorrectionFactor = euclideanLinkLength / link.getLink().getLength();
148                Point2D.Double deltaLinkNorm = new Point2D.Double(deltaLink.x / euclideanLinkLength, deltaLink.y / euclideanLinkLength);
149                Point2D.Double normalizedOrthogonal = new Point2D.Double(deltaLinkNorm.y, - deltaLinkNorm.x);
150                
151                //first calculate the scale of the link based on the node offset, i.e. the link will be shortened at the beginning and the end 
152                double linkScale = 1.0;
153                if ((euclideanLinkLength * 0.2) > (2.0 * nodeOffsetMeter)){ // 2* nodeoffset is more than 20%
154                        linkScale = (euclideanLinkLength - (2.0 * nodeOffsetMeter)) / euclideanLinkLength;
155                }
156                else { // use 80 % as euclidean length
157                        linkScale = euclideanLinkLength * 0.8 / euclideanLinkLength;
158                }
159                
160                //scale the link 
161                Tuple<Double, Double> scaledLink = VectorUtils.scaleVector(linkStart, linkEnd, linkScale);
162                Point2D.Double scaledLinkEnd = scaledLink.getSecond();
163                Point2D.Double scaledLinkStart = scaledLink.getFirst();
164//              log.error("scaledLinkStart: " + scaledLinkStart + " end: " + scaledLinkEnd);
165                
166                VisLinkWLanes lanesLinkData = new VisLinkWLanes(link.getLink().getId().toString());
167                lanesLinkData.setLinkStartEndPoint(scaledLinkStart, scaledLinkEnd);
168                lanesLinkData.setNormalizedLinkVector(deltaLinkNorm);
169                lanesLinkData.setLinkOrthogonalVector(normalizedOrthogonal);
170                lanesLinkData.setNumberOfLanes(link.getLink().getNumberOfLanes());
171
172                if (lanes != null){
173                        int maxAlignment = 0;
174                        for (ModelLane lane : lanes){
175                                VisLane visLane = this.createVisLane(lane, link.getLink().getLength(), linkScale, linkLengthCorrectionFactor);
176                                lanesLinkData.addLaneData(visLane);
177                                if (visLane.getAlignment() > maxAlignment) {
178                                        maxAlignment = visLane.getAlignment();
179                                }
180                        }
181                        lanesLinkData.setMaximalAlignment(maxAlignment);
182
183                        //connect the lanes 
184                        for (ModelLane lane : lanes){
185                                VisLane otfLane = lanesLinkData.getLaneData().get(lane.getLaneData().getId().toString());
186                                if (lane.getToLanes() == null ||  lane.getToLanes().isEmpty()){
187                                        for (Id<Link> id : lane.getLaneData().getToLinkIds()){
188                                                otfLane.addToLinkId(id.toString());
189                                        }
190                                }
191                                else {
192                                        for (ModelLane toLane : lane.getToLanes()){
193                                                VisLane otfToLane = lanesLinkData.getLaneData().get(toLane.getLaneData().getId().toString());
194                                                otfLane.addToLane(otfToLane);
195                                        }
196                                }
197                        }
198                }
199                else {
200                        for (Id<Link> id : link.getLink().getToNode().getOutLinks().keySet()){
201                                lanesLinkData.addToLinkId(id.toString());
202                        }
203                }
204                
205                return lanesLinkData;
206        }

210  private double calculateEuclideanLinkLength(Point2D.Double deltaLink) {
211        return Math.sqrt(Math.pow(deltaLink.x, 2) + Math.pow(deltaLink.y, 2));
212  }
