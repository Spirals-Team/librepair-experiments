020package org.matsim.lanes.vis;
021
022import org.matsim.api.core.v01.Coord;
023import org.matsim.core.mobsim.qsim.interfaces.SignalGroupState;
024import org.matsim.core.utils.collections.Tuple;
025import org.matsim.core.utils.geometry.CoordUtils;
026
027import java.awt.geom.Point2D;
028import java.io.Serializable;
029import java.util.ArrayList;
030import java.util.HashMap;
031import java.util.List;
032import java.util.Map;
033
034
035/**
036 * @author dgrether
037 */
038public class VisLane implements Serializable {
039        
040        private String id = null;
041        private double endPosition;
042        private double startPosition;
043        private int alignment;
044        private double numberOfLanes;
045        private List<VisLane> toLanes = null;
046        private SignalGroupState state = null;
047        private Point2D.Double startPoint = null;
048        private Point2D.Double endPoint = null;
049        private Map<String, VisSignal> signals = null;
050        private List<String> toLinkIds;
051        private transient List<VisLinkWLanes> toLinksData = null;
052        private double euklideanDistance;
053        private Coord startCoord;
054        private Coord endCoord; 
055        private Map<Integer, Tuple<Coord, Coord>> drivingLaneMap = null;
056        
057        public VisLane(String id) {
058                this.id = id;
059        }
060        
061        public void setId(String id){
062                this.id = id;
063        }
064
065        public String getId() {
066                return id;
067        }
068
069        public void setEndPosition(double pos) {
070                this.endPosition = pos;
071        }
072        
073        public double getEndPosition() {
074                return endPosition;
075        }
076
077        
078        public double getStartPosition() {
079                return startPosition;
080        }
081
082        
083        public void setStartPosition(double pos) {
084                this.startPosition = pos;
085        }
086
087        public void setAlignment(int alignment) {
088                this.alignment = alignment;
089        }
090        
091        public int getAlignment(){
092                return this.alignment;
093        }
094        
095        public double getNumberOfLanes() {
096                return this.numberOfLanes;
097        }
098        
099        public void setNumberOfLanes(double noLanes){
100                this.numberOfLanes = noLanes;
101        }
102
103        public void setSignalGroupState(SignalGroupState state) {
104                this.state = state;
105        }
106        
107        public SignalGroupState getSignalGroupState(){
108                return this.state ;
109        }
110
111        public void addToLink(VisLinkWLanes toLink) {
112                if (this.toLinksData == null) {
113                        this.toLinksData = new ArrayList<VisLinkWLanes>();
114                }
115                this.toLinksData.add(toLink);
116        }
117
118        public void addToLane(VisLane toLane) {
119                if (this.toLanes == null){
120                        this.toLanes = new ArrayList<VisLane>();
121                }
122                this.toLanes.add(toLane);
123        }
124
125        
126        public Point2D.Double getStartPoint() {
127                return startPoint;
128        }
129
130        
131        public void setStartEndPoint(Point2D.Double startPoint, Point2D.Double endPoint) {
132                this.startPoint = startPoint;
133                this.endPoint = endPoint;
134                this.calcCoords();
135        }
136        
137        public Point2D.Double getEndPoint() {
138                return endPoint;
139        }
140        
141        private void calcCoords(){
142                this.startCoord = new Coord(startPoint.x, startPoint.y);
143                this.endCoord = new Coord(endPoint.x, endPoint.y);
144                this.euklideanDistance = CoordUtils.calcEuclideanDistance(startCoord, endCoord);
145        }
146
147        public void addDrivingLane(int laneNumber, Point2D.Double  drivingLaneStart, Point2D.Double drivingLaneEnd) {
148                if (this.drivingLaneMap == null){
149                        this.drivingLaneMap = new HashMap<Integer, Tuple<Coord, Coord>>();
150                }
151                Tuple<Coord, Coord> tuple = new Tuple<Coord, Coord>(new Coord(drivingLaneStart.x, drivingLaneStart.y), new Coord(drivingLaneEnd.x, drivingLaneEnd.y));
152                this.drivingLaneMap.put(laneNumber, tuple);
153        }
154        
155        public Tuple<Coord, Coord> getDrivingLaneStartEndCoord(int laneNumber){
156                return this.drivingLaneMap.get(laneNumber);
157        }
158
159        
160        public Coord getStartCoord() {
161                return this.startCoord;
162        }
163
164        public Coord getEndCoord() {
165                return this.endCoord;
166        }
167
168        public void addSignal(VisSignal signal) {
169                if (this.signals == null){
170                        this.signals = new HashMap<String, VisSignal>();
171                }
172                this.signals.put(signal.getId(), signal);
173        }
174        
175        public Map<String, VisSignal> getSignals(){
176                return this.signals;
177        }
178
179        
180        public List<VisLinkWLanes> getToLinks() {
181                return toLinksData;
182        }
183        
184        public void addToLinkId(String toLinkId){
185                if (this.toLinkIds == null)
186                        this.toLinkIds = new ArrayList<>();
187                this.toLinkIds.add(toLinkId);
188        }
189
190        public List<String> getToLinkIds() {
191                return toLinkIds ;
192        }
193        
194        public List<VisLane> getToLanes() {
195                return toLanes;
196        }
197
198        
199        public double getEuklideanDistance() {
200                return euklideanDistance;
201        }
202
203
204
205}
