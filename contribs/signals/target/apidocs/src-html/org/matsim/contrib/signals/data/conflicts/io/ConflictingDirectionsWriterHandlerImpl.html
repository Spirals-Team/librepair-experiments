021package org.matsim.contrib.signals.data.conflicts.io;
022
023import java.io.BufferedWriter;
024import java.io.IOException;
025import java.util.List;
026
027import org.matsim.api.core.v01.Id;
028import org.matsim.api.core.v01.network.Node;
029import org.matsim.contrib.signals.data.conflicts.ConflictData;
030import org.matsim.contrib.signals.data.conflicts.IntersectionDirections;
031import org.matsim.contrib.signals.data.conflicts.Direction;
032import org.matsim.contrib.signals.model.SignalSystem;
033
034/**
035 * @author tthunig
036 */
037public class ConflictingDirectionsWriterHandlerImpl implements ConflictingDirectionsWriterHandler {
038
039        @Override
040        public void writeHeaderAndStartElement(BufferedWriter out) throws IOException {
041                out.write("<?xml version=\"1.0\" encoding=\"utf-8\"?>\n");
042//              out.write("<!DOCTYPE conflictData SYSTEM \"" + MatsimXmlWriter.DEFAULT_DTD_LOCATION + "conflictData.dtd\">\n\n");
043        }
044
045        @Override
046        public void startConflictData(BufferedWriter out) throws IOException {
047                out.write("<conflictData>\n\n");
048        }
049
050        @Override
051        public void endConflictData(BufferedWriter out) throws IOException {
052                out.write("</conflictData>\n");
053        }
054
055        @Override
056        public void writeIntersections(ConflictData conflictData, BufferedWriter out) throws IOException {
057                for (IntersectionDirections conflictingDirections : conflictData.getConflictsPerSignalSystem().values()) {
058                        this.startIntersection(conflictingDirections.getNodeId(), conflictingDirections.getSignalSystemId(), out);
059                        for (Direction direction : conflictingDirections.getDirections().values()) {
060                                this.writeDirection(direction, out);
061                        }
062                        this.endIntersection(out);
063                }
064        }
065
066        private void startIntersection(Id<Node> nodeId, Id<SignalSystem> systemId, BufferedWriter out) throws IOException {
067                out.write("\t<intersection");
068                out.write(" nodeId=\"" + nodeId + "\"");
069                out.write(" signalSystemId=\"" + systemId + "\"");
070                out.write(" >\n");
071        }
072
073        private void endIntersection(BufferedWriter out) throws IOException {
074                out.write("\t</intersection>\n");
075                this.writeSeparator(out);
076                out.flush();
077        }
078
079        @Override
080        public void writeDirection(Direction direction, BufferedWriter out) throws IOException {
081                this.startDirection(direction, out);
082                
083                this.startTag(ConflictingDirectionsReader.CONFLICTING_DIRECTIONS, out);
084                this.writeListOfDirections(direction.getConflictingDirections(), out);
085                this.endTag(ConflictingDirectionsReader.CONFLICTING_DIRECTIONS, out);
086                
087                this.startTag(ConflictingDirectionsReader.DIRECTIONS_WITH_RIGHT_OF_WAY, out);
088                this.writeListOfDirections(direction.getDirectionsWithRightOfWay(), out);
089                this.endTag(ConflictingDirectionsReader.DIRECTIONS_WITH_RIGHT_OF_WAY, out);
090                
091                this.startTag(ConflictingDirectionsReader.DIRECTIONS_WHICH_MUST_YIELD, out);
092                this.writeListOfDirections(direction.getDirectionsWhichMustYield(), out);
093                this.endTag(ConflictingDirectionsReader.DIRECTIONS_WHICH_MUST_YIELD, out);
094                
095                this.startTag(ConflictingDirectionsReader.NON_CONFLICTING_DIRECTIONS, out);
096                this.writeListOfDirections(direction.getNonConflictingDirections(), out);
097                this.endTag(ConflictingDirectionsReader.NON_CONFLICTING_DIRECTIONS, out);
098                
099                this.endDirection(out);
100        }
101
102        private void startDirection(Direction direction, BufferedWriter out) throws IOException {
103                out.write("\t\t<direction");
104                out.write(" id=\"" + direction.getId() + "\"");
105                out.write(" fromLinkId=\"" + direction.getFromLink() + "\"");
106                out.write(" toLinkId=\"" + direction.getToLink() + "\"");
107                out.write(" >\n");
108        }
109
110        private void endDirection(BufferedWriter out) throws IOException {
111                out.write("\t\t</direction>\n\n");
112        }
113        
114        private void startTag(String tagName, BufferedWriter out) throws IOException {
115                out.write("\t\t\t<"+tagName+">");
116        }
117        
118        private void endTag(String tagName, BufferedWriter out) throws IOException {
119                out.write(" </"+tagName+">\n");
120        }
121        
122        private void writeListOfDirections(List<Id<Direction>> directionList, BufferedWriter out) throws IOException {
123                for (Id<Direction> directionId : directionList) {
124                        out.write(" " + directionId);
125                }
126        }
127
128        @Override
129        public void writeSeparator(BufferedWriter out) throws IOException {
130                out.write("<!-- ====================================================================== -->\n\n");
131        }
132
133}
