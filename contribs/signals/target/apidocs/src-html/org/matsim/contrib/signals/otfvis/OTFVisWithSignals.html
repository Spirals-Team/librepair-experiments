057public class OTFVisWithSignals {
058
059        public static void playScenario(final Scenario scenario, boolean startOtfvis){
060                com.google.inject.Injector injector = Injector.createInjector(scenario.getConfig(), AbstractModule.override(Collections.singleton(new AbstractModule() {
061                        @Override
062                        public void install() {
063                                // defaults
064                                install(new NewControlerModule());
065                                install(new ControlerDefaultCoreListenersModule());
066                                install(new ControlerDefaultsModule());
067                                install(new ScenarioByInstanceModule(scenario));
068                        }
069                }), new SignalsModule()));
070        
071                EventsManager events = injector.getInstance(EventsManager.class);
072                
073                PrepareForSimUtils.createDefaultPrepareForSim(scenario).run();
074                
075                QSim qSim = (QSim) injector.getInstance(Mobsim.class);
076
077                if (startOtfvis) {
078                        OnTheFlyServer server = startServerAndRegisterWithQSim(scenario.getConfig(), scenario, events, qSim);
079                        OTFClientLiveWithSignals.run(scenario.getConfig(), server);
080                }
081
082                qSim.run();
083        }
084
085        public static OnTheFlyServer startServerAndRegisterWithQSim(Config config, Scenario scenario, EventsManager events, QSim qSim) {
086                OnTheFlyServer server = OnTheFlyServer.createInstance(scenario, events, qSim);
087                Network network = scenario.getNetwork();
088                TransitSchedule transitSchedule = scenario.getTransitSchedule();
089//                      TransitQSimEngine transitEngine = qSim.getTransitEngine();
090//                      TransitStopAgentTracker agentTracker = transitEngine.getAgentTracker();
091
092//                      AgentSnapshotInfoFactory snapshotInfoFactory = qSim.getVisNetwork().getAgentSnapshotInfoFactory();
093                SnapshotLinkWidthCalculator linkWidthCalculator = new SnapshotLinkWidthCalculator();
094                linkWidthCalculator.setLinkWidthForVis( config.qsim().getLinkWidthForVis() );
095                if (! Double.isNaN(network.getEffectiveLaneWidth())){
096                        linkWidthCalculator.setLaneWidth( network.getEffectiveLaneWidth() );
097                }
098                AgentSnapshotInfoFactory snapshotInfoFactory = new AgentSnapshotInfoFactory(linkWidthCalculator);
099
100                for( AgentTracker agentTracker : qSim.getAgentTrackers() ) {
101                        FacilityDrawer.Writer facilityWriter = new FacilityDrawer.Writer(network, transitSchedule, agentTracker, snapshotInfoFactory);
102                        server.addAdditionalElement(facilityWriter);
103                }
104
105                if ( (config.qsim().isUseLanes() || config.network().getLaneDefinitionsFile()!=null ) 
106                                && (!(boolean) ConfigUtils.addOrGetModule(config, SignalSystemsConfigGroup.GROUPNAME, SignalSystemsConfigGroup.class).isUseSignalSystems())) {
107                        ConfigUtils.addOrGetModule(config, OTFVisConfigGroup.GROUP_NAME, OTFVisConfigGroup.class).setScaleQuadTreeRect(true);
108                        OTFLaneWriter otfLaneWriter = new OTFLaneWriter(qSim.getVisNetwork(), (Lanes) scenario.getScenarioElement(Lanes.ELEMENT_NAME), scenario.getConfig());
109                        server.addAdditionalElement(otfLaneWriter);
110                } else if ((boolean) ConfigUtils.addOrGetModule(config, SignalSystemsConfigGroup.GROUPNAME, SignalSystemsConfigGroup.class).isUseSignalSystems()) {
111                        ConfigUtils.addOrGetModule(config, OTFVisConfigGroup.GROUP_NAME, OTFVisConfigGroup.class).setScaleQuadTreeRect(true);
112                        SignalGroupStateChangeTracker signalTracker = new SignalGroupStateChangeTracker();
113                        events.addHandler(signalTracker);
114                        SignalsData signalsData = (SignalsData) scenario.getScenarioElement(SignalsData.ELEMENT_NAME);
115                        Lanes laneDefs = scenario.getLanes();
116                        SignalSystemsData systemsData = signalsData.getSignalSystemsData();
117                        SignalGroupsData groupsData = signalsData.getSignalGroupsData();
118                        OTFSignalWriter otfSignalWriter = new OTFSignalWriter(qSim.getVisNetwork(), laneDefs, scenario.getConfig(), systemsData, groupsData , signalTracker);
119                        server.addAdditionalElement(otfSignalWriter);
120                }
121                server.pause();
122                return server;
123        }
