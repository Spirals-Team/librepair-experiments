020package org.matsim.contrib.signals.data;
021
022import java.io.IOException;
023import java.net.URL;
024
025import org.apache.log4j.Logger;
026import org.matsim.contrib.signals.SignalSystemsConfigGroup;
027import org.matsim.contrib.signals.data.ambertimes.v10.AmberTimesReader10;
028import org.matsim.contrib.signals.data.conflicts.io.ConflictingDirectionsReader;
029import org.matsim.contrib.signals.data.intergreens.v10.IntergreenTimesReader10;
030import org.matsim.contrib.signals.data.signalcontrol.v20.SignalControlReader20;
031import org.matsim.contrib.signals.data.signalgroups.v20.SignalGroupsReader20;
032import org.matsim.contrib.signals.data.signalsystems.v20.SignalSystemsReader20;
033import org.matsim.core.config.Config;
034import org.matsim.core.config.ConfigGroup;
035import org.matsim.core.config.ConfigUtils;
036import org.matsim.core.utils.io.UncheckedIOException;
037
038
039/**
040 * Loads all data files related to the traffic signal systems model.
041 * 
042 * @author dgrether
043 *
044 */
045public class SignalsDataLoader {
046
047        private static final Logger log = Logger.getLogger(SignalsDataLoader.class);
048
049        private Config config;
050        private SignalSystemsConfigGroup signalConfig;
051
052        public SignalsDataLoader(Config config){
053                this.config = config;
054                this.signalConfig = ConfigUtils.addOrGetModule(config,
055                                SignalSystemsConfigGroup.GROUPNAME, SignalSystemsConfigGroup.class);
056        }
057
058        public SignalsData loadSignalsData() {
059                SignalsData data = new SignalsDataImpl(this.signalConfig);
060                this.loadSystems(data);
061                this.loadGroups(data);
062                this.loadControl(data);
063                if (this.signalConfig.isUseAmbertimes()){
064                        this.loadAmberTimes(data);
065                }
066                if (this.signalConfig.isUseIntergreenTimes()){
067                        this.loadIntergreenTimes(data);
068                }
069                if (this.signalConfig.isUseConflictingDirections()) {
070                        this.loadConflicts(data);
071                }
072                return data;
073        }
074
075
076        private void loadIntergreenTimes(SignalsData data){
077                if (this.signalConfig.getIntergreenTimesFile() != null) {
078                        IntergreenTimesReader10 reader = new IntergreenTimesReader10(data.getIntergreenTimesData());
079                        URL filename = ConfigGroup.getInputFileURL(config.getContext(), this.signalConfig.getIntergreenTimesFile());
080                        try {
081                                reader.readStream(filename.openStream());
082                        } catch (IOException e) {
083                                throw new UncheckedIOException(e);
084                        }
085                }
086        }
087
088
089        private void loadAmberTimes(SignalsData data) {
090                if (this.signalConfig.getAmberTimesFile() != null){
091                        AmberTimesReader10 reader = new AmberTimesReader10(data.getAmberTimesData());
092                        URL filename = ConfigGroup.getInputFileURL(config.getContext(), this.signalConfig.getAmberTimesFile());
093                        reader.readFile(filename.getFile());
094                }
095                else {
096                        log.info("Signals: No amber times file set, can't load amber times!");
097                }
098        }
099
100        private void loadControl(SignalsData data){
101                if (this.signalConfig.getSignalControlFile() != null){
102                        SignalControlReader20 reader = new SignalControlReader20(data.getSignalControlData());
103                        URL filename = ConfigGroup.getInputFileURL(config.getContext(), this.signalConfig.getSignalControlFile());
104                        try {
105                                reader.readStream(filename.openStream());
106                        } catch (IOException e) {
107                                throw new UncheckedIOException(e);
108                        }
109                }
110                else {
111                        log.info("Signals: No signal control file set, can't load signal control data!");
112                }
113        }
114
115        private void loadGroups(SignalsData data) {
116                if (this.signalConfig.getSignalGroupsFile() != null){
117                        SignalGroupsReader20 reader = new SignalGroupsReader20(data.getSignalGroupsData());
118                        URL filename = ConfigGroup.getInputFileURL(config.getContext(), this.signalConfig.getSignalGroupsFile());
119                        try {
120                                reader.readStream(filename.openStream());
121                        } catch (IOException e) {
122                                throw new UncheckedIOException(e);
123                        }
124                }
125                else {
126                        log.info("Signals: No signal groups file set, can't load signal groups!");
127                }
128        }
129
130        private void loadSystems(SignalsData data){
131                if (this.signalConfig.getSignalSystemFile() != null){
132                        SignalSystemsReader20 reader = new SignalSystemsReader20(data.getSignalSystemsData());
133                        URL filename = ConfigGroup.getInputFileURL(config.getContext(), this.signalConfig.getSignalSystemFile());
134                        try {
135                                reader.readStream(filename.openStream());
136                        } catch (IOException e) {
137                                throw new UncheckedIOException(e);
138                        }
139                }
140                else {
141                        log.info("Signals: No signal systems file set, can't load signal systems information!");
142                }
143        }
144        
145        private void loadConflicts(SignalsData data) {
146                if (this.signalConfig.getConflictingDirectionsFile() != null) {
147                        ConflictingDirectionsReader reader = new ConflictingDirectionsReader(data.getConflictingDirectionsData());
148                        URL filename = ConfigGroup.getInputFileURL(config.getContext(), this.signalConfig.getConflictingDirectionsFile());
149                        try {
150                                reader.parse(filename.openStream());
151                        } catch (IOException e) {
152                                throw new UncheckedIOException(e);
153                        }
154                }
155                else {
156                        log.info("Signals: No conflicting directions file set, can't load conflicting directions information!");
157                }
158        }
159
160
161}
