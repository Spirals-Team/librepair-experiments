020package org.matsim.contrib.signals.otfvis;
021
022import java.io.IOException;
023import java.nio.ByteBuffer;
024import java.util.Map;
025
026import org.matsim.api.core.v01.Id;
027import org.matsim.api.core.v01.network.Link;
028import org.matsim.contrib.signals.events.SignalGroupStateChangedEvent;
029import org.matsim.core.config.Config;
030import org.matsim.core.mobsim.qsim.interfaces.SignalGroupState;
031import org.matsim.core.utils.misc.ByteBufferUtils;
032import org.matsim.lanes.data.Lane;
033import org.matsim.lanes.data.Lanes;
034import org.matsim.contrib.signals.data.signalgroups.v20.SignalGroupData;
035import org.matsim.contrib.signals.data.signalgroups.v20.SignalGroupsData;
036import org.matsim.contrib.signals.data.signalgroups.v20.SignalData;
037import org.matsim.contrib.signals.data.signalsystems.v20.SignalSystemsData;
038import org.matsim.contrib.signals.model.Signal;
039import org.matsim.contrib.signals.model.SignalGroup;
040import org.matsim.contrib.signals.model.SignalSystem;
041import org.matsim.vis.snapshotwriters.VisNetwork;
042
043/**
044 * @author dgrether
045 */
046public class OTFSignalWriter extends OTFLaneWriter {
047
048        private transient SignalGroupStateChangeTracker signalTracker;
049        private transient SignalGroupsData signalGroups;
050        private transient SignalSystemsData signalSystems;
051
052        public OTFSignalWriter(VisNetwork visNetwork, Lanes lanes, Config config, SignalSystemsData signalSystemsData, SignalGroupsData signalGroupsData, SignalGroupStateChangeTracker signalTracker) {
053                super(visNetwork, lanes, config);
054                this.signalTracker = signalTracker;
055                this.signalSystems = signalSystemsData;
056                this.signalGroups = signalGroupsData;
057        }
058
059        @Override
060        public void writeConstData(ByteBuffer out) throws IOException {
061                super.writeConstData(out);
062                this.writeSignalSystems(out);
063        }
064        
065        private void writeSignalSystems(ByteBuffer out){
066                out.putInt(this.signalGroups.getSignalGroupDataBySignalSystemId().size());
067                for (Id<SignalSystem> systemId : this.signalGroups.getSignalGroupDataBySignalSystemId().keySet()){
068                        ByteBufferUtils.putString(out, systemId.toString());
069                        Map<Id<SignalGroup>, SignalGroupData> groups = this.signalGroups.getSignalGroupDataBySystemId(systemId);
070                        out.putInt(groups.size());
071                        for (SignalGroupData group : groups.values()){
072                                ByteBufferUtils.putString(out, group.getId().toString());
073                                out.putInt(group.getSignalIds().size());
074                                for (Id<Signal> signalId : group.getSignalIds()){
075                                        ByteBufferUtils.putString(out, signalId.toString());
076                                        SignalData signal = signalSystems.getSignalSystemData().get(systemId).getSignalData().get(signalId);
077                                        ByteBufferUtils.putString(out, signal.getLinkId().toString());
078                                        if (signal.getLaneIds() == null || signal.getLaneIds().isEmpty()){
079                                                out.putInt(0);
080                                        }
081                                        else {
082                                                out.putInt(signal.getLaneIds().size());
083                                                for (Id<Lane> laneId : signal.getLaneIds()){
084                                                        ByteBufferUtils.putString(out, laneId.toString());
085                                                }
086                                        }
087                                        
088                                        if (signal.getTurningMoveRestrictions() == null || signal.getTurningMoveRestrictions().isEmpty()){
089                                                out.putInt(0);
090                                        }
091                                        else {
092                                                out.putInt(signal.getTurningMoveRestrictions().size());
093                                                for (Id<Link> outLinkId : signal.getTurningMoveRestrictions()){
094                                                        ByteBufferUtils.putString(out, outLinkId.toString());
095                                                }
096                                        }
097                                }
098                        }
099                }
100        }
101
102        
103        private void writeSignalGroupStates(ByteBuffer out){
104                out.putInt(this.signalTracker.getSignalGroupEvents().size());
105                for (SignalGroupStateChangedEvent e : this.signalTracker.getSignalGroupEvents()){
106                        ByteBufferUtils.putString(out, e.getSignalSystemId().toString());
107                        ByteBufferUtils.putString(out, e.getSignalGroupId().toString());
108                        SignalGroupState state = e.getNewState();
109                        if (state.equals(SignalGroupState.GREEN)){
110                                out.putInt(1);
111                        }
112                        else if (state.equals(SignalGroupState.RED)){
113                                out.putInt(0);
114                        }
115                        else if (state.equals(SignalGroupState.REDYELLOW)){
116                                out.putInt(2);
117                        }
118                        else if (state.equals(SignalGroupState.YELLOW)){
119                                out.putInt(3);
120                        }
121                        else if (state.equals(SignalGroupState.OFF)){
122                                out.putInt(4);
123                        }
124                        
125                }
126                this.signalTracker.getSignalGroupEvents().clear();
127        }
128        
129        @Override
130        public void writeDynData(ByteBuffer out) throws IOException {
131                this.writeSignalGroupStates(out);
132        }
133
134}
