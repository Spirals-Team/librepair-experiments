020package org.matsim.contrib.signals.data.signalsystems.v20;
021
022import java.io.BufferedWriter;
023import java.io.FileNotFoundException;
024import java.io.IOException;
025
026import javax.xml.bind.JAXBContext;
027import javax.xml.bind.JAXBException;
028import javax.xml.bind.Marshaller;
029
030import org.apache.log4j.Logger;
031import org.matsim.api.core.v01.Id;
032import org.matsim.api.core.v01.network.Link;
033import org.matsim.core.utils.io.IOUtils;
034import org.matsim.core.utils.io.MatsimJaxbXmlWriter;
035import org.matsim.jaxb.signalsystems20.ObjectFactory;
036import org.matsim.jaxb.signalsystems20.XMLIdRefType;
037import org.matsim.jaxb.signalsystems20.XMLSignalSystemType;
038import org.matsim.jaxb.signalsystems20.XMLSignalSystems;
039import org.matsim.jaxb.signalsystems20.XMLSignalType;
040import org.matsim.jaxb.signalsystems20.XMLSignalType.XMLLane;
041import org.matsim.lanes.data.Lane;
042import org.matsim.contrib.signals.data.signalgroups.v20.SignalData;
043import org.matsim.contrib.signals.MatsimSignalSystemsReader;
044
045
046/**
047 * @author dgrether
048 *
049 */
050public class SignalSystemsWriter20 extends MatsimJaxbXmlWriter {
051        
052        private static final Logger log = Logger.getLogger(SignalSystemsWriter20.class);
053        
054        private SignalSystemsData signalData;
055        
056        
057        public SignalSystemsWriter20(SignalSystemsData signalSystemsData){
058                this.signalData = signalSystemsData;
059        }
060        
061        private XMLSignalSystems convertDataToXml() {
062                ObjectFactory fac = new ObjectFactory();
063                XMLSignalSystems xmlContainer = fac.createXMLSignalSystems();
064                
065                for (SignalSystemData ssd : this.signalData.getSignalSystemData().values()){
066                        XMLSignalSystemType xmlsstype = fac.createXMLSignalSystemType();
067                        xmlContainer.getSignalSystem().add(xmlsstype);
068                        xmlsstype.setId(ssd.getId().toString());
069                        
070                        if (ssd.getSignalData() != null){
071                                xmlsstype.setSignals(fac.createXMLSignalSystemTypeXMLSignals());
072                                
073                                for (SignalData sd : ssd.getSignalData().values()){
074                                        XMLSignalType xmlssd = fac.createXMLSignalType();
075                                        xmlsstype.getSignals().getSignal().add(xmlssd);
076                                        xmlssd.setId(sd.getId().toString());
077                                        xmlssd.setLinkIdRef(sd.getLinkId().toString());
078                                        if (sd.getLaneIds() != null){
079                                                for (Id<Lane> id : sd.getLaneIds()){
080                                                        XMLLane xmllane = fac.createXMLSignalTypeXMLLane();
081                                                        xmlssd.getLane().add(xmllane);
082                                                        xmllane.setRefId(id.toString());
083                                                }
084                                        }
085                                        if (sd.getTurningMoveRestrictions() != null){
086                                                xmlssd.setTurningMoveRestrictions(fac.createXMLSignalTypeXMLTurningMoveRestrictions());
087                                                for (Id<Link> id : sd.getTurningMoveRestrictions()){
088                                                        XMLIdRefType xmlid = fac.createXMLIdRefType();
089                                                        xmlssd.getTurningMoveRestrictions().getToLink().add(xmlid);
090                                                        xmlid.setRefId(id.toString());
091                                                }
092                                        }
093                                }
094                        }
095                }
096                return xmlContainer;
097        }
098
099        public void write(final String filename, XMLSignalSystems xmlSignals){
100                log.info("writing file: " + filename);
101                JAXBContext jc;
102                try {
103                        jc = JAXBContext.newInstance(org.matsim.jaxb.signalsystems20.ObjectFactory.class);
104                        Marshaller m = jc.createMarshaller();
105                        super.setMarshallerProperties(MatsimSignalSystemsReader.SIGNALSYSTEMS20, m);
106                        BufferedWriter bufout = IOUtils.getBufferedWriter(filename);
107                        m.marshal(xmlSignals, bufout);
108                        bufout.close();
109                        log.info(filename + " written successfully.");
110                } catch (JAXBException e) {
111                        e.printStackTrace();
112                } catch (FileNotFoundException e) {
113                        e.printStackTrace();
114                } catch (IOException e) {
115                        e.printStackTrace();
116                }
117                
118        }
119        
120        @Override
121        public void write(final String filename) {
122                XMLSignalSystems xmlSignals = convertDataToXml();
123                this.write(filename, xmlSignals);
124        }
125        
126}
