048/**
049 * @author tthunig
050 *
051 */
052public class ConflictingDirectionsLogicImpl implements ConflictingDirectionsLogic {
053
054private static final Logger log = Logger.getLogger(ConflictingDirectionsLogicImpl.class);
055
056        private ActionOnSignalSpecsViolation actionOnConflictingDirectionsViolation;
057        private Map<Id<Signal>, Set<Tuple<Id<Link>, Id<Link>>>> setOfLinkTuplesPerSignal = new HashMap<>();
058        private Map<Id<SignalGroup>, Set<Direction>> setOfDirectionsPerGroup = new HashMap<>();
059        
060        private Map<Id<SignalSystem>, List<Id<SignalGroup>>> greenSignalsPerSystem = new HashMap<>();
061
062        public ConflictingDirectionsLogicImpl(Network network, Lanes lanes, SignalsData signalsData, ActionOnSignalSpecsViolation actionOnConflictingDirectionsViolation) {
063                this.actionOnConflictingDirectionsViolation = actionOnConflictingDirectionsViolation;
064                
065                for (SignalSystemData signalSystem : signalsData.getSignalSystemsData().getSignalSystemData().values()) {
066                        // remember relation of signals to directions
067                        for (SignalData signal : signalSystem.getSignalData().values()) {
068                                setOfLinkTuplesPerSignal.put(signal.getId(), new HashSet<>());
069                                if (signal.getTurningMoveRestrictions() != null && !signal.getTurningMoveRestrictions().isEmpty()) {
070                                        // use turning move restrictions if possible
071                                        for (Id<Link> nextLinkId : signal.getTurningMoveRestrictions()) {
072                                                setOfLinkTuplesPerSignal.get(signal.getId()).add(new Tuple<Id<Link>, Id<Link>>(signal.getLinkId(), nextLinkId));
073                                        }
074                                } else if (signal.getLaneIds() != null && !signal.getLaneIds().isEmpty()) {
075                                        // if no turning move restrictions exist, use lane information
076                                        for (Id<Lane> signalizedLaneId : signal.getLaneIds()) {
077                                                Lane signalizedLane = lanes.getLanesToLinkAssignments().get(signal.getLinkId()).getLanes().get(signalizedLaneId);
078                                                for (Id<Link> toLinkId : signalizedLane.getToLinkIds()) {
079                                                        setOfLinkTuplesPerSignal.get(signal.getId()).add(new Tuple<Id<Link>, Id<Link>>(signal.getLinkId(), toLinkId));
080                                                }
081                                        }
082                                } else {
083                                        // no turning move restrictions and no lanes exist. all outgoing links of the node are possible next links
084                                        for (Id<Link> outgoingLinkId : network.getLinks().get(signal.getLinkId()).getToNode().getOutLinks().keySet()) {
085                                                setOfLinkTuplesPerSignal.get(signal.getId()).add(new Tuple<Id<Link>, Id<Link>>(signal.getLinkId(), outgoingLinkId));
086                                        }
087                                }
088                        }
089                        
090                        // group directions to signal groups
091                        IntersectionDirections conflictsOfThisSystem = signalsData.getConflictingDirectionsData().getConflictsPerSignalSystem().get(signalSystem.getId());
092                        for (SignalGroupData group : signalsData.getSignalGroupsData().getSignalGroupDataBySystemId(signalSystem.getId()).values()) {
093                                setOfDirectionsPerGroup.put(group.getId(), new HashSet<>());
094                                for (Id<Signal> signalIdOfThisGroup : signalsData.getSignalGroupsData().getSignalGroupDataBySystemId(signalSystem.getId()).get(group.getId()).getSignalIds()) {
095                                        for (Tuple<Id<Link>, Id<Link>> from2toLink : setOfLinkTuplesPerSignal.get(signalIdOfThisGroup)) {
096                                                setOfDirectionsPerGroup.get(group.getId()).add(conflictsOfThisSystem.getDirection(from2toLink.getFirst(), from2toLink.getSecond()));
097                                        }
098                                }
099                                log.info("Group " + group.getId() + " corresponds to " + setOfDirectionsPerGroup.get(group.getId()).size() + " directions.");
100                        }
101                }
102        }
103
104        @Override
105        public void reset(int iteration) {
106                this.greenSignalsPerSystem.clear();
107        }
108
109        @Override
110        public void handleEvent(SignalGroupStateChangedEvent event) {
111                if (SignalGroupState.GREEN.equals(event.getNewState())){
112                        // register system if necessary
113                        if (!greenSignalsPerSystem.containsKey(event.getSignalSystemId())) {
114                                greenSignalsPerSystem.put(event.getSignalSystemId(), new LinkedList<>());
115                        }
116                        
117                        // check whether a conflicting direction already shows green
118                        this.checkAndHandleGreenAllowed(event);
119                        
120                        // add signal as green
121                        greenSignalsPerSystem.get(event.getSignalSystemId()).add(event.getSignalGroupId());
122                } else if (SignalGroupState.RED.equals(event.getNewState())) {
123                        // remove signal from the set of green signal groups
124                        if (greenSignalsPerSystem.containsKey(event.getSignalSystemId())) {
125                                greenSignalsPerSystem.get(event.getSignalSystemId()).remove(event.getSignalGroupId());
126                        }
127                }
128        }
129        
130        private void checkAndHandleGreenAllowed(SignalGroupStateChangedEvent event) {
131                // check whether one direction that has green at the moment is conflicting with at least one direction of the group that is switched to green
132                for (Id<SignalGroup> greenGroupId : greenSignalsPerSystem.get(event.getSignalSystemId())) {
133                        for (Direction greenDirection : setOfDirectionsPerGroup.get(greenGroupId)) {
134                                for (Direction directionToSwitchGreen : setOfDirectionsPerGroup.get(event.getSignalGroupId())) {
135                                        if (greenDirection.getConflictingDirections().contains(directionToSwitchGreen.getId()) || directionToSwitchGreen.getConflictingDirections().contains(greenDirection.getId())) {
136                                                String conflictingDirectionViolation = "Signal Group " + event.getSignalGroupId() + " is switched to green although " + greenGroupId 
137                                                                + " already shows green. This is not allowed due to the conflicting directions data defined in the scenario because direction " 
138                                                                + greenDirection.getId() + " from link " + greenDirection.getFromLink() + " to link " + greenDirection.getToLink() 
139                                                                + " is conflicting to direction " + directionToSwitchGreen.getId() + " from link " + directionToSwitchGreen.getFromLink()
140                                                                + " to link " + directionToSwitchGreen.getToLink() + ".";
141                                                if (this.actionOnConflictingDirectionsViolation.equals(SignalSystemsConfigGroup.ActionOnSignalSpecsViolation.WARN)) {
142                                                        log.warn(conflictingDirectionViolation);
143                                                } else if (this.actionOnConflictingDirectionsViolation.equals(SignalSystemsConfigGroup.ActionOnSignalSpecsViolation.EXCEPTION)) {
144                                                        log.warn(conflictingDirectionViolation);
145                                                        throw new RuntimeException(conflictingDirectionViolation);
146                                                }
147                                        }
148                                }
149                        }
150                }       
151        }
