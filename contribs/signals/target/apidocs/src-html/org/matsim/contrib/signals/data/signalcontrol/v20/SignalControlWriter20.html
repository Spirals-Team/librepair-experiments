056 * @author dgrether
057 *
058 */
059public class SignalControlWriter20 extends MatsimJaxbXmlWriter {
060
061        private static final Logger log = Logger.getLogger(SignalControlWriter20.class);
062        
063        private SignalControlData data;
064
065        public SignalControlWriter20(SignalControlData controlData) {
066                this.data = controlData;
067        }
068
069        private XMLSignalControl convertDataToXml() throws DatatypeConfigurationException{
070                ObjectFactory fac = new ObjectFactory();
071                XMLSignalControl xmlSignalControl = fac.createXMLSignalControl();
072                for (SignalSystemControllerData cd : this.data.getSignalSystemControllerDataBySystemId().values()){
073                        XMLSignalSystemType xmlSS = fac.createXMLSignalSystemType();
074                        xmlSignalControl.getSignalSystem().add(xmlSS);
075                        xmlSS.setRefId(cd.getSignalSystemId().toString());
076                        
077                        XMLSignalSystemControllerType xmlCd = fac.createXMLSignalSystemControllerType();
078                        xmlSS.setSignalSystemController(xmlCd);
079                        xmlCd.setControllerIdentifier(cd.getControllerIdentifier());
080                        
081                        //process plans if there are any
082                        if (cd.getSignalPlanData() != null){
083                                for (SignalPlanData planData : cd.getSignalPlanData().values()){
084                                        XMLSignalPlanType xmlPd = fac.createXMLSignalPlanType();
085                                        xmlCd.getSignalPlan().add(xmlPd);
086                                        xmlPd.setId(planData.getId().toString());
087                                        XMLStart xmlStart = fac.createXMLSignalPlanTypeXMLStart();
088                                        xmlPd.setStart(xmlStart);
089                                        xmlStart.setDaytime(this.getXmlGregorianCalendar(planData.getStartTime()));
090                                        XMLStop xmlStop = fac.createXMLSignalPlanTypeXMLStop();
091                                        xmlPd.setStop(xmlStop);
092                                        xmlStop.setDaytime(this.getXmlGregorianCalendar(planData.getEndTime()));
093                                        if (planData.getCycleTime() != null){
094                                                XMLCycleTime xmlCycleTime = fac.createXMLSignalPlanTypeXMLCycleTime();
095                                                xmlPd.setCycleTime(xmlCycleTime);
096                                                xmlCycleTime.setSec(planData.getCycleTime());
097                                        }
098                                        if (planData.getOffset() != null){
099                                                XMLOffset xmlOffset = fac.createXMLSignalPlanTypeXMLOffset();
100                                                xmlPd.setOffset(xmlOffset);
101                                                xmlOffset.setSec(planData.getOffset());
102                                        }
103                                        
104                                        //process signalGroupSettings
105                                        for (SignalGroupSettingsData sgSettings : planData.getSignalGroupSettingsDataByGroupId().values()){
106                                                XMLSignalGroupSettingsType xmlSgSettings = fac.createXMLSignalGroupSettingsType();
107                                                xmlPd.getSignalGroupSettings().add(xmlSgSettings);
108                                                xmlSgSettings.setRefId(sgSettings.getSignalGroupId().toString());
109                                                
110                                                XMLOnset xmlOnset = fac.createXMLSignalGroupSettingsTypeXMLOnset();
111                                                xmlSgSettings.setOnset(xmlOnset);
112                                                xmlOnset.setSec(Integer.valueOf(sgSettings.getOnset()));
113                                                
114                                                XMLDropping xmlDropping = fac.createXMLSignalGroupSettingsTypeXMLDropping();
115                                                xmlSgSettings.setDropping(xmlDropping);
116                                                xmlDropping.setSec(Integer.valueOf(sgSettings.getDropping()));
117                                        }
118                                }
119                        } //end process plans
120                }
121                return xmlSignalControl;
122        }
123        
124        public void write(String filename, XMLSignalControl xmlSignalControl) {
125                log.info("writing file: " + filename);
126                JAXBContext jc;
127                try {
128                        jc = JAXBContext.newInstance(org.matsim.jaxb.signalcontrol20.ObjectFactory.class);
129                        Marshaller m = jc.createMarshaller();
130                        super.setMarshallerProperties(MatsimSignalSystemsReader.SIGNALCONTROL20, m);
131                        BufferedWriter bufout = IOUtils.getBufferedWriter(filename);
132                        m.marshal(xmlSignalControl, bufout);
133                        bufout.close();
134                        log.info(filename + " written successfully.");
135                } catch (JAXBException e) {
136                        e.printStackTrace();
137                } catch (FileNotFoundException e) {
138                        e.printStackTrace();
139                } catch (IOException e) {
140                        e.printStackTrace();
141                }
142        }
143
144        @Override
145        public void write(final String filename) {
146                XMLSignalControl xmlSignalControl;
147                try {
148                        xmlSignalControl = convertDataToXml();
149                        this.write(filename, xmlSignalControl);
150                } catch (DatatypeConfigurationException e) {
151                        log.error("Could not write file " + filename + " due to ");
152                        e.printStackTrace();
153                }
154        }
155        
156        private XMLGregorianCalendar getXmlGregorianCalendar(double seconds) throws DatatypeConfigurationException {
157                XMLGregorianCalendar time = DatatypeFactory.newInstance().newXMLGregorianCalendar();
158                int s = (int) seconds;
159                int h = (s / 3600);
160                s = s % 3600;
161                int m = (s / 60);
162                s = s % 60;
163                time.setSecond(s);
164                time.setMinute(m);
165                time.setHour(h);
166                return time;
167        }
