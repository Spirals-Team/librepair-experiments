052public class SignalUtils {
053
054        /**
055         * Create an empty SignalsData object
056         * 
057         * @param signalSystemsConfigGroup
058         */
059        public static SignalsData createSignalsData(SignalSystemsConfigGroup signalSystemsConfigGroup) {
060                return new SignalsDataImpl(signalSystemsConfigGroup);
061        }
062
063        /**
064         * Creates a signal group for each single signal of the
065         * system and adds it to the given SignalGroupsData container. 
066         * Each created signal group will get the same ID as the signal itself.
067         * 
068         * @param groups
069         *            container to that the signal groups are added
070         * @param system
071         *            contains the signals for which signal groups are created
072         */
073        public static void createAndAddSignalGroups4Signals(SignalGroupsData groups, SignalSystemData system) {
074                for (SignalData signal : system.getSignalData().values()) {
075                        SignalGroupData group4signal = groups.getFactory().createSignalGroupData(system.getId(), 
076                                        Id.create(signal.getId(), SignalGroup.class));
077                        group4signal.addSignalId(signal.getId());
078                        groups.addSignalGroupData(group4signal);
079                }
080        }
081
082        /**
083         * Convenience method to create a signal with the given Id on the link with
084         * the given Id on the lanes with the given Ids. The signal is added to the
085         * SignalSystemData.
086         */
087        public static void createAndAddSignal(SignalSystemData sys,
088                        SignalSystemsDataFactory factory, Id<Signal> signalId,
089                        Id<Link> linkId, List<Id<Lane>> laneIds) {
090                
091                SignalData signal = factory.createSignalData(signalId);
092                sys.addSignalData(signal);
093                signal.setLinkId(linkId);
094                if (laneIds != null) {
095                        for (Id<Lane> laneId : laneIds) {
096                                signal.addLaneId(laneId);
097                        }
098                }
099        }
100        
101        /**
102         * Creates a signal plan with the given cycle time and offset.
103         * The plan gets the default signal plan id 1.
104         * 
105         * @param fac
106         * @param cycleTime
107         * @param offset
108         * @return the signal plan
109         */
110        public static SignalPlanData createSignalPlan(SignalControlDataFactory fac, int cycleTime, int offset) {
111                return createSignalPlan(fac, cycleTime, offset, Id.create(1,SignalPlan.class));
112        }
113        
114        /**
115         * Creates a signal plan with the given cycle time, offset and id.
116         * 
117         * @param fac
118         * @param cycleTime
119         * @param offset
120         * @param signalPlanId
121         * @return the signal plan
122         */
123        public static SignalPlanData createSignalPlan(SignalControlDataFactory fac, int cycleTime, int offset, Id<SignalPlan> signalPlanId) {
124                
125                SignalPlanData signalPlan = fac.createSignalPlanData(signalPlanId);
126                signalPlan.setCycleTime(cycleTime);
127                signalPlan.setOffset(offset);
128                return signalPlan;
129        }
130
131        /**
132         * Creates and returns a signal group setting for the given signal group id with the
133         * given onset and dropping time.
134         * 
135         * @param fac
136         * @param signalGroupId
137         * @param onset
138         * @param dropping
139         * @return the signal group setting
140         */
141        public static SignalGroupSettingsData createSetting4SignalGroup(
142                        SignalControlDataFactory fac, Id<SignalGroup> signalGroupId,
143                        int onset, int dropping) {
144
145                SignalGroupSettingsData signalGroupSettings = fac
146                                .createSignalGroupSettingsData(signalGroupId);
147                signalGroupSettings.setOnset(onset);
148                signalGroupSettings.setDropping(dropping);
149                return signalGroupSettings;
150        }
151        
152        /**
153         * Clones the SignalControlData given as argument and returns a new instance of SignalControlData with the same content.
154         */
155        public static SignalControlData copySignalControlData (SignalControlData oldSignalControlData) {
156                SignalControlData newSignalControlData = new SignalControlDataImpl();
157                for (SignalSystemControllerData oldSystemControl : oldSignalControlData.getSignalSystemControllerDataBySystemId().values()) {
158                        SignalSystemControllerData newSystemControl = new SignalSystemControllerDataImpl(oldSystemControl.getSignalSystemId());
159                        newSystemControl.setControllerIdentifier(oldSystemControl.getControllerIdentifier());
160                        for (SignalPlanData oldPlan : oldSystemControl.getSignalPlanData().values()) {
161                                SignalPlanData newPlan = new SignalPlanDataImpl(oldPlan.getId());
162                                newPlan.setStartTime(oldPlan.getStartTime());
163                                newPlan.setEndTime(oldPlan.getEndTime());
164                                newPlan.setCycleTime(oldPlan.getCycleTime());
165                                newPlan.setOffset(oldPlan.getOffset());
166                                for (SignalGroupSettingsData oldGroupSetting : oldPlan.getSignalGroupSettingsDataByGroupId().values()) {
167                                        SignalGroupSettingsData newGroupSettings = new SignalGroupSettingsDataImpl(oldGroupSetting.getSignalGroupId());
168                                        newGroupSettings.setOnset(oldGroupSetting.getOnset());
169                                        newGroupSettings.setDropping(oldGroupSetting.getDropping());
170                                        newPlan.addSignalGroupSettings(newGroupSettings);
171                                }
172                                newSystemControl.addSignalPlanData(newPlan);
173                        }
174                        newSignalControlData.addSignalSystemControllerData(newSystemControl);
175                }
176                return newSignalControlData;
177        }
