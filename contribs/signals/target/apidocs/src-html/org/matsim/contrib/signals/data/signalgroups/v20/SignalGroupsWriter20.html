020package org.matsim.contrib.signals.data.signalgroups.v20;
021
022import java.io.BufferedWriter;
023import java.io.FileNotFoundException;
024import java.io.IOException;
025import java.util.Map;
026import java.util.Map.Entry;
027
028import javax.xml.bind.JAXBContext;
029import javax.xml.bind.JAXBException;
030import javax.xml.bind.Marshaller;
031
032import org.apache.log4j.Logger;
033import org.matsim.api.core.v01.Id;
034import org.matsim.core.utils.io.IOUtils;
035import org.matsim.core.utils.io.MatsimJaxbXmlWriter;
036import org.matsim.jaxb.signalgroups20.ObjectFactory;
037import org.matsim.jaxb.signalgroups20.XMLIdRefType;
038import org.matsim.jaxb.signalgroups20.XMLSignalGroupType;
039import org.matsim.jaxb.signalgroups20.XMLSignalGroups;
040import org.matsim.jaxb.signalgroups20.XMLSignalSystemSignalGroupType;
041import org.matsim.contrib.signals.model.Signal;
042import org.matsim.contrib.signals.model.SignalGroup;
043import org.matsim.contrib.signals.model.SignalSystem;
044import org.matsim.contrib.signals.MatsimSignalSystemsReader;
045
046/**
047 * @author dgrether
048 * @author jbischoff
049 * 
050 */
051public class SignalGroupsWriter20 extends MatsimJaxbXmlWriter {
052
053        private static final Logger log = Logger.getLogger(SignalGroupsWriter20.class);
054
055        private SignalGroupsData signalGroupsData;
056
057        public SignalGroupsWriter20(SignalGroupsData signalGroupsData) {
058                this.signalGroupsData = signalGroupsData;
059        }
060
061        public void write(final String filename, XMLSignalGroups xmlSignalGroups) {
062                log.info("writing file: " + filename);
063                JAXBContext jc;
064                try {
065                        jc = JAXBContext.newInstance(org.matsim.jaxb.signalgroups20.ObjectFactory.class);
066                        Marshaller m = jc.createMarshaller();
067                        super.setMarshallerProperties(MatsimSignalSystemsReader.SIGNALGROUPS20, m);
068                        BufferedWriter bufout = IOUtils.getBufferedWriter(filename);
069                        m.marshal(xmlSignalGroups, bufout);
070                        bufout.close();
071                        log.info(filename + " written successfully.");
072                } catch (JAXBException e) {
073                        e.printStackTrace();
074                } catch (FileNotFoundException e) {
075                        e.printStackTrace();
076                } catch (IOException e) {
077                        e.printStackTrace();
078                }
079
080        }
081
082        @Override
083        public void write(String filename) {
084                XMLSignalGroups xmlSignalGroups = this.convertDataToXml();
085                this.write(filename, xmlSignalGroups);
086        }
087
088        private XMLSignalGroups convertDataToXml() {
089                ObjectFactory fac = new ObjectFactory();
090                XMLSignalGroups xmlContainer = fac.createXMLSignalGroups();
091
092                for (Entry<Id<SignalSystem>, Map<Id<SignalGroup>, SignalGroupData>> ssid : this.signalGroupsData
093                                .getSignalGroupDataBySignalSystemId().entrySet()) {
094                        // signal System
095                        XMLSignalSystemSignalGroupType xsssgt = fac.createXMLSignalSystemSignalGroupType();
096                        xsssgt.setRefId(ssid.getKey().toString());
097
098                        for (Entry<Id<SignalGroup>, SignalGroupData> signal : ssid.getValue().entrySet()) {
099                                XMLSignalGroupType xsgt = fac.createXMLSignalGroupType();
100                                xsgt.setId(signal.getValue().getId().toString());
101                                for (Id<Signal> ids : signal.getValue().getSignalIds()) {
102                                        XMLIdRefType xmlid = fac.createXMLIdRefType();
103                                        xmlid.setRefId(ids.toString());
104                                        xsgt.getSignal().add(xmlid);
105                                }
106                                xsssgt.getSignalGroup().add(xsgt);
107
108                        }
109                        xmlContainer.getSignalSystem().add(xsssgt);
110                }
111                return xmlContainer;
112        }
113}
