047public class SignalSystemsDataConsistencyChecker {
048
049        
050        private static final Logger log = Logger.getLogger(SignalSystemsDataConsistencyChecker.class);
051        
052        private SignalsData signalsData;
053
054        private Lanes lanes;
055
056        private Network network;
057
058        private List<Tuple<Id<Signal>, Id<SignalSystem>>> malformedSignals = new LinkedList<>();
059
060        private List<Tuple<Id<SignalGroup>, Id<SignalSystem>>> emptyGroups = new LinkedList<>();
061
062        private List<Id<SignalSystem>> emptySystems = new LinkedList<>();
063
064        public SignalSystemsDataConsistencyChecker(Network network, Lanes lanes, SignalsData signalsData) {
065                this.network = network;
066                this.signalsData = signalsData;
067                this.lanes = lanes;
068        }
069        
070        public void checkConsistency() {
071                if (this.signalsData == null) {
072                        log.error("No SignalsData instance found as ScenarioElement of Scenario instance!");
073                        log.error("Nothing to check, aborting!");
074                        return;
075                }
076                this.checkSignalToLinkMatching();
077                this.checkSignalToLaneMatching();
078                this.removeMalformedSignalSystems();
079        }
080
081        private void removeMalformedSignalSystems() {
082
083                for (Tuple<Id<Signal>, Id<SignalSystem>> tuple : malformedSignals){
084                        signalsData.getSignalSystemsData().getSignalSystemData().get(tuple.getSecond()).getSignalData().remove(tuple.getFirst());
085                        for(SignalGroupData sigGroup : this.signalsData.getSignalGroupsData().getSignalGroupDataBySignalSystemId().get(tuple.getSecond()).values()){
086                                if(sigGroup.getSignalIds().contains(tuple.getFirst())){
087                                        sigGroup.getSignalIds().remove(tuple.getFirst());
088                                }
089                        }
090                }
091
092//              SignalControlData control = this.signalsData.getSignalControlData();
093//              for (SignalSystemControllerData  controller : control.getSignalSystemControllerDataBySystemId().values()) {
094//                      Map<Id<SignalGroup>, SignalGroupData> signalGroups = this.signalsData.getSignalGroupsData().getSignalGroupDataBySystemId(controller.getSignalSystemId());
095//                      for (SignalPlanData plan : controller.getSignalPlanData().values()) {
096//                              for (SignalGroupSettingsData settings : plan.getSignalGroupSettingsDataByGroupId().values()) {
097//                                      if (! signalGroups.containsKey(settings.getSignalGroupId())){
098//
099//                                      }
100//                              }
101//                      }
102//              }
103
104
105                for (Map<Id<SignalGroup>, SignalGroupData> sigGroups : this.signalsData.getSignalGroupsData().getSignalGroupDataBySignalSystemId().values()) {
106                        for(SignalGroupData sigGroup : sigGroups.values()){
107                                if(sigGroup.getSignalIds().isEmpty()){
108                                        emptyGroups.add(new Tuple<> (sigGroup.getId(), sigGroup.getSignalSystemId()));
109                                }
110                        }
111                }
112
113                for (Tuple<Id<SignalGroup>, Id<SignalSystem>> tuple : emptyGroups){
114                        signalsData.getSignalGroupsData().getSignalGroupDataBySignalSystemId().get(tuple.getSecond()).remove(tuple.getFirst());
115                        for(SignalPlanData sigPlanData : signalsData.getSignalControlData().getSignalSystemControllerDataBySystemId().get(tuple.getSecond()).getSignalPlanData().values()){
116                                sigPlanData.getSignalGroupSettingsDataByGroupId().remove(tuple.getFirst());
117                        }
118                }
119
120                for(SignalSystemData system : signalsData.getSignalSystemsData().getSignalSystemData().values()){
121                        if (system.getSignalData().isEmpty()) {
122                                emptySystems.add(system.getId());
123                                if (!signalsData.getSignalGroupsData().getSignalGroupDataBySystemId(system.getId()).isEmpty()){
124                                        log.warn("a system contains no signals but groups");
125                                }
126                        }
127                }
128
129                for(Id<SignalSystem> systemId : emptySystems){
130                        signalsData.getSignalSystemsData().getSignalSystemData().remove(systemId);
131                        signalsData.getSignalControlData().getSignalSystemControllerDataBySystemId().remove(systemId);
132                        signalsData.getSignalGroupsData().getSignalGroupDataBySignalSystemId().remove(systemId);
133                }
134        }
135
136
137        private void checkSignalToLinkMatching() {
138                SignalSystemsData signalSystems = this.signalsData.getSignalSystemsData();
139                for (SignalSystemData system : signalSystems.getSignalSystemData().values()) {
140                        for (SignalData signal : system.getSignalData().values()) {
141                                if (! this.network.getLinks().containsKey(signal.getLinkId())){
142                                        log.error("Error: No Link for Signal: "); 
143                                        log.error("\t\tSignalData Id: "  + signal.getId() + " of SignalSystemData Id: " + system.getId() 
144                                                        + " is located at Link Id: " + signal.getLinkId() + " but this link is not existing in the network!");
145                                        this.malformedSignals.add(new Tuple<>(signal.getId(), system.getId()));
146                                }
147                        }
148                }
149                
150        }
151
152        private void checkSignalToLaneMatching() {
153                SignalSystemsData signalSystems = this.signalsData.getSignalSystemsData();
154                
155                for (SignalSystemData system : signalSystems.getSignalSystemData().values()) {
156                        for (SignalData signal : system.getSignalData().values()) {
157                                if (signal.getLaneIds() != null && ! signal.getLaneIds().isEmpty()){
158                                        if (! this.lanes.getLanesToLinkAssignments().containsKey(signal.getLinkId())){
159                                                log.error("Error: No LanesToLinkAssignment for Signals:");
160                                                log.error("\t\tSignalData Id: "  + signal.getId() + " of SignalSystemData Id: " + system.getId() 
161                                                        + " is located at some lanes of Link Id: " + signal.getLinkId() + " but there is no LanesToLinkAssignemt existing in the LaneDefinitions.");
162                                                malformedSignals.add(new Tuple<>(signal.getId(), system.getId()));
163                                        }
164                                        else {
165                                                List<Id<Lane>> lanesToRemove = new LinkedList<>();
166                                                LanesToLinkAssignment l2l = this.lanes.getLanesToLinkAssignments().get(signal.getLinkId());
167                                                for (Id<Lane> laneId : signal.getLaneIds()) {
168                                                        if (! l2l.getLanes().containsKey(laneId)) {
169                                                                log.error("Error: No Lane for Signal: "); 
170                                                                log.error("\t\tSignalData Id: "  + signal.getId() + " of SignalSystemData Id: " + system.getId() 
171                                                                                + " is located at Link Id: " + signal.getLinkId() + " at Lane Id: " + laneId + " but this link is not existing in the network!");
172                                                                lanesToRemove.add(laneId);
173                                                        }
174                                                }
175                                                for (Id<Lane> laneId : lanesToRemove){
176                                                        signal.getLaneIds().remove(laneId);
177                                                }
178                                        }
179                                }
180                        }
181                }
182        }
