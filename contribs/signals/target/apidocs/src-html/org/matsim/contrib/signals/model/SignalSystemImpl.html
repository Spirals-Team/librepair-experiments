020package org.matsim.contrib.signals.model;
021
022import java.util.HashMap;
023import java.util.HashSet;
024import java.util.Map;
025import java.util.PriorityQueue;
026import java.util.Set;
027
028import org.matsim.api.core.v01.Id;
029import org.matsim.contrib.signals.events.SignalGroupStateChangedEvent;
030import org.matsim.core.mobsim.qsim.interfaces.SignalGroupState;
031
032
033/**
034 * @author dgrether
035 *
036 */
037public class SignalSystemImpl implements SignalSystem {
038
039        /*package*/ static final int SWITCH_OFF_SEQUENCE_LENGTH = 5;
040        
041        private SignalController signalController;
042        private Map<Id<SignalGroup>, SignalGroup> signalGroups = new HashMap<>();
043        private SignalSystemsManager signalManager;
044        
045        private Set<SignalGroupStateChangeRequest> requests = new HashSet<SignalGroupStateChangeRequest>();
046        
047        private PriorityQueue<SignalGroupStateChangeRequest> sortedRequests = new PriorityQueue<SignalGroupStateChangeRequest>();
048        private Id<SignalSystem> id;
049        private Map<Id<Signal>, Signal> signals = new HashMap<>();
050        
051        
052        public SignalSystemImpl(Id<SignalSystem> id) {
053                this.id = id;
054        }
055
056        @Override
057        public void setSignalSystemsManager(SignalSystemsManager signalManager) {
058                this.signalManager = signalManager;
059        }
060
061        @Override
062        public void setSignalSystemController(SignalController controller) {
063                this.signalController = controller;
064        }
065        
066        @Override
067        public void scheduleDropping(double timeSeconds, Id<SignalGroup> signalGroupId) {
068//              log.debug("dropping  at time " + timeSeconds + " of  group " + signalGroupId);
069                Set<SignalGroupStateChangeRequest> rqs = this.signalManager.getAmberLogic().processDropping(timeSeconds, this.getId(), signalGroupId);
070                requests.addAll(rqs);
071        }
072        
073        @Override
074        public void scheduleOnset(double timeSeconds, Id<SignalGroup> signalGroupId) {
075//              log.debug("onset at time " + timeSeconds + " of  group " + signalGroupId);
076                Set<SignalGroupStateChangeRequest> rqs = this.signalManager.getAmberLogic().processOnsets(timeSeconds, this.getId(), signalGroupId);
077                requests.addAll(rqs);
078        }
079        
080        @Override
081        public void updateState(double timeSeconds) {
082                this.signalController.updateState(timeSeconds);
083                SignalGroupStateChangedEvent stateEvent;
084                this.sortedRequests.addAll(this.requests);
085                this.requests.clear();
086                SignalGroupStateChangeRequest request = this.sortedRequests.peek();
087                while (request != null && request.getTimeOfDay() <= timeSeconds){
088//                      log.debug("system id " + this.id + " group " + request.getSignalGroupId() + " state " + request.getRequestedState() + " at time " + timeSeconds);
089                        this.signalGroups.get(request.getSignalGroupId()).setState(request.getRequestedState());
090                        stateEvent = new SignalGroupStateChangedEvent(timeSeconds, this.getId(), request.getSignalGroupId(), request.getRequestedState());
091                        this.signalManager.getEventsManager().processEvent(stateEvent);
092                        this.sortedRequests.poll();
093                        request = this.sortedRequests.peek();
094                }
095        }
096        
097        @Override
098        public void switchOff(double timeSeconds) {
099                Set<SignalGroupStateChangeRequest> req = new HashSet<SignalGroupStateChangeRequest>();
100                for (Id<SignalGroup> sgId : this.signalGroups.keySet()){
101                        req.add(new SignalGroupStateChangeRequestImpl(sgId, SignalGroupState.YELLOW, timeSeconds));
102                        req.add(new SignalGroupStateChangeRequestImpl(sgId, SignalGroupState.OFF, timeSeconds + SWITCH_OFF_SEQUENCE_LENGTH));
103                }
104                this.sortedRequests.addAll(req);
105        }
106
107
108        @Override
109        public void simulationInitialized(double simStartTimeSeconds) {
110                this.signalController.simulationInitialized(simStartTimeSeconds);
111        }
112
113        
114        @Override
115        public Id<SignalSystem> getId() {
116                return this.id;
117        }
118
119        @Override
120        public void addSignal(Signal signal) {
121                this.signals.put(signal.getId(), signal);
122        }
123
124        @Override
125        public Map<Id<Signal>, Signal> getSignals() {
126                return this.signals;
127        }
128
129        @Override
130        public void addSignalGroup(SignalGroup group) {
131                this.signalGroups.put(group.getId(), group);
132        }
133        
134        @Override
135        public Map<Id<SignalGroup>, SignalGroup> getSignalGroups(){
136                return this.signalGroups;
137        }
138
139        @Override
140        public SignalController getSignalController() {
141                return this.signalController;
142        }
143
144        @Override
145        public void startPlan(double now) {
146                Set<SignalGroupStateChangeRequest> req = new HashSet<SignalGroupStateChangeRequest>();
147                for (Id<SignalGroup> sgId : this.signalGroups.keySet()){
148                        req.add(new SignalGroupStateChangeRequestImpl(sgId, SignalGroupState.START_PLAN, now));
149                }
150                this.sortedRequests.addAll(req);
151        }
152
153}
