020package org.matsim.contrib.signals.otfvis;
021
022import java.io.IOException;
023import java.nio.ByteBuffer;
024
025import org.matsim.core.mobsim.qsim.interfaces.SignalGroupState;
026import org.matsim.core.utils.misc.ByteBufferUtils;
027import org.matsim.lanes.vis.VisLane;
028import org.matsim.lanes.vis.VisLinkWLanes;
029import org.matsim.lanes.vis.VisSignal;
030import org.matsim.vis.otfvis.caching.SceneGraph;
031
032/**
033 * @author dgrether
034 */
035public class OTFSignalReader extends OTFLaneReader {
036
037        public OTFSignalReader() {
038                super();
039        }
040
041        @Override
042        public void readConstData(ByteBuffer in) throws IOException {
043                super.readConstData(in);
044                this.readSignalSystems(in);
045        }
046
047        private void readSignalSystems(ByteBuffer in){
048                int noSignalSystems = in.getInt();
049                for (int i = 0; i < noSignalSystems; i++){
050                        String systemId = ByteBufferUtils.getString(in);
051                        VisSignalSystem otfsystem = new VisSignalSystem(systemId);
052                        this.drawer.addOTFSignalSystem(otfsystem);
053                        
054                        int noGroups = in.getInt();
055                        for (int j = 0; j < noGroups; j++){
056                                String groupId = ByteBufferUtils.getString(in);
057                                VisSignalGroup otfgroup = new VisSignalGroup(systemId, groupId);
058                                otfsystem.addOTFSignalGroup(otfgroup);
059                                int noSignals = in.getInt();
060                                for (int k = 0; k < noSignals; k++){
061                                        String signalId = ByteBufferUtils.getString(in);
062                                        String linkId = ByteBufferUtils.getString(in);
063                                        VisLinkWLanes link = this.drawer.getLanesLinkData().get(linkId);
064                                        VisSignal signal = new VisSignal(systemId, signalId);
065                                        otfgroup.addSignal(signal);
066                                        int noLanes = in.getInt();
067                                        if (noLanes == 0){
068                                                link.addSignal(signal);
069                                        }
070                                        else {
071                                                for (int l = 0; l < noLanes; l++){
072                                                        String laneId = ByteBufferUtils.getString(in);
073                                                        VisLane laneData = link.getLaneData().get(laneId);
074                                                        laneData.addSignal(signal);
075                                                }
076                                        }
077                                        int noTurningMoveRestrictions = in.getInt();
078                                        for (int l = 0; l < noTurningMoveRestrictions; l++){
079                                                String toLinkId = ByteBufferUtils.getString(in);
080                                                VisLinkWLanes toLink = this.drawer.getLanesLinkData().get(toLinkId);
081                                                signal.addTurningMoveRestriction(toLink);
082                                        }
083                                }
084                        }
085                }
086        }
087        
088
089        @Override
090        public void readDynData(ByteBuffer in, SceneGraph graph) throws IOException {
091                int noEvents = in.getInt();
092                for (int i = 0; i < noEvents; i++){
093                        String systemId = ByteBufferUtils.getString(in);
094                        String groupId = ByteBufferUtils.getString(in);
095                        int stateInt = in.getInt();
096                        SignalGroupState state = null;
097                        if (stateInt == 1){
098                                state = SignalGroupState.GREEN;
099                        }
100                        else if (stateInt == 0){
101                                state = SignalGroupState.RED;
102                        }
103                        else if (stateInt == 2){
104                                state = SignalGroupState.REDYELLOW;
105                        }
106                        else if (stateInt == 3){
107                                state = SignalGroupState.YELLOW;
108                        }
109                        else if (stateInt == 4){
110                                state = SignalGroupState.OFF;
111                        }
112                        this.drawer.updateGreenState(systemId, groupId, state);
113                }
114        }
115}
