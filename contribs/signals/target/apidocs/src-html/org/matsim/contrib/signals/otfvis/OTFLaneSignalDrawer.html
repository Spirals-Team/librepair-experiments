040
041public class OTFLaneSignalDrawer extends OTFGLAbstractDrawableReceiver {
042
043//      private static final Logger log = Logger.getLogger(OTFLaneSignalDrawer.class);
044        
045        private static final int glListName = 2342;
046        
047        private static final double zCoord = 1.0;
048        private static final double quadSizeLinkEnd = 5.0;
049        private static final double quadSizeLinkStart = 3.5;
050        private static final double quadSizeLaneStart = 3.5;
051        private static final double quadSizeLaneEnd = 4;
052
053        private Map<String, VisLinkWLanes> lanesLinkData =  new HashMap<String, VisLinkWLanes>();
054        
055        private Map<String, VisSignalSystem> signalSystems = new HashMap<String, VisSignalSystem>();
056
057        private enum Color {LANECOLOR, GREEN, REDYELLOW, YELLOW, RED};
058        
059        private double currentLinkWidthCorrection = java.lang.Double.NEGATIVE_INFINITY;
060        
061        private int glNetList = -1;
062        
063        private VisLaneModelBuilder laneModelBuilder = new VisLaneModelBuilder();
064        
065        @Override
066        public void onDraw(GL2 gl) {
067                if (this.currentLinkWidthCorrection != OTFClientControl.getInstance().getOTFVisConfig().getLinkWidth()){
068                        this.currentLinkWidthCorrection = OTFClientControl.getInstance().getOTFVisConfig().getLinkWidth();
069                        this.recalculatePositions();
070                        gl.glDeleteLists(this.glNetList, 1);
071                        this.glNetList = -1;
072                }
073                if (this.glNetList < 0){
074                        this.updateNetList(gl);
075                }
076                gl.glCallList(this.glNetList);
077                
078                for (VisLinkWLanes laneLinkData : this.lanesLinkData.values()){
079                        this.drawLinkEndsAndSignals(gl, laneLinkData);
080                }
081        }
082        
083        @Override
084        public void addToSceneGraph(SceneGraph graph) {
085                graph.addItem(this);
086        }
087        
088        private void updateNetList(GL2 gl){
089                this.glNetList = gl.glGenLists(glListName);
090                gl.glNewList(this.glNetList, GL2.GL_COMPILE);
091                for (VisLinkWLanes laneLinkData : this.lanesLinkData.values()){
092                        this.drawLink(gl, laneLinkData);
093                }
094                gl.glEndList();
095        }
096        
097
098        private void drawLinkEndsAndSignals(GL2 gl, VisLinkWLanes link) {
099                if (link.getLaneData() != null) {
100                        for (VisLane ld : link.getLaneData().values()){
101                                if (ld.getSignals() != null){
102                                        this.drawSignals(gl, ld.getSignals(), ld.getEndPoint(), link.getLinkOrthogonalVector(), ld.getToLinks());
103                                }
104                                else {
105                                        this.setColor(gl, Color.LANECOLOR);
106                                        this.drawLaneEnd(gl, ld);
107                                        this.drawToLinks(gl, ld.getEndPoint(), ld.getToLinks());
108                                }
109                        }
110                }
111                else { //link end without lanes
112                        if (link.getSignals() != null){
113                                this.drawSignals(gl, link.getSignals(), link.getLinkEndCenterPoint(), link.getLinkOrthogonalVector(), link.getToLinks());
114                        }
115                        else {
116                                this.setColor(gl, Color.LANECOLOR);
117                                this.drawQuad(gl, link.getLinkEndCenterPoint(), quadSizeLinkEnd);
118                                this.drawToLinks(gl, link.getLinkEndCenterPoint(), link.getToLinks());
119                        }
120                }
121        }
122
123        private void drawLink(GL2 gl, VisLinkWLanes link){
124                //draw a rect around linkStart
125                this.setColor(gl, Color.LANECOLOR);
126                this.drawQuad(gl, link.getLinkStartCenterPoint(), quadSizeLinkStart);
127                if (link.getLaneData() != null) { //draw the lanes
128                        for (VisLane ld : link.getLaneData().values()){
129                                this.drawLane(gl, ld, link);
130                        }
131                }
132                else { //draw the link
133                        this.drawVertex(gl, link.getLinkStartCenterPoint(), link.getLinkEndCenterPoint(), (float) (link.getNumberOfLanes() * 2));
134                }
135        }
136        
137        private void drawSignals(GL2 gl, Map<String, VisSignal> signals, Point2D.Double point, Point2D.Double ortho, List<VisLinkWLanes> toLinks){
138                double dist = signals.size() - 1;
139                Point2D.Double startPoint = this.laneModelBuilder.calcPoint(point, ortho, (quadSizeLinkEnd * -dist));
140                int i = 0;
141                for (VisSignal signal : signals.values()){
142                        i++;
143                        if (SignalGroupState.GREEN.equals(signal.getSignalGroupState())) {
144                                setColor(gl, Color.GREEN);
145                        }
146                        else if (SignalGroupState.RED.equals(signal.getSignalGroupState())) {
147                                this.setColor(gl, Color.RED);
148                        }
149                        else if (SignalGroupState.REDYELLOW.equals(signal.getSignalGroupState())) {
150                                this.setColor(gl, Color.REDYELLOW);
151                        }
152                        else if (SignalGroupState.YELLOW.equals(signal.getSignalGroupState())) {
153                                this.setColor(gl, Color.YELLOW);
154                        }
155                        else if (SignalGroupState.OFF.equals(signal.getSignalGroupState())){
156                                this.setColor(gl, Color.LANECOLOR);
157                        }
158                        this.drawQuad(gl, startPoint, quadSizeLinkEnd);
159                        if (!(signal.getTurningMoveRestrictions() == null || signal.getTurningMoveRestrictions().isEmpty())){
160                                this.drawToLinks(gl, startPoint, signal.getTurningMoveRestrictions());
161                        }
162                        else{
163                                this.drawToLinks(gl, startPoint, toLinks);
164                        }
165                        startPoint = this.laneModelBuilder.calcPoint(point, ortho, (quadSizeLinkEnd * i));
166                }
167        }
168
169        private void recalculatePositions() {
170                SnapshotLinkWidthCalculator linkWidthCalculator = new SnapshotLinkWidthCalculator();
171                linkWidthCalculator.setLaneWidth(OTFClientControl.getInstance().getOTFVisConfig().getEffectiveLaneWidth());
172                linkWidthCalculator.setLinkWidthForVis(OTFClientControl.getInstance().getOTFVisConfig().getLinkWidth());
173                for (VisLinkWLanes linkData : this.lanesLinkData.values()){
174                        this.laneModelBuilder.recalculatePositions(linkData, linkWidthCalculator);
175                }
176        }
177
178        private void drawLaneEnd(GL2 gl, VisLane ld){
179                this.setColor(gl, Color.LANECOLOR);
180                this.drawQuad(gl, ld.getEndPoint(), quadSizeLaneEnd);
181                if (!(ld.getToLanes() == null || ld.getToLanes().isEmpty())){
182                        for (VisLane toLane : ld.getToLanes()){
183                                this.drawVertex(gl, ld.getEndPoint(), toLane.getStartPoint(), 1.0f);
184                        }
185                }
186                
187        }
188        
189
190        private void drawLane(GL2 gl, VisLane ld, VisLinkWLanes laneLinkData){
191                //draw lane start
192                this.setColor(gl, Color.LANECOLOR);
193                this.drawQuad(gl, ld.getStartPoint(), quadSizeLaneStart);
194                //draw line between lane start point and lane end point
195                this.drawVertex(gl, ld.getStartPoint(), ld.getEndPoint(), ((float)ld.getNumberOfLanes()* 2));
196        }
197
198        
199        private void drawToLinks(GL2 gl, Point2D.Double fromPoint, List<VisLinkWLanes> toLinks) {
200                if (!(toLinks == null || toLinks.isEmpty())){
201                        for (VisLinkWLanes toLink : toLinks){
202                                this.drawVertex(gl, fromPoint, toLink.getLinkStartCenterPoint(), 1.0f);
203                        }
204                }
205        }
206
207
208        private void drawVertex(GL2 gl, Point2D.Double startPoint, Point2D.Double endPoint, float lineWidth){
209                gl.glLineWidth(lineWidth);
210                gl.glBegin(GL2.GL_LINES);
211                gl.glVertex3d(startPoint.x, startPoint.y , zCoord);
212                gl.glVertex3d(endPoint.x, endPoint.y, zCoord);
213                gl.glEnd();
214        }
215        
216        private void drawQuad(GL2 gl, Point2D.Double point, double quadSize){
217                gl.glBegin(GL2.GL_QUADS);
218                gl.glVertex3d(point.x - quadSize, point.y - quadSize, zCoord);
219                gl.glVertex3d(point.x - quadSize, point.y + quadSize, zCoord);
220                gl.glVertex3d(point.x + quadSize, point.y + quadSize, zCoord);
221                gl.glVertex3d(point.x + quadSize, point.y - quadSize, zCoord);
222                gl.glEnd();
223        }
224        
225        
226        private void setColor(GL2 gl, Color color){
227                switch (color) {
228                        case GREEN:
229                                gl.glColor3d(0.0, 1.0, 0.0);
230                                break;
231                        case RED:
232                                gl.glColor3d(1.0, 0.0, 0.0);
233                                break;
234                        case REDYELLOW:
235                                gl.glColor3d(1.0, 0.75, 0.0);
236                                break;
237                        case YELLOW:
238                                gl.glColor3d(1.0, 1.0, 0.0);
239                                break;
240                        case LANECOLOR:
241                                gl.glColor3d(1.0, 1.0, 0.8);
242                                break;
243                        default:
244                                gl.glColor3d(1.0, 1.0, 0.8);
245                                break;
246                }
247        }

250        public void addLaneLinkData(VisLinkWLanes laneLinkData){
251                this.lanesLinkData.put(laneLinkData.getLinkId(), laneLinkData);
252        }
253        
254        public Map<String, VisLinkWLanes> getLanesLinkData(){
255                return this.lanesLinkData;
256        }
257
258        public void updateGreenState(String systemId, String groupId, SignalGroupState state) {
259                this.signalSystems.get(systemId).getOTFSignalGroups().get(groupId).setState(state);
260        }
261
262        public void addOTFSignalSystem(VisSignalSystem otfsystem) {
263                this.signalSystems .put(otfsystem.getId(), otfsystem);
264        }
