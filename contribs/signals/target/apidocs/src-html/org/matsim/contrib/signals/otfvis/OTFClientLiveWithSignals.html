052
053public class OTFClientLiveWithSignals {
054
055        public static void run(final Config config, final OTFServer server) {
056                SwingUtilities.invokeLater(new Runnable() {
057                        @Override
058                        public void run() {
059                                OTFVisConfigGroup otfVisConfigGroup = ConfigUtils.addOrGetModule(config, OTFVisConfigGroup.GROUP_NAME, OTFVisConfigGroup.class);
060                                OTFConnectionManager connectionManager = new OTFConnectionManager();
061                                connectionManager.connectLinkToWriter(OTFLinkAgentsHandler.Writer.class);
062                                connectionManager.connectWriterToReader(OTFLinkAgentsHandler.Writer.class, OTFLinkAgentsHandler.class);
063                                connectionManager.connectWriterToReader(OTFAgentsListHandler.Writer.class, OTFAgentsListHandler.class);
064                                
065                                if (config.transit().isUseTransit()) {
066                                        connectionManager.connectWriterToReader(FacilityDrawer.Writer.class, FacilityDrawer.Reader.class);
067                                        connectionManager.connectReaderToReceiver(FacilityDrawer.Reader.class, FacilityDrawer.DataDrawer.class);
068                                        connectionManager.connectReceiverToLayer(FacilityDrawer.DataDrawer.class, SimpleSceneLayer.class);
069                                }
070
071                                if ( (config.qsim().isUseLanes()) 
072                                                && (!ConfigUtils.addOrGetModule(config, SignalSystemsConfigGroup.GROUPNAME, SignalSystemsConfigGroup.class).isUseSignalSystems())) {
073                                        connectionManager.connectWriterToReader(OTFLaneWriter.class, OTFLaneReader.class);
074                                        connectionManager.connectReaderToReceiver(OTFLaneReader.class, OTFLaneSignalDrawer.class);
075                                        connectionManager.connectReceiverToLayer(OTFLaneSignalDrawer.class, SimpleSceneLayer.class);
076                                } else if ((boolean) ConfigUtils.addOrGetModule(config, SignalSystemsConfigGroup.GROUPNAME, SignalSystemsConfigGroup.class).isUseSignalSystems()) {
077                                        connectionManager.connectWriterToReader(OTFSignalWriter.class, OTFSignalReader.class);
078                                        connectionManager.connectReaderToReceiver(OTFSignalReader.class, OTFLaneSignalDrawer.class);
079                                        connectionManager.connectReceiverToLayer(OTFLaneSignalDrawer.class, SimpleSceneLayer.class);
080                                }
081                                Component canvas = OTFOGLDrawer.createGLCanvas(otfVisConfigGroup);
082                                final OTFHostControl hostControl = new OTFHostControl(server, canvas);
083                                SettingsSaver saver = new SettingsSaver("otfsettings");
084                                OTFVisConfigGroup visconf = saver.tryToReadSettingsFile();
085                                if (visconf == null) {
086                                        visconf = server.getOTFVisConfig();
087                                }
088                                OTFClientControl.getInstance().setOTFVisConfig(visconf); // has to be set before OTFClientQuadTree.getConstData() is invoked!
089                                OTFServerQuadTree serverQuadTree = server.getQuad(connectionManager);
090                                OTFClientQuadTree clientQuadTree = serverQuadTree.convertToClient(server, connectionManager);
091                                OTFOGLDrawer mainDrawer = new OTFOGLDrawer(clientQuadTree, otfVisConfigGroup, canvas, hostControl);
092                                OTFControlBar hostControlBar = new OTFControlBar(server, hostControl, mainDrawer);
093                                OTFQueryControl queryControl = new OTFQueryControl(server, visconf);
094                                OTFQueryControlToolBar queryControlBar = new OTFQueryControlToolBar(queryControl, visconf);
095                                queryControl.setQueryTextField(queryControlBar.getTextField());
096                                OTFVisFrame otfVisFrame = new OTFVisFrame(canvas, server, hostControlBar, mainDrawer, saver);
097                                otfVisFrame.getContentPane().add(queryControlBar, BorderLayout.SOUTH);
098                                mainDrawer.setQueryHandler(queryControl);
099                                if (otfVisConfigGroup.isMapOverlayMode()) {
100                                        TileFactory tf;
101                                        if (otfVisConfigGroup.getMapBaseURL().isEmpty()) {
102                                                assertZoomLevel17(config);
103                                                tf = osmTileFactory();
104                                        } else {
105                                                WMSService wms = new WMSService(otfVisConfigGroup.getMapBaseURL(), otfVisConfigGroup.getMapLayer());
106                                                tf = new OTFVisWMSTileFactory(wms, otfVisConfigGroup.getMaximumZoom());
107                                        }
108                                        otfVisFrame.addMapViewer(tf);
109                                }
110                otfVisFrame.pack();
111                                otfVisFrame.setVisible(true);
112                        }
113                });
114        }
115        
116        private static void assertZoomLevel17(Config config) {
117                if(ConfigUtils.addOrGetModule(config, OTFVisConfigGroup.GROUP_NAME, OTFVisConfigGroup.class).getMaximumZoom() != 17) {
118                        throw new RuntimeException("The OSM layer only works with maximumZoomLevel = 17. Please adjust your config.");
119                }
120        }
121        
122        private static TileFactory osmTileFactory() {
123                final int max=17;
124                TileFactoryInfo info = new TileFactoryInfo(0, 17, 17,
125                                256, true, true,
126                                "http://tile.openstreetmap.org",
127                                "x","y","z") {
128                        @Override
129                        public String getTileUrl(int x, int y, int zoom) {
130                                zoom = max-zoom;
131                                String url = this.baseURL +"/"+zoom+"/"+x+"/"+y+".png";
132                                return url;
133                        }
134
135                };
136                TileFactory tf = new DefaultTileFactory(info);
137                return tf;
138        }
139        
140        private static class OTFVisWMSTileFactory extends DefaultTileFactory {
141                public OTFVisWMSTileFactory(final WMSService wms, final int maxZoom) {
142                        super(new TileFactoryInfo(0, maxZoom, maxZoom,
143                                        256, true, true, // tile size and x/y orientation is r2l & t2b
144                                        "","x","y","zoom") {
145                                @Override
146                                public String getTileUrl(int x, int y, int zoom) {
147                                        int zz = maxZoom - zoom;
148                                        int z = (int)Math.pow(2,(double)zz-1);
149                                        return wms.toWMSURL(x-z, z-1-y, zz, getTileSize(zoom));
150                                }
151
152                        });
153                }
154        }
