021package org.matsim.contrib.signals.data.conflicts;
022
023import java.util.HashMap;
024import java.util.Map;
025
026import org.matsim.api.core.v01.Id;
027import org.matsim.api.core.v01.network.Node;
028import org.matsim.contrib.signals.model.SignalSystem;
029
030/**
031 * @author tthunig
032 */
033public class ConflictDataImpl implements ConflictData {
034
035        private ConflictDataFactory factory = new ConflictDataFactoryImpl();
036        
037        private Map<Id<SignalSystem>, IntersectionDirections> conflictsPerSignalSystem = new HashMap<>();
038        private Map<Id<Node>, IntersectionDirections> conflictsPerNode = new HashMap<>();
039        
040        private Map<Id<SignalSystem>, Id<Node>> system2nodeMap = new HashMap<>();
041        private Map<Id<Node>, Id<SignalSystem>> node2systemMap = new HashMap<>();
042        
043        @Override
044        public void addConflictingDirectionsForIntersection(Id<SignalSystem> signalSystemId, Id<Node> nodeId,
045                        IntersectionDirections conflictingDirections) {
046                conflictsPerSignalSystem.put(signalSystemId, conflictingDirections);
047                conflictsPerNode.put(nodeId, conflictingDirections);
048                
049                system2nodeMap.put(signalSystemId, nodeId);
050                node2systemMap.put(nodeId, signalSystemId);
051        }
052
053        @Override
054        public Map<Id<SignalSystem>, IntersectionDirections> getConflictsPerSignalSystem() {
055                return conflictsPerSignalSystem;
056        }
057
058        @Override
059        public Map<Id<Node>, IntersectionDirections> getConflictsPerNode() {
060                return conflictsPerNode;
061        }
062
063        @Override
064        public ConflictDataFactory getFactory() {
065                return factory;
066        }
067
068        @Override
069        public void setFactory(ConflictDataFactory factory) {
070                this.factory = factory;
071        }
072
073        @Override
074        public void removeConflictingDirectionsForSignalSystem(Id<SignalSystem> signalSystemId) {
075                conflictsPerSignalSystem.remove(signalSystemId);
076                conflictsPerNode.remove(system2nodeMap.get(signalSystemId));
077        }
078
079        @Override
080        public void removeConflictingDirectionsForNode(Id<Node> nodeId) {
081                conflictsPerNode.remove(nodeId);
082                conflictsPerSignalSystem.remove(node2systemMap.get(nodeId));
083        }
084
085}
