032/**
033 * Implements a Fixed-time control.
034 * 
035 * @author dgrether, tthunig
036 *
037 */
038public class DefaultPlanbasedSignalSystemController extends AbstractSignalController implements SignalController {
039        
040        private static final Logger log = Logger.getLogger(DefaultPlanbasedSignalSystemController.class);
041        
042        /* an identifier 'PlanbasedSignalControl' would be better analogously to all other identifiers of signal controller
043         * but renaming would cause to much problems with old signal control input files. theresa jan'17
044         */
045        public static final String IDENTIFIER = "DefaultPlanbasedSignalSystemController";
046        private LinkedList<SignalPlan> planQueue = null;
047        private SignalPlan activePlan = null;
048        private double nextActivePlanCheckTime;
049        
050        @Override
051        public void updateState(double timeSeconds) {
052                if (nextActivePlanCheckTime <= timeSeconds){
053                        this.checkActivePlan(timeSeconds);
054                }
055                if (this.activePlan != null){
056                        // process droppings and onsets of the active plan of this second
057                        List<Id<SignalGroup>> droppingGroupIds = this.activePlan.getDroppings(timeSeconds);
058                        this.processDroppingGroupIds(timeSeconds, droppingGroupIds);
059                        
060                        List<Id<SignalGroup>> onsetGroupIds = this.activePlan.getOnsets(timeSeconds);
061                        this.processOnsetGroupIds(timeSeconds, onsetGroupIds);
062                }
063        }
064        
065        private void processOnsetGroupIds(double timeSeconds, List<Id<SignalGroup>> onsetGroupIds) {
066                if (onsetGroupIds != null){
067                        for (Id<SignalGroup> id : onsetGroupIds){
068                                this.system.scheduleOnset(timeSeconds, id);
069                        }
070                }               
071        }
072
073        private void processDroppingGroupIds(double timeSeconds, List<Id<SignalGroup>> droppingGroupIds){
074                if (droppingGroupIds != null){
075                        for (Id<SignalGroup> id : droppingGroupIds){
076                                this.system.scheduleDropping(timeSeconds, id);
077                        }
078                }
079        }
080        
081        /**
082         * Method to be called when a check time is reached.
083         * Checks whether the next signal plan has to start or if not
084         * whether the active signal plan has to end, though.
085         * It also sets next check times.
086         * 
087         * This method may assume some important facts:
088         * 1. there is always a next plan (because activated plans are put to the end of the queue again for the next day, see startFirstPlanInQueue)
089         * 2. start and end times of signal plans always exist (see validateSignalPlans)
090         * 3. the start time of the next plan is always >= the end time of the active plan (or at the next day) (see validateSignalPlans)
091         * 
092         * @param now current time
093         */
094        private void checkActivePlan(double now){
095                SignalPlan nextPlan = this.planQueue.peek();
096                if (nextPlan.getStartTime() == now % (3600*24)) {
097                        /* start time of next signal plan is reached: 
098                         * stop active plan (if it is still running) and start next signal plan */
099                        log.info("Starting signal control plan " + nextPlan.getId());
100                        startFirstPlanInQueue(now);
101                }
102                else if (this.activePlan != null && now % (3600*24) == this.activePlan.getEndTime()) {
103                        /* no next plan has started but active signal plans end time is reached: 
104                         * stop active signal plan. look for next starting time. */
105                        log.info("Switching off active signal control plan with id " + this.activePlan.getId());
106//                      if (nextPlan.getStartTime() > (now % (3600*24) + SignalSystemImpl.SWITCH_OFF_SEQUENCE_LENGTH)){
107                        double diff = nextPlan.getStartTime() - this.activePlan.getEndTime() + 24*3600;
108                        double mod = diff % (24*3600);
109                        if (mod > SignalSystemImpl.SWITCH_OFF_SEQUENCE_LENGTH){
110                                // switch off the signals if next plan is starting in more than 5 seconds
111                                this.system.switchOff(now);
112                        }
113                        waitForNextPlanInQueue(now);
114                }
115                else { // This case is just for completeness. It should not happen when check times are set correctly.
116                        /* no active plan exists and nextPlan has not started yet: determine next check time */
117                        waitForNextPlanInQueue(now);
118                }
119        }
120        
121        /**
122         * Correct next start or end time of a signal plan.
123         * Needed in case a plan goes over midnight or simulation reaches the next day.
124         * 
125         * @param startEndTime start or end time of a signal plan between 00:00:00 and 23:59:59 in seconds
126         * @param now current time in seconds, giving also the current day of simulation
127         * @return the next start or end time of the signal plan in seconds corresponding to the current day
128         */
129        private double adaptTime2Day(double startEndTime, double now) {
130                while (startEndTime <= now) {
131                        startEndTime += 3600 * 24.0;
132                }
133                return startEndTime;
134        }
135
136        @Override
137        public void simulationInitialized(double simStartTime) {                
138                // store all plans in a queue, sort them according to their end times and validate them
139                this.planQueue = new LinkedList<SignalPlan>(this.signalPlans.values());
140                Collections.sort(planQueue, new SignalPlanEndTimeComparator(simStartTime));
141                validateSignalPlans();
142                
143                // check if first plan should already start
144                SignalPlan firstPlan = this.planQueue.peek();
145                if (firstPlan.getStartTime() < firstPlan.getEndTime()){
146                        // plan starts after midnight (i.e. does not go over midnight)
147                        if (firstPlan.getStartTime() <= simStartTime && firstPlan.getEndTime() >= simStartTime){
148                                startFirstPlanInQueue(simStartTime);
149                                /* possible cases:
150                                 * 1. simulation starts in between start and end time of the first signal plan [-|*|---|1**!*|--|**|-]
151                                 */
152                        } else { // startTime > simStartTime or endTime < simStartTime
153                                waitForNextPlanInQueue(simStartTime);
154                                /* possible cases:
155                                 * 1. all signal plans start and end before simStartTime [--|1**|--|**|--!------]
156                                 * 2. first signal plan starts after simStartTime [-|*|--!--|1**|--|**|-]
157                                 */
158                        }
159                } else { // startTime >= endTime
160                        // plan starts before midnight (i.e. goes over midnight)
161                        if (firstPlan.getStartTime() <= simStartTime || firstPlan.getEndTime() >= simStartTime){
162                                startFirstPlanInQueue(simStartTime);
163                                /* possible cases:
164                                 * 1. midnight plan has started this day but before simulation start [**|--|*|---|1*!***]
165                                 * 2. midnight plan has already started last day [**!*|--|*|---|1**]
166                                 */
167                        } else { // startTime > simStartTime && endTime < simStartTime
168                                waitForNextPlanInQueue(simStartTime);
169                                /* possible cases:
170                                 * 1. midnight plan starts later the day [**|--|*|--!--|1***]
171                                 */
172                        }
173                }
174        }
175
176        /**
177         * Activate the signal plan that is first in the queue. 
178         * Moves plan from the queues first position to the last position.
179         * (Necessary to keep simulating signals when plans go over midnight and/or the simulation lasts for more than 24h.)
180         * Determine the time when to check active plans next.
181         * 
182         * @param now current time
183         */
184        private void startFirstPlanInQueue(double now) {
185                this.activePlan = this.planQueue.poll();
186                // shift plan to queue end
187                planQueue.add(activePlan);
188//              this.signalSystem.startPlan(now);
189                this.nextActivePlanCheckTime = adaptTime2Day(activePlan.getEndTime(), now);
190        }
191
192        /**
193         * Reset active plan - no plan is active.
194         * Determine the time when to check active plans next.
195         * 
196         * @param now current time
197         */
198        private void waitForNextPlanInQueue(double now) {
199                this.activePlan = null;
200                this.nextActivePlanCheckTime = adaptTime2Day(this.planQueue.peek().getStartTime(), now);
201        }
202        
203        /**
204         * This method checks whether signal plans overlap each other.
205         * It also throws an exception if start or end times of signal plans are not specified.
206         * 
207         * @throws UnsupportedOperationException if signal plans of one signal system overlap
208         */
209        private void validateSignalPlans() {
210                SignalPlan planI = null;
211                // iterate over all signal plans. check always overlapping between plan i and plan i+1
212                for (SignalPlan planIplus1 : this.planQueue) {
213                        if (planI != null) {
214                                checkOverlapping(planI, planIplus1);
215                        }
216                        planI = planIplus1;
217                }
218                if (this.planQueue.size() > 1) {
219                        // check also overlapping of last and first plan
220                        checkOverlapping(planI, this.planQueue.peek());
221                }
222        }
223
224        /**
225         * Assume that no plan covers the hole day.
226         * Than there are 4 valid cases how plan 1 and 2 can be ordered within the day such that they do not overlap:
227         * 1. [--|1**|--|2**|--] <=> start1 <= end1 <= start2 <= end2
228         * 2. [*|---|1**|--|2**] <=> end2 <= start1 <= end1 <= start2
229         * 3. [--|2**|---|1**|-] <=> start2 <= end2 <= start1 <= end1
230         * 4. [**|--|2**|---|1*] <=> end1 <= start2 <= end2 <= start1
231         * This means:
232         * If no plan covers the hole day it is
233         * 3 of the 4 inequalities are fulfilled <=> plans do not overlap.
234         * 
235         * The method throws an UnsupportedOperationException when less than 3 inequalities are fulfilled, i.e. when the plans overlap.
236         * 
237         * @param plan1
238         * @param plan2
239         */
240        private void checkOverlapping(SignalPlan plan1, SignalPlan plan2) {
241                if (plan1.getStartTime() % (24*3600) == plan1.getEndTime() % (24*3600) || plan2.getStartTime() % (24*3600) == plan2.getEndTime() % (24*3600)){
242                        throw new UnsupportedOperationException("Signal system " + system.getId() + " has multiple plans but at least one of them covers the hole day. "
243                                        + "If multiple signal plans are used, they are not allowed to overlap.");
244                }
245                
246                int counter = 0;
247                if (plan1.getStartTime() <= plan1.getEndTime())
248                        counter++;
249                if (plan1.getEndTime() <= plan2.getStartTime())
250                        counter++;
251                if (plan2.getStartTime() <= plan2.getEndTime())
252                        counter++;
253                if (plan2.getEndTime() <= plan1.getStartTime())
254                        counter++;
255                if (counter < 3) {
256                        throw new UnsupportedOperationException("Signal plans " + plan1.getId() + " and " + plan2.getId() + " of signal system " + system.getId() + " overlap.");
257                }
258        }

266        /**
267         * Sorts Signal Plans according to their end times.
268         * The plan with the end time coming first after (not at!) simulation start gets the first position.
269         * The simulation start time is given in the constructor. 
270         */
271        private static class SignalPlanEndTimeComparator implements Comparator<SignalPlan>, Serializable {
272
273                private static final long serialVersionUID = 1L;
274                private final double simStartTime;
275
276                public SignalPlanEndTimeComparator(double simulationStart) {
277                        this.simStartTime = simulationStart;
278                }
279
280                @Override
281                public int compare(SignalPlan p1, SignalPlan p2) {
282                        if (p1.getEndTime() <= simStartTime && p2.getEndTime() > simStartTime) {
283                                // first end time before and second after simulationStart
284                                return 1;
285                        } else if (p1.getEndTime() > simStartTime && p2.getEndTime() <= simStartTime) {
286                                // first end time after and second before simulationStart
287                                return -1;
288                        } else {
289                                // both before (incl. at) or both after simulationStart
290                                return Double.compare(p1.getEndTime(), p2.getEndTime());
291                        }
292                }
293                
294        }
