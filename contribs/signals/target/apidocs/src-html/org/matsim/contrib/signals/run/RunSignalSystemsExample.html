021package org.matsim.contrib.signals.run;
022
023import org.matsim.api.core.v01.Scenario;
024import org.matsim.contrib.signals.controler.SignalsModule;
025import org.matsim.contrib.signals.data.SignalsData;
026import org.matsim.contrib.signals.data.SignalsDataLoader;
027import org.matsim.core.config.Config;
028import org.matsim.core.config.ConfigUtils;
029import org.matsim.core.controler.Controler;
030import org.matsim.core.scenario.ScenarioUtils;
031
032/**
033 * Minimal example how to start your matsim run with traffic signals
034 * 
035 * @author tthunig
036 */
037public class RunSignalSystemsExample {
038        // do not change name of class; matsim book refers to it.  theresa, apr'18
039
040        /**
041         * @param args the path to your config file
042         */
043        public static void main(String[] args) {
044                if (args.length==0 || args.length>1) {
045                        throw new RuntimeException("Please provide exactly one argument -- the path to your config.xml file.") ;
046                }
047                // --- create the config
048                Config config = ConfigUtils.loadConfig(args[0]);
049
050                run(config); // The run method is extracted so that a test can operate on it.
051        }
052        
053        public static void run(Config config) {
054                // --- create the scenario
055                Scenario scenario = ScenarioUtils.loadScenario(config);
056                // load the information about signals data (i.e. fill the SignalsData object) and add it to the scenario as scenario element
057                scenario.addScenarioElement(SignalsData.ELEMENT_NAME, new SignalsDataLoader(config).loadSignalsData());
058
059                // --- create the controler
060                Controler c = new Controler(scenario);
061                // add the signals module to the simulation such that SignalsData is not only contained in the scenario but also used in the simulation
062                c.addOverridingModule(new SignalsModule());
063
064                // --- run the simulation
065                c.run();
066        }
067
068}
