021package org.matsim.contrib.signals.data.conflicts.io;
022
023import java.util.Arrays;
024import java.util.List;
025import java.util.Stack;
026
027import org.matsim.api.core.v01.Id;
028import org.matsim.api.core.v01.network.Link;
029import org.matsim.api.core.v01.network.Node;
030import org.matsim.contrib.signals.data.conflicts.ConflictData;
031import org.matsim.contrib.signals.data.conflicts.IntersectionDirections;
032import org.matsim.contrib.signals.data.conflicts.Direction;
033import org.matsim.contrib.signals.model.SignalSystem;
034import org.matsim.core.utils.io.MatsimXmlParser;
035import org.xml.sax.Attributes;
036
037/**
038 * @author tthunig
039 */
040public class ConflictingDirectionsReader extends MatsimXmlParser {
041
042        final static String INTERSECTION = "intersection";
043        final static String SIGNAL_SYSTEM_ID = "signalSystemId";
044        final static String NODE_ID = "nodeId";
045        final static String DIRECTION = "direction";
046        final static String ID = "id";
047        final static String FROM_LINK_ID = "fromLinkId";
048        final static String TO_LINK_ID = "toLinkId";
049        final static String CONFLICTING_DIRECTIONS = "conflictingDirections";
050        final static String DIRECTIONS_WITH_RIGHT_OF_WAY = "directionsWithRightOfWay";
051        final static String DIRECTIONS_WHICH_MUST_YIELD = "directionsWhichMustYield";
052        final static String NON_CONFLICTING_DIRECTIONS = "nonConflictingDirections";
053        
054        private ConflictData conflictData;
055        
056        private IntersectionDirections currentIntersection;
057        private Direction currentDirection;
058        
059        public ConflictingDirectionsReader(ConflictData conflictData) {
060                this.conflictData = conflictData;
061                this.setValidating(false);
062        }
063        
064        
065        @Override
066        public void startTag(String name, Attributes atts, Stack<String> context) {
067                switch (name) {
068                case INTERSECTION:
069                        startIntersection(Id.create(atts.getValue(SIGNAL_SYSTEM_ID), SignalSystem.class), Id.createNodeId(atts.getValue(NODE_ID)));
070                        break;
071                case DIRECTION:
072                        startDirection(Id.create(atts.getValue(ID), Direction.class), Id.createLinkId(atts.getValue(FROM_LINK_ID)), Id.createLinkId(atts.getValue(TO_LINK_ID)));
073                        break;
074                default:
075                        break;
076                }
077        }
078
079        private void startDirection(Id<Direction> directionId, Id<Link> fromLinkId, Id<Link> toLinkId) {
080                currentDirection = conflictData.getFactory().createDirection(currentIntersection.getSignalSystemId(), currentIntersection.getNodeId(), fromLinkId, toLinkId, directionId);
081                currentIntersection.addDirection(currentDirection);
082        }
083
084
085        private void startIntersection(Id<SignalSystem> signalSystemId, Id<Node> nodeId) {
086                currentIntersection = conflictData.getFactory().createConflictingDirectionsContainerForIntersection(signalSystemId, nodeId);
087                conflictData.addConflictingDirectionsForIntersection(signalSystemId, nodeId, currentIntersection);
088        }
089
090
091        @Override
092        public void endTag(String name, String content, Stack<String> context) {
093                switch (name) {
094                case CONFLICTING_DIRECTIONS:
095                case DIRECTIONS_WITH_RIGHT_OF_WAY:
096                case DIRECTIONS_WHICH_MUST_YIELD:
097                case NON_CONFLICTING_DIRECTIONS:
098                        if (content != null && !content.equals("") && !content.equals(" ")) {
099                                addConflictsForCurrentDirection(name, content);
100                        }
101                        break;
102                default:
103                        break;
104                }
105        }
106
107
108        private void addConflictsForCurrentDirection(String name, String directions) {
109                List<String> directionsList = Arrays.asList(directions.trim().split(" "));
110                for (String direction : directionsList) {
111                        Id<Direction> directionId = Id.create(direction, Direction.class);
112                        switch (name) {
113                        case CONFLICTING_DIRECTIONS:
114                                currentDirection.addConflictingDirection(directionId);
115                                break;
116                        case DIRECTIONS_WITH_RIGHT_OF_WAY:
117                                currentDirection.addDirectionWithRightOfWay(directionId);
118                                break;
119                        case DIRECTIONS_WHICH_MUST_YIELD:
120                                currentDirection.addDirectionWhichMustYield(directionId);
121                                break;
122                        case NON_CONFLICTING_DIRECTIONS:
123                                currentDirection.addNonConflictingDirection(directionId);
124                                break;
125                        }
126                }
127        }
128
129}
