039/**
040 * Flexible writer for all kind of signals data output. Easiest way to write all existing data is
041 * new SignalsScenarioWriter("myOutputDirectory").writeSignalsData(mySignalsDataInstance);
042 * 
043 * @author dgrether, tthunig
044 */
045public class SignalsScenarioWriter {
046        
047        private static final Logger log = Logger.getLogger(SignalsScenarioWriter.class);
048        
049        public static final String FILENAME_SIGNAL_SYSTEMS = "output_signal_systems_v2.0.xml.gz";
050        public static final String FILENAME_SIGNAL_GROUPS = "output_signal_groups_v2.0.xml.gz";
051        public static final String FILENAME_SIGNAL_CONTROL = "output_signal_control_v2.0.xml.gz";
052        public static final String FILENAME_AMBER_TIMES = "output_amber_times_v1.0.xml.gz";
053        private static final String FILENAME_INTERGREEN_TIMES = "output_intergreen_times_v1.0.xml.gz";
054        private static final String FILENAME_CONFLICTING_DIRECTIONS = "output_conflicting_directions_v1.0.xml.gz";
055        private static final String FILENAME_SIGNAL_CSV = "via_signals.csv.gz";
056
057        private String pathToSignalSystemsOutputFilename = null;
058        private String pathToSignalGroupsOutputFilename = null;
059        private String pathToSignalControlOutputFilename = null;
060        private String pathToAmberTimesOutputFilename = null;
061        private String pathToIntergreenTimesOutputFilename = null;
062        private String pathToSignalCSVOutputFilename = null;
063        private String pathToConflictingDirectionsOutputFilename = null;
064
065        public SignalsScenarioWriter(){
066        }
067        
068        public SignalsScenarioWriter(String outputDirectoryPath){
069                this.pathToSignalSystemsOutputFilename = outputDirectoryPath + FILENAME_SIGNAL_SYSTEMS;
070                this.pathToSignalGroupsOutputFilename = outputDirectoryPath + FILENAME_SIGNAL_GROUPS;
071                this.pathToSignalControlOutputFilename = outputDirectoryPath + FILENAME_SIGNAL_CONTROL;
072                this.pathToAmberTimesOutputFilename = outputDirectoryPath + FILENAME_AMBER_TIMES;
073                this.pathToIntergreenTimesOutputFilename =  outputDirectoryPath + FILENAME_INTERGREEN_TIMES;
074                this.pathToSignalCSVOutputFilename = outputDirectoryPath + FILENAME_SIGNAL_CSV;
075                this.pathToConflictingDirectionsOutputFilename = outputDirectoryPath + FILENAME_CONFLICTING_DIRECTIONS;
076        }
077        
078        
079        public SignalsScenarioWriter(final OutputDirectoryHierarchy controlerIo){
080                this.pathToSignalSystemsOutputFilename = controlerIo.getOutputFilename(FILENAME_SIGNAL_SYSTEMS);
081                this.pathToSignalGroupsOutputFilename = controlerIo.getOutputFilename(FILENAME_SIGNAL_GROUPS);
082                this.pathToSignalControlOutputFilename = controlerIo.getOutputFilename(FILENAME_SIGNAL_CONTROL);
083                this.pathToAmberTimesOutputFilename = controlerIo.getOutputFilename(FILENAME_AMBER_TIMES);
084                this.pathToIntergreenTimesOutputFilename = controlerIo.getOutputFilename(FILENAME_INTERGREEN_TIMES);
085                this.pathToSignalCSVOutputFilename = controlerIo.getOutputFilename(FILENAME_SIGNAL_CSV);
086                this.pathToConflictingDirectionsOutputFilename = controlerIo.getOutputFilename(FILENAME_CONFLICTING_DIRECTIONS);
087        }
088        
089        public void writeSignalsData(Scenario scenario){
090                SignalsData signalsData = (SignalsData) scenario.getScenarioElement(SignalsData.ELEMENT_NAME);
091                
092                if (signalsData.getSignalSystemsData() != null){
093                        this.writeSignalSystemsData(signalsData.getSignalSystemsData());
094                }
095                else {
096                        log.info("No SignalSystemsData object set to write!");
097                }
098                if (signalsData.getSignalGroupsData() != null){
099                        writeSignalGroupsData(signalsData.getSignalGroupsData());
100                }
101                else {
102                        log.info("No SignalGroupsData object set to write!");
103                }
104                if (signalsData.getSignalControlData() != null){
105                        writeSignalControlData(signalsData.getSignalControlData());
106                }
107                else {
108                        log.info("No SignalControlData object set to write!");
109                }
110                if (signalsData.getAmberTimesData() != null){
111                        writeAmberTimesData(signalsData.getAmberTimesData());
112                }
113                else {
114                        log.info("No AmberTimesData object set to write!");
115                }
116                if (signalsData.getIntergreenTimesData() != null){
117                        writeIntergreenTimesData(signalsData.getIntergreenTimesData());
118                }
119                else{
120                        log.info("No IntergreenTimesData object set to write!");
121                }
122                if (signalsData.getConflictingDirectionsData() != null){
123                        writeConflictingDirectionsData(signalsData.getConflictingDirectionsData());
124                }
125                else{
126                        log.info("No ConflictData object set to write!");
127                }
128        }
129
130        public void writeSignalSystemsData(SignalSystemsData signalSystemsData){
131                if (this.pathToSignalSystemsOutputFilename != null) {
132                        SignalSystemsWriter20 writer = new SignalSystemsWriter20(signalSystemsData);
133                        writer.write(this.pathToSignalSystemsOutputFilename);
134                }
135                else {
136                        log.warn("No path to signal systems output file set!");
137                }
138        }
139        
140        public void writeSignalGroupsData(SignalGroupsData signalGroupsData){
141                if (this.pathToSignalGroupsOutputFilename != null){
142                        SignalGroupsWriter20 writer = new SignalGroupsWriter20(signalGroupsData);
143                        writer.write(this.pathToSignalGroupsOutputFilename);
144                }
145                else {
146                        log.warn("No path to signal groups output file set!");
147                }
148        }
149        
150        public void writeSignalControlData(SignalControlData controlData){
151                if (this.pathToSignalControlOutputFilename != null) {
152                        SignalControlWriter20 writer = new SignalControlWriter20(controlData);
153                        writer.write(this.pathToSignalControlOutputFilename);
154                }
155                else {
156                        log.warn("No path to signal control output file set!");
157                }
158        }
159        
160        public void writeAmberTimesData(AmberTimesData amberTimesData){
161                if (this.pathToAmberTimesOutputFilename != null) {
162                        AmberTimesWriter10 writer = new AmberTimesWriter10(amberTimesData);
163                        writer.write(this.pathToAmberTimesOutputFilename);
164                }
165                else {
166                        log.warn("No path to amber times output file set!");
167                }
168        }
169        
170        public void writeIntergreenTimesData(IntergreenTimesData intergreenTimesData){
171                if (this.pathToIntergreenTimesOutputFilename != null){
172                        IntergreenTimesWriter10 writer = new IntergreenTimesWriter10(intergreenTimesData);
173                        writer.write(this.pathToIntergreenTimesOutputFilename);
174                }
175                else {
176                        log.warn("No path to intergreen times output file set!");
177                }
178        }
179        
180        public void writeConflictingDirectionsData(ConflictData conflictData){
181                if (this.pathToConflictingDirectionsOutputFilename != null){
182                        ConflictingDirectionsWriter writer = new ConflictingDirectionsWriter(conflictData);
183                        writer.write(this.pathToConflictingDirectionsOutputFilename);
184                }
185                else {
186                        log.warn("No path to conflicting directions output file set!");
187                }
188        }
189
190        
191        public String getSignalSystemsOutputFilename() {
192                return pathToSignalSystemsOutputFilename;
193        }
194
195        
196        public void setSignalSystemsOutputFilename(String signalSystemsOutputFilename) {
197                this.pathToSignalSystemsOutputFilename = signalSystemsOutputFilename;
198        }
199
200        
201        public String getSignalGroupsOutputFilename() {
202                return pathToSignalGroupsOutputFilename;
203        }
204
205        
206        public void setSignalGroupsOutputFilename(String signalGroupsOutputFilename) {
207                this.pathToSignalGroupsOutputFilename = signalGroupsOutputFilename;
208        }
209
210        
211        public String getSignalControlOutputFilename() {
212                return pathToSignalControlOutputFilename;
213        }
214
215        
216        public void setSignalControlOutputFilename(String signalControlOutputFilename) {
217                this.pathToSignalControlOutputFilename = signalControlOutputFilename;
218        }
219
220        
221        public String getAmberTimesOutputFilename() {
222                return pathToAmberTimesOutputFilename;
223        }
224
225        
226        public void setAmberTimesOutputFilename(String amberTimesOutputFilename) {
227                this.pathToAmberTimesOutputFilename = amberTimesOutputFilename;
228        }
229
230        
231        public String getIntergreenTimesOutputFilename() {
232                return pathToIntergreenTimesOutputFilename;
233        }
234
235        
236        public void setIntergreenTimesOutputFilename(String intergreenTimesOutputFilename) {
237                this.pathToIntergreenTimesOutputFilename = intergreenTimesOutputFilename;
238        }
