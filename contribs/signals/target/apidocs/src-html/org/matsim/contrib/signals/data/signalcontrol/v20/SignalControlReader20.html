051
052/**
053 * @author dgrether
054 *
055 */
056public class SignalControlReader20 implements MatsimReader {
057
058        private SignalControlData signalControlData;
059
060        public SignalControlReader20(SignalControlData signalControlData){
061                this.signalControlData = signalControlData;
062        }
063
064        private XMLSignalControl readSignalControl20Stream(InputStream stream) {
065                try {
066                        JAXBContext jc = JAXBContext.newInstance(org.matsim.jaxb.signalcontrol20.ObjectFactory.class);
067                        Unmarshaller u = jc.createUnmarshaller();
068                        u.setSchema(SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI).newSchema(getClass().getResource("/dtd/signalControl_v2.0.xsd")));
069                        return (XMLSignalControl) u.unmarshal(stream);
070                } catch (JAXBException | SAXException e) {
071                        throw new UncheckedIOException(e);
072                }
073        }
074
075        @Override
076        public void readFile(String filename) {
077                readStream(IOUtils.getInputStream(filename));
078        }
079
080        private double getSeconds(XMLGregorianCalendar daytime) {
081                double sec = daytime.getHour() * 3600.0;
082                sec += daytime.getMinute() * 60.0;
083                sec += daytime.getSecond();
084                return sec;
085        }
086
087        public void readStream(InputStream inputStream) {
088                XMLSignalControl xmlSignalControl = this.readSignalControl20Stream(inputStream);
089                SignalControlDataFactory factory = this.signalControlData.getFactory();
090                for (XMLSignalSystemType xmlSystem : xmlSignalControl.getSignalSystem()){
091                        SignalSystemControllerData controllerData = factory.createSignalSystemControllerData(Id.create(xmlSystem.getRefId(), SignalSystem.class));
092                        this.signalControlData.addSignalSystemControllerData(controllerData);
093                        XMLSignalSystemControllerType xmlController = xmlSystem.getSignalSystemController();
094                        controllerData.setControllerIdentifier(xmlController.getControllerIdentifier());
095                        for (XMLSignalPlanType xmlPlan : xmlController.getSignalPlan()){
096                                SignalPlanData plan = factory.createSignalPlanData(Id.create(xmlPlan.getId(), SignalPlan.class));
097                                controllerData.addSignalPlanData(plan);
098                                if (xmlPlan.getCycleTime() != null){
099                                        plan.setCycleTime(xmlPlan.getCycleTime().getSec());
100                                }
101                                if (xmlPlan.getOffset() != null){
102                                        plan.setOffset(xmlPlan.getOffset().getSec());
103                                }
104                                if (xmlPlan.getStart() != null){
105                                        plan.setStartTime(this.getSeconds(xmlPlan.getStart().getDaytime()));
106                                }
107                                if (xmlPlan.getStop() != null){
108                                        plan.setEndTime(this.getSeconds(xmlPlan.getStop().getDaytime()));
109                                }
110                                for (XMLSignalGroupSettingsType xmlSettings: xmlPlan.getSignalGroupSettings()){
111                                        SignalGroupSettingsData settings = factory.createSignalGroupSettingsData(Id.create(xmlSettings.getRefId(), SignalGroup.class));
112                                        plan.addSignalGroupSettings(settings);
113                                        settings.setDropping(xmlSettings.getDropping().getSec());
114                                        settings.setOnset(xmlSettings.getOnset().getSec());
115                                }
116                        }
117                }
118        }
119
120}
