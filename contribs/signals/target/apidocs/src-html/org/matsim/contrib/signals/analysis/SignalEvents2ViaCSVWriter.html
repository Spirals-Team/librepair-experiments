025import java.io.IOException;

053/**
054 * Write a csv file for visualizing signals in via based on the events.
055 * 
056 * @author tthunig
057 *
058 */
059public class SignalEvents2ViaCSVWriter implements SignalGroupStateChangedEventHandler, IterationEndsListener, IterationStartsListener {
060
061        private static final Logger log = Logger.getLogger(SignalEvents2ViaCSVWriter.class);
062
063        private static final String SIGNAL_ID = "signal id";
064        private static final String X_COORD = "x";
065        private static final String Y_COORD = "y";
066        private static final String TIME = "time";
067        private static final String SIGNAL_STATE = "signal state";
068        
069        private int countWarnings = 0;
070
071        /**
072         * distance between signal (the coordinate that will be drawn) and node in direction of the link
073         */
074        private static final double SIGNAL_COORD_NODE_OFFSET = 20.0;
075        /**
076         * distance between signal (the coordinate that will be drawn) and link in orthogonal direction to the link
077         */
078        private static final double SIGNAL_COORD_LINK_OFFSET = 5.0;
079
080        private BufferedWriter signalsCSVWriter;
081        private SignalsData signalsData;
082        Scenario scenario;
083
084        @Inject
085        public SignalEvents2ViaCSVWriter(Scenario scenario, ControlerListenerManager clm, EventsManager em) {
086                this.scenario = scenario;
087                clm.addControlerListener(this);
088                em.addHandler(this);
089        }
090
091        private Map<Id<Signal>, Coord> signal2Coord = new HashMap<>();
092
093        @Override
094        public void reset(int iteration) {
095        }
096
097        @Override
098        public void notifyIterationStarts(IterationStartsEvent event) {
099                if (event.getIteration() == scenario.getConfig().controler().getFirstIteration()) {
100                        /*
101                         * do all the stuff that is needed only once a simulation: - calculating coordinations for the via file - getting the signals data out of the scenario
102                         */
103                        init();
104                }
105
106                String signalCSVFilename = scenario.getConfig().controler().getOutputDirectory() + "/ITERS/it." + event.getIteration() + "/signalEvents2Via.csv";
107
108                // log.info("Initializing SignalsCSVWriter ...");
109                signalsCSVWriter = IOUtils.getBufferedWriter(signalCSVFilename);
110                log.info("Writing signal events of iteration " + event.getIteration() + " as csv file for via to " + signalCSVFilename + " ...");
111
112                // create header
113                try {
114                        signalsCSVWriter.write(SIGNAL_ID + ";" + X_COORD + ";" + Y_COORD + ";" + TIME + ";" + SIGNAL_STATE);
115                        signalsCSVWriter.newLine();
116                } catch (IOException e) {
117                        e.printStackTrace();
118                        log.error("Something went wrong while writing the header of the signals csv file.");
119                }
120        }
121
122        private void init() {
123                this.signalsData = (SignalsData) scenario.getScenarioElement(SignalsData.ELEMENT_NAME);
124
125                // calculate coordinates of all signals and remember them
126                for (SignalSystemData signalSystem : signalsData.getSignalSystemsData().getSignalSystemData().values()) {
127                        for (SignalData signalData : signalSystem.getSignalData().values()) {
128                                Coord signalCoord = calculateSignalCoordinate(signalData);
129                                signal2Coord.put(signalData.getId(), signalCoord);
130                        }
131                }
132        }
133
134        @Override
135        public void handleEvent(SignalGroupStateChangedEvent event) {
136                if (signalsData == null){
137                        if (countWarnings > 9){
138                                log.warn("You are using the SignalsModule without Controler. No output is written out. (This warning is only given 10 times.)");
139                                countWarnings++;
140                        }
141                        return;
142                }
143                SignalGroupData signalGroupData = signalsData.getSignalGroupsData().getSignalGroupDataBySystemId(event.getSignalSystemId()).get(event.getSignalGroupId());
144                // write a line for each signal of the group
145                for (Id<Signal> signalId : signalGroupData.getSignalIds()) {
146                        Coord signalCoord = signal2Coord.get(signalId);
147                        try {
148                                signalsCSVWriter.write(signalId + ";" + signalCoord.getX() + ";" + signalCoord.getY() + ";" + event.getTime() + ";" + event.getNewState());
149                                signalsCSVWriter.newLine();
150                        } catch (IOException e) {
151                                e.printStackTrace();
152                                log.error("Something went wrong while adding a line for the signal state switch of signal " + signalId);
153                        }
154                }
155        }
156
157        private Coord calculateSignalCoordinate(SignalData signalData) {
158                // log.info("Calculating coordinate for signal " + signalId + " of signal system " + signalSystemId + " ...");
159
160                Id<Link> signalLinkId = signalData.getLinkId();
161                Link signalLink = scenario.getNetwork().getLinks().get(signalLinkId);
162                Coord toNodeCoord = signalLink.getToNode().getCoord();
163                Coord fromNodeCoord = signalLink.getFromNode().getCoord();
164
165                int stepNumber = 0;
166                if (signalData.getLaneIds() != null) {
167                        // if the signal belongs to lanes, different signals may exist for one link -> determine step number for visualization
168                        Lanes lanes = scenario.getLanes();
169                        Id<Lane> firstSignalLaneId = (Id<Lane>) signalData.getLaneIds().toArray()[0];
170                        Lane firstSignalLane = lanes.getLanesToLinkAssignments().get(signalLinkId).getLanes().get(firstSignalLaneId);
171                        stepNumber = firstSignalLane.getAlignment();
172                }
173
174                // calculate delta X and Y depending on the node offset
175                double deltaX = 0;
176                double deltaY = 0;
177                // handle different cases of links
178                if (toNodeCoord.getX() == fromNodeCoord.getX()) {
179                        // vertical link
180                        deltaX = 0;
181                        if (toNodeCoord.getY() < fromNodeCoord.getY()) {
182                                deltaY = 1;
183                        } else {
184                                deltaY = -1;
185                        }
186
187                } else {
188                        // this case includes the case when the link is horizontal
189                        double m = (toNodeCoord.getY() - fromNodeCoord.getY()) / (toNodeCoord.getX() - fromNodeCoord.getX());
190                        deltaX = 1 / (Math.sqrt(1 + m * m));
191                        if (toNodeCoord.getX() > fromNodeCoord.getX()) {
192                                // link is oriented to the right -> coordinates has to be shifted to the left
193                                deltaX *= -1;
194                        }
195                        deltaY = m * deltaX;
196                }
197
198                // calculate x and y coord where the signal should be drawn
199                double x = toNodeCoord.getX() + SIGNAL_COORD_NODE_OFFSET * deltaX + stepNumber * SIGNAL_COORD_LINK_OFFSET * deltaY;
200                double y = toNodeCoord.getY() + SIGNAL_COORD_NODE_OFFSET * deltaY - stepNumber * SIGNAL_COORD_LINK_OFFSET * deltaX;
201
202                // log.info("... done!");
203                return new Coord(x, y);
204        }
205
206        @Override
207        public void notifyIterationEnds(IterationEndsEvent event) {
208                // close the stream of this iteration
209                try {
210                        signalsCSVWriter.flush();
211                        signalsCSVWriter.close();
212                } catch (IOException e1) {
213                        e1.printStackTrace();
214                        log.error("Something went wrong while closing the signals csv writer stream.");
215                }
216                log.info("... done for this iteration");
217        }
