020package org.matsim.contrib.signals;
021
022import org.matsim.core.config.Config;
023import org.matsim.core.config.ReflectiveConfigGroup;
024
025/**
026 * This config Module can be used to specify the paths to the
027 * xml files configuring the signals.
028 *
029 * @author dgrether
030 *
031 */
032public final class SignalSystemsConfigGroup extends ReflectiveConfigGroup {
033
034        public static final String GROUPNAME = "signalsystems";
035        public static final String USE_SIGNALSYSTEMS = "useSignalsystems";
036        public static final String SIGNALSYSTEM_FILE = "signalsystems";
037        public static final String SIGNALCONTROL_FILE = "signalcontrol";
038        public static final String SIGNALGROUPS_FILE = "signalgroups";
039        public static final String USE_AMBER_TIMES = "useAmbertimes";
040        public static final String AMBERTIMES_FILE = "ambertimes";
041        public static final String INTERGREENTIMES_FILE = "intergreentimes";
042        public static final String USE_INTERGREEN_TIMES = "useIntergreentimes";
043        public static final String ACTION_ON_INTERGREEN_VIOLATION = "actionOnIntergreenViolation";
044        public enum ActionOnIntergreenViolation{
045                WARN, EXCEPTION
046        }
047
048        private String signalSystemFile;
049        private String signalControlFile;
050        private String signalGroupsFile;
051        private String amberTimesFile;
052        private String intergreenTimesFile;
053        private boolean useIntergreens = false;
054        private boolean useAmbertimes = false;
055        private boolean useSignalSystems = false;
056        private ActionOnIntergreenViolation actionOnIntergreenViolation = ActionOnIntergreenViolation.WARN;
057        
058        public SignalSystemsConfigGroup() {
059                super(GROUPNAME);
060        }
061
062        @Override
063        protected void checkConsistency(Config config) {
064                if ((this.signalSystemFile == null) && (this.signalControlFile != null)) {
065                        throw new IllegalStateException("For using a SignalSystemConfiguration a definition of the signal systems must exist!");
066                }
067        }
068
069        @StringGetter( SIGNALSYSTEM_FILE )
070        public String getSignalSystemFile() {
071                return this.signalSystemFile;
072        }
073
074        @StringSetter( SIGNALSYSTEM_FILE )
075        public void setSignalSystemFile(final String signalSystemFile) {
076                this.signalSystemFile = signalSystemFile;
077        }
078
079
080        @StringGetter( SIGNALGROUPS_FILE )
081        public String getSignalGroupsFile() {
082                return this.signalGroupsFile;
083        }
084        
085        @StringSetter( SIGNALGROUPS_FILE )
086        public void setSignalGroupsFile(String signalGroupsFile){
087                this.signalGroupsFile = signalGroupsFile;
088        }
089
090        @StringGetter( AMBERTIMES_FILE )
091        public String getAmberTimesFile() {
092                return this.amberTimesFile;
093        }
094        
095        @StringSetter( AMBERTIMES_FILE )
096        public void setAmberTimesFile(String amberTimesFile){
097                this.amberTimesFile = amberTimesFile;
098        }
099        
100        @StringGetter( INTERGREENTIMES_FILE )
101        public String getIntergreenTimesFile() {
102                return intergreenTimesFile;
103        }
104        
105        @StringSetter( INTERGREENTIMES_FILE )
106        public void setIntergreenTimesFile(String intergreenTimesFile) {
107                this.intergreenTimesFile = intergreenTimesFile;
108        }
109
110        @StringGetter( SIGNALCONTROL_FILE )
111        public String getSignalControlFile() {
112                return this.signalControlFile;
113        }
114        
115        @StringSetter( SIGNALCONTROL_FILE )
116        public void setSignalControlFile(String signalControlFile){
117                this.signalControlFile = signalControlFile;
118        }
119
120        @StringGetter( USE_INTERGREEN_TIMES )
121        public boolean isUseIntergreenTimes() {
122                return this.useIntergreens;
123        }
124        
125        @StringSetter( USE_INTERGREEN_TIMES )
126        public void setUseIntergreenTimes(boolean useIntergreens){
127                this.useIntergreens = useIntergreens;
128        }
129        
130        @StringGetter( ACTION_ON_INTERGREEN_VIOLATION )
131        public ActionOnIntergreenViolation getActionOnIntergreenViolation() {
132                return actionOnIntergreenViolation;
133        }
134
135        @StringSetter( ACTION_ON_INTERGREEN_VIOLATION )
136        public void setActionOnIntergreenViolation(ActionOnIntergreenViolation actionOnIntergreenViolation) {
137                switch (actionOnIntergreenViolation){
138                // set the value for the supported actions
139                case WARN:
140                case EXCEPTION:
141                        this.actionOnIntergreenViolation = actionOnIntergreenViolation;
142                        break;
143                // throw an exception if the value is not supported
144                default:
145                        throw new IllegalArgumentException("The value " + actionOnIntergreenViolation 
146                                        + " for key : " + ACTION_ON_INTERGREEN_VIOLATION + " is not supported by this config group");
147                }
148        }
149        
150        @StringGetter( USE_AMBER_TIMES )
151        public boolean isUseAmbertimes() {
152                return useAmbertimes;
153        }
154        
155        @StringSetter( USE_AMBER_TIMES )
156        public void setUseAmbertimes(boolean useAmbertimes) {
157                this.useAmbertimes = useAmbertimes;
158        }
159
160        @StringGetter( USE_SIGNALSYSTEMS )
161        public boolean isUseSignalSystems() {
162                return this.useSignalSystems;
163        }
164
165        @StringSetter( USE_SIGNALSYSTEMS )
166        public void setUseSignalSystems(final boolean useSignalSystems) {
167                this.useSignalSystems = useSignalSystems;
168        }
169}
