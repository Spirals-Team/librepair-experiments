020package org.matsim.contrib.signals;
021
022import org.matsim.core.config.Config;
023import org.matsim.core.config.ReflectiveConfigGroup;
024
025/**
026 * This config Module can be used to specify the paths to the
027 * xml files configuring the signals.
028 *
029 * @author dgrether
030 *
031 */
032public final class SignalSystemsConfigGroup extends ReflectiveConfigGroup {
033
034        public static final String GROUPNAME = "signalsystems";
035        public static final String USE_SIGNALSYSTEMS = "useSignalsystems";
036        public static final String SIGNALSYSTEM_FILE = "signalsystems";
037        public static final String SIGNALCONTROL_FILE = "signalcontrol";
038        public static final String SIGNALGROUPS_FILE = "signalgroups";
039        public static final String USE_AMBER_TIMES = "useAmbertimes";
040        public static final String AMBERTIMES_FILE = "ambertimes";
041        public static final String CONFLICTING_DIRECTIONS_FILE = "conflictingDirections";
042        public static final String USE_CONFLICTING_DIRECTIONS = "useConflictingDirections";     
043        public static final String INTERGREENTIMES_FILE = "intergreentimes";
044        public static final String USE_INTERGREEN_TIMES = "useIntergreentimes";
045        public static final String ACTION_ON_INTERGREEN_VIOLATION = "actionOnIntergreenViolation";
046        public static final String ACTION_ON_CONFLICTING_DIRECTION_VIOLATION = "actionOnConflictingDirectionViolation";
047        public enum ActionOnSignalSpecsViolation{
048                WARN, EXCEPTION
049        }
050
051        private String signalSystemFile;
052        private String signalControlFile;
053        private String signalGroupsFile;
054        private String amberTimesFile;
055        private String intergreenTimesFile;
056        private String conflictingDirectionsFile;
057        private boolean useIntergreens = false;
058        private boolean useAmbertimes = false;
059        private boolean useSignalSystems = false;
060        private boolean useConflictingDirections = false;
061        private ActionOnSignalSpecsViolation actionOnIntergreenViolation = ActionOnSignalSpecsViolation.WARN;
062        private ActionOnSignalSpecsViolation actionOnConflictingDirectionViolation = ActionOnSignalSpecsViolation.WARN;
063        
064        public SignalSystemsConfigGroup() {
065                super(GROUPNAME);
066        }
067
068        @Override
069        protected void checkConsistency(Config config) {
070                if ((this.signalSystemFile == null) && (this.signalControlFile != null)) {
071                        throw new IllegalStateException("For using a SignalSystemConfiguration a definition of the signal systems must exist!");
072                }
073        }
074
075        @StringGetter( SIGNALSYSTEM_FILE )
076        public String getSignalSystemFile() {
077                return this.signalSystemFile;
078        }
079
080        @StringSetter( SIGNALSYSTEM_FILE )
081        public void setSignalSystemFile(final String signalSystemFile) {
082                this.signalSystemFile = signalSystemFile;
083        }
084
085
086        @StringGetter( SIGNALGROUPS_FILE )
087        public String getSignalGroupsFile() {
088                return this.signalGroupsFile;
089        }
090        
091        @StringSetter( SIGNALGROUPS_FILE )
092        public void setSignalGroupsFile(String signalGroupsFile){
093                this.signalGroupsFile = signalGroupsFile;
094        }
095
096        @StringGetter( AMBERTIMES_FILE )
097        public String getAmberTimesFile() {
098                return this.amberTimesFile;
099        }
100        
101        @StringSetter( AMBERTIMES_FILE )
102        public void setAmberTimesFile(String amberTimesFile){
103                this.amberTimesFile = amberTimesFile;
104        }
105        
106        @StringGetter( INTERGREENTIMES_FILE )
107        public String getIntergreenTimesFile() {
108                return intergreenTimesFile;
109        }
110        
111        @StringSetter( INTERGREENTIMES_FILE )
112        public void setIntergreenTimesFile(String intergreenTimesFile) {
113                this.intergreenTimesFile = intergreenTimesFile;
114        }
115
116        @StringGetter( SIGNALCONTROL_FILE )
117        public String getSignalControlFile() {
118                return this.signalControlFile;
119        }
120        
121        @StringSetter( SIGNALCONTROL_FILE )
122        public void setSignalControlFile(String signalControlFile){
123                this.signalControlFile = signalControlFile;
124        }
125        
126        @StringGetter( CONFLICTING_DIRECTIONS_FILE )
127        public String getConflictingDirectionsFile() {
128                return this.conflictingDirectionsFile;
129        }
130        
131        @StringSetter( CONFLICTING_DIRECTIONS_FILE )
132        public void setConflictingDirectionsFile(String conflictingDirectionsFile){
133                this.conflictingDirectionsFile = conflictingDirectionsFile;
134        }
135
136        @StringGetter( USE_INTERGREEN_TIMES )
137        public boolean isUseIntergreenTimes() {
138                return this.useIntergreens;
139        }
140        
141        @StringSetter( USE_INTERGREEN_TIMES )
142        public void setUseIntergreenTimes(boolean useIntergreens){
143                this.useIntergreens = useIntergreens;
144        }
145        
146        @StringGetter( ACTION_ON_INTERGREEN_VIOLATION )
147        public ActionOnSignalSpecsViolation getActionOnIntergreenViolation() {
148                return actionOnIntergreenViolation;
149        }
150
151        @StringSetter( ACTION_ON_INTERGREEN_VIOLATION )
152        public void setActionOnIntergreenViolation(ActionOnSignalSpecsViolation actionOnIntergreenViolation) {
153                switch (actionOnIntergreenViolation){
154                // set the value for the supported actions
155                case WARN:
156                case EXCEPTION:
157                        this.actionOnIntergreenViolation = actionOnIntergreenViolation;
158                        break;
159                // throw an exception if the value is not supported
160                default:
161                        throw new IllegalArgumentException("The value " + actionOnIntergreenViolation 
162                                        + " for key : " + ACTION_ON_INTERGREEN_VIOLATION + " is not supported by this config group");
163                }
164        }
165        
166        @StringGetter( ACTION_ON_CONFLICTING_DIRECTION_VIOLATION )
167        public ActionOnSignalSpecsViolation getActionOnConflictingDirectionViolation() {
168                return actionOnConflictingDirectionViolation;
169        }
170
171        @StringSetter( ACTION_ON_CONFLICTING_DIRECTION_VIOLATION )
172        public void setActionOnConflictingDirectionViolation(ActionOnSignalSpecsViolation actionOnConflictingDirectionViolation) {
173                switch (actionOnConflictingDirectionViolation){
174                // set the value for the supported actions
175                case WARN:
176                case EXCEPTION:
177                        this.actionOnConflictingDirectionViolation = actionOnConflictingDirectionViolation;
178                        break;
179                // throw an exception if the value is not supported
180                default:
181                        throw new IllegalArgumentException("The value " + actionOnConflictingDirectionViolation 
182                                        + " for key : " + ACTION_ON_CONFLICTING_DIRECTION_VIOLATION + " is not supported by this config group");
183                }
184        }
185        
186        @StringGetter( USE_AMBER_TIMES )
187        public boolean isUseAmbertimes() {
188                return useAmbertimes;
189        }
190        
191        @StringSetter( USE_AMBER_TIMES )
192        public void setUseAmbertimes(boolean useAmbertimes) {
193                this.useAmbertimes = useAmbertimes;
194        }
195
196        @StringGetter( USE_SIGNALSYSTEMS )
197        public boolean isUseSignalSystems() {
198                return this.useSignalSystems;
199        }
200
201        @StringSetter( USE_SIGNALSYSTEMS )
202        public void setUseSignalSystems(final boolean useSignalSystems) {
203                this.useSignalSystems = useSignalSystems;
204        }
205        
206        @StringGetter( USE_CONFLICTING_DIRECTIONS )
207        public boolean isUseConflictingDirections() {
208                return this.useConflictingDirections;
209        }
210        
211        @StringSetter( USE_CONFLICTING_DIRECTIONS )
212        public void setUseConflictingDirections(boolean useConflictingDirections){
213                this.useConflictingDirections = useConflictingDirections;
214        }
215}
