020package org.matsim.contrib.signals.model;
021
022import java.util.ArrayList;
023import java.util.HashMap;
024import java.util.List;
025import java.util.Map;
026
027import org.matsim.api.core.v01.Id;
028import org.matsim.contrib.signals.data.signalgroups.v20.SignalGroupSettingsData;
029import org.matsim.contrib.signals.data.signalgroups.v20.SignalPlanData;
030
031
032/**
033 * @author dgrether
034 *
035 */
036public class DatabasedSignalPlan implements SignalPlan {
037//      private static final Logger log = Logger.getLogger(DatabasedSignalPlan.class);
038        
039        private SignalPlanData data;
040        private int cycle;
041        
042        private Map<Integer, List<Id<SignalGroup>>> secondInPlanOnsetsMap = new HashMap<>();
043
044        private Map<Integer, List<Id<SignalGroup>>> secondInPlanDroppingsMap = new HashMap<>();
045        
046        public DatabasedSignalPlan(SignalPlanData planData) {
047                this.data = planData;
048                this.init();
049        }
050        
051        private void init(){
052                if (this.data.getCycleTime() != null){
053                        this.cycle = data.getCycleTime();
054                }
055                else {
056                        throw new IllegalStateException("This implementation of SignalPlan works only with a cycle time");
057                }
058        
059                int offset = 0;
060                if (this.data.getOffset() != null){
061                        offset = this.data.getOffset();
062                }
063                
064                for (SignalGroupSettingsData sgdata : this.data.getSignalGroupSettingsDataByGroupId().values()){
065//                      log.error("  SignalGroup " +  sgdata.getSignalGroupId());
066                        //do nothing if onset == dropping or all time green is set
067                        if (! ( (sgdata.getOnset() == sgdata.getDropping()) || 
068                                        (sgdata.getOnset() % this.cycle == 0 && sgdata.getDropping() % this.cycle == 0))){
069                                int onset = sgdata.getOnset();
070                                onset = getPositiveModuloByCycleTime(offset+onset);
071                                //onsets
072                                List<Id<SignalGroup>> onsetsSgIds = this.secondInPlanOnsetsMap.get(onset);
073                                if (onsetsSgIds == null){
074                                        onsetsSgIds = new ArrayList<>();
075                                        this.secondInPlanOnsetsMap.put(onset, onsetsSgIds);
076                                }
077                                onsetsSgIds.add(sgdata.getSignalGroupId());
078                                //dropping
079                                int dropping = sgdata.getDropping();
080                                dropping = getPositiveModuloByCycleTime(offset+dropping);
081                                List<Id<SignalGroup>> droppingSgIds = this.secondInPlanDroppingsMap.get(dropping);
082                                if (droppingSgIds == null){
083                                        droppingSgIds = new ArrayList<>();
084                                        this.secondInPlanDroppingsMap.put(dropping, droppingSgIds);
085                                }
086                                droppingSgIds.add(sgdata.getSignalGroupId());
087                        }
088                }
089        }
090
091        private int getPositiveModuloByCycleTime(int dividend) {
092                int modulo = dividend % this.cycle;
093                if (modulo < 0) modulo += this.cycle;
094                return modulo;
095        }
096        
097
098        @Override
099        public List<Id<SignalGroup>> getDroppings(double timeSeconds) {
100                Integer currentSecondInPlan = ((int) (timeSeconds % this.cycle));
101                return this.secondInPlanDroppingsMap.get(currentSecondInPlan);
102        }
103
104        @Override
105        public List<Id<SignalGroup>> getOnsets(double timeSeconds) {
106                Integer currentSecondInPlan = ((int) (timeSeconds  % this.cycle));
107                return this.secondInPlanOnsetsMap.get(currentSecondInPlan);
108        }
109
110        
111        @Override
112        public double getEndTime() {
113                return this.data.getEndTime();
114        }
115
116        @Override
117        public double getStartTime() {
118                return this.data.getStartTime();
119        }
120
121        @Override
122        public Id<SignalPlan> getId() {
123                return this.data.getId();
124        }
125
126        @Override
127        public Integer getOffset() {
128                return this.data.getOffset();
129        }
130
131        @Override
132        public Integer getCycleTime() {
133                return this.data.getCycleTime();
134        }
135
136        public SignalPlanData getPlanData(){
137                return this.data;
138        }
139        
140}
