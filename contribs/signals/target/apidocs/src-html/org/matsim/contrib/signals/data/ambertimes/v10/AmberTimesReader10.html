020package org.matsim.contrib.signals.data.ambertimes.v10;
021
022import java.io.IOException;
023import java.io.InputStream;
024
025import javax.xml.XMLConstants;
026import javax.xml.bind.JAXBContext;
027import javax.xml.bind.JAXBException;
028import javax.xml.bind.Unmarshaller;
029import javax.xml.validation.SchemaFactory;
030
031import org.apache.log4j.Logger;
032import org.matsim.api.core.v01.Id;
033import org.matsim.contrib.signals.model.Signal;
034import org.matsim.contrib.signals.model.SignalSystem;
035import org.matsim.core.api.internal.MatsimReader;
036import org.matsim.core.utils.io.IOUtils;
037import org.matsim.core.utils.io.UncheckedIOException;
038import org.matsim.jaxb.amberTimes10.XMLAmberTimes;
039import org.matsim.jaxb.amberTimes10.XMLAmberTimes.XMLSignalSystem;
040import org.matsim.jaxb.amberTimes10.XMLAmberTimes.XMLSignalSystem.XMLSignal;
041import org.xml.sax.SAXException;
042
043/**
044 * @author jbischoff
045 * @author dgrether
046 */
047public class AmberTimesReader10 implements MatsimReader {
048
049        private static final Logger log = Logger.getLogger(AmberTimesReader10.class);
050        private AmberTimesData amberTimesData;
051
052        public AmberTimesReader10(AmberTimesData amberTimesData) {
053                this.amberTimesData = amberTimesData;
054        }
055
056
057        @Override
058        public void readFile(final String filename) {
059                // create jaxb infrastructure
060                JAXBContext jc;
061                XMLAmberTimes xmlatdefs;
062                try (InputStream stream = IOUtils.getInputStream(filename)) {
063                        jc = JAXBContext.newInstance(org.matsim.jaxb.amberTimes10.ObjectFactory.class);
064                        Unmarshaller u = jc.createUnmarshaller();
065                        // validate XML file
066                        log.info("starting to validate " + filename);
067                        u.setSchema(SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI).newSchema(getClass().getResource("/dtd/amberTimes_v1.0.xsd")));
068                        log.info("starting unmarshalling " + filename);
069                        xmlatdefs = (XMLAmberTimes) u.unmarshal(stream);
070                } catch (JAXBException | SAXException | IOException e) {
071                        throw new UncheckedIOException(e);
072                }
073
074                // convert from Jaxb types to MATSim-API conform types
075
076                // Global Defaults
077                if (xmlatdefs.getGlobalDefaults() != null){
078                        if (xmlatdefs.getGlobalDefaults().getAmber() != null){
079                                amberTimesData
080                                .setDefaultAmber(xmlatdefs.getGlobalDefaults().getAmber().getSeconds().intValue());
081                        }
082                        if (xmlatdefs.getGlobalDefaults().getAmberTimeGreen() != null){
083                                amberTimesData.setDefaultAmberTimeGreen(xmlatdefs.getGlobalDefaults().getAmberTimeGreen()
084                                                .getProportion().doubleValue());
085                        }
086                        if (xmlatdefs.getGlobalDefaults().getRedAmber() != null){
087                                amberTimesData.setDefaultRedAmber(xmlatdefs.getGlobalDefaults().getRedAmber().getSeconds()
088                                                .intValue());
089                        }
090                }
091
092                for (XMLSignalSystem xmlss : xmlatdefs.getSignalSystem()) {
093                        Id<SignalSystem> ssid = Id.create(xmlss.getRefId(), SignalSystem.class);
094                        AmberTimeData atdata = new AmberTimeDataImpl(ssid);
095
096                        // Signalsystem Defaults
097                        if (xmlss.getSystemDefaults() != null){
098                                if (xmlss.getSystemDefaults().getAmber() != null){
099                                        atdata.setDefaultAmber(xmlss.getSystemDefaults().getAmber().getSeconds().intValue());
100                                }
101                                if (xmlss.getSystemDefaults().getRedAmber() != null){
102                                        atdata.setDefaultRedAmber(xmlss.getSystemDefaults().getRedAmber().getSeconds().intValue());
103                                }
104                        }
105
106                        for (XMLSignal xmls : xmlss.getSignal()) {
107
108                                Id<Signal> sid = Id.create(xmls.getRefId(), Signal.class);
109                                atdata.setAmberTimeOfSignal(sid, xmls.getAmber().getSeconds().intValue());
110                                atdata.setRedAmberTimeOfSignal(sid, xmls.getRedAmber().getSeconds().intValue());
111
112                        }
113
114                        amberTimesData.addAmberTimeData(atdata);
115                }
116
117        }
118
119}
