020package org.matsim.contrib.signals;
021
022import org.apache.log4j.Logger;
023import org.matsim.api.core.v01.Id;
024import org.matsim.api.core.v01.TransportMode;
025import org.matsim.api.core.v01.network.Link;
026import org.matsim.contrib.signals.data.signalgroups.v20.SignalData;
027import org.matsim.contrib.signals.data.signalsystems.v20.SignalSystemData;
028import org.matsim.contrib.signals.data.signalsystems.v20.SignalSystemsData;
029import org.matsim.core.network.algorithms.NetworkExpandNode.TurnInfo;
030
031import java.util.*;
032
033
034/**
035 * Creates TurnInfo objects from traffic signal data.
036 * 
037 * @author dgrether
038 *
039 */
040public class SignalsTurnInfoBuilder {
041        
042        private static final Logger log = Logger.getLogger(SignalsTurnInfoBuilder.class);
043
044        private static int warnCount = 0;
045
046        
047        public static Map<Id<Link>, List<TurnInfo>> createSignalsTurnInfos(SignalSystemsData ssd) {
048                Map<Id<Link>, List<TurnInfo>> inLinkIdTurnInfoMap = new HashMap<Id<Link>, List<TurnInfo>>();
049                for (SignalSystemData signalSystem : ssd.getSignalSystemData().values()){
050                        for (SignalData signal : signalSystem.getSignalData().values()){
051                                if (signal.getTurningMoveRestrictions() != null && ! signal.getTurningMoveRestrictions().isEmpty()){
052                                        if (warnCount < 1){
053                                                log.warn("Turning move restrictions for signals are implemented for TransportMode.car only, yet!");
054                                                warnCount++;
055                                        }
056                                        if (!inLinkIdTurnInfoMap.containsKey(signal.getLinkId())){
057                                                inLinkIdTurnInfoMap.put(signal.getLinkId(), new ArrayList<TurnInfo>());
058                                        }
059                                        Set<String> modeCar = new HashSet<String>();
060                                        modeCar.add(TransportMode.car);
061                                        for (Id<Link> toLinkId : signal.getTurningMoveRestrictions()){
062                                                TurnInfo ti = new TurnInfo(signal.getLinkId(), toLinkId, modeCar);
063                                                inLinkIdTurnInfoMap.get(signal.getLinkId()).add(ti);
064                                        }
065                                }
066                        }
067                }
068                return inLinkIdTurnInfoMap;
069        }
070}
