054/**
055 * @author dgrether
056 *
057 */
058public class FromDataBuilder implements SignalSystemsModelBuilder{
059
060        private SignalsData signalsData;
061        private SignalModelFactory factory;
062        private EventsManager events;
063        private Scenario scenario;
064
065        @Inject
066        private FromDataBuilder(Scenario scenario, SignalModelFactory factory, EventsManager events){
067                this.signalsData = (SignalsData) scenario.getScenarioElement(SignalsData.ELEMENT_NAME);
068                this.scenario = scenario;
069                this.factory = factory;
070                this.events = events;
071        }
072        
073        private void createAndAddSignals(SignalSystem system){
074                SignalSystemData ssData = signalsData.getSignalSystemsData().getSignalSystemData().get(system.getId());
075                for (SignalData signalData : ssData.getSignalData().values()){
076                        Signal signal = new DatabasedSignal(signalData);
077                        system.addSignal(signal);
078                }
079        }
080        
081        private void createAndAddSignalSystemsFromData(SignalSystemsManager manager){
082                //process information of SignalSystemsData object
083                for (SignalSystemData ssData : this.signalsData.getSignalSystemsData().getSignalSystemData().values()){
084                        SignalSystem system = this.factory.createSignalSystem(ssData.getId());
085                        manager.addSignalSystem(system);
086                        system.setSignalSystemsManager(manager);
087                }
088        }
089        
090        private void createAndAddSignalGroupsFromData(SignalSystem system){
091                //process information of  SignalGroupsData object and create the signal groups
092                Map<Id<SignalGroup>, SignalGroupData> signalGroupDataMap = this.signalsData.getSignalGroupsData().getSignalGroupDataBySystemId(system.getId());
093                for (SignalGroupData signalGroupData : signalGroupDataMap.values()){
094                        SignalGroup group = new SignalGroupImpl(signalGroupData.getId());
095                        for (Id<Signal> signalId : signalGroupData.getSignalIds()){
096                                Signal signal = system.getSignals().get(signalId);
097                                group.addSignal(signal);
098                        }
099                        system.addSignalGroup(group);
100                }
101        }
102        
103        private void createAndAddSignalSystemControllerFromData(SignalSystem system){
104                //process information of SignalControlData
105                SignalSystemControllerData systemControlData = signalsData.getSignalControlData().getSignalSystemControllerDataBySystemId().get(system.getId());
106                SignalController controller = this.factory.createSignalSystemController(systemControlData.getControllerIdentifier(), system);
107//              controller.setSignalSystem(system);
108                system.setSignalSystemController(controller);
109                if (systemControlData.getSignalPlanData() != null) { 
110                        for (SignalPlanData planData : systemControlData.getSignalPlanData().values()){
111                                SignalPlan plan = this.factory.createSignalPlan(planData);
112                                controller.addPlan(plan);
113                        }
114                }
115        }
116        
117        private void createAndAddAmberLogic(SignalSystemsManager manager){
118                //process information of AmberTimesData object
119                if (ConfigUtils.addOrGetModule(this.scenario.getConfig(), SignalSystemsConfigGroup.GROUPNAME, SignalSystemsConfigGroup.class).isUseAmbertimes()){
120                        AmberLogic amberLogic = new AmberLogicImpl(this.signalsData.getAmberTimesData());
121                        manager.setAmberLogic(amberLogic);
122                }
123        }
124        
125        private void createAndAddIntergreenTimesLogic(SignalSystemsManager manager){
126                if (ConfigUtils.addOrGetModule(this.scenario.getConfig(), SignalSystemsConfigGroup.GROUPNAME, SignalSystemsConfigGroup.class).isUseIntergreenTimes()){
127                        IntergreensLogic intergreensLogic = new IntergreensLogicImpl(this.signalsData.getIntergreenTimesData(), ConfigUtils.addOrGetModule(this.scenario.getConfig(), SignalSystemsConfigGroup.GROUPNAME, SignalSystemsConfigGroup.class));
128                        this.events.addHandler(intergreensLogic);
129                }
130        }
131        
132        private void createAndAddConflictingDirectionsLogic(SignalSystemsManager manager) {
133                if (ConfigUtils.addOrGetModule(this.scenario.getConfig(), SignalSystemsConfigGroup.GROUPNAME, SignalSystemsConfigGroup.class).isUseConflictingDirections()){
134                        ConflictingDirectionsLogic conflictLogic = new ConflictingDirectionsLogicImpl(this.scenario.getNetwork(), this.scenario.getLanes(), this.signalsData, 
135                                        ConfigUtils.addOrGetModule(this.scenario.getConfig(), SignalSystemsConfigGroup.GROUPNAME, SignalSystemsConfigGroup.class).getActionOnConflictingDirectionViolation());
136                        this.events.addHandler(conflictLogic);
137                }
138        }
139
140        @Override
141        public SignalSystemsManager createAndInitializeSignalSystemsManager() {
142                //1.) SignalSystemsManager
143                SignalSystemsManager manager = new SignalSystemsManagerImpl(signalsData, events);
144                //2.) SignalSystems
145                this.createAndAddSignalSystemsFromData(manager);
146                //3.) Signals then SignalGroups then SignalController
147                for (SignalSystem system : manager.getSignalSystems().values()){
148                        this.createAndAddSignals(system);
149                        this.createAndAddSignalGroupsFromData(system);
150                        this.createAndAddSignalSystemControllerFromData(system);
151                }
152                //4.) AmberLogic
153                this.createAndAddAmberLogic(manager);
154                //5.) IntergreenTimesLogic 
155                this.createAndAddIntergreenTimesLogic(manager);
156                //6.) ConflictingDirectionsLogic
157                this.createAndAddConflictingDirectionsLogic(manager);
158                return manager;
159        }
160}
