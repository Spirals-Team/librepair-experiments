020package org.matsim.contrib.signals.otfvis;
021
022import java.io.IOException;
023import java.nio.ByteBuffer;
024import java.util.List;
025
026import org.matsim.core.config.Config;
027import org.matsim.core.utils.misc.ByteBufferUtils;
028import org.matsim.lanes.data.ModelLane;
029import org.matsim.lanes.data.Lanes;
030import org.matsim.lanes.data.LanesToLinkAssignment;
031import org.matsim.lanes.data.LanesUtils;
032import org.matsim.lanes.vis.VisLaneModelBuilder;
033import org.matsim.lanes.vis.VisLinkWLanes;
034import org.matsim.vis.otfvis.data.OTFDataWriter;
035import org.matsim.vis.otfvis.data.OTFServerQuadTree;
036import org.matsim.vis.snapshotwriters.VisLink;
037import org.matsim.vis.snapshotwriters.VisNetwork;
038
039
040/**
041 * @author dgrether
042 *
043 */
044public class OTFLaneWriter extends OTFDataWriter<Void> {
045
046        private final transient VisNetwork network;
047
048        private final transient Lanes lanes;
049        
050        private transient VisLaneModelBuilder laneModelBuilder = new VisLaneModelBuilder();
051
052        private Config config;
053        
054        public OTFLaneWriter(VisNetwork visNetwork, Lanes laneDefinitions, Config config){
055                this.network = visNetwork;
056                this.lanes = laneDefinitions;
057                this.config = config;
058        }
059        
060        @Override
061        public void writeConstData(ByteBuffer out) throws IOException {
062                //write the data for the links
063                out.putInt(this.network.getVisLinks().size());
064                for (VisLink visLink : this.network.getVisLinks().values()) {
065                        LanesToLinkAssignment l2l = null;
066                        if (this.lanes != null){
067                                l2l = this.lanes.getLanesToLinkAssignments().get(visLink.getLink().getId());
068                        }
069                        List<ModelLane> la = null;
070                        if (l2l != null) {
071                                la = LanesUtils.createLanes(visLink.getLink(), l2l);
072                        }
073                        VisLinkWLanes otfLink = this.laneModelBuilder.createVisLinkLanes(OTFServerQuadTree.getOTFTransformation(), visLink, config.qsim().getNodeOffset(), la);
074                        //write link data
075                        ByteBufferUtils.putObject(out, otfLink);
076                }
077        }
078        
079        @Override
080        public void writeDynData(ByteBuffer out) throws IOException {
081                //nothing to do as lanes are not dynamic
082        }
083
084
085}
