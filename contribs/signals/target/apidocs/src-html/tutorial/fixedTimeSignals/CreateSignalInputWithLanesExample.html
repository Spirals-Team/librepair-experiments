060/**
061 * Example for how to create signal input files for a scenario with lanes from code.
062 * 
063 * @link VisualizeSignalScenarioWithLanes for how to visualize this scenario.
064 * 
065 * @author dgrether
066 * @author tthunig
067 */
068public class CreateSignalInputWithLanesExample {
069
070        private static final Logger log = Logger.getLogger(CreateSignalInputWithLanesExample.class);
071        private static final String INPUT_DIR = "./examples/tutorial/example90TrafficLights/createSignalInput/";
072        
073        private static final int ONSET1 = 0;
074        private static final int DROPPING1 = 55;
075        private static final int ONSET2 = 60;
076        private static final int DROPPING2 = 115;
077        private static final int CYCLE = 120;
078        private static final double LANE_LENGTH = 150.0;
079        private static final int LANE_CAPACITY = 1800;
080        private static final int NO_LANES = 1; // number of represented lanes per lane
081        private static final double LINK_LENGTH = 300;
082
083        private void createGroupsAndSystem2(Scenario scenario, SignalSystemsData systems, SignalGroupsData groups) {
084                SignalSystemData sys = systems.getFactory().createSignalSystemData(Id.create("2", SignalSystem.class));
085                systems.addSignalSystemData(sys);
086                SignalSystemsDataFactory factory = systems.getFactory();
087                
088                SignalUtils.createAndAddSignal(sys, factory, Id.create("1", Signal.class), Id.createLinkId("12"),
089                                Arrays.asList(Id.create("12.l", Lane.class)));
090                SignalUtils.createAndAddSignal(sys, factory, Id.create("2", Signal.class), Id.createLinkId("12"),
091                                Arrays.asList(Id.create("12.r", Lane.class)));
092                SignalUtils.createAndAddSignal(sys, factory, Id.create("3", Signal.class), Id.createLinkId("32"), null);
093                SignalUtils.createAndAddSignal(sys, factory, Id.create("4", Signal.class), Id.createLinkId("72"), null);
094                
095                // create a signal group for every signal
096                SignalUtils.createAndAddSignalGroups4Signals(groups, sys);
097        }
098
099        private void createGroupsAndSystem5(Scenario scenario, SignalSystemsData systems, SignalGroupsData groups) {
100                SignalSystemData sys = systems.getFactory().createSignalSystemData(Id.create("5", SignalSystem.class));
101                systems.addSignalSystemData(sys);
102                SignalSystemsDataFactory factory = systems.getFactory();
103
104                SignalUtils.createAndAddSignal(sys, factory, Id.create("1", Signal.class), Id.createLinkId("65"),
105                                Arrays.asList(Id.create("65.r", Lane.class)));
106                SignalUtils.createAndAddSignal(sys, factory, Id.create("2", Signal.class), Id.createLinkId("65"),
107                                Arrays.asList(Id.create("65.l", Lane.class)));
108                SignalUtils.createAndAddSignal(sys, factory, Id.create("3", Signal.class), Id.createLinkId("45"), null);
109                SignalUtils.createAndAddSignal(sys, factory, Id.create("4", Signal.class), Id.createLinkId("85"), null);
110
111                // create a signal group for every signal
112                SignalUtils.createAndAddSignalGroups4Signals(groups, sys);
113        }
114        
115        private void createSystemControl(SignalControlData control, Id<SignalSystem> signalSystemId, 
116                        int onset1, int dropping1, int onset2, int dropping2) {
117                SignalControlDataFactory fac = control.getFactory();
118                
119                // create and add signal control for the given system id
120                SignalSystemControllerData controller = 
121                                fac.createSignalSystemControllerData(signalSystemId);
122                control.addSignalSystemControllerData(controller);
123                controller.setControllerIdentifier(DefaultPlanbasedSignalSystemController.IDENTIFIER);
124                
125                // create and add signal plan with defined cycle time and offset 0              
126                SignalPlanData plan = SignalUtils.createSignalPlan(fac, CYCLE, 0);
127                controller.addSignalPlanData(plan);
128                
129                // create and add control settings for signal groups
130                plan.addSignalGroupSettings(SignalUtils.createSetting4SignalGroup(fac,
131                                Id.create("1", SignalGroup.class), onset1, dropping1));
132                plan.addSignalGroupSettings(SignalUtils.createSetting4SignalGroup(fac,
133                                Id.create("2", SignalGroup.class), onset1, dropping1));
134                plan.addSignalGroupSettings(SignalUtils.createSetting4SignalGroup(fac,
135                                Id.create("3", SignalGroup.class), onset2, dropping2));
136                plan.addSignalGroupSettings(SignalUtils.createSetting4SignalGroup(fac,
137                                Id.create("4", SignalGroup.class), onset2, dropping2));
138        }
139
140        private void createLanes(Lanes lanes) {
141                LanesFactory factory = lanes.getFactory();
142                
143                // create lanes for link 12
144                LanesToLinkAssignment lanesForLink12 = factory
145                                .createLanesToLinkAssignment(Id.createLinkId("12"));
146                lanes.addLanesToLinkAssignment(lanesForLink12);
147                
148                // original lane, i.e. lane that starts at the link from node and leads to all other lanes of the link
149                LanesUtils.createAndAddLane(lanesForLink12, factory, 
150                                Id.create("12.ol", Lane.class), LANE_CAPACITY, LINK_LENGTH, 0, NO_LANES, 
151                                null, Arrays.asList(Id.create("12.l", Lane.class), Id.create("12.r", Lane.class)));
152                
153                // left turning lane (alignment 1)
154                LanesUtils.createAndAddLane(lanesForLink12, factory,
155                                Id.create("12.l", Lane.class), LANE_CAPACITY, LANE_LENGTH, 1, NO_LANES,
156                                Collections.singletonList(Id.create("23", Link.class)), null);
157
158                // right turning lane (alignment -1)
159                LanesUtils.createAndAddLane(lanesForLink12, factory,
160                                Id.create("12.r", Lane.class), LANE_CAPACITY, LANE_LENGTH, -1, NO_LANES,
161                                Collections.singletonList(Id.create("27", Link.class)), null);
162
163                // create lanes for link 65
164                LanesToLinkAssignment lanesForLink65 = factory
165                                .createLanesToLinkAssignment(Id.create("65", Link.class));
166                lanes.addLanesToLinkAssignment(lanesForLink65);
167
168                // original lane, i.e. lane that starts at the link from node and leads to all other lanes of the link
169                LanesUtils.createAndAddLane(lanesForLink65, factory, 
170                                Id.create("65.ol", Lane.class), LANE_CAPACITY, LINK_LENGTH, 0, NO_LANES, null,
171                                Arrays.asList(Id.create("65.l", Lane.class), Id.create("65.r", Lane.class)));           
172                
173                // right turning lane (alignment -1)
174                LanesUtils.createAndAddLane(lanesForLink65, factory,
175                                Id.create("65.r", Lane.class), LANE_CAPACITY, LANE_LENGTH, -1, NO_LANES,
176                                Collections.singletonList(Id.create("54", Link.class)), null);
177
178                // left turning lane (alignment 1)
179                LanesUtils.createAndAddLane(lanesForLink65, factory,
180                                Id.create("65.l", Lane.class), LANE_CAPACITY, LANE_LENGTH, 1, NO_LANES,
181                                Collections.singletonList(Id.create("58", Link.class)), null);
182        }

184        public void run(String outputDir) throws IOException {          
185                // create an empty config
186                Config config = ConfigUtils.createConfig();
187
188                // set network and population files
189                config.network().setInputFile(INPUT_DIR + "network.xml.gz");
190                config.plans().setInputFile(INPUT_DIR + "population.xml.gz");
191                
192                // enable lanes
193                config.qsim().setUseLanes(true);
194                
195                // add the signal config group to the config file
196                SignalSystemsConfigGroup signalSystemsConfigGroup = 
197                                ConfigUtils.addOrGetModule(config, SignalSystemsConfigGroup.GROUPNAME, SignalSystemsConfigGroup.class);
198                
199                /* the following makes the contrib load the signal input files, but not to do anything with them
200                 * (this switch will eventually go away) */
201                signalSystemsConfigGroup.setUseSignalSystems(true);
202                
203                // specify some details for the visualization
204                config.qsim().setNodeOffset(20.0);
205                config.qsim().setSnapshotStyle(SnapshotStyle.queue);
206                
207                // --- create the scenario
208                Scenario scenario = ScenarioUtils.loadScenario(config);
209                /* create the information about signals data (i.e. create an empty SignalsData object)
210                 * and add it to the scenario as scenario element */
211                SignalsData signalsData = SignalUtils.createSignalsData(signalSystemsConfigGroup);
212                scenario.addScenarioElement(SignalsData.ELEMENT_NAME, signalsData);
213                
214                // create lanes for the scenario
215                this.createLanes(scenario.getLanes());
216
217                /* fill the SignalsData object with information:
218                 * signal systems - specify signalized intersections
219                 * signal groups - specify signals that always have the same signal control
220                 * signal control - specify cycle time, onset and dropping time, offset... for all signal groups */
221                this.createGroupsAndSystem2(scenario, signalsData.getSignalSystemsData(), signalsData.getSignalGroupsData());
222                this.createGroupsAndSystem5(scenario, signalsData.getSignalSystemsData(), signalsData.getSignalGroupsData());
223                this.createSystemControl(signalsData.getSignalControlData(), Id.create("2", SignalSystem.class), 
224                                ONSET1, DROPPING1, ONSET2, DROPPING2);
225                this.createSystemControl(signalsData.getSignalControlData(), Id.create("5", SignalSystem.class), 
226                                ONSET2, DROPPING2, ONSET1, DROPPING1);
227
228                // create the path to the output directory if it does not exist yet
229                Files.createDirectories(Paths.get(outputDir));
230                
231                // set output filenames
232                config.network().setLaneDefinitionsFile(outputDir + "lane_definitions_v2.0.xml");
233                signalSystemsConfigGroup.setSignalSystemFile(outputDir + "signal_systems.xml");
234                signalSystemsConfigGroup.setSignalGroupsFile(outputDir + "signal_groups.xml");
235                signalSystemsConfigGroup.setSignalControlFile(outputDir + "signal_control.xml");
236
237                //write config to file
238                String configFile = outputDir + "config.xml";
239                ConfigWriter configWriter = new ConfigWriter(config);
240                configWriter.write(configFile);
241                
242                // write lanes to file
243                LanesWriter writerDelegate = new LanesWriter(scenario.getLanes());
244                writerDelegate.write(config.network().getLaneDefinitionsFile());
245
246                // write signal information to file
247                SignalsScenarioWriter signalsWriter = new SignalsScenarioWriter();
248                signalsWriter.setSignalSystemsOutputFilename(signalSystemsConfigGroup.getSignalSystemFile());
249                signalsWriter.setSignalGroupsOutputFilename(signalSystemsConfigGroup.getSignalGroupsFile());
250                signalsWriter.setSignalControlOutputFilename(signalSystemsConfigGroup.getSignalControlFile());
251                signalsWriter.writeSignalsData(scenario);
252
253                log.info("Config of traffic light scenario with lanes is written to " + configFile);
254                log.info("Visualize scenario by calling VisTrafficSignalScenarioWithLanes in the same package.");
255        }
256
257        public static void main(String[] args) throws IOException {
258                new CreateSignalInputWithLanesExample().run("output/example90TrafficLights/");
259        }
