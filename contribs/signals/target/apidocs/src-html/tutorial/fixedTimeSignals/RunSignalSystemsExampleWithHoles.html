028import org.matsim.core.config.Config;

030import org.matsim.core.config.groups.QSimConfigGroup.SnapshotStyle;
031import org.matsim.core.config.groups.QSimConfigGroup.TrafficDynamics;
032import org.matsim.core.controler.Controler;
033import org.matsim.core.controler.OutputDirectoryHierarchy.OverwriteFileSetting;
034import org.matsim.core.scenario.ScenarioUtils;
035import org.matsim.core.utils.io.IOUtils;
036import org.matsim.examples.ExamplesUtils;
037import org.matsim.vis.otfvis.OTFVisConfigGroup;
038import org.matsim.vis.otfvis.OTFVisConfigGroup.ColoringScheme;

040
041/**
042 * Configures and runs MATSim with traffic signals from input files
043 * and visualizes it with OTFVis. 
044 * 
045 * @author dgrether, tthunig
046 */
047public class RunSignalSystemsExampleWithHoles {
048        
049        public static void main(String[] args) {
050                run(true);
051        }
052
053        static void run(boolean useOTFVis) {
054                // load a config (without signal information)
055                Config config = ConfigUtils.loadConfig(IOUtils.newUrl(ExamplesUtils.getTestScenarioURL("equil-extended"), "config.xml"));
056                config.controler().setOverwriteFileSetting( OverwriteFileSetting.deleteDirectoryIfExists);
057                
058                // use higher values if you want to iterate
059                config.controler().setLastIteration(0); 
060                
061                config.network().setInputFile("network.xml");
062                config.plans().setInputFile("plans2000.xml.gz");
063                
064                // simulate traffic dynamics with holes (default would be without)
065                config.qsim().setTrafficDynamics(TrafficDynamics.withHoles);
066                config.qsim().setSnapshotStyle(SnapshotStyle.withHoles);
067                config.qsim().setNodeOffset(5.);
068        config.qsim().setUsingFastCapacityUpdate(false);
069                
070                // add the signal config group to the config file
071                SignalSystemsConfigGroup signalConfig = 
072                                ConfigUtils.addOrGetModule(config, SignalSystemsConfigGroup.GROUPNAME, SignalSystemsConfigGroup.class ) ;
073                
074                // the following makes the contrib load the signal input files, but not to do anything with them
075                // (this switch will eventually go away)
076                signalConfig.setUseSignalSystems(true);
077
078                // set the paths to the signal systems definition files
079                signalConfig.setSignalSystemFile("signalSystems_v2.0.xml");
080                signalConfig.setSignalGroupsFile("signalGroups_v2.0.xml");
081                signalConfig.setSignalControlFile("signalControl_v2.0.xml");
082                
083//              // here is how to also use intergreen and amber times:
084//              signalConfig.setUseIntergreenTimes(true);
085//              signalConfig.setIntergreenTimesFile(intergreenTimesFile);
086//              // set a suitable action for the case that intergreens are violated
087//              signalConfig.setActionOnIntergreenViolation(SignalSystemsConfigGroup.ActionOnIntergreenViolation.WARN);
088//              signalConfig.setUseAmbertimes(true);
089//              signalConfig.setAmberTimesFile(amberTimesFile);
090                
091//              // here is how to switch on link to link travel times if lanes are used:
092//              config.travelTimeCalculator().setCalculateLinkToLinkTravelTimes(true);
093//              config.controler().setLinkToLinkRoutingEnabled(true);
094                
095                if (useOTFVis) {
096                        // add the OTFVis config group
097                        OTFVisConfigGroup otfvisConfig = 
098                                        ConfigUtils.addOrGetModule(config, OTFVisConfigGroup.GROUP_NAME, OTFVisConfigGroup.class);
099                        // make links visible beyond screen edge
100                        otfvisConfig.setScaleQuadTreeRect(true); 
101                        otfvisConfig.setColoringScheme(ColoringScheme.byId);
102                        otfvisConfig.setAgentSize(240);
103                }
104
105                // --- create the scenario
106                Scenario scenario = ScenarioUtils.loadScenario( config ) ;
107                /* load the information about signals data (i.e. fill the SignalsData object)
108                 * and add it to the scenario as scenario element */
109                scenario.addScenarioElement(SignalsData.ELEMENT_NAME, new SignalsDataLoader(config).loadSignalsData());
110                
111                // --- create the controler
112                Controler c = new Controler( scenario );
113
114                /* add the signals module to the simulation
115                 * such that SignalsData is not only contained in the scenario 
116                 * but also used in the simulation */
117                c.addOverridingModule(new SignalsModule());
118                
119                /* add the visualization module to the simulation
120                 * such that it is used */
121                if ( useOTFVis ) {
122                        c.addOverridingModule( new OTFVisWithSignalsLiveModule() );
123                }
124                
125                // run the simulation
126                c.run();
127        }
128}
