055/**
056 * Example for how to create signal input files from code.
057 * 
058 * See also the picture of the scenario in the same package as the input files
059 * (examples/tutorial/example90TrafficLights/).
060 * 
061 * @link VisualizeSignalScenario for how to visualize this scenario.
062 * 
063 * @author dgrether
064 */
065public class CreateSignalInputExample {
066
067        private static final Logger log = Logger.getLogger(CreateSignalInputExample.class);
068        private static final String INPUT_DIR = "./examples/tutorial/example90TrafficLights/createSignalInput/";
069        
070        /**
071         * This method creates the locations of signals, i.e. it specifies signalized intersections.
072         * Furthermore groups for the signals are created that specify which signals will always have the same control.
073         * 
074         * @param systems the so far empty object for information about signalized intersections
075         * @param groups the so far empty object for information about signals that are controlled together as groups
076         */
077        private void createSignalSystemsAndGroups(SignalSystemsData systems, SignalGroupsData groups){          
078                // create signal system 3 (at node 3)
079                SignalSystemData sys = systems.getFactory().createSignalSystemData(Id.create("3", SignalSystem.class));
080                // add signal system 3 to the overall signal systems container
081                systems.addSignalSystemData(sys);
082                // create signal 1
083                SignalData signal = systems.getFactory().createSignalData(Id.create("1", Signal.class));
084                // add signal 1 to signal system 3, such that it belongs to node 3
085                sys.addSignalData(signal);
086                // specify the link at which signal 1 is located
087                signal.setLinkId(Id.create("23", Link.class));
088                // create signal 2
089                signal = systems.getFactory().createSignalData(Id.create("2", Signal.class));
090                // add signal 2 to signal system 3, such that it also belongs to node 3
091                sys.addSignalData(signal);
092                // specify the link at which signal 2 is located
093                signal.setLinkId(Id.create("43", Link.class));
094                // create a single signal group for each signal of system 3, i.e. for signal 1 and 2
095                SignalUtils.createAndAddSignalGroups4Signals(groups, sys);
096                
097                // do the same with signal system 4 (node 4)
098                sys = systems.getFactory().createSignalSystemData(Id.create("4", SignalSystem.class));
099                systems.addSignalSystemData(sys);
100                signal = systems.getFactory().createSignalData(Id.create("1", Signal.class));
101                sys.addSignalData(signal);
102                signal.setLinkId(Id.create("34", Link.class));
103                signal = systems.getFactory().createSignalData(Id.create("2", Signal.class));
104                sys.addSignalData(signal);
105                signal.setLinkId(Id.create("54", Link.class));
106                SignalUtils.createAndAddSignalGroups4Signals(groups, sys);
107
108                // signal system 7
109                sys = systems.getFactory().createSignalSystemData(Id.create("7", SignalSystem.class));
110                systems.addSignalSystemData(sys);
111                signal = systems.getFactory().createSignalData(Id.create("1", Signal.class));
112                sys.addSignalData(signal);
113                signal.setLinkId(Id.create("27", Link.class));
114                signal = systems.getFactory().createSignalData(Id.create("2", Signal.class));
115                sys.addSignalData(signal);
116                signal.setLinkId(Id.create("87", Link.class));
117                SignalUtils.createAndAddSignalGroups4Signals(groups, sys);
118                
119                // signal system 8
120                sys = systems.getFactory().createSignalSystemData(Id.create("8", SignalSystem.class));
121                systems.addSignalSystemData(sys);
122                signal = systems.getFactory().createSignalData(Id.create("1", Signal.class));
123                sys.addSignalData(signal);
124                signal.setLinkId(Id.create("78", Link.class));
125                signal = systems.getFactory().createSignalData(Id.create("2", Signal.class));
126                sys.addSignalData(signal);
127                signal.setLinkId(Id.create("58", Link.class));
128                SignalUtils.createAndAddSignalGroups4Signals(groups, sys);
129        }
130        
131        /**
132         * Create a fixed time traffic signal control for all signal groups in the scenario,
133         * i.e. specify when their signals show green or red.
134         * 
135         * Each signal system (signalized intersection) is equipped with a control,
136         * namely each with the same. The control contains the following information.
137         * - A cylce time of 120 seconds.
138         * - An offset (for green waves) of 0 seconds.
139         * - Each direction gets green for second 0 to 55 within the cycle.
140         * 
141         * @param control the so far empty object for information about when to show green and red
142         */
143        private void createSignalControl(SignalControlData control) {
144                // specify overall cycle time
145                int cycle = 120;
146                
147                // create signal control for systems 3, 4, 7 and 8
148                List<Id<SignalSystem>> ids = new LinkedList<Id<SignalSystem>>();
149                ids.add(Id.create("3", SignalSystem.class));
150                ids.add(Id.create("4", SignalSystem.class));
151                ids.add(Id.create("7", SignalSystem.class));
152                ids.add(Id.create("8", SignalSystem.class));
153                for (Id<SignalSystem> id : ids){
154                        // create a signal control for the system
155                        SignalSystemControllerData controller = control.getFactory().createSignalSystemControllerData(id);
156                        // add it to the overall signal control container
157                        control.addSignalSystemControllerData(controller);
158                        // declare the control as a fixed time control
159                        controller.setControllerIdentifier(DefaultPlanbasedSignalSystemController.IDENTIFIER);
160                        
161                        /* create a first signal plan for the system control (a signal system control (i.e. an intersection) 
162                         * can have different (non-overlapping) plans for different times of the day) */
163                        SignalPlanData plan1 = control.getFactory().createSignalPlanData(Id.create("1", SignalPlan.class));
164                        // add the (first) plan to the system control
165                        controller.addSignalPlanData(plan1);
166                        // fill the plan with information: cycle time, offset, signal settings
167                        plan1.setStartTime(0.0);
168                        /* note: use start and end time as 0.0 if you want to define a signal plan that is valid all day. */
169                        plan1.setEndTime(6.0*3600);
170                        plan1.setCycleTime(cycle);
171                        plan1.setOffset(0);
172                        SignalGroupSettingsData settings1 = control.getFactory().createSignalGroupSettingsData(Id.create("1", SignalGroup.class));
173                        plan1.addSignalGroupSettings(settings1);
174                        settings1.setOnset(0);
175                        settings1.setDropping(55);
176                        SignalGroupSettingsData settings2 = control.getFactory().createSignalGroupSettingsData(Id.create("2", SignalGroup.class));
177                        plan1.addSignalGroupSettings(settings2);
178                        settings2.setOnset(0);
179                        settings2.setDropping(55);
180                        
181                        // create another signal plan for the system control
182                        SignalPlanData plan2 = control.getFactory().createSignalPlanData(Id.create("2", SignalPlan.class));
183                        controller.addSignalPlanData(plan2);
184                        /* note: end and start time of the two signal plans may be equal but not overlapping */
185                        plan2.setStartTime(6.0*3600);
186                        /* note: signal plans of a group do not have to cover the hole day.
187                         * when no signal plan is defined for a time period, signals are switched off */
188                        plan2.setEndTime(18.0 * 3600);
189                        plan2.setCycleTime(cycle/2);
190                        plan2.setOffset(0);
191                        SignalGroupSettingsData settings1_2 = control.getFactory().createSignalGroupSettingsData(Id.create("1", SignalGroup.class));
192                        plan2.addSignalGroupSettings(settings1_2);
193                        settings1_2.setOnset(0);
194                        settings1_2.setDropping(25);
195                        SignalGroupSettingsData settings2_2 = control.getFactory().createSignalGroupSettingsData(Id.create("2", SignalGroup.class));
196                        plan2.addSignalGroupSettings(settings2_2);
197                        settings2_2.setOnset(0);
198                        settings2_2.setDropping(25);
199                }
200        }

202        /**
203         * Set up the config and scenario, create signal information 
204         * and write them to file as input for further simulations.
205         * 
206         * @throws IOException
207         */
208        public void run(String outputDir) throws IOException {
209                // create an empty config
210                Config config = ConfigUtils.createConfig();
211                
212                // set network and population files
213                config.network().setInputFile(INPUT_DIR + "network.xml.gz");
214                config.plans().setInputFile(INPUT_DIR + "population.xml.gz");
215                
216                // add the signal config group to the config file
217                SignalSystemsConfigGroup signalSystemsConfigGroup = 
218                                ConfigUtils.addOrGetModule(config, SignalSystemsConfigGroup.GROUPNAME, SignalSystemsConfigGroup.class);
219                
220                /* the following makes the contrib load the signal input files, but not to do anything with them
221                 * (this switch will eventually go away) */
222                signalSystemsConfigGroup.setUseSignalSystems(true);
223                
224                // specify some details for the visualization
225                config.qsim().setNodeOffset(20.0);
226                config.qsim().setSnapshotStyle(SnapshotStyle.queue);
227                
228                // --- create the scenario
229                Scenario scenario = ScenarioUtils.loadScenario(config);
230                /* create the information about signals data (i.e. create an empty SignalsData object)
231                 * and add it to the scenario as scenario element */
232                SignalsData signalsData = SignalUtils.createSignalsData(signalSystemsConfigGroup);
233                scenario.addScenarioElement(SignalsData.ELEMENT_NAME, signalsData);
234                
235                /* fill the SignalsData object with information:
236                 * signal systems - specify signalized intersections
237                 * signal groups - specify signals that always have the same signal control
238                 * signal control - specify cycle time, onset and dropping time, offset... for all signal groups
239                 */
240                this.createSignalSystemsAndGroups(signalsData.getSignalSystemsData(), signalsData.getSignalGroupsData());
241                this.createSignalControl(signalsData.getSignalControlData());
242                
243                // create the path to the output directory if it does not exist yet
244                Files.createDirectories(Paths.get(outputDir));
245                
246                // set output filenames
247                signalSystemsConfigGroup.setSignalSystemFile(outputDir + "signal_systems.xml");
248                signalSystemsConfigGroup.setSignalGroupsFile(outputDir + "signal_groups.xml");
249                signalSystemsConfigGroup.setSignalControlFile(outputDir + "signal_control.xml");
250                
251                //write config to file
252                String configFile = outputDir  + "config.xml";
253                ConfigWriter configWriter = new ConfigWriter(config);
254                configWriter.write(configFile);         
255                
256                // write signal information to file
257                SignalsScenarioWriter signalsWriter = new SignalsScenarioWriter();
258                signalsWriter.setSignalSystemsOutputFilename(signalSystemsConfigGroup.getSignalSystemFile());
259                signalsWriter.setSignalGroupsOutputFilename(signalSystemsConfigGroup.getSignalGroupsFile());
260                signalsWriter.setSignalControlOutputFilename(signalSystemsConfigGroup.getSignalControlFile());
261                signalsWriter.writeSignalsData(scenario);
262                
263                log.info("Config of simple traffic light scenario is written to " + configFile);
264                log.info("Visualize scenario by calling VisSimpleTrafficSignalScenario in the same package.");
265        }
266        
267        public static void main(String[] args) throws IOException {
268                new CreateSignalInputExample().run("output/example90TrafficLights/");
269        }
