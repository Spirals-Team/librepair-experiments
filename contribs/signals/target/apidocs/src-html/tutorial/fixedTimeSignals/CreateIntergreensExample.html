020package tutorial.fixedTimeSignals;
021
022import java.io.IOException;
023
024import org.apache.log4j.Logger;
025import org.matsim.api.core.v01.Id;
026import org.matsim.contrib.signals.data.SignalsDataLoader;
027import org.matsim.contrib.signals.data.intergreens.v10.IntergreenTimesData;
028import org.matsim.contrib.signals.data.intergreens.v10.IntergreenTimesDataFactory;
029import org.matsim.contrib.signals.data.intergreens.v10.IntergreenTimesWriter10;
030import org.matsim.contrib.signals.data.intergreens.v10.IntergreensForSignalSystemData;
031import org.matsim.core.config.Config;
032import org.matsim.core.config.ConfigUtils;
033import org.matsim.contrib.signals.data.SignalsData;
034import org.matsim.contrib.signals.model.SignalGroup;
035import org.matsim.contrib.signals.model.SignalSystem;
036import org.matsim.contrib.signals.SignalSystemsConfigGroup;
037
038/**
039 * This example shows how to create an intergreens input file for a given scenario.
040 *
041 * If there is no intergreen data available for your scenario you may use the
042 * intergreen times of a correct fixed time signal control, i.e. a signal control
043 * with realistic intergreen times where no collisions may occur.
044 * @link TtCalculateSimplifiedIntergreens for how to extract intergreen times from a fixed time control
045 *  
046 * @author dgrether
047 */
048public class CreateIntergreensExample {
049
050        private static final Logger log = Logger.getLogger(CreateIntergreensExample.class);
051        
052        private static final String INPUT_DIR = "examples/tutorial/example90TrafficLights/useSignalInput/woLanes/";
053        private static String outputDir = "output/example90TrafficLights/";
054        
055        private static void createIntergreens(SignalsData sd){
056                IntergreenTimesData ig = sd.getIntergreenTimesData();
057                IntergreenTimesDataFactory igdf = ig.getFactory();
058                // Create a data object for signal system with id 3
059                IntergreensForSignalSystemData ig3 = igdf.createIntergreensForSignalSystem(Id.create("3", SignalSystem.class));
060                // Request at least 10 seconds red between the end of green of signal group 1 
061                // and the beginning of green of signal group 2 (signal system 3)
062                ig3.setIntergreenTime(10, Id.create("1", SignalGroup.class), Id.create("2", SignalGroup.class));
063                // add the data object to the container
064                ig.addIntergreensForSignalSystem(ig3);
065                // same as above for signal system 4...
066                IntergreensForSignalSystemData ig4 = igdf.createIntergreensForSignalSystem(Id.create("4", SignalSystem.class));
067                ig4.setIntergreenTime(10, Id.create("1", SignalGroup.class), Id.create("2", SignalGroup.class));
068                ig.addIntergreensForSignalSystem(ig4);
069        }
070        
071        /**
072         * @param args if not null it gives the output directory for the intergreens file
073         * @throws IOException
074         */
075        public static void main(String[] args) throws IOException {
076                if (args != null){
077                        // use the given output if args is not null
078                        outputDir = args[0];
079                }
080                // read in the existing scenario and add the intergreens data
081                Config config = ConfigUtils.loadConfig(INPUT_DIR + "config.xml");
082                SignalSystemsConfigGroup signalSystemsConfigGroup = 
083                                ConfigUtils.addOrGetModule(config, SignalSystemsConfigGroup.GROUPNAME, SignalSystemsConfigGroup.class);
084                signalSystemsConfigGroup.setUseIntergreenTimes(true);
085                SignalsDataLoader signalsDataLoader = new SignalsDataLoader(config);
086                SignalsData signalsData = signalsDataLoader.loadSignalsData();
087                createIntergreens(signalsData);
088                // write the intergreens file
089                IntergreenTimesWriter10 writer = new IntergreenTimesWriter10(signalsData.getIntergreenTimesData());
090                String intergreensFilename = outputDir + "intergreens.xml";
091                writer.write(intergreensFilename);
092                log.info("Intergreens written to " + intergreensFilename);
093        }
094}
