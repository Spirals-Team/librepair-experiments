046/**
047 * Example for a responsive signal.
048 * 
049 * It updates the fixed signal control depending on the average delay every i-th iteration, 
050 * whereby i is defined in the variable INTERVAL.
051 * 
052 * @author tthunig
053 *
054 */
055public class SimpleResponsiveSignal implements AfterMobsimListener{
056
057        private static final Logger LOG = Logger.getLogger(SimpleResponsiveSignal.class);
058        
059        // increase this if agents should have "time" (iterations) to react to the changed signal control
060        private static final int INTERVAL = 1;
061        
062        private Map<Id<Link>, Double> link2avgDelay = new HashMap<>();
063        
064        @Inject Scenario scenario;
065        
066        @Override
067        public void notifyAfterMobsim(AfterMobsimEvent event) {
068                // change signal green split every INTERVAL iteration
069                if (event.getIteration() % INTERVAL == 0 && event.getIteration() != scenario.getConfig().controler().getFirstIteration()){
070                        computeDelays(event);
071                        LOG.info("+++ Iteration " + event.getIteration() + ". Update signal green split...");
072                        updateSignals();
073                }
074        }
075
076        private void computeDelays(AfterMobsimEvent event) {
077                TravelTime travelTime = event.getServices().getLinkTravelTimes();
078                int timeBinSize = scenario.getConfig().travelTimeCalculator().getTraveltimeBinSize();
079                
080                for (Link link : scenario.getNetwork().getLinks().values()){    
081                        double freespeedTT = link.getLength() / link.getFreespeed();
082                        
083                        int timeBinCounter = 0;
084                        double summedDelay = 0.0;
085                        for (int endTime = timeBinSize ; endTime <= scenario.getConfig().travelTimeCalculator().getMaxTime(); endTime = endTime + timeBinSize ) {
086                                double avgDelay = travelTime.getLinkTravelTime(link, (endTime - timeBinSize/2.), null, null) - freespeedTT;
087                                summedDelay += avgDelay;
088                                timeBinCounter++;
089                        }
090                        link2avgDelay.put(link.getId(), summedDelay/timeBinCounter);
091                        LOG.info("Link id: " + link.getId() + ", avg delay: " + summedDelay/timeBinCounter);
092                }
093                
094        }
095
096        private void updateSignals() {
097                SignalsData signalsData = (SignalsData) scenario.getScenarioElement(SignalsData.ELEMENT_NAME);
098                SignalControlData signalControl = signalsData.getSignalControlData();
099                
100                SignalSystemControllerData signalControlSystem1 = signalControl.getSignalSystemControllerDataBySystemId().get(Id.create("SignalSystem1", SignalSystem.class));
101                SignalPlanData signalPlan = signalControlSystem1.getSignalPlanData().get(Id.create("SignalPlan1", SignalPlan.class));
102                SortedMap<Id<SignalGroup>, SignalGroupSettingsData> signalGroupSettings = signalPlan.getSignalGroupSettingsDataByGroupId();
103                SignalGroupSettingsData group1Setting = signalGroupSettings.get(Id.create("SignalGroup1", SignalGroup.class));
104                SignalGroupSettingsData group2Setting = signalGroupSettings.get(Id.create("SignalGroup2", SignalGroup.class));
105                
106                // shift green time by one second depending on which delay is higher
107                double delaySignalGroup1 = link2avgDelay.get(Id.createLinkId("2_3")) + link2avgDelay.get(Id.createLinkId("4_3"));
108                double delaySignalGroup2 = link2avgDelay.get(Id.createLinkId("7_3")) + link2avgDelay.get(Id.createLinkId("8_3"));
109                int greenTimeShift = (int) Math.signum(delaySignalGroup1 - delaySignalGroup2);
110
111                // group1 onset = 0, group2 dropping = 55. signal switch should stay inside this interval
112                if (greenTimeShift != 0 && group1Setting.getDropping() + greenTimeShift > 0 && group2Setting.getOnset() + greenTimeShift < 55){
113                        group1Setting.setDropping(group1Setting.getDropping() + greenTimeShift);
114                        group2Setting.setOnset(group2Setting.getOnset() + greenTimeShift);
115                        LOG.info("SignalGroup1: onset " + group1Setting.getOnset() + ", dropping " + group1Setting.getDropping());
116                        LOG.info("SignalGroup2: onset " + group2Setting.getOnset() + ", dropping " + group2Setting.getDropping());
117                } else {
118                        // do nothing
119                        LOG.info("Signal control unchanged.");
120                }
121        }
122
123}
