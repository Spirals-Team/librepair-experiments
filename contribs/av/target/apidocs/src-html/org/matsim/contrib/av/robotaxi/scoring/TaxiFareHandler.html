023package org.matsim.contrib.av.robotaxi.scoring;
024
025import java.util.HashMap;
026import java.util.HashSet;
027import java.util.Map;
028import java.util.Set;
029
030import org.apache.commons.lang.mutable.MutableDouble;
031import org.matsim.api.core.v01.Id;
032import org.matsim.api.core.v01.events.LinkEnterEvent;
033import org.matsim.api.core.v01.events.PersonArrivalEvent;
034import org.matsim.api.core.v01.events.PersonDepartureEvent;
035import org.matsim.api.core.v01.events.PersonEntersVehicleEvent;
036import org.matsim.api.core.v01.events.PersonMoneyEvent;
037import org.matsim.api.core.v01.events.handler.LinkEnterEventHandler;
038import org.matsim.api.core.v01.events.handler.PersonArrivalEventHandler;
039import org.matsim.api.core.v01.events.handler.PersonDepartureEventHandler;
040import org.matsim.api.core.v01.events.handler.PersonEntersVehicleEventHandler;
041import org.matsim.api.core.v01.network.Network;
042import org.matsim.api.core.v01.population.Person;
043import org.matsim.contrib.dvrp.run.DvrpConfigGroup;
044import org.matsim.core.api.experimental.events.EventsManager;
045import org.matsim.core.config.Config;
046import org.matsim.vehicles.Vehicle;
047
048import com.google.inject.Inject;
049
050/**
051 * @author  jbischoff
052 *      A simple implementation for taxi fares.
053 *  Note that these fares are scored in excess to anything set in the modeparams in the config file.
054 */
055/**
056 *
057 */
058
059public class TaxiFareHandler implements LinkEnterEventHandler, PersonEntersVehicleEventHandler,
060                PersonDepartureEventHandler, PersonArrivalEventHandler {
061        
062        
063        private final EventsManager events;
064        private final Network network;
065        
066        private final double distanceFare_Meter;
067        private final double baseFare;
068        private final double timeFare_hour;
069        private final double dailyFee;
070        
071        Map<Id<Vehicle>,MutableDouble> currentRideDistance = new HashMap<>();
072        Map<Id<Person>,Id<Vehicle>> currentVehicle = new HashMap<>();
073        Set<Id<Person>> waitingPax = new HashSet<>();
074        Map<Id<Person>, Double> vehicleEnterTime = new HashMap<>();
075        Set<Id<Person>> dailyFeeCharged = new HashSet<>();
076        private String mode;
077        
078        /**
079         *  @params config: A Matsim config that contains a TaxiFareConfigGroup
080         */
081        @Inject
082        public TaxiFareHandler(Config config, EventsManager events, Network network) {
083                TaxiFareConfigGroup taxiFareConfigGroup = TaxiFareConfigGroup.get(config);
084                this.mode = DvrpConfigGroup.get(config).getMode();
085                this.events= events;
086                this.network = network;
087                this.distanceFare_Meter = taxiFareConfigGroup.getDistanceFare_m();
088                this.baseFare = taxiFareConfigGroup.getBasefare();
089                this.dailyFee = taxiFareConfigGroup.getDailySubscriptionFee();
090                this.timeFare_hour = taxiFareConfigGroup.getTimeFare_h();
091        }
092        
093        
094
095        /* (non-Javadoc)
096         * @see org.matsim.core.events.handler.EventHandler#reset(int)
097         */
098        @Override
099        public void reset(int iteration) {
100                waitingPax.clear();
101                currentVehicle.clear();
102                currentRideDistance.clear();
103                dailyFeeCharged.clear();
104                vehicleEnterTime.clear();
105        }
106
107        /* (non-Javadoc)
108         * @see org.matsim.api.core.v01.events.handler.PersonArrivalEventHandler#handleEvent(org.matsim.api.core.v01.events.PersonArrivalEvent)
109         */
110        @Override
111        public void handleEvent(PersonArrivalEvent event) {
112                if (currentVehicle.containsKey(event.getPersonId())){
113                        Id<Vehicle> vid = currentVehicle.remove(event.getPersonId());
114                        double distance = currentRideDistance.remove(vid).doubleValue();
115                        double rideTime = (event.getTime() - vehicleEnterTime.remove(event.getPersonId()))/3600.0 ;
116                        double fare = -(baseFare + distance*distanceFare_Meter + rideTime*timeFare_hour);
117                        events.processEvent(new PersonMoneyEvent(event.getTime(), event.getPersonId(), fare));
118                }
119        }
120
121        /* (non-Javadoc)
122         * @see org.matsim.api.core.v01.events.handler.PersonDepartureEventHandler#handleEvent(org.matsim.api.core.v01.events.PersonDepartureEvent)
123         */
124        @Override
125        public void handleEvent(PersonDepartureEvent event) {
126                if (event.getLegMode().equals(mode)){
127                        waitingPax.add(event.getPersonId());
128                        if (!dailyFeeCharged.contains(event.getPersonId())){
129                                dailyFeeCharged.add(event.getPersonId());
130                                events.processEvent(new PersonMoneyEvent(event.getTime(), event.getPersonId(), -dailyFee));
131                        }
132                }
133        }
134
135        /* (non-Javadoc)
136         * @see org.matsim.api.core.v01.events.handler.PersonEntersVehicleEventHandler#handleEvent(org.matsim.api.core.v01.events.PersonEntersVehicleEvent)
137         */
138        @Override
139        public void handleEvent(PersonEntersVehicleEvent event) {
140                if (waitingPax.contains(event.getPersonId())){
141                        currentVehicle.put(event.getPersonId(), event.getVehicleId());
142                        currentRideDistance.put(event.getVehicleId(), new MutableDouble(0.0));
143                        waitingPax.remove(event.getPersonId());
144                        vehicleEnterTime.put(event.getPersonId(), event.getTime());
145                }
146        }
147
148        /* (non-Javadoc)
149         * @see org.matsim.api.core.v01.events.handler.LinkEnterEventHandler#handleEvent(org.matsim.api.core.v01.events.LinkEnterEvent)
150         */
151        @Override
152        public void handleEvent(LinkEnterEvent event) {
153                if (currentRideDistance.containsKey(event.getVehicleId())){
154                        double length = network.getLinks().get(event.getLinkId()).getLength();
155                        currentRideDistance.get(event.getVehicleId()).add(length);
156                }
157        }
158
159}
