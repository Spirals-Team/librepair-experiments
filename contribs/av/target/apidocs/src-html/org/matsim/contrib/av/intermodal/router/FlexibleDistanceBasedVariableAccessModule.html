057public class FlexibleDistanceBasedVariableAccessModule implements VariableAccessEgressTravelDisutility {
058
059        
060        private Map<String,Boolean> teleportedModes = new HashMap<>();
061        private Map<String,Integer> distanceMode = new TreeMap<>();
062        private Map<String, LeastCostPathCalculator> lcpPerNonTeleportedMode = new HashMap<>();
063        
064        private final Network carnetwork;
065        private final Config config;
066        
067        private boolean checkCarAvail = true;
068        private final Random rand = MatsimRandom.getRandom();
069        /**
070         * 
071         */
072        public FlexibleDistanceBasedVariableAccessModule(Network carnetwork, Config config) {
073                this.config = config;
074                this.carnetwork = carnetwork;
075        }
076        /**
077         * 
078         * @param mode the mode to register
079         * @param maximumAccessDistance maximum beeline Distance for using this mode
080         * @param isTeleported defines whether this is a teleported mode
081         * @param lcp for non teleported modes, some travel time assumption is required
082         */
083        public void registerMode(String mode, int maximumAccessDistance, boolean isTeleported){
084                if (this.distanceMode.containsKey(mode)){
085                        throw new RuntimeException("mode "+mode+"is already registered. Check your config four double entries..");
086                }
087                if (isTeleported){
088                        teleportedModes.put(mode, true);
089                        distanceMode.put(mode, maximumAccessDistance);
090                }
091                else {
092                        teleportedModes.put(mode, false);
093                        distanceMode.put(mode, maximumAccessDistance);
094                        
095                        
096                }
097                
098        }
099        
100        
101        /* (non-Javadoc)
102         * @see playground.jbischoff.pt.VariableAccessEgressTravelDisutility#getAccessEgressModeAndTraveltime(org.matsim.api.core.v01.population.Person, org.matsim.api.core.v01.Coord, org.matsim.api.core.v01.Coord)
103         */
104        @Override
105        public Leg getAccessEgressModeAndTraveltime(Person person, Coord coord, Coord toCoord, double time) {
106                double egressDistance = CoordUtils.calcEuclideanDistance(coord, toCoord);
107                
108                String mode = getModeForDistance(egressDistance, person);
109                Leg leg = PopulationUtils.createLeg(mode);
110                Link startLink = NetworkUtils.getNearestLink(carnetwork, coord);
111                Link endLink = NetworkUtils.getNearestLink(carnetwork, toCoord);
112                Route route = RouteUtils.createGenericRouteImpl(startLink.getId(), endLink.getId());
113                leg.setRoute(route);
114                if (this.teleportedModes.get(mode)){
115                        double distf = config.plansCalcRoute().getModeRoutingParams().get(mode).getBeelineDistanceFactor();
116                        double speed = config.plansCalcRoute().getModeRoutingParams().get(mode).getTeleportedModeSpeed();
117                        double distance = egressDistance*distf;
118                        double travelTime = distance / speed;
119                        leg.setTravelTime(travelTime);
120                        route.setDistance(distance);
121                        leg.setDepartureTime(time);
122                        
123                                                
124                } else {
125                        double distance = egressDistance*1.3;
126                        double travelTime = distance / 7.25;
127                        leg.setTravelTime(travelTime);
128                        route.setDistance(distance);
129                        
130//                      too expensive
131//                      Path path = this.lcpPerNonTeleportedMode.get(mode).calcLeastCostPath(startLink.getFromNode(), endLink.getToNode(), 0, person, null);
132//                      route.setDistance(path.travelCost);
133//                      route.setTravelTime(path.travelTime);
134                }
135                return leg;
136        }
137
138        /**
139         * @param egressDistance
140         * @return
141         */
142        private String getModeForDistance(double egressDistance, Person p) {
143                if (distanceMode.containsKey(TransportMode.walk))
144                if (egressDistance<=distanceMode.get(TransportMode.walk)) return TransportMode.walk;
145                //TODO: MAke Config switch
146                List<String> possibleModes = new ArrayList<>();
147                for (Entry<String,Integer> e : this.distanceMode.entrySet()){
148                        if (e.getValue()>=egressDistance){
149                                if (e.getKey().equals(TransportMode.car)){
150                                        if ((checkCarAvail)&&(p!=null)&&(p.getCustomAttributes()!=null)){
151                                                
152                                                String carA  = PersonUtils.getCarAvail(p);
153                                                if (carA!=null){
154                                                if (carA.equals("always")||carA.equals("sometimes")){
155                                                        possibleModes.add(e.getKey());
156                                                
157                                                }
158                                                }
159                                                else {                          
160                                                        possibleModes.add(e.getKey());
161                                                }
162                                        }
163                                        else {
164                                                possibleModes.add(e.getKey());
165                                        }
166                                }
167                                else {
168                                        possibleModes.add(e.getKey());
169                                }
170                        }
171                }
172                if (possibleModes.size()<1){
173//                      Logger.getLogger(getClass()).warn("Egress distance "+egressDistance+ " is not covered by any available mode mode for person "+p.getId()+". Assuming walk.");
174                        return (TransportMode.walk);
175                }
176                return possibleModes.get(rand.nextInt(possibleModes.size()));
177        }
178
179
180        /* (non-Javadoc)
181         * @see playground.jbischoff.pt.VariableAccessEgressTravelDisutility#isTeleportedAccessEgressMode(java.lang.String)
182         */
183        @Override
184        public boolean isTeleportedAccessEgressMode(String mode) {
185                return this.teleportedModes.get(mode);
186        }
