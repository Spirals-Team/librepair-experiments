020package org.matsim.contrib.av.robotaxi.run;
021
022import org.matsim.api.core.v01.Scenario;
023import org.matsim.contrib.av.robotaxi.scoring.TaxiFareConfigGroup;
024import org.matsim.contrib.av.robotaxi.scoring.TaxiFareHandler;
025import org.matsim.contrib.dvrp.run.DvrpConfigGroup;
026import org.matsim.contrib.otfvis.OTFVisLiveModule;
027import org.matsim.contrib.taxi.run.TaxiConfigConsistencyChecker;
028import org.matsim.contrib.taxi.run.TaxiConfigGroup;
029import org.matsim.contrib.taxi.run.TaxiModule;
030import org.matsim.contrib.taxi.run.examples.TaxiDvrpModules;
031import org.matsim.core.config.Config;
032import org.matsim.core.config.ConfigUtils;
033import org.matsim.core.controler.AbstractModule;
034import org.matsim.core.controler.Controler;
035import org.matsim.core.scenario.ScenarioUtils;
036import org.matsim.vis.otfvis.OTFVisConfigGroup;
037
038/**
039 * This class runs an example robotaxi scenario including scoring. The simulation runs for 10 iterations, this takes
040 * quite a bit time (25 minutes or so). You may switch on OTFVis visualisation in the main method below. The scenario
041 * should run out of the box without any additional files. If required, you may find all input files in the resource
042 * path or in the jar maven has downloaded). There are two vehicle files: 2000 vehicles and 5000, which may be set in
043 * the config. Different fleet sizes can be created using
044 * {@link org.matsim.contrib.robotaxi.vehicles.CreateTaxiVehicles}
045 * 
046 * 
047 */
048public class RunRobotaxiExample {
049
050        public static void main(String[] args) {
051                String configFile = "cottbus_robotaxi/config.xml";
052                RunRobotaxiExample.run(configFile, false);
053        }
054
055        public static void run(String configFile, boolean otfvis) {
056                Config config = ConfigUtils.loadConfig(configFile, new DvrpConfigGroup(), new TaxiConfigGroup(),
057                                new OTFVisConfigGroup(), new TaxiFareConfigGroup());
058                createControler(config, otfvis).run();
059        }
060
061        public static Controler createControler(Config config, boolean otfvis) {
062                config.addConfigConsistencyChecker(new TaxiConfigConsistencyChecker());
063                config.checkConsistency();
064
065                Scenario scenario = ScenarioUtils.loadScenario(config);
066
067                Controler controler = new Controler(scenario);
068                controler.addOverridingModule(new AbstractModule() {
069                        @Override
070                        public void install() {
071                                addEventHandlerBinding().to(TaxiFareHandler.class).asEagerSingleton();
072                        }
073                });
074                controler.addOverridingModule(TaxiDvrpModules.create());
075                controler.addOverridingModule(new TaxiModule());
076
077                if (otfvis) {
078                        controler.addOverridingModule(new OTFVisLiveModule());
079                }
080
081                return controler;
082        }
083}
