001package org.matsim.contrib.av.intermodal.router;
002
003import com.google.inject.name.Named;
004import org.matsim.api.core.v01.Scenario;
005import org.matsim.api.core.v01.TransportMode;
006import org.matsim.contrib.dvrp.router.TimeAsTravelDisutility;
007import org.matsim.core.router.RoutingModule;
008import org.matsim.core.router.util.LeastCostPathCalculator;
009import org.matsim.core.router.util.LeastCostPathCalculatorFactory;
010import org.matsim.core.router.util.TravelDisutility;
011import org.matsim.core.router.util.TravelTime;
012import org.matsim.pt.router.TransitRouter;
013
014import java.util.Map;
015
016import javax.inject.Inject;
017import javax.inject.Provider;
018
019public class VariableAccessTransit implements Provider<RoutingModule> {
020
021        private final TransitRouter transitRouter;
022
023        private final Scenario scenario;
024
025        private final RoutingModule transitWalkRouter;
026        private final LeastCostPathCalculator routeAlgo;
027
028        @Inject
029    VariableAccessTransit(@Named("variableAccess") TransitRouter transitRouter, Scenario scenario, @Named(TransportMode.transit_walk) RoutingModule transitWalkRouter, LeastCostPathCalculatorFactory routeAlgoF, Map<String,TravelTime> travelTimes) {
030                this.transitRouter = transitRouter;
031                this.scenario = scenario;
032                this.transitWalkRouter = transitWalkRouter;
033                TravelTime travelTime = travelTimes.get(TransportMode.car);
034        TravelDisutility travelDisutility = new TimeAsTravelDisutility(travelTime);
035
036                this.routeAlgo = routeAlgoF.createPathCalculator(scenario.getNetwork(), travelDisutility, travelTime);
037        }
038
039        @Override
040        public RoutingModule get() {
041                return new VariableAccessTransitRouterWrapper(transitRouter,
042                                        scenario.getTransitSchedule(),
043                                        scenario.getNetwork(),
044                                        transitWalkRouter, routeAlgo);
045        }
046}
