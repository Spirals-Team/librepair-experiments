089        private Map<Node, InitialNode> locateWrappedNearestTransitNodes(Person person, Coord coord, double departureTime) {
090                Collection<TransitRouterNetworkNode> nearestNodes = this.transitNetwork.getNearestNodes(coord, this.trConfig.getSearchRadius());
091                if (nearestNodes.size() < 2) {
092                        // also enlarge search area if only one stop found, maybe a second one is near the border of the search area
093                        TransitRouterNetworkNode nearestNode = this.transitNetwork.getNearestNode(coord);
094                        double distance = CoordUtils.calcEuclideanDistance(coord, nearestNode.stop.getStopFacility().getCoord());
095                        nearestNodes = this.transitNetwork.getNearestNodes(coord, distance + this.trConfig.getExtensionRadius());
096                }
097                Map<Node, InitialNode> wrappedNearestNodes = new LinkedHashMap<>();
098                for (TransitRouterNetworkNode node : nearestNodes) {
099                        Coord toCoord = node.stop.getStopFacility().getCoord();
100                        Leg initialLeg = getAccessEgressLeg(person, coord, toCoord, departureTime);
101                        double initialTime = initialLeg.getTravelTime();
102                        double marginalUtilityOfDistance_utl_m = 
103                                        pcsConfig.getModes().get(initialLeg.getMode()).getMonetaryDistanceRate() * pcsConfig.getMarginalUtilityOfMoney() +
104                                        pcsConfig.getModes().get(initialLeg.getMode()).getMarginalUtilityOfDistance();
105                        double marginalUtilityOfTravelTime_utl_s = 
106                                        pcsConfig.getModes().get(initialLeg.getMode()).getMarginalUtilityOfTraveling()/3600.0 -
107                                        pcsConfig.getPerforming_utils_hr()/3600.;
108                        //  getMarginalUtilityOfTravelTimeWalk includes the opportunity cost of time.
109                        double timeCost = - initialLeg.getTravelTime() * marginalUtilityOfTravelTime_utl_s ;
110                        // (sign: margUtl is negative; overall it should be positive because it is a cost.)
111                        double distanceCost = - initialLeg.getRoute().getDistance() * marginalUtilityOfDistance_utl_m ;
112                        // (sign: same as above)
113                        double initialCost = timeCost + distanceCost;
114                        wrappedNearestNodes.put(node, new InitialNode(initialCost, initialTime + departureTime));
115                }
116                return wrappedNearestNodes;
117        }

119        private Leg getAccessEgressLeg(Person person, Coord coord, Coord toCoord, double time) {
120                return variableAccessEgressTravelDisutility.getAccessEgressModeAndTraveltime(person, coord, toCoord, time);

124        private double getTransferTime(Person person, Coord coord, Coord toCoord) {
125                return travelDisutility.getWalkTravelTime(person, coord, toCoord) + this.trConfig.getAdditionalTransferTime();
126        }

128        private double getAccessEgressDisutility(Person person, Coord coord, Coord toCoord) {
129                return travelDisutility.getWalkTravelDisutility(person, coord, toCoord);
130        }

132        @Override
133        public List<Leg> calcRoute(final Facility<?> fromFacility, final Facility<?> toFacility, final double departureTime, final Person person) {
134                // find possible start stops
135                Map<Node, InitialNode> wrappedFromNodes = this.locateWrappedNearestTransitNodes(person, fromFacility.getCoord(), departureTime);
136                // find possible end stops
137                Map<Node, InitialNode> wrappedToNodes = this.locateWrappedNearestTransitNodes(person, toFacility.getCoord(), departureTime);
138
139                TransitLeastCostPathTree tree = new TransitLeastCostPathTree(transitNetwork, travelDisutility, travelTime,
140                                wrappedFromNodes, wrappedToNodes, person);
141
142                // find routes between start and end stop
143                Path p = tree.getPath(wrappedToNodes);
144
145                if (p == null) {
146                        return null;
147                }
148
149                double directWalkCost = getAccessEgressDisutility(person, fromFacility.getCoord(), toFacility.getCoord());
150                double pathCost = p.travelCost + wrappedFromNodes.get(p.nodes.get(0)).initialCost + wrappedToNodes.get(p.nodes.get(p.nodes.size() - 1)).initialCost;
151
152                if (directWalkCost * trConfig.getDirectWalkFactor() < pathCost) {
153                        return this.createDirectAccessEgressModeLegList(null, fromFacility.getCoord(), toFacility.getCoord(), departureTime);
154                }
155                return convertPathToLegList(departureTime, p, fromFacility.getCoord(), toFacility.getCoord(), person);
156        }

158        private List<Leg> createDirectAccessEgressModeLegList(Person person, Coord fromCoord, Coord toCoord, double time) {
159                List<Leg> legs = new ArrayList<>();
160                Leg leg = getAccessEgressLeg(person, fromCoord, toCoord, time);
161                legs.add(leg);
162                return legs;
163        }

165        protected List<Leg> convertPathToLegList(double departureTime, Path path, Coord fromCoord, Coord toCoord, Person person) {
166                // yy would be nice if the following could be documented a bit better.  kai, jul'16
167                
168                // now convert the path back into a series of legs with correct routes
169                double time = departureTime;
170                List<Leg> legs = new ArrayList<>();
171                Leg leg;
172                TransitLine line = null;
173                TransitRoute route = null;
174                TransitStopFacility accessStop = null;
175                TransitRouteStop transitRouteStart = null;
176                TransitRouterNetworkLink prevLink = null;
177                double currentDistance = 0;
178                int transitLegCnt = 0;
179                for (Link ll : path.links) {
180                        TransitRouterNetworkLink link = (TransitRouterNetworkLink) ll;
181                        if (link.getLine() == null) {
182                                // (it must be one of the "transfer" links.) finish the pt leg, if there was one before...
183                                TransitStopFacility egressStop = link.fromNode.stop.getStopFacility();
184                                if (route != null) {
185                                        leg = PopulationUtils.createLeg(TransportMode.pt);
186                                        ExperimentalTransitRoute ptRoute = new ExperimentalTransitRoute(accessStop, line, route, egressStop);
187                                        double arrivalOffset = (link.getFromNode().stop.getArrivalOffset() != Time.UNDEFINED_TIME) ? link.fromNode.stop.getArrivalOffset() : link.fromNode.stop.getDepartureOffset();
188                                        double arrivalTime = this.preparedTransitSchedule.getNextDepartureTime(route, transitRouteStart, time) + (arrivalOffset - transitRouteStart.getDepartureOffset());
189                                        ptRoute.setTravelTime(arrivalTime - time);
190
191//                                      ptRoute.setDistance( currentDistance );
192                                        ptRoute.setDistance( link.getLength() );
193                                        // (see MATSIM-556)
194
195                                        leg.setRoute(ptRoute);
196                                        leg.setTravelTime(arrivalTime - time);
197                                        time = arrivalTime;
198                                        legs.add(leg);
199                                        transitLegCnt++;
200                                        accessStop = egressStop;
201                                }
202                                line = null;
203                                route = null;
204                                transitRouteStart = null;
205                                currentDistance = link.getLength();
206                        } else {
207                                // (a real pt link)
208                                currentDistance += link.getLength();
209                                if (link.getRoute() != route) {
210                                        // the line changed
211                                        TransitStopFacility egressStop = link.fromNode.stop.getStopFacility();
212                                        if (route == null) {
213                                                // previously, the agent was on a transfer, add the walk leg
214                                                transitRouteStart = ((TransitRouterNetworkLink) ll).getFromNode().stop;
215                                                if (accessStop != egressStop) {
216                                                        if (accessStop != null) {
217                                                                leg = PopulationUtils.createLeg(TransportMode.transit_walk);
218                                                                //                                                          double walkTime = getWalkTime(person, accessStop.getCoord(), egressStop.getCoord());
219                                                                double transferTime = getTransferTime(person, accessStop.getCoord(), egressStop.getCoord());
220                                                                Route walkRoute = RouteUtils.createGenericRouteImpl(accessStop.getLinkId(),
221                                                                                egressStop.getLinkId());
222                                                                // (yy I would have expected this from egressStop to accessStop. kai, jul'16)
223                                                                
224                                                                //                                                          walkRoute.setTravelTime(walkTime);
225                                                                walkRoute.setTravelTime(transferTime);
226                                                                
227//                                                              walkRoute.setDistance( currentDistance );
228                                                                walkRoute.setDistance( trConfig.getBeelineDistanceFactor() * 
229                                                                                NetworkUtils.getEuclideanDistance(accessStop.getCoord(), egressStop.getCoord()) );
230                                                                // (see MATSIM-556)
231
232                                                                leg.setRoute(walkRoute);
233                                                                //                                                          leg.setTravelTime(walkTime);
234                                                                leg.setTravelTime(transferTime);
235                                                                //                                                          time += walkTime;
236                                                                time += transferTime;
237                                                                legs.add(leg);
238                                                        } else {
239                                                                // accessStop == null, so it must be the first access-leg. If mode is e.g. taxi, we need a transit_walk to get to pt link
240                                                                leg = getAccessEgressLeg(person, fromCoord, egressStop.getCoord(),time);
241                                                                if (variableAccessEgressTravelDisutility.isTeleportedAccessEgressMode(leg.getMode()))
242                                                                {
243                                                                        leg.getRoute().setEndLinkId(egressStop.getLinkId());
244                                                                        time += leg.getTravelTime();
245                                                                        legs.add(leg);
246
247
248                                                                } else {
249                                                                        legs.add(leg); //access leg
250                                                                        time += leg.getTravelTime();
251                                                                        
252                                                                        Route walkRoute = RouteUtils.createGenericRouteImpl(
253                                                                                        leg.getRoute().getEndLinkId(), egressStop.getLinkId());
254                                                                        double walkTime = getTransferTime(person, network.getLinks().get(leg.getRoute().getEndLinkId()).getCoord(), egressStop.getCoord());
255                                                                        walkRoute.setTravelTime(walkTime);
256                                                                        walkRoute.setDistance(trConfig.getBeelineDistanceFactor() * 
257                                                                                        NetworkUtils.getEuclideanDistance(network.getLinks().get(leg.getRoute().getEndLinkId()).getCoord(), egressStop.getCoord()) );
258                                                                
259                                                                        Leg walkleg = PopulationUtils.createLeg(TransportMode.transit_walk);
260                                                                        walkleg.setTravelTime(walkTime);
261                                                                        walkleg.setRoute(walkRoute);
262                                                                        time += walkTime;
263                                                                        legs.add(walkleg);
264                                                                        
265                                                                }
266                                                                
267//                                                              walkRoute.setDistance( currentDistance );
268                                                                // (see MATSIM-556)
269
270                                                        }
271                                                }
272                                                currentDistance = 0;
273                                        }
274                                        line = link.getLine();
275                                        route = link.getRoute();
276                                        accessStop = egressStop;
277                                }
278                        }
279                        prevLink = link;
280                }
281                if (route != null) {
282                        // the last part of the path was with a transit route, so add the pt-leg and final walk-leg
283                        leg = PopulationUtils.createLeg(TransportMode.pt);
284                        TransitStopFacility egressStop = prevLink.toNode.stop.getStopFacility();
285                        ExperimentalTransitRoute ptRoute = new ExperimentalTransitRoute(accessStop, line, route, egressStop);
286//                      ptRoute.setDistance( currentDistance );
287                        ptRoute.setDistance( trConfig.getBeelineDistanceFactor() * NetworkUtils.getEuclideanDistance(accessStop.getCoord(), egressStop.getCoord() ) );
288                        // (see MATSIM-556)
289                        leg.setRoute(ptRoute);
290                        double arrivalOffset = ((prevLink).toNode.stop.getArrivalOffset() != Time.UNDEFINED_TIME) ?
291                                        (prevLink).toNode.stop.getArrivalOffset()
292                                        : (prevLink).toNode.stop.getDepartureOffset();
293                                        double arrivalTime = this.preparedTransitSchedule.getNextDepartureTime(route, transitRouteStart, time) + (arrivalOffset - transitRouteStart.getDepartureOffset());
294                                        leg.setTravelTime(arrivalTime - time);
295                                        ptRoute.setTravelTime( arrivalTime - time );
296                                        legs.add(leg);
297                                        transitLegCnt++;
298                                        accessStop = egressStop;
299                }
300                if (prevLink != null) {
301                        if (accessStop == null) {
302                                // no use of pt
303                                leg = getAccessEgressLeg(person, fromCoord, toCoord, time);
304                                legs.add(leg);
305
306                        } else {
307                                Leg eleg = getAccessEgressLeg(person, accessStop.getCoord(), toCoord, time);
308                                
309                                if (variableAccessEgressTravelDisutility.isTeleportedAccessEgressMode(eleg.getMode())){
310                                        leg = eleg;
311                                        leg.getRoute().setStartLinkId(accessStop.getLinkId());
312                                        legs.add(leg);
313
314                                }
315                                else {
316                                        leg = PopulationUtils.createLeg(TransportMode.transit_walk);
317                                        double walkTime = getTransferTime(person, accessStop.getCoord(), network.getLinks().get(eleg.getRoute().getStartLinkId()).getCoord());
318                                        leg.setTravelTime(walkTime);
319                                        Route walkRoute = RouteUtils.createGenericRouteImpl(accessStop.getLinkId(), eleg.getRoute().getStartLinkId());
320                                        walkRoute.setDistance(trConfig.getBeelineDistanceFactor() * 
321                                                                                        NetworkUtils.getEuclideanDistance(accessStop.getCoord(),network.getLinks().get(eleg.getRoute().getStartLinkId()).getCoord()));
322                                        leg.setRoute(walkRoute);
323                                        legs.add(leg);
324                                        legs.add(eleg);
325                                }
326                        }
327                }
328                if (transitLegCnt == 0) {
329                        // it seems, the agent only walked
330                        legs.clear();
331                        try{
332                        leg = getAccessEgressLeg(person, fromCoord, toCoord, time);
333
334                        legs.add(leg);
335                        } catch (NullPointerException e){
336                                throw new RuntimeException(" npe: person"+ person + fromCoord + toCoord + time);
337                        }
338                }
339                return legs;
340        }
