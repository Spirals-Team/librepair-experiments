043 * @author  jbischoff

045 */
046/**
047 *
048 */
049public class FixedDistanceBasedVariableAccessModule implements VariableAccessEgressTravelDisutility {
050
051        
052        private Map<String,Boolean> teleportedModes = new HashMap<>();
053        private Map<Integer,String> distanceMode = new TreeMap<>();
054        
055        private final Network carnetwork;
056        private final Config config;
057        
058        /**
059         * 
060         */
061        public FixedDistanceBasedVariableAccessModule(Network carnetwork, Config config) {
062                this.config = config;
063                this.carnetwork = carnetwork;
064        }
065        /**
066         * 
067         * @param mode the mode to register
068         * @param maximumAccessDistance maximum beeline Distance for using this mode
069         * @param isTeleported defines whether this is a teleported mode
070         * @param lcp for non teleported modes, some travel time assumption is required
071         */
072        public void registerMode(String mode, int maximumAccessDistance, boolean isTeleported){
073                if (this.distanceMode.containsKey(maximumAccessDistance)){
074                        throw new RuntimeException("Maximum distance of "+maximumAccessDistance+" is already registered to mode "+distanceMode.get(maximumAccessDistance)+" and cannot be re-registered to mode: "+mode);
075                }
076                if (isTeleported){
077                        teleportedModes.put(mode, true);
078                        distanceMode.put(maximumAccessDistance, mode);
079                } else {
080                        teleportedModes.put(mode, false);
081                        distanceMode.put(maximumAccessDistance,mode);
082                        
083                        
084                }
085        }
086        
087        
088        /* (non-Javadoc)
089         * @see playground.jbischoff.pt.VariableAccessEgressTravelDisutility#getAccessEgressModeAndTraveltime(org.matsim.api.core.v01.population.Person, org.matsim.api.core.v01.Coord, org.matsim.api.core.v01.Coord)
090         */
091        @Override
092        public Leg getAccessEgressModeAndTraveltime(Person person, Coord coord, Coord toCoord, double time) {
093                double egressDistance = CoordUtils.calcEuclideanDistance(coord, toCoord);
094                String mode = getModeForDistance(egressDistance);
095                Leg leg = PopulationUtils.createLeg(mode);
096                Link startLink = NetworkUtils.getNearestLink(carnetwork, coord);
097                Link endLink = NetworkUtils.getNearestLink(carnetwork, toCoord);
098                Route route = RouteUtils.createGenericRouteImpl(startLink.getId(), endLink.getId());
099                leg.setRoute(route);
100                if (this.teleportedModes.get(mode)){
101                        double distf = config.plansCalcRoute().getModeRoutingParams().get(mode).getBeelineDistanceFactor();
102                        double speed = config.plansCalcRoute().getModeRoutingParams().get(mode).getTeleportedModeSpeed();
103                        double distance = egressDistance*distf;
104                        double travelTime = distance / speed;
105                        leg.setTravelTime(travelTime);
106                        route.setDistance(distance);
107                        leg.setDepartureTime(time);
108                        
109                                                
110                } else {
111                        double distance = egressDistance*1.3;
112                        double travelTime = distance / 7.25;
113                        leg.setTravelTime(travelTime);
114                        route.setDistance(distance);
115                        
116
117                }
118                return leg;
119        }
120
121        /**
122         * @param egressDistance
123         * @return
124         */
125        private String getModeForDistance(double egressDistance) {
126                for (Entry<Integer, String> e : this.distanceMode.entrySet()){
127                        if (e.getKey()>=egressDistance){
128//                              System.out.println("Mode" + e.getValue()+" "+egressDistance);
129                                return e.getValue();
130                        }
131                }
132                throw new RuntimeException(egressDistance + " m is not covered by any egress / access mode.");
133                
134        }
135
136
137        /* (non-Javadoc)
138         * @see playground.jbischoff.pt.VariableAccessEgressTravelDisutility#isTeleportedAccessEgressMode(java.lang.String)
139         */
140        @Override
141        public boolean isTeleportedAccessEgressMode(String mode) {
142                return this.teleportedModes.get(mode);
143        }
