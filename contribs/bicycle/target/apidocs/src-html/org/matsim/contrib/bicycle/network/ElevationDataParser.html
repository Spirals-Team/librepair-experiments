072        
073        public ElevationDataParser(String tiffFile, String scenarioCRS) {
074                this.ct = TransformationFactory.getCoordinateTransformation(scenarioCRS, "EPSG:4326");
075                
076                GeoTiffReader reader = null;
077                try {
078                        reader = new GeoTiffReader(tiffFile);
079                } catch (DataSourceException e) {
080                        e.printStackTrace();
081                }
082
083                try {
084                        grid = reader.read(null);
085                } catch (IOException e) {
086                        e.printStackTrace();
087                }
088                gridData = grid.getRenderedImage().getData();
089        }
090        
091        
092        public double getElevation(double x, double y) {
093                return getElevation(CoordUtils.createCoord(x, y));
094        }
095        
096        
097        public double getElevation(Coord coord) {
098                GridGeometry2D gg = grid.getGridGeometry();
099                
100                Coord transformedCoord = ct.transform(coord);
101                
102                GridCoordinates2D posGrid = null;
103                try {
104                        posGrid = gg.worldToGrid(new DirectPosition2D(transformedCoord.getX(), transformedCoord.getY()));
105                } catch (InvalidGridGeometryException e) {
106                        e.printStackTrace();
107                } catch (TransformException e) {
108                        e.printStackTrace();
109                }
110
111                double[] pixel = new double[1];
112                double[] data = gridData.getPixel(posGrid.x, posGrid.y, pixel);
113                return data[0];
114        }
