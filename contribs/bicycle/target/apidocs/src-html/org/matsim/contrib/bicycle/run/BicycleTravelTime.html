019package org.matsim.contrib.bicycle.run;
020
021import org.matsim.api.core.v01.network.Link;
022import org.matsim.api.core.v01.population.Person;
023import org.matsim.contrib.bicycle.BicycleLabels;
024import org.matsim.contrib.bicycle.BicycleUtils;
025import org.matsim.core.router.util.TravelTime;
026import org.matsim.vehicles.Vehicle;
027
028/**
029 * @author dziemke
030 */
031public class BicycleTravelTime implements TravelTime {
032
033        @Override
034        public double getLinkTravelTime(Link link, double time, Person person, Vehicle vehicle) {
035                // TODO make this speed adjustable
036                double minimumSpeedForDedicatedCyclingInfrastructure = 15.0/3.6;
037                
038                String type = (String) link.getAttributes().getAttribute("type");
039                String cycleway = (String) link.getAttributes().getAttribute(BicycleLabels.CYCLEWAY);
040                double cyclingInfrastructureSpeed = computeMinimumCyclingInfrastructureSpeed(type,
041                                cycleway, minimumSpeedForDedicatedCyclingInfrastructure);
042                double infrastructureSpeed = Math.max(link.getFreespeed(), cyclingInfrastructureSpeed);
043
044                double vehicleLinkSpeed = Math.min(infrastructureSpeed, BicycleUtils.getSpeed("bicycle"));
045
046                double gradientSpeed = computeGradientSpeed(link, vehicleLinkSpeed);
047
048                String surface = (String) link.getAttributes().getAttribute(BicycleLabels.SURFACE);
049                double surfaceSpeed = vehicleLinkSpeed;
050                if (surface != null) {
051                        surfaceSpeed = computeSurfaceSpeed(vehicleLinkSpeed, surface, type);
052                }
053                
054                double effectiveSpeed = Math.min(gradientSpeed, surfaceSpeed);
055                
056                return (link.getLength() / effectiveSpeed);
057        }
058        
059        /**
060         * If there is a dedicated cycling infrastructure, it is unlikely that the speed (before later consideration
061         * of slopes) falls under a certain minimum
062         * @return
063         */
064        private double computeMinimumCyclingInfrastructureSpeed(String type, String cycleway, double minimumSpeedForDedicatedCyclingInfrastructure) {
065                double cyclingInfrastructureSpeed = 0.;
066                if (type.equals("cycleway")) {
067                        cyclingInfrastructureSpeed = minimumSpeedForDedicatedCyclingInfrastructure; // Assume that this speed is always feasible on a dedicated cycleway
068                }
069                
070                if (cycleway != null) {
071                        if (cycleway.equals("track") || cycleway.equals("track")) {
072                                cyclingInfrastructureSpeed = minimumSpeedForDedicatedCyclingInfrastructure; // Assume that this speed is always feasible on a dedicated cycleway
073                        }
074                }
075                return cyclingInfrastructureSpeed;
076        }
077
078        /**
079         * Based on "Fl√ºgel et al. -- Empirical speed models for cycling in the Oslo road network" (not yet published!)
080         * Positive gradients (uphill): Roughly linear decrease in speed with increasing gradient
081         * At 9% gradient, cyclists are 42.7% slower
082         * Negative gradients (downhill):
083         * Not linear; highest speeds at 5% or 6% gradient; at gradients higher than 6% braking
084         */
085        private double computeGradientSpeed(Link link, double vehicleLinkSpeed) {
086                double gradientSpeedFactor = 1.;
087                Double fromNodeZ = link.getFromNode().getCoord().getZ();
088                Double toNodeZ = link.getToNode().getCoord().getZ();
089                if ((fromNodeZ != null) && (toNodeZ != null)) {
090                        if (toNodeZ > fromNodeZ) { // No positive speed increase for downhill, only decrease for uphill
091                                gradientSpeedFactor = 1 - 5. * ((toNodeZ - fromNodeZ) / link.getLength()); // 50% reducation at 10% up-slope
092                        }
093                }
094                if (gradientSpeedFactor < 0.1) {
095                        gradientSpeedFactor = 0.1;
096                }
097                return vehicleLinkSpeed * gradientSpeedFactor;
098        }
099        
100        // TODO combine this with comfort
101        private double computeSurfaceSpeed(double vehicleLinkSpeed, String surface, String type) {
102                if (type.equals("cycleway")) {
103                        return vehicleLinkSpeed; // Assuming that dedicated cycleways are on good surface like ashalt
104                }
105                double surfaceSpeedFactor;
106                switch (surface) {
107                        case "paved":
108                        case "asphalt": surfaceSpeedFactor = 1.; break;
109                        case "cobblestone": surfaceSpeedFactor = 0.5; break;
110                        case "cobblestone (bad)": surfaceSpeedFactor = 0.4; break;
111                        case "cobblestone;flattened":
112                        case "cobblestone:flattened":
113                        case "sett": surfaceSpeedFactor = 0.6; break;
114                        case "concrete": surfaceSpeedFactor =  0.9; break;
115                        case "concrete:lanes":
116                        case "concrete_plates":
117                        case "concrete:plates": surfaceSpeedFactor = 0.8; break;
118                        case "paving_stones":
119                        case "paving_stones:35":
120                        case "paving_stones:30": surfaceSpeedFactor = 0.7; break;
121                        case "unpaved": surfaceSpeedFactor = 0.5; break;
122                        case "compacted": surfaceSpeedFactor = 0.9; break;
123                        case "dirt": surfaceSpeedFactor = 0.5; break;
124                        case "earth": surfaceSpeedFactor = 0.6; break;
125                        case "fine_gravel":
126                        case "gravel":
127                        case "ground": surfaceSpeedFactor = 0.7; break;
128                        case "wood": surfaceSpeedFactor =  0.5; break;
129                        case "pebblestone": surfaceSpeedFactor = 0.7; break;
130                        case "sand": surfaceSpeedFactor = 0.2; break;
131                        case "bricks":
132                        case "stone": surfaceSpeedFactor = 0.7; break;
133                        case "grass": surfaceSpeedFactor = 0.4; break;
134                        case "compressed": surfaceSpeedFactor =  0.7; break;
135                        case "asphalt;paving_stones:35": surfaceSpeedFactor = 0.9; break;
136                        case "paving_stones:3": surfaceSpeedFactor = 0.8; break;
137                        default: surfaceSpeedFactor = 0.5;
138                }
139                return vehicleLinkSpeed * surfaceSpeedFactor;
140        }
141}
