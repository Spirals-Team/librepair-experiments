049                QSimConfigGroup conf = scenario.getConfig().qsim();

054                // construct the QSim:

057                // add the activity engine:

060                qSim.addActivityHandler(activityEngine);
061
062                // add the netsim engine:

069//                              if ( vehicle.getVehicle().getType().equals( "bicycle" ) ) {
070//                                      return MixedTrafficVehiclesUtils.getSpeed("bike"); // compute bicycle speed instead
071//                              } else {

075//              });
076//              QNetsimEngine netsimEngine = new QNetsimEngine(qSim, qNetworkFactory ) ;
077
078                QNetsimEngine netsimEngine = new QNetsimEngine(qSim ) ;
079                qSim.addMobsimEngine(netsimEngine);
080                qSim.addDepartureHandler(netsimEngine.getDepartureHandler());
081
082                DefaultTeleportationEngine teleportationEngine = new DefaultTeleportationEngine(scenario, eventsManager);
083                qSim.addMobsimEngine(teleportationEngine);
084
085                AgentFactory agentFactory = new DefaultAgentFactory(qSim);
086
087                PopulationAgentSource agentSource = new PopulationAgentSource(scenario.getPopulation(), agentFactory, qSim);
088                qSim.addAgentSource(agentSource);
089
090                return qSim ;
091        }
