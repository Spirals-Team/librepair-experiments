019package org.matsim.contrib.bicycle.run;
020
021import org.apache.log4j.Logger;
022import org.matsim.api.core.v01.Id;
023import org.matsim.api.core.v01.Scenario;
024import org.matsim.api.core.v01.events.Event;
025import org.matsim.api.core.v01.events.LinkEnterEvent;
026import org.matsim.api.core.v01.events.VehicleEntersTrafficEvent;
027import org.matsim.api.core.v01.events.VehicleLeavesTrafficEvent;
028import org.matsim.api.core.v01.network.Link;
029import org.matsim.api.core.v01.population.Person;
030import org.matsim.contrib.bicycle.MotorizedInteractionEvent;
031import org.matsim.contrib.bicycle.MotorizedInteractionEventHandler;
032import org.matsim.core.events.algorithms.Vehicle2DriverEventHandler;
033import org.matsim.core.scoring.SumScoringFunction;
034import org.matsim.vehicles.Vehicle;
035
036/**
037 * @author dziemke
038 */
039public class BicycleScoring implements SumScoringFunction.ArbitraryEventScoring, MotorizedInteractionEventHandler {
040        private static final Logger LOG = Logger.getLogger(BicycleScoring.class);
041        
042        private Scenario scenario;
043        private BicycleTravelDisutility bicycleTravelDisutility;
044        private Vehicle2DriverEventHandler delegate = new Vehicle2DriverEventHandler();
045        private Id<Link> previousLink;
046        private double previousLinkRelativePosition;
047        private double previousLinkEnterTime;
048        private double score;
049        private int carCountOnLink;
050        
051        public BicycleScoring(Scenario scenario, BicycleTravelTime bicycleTravelTime, BicycleTravelDisutilityFactory bicycleTravelDisutilityFactory) {
052                                this.scenario = scenario;
053                this.bicycleTravelDisutility = (BicycleTravelDisutility) bicycleTravelDisutilityFactory.createTravelDisutility(bicycleTravelTime);
054        }
055
056        @Override public void finish() {}
057        
058        @Override
059        public double getScore() {
060                return score;
061        }
062        
063        @Override
064        public void handleEvent(Event event) {
065                if (event instanceof VehicleEntersTrafficEvent) {
066//                      LOG.warn(event.toString());
067                        VehicleEntersTrafficEvent vehEvent = (VehicleEntersTrafficEvent) event;
068                        
069                        // Establish connection between driver and vehicle
070                        delegate.handleEvent(vehEvent);
071                        
072                        // No LinkEnterEvent on first link of a leg
073                        previousLink = vehEvent.getLinkId();
074                        carCountOnLink = 0;
075                        previousLinkRelativePosition = vehEvent.getRelativePositionOnLink();
076                        previousLinkEnterTime =vehEvent.getTime();
077                        
078                }
079                if (event instanceof VehicleLeavesTrafficEvent) {
080//                      LOG.warn(event.toString());
081                        VehicleLeavesTrafficEvent vehEvent = (VehicleLeavesTrafficEvent) event;
082                        
083                        Id<Vehicle> vehId = vehEvent.getVehicleId();
084                        double enterTime = previousLinkEnterTime;
085                        double travelTime = vehEvent.getTime() - enterTime;
086                        calculateScoreForPreviousLink(vehEvent.getLinkId(), enterTime, vehId, travelTime, previousLinkRelativePosition);
087                        
088                        // End connection between driver and vehicle
089                        delegate.handleEvent(vehEvent);
090                }
091                if (event instanceof LinkEnterEvent) {
092                        // This only works because setPassLinkEventsToPerson is activated (via ScoringFunctionsForPopulation)
093                        // Otherwise ArbitraryEventScoring only handles events that are instance of HasPersonId, which is not the case for LinkEnterEvents
094//                      LOG.warn(event.toString());
095                        LinkEnterEvent linkEnterEvent = (LinkEnterEvent) event;
096                        
097                        Id<Vehicle> vehId = linkEnterEvent.getVehicleId();
098                        double enterTime = previousLinkEnterTime;
099                        double travelTime = linkEnterEvent.getTime() - enterTime;
100                        calculateScoreForPreviousLink(previousLink, enterTime, vehId, travelTime, previousLinkRelativePosition);
101                        
102                        previousLink = linkEnterEvent.getLinkId();
103                        carCountOnLink = 0;
104                        previousLinkRelativePosition = 0.;
105                        previousLinkEnterTime = linkEnterEvent.getTime();
106
107                }       
108        }
109        
110        private void calculateScoreForPreviousLink(Id<Link> linkId, Double enterTime, Id<Vehicle> vehId, double travelTime, double relativeLinkEnterPosition) {
111                if (relativeLinkEnterPosition != 1.0) {
112                        Link link = scenario.getNetwork().getLinks().get(linkId);
113                        Person person = scenario.getPopulation().getPersons().get(delegate.getDriverOfVehicle(vehId));
114                        Vehicle vehicle = scenario.getVehicles().getVehicles().get(vehId);
115                        
116                        double carScoreOffset = -(this.carCountOnLink * 0.04);
117                        this.score += carScoreOffset;
118                        LOG.warn("----- link = " + linkId + " -- car score offset = " + carScoreOffset);
119                        
120                        this.score += bicycleTravelDisutility.getTravelDisutilityBasedOnTTime(link, enterTime, person, vehicle, travelTime);
121//                      LOG.warn("score = " + score + " -- linkId = " + link.getId() + " -- enterTime = " + enterTime + " -- personId = " + person.getId() + " -- travelTime = " + travelTime);
122                }
123                else {
124                        // If agent was already at the end of the link and thus did not travel on it, do nothing
125                }
126                // TODO Use relative position in a more sophisticated way.
127        }
128
129        @Override
130        public void handleEvent(MotorizedInteractionEvent event) {
131//              LOG.info("event received from link " + event.getLinkId());
132                if (event.getLinkId().equals(previousLink)) {
133//                      LOG.warn("USED: Event from link " + event.getLinkId());
134                        this.carCountOnLink++;
135                }
136        }
137}
