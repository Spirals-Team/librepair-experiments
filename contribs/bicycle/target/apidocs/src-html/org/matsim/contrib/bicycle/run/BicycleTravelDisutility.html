038public class BicycleTravelDisutility implements TravelDisutility {
039        private static final Logger LOG = Logger.getLogger(BicycleTravelDisutility.class);

041        private final double marginalCostOfTime_s;
042        private final double marginalCostOfDistance_m;
043        private final double marginalCostOfInfrastructure_m;
044        private final double marginalCostOfComfort_m;
045        private final double marginalCostOfGradient_m_100m;
046        private final double sigma;
047        private final TravelTime timeCalculator;
048        private final Network network;
049
050        private static int normalisationWrnCnt = 0;
051
052        BicycleTravelDisutility(BicycleConfigGroup bicycleConfigGroup, PlanCalcScoreConfigGroup cnScoringGroup,
053                        PlansCalcRouteConfigGroup plansCalcRouteConfigGroup, TravelTime timeCalculator, Network network) {
054                final PlanCalcScoreConfigGroup.ModeParams bicycleParams = cnScoringGroup.getModes().get("bicycle");
055                if (bicycleParams == null) {
056                        throw new NullPointerException("Bicycle is not part of the valid mode parameters " + cnScoringGroup.getModes().keySet());
057                }
058
059                this.marginalCostOfDistance_m = -(bicycleParams.getMonetaryDistanceRate() * cnScoringGroup.getMarginalUtilityOfMoney())
060                                - bicycleParams.getMarginalUtilityOfDistance();
061                this.marginalCostOfTime_s = -(bicycleParams.getMarginalUtilityOfTraveling() / 3600.0) + cnScoringGroup.getPerforming_utils_hr() / 3600.0;
062
063                this.marginalCostOfInfrastructure_m = -(bicycleConfigGroup.getMarginalUtilityOfInfrastructure_m());
064                this.marginalCostOfComfort_m = -(bicycleConfigGroup.getMarginalUtilityOfComfort_m());
065                this.marginalCostOfGradient_m_100m = -(bicycleConfigGroup.getMarginalUtilityOfGradient_m_100m());
066
067                // Does not seem to be implemented yet
068                // this.sigma = plansCalcRouteConfigGroup.getRoutingRandomness();
069                this.sigma = 0.2;
070                //this.sigma = 0.0;
071
072                this.timeCalculator = timeCalculator;
073                
074                // TODO only needed as long as network mode filtering kicks out attributes; remove when possible, dz, sep'17
075                this.network = network;
076        }
077
078        @Override
079        public double getLinkTravelDisutility(Link link, double time, Person person, Vehicle vehicle) {
080                Link linkWithAttributes = network.getLinks().get(link.getId());
081                double travelTime = timeCalculator.getLinkTravelTime(linkWithAttributes, time, person, vehicle);
082                return getTravelDisutilityBasedOnTTime(linkWithAttributes, time, person, vehicle, travelTime);
083        }
084
085        public double getTravelDisutilityBasedOnTTime(Link link, double enterTime, Person person, Vehicle vehicle, double travelTime) {
086                String surface = (String) link.getAttributes().getAttribute(BicycleLabels.SURFACE);
087                String type = (String) link.getAttributes().getAttribute("type");
088                String cyclewaytype = (String) link.getAttributes().getAttribute(BicycleLabels.CYCLEWAY);
089
090                double distance = link.getLength();
091                
092                double travelTimeDisutility = marginalCostOfTime_s * travelTime;
093                double distanceDisutility = marginalCostOfDistance_m * distance;
094                
095                double comfortFactor = getComfortFactor(surface, type);
096                double comfortDisutility = marginalCostOfComfort_m * (1. - comfortFactor) * distance;
097                
098                double infrastructureFactor = getInfrastructureFactor(type, cyclewaytype);
099                double infrastructureDisutility = marginalCostOfInfrastructure_m * (1. - infrastructureFactor) * distance;
100                
101                double gradientFactor = getGradientFactor(link);
102                double gradientDisutility = marginalCostOfGradient_m_100m * gradientFactor * distance;
103                
104//              LOG.warn("link = " + link.getId() + "-- travelTime = " + travelTime + " -- distance = " + distance + " -- comfortFactor = "
105//                              + comfortFactor + " -- infraFactor = "+ infrastructureFactor + " -- gradient = " + gradientFactor);
106                 
107                // TODO Gender
108                // TODO Activity
109                // TODO Other influence factors
110
111                 double normalization = 1;
112                 if (sigma != 0.) {
113                         normalization = 1. / Math.exp(this.sigma * this.sigma / 2);
114                         if (normalisationWrnCnt < 10) {
115                                 normalisationWrnCnt++;
116                                 LOG.info("Sigma = " + this.sigma + " -- resulting normalization: " + normalization);
117                         }
118                }
119                Random random2 = MatsimRandom.getLocalInstance(); // Make sure that stream of random variables is reproducible. dz, aug'17
120                double logNormalRnd = Math.exp(sigma * random2.nextGaussian());
121                logNormalRnd *= normalization;
122
123//              LOG.warn("link = " + link.getId() + " -- travelTimeDisutility = " + travelTimeDisutility + " -- distanceDisutility = "+ distanceDisutility
124//                              + " -- infrastructureDisutility = " + infrastructureDisutility + " -- comfortDisutility = "
125//                              + comfortDisutility + " -- gradientDisutility = " + gradientDisutility + " -- randomfactor = " + logNormalRnd);
126                return (travelTimeDisutility + logNormalRnd * (distanceDisutility + infrastructureDisutility + comfortDisutility + gradientDisutility));
127        }
128
129        private double getGradientFactor(Link link) {
130                double gradient = 0.;
131                Double fromNodeZ = link.getFromNode().getCoord().getZ();
132                Double toNodeZ = link.getToNode().getCoord().getZ();
133                if ((fromNodeZ != null) && (toNodeZ != null)) {
134                        if (toNodeZ > fromNodeZ) { // No positive utility for downhill, only negative for uphill
135                                gradient = (toNodeZ - fromNodeZ) / link.getLength();
136                        }
137                }
138                return gradient;
139        }

141        // TODO combine this with speeds
142        private double getComfortFactor(String surface, String type) {
143                double comfortFactor = 1.0;
144                if (surface != null) {
145                        switch (surface) {
146                        case "paved":
147                        case "asphalt": comfortFactor = 1.0; break;
148                        case "cobblestone": comfortFactor = .40; break;
149                        case "cobblestone (bad)": comfortFactor = .30; break;
150                        case "sett": comfortFactor = .50; break;
151                        case "cobblestone;flattened":
152                        case "cobblestone:flattened": comfortFactor = .50; break;
153                        case "concrete": comfortFactor = .100; break;
154                        case "concrete:lanes": comfortFactor = .95; break;
155                        case "concrete_plates":
156                        case "concrete:plates": comfortFactor = .90; break;
157                        case "paving_stones": comfortFactor = .80; break;
158                        case "paving_stones:35":
159                        case "paving_stones:30": comfortFactor = .80; break;
160                        case "unpaved": comfortFactor = .60; break;
161                        case "compacted": comfortFactor = .70; break;
162                        case "dirt":
163                        case "earth": comfortFactor = .30; break;
164                        case "fine_gravel": comfortFactor = .90; break;
165                        case "gravel":
166                        case "ground": comfortFactor = .60; break;
167                        case "wood":
168                        case "pebblestone":
169                        case "sand": comfortFactor = .30; break;
170                        case "bricks": comfortFactor = .60; break;
171                        case "stone":
172                        case "grass":
173                        case "compressed": comfortFactor = .40; break;
174                        case "asphalt;paving_stones:35": comfortFactor = .60; break;
175                        case "paving_stones:3": comfortFactor = .40; break;
176                        default: comfortFactor = .85;
177                        }
178                } else {
179                        // For many primary and secondary roads, no surface is specified because they are by default assumed to be is asphalt.
180                        // For tertiary roads street this is not true, e.g. Friesenstr. in Kreuzberg
181                        if (type != null) {
182                                if (type.equals("primary") || type.equals("primary_link") || type.equals("secondary") || type.equals("secondary_link")) {
183                                        comfortFactor = 1.0;
184                                }
185                        }
186                }
187                return comfortFactor;
188        }
189
190        private double getInfrastructureFactor(String type, String cyclewaytype) {
191                double infrastructureFactor = 1.0;
192                if (type != null) {
193                        if (type.equals("trunk")) {
194                                if (cyclewaytype == null || cyclewaytype.equals("no") || cyclewaytype.equals("none")) { // No cycleway
195                                        infrastructureFactor = .05;
196                                } else { // Some kind of cycleway
197                                        infrastructureFactor = .95;
198                                }
199                        } else if (type.equals("primary") || type.equals("primary_link")) {
200                                if (cyclewaytype == null || cyclewaytype.equals("no") || cyclewaytype.equals("none")) { // No cycleway
201                                        infrastructureFactor = .10;
202                                } else { // Some kind of cycleway
203                                        infrastructureFactor = .95;
204                                }
205                        } else if (type.equals("secondary") || type.equals("secondary_link")) {
206                                if (cyclewaytype == null || cyclewaytype.equals("no") || cyclewaytype.equals("none")) { // No cycleway
207                                        infrastructureFactor = .30;
208                                } else { // Some kind of cycleway
209                                        infrastructureFactor = .95;
210                                }
211                        } else if (type.equals("tertiary") || type.equals("tertiary_link")) {
212                                if (cyclewaytype == null || cyclewaytype.equals("no") || cyclewaytype.equals("none")) { // No cycleway
213                                        infrastructureFactor = .40;
214                                } else { // Some kind of cycleway
215                                        infrastructureFactor = .95;
216                                }
217                        } else if (type.equals("unclassified")) {
218                                if (cyclewaytype == null || cyclewaytype.equals("no") || cyclewaytype.equals("none")) { // No cycleway
219                                        infrastructureFactor = .90;
220                                } else { // Some kind of cycleway
221                                        infrastructureFactor = .95;
222                                }
223                        } else if (type.equals("unclassified")) {
224                                infrastructureFactor = .95;
225                        } else if (type.equals("service") || type.equals("living_street") || type.equals("minor")) {
226                                infrastructureFactor = .95;
227                        } else if (type.equals("cycleway") || type.equals("path")) {
228                                infrastructureFactor = 1.00;
229                        } else if (type.equals("footway") || type.equals("track") || type.equals("pedestrian")) {
230                                infrastructureFactor = .95;
231                        } else if (type.equals("steps")) {
232                                infrastructureFactor = .10;
233                        }
234                } else {
235                        infrastructureFactor = .85;
236                }
237                return infrastructureFactor;
238        }
