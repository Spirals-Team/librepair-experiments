020package org.matsim.contrib.bicycle.run;
021
022import org.matsim.api.core.v01.Scenario;
023import org.matsim.api.core.v01.population.Person;
024import org.matsim.contrib.bicycle.MotorizedInteractionEvent;
025import org.matsim.contrib.bicycle.MotorizedInteractionEventHandler;
026import org.matsim.core.api.experimental.events.EventsManager;
027import org.matsim.core.scoring.ScoringFunction;
028import org.matsim.core.scoring.ScoringFunctionFactory;
029import org.matsim.core.scoring.ScoringFunctionsForPopulation;
030import org.matsim.core.scoring.SumScoringFunction;
031import org.matsim.core.scoring.functions.CharyparNagelActivityScoring;
032import org.matsim.core.scoring.functions.CharyparNagelAgentStuckScoring;
033import org.matsim.core.scoring.functions.CharyparNagelLegScoring;
034import org.matsim.core.scoring.functions.ScoringParameters;
035import org.matsim.core.scoring.functions.ScoringParametersForPerson;
036
037import com.google.inject.Inject;
038
039/**
040 * @author dziemke
041 */
042public class BicycleScoringFunctionFactory implements ScoringFunctionFactory {
043        @Inject ScoringParametersForPerson parameters;
044
045        @Inject ScoringFunctionsForPopulation scoringFunctionsForPopulation;
046        
047        
048        @Inject Scenario scenario;
049        @Inject BicycleTravelTime bicycleTravelTime;
050        @Inject BicycleTravelDisutilityFactory bicycleTravelDisutilityFactory;
051        
052        @Inject EventsManager eventsManager;
053
054
055        @Override
056        public ScoringFunction createNewScoringFunction(Person person) {                
057                SumScoringFunction sumScoringFunction = new SumScoringFunction();
058
059                final ScoringParameters params = parameters.getScoringParameters(person);
060                // No leg scoring here as it would only double-count
061                // sumScoringFunction.addScoringFunction(new CharyparNagelLegScoring(params, scenario.getNetwork()));
062                sumScoringFunction.addScoringFunction(new CharyparNagelActivityScoring(params)) ;
063                sumScoringFunction.addScoringFunction(new CharyparNagelAgentStuckScoring(params));
064
065                scoringFunctionsForPopulation.setPassLinkEventsToPerson(true);
066                // yyyyyy this is the ONLY place where we need to make ScoringFunctionsForPopulation public.  Somehow,
067                // this would imply to me that we should attach this switch to something else. kai, sep'17
068                
069                BicycleScoring bicycleScoring = new BicycleScoring(scenario, bicycleTravelTime, bicycleTravelDisutilityFactory);
070                sumScoringFunction.addScoringFunction(bicycleScoring);
071                
072                CarCounter carCounter = new CarCounter(bicycleScoring);
073                eventsManager.addHandler(carCounter);
074
075                return sumScoringFunction;
076        }
077        
078        private class CarCounter implements MotorizedInteractionEventHandler {
079                private BicycleScoring bicycleScoring;
080
081                public CarCounter(BicycleScoring bicycleScoring) {
082                        this.bicycleScoring = bicycleScoring;
083                }
084
085                @Override
086                public void handleEvent(MotorizedInteractionEvent event) {
087                        bicycleScoring.handleEvent(event);
088                }
089        }
090}
