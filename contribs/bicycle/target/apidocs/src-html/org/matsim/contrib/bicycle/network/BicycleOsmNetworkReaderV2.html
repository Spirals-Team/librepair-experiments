114                public BicycleOsmNetworkReaderV2(final Network network, final CoordinateTransformation transformation, ElevationDataParser elevationDataParser,
115                        final boolean useHighwayDefaults, String bicycleAsTransportModeName, double bicyclePCU, boolean useVspAdjustments) {
116                // If "useHighwayDefaults" is set to true, super sets defaults for all "roads" ("motorway" to "residential", except "service")
117                // and all "link roads" and "living_street" (part of "special road types"). Hierachies 1 to 6 are used.
118                super(network, transformation, useHighwayDefaults, useVspAdjustments);
119                super.addWayTags(bicycleWayTags);
120                
121//              double bicyclePCU = 0.2; // Use the same in your run
122                        this.bicycleAsTransportModeName = bicycleAsTransportModeName;
123                
124                if (useHighwayDefaults) {
125                        LOG.info("Also falling back to bicycle-specific default values.");
126                        // All highway types whose defaults are no set via the super class, see above.
127                        // Cf. http://wiki.openstreetmap.org/wiki/Key:highway
128                        // hierarchy , highwayType, lanesPerDirection, freespeed, freespeedFactor, laneCapacity_vehPerHour
129                        
130                        // Capacity of a track for bicycles analogous to that of a primary road for cars, dz, aug'18
131                        // this.setHighwayDefaults(7, "track",  1, 10.0/3.6, 1.0, 50); // Simon's values
132                        this.setHighwayDefaults(BICYCLE_TRACK_HIERARCHY, "track", 1, 30.0/3.6, 1.0, 1500 * bicyclePCU);
133                        
134                        //this.setHighwayDefaults(7, "cycleway", 1, 10.0/3.6, 1.0, 50); // Simon's values
135                        this.setHighwayDefaults(BICYCLE_TRACK_HIERARCHY, "cycleway", 1, 30.0/3.6, 1.0, 1500 * bicyclePCU);
136                        
137                        // this.setHighwayDefaults(8, "service", 1, 10.0/3.6, 1.0, 50); // Simon's values
138                        this.setHighwayDefaults(BICYCLE_TRACK_HIERARCHY, "service", 1, 10.0/3.6, 1.0, 1000 * bicyclePCU);
139                        
140                        // Capacity of a footway/pedestrian/path zone for bicycles analogous to that of a residential road for cars, dz, aug'18
141                        // this.setHighwayDefaults(8, "footway", 1, 10.0/3.6, 1.0, 50); // Simon's values
142                        this.setHighwayDefaults(FOOTPATH_TRACK_HIERARCHY, "footway", 1, 10.0/3.6, 1.0, 600 * bicyclePCU); // Simon's values
143                        // this.setHighwayDefaults(8, "pedestrian", 1, 10.0/3.6, 1.0, 50); // Simon's values
144                        this.setHighwayDefaults(FOOTPATH_TRACK_HIERARCHY, "pedestrian", 1, 10.0/3.6, 1.0, 600 * bicyclePCU); // Simon's values
145                        // this.setHighwayDefaults(8, "path", 1, 10.0/3.6, 1.0, 50); // Simon's values
146                        this.setHighwayDefaults(FOOTPATH_TRACK_HIERARCHY, "path", 1, 20.0/3.6, 1.0, 600 * bicyclePCU);
147                        
148                        // "bridleway" is exclusively for equestrians; if others are allowed to use it, it should not be classified as bridleway
149                        
150                        // this.setHighwayDefaults(10, "steps", 1, 2.0/3.6, 1.0, 50); // Simon's values
151                        this.setHighwayDefaults(STEPS_TRACK_HIERARCHY, "steps", 1, 1.0/3.6, 1.0, 50);
152                } else {
153                        LOG.error("The way this is written assumes that highway defaults are used! This is, however, not done here."
154                                        + "Reconsider this. It will likely lead to awkward resuklts otherwise.");
155                }
156
157                if (elevationDataParser==null){
158                        LOG.warn("Elevation parser is not provided, thus, elevation can not be computed.");
159                } else {
160                        this.elevationDataParser = elevationDataParser;
161                }
162        }

164        private void stats(Network network) {
165                LOG.info("Smoothness created: " + countSmoothness + " which is " + ((float) countSmoothness / network.getLinks().size()) * 100 + "%.");
166                LOG.info("Surface (direct) created: " + countSurfaceDirect  + " which is " + ((float) countSurfaceDirect / network.getLinks().size()) * 100 + "%.");
167                LOG.info("Surface (inferred) created: " + countSurfaceInferred  + " which is " + ((float) countSurfaceInferred / network.getLinks().size()) * 100 + "%.");
168//              LOG.info("Highway type created: " + countHighwayType + " which is " + ((float) countHighwayType / network.getLinks().size()) * 100 + "%.");
169                LOG.info("Cycleway type created: " + countCyclewayType + " which is " + ((float) countCyclewayType / network.getLinks().size()) * 100 + "%.");
170                LOG.info("Bicyle created: " + countBicycle + " which is " + ((float) countBicycle / network.getLinks().size()) * 100 + "%.");
171        }

173        @Override
174        protected void setOrModifyNodeAttributes(Node n, OsmNode node) {
175                if (this.elevationDataParser==null) return;
176                Coord coord = n.getCoord();
177                double elevation = elevationDataParser.getElevation(n.getCoord());
178                Coord elevationCoord = CoordUtils.createCoord(coord.getX(), coord.getY(), elevation);
179                n.setCoord(elevationCoord);
180        }

182        @Override
183        protected void setOrModifyLinkAttributes(Link l, OsmWay way, boolean forwardDirection) {
184                Set<String> modes = new HashSet<String>();
185                
186                String highwayType = way.tags.get(TAG_HIGHWAY);
187                OsmHighwayDefaults defaults = this.highwayDefaults.get(way.tags.get(TAG_HIGHWAY));
188                
189                // Allow bicycles on all infrastructures except motorways (hierarchy 1) and trunk roads (hierarchy 2)
190//              if (defaults.hierarchy == 3 || defaults.hierarchy == 4 || defaults.hierarchy == 5 || defaults.hierarchy == 6
191//                      || defaults.hierarchy == 7 || defaults.hierarchy == 8 || defaults.hierarchy == 9) {
192                if (! (defaults.hierarchy == 1 || defaults.hierarchy == 2) ) {
193                        modes.add(this.bicycleAsTransportModeName); // TODO add TransportMode "bicycle"
194                }
195
196                // Allow cars if the direction under consideration is open to cars and if we are not on an infrastructure mainly intended
197                // for bicyclists (hierarchy 7) or pedestrians (hierarchy 8)
198                /* use of following condition (direction/oneway) will set different allowed modes on links which are continuous (continuous hierarchy)
199                 * this mean, a continuous routes will not be available and router may throw an exception. Amit Feb'18
200                  */
201//              if ( (forwardDirection && !isOnewayReverse(way)) || (!forwardDirection && !isOneway(way)) ) {
202                // above condition is commented but one can discuss it again. AA, DZ. Feb'18
203                        if (defaults.hierarchy != BICYCLE_TRACK_HIERARCHY
204                                        && defaults.hierarchy != FOOTPATH_TRACK_HIERARCHY
205                                        && defaults.hierarchy != STEPS_TRACK_HIERARCHY) {
206                                modes.add(TransportMode.car);
207                        }
208//              }
209                l.setAllowedModes(modes);
210
211                if (  this.elevationDataParser!=null ){
212                        // Gradient
213//              double gradient = (l.getToNode().getCoord().getZ() - l.getFromNode().getCoord().getZ()) / l.getLength();
214//              l.getAttributes().putAttribute(BicycleLabels.GRADIENT, gradient);
215
216                        // Elevation
217                        double averageElevation = (l.getToNode().getCoord().getZ() + l.getFromNode().getCoord().getZ()) / 2;
218                        l.getAttributes().putAttribute(BicycleLabels.AVERAGE_ELEVATION, averageElevation);
219                }
220
221                // Smoothness
222                String smoothness = way.tags.get(BicycleLabels.SMOOTHNESS);
223                if (smoothness != null) {
224                        l.getAttributes().putAttribute(BicycleLabels.SMOOTHNESS, smoothness);
225                        this.countSmoothness++;
226                }
227
228                // Surface
229                String surface = way.tags.get(BicycleLabels.SURFACE);
230                if (surface != null) {
231                        l.getAttributes().putAttribute(BicycleLabels.SURFACE, surface);
232                        this.countSurfaceDirect++;
233                } else {
234                        if (highwayType != null) {
235                                // it used to be '&&' instead of '||' which will always be false. Most likely, it must be '||'. Amit Feb'18
236                                if (defaults.hierarchy == 3 || defaults.hierarchy == 4) { // 3 = primary, 4 = secondary
237                                        l.getAttributes().putAttribute(BicycleLabels.SURFACE, "asphalt");
238                                        this.countSurfaceInferred++;
239                                } else {
240                                        if (warnCount <= maxWarnCount){
241                                                LOG.warn("Link did not get a surface.");
242                                                LOG.warn(Gbl.FUTURE_SUPPRESSED);
243                                                warnCount++;
244                                        }
245                                }
246                        } else {
247                                if (warnCount <= maxWarnCount ){
248                                        LOG.warn("Link did not get a surface.");
249                                        LOG.warn(Gbl.FUTURE_SUPPRESSED);
250                                        warnCount++;
251                                }
252                        }
253                }
254
255                // Cycleway
256                String cyclewayType = way.tags.get(BicycleLabels.CYCLEWAY);
257                if (cyclewayType != null) {
258                        l.getAttributes().putAttribute(BicycleLabels.CYCLEWAY, cyclewayType);
259                        this.countCyclewayType++;
260                }
261
262                // Bicycle restrictions
263                String bicycle = way.tags.get(TAG_BICYCLE);
264                if (bicycle != null) {
265                        l.getAttributes().putAttribute(this.bicycleAsTransportModeName, bicycle);
266                        countBicycle++;
267                }
268        }

270        @Override
271        protected boolean reverseDirectionExists(OsmWay way) {
272                return !isOneway(way) || oppositeDirectionAllowedForBicycles(way);
273        }
274        
275        @Override
276        protected boolean forwardDirectionExists(OsmWay way) {
277                return !isOnewayReverse(way) || oppositeDirectionAllowedForBicycles(way);
278        }
279
280        private boolean oppositeDirectionAllowedForBicycles(OsmWay way) {
281                boolean reverseDirectionAllowedForBicycles = false;
282                String onewayTagBicycle = way.tags.get(TAG_ONEWAYBICYCLE);
283                if (onewayTagBicycle != null) {
284                        if ("no".equals(onewayTagBicycle)) { // This means cyclists are allowed to travel in the opposite direction
285                                reverseDirectionAllowedForBicycles = true;
286                        } 
287                }
288                String cyclewayType = way.tags.get(BicycleLabels.CYCLEWAY);
289                if (cyclewayType != null) {
290                        if ("opposite".equals(cyclewayType) || "opposite_track".equals(cyclewayType) || "opposite_lane".equals(cyclewayType)) {
291                                reverseDirectionAllowedForBicycles = true;
292                        }
293                }
294                return reverseDirectionAllowedForBicycles;
295        }
