108                public BicycleOsmNetworkReaderV2(final Network network, final CoordinateTransformation transformation, ElevationDataParser elevationDataParser,
109                        final boolean useHighwayDefaults, String bicycleAsTransportModeName, double bicyclePCU) {
110                // If "useHighwayDefaults" is set to true, super sets defaults for all "roads" ("motorway" to "residential", except "service")
111                // and all "link roads" and "living_street" (part of "special road types"). Hierachies 1 to 6 are used.
112                super(network, transformation, useHighwayDefaults);
113                super.addWayTags(bicycleWayTags);
114                
115//              double bicyclePCU = 0.2; // Use the same in your run
116                        this.bicycleAsTransportModeName = bicycleAsTransportModeName;
117                
118                if (useHighwayDefaults) {
119                        LOG.info("Also falling back to bicycle-specific default values.");
120                        // All highway types whose defaults are no set via the super class, see above.
121                        // Cf. http://wiki.openstreetmap.org/wiki/Key:highway
122                        // hierarchy , highwayType, lanesPerDirection, freespeed, freespeedFactor, laneCapacity_vehPerHour
123                        
124                        // Capacity of a track for bicycles analogous to that of a primary road for cars, dz, aug'18
125                        // this.setHighwayDefaults(7, "track",  1, 10.0/3.6, 1.0, 50); // Simon's values
126                        this.setHighwayDefaults(BICYCLE_TRACK_HIERARCHY, "track", 1, 30.0/3.6, 1.0, 1500 * bicyclePCU);
127                        
128                        //this.setHighwayDefaults(7, "cycleway", 1, 10.0/3.6, 1.0, 50); // Simon's values
129                        this.setHighwayDefaults(BICYCLE_TRACK_HIERARCHY, "cycleway", 1, 30.0/3.6, 1.0, 1500 * bicyclePCU);
130                        
131                        // this.setHighwayDefaults(8, "service", 1, 10.0/3.6, 1.0, 50); // Simon's values
132                        this.setHighwayDefaults(BICYCLE_TRACK_HIERARCHY, "service", 1, 10.0/3.6, 1.0, 1000 * bicyclePCU);
133                        
134                        // Capacity of a footway/pedestrian/path zone for bicycles analogous to that of a residential road for cars, dz, aug'18
135                        // this.setHighwayDefaults(8, "footway", 1, 10.0/3.6, 1.0, 50); // Simon's values
136                        this.setHighwayDefaults(FOOTPATH_TRACK_HIERARCHY, "footway", 1, 10.0/3.6, 1.0, 600 * bicyclePCU); // Simon's values
137                        // this.setHighwayDefaults(8, "pedestrian", 1, 10.0/3.6, 1.0, 50); // Simon's values
138                        this.setHighwayDefaults(FOOTPATH_TRACK_HIERARCHY, "pedestrian", 1, 10.0/3.6, 1.0, 600 * bicyclePCU); // Simon's values
139                        // this.setHighwayDefaults(8, "path", 1, 10.0/3.6, 1.0, 50); // Simon's values
140                        this.setHighwayDefaults(FOOTPATH_TRACK_HIERARCHY, "path", 1, 20.0/3.6, 1.0, 600 * bicyclePCU);
141                        
142                        // "bridleway" is exclusively for equestrians; if others are allowed to use it, it should not be classified as bridleway
143                        
144                        // this.setHighwayDefaults(10, "steps", 1, 2.0/3.6, 1.0, 50); // Simon's values
145                        this.setHighwayDefaults(STEPS_TRACK_HIERARCHY, "steps", 1, 1.0/3.6, 1.0, 50);
146                } else {
147                        LOG.error("The way this is written assumes that highway defaults are used! This is, however, not done here."
148                                        + "Reconsider this. It will likely lead to awkward resuklts otherwise.");
149                }
150
151                if (elevationDataParser==null){
152                        LOG.warn("Elevation parser is not provided, thus, elevation can not be computed.");
153                } else {
154                        this.elevationDataParser = elevationDataParser;
155                }
156        }

158        private void stats(Network network) {
159                LOG.info("Smoothness created: " + countSmoothness + " which is " + ((float) countSmoothness / network.getLinks().size()) * 100 + "%.");
160                LOG.info("Surface (direct) created: " + countSurfaceDirect  + " which is " + ((float) countSurfaceDirect / network.getLinks().size()) * 100 + "%.");
161                LOG.info("Surface (inferred) created: " + countSurfaceInferred  + " which is " + ((float) countSurfaceInferred / network.getLinks().size()) * 100 + "%.");
162//              LOG.info("Highway type created: " + countHighwayType + " which is " + ((float) countHighwayType / network.getLinks().size()) * 100 + "%.");
163                LOG.info("Cycleway type created: " + countCyclewayType + " which is " + ((float) countCyclewayType / network.getLinks().size()) * 100 + "%.");
164                LOG.info("Bicyle created: " + countBicycle + " which is " + ((float) countBicycle / network.getLinks().size()) * 100 + "%.");
165        }

167        @Override
168        protected void setOrModifyNodeAttributes(Node n, OsmNode node) {
169                if (this.elevationDataParser==null) return;
170                Coord coord = n.getCoord();
171                double elevation = elevationDataParser.getElevation(n.getCoord());
172                Coord elevationCoord = CoordUtils.createCoord(coord.getX(), coord.getY(), elevation);
173                n.setCoord(elevationCoord);
174        }

176        @Override
177        protected void setOrModifyLinkAttributes(Link l, OsmWay way, boolean forwardDirection) {
178                Set<String> modes = new HashSet<String>();
179                
180                String highwayType = way.tags.get(TAG_HIGHWAY);
181                OsmHighwayDefaults defaults = this.highwayDefaults.get(way.tags.get(TAG_HIGHWAY));
182                
183                // Allow bicycles on all infrastructures except motorways (hierarchy 1) and trunk roads (hierarchy 2)
184//              if (defaults.hierarchy == 3 || defaults.hierarchy == 4 || defaults.hierarchy == 5 || defaults.hierarchy == 6
185//                      || defaults.hierarchy == 7 || defaults.hierarchy == 8 || defaults.hierarchy == 9) {
186                if (! (defaults.hierarchy == 1 || defaults.hierarchy == 2) ) {
187                        modes.add(this.bicycleAsTransportModeName); // TODO add TransportMode "bicycle"
188                }
189
190                // Allow cars if the direction under consideration is open to cars and if we are not on an infrastructure mainly intended
191                // for bicyclists (hierarchy 7) or pedestrians (hierarchy 8)
192                /* use of following condition (direction/oneway) will set different allowed modes on links which are continuous (continuous hierarchy)
193                 * this mean, a continuous routes will not be available and router may throw an exception. Amit Feb'18
194                  */
195//              if ( (forwardDirection && !isOnewayReverse(way)) || (!forwardDirection && !isOneway(way)) ) {
196                // above condition is commented but one can discuss it again. AA, DZ. Feb'18
197                        if (defaults.hierarchy != BICYCLE_TRACK_HIERARCHY
198                                        && defaults.hierarchy != FOOTPATH_TRACK_HIERARCHY
199                                        && defaults.hierarchy != STEPS_TRACK_HIERARCHY) {
200                                modes.add(TransportMode.car);
201                        }
202//              }
203                l.setAllowedModes(modes);
204
205                if (  this.elevationDataParser!=null ){
206                        // Gradient
207//              double gradient = (l.getToNode().getCoord().getZ() - l.getFromNode().getCoord().getZ()) / l.getLength();
208//              l.getAttributes().putAttribute(BicycleLabels.GRADIENT, gradient);
209
210                        // Elevation
211                        double averageElevation = (l.getToNode().getCoord().getZ() + l.getFromNode().getCoord().getZ()) / 2;
212                        l.getAttributes().putAttribute(BicycleLabels.AVERAGE_ELEVATION, averageElevation);
213                }
214
215                // Smoothness
216                String smoothness = way.tags.get(BicycleLabels.SMOOTHNESS);
217                if (smoothness != null) {
218                        l.getAttributes().putAttribute(BicycleLabels.SMOOTHNESS, smoothness);
219                        this.countSmoothness++;
220                }
221
222                // Surface
223                String surface = way.tags.get(BicycleLabels.SURFACE);
224                if (surface != null) {
225                        l.getAttributes().putAttribute(BicycleLabels.SURFACE, surface);
226                        this.countSurfaceDirect++;
227                } else {
228                        if (highwayType != null) {
229                                // it used to be '&&' instead of '||' which will always be false. Most likely, it must be '||'. Amit Feb'18
230                                if (defaults.hierarchy == 3 || defaults.hierarchy == 4) { // 3 = primary, 4 = secondary
231                                        l.getAttributes().putAttribute(BicycleLabels.SURFACE, "asphalt");
232                                        this.countSurfaceInferred++;
233                                } else {
234                                        if (warnCount <= maxWarnCount){
235                                                LOG.warn("Link did not get a surface.");
236                                                LOG.warn(Gbl.FUTURE_SUPPRESSED);
237                                                warnCount++;
238                                        }
239                                }
240                        } else {
241                                if (warnCount <= maxWarnCount ){
242                                        LOG.warn("Link did not get a surface.");
243                                        LOG.warn(Gbl.FUTURE_SUPPRESSED);
244                                        warnCount++;
245                                }
246                        }
247                }
248
249                // Cycleway
250                String cyclewayType = way.tags.get(BicycleLabels.CYCLEWAY);
251                if (cyclewayType != null) {
252                        l.getAttributes().putAttribute(BicycleLabels.CYCLEWAY, cyclewayType);
253                        this.countCyclewayType++;
254                }
255
256                // Bicycle restrictions
257                String bicycle = way.tags.get(TAG_BICYCLE);
258                if (bicycle != null) {
259                        l.getAttributes().putAttribute(this.bicycleAsTransportModeName, bicycle);
260                        countBicycle++;
261                }
262        }

264        @Override
265        protected boolean reverseDirectionExists(OsmWay way) {
266                return !isOneway(way) || oppositeDirectionAllowedForBicycles(way);
267        }
268        
269        @Override
270        protected boolean forwardDirectionExists(OsmWay way) {
271                return !isOnewayReverse(way) || oppositeDirectionAllowedForBicycles(way);
272        }
273
274        private boolean oppositeDirectionAllowedForBicycles(OsmWay way) {
275                boolean reverseDirectionAllowedForBicycles = false;
276                String onewayTagBicycle = way.tags.get(TAG_ONEWAYBICYCLE);
277                if (onewayTagBicycle != null) {
278                        if ("no".equals(onewayTagBicycle)) { // This means cyclists are allowed to travel in the opposite direction
279                                reverseDirectionAllowedForBicycles = true;
280                        } 
281                }
282                String cyclewayType = way.tags.get(BicycleLabels.CYCLEWAY);
283                if (cyclewayType != null) {
284                        if ("opposite".equals(cyclewayType) || "opposite_track".equals(cyclewayType) || "opposite_lane".equals(cyclewayType)) {
285                                reverseDirectionAllowedForBicycles = true;
286                        }
287                }
288                return reverseDirectionAllowedForBicycles;
289        }
