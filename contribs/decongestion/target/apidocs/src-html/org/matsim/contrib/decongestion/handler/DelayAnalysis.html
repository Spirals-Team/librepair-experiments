020package org.matsim.contrib.decongestion.handler;
021
022import java.util.HashMap;
023import java.util.Map;
024
025import org.apache.log4j.Logger;
026import org.matsim.api.core.v01.Id;
027import org.matsim.api.core.v01.Scenario;
028import org.matsim.api.core.v01.events.LinkEnterEvent;
029import org.matsim.api.core.v01.events.LinkLeaveEvent;
030import org.matsim.api.core.v01.events.PersonArrivalEvent;
031import org.matsim.api.core.v01.events.handler.LinkEnterEventHandler;
032import org.matsim.api.core.v01.events.handler.LinkLeaveEventHandler;
033import org.matsim.api.core.v01.events.handler.PersonArrivalEventHandler;
034import org.matsim.vehicles.Vehicle;
035
036import com.google.inject.Inject;
037
038/**
039 * 
040 * Computes the total delay and travel time.
041 * 
042 * WARNING: Link-based analysis. Ignores the travel time and delay on the first link of each trip, i.e. the link on which the trip has started.
043 * 
044 * @author ikaddoura
045 */
046
047public class DelayAnalysis implements LinkEnterEventHandler, LinkLeaveEventHandler, PersonArrivalEventHandler {
048        private static final Logger log = Logger.getLogger(DelayAnalysis.class);
049
050        private Map<Id<Vehicle>, Double> vehicleId2enterTime = new HashMap<>();
051        
052        @Inject
053        private Scenario scenario;
054        
055        // some aggregated numbers for analysis purposes
056        private double totalDelayPerDay_sec = 0.;
057        private double totalTravelTimePerDay_sec = 0.;
058        private int warnCnt = 0;
059        private int warnCnt2 = 0;
060
061        public void setScenario(Scenario scenario) {
062                this.scenario = scenario;
063        }
064
065        @Override
066        public void reset(int iteration) {
067                this.totalDelayPerDay_sec = 0.;
068                this.totalTravelTimePerDay_sec = 0.;
069                this.vehicleId2enterTime.clear();
070        }
071
072        @Override
073        public void handleEvent(LinkLeaveEvent event) {
074                
075                if (this.vehicleId2enterTime.get(event.getVehicleId()) != null) {
076                        
077                        // compute the travel time
078                        double traveltimeThisAgent = event.getTime() - this.vehicleId2enterTime.get(event.getVehicleId());                      
079                        double freespeedTravelTime = 1 + Math.ceil(this.scenario.getNetwork().getLinks().get(event.getLinkId()).getLength() / this.scenario.getNetwork().getLinks().get(event.getLinkId()).getFreespeed());
080                        double delayThisAgent = traveltimeThisAgent - freespeedTravelTime;              
081                        
082                        if (delayThisAgent < -1.)  { 
083                                if (warnCnt2 <= 5) {
084                                        log.warn("The delay is negative! Delay:" + delayThisAgent + " - traveltime: " + traveltimeThisAgent + " - freespeed traveltime: " + freespeedTravelTime + " - link: " + event.getLinkId() );
085                                        log.warn(event.toString());
086                                        if (warnCnt2 == 5) {
087                                                log.warn("Further warnings of this type will not be printed out.");
088                                        }
089                                        warnCnt2++;
090                                }
091                        } else if (delayThisAgent < 0.) {
092                                if (warnCnt  == 0) {
093                                        log.warn("Delay is " + delayThisAgent + ". A delay of 1 sec may result from rounding errors. Therefore a delay of 1 sec is ignored and set to zero.");
094                                        warnCnt ++;
095                                }
096                                delayThisAgent = 0.;
097                                
098                        } else if (delayThisAgent > 0.) {                       
099                                this.totalDelayPerDay_sec = this.totalDelayPerDay_sec + delayThisAgent;
100                        }
101                                                
102                        this.totalTravelTimePerDay_sec = this.totalTravelTimePerDay_sec + traveltimeThisAgent;
103                }
104        }
105        
106        @Override
107        public void handleEvent(LinkEnterEvent event) {
108                this.vehicleId2enterTime.put(event.getVehicleId(), event.getTime());
109        }
110        
111        @Override
112        public void handleEvent(PersonArrivalEvent event) {
113                this.vehicleId2enterTime.remove(event.getPersonId());
114        }
115
116        public double getTotalDelay() {
117                return totalDelayPerDay_sec;
118        }
119
120        public double getTotalTravelTime() {
121                return totalTravelTimePerDay_sec;
122        }
123
124}
