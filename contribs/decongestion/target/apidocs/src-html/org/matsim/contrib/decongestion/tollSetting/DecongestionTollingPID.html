020package org.matsim.contrib.decongestion.tollSetting;
021
022import java.util.HashMap;
023import java.util.Map;
024
025import org.apache.log4j.Logger;
026import org.matsim.api.core.v01.Id;
027import org.matsim.api.core.v01.events.LinkLeaveEvent;
028import org.matsim.api.core.v01.events.handler.LinkLeaveEventHandler;
029import org.matsim.api.core.v01.network.Link;
030
031import com.google.inject.Inject;
032
033import org.matsim.contrib.decongestion.DecongestionConfigGroup.IntegralApproach;
034import org.matsim.contrib.decongestion.data.DecongestionInfo;
035import org.matsim.contrib.decongestion.data.LinkInfo;
036import org.matsim.core.utils.misc.Counter;
037
038/**
039 * 
040 * PDI-based toll adjustment, where e(t) = average delay
041 * 
042 * @author ikaddoura
043 */
044
045public class DecongestionTollingPID implements DecongestionTollSetting, LinkLeaveEventHandler {
046        private static final Logger log = Logger.getLogger(DecongestionTollingPID.class);
047        
048        @Inject
049        private DecongestionInfo congestionInfo;
050        
051        private Map<Id<Link>, LinkInfo> linkId2infoPreviousTollComputation = new HashMap<>();   
052        private int tollUpdateCounter = 0;
053        private final Map<Id<Link>, Map<Integer, Double>> linkId2time2totalDelayAllIterations = new HashMap<>();        
054        
055        private final Map<Id<Link>, Map<Integer, Double>> linkId2time2avgDelayAllIterations = new HashMap<>();  
056        private final Map<Id<Link>, Map<Integer, Integer>> linkId2time2leavingAgents = new HashMap<>();
057
058        @Override
059        public void updateTolls() {
060                
061                final double K_p = congestionInfo.getDecongestionConfigGroup().getKp();
062                final double K_i = congestionInfo.getDecongestionConfigGroup().getKi();
063                final double K_d = congestionInfo.getDecongestionConfigGroup().getKd();
064                
065                final String integralApproach = congestionInfo.getDecongestionConfigGroup().getIntegralApproach().toString();
066                
067                final double timeBinSize = (double) this.congestionInfo.getScenario().getConfig().travelTimeCalculator().getTraveltimeBinSize();
068                final double capacityPeriod = this.congestionInfo.getScenario().getNetwork().getCapacityPeriod();
069                final double flowCapacityFactor = this.congestionInfo.getScenario().getConfig().qsim().getFlowCapFactor();
070                final double toleratedAvgDelay = this.congestionInfo.getDecongestionConfigGroup().getToleratedAverageDelaySec();
071                final boolean msa = this.congestionInfo.getDecongestionConfigGroup().isMsa();
072                final double blendFactorFromConfig = this.congestionInfo.getDecongestionConfigGroup().getTollBlendFactor();
073                
074                Counter counter = new Counter( "link # ", "" ) ;
075
076                for (Id<Link> linkId : this.congestionInfo.getlinkInfos().keySet()) {
077
078                        counter.incCounter();
079                        
080                        double flowCapacityHeadwaySec = Double.NEGATIVE_INFINITY;
081                        if (K_i != 0. && integralApproach.equals(IntegralApproach.UnusedHeadway.toString())) {
082                                flowCapacityHeadwaySec = capacityPeriod / ( this.congestionInfo.getScenario().getNetwork().getLinks().get(linkId).getCapacity() * flowCapacityFactor);
083                        }
084
085                        LinkInfo linkInfo = this.congestionInfo.getlinkInfos().get(linkId);
086                        for (Integer intervalNr : linkInfo.getTime2avgDelay().keySet()) {
087                                
088                                // 0) average delay
089                                
090                                double averageDelay = linkInfo.getTime2avgDelay().get(intervalNr);      
091                                if (averageDelay <= toleratedAvgDelay) {
092                                        averageDelay = 0.0;
093                                }
094                                                                
095                                double toll = 0.;
096                                
097                                // 1) proportional term
098                                
099                                if (K_p != 0.) {
100                                        toll += K_p * averageDelay;
101                                }
102                
103                                // 2) integral term
104                                
105//                              // Ideen:
106                                // --> letzten positiven averageDelay merken, und dann totalDelay -= lastAverageDelay ; --> DONE (alpha = 1.0)
107                                // --> Mittel Ã¼ber die letzten positven averageDelays (exponential smoothing probably ok: lastAverageDelay = (1-alpha)*lastAvDelay + alpha*averageDelay ; ) --> DONE
108                                // --> \propto * (1/flow - 1/cap, so etwas wie die "headway reserve" oder "unused time headway") --> DONE
109
110                                if (K_i != 0.) {
111                                        
112                                        // 2a) average approach
113                                        
114                                        double avgDelayAllIterations = 0.;
115                                        if (integralApproach.equals(IntegralApproach.Average.toString())) {
116                                                if (averageDelay > congestionInfo.getDecongestionConfigGroup().getToleratedAverageDelaySec()) {
117                                                        if (this.linkId2time2avgDelayAllIterations.get(linkId) == null) {
118                                                                avgDelayAllIterations = averageDelay;
119                                                                this.linkId2time2avgDelayAllIterations.put(linkId, new HashMap<>());
120                                                        } else {
121                                                                if (this.linkId2time2avgDelayAllIterations.get(linkId).get(intervalNr) == null) {
122                                                                        avgDelayAllIterations = averageDelay;
123                                                                } else {
124                                                                        avgDelayAllIterations =  (1 - congestionInfo.getDecongestionConfigGroup().getIntegralApproachAverageAlpha()) * this.linkId2time2avgDelayAllIterations.get(linkId).get(intervalNr)
125                                                                                        + congestionInfo.getDecongestionConfigGroup().getIntegralApproachAverageAlpha() * averageDelay;
126                                                                }
127                                                        }
128                                                        this.linkId2time2avgDelayAllIterations.get(linkId).put(intervalNr, avgDelayAllIterations);
129                                                }       
130                                        }
131                                        
132                                        // 2b) unused headway approach
133                                        
134                                        double unusedHeadway = 0.;
135                                        if (integralApproach.equals(IntegralApproach.UnusedHeadway.toString())) {
136                                                double flowHeadwaySec = timeBinSize;
137                                                if (this.linkId2time2leavingAgents.get(linkId) != null && this.linkId2time2leavingAgents.get(linkId).get(intervalNr) != null) {
138                                                        flowHeadwaySec = timeBinSize / this.linkId2time2leavingAgents.get(linkId).get(intervalNr);
139                                                }
140                                                unusedHeadway = flowHeadwaySec - flowCapacityHeadwaySec;
141                                                if (unusedHeadway < 0.) unusedHeadway = 0.; // there is no unused Headway
142                                        }
143
144                                        // update the total delay over all iterations
145                                        double totalDelayAllIterations = 0.;
146                                        if (linkId2time2totalDelayAllIterations.get(linkId) == null) {  
147                                                totalDelayAllIterations = averageDelay;
148                                                this.linkId2time2totalDelayAllIterations.put(linkId, new HashMap<>());
149                                        
150                                        } else {
151                                                
152                                                if (this.linkId2time2totalDelayAllIterations.get(linkId).get(intervalNr) == null) {
153                                                        totalDelayAllIterations = averageDelay;
154                                                
155                                                } else {        
156                                                                                                        
157                                                        if (averageDelay <= congestionInfo.getDecongestionConfigGroup().getToleratedAverageDelaySec()) {
158                                                                
159                                                                if (integralApproach.equals(IntegralApproach.Average.toString())) {
160                                                                        totalDelayAllIterations = this.linkId2time2totalDelayAllIterations.get(linkId).get(intervalNr)
161                                                                                        - avgDelayAllIterations;
162                                                                } else if (integralApproach.equals(IntegralApproach.UnusedHeadway.toString())) {
163                                                                        totalDelayAllIterations = this.linkId2time2totalDelayAllIterations.get(linkId).get(intervalNr)
164                                                                                        - (congestionInfo.getDecongestionConfigGroup().getIntegralApproachUnusedHeadwayFactor() * unusedHeadway);
165                                                                } else if (integralApproach.equals(IntegralApproach.Zero.toString())) {
166                                                                        totalDelayAllIterations = 0.;
167                                                                } else {
168                                                                        throw new RuntimeException("Unknown integral approach. Aborting...");
169                                                                }
170                                                                        
171                                                        } else {
172                                                                totalDelayAllIterations = this.linkId2time2totalDelayAllIterations.get(linkId).get(intervalNr) 
173                                                                                + averageDelay;
174                                                        }
175                                                }
176                                        }
177                                        this.linkId2time2totalDelayAllIterations.get(linkId).put(intervalNr, totalDelayAllIterations);                          
178                                        toll += K_i * totalDelayAllIterations;
179                                }
180                                
181                                // 3) derivative term
182                                
183                                if (K_d != 0.) {
184                                        double previousDelay = 0.;
185                                        if (this.linkId2infoPreviousTollComputation.get(linkId) != null 
186                                                        && this.linkId2infoPreviousTollComputation.get(linkId).getTime2avgDelay().get(intervalNr) != null) {
187                                                previousDelay = this.linkId2infoPreviousTollComputation.get(linkId).getTime2avgDelay().get(intervalNr);
188                                        }
189                                
190                                        double deltaDelay = averageDelay - previousDelay;
191                                        toll += K_d * deltaDelay;
192                                }
193                                
194                                // 4) prevent negative tolls
195                                
196                                if (toll < 0) {
197                                        toll = 0;
198                                }
199                                
200                                // 5) smoothen the tolls
201                                
202                                Double previousToll = linkInfo.getTime2toll().get(intervalNr);
203
204                                double blendFactor;
205                                if (msa) {
206                                        if (this.tollUpdateCounter > 0) {
207                                                blendFactor = 1.0 / (double) this.tollUpdateCounter;
208                                        } else {
209                                                blendFactor = 1.0;
210                                        }
211                                } else {
212                                        blendFactor = blendFactorFromConfig;
213                                }
214                                
215                                double smoothedToll;
216                                if (previousToll != null && previousToll >= 0.) {
217                                        smoothedToll = toll * blendFactor + previousToll * (1 - blendFactor);
218                                } else {
219                                        smoothedToll = toll;
220                                }
221                                                                
222                                // 6) store the updated toll
223                                
224                                linkInfo.getTime2toll().put(intervalNr, smoothedToll);
225
226                        }       
227                }
228                
229                log.info("Updating tolls completed.");
230                this.tollUpdateCounter++;
231                
232                // store the current link information for the next toll computation
233                
234                linkId2infoPreviousTollComputation = new HashMap<>();
235                for ( Map.Entry<Id<Link>,LinkInfo> entry : this.congestionInfo.getlinkInfos().entrySet()) {
236                        LinkInfo linkInfo = entry.getValue();;
237                        
238                        Map<Integer, Double> time2previousDelay = new HashMap<>();
239                        for (Integer intervalNr : linkInfo.getTime2avgDelay().keySet()) {
240                                time2previousDelay.put(intervalNr, linkInfo.getTime2avgDelay().get(intervalNr));
241                        }
242                        
243                        LinkInfo linkInfoPreviousTollComputation = new LinkInfo(linkInfo.getLink());
244                        linkInfoPreviousTollComputation.setTime2avgDelay(time2previousDelay);
245                        linkId2infoPreviousTollComputation.put(linkInfo.getLink().getId(), linkInfoPreviousTollComputation);
246                }
247        }
248
249        @Override
250        public void reset(int iteration) {
251                this.linkId2time2leavingAgents.clear();
252        }
253
254        @Override
255        public void handleEvent(LinkLeaveEvent event) {
256                int timeBinNr = getIntervalNr(event.getTime());
257
258                Id<Link> linkId = event.getLinkId();
259                
260                if (linkId2time2leavingAgents.get(linkId) != null) {
261                        
262                        if (linkId2time2leavingAgents.get(linkId).get(timeBinNr) != null) {
263                                int leavingAgents = linkId2time2leavingAgents.get(linkId).get(timeBinNr) + 1;
264                                linkId2time2leavingAgents.get(linkId).put(timeBinNr, leavingAgents);
265                                
266                        } else {
267                                linkId2time2leavingAgents.get(linkId).put(timeBinNr, 1);
268                        }
269                        
270                } else {
271                        Map<Integer, Integer> time2leavingAgents = new HashMap<>();
272                        time2leavingAgents.put(timeBinNr, 1);
273                        linkId2time2leavingAgents.put(linkId, time2leavingAgents);
274                        
275                }
276        }
277        
278        private int getIntervalNr(double time) {
279                double timeBinSize = congestionInfo.getScenario().getConfig().travelTimeCalculator().getTraveltimeBinSize();            
280                return (int) (time / timeBinSize);
281        }
