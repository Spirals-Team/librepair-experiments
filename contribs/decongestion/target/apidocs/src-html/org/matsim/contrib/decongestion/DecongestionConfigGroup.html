020package org.matsim.contrib.decongestion;
021
022import org.matsim.core.config.ReflectiveConfigGroup;
023
024/**
025 * Contains the input parameters, e.g. how often the interval-based output is written out, the number of iterations for which the price is kept constant, ...
026 * 
027 * @author ikaddoura
028 */
029
030public class DecongestionConfigGroup extends ReflectiveConfigGroup {
031        public static final String GROUP_NAME = "decongestion" ;
032
033        private static final String Kp="Kp" ;
034        private static final String Ki="Ki" ;
035        private static final String Kd = "Kd";
036        private static final String WRITE_OUTPUT_ITERATION = "writeOutputIteration";
037        private static final String UPDATE_PRICE_INTERVAL = "updatePriceInterval";
038        private static final String TOLERATED_AVERAGE_DELAY_SEC = "toleratedAverageDelaySec";
039        private static final String TOLL_ADJUSTMENT = "tollAdjustment";
040        private static final String FRACTION_OF_ITERATIONS_TO_END_PRICE_ADJUSTMENT = "fractionOfIterationsToEndPriceAdjustment";
041        private static final String TOLL_BLEND_FACTOR = "tollBlendFactor";
042        private static final String INITIAL_TOLL = "initialToll";
043        private static final String FRACTION_OF_ITERATIONS_TO_START_PRICE_ADJUSTMENT = "fractionOfIterationsToStartPriceAdjustment";
044        private static final String RUN_FINAL_ANALYSIS = "runFinalAnalysis";
045        private static final String WRITE_LINK_INFO_CHARTS = "writeLinkInfoCharts";
046        private static final String MSA = "usingMsa";
047        private static final String INTEGRAL_APPROACH = "integralApproach";
048        private static final String INTEGRAL_APPROACH_AVERAGE_ALPHA = "integralApproachAverageAlpha";
049        private static final String INTEGRAL_APPROACH_UNUSED_HEADWAY_FACTOR = "integralApproachUnusedHeadwayFactor";
050        private static final String DECONGESTION_APPROACH = "decongestionApproach";
051        private static final String ENABLE_DECONGESTION_PRICING = "enableDecongestionPricing";
052        
053        public DecongestionConfigGroup() {
054                super(GROUP_NAME);
055        }
056        
057        // General parameters
058        private boolean enableDecongestionPricing = true;
059        private DecongestionApproach decongestionApproach = DecongestionApproach.PID;
060        private boolean runFinalAnalysis = true;
061        private int updatePriceInterval = 1;
062        private int writeOutputIteration = 1;
063        private boolean writeLinkInfoCharts = false; // set to false for big networks
064        private double toleratedAverageDelaySec = 1.0; // set to 1.0 to account for rounding errors
065        private double fractionOfIterationsToStartPriceAdjustment = 0.1; // set above 0.0 to disable pricing in the previous iterations
066        private double fractionOfIterationsToEndPriceAdjustment = 1.0; // set below 1.0 to disable price adjustment for final iterations
067        private double tollBlendFactor = 1.0; // default: 1.0
068        private boolean msa = false;
069        
070        // BangBang approach
071        private double initialToll = 10.0;
072        private double tollAdjustment = 1.0;
073
074        // PID approach
075        private double kp = 1.;
076        private double kd = 1.;
077        private double ki = 1.;
078        private IntegralApproach integralApproach = IntegralApproach.Zero;
079        private double integralApproachAverageAlpha = 0.1;
080        private double integralApproachUnusedHeadwayFactor = 10.;
081        
082        // ######################################################################################
083        
084        public enum IntegralApproach {
085                Average, UnusedHeadway, Zero
086        }
087        
088        public enum DecongestionApproach {
089                BangBang, PID, P_MC
090        }
091        
092        @StringGetter( Kp )
093        public double getKp() {
094                return kp;
095        }
096
097        @StringSetter( Kp )
098        public void setKp(double kp) {
099                this.kp = kp;
100        }
101
102        @StringGetter( Ki )
103        public double getKi() {
104                return ki;
105        }
106
107        @StringSetter( Ki )
108        public void setKi(double ki) {
109                this.ki = ki;
110        }
111
112        @StringGetter( Kd )
113        public double getKd() {
114                return kd;
115        }
116
117        @StringSetter( Kd )
118        public void setKd(double kd) {
119                this.kd = kd;
120        }
121        
122        @StringGetter( WRITE_OUTPUT_ITERATION )
123        public int getWriteOutputIteration() {
124                return writeOutputIteration;
125        }
126
127        @StringGetter( UPDATE_PRICE_INTERVAL )
128        public int getUpdatePriceInterval() {
129                return updatePriceInterval;
130        }
131
132        @StringGetter( TOLERATED_AVERAGE_DELAY_SEC )
133        public double getToleratedAverageDelaySec() {
134                return toleratedAverageDelaySec;
135        }
136
137        @StringGetter( TOLL_ADJUSTMENT )
138        public double getTollAdjustment() {
139                return tollAdjustment;
140        }
141
142        @StringGetter( FRACTION_OF_ITERATIONS_TO_END_PRICE_ADJUSTMENT )
143        public double getFractionOfIterationsToEndPriceAdjustment() {
144                return fractionOfIterationsToEndPriceAdjustment;
145        }
146
147        @StringGetter( TOLL_BLEND_FACTOR )
148        public double getTollBlendFactor() {
149                return tollBlendFactor;
150        }
151
152        @StringSetter( TOLL_ADJUSTMENT )
153        public void setTollAdjustment(double tOLL_ADJUSTMENT) {
154                tollAdjustment = tOLL_ADJUSTMENT;
155        }
156        
157        @StringSetter( UPDATE_PRICE_INTERVAL )
158        public void setUpdatePriceInterval(int uPDATE_PRICE_INTERVAL) {
159                updatePriceInterval = uPDATE_PRICE_INTERVAL;
160        }
161
162        @StringSetter( TOLL_BLEND_FACTOR )
163        public void setTollBlendFactor(double tOLL_BLEND_FACTOR) {
164                tollBlendFactor = tOLL_BLEND_FACTOR;
165        }
166
167        @StringSetter( WRITE_OUTPUT_ITERATION )
168        public void setWriteOutputIteration(int wRITE_OUTPUT_ITERATION) {
169                writeOutputIteration = wRITE_OUTPUT_ITERATION;
170        }
171
172        @StringSetter( TOLERATED_AVERAGE_DELAY_SEC )
173        public void setToleratedAverageDelaySec(double tOLERATED_AVERAGE_DELAY_SEC) {
174                toleratedAverageDelaySec = tOLERATED_AVERAGE_DELAY_SEC;
175        }
176
177        @StringSetter( FRACTION_OF_ITERATIONS_TO_END_PRICE_ADJUSTMENT )
178        public void setFractionOfIterationsToEndPriceAdjustment(double fRACTION_OF_ITERATIONS_TO_END_PRICE_ADJUSTMENT) {
179                fractionOfIterationsToEndPriceAdjustment = fRACTION_OF_ITERATIONS_TO_END_PRICE_ADJUSTMENT;
180        }
181
182        @StringGetter( INITIAL_TOLL )
183        public double getInitialToll() {
184                return initialToll;
185        }
186
187        @StringSetter( INITIAL_TOLL )
188        public void setInitialToll(double iNITIAL_TOLL) {
189                initialToll = iNITIAL_TOLL;
190        }
191
192        @StringGetter( FRACTION_OF_ITERATIONS_TO_START_PRICE_ADJUSTMENT )
193        public double getFractionOfIterationsToStartPriceAdjustment() {
194                return fractionOfIterationsToStartPriceAdjustment;
195        }
196
197        @StringSetter( FRACTION_OF_ITERATIONS_TO_START_PRICE_ADJUSTMENT )
198        public void setFractionOfIterationsToStartPriceAdjustment(
199                        double fRACTION_OF_ITERATIONS_TO_START_PRICE_ADJUSTMENT) {
200                fractionOfIterationsToStartPriceAdjustment = fRACTION_OF_ITERATIONS_TO_START_PRICE_ADJUSTMENT;
201        }
202
203        @StringGetter( RUN_FINAL_ANALYSIS )
204        public boolean isRunFinalAnalysis() {
205                return runFinalAnalysis;
206        }
207
208        @StringSetter( RUN_FINAL_ANALYSIS )
209        public void setRunFinalAnalysis(boolean rUN_FINAL_ANALYSIS) {
210                runFinalAnalysis = rUN_FINAL_ANALYSIS;
211        }
212
213        @StringGetter( WRITE_LINK_INFO_CHARTS )
214        public boolean isWriteLinkInfoCharts() {
215                return writeLinkInfoCharts;
216        }
217
218        @StringSetter( WRITE_LINK_INFO_CHARTS )
219        public void setWriteLinkInfoCharts(boolean wRITE_CHARTS) {
220                writeLinkInfoCharts = wRITE_CHARTS;
221        }
222
223        @StringGetter( MSA )
224        public boolean isMsa() {
225                return msa;
226        }
227
228        @StringSetter( MSA )
229        public void setMsa(boolean msa) {
230                this.msa = msa;
231        }
232
233        @StringGetter( INTEGRAL_APPROACH )
234        public IntegralApproach getIntegralApproach() {
235                return integralApproach;
236        }
237
238        @StringSetter( INTEGRAL_APPROACH )
239        public void setIntegralApproach(IntegralApproach integralApproach) {
240                this.integralApproach = integralApproach;
241        }
242
243        @StringGetter( INTEGRAL_APPROACH_AVERAGE_ALPHA )
244        public double getIntegralApproachAverageAlpha() {
245                return integralApproachAverageAlpha;
246        }
247
248        @StringSetter( INTEGRAL_APPROACH_AVERAGE_ALPHA )
249        public void setIntegralApproachAverageAlpha(double integralApproachAverageAlpha) {
250                this.integralApproachAverageAlpha = integralApproachAverageAlpha;
251        }
252
253        @StringGetter( INTEGRAL_APPROACH_UNUSED_HEADWAY_FACTOR )
254        public double getIntegralApproachUnusedHeadwayFactor() {
255                return integralApproachUnusedHeadwayFactor;
256        }
257
258        @StringSetter( INTEGRAL_APPROACH_UNUSED_HEADWAY_FACTOR )
259        public void setIntegralApproachUnusedHeadwayFactor(double integralApproachUnusedHeadwayFactor) {
260                this.integralApproachUnusedHeadwayFactor = integralApproachUnusedHeadwayFactor;
261        }
262
263        @StringGetter( DECONGESTION_APPROACH )
264        public DecongestionApproach getDecongestionApproach() {
265                return decongestionApproach;
266        }
267
268        @StringSetter( DECONGESTION_APPROACH )
269        public void setDecongestionApproach(DecongestionApproach decongestionApproach) {
270                this.decongestionApproach = decongestionApproach;
271        }
272
273        @StringGetter( ENABLE_DECONGESTION_PRICING )
274        public boolean isEnableDecongestionPricing() {
275                return enableDecongestionPricing;
276        }
277
278        @StringSetter( ENABLE_DECONGESTION_PRICING )
279        public void setEnableDecongestionPricing(boolean enableDecongestionPricing) {
280                this.enableDecongestionPricing = enableDecongestionPricing;
281        }
