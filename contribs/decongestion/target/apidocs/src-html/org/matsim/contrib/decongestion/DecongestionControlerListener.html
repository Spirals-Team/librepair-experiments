058/**
059 * Interval-based decongestion pricing approach:
060 * 
061 * (1) Identify congested links and time intervals and set an initial toll for these links and time intervals.
062 * (2) Let the demand adjust for x iterations.
063 * (3) Adjust the tolls (different implementations of {@link DecongestionTollSetting})
064 * (4) GOTO (2)
065 * 
066 * All relevant parameters are specified in {@link DecongestionInfo}.
067 * 
068 * 
069 * @author ikaddoura
070 *
071 */
072
073public class DecongestionControlerListener implements StartupListener, AfterMobsimListener, IterationStartsListener, IterationEndsListener {
074

080        private final SortedMap<Integer, Double> iteration2userBenefits = new TreeMap<>();

099        @Override
100        public void notifyStartup(StartupEvent event) {
101                
102                log.info("decongestion settings: " + congestionInfo.getDecongestionConfigGroup().toString());
103                
104                this.outputDirectory = this.congestionInfo.getScenario().getConfig().controler().getOutputDirectory();
105                if (!outputDirectory.endsWith("/")) {
106                        log.info("Adjusting output directory.");
107                        outputDirectory = outputDirectory + "/";
108                }               
109        }

111        @Override
112        public void notifyAfterMobsim(AfterMobsimEvent event) {
113                
114                if (event.getIteration() % this.congestionInfo.getDecongestionConfigGroup().getWriteOutputIteration() == 0. || event.getIteration() % this.congestionInfo.getDecongestionConfigGroup().getUpdatePriceInterval() == 0.) {
115                        computeDelays(event);
116                }
117                
118                if (event.getIteration() == this.congestionInfo.getScenario().getConfig().controler().getFirstIteration()) {
119                        // skip first iteration
120                
121                } else if (event.getIteration() % this.congestionInfo.getDecongestionConfigGroup().getUpdatePriceInterval() == 0.) {
122                        
123                        int totalNumberOfIterations = this.congestionInfo.getScenario().getConfig().controler().getLastIteration() - this.congestionInfo.getScenario().getConfig().controler().getFirstIteration();
124                        int iterationCounter = event.getIteration() - this.congestionInfo.getScenario().getConfig().controler().getFirstIteration();
125                        
126                        if (iterationCounter < this.congestionInfo.getDecongestionConfigGroup().getFractionOfIterationsToEndPriceAdjustment() * totalNumberOfIterations
127                                        && iterationCounter > this.congestionInfo.getDecongestionConfigGroup().getFractionOfIterationsToStartPriceAdjustment() * totalNumberOfIterations) {
128                                
129                                if (tollComputation != null) {
130                                        log.info("+++ Iteration " + event.getIteration() + ". Update tolls per link and time bin.");
131                                        tollComputation.updateTolls();
132                                }
133                        }
134                }
135                
136                if (event.getIteration() % this.congestionInfo.getDecongestionConfigGroup().getWriteOutputIteration() == 0.) {
137                        CongestionInfoWriter.writeDelays(congestionInfo, event.getIteration(), this.outputDirectory + "ITERS/it." + event.getIteration() + "/");
138                        CongestionInfoWriter.writeTolls(congestionInfo, event.getIteration(), this.outputDirectory + "ITERS/it." + event.getIteration() + "/");
139                }
140        }

142        private void computeDelays(AfterMobsimEvent event) {
143//              log.warn("entering computeDelays ...") ;
144                
145                TravelTime travelTime = event.getServices().getLinkTravelTimes();
146                int timeBinSize = this.congestionInfo.getScenario().getConfig().travelTimeCalculator().getTraveltimeBinSize();
147                
148                for (Link link : this.congestionInfo.getScenario().getNetwork().getLinks().values()) {
149//                      log.warn("linkID=" + link.getId() ) ;
150                        
151                        Map<Integer, Double> time2avgDelay = new HashMap<>();
152                        
153                        int timeBinCounter = 0;
154                        for (int endTime = timeBinSize ; endTime <= this.congestionInfo.getScenario().getConfig().travelTimeCalculator().getMaxTime(); endTime = endTime + timeBinSize ) {
155                                final double probedTime = endTime - timeBinSize / 2.;
156                                double freespeedTravelTime = link.getLength() / link.getFreespeed( probedTime ) ;
157                                final double congestedTravelTime = travelTime.getLinkTravelTime(link, probedTime, null, null);
158                                double avgDelay = congestedTravelTime - freespeedTravelTime;
159//                              if ( link.getId().equals(Id.createLinkId("24007-24006"))) {
160//                                      log.warn("endTime=" + endTime / 3600. + "; freeTravelTime=" + freespeedTravelTime + "; congTravelTime=" + congestedTravelTime
161//                                                                       + "; avgDelay=" + avgDelay);
162//                              }
163//                              if ( avgDelay > 0 ) {
164//                                      log.warn( "time=" + probedTime + "; linkID=" + link.getId() + "; avgDelay=" + avgDelay  ) ;
165////                                    throw new RuntimeException("found a delay") ;
166//                              }
167                                time2avgDelay.put(timeBinCounter, avgDelay);                            
168                                timeBinCounter++;
169                        }
170                        
171                        if (this.congestionInfo.getlinkInfos().get(link.getId()) != null) {
172                                this.congestionInfo.getlinkInfos().get(link.getId()).setTime2avgDelay(time2avgDelay);
173                        } else {
174                                LinkInfo linkInfo = new LinkInfo(link);
175                                linkInfo.setTime2avgDelay(time2avgDelay);
176                                this.congestionInfo.getlinkInfos().put(link.getId(), linkInfo);
177                        }
178                }
179//              log.warn("... done with computeDelays.") ;
180        }

182        @Override
183        public void notifyIterationEnds(IterationEndsEvent event) {
184                                
185                if (this.delayComputation != null) {
186                        // Store and write out some aggregated numbers for analysis purposes.
187                        
188                        this.iteration2totalDelay.put(event.getIteration(), this.delayComputation.getTotalDelay());
189                        double totalPayments = 0.;
190                        if (this.intervalBasedTolling != null) totalPayments = this.intervalBasedTolling.getTotalTollPayments();
191                        this.iteration2totalTollPayments.put(event.getIteration(), totalPayments);
192                        this.iteration2totalTravelTime.put(event.getIteration(), this.delayComputation.getTotalTravelTime());
193                        
194                        double monetizedUserBenefits = 0.;
195                        for (Person person : this.congestionInfo.getScenario().getPopulation().getPersons().values()) {
196                                monetizedUserBenefits = monetizedUserBenefits + person.getSelectedPlan().getScore() / this.congestionInfo.getScenario().getConfig().planCalcScore().getMarginalUtilityOfMoney();
197                        }
198                        this.iteration2userBenefits.put(event.getIteration(), monetizedUserBenefits);
199                        
200                        CongestionInfoWriter.writeIterationStats(
201                                        this.iteration2totalDelay,
202                                        this.iteration2totalTollPayments,
203                                        this.iteration2totalTravelTime,
204                                        this.iteration2userBenefits,
205                                        outputDirectory
206                                        );
207                        
208                        XYLineChart chart1 = new XYLineChart("Total travel time and total delay", "Iteration", "Hours");
209                        double[] iterations1 = new double[event.getIteration() + 1];
210                        double[] values1a = new double[event.getIteration() + 1];
211                        double[] values1b = new double[event.getIteration() + 1];
212                        for (int i = this.congestionInfo.getScenario().getConfig().controler().getFirstIteration(); i <= event.getIteration(); i++) {
213                                iterations1[i] = i;
214                                values1a[i] = this.iteration2totalDelay.get(i) / 3600.;
215                                values1b[i] = this.iteration2totalTravelTime.get(i) / 3600.;
216                        }
217                        chart1.addSeries("Total delay", iterations1, values1a);
218                        chart1.addSeries("Total travel time", iterations1, values1b);
219                        chart1.saveAsPng(outputDirectory + "travelTime_delay.png", 800, 600);
220                        
221                        XYLineChart chart2 = new XYLineChart("user benefits and toll revenues", "Iteration", "Monetary units");
222                        double[] iterations2 = new double[event.getIteration() + 1];
223                        double[] values2b = new double[event.getIteration() + 1];
224                        double[] values2c = new double[event.getIteration() + 1];
225
226                        for (int i = this.congestionInfo.getScenario().getConfig().controler().getFirstIteration(); i <= event.getIteration(); i++) {
227                                iterations2[i] = i;
228                                values2b[i] = this.iteration2userBenefits.get(i);
229                                values2c[i] = this.iteration2totalTollPayments.get(i);
230                        }
231                        chart2.addSeries("User benefits", iterations2, values2b);
232                        chart2.addSeries("Toll revenues", iterations2, values2c);
233                        chart2.saveAsPng(outputDirectory + "userBenefits_tollRevenues.png", 800, 600);
234                }
235        }

237        @Override
238        public void notifyIterationStarts(IterationStartsEvent event) {
239                        
240                if (congestionInfo.getDecongestionConfigGroup().getUpdatePriceInterval() > 1) {
241                        if (event.getIteration() == this.congestionInfo.getScenario().getConfig().controler().getFirstIteration()) {
242                                
243                                this.nextDisableInnovativeStrategiesIteration = (int) (congestionInfo.getScenario().getConfig().strategy().getFractionOfIterationsToDisableInnovation() * congestionInfo.getDecongestionConfigGroup().getUpdatePriceInterval());
244                                log.info("next disable innovative strategies iteration: " + this.nextDisableInnovativeStrategiesIteration);
245
246                                if (this.nextDisableInnovativeStrategiesIteration != 0) {
247                                        this.nextEnableInnovativeStrategiesIteration = (int) (congestionInfo.getDecongestionConfigGroup().getUpdatePriceInterval() + 1);
248                                }
249                                log.info("next enable innovative strategies iteration: " + this.nextEnableInnovativeStrategiesIteration);
250
251                        } else {
252                                
253                                if (event.getIteration() == this.nextDisableInnovativeStrategiesIteration) {
254                                        // set weight to zero
255                                        log.warn("Strategy weight adjustment (set to zero) in iteration " + event.getIteration());
256                                                                        
257                                        for (GenericPlanStrategy<Plan, Person> strategy : event.getServices().getStrategyManager().getStrategies(null)) {
258                                                
259                                                String strategyName = strategy.toString();
260                                                if (isInnovativeStrategy(strategy)) {
261                                                        log.info("Setting weight for " + strategyName + " to zero.");
262                                                        event.getServices().getStrategyManager().changeWeightOfStrategy(strategy, null, 0.0);
263                                                }
264                                        }
265                                        
266                                        this.nextDisableInnovativeStrategiesIteration += congestionInfo.getDecongestionConfigGroup().getUpdatePriceInterval();
267                                        log.info("next disable innovative strategies iteration: " + this.nextDisableInnovativeStrategiesIteration);
268                                        
269                                } else if (event.getIteration() == this.nextEnableInnovativeStrategiesIteration) {
270                                        // set weight back to original value
271
272                                        if (event.getIteration() >= congestionInfo.getScenario().getConfig().strategy().getFractionOfIterationsToDisableInnovation() * (congestionInfo.getScenario().getConfig().controler().getLastIteration() - congestionInfo.getScenario().getConfig().controler().getFirstIteration())) {
273                                                
274                                                log.info("Strategies are switched off by global settings. Do not set back the strategy parameters to original values...");
275                                        
276                                        } else {
277                                                
278                                                log.info("Strategy weight adjustment (set back to original value) in iteration " + event.getIteration());
279                                                
280                                                for (GenericPlanStrategy<Plan, Person> strategy : event.getServices().getStrategyManager().getStrategies(null)) {
281                                                        
282                                                        String strategyName = strategy.toString();
283                                                        if (isInnovativeStrategy(strategy)) {
284                                                                
285                                                                double originalValue = -1.0;
286                                                                for (StrategySettings setting : event.getServices().getConfig().strategy().getStrategySettings()) {
287                                                                        log.info("setting: " + setting.getStrategyName());
288                                                                        log.info("strategyName: " + strategyName);
289
290                                                                        if (strategyName.contains(setting.getStrategyName())) {
291                                                                                originalValue = setting.getWeight();
292                                                                        }
293                                                                }               
294                                                                
295                                                                if (originalValue == -1.0) {
296                                                                        throw new RuntimeException("Aborting...");
297                                                                }
298                                                                
299                                                                log.warn("Setting weight for " + strategyName + " back to original value: " + originalValue);
300                                                                event.getServices().getStrategyManager().changeWeightOfStrategy(strategy, null, originalValue);
301                                                        }
302                                                }                       
303                                                this.nextEnableInnovativeStrategiesIteration += congestionInfo.getDecongestionConfigGroup().getUpdatePriceInterval();
304                                        }
305                                }
306                        }       
307                }
308        }
