020package org.matsim.contrib.decongestion.handler;
021
022import org.matsim.api.core.v01.events.LinkLeaveEvent;
023import org.matsim.api.core.v01.events.PersonMoneyEvent;
024import org.matsim.api.core.v01.events.handler.LinkLeaveEventHandler;
025import org.matsim.core.api.experimental.events.EventsManager;
026
027import com.google.inject.Inject;
028
029import org.matsim.contrib.decongestion.data.DecongestionInfo;
030
031/**
032 * Throws agent money events for the tolled links and time bins.
033 * 
034 * @author ikaddoura
035 */
036
037public class IntervalBasedTollingAll implements LinkLeaveEventHandler, IntervalBasedTolling {
038
039        @Inject
040        private EventsManager eventsManager;
041        
042        @Inject
043        private DecongestionInfo decongestionInfo;
044        
045        private double totalTollPayments;
046
047        @Override
048        public void reset(int iteration) {
049                this.totalTollPayments = 0.;
050        }
051
052        @Override
053        public void handleEvent(LinkLeaveEvent event) {
054                if (decongestionInfo.getlinkInfos().get(event.getLinkId()) != null) {
055                                                
056                        int currentTimeBin = (int) (event.getTime() / this.decongestionInfo.getScenario().getConfig().travelTimeCalculator().getTraveltimeBinSize());
057                        
058                        if (decongestionInfo.getlinkInfos().get(event.getLinkId()).getTime2toll().get(currentTimeBin) != null) {
059                                double toll = decongestionInfo.getlinkInfos().get(event.getLinkId()).getTime2toll().get(currentTimeBin);
060                                this.eventsManager.processEvent(new PersonMoneyEvent(event.getTime(), this.decongestionInfo.getVehicleId2personId().get(event.getVehicleId()), -1. * toll));
061//                              this.eventsManager.processEvent(new PersonLinkMoneyEvent(event.getTime(), this.decongestionInfo.getVehicleId2personId().get(event.getVehicleId()), event.getLinkId(), -1. * toll, event.getTime(), "congestion"));
062                                this.totalTollPayments = this.totalTollPayments + toll;
063                        }               
064                }
065        }
066
067        @Override
068        public double getTotalTollPayments() {
069                return totalTollPayments;
070        }
071
072}
