020package org.matsim.contrib.decongestion.tollSetting;
021
022import java.util.HashMap;
023import java.util.Map;
024
025import org.apache.log4j.Logger;
026import org.matsim.api.core.v01.Id;
027import org.matsim.api.core.v01.TransportMode;
028import org.matsim.api.core.v01.events.LinkLeaveEvent;
029import org.matsim.api.core.v01.events.handler.LinkLeaveEventHandler;
030import org.matsim.api.core.v01.network.Link;
031
032import com.google.inject.Inject;
033
034import org.matsim.contrib.decongestion.data.DecongestionInfo;
035import org.matsim.contrib.decongestion.data.LinkInfo;
036
037/**
038 * 
039 * P-based toll adjustment, where e(t) = average delay and K_p = VTTS * number of delayed agents
040 * 
041 * @author ikaddoura
042 */
043
044public class DecongestionTollingP_MCP implements DecongestionTollSetting, LinkLeaveEventHandler {
045        private static final Logger log = Logger.getLogger(DecongestionTollingP_MCP.class);
046        
047        @Inject
048        private DecongestionInfo congestionInfo;
049        
050        private Map<Id<Link>, LinkInfo> linkId2infoPreviousTollComputation = new HashMap<>();   
051        private int tollUpdateCounter = 0;
052        private final Map<Id<Link>, Map<Integer, Integer>> linkId2time2leavingAgents = new HashMap<>();
053
054        @Override
055        public void updateTolls() {
056                
057                final double vtts = ( this.congestionInfo.getScenario().getConfig().planCalcScore().getPerforming_utils_hr()
058                                - this.congestionInfo.getScenario().getConfig().planCalcScore().getModes().get(TransportMode.car).getMarginalUtilityOfTraveling() )
059                                / this.congestionInfo.getScenario().getConfig().planCalcScore().getMarginalUtilityOfMoney();
060
061                        
062                final double toleratedAvgDelay = this.congestionInfo.getDecongestionConfigGroup().getToleratedAverageDelaySec();
063                final boolean msa = this.congestionInfo.getDecongestionConfigGroup().isMsa();
064                final double blendFactorFromConfig = this.congestionInfo.getDecongestionConfigGroup().getTollBlendFactor();
065                
066                for (Id<Link> linkId : this.congestionInfo.getlinkInfos().keySet()) {
067                        
068                        LinkInfo linkInfo = this.congestionInfo.getlinkInfos().get(linkId);
069                        for (Integer intervalNr : linkInfo.getTime2avgDelay().keySet()) {
070                                
071                                // average delay
072                                
073                                double averageDelay = linkInfo.getTime2avgDelay().get(intervalNr);      
074                                if (averageDelay <= toleratedAvgDelay) {
075                                        averageDelay = 0.0;
076                                }
077                                                                
078                                // toll
079                                
080                                double demand = 1.0;
081                                if (this.linkId2time2leavingAgents.get(linkId) != null && this.linkId2time2leavingAgents.get(linkId).get(intervalNr) != null) {
082                                        demand = this.linkId2time2leavingAgents.get(linkId).get(intervalNr);
083                                }
084                                
085                                double toll = vtts * demand * averageDelay / 3600.;
086
087                                // prevent negative tolls
088                                
089                                if (toll < 0) {
090                                        log.warn("Negative tolls... Are you sure everything works fine?");
091                                        toll = 0;
092                                }
093                                
094                                // smoothen the tolls
095                                
096                                Double previousToll = linkInfo.getTime2toll().get(intervalNr);
097
098                                double blendFactor;
099                                if (msa) {
100                                        if (this.tollUpdateCounter > 0) {
101                                                blendFactor = 1.0 / (double) this.tollUpdateCounter;
102                                        } else {
103                                                blendFactor = 1.0;
104                                        }
105                                } else {
106                                        blendFactor = blendFactorFromConfig;
107                                }
108                                
109                                double smoothedToll;
110                                if (previousToll != null && previousToll >= 0.) {
111                                        smoothedToll = toll * blendFactor + previousToll * (1 - blendFactor);
112                                } else {
113                                        smoothedToll = toll;
114                                }
115                                                                
116                                // store the updated toll
117                                
118                                linkInfo.getTime2toll().put(intervalNr, smoothedToll);
119
120                        }       
121                }
122                
123                log.info("Updating tolls completed.");
124                this.tollUpdateCounter++;
125                
126                // store the current link information for the next toll computation
127                
128                linkId2infoPreviousTollComputation = new HashMap<>();
129                for ( Map.Entry< Id<Link>,LinkInfo> entry : this.congestionInfo.getlinkInfos().entrySet() ) {
130                        LinkInfo linkInfo = entry.getValue() ;
131                        
132                        Map<Integer, Double> time2previousDelay = new HashMap<>();
133                        for (Integer intervalNr : linkInfo.getTime2avgDelay().keySet()) {
134                                time2previousDelay.put(intervalNr, linkInfo.getTime2avgDelay().get(intervalNr));
135                        }
136                        
137                        LinkInfo linkInfoPreviousTollComputation = new LinkInfo(linkInfo.getLink());
138                        linkInfoPreviousTollComputation.setTime2avgDelay(time2previousDelay);
139                        linkId2infoPreviousTollComputation.put(linkInfo.getLink().getId(), linkInfoPreviousTollComputation);
140                }
141        }
142
143        @Override
144        public void reset(int iteration) {
145                this.linkId2time2leavingAgents.clear();
146        }
147
148        @Override
149        public void handleEvent(LinkLeaveEvent event) {
150                int timeBinNr = getIntervalNr(event.getTime());
151
152                Id<Link> linkId = event.getLinkId();
153                
154                if (linkId2time2leavingAgents.get(linkId) != null) {
155                        
156                        if (linkId2time2leavingAgents.get(linkId).get(timeBinNr) != null) {
157                                int leavingAgents = linkId2time2leavingAgents.get(linkId).get(timeBinNr) + 1;
158                                linkId2time2leavingAgents.get(linkId).put(timeBinNr, leavingAgents);
159                                
160                        } else {
161                                linkId2time2leavingAgents.get(linkId).put(timeBinNr, 1);
162                        }
163                        
164                } else {
165                        Map<Integer, Integer> time2leavingAgents = new HashMap<>();
166                        time2leavingAgents.put(timeBinNr, 1);
167                        linkId2time2leavingAgents.put(linkId, time2leavingAgents);
168                        
169                }
170        }
171        
172        private int getIntervalNr(double time) {
173                double timeBinSize = congestionInfo.getScenario().getConfig().travelTimeCalculator().getTraveltimeBinSize();            
174                return (int) (time / timeBinSize);
175        }
176
177}
