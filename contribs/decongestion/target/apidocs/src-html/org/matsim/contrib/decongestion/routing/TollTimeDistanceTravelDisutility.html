021package org.matsim.contrib.decongestion.routing;
022
023import org.apache.log4j.Logger;
024import org.matsim.api.core.v01.TransportMode;
025import org.matsim.api.core.v01.network.Link;
026import org.matsim.api.core.v01.population.Person;
027import org.matsim.core.config.groups.PlanCalcScoreConfigGroup;
028import org.matsim.core.router.costcalculators.RandomizingTimeDistanceTravelDisutilityFactory;
029import org.matsim.core.router.util.TravelDisutility;
030import org.matsim.core.router.util.TravelTime;
031import org.matsim.vehicles.Vehicle;
032
033import org.matsim.contrib.decongestion.data.DecongestionInfo;
034
035/**
036 * A cost calculator which respects time, distance and decongestion tolls. 
037 *
038 * @author ikaddoura
039 */
040public final class TollTimeDistanceTravelDisutility implements TravelDisutility {
041        private static final Logger log = Logger.getLogger(TollTimeDistanceTravelDisutility.class);
042
043        private final TravelDisutility delegate;
044        private final PlanCalcScoreConfigGroup cnScoringGroup;
045        private final DecongestionInfo info;
046        private final double sigma;
047
048        TollTimeDistanceTravelDisutility(final TravelTime timeCalculator, PlanCalcScoreConfigGroup cnScoringGroup, double sigma, DecongestionInfo info) {
049                this.info = info;
050                this.cnScoringGroup = cnScoringGroup;
051                this.sigma = sigma;
052                
053                final RandomizingTimeDistanceTravelDisutilityFactory builder = new RandomizingTimeDistanceTravelDisutilityFactory( TransportMode.car, cnScoringGroup );
054                builder.setSigma(sigma);
055                this.delegate = builder.createTravelDisutility(timeCalculator);
056
057                log.info("Using the toll-adjusted travel disutility in the decongestion package.");
058        }
059
060        @Override
061        public double getLinkTravelDisutility(final Link link, final double time, final Person person, final Vehicle vehicle) {
062                int timeBin = (int) (time / info.getScenario().getConfig().travelTimeCalculator().getTraveltimeBinSize());
063
064                double timeDistanceTravelDisutilityFromDelegate = this.delegate.getLinkTravelDisutility(link, time, person, vehicle);
065                                
066                double logNormalRnd = 1. ;
067                if ( sigma != 0. ) {
068                        logNormalRnd = (double) person.getCustomAttributes().get("logNormalRnd") ;
069                }
070
071                // adjust the travel disutility for the toll
072                double toll = 0.;
073                if (info.getlinkInfos().containsKey(link.getId()) && info.getlinkInfos().get(link.getId()).getTime2toll().containsKey(timeBin)) {
074                        toll = info.getlinkInfos().get(link.getId()).getTime2toll().get(timeBin);
075                }
076                double tollAdjustedLinkTravelDisutility = Double.NEGATIVE_INFINITY;
077                tollAdjustedLinkTravelDisutility = timeDistanceTravelDisutilityFromDelegate + logNormalRnd * cnScoringGroup.getMarginalUtilityOfMoney() * toll;
078                
079                return tollAdjustedLinkTravelDisutility;
080        }
081
082        @Override
083        public double getLinkMinimumTravelDisutility(final Link link) {
084                return this.delegate.getLinkMinimumTravelDisutility(link);
085        }
086
087}
