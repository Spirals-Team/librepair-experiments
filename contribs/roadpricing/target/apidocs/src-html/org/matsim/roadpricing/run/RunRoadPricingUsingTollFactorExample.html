019package org.matsim.roadpricing.run;

021import org.matsim.api.core.v01.Id;
022import org.matsim.api.core.v01.Scenario;
023import org.matsim.api.core.v01.network.Link;
024import org.matsim.api.core.v01.population.Person;
025import org.matsim.core.config.Config;
026import org.matsim.core.config.ConfigUtils;
027import org.matsim.core.controler.Controler;
028import org.matsim.core.scenario.ScenarioUtils;
029import org.matsim.roadpricing.RoadPricingConfigGroup;
030import org.matsim.roadpricing.RoadPricingModule;
031import org.matsim.roadpricing.RoadPricingSchemeUsingTollFactor;
032import org.matsim.roadpricing.TollFactor;
033import org.matsim.vehicles.Vehicle;
034import org.matsim.vehicles.VehicleType;
035
036/**
037 * Illustrative example how to first read a "base" toll file, and then make it dependent on the vehicle type.
038 * <br/><br/>
039 * Not tested.
040 * 
041 * @author nagel
042 *
043 */
044public class RunRoadPricingUsingTollFactorExample {
045
046        public static void main(String[] args) {
047
048                // load the config from file:
049                Config config = ConfigUtils.loadConfig(args[0]) ;
050                
051                // "materialize" the road pricing config group:
052                RoadPricingConfigGroup rpConfig = ConfigUtils.addOrGetModule(config, RoadPricingConfigGroup.class) ;
053                
054                // load the scenario from file.  This will _not_ load the road pricing file, since it is not part of the main distribution.
055                final Scenario scenario = ScenarioUtils.loadScenario(config) ;
056                
057                // define the toll factor as an anonymous class.  If more flexibility is needed, convert to "full" class.
058                TollFactor tollFactor = new TollFactor(){
059                        @Override public double getTollFactor(Id<Person> personId, Id<Vehicle> vehicleId, Id<Link> linkId, double time) {
060                                VehicleType someVehicleType = null ; // --> replace by something meaningful <-- 
061                                if ( scenario.getVehicles().getVehicles().get( vehicleId ).getType().equals( someVehicleType ) ) {
062                                        return 2 ;
063                                } else {
064                                        return 1 ; 
065                                }
066                        }
067                } ;
068                
069                // instantiate the road pricing scheme, with the toll factor inserted:
070                RoadPricingSchemeUsingTollFactor scheme = new RoadPricingSchemeUsingTollFactor(rpConfig.getTollLinksFile(), tollFactor) ;
071                
072                // instantiate the control(l)er:
073                Controler controler = new Controler( scenario ) ;
074                
075                // add the road pricing module, with our scheme from above inserted:
076                controler.addOverridingModule( new RoadPricingModule( scheme ) ) ;
077                
078                // run everything:
079                controler.run();
080                
081        }
082
083}
