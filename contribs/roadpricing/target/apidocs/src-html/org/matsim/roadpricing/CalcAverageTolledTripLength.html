044/**
045 * Calculates the distance of a trip which occurred on tolled links.
046 * Requires roadpricing to be on.
047 *
048 * @author mrieser
049 */
050public final class CalcAverageTolledTripLength implements LinkEnterEventHandler, PersonArrivalEventHandler, VehicleEntersTrafficEventHandler, VehicleLeavesTrafficEventHandler {
051
052        @SuppressWarnings("unused")
053        private static final Logger log = Logger.getLogger(CalcAverageTolledTripLength.class);
054
055        private double sumLength = 0.0;
056        private int cntTrips = 0;
057        private RoadPricingScheme scheme = null;
058        private Network network = null;
059        private TreeMap<Id<Person>, Double> agentDistance = null;
060
061        private Vehicle2DriverEventHandler delegate = new Vehicle2DriverEventHandler();
062        
063        private static Double zero = 0.0;
064
065    @Inject
066        public CalcAverageTolledTripLength(final Network network, final RoadPricingScheme scheme, EventsManager events) {
067                this.scheme = scheme;
068                this.network = network;
069                this.agentDistance = new TreeMap<>();
070                events.addHandler(this);
071        }
072
073        @Override
074        public void handleEvent(final LinkEnterEvent event) {
075                
076                Id<Person> driverId = delegate.getDriverOfVehicle(event.getVehicleId());
077                
078                // getting the (monetary? generalized?) cost of the link
079                Cost cost = this.scheme.getLinkCostInfo(event.getLinkId(), event.getTime(), driverId, event.getVehicleId() );
080                
081                if (cost != null) {
082                        // i.e. if there is a toll on the link
083                        
084                        Link link = this.network.getLinks().get(event.getLinkId());
085                        if (link != null) {
086                                
087                                // get some distance that has been accumulated (how?) up to this point:
088                                Double length = this.agentDistance.get(driverId);
089                                
090                                // if nothing has been accumlated so far, initialize this at zero:
091                                if (length == null) {
092                                        length = zero;
093                                }
094                                
095                                // add the new length to the already accumulated length:
096                                length = length + link.getLength();
097                                
098                                // put the result again in the "memory":
099                                this.agentDistance.put(driverId, length);
100                        }
101                }
102        }
103
104        @Override
105        public void handleEvent(final PersonArrivalEvent event) {
106                // at arrival of the agent ...
107                
108                // get the accumulated "tolled" length from the agent
109                Double length = this.agentDistance.get(event.getPersonId());
110                if (length != null) {
111                        // if this is not zero, accumulate it into some global accumulated length ...
112                        this.sumLength += length;
113                        
114                        // ... and reset the agent-individual accumlated length to zero:
115                        this.agentDistance.put(event.getPersonId(), zero);
116                        this.cntTrips++;
117                }
118
119                // count _all_ finished trips (independent off toll payment):
120//              this.cntTrips++;
121        }
122
123        @Override
124        public void reset(final int iteration) {
125                this.sumLength = 0.0;
126                this.cntTrips = 0;
127                delegate.reset(iteration);
128        }
129
130        public double getAverageTripLength() {
131                // public is currently needed. kai, sep'13
132
133                if (this.cntTrips == 0) return 0;
134//              log.warn("NOTE: The result of this calculation has been changed from 'av over all trips' to 'av over tolled trips'.  kai/benjamin, apr'10") ;
135                // commenting this out.  kai, mar'12
136                return (this.sumLength / this.cntTrips);
137        }
138
139        @Override
140        public void handleEvent(VehicleLeavesTrafficEvent event) {
141                delegate.handleEvent(event);
142        }
143
144        @Override
145        public void handleEvent(VehicleEntersTrafficEvent event) {
146                delegate.handleEvent(event);
147        }
148}
