021package org.matsim.roadpricing;
022
023import java.util.List;
024import java.util.Map;
025import java.util.Set;
026
027import org.apache.log4j.Logger;
028import org.matsim.api.core.v01.Id;
029import org.matsim.api.core.v01.network.Link;
030import org.matsim.api.core.v01.population.Person;
031import org.matsim.roadpricing.RoadPricingSchemeImpl.Cost;
032import org.matsim.vehicles.Vehicle;
033
034/**
035 * @author nagel
036 *
037 */
038public final class RoadPricingSchemeUsingTollFactor implements RoadPricingScheme {
039        // needs to be public. kai, sep'14
040        
041        @SuppressWarnings("unused")
042        private static final Logger log = Logger.getLogger( RoadPricingSchemeUsingTollFactor.class ) ;
043
044        private RoadPricingScheme delegate = null ;
045        private final TollFactor tollFactor ;
046
047        public RoadPricingSchemeUsingTollFactor(RoadPricingScheme scheme, TollFactor tollFactor ) {
048                this.delegate = scheme;
049                this.tollFactor = tollFactor;
050
051        }
052
053        public RoadPricingSchemeUsingTollFactor( String pricingSchemeFileName, TollFactor tollFactor ) {
054                
055                // read the road pricing scheme from file
056                RoadPricingSchemeImpl scheme = new RoadPricingSchemeImpl();
057                RoadPricingReaderXMLv1 rpReader = new RoadPricingReaderXMLv1(scheme);
058                try {
059                        rpReader.readFile( pricingSchemeFileName  );
060                } catch (Exception e) {
061                        throw new RuntimeException(e);
062                }
063                this.delegate = scheme ;
064                this.tollFactor = tollFactor ;
065                                
066        }
067
068        @Override
069        public String getDescription() {
070                return delegate.getDescription();
071        }
072
073        @Override
074        public Cost getLinkCostInfo(Id<Link> linkId, double time, Id<Person> personId, Id<Vehicle> vehicleId) {
075                Cost baseToll = delegate.getLinkCostInfo(linkId, time, personId, vehicleId );
076                if (baseToll == null) {
077                        return null ;
078                }
079                final double tollFactorVal = tollFactor.getTollFactor(personId, vehicleId, linkId, time);
080                return new Cost( baseToll.startTime, baseToll.endTime, baseToll.amount * tollFactorVal );
081        }
082        
083        @Override
084        public Cost getTypicalLinkCostInfo( Id<Link> linkId, double time ) {
085                return delegate.getTypicalLinkCostInfo(linkId, time) ;
086        }
087
088        @Override
089        public Set<Id<Link>> getTolledLinkIds() {
090                return delegate.getTolledLinkIds();
091        }
092
093        @Override
094        public String getName() {
095                return delegate.getName();
096        }
097
098        @Override
099        public String getType() {
100                return delegate.getType();
101        }
102
103        @Override
104        public Iterable<Cost> getTypicalCosts() {
105                return delegate.getTypicalCosts() ;
106        }
107
108        @Override
109        public Map<Id<Link>, List<Cost>> getTypicalCostsForLink() {
110                return delegate.getTypicalCostsForLink() ;
111        }
112
113}
