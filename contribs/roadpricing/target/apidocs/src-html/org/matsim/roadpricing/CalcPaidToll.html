043/**
044 * Calculates the toll agents pay during a simulation by analyzing events. 
045 * Add an instance of this class as an EventHandler.
046 * 
047 * Users of this class can get to the amounts which have to be paid by the 
048 * agents by using the getter methods and/or by asking for a stream of 
049 * AgentMoneyEvent instances.
050 *
051 * @author mrieser
052 */
053public final class CalcPaidToll implements LinkEnterEventHandler, VehicleEntersTrafficEventHandler, VehicleLeavesTrafficEventHandler {
054
055        Logger log = Logger.getLogger( CalcPaidToll.class ) ;
056
057        static class AgentTollInfo {
058                public double toll = 0.0;
059                public boolean insideCordonArea = true;
060        }
061
062        final RoadPricingScheme scheme;
063        final TreeMap<Id<Person>, AgentTollInfo> agents = new TreeMap<>();
064        private final Network network;
065
066        private final TollBehaviourI handler;
067        private Vehicle2DriverEventHandler delegate = new Vehicle2DriverEventHandler();
068
069    @Inject
070        CalcPaidToll(final Network network, final RoadPricingScheme scheme, EventsManager events ) {
071                super();
072                events.addHandler(this);
073                this.network = network;
074                this.scheme = scheme;
075                if (RoadPricingScheme.TOLL_TYPE_DISTANCE.equals(scheme.getType())) {
076                        this.handler = new DistanceTollBehaviour(events);
077                        log.info("just instantiated DistanceTollBehavior") ;
078                } else if (RoadPricingScheme.TOLL_TYPE_AREA.equals(scheme.getType())) {
079                        this.handler = new AreaTollBehaviour(events);
080                        log.info("just instantiated AreaTollBehavior") ;
081                        log.warn("area toll does not work if you have different toll amounts on different " +
082                                        "links.  Make sure you get what you want.  kai, mar'12");
083                } else if (RoadPricingScheme.TOLL_TYPE_CORDON.equals(scheme.getType())) {
084                        this.handler = new CordonTollBehaviour(events);
085                        log.info("just instantiated CordonTollBehavior") ;
086                        log.warn("cordon toll only charges at transition from untolled to tolled. " +
087                                        "Make sure this is what you want. kai, mar'12") ;
088                } else if (RoadPricingScheme.TOLL_TYPE_LINK.equals(scheme.getType())) {
089                        this.handler = new LinkTollBehaviour(events);
090                        log.info("just instantiated LinkTollBehavior") ;
091                } else {
092                        throw new IllegalArgumentException("RoadPricingScheme of type \"" + scheme.getType() + "\" is not supported.");
093                }
094        }
095
096        @Override
097        public void handleEvent(final LinkEnterEvent event) {
098                Link link = this.network.getLinks().get(event.getLinkId());
099                this.handler.handleEvent(event, link);
100        }
101
102        @Override
103        public void handleEvent(final VehicleEntersTrafficEvent event) {
104                delegate.handleEvent(event);
105                
106                Link link = this.network.getLinks().get(event.getLinkId());
107                if (handler instanceof DistanceTollBehaviour || handler instanceof LinkTollBehaviour) {
108                        /* we do not handle wait2link-events for these tolls, because the agent
109                         * should not pay twice for this link, and he (likely) paid already when
110                         * arriving at this link.  */
111                        return;
112                } else {
113                        // Just like a LinkEnterEvent
114                        this.handler.handleEvent(new LinkEnterEvent(event.getTime(), event.getVehicleId(), event.getLinkId()), link);
115                }
116        }

145        /**
146         * Returns the toll the specified agent has paid in the course of the
147         * simulation so far.
148         *
149         * @param agentId
150         * @return The toll paid by the specified agent, 0.0 if no toll was paid.
151         */
152        double getAgentToll(final Id<Person> agentId) {
153                AgentTollInfo info = this.agents.get(agentId);
154                if (info == null) {
155                        return 0.0;
156                }
157                return info.toll;
158        }

160        /**
161         * @return The toll paid by all the agents.
162         */
163        public double getAllAgentsToll() {
164                // public is currently needed. kai, sep'13
165
166                double tolls = 0;
167                for (AgentTollInfo ai : this.agents.values()) {
168                        tolls += (ai == null) ? 0.0 : ai.toll;
169                }
170                return tolls;
171        }

173        /**
174         * @return The Number of all the Drawees.
175         */
176        public int getDraweesNr() {
177                // public is currently needed. kai, sep'13
178
179                int dwCnt = 0;
180                for (AgentTollInfo ai : this.agents.values()) {
181                        if ((ai != null) && (ai.toll > 0.0)) {
182                                dwCnt++;
183                        }
184                }
185                return dwCnt;
186        }

188        /**
189         * A simple interface to implement different toll schemes.
190         */
191        private interface TollBehaviourI {
192                public void handleEvent(LinkEnterEvent event, Link link );
193        }
194
195        /**
196         * Handles the calculation of the distance toll. If an agent enters a link at
197         * a time the toll has to be paid, the toll amount is added to the agent. The
198         * agent does not have to pay the toll for a link if it starts on the link,
199         * as it may have paid already when arriving on the link.
200         */
201        class DistanceTollBehaviour implements TollBehaviourI {
202                private EventsManager events;
203                /**
204                 * @param events The EventsManager to send money events
205                 */
206                public DistanceTollBehaviour(EventsManager events) {
207                        this.events = events;
208                }
209
210                @Override
211                public void handleEvent(final LinkEnterEvent event, final Link link) {
212                        Id<Person> driverId = delegate.getDriverOfVehicle(event.getVehicleId());
213                        Cost cost = CalcPaidToll.this.scheme.getLinkCostInfo(link.getId(),
214                                        event.getTime(), driverId, event.getVehicleId());
215                        if (cost != null) {
216                                double newToll = link.getLength() * cost.amount;
217                                AgentTollInfo info = CalcPaidToll.this.agents.get(driverId);
218                                if (info == null) {
219                                        /* The agent is not yet "registered". */
220                                        
221                                        /* Generate a "registration object. */
222                                        info = new AgentTollInfo();
223                                        
224                                        /* Register it. */
225                                        CalcPaidToll.this.agents.put(driverId, info);
226                                }
227                                info.toll += newToll;
228                                events.processEvent(new PersonMoneyEvent(event.getTime(),driverId,-newToll));
229
230                        }
231                }
232        }

235        class LinkTollBehaviour implements TollBehaviourI {
236                private EventsManager events;
237                /**
238                 * @param events The EventsManager to send money events
239                 */
240                public LinkTollBehaviour(EventsManager events) {
241                        this.events = events;
242
243                }
244
245                @Override
246                public void handleEvent(final LinkEnterEvent event, final Link link) {
247
248                        Id<Person> driverId = delegate.getDriverOfVehicle(event.getVehicleId());
249                        
250                        Cost cost = CalcPaidToll.this.scheme.getLinkCostInfo(link.getId(), event.getTime(), driverId, event.getVehicleId() );
251                        if (cost != null) {
252
253                                AgentTollInfo info = CalcPaidToll.this.agents.get(driverId);
254                                if (info == null) {
255                                        /* The agent is not yet "registered" */
256
257                                        /* Generate a "registration object" */
258                                        info = new AgentTollInfo();
259
260                                        /* Register it. */
261                                        CalcPaidToll.this.agents.put(driverId, info);
262                                }
263                                info.toll += cost.amount;
264                                events.processEvent(new PersonMoneyEvent(event.getTime(),driverId,-cost.amount));
265                        }
266                }
267        }

269        /** Handles the calculation of the area toll. Whenever the agent is seen on
270         * one of the tolled link, the constant toll amount has to be paid once.
271         * <br>
272         * Design comments:
273         * <ul>
274         *              <li> This implementation becomes a problem if someone tries to 
275         *                       implement more than one area which do not share the same flat 
276         *                       fee.  kai, mar'12
277         * </ul> 
278         */
279        class AreaTollBehaviour implements TollBehaviourI {
280                private EventsManager events;
281                /**
282                 * @param events The EventsManager to send money events
283                 */
284                public AreaTollBehaviour(EventsManager events) {
285                        this.events = events;
286                }
287
288                @Override
289                public void handleEvent(final LinkEnterEvent event, final Link link) {
290                        Id<Person> driverId = delegate.getDriverOfVehicle(event.getVehicleId());
291                        Cost cost = CalcPaidToll.this.scheme.getLinkCostInfo(link.getId(), event.getTime(), driverId, event.getVehicleId() );
292                        if (cost != null) {
293                                AgentTollInfo info = CalcPaidToll.this.agents.get(driverId);
294                                if (info == null) {
295                                        /* The agent is not yet "registered" */
296
297                                        /* Generate a "registration object" */
298                                        info = new AgentTollInfo();
299
300                                        /* Register it. */
301                                        CalcPaidToll.this.agents.put(driverId, info);
302
303                                        /* The toll amount comes from the current link, but should 
304                                         * be the same for all links. */
305                                        info.toll = cost.amount;
306                                        events.processEvent(new PersonMoneyEvent(event.getTime(),driverId,-cost.amount));
307                                }
308                        }
309                }
310        }

312        /**
313         * Handles the calculation of the cordon toll. An agent has only to pay if he
314         * crosses the cordon from the outside to the inside.
315         */
316        class CordonTollBehaviour implements TollBehaviourI {
317                private EventsManager events;
318                /**
319                 * @param events The EventsManager to send money events
320                 */
321                public CordonTollBehaviour(EventsManager events) {
322                        this.events = events;
323                }
324
325                @Override
326                public void handleEvent(final LinkEnterEvent event, final Link link) {
327                        Id<Person> driverId = delegate.getDriverOfVehicle(event.getVehicleId());
328                        Cost cost = CalcPaidToll.this.scheme.getLinkCostInfo(link.getId(), event.getTime(), driverId, event.getVehicleId() );
329                        if (cost != null) {
330                                // this is a link inside the toll area.
331                                // [[I guess this assumes that all links inside the cordon are listed in the toll scheme, similar to an area
332                                // toll.  Conventionally, one would not do it in this way, but one would just name those links where
333                                // the cordon toll is charged.  kai, mar'12]]
334                                AgentTollInfo info = CalcPaidToll.this.agents.get(driverId);
335                                if (info == null) {
336                                        // (the agent is not yet "registered")
337                                        // [[yyyy this would refer to any toll, so if we have two cordons, it does not work.  kai, mar'12]]
338
339                                        // generate a "registration object":
340                                        info = new AgentTollInfo();
341
342                                        // register it:
343                                        CalcPaidToll.this.agents.put(driverId, info);
344
345                                        // we start in the area, do not toll:
346                                        info.toll = 0.0;
347
348                                        // info.insideCordonArea is implicitly initialized with `true'. kai, mar'12
349                                } else if (!info.insideCordonArea) {
350                                        // (info is != null, and insideCordonArea is false)
351                                        // agent was outside before, now inside the toll area --> agent has to pay
352                                        info.insideCordonArea = true;
353                                        info.toll += cost.amount;
354                                        events.processEvent(new PersonMoneyEvent(event.getTime(),driverId,-cost.amount));
355                                }
356                                // else: agent was already in toll area, does not have to pay again (this implementation is a bit unusual!)
357                        } else {
358                                // this is a link outside the toll area; just need to memorize that the agent is outside the toll area.
359                                AgentTollInfo info = CalcPaidToll.this.agents.get(driverId);
360                                if (info == null) {
361                                        // (the agent is not yet "registered")
362
363                                        // generate a "registration object":
364                                        info = new AgentTollInfo();
365
366                                        // register it:
367                                        CalcPaidToll.this.agents.put(driverId, info);
368                                }
369                                // memorize that agent is outside toll area:
370                                info.insideCordonArea = false;
371                        }
372                }
373        }
374
375        @Override
376        public void handleEvent(VehicleLeavesTrafficEvent event) {
377                delegate.handleEvent(event);
378        }
