001package org.matsim.roadpricing;
002
003import org.matsim.api.core.v01.TransportMode;
004import org.matsim.core.config.ConfigUtils;
005import org.matsim.core.controler.AbstractModule;
006import org.matsim.core.router.costcalculators.RandomizingTimeDistanceTravelDisutilityFactory;
007import org.matsim.roadpricing.ControlerDefaultsWithRoadPricingModule.RoadPricingInitializer;
008import org.matsim.roadpricing.ControlerDefaultsWithRoadPricingModule.RoadPricingSchemeProvider;
009import org.matsim.roadpricing.ControlerDefaultsWithRoadPricingModule.TravelDisutilityIncludingTollFactoryProvider;
010
011import com.google.inject.Singleton;
012
013public final class RoadPricingModule extends AbstractModule {
014        
015        private RoadPricingScheme scheme;
016
017        public RoadPricingModule() {
018                
019        }
020        
021        public RoadPricingModule( RoadPricingScheme scheme ) {
022                this.scheme = scheme;
023                
024        }
025        
026        @Override
027        public void install() {
028                ConfigUtils.addOrGetModule(getConfig(), RoadPricingConfigGroup.GROUP_NAME, RoadPricingConfigGroup.class);
029                
030                
031                // TODO sort out different ways to set toll schemes; reduce automagic 
032                if ( scheme != null) {
033                        // scheme has come in from the constructor, use that one:
034                        bind(RoadPricingScheme.class).toInstance(scheme);
035                } else {
036                        // no scheme has come in from the constructor, use a class that reads it from file:
037                        bind(RoadPricingScheme.class).toProvider(RoadPricingSchemeProvider.class).in(Singleton.class);
038                }
039                // also add RoadPricingScheme as ScenarioElement.  yyyy TODO might try to get rid of this; binding it is safer
040                bind(RoadPricingInitializer.class).asEagerSingleton();
041                
042                // add the toll to the routing disutility.  also includes "randomizing":
043                addTravelDisutilityFactoryBinding(TransportMode.car).toProvider(TravelDisutilityIncludingTollFactoryProvider.class).asEagerSingleton();
044
045                // specific re-routing strategy for area toll:
046                // yyyy TODO could probably combine them somewhat
047                bind(PlansCalcRouteWithTollOrNot.class);
048                addPlanStrategyBinding("ReRouteAreaToll").toProvider(ReRouteAreaToll.class);
049                addTravelDisutilityFactoryBinding("car_with_payed_area_toll").toInstance(new RandomizingTimeDistanceTravelDisutilityFactory(TransportMode.car, getConfig().planCalcScore()));
050                addRoutingModuleBinding("car_with_payed_area_toll").toProvider(new RoadPricingNetworkRouting());
051                
052                // yyyy TODO It might be possible that the area stuff is adequatly resolved by the randomizing approach.  Would need to try 
053                // that out.  kai, sep'16
054
055                // this is what makes the mobsim compute tolls and generate money events
056                // TODO yyyy could probably combine the following two:
057                addControlerListenerBinding().to(RoadPricingControlerListener.class);
058                bind(CalcPaidToll.class).in(Singleton.class);
059
060                // this is for analysis only:
061                bind(CalcAverageTolledTripLength.class).in(Singleton.class);
062        }
063}
