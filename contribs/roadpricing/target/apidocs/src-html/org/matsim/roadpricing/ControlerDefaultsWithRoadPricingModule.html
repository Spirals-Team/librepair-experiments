040public final class ControlerDefaultsWithRoadPricingModule extends AbstractModule {
041
042        final RoadPricingScheme roadPricingScheme;
043
044        public ControlerDefaultsWithRoadPricingModule() {
045                this.roadPricingScheme = null;
046        }
047
048        public ControlerDefaultsWithRoadPricingModule(RoadPricingScheme roadPricingScheme) {
049                this.roadPricingScheme = roadPricingScheme;
050        }
051
052        @Override
053        public void install() {
054                // This is not optimal yet. Modules should not need to have parameters.
055                // But I am not quite sure yet how to best handle custom scenario elements. mz
056
057                // use ControlerDefaults configuration, replacing the TravelDisutility with a toll-dependent one
058                install(AbstractModule.override(Arrays.<AbstractModule>asList(new ControlerDefaultsModule()), new RoadPricingModule(roadPricingScheme)));
059        }
060
061        static class RoadPricingInitializer {
062                @Inject
063                RoadPricingInitializer(RoadPricingScheme roadPricingScheme, Scenario scenario) {
064                        RoadPricingScheme scenarioRoadPricingScheme = (RoadPricingScheme) scenario.getScenarioElement(RoadPricingScheme.ELEMENT_NAME);
065                        if (scenarioRoadPricingScheme == null) {
066                                scenario.addScenarioElement(RoadPricingScheme.ELEMENT_NAME, roadPricingScheme);
067                        } else {
068                                if (roadPricingScheme != scenarioRoadPricingScheme) {
069                                        throw new RuntimeException();
070                                }
071                        }
072                }
073        }
074
075
076        static class RoadPricingSchemeProvider implements Provider<RoadPricingScheme> {
077
078                private final Config config;
079                private Scenario scenario;
080
081                @Inject
082                RoadPricingSchemeProvider(Config config, Scenario scenario) {
083                        this.config = config;
084                        this.scenario = scenario;
085                }
086
087                @Override
088                public RoadPricingScheme get() {
089                        RoadPricingScheme scenarioRoadPricingScheme = (RoadPricingScheme) scenario.getScenarioElement(RoadPricingScheme.ELEMENT_NAME);
090                        if (scenarioRoadPricingScheme != null) {
091                                return scenarioRoadPricingScheme;
092                        } else {
093                                RoadPricingConfigGroup rpConfig = ConfigUtils.addOrGetModule(config, RoadPricingConfigGroup.GROUP_NAME, RoadPricingConfigGroup.class);
094
095                                if ( rpConfig.getTollLinksFile() == null ) {
096                                        throw new RuntimeException("Road pricing inserted but neither toll links file nor RoadPricingScheme given.  "
097                                                        + "Such an execution path is not allowed.  If you want a base case without toll, "
098                                                        + "construct a zero toll file and insert that. ") ;
099                                }
100                                URL tollLinksFile = ConfigGroup.getInputFileURL(this.config.getContext(), rpConfig.getTollLinksFile());
101                                RoadPricingSchemeImpl rpsImpl = new RoadPricingSchemeImpl() ;
102                                new RoadPricingReaderXMLv1(rpsImpl).parse(tollLinksFile);
103                                return rpsImpl;
104                        }
105                }
106        }
107
108        static class TravelDisutilityIncludingTollFactoryProvider implements Provider<TravelDisutilityFactory> {
109
110                private final Scenario scenario;
111                private final RoadPricingScheme scheme;
112
113                @Inject
114                TravelDisutilityIncludingTollFactoryProvider(Scenario scenario, RoadPricingScheme scheme) {
115                        this.scenario = scenario;
116                        this.scheme = scheme;
117                }
118
119                @Override
120                public TravelDisutilityFactory get() {
121                        final Config config = scenario.getConfig();
122                        final TravelDisutilityFactory originalTravelDisutilityFactory = ControlerDefaults.createDefaultTravelDisutilityFactory(scenario);
123                        //                      if (!scheme.getType().equals(RoadPricingScheme.TOLL_TYPE_AREA)) {
124                        RoadPricingTravelDisutilityFactory travelDisutilityFactory = new RoadPricingTravelDisutilityFactory(
125                                        originalTravelDisutilityFactory, scheme, config.planCalcScore().getMarginalUtilityOfMoney()
126                                        );
127                        travelDisutilityFactory.setSigma(config.plansCalcRoute().getRoutingRandomness());
128                        return travelDisutilityFactory;
129                        //            } else {
130                        //                return originalTravelDisutilityFactory;
131                        //            }
132                }
133
134        }
