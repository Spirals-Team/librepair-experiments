022
023import java.util.ArrayList;
024import java.util.HashMap;
025import java.util.List;
026import java.util.Map;
027import java.util.Set;
028
029import org.apache.log4j.Logger;
030import org.matsim.api.core.v01.Id;

036/**
037 * A road pricing scheme (sometimes also called toll scheme) contains the type of the toll, a list of the
038 * tolled links and the (time-dependent) toll amount agents have to pay.
039 *
040 * @author mrieser
041 */
042public final class RoadPricingSchemeImpl implements RoadPricingScheme {
043        // currently needs to be public. kai, sep'14
044        
045        private static Logger log = Logger.getLogger( RoadPricingSchemeImpl.class ) ;
046
047        private Map<Id<Link>, List<Cost>> linkIds = null;
048
049        private String name = null;
050        private String type = null;
051        private String description = null;
052        private final ArrayList<Cost> costs ;
053
054        private boolean cacheIsInvalid = true;
055        private Cost[] costCache = null;
056
057        public RoadPricingSchemeImpl() {
058                this.linkIds = new HashMap<Id<Link>, List<Cost>>();
059                this.costs = new ArrayList<Cost>();
060        }
061
062        public void addLink(final Id<Link> linkId) {
063                this.linkIds.put(linkId, null);
064        }
065
066        public void setName(final String name) {
067                this.name = name;
068        }
069
070        @Override
071        public String getName() {
072                return this.name;
073        }
074
075        public void setType(final String type) {
076                this.type = type.intern();
077        }
078
079        @Override
080        public String getType() {
081                return this.type;
082        }
083
084        public void setDescription(final String description) {
085                this.description = description;
086        }
087
088        @Override
089        public String getDescription() {
090                return this.description;
091        }
092
093        private static int wrnCnt = 0 ;
094        
095        /**
096         * This is (if I am right) adding a possible toll for <i>all</i> links.  kai, oct'14
097         */
098        public Cost createAndAddCost(final double startTime, final double endTime, final double amount) {
099                if ( startTime==0. && endTime == 24.*3600. ) {
100                        if (wrnCnt < 1) {
101                                wrnCnt++ ;
102                                log.warn("startTime=0:00 and endTime=24:00 means NO toll after 24h (no wrap-around); make sure this is what you want" ) ;
103                                if ( wrnCnt==1 ) {
104                                        log.warn( Gbl.ONLYONCE ) ;
105                                }
106                        }
107                }
108                Cost cost = new Cost(startTime, endTime, amount);
109                this.costs.add(cost);
110                this.cacheIsInvalid = true;
111                return cost;
112        }
113
114        public void addLinkCost(Id<Link> linkId, double startTime, double endTime, double amount) {
115                if ( startTime==0. && endTime == 24.*3600. ) {
116                        if (wrnCnt < 1) {
117                                wrnCnt++ ;
118                                log.warn("startTime=0:00 and endTime=24:00 means NO toll after 24h (no wrap-around); make sure this is what you want" ) ;
119                                if ( wrnCnt==1 ) {
120                                        log.warn( Gbl.ONLYONCE ) ;
121                                }
122                        }
123                }
124                Cost cost = new Cost(startTime, endTime, amount);
125                List<Cost> cs = this.linkIds.get(linkId);
126                if (cs == null) {
127                        cs = new ArrayList<Cost>();
128                        this.linkIds.put(linkId, cs);
129                }
130                cs.add(cost);
131        }
132
133        public boolean removeCost(final Cost cost) {
134                this.cacheIsInvalid = true; // added this without testing it.  kai, nov'13
135                return this.costs.remove(cost);
136        }
137
138        public boolean removeLinkCost(final Id<Link> linkId, final Cost cost){
139                List<Cost> c = this.linkIds.get(linkId);
140                return (c != null) ? c.remove(cost) : false;
141        }
142
143        @Override
144        public Set<Id<Link>> getTolledLinkIds() {
145                return this.linkIds.keySet();
146        }
147        @Override
148        public Map<Id<Link>, List<Cost>> getTypicalCostsForLink(){
149                return this.linkIds;
150        }
151        @Override
152        public Iterable<Cost> getTypicalCosts() {
153                return this.costs;
154        }
155
156        /** @return all Cost objects as an array for faster iteration. */
157        public Cost[] getCostArray() {
158                if (this.cacheIsInvalid) buildCache();
159                return this.costCache.clone();
160        }
161
162        @Override
163        public Cost getLinkCostInfo(final Id<Link> linkId, final double time, Id<Person> personId, Id<Vehicle> vehicleId) {
164                // this is the default road pricing scheme, which ignores the person.  kai, mar'12
165                // Now also added vehicleId as an argument, which is also ignored at the default level. kai, apr'14
166
167                if (this.cacheIsInvalid) buildCache(); //(*)
168
169                if (this.linkIds.containsKey(linkId)) {
170                        // (linkId is in contained in list of tolled links)
171                        
172                        List<Cost> linkSpecificCosts = this.linkIds.get(linkId);
173                        if (linkSpecificCosts == null) {
174                                // (It is expected to have links in the map with null as "value")
175                                // no link specific info found, apply "general" cost (which is in costCache after (*)):
176                                for (Cost cost : this.costCache) {
177                                        if ((time >= cost.startTime) && (time < cost.endTime)) {
178                                                return cost;
179                                        }
180                                }
181                        }
182                        else {
183                                for (Cost cost : linkSpecificCosts){
184                                        if ((time >= cost.startTime) && (time < cost.endTime)) {
185                                                return cost;
186                                        }
187                                }
188                        }
189                }
190                return null;
191        }
192        
193        @Override
194        public Cost getTypicalLinkCostInfo( Id<Link> linkId, double time ) {
195                return this.getLinkCostInfo( linkId, time, null, null ) ;
196        }
197
198        private void buildCache() {
199                this.costCache = new Cost[this.costs.size()];
200                this.costCache = this.costs.toArray(this.costCache);
201                this.cacheIsInvalid = false;
202        }
203        
204        /**
205         * A single, time-dependent toll-amount for a roadpricing scheme.
206         *
207         * @author mrieser
208         */
209        static public class Cost {
210                public final double startTime;
211                public final double endTime;
212                public final double amount;
213
214                public Cost(final double startTime, final double endTime, final double amount) {
215                        this.startTime = startTime;
216                        this.endTime = endTime;
217                        this.amount = amount;
218                }
219                
220                @Override
221                public String toString() {
222                        return "startTime: " + this.startTime + " endTime: " + this.endTime + " amount: " + this.amount ;
223                }
224        }
225}
