021package org.matsim.roadpricing;
022
023import java.util.List;
024import java.util.Map;
025import java.util.Set;
026
027import org.matsim.api.core.v01.Id;
028import org.matsim.api.core.v01.network.Link;
029import org.matsim.api.core.v01.population.Person;
030import org.matsim.roadpricing.RoadPricingSchemeImpl.Cost;
031import org.matsim.vehicles.Vehicle;
032
033/**
034 *
035 * @author nagel, mrieser, michaz
036 */
037public interface RoadPricingScheme {
038        /** the name to which schemes should be associated in Scenario */
039        public static final String ELEMENT_NAME = "roadPricingScheme";
040
041        /**
042         *  The type to be returned by getType() if this is a distance toll,
043         *  i.e. getLinkCostInfo() returns how much must be paid for each m.
044         *
045         */
046        public static final String TOLL_TYPE_DISTANCE = "distance";
047        
048        /** The type to be used for cordon tolls. */
049        public static final String TOLL_TYPE_CORDON = "cordon";
050        
051        /** The type to be used for area tolls. */
052        public static final String TOLL_TYPE_AREA = "area";
053        
054        /** The type to be used for link toll (the toll scheme gives a toll per link). */
055        public static final String TOLL_TYPE_LINK = "link";
056
057        public String getName();
058
059        /**
060         * @return Must return one of the four constants specified above.
061         */
062        public String getType();
063
064        public String getDescription();
065
066        public Set<Id<Link>> getTolledLinkIds();
067
068        /**
069         * Returns the Cost object that contains the active costs for the given link
070         * at the specified time for a given person.  There is no need that the toll 
071         * scheme indeed uses the person information (or the time), and indeed the 
072         * default implementation of this interface does not use it.
073         * 
074         * Note: Implementations of this interface are expected to be pure, stateless 
075         * data containers. Do not, for example, implement a toll which a user must 
076         * only pay once per day by giving a different cost depending on how often 
077         * his method is called. The interpretation ("when do you pay this amount") 
078         * is done by the classes which use this interface, depending on the value 
079         * of getType().
080         * 
081         * <br>
082         * Design issues:<ul>
083         *              <li> I have decided to put the person into the method call rather 
084         *                       than the setPerson construction in TravelDisutility etc.
085         *                       Reason: A big advantage of agent-based simulation over traditional 
086         *                       methods is heterogeneity of agent population. But if we make 
087         *                       this hard to use, the advantage shrinks.  kai, mar'12
088         * 
089         *              <li> It should truly be based on the vehicle (type).  But since 
090         *                       vehicles have not yet universally annealed to a robust state, 
091         *                       I am sticking with persons.  kai, mar'12
092         * 
093         *              <li> In jan'14 Johan is far enough to use this with vehicles rather 
094         *                       than persons.  However, I can imagine situations where there 
095         *                       is also a discount for persons, e.g. disabled or low income.  
096         *                       Since it is also easier to refactor by just adding the vehId, 
097         *                       this is what I will do.  kai, jan'14
098         * </ul>
099         *
100         * @param linkId
101         * @param time
102         * @param vehicleId TODO
103         * @param person TODO
104         * @return The Cost object for the given link at the specified time,
105         * <code>null</code> if the link is either not part of the tolling scheme
106         * or there is no toll at the specified time for the link.
107         */
108        public Cost getLinkCostInfo(final Id<Link> linkId, final double time, Id<Person> personId, Id<Vehicle> vehicleId);
109        
110        /**
111         * This is essentially to enforce that getLinkCostInfo( linkId, time, null, null ) 
112         * is there ... to allow mental modules to work on "approximate" numbers.
113         */
114        public Cost getTypicalLinkCostInfo( final Id<Link> linkId, final double time ) ;
115
116        Iterable<Cost> getTypicalCosts();
117        // yy seems to me that this is not needed; getting Map<Id, List<Cost>> should be enough.  Has something to do with cache and
118        // access speeds.
119
120        Map<Id<Link>, List<Cost>> getTypicalCostsForLink();
121
122}
