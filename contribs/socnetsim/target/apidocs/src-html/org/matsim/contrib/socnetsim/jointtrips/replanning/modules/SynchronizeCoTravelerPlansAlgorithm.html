020package org.matsim.contrib.socnetsim.jointtrips.replanning.modules;
021
022import java.util.List;
023
024import org.apache.log4j.Logger;
025import org.matsim.api.core.v01.population.Activity;
026import org.matsim.api.core.v01.population.Leg;
027import org.matsim.api.core.v01.population.PlanElement;
028import org.matsim.api.core.v01.population.Route;
029import org.matsim.core.router.CompositeStageActivityTypes;
030import org.matsim.core.router.StageActivityTypes;
031import org.matsim.core.utils.misc.Time;
032
033import org.matsim.contrib.socnetsim.jointtrips.population.JointActingTypes;
034import org.matsim.contrib.socnetsim.framework.population.JointPlan;
035import org.matsim.contrib.socnetsim.framework.replanning.GenericPlanAlgorithm;
036import org.matsim.contrib.socnetsim.jointtrips.JointTravelUtils;
037import org.matsim.contrib.socnetsim.jointtrips.JointTravelUtils.JointTravelStructure;
038import org.matsim.contrib.socnetsim.jointtrips.JointTravelUtils.JointTrip;
039
040/**
041 * An algorithm which attempts to synchronize the plans of passengers
042 * with the plans of drivers.
043 * It relies on the travel times stored in the legs for its correctness,
044 * and assumes departure times correspond to activity end times.
045 *
046 * @author thibautd
047 */
048public class SynchronizeCoTravelerPlansAlgorithm implements GenericPlanAlgorithm<JointPlan> {
049        private static final Logger log =
050                Logger.getLogger(SynchronizeCoTravelerPlansAlgorithm.class);
051
052        private final StageActivityTypes stageTypes;
053
054        public SynchronizeCoTravelerPlansAlgorithm(
055                        final StageActivityTypes externalTypes) {
056                final CompositeStageActivityTypes composite = new CompositeStageActivityTypes();
057                composite.addActivityTypes( externalTypes );
058                composite.addActivityTypes( JointActingTypes.JOINT_STAGE_ACTS );
059                this.stageTypes = composite;
060        }
061
062        @Override
063        public void run(final JointPlan plan) {
064                final JointTravelStructure jointTravelStructure =
065                        JointTravelUtils.analyseJointTravel(plan);
066
067                for ( JointTrip jt : jointTravelStructure.getJointTrips() ) {
068                        final double pickUpTime = inferPickUpTime( jt , plan );
069                        if ( pickUpTime < 0 ) {
070                                throw new RuntimeException( "got negative PU time for joint trip "+jt+" in plan "+plan );
071                        }
072                        setPassengerDepartureTime( jt , plan , pickUpTime );
073                }
074        }
075
076        private void setPassengerDepartureTime(
077                        final JointTrip jt,
078                        final JointPlan plan,
079                        final double pickUpTime) {
080                final List<PlanElement> passengerElements =
081                        plan.getIndividualPlan( jt.getPassengerId() ).getPlanElements();
082
083                int ind = passengerElements.indexOf( jt.getPassengerLeg() );
084                if ( ind < 0 ) throw new RuntimeException( "did not found the passenger leg in the plan" );
085
086                double now = pickUpTime;
087                while ( ind > 0 ) {
088                        final PlanElement pe = passengerElements.get( --ind );
089
090                        // assume stage activities have 0 duration
091                        if ( pe instanceof Activity &&
092                                        !stageTypes.isStageActivity( ((Activity) pe).getType() ) ) {
093                                ((Activity) pe).setMaximumDuration( Time.UNDEFINED_TIME );
094                                ((Activity) pe).setEndTime( now > 0 ? now : 0 );
095                                return;
096                        }
097
098                        if ( pe instanceof Leg ) {
099                                final Leg leg = (Leg) pe;
100                                final Route route = leg.getRoute();
101
102                                final double legDur = route != null && route.getTravelTime() != Time.UNDEFINED_TIME ?
103                                        route.getTravelTime() : leg.getTravelTime();
104
105                                if ( legDur != Time.UNDEFINED_TIME ) {
106                                        now -= legDur;
107                                }
108                                else {
109                                        log.warn( "no time in leg "+leg );
110                                }
111                        }
112                }
113
114                throw new RuntimeException( "did not found a passenger departure" );
115        }
116
117        private double inferPickUpTime(
118                        final JointTrip jt,
119                        final JointPlan plan) {
120                final Leg firstDriverLeg = jt.getDriverLegs().get( 0 );
121                final List<PlanElement> driverPlanElements = plan.getIndividualPlan( jt.getDriverId() ).getPlanElements();
122
123                int ind = driverPlanElements.indexOf( firstDriverLeg );
124                if ( ind < 0 ) throw new RuntimeException( "did not found the driver leg in the plan" );
125
126                // parse plan elements backwards, stop at the first act
127                double tt = 0;
128                while (ind > 0) {
129                        final PlanElement pe = driverPlanElements.get( --ind );
130
131                        // assume stage activities have 0 duration
132                        if ( pe instanceof Activity &&
133                                        !stageTypes.isStageActivity( ((Activity) pe).getType() ) ) {
134                                final double endTime = ((Activity) pe).getEndTime();
135                                if ( endTime == Time.UNDEFINED_TIME ) throw new RuntimeException( "undefined end time" );
136                                return endTime + tt;
137                        }
138
139                        if ( pe instanceof Leg ) {
140                                final Leg leg = (Leg) pe;
141                                final Route route = leg.getRoute();
142
143                                final double legDur = route != null && route.getTravelTime() != Time.UNDEFINED_TIME ?
144                                        route.getTravelTime() : leg.getTravelTime();
145
146                                if ( legDur != Time.UNDEFINED_TIME ) {
147                                        tt += legDur;
148                                }
149                                else {
150                                        log.warn( "no time in leg "+leg );
151                                }
152                        }
153                }
154                
155                throw new RuntimeException( "did not found a driver departure" );
156        }
157}
