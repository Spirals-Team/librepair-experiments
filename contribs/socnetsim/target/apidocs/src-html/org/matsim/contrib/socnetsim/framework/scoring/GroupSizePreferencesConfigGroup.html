020package org.matsim.contrib.socnetsim.framework.scoring;
021
022import java.util.Random;
023
024import org.matsim.api.core.v01.population.Person;
025import org.matsim.core.config.Config;
026import org.matsim.core.config.ReflectiveConfigGroup;
027
028/**
029 * @author thibautd
030 */
031public class GroupSizePreferencesConfigGroup extends ReflectiveConfigGroup {
032        public static final String GROUP_NAME = "groupSizePreferences";
033
034        private String activityType = "leisure";
035        private int seed = 1234;
036        private int minPref = 0;
037        private int maxPref = 0;
038
039        private double utilityOfMissingContact_util_s = -6 / 3600d;
040
041        public GroupSizePreferencesConfigGroup() {
042                super( GROUP_NAME );
043        }
044
045        @StringGetter( "activityType" )
046        public String getActivityType() {
047                return activityType;
048        }
049
050        @StringSetter( "activityType" )
051        public void setActivityType( String activityType ) {
052                this.activityType = activityType;
053        }
054
055        @StringGetter( "seed" )
056        public int getSeed() {
057                return seed;
058        }
059
060        @StringSetter( "seed" )
061        public void setSeed( int seed ) {
062                this.seed = seed;
063        }
064
065        @StringGetter( "minPref" )
066        public int getMinPref() {
067                return minPref;
068        }
069
070        @StringSetter( "minPref" )
071        public void setMinPref( int minPref ) {
072                this.minPref = minPref;
073        }
074
075        @StringGetter( "maxPref" )
076        public int getMaxPref() {
077                return maxPref;
078        }
079
080        @StringSetter( "maxPref" )
081        public void setMaxPref( int maxPref ) {
082                this.maxPref = maxPref;
083        }
084
085        public double getUtilityOfMissingContact_util_s() {
086                return utilityOfMissingContact_util_s;
087        }
088
089        @StringGetter( "utilityOfMissingContact_util_h" )
090        public double getUtilityOfMissingContact_util_h() {
091                return getUtilityOfMissingContact_util_s() * 3600d;
092        }
093
094        public void setUtilityOfMissingContact_util_s( double utilityOfMissingContact_util_s ) {
095                this.utilityOfMissingContact_util_s = utilityOfMissingContact_util_s;
096        }
097
098        @StringSetter( "utilityOfMissingContact_util_h" )
099        public void setUtilityOfMissingContact_util_h( double utilityOfMissingContact_util_h ) {
100                setUtilityOfMissingContact_util_s( utilityOfMissingContact_util_h / 3600d );
101        }
102
103        @Override
104        protected void checkConsistency(Config config) {
105                super.checkConsistency(config);
106                if ( maxPref < minPref ) throw new IllegalStateException( "max pref "+maxPref+" lower than min "+minPref+"!" );
107        }
108
109        public int getPersonPreference( final Person person, Config config ) {
110                checkConsistency(config);
111
112                final int personalSeed = seed + person.getId().toString().hashCode();
113                final Random random = new Random( personalSeed );
114                // AH argues this is necessary to be closer to random...
115                for ( int i=0; i < 5; i++ ) random.nextLong();
116
117                return minPref + ((maxPref != minPref) ? random.nextInt( maxPref - minPref ) : 0);
118        }
119}
