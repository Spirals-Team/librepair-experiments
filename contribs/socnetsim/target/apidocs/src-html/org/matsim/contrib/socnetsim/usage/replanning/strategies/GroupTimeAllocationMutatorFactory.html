020package org.matsim.contrib.socnetsim.usage.replanning.strategies;
021
022import javax.inject.Provider;
023
024import org.apache.log4j.Logger;
025import org.matsim.api.core.v01.Scenario;
026import org.matsim.core.config.Config;
027import org.matsim.core.gbl.MatsimRandom;
028import org.matsim.core.population.algorithms.PlanAlgorithm;
029import org.matsim.core.replanning.modules.AbstractMultithreadedModule;
030import org.matsim.core.router.CompositeStageActivityTypes;
031import org.matsim.core.router.TripRouter;
032
033import com.google.inject.Inject;
034
035import org.matsim.contrib.socnetsim.framework.PlanRoutingAlgorithmFactory;
036import org.matsim.contrib.socnetsim.framework.replanning.modules.BlackListedTimeAllocationMutator;
037import org.matsim.contrib.socnetsim.jointtrips.population.JointActingTypes;
038import org.matsim.contrib.socnetsim.framework.population.JointPlans;
039import org.matsim.contrib.socnetsim.framework.replanning.GroupPlanStrategy;
040import org.matsim.contrib.socnetsim.usage.replanning.GroupPlanStrategyFactoryUtils;
041import org.matsim.contrib.socnetsim.framework.replanning.IndividualBasedGroupStrategyModule;
042import org.matsim.contrib.socnetsim.framework.replanning.modules.PlanLinkIdentifier;
043import org.matsim.contrib.socnetsim.framework.replanning.modules.PlanLinkIdentifier.Strong;
044
045public class GroupTimeAllocationMutatorFactory extends AbstractConfigurableSelectionStrategy {
046        private static final Logger log =
047                Logger.getLogger(GroupTimeAllocationMutatorFactory.class);
048
049        private final Scenario sc;
050        private final PlanRoutingAlgorithmFactory planRoutingAlgorithmFactory;
051        private final Provider<TripRouter> tripRouterFactory;
052        private final PlanLinkIdentifier planLinkIdentifier;
053
054        // TODO make configurable again
055        private final double maxTemp = 1;
056
057        @Inject
058        public GroupTimeAllocationMutatorFactory( final Scenario sc , final PlanRoutingAlgorithmFactory planRoutingAlgorithmFactory , final Provider<TripRouter> tripRouterFactory ,
059                        @Strong final PlanLinkIdentifier planLinkIdentifier ) {
060                this.sc = sc;
061                this.planRoutingAlgorithmFactory = planRoutingAlgorithmFactory;
062                this.tripRouterFactory = tripRouterFactory;
063                this.planLinkIdentifier = planLinkIdentifier;
064        }
065
066
067        @Override
068        public GroupPlanStrategy get() {
069                final Config config = sc.getConfig();
070                final GroupPlanStrategy strategy = instantiateStrategy( config );
071                final PlanRoutingAlgorithmFactory planRouterFactory = planRoutingAlgorithmFactory;
072
073                strategy.addStrategyModule(
074                                new IndividualBasedGroupStrategyModule(
075                                        new AbstractMultithreadedModule( config.global().getNumberOfThreads() ) {
076                                                @Override
077                                                public PlanAlgorithm getPlanAlgoInstance() {
078                                                        final CompositeStageActivityTypes blackList = new CompositeStageActivityTypes();
079                                                        blackList.addActivityTypes( tripRouterFactory.get().getStageActivityTypes() );
080                                                        blackList.addActivityTypes( JointActingTypes.JOINT_STAGE_ACTS );
081
082                                                        final int iteration = getReplanningContext().getIteration();
083                                                        final int firstIteration = config.controler().getFirstIteration();
084                                                        final double nIters = config.controler().getLastIteration() - firstIteration;
085                                                        final double minTemp = 1;
086                                                        final double startMin = (2 / 3.) * nIters;
087                                                        final double progress = (iteration - firstIteration) / startMin;
088                                                        final double temp = minTemp + Math.max(1 - progress , 0) * (maxTemp - minTemp);
089                                                        log.debug( "temperature in iteration "+iteration+": "+temp );
090                                                        final BlackListedTimeAllocationMutator algo =
091                                                                        new BlackListedTimeAllocationMutator(
092                                                                                blackList,
093                                                                                config.timeAllocationMutator().getMutationRange() * temp,
094                                                                                MatsimRandom.getLocalInstance() );
095                                                        return algo;
096                                                }
097                                        }));
098
099                strategy.addStrategyModule(
100                                GroupPlanStrategyFactoryUtils.createReRouteModule(
101                                                config,
102                                                planRouterFactory,
103                                                tripRouterFactory) );
104
105                strategy.addStrategyModule(
106                                GroupPlanStrategyFactoryUtils.createSynchronizerModule(
107                                        config,
108                                        tripRouterFactory) );
109
110                strategy.addStrategyModule(
111                                GroupPlanStrategyFactoryUtils.createRecomposeJointPlansModule(
112                                        config,
113                                        ((JointPlans) sc.getScenarioElement( JointPlans.ELEMENT_NAME  )).getFactory(),
114                                        planLinkIdentifier ) );
115
116                return strategy;
117        }
118}
