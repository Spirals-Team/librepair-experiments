020package org.matsim.contrib.socnetsim.utils;
021
022import org.apache.log4j.Logger;
023
024import java.lang.ref.WeakReference;
025import java.util.HashMap;
026import java.util.Map;
027import java.util.WeakHashMap;
028
029/**
030 * A simple object to avoid multiplying identical instances of immutable objects.
031 * It uses a WeakHashMap internally, so that no unnecessary instances are remembered.
032 * It helps achieving nice memory consumption reduction when using with coordinates,
033 * for instance.
034 * <br>
035 * Do not use with mutable objects!
036 *
037 * @author thibautd
038 */
039public class ObjectPool<T extends Object> {
040        private static final Logger log =
041                Logger.getLogger(ObjectPool.class);
042
043        private final Map<T, WeakReference<T>> pool;
044        private long queries = 0;
045        private long unknown = 0;
046
047        public ObjectPool() {
048                this( true );
049        }
050
051        /**
052         * @param weaklyReference if true, pooled instances can be garbage collected if not
053         * referenced anywhere anymore (default)
054         */
055        public ObjectPool( final boolean weaklyReference ) {
056                this.pool = weaklyReference ?
057                        new WeakHashMap<T, WeakReference<T>>() :
058                        // this is sufficient: the Key is a hard reference of the object,
059                        // so the value is not garbage collected.
060                        new HashMap<T, WeakReference<T>>();
061        }
062
063        /**
064         * Get the pooled instance that is equal to the object passed as a parameter,
065         * or pool the object.
066         *
067         * @param object the object for which an instance is searched
068         * @return the pooled instance equal to the object, if any, or the object itself,
069         * after being pooled.
070         */
071        public T getPooledInstance( final T object ) {
072                queries++;
073                WeakReference<T> pooled = pool.get( object );
074
075                if (pooled == null) {
076                        unknown++;
077                        pooled = new WeakReference<T>( object );
078                        pool.put( object , pooled );
079                }
080
081                return pooled.get();
082        }
083
084        public void printStats(final String poolName) {
085                log.info( "["+poolName+"]: "+queries+" queries handled" );
086                log.info( "["+poolName+"]: "+unknown+" instances pooled" );
087                log.info( "["+poolName+"]: => "+(queries - unknown) * 100 / ((double) queries)+"% of the queries returned pooled instances" );
088        }
089}
