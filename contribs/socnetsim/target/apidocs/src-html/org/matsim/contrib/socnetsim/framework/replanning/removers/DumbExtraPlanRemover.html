020package org.matsim.contrib.socnetsim.framework.replanning.removers;
021
022import java.util.ArrayList;
023import java.util.Collection;
024import java.util.Collections;
025import java.util.List;
026
027import org.matsim.api.core.v01.population.Person;
028import org.matsim.api.core.v01.population.Plan;
029
030import org.matsim.contrib.socnetsim.framework.population.JointPlan;
031import org.matsim.contrib.socnetsim.framework.population.JointPlans;
032import org.matsim.contrib.socnetsim.framework.replanning.ExtraPlanRemover;
033import org.matsim.contrib.socnetsim.framework.replanning.grouping.GroupPlans;
034import org.matsim.contrib.socnetsim.framework.replanning.grouping.ReplanningGroup;
035import org.matsim.contrib.socnetsim.framework.replanning.selectors.GroupLevelPlanSelector;
036
037/**
038 * @author thibautd
039 */
040public class DumbExtraPlanRemover implements ExtraPlanRemover {
041        private final GroupLevelPlanSelector selectorForRemoval;
042        private final int maxPlanPerAgent;
043
044        public DumbExtraPlanRemover(
045                        final GroupLevelPlanSelector selectorForRemoval,
046                        final int maxPlanPerAgent ) {
047                this.selectorForRemoval = selectorForRemoval;
048                this.maxPlanPerAgent = maxPlanPerAgent;
049        }
050
051        @Override
052        public boolean removePlansInGroup(
053                        final JointPlans jointPlans,
054                        final ReplanningGroup group) {
055                int c = 0;
056                while ( removeOneExtraPlan( jointPlans , group ) ) c++;
057                if (c > 1) {
058                        // in the clique setting, it is impossible.
059                        // replace by a warning (or nothing) when there is a setting
060                        // in which this makes sense.
061                        throw new RuntimeException( c+" removal iterations were performed for group "+group );
062                }
063                return c > 0;
064        }
065
066        private final boolean removeOneExtraPlan(
067                        final JointPlans jointPlans,
068                        final ReplanningGroup group) {
069                if (maxPlanPerAgent <= 0) return false;
070
071                GroupPlans toRemove = null;
072                boolean somethingDone = false;
073                final List<Person> personsToHandle = new ArrayList<Person>( group.getPersons() );
074                while ( !personsToHandle.isEmpty() ) {
075                        final Person person = personsToHandle.remove( 0 );
076                        if (person.getPlans().size() <= maxPlanPerAgent) continue;
077
078                        somethingDone = true;
079                        if (toRemove == null) {
080                                toRemove = selectorForRemoval.selectPlans( jointPlans , group );
081                                if ( toRemove == null ) {
082                                        throw new RuntimeException(
083                                                        "The selector for removal returned no plan "
084                                                        +"for group "+group );
085                                }
086                        }
087
088                        for (Plan plan : toRemove( jointPlans , person , toRemove )) {
089                                final Person personToHandle = plan.getPerson();
090
091                                if ( personToHandle != person ) {
092                                        final boolean removed = personsToHandle.remove( personToHandle );
093                                        if ( !removed ) throw new RuntimeException( "person "+personToHandle+" is not part of the persons to handle when processing group "+group );
094                                }
095
096                                final boolean removed = personToHandle.getPlans().remove( plan );
097                                if ( !removed ) throw new RuntimeException( "could not remove plan "+plan+" of person "+personToHandle );
098
099                                final JointPlan jpToRemove = jointPlans.getJointPlan( plan );
100                                if ( jpToRemove != null ) jointPlans.removeJointPlan( jpToRemove );
101                        }
102                }
103
104                return somethingDone;
105        }
106
107        private static final Collection<Plan> toRemove(
108                        final JointPlans jointPlans,
109                        final Person person,
110                        final GroupPlans toRemove) {
111                for (Plan plan : person.getPlans()) {
112                        JointPlan jp = jointPlans.getJointPlan( plan );
113
114                        if (jp != null && toRemove.getJointPlans().contains( jp )) {
115                                return jp.getIndividualPlans().values();
116                        }
117
118                        if (jp == null && toRemove.getIndividualPlans().contains( plan )) {
119                                return Collections.singleton( plan );
120                        }
121                }
122                throw new IllegalArgumentException();
123        }
124
125}
