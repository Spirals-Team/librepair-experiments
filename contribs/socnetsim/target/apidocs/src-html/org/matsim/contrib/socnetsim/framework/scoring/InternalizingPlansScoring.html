048/**
049 * @author thibautd
050 */
051public class InternalizingPlansScoring implements PlansScoring, ScoringListener, IterationStartsListener {
052        private static final Logger log =
053                Logger.getLogger(InternalizingPlansScoring.class);
054
055
056        private EventsToScore eventsToScore;
057
058        private final Scenario sc;
059        private final EventsManager events;
060
061        private final InternalizationSettings ratioCalculator;
062
063        @Inject
064        public InternalizingPlansScoring(
065                        final InternalizationSettings ratio,
066                        final Scenario sc,
067                        final EventsManager events,
068                        final ScoringFunctionFactory scoringFunctionFactory) {
069                this.ratioCalculator = ratio;
070                this.sc = sc ;
071                this.events = events ;
072                this.eventsToScore = EventsToScore.createWithScoreUpdating(this.sc, scoringFunctionFactory, events);
073
074        }
075
076
077        @Override
078        public void notifyScoring(final ScoringEvent event) {
079                this.eventsToScore.finish();
080                internalizeAltersScores();
081        }
082
083        private void internalizeAltersScores() {
084                if ( log.isTraceEnabled() ) log.trace( "internalizing alter's scores" );
085
086                // first need to let the scores unmodified, to "internalize" "raw" scores
087                final Map<Id, Double> internalizedScores = new HashMap<Id, Double>();
088
089                for ( Person ego : sc.getPopulation().getPersons().values() ) {
090                        final Collection<? extends Person> alters =
091                                MapUtils.get(
092                                                ratioCalculator.getInternalizationNetwork().getAlters( ego.getId() ),
093                                                sc.getPopulation().getPersons() );
094
095                        double internalizedScore = getScore( ego );
096
097                        if ( log.isTraceEnabled() ) {
098                                log.trace( "internalizing alter's scores for ego "+ego );
099                                log.trace( alters.size()+" alters" );
100                                log.trace( "initial score "+internalizedScore );
101                        }
102
103                        for ( Person alter : alters ) {
104                                if ( alter == null ) throw new NullPointerException( "not all alters part of the population?" );
105                                final double ratio = ratioCalculator.getInternalizationRatio( ego.getId() , alter.getId() );
106                                internalizedScore += ratio * getScore( alter );
107                        }
108
109                        assert !internalizedScores.containsKey( ego.getId() );
110
111                        internalizedScores.put(
112                                        ego.getId(), 
113                                        internalizedScore );
114
115                        if ( log.isTraceEnabled() ) {
116                                log.trace( "FINISHED internalizing alter's scores for ego "+ego );
117                                log.trace( "new score "+internalizedScore );
118                        }
119                }
120
121                for ( Person p : sc.getPopulation().getPersons().values() ) {
122                        assert internalizedScores.containsKey( p.getId() );
123                        setScore( p , internalizedScores.remove( p.getId() ) );
124                }
125
126                assert internalizedScores.isEmpty();
127                if ( log.isTraceEnabled() ) log.trace( "internalizing alter's scores: DONE" );
128        }
129
130        private static void setScore(
131                        final Person p,
132                        final Double score) {
133                if ( score == null ) throw new NullPointerException( "got null score for person "+p );
134                if ( score.isNaN() ) throw new RuntimeException( "got NaN score for person "+p );
135
136                p.getSelectedPlan().setScore( score );
137        }
138
139        private static double getScore(final Person p) {
140                final Double score = p.getSelectedPlan().getScore();
141
142                if ( score == null ) throw new NullPointerException( "got null score for person "+p );
143                if ( score.isNaN() ) throw new RuntimeException( "got NaN score for person "+p );
144
145                return score.doubleValue();
146        }
147
148        @Override
149        public void notifyIterationStarts( final IterationStartsEvent event ) {
150                this.eventsToScore.beginIteration( event.getIteration() );
151        }
152
153        public interface InternalizationSettings {
154                SocialNetwork getInternalizationNetwork();
155                double getInternalizationRatio( Id ego , Id alter );
156        }
157
158        @Singleton
159        public static class ConfigBasedInternalizationSettings implements InternalizationSettings {
160                private final double ratio;
161                private final SocialNetwork network;
162
163                @Inject
164                public ConfigBasedInternalizationSettings(final Scenario sc) {
165                        final InternalizationConfigGroup group = (InternalizationConfigGroup)
166                                        sc.getConfig().getModule( InternalizationConfigGroup.GROUP_NAME );
167
168                        this.network = group.getInternalizationSocialNetworkFile() == null ?
169                                        (SocialNetwork) sc.getScenarioElement( SocialNetwork.ELEMENT_NAME ) :
170                                        readSocialNetwork( group );
171                        this.ratio = calc( group );
172                }
173
174                private SocialNetwork readSocialNetwork(final InternalizationConfigGroup group) {
175                        final Scenario sc = ScenarioUtils.createScenario(ConfigUtils.createConfig() );
176                        // could be added to global scenario somehow
177                        new SocialNetworkReader( "sn" , sc ).readFile(group.getInternalizationSocialNetworkFile());
178                        return (SocialNetwork) sc.getScenarioElement( "sn" );
179                }
180
181                private static double calc( final InternalizationConfigGroup group ) {
182
183                        if ( group == null ) {
184                                log.warn( "no "+InternalizationConfigGroup.GROUP_NAME+" module found in config" );
185                                log.warn( "using null internalization ratio as a consequence" );
186                                return 0;
187                        }
188
189                        final double ratio = group.getInternalizationRatio();
190
191                        final double epsilon = 1E-9;
192
193                        if ( ratio < -epsilon ) {
194                                log.warn( "internalization ratio "+ratio+" is negative!" );
195                                log.warn( "agents will try to DECREASE the score of their alters!" );
196                        }
197
198                        if ( Math.abs( ratio ) < epsilon ) {
199                                log.info( "null internalisation ratio: egoistic agents" );
200                        }
201
202                        if ( Math.abs( ratio - 1 ) < epsilon ) {
203                                log.info( "unit internalisation ratio: agents try to maximize group average utility" );
204                        }
205
206                        if ( ratio > 1 + epsilon ) {
207                                log.warn( "internalization ratio "+ratio+" is greater than one!" );
208                                log.warn( "agents will value the utility of alters higher than their own!" );
209                        }
210
211                        log.info( "setting internalization ratio to "+ratio );
212                        return ratio;
213                }
214
215                @Override
216                public SocialNetwork getInternalizationNetwork() {
217                        return network;
218                }
219
220                @Override
221                public double getInternalizationRatio(final Id ego, final Id alter) {
222                        return ratio;
223                }
224        }
225}
