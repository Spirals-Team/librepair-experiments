020package org.matsim.contrib.socnetsim.usage.replanning.strategies;
021
022import com.google.inject.Inject;
023import com.google.inject.Provider;
024import org.matsim.core.config.Config;
025import org.matsim.contrib.socnetsim.framework.replanning.GroupPlanStrategy;
026import org.matsim.contrib.socnetsim.framework.replanning.selectors.GroupLevelPlanSelector;
027import org.matsim.contrib.socnetsim.usage.replanning.GroupReplanningConfigGroup;
028
029import java.util.Map;
030
031/**
032 * @author thibautd
033 */
034public abstract class AbstractConfigurableSelectionStrategy implements Provider<GroupPlanStrategy> {
035        @Inject
036        private Map<String, Provider<GroupLevelPlanSelector>> factoryRegistry;
037
038        protected GroupPlanStrategy instantiateStrategy( final Config config ) {
039                final GroupReplanningConfigGroup conf = (GroupReplanningConfigGroup)
040                        config.getModule( GroupReplanningConfigGroup.GROUP_NAME );
041
042                return new GroupPlanStrategy( factoryRegistry.get( conf.getSelectorForModification() ).get() );
043        }
044}
