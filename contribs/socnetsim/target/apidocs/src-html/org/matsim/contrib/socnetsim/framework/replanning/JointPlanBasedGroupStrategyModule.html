020package org.matsim.contrib.socnetsim.framework.replanning;
021
022import java.util.ArrayList;
023import java.util.Collection;
024import java.util.HashMap;
025import java.util.List;
026import java.util.Map;
027
028import org.matsim.api.core.v01.Id;
029import org.matsim.api.core.v01.population.Person;
030import org.matsim.api.core.v01.population.Plan;
031import org.matsim.api.core.v01.replanning.PlanStrategyModule;
032import org.matsim.core.replanning.ReplanningContext;
033import org.matsim.core.replanning.modules.AbstractMultithreadedModule;
034
035import org.matsim.contrib.socnetsim.framework.population.JointPlan;
036import org.matsim.contrib.socnetsim.framework.population.JointPlanFactory;
037import org.matsim.contrib.socnetsim.framework.replanning.grouping.GroupPlans;
038
039/**
040 * Delegates to a {@link PlanStrategyModule} which handles JointPlans.
041 * @author thibautd
042 */
043public class JointPlanBasedGroupStrategyModule implements GenericStrategyModule<GroupPlans> {
044        private final boolean wrapIndividualPlansAndActOnThem;
045        private final GenericStrategyModule<JointPlan> delegate;
046        private final JointPlanFactory jointPlanFactory = new JointPlanFactory();
047
048        public JointPlanBasedGroupStrategyModule(
049                        final GenericStrategyModule<JointPlan> module) {
050                this( true , module );
051        }
052
053        public JointPlanBasedGroupStrategyModule(
054                        final boolean wrapIndividualPlansAndActOnThem,
055                        final GenericStrategyModule<JointPlan> module) {
056                this.wrapIndividualPlansAndActOnThem = wrapIndividualPlansAndActOnThem;
057                this.delegate = module;
058        }
059
060        @Override
061        public void handlePlans(
062                        final ReplanningContext replanningContext,
063                        final Collection<GroupPlans> groupPlans) {
064                final List<JointPlan> jointPlans = new ArrayList<JointPlan>();
065                
066                for (GroupPlans groupPlan : groupPlans) {
067                        jointPlans.addAll( groupPlan.getJointPlans() );
068        
069                        if (!wrapIndividualPlansAndActOnThem) return;
070        
071                        for (Plan p : groupPlan.getIndividualPlans()) {
072                                Map<Id<Person>, Plan> fakeJointPlanMap = new HashMap< >();
073                                fakeJointPlanMap.put( p.getPerson().getId() , p );
074                                JointPlan jp = jointPlanFactory.createJointPlan( fakeJointPlanMap , false );
075                                jointPlans.add( jp );
076                        }
077                }
078                
079                delegate.handlePlans( replanningContext , jointPlans );
080        }
081
082        @Override
083        public String toString() {
084                String delegateName = delegate.getClass().getSimpleName();
085                if (delegateName.length() == 0 && delegate instanceof AbstractMultithreadedModule) {
086                        // anonymous class
087                        delegateName = ((AbstractMultithreadedModule) delegate).getPlanAlgoInstance().getClass().getSimpleName();
088                }
089
090                return "["+getClass().getSimpleName()+": "+delegateName+"]";
091        }
092}
