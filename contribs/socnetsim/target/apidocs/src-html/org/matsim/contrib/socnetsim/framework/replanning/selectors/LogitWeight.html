020package org.matsim.contrib.socnetsim.framework.replanning.selectors;
021
022import java.util.Random;
023
024import org.matsim.api.core.v01.population.Plan;
025
026import org.matsim.contrib.socnetsim.framework.replanning.grouping.ReplanningGroup;
027
028public class LogitWeight implements WeightCalculator {
029        private final WeightCalculator baseWeight;
030        private final Random random;
031
032        // this needs to be modifiable to use "annealing"
033        private double scaleParameter;
034        
035        public LogitWeight(
036                        final Random random,
037                        final double scale) {
038                this( new ScoreWeight() , random , scale );
039        }
040
041        public LogitWeight(
042                        final WeightCalculator baseWeight,
043                        final Random random,
044                        final double scale) {
045                this.baseWeight = baseWeight;
046                this.random = random;
047                this.scaleParameter = scale;
048        }
049        
050        @Override
051        public double getWeight(
052                        final Plan indivPlan,
053                        final ReplanningGroup group) {
054                return baseWeight.getWeight( indivPlan , group ) + nextErrorTerm();
055        }
056
057        private double nextErrorTerm() {
058                // "inversion sampling": sample a number between 0 and 1,
059                // and apply the inverse of the CDF to it.
060                final double choice = random.nextDouble();
061
062                double value = Math.log( choice );
063                checkDouble( value );
064
065                value = Math.log( -value );
066                checkDouble( value );
067
068                value = -value / scaleParameter;
069                checkDouble( value );
070
071                return value;
072        }
073
074        private static void checkDouble( final double value ) {
075                if ( Double.isInfinite( value ) || Double.isNaN( value ) || Double.MIN_VALUE == value ) {
076                        throw new RuntimeException( "problematic value "+value );
077                }
078        }
079
080        public void setScaleParameter(final double scaleParameter) {
081                this.scaleParameter = scaleParameter;
082        }
083}
