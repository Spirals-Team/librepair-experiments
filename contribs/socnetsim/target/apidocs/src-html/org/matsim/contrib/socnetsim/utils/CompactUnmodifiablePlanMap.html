019package org.matsim.contrib.socnetsim.utils;
020
021import org.matsim.api.core.v01.Id;
022import org.matsim.api.core.v01.population.Person;
023import org.matsim.api.core.v01.population.Plan;
024
025import java.util.AbstractSet;
026import java.util.Arrays;
027import java.util.Collection;
028import java.util.Comparator;
029import java.util.Iterator;
030import java.util.Map;
031import java.util.Set;
032
033/**
034 * A memory optimized map to store a relation Id to plan.
035 * Designed specifically for the case of joint plans.
036 *
037 * @author thibautd
038 */
039// TODO: make more generic when language level is java 8 (too much boilerplate in 7)
040// would simply need to store a lambda that maps values to Comparable keys.
041public class CompactUnmodifiablePlanMap implements Map<Id<Person>,Plan> {
042        private final Plan[] array;
043
044        public CompactUnmodifiablePlanMap( final Collection<? extends Plan> plans ) {
045                array = plans.toArray( new Plan[ plans.size() ] );
046                Arrays.sort( array, new PlanComparator() );
047        }
048
049        @Override
050        public int size() {
051                return array.length;
052        }
053
054        @Override
055        public boolean isEmpty() {
056                return array.length == 0;
057        }
058
059        @Override
060        public boolean containsKey( final Object key ) {
061                if ( !(key instanceof Id) ) return false;
062                return binarySearch( (Id<Person>) key ) >= 0;
063        }
064
065
066        @Override
067        public boolean containsValue( final Object value ) {
068                if ( !(value instanceof Plan) ) return false;
069                final Plan plan = get( getId( (Plan) value ) );
070                return plan != null && plan.equals( value );
071        }
072
073        @Override
074        public Plan get( final Object key ) {
075                if ( !(key instanceof Id) ) return null;
076                final int index = binarySearch( (Id<Person>) key );
077                return index < 0 ? null : array[ index ];
078        }
079
080        @Override
081        public Plan put( final Id<Person> key, final Plan value ) {
082                throw new UnsupportedOperationException();
083        }
084
085        @Override
086        public Plan remove( final Object key ) {
087                throw new UnsupportedOperationException();
088        }
089
090        @Override
091        public void putAll( final Map<? extends Id<Person>, ? extends Plan> m ) {
092                throw new UnsupportedOperationException();
093        }
094
095        @Override
096        public void clear() {
097                throw new UnsupportedOperationException();
098        }
099
100        @Override
101        public Set<Id<Person>> keySet() {
102                return new AbstractSet<Id<Person>>() {
103                        @Override
104                        public boolean contains( final Object id ) {
105                                return containsKey( id );
106                        }
107
108                        @Override
109                        public Iterator<Id<Person>> iterator() {
110                                return new Iterator<Id<Person>>() {
111                                        int i = 0;
112                                        @Override
113                                        public boolean hasNext() {
114                                                return i < size();
115                                        }
116
117                                        @Override
118                                        public Id<Person> next() {
119                                                return getId( array[ i++ ] );
120                                        }
121
122                                        @Override
123                                        public void remove() {
124                                                throw new UnsupportedOperationException();
125                                        }
126                                };
127                        }
128
129                        @Override
130                        public int size() {
131                                return CompactUnmodifiablePlanMap.this.size();
132                        }
133                };
134        }
135
136        @Override
137        public Collection<Plan> values() {
138                return Arrays.asList( array );
139        }
140
141        @Override
142        public Set<Entry<Id<Person>, Plan>> entrySet() {
143                return new AbstractSet<Entry<Id<Person>,Plan>>() {
144                        @Override
145                        public Iterator<Entry<Id<Person>, Plan>> iterator() {
146                                return new Iterator<Entry<Id<Person>, Plan>>() {
147                                        int i = 0;
148                                        @Override
149                                        public boolean hasNext() {
150                                                return i < size();
151                                        }
152
153                                        @Override
154                                        public Entry<Id<Person>, Plan> next() {
155                                                final int index = i++;
156                                                return new Entry<Id<Person>, Plan>() {
157                                                        @Override
158                                                        public Id<Person> getKey() {
159                                                                return getId( array[ index ] );
160                                                        }
161
162                                                        @Override
163                                                        public Plan getValue() {
164                                                                return array[ index ];
165                                                        }
166
167                                                        @Override
168                                                        public Plan setValue( final Plan value ) {
169                                                                throw new UnsupportedOperationException();
170                                                        }
171                                                };
172                                        }
173
174                                        @Override
175                                        public void remove() {
176                                                throw new UnsupportedOperationException();
177                                        }
178                                };
179                        }
180
181                        @Override
182                        public int size() {
183                                return CompactUnmodifiablePlanMap.this.size();
184                        }
185                };
186        }
187
188        private int binarySearch( final Id<Person> key ) {
189                // adapted from Arrays.binarySearch
190                int low = 0;
191        int high = array.length - 1;
192
193        while (low <= high) {
194            int mid = (low + high) >>> 1;
195            Plan midVal = array[mid];
196            int cmp = midVal.getPerson().getId().compareTo( key );
197            if (cmp < 0)
198                low = mid + 1;
199            else if (cmp > 0)
200                high = mid - 1;
201            else
202                return mid; // key found
203        }
204        return -(low + 1);  // key not found.
205        }
206
207        private Id<Person> getId( final Plan plan ) {
208                return plan.getPerson().getId();
209        }
210
211        private static class PlanComparator implements Comparator<Plan> {
212                @Override
213                public int compare( final Plan o1, final Plan o2 ) {
214                        return o1.getPerson().getId().compareTo( o2.getPerson().getId() );
215                }
216        }
217
218        @Override
219        public String toString() {
220                return "CompactUnmodifiablePlanMap{" +
221                                        "size="+array.length+", " +
222                                        "plans="+Arrays.asList( array )+"}";
223        }
224}
