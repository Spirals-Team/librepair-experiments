054/**
055 * @author thibautd
056 */
057public class JointTripRemoverAlgorithm implements GenericPlanAlgorithm<JointPlan> {
058        private static final Logger log =
059                Logger.getLogger(JointTripRemoverAlgorithm.class);
060
061        private final Random random;
062        private final StageActivityTypes stages;
063        private final StageActivityTypes stagesWithJointTypes;
064        private final MainModeIdentifier mainModeIdentifier;
065
066        public JointTripRemoverAlgorithm(
067                        final Random random,
068                        final StageActivityTypes stages,
069                        final MainModeIdentifier mainModeIdentifier) {
070                this.random = random;
071                this.stages = stages;
072
073                final CompositeStageActivityTypes compositeStages = new CompositeStageActivityTypes();
074                compositeStages.addActivityTypes( stages );
075                compositeStages.addActivityTypes( JointActingTypes.JOINT_STAGE_ACTS );
076                this.stagesWithJointTypes = compositeStages;
077                this.mainModeIdentifier = mainModeIdentifier;
078        }
079
080        @Override
081        public void run(final JointPlan plan) {
082                run( plan , Collections.<Id<Person>>emptyList() );
083        }
084
085        public ActedUponInformation run( final JointPlan plan , final Collection<Id<Person>> agentsToIgnore ) {
086                final JointTravelStructure structure = JointTravelUtils.analyseJointTravel(plan);
087
088                if (structure.getJointTrips().size() == 0) {
089                        log.warn( getClass().getSimpleName()+" was called on a plan with no joint trips."
090                                        +" Make sure it is what you want!" );
091                        return null;
092                }
093
094                final List<JointTrip> choiceSet = getChoiceSet( structure , agentsToIgnore );
095                if (choiceSet.isEmpty()) return null;
096
097                final JointTrip toRemove = choiceSet.get( random.nextInt( choiceSet.size() ) );
098
099                removePassengerTrip( toRemove , plan );
100                removeDriverTrip( toRemove , plan );
101                return new ActedUponInformation(
102                                toRemove.getDriverId(),
103                                toRemove.getPassengerId() );
104        }
105
106        private static List<JointTrip> getChoiceSet(
107                        final JointTravelStructure structure,
108                        final Collection<Id<Person>> agentsToIgnore) {
109                if (agentsToIgnore.isEmpty()) return structure.getJointTrips();
110
111                final List<JointTrip> choiceSet = new ArrayList<JointTrip>();
112
113                for (JointTrip t : structure.getJointTrips()) {
114                        if ( agentsToIgnore.contains( t.getDriverId() ) ) continue;
115                        if ( agentsToIgnore.contains( t.getPassengerId() ) ) continue;
116                        choiceSet.add( t );
117                }
118
119                return choiceSet;
120        }
121
122        // package protected for tests
123        final void removePassengerTrip(
124                        final JointTrip toRemove,
125                        final JointPlan jointPlan) {
126                final Plan passengerPlan = jointPlan.getIndividualPlan( toRemove.getPassengerId() );
127
128                final Trip tripWithLeg =
129                        getTripWithLeg(
130                                        passengerPlan,
131                                        toRemove.getPassengerLeg(),
132                                        stagesWithJointTypes );
133
134                TripRouter.insertTrip(
135                                passengerPlan , 
136                                tripWithLeg.getOriginActivity(),
137                                Collections.singletonList( PopulationUtils.createLeg(TransportMode.pt) ),
138                                tripWithLeg.getDestinationActivity() );
139        }
140
141        private static Trip getTripWithLeg(
142                        final Plan plan,
143                        final Leg leg,
144                        final StageActivityTypes stages) {
145                for ( Trip t : TripStructureUtils.getTrips( plan , stages ) ) {
146                        if ( t.getTripElements().contains( leg ) ) return t;
147                }
148                throw new RuntimeException( plan.getPlanElements() +" doesn't contain "+leg );
149        }
150
151        // package protected for tests
152        final void removeDriverTrip(
153                        final JointTrip toRemove,
154                        final JointPlan plan) {
155                unregisterPassengerFromDriverRoutes( toRemove );
156                repareDriverTrips( toRemove , plan );
157        }
158
159        private void repareDriverTrips(
160                        final JointTrip toRemove,
161                        final JointPlan plan) {
162                final List<TripStructureUtils.Trip> subtrips = getDriverTrip( toRemove , plan.getIndividualPlan( toRemove.getDriverId() ) );
163                final List<PlanElement> newTrip = new ArrayList<PlanElement>();
164                newTrip.add( PopulationUtils.createLeg(TransportMode.car) );
165
166                // "state" variables, changed in the loop:
167                // - keeps track of the passengers currently in the vehicle.
168                //   Pick-up or drop-offs are created at each change
169                Set<Id> currentPassengers = Collections.<Id>emptySet();
170                for ( TripStructureUtils.Trip subtrip : subtrips ) {
171                        final Leg leg = getDriverLegIfItIs( subtrip );
172                        final Route route = leg == null ? null : leg.getRoute();
173
174                        final Set<Id> newPassengers = route != null ?
175                                new HashSet<Id>( ((DriverRoute) route).getPassengersIds() ) :
176                                Collections.<Id>emptySet();
177                        // note that no check of the mode is done...
178                        if ( !newPassengers.equals( currentPassengers ) ) {
179                                newTrip.add(
180                                                PopulationUtils.createActivityFromLinkId(JointActingTypes.INTERACTION, route != null ?
181                                                        route.getStartLinkId() :
182                                                                subtrip.getOriginActivity().getLinkId()) );
183
184                                // as the spatial structure of the trip is modified, it is possible
185                                // that some pre-existing subtours are removed. Thus, a driver that may
186                                // have walked to a pick up (because at the same location as its departure)
187                                // may then have to drive to pick up the second passenger directly if
188                                // the first passenger was removed. Setting all non-driver legs
189                                // to car ensures to have a consistent mode chain.
190                                // XXX It could be done in a smarter way, so that if no subtour is removed, no modification is done
191                                // For instance, when removing an "intern" trip, first PU and last DO are
192                                // left untouched, and thus access and egress leg need not be touched.
193                                newTrip.add( leg != null ? leg : PopulationUtils.createLeg(TransportMode.car) );
194                                currentPassengers = newPassengers;
195                        }
196                }
197
198                TripRouter.insertTrip(
199                                plan.getIndividualPlan( toRemove.getDriverId() ),
200                                subtrips.get( 0 ).getOriginActivity(),
201                                newTrip,
202                                subtrips.get( subtrips.size() - 1 ).getDestinationActivity() );
203        }
204
205        private Leg getDriverLegIfItIs(final Trip subtrip) {
206                if ( !mainModeIdentifier.identifyMainMode( subtrip.getTripElements() ).equals( JointActingTypes.DRIVER ) ) return null;
207                if ( subtrip.getLegsOnly().size() != 1 ) throw new RuntimeException( "unexpected driver subtrip length: "+subtrip );
208                return subtrip.getLegsOnly().get( 0 );
209        }
210
211        private List<TripStructureUtils.Trip> getDriverTrip(
212                        final JointTrip toRemove,
213                        final Plan driverPlan) {
214                final TripStructureUtils.Trip driverTrip = getTripWithLeg( driverPlan , toRemove.getDriverLegs().get( 0 ) , stagesWithJointTypes );
215                assert driverTrip.getTripElements().containsAll( toRemove.getDriverLegs() );
216
217                final List<PlanElement> elements = new ArrayList<PlanElement>();
218                elements.add( driverTrip.getOriginActivity() );
219                elements.addAll( driverTrip.getTripElements() );
220                elements.add( driverTrip.getDestinationActivity() );
221                return TripStructureUtils.getTrips( elements , stages );
222        }
223
224        private static void unregisterPassengerFromDriverRoutes(
225                        final JointTrip toRemove) {
226                for (Leg driverLeg : toRemove.getDriverLegs()) {
227                        final DriverRoute route = (DriverRoute) driverLeg.getRoute();
228                        route.removePassenger( toRemove.getPassengerId() );
229                        if ( route.getPassengersIds().isEmpty() ) {
230                                driverLeg.setMode( TransportMode.car );
231                                driverLeg.setRoute( null );
232                        }
233                }
234        }
235}
