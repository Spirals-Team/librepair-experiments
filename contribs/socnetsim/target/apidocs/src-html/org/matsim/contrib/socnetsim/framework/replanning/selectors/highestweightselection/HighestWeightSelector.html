046/**
047 * Selects the plan combination with the highest (implementation specific)
048 * weight.
049 * <br>
050 * To do so, it iteratively constructs the joint plan using a branch-and-bound
051 * approach, which avoids exploring the full set of combinations.
052 * @author thibautd
053 */
054public final class HighestWeightSelector implements GroupLevelPlanSelector {
055        private static final double EPSILON = 1E-7;
056        private final boolean forbidBlockingCombinations;
057        private final WeightCalculator weightCalculator;
058        private final IncompatiblePlansIdentifierFactory incompFactory;
059
060        public HighestWeightSelector(
061                        final IncompatiblePlansIdentifierFactory incompFactory,
062                        final WeightCalculator weightCalculator) {
063                this( false , incompFactory , weightCalculator );
064        }
065
066        public HighestWeightSelector(
067                        final boolean isForRemoval,
068                        final IncompatiblePlansIdentifierFactory incompFactory,
069                        final WeightCalculator weightCalculator) {
070                this.forbidBlockingCombinations = isForRemoval;
071                this.weightCalculator = weightCalculator;
072                this.incompFactory = incompFactory;
073        }
074        
075        // /////////////////////////////////////////////////////////////////////////
076        // interface and abstract method
077        // /////////////////////////////////////////////////////////////////////////
078        @Override
079        public final GroupPlans selectPlans(
080                        final JointPlans jointPlans,
081                        final ReplanningGroup group) {
082                final IncompatiblePlansIdentifier incompatiblePlansIdentifier =
083                        incompFactory.createIdentifier( jointPlans , group );
084                final Map<Id, PersonRecord> personRecords =
085                        getPersonRecords(
086                                        jointPlans,
087                                        group );
088                final IncompatiblePlanRecords incompatibleRecords =
089                        new IncompatiblePlanRecords(
090                                        incompatiblePlansIdentifier,
091                                        personRecords );
092
093                PersonRecordsPlansPruner.prunePlans(
094                                incompatibleRecords,
095                                personRecords );
096
097                final PlanAllocation allocation = buildPlanString(
098                                new KnownStates(),
099                                new KnownFeasibleAllocations( 20 ),
100                                incompatibleRecords,
101                                SelectorUtils.toSortedList( incompatibleRecords , personRecords ),
102                                personRecords,
103                                forbidBlockingCombinations ? new PlanAllocation() : null,
104                                incompatibleRecords.getAllIncompatibilityGroupIds(),
105                                Double.NEGATIVE_INFINITY).allocation;
106
107                assert allocation == null || allocation.getPlans().size() == group.getPersons().size() :
108                                allocation.getPlans().size()+" != "+group.getPersons().size();
109
110                final GroupPlans plans = allocation == null ? null : SelectorUtils.toGroupPlans( allocation );
111
112                return plans;
113        }

115        // /////////////////////////////////////////////////////////////////////////
116        // "translation" to and from the internal data structures
117        // /////////////////////////////////////////////////////////////////////////

119        private Map<Id, PersonRecord> getPersonRecords(
120                        final JointPlans jointPlans,
121                        final ReplanningGroup group) {
122                final Map<Id, PersonRecord> map = new LinkedHashMap<Id, PersonRecord>();
123                final Map<Plan, Double> weights = new HashMap<Plan, Double>();
124
125                for (Person person : group.getPersons()) {
126                        for (Plan plan : person.getPlans()) {
127                                final double w = weightCalculator.getWeight( plan , group );
128                                if ( Double.isNaN( w ) ) throw new IllegalArgumentException( "NaN weights are not allowed" );
129                                weights.put( plan , w );
130                        }
131                }
132
133                final Map<JointPlan, Collection<PlanRecord>> recordsPerJp = new HashMap<JointPlan, Collection<PlanRecord>>();
134                for (Person person : group.getPersons()) {
135                        final List<PlanRecord> plans = new ArrayList<PlanRecord>();
136                        for (Plan plan : person.getPlans()) {
137                                double w = weights.get( plan );
138                                final JointPlan jp = jointPlans.getJointPlan( plan );
139
140                                if (jp != null) {
141                                        for (Plan p : jp.getIndividualPlans().values()) {
142                                                if (p == plan) continue;
143                                                w += weights.get( p );
144                                        }
145                                        w /= jp.getIndividualPlans().size();
146                                }
147                                
148                                final PlanRecord r = new PlanRecord(
149                                                        plan,
150                                                        jp,
151                                                        w);
152                                plans.add( r );
153                                if ( jp != null ) {
154                                        Collection<PlanRecord> rs = recordsPerJp.get( jp );
155                                        if ( rs == null ) {
156                                                rs = new ArrayList<PlanRecord>();
157                                                recordsPerJp.put( jp , rs );
158                                        }
159                                        rs.add( r );
160                                }
161                        }
162                        final PersonRecord pr = new PersonRecord( person , plans );
163                        map.put(
164                                        person.getId(),
165                                        pr );
166                        for ( PlanRecord p : plans ) {
167                                p.person = pr;
168                        }
169                }
170
171                for (PersonRecord personRecord : map.values()) {
172                        for ( PlanRecord pr : personRecord.plans ) {
173                                if ( pr.jointPlan == null ) continue;
174                                pr.linkedPlans.addAll( recordsPerJp.get( pr.jointPlan ) );
175                                pr.linkedPlans.remove( pr );
176                        }
177                }
178
179                return map;
180        }

182        // /////////////////////////////////////////////////////////////////////////
183        // actual branching and bounding methods
184        // /////////////////////////////////////////////////////////////////////////
185        /**
186         * Recursively decends in the tree of possible joint plans.
187         *
188         * @param allPersonRecord helper map, just links persons to ids
189         * @param personsStillToAllocate in the name
190         * @param alreadyAllocatedPersons set of the ids of the already allocated persons,
191         * used to determine which joint plans are stil possible
192         * @param str the PlanString of the plan constructed until now
193         */
194        private AllocationRecord buildPlanString(
195                        final KnownStates knownStates,
196                        final KnownFeasibleAllocations knownFeasibleAllocations,
197                        final IncompatiblePlanRecords incompatibleRecords,
198                        final List<PersonRecord> personsStillToAllocate,
199                        final Map<Id, PersonRecord> allPersons,
200                        // needed to check whether a leave is forbidden or not
201                        final PlanAllocation currentAllocation,
202                        final Set<Id> allowedIncompatibilityGroups,
203                        final double minimalWeightToObtain) {
204                if ( knownStates != null &&
205                                knownStates.isUnfeasible(
206                                        personsStillToAllocate,
207                                        allowedIncompatibilityGroups,
208                                        minimalWeightToObtain) ) {
209                        assert null ==
210                                        buildPlanString(
211                                                null,
212                                                new KnownFeasibleAllocations( 0 ),
213                                                incompatibleRecords,
214                                                personsStillToAllocate,
215                                                allPersons,
216                                                currentAllocation,
217                                                allowedIncompatibilityGroups,
218                                                minimalWeightToObtain).allocation;
219                        return new AllocationRecord( null , false );
220                }
221
222                /*scope of cachedAlloc*/ {
223                        final PlanAllocation cachedAlloc = knownStates == null ? null :
224                                knownStates.getCached(
225                                        personsStillToAllocate,
226                                        allowedIncompatibilityGroups );
227
228                        if ( cachedAlloc != null ) {
229                                if ( !forbidBlockingCombinations ||
230                                                !SelectorUtils.isBlocking(
231                                                        knownFeasibleAllocations,
232                                                        incompatibleRecords,
233                                                        allPersons,
234                                                        SelectorUtils.toGroupPlans(
235                                                                SelectorUtils.merge(
236                                                                        currentAllocation,
237                                                                        cachedAlloc ) ) ) ) {
238                                        //assert cachedAlloc.equals(
239                                        //              buildPlanString(
240                                        //                      null,
241                                        //                      new KnownFeasibleAllocations( 0 ),
242                                        //                      incompatibleRecords,
243                                        //                      personsStillToAllocate,
244                                        //                      allPersons,
245                                        //                      currentAllocation,
246                                        //                      allowedIncompatibilityGroups,
247                                        //                      Double.NEGATIVE_INFINITY).allocation ):
248                                        //      "cachedWeight="+cachedAlloc.getWeight()+
249                                        //       " searchWeight="+
250                                        //              buildPlanString(
251                                        //                      null,
252                                        //                      new KnownFeasibleAllocations( 0 ),
253                                        //                      incompatibleRecords,
254                                        //                      personsStillToAllocate,
255                                        //                      allPersons,
256                                        //                      currentAllocation,
257                                        //                      allowedIncompatibilityGroups,
258                                        //                      Double.NEGATIVE_INFINITY).allocation.getWeight();
259                                        return new AllocationRecord( cachedAlloc , false );
260                                }
261                        }
262                }
263
264                // do one step forward: "point" to the next person
265                final List<PersonRecord> remainingPersons =
266                        new ArrayList<PersonRecord>( personsStillToAllocate );
267                final PersonRecord currentPerson = remainingPersons.remove(0);
268                assert remainingPersons.size() == personsStillToAllocate.size() - 1;
269
270                // get a list of plans in decreasing order of maximum possible weight.
271                // The weight is always computed on the full joint plan, and thus consists
272                // of the weight until now plus the upper bound
273                final List<PlanRecord> records = new ArrayList<PlanRecord>();
274                for ( PlanRecord r : currentPerson.prunedPlans ) {
275                        if ( r.isStillFeasible ) records.add( r );
276                }
277
278                if ( records.isEmpty() ) return new AllocationRecord( null , false );
279
280                weightPlanRecords(
281                                incompatibleRecords,
282                                records,
283                                remainingPersons );
284
285                // Sort in decreasing order of upper bound: we can stop as soon
286                // as the constructed plan has weight greater than the upper bound
287                // of the next branch.
288                Collections.sort(
289                                records,
290                                new Comparator<PlanRecord>() {
291                                        @Override
292                                        public int compare(
293                                                        final PlanRecord o1,
294                                                        final PlanRecord o2) {
295                                                // sort in DECREASING order
296                                                return Double.compare(
297                                                        o2.cachedMaximumWeight,
298                                                        o1.cachedMaximumWeight );
299                                        }
300                                });
301
302                // get the actual allocation, and stop when the allocation
303                // is better than the maximum possible in remaining plans
304                // or worst than the worst possible at a higher level
305                PlanAllocation constructedString = null;
306                boolean resultsFromBlocking = false;
307                for (PlanRecord r : records) {
308                        if ( constructedString != null &&
309                                        r.cachedMaximumWeight <= constructedString.getWeight()) {
310                                break;
311                        }
312
313                        if ( r.cachedMaximumWeight <= minimalWeightToObtain) {
314                                break;
315                        }
316
317                        assert r.isStillFeasible;
318
319                        final PlanAllocation newAllocation = new PlanAllocation();
320                        newAllocation.add( r );
321                        List<PersonRecord> actuallyRemainingPersons = remainingPersons;
322                        if (r.jointPlan != null) {
323                                assert currentAllocation == null ||
324                                                !SelectorUtils.intersects( r.jointPlan.getIndividualPlans().keySet() , currentAllocation );
325                                assert r.linkedPlans.size() == r.jointPlan.getIndividualPlans().size() -1;
326
327                                newAllocation.addAll( r.linkedPlans );
328                                actuallyRemainingPersons = SelectorUtils.filter( remainingPersons , r.jointPlan );
329
330                                assert actuallyRemainingPersons.size() + r.jointPlan.getIndividualPlans().size() == personsStillToAllocate.size();
331                        }
332
333                        PlanAllocation newString = null;
334
335                        if ( !actuallyRemainingPersons.isEmpty() ) {
336                                final FeasibilityChanger localFeasibilityChanger = new FeasibilityChanger();
337                                SelectorUtils.tagIncompatiblePlansAsInfeasible(
338                                                r,
339                                                incompatibleRecords,
340                                                localFeasibilityChanger);
341
342                                if ( currentAllocation != null ) currentAllocation.addAll( newAllocation.getPlans() );
343                                final AllocationRecord record = buildPlanString(
344                                                knownStates,
345                                                knownFeasibleAllocations,
346                                                incompatibleRecords,
347                                                actuallyRemainingPersons,
348                                                allPersons,
349                                                currentAllocation,
350                                                newIncompatibilityGroups(
351                                                        incompatibleRecords,
352                                                        allowedIncompatibilityGroups,
353                                                        r ),
354                                                actualMinToObtain(
355                                                        constructedString,
356                                                        minimalWeightToObtain)
357                                                        - newAllocation.getWeight() );
358                                if ( currentAllocation != null ) currentAllocation.removeAll( newAllocation.getPlans() );
359                                localFeasibilityChanger.resetFeasibilities();
360
361                                newString = record.allocation;
362
363                                if ( record.resultsFromBlocking ) {
364                                        resultsFromBlocking = true;
365
366                                        assert forbidBlockingCombinations;
367                                        if ( record.allocation == null &&
368                                                        constructedString == null &&
369                                                        knownFeasibleAllocations.isKownAsBlocking(
370                                                                SelectorUtils.toGroupPlans(
371                                                                                currentAllocation ) ) ) {
372                                                // early cuttoff if this branch is always blocking.
373                                                return new AllocationRecord( null , true );
374                                        }
375                                }
376
377                                if ( newString != null ) {
378                                        assert newString.getPlans().size() == actuallyRemainingPersons.size() : newString.getPlans().size()+" plans for "+actuallyRemainingPersons.size()+" agents";
379                                        newString.addAll( newAllocation.getPlans() );
380                                        assert newString.getPlans().size() == personsStillToAllocate.size() : newString.getPlans().size()+" plans for "+personsStillToAllocate.size()+" agents";
381                                }
382                        }
383                        else {
384                                if ( newAllocation.getWeight() <= actualMinToObtain( constructedString , minimalWeightToObtain ) ) continue;
385                                assert !forbidBlockingCombinations || currentAllocation != null;
386
387                                if (forbidBlockingCombinations &&
388                                        SelectorUtils.isBlocking(
389                                                        knownFeasibleAllocations,
390                                                        incompatibleRecords,
391                                                        allPersons,
392                                                        SelectorUtils.toGroupPlans(
393                                                                SelectorUtils.merge(
394                                                                        currentAllocation,
395                                                                        newAllocation ) ) ) ) {
396                                        resultsFromBlocking = true;
397
398                                        if ( constructedString == null &&
399                                                        knownFeasibleAllocations.isKownAsBlocking(
400                                                                SelectorUtils.toGroupPlans(
401                                                                                currentAllocation ) ) ) {
402                                                // early cuttoff if this branch is always blocking
403                                                return new AllocationRecord( null , true );
404                                        }
405                                }
406                                else {
407                                        newString = newAllocation;
408                                }
409                        }
410
411                        if ( newString == null ||
412                                        newString.getWeight() <= actualMinToObtain( constructedString , minimalWeightToObtain ) ) continue;
413
414                        assert newString.getWeight() <= r.cachedMaximumWeight :
415                                getClass()+" weight higher than estimated max: "+newString.getPlans()+" has weight "+newString.getWeight()+" > "+r.cachedMaximumWeight;
416                        assert newString.getWeight() > minimalWeightToObtain :
417                                newString.getWeight() +"<="+ minimalWeightToObtain;
418                        assert constructedString == null || newString.getWeight() > constructedString.getWeight() :
419                                newString.getWeight() +"<="+ minimalWeightToObtain;
420
421                        constructedString = newString;
422                }
423
424                assert constructedString == null || constructedString.getPlans().size() == personsStillToAllocate.size() :
425                        constructedString.getPlans().size()+" plans for "+personsStillToAllocate.size()+" agents";
426
427                if ( knownStates != null && !resultsFromBlocking ) {
428                        knownStates.cache(
429                                        personsStillToAllocate,
430                                        allowedIncompatibilityGroups,
431                                        constructedString ,
432                                        minimalWeightToObtain);
433                }
434
435                return new AllocationRecord( constructedString , resultsFromBlocking );
436        }

438        private static double actualMinToObtain(
439                        final PlanAllocation constructedString,
440                        final double minimalWeightToObtain) {
441                assert constructedString == null || constructedString.getWeight() > minimalWeightToObtain;
442                return Math.max(
443                        minimalWeightToObtain,
444                        constructedString != null ?
445                                constructedString.getWeight() :
446                                Double.NEGATIVE_INFINITY);
447        }

449        private static Set<Id> newIncompatibilityGroups(
450                        final IncompatiblePlanRecords incompatibleRecords,
451                        final Set<Id> allowedIncompatibilityGroups,
452                        final PlanRecord r ) {
453                final Set<Id> forbid = incompatibleRecords.getIncompatibilityGroups( r );
454
455                if ( forbid.isEmpty() ) return allowedIncompatibilityGroups;
456
457                final Set<Id> newSet = new HashSet<Id>( allowedIncompatibilityGroups );
458                newSet.removeAll( forbid );
459
460                return newSet;
461        }

463        private static void weightPlanRecords(
464                        final IncompatiblePlanRecords incompatibleRecords,
465                        final Collection<PlanRecord> records,
466                        final List<PersonRecord> remainingPersons) {
467                final FeasibilityChanger localFeasibilityChanger = new FeasibilityChanger();
468
469                for (PlanRecord r : records) {
470                        if ( r.isStillFeasible ) {
471                                SelectorUtils.tagIncompatiblePlansAsInfeasible(
472                                                r,
473                                                incompatibleRecords,
474                                                localFeasibilityChanger);
475
476                                r.cachedMaximumWeight =
477                                                getMaxWeightFromPersons(
478                                                                r,
479                                                                remainingPersons );
480
481                                localFeasibilityChanger.resetFeasibilities();
482                        }
483                        else {
484                                r.cachedMaximumWeight = Double.NEGATIVE_INFINITY;
485                        }
486                }
487        }

489        /**
490         * Gets the maximum plan weight that can be obtained from the
491         * plans of remainingPersons, given the alradySelected has been
492         * selected, and that planToSelect is about to be selected.
493         */
494        private static double getMaxWeightFromPersons(
495                        final PlanRecord planToSelect,
496                        final List<PersonRecord> remainingPersons) {
497                double score = planToSelect.avgJointPlanWeight;
498
499                for (PersonRecord record : remainingPersons) {
500                        final double max = getMaxWeight(
501                                        planToSelect ,
502                                        record );
503                        // if negative, no need to continue
504                        // moreover, returning here makes sure the branch has infinitely negative
505                        // weight, even if plans in it have infinitely positive weights
506                        if (max == Double.NEGATIVE_INFINITY) return Double.NEGATIVE_INFINITY;
507                        // avoid making the bound too tight, to avoid messing up with the rounding error
508                        score += max + EPSILON;
509                }
510
511                return score;
512        }
513
514        /**
515         * @return the highest weight of a plan wich does not pertains to a joint
516         * plan shared with agents in personsSelected
517         */
518        private static double getMaxWeight(
519                        final PlanRecord planToSelect,
520                        final PersonRecord record) {
521                // case in jp: plan is fully determined
522                if (planToSelect.jointPlan != null) {
523                        final Plan plan = planToSelect.jointPlan.getIndividualPlan( record.person.getId() );
524                        if (plan != null) {
525                                // the "==" part allows to take into account infinite weights
526                                assert record.getRecord( plan ).avgJointPlanWeight == planToSelect.avgJointPlanWeight ||
527                                        Math.abs( record.getRecord( plan ).avgJointPlanWeight - planToSelect.avgJointPlanWeight ) < EPSILON;
528                                return planToSelect.avgJointPlanWeight;
529                        }
530                }
531
532                for (PlanRecord plan : record.prunedPlans ) {
533                        // the plans are sorted by decreasing weight:
534                        // consider the first valid plan
535
536                        if ( plan.isStillFeasible ) return plan.avgJointPlanWeight;
537                        // no need to continue if we now the result can only be infinitely neg.
538                        if ( plan.avgJointPlanWeight == Double.NEGATIVE_INFINITY ) break;
539                }
540
541                // this combination is impossible
542                return Double.NEGATIVE_INFINITY;
543        }

550        public AllocationRecord(
551                        final PlanAllocation allocation,
552                        final boolean resultsFromBlocking) {
553                this.allocation = allocation;
554                this.resultsFromBlocking = resultsFromBlocking;
555        }
