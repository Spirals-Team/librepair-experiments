045/**
046 * @author thibautd
047 */
048@Singleton
049public class CourtesyHistogramListener  implements IterationEndsListener, IterationStartsListener {
050
051        private final CourtesyHistogram histogram;
052        private final boolean outputGraph;
053
054        private static final Logger log = Logger.getLogger(CourtesyHistogramListener.class);
055        private final OutputDirectoryHierarchy controlerIO;
056
057    @Inject
058    CourtesyHistogramListener(Config config, CourtesyHistogram histogram, OutputDirectoryHierarchy controlerIO) {
059        this.controlerIO = controlerIO;
060                this.histogram = histogram;
061                this.outputGraph = config.controler().isCreateGraphs();
062        }
063
064        @Override
065        public void notifyIterationStarts(final IterationStartsEvent event) {
066                this.histogram.reset(event.getIteration());
067        }
068
069        @Override
070        public void notifyIterationEnds(final IterationEndsEvent event) {
071                this.histogram.write(controlerIO.getIterationFilename(event.getIteration(), "courtesyHistogram.txt"));
072                if (this.outputGraph) {
073                        for ( String type : histogram.getDataFrames().keySet() ) {
074                                writeGraphic(
075                                                this.histogram,
076                                                type,
077                                                controlerIO.getIterationFilename(event.getIteration(), "courtesyHistogram_"+type+".png"));
078                        }
079                }
080        }
081
082        static JFreeChart getGraphic(final CourtesyHistogram.DataFrame dataFrame, int iteration, String actType) {
083                final XYSeriesCollection xyData = new XYSeriesCollection();
084                final XYSeries helloSeries = new XYSeries("hello", false, true);
085                final XYSeries goodbyeSerie = new XYSeries("goodbye", false, true);
086                final XYSeries togetherSerie = new XYSeries("pairs together", false, true);
087                int together = 0;
088                for (int i = 0; i < dataFrame.countsHello.length; i++) {
089                        together = together + dataFrame.countsHello[i] - dataFrame.countsGoodbye[i];
090                        double hour = i*dataFrame.binSize / 60.0 / 60.0;
091                        helloSeries.add(hour, dataFrame.countsHello[i]);
092                        goodbyeSerie.add(hour, dataFrame.countsGoodbye[i]);
093                        togetherSerie.add(hour, together);
094                }
095
096                xyData.addSeries(helloSeries);
097                xyData.addSeries(goodbyeSerie);
098                xyData.addSeries(togetherSerie);
099
100        final JFreeChart chart = ChartFactory.createXYStepChart(
101                                "Courtesy Statistics," +
102                                "actType "+actType+
103                                " it." + iteration,
104                                "time", "# persons",
105                                xyData,
106                                PlotOrientation.VERTICAL,
107                                true,   // legend
108                                false,   // tooltips
109                                false   // urls
110                );
111
112                XYPlot plot = chart.getXYPlot();
113
114                final CategoryAxis axis1 = new CategoryAxis("hour");
115                axis1.setTickLabelFont(new Font("SansSerif", Font.PLAIN, 7));
116                plot.setDomainAxis(new NumberAxis("time"));
117
118                plot.getRenderer().setSeriesStroke(0, new BasicStroke(2.0f));
119                plot.getRenderer().setSeriesStroke(1, new BasicStroke(2.0f));
120                plot.getRenderer().setSeriesStroke(2, new BasicStroke(2.0f));
121                plot.setBackgroundPaint(Color.white);
122                plot.setRangeGridlinePaint(Color.gray);
123                plot.setDomainGridlinePaint(Color.gray);
124
125                return chart;
126        }
127
128        public static void writeGraphic(CourtesyHistogram courtesyHistogram, final String actType, final String filename) {
129                try {
130            ChartUtilities.saveChartAsPNG(
131                                        new File(filename),
132                                        getGraphic(
133                                                        courtesyHistogram.getDataFrames().get( actType ),
134                                                        courtesyHistogram.getIteration(),
135                                                        actType), 1024, 768);
136                } catch (IOException e) {
137            throw new UncheckedIOException(e);
138                }
139        }
140}
