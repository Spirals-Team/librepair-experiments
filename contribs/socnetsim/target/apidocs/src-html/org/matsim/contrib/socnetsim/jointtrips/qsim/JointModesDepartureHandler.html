051public class JointModesDepartureHandler implements DepartureHandler , MobsimEngine {
052        private static final Logger log =
053                Logger.getLogger(JointModesDepartureHandler.class);
054
055        private final QVehicleProvider vehicleProvider;
056        private final DepartureHandler departureHandler;
057
058        private final PassengersWaitingPerDriver passengersWaitingPerDriver = new PassengersWaitingPerDriver();
059        // map driverId -> driver info
060        private final Map<Id , WaitingDriver> waitingDrivers =
061                new LinkedHashMap<Id , WaitingDriver>();
062        
063        public JointModesDepartureHandler(
064                        final QNetsimEngine netsimEngine) {
065                this( new NetsimWrappingQVehicleProvider( netsimEngine ),
066                                netsimEngine.getDepartureHandler() );
067        }
068
069        public JointModesDepartureHandler(
070                        final QVehicleProvider vehicleProvider,
071                        final DepartureHandler departureHandler) {
072                this.vehicleProvider = vehicleProvider;
073                this.departureHandler = departureHandler;
074        }
075
076        // /////////////////////////////////////////////////////////////////////////
077        // departure handler
078        // /////////////////////////////////////////////////////////////////////////
079        @Override
080        public boolean handleDeparture(
081                        final double now,
082                        final MobsimAgent agent,
083                        final Id linkId) {
084                final String mode = agent.getMode();
085
086                if ( mode.equals( JointActingTypes.DRIVER ) ) {
087                        if ( log.isTraceEnabled() ) {
088                                log.trace( "Handling DRIVER departure for agent "+agent );
089                        }
090                        handleDriverDeparture( now , agent , linkId );
091                        return true;
092                }
093
094                if ( mode.equals( JointActingTypes.PASSENGER ) ) {
095                        if ( log.isTraceEnabled() ) {
096                                log.trace( "Handling PASSENGER departure for agent "+agent );
097                        }
098                        handlePassengerDeparture( now , agent , linkId );
099                        return true;
100                }
101
102                return departureHandler.handleDeparture( now , agent , linkId );
103        }
104
105        private void handleDriverDeparture(
106                        final double now,
107                        final MobsimAgent agent,
108                        final Id linkId) {
109                assert agent.getCurrentLinkId().equals( linkId ) : agent+" not at link "+linkId;
110
111                final Id<Person> driverId = agent.getId();
112                final Collection<Id<Person>> passengerIds = getPassengerIds( agent );
113                final Id vehicleId = getVehicleId( agent );
114                final MobsimVehicle vehicle = vehicleProvider.getVehicle( vehicleId );
115
116                final Map<Id, PassengerAgent> passengersWaiting =
117                        passengersWaitingPerDriver.getPassengersWaitingDriverAtLink(
118                                        driverId , linkId );
119                final Collection<Id> presentPassengers = new ArrayList<Id>();
120                presentPassengers.addAll( passengersWaiting.keySet() );
121                IdentifiableCollectionsUtils.addAll( presentPassengers , vehicle.getPassengers() );
122
123                // all persons in vehicle should be identified as passengers
124                assert IdentifiableCollectionsUtils.containsAll( passengerIds , vehicle.getPassengers() ) :
125                        passengerIds+" does not contains all of "+vehicle.getPassengers()+" with present passengers "+presentPassengers;
126
127                if ( presentPassengers.containsAll( passengerIds ) ) {
128                        if ( log.isTraceEnabled() ) {
129                                log.trace( "All passengers "+passengerIds+" in present passengers "+presentPassengers );
130                                log.trace( "Processing to departure for driver "+driverId );
131                        }
132
133                        // all passengers are or already in the car,
134                        // or waiting. Board waiting passengers and depart.
135                        for (Id passengerId : passengerIds) {
136                                final PassengerAgent passenger = passengersWaiting.remove( passengerId );
137                                if ( passenger != null ) {
138                                        // say to the driver to board the passenger before leaving the first
139                                        // link. We cannot add the passengers to the vehicle here, as the driver
140                                        // may have to wait for the vehicle to come before departing.
141                                        ((PassengerUnboardingDriverAgent) agent).addPassengerToBoard( passenger );
142                                }
143                                else assert IdentifiableCollectionsUtils.contains( vehicle.getPassengers(), passengerId );
144                        }
145
146                        // this needs to happen before calling the network departure handler!
147                        // in case of A-to-A legs or zero duration activities,
148                        // the departure handler will call arangeAgentNextState before returning.
149                        // This caused problems when there were other joint trips in this sequence.
150                        waitingDrivers.remove( driverId );
151                        if ( log.isTraceEnabled() ) {
152                                log.trace( "driver "+driverId+" removed from waiting list." );
153                                log.trace( "waiting list is now "+waitingDrivers.keySet() );
154                        }
155
156                        // it is possible that not all passengers boarded yet,
157                        // but the car mustn't contain passengers that are not
158                        // identified as such.
159                        assert IdentifiableCollectionsUtils.containsAll( passengerIds , vehicle.getPassengers() ) :
160                                passengerIds+" does not contains all of "+vehicle.getPassengers()+" with present passengers "+presentPassengers;
161
162                        final boolean handled =
163                                departureHandler.handleDeparture(
164                                                now,
165                                                agent,
166                                                linkId );
167
168                        if ( !handled ) {
169                                throw new RuntimeException( "failed to handle departure. Check the main modes?" );
170                        }
171
172                        if ( log.isTraceEnabled() ) {
173                                log.trace( "departure of driver "+driverId+" handled succesfully" );
174                        }
175                }
176                else {
177                        if ( log.isTraceEnabled() ) {
178                                log.trace( "NOT all passengers "+passengerIds+" in present passengers "+presentPassengers );
179                                log.trace( "NOT (yet) processing to departure for driver "+driverId );
180                        }
181
182                        waitingDrivers.put( driverId , new WaitingDriver( agent , linkId ) );
183
184                        if ( log.isTraceEnabled() ) {
185                                log.trace( "driver "+driverId+" added to waiting list." );
186                                log.trace( "waiting list is now "+waitingDrivers.keySet() );
187                        }
188                }
189        }
190
191        private static Id getVehicleId(final MobsimAgent agent) {
192                if ( !(agent instanceof VehicleUsingAgent) ) throw new RuntimeException( agent.getClass().toString() );
193                return ((VehicleUsingAgent) agent).getPlannedVehicleId();
194        }
195
196        private static Collection<Id<Person>> getPassengerIds(final MobsimAgent agent) {
197                if ( !(agent instanceof PlanAgent) ) throw new RuntimeException( agent.getClass().toString() );
198                final Leg currentLeg = (Leg) ((PlanAgent) agent).getCurrentPlanElement();
199                final DriverRoute route = (DriverRoute) currentLeg.getRoute();
200                return route.getPassengersIds();
201        }
202
203        private void handlePassengerDeparture(
204                        final double now,
205                        final MobsimAgent agent,
206                        final Id linkId) {
207                final Id driverId = getDriverId( agent );
208
209                // go in the "queue"
210                final Map<Id, PassengerAgent> waiting =
211                        passengersWaitingPerDriver.getPassengersWaitingDriverAtLink(
212                                        driverId,
213                                        linkId );
214
215                waiting.put( agent.getId() , (PassengerAgent) agent );
216
217                // if the driver is waiting, as him to depart.
218                // departure will succeed only of all passengers are here;
219                // otherwise, everybody will wait for everybody to be here.
220                final WaitingDriver wDriver = waitingDrivers.get( driverId );
221                if ( wDriver != null && wDriver.linkId.equals( linkId ) ) {
222                        if ( log.isTraceEnabled() ) {
223                                log.trace( "driver "+driverId+" found waiting: ask him to depart for passenger "+agent.getId() );
224                        }
225                        handleDriverDeparture(
226                                        now,
227                                        wDriver.driverAgent,
228                                        linkId );
229                }
230                else if ( log.isTraceEnabled() ) {
231                        log.trace( "driver "+driverId+" NOT found waiting: DO NOTHING for passenger "+agent.getId() );
232                        log.trace( "waiting list is "+waitingDrivers.keySet() );
233                }
234        }
235
236        private static Id getDriverId(final MobsimAgent agent) {
237                if ( !(agent instanceof PlanAgent) ) throw new RuntimeException( agent.getClass().toString() );
238                final Leg currentLeg = (Leg) ((PlanAgent) agent).getCurrentPlanElement();
239                final PassengerRoute route = (PassengerRoute) currentLeg.getRoute();
240                return route.getDriverId();
241        }

265        // /////////////////////////////////////////////////////////////////////////

268        private final class WaitingDriver {
269                public final MobsimAgent driverAgent;
270                public final Id linkId;
271
272                public WaitingDriver(
273                                final MobsimAgent agent,
274                                final Id link) {
275                        this.driverAgent = agent;
276                        this.linkId = link;
277                }
278        }
279
280        private final class PassengersWaitingPerDriver {
281                private final Map<Id, PassengersWaitingForDriver> map =
282                        new LinkedHashMap<Id, PassengersWaitingForDriver>();
283
284                public Map<Id, PassengerAgent> getPassengersWaitingDriverAtLink(
285                                final Id driverId,
286                                final Id linkId) {
287                        PassengersWaitingForDriver ps = map.get( driverId );
288
289                        if ( ps == null ) {
290                                ps = new PassengersWaitingForDriver();
291                                map.put( driverId , ps );
292                        }
293
294                        return ps.getPassengersWaitingAtLink( linkId );
295                }
296        }
297
298        private final class PassengersWaitingForDriver {
299                private final Map<Id, Map<Id, PassengerAgent>> agentsAtLink = new LinkedHashMap<Id, Map<Id, PassengerAgent>>();
300
301                public Map<Id, PassengerAgent> getPassengersWaitingAtLink(final Id linkId) {
302                        Map<Id, PassengerAgent> ps = agentsAtLink.get( linkId );
303
304                        if ( ps == null ) {
305                                ps = new LinkedHashMap<Id, PassengerAgent>();
306                                agentsAtLink.put( linkId , ps );
307                        }
308
309                        return ps;
310                }
311        }
