050/**
051 * Randomly mutates activity locations, with a high probability that social contacts
052 * go to the same location.
053 *
054 * It uses the following random local search method:
055 * <ul>
056 * <li> group agents with random social contacts from the replanning group
057 * <li> for each of those group, choose a random flexible type to mutate
058 * <li> choose a random activity of this type (if any) for each agent
059 * <li> mutate the location of those activities to the same random location
060 * </ul>
061 *
062 * the random location is choosen the following way:
063 * <ul>
064 * <li> compute the center of gravity of the current locations of the activities
065 * <li> choose a random direction
066 * <li> sample a distance from a normal distribution, to that in 95% of the case,
067 * it falls bellow the upperBound95 parameter of the config group
068 * <li> choose the facility closest to the corresponding point.
069 * </ul>
070 * @author thibautd
071 */
072public class RandomJointLocationChoiceAlgorithm implements GenericPlanAlgorithm<GroupPlans> {
073        private final Map<String, QuadTree<ActivityFacility>> quadTreePerType;
074        private final ActivityFacilities facilities;
075        private final SocialNetwork socialNetwork;
076        private final RandomJointLocationChoiceConfigGroup config;
077        private final StageActivityTypes activityFilter =
078                new StageActivityTypes() {
079                        @Override
080                        public boolean isStageActivity(final String activityType) {
081                                return !config.getTypes().contains( activityType );
082                        }
083                };
084        final Random random;
085
086        public RandomJointLocationChoiceAlgorithm(
087                        final RandomJointLocationChoiceConfigGroup config,
088                        final ActivityFacilities facilities,
089                        final SocialNetwork socialNetwork) {
090                this.config = config;
091                this.socialNetwork = socialNetwork;
092                this.facilities = facilities;
093                this.quadTreePerType = createQuadTrees( config , facilities );
094                this.random = MatsimRandom.getLocalInstance();
095        }
096
097        private static Map<String, QuadTree<ActivityFacility>> createQuadTrees(
098                        final RandomJointLocationChoiceConfigGroup config,
099                        final ActivityFacilities facilities) {
100                final Map<String, QuadTreeRebuilder<ActivityFacility>> builders = new HashMap<String, QuadTreeRebuilder<ActivityFacility>>();
101
102                final MapUtils.Factory<QuadTreeRebuilder<ActivityFacility>> factory =
103                        new MapUtils.Factory<QuadTreeRebuilder<ActivityFacility>>() {
104                                @Override
105                                public QuadTreeRebuilder<ActivityFacility> create() {
106                                        return new QuadTreeRebuilder<ActivityFacility>();
107                                }
108                        };
109
110                for ( ActivityFacility facility : facilities.getFacilities().values() ) {
111                        final Collection<String> types =
112                                CollectionUtils.intersect(
113                                                facility.getActivityOptions().keySet(),
114                                                config.getTypes() );
115
116                        for ( String t : types ) {
117                                final QuadTreeRebuilder<ActivityFacility> builder =
118                                        MapUtils.getArbitraryObject(
119                                                        t,
120                                                        builders,
121                                                        factory );
122                                builder.put( facility.getCoord() , facility );
123                        }
124                }
125
126                final Map<String, QuadTree<ActivityFacility>> qts = new HashMap<String, QuadTree<ActivityFacility>>();
127                for ( Map.Entry<String, QuadTreeRebuilder<ActivityFacility>> e : builders.entrySet() ) {
128                        qts.put(
129                                        e.getKey(),
130                                        e.getValue().getQuadTree() );
131                }
132
133                return qts;
134        }
135
136        @Override
137        public void run(final GroupPlans plan) {
138                // TODO: do not split joint plans
139                final Map<Id<Person>, Plan> planPerAgent = getPlansMap( plan );
140
141                while ( !planPerAgent.isEmpty() ) {
142                        mutateJointly( removeRandomPlanGroup( planPerAgent ) );
143                }
144        }
145
146        private Collection<Plan> removeRandomPlanGroup(final Map<Id<Person>, Plan> planPerAgent) {
147                final List<Plan> plans = new ArrayList<Plan>();
148
149                final Plan seed = CollectionUtils.removeRandomElement( random , planPerAgent ).getValue();
150                plans.add( seed );
151
152                // TODO: recurse (requires some care)
153                final Set<Id<Person>> availableAlters = CollectionUtils.intersectSorted(
154                                socialNetwork.getAlters( seed.getPerson().getId() ),
155                                planPerAgent.keySet() );
156
157                for ( Id alter : availableAlters ) {
158                        if ( random.nextDouble() < config.getTieActivationProb() ) {
159                                plans.add( planPerAgent.remove( alter ) ); 
160                        }
161                }
162
163                return plans;
164        }
165
166        private static Map<Id<Person>, Plan> getPlansMap(final GroupPlans plan) {
167                final Map<Id<Person>, Plan> planPerAgent = new LinkedHashMap<>();
168
169                for ( Plan p : plan.getAllIndividualPlans() ) {
170                        planPerAgent.put( p.getPerson().getId() , p );
171                }
172
173                return planPerAgent;
174        }
175
176        private void mutateJointly(
177                        final Collection<Plan> plans) {
178                final AgentAndTypeActivityMap actsPerType = new AgentAndTypeActivityMap();
179                for ( Plan p : plans ) {
180                        for ( Activity act : TripStructureUtils.getActivities( p , activityFilter ) ) {
181                                actsPerType.addActivity( p.getPerson().getId() , act );
182                        }
183                }
184
185                if ( actsPerType.getTypes().isEmpty() ) return;
186                final String type = actsPerType.getType( random.nextInt( actsPerType.getTypes().size() ) );
187                assert config.getTypes().contains( type );
188                final Map<Id, List<Activity>> activitiesPerPerson = actsPerType.getActivitiesOfType( type );
189
190                final List<Activity> activitiesToMutate = new ArrayList<Activity>();
191                for ( List<Activity> activitiesOfAgent : activitiesPerPerson.values() ) {
192                        assert !activitiesOfAgent.isEmpty();
193                        activitiesToMutate.add(
194                                        activitiesOfAgent.get(
195                                                random.nextInt(
196                                                        activitiesOfAgent.size() ) ) );
197                }
198
199                mutateLocations(
200                                type,
201                                activitiesToMutate );
202        }
203
204        private void mutateLocations(
205                        final String type,
206                        final List<Activity> activitiesToMutate) {
207                final Coord coordBarycenter = calcBarycenterCoord( activitiesToMutate );
208                final double angle = random.nextDouble() * Math.PI;
209                final double distance = nextNormalDouble() * config.getStandardDeviation();
210
211                final ActivityFacility fac = getFacility(
212                                type,
213                                coordBarycenter,
214                                angle,
215                                distance );
216
217                for ( Activity act : activitiesToMutate ) {
218                        ((Activity) act).setFacilityId( fac.getId() );
219                        ((Activity) act).setLinkId( fac.getLinkId() );
220                        ((Activity) act).setCoord( fac.getCoord() );
221                }
222        }
223
224        /* package (tests) */
225        final Coord calcBarycenterCoord(
226                        final List<Activity> activitiesToMutate ) {
227                double sumX = 0;
228                double sumY = 0;
229
230                for ( Activity act : activitiesToMutate ) {
231                        final Id facilityId = act.getFacilityId();
232                        if ( facilityId == null ) throw new RuntimeException( "no facility for act "+act );
233
234                        final ActivityFacility fac = facilities.getFacilities().get( facilityId );
235                        sumX += fac.getCoord().getX();
236                        sumY += fac.getCoord().getY();
237                }
238
239                return new Coord(sumX / activitiesToMutate.size(), sumY / activitiesToMutate.size());
240        }
241
242        /* package (tests) */
243        final ActivityFacility getFacility(
244                        final String type,
245                        final Coord coordBarycenter,
246                        final double angle,
247                        final double distance ) {
248                final double xLoc = coordBarycenter.getX() + distance * Math.cos( angle );
249                final double yLoc = coordBarycenter.getY() + distance * Math.sin( angle );
250
251                return quadTreePerType.get( type ).getClosest(xLoc, yLoc);
252        }
253
254        private double cachedNormalDouble = Double.NaN;
255        private double nextNormalDouble() {
256                if ( !Double.isNaN( cachedNormalDouble ) ) {
257                        final double toReturn = cachedNormalDouble;
258                        cachedNormalDouble = Double.NaN;
259                        return toReturn;
260                }
261
262                // box-muller transform
263                final double u = random.nextDouble();
264                final double v = random.nextDouble();
265
266                final double r = Math.sqrt( -2 * Math.log( u ) );
267                final double t = v * 2 * Math.PI;
268
269                cachedNormalDouble = r * Math.cos( t );
270
271                return r * Math.sin( t );
272        }

274        private final class AgentAndTypeActivityMap {
275                private final Map<String, Map<Id, List<Activity>>> map = new LinkedHashMap<String, Map<Id, List<Activity>>>();
276
277                public void addActivity( final Id person , final Activity activity ) {
278                        final Map<Id, List<Activity>> activitiesPerPerson =
279                                MapUtils.getMap(
280                                                activity.getType(),
281                                                map );
282                        final List<Activity> activities =
283                                MapUtils.getList(
284                                                person,
285                                                activitiesPerPerson );
286                        activities.add( activity );
287                }
288
289                public Collection<String> getTypes() {
290                        return map.keySet();
291                }
292
293                public String getType(final int index) {
294                        int i=0;
295                        for ( String type : getTypes() ) {
296                                if ( i++ == index ) return type;
297                        }
298                        throw new RuntimeException( "index "+index+" > "+getTypes().size() );
299                }
300
301                public Map<Id, List<Activity>> getActivitiesOfType(final String type) {
302                        return map.get( type );
303                }
304        }
