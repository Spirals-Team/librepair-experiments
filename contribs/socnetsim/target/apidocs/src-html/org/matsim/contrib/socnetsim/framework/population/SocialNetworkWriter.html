020package org.matsim.contrib.socnetsim.framework.population;
021
022import org.apache.log4j.Logger;
023import org.matsim.api.core.v01.Id;
024import org.matsim.api.core.v01.population.Person;
025import org.matsim.core.utils.collections.Tuple;
026import org.matsim.core.utils.io.MatsimXmlWriter;
027import org.matsim.core.utils.misc.Counter;
028
029import java.util.*;
030
031/**
032 * @author thibautd
033 */
034public class SocialNetworkWriter extends MatsimXmlWriter {
035        private static final Logger log =
036                Logger.getLogger(SocialNetworkWriter.class);
037
038        public static final String ROOT_TAG = "socialnet";
039        public static final String EGO_TAG = "ego";
040        public static final String TIE_TAG = "tie";
041
042        public static final String REFLECTIVE_ATT = "isReflective";
043        public static final String EGO_ATT = "egoId";
044        public static final String ALTER_ATT = "alterId";
045
046        public static final String METADATA_TAG = "metadata";
047        public static final String ATTRIBUTE_TAG = "attribute";
048        public static final String NAME_ATT = "name";
049        public static final String VALUE_ATT = "value";
050
051        private final SocialNetwork network;
052
053        public SocialNetworkWriter( final SocialNetwork network ) {
054                this.network = network;
055        }
056
057        public void write( final String file ) {
058                this.openFile( file );
059                log.info( "Start writing social network in file "+file );
060                this.writeXmlHead();
061                this.writeDoctype( ROOT_TAG , "socialnetwork_v1.dtd" );
062                this.writeStartTag(
063                                ROOT_TAG,
064                                Collections.singletonList(
065                                        createTuple(
066                                                REFLECTIVE_ATT,
067                                                network.isReflective() ) ) );
068                writeMetadata();
069                writeEgos();
070                writeNetwork( );
071                this.writeEndTag( ROOT_TAG );
072                this.close();
073                log.info( "Finished writing social network in file "+file );
074        }
075
076        private void writeMetadata() {
077                if ( network.getMetadata().isEmpty() ) return;
078                writeStartTag( METADATA_TAG , Collections.<Tuple<String, String>>emptyList() );
079
080                for ( Map.Entry<String, String> e : network.getMetadata().entrySet() ) {
081                        writeStartTag(
082                                        ATTRIBUTE_TAG,
083                                        Arrays.asList(
084                                                createTuple(
085                                                        NAME_ATT,
086                                                        e.getKey() ),
087                                                createTuple(
088                                                        VALUE_ATT,
089                                                        e.getValue() ) ),
090                                        true);
091                }
092
093                writeEndTag( METADATA_TAG );
094                writeContent( "" , true ); // to skip a line
095        }
096
097        private void writeEgos() {
098                log.info( "start writing egos..." );
099                final Counter counter = new Counter( "[SocialNetworkWriter] writing ego # " );
100                for ( Id ego : network.getEgos() ) {
101                        counter.incCounter();
102                        final List<Tuple<String,String>> atts = new ArrayList<Tuple<String, String>>();
103                        atts.add( createTuple( EGO_ATT , ego.toString() ) );
104                        writeStartTag( EGO_TAG , atts , true );
105                }
106                counter.printCounter();
107                log.info( "finished writing egos." );
108        }
109
110        private void writeNetwork() {
111                log.info( "start writing ties..." );
112                final Counter counter = new Counter( "[SocialNetworkWriter] writing tie # " );
113                final Set<Id> dumpedEgos = new HashSet<Id>();
114                final boolean reflective = network.isReflective();
115                for ( Id ego : network.getEgos() ) {
116                        final Iterable<Id<Person>> alters = network.getAlters( ego );
117                        for ( Id<Person> alter : alters ) {
118                                if ( reflective && dumpedEgos.contains( alter ) ) continue;
119
120                                counter.incCounter();
121                                final List<Tuple<String,String>> atts = new ArrayList<Tuple<String, String>>();
122                                atts.add( createTuple( EGO_ATT , ego.toString() ) );
123                                atts.add( createTuple( ALTER_ATT , alter.toString() ) );
124                                writeStartTag( TIE_TAG , atts , true );
125                        }
126                        dumpedEgos.add( ego );
127                }
128                counter.printCounter();
129                log.info( "finished writing ties." );
130        }
131}
