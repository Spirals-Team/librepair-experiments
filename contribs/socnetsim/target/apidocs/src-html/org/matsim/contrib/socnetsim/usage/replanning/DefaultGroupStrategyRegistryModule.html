020package org.matsim.contrib.socnetsim.usage.replanning;
021
022import com.google.inject.Inject;
023import com.google.inject.Provider;
024import com.google.inject.Singleton;
025import org.apache.log4j.Logger;
026import org.matsim.api.core.v01.Scenario;
027import org.matsim.core.config.Config;
028import org.matsim.core.controler.AbstractModule;
029import org.matsim.contrib.socnetsim.framework.replanning.ExtraPlanRemover;
030import org.matsim.contrib.socnetsim.framework.replanning.GroupPlanStrategy;
031import org.matsim.contrib.socnetsim.framework.replanning.GroupStrategyRegistry;
032import org.matsim.contrib.socnetsim.usage.replanning.GroupReplanningConfigGroup.StrategyParameterSet;
033
034import java.util.Map;
035
036/**
037 * @author thibautd
038 */
039public class DefaultGroupStrategyRegistryModule extends AbstractModule {
040        private static Logger log = Logger.getLogger( DefaultGroupStrategyRegistryModule.class );
041
042        @Override
043    public void install() {
044        install(new DefaultJointStrategiesModule());
045        bind(GroupStrategyRegistry.class).toProvider(GroupStrategyRegistryFiller.class).in(Singleton.class);
046    }
047
048    private static class GroupStrategyRegistryFiller implements Provider<GroupStrategyRegistry> {
049
050                private final Map<String, Provider<GroupPlanStrategy>> strategies;
051                private final Map<String, Provider<ExtraPlanRemover>> removers;
052
053                private final Scenario sc;
054
055                @Inject
056                public GroupStrategyRegistryFiller(
057                                final Map<String, Provider<GroupPlanStrategy>> strategies,
058                                final Map<String, Provider<ExtraPlanRemover>> removers,
059                                final Scenario sc ) {
060                        this.strategies = strategies;
061                        this.removers = removers;
062                        this.sc = sc;
063                }
064
065        @Override
066        public GroupStrategyRegistry get() {
067                        final GroupStrategyRegistry strategyRegistry  = new GroupStrategyRegistry();
068                        final Config config = sc.getConfig();
069                        final GroupReplanningConfigGroup weights = (GroupReplanningConfigGroup) config.getModule( GroupReplanningConfigGroup.GROUP_NAME );
070
071                        strategyRegistry.setExtraPlanRemover( removers.get( weights.getSelectorForRemoval() ).get() );
072
073                        for ( StrategyParameterSet set : weights.getStrategyParameterSets() ) {
074                                final Provider<GroupPlanStrategy> provider = strategies.get( set.getStrategyName() );
075
076                                if ( provider == null ) {
077                                        log.error( "Could not find strategy "+set.getStrategyName() );
078                                        log.error( "valid strategy names are "+strategies.keySet() );
079                                        throw new RuntimeException( "Could not find strategy "+set.getStrategyName() );
080                                }
081                                strategyRegistry.addStrategy(
082                                                provider.get(),
083                                                set.getSubpopulation(),
084                                                set.getWeight(),
085                                                set.isInnovative() ?
086                                                        weights.getDisableInnovationAfterIter() :
087                                                        -1 );
088                        }
089
090            return strategyRegistry;
091        }
092    }
093
094}
