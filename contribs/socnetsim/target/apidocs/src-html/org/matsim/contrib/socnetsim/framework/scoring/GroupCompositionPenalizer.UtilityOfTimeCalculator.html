023import org.matsim.api.core.v01.events.ActivityEndEvent;
024import org.matsim.api.core.v01.events.ActivityStartEvent;
025import org.matsim.api.core.v01.events.Event;
026import org.matsim.contrib.socnetsim.framework.events.CourtesyEvent;
027import org.matsim.core.scoring.SumScoringFunction.ArbitraryEventScoring;
028import org.matsim.core.utils.misc.Time;
029
030import java.util.ArrayDeque;
031import java.util.ArrayList;

037/**
038 * Scoring element that computes a penalty for "joint activities"
039 * if they do not fulfill the desired number of participants.
040 * Each agent has preferences on the number of social contacts with whom it wants
041 * to perform a joint activity with.
042 * A penalty of time, dependent on the number of participants, is applied.
043 *
044 * @author thibautd
045 */
046public class GroupCompositionPenalizer implements ArbitraryEventScoring {
047        private static final Logger log = Logger.getLogger( GroupCompositionPenalizer.class );
048        private final UtilityOfTimeCalculator utilityCalculator;
049
050        private final String activityType;
051
052        private double lastChangeInNCoparticipants = -1;
053        private int currentNCoparticipants = 0;
054
055        private double score = 0;
056
057        public GroupCompositionPenalizer(
058                        final String activityType,
059                        final UtilityOfTimeCalculator utilityCalculator ) {
060                this.activityType = activityType;
061                this.utilityCalculator = utilityCalculator;
062        }
063
064        @Override
065        public void finish() {}
066
067        @Override
068        public double getScore() {
069                return score;
070        }
071
072        @Override
073        public void handleEvent( final Event event ) {
074                if ( event instanceof ActivityStartEvent ) {
075                        startActivity( (ActivityStartEvent) event );
076                }
077                if ( event instanceof ActivityEndEvent ) {
078                        endActivity( (ActivityEndEvent) event );
079                }
080                if ( event instanceof CourtesyEvent ) {
081                        handleCourtesy( (CourtesyEvent) event );
082                }
083        }
084
085        private void startActivity( final ActivityStartEvent event ) {
086                if ( !event.getActType().equals( activityType ) ) {
087                        assert currentNCoparticipants == 0 : currentNCoparticipants;
088                        return;
089                }
090
091                if ( log.isTraceEnabled() ) {
092                        log.trace( "Starting activity at time "+Time.writeTime( event.getTime() ) );
093                }
094
095                lastChangeInNCoparticipants = event.getTime();
096        }
097
098        private void handleCourtesy( final CourtesyEvent event ) {
099                if ( !event.getActType().equals( activityType ) ) return;
100
101                if ( log.isTraceEnabled() ) {
102                        log.trace( "Handle "+event.getType()+" at time "+Time.writeTime( event.getTime() ) );
103                }
104
105                updateScore( event.getTime() );
106
107                lastChangeInNCoparticipants = event.getTime();
108                switch ( event.getType() ) {
109                        case sayGoodbyeEvent:
110                                currentNCoparticipants--;
111                                break;
112                        case sayHelloEvent:
113                                currentNCoparticipants++;
114                                break;
115                        default:
116                                throw new RuntimeException( event.getType()+"?" );
117                }
118
119                if ( log.isTraceEnabled() ) {
120                        log.trace( "Now "+currentNCoparticipants+" participants" );
121                }
122
123                assert currentNCoparticipants >= 0 : event +" -> "+ currentNCoparticipants;
124        }
125
126        private void endActivity( final ActivityEndEvent event ) {
127                if ( !event.getActType().equals( activityType ) ) return;
128
129                if ( log.isTraceEnabled() ) {
130                        log.trace( "Ending activity at time "+Time.writeTime( event.getTime() ) );
131                }
132
133                updateScore( event.getTime() );
134                lastChangeInNCoparticipants = -1;
135        }
136
137        private void updateScore( final double time ) {
138                if (lastChangeInNCoparticipants < 0) return;
139                assert time >= lastChangeInNCoparticipants : time +" < "+ lastChangeInNCoparticipants;
140
141                this.score += utilityCalculator.getUtilityOfTime( currentNCoparticipants ) *
142                        (time - lastChangeInNCoparticipants );
143
144                if ( log.isTraceEnabled() ) {
145                        log.trace( "update score "+Time.writeTime( lastChangeInNCoparticipants )+"->"+Time.writeTime( time )+
146                                        ": currently "+currentNCoparticipants+" participants, score="+score );
147                }
148        }
149
150        public interface UtilityOfTimeCalculator {
151                double getUtilityOfTime( int nCoParticipants );
152        }
153
154        public static class MinGroupSizeLinearUtilityOfTime implements UtilityOfTimeCalculator {
155                private final int minGroupSize;
156                private final double utilOfMissingContact;
157
158                public MinGroupSizeLinearUtilityOfTime(
159                                final int minGroupSize,
160                                final double utilOfMissingContact ) {
161                        this.minGroupSize = minGroupSize;
162
163                        if ( utilOfMissingContact > 0 ) throw new IllegalArgumentException( "util of missing contact expected to be negative. Was "+utilOfMissingContact );
164                        this.utilOfMissingContact = utilOfMissingContact;
165                }
166
167                @Override
168                public double getUtilityOfTime( final int nCoParticipants ) {
169                        if ( nCoParticipants >= minGroupSize ) return 0;
170
171                        final double util = (minGroupSize - nCoParticipants) * utilOfMissingContact;
172                        assert util <= 0 : util;
173
174                        return util;
175                }
176        }
177}
