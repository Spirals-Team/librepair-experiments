020package org.matsim.contrib.socnetsim.jointtrips.scoring;
021
022import org.apache.log4j.Logger;
023import org.matsim.api.core.v01.TransportMode;
024import org.matsim.api.core.v01.network.Network;
025import org.matsim.api.core.v01.population.Leg;
026import org.matsim.api.core.v01.population.Route;
027import org.matsim.core.population.routes.NetworkRoute;
028import org.matsim.core.population.routes.RouteUtils;
029import org.matsim.core.scoring.functions.ScoringParameters;
030import org.matsim.deprecated.scoring.ScoringFunctionAccumulator;
031import org.matsim.core.scoring.SumScoringFunction.LegScoring;
032
033/**
034 * @author thibautd
035 */
036public class ElementalCharyparNagelLegScoringFunction implements LegScoring, ScoringFunctionAccumulator.LegScoring {
037        private static final Logger log =
038                Logger.getLogger(ElementalCharyparNagelLegScoringFunction.class);
039
040        private double score = 0;
041
042        private final String mode;
043        private final LegScoringParameters params;
044        private final Network network;
045
046        public ElementalCharyparNagelLegScoringFunction(
047                        final String mode,
048                        final LegScoringParameters params,
049                        final Network network) {
050                this.mode = mode;
051                this.params = params;
052                this.network = network;
053                this.reset();
054        }
055
056        @Override
057        public void reset() {
058                this.score = 0;
059        }
060
061        @Override
062        public void handleLeg( final Leg leg ) {
063                score += calcLegScore( leg );
064                if ( log.isTraceEnabled() ) {
065                        log.trace( "new score for mode "+mode+" for leg "+leg+": "+score );
066                }
067        }
068
069        @Override
070        public void startLeg(final double time, final Leg leg) {
071                handleLeg( leg );
072        }
073
074        @Override
075        public void endLeg(final double time) {}
076
077        @Override
078        public void finish() {}
079
080        @Override
081        public double getScore() {
082                return this.score;
083        }
084
085        private double calcLegScore(final Leg leg) {
086                if ( !leg.getMode().equals( mode ) ) return 0;
087
088                final double dist =
089                        // I took this because it was in the core function,
090                        // but I am not sure it does what it should,
091                        // as there are several values
092                        // of zero in the IEEE standard...
093                        params.marginalUtilityOfDistance_m != 0.0 ?
094                                getDistance( leg.getRoute() ) :
095                                0;
096
097                // we really do not want that, even if marginal utility of
098                // distance is 0, as any computation including NaN gives NaN.
099                if ( Double.isNaN( dist ) ) throw new RuntimeException( "NaN dist with leg "+leg );
100
101                return params.constant +
102                        params.marginalUtilityOfTraveling_s * leg.getTravelTime() +
103                        params.marginalUtilityOfDistance_m * dist;
104        }
105
106        protected double getDistance(Route route) {
107                if (route instanceof NetworkRoute) {
108                        return  RouteUtils.calcDistanceExcludingStartEndLink((NetworkRoute) route, network);
109                }
110
111                return route.getDistance();
112        }
113
114        public static class LegScoringParameters {
115                private final double constant;
116                private final double marginalUtilityOfTraveling_s;
117                private final double marginalUtilityOfDistance_m;
118
119                public LegScoringParameters(
120                                final double constant,
121                                final double marginalUtilityOfTraveling_s,
122                                final double marginalUtilityOfDistance_m) {
123                        this.constant = constant;
124                        this.marginalUtilityOfTraveling_s = marginalUtilityOfTraveling_s;
125                        this.marginalUtilityOfDistance_m = marginalUtilityOfDistance_m;
126                }
127
128                public static LegScoringParameters createForCar(final ScoringParameters params) {
129                        return new LegScoringParameters(
130                                        params.modeParams.get(TransportMode.car).constant,
131                                        params.modeParams.get(TransportMode.car).marginalUtilityOfTraveling_s,
132                                        params.modeParams.get(TransportMode.car).marginalUtilityOfDistance_m);
133                }
134
135                public static LegScoringParameters createForPt(final ScoringParameters params) {
136                        return new LegScoringParameters(
137                                        params.modeParams.get(TransportMode.pt).constant,
138                                        params.modeParams.get(TransportMode.pt).marginalUtilityOfTraveling_s,
139                                        params.modeParams.get(TransportMode.pt).marginalUtilityOfDistance_m);
140                }
141
142                public static LegScoringParameters createForBike(final ScoringParameters params) {
143                        return new LegScoringParameters(
144                                        params.modeParams.get(TransportMode.bike).constant,
145                                        params.modeParams.get(TransportMode.bike).marginalUtilityOfTraveling_s,
146                                        // Bike has no such setting.
147                                        0 );
148                }
149
150                public static LegScoringParameters createForWalk(final ScoringParameters params) {
151                        return new LegScoringParameters(
152                                        params.modeParams.get(TransportMode.walk).constant,
153                                        params.modeParams.get(TransportMode.walk).marginalUtilityOfTraveling_s,
154                                        params.modeParams.get(TransportMode.walk).marginalUtilityOfDistance_m);
155                }
156        }
157}
