020package org.matsim.contrib.socnetsim.framework.replanning.grouping;
021
022import java.util.ArrayList;
023import java.util.Collection;
024import java.util.Collections;
025import java.util.List;
026
027import org.matsim.api.core.v01.population.Plan;
028import org.matsim.api.core.v01.population.Plan;
029import org.matsim.contrib.socnetsim.framework.population.JointPlan;
030import org.matsim.contrib.socnetsim.framework.population.JointPlanFactory;
031import org.matsim.contrib.socnetsim.framework.population.PlanWithCachedJointPlan;
032
033/**
034 * Stores plans for a group.
035 * For plans pertaining to a joint plan, the joint plan is stored;
036 * for individual plans, the individual plan is stored.
037 * <br>
038 * The collections returned are the interal references.
039 * GroupStrategyModules may modify them if needed.
040 * @author thibautd
041 */
042public class GroupPlans {
043        private final List<JointPlan> jointPlans;
044        private final Collection<JointPlan> unmodifiableJointPlans;
045        private final List<Plan> individualPlans;
046        private final Collection<Plan> unmodifiablePlans;
047
048        public GroupPlans() {
049                this(
050                                Collections.<JointPlan>emptyList(),
051                                Collections.<Plan>emptyList() );
052        }
053
054        public GroupPlans(
055                        final Collection<JointPlan> jointPlans,
056                        final Collection<Plan> individualPlans) {
057                this.jointPlans = new ArrayList<JointPlan>(jointPlans);
058                this.unmodifiableJointPlans = Collections.unmodifiableList( this.jointPlans );
059                this.individualPlans = new ArrayList<Plan>(individualPlans);
060                this.unmodifiablePlans = Collections.unmodifiableList( this.individualPlans );
061        }
062
063        public Collection<JointPlan> getJointPlans() {
064                return this.unmodifiableJointPlans;
065        }
066
067        public Collection<Plan> getIndividualPlans() {
068                return this.unmodifiablePlans;
069        }
070
071        public Collection<Plan> getAllIndividualPlans() {
072                final Collection<Plan> plans = new ArrayList<Plan>();
073
074                plans.addAll( individualPlans );
075                for ( JointPlan jp : jointPlans ) {
076                        plans.addAll( jp.getIndividualPlans().values() );
077                }
078
079                return plans;
080        }
081
082        public void addJointPlan(final JointPlan jp) {
083                jointPlans.add( jp );
084        }
085
086        public void removeJointPlan(final JointPlan jp) {
087                jointPlans.remove( jp );
088        }
089
090        public void addJointPlans(final Collection<JointPlan> jps) {
091                jointPlans.addAll( jps );
092        }
093
094        public void addIndividualPlan(final Plan p) {
095                individualPlans.add( p );
096        }
097
098        public void removeIndividualPlan(final Plan p) {
099                individualPlans.remove( p );
100        }
101
102        public void addIndividualPlans(final Collection<Plan> ps) {
103                individualPlans.addAll( ps );
104        }
105
106        public void clear() {
107                jointPlans.clear();
108                individualPlans.clear();
109        }
110
111        public void clearJointPlans() {
112                jointPlans.clear();
113        }
114
115        public void clearIndividualPlans() {
116                individualPlans.clear();
117        }
118
119        @Override
120        public int hashCode() {
121                return jointPlans.hashCode() + individualPlans.hashCode();
122        }
123
124        @Override
125        public boolean equals(final Object o) {
126                return o instanceof GroupPlans &&
127                        ((GroupPlans) o).jointPlans.size() == jointPlans.size() &&
128                        ((GroupPlans) o).jointPlans.containsAll( jointPlans ) &&
129                        ((GroupPlans) o).individualPlans.size() == individualPlans.size() &&
130                        ((GroupPlans) o).individualPlans.containsAll( individualPlans );
131        }
132
133        @Override
134        public String toString() {
135                return "{GroupPlans: jointPlans="+jointPlans+"; individualPlans="+individualPlans+"}";
136        }
137
138        public static GroupPlans copyPlans(
139                        final JointPlanFactory jointPlanFactory,
140                        final GroupPlans plans) {
141                List<JointPlan> jps = new ArrayList<JointPlan>();
142                List<Plan> ps = new ArrayList<Plan>();
143
144                for (JointPlan jp : plans.getJointPlans()) {
145                        jps.add( jointPlanFactory.copyJointPlan( jp ) );
146                }
147
148                for (Plan p : plans.getIndividualPlans()) {
149                        Plan newPlan = JointPlanFactory.createIndividualPlan( p.getPerson() );
150                        ((PlanWithCachedJointPlan)newPlan).copyFrom( p );
151                        // I think that the above cast will now fail.  It probably worked originally, since JointPlan was an extension of PlanImpl, which is
152                        // no longer allowed. I would, however, say that it was not a clean copy anyways, since it ignored the additional fields
153                        // of JointPlan.  Thibaut, I am confident that you can resolve this if you encounter it, but please let me know if you want to dicuss.
154                        // kai, nov'15
155                        
156                        p.getPerson().addPlan( newPlan );
157                        ps.add( newPlan );
158                }
159
160                return new GroupPlans( jps , ps );
161        }
162}
