020package org.matsim.contrib.socnetsim.framework.replanning.grouping;
021
022import org.matsim.api.core.v01.Id;
023import org.matsim.api.core.v01.population.Person;
024import org.matsim.core.utils.io.IOUtils;
025import org.matsim.core.utils.misc.Counter;
026
027import javax.xml.stream.XMLInputFactory;
028import javax.xml.stream.XMLStreamConstants;
029import javax.xml.stream.XMLStreamException;
030import javax.xml.stream.XMLStreamReader;
031import java.io.InputStream;
032import java.util.ArrayList;
033import java.util.Collection;
034import java.util.List;
035
036/**
037 * @author thibautd
038 */
039public class FixedGroupsIdentifierFileParser {
040        public static FixedGroupsIdentifier readCliquesFile(final String fileName) {
041                return new FixedGroupsIdentifierFileParser().readFile( fileName );
042        }
043
044        private static XMLStreamReader getStreamReader(final String fileName) {
045                final XMLInputFactory xmlif = XMLInputFactory.newInstance();
046                try {
047                        final InputStream stream = IOUtils.getInputStream(fileName);
048                        return xmlif.createXMLStreamReader( stream );
049                }
050                catch (Exception e) {
051                        throw new ParsingException( e );
052                }
053        }
054
055        protected  FixedGroupsIdentifier parse(
056                        final XMLStreamReader streamReader) throws XMLStreamException {
057                final Counter counter = new Counter( "parsing group # " );
058                final List<Collection<Id<Person>>> groups = new ArrayList<>();
059
060                List<Id<Person>> currentGroup = new ArrayList<>();
061                groups.add( currentGroup );
062                while ( streamReader.hasNext() ) {
063                        if ( streamReader.next() != XMLStreamConstants.START_ELEMENT ) continue;
064
065                        if ( streamReader.getLocalName().equals( "clique" ) && 
066                                        currentGroup.size() > 0 ) {
067                                counter.incCounter();
068                                currentGroup = new ArrayList<>();
069                                groups.add( currentGroup );
070                        }
071
072                        if ( streamReader.getLocalName().equals( "person" ) ) {
073                                currentGroup.add( parseId( streamReader , Person.class ) );
074                        }
075                }
076                counter.printCounter();
077
078                return new FixedGroupsIdentifier( groups );
079        }
080
081        private static <T> Id<T> parseId(final XMLStreamReader streamReader, final Class<T> idType ) {
082                if ( streamReader.getAttributeCount() != 1 ) {
083                        throw new ParsingException( "unexpected attribute count "+streamReader.getAttributeCount() );
084                }
085
086                if ( !streamReader.getAttributeLocalName( 0 ).equals( "id" ) ) {
087                        throw new ParsingException( "unexpected attribute name "+streamReader.getAttributeLocalName( 0 ) );
088                }
089
090                return Id.create( streamReader.getAttributeValue( 0 ).intern() , idType);
091        }
092
093        protected FixedGroupsIdentifier readFile(final String fileName) {
094                final XMLStreamReader streamReader = getStreamReader( fileName );
095                try {
096                        return parse( streamReader );
097                }
098                catch (XMLStreamException e) {
099                        throw new ParsingException( e );
100                }
101                finally {
102                        try {
103                                streamReader.close();
104                        } catch (XMLStreamException e) {
105                                throw new ParsingException( e );
106                        }
107                }
108        }
109
110        public static final class ParsingException extends RuntimeException {
111                private static final long serialVersionUID = 1L;
112
113                public ParsingException(final String m) {
114                        super( m );
115                }
116
117                public ParsingException(final Throwable e) {
118                        super( e );
119                }
120
121
122                public ParsingException(final String m, final Throwable e) {
123                        super( m , e );
124                }
125        }
126}
