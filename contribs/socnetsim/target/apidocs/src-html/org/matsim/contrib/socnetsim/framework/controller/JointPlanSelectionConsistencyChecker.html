020package org.matsim.contrib.socnetsim.framework.controller;
021
022import com.google.inject.Inject;
023import org.apache.log4j.Logger;
024import org.matsim.api.core.v01.Scenario;
025import org.matsim.api.core.v01.population.Person;
026import org.matsim.api.core.v01.population.Plan;
027import org.matsim.api.core.v01.population.Population;
028import org.matsim.core.controler.events.IterationEndsEvent;
029import org.matsim.core.controler.events.IterationStartsEvent;
030import org.matsim.core.controler.listener.IterationEndsListener;
031import org.matsim.core.controler.listener.IterationStartsListener;
032import org.matsim.contrib.socnetsim.framework.population.JointPlan;
033import org.matsim.contrib.socnetsim.framework.population.JointPlans;
034
035import java.util.HashMap;
036import java.util.HashSet;
037import java.util.Map;
038import java.util.Set;
039
040public final class JointPlanSelectionConsistencyChecker implements IterationEndsListener, IterationStartsListener {
041        private static final Logger log = Logger.getLogger(JointPlanCompositionMinimalityChecker.class);
042
043        private boolean gotError;
044        private final Population population;
045        private final JointPlans jointPlans;
046
047        @Inject
048        public JointPlanSelectionConsistencyChecker( final Scenario sc ) {
049                this.population = sc.getPopulation();
050                this.jointPlans = (JointPlans) sc.getScenarioElement( JointPlans.ELEMENT_NAME );
051        }
052
053        @Override
054        public void notifyIterationEnds(final IterationEndsEvent event) {
055                log.info( "Checking consistency of joint plan selection" );
056                final Map<JointPlan, Set<Plan>> plansOfJointPlans = new HashMap<JointPlan, Set<Plan>>();
057
058                for ( Person person : population.getPersons().values() ) {
059                        final Plan plan = person.getSelectedPlan();
060                        final JointPlan jp = jointPlans.getJointPlan( plan );
061
062                        if ( jp != null ) {
063                                Set<Plan> plans = plansOfJointPlans.get( jp );
064
065                                if ( plans == null ) {
066                                        plans = new HashSet<Plan>();
067                                        plansOfJointPlans.put( jp , plans );
068                                }
069
070                                plans.add( plan );
071                        }
072                }
073
074                for ( Map.Entry<JointPlan , Set<Plan>> entry : plansOfJointPlans.entrySet() ) {
075                        if ( entry.getKey().getIndividualPlans().size() != entry.getValue().size() ) {
076                                log.error( "joint plan "+entry.getKey()+
077                                                " of size "+entry.getKey().getIndividualPlans().size()+
078                                                " has only the "+entry.getValue().size()+" following plans selected: "+
079                                                entry.getValue() );
080                                gotError = true;
081                        }
082                }
083        }
084
085        @Override
086        public void notifyIterationStarts( IterationStartsEvent event ) {
087                if ( gotError ) throw new RuntimeException( "inconsistency detected. Look at error messages for details" );
088        }
089}
