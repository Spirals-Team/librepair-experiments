044import com.google.inject.Provider;

046/**
047 * @author thibautd
048 */
049public class JointPrismLocationChoiceWithJointTripInsertionStrategyFactory extends AbstractConfigurableSelectionStrategy {
050
051        private final Scenario sc;
052        private final PlanRoutingAlgorithmFactory planRoutingAlgorithmFactory;
053        private final Provider<TripRouter> tripRouterFactory;
054        private final PlanLinkIdentifier planLinkIdentifier;
055        private javax.inject.Provider<TripRouter> tripRouterProvider;
056
057        @Inject
058        public JointPrismLocationChoiceWithJointTripInsertionStrategyFactory(Scenario sc, PlanRoutingAlgorithmFactory planRoutingAlgorithmFactory,
059                                                                                                                                                 Provider<TripRouter> tripRouterFactory, @Strong PlanLinkIdentifier planLinkIdentifier, javax.inject.Provider<TripRouter> tripRouterProvider) {
060                this.sc = sc;
061                this.planRoutingAlgorithmFactory = planRoutingAlgorithmFactory;
062                this.tripRouterFactory = tripRouterFactory;
063                this.planLinkIdentifier = planLinkIdentifier;
064                this.tripRouterProvider = tripRouterProvider;
065        }
066
067
068        @Override
069        public GroupPlanStrategy get() {
070                final GroupPlanStrategy strategy = instantiateStrategy( sc.getConfig() );
071
072                strategy.addStrategyModule(
073                                new PrismicLocationChoiceModule(
074                                        sc, tripRouterProvider) );
075
076                strategy.addStrategyModule(
077                                GroupPlanStrategyFactoryUtils.createJointTripAwareTourModeUnifierModule(
078                                                sc.getConfig(),
079                                                tripRouterFactory) );
080                
081                strategy.addStrategyModule(
082                                GroupPlanStrategyFactoryUtils.createRecomposeJointPlansModule(
083                                        sc.getConfig(),
084                                        ((JointPlans) sc.getScenarioElement( JointPlans.ELEMENT_NAME  )).getFactory(),
085                                        planLinkIdentifier ) );
086                
087                strategy.addStrategyModule(
088                        new JointPlanBasedGroupStrategyModule(
089                                        new AbstractMultithreadedGenericStrategyModule<JointPlan>( sc.getConfig().global() ) {
090                                                @Override
091                                                public GenericPlanAlgorithm<JointPlan> createAlgorithm(ReplanningContext replanningContext) {
092                                                        return new JointTripInsertorAlgorithm(
093                                                                MatsimRandom.getLocalInstance(),
094                                                                (SocialNetwork) sc.getScenarioElement(  SocialNetwork.ELEMENT_NAME ),
095                                                                (JointTripInsertorConfigGroup) sc.getConfig().getModule( JointTripInsertorConfigGroup.GROUP_NAME ),
096                                                                tripRouterFactory.get());
097                                                }
098                                                
099                                                @Override
100                                                protected String getName() {
101                                                        return "JointTripMutator";
102                                                }
103                                        }));
104
105                // TODO: add an option to enable or disable this part?
106                final VehicleRessources vehicles =
107                                (VehicleRessources) sc.getScenarioElement(
108                                        VehicleRessources.ELEMENT_NAME );
109                if ( vehicles != null ) {
110                        strategy.addStrategyModule(
111                                        GroupPlanStrategyFactoryUtils.createVehicleAllocationModule(
112                                                sc.getConfig(),
113                                                vehicles ) );
114                }
115
116                strategy.addStrategyModule(
117                                GroupPlanStrategyFactoryUtils.createReRouteModule(
118                                        sc.getConfig(),
119                                        planRoutingAlgorithmFactory,
120                                        tripRouterFactory ) );
121
122                strategy.addStrategyModule(
123                                GroupPlanStrategyFactoryUtils.createSynchronizerModule(
124                                        sc.getConfig(),
125                                        tripRouterFactory) );
126
127                strategy.addStrategyModule(
128                                GroupPlanStrategyFactoryUtils.createRecomposeJointPlansModule(
129                                        sc.getConfig(),
130                                        ((JointPlans) sc.getScenarioElement( JointPlans.ELEMENT_NAME  )).getFactory(),
131                                        planLinkIdentifier ) );
132
133                return strategy;
134        }
135}
