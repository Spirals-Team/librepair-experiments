046/**
047 *
048 * @author thibautd
049 */
050public class JointScenarioUtils {
051        private JointScenarioUtils() {}
052
053        /**
054         * binds to createScenario(createConfig(configFile));
055         *
056         * @param configFile the path to the configFile
057         * @return a ready to use scenario
058         */
059        public static Scenario loadScenario(final String configFile) {
060                return loadScenario(loadConfig(configFile));
061        }
062
063        public static Scenario createScenario(final Config config) {
064                final Scenario sc = ScenarioUtils.createScenario( config );
065                final RouteFactories rFactory = ((PopulationFactory) sc.getPopulation().getFactory()).getRouteFactories();
066                rFactory.setRouteFactory(
067                                DriverRoute.class,//JointActingTypes.DRIVER,
068                                new DriverRouteFactory());
069                rFactory.setRouteFactory(
070                                PassengerRoute.class,//JointActingTypes.PASSENGER,
071                                new PassengerRouteFactory());
072                return sc;
073        }
074
075        /**
076         * @param config a loaded config
077         * @return a ready to use scenario
078         */
079        public static Scenario loadScenario(final Config config) {
080                final Scenario scenario = createScenario( config );
081                ScenarioUtils.loadScenario( scenario );
082                enrichScenario( scenario );
083                return scenario;
084        }
085
086        public static void enrichScenario(final Scenario scenario) {
087                final Config config = scenario.getConfig();
088                final JointPlansConfigGroup jpConfig = (JointPlansConfigGroup)
089                        config.getModule( JointPlansConfigGroup.GROUP_NAME );
090                if ( jpConfig.getFileName() != null) {
091                        new JointPlansXmlReader( scenario ).readFile( jpConfig.getFileName() );
092                }
093                else {
094                        scenario.addScenarioElement(JointPlans.ELEMENT_NAME, new JointPlans());
095                }
096        }
097
098        public static Config createConfig() {
099                final Config config = ConfigUtils.createConfig();
100
101                addConfigGroups( config );
102
103                return config;
104        }
105
106        public static void addConfigGroups(final Config config) {
107                config.addModule( new CliquesConfigGroup());
108                config.addModule( new JointTripsMutatorConfigGroup());
109                config.addModule( new JointTimeModeChooserConfigGroup());
110                config.addModule( new JointTripInsertorConfigGroup());
111                config.addModule( new JointPlansConfigGroup());
112                config.addModule( new GroupReplanningConfigGroup() );
113                config.addModule( new SocialNetworkConfigGroup() );
114                config.addModule( new RandomJointLocationChoiceConfigGroup() );
115                config.addModule( new PlanLinkConfigGroup() );
116                config.addModule( new PrismicLocationChoiceConfigGroup() );
117                config.addModule( new InternalizationConfigGroup() );
118        }
119
120        public static Config loadConfig(final String configFile) {
121                final Config config = createConfig();
122                new ConfigReader( config ).readFile( configFile );
123                return config;
124        }
125}
