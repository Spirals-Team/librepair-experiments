060/**
061 * An algorithm which creates joint trips from nothing,
062 * by grouping a car trip with a non-chain-based-mode trip.
063 * @author thibautd
064 */
065public class JointTripInsertorAlgorithm implements GenericPlanAlgorithm<JointPlan> {
066        private final TripRouter router;
067        private final List<String> chainBasedModes;
068        private final double betaDetour;
069        private final double scale;
070        private final Random random;
071
072        private final SocialNetwork socialNetwork;
073
074        public JointTripInsertorAlgorithm(
075                        final Random random,
076                        final SocialNetwork socialNetwork,
077                        final JointTripInsertorConfigGroup config,
078                        final TripRouter router) {
079                this.router = router;
080                this.socialNetwork = socialNetwork;
081                chainBasedModes = config.getChainBasedModes();
082                betaDetour = config.getBetaDetour();
083                scale = config.getScale();
084                this.random = random;
085        }
086
087        @Override
088        public void run(final JointPlan plan) {
089                run( plan , Collections.<Id<Person>>emptyList() );
090        }
091
092        public ActedUponInformation run(
093                        final JointPlan jointPlan,
094                        final Collection<Id<Person>> agentsToIgnore) {
095                final ClassifiedTrips trips = extractClassifiedTrips( jointPlan , agentsToIgnore );
096                final List<Match> matches =
097                        extractMatches(
098                                        jointPlan,
099                                        trips );
100                if (matches.size() == 0) return null;
101
102                final Match match = chooseMatch( matches );
103                insertMatch( jointPlan , match );
104                return match.toInformation();
105        }
106
107        private ClassifiedTrips extractClassifiedTrips(
108                        final JointPlan jointPlan,
109                        final Collection<Id<Person>> agentsToIgnore) {
110                final ClassifiedTrips trips = new ClassifiedTrips();
111
112                for (Map.Entry<Id<Person>, Plan> entry : jointPlan.getIndividualPlans().entrySet()) {
113                        final Id id = entry.getKey();
114                        if ( agentsToIgnore.contains( id ) ) continue;
115                        final Plan plan = entry.getValue();
116
117                        // identify the joint trips as one single trips.
118                        // Otherwise, the process will insert joint trips to access pick-ups
119                        // or go from drop offs...
120                        final CompositeStageActivityTypes types = new CompositeStageActivityTypes();
121                        types.addActivityTypes( router.getStageActivityTypes() );
122                        types.addActivityTypes( JointActingTypes.JOINT_STAGE_ACTS );
123
124                        final MainModeIdentifier mainModeIdentifier =
125                                new JointMainModeIdentifier(
126                                                router.getMainModeIdentifier() );
127
128                        for ( TripStructureUtils.Trip trip : TripStructureUtils.getTrips( plan , types ) ) {
129                                final String mode = mainModeIdentifier.identifyMainMode( trip.getTripElements() );
130
131                                if ( mode.equals( TransportMode.car ) ) {
132                                        trips.addCarTrip(
133                                                        new Trip(
134                                                                trip.getOriginActivity(),
135                                                                trip.getDestinationActivity(),
136                                                                calcEndOfActivity( trip.getOriginActivity() , plan ),
137                                                                id) );
138                                }
139                                else if (
140                                                !JointActingTypes.JOINT_MODES.contains( mode ) &&
141                                                !chainBasedModes.contains( mode ) ) {
142                                        trips.addNonChainBasedTrip(
143                                                        new Trip(
144                                                                trip.getOriginActivity(),
145                                                                trip.getDestinationActivity(),
146                                                                calcEndOfActivity( trip.getOriginActivity() , plan ),
147                                                                id) );
148                                }
149                        }
150                }
151
152                return trips;
153        }
154
155        private List<Match> extractMatches(
156                        final JointPlan jointPlan,
157                        final ClassifiedTrips trips) {
158                final List<Match> matches = new ArrayList<Match>();
159                final JointTravelStructure structure = JointTravelUtils.analyseJointTravel(jointPlan);
160
161                for ( Map.Entry<Id<Person>, List<Trip>> eDriver : trips.carTrips.entrySet() ) {
162                        final Id<Person> driverId = eDriver.getKey();
163                        for ( Map.Entry<Id<Person>, List<Trip>> ePass : trips.nonChainBasedModeTrips.entrySet() ) {
164                                final Id<Person> passengerId = ePass.getKey();
165                                if ( acceptAgentMatch( driverId , passengerId ) ) {
166                                        addMatches( matches , jointPlan , structure , eDriver.getValue() , ePass.getValue() );
167                                }
168                        }
169                }
170
171                return matches;
172        }
173
174        private void addMatches(
175                        final List<Match> matches,
176                        final JointPlan jointPlan,
177                        final JointTravelStructure structure,
178                        final List<Trip> driverTrips,
179                        final List<Trip> passengerTrips) {
180                for ( Trip driverTrip : driverTrips ) {
181                        for (Trip passengerTrip : passengerTrips ) {
182                                if ( isInCorrectSequence( jointPlan , structure , driverTrip , passengerTrip ) ) {
183                                        matches.add(
184                                                        new Match(
185                                                                driverTrip,
186                                                                passengerTrip,
187                                                                calcMatchCost(
188                                                                        driverTrip,
189                                                                        passengerTrip )));
190                                }
191                        }
192                }
193        }
194
195        private boolean acceptAgentMatch(
196                        final Id driver,
197                        final Id passenger ) {
198                if ( driver.equals( passenger ) ) return false;
199                if ( socialNetwork == null ) return true;
200
201                final boolean passengerAlterOfDriver =
202                        socialNetwork.getAlters( driver ).contains( passenger );
203
204                if ( socialNetwork.isReflective() ) return passengerAlterOfDriver;
205
206                return passengerAlterOfDriver || 
207                        socialNetwork.getAlters( passenger ).contains( driver );
208        }
209
210        private static boolean isInCorrectSequence(
211                        final JointPlan jointPlan,
212                        final JointTravelStructure structure,
213                        final Trip driverTrip,
214                        final Trip passengerTrip) {
215                final List<JointTrip> jointTrips =
216                                structure.getJointTripsForCotravelers(
217                                                driverTrip.agentId,
218                                                passengerTrip.agentId);
219                final int positionInDriverPlan = getPosition( jointPlan , jointTrips , driverTrip );
220                final int positionInPassengerPlan = getPosition( jointPlan , jointTrips , passengerTrip );
221                return positionInDriverPlan == positionInPassengerPlan;
222        }
223
224        private static int getPosition(
225                        final JointPlan jointPlan,
226                        final List<JointTrip> jointTrips,
227                        final Trip trip) {
228                final Plan plan = jointPlan.getIndividualPlan( trip.agentId );
229
230                final int indexOfTrip = plan.getPlanElements().indexOf( trip.departure );
231
232                // count joint trips occuring before the candidate trip
233                int pos = 0;
234                for ( JointTrip jt : jointTrips ) {
235                        final int indexOfJointTrip =
236                                Math.max(
237                                                plan.getPlanElements().indexOf( jt.getPassengerLeg() ),
238                                                plan.getPlanElements().indexOf( jt.getDriverLegs().get( 0 ) ) );
239                        assert indexOfJointTrip >= 0;
240                        if ( indexOfJointTrip < indexOfTrip ) pos++;
241                }
242
243                return pos;
244        }
245
246        private double calcMatchCost(
247                        final Trip driverTrip,
248                        final Trip passengerTrip) {
249                final double timeDiff = Math.abs( driverTrip.departureTime - passengerTrip.departureTime );
250                final double detourDist = 
251                        CoordUtils.calcEuclideanDistance( driverTrip.departure.getCoord() , passengerTrip.departure.getCoord() ) +
252                        CoordUtils.calcEuclideanDistance( driverTrip.arrival.getCoord() , passengerTrip.arrival.getCoord() ) +
253                        passengerTrip.length - driverTrip.length;
254                        
255                return scale * (timeDiff + betaDetour * detourDist);
256        }
257
258        private Match chooseMatch(final List<Match> matches) {
259                final double[] thresholds = new double[ matches.size() ];
260
261                double sum = 0;
262                int i=0;
263                for (Match match : matches) {
264                        sum += Math.exp( -match.cost );
265                        thresholds[i] = sum;
266                        i++;
267                }
268
269                final double choice = random.nextDouble() * sum;
270
271                for (i=0; i < thresholds.length; i++) {
272                        if (choice <= thresholds[i]) {
273                                return matches.get( i );
274                        }
275                }
276
277                throw new RuntimeException( "choice procedure failed! this should not happen! choice="+choice+" in thresholds="+Arrays.toString(thresholds) );
278        }

280        private static void insertMatch(
281                        final JointPlan jointPlan,
282                        final Match match) {
283                final Plan driverPlan = jointPlan.getIndividualPlans().get( match.tripDriver.agentId );
284                final Plan passengerPlan = jointPlan.getIndividualPlans().get( match.tripPassenger.agentId );
285
286                /* scope of driver-specific variables */ {
287                        // insert in driver plan
288                        final List<PlanElement> driverTrip = new ArrayList<PlanElement>();
289                        driverTrip.add( PopulationUtils.createLeg(TransportMode.car) );
290                        /* scope of firstAct */ {
291                                final Activity firstAct = PopulationUtils.createActivityFromCoordAndLinkId(JointActingTypes.INTERACTION, match.tripPassenger.departure.getCoord(), match.tripPassenger.departure.getLinkId());
292                                firstAct.setMaximumDuration( 0 );
293                                driverTrip.add( firstAct );
294                        }
295                        /* scope of leg */ {
296                                final Leg leg =  PopulationUtils.createLeg(JointActingTypes.DRIVER);
297                                final DriverRoute dRoute = new DriverRoute(
298                                                match.tripPassenger.departure.getLinkId(),
299                                                match.tripPassenger.arrival.getLinkId());
300                                dRoute.addPassenger( match.tripPassenger.agentId );
301                                leg.setRoute( dRoute );
302                                driverTrip.add( leg );
303                        }
304                        /* scope of secondAct */ {
305                                final Activity secondAct = PopulationUtils.createActivityFromCoordAndLinkId(JointActingTypes.INTERACTION, match.tripPassenger.arrival.getCoord(), match.tripPassenger.arrival.getLinkId());
306                                secondAct.setMaximumDuration( 0 );
307                                driverTrip.add( secondAct );
308                        }
309                        driverTrip.add( PopulationUtils.createLeg(TransportMode.car) );
310
311                        TripRouter.insertTrip(
312                                        driverPlan,
313                                        match.tripDriver.departure,
314                                        driverTrip,
315                                        match.tripDriver.arrival );
316                }
317
318                /* scope of passenger-specific variables */ {
319                        // insert in passenger plan
320                        final Leg pLeg =  PopulationUtils.createLeg(JointActingTypes.PASSENGER);
321                        final PassengerRoute pRoute = new PassengerRoute(
322                                        match.tripPassenger.departure.getLinkId(),
323                                        match.tripPassenger.arrival.getLinkId());
324                        pRoute.setDriverId( match.tripDriver.agentId );
325                        pLeg.setRoute( pRoute );
326
327
328                        TripRouter.insertTrip(
329                                        passengerPlan,
330                                        match.tripPassenger.departure,
331                                        Collections.singletonList( pLeg ),
332                                        match.tripPassenger.arrival );
333                }
334        }

336        private static double calcEndOfActivity(
337                        final Activity activity,
338                        final Plan plan) {
339                if (activity.getEndTime() != Time.UNDEFINED_TIME) return activity.getEndTime();
340
341                // no sufficient information in the activity...
342                // do it the long way.
343                // XXX This is inefficient! Using a cache for each plan may be an option
344                // (knowing that plan elements are iterated in proper sequence,
345                // no need to re-examine the parts of the plan already known)
346                double now = 0;
347
348                for (PlanElement pe : plan.getPlanElements()) {
349                        now = updateNow( now , pe );
350                        if (pe == activity) return now;
351                }
352
353                throw new RuntimeException( "activity "+activity+" not found in "+plan.getPlanElements() );
354        }
355
356        private static double updateNow(
357                        final double now,
358                        final PlanElement pe) {
359                if (pe instanceof Activity) {
360                        Activity act = (Activity) pe;
361                        double endTime = act.getEndTime();
362                        double startTime = act.getStartTime();
363                        double dur = (act instanceof Activity ? ((Activity) act).getMaximumDuration() : Time.UNDEFINED_TIME);
364                        if (endTime != Time.UNDEFINED_TIME) {
365                                // use fromAct.endTime as time for routing
366                                return endTime;
367                        }
368                        else if ((startTime != Time.UNDEFINED_TIME) && (dur != Time.UNDEFINED_TIME)) {
369                                // use fromAct.startTime + fromAct.duration as time for routing
370                                return startTime + dur;
371                        }
372                        else if (dur != Time.UNDEFINED_TIME) {
373                                // use last used time + fromAct.duration as time for routing
374                                return now + dur;
375                        }
376                        else {
377                                throw new RuntimeException("activity has neither end-time nor duration." + act);
378                        }
379                }
380                double tt = ((Leg) pe).getTravelTime();
381                return now + (tt != Time.UNDEFINED_TIME ? tt : 0);
382        }

387        private static class ClassifiedTrips {
388                public final Map<Id<Person>, List<Trip>> carTrips = new HashMap< >();
389                public final Map<Id<Person>, List<Trip>> nonChainBasedModeTrips = new HashMap< >();
390
391                public void addCarTrip( final Trip t ) {
392                        MapUtils.getList( t.agentId , carTrips ).add( t );
393                }
394
395                public void addNonChainBasedTrip( final Trip t ) {
396                        MapUtils.getList( t.agentId , nonChainBasedModeTrips ).add( t );
397                }
398        }

400        private static class Trip {
401                final Activity departure;
402                final Activity arrival;
403                final double departureTime;
404                final double length;
405                final Id<Person> agentId;
406
407                public Trip(
408                                final Activity departure,
409                                final Activity arrival,
410                                final double departureTime,
411                                final Id<Person> agentId) {
412                        this.departure = departure;
413                        this.arrival = arrival;
414                        this.departureTime = departureTime;
415                        this.length = CoordUtils.calcEuclideanDistance( departure.getCoord() , arrival.getCoord() );
416                        this.agentId = agentId;
417                }
418        }

423        private static class Match {
424                final Trip tripDriver;
425                final Trip tripPassenger;
426                final double cost;
427
428                public Match(
429                                final Trip tripDriver,
430                                final Trip tripPassenger,
431                                final double cost) {
432                        this.tripDriver = tripDriver;
433                        this.tripPassenger = tripPassenger;
434                        this.cost = cost;
435                }
436
437                public ActedUponInformation toInformation() {
438                        return new ActedUponInformation(
439                                        tripDriver.agentId,
440                                        tripPassenger.agentId);
441                }
