049/**
050 * Optimizes vehicle allocation at the tour level, by minimizing the estimated
051 * waiting times.
052 * It assumes all persons in the group start and end their plans at the same location ("home"),
053 * and get the vehicles at the "home" location (ie household-like).
054 * If those assumptions are not verified, the algorithm falls back to random allocation.
055 *
056 * <br>
057 * It also assumes departure times are determined by activity end times and routes
058 * (or legs) contain good estimation of the travel time.
059 * @author thibautd
060 */
061public class OptimizeVehicleAllocationAtTourLevelAlgorithm implements GenericPlanAlgorithm<GroupPlans> {
062        private final GenericPlanAlgorithm<GroupPlans> randomAllocator;
063        private final StageActivityTypes stageActs;
064        private final Collection<String> vehicularModes;
065        private final boolean allowNullRoutes;
066        private final VehicleRessources vehicleRessources;
067
068        private boolean anchorAtFacilities = false;
069
070        public OptimizeVehicleAllocationAtTourLevelAlgorithm(
071                        final StageActivityTypes stageActivitiesForSubtourDetection,
072                        final Random random,
073                        final VehicleRessources vehicleRessources,
074                        final Collection<String> modes,
075                        final boolean allowNullRoutes) {
076                this.randomAllocator = new AllocateVehicleToPlansInGroupPlanAlgorithm(
077                        random,
078                        vehicleRessources,
079                        modes,
080                        allowNullRoutes,
081                        false);
082                this.vehicularModes = modes;
083                this.allowNullRoutes = allowNullRoutes;
084                this.stageActs = stageActivitiesForSubtourDetection;
085                this.vehicleRessources = vehicleRessources;
086        }
087
088        @Override
089        public void run(final GroupPlans plan) {
090                final List<SubtourRecord> vehicularTours = getVehicularToursSortedByStartTime( plan );
091                if ( vehicularTours == null ) {
092                        randomAllocator.run( plan );
093                        return;
094                }
095
096                allocateVehicles( vehicularTours );
097
098                processAllocation( vehicularTours );
099        }
100
101        private void processAllocation(final List<SubtourRecord> vehicularTours) {
102                for ( final SubtourRecord r : vehicularTours ) {
103                        for ( final Trip t : r.subtour.getTrips() ) {
104                                for ( final Leg leg : t.getLegsOnly() ) {
105                                        if ( !vehicularModes.contains( leg.getMode() ) ) continue;
106
107                                        if ( allowNullRoutes && leg.getRoute() == null ) {
108                                                // this is not so nice...
109                                                leg.setRoute( new VehicleOnlyNetworkRoute() );
110                                        }
111
112                                        if ( !( leg.getRoute() instanceof NetworkRoute ) ) {
113                                                throw new RuntimeException( "route for mode "+leg.getMode()+" has non-network route "+leg.getRoute() );
114                                        }
115
116                                        ((NetworkRoute) leg.getRoute()).setVehicleId( r.allocatedVehicle );
117                                }
118                        }
119                }
120        }
121
122        private void allocateVehicles(
123                        final List<SubtourRecord> toursToAllocate) {
124                // greedy algo. Should be ok, but didn't formally prove it
125                if ( toursToAllocate.isEmpty() ) return;
126                final List<SubtourRecord> remainingSubtours = new ArrayList<SubtourRecord>( toursToAllocate );
127                final SubtourRecord currentSubtour = remainingSubtours.remove( 0 );
128
129                final VehicleRecord firstAvailableVehicle =
130                        Collections.min(
131                                currentSubtour.possibleVehicles,
132                                new Comparator<VehicleRecord>() {
133                                        @Override
134                                        public int compare(
135                                                        final VehicleRecord o1,
136                                                        final VehicleRecord o2) {
137                                                final int timeComp = Double.compare(
138                                                        o1.availableFrom,
139                                                        o2.availableFrom );
140                                                return timeComp != 0 ? timeComp :
141                                                        o1.nAllocs - o2.nAllocs;
142                                        }
143                                });
144
145                if ( firstAvailableVehicle.availableFrom < currentSubtour.endTime ) {
146                        firstAvailableVehicle.availableFrom = currentSubtour.endTime;
147                }
148                firstAvailableVehicle.nAllocs++;
149                currentSubtour.allocatedVehicle = firstAvailableVehicle.id;
150
151                allocateVehicles( remainingSubtours );
152        }
153
154        /*for tests*/ double calcOverlap(final GroupPlans gps) {
155                final List<SubtourRecord> tours = getVehicularToursSortedByStartTime( gps );
156
157                double overlap = 0;
158                for ( final SubtourRecord tour : tours ) {
159                        Id veh = null;
160                        for ( final Trip t : tour.subtour.getTrips() ) {
161                                veh = getVehicle( t );
162                                if ( veh != null ) break;
163                        }
164                        
165                        for ( final VehicleRecord vr : tour.possibleVehicles ) {
166                                if ( vr.id.equals( veh ) ) {
167                                        overlap += Math.max( 0 , vr.availableFrom - tour.startTime );
168                                        if ( vr.availableFrom < tour.endTime ) {
169                                                vr.availableFrom = tour.endTime;
170                                        }
171                                        break;
172                                }
173                        }
174                }
175
176                return overlap;
177        }

179        // /////////////////////////////////////////////////////////////////////////
180        // subtour-related methods
181        // /////////////////////////////////////////////////////////////////////////
182        private List<SubtourRecord> getVehicularToursSortedByStartTime(final GroupPlans plans) {
183                final List<SubtourRecord> vehicularTours = new ArrayList<SubtourRecord>();
184                final VehicleRecordFactory factory = new VehicleRecordFactory();
185
186                for ( final Plan p : plans.getAllIndividualPlans() ) {
187                        final Collection<Subtour> subtours =
188                                TripStructureUtils.getSubtours(
189                                                p,
190                                                stageActs,
191                                                anchorAtFacilities );
192                        for ( final Subtour s : subtours ) {
193                                if ( s.getParent() != null ) continue; // is not a root tour
194                                boolean isFirstTrip = true;
195                                for ( final Trip t : s.getTrips() ) {
196                                        // TODO: check that the sequence of vehicular movements come back to origin
197                                        if ( isFirstTrip && isVehicular( t ) ) {
198                                                vehicularTours.add(
199                                                                new SubtourRecord(
200                                                                        s,
201                                                                        factory.getRecords(
202                                                                                vehicleRessources.identifyVehiclesUsableForAgent(
203                                                                                        p.getPerson().getId() ))) );
204                                                break;
205                                        }
206                                        if ( !isFirstTrip && isVehicular( t ) ) {
207                                                // invalid structure
208                                                return null;
209                                        }
210                                        isFirstTrip = false;
211                                }
212                        }
213                }
214
215                // check validity
216                Id homeLoc = null;
217                for ( final SubtourRecord record : vehicularTours ) {
218                        final Subtour s = record.subtour;
219                        assert s.getParent() == null;
220                        final Id anchor = anchorAtFacilities ?
221                                s.getTrips().get( 0 ).getOriginActivity().getFacilityId() :
222                                s.getTrips().get( 0 ).getOriginActivity().getLinkId();
223
224                        if ( anchor == null ) throw new NullPointerException( "null anchor location" );
225                        if ( homeLoc == null ) {
226                                homeLoc = anchor;
227                        }
228                        else if ( !homeLoc.equals( anchor ) ) {
229                                // invalid
230                                return null;
231                        }
232                }
233
234                Collections.sort(
235                                vehicularTours,
236                                new Comparator<SubtourRecord>() {
237                                        @Override
238                                        public int compare(
239                                                        final SubtourRecord o1,
240                                                        final SubtourRecord o2) {
241                                                return Double.compare( o1.startTime , o2.startTime );
242                                        }
243                                });
244
245
246                return vehicularTours;
247        }
248
249        private boolean isVehicular(final Trip t) {
250                // note: checking that getVehicle returns null doen't work
251                // when allowing for null routes. Hence the duplication of the logic...
252                final List<Leg> legs = t.getLegsOnly();
253                if ( legs.isEmpty() ) return false;
254
255                // XXX what to do if several legs???
256                final Leg l = legs.get( 0 );
257                if ( !vehicularModes.contains( l.getMode() ) ) return false;
258                if ( !allowNullRoutes && l.getRoute() == null ) return false;
259                if ( l.getRoute() != null && !(l.getRoute() instanceof NetworkRoute) ) return false; 
260                return true;
261        }
262
263        private Id getVehicle( final Trip t ) {
264                final List<Leg> legs = t.getLegsOnly();
265                if ( legs.isEmpty() ) return null;
266
267                // XXX what to do if several legs???
268                final Leg l = legs.get( 0 );
269                if ( !vehicularModes.contains( l.getMode() ) ) return null;
270                if ( !allowNullRoutes && l.getRoute() == null ) return null;
271                if ( l.getRoute() != null && !(l.getRoute() instanceof NetworkRoute) ) return null; 
272                return ((NetworkRoute) l.getRoute()).getVehicleId();
273        }

278        private static class SubtourRecord {
279                public final double startTime, endTime;
280                public final List<VehicleRecord> possibleVehicles;
281                public final Subtour subtour;
282                public Id allocatedVehicle = null;
283
284                public SubtourRecord(
285                                final Subtour subtour,
286                                final List<VehicleRecord> possibleVehicles) {
287                        this.possibleVehicles = possibleVehicles; 
288                        this.subtour = subtour;
289                        
290                        final Trip firstTrip = subtour.getTrips().get( 0 );
291                        this.startTime = firstTrip.getOriginActivity().getEndTime();
292                        if ( startTime == Time.UNDEFINED_TIME ) throw new RuntimeException( "no end time in "+firstTrip.getOriginActivity() );
293
294                        final Trip lastTrip = subtour.getTrips().get( subtour.getTrips().size() - 1 );
295                        this.endTime = calcArrivalTime( lastTrip );
296                }
297        }
298
299        private static class VehicleRecord {
300                public final Id id;
301                public int nAllocs = 0;
302                public double availableFrom = Double.NEGATIVE_INFINITY;
303
304                public VehicleRecord(final Id id) {
305                        this.id = id;
306                }
307        }
308
309        private static class VehicleRecordFactory {
310                private final Map<Id<Vehicle>, VehicleRecord> records = new HashMap<>();
311
312                public List<VehicleRecord> getRecords(final Collection<Id<Vehicle>> ids) {
313                        final List<VehicleRecord> list = new ArrayList<VehicleRecord>();
314
315                        for ( final Id<Vehicle> id : ids ) {
316                                VehicleRecord r = records.get( id );
317
318                                if ( r == null ) {
319                                        r = new VehicleRecord( id );
320                                        records.put( id , r );
321                                }
322
323                                list.add( r );
324                        }
325
326                        return list;
327                }
328        }
329
330        private static double calcArrivalTime(final Trip trip) {
331                double now = trip.getOriginActivity().getEndTime();
332                for ( final PlanElement pe : trip.getTripElements() ) {
333                        if ( pe instanceof Activity ) {
334                                final double end = ((Activity) pe).getEndTime();
335                                now = end != Time.UNDEFINED_TIME ? end : now + ((Activity) pe).getMaximumDuration();
336                                // TODO: do not fail *that* badly, but just revert to random alloc
337                                if ( now == Time.UNDEFINED_TIME ) throw new RuntimeException( "could not get time from "+pe );
338                        }
339                        else if ( pe instanceof Leg ) {
340                                final Route r = ((Leg) pe).getRoute();
341                                if ( r != null && r.getTravelTime() != Time.UNDEFINED_TIME ) {
342                                        now += r.getTravelTime();
343                                }
344                                else {
345                                        now += ((Leg) pe).getTravelTime() != Time.UNDEFINED_TIME ?
346                                                        ((Leg) pe).getTravelTime() :
347                                                        0; // no info: just assume instantaneous. This will give poor results!
348                                }
349                        }
350                }
351                return now;
352        }
