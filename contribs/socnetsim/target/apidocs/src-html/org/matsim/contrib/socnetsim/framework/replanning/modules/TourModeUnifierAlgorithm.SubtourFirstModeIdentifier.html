020package org.matsim.contrib.socnetsim.framework.replanning.modules;
021
022import java.util.Collections;
023
024import org.matsim.api.core.v01.population.Plan;
025import org.matsim.core.population.PopulationUtils;
026import org.matsim.core.population.algorithms.PlanAlgorithm;
027import org.matsim.core.router.MainModeIdentifier;
028import org.matsim.core.router.StageActivityTypes;
029import org.matsim.core.router.TripRouter;
030import org.matsim.core.router.TripStructureUtils;
031import org.matsim.core.router.TripStructureUtils.Trip;
032import org.matsim.core.router.TripStructureUtils.Subtour;
033
034/**
035 * sets the mode of all trips in full tours to the same mode.
036 * Useful after operations which may change the tour structure
037 * (for instance location choice or sequence mutation)
038 * @author thibautd
039 */
040public class TourModeUnifierAlgorithm implements PlanAlgorithm {
041        private final StageActivityTypes stages;
042        private final SubtourModeIdentifier modeIdentifier;
043
044        public TourModeUnifierAlgorithm(
045                        final StageActivityTypes stages,
046                        final MainModeIdentifier modeIdentifier) {
047                this( stages,
048                                new SubtourFirstModeIdentifier(
049                                        modeIdentifier ) );
050        }
051
052        public TourModeUnifierAlgorithm(
053                        final StageActivityTypes stages,
054                        final SubtourModeIdentifier modeIdentifier) {
055                this.stages = stages;
056                this.modeIdentifier = modeIdentifier;
057        }
058
059        @Override
060        public void run(final Plan plan) {
061                for ( Subtour subtour : TripStructureUtils.getSubtours( plan , stages ) ) {
062                        // not clear what we should do with open tours
063                        if ( !subtour.isClosed() ) continue;
064                        // only consider "root" tours: tours without (closed) parent
065                        if ( subtour.getParent() != null && subtour.getParent().isClosed() ) continue;
066
067                        final String mode = modeIdentifier.identifyMode( subtour );
068
069                        for ( Trip trip : subtour.getTrips() ) {
070                                TripRouter.insertTrip(
071                                                plan,
072                                                trip.getOriginActivity(),
073                                                Collections.singletonList( PopulationUtils.createLeg(mode) ),
074                                                trip.getDestinationActivity() );
075                        }
076                }
077        }
078
079        public static interface SubtourModeIdentifier {
080                public String identifyMode( final Subtour subtour );
081        }
082
083        public static final class SubtourFirstModeIdentifier implements SubtourModeIdentifier {
084                private final MainModeIdentifier modeIdentifier;
085
086                public SubtourFirstModeIdentifier(
087                                final MainModeIdentifier modeIdentifier) {
088                        this.modeIdentifier = modeIdentifier;
089                }
090
091                @Override
092                public String identifyMode(final Subtour subtour) {
093                        return modeIdentifier.identifyMainMode(
094                                        subtour.getTrips().get( 0 ).getTripElements() );
095                }
096        }
097}
