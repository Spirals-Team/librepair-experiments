025import org.matsim.api.core.v01.Id;

031/**
032 * <b>Static</b> factory to create joint plans.
033 * The fact that it is static allows to track which joint plan is associated
034 * to each individual plan in a global way.
035 *
036 * @author thibautd
037 */
038public class JointPlanFactory implements MatsimFactory {
039        public JointPlan createJointPlan(
040                        final Map<Id<Person>, ? extends Plan> plans) {
041                return createJointPlan( plans, true );
042        }
043
044        /**
045         * Creates a joint plan from individual plans.
046         * Two individual trips to be shared must have their Pick-Up activity type set
047         * to 'pu_i', where i is an integer which identifies the joint trip.
048         * @param plans the individual plans. If they consist of Joint activities, 
049         * those activities are referenced, otherwise, they are copied in a joint activity.
050         * @param addAtIndividualLevel if true, the plans are added to the Person's plans.
051         * set to false for a temporary plan (in a replaning for example).
052         */
053        public JointPlan createJointPlan(
054                        final Map<Id<Person>, ? extends Plan> plans,
055                        final boolean addAtIndividualLevel) {
056                JointPlan jointPlan = new JointPlan( plans );
057
058                for (Plan plan : plans.values()) {
059                        if (addAtIndividualLevel) {
060                                final Person person = plan.getPerson();
061                                if (person == null) {
062                                        throw new NullPointerException(
063                                                        "the person backpointed by the plan"+
064                                                        " must not be null for the plan to be added" );
065                                }
066                                if ( !person.getPlans().contains( plan ) ) person.addPlan( plan );
067                        }
068                }
069
070                return jointPlan;
071        }
072
073        public JointPlan copyJointPlan(
074                        final JointPlan toCopy) {
075                return copyJointPlan( toCopy , true );
076        }
077
078        public JointPlan copyJointPlan(
079                        final JointPlan toCopy,
080                        final boolean addAtIndividualLevel) {
081                return createJointPlan(
082                                cloneIndividualPlans( toCopy ),
083                                addAtIndividualLevel );
084        }
085
086        private static Map<Id<Person>, Plan> cloneIndividualPlans(final JointPlan plan) {
087                final Map<Id<Person> , Plan> plans = new LinkedHashMap< >();
088
089                for (Map.Entry<Id<Person>, Plan> indiv : plan.getIndividualPlans().entrySet()) {
090                        final Plan newPlan = createIndividualPlan( indiv.getValue().getPerson() );
091                        ((PlanWithCachedJointPlan)newPlan).copyFrom( indiv.getValue() );
092                        // I think that the above cast will now fail.  It probably worked originally, since PlanWithCachedJointPlan was an extension of PlanImpl, which is
093                        // no longer allowed. I would, however, say that it was not a clean copy anyways, since it ignored the additional fields
094                        // of JointPlan.  Thibaut, I am confident that you can resolve this if you encounter it, but please let me know if you want to discuss.
095                        // kai, nov'15
096                        plans.put( indiv.getKey() , newPlan );
097                }
098                
099                return plans;
100        }
101
102        public static Plan createIndividualPlan( final Person person) {
103                return new PlanWithCachedJointPlan( person );
104        }
105}
