020package org.matsim.contrib.socnetsim.framework.controller;
021
022import java.util.Collection;
023import java.util.HashSet;
024import java.util.Set;
025
026import org.apache.log4j.Logger;
027import org.matsim.api.core.v01.Scenario;
028import org.matsim.api.core.v01.population.Person;
029import org.matsim.api.core.v01.population.Plan;
030import org.matsim.api.core.v01.population.Population;
031import org.matsim.core.controler.events.IterationEndsEvent;
032import org.matsim.core.controler.events.IterationStartsEvent;
033import org.matsim.core.controler.listener.IterationEndsListener;
034import org.matsim.core.controler.listener.IterationStartsListener;
035
036import org.matsim.contrib.socnetsim.framework.population.JointPlan;
037import org.matsim.contrib.socnetsim.framework.population.JointPlans;
038import org.matsim.contrib.socnetsim.framework.replanning.modules.PlanLinkIdentifier;
039import org.matsim.contrib.socnetsim.framework.replanning.modules.PlanLinkIdentifier.Strong;
040
041import com.google.inject.Inject;
042
043public final class JointPlanCompositionMinimalityChecker implements IterationEndsListener, IterationStartsListener {
044        private static final Logger log = Logger.getLogger( JointPlanCompositionMinimalityChecker.class );
045
046        // Fail only on start of next iteration, to let all consistency checkers print their error, if any.
047        private boolean gotError = false;
048        private final PlanLinkIdentifier linkIdentifier;
049        private final Population population;
050        private final JointPlans jointPlansStruct;
051
052        @Inject
053        public JointPlanCompositionMinimalityChecker(
054                        final @Strong PlanLinkIdentifier linkIdentifier,
055                        final Scenario sc ) {
056                this.linkIdentifier = linkIdentifier;
057                this.population = sc.getPopulation();
058                this.jointPlansStruct = (JointPlans) sc.getScenarioElement( JointPlans.ELEMENT_NAME );
059        }
060
061        @Override
062        public void notifyIterationEnds(final IterationEndsEvent event) {
063                log.info( "Checking minimality of joint plan composition" );
064                final PlanLinkIdentifier links = linkIdentifier;
065                final Set<JointPlan> jointPlans = new HashSet<JointPlan>();
066
067                for ( Person person : population.getPersons().values() ) {
068                        final Plan plan = person.getSelectedPlan();
069                        final JointPlan jp = jointPlansStruct.getJointPlan( plan );
070
071                        if ( jp != null ) {
072                                jointPlans.add( jp );
073                        }
074                }
075
076                for ( JointPlan jp : jointPlans ) {
077                        for ( Plan p : jp.getIndividualPlans().values() ) {
078                                if ( !hasLinkedPlan( links , p , jp.getIndividualPlans().values() ) ) {
079                                        log.error( "plan "+p+" is in "+jp+" but is not linked with any plan" );
080                                        gotError = true ;
081                                }
082                        }
083                }
084        }
085
086        private boolean hasLinkedPlan(
087                        final PlanLinkIdentifier links,
088                        final Plan p,
089                        final Collection<Plan> plans) {
090                for ( Plan p2 : plans ) {
091                        if ( p == p2 ) continue;
092                        final boolean l = links.areLinked( p , p2 );
093                        if ( l != links.areLinked( p2 , p ) ) {
094                                log.error( "inconsistent plan link identifier revealed by "+p+" and "+p2 );
095                                log.error( p.getPerson().getId()+" is "+(l ? "" : "NOT")+" linked with "+p2.getPerson().getId() );
096                                log.error( p2.getPerson().getId()+" is "+(l ? "NOT" : "")+" linked with "+p.getPerson().getId() );
097                                gotError = true;
098                        }
099                        if ( l ) return true;
100                }
101                return false;
102        }
103
104        @Override
105        public void notifyIterationStarts( IterationStartsEvent event ) {
106                if ( gotError ) throw new RuntimeException( "inconsistency detected. Look at error messages for details" );
107        }
108}
