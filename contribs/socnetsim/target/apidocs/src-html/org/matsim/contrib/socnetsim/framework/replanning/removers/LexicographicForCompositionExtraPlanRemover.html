046/**
047 * Considers the "maximum plans per agent" is actually a "maximum plans per
048 * joint plan composition".
049 *
050 * If there are too many plans of a given composition, the joint plan maximizing
051 * the number of first worst, then second worst, etc. plans is removed (lexicographic
052 * order).
053 * The first, second, etc is defined on the set of *all* plans of an agent.
054 * @author thibautd
055 */
056public class LexicographicForCompositionExtraPlanRemover implements ExtraPlanRemover {
057        private static final Logger log = Logger.getLogger( LexicographicForCompositionExtraPlanRemover.class );
058        private final Random random;
059        private final int maxPlansPerComposition;
060        private final int maxPlansPerAgent;
061
062        @Inject
063        public LexicographicForCompositionExtraPlanRemover(
064                        final GroupReplanningConfigGroup config ) {
065                this( config.getMaxPlansPerComposition() , config.getMaxPlansPerAgent() );
066        }
067
068        public LexicographicForCompositionExtraPlanRemover(
069                        final int maxPlansPerComposition,
070                        final int maxPlansPerAgent) {
071                this.maxPlansPerComposition = maxPlansPerComposition;
072                this.maxPlansPerAgent = maxPlansPerAgent;
073
074                this.random = MatsimRandom.getLocalInstance();
075        }
076
077        @Override
078        public boolean removePlansInGroup(
079                        final JointPlans jointPlans,
080                        final ReplanningGroup group) {
081                if ( log.isTraceEnabled() ) log.trace( "remove plans in group of size "+group.getPersons().size() );
082                boolean somethingDone = false;
083
084                // check condition per composition
085                final int maxRank = maxRank( group );
086                for ( Person person : group.getPersons() ) {
087                        if ( person.getPlans().size() <= maxPlansPerAgent ) continue;
088                        final PlansPerComposition plansPerComposition = getPlansPerComposition( jointPlans , person );
089
090                        // too many individual plans?
091                        int rem = 0;
092                        while ( plansPerComposition.getIndividualPlans().size() - rem > maxPlansPerComposition ) {
093                                rem++;
094                                person.removePlan(
095                                                identifyWorst(
096                                                                plansPerComposition.getIndividualPlans()));
097                                somethingDone = true;
098                        }
099                        if ( log.isTraceEnabled() && rem > 0 ) log.trace( "removed "+rem+" individual plans for person "+person.getId() );
100
101                        for ( Collection<JointPlan> structure : plansPerComposition.getJointPlansGroupedPerStructure() ) {
102                                rem = 0;
103                                int size = 0;
104                                while ( structure.size() - rem > maxPlansPerComposition ) {
105                                        rem++;
106                                        final JointPlan toRemove = identifyWorst( maxRank , structure );
107                                        size = toRemove.getIndividualPlans().size();
108
109                                        for ( Plan p : toRemove.getIndividualPlans().values() ) {
110                                                p.getPerson().removePlan(p);
111                                        }
112                                        jointPlans.removeJointPlan( toRemove );
113                                        structure.remove( toRemove );
114                                        somethingDone = true;
115                                }
116                                if ( log.isTraceEnabled() && rem > 0 ) log.trace( "removed "+rem+" joint plans for structure of size "+size );
117                        }
118                }
119
120                // now that maxima per composition are processed, check whether
121                // there are still agents with too many plans. This is mainly to
122                // avoid memory consumption getting out of control, but the number should
123                // be large, as this way of doing is problematic (it does not consider conflicts
124                // at all).
125                // randomize order to avoid the existence of "dictators" (agent for which their
126                // worst plan is always removed, whatever the ranking of the others is)
127                final List<Person> persons = new ArrayList<>( group.getPersons() );
128                Collections.shuffle( persons , random );
129                boolean loop = true;
130                int rem = 0;
131                // is useful for performance if lots of plans to remove, which is not the typical usecase...
132                final Map<Id<Person>, List<? extends Plan>> sortedPlans = new HashMap<>();
133                while ( loop ) {
134                        loop = false;
135                        for ( Person person : persons ) {
136                                if ( person.getPlans().size() > maxPlansPerAgent ) {
137                                        if ( log.isTraceEnabled() ) log.trace( "person "+person.getId()+" has "+(person.getPlans().size() - maxPlansPerAgent)+" excedentary plans" );
138                                        final Plan toRemove = getWorstNonUniquelyIndividualPlan( getSortedPlans( sortedPlans , person ) , jointPlans );
139
140                                        rem++;
141                                        final JointPlan jpToRemove = jointPlans.getJointPlan( toRemove );
142
143                                        if ( jpToRemove != null ) {
144                                                for ( Plan p : jpToRemove.getIndividualPlans().values() ) {
145                                                        p.getPerson().removePlan( p );
146                                                }
147                                                jointPlans.removeJointPlan( jpToRemove );
148                                        }
149                                        else {
150                                                person.removePlan( toRemove );
151                                        }
152
153                                        somethingDone = true;
154                                }
155                                // continue looping if still too many plans
156                                loop = loop || person.getPlans().size() > maxPlansPerAgent;
157                        }
158                }
159                if ( log.isTraceEnabled() && rem > 0 ) log.trace( "removed "+rem+" excedentary plans" );
160
161                return somethingDone;
162        }
163
164        private List<? extends Plan> getSortedPlans( final Map<Id<Person>, List<? extends Plan>> sortedPlans, final Person person ) {
165                if ( sortedPlans.containsKey( person.getId() ) ) return sortedPlans.get( person.getId() );
166                // side effect. is it ok?
167                // necessary to avoid having to remove plans here as well
168                // works only if person returns internal instance
169                final List<? extends Plan> plans = person.getPlans();
170                Collections.sort( plans, new Comparator<Plan>() {
171                        @Override
172                        public int compare( final Plan o1, final Plan o2 ) {
173                                return Double.compare( o1.getScore() , o2.getScore() );
174                        }
175                } );
176                sortedPlans.put( person.getId() , plans );
177                return plans;
178        }
179
180        private static Plan getWorstNonUniquelyIndividualPlan(
181                        final List<? extends Plan> plans,
182                        final JointPlans jointPlans) {
183                Plan worst = null;
184                Plan secondWorst = null;
185
186                int nIndividualPlans = 0;
187                for ( Plan plan : plans ) {
188                        final JointPlan jp = jointPlans.getJointPlan( plan );
189
190                        if ( isIndividual( jp ) ) nIndividualPlans++;
191
192                        if ( worst == null ) {
193                                if ( !isIndividual( jp ) || nIndividualPlans > 1 ) return plan;
194
195                                worst = plan;
196                        }
197                        else {
198                                secondWorst = plan;
199                                break;
200                        }
201                }
202
203                // if there is only one individual plan, do not consider it "worse"
204                // assert nIndividualPlans > 0; // this assertion only makes sense if all persons have at least 1 individual plan
205                return ( nIndividualPlans == 1 ) ? secondWorst : worst;
206        }
207
208        private static boolean isIndividual( final JointPlan jp ) {
209                return jp == null || jp.getIndividualPlans().size() == 1;
210        }
211
212        private static int maxRank(final ReplanningGroup group) {
213                int maxRank = 0;
214
215                for ( Person p : group.getPersons() ) {
216                        maxRank = Math.max( maxRank , p.getPlans().size() );
217                }
218
219                return maxRank;
220        }
221
222        private static JointPlan identifyWorst(
223                        final int maxRank,
224                        final Collection<JointPlan> jointPlans) {
225                final Collection<ValuedJointPlan> coll = new ArrayList<>();
226                for ( JointPlan jp : jointPlans ) {
227                        coll.add(
228                                        new ValuedJointPlan(
229                                                maxRank,
230                                                jp ) );
231                }
232                return Collections.min( coll ).jointPlan;
233        }
234
235        private final static Comparator<Plan> individualPlanComparator =
236                new Comparator<Plan>() {
237                        @Override
238                        public int compare(
239                                        final Plan o1,
240                                        final Plan o2) {
241                                if ( o1.getScore() == null || o2.getScore() == null ) {
242                                        throw new NullPointerException( "got plan with null score "+o1+" or "+o2 );
243                                }
244                                return Double.compare(
245                                                o1.getScore(),
246                                                o2.getScore() );
247                        }
248                };
249        private static Plan identifyWorst(
250                        final Collection<Plan> plans) {
251                return Collections.min( plans , individualPlanComparator );
252        }
253
254        private static class ValuedJointPlan implements Comparable<ValuedJointPlan> {
255                public final JointPlan jointPlan;
256                private final int[] nPlansOfRank;
257
258                public ValuedJointPlan(
259                                final int maxRank,
260                                final JointPlan jp) {
261                        this.jointPlan = jp;
262
263                        nPlansOfRank = new int[ maxRank ];
264                        for ( int i=0; i < nPlansOfRank.length; i++ ) {
265                                nPlansOfRank[ i ] = 0;
266                        }
267
268                        for ( Plan p : jp.getIndividualPlans().values() ) {
269                                nPlansOfRank[ rank( p ) ]++;
270                        }
271                }
272
273                private int rank(final Plan plan) {
274                        int nWorsePlans = 0;
275
276                        for ( Plan other : plan.getPerson().getPlans() ) {
277                                if ( other != plan && other.getScore() < plan.getScore() ) nWorsePlans++;
278
279                        }
280
281                        return nWorsePlans;
282                }
283
284                @Override
285                public int compareTo(final ValuedJointPlan o) {
286                        assert nPlansOfRank.length == o.nPlansOfRank.length;
287
288                        for ( int rank = 0; rank < nPlansOfRank.length; rank++ ) {
289                                // less worst plans is better
290                                if ( nPlansOfRank[ rank ] < o.nPlansOfRank[ rank ] ) return 1;
291                                // more worst plans is worst
292                                if ( nPlansOfRank[ rank ] > o.nPlansOfRank[ rank ] ) return -1;
293                        }
294
295                        // all was the same
296                        return 0;
297                }
298        }
299
300        private static final PlansPerComposition getPlansPerComposition(
301                        final JointPlans jointPlans,
302                        final Person person ) {
303                final PlansPerComposition plans = new PlansPerComposition();
304
305                for ( Plan plan : person.getPlans() ) {
306                        final JointPlan jp = jointPlans.getJointPlan( plan );
307
308                        if ( jp == null ) plans.addPlan( plan );
309                        else plans.addJointPlan( jp );
310                }
311
312                return plans;
313        }
314
315        private static class PlansPerComposition {
316                private final Collection<Plan> individualPlans = new ArrayList<>();
317                private final Map< Set<Id<Person>> , Collection<JointPlan> > jointPlans = new LinkedHashMap< >();
318
319                public void addPlan( final Plan p ) {
320                        individualPlans.add( p );
321                }
322
323                public void addJointPlan( final JointPlan jp ) {
324                        MapUtils.getCollection(
325                                        jp.getIndividualPlans().keySet(),
326                                        jointPlans ).add( jp );
327                }
328
329                public Collection<Plan> getIndividualPlans() {
330                        return individualPlans;
331                }
332
333                public Collection< Collection<JointPlan> > getJointPlansGroupedPerStructure() {
334                        return jointPlans.values();
335                }
336        }
