020package org.matsim.contrib.socnetsim.utils;
021
022import org.matsim.api.core.v01.Coord;
023import org.matsim.core.utils.collections.QuadTree;
024import org.matsim.core.utils.collections.Tuple;
025
026import java.util.ArrayList;
027import java.util.List;
028
029/**
030 * Helps working with quad trees, by re-building a new quad tree
031 * if necessary (avoids having to care about bounds).
032 * @author thibautd
033 */
034public class QuadTreeRebuilder<T> {
035        private static final double EPSILON = 1E-7;
036        private double minX = Double.POSITIVE_INFINITY;
037        private double minY = Double.POSITIVE_INFINITY;
038        private double maxX = Double.NEGATIVE_INFINITY;
039        private double maxY = Double.NEGATIVE_INFINITY;
040
041        private final List< Tuple<Coord, T> > elements = new ArrayList< Tuple<Coord, T> >();
042        private QuadTree<T> quadTree = null;
043
044        public synchronized QuadTree<T> getQuadTree() {
045                if ( quadTree == null ) buildQuadTree();
046                return quadTree;
047        }
048
049        private void buildQuadTree() {
050                assert minX <= maxX : minX +" > "+maxX;
051                assert minY <= maxY : minY +" > "+maxY;
052                this.quadTree = new QuadTree<T>(
053                                minX - EPSILON ,
054                                minY - EPSILON ,
055                                maxX + EPSILON ,
056                                maxY + EPSILON );
057
058                for ( Tuple< Coord , T > element : elements ) {
059                        try {
060                                quadTree.put(
061                                                element.getFirst().getX(),
062                                                element.getFirst().getY(),
063                                                element.getSecond() );
064                        }
065                        catch ( Exception e ) {
066                                // if insertion fail, give more informative error message.
067                                throw new RuntimeException( "problem at inserting element "
068                                                +elements.indexOf( element )+" of "
069                                                +elements.size()+": "
070                                                +element+" with bounds"
071                                                +" minX="+minX
072                                                +" minY="+minY
073                                                +" maxX="+maxX
074                                                +" maxY="+maxY ,
075                                                e );
076                        }
077                }
078        }
079
080        public void put( final double x , final double y , final T element ) {
081                put(new Coord(x, y), element );
082        }
083
084        public void put( final Coord coord , final T element ) {
085                elements.add( new Tuple< Coord , T >( coord , element ) );
086                if ( quadTree != null && inBounds( coord ) ) {
087                        quadTree.put( coord.getX() , coord.getY() , element );
088                }
089                else {
090                        minX = Math.min( coord.getX() , minX );
091                        minY = Math.min( coord.getY() , minY );
092                        maxX = Math.max( coord.getX() , maxX );
093                        maxY = Math.max( coord.getY() , maxY );
094                        quadTree = null;
095                }
096        }
097
098        private boolean inBounds(final Coord coord) {
099                return coord.getX() > minX + EPSILON &&
100                        coord.getX() < maxX - EPSILON &&
101                        coord.getY() > minY + EPSILON &&
102                        coord.getY() < maxY - EPSILON ;
103        }
104}
