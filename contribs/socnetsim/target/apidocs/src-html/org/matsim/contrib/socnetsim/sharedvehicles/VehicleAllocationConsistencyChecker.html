020package org.matsim.contrib.socnetsim.sharedvehicles;
021
022import com.google.inject.Inject;
023import org.apache.log4j.Logger;
024import org.matsim.api.core.v01.Id;
025import org.matsim.api.core.v01.Scenario;
026import org.matsim.api.core.v01.population.Person;
027import org.matsim.api.core.v01.population.Plan;
028import org.matsim.api.core.v01.population.Population;
029import org.matsim.core.controler.events.IterationEndsEvent;
030import org.matsim.core.controler.events.IterationStartsEvent;
031import org.matsim.core.controler.listener.IterationEndsListener;
032import org.matsim.core.controler.listener.IterationStartsListener;
033import org.matsim.contrib.socnetsim.framework.population.JointPlan;
034import org.matsim.contrib.socnetsim.framework.population.JointPlans;
035
036import java.util.Collections;
037import java.util.HashSet;
038import java.util.Set;
039
040public final class VehicleAllocationConsistencyChecker implements IterationEndsListener, IterationStartsListener {
041        private final static Logger log = Logger.getLogger( VehicleAllocationConsistencyChecker.class );
042
043        private boolean gotError = false;
044
045        private final Population population;
046        private final JointPlans jointPlans;
047
048        @Inject
049        public VehicleAllocationConsistencyChecker( final Scenario sc ) {
050                this.population = sc.getPopulation();
051                this.jointPlans = (JointPlans) sc.getScenarioElement( JointPlans.ELEMENT_NAME );
052        }
053
054        @Override
055        public void notifyIterationEnds(final IterationEndsEvent event) {
056                log.info( "Checking consistency of vehicle allocation" );
057                final Set<Id> knownVehicles = new HashSet<Id>();
058                final Set<JointPlan> knownJointPlans = new HashSet<JointPlan>();
059
060                boolean hadNull = false;
061                boolean hadNonNull = false;
062                for ( Person person : population.getPersons().values() ) {
063                        final Plan plan = person.getSelectedPlan();
064                        final JointPlan jp = jointPlans.getJointPlan( plan );
065
066                        final Set<Id> vehsOfPlan = 
067                                 jp != null && knownJointPlans.add( jp ) ?
068                                        SharedVehicleUtils.getVehiclesInJointPlan(
069                                                        jp,
070                                                        SharedVehicleUtils.DEFAULT_VEHICULAR_MODES) :
071                                        (jp == null ?
072                                                SharedVehicleUtils.getVehiclesInPlan(
073                                                                plan,
074                                                                SharedVehicleUtils.DEFAULT_VEHICULAR_MODES) :
075                                                Collections.<Id>emptySet());
076
077                        for ( Id v : vehsOfPlan ) {
078                                if ( v == null ) {
079                                        if ( hadNonNull ) {
080                                                log.error( "got null and non-null vehicles" );
081                                                gotError = true;
082                                        }
083
084                                        hadNull = true;
085                                }
086                                else {
087                                        if ( hadNull ) {
088                                                log.error( "got null and non-null vehicles" );
089                                                gotError = true;
090                                        }
091                                        if ( !knownVehicles.add( v ) ) {
092                                                log.error( "inconsistent allocation of vehicle "+v+" (found in several distinct joint plans)" );
093                                                gotError = true;
094                                        }
095                                        hadNonNull = true;
096                                }
097                        }
098                }
099        }
100
101        @Override
102        public void notifyIterationStarts( IterationStartsEvent event ) {
103                if ( gotError ) throw new RuntimeException( "inconsistency detected. Look at error messages for details" );
104        }
105}
