021

023import java.util.Collection;
024import java.util.LinkedHashMap;
025import java.util.List;
026import java.util.Map;
027import java.util.Random;
028
029import org.apache.log4j.Logger;
030import org.matsim.analysis.IterationStopWatch;
031import org.matsim.api.core.v01.Scenario;
032import org.matsim.api.core.v01.population.Person;
033import org.matsim.api.core.v01.population.Population;
034import org.matsim.core.gbl.MatsimRandom;
035import org.matsim.core.replanning.ReplanningContext;
036import org.matsim.utils.objectattributes.ObjectAttributes;
037
038import org.matsim.core.utils.collections.MapUtils;
039import org.matsim.contrib.socnetsim.framework.population.JointPlans;
040import org.matsim.contrib.socnetsim.framework.replanning.grouping.GroupIdentifier;
041import org.matsim.contrib.socnetsim.framework.replanning.grouping.ReplanningGroup;
042
043/**
044 * Implements the group-level replanning logic.
045 * Not very different from the standard StrategyManager.
046 * @author thibautd
047 */
048public class GroupStrategyManager {
049        private static final Logger log =
050                Logger.getLogger(GroupStrategyManager.class);
051
052        private IterationStopWatch stopWatch;
053
054        private final GroupIdentifier groupIdentifier;
055        private final GroupStrategyRegistry registry;
056
057        private final Random random;
058        private final List<Listener> listeners = new ArrayList<Listener>( 1 );
059
060        public GroupStrategyManager(
061                        final IterationStopWatch stopWatch,
062                        final GroupIdentifier groupIdentifier,
063                        final GroupStrategyRegistry registry ) {
064                this.stopWatch = stopWatch;
065                this.groupIdentifier = groupIdentifier;
066                this.registry = registry;
067                this.random = MatsimRandom.getLocalInstance();
068        }
069
070        public void addListener(final Listener l) {
071                listeners.add( l );
072        }
073
074        public final void run(
075                        final ReplanningContext context,
076                        final Scenario scenario) {
077                final Population population = scenario.getPopulation();
078                final JointPlans jointPlans = (JointPlans) scenario.getScenarioElement( JointPlans.ELEMENT_NAME );
079                final Collection<ReplanningGroup> groups = groupIdentifier.identifyGroups( population );
080
081                final Map<GroupPlanStrategy, List<ReplanningGroup>> strategyAllocations =
082                        new LinkedHashMap<GroupPlanStrategy, List<ReplanningGroup>>();
083                if ( stopWatch != null ) stopWatch.beginOperation( "remove plans alloc strategy" );
084                for (ReplanningGroup g : groups) {
085                        registry.getExtraPlanRemover().removePlansInGroup(
086                                        jointPlans,
087                                        g );
088
089                        final String subpop = identifySubpopulation( g , scenario );
090                        final GroupPlanStrategy strategy = registry.chooseStrategy( context.getIteration() , subpop , random.nextDouble() );
091                        final List<ReplanningGroup> alloc = MapUtils.getList( strategy , strategyAllocations );
092
093                        notifyAlloc( context.getIteration() , g , strategy );
094                        alloc.add( g );
095                }
096                if ( stopWatch != null ) stopWatch.endOperation( "remove plans alloc strategy" );
097
098                for (Map.Entry<GroupPlanStrategy, List<ReplanningGroup>> e : strategyAllocations.entrySet()) {
099                        final GroupPlanStrategy strategy = e.getKey();
100                        final List<ReplanningGroup> toHandle = e.getValue();
101                        log.info( "passing "+toHandle.size()+" groups to strategy "+strategy );
102
103                        if ( log.isTraceEnabled() ) {
104                                for ( ReplanningGroup group : toHandle ) {
105                                        log.trace( "Handling group "+group+" with strategy "+strategy );
106                                }
107                        }
108                        if ( stopWatch != null ) stopWatch.beginOperation( "strategy "+strategy );
109                        strategy.run(
110                                        context,
111                                        jointPlans,
112                                        toHandle );
113                        if ( stopWatch != null ) stopWatch.endOperation( "strategy "+strategy );
114                        log.info( "strategy "+strategy+" finished" );
115                }
116        }
117
118        private static String identifySubpopulation(
119                        final ReplanningGroup g,
120                        final Scenario sc) {
121                final String attName = sc.getConfig().plans().getSubpopulationAttributeName();
122                final ObjectAttributes atts = sc.getPopulation().getPersonAttributes();
123
124                String name = null;
125
126                for ( Person p : g.getPersons() ) {
127                        final String persSubPop = (String) atts.getAttribute( p.getId().toString() , attName );
128                        if ( persSubPop == null && name != null ) throw new RuntimeException( "inconsistent subpopulations in group "+g );
129                        if ( name != null && !name.equals( persSubPop ) ) throw new RuntimeException( "inconsistent subpopulations in group "+g );
130                        name = persSubPop;
131                }
132
133                return name;
134        }
135
136        private void notifyAlloc(
137                        final int i,
138                        final ReplanningGroup g,
139                        final GroupPlanStrategy strategy) {
140                for ( Listener l : listeners ) l.notifyAlloc( i , g , strategy );
141        }
142
143        public static interface Listener {
144                public void notifyAlloc(
145                                final int iteration,
146                                final ReplanningGroup g,
147                                final GroupPlanStrategy strategy );
148        }
149}
