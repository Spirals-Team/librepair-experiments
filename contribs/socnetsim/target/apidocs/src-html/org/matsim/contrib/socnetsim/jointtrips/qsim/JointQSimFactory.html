020package org.matsim.contrib.socnetsim.jointtrips.qsim;

022import org.apache.log4j.Logger;
023import org.matsim.api.core.v01.Scenario;
024import org.matsim.core.api.experimental.events.EventsManager;
025import org.matsim.core.config.groups.QSimConfigGroup;
026import org.matsim.core.mobsim.framework.AgentSource;
027import org.matsim.core.mobsim.framework.MobsimFactory;
028import org.matsim.core.mobsim.qsim.ActivityEngine;
029import org.matsim.core.mobsim.qsim.QSim;

031import org.matsim.core.mobsim.qsim.agents.DefaultAgentFactory;
032import org.matsim.core.mobsim.qsim.agents.TransitAgentFactory;
033import org.matsim.core.mobsim.qsim.pt.ComplexTransitStopHandlerFactory;
034import org.matsim.core.mobsim.qsim.pt.TransitQSimEngine;
035import org.matsim.core.mobsim.qsim.qnetsimengine.QNetsimEngine;
036
037import com.google.inject.Inject;
038import com.google.inject.Provider;
039
040import org.matsim.contrib.socnetsim.qsim.NetsimWrappingQVehicleProvider;
041import org.matsim.contrib.socnetsim.jointtrips.population.JointActingTypes;
042import org.matsim.contrib.socnetsim.sharedvehicles.qsim.PopulationAgentSourceWithVehicles;
043
044import java.util.ArrayList;
045import java.util.List;
046
047/**
048 * @author thibautd
049 */
050public class JointQSimFactory implements MobsimFactory, Provider<QSim> {
051        private static final Logger log =
052                Logger.getLogger(JointQSimFactory.class);
053        
054        private final Scenario sc;
055        private final EventsManager events;
056
057        @Inject
058        public JointQSimFactory( final Scenario sc, final EventsManager events ) {
059                this.sc = sc;
060                this.events = events;
061        }
062        
063        public JointQSimFactory() {
064                this( null , null );
065        }
066        
067        @Override
068        public QSim get() {
069                return createMobsim( sc , events );
070        }
071
072        @Override
073        public QSim createMobsim(
074                        final Scenario sc1,
075                        final EventsManager eventsManager) {
076        final QSimConfigGroup conf = sc1.getConfig().qsim();
077        if (conf == null) {
078            throw new NullPointerException("There is no configuration set for the QSim. Please add the module 'qsim' to your config file.");
079        }
080
081                if ( !conf.getMainModes().contains( JointActingTypes.DRIVER ) ) {
082                        log.warn( "adding the driver mode as a main mode in the config at "+getClass()+" initialisation!" );
083                        final List<String> ms = new ArrayList<String>( conf.getMainModes() );
084                        ms.add( JointActingTypes.DRIVER );
085                        conf.setMainModes( ms );
086                }
087
088                // default initialisation
089                final QSim qSim = new QSim( sc1 , eventsManager );
090
091                final ActivityEngine activityEngine = new ActivityEngine(eventsManager, qSim.getAgentCounter());
092                qSim.addMobsimEngine( activityEngine );
093                qSim.addActivityHandler( activityEngine );
094
095        final QNetsimEngine netsimEngine = new QNetsimEngine(qSim);
096                qSim.addMobsimEngine( netsimEngine );
097                // DO NOT ADD DEPARTURE HANDLER: it is done by the joint departure handler
098
099                final JointModesDepartureHandler jointDepHandler = new JointModesDepartureHandler( netsimEngine );
100                qSim.addDepartureHandler( jointDepHandler );
101                qSim.addMobsimEngine( jointDepHandler );
102
103                final DefaultTeleportationEngine teleportationEngine = new DefaultTeleportationEngine(sc1, eventsManager);
104                qSim.addMobsimEngine( teleportationEngine );
105
106        if (sc1.getConfig().transit().isUseTransit()) {
107            final TransitQSimEngine transitEngine = new TransitQSimEngine(qSim);
108            transitEngine.setTransitStopHandlerFactory(new ComplexTransitStopHandlerFactory());
109            qSim.addDepartureHandler(transitEngine);
110            qSim.addAgentSource(transitEngine);
111            qSim.addMobsimEngine(transitEngine);
112        }
113
114                final PassengerUnboardingAgentFactory passAgentFactory =
115                                        new PassengerUnboardingAgentFactory(
116                                                sc1.getConfig().transit().isUseTransit() ?
117                                                        new TransitAgentFactory(qSim) :
118                                                        new DefaultAgentFactory(qSim) ,
119                                                new NetsimWrappingQVehicleProvider(
120                                                        netsimEngine) );
121        final AgentSource agentSource =
122                        new PopulationAgentSourceWithVehicles(
123                                        sc1.getPopulation(),
124                                        passAgentFactory,
125                                        qSim);
126                qSim.addMobsimEngine( passAgentFactory );
127        qSim.addAgentSource(agentSource);
128        return qSim;
129        }
130}
