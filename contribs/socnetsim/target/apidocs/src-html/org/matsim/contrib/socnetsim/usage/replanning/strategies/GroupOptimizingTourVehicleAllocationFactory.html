020package org.matsim.contrib.socnetsim.usage.replanning.strategies;
021
022import org.matsim.api.core.v01.Scenario;
023import org.matsim.core.router.CompositeStageActivityTypes;
024import org.matsim.core.router.TripRouter;
025
026import org.matsim.contrib.socnetsim.framework.PlanRoutingAlgorithmFactory;
027import org.matsim.contrib.socnetsim.jointtrips.population.JointActingTypes;
028import org.matsim.contrib.socnetsim.framework.population.JointPlans;
029import org.matsim.contrib.socnetsim.framework.replanning.GroupPlanStrategy;
030import org.matsim.contrib.socnetsim.usage.replanning.GroupPlanStrategyFactoryUtils;
031import org.matsim.contrib.socnetsim.framework.replanning.modules.PlanLinkIdentifier;
032import org.matsim.contrib.socnetsim.framework.replanning.modules.PlanLinkIdentifier.Strong;
033import org.matsim.contrib.socnetsim.sharedvehicles.SharedVehicleUtils;
034import org.matsim.contrib.socnetsim.sharedvehicles.VehicleRessources;
035import org.matsim.contrib.socnetsim.sharedvehicles.replanning.OptimizeVehicleAllocationAtTourLevelModule;
036
037import com.google.inject.Inject;
038import com.google.inject.Provider;
039
040/**
041 * @author thibautd
042 */
043public class GroupOptimizingTourVehicleAllocationFactory extends AbstractConfigurableSelectionStrategy {
044
045        private final Scenario sc;
046        private final PlanRoutingAlgorithmFactory planRoutingAlgorithmFactory;
047        private final Provider<TripRouter> tripRouterFactory;
048        private final PlanLinkIdentifier planLinkIdentifier;
049
050        @Inject
051        public GroupOptimizingTourVehicleAllocationFactory( Scenario sc , PlanRoutingAlgorithmFactory planRoutingAlgorithmFactory ,
052                        Provider<TripRouter> tripRouterFactory , @Strong PlanLinkIdentifier planLinkIdentifier ) {
053                this.sc = sc;
054                this.planRoutingAlgorithmFactory = planRoutingAlgorithmFactory;
055                this.tripRouterFactory = tripRouterFactory;
056                this.planLinkIdentifier = planLinkIdentifier;
057        }
058
059        @Override
060        public GroupPlanStrategy get() {
061                final GroupPlanStrategy strategy = instantiateStrategy( sc.getConfig() );
062
063                final CompositeStageActivityTypes stageActs = new CompositeStageActivityTypes();
064                stageActs.addActivityTypes( tripRouterFactory.get().getStageActivityTypes() );
065                stageActs.addActivityTypes( JointActingTypes.JOINT_STAGE_ACTS );
066                strategy.addStrategyModule(
067                                //new AllocateVehicleToPlansInGroupPlanModule(
068                                new OptimizeVehicleAllocationAtTourLevelModule(
069                                                sc.getConfig().global().getNumberOfThreads(),
070                                                stageActs,
071                                                (VehicleRessources) sc.getScenarioElement(
072                                                        VehicleRessources.ELEMENT_NAME ),
073                                                SharedVehicleUtils.DEFAULT_VEHICULAR_MODES,
074                                                true));
075
076                strategy.addStrategyModule(
077                                GroupPlanStrategyFactoryUtils.createReRouteModule(
078                                        sc.getConfig(),
079                                        planRoutingAlgorithmFactory,
080                                        tripRouterFactory ) );
081                
082
083                strategy.addStrategyModule(
084                                GroupPlanStrategyFactoryUtils.createRecomposeJointPlansModule(
085                                        sc.getConfig(),
086                                        ((JointPlans) sc.getScenarioElement( JointPlans.ELEMENT_NAME )).getFactory(),
087                                        planLinkIdentifier ) );
088
089                return strategy;
090
091        }
092}
