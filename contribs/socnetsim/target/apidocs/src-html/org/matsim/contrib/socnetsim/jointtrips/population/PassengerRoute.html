020package org.matsim.contrib.socnetsim.jointtrips.population;
021
022import org.matsim.api.core.v01.Id;
023import org.matsim.api.core.v01.network.Link;
024import org.matsim.api.core.v01.population.Person;
025import org.matsim.api.core.v01.population.Route;
026import org.matsim.core.utils.misc.Time;
027
028/**
029 * A route for passenger trips.
030 * @author thibautd
031 */
032public class PassengerRoute implements Route {
033        private double distance = Double.NaN;
034        private double travelTime = Time.UNDEFINED_TIME;
035        private Id<Link> startLink = null;
036        private Id<Link> endLink = null;
037        private Id<Person> driver = null;
038
039        private PassengerRoute() {}
040
041        public PassengerRoute(
042                        final Id<Link> startLink,
043                        final Id<Link> endLink) {
044                this.startLink = startLink;
045                this.endLink = endLink;
046        }
047
048        public Id<Person> getDriverId() {
049                return driver;
050        }
051
052        public void setDriverId(final Id<Person> d) {
053                driver = d;
054        }
055
056        @Override
057        public double getDistance() {
058                return distance;
059        }
060
061        @Override
062        public void setDistance(final double distance) {
063                this.distance = distance;
064        }
065
066        @Override
067        public double getTravelTime() {
068                return travelTime;
069        }
070
071        @Override
072        public void setTravelTime(final double travelTime) {
073                this.travelTime = travelTime;
074        }
075
076        @Override
077        public Id<Link> getStartLinkId() {
078                return startLink;
079        }
080
081        @Override
082        public Id<Link> getEndLinkId() {
083                return endLink;
084        }
085
086        @Override
087        public void setStartLinkId(final Id<Link> linkId) {
088                startLink = linkId;
089        }
090
091        @Override
092        public void setEndLinkId(final Id<Link> linkId) {
093                endLink = linkId;
094        }
095
096        @Override
097        public void setRouteDescription(
098                        final String routeDescription
099                        ) {
100                driver = Id.create(routeDescription.trim(), Person.class);
101        }
102
103        @Override
104        public String getRouteDescription() {
105                return driver != null ? driver.toString() : "";
106        }
107
108        @Override
109        public String getRouteType() {
110                return "passenger";
111        }
112
113        @Override
114        public PassengerRoute clone() {
115                PassengerRoute c = new PassengerRoute();
116                c.distance = distance;
117                c.travelTime= travelTime;
118                c.startLink = startLink;
119                c.endLink = endLink;
120                c.driver = driver;
121                return c;
122        }
123
124        @Override
125        public String toString() {
126                return "[PassengerRoute: "+
127                        "distance="+distance+"; "+
128                        "travelTime="+travelTime+"; "+
129                        "startLink="+startLink+"; "+
130                        "endLink="+endLink+"; "+
131                        "driver="+driver+"]";
132        }
133}
