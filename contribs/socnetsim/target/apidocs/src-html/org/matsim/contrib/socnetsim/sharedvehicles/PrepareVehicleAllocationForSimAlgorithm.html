020package org.matsim.contrib.socnetsim.sharedvehicles;
021
022import java.util.HashSet;
023import java.util.Random;
024import java.util.Set;
025
026import org.matsim.api.core.v01.population.Person;
027import org.matsim.api.core.v01.population.Plan;
028
029import org.matsim.contrib.socnetsim.framework.population.JointPlan;
030import org.matsim.contrib.socnetsim.framework.population.JointPlans;
031import org.matsim.contrib.socnetsim.framework.replanning.GenericPlanAlgorithm;
032import org.matsim.contrib.socnetsim.framework.replanning.grouping.GroupPlans;
033import org.matsim.contrib.socnetsim.framework.replanning.grouping.ReplanningGroup;
034import org.matsim.contrib.socnetsim.framework.replanning.modules.PlanLinkIdentifier;
035import org.matsim.contrib.socnetsim.framework.replanning.modules.RecomposeJointPlanAlgorithm;
036import org.matsim.contrib.socnetsim.sharedvehicles.replanning.AllocateVehicleToPlansInGroupPlanAlgorithm;
037
038/**
039 * @author thibautd
040 */
041public class PrepareVehicleAllocationForSimAlgorithm implements GenericPlanAlgorithm<ReplanningGroup> {
042        private final JointPlans jointPlans;
043
044        private final GenericPlanAlgorithm<GroupPlans> allocateVehiclesAlgo;
045        private final GenericPlanAlgorithm<GroupPlans> recomposeJointPlansAlgo;
046
047        public PrepareVehicleAllocationForSimAlgorithm(
048                        final Random random,
049                        final JointPlans jointPlans,
050                        final VehicleRessources vehicles,
051                        final PlanLinkIdentifier planLinkIdentifier) {
052                this.jointPlans = jointPlans;
053
054                this.allocateVehiclesAlgo =
055                        new AllocateVehicleToPlansInGroupPlanAlgorithm(
056                                        random,
057                                        vehicles,
058                                        SharedVehicleUtils.DEFAULT_VEHICULAR_MODES,
059                                        true,
060                                        true);
061
062                this.recomposeJointPlansAlgo =
063                        planLinkIdentifier == null ? null :
064                        new RecomposeJointPlanAlgorithm(
065                                        jointPlans.getFactory(),
066                                        planLinkIdentifier);
067        }
068
069        @Override
070        public void run(final ReplanningGroup group) {
071                final Set<Plan> ps = new HashSet<Plan>();
072                final Set<JointPlan> jps = new HashSet<JointPlan>();
073
074                for ( Person person : group.getPersons() ) {
075                        if ( person.getPlans().size() > 1 ) return;
076                        final Plan plan = person.getSelectedPlan();
077                        final JointPlan jp = jointPlans.getJointPlan( plan );
078
079                        if ( jp != null ) {
080                                // TODO: check that all members of JP are in group
081                                jps.add( jp );
082                        }
083                        else {
084                                if ( plan == null ) throw new NullPointerException();
085                                ps.add( plan );
086                        }
087                }
088
089                final GroupPlans groupPlans = new GroupPlans( jps , ps );
090                allocateVehiclesAlgo.run( groupPlans );
091                if ( recomposeJointPlansAlgo != null ) {
092                        recomposeJointPlansAlgo.run( groupPlans );
093                }
094
095                for ( JointPlan jp : jps ) jointPlans.removeJointPlan( jp );
096                jointPlans.addJointPlans( groupPlans.getJointPlans() );
097        }
098}
