023import java.util.ArrayList;
024import java.util.Collection;
025import java.util.HashMap;
026import java.util.HashSet;
027import java.util.List;
028import java.util.Map;
029import java.util.Queue;
030import java.util.Random;
031import java.util.Set;

033import org.apache.log4j.Logger;

035import org.matsim.api.core.v01.Scenario;
036import org.matsim.api.core.v01.population.Person;
037import org.matsim.api.core.v01.population.Plan;

039import org.matsim.core.gbl.MatsimRandom;
040
041import org.matsim.contrib.socnetsim.framework.population.JointPlan;
042import org.matsim.contrib.socnetsim.framework.population.JointPlans;

044
045import com.google.inject.Inject;
046import com.google.inject.Singleton;
047
048/**
049 * @author thibautd
050 */
051@Singleton
052public class DynamicGroupIdentifier implements GroupIdentifier {
053        private static final Logger log =
054                Logger.getLogger(DynamicGroupIdentifier.class);
055
056        private final Scenario scenario;
057        private final Random random;
058
059        @Inject
060        public DynamicGroupIdentifier(final Scenario scenario) {
061                this.scenario = scenario;
062                this.random = MatsimRandom.getLocalInstance();
063        }
064
065        @Override
066        public Collection<ReplanningGroup> identifyGroups(
067                        final Population population) {
068                final SocialNetwork socialNetwork = (SocialNetwork)
069                        scenario.getScenarioElement( SocialNetwork.ELEMENT_NAME );
070                final JointPlans jointPlans = (JointPlans)
071                        scenario.getScenarioElement( JointPlans.ELEMENT_NAME );
072
073                final Queue<TaggedReplanningGroup> jointPlansGroups =
074                        identifyGroupsBasedOnJointPlans( population , jointPlans );
075                final List<ReplanningGroup> mergedGroups =
076                        randomlyMergeGroups( jointPlansGroups , socialNetwork );
077
078                return mergedGroups;
079        }
080
081        private List<ReplanningGroup> randomlyMergeGroups(
082                        final Queue<TaggedReplanningGroup> groupsToMerge,
083                        final SocialNetwork socialNetwork) {
084                final Map<Id, TaggedReplanningGroup> groupPerPerson = new HashMap<Id, TaggedReplanningGroup>();
085
086                for ( TaggedReplanningGroup g : groupsToMerge ) {
087                        for ( Person p : g.group.getPersons() ) {
088                                final TaggedReplanningGroup rem = groupPerPerson.put( p.getId() , g );
089                                assert rem == null;
090                        }
091                }
092
093                final List<ReplanningGroup> merged = new ArrayList<ReplanningGroup>();
094                while ( true ) {
095                        final TaggedReplanningGroup g = pollNonAllocated( groupsToMerge );
096
097                        if ( g == null || groupsToMerge.isEmpty() ) {
098                                if ( g != null ) merged.add( g.group );
099                                return merged;
100                        }
101                        for ( Person p : g.group.getPersons() ) groupPerPerson.remove( p.getId() );
102
103                        final List<Id> groupAlters = getGroupAlters( g.group , socialNetwork , groupPerPerson.keySet() );
104
105                        if ( groupAlters.isEmpty() ) {
106                                // no need to merge
107                                if ( log.isTraceEnabled() ) {
108                                        log.trace( "add non-merged group "+g );
109                                }
110                                merged.add( g.group );
111                                continue;
112                        }
113
114                        final Id selectedAlter = groupAlters.get( random.nextInt( groupAlters.size() ) );
115                        final TaggedReplanningGroup selectedGroup = groupPerPerson.remove( selectedAlter );
116                        assert selectedGroup != null;
117                        for ( Person p : selectedGroup.group.getPersons() ) groupPerPerson.remove( p.getId() );
118                        selectedGroup.isAllocated = true;
119
120                        final ReplanningGroup newGroup = new ReplanningGroup();
121                        for ( Person p : g.group.getPersons() ) newGroup.addPerson( p );
122                        for ( Person p : selectedGroup.group.getPersons() ) newGroup.addPerson( p );
123
124                        if ( log.isTraceEnabled() ) {
125                                log.trace( "add merged groups "+g+" and "+selectedGroup );
126                        }
127                        merged.add( newGroup );
128                }
129        }
130
131        private static TaggedReplanningGroup pollNonAllocated(
132                        final Queue<TaggedReplanningGroup> groupsToMerge) {
133                while ( !groupsToMerge.isEmpty() ) {
134                        final TaggedReplanningGroup g = groupsToMerge.poll();
135                        if ( !g.isAllocated ) return g;
136                }
137                return null;
138        }
139
140        private static List<Id> getGroupAlters(
141                        final ReplanningGroup g,
142                        final SocialNetwork socialNetwork,
143                        final Set<Id> allowedPersonIds) {
144                final List<Id> groupAlters = new ArrayList<Id>();
145
146                final Set<Id<Person>> groupIds = new HashSet<>( g.getPersons().size() * 2 );
147                for ( Person p : g.getPersons() ) groupIds.add( p.getId() );
148
149                for ( Person p : g.getPersons() ) {
150                        final Set<Id<Person>> alters = socialNetwork.getAlters( p.getId() );
151                        for ( Id<Person> alter : alters ) {
152                                // group alters are agents which are not in the group but alters
153                                // of a member of the group.
154                                // Alters are added several times if they are alters of several egos
155                                // in the group, so that the probability of selecting them
156                                // is proportionnal to their "popularity" in the group.
157                                if ( !groupIds.contains( alter ) && allowedPersonIds.contains( alter ) ) groupAlters.add( alter );
158                        }
159                }
160
161                return groupAlters;
162        }
163
164        private Queue<TaggedReplanningGroup> identifyGroupsBasedOnJointPlans(
165                        final Population population,
166                        final JointPlans jointPlans) {
167                final Queue<Person> remainingPersons = new ArrayDeque<Person>( population.getPersons().values() );
168                final Queue<TaggedReplanningGroup> groups = new ArrayDeque<TaggedReplanningGroup>();
169
170                while ( !remainingPersons.isEmpty() ) {
171                        final ReplanningGroup group = new ReplanningGroup();
172                        groups.add( new TaggedReplanningGroup( group ) );
173                        final Person person = remainingPersons.remove();
174                        group.addPerson( person );
175                        addPersonsLinkedByJointPlans(
176                                        group,
177                                        person,
178                                        remainingPersons,
179                                        jointPlans );
180                }
181
182                return groups;
183        }
184
185        private void addPersonsLinkedByJointPlans(
186                        final ReplanningGroup group,
187                        final Person person,
188                        final Queue<Person> remainingPersons,
189                        final JointPlans jointPlans) {
190                for ( Plan plan : person.getPlans() ) {
191                        final JointPlan jp = jointPlans.getJointPlan( plan );
192                        if ( jp == null ) continue;
193
194                        for ( Plan planInJp : jp.getIndividualPlans().values() ) {
195                                final Person alter = planInJp.getPerson();
196                                if ( alter == person ) continue;
197                                // should be more efficient than checking if in remaining persons
198                                if ( group.getPersons().contains( alter ) ) continue;
199                                if ( !remainingPersons.remove( alter ) ) {
200                                        // should not happen
201                                        throw new RuntimeException( "alter "+alter+" was not available and not in group "+group+" for person "+person );
202                                }
203
204                                group.addPerson( alter );
205                                // TODO: avoid recursion. we fill the stack with redundant information.
206                                addPersonsLinkedByJointPlans(
207                                                group,
208                                                alter,
209                                                remainingPersons,
210                                                jointPlans );
211                        }
212                }
213        }
214
215        // the groups are put in a queue, but we also need to remove
216        // specified elements. Tagging them as "allocated" is much faster than
217        // removing them.
218        private static class TaggedReplanningGroup {
219                public final ReplanningGroup group;
220                public boolean isAllocated = false;
221
222                public TaggedReplanningGroup(final ReplanningGroup r) {
223                        this.group = r;
224                }
225        }
226}
