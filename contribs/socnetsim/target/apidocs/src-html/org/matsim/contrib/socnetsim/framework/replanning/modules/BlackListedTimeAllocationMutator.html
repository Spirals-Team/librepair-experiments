020package org.matsim.contrib.socnetsim.framework.replanning.modules;
021
022import org.apache.log4j.Logger;
023import org.matsim.api.core.v01.population.Activity;
024import org.matsim.api.core.v01.population.Activity;
025import org.matsim.api.core.v01.population.Plan;
026import org.matsim.core.population.algorithms.PlanAlgorithm;
027import org.matsim.core.router.StageActivityTypes;
028import org.matsim.core.router.TripStructureUtils;
029import org.matsim.core.utils.misc.Time;
030
031import java.util.List;
032import java.util.Random;
033
034/**
035 * A time allocation mutator to use with multi-leg routing.
036 *
037 * @author thibautd
038 */
039public class BlackListedTimeAllocationMutator implements PlanAlgorithm {
040        private static final Logger log =
041                Logger.getLogger(BlackListedTimeAllocationMutator.class);
042
043        private final double mutationRange;
044        private final StageActivityTypes blackList;
045        private final Random random;
046        private Setting setting = Setting.MUTATE_END;
047
048        public enum Setting {
049                MUTATE_DUR,
050                MUTATE_END,
051                MUTATE_END_AS_DUR;
052        }
053
054        public BlackListedTimeAllocationMutator(
055                        final StageActivityTypes blackList,
056                        final double mutationRange,
057                        final Random random) {
058                this.blackList = blackList;
059                this.mutationRange = mutationRange;
060                this.random = random;
061                log.debug( "setting initialized to "+setting );
062        }
063
064        @Override
065        public void run(final Plan plan) {
066                final List<Activity> activities = TripStructureUtils.getActivities( plan , blackList );
067                final int nActs = activities.size();
068                // when mutating durations "blindly", avoid creating activities ending before
069                // the previous activity.
070                double lastEndTime = Double.NEGATIVE_INFINITY;
071                for ( Activity a : activities ) {
072                        switch ( setting ) {
073                                case MUTATE_DUR:
074                                        ((Activity) a).setMaximumDuration( mutateTime( a.getMaximumDuration() ) );
075                                        break;
076                                case MUTATE_END:
077                                        a.setEndTime( mutateTime( a.getEndTime() ) );
078                                        if ( a.getEndTime() != Time.UNDEFINED_TIME &&
079                                                        a.getEndTime() < lastEndTime ) {
080                                                a.setEndTime( lastEndTime );
081                                        }
082                                        lastEndTime = a.getEndTime();
083                                        break;
084                                case MUTATE_END_AS_DUR:
085                                        final double oldTime = a.getEndTime();
086                                        if ( oldTime == Time.UNDEFINED_TIME ) break;
087                                        final double newTime = mutateTime( oldTime );
088                                        // doing this so rather than sampling mut directly allows
089                                        // to avoid negative times
090                                        final double mut = newTime - oldTime;
091                                        // shift all times after the mutated time (as if we were working on durations)
092                                        for ( Activity currAct : activities.subList( activities.indexOf( a ) , nActs ) ) {
093                                                currAct.setEndTime( currAct.getEndTime() + mut );
094                                        }
095                                        break;
096                                default:
097                                        throw new RuntimeException( "what is that? "+setting );
098                        }
099                }
100        }
101
102        private double mutateTime(final double time) {
103                // do not do anything if time is undefined
104                if ( time == Time.UNDEFINED_TIME ) return time;
105                if ( Double.isNaN( time ) ) throw new IllegalArgumentException( ""+time );
106
107                final double t = time + (int)((this.random.nextDouble() * 2.0 - 1.0) * mutationRange);
108                assert !Double.isNaN( t ) : t;
109                assert !Double.isInfinite( t ) : t;
110                return t < 0 ? 0 : t;
111        }
112
113        public void setSetting(final Setting setting) {
114                log.debug( "setting set to "+setting );
115                this.setting = setting;
116        }
117}
