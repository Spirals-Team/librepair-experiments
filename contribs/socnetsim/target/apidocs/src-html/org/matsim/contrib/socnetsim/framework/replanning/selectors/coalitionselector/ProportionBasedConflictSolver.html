019package org.matsim.contrib.socnetsim.framework.replanning.selectors.coalitionselector;
020
021import org.apache.log4j.Logger;
022import org.matsim.api.core.v01.Id;
023import org.matsim.api.core.v01.population.Person;
024import org.matsim.api.core.v01.population.Plan;
025import org.matsim.contrib.socnetsim.framework.population.SocialNetwork;
026
027import java.util.ArrayList;
028import java.util.Collection;
029import java.util.HashSet;
030import java.util.Set;
031
032/**
033 * @author thibautd
034 */
035public class ProportionBasedConflictSolver implements CoalitionSelector.ConflictSolver {
036        private static final Logger log = Logger.getLogger( ProportionBasedConflictSolver.class );
037        private final SocialNetwork socialNetwork;
038        private final boolean tabuFriends;
039
040        public ProportionBasedConflictSolver( final SocialNetwork socialNetwork ) {
041                this.socialNetwork = socialNetwork;
042                // make configurable?
043                this.tabuFriends = false;
044        }
045
046        @Override
047        public void attemptToSolveConflicts(
048                        final CoalitionSelector.RecordsOfJointPlan recordsPerJointPlan) {
049                double maxProportion = Double.NEGATIVE_INFINITY;
050
051                final Collection<Collection<PlanRecord>> currentMostPointed = new ArrayList<>();
052
053                for ( final Collection<PlanRecord> records : recordsPerJointPlan.values() ) {
054                        final double proportion = getProportionPointed( records , maxProportion - 1E-9 );
055
056                        if ( proportion > maxProportion + 1E-9 ) {
057                                currentMostPointed.clear();
058                                maxProportion = proportion;
059                        }
060                        if ( proportion >= maxProportion - 1E-9 ) {
061                                currentMostPointed.add( records );
062                        }
063                }
064
065                final TabuList tabu = new TabuList();
066                int c = 0;
067                int ignoredIndiv = 0;
068                int ignoredJoint = 0;
069                boolean loop = true;
070                while ( loop ) {
071                        loop = false;
072                        for ( final Collection<PlanRecord> records : currentMostPointed ) {
073                                if ( tabuFriends && tabu.addAndCheckTabu( records ) ) continue;
074
075                                boolean doneSomething = false;
076                                for ( final PlanRecord r : records ) {
077                                        if ( r.isPointed() ) continue;
078                                        // for each agent in the joint plan that does not point to the joint plan,
079                                        // we remove the pointed joint plan.
080                                        final Plan pointed = r.getAgent().getPointedPlan();
081                                        final Collection<PlanRecord> recordsInJointPlan = recordsPerJointPlan.getRecords( pointed );
082
083                                        // can happen if the initially pointed joint plan was removed for a friend,
084                                        // when ignoring tabu restrictions.
085                                        if ( recordsInJointPlan.size() == 1 ) {
086                                                ignoredIndiv++;
087                                                continue;
088                                        }
089                                        // test if greater than max proportion plus small tolerance, to avoid eg problems with proportion 0.5 with
090                                        // plans of size 2
091                                        if ( !tabuFriends && proportionPointedGreaterThan( recordsInJointPlan, maxProportion + 1E-9 ) ) {
092                                                ignoredJoint++;
093                                                continue;
094                                        }
095
096                                        for ( PlanRecord inPointed : recordsInJointPlan ) {
097                                                inPointed.setInfeasible();
098                                        }
099
100                                        final Collection<PlanRecord> removed = recordsPerJointPlan.removeJointPlan( pointed );
101                                        assert removed != null;
102                                        doneSomething = true;
103                                        // as long as we can "resolve" something for a plan, we continue...
104                                        loop = true;
105                                }
106                                if ( doneSomething ) c++;
107                        }
108                }
109                if ( log.isTraceEnabled() ) {
110                        log.trace( "Handled "+c+" of "+currentMostPointed.size()+" possibly conflicting joint plans" );
111                        log.trace( ignoredIndiv+" agents were not corrected because they pointed to an individual plan" );
112                        log.trace( ignoredJoint+" agents were not corrected because they pointed to joint plan with a pointing proportion greater than "+maxProportion );
113                }
114        }
115
116        private boolean proportionPointedGreaterThan(
117                        final Collection<PlanRecord> records,
118                        final double v ) {
119                final double minNPointed = v * records.size();
120
121                int nPoint = 0;
122                int remaining = records.size();
123                for ( final PlanRecord r : records ) {
124                        remaining--;
125                        if ( r.getAgent().getPointedPlan() == r.getPlan()  ) nPoint++;
126                        if ( nPoint >= minNPointed ) return true;
127                        if ( remaining + nPoint < minNPointed ) return false;
128                }
129
130                return false;
131        }
132
133        private double getProportionPointed(
134                        final Collection<PlanRecord> records,
135                        final double min ) {
136                int nPoint = 0;
137
138                final double minNPointed = min * records.size();
139                int remaining = records.size();
140
141                // if plan comes in here, it is not fully pointed
142                if ( minNPointed > remaining - 1 ) return Double.NEGATIVE_INFINITY;
143
144                for ( final PlanRecord r : records ) {
145                        remaining--;
146                        if ( r.getAgent().getPointedPlan() == r.getPlan()  ) nPoint++;
147
148                        // we know we will be lower than the minimum to find: abort
149                        if ( nPoint + remaining < minNPointed ) return Double.NEGATIVE_INFINITY;
150                }
151
152                return nPoint / ((double) records.size());
153        }
154
155        private class TabuList {
156                final Set<Id<Person>> tabuAgents = new HashSet<>();
157
158                public boolean addAndCheckTabu( final Iterable<PlanRecord> records ) {
159                        boolean tabu = areAltersTabu( records );
160
161                        for ( PlanRecord r :records ) {
162                                tabu = !tabuAgents.add( r.getPlan().getPerson().getId() ) || tabu;
163                        }
164
165                        return tabu;
166                }
167
168                // this is done for improving the number of pruned possible conflicts for very big groups.
169                private boolean areAltersTabu( final Iterable<PlanRecord> records ) {
170                        for ( PlanRecord r :records ) {
171                                final Set<Id<Person>> alters = socialNetwork.getAlters( r.getPlan().getPerson().getId() );
172                                for ( Id<Person> alter : alters ) {
173                                        if ( tabuAgents.contains( alter ) ) {
174                                                return true;
175                                        }
176                                }
177                        }
178                        return false;
179                }
180        }
181}
