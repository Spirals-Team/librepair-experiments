020package org.matsim.contrib.socnetsim.usage.replanning.removers;
021
022import org.matsim.api.core.v01.Scenario;
023
024import org.matsim.contrib.socnetsim.framework.replanning.removers.LexicographicForCompositionExtraPlanRemover;
025import org.matsim.contrib.socnetsim.usage.replanning.GroupReplanningConfigGroup;
026import org.matsim.contrib.socnetsim.framework.replanning.ExtraPlanRemover;
027import com.google.inject.Inject;
028import com.google.inject.Provider;
029
030/**
031 * @author thibautd
032 */
033public class LexicographicRemoverFactory implements Provider<ExtraPlanRemover> {
034
035        private final Scenario sc;
036
037        @Inject
038        public LexicographicRemoverFactory( final Scenario sc ) {
039                this.sc = sc;
040        }
041
042        @Override
043        public ExtraPlanRemover get() {
044                final GroupReplanningConfigGroup conf = (GroupReplanningConfigGroup)
045                                sc.getConfig().getModule(
046                                                GroupReplanningConfigGroup.GROUP_NAME );
047
048                return new LexicographicForCompositionExtraPlanRemover(
049                                conf.getMaxPlansPerComposition(),
050                                conf.getMaxPlansPerAgent() );
051        }
052}
