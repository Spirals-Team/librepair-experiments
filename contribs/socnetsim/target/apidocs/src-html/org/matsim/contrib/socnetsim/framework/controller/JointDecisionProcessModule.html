049/**
050 * Defines the basic elements for the matsim process with joint plans.
051 * Should be enough to run without crash, but not very useful as such.
052 *
053 * A useful module would first add this module, and then an "overriding"
054 * module defining more useful elements.
055 *
056 * @author thibautd
057 */
058public class JointDecisionProcessModule extends AbstractModule {
059
060        @Override
061        public void install() {
062                // process as such
063                bind(PlansReplanning.class).to( GroupReplanningListenner.class );
064                bind(PlansScoring.class).to( InternalizingPlansScoring.class );
065                addControlerListenerBinding().to( InternalizingPlansScoring.class );
066                addControlerListenerBinding().to( DumpJointDataAtEnd.class );
067
068                bind( InternalizationSettings.class ).to( ConfigBasedInternalizationSettings.class );
069
070                addControlerListenerBinding().to(JointPlansDumping.class);
071
072                addEventHandlerBinding().to(CourtesyEventsGenerator.class);
073
074                // consistency
075                addControlerListenerBinding().to(JointPlanCompositionMinimalityChecker.class);
076                addControlerListenerBinding().to(JointPlanSelectionConsistencyChecker.class);
077
078                // default elements
079                bind(GroupIdentifier.class).toInstance(
080                                // by default, no groups (results in individual replanning)
081                                new GroupIdentifier() {
082                                        @Override
083                                        public Collection<ReplanningGroup> identifyGroups(
084                                                        final Population population) {
085                                                return Collections.<ReplanningGroup>emptyList();
086                                        }
087                                });
088
089                bind(PlanRoutingAlgorithmFactory.class).toInstance(
090                                new PlanRoutingAlgorithmFactory() {
091                                        @Override
092                                        public PlanAlgorithm createPlanRoutingAlgorithm(
093                                                        final TripRouter tripRouter) {
094                                                return new PlanRouter(tripRouter);
095                                        }
096                                });
097
098                bind( PlanLinkIdentifier.class ).annotatedWith( PlanLinkIdentifier.Strong.class ).toInstance(new CompositePlanLinkIdentifier());
099                bind( PlanLinkIdentifier.class ).annotatedWith( PlanLinkIdentifier.Weak.class ).toInstance(new CompositePlanLinkIdentifier());
100                bind( IncompatiblePlansIdentifierFactory.class ).toInstance( new EmptyIncompatiblePlansIdentifierFactory() );
101
102                //addControlerListenerBinding().to( TripModeShares.class );
103                //final CompositeStageActivityTypes actTypesForAnalysis = new CompositeStageActivityTypes();
104                //actTypesForAnalysis.addActivityTypes(
105                //              controller.getRegistry().getTripRouterFactory().get().getStageActivityTypes() );
106                //actTypesForAnalysis.addActivityTypes( JointActingTypes.JOINT_STAGE_ACTS );
107                //controller.addControlerListener(
108                //              new TripModeShares(
109                //                      graphWriteInterval,
110                //                      controller.getControlerIO(),
111                //                      controller.getRegistry().getScenario(),
112                //                      new JointMainModeIdentifier( new MainModeIdentifierImpl() ),
113                //                      actTypesForAnalysis));
114
115                // For convenience
116                bind( JointPlans.class ).toProvider( new ScenarioElementProvider<JointPlans>( JointPlans.ELEMENT_NAME ) );
117                bind( SocialNetwork.class ).toProvider( new ScenarioElementProvider<SocialNetwork>( SocialNetwork.ELEMENT_NAME ) );
118        }
119}
