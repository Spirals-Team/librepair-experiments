091        public static OverlapScorerFactory getPersonOverlapScorerFactory(
092                        final Scenario scenario ) {
093                final ScoringFunctionConfigGroup scoringFunctionConf = (ScoringFunctionConfigGroup)
094                        scenario.getConfig().getModule( ScoringFunctionConfigGroup.GROUP_NAME );
095                switch ( scoringFunctionConf.getTogetherScoringForm() ) {
096                        case linear:
097                                return new OverlapScorerFactory() {
098                                                @Override
099                                                public BeingTogetherScoring.PersonOverlapScorer createScorer(final Id id) {
100                                                        return new BeingTogetherScoring.LinearOverlapScorer(
101                                                                        scoringFunctionConf.getMarginalUtilityOfBeingTogether_s() );
102                                                }
103                                        };
104                        case logarithmic:
105                                return new OverlapScorerFactory() {
106                                                @Override
107                                                public BeingTogetherScoring.PersonOverlapScorer createScorer(final Id id) {
108                                                        final Person person = (Person) scenario.getPopulation().getPersons().get( id );
109                                                        if ( person == null ) {
110                                                                // eg transit agent
111                                                                return new BeingTogetherScoring.LinearOverlapScorer( 0 );
112                                                        }
113                                                        final double typicalDuration =
114                                                                getTypicalDuration( 
115                                                                                scenario,
116                                                                                person,
117                                                                                scoringFunctionConf.getActivityTypeForContactInDesires() );
118                                                        final double zeroDuration = typicalDuration * Math.exp( -10.0 / typicalDuration );
119                                                        return new BeingTogetherScoring.LogOverlapScorer(
120                                                                        scoringFunctionConf.getMarginalUtilityOfBeingTogether_s(),
121                                                                        typicalDuration,
122                                                                        zeroDuration);
123                                                }
124                                        };
125                        default:
126                                throw new RuntimeException( ""+scoringFunctionConf.getTogetherScoringForm() );
127                }
128        }
129
130        public static double getTypicalDuration(
131                        final Scenario scenario,
132                        final Person person,
133                        final String type ) {
134                final Double typicalDuration =
135                                        (Double) scenario.getPopulation().getPersonAttributes().getAttribute(
136                                                person.getId().toString(),
137                                                "typicalDuration_"+type );
138
139                if ( typicalDuration != null ) return typicalDuration.doubleValue();
140
141                final ActivityParams params = scenario.getConfig().planCalcScore().getActivityParams( type );
142                
143                if ( params == null ) {
144                        //throw new RuntimeException( "could not find typical duration for Person "+person.getId()+" for type "+type );
145                        // not that nice, but needed for agents that might not have a preference. BeingTogetherScoring knows how to handle that
146                        return Double.NEGATIVE_INFINITY;
147                }
148                
149                return params.getTypicalDuration();
150        }

204        private <T extends Event & HasPersonId> void transmitEventToRelevantPersons( final T event ) {
205                final Id<Person> ego = event.getPersonId();
206                if ( !socialNetwork.getEgos().contains( ego ) ) return;
207                final Iterable<Id<Person>> ids = cat( ego , socialNetwork.getAlters( ego ) );
208                for ( Id<Person> id : ids ) {
209                        final Id finalId = id;
210                        final BeingTogetherScoring scoring =
211                                MapUtils.getArbitraryObject(
212                                                id,
213                                                scorings,
214                                                new MapUtils.Factory<BeingTogetherScoring>() {
215                                                        @Override
216                                                        public BeingTogetherScoring create() {
217                                                                return new BeingTogetherScoring(
218                                                                                facilities,
219                                                                                actTypeFilter,
220                                                                                modeFilter,
221                                                                                scorerFactory.createScorer(finalId),
222                                                                                finalId,
223                                                                                socialNetwork.getAlters( finalId ) );
224                                                        }
225                                                });
226                        scoring.handleEvent( event );
227                }
228        }
229
230        public Iterable<Id<Person>> cat(final Id<Person> ego, final Set<Id<Person>> alters) {
231                final Collection<Id<Person>> ids = new HashSet<>( alters );
232                ids.add( ego );
233                return ids;
234        }
235
236        @Override
237        public void notifyAfterMobsim(final AfterMobsimEvent event) {
238                for ( Map.Entry<Id, BeingTogetherScoring> e : scorings.entrySet() ) {
239                        final Id id = e.getKey();
240                        final BeingTogetherScoring scoring = e.getValue();
241
242                        events.processEvent(
243                                        new PersonMoneyEvent(Time.MIDNIGHT, id, scoring.getScore() / marginalUtilityOfMoney) );
244                }
245        }
