022import java.util.HashMap;
023import java.util.Map;
024import java.util.Set;

026import org.matsim.api.core.v01.Id;
027import org.matsim.api.core.v01.Scenario;
028import org.matsim.api.core.v01.events.ActivityEndEvent;
029import org.matsim.api.core.v01.events.ActivityStartEvent;
030import org.matsim.api.core.v01.events.handler.ActivityEndEventHandler;
031import org.matsim.api.core.v01.events.handler.ActivityStartEventHandler;
032import org.matsim.api.core.v01.population.Person;

039import com.google.inject.Inject;

041/**
042 * Creates "courtesy" events: agents say "hello" and "goodbye" to their
043 * social contacts located in the same facility as them.
044 *
045 * This is needed for scoring, because scoring functions cannot listen to the events of social contacts.
046 * @author thibautd
047 */
048public class CourtesyEventsGenerator implements ActivityStartEventHandler, ActivityEndEventHandler {
049
050        private final EventsManager events;
051        private final SocialNetwork socialNetwork;
052        private final Map< Id<ActivityFacility> , Map<String, Set< Id<Person> > > > personsAtFacility = new HashMap< >();
053
054        @Inject
055        public CourtesyEventsGenerator(
056                        final EventsManager events,
057                        final Scenario sc) {
058                this.events = events;
059                this.socialNetwork = (SocialNetwork) sc.getScenarioElement( SocialNetwork.ELEMENT_NAME );
060        }
061
062        public CourtesyEventsGenerator(
063                        final EventsManager events,
064                        final SocialNetwork socialNetwork) {
065                this.events = events;
066                this.socialNetwork = socialNetwork;
067        }
068
069        @Override
070        public void reset(int iteration) {
071                personsAtFacility.clear();
072        }
073
074        @Override
075        public void handleEvent(final ActivityStartEvent event) {
076                handleEvent(
077                                CourtesyEvent.Type.sayHelloEvent,
078                                event.getActType(),
079                                event.getPersonId(),
080                                event.getFacilityId(),
081                                event.getTime());
082        }
083
084        @Override
085        public void handleEvent(final ActivityEndEvent event) {
086                handleEvent(
087                                CourtesyEvent.Type.sayGoodbyeEvent,
088                                event.getActType(),
089                                event.getPersonId(),
090                                event.getFacilityId(),
091                                event.getTime() );
092        }
093
094        private void handleEvent(
095                        final CourtesyEvent.Type type,
096                        final String actType,
097                        final Id<Person> ego,
098                        final Id<ActivityFacility> facility,
099                        final double time) {
100                // TODO: handle wraparound (done improperly because we track people from their second act...)
101                final Set< Id<Person> > alters = socialNetwork.getAlters( ego );
102
103                switch ( type ) {
104                        case sayGoodbyeEvent:
105                                // avoid problems with wrap-around: do not say goodbye before being tracked.
106                                // this caused problems with agents having leisure at home.
107                                // solution would be to track the agents from the start of the simulation
108                                if ( !getPersonsAtFacilityForType( facility , actType ).remove( ego ) ) return;
109                                break;
110                        case sayHelloEvent:
111                                getPersonsAtFacilityForType( facility , actType ).add( ego );
112                                break;
113                        default:
114                                throw new RuntimeException( type+"?" );
115                }
116
117                for ( Id<Person> present : getPersonsAtFacilityForType( facility , actType ) ) {
118                        if ( alters.contains( present ) ) {
119                                events.processEvent(
120                                                new CourtesyEvent(
121                                                        time,
122                                                        actType,
123                                                        ego,
124                                                        present,
125                                                        type ) );
126                                events.processEvent(
127                                                new CourtesyEvent(
128                                                        time,
129                                                        actType,
130                                                        present,
131                                                        ego,
132                                                        type ) );
133                        }
134                }
135        }
136
137        private Set<Id<Person>> getPersonsAtFacilityForType( Id<ActivityFacility> facility, String actType ) {
138                return MapUtils.getSet(
139                                actType,
140                                MapUtils.getMap(
141                                                facility,
142                                                personsAtFacility ) );
143        }
144}
