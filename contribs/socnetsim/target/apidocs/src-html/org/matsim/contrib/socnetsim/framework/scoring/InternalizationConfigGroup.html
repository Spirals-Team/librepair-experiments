019package org.matsim.contrib.socnetsim.framework.scoring;
020
021import org.matsim.core.config.ReflectiveConfigGroup;
022
023import java.util.Map;
024
025/**
026 * @author thibautd
027 */
028public class InternalizationConfigGroup extends ReflectiveConfigGroup {
029
030    public static final String GROUP_NAME = "scoreInternalization";
031
032    private String internalizationSocialNetworkFile = null;
033    private double internalizationRatio = 0;
034
035    public InternalizationConfigGroup() {
036        super( GROUP_NAME );
037    }
038
039    @Override
040    public Map<String,String> getComments() {
041        final Map<String,String> map = super.getComments();
042        map.put( "internalizationSocialNetworkFile" ,
043                "path to the file defining the score internalization social network, if any. If null, the social network of the simulation will be used.");
044        map.put( "internalizationRatio" ,
045                "ratio by which to multiply the scores of social contacts before suming them with the ego's score. 0 means egoism, 1 average group score maximisation." );
046        return map;
047    }
048
049    @StringGetter( "internalizationSocialNetworkFile" )
050    public String getInternalizationSocialNetworkFile() {
051        return internalizationSocialNetworkFile;
052    }
053
054    @StringSetter( "internalizationSocialNetworkFile" )
055    public void setInternalizationSocialNetworkFile(final String internalizationSocialNetwork) {
056        this.internalizationSocialNetworkFile = internalizationSocialNetwork;
057    }
058
059    @StringGetter( "internalizationRatio" )
060    public double getInternalizationRatio() {
061        return internalizationRatio;
062    }
063
064    @StringSetter( "internalizationRatio" )
065    public void setInternalizationRatio(final double internalizationRatio) {
066        this.internalizationRatio = internalizationRatio;
067    }
068}
