020package org.matsim.contrib.socnetsim.jointtrips.router;
021
022import java.util.Collections;
023import java.util.List;
024
025import org.matsim.api.core.v01.Id;
026import org.matsim.api.core.v01.population.Leg;
027import org.matsim.api.core.v01.population.Person;
028import org.matsim.api.core.v01.population.PlanElement;
029import org.matsim.api.core.v01.population.PopulationFactory;
030import org.matsim.core.population.routes.NetworkRoute;
031import org.matsim.core.router.EmptyStageActivityTypes;
032import org.matsim.core.router.RoutingModule;
033import org.matsim.core.router.StageActivityTypes;
034import org.matsim.facilities.Facility;
035
036import org.matsim.contrib.socnetsim.jointtrips.population.DriverRoute;
037
038/**
039 * @author thibautd
040 */
041public class DriverRoutingModule implements RoutingModule {
042        private final RoutingModule carRoutingModule;
043        private final String mode;
044        private final PopulationFactory popFactory;
045
046        public DriverRoutingModule(
047                        final String mode,
048                        final PopulationFactory popFactory,
049                        final RoutingModule carRoutingModule) {
050                this.mode = mode;
051                this.popFactory = popFactory;
052                this.carRoutingModule = carRoutingModule;
053        }
054
055        @Override
056        public List<? extends PlanElement> calcRoute(
057                        final Facility fromFacility,
058                        final Facility toFacility, 
059                        final double departureTime,
060                        final Person person) {
061                List<? extends PlanElement> trip =
062                        carRoutingModule.calcRoute(
063                                        fromFacility,
064                                        toFacility, 
065                                        departureTime,
066                                        person);
067
068                if (trip.size() != 1) {
069                        throw new RuntimeException( "unexpected trip size for trip "+trip+" for mode "+mode );
070                }
071
072                Leg carLeg = (Leg) trip.get( 0 );
073                NetworkRoute netRoute = (NetworkRoute) carLeg.getRoute();
074
075                Leg leg = popFactory.createLeg( mode );
076                DriverRoute dRoute = new DriverRoute( netRoute , Collections.<Id<Person>>emptyList() );
077                leg.setRoute( dRoute );
078                leg.setDepartureTime( departureTime );
079                leg.setTravelTime( dRoute.getTravelTime() );
080
081                return Collections.singletonList( leg );
082        }
083
084        @Override
085        public StageActivityTypes getStageActivityTypes() {
086                return EmptyStageActivityTypes.INSTANCE;
087        }
088}
