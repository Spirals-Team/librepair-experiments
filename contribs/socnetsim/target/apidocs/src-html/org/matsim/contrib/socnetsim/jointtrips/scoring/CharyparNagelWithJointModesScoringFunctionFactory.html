020package org.matsim.contrib.socnetsim.jointtrips.scoring;
021
022import java.util.HashMap;
023import java.util.Map;
024
025import org.apache.log4j.Logger;
026import org.matsim.api.core.v01.Scenario;
027import org.matsim.api.core.v01.TransportMode;
028import org.matsim.api.core.v01.population.Person;
029import org.matsim.contrib.socnetsim.jointtrips.population.JointActingTypes;
030import org.matsim.contrib.socnetsim.jointtrips.scoring.ElementalCharyparNagelLegScoringFunction.LegScoringParameters;
031import org.matsim.core.router.StageActivityTypes;
032import org.matsim.core.scoring.ScoringFunction;
033import org.matsim.core.scoring.ScoringFunctionFactory;
034import org.matsim.core.scoring.SumScoringFunction;
035import org.matsim.core.scoring.functions.CharyparNagelActivityScoring;
036import org.matsim.core.scoring.functions.CharyparNagelAgentStuckScoring;
037import org.matsim.core.scoring.functions.CharyparNagelMoneyScoring;
038import org.matsim.core.scoring.functions.ScoringParameters;
039
040import com.google.inject.Inject;
041
042/**
043 * @author thibautd
044 */
045public class CharyparNagelWithJointModesScoringFunctionFactory implements ScoringFunctionFactory {
046        private static final Logger log =
047                Logger.getLogger(CharyparNagelWithJointModesScoringFunctionFactory.class);
048
049
050        private final StageActivityTypes blackList;
051        private final ScoringParameters params;
052        private final Map<String, LegScoringParameters> parametersPerMode = new HashMap<String, LegScoringParameters>();
053    private final Scenario scenario;
054
055        // /////////////////////////////////////////////////////////////////////////
056        // constructors
057        // /////////////////////////////////////////////////////////////////////////
058        @Inject
059    public CharyparNagelWithJointModesScoringFunctionFactory(
060                final Scenario sc ) {
061        this( JointActingTypes.JOINT_STAGE_ACTS , sc );
062    }
063
064    public CharyparNagelWithJointModesScoringFunctionFactory(
065                        final StageActivityTypes typesNotToScore,
066                        final Scenario scenario) {
067                log.warn( "initializing the \"CharyparNagel\" scoring function: this has no special parameters for joint modes!" );
068                this.params = new ScoringParameters.Builder(scenario.getConfig().planCalcScore(), scenario.getConfig().planCalcScore().getScoringParameters(null), scenario.getConfig().scenario()).build();
069                this.scenario = scenario;
070                this.blackList = typesNotToScore;
071
072                // standard modes
073                parametersPerMode.put(
074                                        TransportMode.car,
075                                        LegScoringParameters.createForCar(
076                                                params ));
077                parametersPerMode.put(
078                                        TransportMode.pt,
079                                        LegScoringParameters.createForPt(
080                                                params ));
081                parametersPerMode.put(
082                                        TransportMode.walk,
083                                        LegScoringParameters.createForWalk(
084                                                params ));
085                parametersPerMode.put(
086                                        TransportMode.bike,
087                                        LegScoringParameters.createForBike(
088                                                params ));
089                parametersPerMode.put(
090                                        TransportMode.transit_walk,
091                                        LegScoringParameters.createForWalk(
092                                                params ));
093
094                // joint modes
095                parametersPerMode.put(
096                                        JointActingTypes.DRIVER,
097                                        LegScoringParameters.createForCar(
098                                                params ));
099                parametersPerMode.put(
100                                        JointActingTypes.PASSENGER,
101                                        new LegScoringParameters(
102                                                params.modeParams.get(TransportMode.car).constant,
103                                                params.modeParams.get(TransportMode.car).marginalUtilityOfTraveling_s,
104                                                // passenger doesn't pay gasoline
105                                                0 ));
106        }
107
108        @Override
109        public ScoringFunction createNewScoringFunction(final Person person) {
110                final SumScoringFunction scoringFunctionAccumulator = new SumScoringFunction();
111
112                scoringFunctionAccumulator.addScoringFunction(
113                                new BlackListedActivityScoringFunction(
114                                        blackList,
115                                        new CharyparNagelActivityScoring(
116                                                params ) ) );
117
118                for ( Map.Entry<String, LegScoringParameters> entry : parametersPerMode.entrySet() ) {
119                        scoringFunctionAccumulator.addScoringFunction(
120                                        new ElementalCharyparNagelLegScoringFunction(
121                                                entry.getKey(),
122                                                entry.getValue(),
123                                                scenario.getNetwork()));
124                }
125
126                // other standard stuff
127                scoringFunctionAccumulator.addScoringFunction(
128                                new CharyparNagelMoneyScoring( params ));
129                scoringFunctionAccumulator.addScoringFunction(
130                                new CharyparNagelAgentStuckScoring( params ));
131
132                return scoringFunctionAccumulator;
133        }
134}
