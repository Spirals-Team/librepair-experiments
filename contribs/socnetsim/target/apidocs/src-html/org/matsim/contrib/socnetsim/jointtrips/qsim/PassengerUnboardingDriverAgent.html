060        public PassengerUnboardingDriverAgent(
061                        final MobsimAgent delegate,
062                        final QVehicleProvider vehicleProvider,
063                        final InternalInterface internalInterface) {
064                if ( delegate == null ) throw new IllegalArgumentException( "delegate cannot be null" );
065                this.delegate = (MobsimDriverAgent) delegate;
066                this.ptDelegate = delegate instanceof PTPassengerAgent ? (PTPassengerAgent) delegate : null;
067                this.planDelegate = (PlanAgent) delegate;
068                this.vehicleProvider = vehicleProvider;
069                this.internalInterface = internalInterface;
070        }

072        // /////////////////////////////////////////////////////////////////////////
073        // modified methods
074        // /////////////////////////////////////////////////////////////////////////
075        @Override
076        public void endLegAndComputeNextState(final double now) {
077                final String mode = delegate.getMode();
078
079                if ( mode == null ) {
080                        throw new IllegalStateException(
081                                        "delegate "+delegate+
082                                        " for agent "+delegate.getId()+
083                                        " returned a null mode. Probably in a wrong state: "
084                                        +delegate.getState()+" with current plan element "+getCurrentPlanElement() );
085                }
086
087                if ( mode.equals( JointActingTypes.DRIVER ) ) {
088                        if ( !passengersToBoard.isEmpty() ) {
089                                assert ((NetworkRoute) ((Leg) getCurrentPlanElement()).getRoute()).getLinkIds().isEmpty();
090                                boardPassengers();
091                                assert passengersToBoard.isEmpty();
092                        }
093
094                        final MobsimVehicle vehicle = vehicleProvider.getVehicle( delegate.getPlannedVehicleId() );
095                        final Id linkId = delegate.getCurrentLinkId();
096                        final Collection<PassengerAgent> passengersToUnboard = new ArrayList<PassengerAgent>();
097                        assert vehicle != null;
098                        for ( PassengerAgent p : vehicle.getPassengers() ) {
099                                if ( p.getDestinationLinkId().equals( linkId ) ) {
100                                        passengersToUnboard.add( p );
101                                }
102                        }
103
104                        final EventsManager events = ((QSim) internalInterface.getMobsim()).getEventsManager();
105                        for (PassengerAgent p : passengersToUnboard) {
106                                assert p != this;
107                                assert !p.getId().equals( getId() );
108
109                                vehicle.removePassenger( p );
110                                ((MobsimAgent) p).notifyArrivalOnLinkByNonNetworkMode( delegate.getCurrentLinkId() );
111                                ((MobsimAgent) p).endLegAndComputeNextState( now );
112                                events.processEvent(
113                                                new PersonLeavesVehicleEvent(now, p.getId(), vehicle.getId()));
114                                internalInterface.arrangeNextAgentState( (MobsimAgent) p );
115                        }
116                }
117
118                delegate.endLegAndComputeNextState( now );
119        }

124                delegate.notifyMoveOverNode(newLinkId);
125        }

127        private void boardPassengers() {
128                // using the vehicle provider might give access to a vehicle which is
129                // used by somebody else, if the sequence is incorrect.
130                // Using getVehicle() looks safer, as it should fail if the vehicle
131                // is somewhere else.
132                // It remains however problematic, as it is not clear whether agents will continue
133                // to be given the vehicle or not...
134                final MobsimVehicle vehicle = getVehicle();// vehicleProvider.getVehicle( delegate.getPlannedVehicleId() );
135                final EventsManager events = ((QSim) internalInterface.getMobsim()).getEventsManager();
136                for ( PassengerAgent passenger : passengersToBoard ) {
137                        assert passenger.getCurrentLinkId().equals( getCurrentLinkId() ) : passenger+" is at "+passenger.getCurrentLinkId()+" instead of "+getCurrentLinkId()+" for driver "+this;
138                        assert ((Leg) getCurrentPlanElement()).getMode().equals( JointActingTypes.DRIVER ) : getCurrentPlanElement();
139                        assert ((DriverRoute) ((Leg) getCurrentPlanElement()).getRoute()).getPassengersIds().contains( passenger.getId() ) :
140                                passenger+" not in "+((DriverRoute) ((Leg) getCurrentPlanElement()).getRoute()).getPassengersIds()+" for driver "+this;
141
142                        final boolean isAdded = vehicle.addPassenger( passenger );
143                        if ( !isAdded ) {
144                                // do not know how to handle that...
145                                throw new RuntimeException( passenger+" could not be added to vehicle "+vehicle );
146                        }
147
148                        events.processEvent(
149                                        new PersonEntersVehicleEvent(internalInterface.getMobsim().getSimTimer().getTimeOfDay(), passenger.getId(), vehicle.getId()));
150                }
151                passengersToBoard.clear();
152
153                final DriverRoute dr = (DriverRoute) ((Leg) getCurrentPlanElement()).getRoute();
154                assert vehicle.getPassengers().size() == dr.getPassengersIds().size() &&
155                        IdentifiableCollectionsUtils.containsAll( dr.getPassengersIds() , vehicle.getPassengers() ) :
156                                vehicle.getPassengers()+" != "+dr.getPassengersIds()+" for driver "+this;
157        }
158
159        public void addPassengerToBoard(final PassengerAgent passenger) {
160                passengersToBoard.add( passenger );
161        }
162
163        @Override
164        public void setVehicle(final MobsimVehicle veh) {
165                delegate.setVehicle(veh);
166                if ( !passengersToBoard.isEmpty() ) {
167                        boardPassengers();
168                        assert passengersToBoard.isEmpty();
169                }
170        }
