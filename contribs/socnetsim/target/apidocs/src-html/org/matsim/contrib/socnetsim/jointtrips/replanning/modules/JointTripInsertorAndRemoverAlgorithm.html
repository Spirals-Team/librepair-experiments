023import java.util.Collection;
024import java.util.Collections;
025import java.util.List;
026import java.util.Random;
027
028import org.matsim.api.core.v01.Id;
029import org.matsim.api.core.v01.population.Leg;
030import org.matsim.api.core.v01.population.Person;
031import org.matsim.api.core.v01.population.Plan;
032import org.matsim.api.core.v01.Scenario;
033import org.matsim.core.router.CompositeStageActivityTypes;
034import org.matsim.core.router.TripRouter;
035import org.matsim.core.router.TripStructureUtils;
036import org.matsim.core.router.TripStructureUtils.Trip;
037
038import org.matsim.contrib.socnetsim.framework.cliques.config.JointTripInsertorConfigGroup;
039import org.matsim.contrib.socnetsim.jointtrips.population.JointActingTypes;
040import org.matsim.contrib.socnetsim.framework.population.JointPlan;

042import org.matsim.contrib.socnetsim.framework.replanning.GenericPlanAlgorithm;

044/**
045 * @author thibautd
046 */
047public class JointTripInsertorAndRemoverAlgorithm implements GenericPlanAlgorithm<JointPlan> {
048        private final TripRouter tripRouter;
049        private final Random random;
050        private final JointTripInsertorAlgorithm insertor;
051        private final JointTripRemoverAlgorithm remover;
052        private final boolean iterative;
053
054        public JointTripInsertorAndRemoverAlgorithm(
055                        final Scenario scenario,
056                        final TripRouter tripRouter,
057                        final Random random,
058                        final boolean iterative) {
059                this.tripRouter = tripRouter;
060                this.random = random;
061                this.insertor = new JointTripInsertorAlgorithm(
062                                random,
063                                (SocialNetwork) scenario.getScenarioElement( SocialNetwork.ELEMENT_NAME ),
064                                (JointTripInsertorConfigGroup) scenario.getConfig().getModule( JointTripInsertorConfigGroup.GROUP_NAME ),
065                                tripRouter);
066                this.remover = new JointTripRemoverAlgorithm(
067                                random,
068                                tripRouter.getStageActivityTypes(),
069                                tripRouter.getMainModeIdentifier());
070                this.iterative = iterative;
071        }
072
073        @Override
074        public void run(final JointPlan plan) {
075                final List<Id<Person>> agentsToIgnore = new ArrayList< >();
076
077                do {
078                        final ActedUponInformation actedUpon =
079                                        random.nextDouble() < getProbRemoval( plan , agentsToIgnore ) ?
080                                                remover.run( plan , agentsToIgnore ) :
081                                                insertor.run( plan , agentsToIgnore );
082
083                        if (actedUpon == null) return;
084                        assert !agentsToIgnore.contains( actedUpon.getDriverId() ) : actedUpon+" <- "+agentsToIgnore;
085                        agentsToIgnore.add( actedUpon.getDriverId() );
086
087                        assert !agentsToIgnore.contains( actedUpon.getPassengerId() ) : actedUpon+" <- "+agentsToIgnore;
088                        agentsToIgnore.add( actedUpon.getPassengerId() );
089                } while ( iterative );
090        }
091
092        private double getProbRemoval(
093                        final JointPlan plan,
094                        final List<Id<Person>> agentsToIgnore) {
095                int countPassengers = 0;
096                int countEgoists = 0;
097
098                final CompositeStageActivityTypes stageTypes = new CompositeStageActivityTypes();
099                stageTypes.addActivityTypes( tripRouter.getStageActivityTypes() );
100                stageTypes.addActivityTypes( JointActingTypes.JOINT_STAGE_ACTS );
101
102                for (Plan indivPlan : plan.getIndividualPlans().values()) {
103                        if ( agentsToIgnore.contains( indivPlan.getPerson().getId() ) ) continue;
104
105                        // parse trips, and count "egoists" (non-driver non-passenger) and
106                        // passengers. Some care is needed: joint trips are not identified as
107                        // trips by the router!
108                        for ( Trip trip : TripStructureUtils.getTrips( indivPlan , stageTypes ) ) {
109                                if ( tripContainsOneOfThoseModes( trip , Collections.singleton( JointActingTypes.PASSENGER ) ) ) countPassengers++;
110                                if ( !tripContainsOneOfThoseModes( trip , JointActingTypes.JOINT_MODES ) ) countEgoists++;
111                        }
112                }
113
114                return ((double) countPassengers) / (countEgoists + countPassengers);
115        }
116
117        private static boolean tripContainsOneOfThoseModes(final Trip trip, final Collection<String> modes) {
118                for ( Leg leg : trip.getLegsOnly() ) {
119                        if ( modes.contains( leg.getMode() ) ) return true;
120                }
121                return false;
122        }
123}
