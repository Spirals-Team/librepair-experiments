020package org.matsim.contrib.socnetsim.sharedvehicles.replanning;
021
022import java.util.ArrayList;
023import java.util.Collection;
024import java.util.Collections;
025import java.util.List;
026import java.util.Random;
027
028import org.matsim.api.core.v01.Id;
029import org.matsim.api.core.v01.population.Leg;
030import org.matsim.api.core.v01.population.Plan;
031import org.matsim.core.population.algorithms.PlanAlgorithm;
032import org.matsim.core.population.routes.NetworkRoute;
033import org.matsim.core.router.TripRouter;
034import org.matsim.core.router.TripStructureUtils;
035import org.matsim.core.router.TripStructureUtils.Subtour;
036import org.matsim.core.router.TripStructureUtils.Trip;
037import org.matsim.contrib.socnetsim.sharedvehicles.VehicleRessources;
038
039/**
040 * Allocates a random vehicle to a random tour ( "root subtour" ) containing a given
041 * <u>main</u> mode. The vehicle is allocated to all NetworkRoutes in the trips with
042 * the given main mode in the choosen tour.
043 * @author thibautd
044 */
045public class AllocateVehicleToSubtourAlgorithm implements PlanAlgorithm {
046        private final String mode;
047        private final VehicleRessources vehicleRessources;
048        private final TripRouter tripRouter;
049        private final Random random;
050
051        public AllocateVehicleToSubtourAlgorithm(
052                        final Random random,
053                        final String mode,
054                        final TripRouter router,
055                        final VehicleRessources vehicleRessources) {
056                this.random = random;
057                this.mode = mode;
058                this.tripRouter = router;
059                this.vehicleRessources = vehicleRessources;
060        }
061
062        @Override
063        public void run(final Plan plan) {
064                final List<Id> vehs = new ArrayList<Id>(
065                        vehicleRessources.identifyVehiclesUsableForAgent(
066                                        plan.getPerson().getId() ) );
067                // make sure order is deterministic
068                Collections.sort( vehs );
069
070                final List<Subtour> rootSubtours = getRootSubtoursWithMode( plan );
071                if ( rootSubtours.isEmpty() ) return;
072
073                final Subtour toHandle = rootSubtours.get( random.nextInt( rootSubtours.size() ) );
074                final Id veh = vehs.get( random.nextInt( vehs.size() ) );
075
076                for ( Trip t : toHandle.getTrips() ) {
077                        if ( !identifyMainMode( t ).equals( mode ) ) continue;
078                        for ( Leg l : t.getLegsOnly() ) {
079                                if ( !(l.getRoute() instanceof NetworkRoute) ) continue;
080                                final NetworkRoute route = (NetworkRoute) l.getRoute();
081                                route.setVehicleId( veh );
082                        }
083                }
084        }
085
086        private String identifyMainMode(final Trip t) {
087                return tripRouter.getMainModeIdentifier().identifyMainMode( t.getTripElements() );
088        }
089
090        private List<Subtour> getRootSubtoursWithMode(final Plan plan) {
091                final Collection<Subtour> allSubtours = TripStructureUtils.getSubtours( plan , tripRouter.getStageActivityTypes() );
092                final List<Subtour> roots = new ArrayList<Subtour>();
093
094                for ( Subtour s : allSubtours ) {
095                        if ( s.getParent() != null ) continue;
096                        if ( !containsMode( s ) ) continue;
097                        roots.add( s );
098                }
099
100                return roots;
101        }
102
103        private boolean containsMode( final Subtour s ) {
104                for ( Trip t : s.getTrips() ) {
105                        if ( tripRouter.getMainModeIdentifier().identifyMainMode(
106                                                t.getTripElements() ).equals( mode ) ) {
107                                return true;
108                        }
109                }
110                return false;
111        }
112}
