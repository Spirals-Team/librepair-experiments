020package org.matsim.contrib.socnetsim.jointactivities.replanning;
021
022import java.util.PriorityQueue;
023import java.util.Queue;
024
025import org.apache.log4j.Logger;
026import org.matsim.api.core.v01.Id;
027import org.matsim.api.core.v01.population.Activity;
028import org.matsim.api.core.v01.population.Plan;
029import org.matsim.core.router.CompositeStageActivityTypes;
030import org.matsim.core.router.StageActivityTypes;
031import org.matsim.core.router.StageActivityTypesImpl;
032import org.matsim.core.router.TripStructureUtils;
033import org.matsim.core.utils.misc.Time;
034import org.matsim.pt.PtConstants;
035
036import org.matsim.contrib.socnetsim.jointtrips.population.JointActingTypes;
037import org.matsim.contrib.socnetsim.framework.replanning.modules.PlanLinkIdentifier;
038
039/**
040 * Links plans with "joinable" activities, that is,
041 * plans of social contacts with activities of a given type
042 * at the same location.
043 * Ideally, time overlap should be included.
044 * @author thibautd
045 */
046public class JoinableActivitiesPlanLinkIdentifier implements PlanLinkIdentifier {
047        private static final Logger log =
048                Logger.getLogger(JoinableActivitiesPlanLinkIdentifier.class);
049
050        // XXX should be passed from outside, but not available at construction
051        private final StageActivityTypes stages =
052                        new CompositeStageActivityTypes(
053                                        JointActingTypes.JOINT_STAGE_ACTS,
054                                        new StageActivityTypesImpl( PtConstants.TRANSIT_ACTIVITY_TYPE ) );
055        private final String type;
056
057        public JoinableActivitiesPlanLinkIdentifier(
058                        final String type) {
059                this.type = type;
060        }
061
062        @Override
063        public boolean areLinked(
064                        final Plan p1,
065                        final Plan p2) {
066                final Queue<LocationEvent> events = new PriorityQueue<LocationEvent>( p1.getPlanElements().size() + p2.getPlanElements().size() );
067
068                fillEvents( p1 , events );
069                fillEvents( p2 , events );
070
071                final Id pers1 = p1.getPerson().getId();
072                Id loc1 = null;
073                String type1 = null;
074                Id loc2 = null;
075                String type2 = null;
076
077                while ( !events.isEmpty() ) {
078                        final LocationEvent event = events.remove();
079
080                        if ( log.isTraceEnabled() ) {
081                                log.trace( "got event "+event+" from queue" );
082                        }
083
084                        if ( pers1.equals( event.getPersonId() ) ) {
085                                loc1 = event.getLocationId();
086                                type1 = event.getActType();
087                        }
088                        else {
089                                loc2 = event.getLocationId();
090                                type2 = event.getActType();
091                        }
092
093                        if ( loc1 != null && loc2 != null &&
094                                        loc1.equals( loc2 ) &&
095                                        type1.equals( type ) && type2.equals( type ) ) {
096                                return true;
097                        }
098                }
099
100                return false;
101        }
102
103        private void fillEvents(
104                        final Plan plan,
105                        final Queue<LocationEvent> events) {
106                final Id personId = plan.getPerson().getId();
107                double lastEnd = 0;
108                int ind = 0;
109                for ( Activity act : TripStructureUtils.getActivities( plan , stages ) ) {
110                        final Id loc = act.getFacilityId();
111
112                        final LocationEvent event =
113                                new LocationEvent(
114                                                ind++,
115                                                personId,
116                                                act.getType(),
117                                                loc,
118                                                lastEnd );
119
120                        // correct times if inconsistent
121                        lastEnd = Math.max(
122                                lastEnd,
123                                act.getEndTime() != Time.UNDEFINED_TIME ?
124                                        act.getEndTime() :
125                                        lastEnd + act.getMaximumDuration() );
126
127                        if ( log.isTraceEnabled() ) {
128                                log.trace( "add event "+event+" to queue" );
129                        }
130
131                        events.add( event );
132                }
133        }
134
135        private static class LocationEvent implements Comparable<LocationEvent> {
136                private final String type;
137                private final Id locId;
138                private final Id personId;
139                private final double startTime;
140                private int index;
141
142                public LocationEvent(
143                                final int index,
144                                final Id personId,
145                                final String type,
146                                final Id locId,
147                                final double startTime ) {
148                        if ( locId == null ) throw new NullPointerException( "null location for person "+personId+" activity type "+type );
149                        this.index = index;
150                        this.type = type;
151                        this.locId = locId;
152                        this.personId = personId;
153                        this.startTime = startTime;
154                }
155
156                public Id getLocationId() { return locId; }
157                public Id getPersonId() { return personId; }
158                public String getActType() { return type; }
159
160                @Override
161                public int compareTo(final LocationEvent o) {
162                        final int comp = Double.compare( startTime , o.startTime );
163                        if ( comp != 0 ) return comp;
164
165                        // order as in plan.
166                        if ( personId.equals( o.personId ) ) return index - o.index;
167
168                        // ugly, but could not find a better way to enforce consistency
169                        return personId.compareTo( o.personId );
170                }
171
172                @Override
173                public String toString() {
174                        return "[person="+personId+
175                                "; type="+type+
176                                "; location="+locId+
177                                "; startTime="+startTime+"]";
178                }
179        }
180}
