051/**
052 * @author thibautd
053 */
054public class CountTripsPerTypeOfOriginAndDestination {
055        private static final MainModeIdentifier MODE_IDENTIFIER = 
056                new MainModeIdentifier() {
057                        final MainModeIdentifier delegate = new JointMainModeIdentifier( new MainModeIdentifierImpl() );
058
059                        @Override
060                        public String identifyMainMode(final List<? extends PlanElement> tripElements) {
061                                if ( tripElements.size() == 1 &&
062                                                ((Leg) tripElements.get( 0 )).getMode().equals( TransportMode.transit_walk ) ) {
063                                        return TransportMode.walk;
064                                }
065                                return delegate.identifyMainMode( tripElements );
066                        }
067                };
068        private static final StageActivityTypes STAGES =
069                new StageActivityTypesImpl(
070                                Arrays.asList(
071                                        PtConstants.TRANSIT_ACTIVITY_TYPE,
072                                        JointActingTypes.INTERACTION ) );
073
074
075        public static void main(final String[] args) throws IOException {
076                final String inPlansFile = args[ 0 ];
077                final String outFile = args[ 1 ];
078
079                final Population population = readPopulation( inPlansFile );
080                final Iterable<Record> counts = countTrips( population );
081                writeCounts( counts , outFile );
082        }
083
084        private static void writeCounts(
085                        final Iterable<Record> counts,
086                        final String outFile) throws IOException {
087                final BufferedWriter writer = IOUtils.getBufferedWriter( outFile );
088                writer.write( "origType\tdestType\tmode\tcount" );
089                for ( Record r : counts ) {
090                        writer.newLine();
091                        writer.write( r.originType+"\t"+
092                                        r.destinationType+"\t"+
093                                        r.mode+"\t"+
094                                        r.count );
095                }
096                writer.close();
097        }
098
099        private static Iterable<Record> countTrips(final Population population) {
100                final QueryableSet records = new QueryableSet();
101                for ( Person person : population.getPersons().values() ) {
102                        final Plan plan = person.getSelectedPlan();
103                        for ( Trip trip : TripStructureUtils.getTrips( plan , STAGES ) ) {
104                                final Record r = records.getOrAddIfNotThere(
105                                                new Record(
106                                                        trip.getOriginActivity().getType(),
107                                                        trip.getDestinationActivity().getType(),
108                                                        MODE_IDENTIFIER.identifyMainMode( trip.getTripElements() ) ) );
109                                r.count++;
110                        }
111                }
112                return records;
113        }
114
115        private static Population readPopulation(final String inPlansFile) {
116                final Scenario s = ScenarioUtils.createScenario( ConfigUtils.createConfig() );
117                new PopulationReader( s ).readFile( inPlansFile );
118                return s.getPopulation();
119        }
120
121        private static class QueryableSet implements Iterable<Record> {
122                private final HashMap<Record, Record> map = new HashMap<Record, Record>();
123
124                public Record getOrAddIfNotThere( final Record record ) {
125                        final Record alreadyThere = map.get( record );
126                        if ( alreadyThere != null ) return alreadyThere;
127
128                        map.put( record , record );
129                        return record;
130                }
131
132                @Override
133                public Iterator<Record> iterator() {
134                        return map.keySet().iterator();
135                }
136        }
137
138        private static class Record {
139                final String originType, destinationType, mode;
140                int count = 0;
141
142                public Record(final String ot, final String dt, final String mode) {
143                        this.originType = ot;
144                        this.destinationType = dt;
145                        this.mode = mode;
146                }
147
148                @Override
149                public boolean equals(final Object o) {
150                        return o instanceof Record &&
151                                originType.equals( ((Record) o).originType ) &&
152                                destinationType.equals( ((Record) o).destinationType ) &&
153                                mode.equals( ((Record) o).mode );
154                }
155
156                @Override
157                public int hashCode() {
158                        return originType.hashCode() + destinationType.hashCode() + mode.hashCode();
159                }
160        }
161}
