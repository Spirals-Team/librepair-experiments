020package org.matsim.contrib.socnetsim.usage.analysis.scripts;
021
022import java.io.BufferedWriter;
023import java.io.IOException;
024import java.util.List;
025
026import org.matsim.api.core.v01.Scenario;
027import org.matsim.api.core.v01.population.Leg;
028import org.matsim.api.core.v01.population.Person;
029import org.matsim.core.config.ConfigUtils;
030import org.matsim.core.population.io.PopulationReader;
031import org.matsim.core.router.EmptyStageActivityTypes;
032import org.matsim.core.router.TripStructureUtils;
033import org.matsim.core.router.TripStructureUtils.Trip;
034import org.matsim.core.scenario.ScenarioUtils;
035import org.matsim.core.utils.io.IOUtils;
036import org.matsim.core.utils.io.UncheckedIOException;
037
038import org.matsim.contrib.socnetsim.jointtrips.population.JointActingTypes;
039
040/**
041 * @author thibautd
042 */
043public class CountNSelectedJointTripsPerAgent {
044        public static void main(final String[] args) {
045                final String popFile = args[ 0 ];
046                final String outFile = args[ 1 ];
047
048                final Scenario sc = ScenarioUtils.createScenario( ConfigUtils.createConfig() );
049                new PopulationReader( sc ).readFile( popFile );
050
051                final BufferedWriter writer = IOUtils.getBufferedWriter( outFile );
052
053                try {
054                        writer.write( "id\tnPassenger\tnDriver\tnJoint" );
055
056                        for (Person p : sc.getPopulation().getPersons().values()) {
057                                final List<Trip> trips =
058                                        TripStructureUtils.getTrips(
059                                                        p.getSelectedPlan(),
060                                                        EmptyStageActivityTypes.INSTANCE );
061
062                                int countPassenger = 0;
063                                int countDriver = 0;
064                                for (Trip trip : trips) {
065                                        final boolean isPassenger = containsMode( trip , JointActingTypes.PASSENGER );
066                                        final boolean isDriver = containsMode( trip , JointActingTypes.DRIVER );
067                                        assert !(isPassenger && isDriver);
068                                        if (isPassenger) countPassenger++;
069                                        if (isDriver) countDriver++;
070                                }
071
072                                writer.newLine();
073                                writer.write( p.getId()+"\t"+
074                                                countPassenger+"\t"+
075                                                countDriver+"\t"+
076                                                (countPassenger + countDriver) );
077                        }
078
079                        writer.close();
080                } catch (IOException e) {
081                        throw new UncheckedIOException( e );
082                }
083        }
084
085        private static boolean containsMode(
086                        final Trip trip,
087                        final String mode) {
088                for (Leg leg : trip.getLegsOnly()) {
089                        if (leg.getMode().equals( mode )) return true;
090                }
091                return false;
092        }
093}
