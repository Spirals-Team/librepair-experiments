040        public interface GroupingParameters {
041                double getTieActivationProbability();
042                double getJointPlanBreakingProbability();
043                double getMaxGroupSize();
044        }

046        public static Collection<Collection<Plan>> randomlyGroup(
047                        final GroupingParameters params,
048                        final Random random,
049                        final GroupPlans groupPlans,
050                        final SocialNetwork socialNetwork) {
051                final Map<Id<Person>, Plan> planPerPerson = new LinkedHashMap<>();
052                for ( Plan p : groupPlans.getAllIndividualPlans() ) {
053                        planPerPerson.put( p.getPerson().getId() , p );
054                }
055
056                final Map<Id, Set<Id>> jpTies = getJointPlanLinks( groupPlans );
057                final Map<Id<Person>, Set<Id<Person>>> subnet =
058                        SocialNetworkUtils.getSubnetwork(
059                                        socialNetwork,
060                                        planPerPerson.keySet() );
061
062                assert planPerPerson.keySet().containsAll( jpTies.keySet() ) : planPerPerson+" "+jpTies;
063                assert planPerPerson.keySet().equals( subnet.keySet() ) : planPerPerson +" != "+ subnet;
064
065                final Collection<Collection<Plan>> groups = new ArrayList<Collection<Plan>>();
066                while ( !subnet.isEmpty() ) {
067                        final Set<Id> group =
068                                        getRandomGroup(
069                                                        params,
070                                                        random,
071                                                        subnet,
072                                                        jpTies );
073
074                        assert planPerPerson.keySet().containsAll( group ) : planPerPerson +" - "+group+" - "+groups;
075
076                        final Collection<Plan> plans = new ArrayList<Plan>();
077                        for ( Id id : group ) {
078                                plans.add( planPerPerson.remove( id ) );
079                        }
080
081                        assert !CollectionUtils.intersects( planPerPerson.keySet() , group ) : planPerPerson+" intersect "+group;
082                        assert planPerPerson.keySet().containsAll( jpTies.keySet() ) : planPerPerson+" "+jpTies;
083                        assert planPerPerson.keySet().equals( subnet.keySet() ) : planPerPerson+"  "+subnet;
084
085                        groups.add( plans );
086                }
087
088                return groups;
089        }
090
091        public static Map<Id, Set<Id>> getJointPlanLinks(final GroupPlans groupPlans) {
092                final Map<Id, Set<Id>> links = new LinkedHashMap<Id, Set<Id>>();
093
094                for ( JointPlan jp : groupPlans.getJointPlans() ) {
095                        for ( Id id : jp.getIndividualPlans().keySet() ) {
096                                final Set<Id> alters = new LinkedHashSet<Id>( jp.getIndividualPlans().keySet() );
097                                alters.remove( id );
098                                links.put( id , alters );
099                        }
100                }
101
102                return links;
103        }

105        public static Collection<ReplanningGroup> randomlyGroupPersons(
106                        final GroupingParameters params,
107                        final Random random,
108                        final Population population,
109                        final JointPlans jointPlans,
110                        final SocialNetwork socialNetwork) {
111
112                final Map<Id, Set<Id>> jpTies = getJointPlanNetwork( population , jointPlans );
113                final Map<Id<Person>, Set<Id<Person>>> netmap = new LinkedHashMap<>( socialNetwork.getMapRepresentation() );
114
115                final Collection<ReplanningGroup> groups = new ArrayList<ReplanningGroup>();
116                while ( !netmap.isEmpty() ) {
117                        final Set<Id> ids =
118                                        getRandomGroup(
119                                                        params,
120                                                        random,
121                                                        netmap,
122                                                        jpTies );
123
124                        final ReplanningGroup group = new ReplanningGroup();
125                        groups.add( group );
126                        for ( Id id : ids ) group.addPerson( population.getPersons().get( id ) );
127                }
128
129                return groups;
130        }
131
132        public static Map<Id, Set<Id>> getJointPlanNetwork(
133                        final Population population,
134                        final JointPlans jointPlans ) {
135                final Map<Id, Set<Id>> links = new LinkedHashMap<Id, Set<Id>>();
136
137                for ( Person person : population.getPersons().values() ) {
138                        final Set<Id> alters = new LinkedHashSet<Id>();
139                        links.put( person.getId() , alters );
140
141                        for ( Plan plan : person.getPlans() ) {
142                                final JointPlan jp = jointPlans.getJointPlan( plan );
143                                if ( jp == null ) continue;
144                                alters.addAll( jp.getIndividualPlans().keySet() );
145                        }
146
147                        alters.remove( person.getId() );
148                }
149
150                return links;
151        }

153        private static Set<Id> getRandomGroup(
154                        final GroupingParameters params,
155                        final Random random,
156                        final Map<Id<Person>, Set<Id<Person>>> netmap,
157                        final Map<Id, Set<Id>> jpTies) {
158                assert netmap.keySet().containsAll( jpTies.keySet() ) : netmap+" != "+jpTies;
159                final Set<Id> group = new LinkedHashSet<Id>();
160
161                final Queue<Id> egoStack = Collections.asLifoQueue( new ArrayDeque<Id>( netmap.size() ) );
162                egoStack.add( CollectionUtils.getElement( 0 , netmap.keySet() ) );
163
164                while ( !egoStack.isEmpty() /*&& group.size() < params.getMaxGroupSize()*/ ) {
165                        final Id ego = egoStack.remove();
166                        final Set<Id<Person>> alters = netmap.remove( ego );
167                        final Set<Id> jpAlters = jpTies.remove( ego );
168
169                        if ( alters == null ) continue;
170                        group.add( ego );
171
172                        if ( jpAlters != null && random.nextDouble() >= params.getJointPlanBreakingProbability() ) {
173                                // keep jp
174                                for ( Id alter : jpAlters ) {
175                                        alters.remove( alter );
176                                        egoStack.add(alter);
177                                        // already add, so that joint plan not broken by early abort
178                                        // group.add( alter );
179                                }
180                        }
181
182                        for ( Id alter : alters ) {
183                                if ( random.nextDouble() < params.getTieActivationProbability() ) {
184                                        egoStack.add( alter );
185                                }
186                        }
187                }
188
189                for ( Id<Person> p : group )  {
190                        netmap.remove( p );
191                        jpTies.remove( p );
192                }
193
194                assert netmap.keySet().containsAll( jpTies.keySet() ) : netmap+" != "+jpTies;
195
196                return group;
197        }
