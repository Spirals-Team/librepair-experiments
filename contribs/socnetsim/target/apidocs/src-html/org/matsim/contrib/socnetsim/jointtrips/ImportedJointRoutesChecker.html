020package org.matsim.contrib.socnetsim.jointtrips;
021
022import java.util.Iterator;
023import java.util.List;
024
025import org.matsim.api.core.v01.population.Activity;
026import org.matsim.api.core.v01.population.Leg;
027import org.matsim.api.core.v01.population.Person;
028import org.matsim.api.core.v01.population.Plan;
029import org.matsim.api.core.v01.population.PlanElement;
030import org.matsim.core.population.algorithms.PersonAlgorithm;
031import org.matsim.core.population.algorithms.PlanAlgorithm;
032import org.matsim.core.router.ActivityWrapperFacility;
033import org.matsim.core.router.TripRouter;
034import org.matsim.core.utils.misc.Time;
035import org.matsim.contrib.socnetsim.jointtrips.population.DriverRoute;
036
037/**
038 * Checks driver routes, in case only passengers were provided in the plan file
039 * @author thibautd
040 */
041public class ImportedJointRoutesChecker implements PlanAlgorithm, PersonAlgorithm {
042        private final TripRouter router;
043
044        public ImportedJointRoutesChecker(final TripRouter router) {
045                this.router = router;
046        }
047
048        @Override
049        public void run(final Person person) {
050                for ( Plan plan : person.getPlans() ) {
051                        run( plan );
052                }
053        }
054
055        @Override
056        public void run(final Plan plan) {
057                Iterator<PlanElement> pes = plan.getPlanElements().iterator();
058
059                Activity origin = (Activity) pes.next();
060                double now = 0;
061                while (pes.hasNext()) {
062                        // FIXME: relies on the assumption of strict alternance leg/act
063                        Leg l = (Leg) pes.next();
064                        Activity dest = (Activity) pes.next();
065
066                        now = updateTime( now , origin );
067                        if (l.getRoute() != null && l.getRoute() instanceof DriverRoute) {
068                                List<? extends PlanElement> trip =
069                                        router.calcRoute(
070                                                        l.getMode(),
071                                                        new ActivityWrapperFacility( origin ),
072                                                        new ActivityWrapperFacility( dest ),
073                                                        now,
074                                                        plan.getPerson());
075
076                                if (trip.size() != 1) {
077                                        throw new RuntimeException( "unexpected trip length "+trip.size()+" for "+trip+" for mode "+l.getMode());
078                                }
079
080                                DriverRoute newRoute = (DriverRoute) ((Leg) trip.get( 0 )).getRoute();
081                                newRoute.setPassengerIds( ((DriverRoute) l.getRoute()).getPassengersIds() );
082                                l.setRoute( newRoute );
083                        }
084
085                        now = updateTime( now , l );
086                        origin = dest;
087                }
088        }
089
090        private static double updateTime(
091                        final double currTime,
092                        final Activity act) {
093                double e = act.getEndTime();
094                double d = act.getMaximumDuration();
095                return e != Time.UNDEFINED_TIME ? e :
096                        currTime + ( d != Time.UNDEFINED_TIME ? d : 0 );
097        }
098
099        private static double updateTime(
100                        final double currTime,
101                        final Leg leg) {
102                double tt = leg.getTravelTime();
103                return tt != Time.UNDEFINED_TIME ? currTime + tt : currTime;
104        }
105}
