020package org.matsim.contrib.socnetsim.usage.replanning.strategies;
021
022import org.matsim.api.core.v01.Scenario;
023import org.matsim.core.router.TripRouter;
024
025import org.matsim.contrib.socnetsim.framework.PlanRoutingAlgorithmFactory;
026import org.matsim.contrib.socnetsim.framework.population.JointPlans;
027import org.matsim.contrib.socnetsim.framework.replanning.GroupPlanStrategy;
028import org.matsim.contrib.socnetsim.usage.replanning.GroupPlanStrategyFactoryUtils;
029import org.matsim.contrib.socnetsim.framework.replanning.modules.PlanLinkIdentifier;
030import org.matsim.contrib.socnetsim.framework.replanning.modules.PlanLinkIdentifier.Strong;
031import org.matsim.contrib.socnetsim.jointactivities.replanning.modules.randomlocationchoice.RandomJointLocationChoiceModule;
032import org.matsim.contrib.socnetsim.sharedvehicles.VehicleRessources;
033
034import com.google.inject.Inject;
035import com.google.inject.Provider;
036
037/**
038 * @author thibautd
039 */
040public class RandomJointLocationChoiceStrategyFactory extends AbstractConfigurableSelectionStrategy {
041
042        private final Scenario sc;
043        private final PlanRoutingAlgorithmFactory planRoutingAlgorithmFactory;
044        private final Provider<TripRouter> tripRouterFactory;
045        private final PlanLinkIdentifier planLinkIdentifier;
046
047        @Inject
048        public RandomJointLocationChoiceStrategyFactory( Scenario sc , PlanRoutingAlgorithmFactory planRoutingAlgorithmFactory ,
049                        Provider<TripRouter> tripRouterFactory , @Strong PlanLinkIdentifier planLinkIdentifier ) {
050                this.sc = sc;
051                this.planRoutingAlgorithmFactory = planRoutingAlgorithmFactory;
052                this.tripRouterFactory = tripRouterFactory;
053                this.planLinkIdentifier = planLinkIdentifier;
054        }
055
056
057        @Override
058        public GroupPlanStrategy get() {
059                final GroupPlanStrategy strategy = instantiateStrategy( sc.getConfig() );
060
061                strategy.addStrategyModule(
062                                new RandomJointLocationChoiceModule(
063                                        sc ) );
064
065                strategy.addStrategyModule(
066                                GroupPlanStrategyFactoryUtils.createJointTripAwareTourModeUnifierModule(
067                                        sc.getConfig(),
068                                        tripRouterFactory ) );
069
070                // TODO: add an option to enable or disable this part?
071                final VehicleRessources vehicles =
072                                (VehicleRessources) sc.getScenarioElement(
073                                        VehicleRessources.ELEMENT_NAME );
074                if ( vehicles != null ) {
075                        strategy.addStrategyModule(
076                                        GroupPlanStrategyFactoryUtils.createVehicleAllocationModule(
077                                                sc.getConfig(),
078                                                vehicles ) );
079                }
080
081                strategy.addStrategyModule(
082                                GroupPlanStrategyFactoryUtils.createReRouteModule(
083                                        sc.getConfig(),
084                                        planRoutingAlgorithmFactory,
085                                        tripRouterFactory ) );
086
087                strategy.addStrategyModule(
088                                GroupPlanStrategyFactoryUtils.createRecomposeJointPlansModule(
089                                        sc.getConfig(),
090                                        ((JointPlans) sc.getScenarioElement( JointPlans.ELEMENT_NAME  )).getFactory(),
091                                        planLinkIdentifier ) );
092
093                return strategy;
094        }
095}
