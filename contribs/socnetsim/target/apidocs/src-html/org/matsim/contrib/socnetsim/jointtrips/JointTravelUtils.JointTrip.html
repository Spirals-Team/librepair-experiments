039        public static JointTravelStructure analyseJointTravel(final JointPlan plan) {
040                // this is a two-passes algorithm:
041                // 1- parse the plans, and identify who drives who from where to where
042                // 2- parse the plans to identify the corresponding passenger legs.
043                // This approach is the best I could find, as it may take several driver
044                // legs to drive one passenger to destination, e.g.:
045                //                    p={1,2}       p={1}
046                // driver ...-----|############|#############|------...
047                // pass 1 ...-----|############|--------------------...
048                // pass 2 ...-----|##########################|------...
049                // (TD, sept. 2012)
050                try {
051                        final List<DriverTrip> driverTrips = parseDriverTrips( plan );
052                        final List<JointTrip> jointTrips = reconstructJointTrips( driverTrips, plan );
053
054                        return new JointTravelStructure( jointTrips );
055                }
056                catch (Exception e) {
057                        throw new RuntimeException( "Problem analyzing joint plan "+plan , e );
058                }
059        }
060
061        private static List<JointTrip> reconstructJointTrips(
062                        final List<DriverTrip> driverTrips,
063                        final JointPlan plan) {
064                final List<JointTrip> jointTrips = new ArrayList<JointTrip>();
065                final PlanElementIterators iterators = new PlanElementIterators( plan );
066
067                for ( final DriverTrip driverTrip : driverTrips ) {
068                        for (final Map.Entry<Id, Id> e : driverTrip.passengerOrigins.entrySet()) {
069                                final Id passengerId = e.getKey();
070                                final Id originId = e.getValue();
071                                final Id destinationId = driverTrip.passengerDestinations.get( passengerId );
072
073                                jointTrips.add(
074                                                getNextPassengerTrip(
075                                                        iterators,
076                                                        driverTrip,
077                                                        passengerId,
078                                                        originId,
079                                                        destinationId) );
080                        }
081                }
082
083                return jointTrips;
084        }
085
086        private static JointTrip getNextPassengerTrip(
087                        final PlanElementIterators iterators,
088                        final DriverTrip driverTrip,
089                        final Id passengerId,
090                        final Id originId,
091                        final Id destinationId) {
092                final Iterator<PlanElement> it =
093                        iterators.getIterator(
094                                        driverTrip.driverId,
095                                        passengerId,
096                                        originId,
097                                        destinationId);
098                while ( it.hasNext() ) {
099                        final PlanElement pe = it.next();
100                        if (pe instanceof Leg &&
101                                        ((Leg) pe).getMode().equals( JointActingTypes.PASSENGER )) {
102                                final PassengerRoute route = (PassengerRoute) ((Leg) pe).getRoute();
103                                if ( route.getDriverId().equals( driverTrip.driverId ) &&
104                                                route.getStartLinkId().equals( originId ) &&
105                                                route.getEndLinkId().equals( destinationId )) {
106                                        return new JointTrip(
107                                                                driverTrip.driverId,
108                                                                extractDriverSubTrip(
109                                                                        originId,
110                                                                        destinationId,
111                                                                        driverTrip.driverTrip ),
112                                                                passengerId,
113                                                                (Leg) pe );
114                                }
115                        }
116                }
117                throw new RuntimeException( "no valid trip found for passenger "+passengerId+" in driver trip "+driverTrip );
118        }
119
120        private static List<Leg> extractDriverSubTrip(
121                        final Id originId,
122                        final Id destinationId,
123                        final List<Leg> driverTrip) {
124                final ArrayList<Leg> subTrip = new ArrayList<Leg>();
125                boolean inSubTrip = false;
126
127                for ( final Leg l : driverTrip ) {
128                        if (l.getRoute().getStartLinkId().equals( originId )) inSubTrip = true;
129                        if (inSubTrip) subTrip.add( l );
130                        if (l.getRoute().getEndLinkId().equals( destinationId )) break;
131                }
132
133                return subTrip;
134        }

136        /**
137         * Parses the plans and get information about driver trips, ie who drives
138         * whom from where to where.
139         * Package protected to be callable from tests
140         */
141        final static List<DriverTrip> parseDriverTrips( final JointPlan plan ) {
142                final List<DriverTrip> driverTrips = new ArrayList<DriverTrip>();
143
144                for ( final Plan indivPlan : plan.getIndividualPlans().values() ) {
145                        final Id driverId = indivPlan.getPerson().getId();
146                        final List<Id> currentPassengers = new ArrayList<Id>();
147
148                        // store the driver trips we are in, if any
149                        DriverTrip currentDriverTrip = null;
150                        for ( final PlanElement pe : indivPlan.getPlanElements() ) {
151                                if ( pe instanceof Leg &&
152                                                JointActingTypes.DRIVER.equals( ((Leg) pe).getMode() ) ) {
153                                        if ( currentDriverTrip == null ) {
154                                                currentDriverTrip = new DriverTrip( driverId );
155                                                driverTrips.add( currentDriverTrip );
156                                                currentPassengers.clear();
157                                        }
158                                        currentDriverTrip.driverTrip.add( (Leg) pe );
159
160                                        final DriverRoute dRoute = (DriverRoute) ((Leg) pe).getRoute();
161                                        final Id origin = dRoute.getStartLinkId();
162                                        final Id destination = dRoute.getEndLinkId();
163                                        final Collection<Id<Person>> passengerIds = dRoute.getPassengersIds();
164
165                                        for ( final Id passengerId : passengerIds ) {
166                                                if ( !currentPassengers.contains( passengerId ) ) {
167                                                        currentDriverTrip.passengerOrigins.put(
168                                                                        passengerId,
169                                                                        origin );
170                                                        currentPassengers.add( passengerId );
171                                                }
172                                        }
173
174                                        final Iterator<Id> currPassengersIter = currentPassengers.iterator();
175                                        while ( currPassengersIter.hasNext() ) {
176                                                final Id p = currPassengersIter.next();
177                                                if ( !passengerIds.contains( p ) ) {
178                                                        // passenger is arrived
179                                                        currPassengersIter.remove();
180                                                }
181                                                else {
182                                                        // update destination
183                                                        currentDriverTrip.passengerDestinations.put(
184                                                                        p,
185                                                                        destination );
186                                                }
187                                        }
188                                }
189                                else if ( pe instanceof Leg ||
190                                                !JointActingTypes.JOINT_STAGE_ACTS.isStageActivity(
191                                                        ((Activity) pe).getType() ) ) {
192                                        currentDriverTrip = null;
193                                }
194                        }
195                }
196
197                return driverTrips;
198        }

200        // package protected for tests
201        static final class DriverTrip {
202                final Id driverId;
203                final List<Leg> driverTrip = new ArrayList<Leg>();
204                final Map<Id, Id> passengerOrigins = new HashMap<Id, Id>();
205                final Map<Id, Id> passengerDestinations = new HashMap<Id, Id>();
206
207                public DriverTrip(final Id driverId) {
208                        this.driverId = driverId;
209                }
210
211                @Override
212                public int hashCode() {
213                        int c = 0;
214
215                        c += driverId.hashCode();
216                        c += driverTrip.hashCode();
217                        c += passengerOrigins.hashCode();
218                        c += passengerDestinations.hashCode();
219
220                        return c;
221                }
222
223                @Override
224                public boolean equals(final Object o) {
225                        if (o != null && o instanceof DriverTrip) {
226                                final DriverTrip other = (DriverTrip) o;
227                                return other.driverId.equals( driverId ) &&
228                                        other.driverTrip.equals( driverTrip ) &&
229                                        other.passengerOrigins.equals( passengerOrigins ) &&
230                                        other.passengerDestinations.equals( passengerDestinations );
231                        }
232                        return false;
233                }
234
235                @Override
236                public String toString() {
237                        return "[DriverTrip: driverId="+driverId
238                                +", driverTrip="+driverTrip
239                                +", passengerOrigins="+passengerOrigins
240                                +", passengerDestinations="+passengerDestinations+"]";
241                }
242        }

246        /**
247         * Represents the "joint travel structure" of the plan, that is,
248         * who travels with whom from where to where, with references
249         * to the relevant plan elements.
250         */
251        public static final class JointTravelStructure {
252                private final List<JointTrip> jointTrips;
253                private final Map<Couple, List<JointTrip>> jointTripsPerCouple = new HashMap<Couple, List<JointTrip>>();
254
255                public JointTravelStructure(
256                                final List<JointTrip> jointTrips) {
257                        this.jointTrips = jointTrips;
258
259                        for (final JointTrip jt : jointTrips) {
260                                final List<JointTrip> trips =
261                                        MapUtils.getList(
262                                                        new Couple(
263                                                                jt.getDriverId(),
264                                                                jt.getPassengerId() ),
265                                                        jointTripsPerCouple );
266                                trips.add( jt );
267                        }
268                }       
269
270                public List<JointTrip> getJointTrips() {
271                        return jointTrips;
272                }
273
274                public List<JointTrip> getJointTripsForCotravelers(final Id id1, final Id id2) {
275                        final List<JointTrip> trips =
276                                jointTripsPerCouple.get(
277                                                new Couple(
278                                                        id1,
279                                                        id2 ) );
280
281                        return trips == null ?
282                                Collections.<JointTrip>emptyList() :
283                                Collections.unmodifiableList( trips );
284                }
285
286                @Override
287                public int hashCode() {
288                        return getJointTrips().hashCode();
289                }
290
291                @Override
292                public boolean equals(final Object o) {
293                        if (o != null && o instanceof JointTravelStructure) {
294                                final JointTravelStructure other = (JointTravelStructure) o;
295                                return other.getJointTrips().size() == getJointTrips().size() &&
296                                        other.getJointTrips().containsAll( getJointTrips() );
297                        }
298                        return false;
299                }
300
301                @Override
302                public String toString() {
303                        return "[JointTravelStructure: "+getJointTrips()+"]";
304                }
305        }

307        /**
308         * Gathers information releted to one joint trip, that is,
309         * one (and <b>only one</b>) passenger being driven.
310         * Note that driver legs may pertain to several joint trips,
311         * if the driver drives several passengers at the same time.
312         */
313        public static final class JointTrip {
314                private final Id driverId;
315                private final Id passengerId;
316                private final List<Leg> driverLegs;
317                private final Leg passengerLeg;
318
319                public JointTrip(
320                                final Id driverId,
321                                final List<Leg> driverLegs,
322                                final Id passengerId,
323                                final Leg passengerLeg) {
324                        this.driverId = driverId;
325                        this.passengerId = passengerId;
326                        this.driverLegs = Collections.unmodifiableList( driverLegs );
327                        this.passengerLeg = passengerLeg;
328                }
329
330                public List<Leg> getDriverLegs() {
331                        return driverLegs;
332                }
333
334                public Id getDriverId() {
335                        return driverId;
336                }
337
338                public Id getPassengerId() {
339                        return passengerId;
340                }
341
342                public Leg getPassengerLeg() {
343                        return passengerLeg;
344                }
345
346                @Override
347                public int hashCode() {
348                        int c = 0;
349
350                        c += getDriverId().hashCode();
351                        c += getPassengerId().hashCode();
352                        c += getDriverLegs().hashCode();
353                        c += getPassengerLeg().hashCode(); 
354
355                        return c;
356                }
357
358                @Override
359                public boolean equals(final Object o) {
360                        if (o != null && o instanceof JointTrip) {
361                                final JointTrip other = (JointTrip) o;
362                                return other.getDriverId().equals( getDriverId() ) &&
363                                        other.getPassengerId().equals( getPassengerId() ) &&
364                                        other.getDriverLegs().equals( getDriverLegs() ) &&
365                                        other.getPassengerLeg().equals( getPassengerLeg() );
366                        }
367                        return false;
368                }
369
370                @Override
371                public String toString() {
372                        return "[JointTrip: driver="+getDriverId()
373                                +", driverTrip="+getDriverLegs()
374                                +", passenger="+getPassengerId()
375                                +", passengerLeg="+getPassengerLeg()+"]";
376                }
377        }
378}

380class PlanElementIterators {
381        private final JointPlan jointPlan;
382        private final Map<TripCharacteristics, Iterator<PlanElement>> iterators = new HashMap<TripCharacteristics, Iterator<PlanElement>>();
383
384        public PlanElementIterators(final JointPlan jointPlan) {
385                this.jointPlan = jointPlan;
386        }
387
388        public Iterator<PlanElement> getIterator(
389                        final Id driver,
390                        final Id passenger,
391                        final Id origin,
392                        final Id destination) {
393                final TripCharacteristics key = new TripCharacteristics( driver, passenger , origin , destination );
394                Iterator<PlanElement> it = iterators.get( key );
395
396                if ( it == null ) {
397                        final Plan plan = jointPlan.getIndividualPlan( passenger );
398                        if ( plan == null ) throw new RuntimeException( "no plan for passenger "+passenger+" in joint plan "+jointPlan );
399                        it = plan.getPlanElements().iterator();
400                        iterators.put( key , it );
401                }
402
403                return it;
404        }
405}

407class TripCharacteristics {
408        private final Id driver, passenger, origin, destination;
409
410        public TripCharacteristics(
411                        final Id driver,
412                        final Id passenger,
413                        final Id origin,
414                        final Id destination) {
415                this.driver = driver;
416                this.passenger = passenger;
417                this.origin = origin;
418                this.destination = destination;
419        }
420
421        @Override
422        public boolean equals(final Object o) {
423                return o instanceof TripCharacteristics &&
424                        ((TripCharacteristics) o).driver.equals( driver ) &&
425                        ((TripCharacteristics) o).passenger.equals( passenger ) &&
426                        ((TripCharacteristics) o).origin.equals( origin ) &&
427                        ((TripCharacteristics) o).destination.equals( destination );
428        }
429
430        @Override
431        public int hashCode() {
432                return driver.hashCode() + passenger.hashCode() + origin.hashCode() + destination.hashCode();
433        }
