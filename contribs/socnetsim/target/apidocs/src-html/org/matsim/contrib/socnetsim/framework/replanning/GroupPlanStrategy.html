020package org.matsim.contrib.socnetsim.framework.replanning;
021
022import java.util.ArrayList;
023import java.util.Collection;
024import java.util.List;
025
026import org.matsim.api.core.v01.population.Plan;
027import org.matsim.core.replanning.ReplanningContext;
028import org.matsim.core.utils.misc.Counter;
029
030import org.matsim.contrib.socnetsim.framework.population.JointPlan;
031import org.matsim.contrib.socnetsim.framework.population.JointPlans;
032import org.matsim.contrib.socnetsim.framework.replanning.grouping.GroupPlans;
033import org.matsim.contrib.socnetsim.framework.replanning.grouping.ReplanningGroup;
034import org.matsim.contrib.socnetsim.framework.replanning.selectors.GroupLevelPlanSelector;
035
036/**
037 * Generalizes the PlanStrategy concept to joint replanning.
038 * @author thibautd
039 */
040public class GroupPlanStrategy {
041        private final GroupLevelPlanSelector selector;
042        private final List<GenericStrategyModule<GroupPlans>> strategyModules =
043                        new ArrayList<GenericStrategyModule<GroupPlans>>();
044
045        public GroupPlanStrategy(final GroupLevelPlanSelector selector) {
046                this.selector = selector;
047        }
048
049        public void addStrategyModule(final GenericStrategyModule<GroupPlans> module) {
050                strategyModules.add( module );
051        }
052
053        public void run(
054                        // XXX what is the best at this level: ReplanningContext or ControllerRegistry?
055                        final ReplanningContext replanningContext,
056                        final JointPlans jointPlans,
057                        final Collection<ReplanningGroup> groups) {
058                List<GroupPlans> plansToHandle = new ArrayList<GroupPlans>();
059
060                Counter selectCounter = new Counter( "["+selector.getClass().getSimpleName()+"] selecting plan # " );
061                for (ReplanningGroup group : groups) {
062                        selectCounter.incCounter();
063                        GroupPlans plans = selector.selectPlans(
064                                        jointPlans,
065                                        group );
066
067                        if (plans == null) {
068                                // this is a valid output from the selector,
069                                // if no plans combination is possible.
070                                selectCounter.printCounter();
071                                throw new RuntimeException( "no plan returned by the selector for group "+group );
072                        }
073
074                        if (strategyModules.size() > 0) {
075                                plans = GroupPlans.copyPlans( jointPlans.getFactory() , plans );
076                                plansToHandle.add( plans );
077                                assert !jointPlansAreRegistered( plans , jointPlans );
078                        }
079
080                        select( plans );
081                }
082                selectCounter.printCounter();
083
084                for (GenericStrategyModule<GroupPlans> module : strategyModules) {
085                        module.handlePlans( replanningContext , plansToHandle );
086                }
087
088                // the modules are allowed to modify joint structure:
089                // register joint plans afterhand only.
090                final Counter jpCounter = new Counter( "Register joint plans for group # " , " / "+plansToHandle.size() );
091                for (GroupPlans groupPlans : plansToHandle) {
092                        jpCounter.incCounter();
093                        jointPlans.addJointPlans( groupPlans.getJointPlans() );
094                }
095                jpCounter.printCounter();
096        }
097
098        private static boolean jointPlansAreRegistered(
099                        final GroupPlans plans,
100                        final JointPlans jointPlans) {
101                for (JointPlan jp : plans.getJointPlans()) {
102                        if (jointPlans.contains( jp )) return true;
103                }
104                return false;
105        }
106
107        private static void select(final GroupPlans plans) {
108                for (JointPlan jp : plans.getJointPlans()) {
109                        for (Plan p : jp.getIndividualPlans().values()) {
110                                p.getPerson().setSelectedPlan(p);
111                        }
112                }
113
114                for (Plan p : plans.getIndividualPlans()) {
115                        p.getPerson().setSelectedPlan(p);
116                }
117        }
118
119        @Override
120        public String toString() {
121                return "["+getClass().getSimpleName()+": "
122                        +selector.getClass().getSimpleName()
123                        +";"+strategyModules+"]";
124        }
125}
