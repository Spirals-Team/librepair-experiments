030import java.util.SortedSet;
031import java.util.TreeSet;

033/**
034 * @author thibautd
035 */
036public class CollectionUtils {
037        public static <T> List<T> getRandomDistinctElements(
038                        final Random random,
039                        final List<T> list,
040                        final int nElements ) {
041                if ( list.size() < nElements ) throw new IllegalArgumentException( "cannot sample "+nElements+" elements from "+list.size() );
042
043                // TODO: avoid creating "bowl" collection
044                final List<T> bowl = new ArrayList<T>( list );
045                final List<T> sample = new ArrayList<T>( nElements );
046
047                for ( int i=0; i < nElements; i++ ) {
048                        sample.add(
049                                        bowl.remove(
050                                                random.nextInt(
051                                                        bowl.size() ) ) );
052                }
053
054                return sample;
055        }
056
057        public static boolean intersects(
058                        final Collection<?> c1,
059                        final Collection<?> c2) {
060                for ( Object o : c1 ) {
061                        if ( c2.contains( o ) ) return true;
062                }
063                return false;
064        }
065
066        public static <T> Set<T> intersect(
067                        final Collection<? extends T> c1,
068                        final Collection<? extends T> c2) {
069                final Set<T> set = new HashSet<T>();
070                for ( T t : c1 ) {
071                        if ( c2.contains( t ) ) set.add( t );
072                }
073                return set;
074        }
075
076        public static <T extends Comparable<T>> SortedSet<T> intersectSorted(
077                        final Collection<? extends T> c1,
078                        final Collection<? extends T> c2) {
079                final SortedSet<T> set = new TreeSet<T>();
080                for ( T t : c1 ) {
081                        if ( c2.contains( t ) ) set.add( t );
082                }
083                return set;
084        }
085
086        /**
087         * makes sense only if iteration order deterministic!
088         */
089        public static <K,V> Map.Entry<K,V> getRandomElement(
090                        final Random random,
091                        final Map<K,V> map) {
092                return getRandomElement( false , random , map );
093        }
094
095        /**
096         * makes sense only if iteration order deterministic!
097         */
098        public static <K,V> Map.Entry<K,V> removeRandomElement(
099                        final Random random,
100                        final Map<K,V> map) {
101                return getRandomElement( true , random , map );
102        }
103
104
105        /**
106         * makes sense only if iteration order deterministic!
107         */
108        private static <K,V> Map.Entry<K,V> getRandomElement(
109                        final boolean remove,
110                        final Random random,
111                        final Map<K,V> map) {
112                if ( map.isEmpty() ) throw new IllegalArgumentException( "map is empty!" );
113                final int index = random.nextInt( map.size() );
114
115                final Iterator<Map.Entry<K,V>> it = map.entrySet().iterator();
116                int i=0;
117
118                while ( i++ < index ) it.next();
119
120                final Map.Entry<K,V> elem = it.next();
121                if ( remove ) it.remove();
122                return elem;
123        }
124
125        /**
126         * makes sense only if iteration order deterministic!
127         */
128        public static <T> T removeElement(
129                        final int index,
130                        final Collection<T> coll) {
131                return getElement( true , index , coll );
132        }
133
134        /**
135         * makes sense only if iteration order deterministic!
136         */
137        public static <T> T getElement(
138                        final int index,
139                        final Collection<T> coll) {
140                return getElement( false , index , coll );
141        }
142
143        /**
144         * makes sense only if iteration order deterministic!
145         */
146        private static <T> T getElement(
147                        final boolean remove,
148                        final int index,
149                        final Collection<T> coll) {
150                if ( index >= coll.size() ) throw new IndexOutOfBoundsException( index+" >= "+coll.size() );
151
152                final Iterator<T> it = coll.iterator();
153                int i=0;
154
155                while ( i++ < index ) it.next();
156
157                final T elem = it.next();
158                if ( remove ) it.remove();
159                return elem;
160        }
161
162        public static List<String> toString(
163                        final List<? extends Object> list) {
164                final List<String> strings = new ArrayList<String>( list.size() );
165                for ( Object o : list ) strings.add( o.toString() );
166                return strings;
167        }
