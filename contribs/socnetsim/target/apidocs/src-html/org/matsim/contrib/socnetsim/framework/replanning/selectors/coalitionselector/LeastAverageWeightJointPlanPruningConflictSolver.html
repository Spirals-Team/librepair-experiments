020package org.matsim.contrib.socnetsim.framework.replanning.selectors.coalitionselector;
021
022import java.util.Collection;
023import java.util.Map;
024
025import org.matsim.contrib.socnetsim.framework.population.JointPlan;
026import org.matsim.contrib.socnetsim.framework.replanning.selectors.coalitionselector.CoalitionSelector.ConflictSolver;
027
028/**
029 * @author thibautd
030 */
031public class LeastAverageWeightJointPlanPruningConflictSolver implements ConflictSolver {
032
033        @Override
034        public void attemptToSolveConflicts(
035                        final CoalitionSelector.RecordsOfJointPlan recordsPerJointPlan) {
036                double minAvg = Double.POSITIVE_INFINITY;
037                Map.Entry<JointPlan , Collection<PlanRecord>> toMark = null;
038
039                for ( final Map.Entry<JointPlan , Collection<PlanRecord>> entry : recordsPerJointPlan.entrySet() ) {
040                        final Collection<PlanRecord> records = entry.getValue();
041
042                        double avg = 0;
043                        boolean isPointed = false;
044                        for ( final PlanRecord r : records ) {
045                                if ( !r.isFeasible() ) throw new RuntimeException( "should not get unfeasible plans!" );
046                                isPointed = isPointed || r.getAgent().getPointedPlan() == r.getPlan();
047                                avg += r.getWeight();
048                        }
049                        avg /= records.size();
050
051                        if ( avg < minAvg ) {
052                                minAvg = avg;
053                                toMark = entry;
054                        }
055                }
056
057                if ( toMark == null ) {
058                        throw new RuntimeException( "could not find pointed joint plan of minimum average weight in "+recordsPerJointPlan );
059                }
060
061                for ( final PlanRecord r : toMark.getValue() ) r.setInfeasible();
062                recordsPerJointPlan.remove( toMark.getKey() );
063        }
064}
