020package org.matsim.contrib.socnetsim.usage.analysis.scripts;
021
022import java.io.BufferedWriter;
023import java.io.IOException;
024import java.util.ArrayList;
025import java.util.HashMap;
026import java.util.HashSet;
027import java.util.List;
028import java.util.Map;
029import java.util.Set;
030import java.util.Stack;
031
032import org.matsim.api.core.v01.Id;
033import org.matsim.api.core.v01.population.Person;
034import org.matsim.core.utils.io.IOUtils;
035import org.matsim.core.utils.io.MatsimXmlParser;
036import org.matsim.core.utils.io.UncheckedIOException;
037import org.matsim.core.utils.misc.Counter;
038import org.xml.sax.Attributes;
039
040/**
041 * @author thibautd
042 */
043public class DumpSizeOfSelectedJointPlans {
044        public static void main(final String[] args) throws IOException {
045                final String plansFile = args[ 0 ];
046                final String jointPlansFile = args[ 1 ];
047                final String outFile = args[ 2 ];
048
049                final Map<Id, Integer> selectedIndices = parseSelectedIndices( plansFile );
050                final Set<Id> agentWithNoJointPlan = new HashSet<Id>( selectedIndices.keySet() );
051
052                final BufferedWriter writer = IOUtils.getBufferedWriter( outFile );
053                writer.write( "personId\tsize" );
054
055                final Counter counter = new Counter( "parse joint plan # " );
056                new MatsimXmlParser( ) {
057                        {
058                                setValidating(false) ;
059                        }
060                        int size = -1;
061                        boolean isSelected = false;
062                        final List<Id> persons = new ArrayList<Id>();
063                        @Override
064                        public void startTag(
065                                        final String name,
066                                        final Attributes atts,
067                                        final Stack<String> context) {
068                                if ( name.equals( "jointPlan" ) ) {
069                                        counter.incCounter();
070                                        size = 0;
071                                        isSelected = false;
072                                        persons.clear();
073                                }
074                                if ( name.equals( "individualPlan" ) ) {
075                                        size++;
076                                        final Id id = Id.create(
077                                                                        atts.getValue(
078                                                                                "personId" ).trim() , Person.class );
079                                        final Integer selectedIndex = selectedIndices.get( id );
080                                        persons.add( id );
081                                        if ( selectedIndex.equals(
082                                                                        Integer.valueOf(
083                                                                                atts.getValue(
084                                                                                        "planNr" ) ) ) ) {
085                                                if ( size > 1 && !isSelected ) throw new RuntimeException();
086                                                isSelected = true;
087                                        }
088                                }
089                        }
090
091                        @Override
092                        public void endTag(
093                                        final String name,
094                                        final String content,
095                                        final Stack<String> context) {
096                                if ( name.equals( "jointPlan" ) && isSelected ) {
097                                        agentWithNoJointPlan.removeAll( persons );
098                                        try {
099                                                for ( Id p : persons ) {
100                                                        writer.newLine();
101                                                        writer.write( p+"\t"+size );
102                                                }
103                                        }
104                                        catch (IOException e) {
105                                                throw new UncheckedIOException( e );
106                                        }
107                                }
108                        }
109                }.readFile( jointPlansFile );
110
111                for ( Id p : agentWithNoJointPlan ) {
112                        writer.newLine();
113                        writer.write( p+"\t"+1 );
114                }
115
116                counter.printCounter();
117                writer.close();
118        }
119
120        private static Map<Id, Integer> parseSelectedIndices(
121                        final String plansFile) {
122                final Map<Id, Integer> map = new HashMap<Id, Integer>();
123
124                final Counter counter = new Counter( "parse person # " );
125                new MatsimXmlParser() {
126                        private Id<Person> currentAgent = null;
127                        private int currIndex = -1;
128
129                        @Override
130                        public void startTag(
131                                        final String name,
132                                        final Attributes atts,
133                                        final Stack<String> context) {
134                                if ( name.equals( "person" ) ) {
135                                        counter.incCounter();
136                                        currentAgent = Id.create( atts.getValue( "id" ).trim() , Person.class );
137                                        currIndex = -1;
138                                }
139                                if ( name.equals( "plan" ) ) {
140                                        currIndex++;
141                                        if ( atts.getValue( "selected" ).equals( "yes" ) ) {
142                                                final Integer old = map.put( currentAgent , currIndex );
143                                                if ( old != null ) throw new RuntimeException();
144                                        }
145                                }
146                        }
147
148                        @Override
149                        public void endTag(
150                                        final String name,
151                                        final String content,
152                                        final Stack<String> context) {
153                        }
154                }.readFile( plansFile );
155
156                counter.printCounter();
157
158                return map;
159        }
160}
