020package org.matsim.contrib.socnetsim.framework.replanning.selectors;
021
022import java.util.ArrayList;
023import java.util.Collection;
024import java.util.HashMap;
025import java.util.HashSet;
026import java.util.Iterator;
027import java.util.List;
028import java.util.Map;
029import java.util.Set;
030
031import org.matsim.api.core.v01.Id;
032import org.matsim.api.core.v01.population.Person;
033import org.matsim.api.core.v01.population.Plan;
034
035import org.matsim.contrib.socnetsim.framework.population.JointPlan;
036import org.matsim.contrib.socnetsim.framework.population.JointPlans;
037import org.matsim.contrib.socnetsim.framework.replanning.grouping.GroupPlans;
038import org.matsim.contrib.socnetsim.framework.replanning.grouping.ReplanningGroup;
039import org.matsim.contrib.socnetsim.framework.replanning.modules.PlanLinkIdentifier;

041/**
042 * create a "weak" selector (only considering strong ties) from a "normal"
043 * selector (considering joint plans).
044 * <br>
045 * This works as expected only if the joint plans were created using the "strong"
046 * version of the plan link identifier used here.
047 * @author thibautd
048 */
049public class WeakSelector implements GroupLevelPlanSelector {
050        private final GroupLevelPlanSelector delegate;
051        private final PlanLinkIdentifier weakPlanLinkIdentifier;
052
053        public WeakSelector(
054                        final PlanLinkIdentifier weakPlanLinkIdentifier,
055                        final GroupLevelPlanSelector delegate ) {
056                this.weakPlanLinkIdentifier = weakPlanLinkIdentifier;
057                this.delegate = delegate;
058        }
059
060        /**
061         * Note that the joint plans in the group plans will not be joint plans
062         * referenced in the container passed as parameter!
063         */
064        @Override
065        public GroupPlans selectPlans(
066                        final JointPlans jointPlans,
067                        final ReplanningGroup group) {
068                final Collection<JointPlan> jpCollection = getJointPlans( group , jointPlans );
069                final JointPlans jointPlansInstance = new JointPlans();
070                fillWithWeakJointPlans( jointPlansInstance , jpCollection );
071
072                final GroupPlans selected = delegate.selectPlans( jointPlansInstance , group );
073
074                // this is necessary in order to avoid ridiculous memory consumption, with
075                // joint plan caches being full of unused joint plans.
076                // As we clean it, we could as well re-use it, to avoid having thousands of
077                // loose objects lying around (not done, as the actual effect on performance
078                // is probably minimal)
079                jointPlansInstance.clear();
080                return selected;
081        }
082
083        private Collection<JointPlan> getJointPlans(
084                        final ReplanningGroup group,
085                        final JointPlans jointPlans) {
086                final Set<JointPlan> set = new HashSet<JointPlan>();
087
088                for ( Person person : group.getPersons() ) {
089                        for ( Plan plan : person.getPlans() ) {
090                                final JointPlan jp = jointPlans.getJointPlan( plan );
091                                if ( jp != null ) set.add( jp );
092                        }
093                }
094                return set;
095        }
096
097        public void fillWithWeakJointPlans(
098                        final JointPlans weakPlans,
099                        final Collection<JointPlan> jointPlans) {
100                for ( JointPlan fullJointPlan : jointPlans ) {
101                        final Map<Id<Person>, Plan> plansMap = new HashMap< >( fullJointPlan.getIndividualPlans() );
102
103                        while ( ! plansMap.isEmpty() ) {
104                                final Plan plan = plansMap.remove( plansMap.keySet().iterator().next() );
105                                final Map<Id<Person>, Plan> jpMap = new HashMap< >();
106                                jpMap.put( plan.getPerson().getId() , plan );
107
108                                findDependentPlans( plan , jpMap , plansMap );
109
110                                if ( jpMap.size() > 1 ) {
111                                        weakPlans.addJointPlan(
112                                                        weakPlans.getFactory().createJointPlan( jpMap ) );
113                                }
114                        }
115                }
116        }
117
118        // DFS
119        private void findDependentPlans(
120                        final Plan plan,
121                        final Map<Id<Person>, Plan> dependantPlans,
122                        final Map<Id<Person>, Plan> plansToLook) {
123                final List<Plan> dependentPlansList = new ArrayList<Plan>();
124
125                final Iterator<Plan> toLookIt = plansToLook.values().iterator();
126                while ( toLookIt.hasNext() ) {
127                        final Plan toLook = toLookIt.next();
128                        if ( weakPlanLinkIdentifier.areLinked( plan , toLook ) ) {
129                                dependentPlansList.add( toLook );
130                                toLookIt.remove();
131                        }
132                }
133
134                for (Plan depPlan : dependentPlansList) {
135                        dependantPlans.put( depPlan.getPerson().getId() , depPlan );
136                        findDependentPlans(
137                                        depPlan,
138                                        dependantPlans,
139                                        plansToLook);
140                }
141        }
142}
