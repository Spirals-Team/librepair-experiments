001package org.matsim.contrib.accessibility;
002
003import org.matsim.api.core.v01.network.Node;
004import org.matsim.contrib.accessibility.utils.AggregationObject;
005import org.matsim.core.config.groups.PlanCalcScoreConfigGroup;
006import org.matsim.core.router.util.LeastCostPathCalculator;
007import org.matsim.facilities.ActivityFacility;
008
009/**
010 * @author thibautd
011 * @author michaz
012 */
013public final class LeastCostPathCalculatorAccessibilityContributionCalculator implements AccessibilityContributionCalculator {
014        private final LeastCostPathCalculator leastCostPathCalculator;
015        private final PlanCalcScoreConfigGroup planCalcScoreConfigGroup;
016        private Node fromNode;
017        private Double departureTime;
018
019        public LeastCostPathCalculatorAccessibilityContributionCalculator(PlanCalcScoreConfigGroup planCalcScoreConfigGroup, LeastCostPathCalculator leastCostPathCalculator) {
020                this.planCalcScoreConfigGroup = planCalcScoreConfigGroup;
021                this.leastCostPathCalculator = leastCostPathCalculator;
022        }
023
024        @Override
025        public void notifyNewOriginNode(Node fromNode, Double departureTime) {
026                this.fromNode = fromNode;
027                this.departureTime = departureTime;
028        }
029
030        @Override
031        public double computeContributionOfOpportunity(ActivityFacility origin, AggregationObject destination, Double departureTime) {
032                LeastCostPathCalculator.Path path = leastCostPathCalculator.calcLeastCostPath(fromNode, destination.getNearestNode(), departureTime, null, null);
033                return destination.getSum() * Math.exp(planCalcScoreConfigGroup.getBrainExpBeta() * path.travelCost);
034        }
035}
