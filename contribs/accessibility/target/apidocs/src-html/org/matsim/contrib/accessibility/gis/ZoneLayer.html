020package org.matsim.contrib.accessibility.gis;
021
022import java.util.ArrayList;
023import java.util.Collections;
024import java.util.List;
025import java.util.Set;
026
027import org.geotools.referencing.CRS;
028import org.opengis.referencing.FactoryException;
029import org.opengis.referencing.crs.CoordinateReferenceSystem;
030import org.opengis.referencing.operation.MathTransform;
031
032import com.vividsolutions.jts.geom.Point;
033import com.vividsolutions.jts.index.SpatialIndex;
034import com.vividsolutions.jts.index.quadtree.Quadtree;
035
036/**
037 * Representation of a spatial index containing zones backed by a quadtree.
038 * 
039 * @author illenberger
040 *
041 */
042public final class ZoneLayer<T> {
043
044        private final SpatialIndex quadtree;
045        
046        private final Set<Zone<T>> zones;
047        
048        private CoordinateReferenceSystem crs;
049        
050        private int srid = -1;
051        
052        /**
053         * Creates a new zone layer containing the zones in <tt>zones</tt>.
054         * 
055         * @param zones a set of zones.
056         */
057        public ZoneLayer(Set<Zone<T>> zones) {
058                for(Zone<T> z : zones) {
059                        if(srid < 0) {
060                                srid = z.getGeometry().getSRID();
061                                crs = CRSUtils.getCRS(srid);
062                        } else {
063                                if(z.getGeometry().getSRID() != srid)
064                                        throw new RuntimeException("Cannot build a spatial index with zones that have different coordinate reference systems.");
065                        }
066                }
067                
068                this.zones = Collections.unmodifiableSet(zones);
069                quadtree = new Quadtree();
070                for(Zone<T> zone : zones) {
071                        quadtree.insert(zone.getGeometry().getEnvelopeInternal(), zone);
072                }
073        }
074
075        /**
076         * Allows to manually overwrite the coordinate reference system (e.g. if the
077         * ZoneLayer is read from a shape file without crs information).
078         * 
079         * @param crs a coordinate reference system.
080         */
081        public void overwriteCRS(CoordinateReferenceSystem crs) {
082                this.crs = crs;
083                this.srid = CRSUtils.getSRID(crs);
084                for(Zone<?> zone : zones) {
085                        zone.getGeometry().setSRID(srid);
086                }
087        }
088
089        @SuppressWarnings("unchecked")
090        private List<Zone<T>> getZones(Point point) {
091                if(point.getSRID() != srid)
092                        point = transformPoint(point);
093                
094                List<Zone<T>> result = quadtree.query(point.getEnvelopeInternal());
095                List<Zone<T>> zones = new ArrayList<Zone<T>>(result.size());
096                for(Zone<T> z : result) {
097                        if(z.getGeometry().contains(point))
098                                zones.add(z);
099                }
100                return zones;
101        }
102        
103        private Point transformPoint(Point point) {
104                CoordinateReferenceSystem sourceCRS = CRSUtils.getCRS(point.getSRID());
105                CoordinateReferenceSystem targetCRS = crs;
106
107                try {
108                        MathTransform transform = CRS.findMathTransform(sourceCRS, targetCRS);
109                        return CRSUtils.transformPoint(point, transform);
110                } catch (FactoryException e) {
111                        e.printStackTrace();
112                        return null;
113                }       
114        }
115        /**
116         * Returns the zone containing <tt>point</tt>. If multiple zones contain
117         * <tt>point</tt> one random zone is returned.
118         * 
119         * @param point a point geometry
120         * @return the zone containing <tt>point</tt>, or <tt>null</tt> if no zone contains <tt>point</tt>.
121         */
122        public Zone<T> getZone(Point point) {
123                List<Zone<T>> zones = getZones(point);
124                if(zones.isEmpty())
125                        return null;
126                else
127                        return zones.get(0);
128        }
129        
130        /**
131         * Returns a set of all zones.
132         * 
133         * @return a set of all zones.
134         */
135        public Set<Zone<T>> getZones() {
136                return zones;
137        }
138}
