065        public AccessibilityCalculator(Scenario scenario, ActivityFacilities measuringPoints) {
066                this.network = scenario.getNetwork();
067                this.measuringPoints = measuringPoints;
068                this.acg = ConfigUtils.addOrGetModule(scenario.getConfig(), AccessibilityConfigGroup.GROUP_NAME, AccessibilityConfigGroup.class);
069                this.cnScoringGroup = scenario.getConfig().planCalcScore();
070
071                if (cnScoringGroup.getOrCreateModeParams(TransportMode.car).getMarginalUtilityOfDistance() != 0.) {
072                        LOG.error("marginal utility of distance for car different from zero but not used in accessibility computations");
073                }
074                if (cnScoringGroup.getOrCreateModeParams(TransportMode.pt).getMarginalUtilityOfDistance() != 0.) {
075                        LOG.error("marginal utility of distance for pt different from zero but not used in accessibility computations");
076                }
077                if (cnScoringGroup.getOrCreateModeParams(TransportMode.bike).getMonetaryDistanceRate() != 0.) {
078                        LOG.error("monetary distance cost rate for bike different from zero but not used in accessibility computations");
079                }
080                if (cnScoringGroup.getOrCreateModeParams(TransportMode.walk).getMonetaryDistanceRate() != 0.) {
081                        LOG.error("monetary distance cost rate for walk different from zero but not used in accessibility computations");
082                }
083
084                this.walkSpeed_m_h = scenario.getConfig().plansCalcRoute().getTeleportedModeSpeeds().get(TransportMode.walk) * 3600.;
085        }

087        public final void computeAccessibilities(Double departureTime, ActivityFacilities opportunities) {
088                AggregationObject[] aggregatedOpportunities = aggregateOpportunities(opportunities, network);
089
090                Map<String,Double> expSums = new HashMap<>();
091                for (String mode : calculators.keySet()) {
092                        expSums.put(mode, 0.);
093                }
094
095                // Condense measuring points (origins) that have the same nearest node on the network
096                Map<Id<Node>, ArrayList<ActivityFacility>> aggregatedOrigins = aggregateMeasurePointsWithSameNearestNode();
097
098                LOG.info("Iterating over all aggregated measuring points...");
099                ProgressBar progressBar = new ProgressBar(aggregatedOrigins.size());
100                
101                // Go through all nodes (keys) that have a measuring point (origin) assigned
102                for (Id<Node> nodeId : aggregatedOrigins.keySet()) {
103                        progressBar.update();
104
105                        Node fromNode = network.getNodes().get(nodeId);
106
107                        for (AccessibilityContributionCalculator calculator : calculators.values()) {
108                                Gbl.assertNotNull(calculator);
109                                calculator.notifyNewOriginNode(fromNode, departureTime);
110                        }
111
112                        // Get list with measuring points that are assigned to "fromNode"
113                        // Go through all measuring points assigned to current node
114                        for (ActivityFacility origin : aggregatedOrigins.get(nodeId)) {
115                                assert(origin.getCoord() != null);
116
117                                for (String key : expSums.keySet()) { // Is it really necessary to reset here?
118                                        expSums.put(key, 0.);
119                                }
120                                
121                                // Gbl.assertIf(aggregatedOpportunities.length > 0);
122                                // yyyyyy a test fails when this line is made active; cannot say why an execution path where there are now opportunities can make sense for a test.  kai, mar'17
123                                
124                                // Go through all aggregated facilities (i.e. network nodes to which at least one facility is assigned)
125                                for (final AggregationObject aggregatedFacility : aggregatedOpportunities) {
126                                        // Go through all calculators
127                                        for (String mode : calculators.keySet()) {
128                                                final double expVhk = calculators.get(mode).computeContributionOfOpportunity(origin, aggregatedFacility, departureTime);
129                                                expSums.put(mode, expSums.get(mode) + expVhk);
130                                        }
131                                }
132                                // What does the aggregation of the starting locations save if we do the just ended loop for all starting
133                                // points separately anyways? Answer: The trees need to be computed only once. (But one could save more.) kai, feb'14
134
135                                // aggregated value
136                                Map<String, Double> accessibilities  = new LinkedHashMap<>();
137
138                                for (String mode : calculators.keySet()) {
139                                        if (acg.getAccessibilityMeasureType() == AccessibilityMeasureType.logSum) {
140                                                accessibilities.put(mode, (1/this.cnScoringGroup.getBrainExpBeta()) * Math.log(expSums.get(mode)));
141                                        } else if (acg.getAccessibilityMeasureType() == AccessibilityMeasureType.rawSum) {
142                                                // this was used by IVT within SustainCity. Not sure if we should maintain this; they could, after all, just exp the log results. kai, may'15
143                                                // The above comment is from the time when the switch "isUsingRawSumsWithoutLn" was a "special case". I think the question is
144                                                // now resolved as "rawSum" has become one of the "AccessibilityMeasureType" options aiming to provide a means to use
145                                                // other potentially useful measure types. dz, july'17
146                                                accessibilities.put(mode, expSums.get(mode));
147                                        } else if (acg.getAccessibilityMeasureType() == AccessibilityMeasureType.gravity) {
148                                                throw new IllegalArgumentException("This accessibility measure is not yet implemented.");
149                                        } else {
150                                                throw new IllegalArgumentException("No valid accessibility measure type chosen.");
151                                        }
152                                }
153                                
154                                for (FacilityDataExchangeInterface zoneDataExchangeInterface : this.zoneDataExchangeListeners) {
155                                        zoneDataExchangeInterface.setFacilityAccessibilities(origin, departureTime, accessibilities);
156                                }
157                        }
158                }
159                for (FacilityDataExchangeInterface zoneDataExchangeInterface : this.zoneDataExchangeListeners) {
160                        zoneDataExchangeInterface.finish();
161                }
162        }
163        
164        /**
165         * Aggregates disutilities Vjk to get from node j to all k that are attached to j and assign sum(Vjk) is to node j.
166         * 
167         *     j---k1 
168         *     |\
169         *     | \
170         *     k2 k3
171         */
172        private final AggregationObject[] aggregateOpportunities(final ActivityFacilities opportunities, Network network) {
173                // yyyy this method ignores the "capacities" of the facilities. kai, mar'14
174                // for now, we decided not to add "capacities" as it is not needed for current projects. dz, feb'16
175
176                LOG.info("Aggregating " + opportunities.getFacilities().size() + " opportunities with identical nearest node...");
177                Map<Id<Node>, AggregationObject> opportunityClusterMap = new ConcurrentHashMap<>();
178                ProgressBar progressBar = new ProgressBar(opportunities.getFacilities().size());
179
180                for (ActivityFacility opportunity : opportunities.getFacilities().values()) {
181                        progressBar.update();
182
183                        Node nearestNode = NetworkUtils.getNearestNode(network, opportunity.getCoord());
184                        double distance_m = NetworkUtils.getEuclideanDistance(opportunity.getCoord(), nearestNode.getCoord());
185                        
186                        // in MATSim this is [utils/h]: cnScoringGroup.getTravelingWalk_utils_hr() - cnScoringGroup.getPerforming_utils_hr()
187                        double walkBetaTT_utils_h = this.cnScoringGroup.getModes().get(TransportMode.walk).getMarginalUtilityOfTraveling()
188                                        - this.cnScoringGroup.getPerforming_utils_hr(); // default values: -12 = (-6.) - (6.)
189                        double VjkWalkTravelTime = walkBetaTT_utils_h * (distance_m / this.walkSpeed_m_h);
190                        // System.out.println("VjkWalkTravelTime = " + VjkWalkTravelTime);
191                        
192                        // in MATSim this is 0 !!! since getMonetaryDistanceCostRateWalk doesn't exist:
193                        double walkBetaTD_utils_m = cnScoringGroup.getModes().get(TransportMode.walk).getMarginalUtilityOfDistance(); // default value: 0.
194                        double VjkWalkDistance = walkBetaTD_utils_m * distance_m;
195                        // System.out.println("VjkWalkDistance = " + VjkWalkDistance);
196                        
197                        double expVjk = Math.exp(this.cnScoringGroup.getBrainExpBeta() * (VjkWalkTravelTime + VjkWalkDistance));
198
199                        // add Vjk to sum
200                        AggregationObject jco = opportunityClusterMap.get(nearestNode.getId()); // Why "jco"?
201                        if (jco == null) {
202                                jco = new AggregationObject(opportunity.getId(), null, null, nearestNode, 0.); // Important: Initialize with zero!
203                                // This is a bit counter-intuitive. The first opportunity is added twice to the aggregation object, but the first time with zero
204                                // "impact". Leave it as is for the time being as urbanSim code uses this, dz, july'17
205                                opportunityClusterMap.put(nearestNode.getId(), jco); 
206                        }
207                        if (acg.getUseOpportunityWeights()) {
208                                if (opportunity.getCustomAttributes().get("weight") == null) {
209                                        throw new RuntimeException("If option \"useOpportunityWeights\" is used, the facilities must have an attribute with key \"weight\"");
210                                } else {
211                                        double weight = (double) opportunity.getCustomAttributes().get("weight");
212                                        jco.addObject(opportunity.getId(), expVjk * Math.pow(weight, acg.getWeightExponent()));
213                                }
214                        } else {
215                                jco.addObject(opportunity.getId(), expVjk);
216                        }
217//                      LOG.info("--- numberOfObjects = " + jco.getNumberOfObjects() + " --- objectIds = " + jco.getObjectIds());
218                }
219                LOG.warn("Here is an error in the aggregation! Fix this soon with other changes that alter test reults!");
220                LOG.info("Aggregated " + opportunities.getFacilities().size() + " opportunities to " + opportunityClusterMap.size() + " nodes.");
221                return opportunityClusterMap.values().toArray(new AggregationObject[opportunityClusterMap.size()]);
222        }

224        private Map<Id<Node>, ArrayList<ActivityFacility>> aggregateMeasurePointsWithSameNearestNode() {
225                Map<Id<Node>,ArrayList<ActivityFacility>> aggregatedOrigins = new ConcurrentHashMap<>();
226
227                Gbl.assertNotNull(measuringPoints);
228                Gbl.assertNotNull(measuringPoints.getFacilities()) ;
229                for (ActivityFacility measuringPoint : measuringPoints.getFacilities().values()) {
230
231                        // Determine nearest network node (from- or toNode) based on the link
232                        Node nearestNode = NetworkUtils.getCloserNodeOnLink(measuringPoint.getCoord(),  NetworkUtils.getNearestLinkExactly(network, measuringPoint.getCoord()));
233                        Id<Node> nearestNodeId = nearestNode.getId();
234
235                        // Create new entry if key does not exist!
236                        if(!aggregatedOrigins.containsKey(nearestNodeId)) {
237                                aggregatedOrigins.put(nearestNodeId, new ArrayList<ActivityFacility>());
238                        }
239                        // Assign measure point (origin) to it's nearest node
240                        aggregatedOrigins.get(nearestNodeId).add(measuringPoint);
241                }
242                LOG.info("Number of measuring points: " + measuringPoints.getFacilities().values().size());
243                LOG.info("Number of aggregated measuring points: " + aggregatedOrigins.size());
244                return aggregatedOrigins;
245        }
