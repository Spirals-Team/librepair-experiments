001package org.matsim.contrib.accessibility.interpolation;
002
003import org.apache.log4j.Logger;
004import org.matsim.api.core.v01.Coord;
005import org.matsim.contrib.accessibility.gis.SpatialGrid;
006
007/**
008 * Wrapper class for interpolation.
009 * Interpolates the value at one point from grid data as SpatialGrid.
010 * Offers bilinear interpolation, bicubic spline interpolation and inverse distance weighting.
011 * 
012 * Bilinear interpolation:
013 *      Uses linear spline interpolation with separation (first horizontal then vertical). Own implementation.
014 * 
015 * Bicubic spline interpolation:
016 *      Uses BicubicSplineInterpolator from apache (http://commons.apache.org/math/apidocs/org/apache/commons/math3/analysis/interpolation/BicubicSplineInterpolator.html).
017 *      Wave effects may occur.
018 * 
019 * Inverse distance weighting:
020 *  Uses inverse distance weighting for the 4 nearest known values. Own implementation.
021 *  Peaks and valleys occur.
022 *  For more information see e.g.: http://www.geography.hunter.cuny.edu/~jochen/GTECH361/lectures/lecture11/concepts/Inverse%20Distance%20Weighted.htm
023 *  or: http://gisbsc.gis-ma.org/GISBScL7/de/html/VL7a_V_lo7.html (in German).
024 * 
025 * @author tthunig
026 *
027 */
028public class Interpolation {
029
030        private static final Logger log = Logger.getLogger(Interpolation.class);
031        
032        public static final int BILINEAR = 0;
033        public static final int BICUBIC = 1;
034        public static final int INVERSE_DISTANCE_WEIGHTING = 2;
035        
036        private SpatialGrid sg = null;
037        private BiCubicInterpolator biCubicInterpolator = null;
038        private BiLinearInterpolator biLinearInterpolator = null;
039        private InverseDistanceWeighting inverseDistanceWeighting = null;
040        
041        /** exponent only necessary for the inverse distance weighting method **/
042        private double exp = 1.;
043        private int interpolationMethod = -1;
044        
045        /**
046         * Prepares interpolation with the selected interpolation method.
047         * 
048         * If inverse distance weighting is chosen, the exponent for weights will be the default value 1.
049         * 
050         * @param sg the SpatialGrid to interpolate
051         * @param method the interpolation method
052         */
053        public Interpolation(SpatialGrid sg, final int method ){
054                
055                this(sg, method, 1);
056        }
057        
058        /**
059         * Prepares interpolation with the selected interpolation method.
060         * 
061         * @param sg the SpatialGrid to interpolate
062         * @param method the interpolation method
063         * @param exp the exponent for weights. only considered if interpolation method is inverse distance weighting. standard values are one or two.
064         */
065        public Interpolation(SpatialGrid sg, final int method, final double exp ){
066                
067                this.sg = sg;
068                this.interpolationMethod = method;
069                this.exp = exp;
070                if(this.interpolationMethod == BILINEAR){
071                        log.info("Preparing bilinear interpolation ...");
072                        this.biLinearInterpolator = new BiLinearInterpolator(this.sg);
073                }
074                if(this.interpolationMethod == BICUBIC){
075                        log.info("Preparing bicubic interpolation ...");
076                        this.biCubicInterpolator = new BiCubicInterpolator(this.sg);
077                }
078                if(this.interpolationMethod == INVERSE_DISTANCE_WEIGHTING){
079                        log.info("Preparing interpolation with the inverse distance weighting method ...");
080                        this.inverseDistanceWeighting = new InverseDistanceWeighting(this.sg);
081                }
082        }
083
084        /**
085         * Interpolates the value at the given coordinate.
086         * 
087         * @param coord
088         * @return the interpolated value
089         */
090        public double interpolate(Coord coord){
091                if(sg != null && coord != null)
092                        return interpolate(coord.getX(), coord.getY());
093                log.warn("Either the spatial grid is not initialized or the coordinates are zero!");
094                return Double.NaN;
095        }
096        
097        /**
098         * Interpolates the value at the given coordinate (x,y).
099         * 
100         * @param x the x-coordinate
101         * @param y the y-coordinate
102         * @return the interpolated value at (x,y)
103         */
104        public double interpolate(double x, double y){
105                
106                switch(this.interpolationMethod){
107                case 0: return this.biLinearInterpolator.biLinearInterpolation(x, y);
108                case 1: return this.biCubicInterpolator.biCubicInterpolation(x, y);
109                case 2: return this.inverseDistanceWeighting.inverseDistanceWeighting(x, y, this.exp);          
110                }
111                return Double.NaN;
112        }
113
114        public int getInterpolationMethod() {
115                return interpolationMethod;
116        }       
117        
118}
