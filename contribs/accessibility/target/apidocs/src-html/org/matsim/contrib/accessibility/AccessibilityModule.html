058/**
059 * @author dziemke
060 */
061public final class AccessibilityModule extends AbstractModule {
062        private static final Logger LOG = Logger.getLogger(AccessibilityModule.class);
063
064        private List<SpatialGridDataExchangeInterface> spatialGridDataListeners = new ArrayList<>() ;
065        private List<FacilityDataExchangeInterface> facilityDataListeners = new ArrayList<>() ; 
066        private List<ActivityFacilities> additionalFacs = new ArrayList<>() ;
067        private String activityType;
068        private boolean pushing2Geoserver;
069        private String crs;
070        
071        /**
072         * If this class does not provide you with enough flexibility, do your own new AbstractModule(){...}, copy the install part from this class
073         * into that, and go from there. 
074         */
075        public AccessibilityModule() {
076        }
077        
078        
079        @Override
080        public void install() {
081                addControlerListenerBinding().toProvider(new Provider<ControlerListener>() {
082                        // yy not sure if this truly needs to be a provider.  kai, dec'16
083                        
084                        @Inject private Config config ;
085                        @Inject private Network network ;
086                        @Inject private Scenario scenario;
087
088                        @Inject (optional = true) PtMatrix ptMatrix = null; // Downstream code knows how to handle a null PtMatrix
089                        @Inject private Map<String,TravelDisutilityFactory> travelDisutilityFactories ;
090                        @Inject private Map<String,TravelTime> travelTimes ;
091                        
092                        @Inject TripRouter tripRouter ;
093                        
094                        @Override
095                        public ControlerListener get() {
096                                AccessibilityConfigGroup acg = ConfigUtils.addOrGetModule(scenario.getConfig(), AccessibilityConfigGroup.class);
097                                double cellSize_m = acg.getCellSizeCellBasedAccessibility();
098                                if (cellSize_m <= 0) {
099                                        LOG.error("Cell Size needs to be assigned a value greater than zero.");
100                                }
101                                crs = acg.getOutputCrs() ;
102                                
103                                ActivityFacilities opportunities = AccessibilityUtils.collectActivityFacilitiesWithOptionOfType(scenario, activityType) ;
104                                
105                                final BoundingBox boundingBox;
106                                final ActivityFacilities measuringPoints;
107                                
108                                if (acg.getAreaOfAccessibilityComputation() == AreaOfAccesssibilityComputation.fromShapeFile) {
109                                        Geometry boundary = GridUtils.getBoundary(acg.getShapeFileCellBasedAccessibility());
110                                        Envelope envelope = boundary.getEnvelopeInternal();
111                                        boundingBox = BoundingBox.createBoundingBox(envelope.getMinX(), envelope.getMinY(), envelope.getMaxX(), envelope.getMaxY());
112                                        measuringPoints = GridUtils.createGridLayerByGridSizeByShapeFileV2(boundary, cellSize_m);
113                                        LOG.info("Using shape file to determine the area for accessibility computation.");
114                                
115                                } else if (acg.getAreaOfAccessibilityComputation() == AreaOfAccesssibilityComputation.fromBoundingBox) {
116                                        boundingBox = BoundingBox.createBoundingBox(acg.getBoundingBoxLeft(), acg.getBoundingBoxBottom(), acg.getBoundingBoxRight(), acg.getBoundingBoxTop());
117                                        measuringPoints = GridUtils.createGridLayerByGridSizeByBoundingBoxV2(boundingBox, cellSize_m);
118                                        LOG.info("Using custom bounding box to determine the area for accessibility computation.");
119                                
120                                } else if (acg.getAreaOfAccessibilityComputation() == AreaOfAccesssibilityComputation.fromFile) {
121                                        boundingBox = BoundingBox.createBoundingBox(scenario.getNetwork());
122                                        LOG.info("Using the boundary of the network file to determine the area for accessibility computation.");
123                                        LOG.warn("This can lead to memory issues when the network is large and/or the cell size is too fine!");
124                                        Scenario measuringPointsSc = ScenarioUtils.createScenario(ConfigUtils.createConfig());
125                                        String measuringPointsFile = ConfigUtils.addOrGetModule(config, AccessibilityConfigGroup.class ).getMeasuringPointsFile() ;
126                                        new FacilitiesReaderMatsimV1(measuringPointsSc).readFile(measuringPointsFile);
127                                        measuringPoints = (ActivityFacilitiesImpl) AccessibilityUtils.collectActivityFacilitiesWithOptionOfType(measuringPointsSc, activityType);
128                                        LOG.info("Using measuring points from file: " + measuringPointsFile);
129                                
130                                } else {
131                                        boundingBox = BoundingBox.createBoundingBox(scenario.getNetwork());
132                                        measuringPoints = GridUtils.createGridLayerByGridSizeByBoundingBoxV2(boundingBox, cellSize_m) ;
133                                        LOG.info("Using the boundary of the network file to determine the area for accessibility computation.");
134                                        LOG.warn("This can lead to memory issues when the network is large and/or the cell size is too fine!");
135                                }
136                                
137                                LOG.warn("boundingBox = " + boundingBox);
138                                
139                                AccessibilityCalculator accessibilityCalculator = new AccessibilityCalculator(scenario, measuringPoints);
140                                for (Modes4Accessibility mode : acg.getIsComputingMode()) {
141                                        AccessibilityContributionCalculator calculator = null ;
142                                        switch(mode) {
143                                        case bike:
144                                                calculator = new ConstantSpeedAccessibilityExpContributionCalculator(mode.name(), config, network);
145                                                break;
146                                        case car: {
147                                                final TravelTime travelTime = travelTimes.get(mode.name());
148                                                Gbl.assertNotNull(travelTime);
149                                                final TravelDisutilityFactory travelDisutilityFactory = travelDisutilityFactories.get(mode.name());
150                                                calculator = new NetworkModeAccessibilityExpContributionCalculator(travelTime, travelDisutilityFactory, scenario) ;
151                                                break; }
152                                        case freespeed: {
153                                                final TravelDisutilityFactory travelDisutilityFactory = travelDisutilityFactories.get(TransportMode.car);
154                                                Gbl.assertNotNull(travelDisutilityFactory);
155                                                calculator = new NetworkModeAccessibilityExpContributionCalculator(new FreeSpeedTravelTime(), travelDisutilityFactory, scenario) ;
156                                                break; }
157                                        case walk:
158                                                calculator = new ConstantSpeedAccessibilityExpContributionCalculator(mode.name(), config, network);
159                                                break;
160                                        case matrixBasedPt:
161                                                calculator = new LeastCostPathCalculatorAccessibilityContributionCalculator(
162                                                                config.planCalcScore(), ptMatrix.asPathCalculator(config.planCalcScore()));
163                                                break;
164                                                //$CASES-OMITTED$
165                                        default:
166//                                              TravelTime timeCalculator = this.travelTimes.get( mode.toString() ) ;
167//                                              TravelDisutility travelDisutility = this.travelDisutilityFactories.get(mode.toString()).createTravelDisutility(timeCalculator) ;
168                                                calculator = new TripRouterAccessibilityContributionCalculator(mode.toString(), tripRouter, config.planCalcScore());
169                                        }
170                                        accessibilityCalculator.putAccessibilityContributionCalculator(mode.name(), calculator);
171                                }
172                                
173                                if (pushing2Geoserver == true) {
174                                        accessibilityCalculator.addFacilityDataExchangeListener(new GeoserverUpdater(crs,
175                                                        config.controler().getRunId() + "_" + activityType, acg.getCellSizeCellBasedAccessibility()));
176                                }
177
178                                GridBasedAccessibilityShutdownListenerV3 gbasl = new GridBasedAccessibilityShutdownListenerV3(accessibilityCalculator, 
179                                                opportunities, ptMatrix, scenario, boundingBox, cellSize_m);
180                                
181                                for (ActivityFacilities fac : additionalFacs) {
182                                        gbasl.addAdditionalFacilityData(fac);
183                                }
184
185                                for (SpatialGridDataExchangeInterface listener : spatialGridDataListeners) {
186                                        gbasl.addSpatialGridDataExchangeListener(listener) ;
187                                }
188                                
189                                for (FacilityDataExchangeInterface listener : facilityDataListeners) {
190                                        gbasl.addFacilityDataExchangeListener(listener);
191                                }
192                                
193                                gbasl.writeToSubdirectoryWithName(activityType);
194                                
195                                return gbasl;
196                        }
197                });
198        }
199        
200        public final void setPushing2Geoserver( boolean pushing2Geoserver ) {
201                this.pushing2Geoserver = pushing2Geoserver ;
202        }
203        
204        @Deprecated
205        public final void addSpatialGridDataExchangeListener(SpatialGridDataExchangeInterface listener) {
206                spatialGridDataListeners.add(listener) ;
207        }
208        
209        public final void addFacilityDataExchangeListener(FacilityDataExchangeInterface listener) {
210                this.facilityDataListeners.add(listener) ;
211        }
212
213        /**
214         * Add additional facility data that will generate an additional column for each (x,y,t)-Entry. The facilities are aggregated to
215         * the measurement points in downstream code.
216         */
217        public void addAdditionalFacilityData(ActivityFacilities facilities) {
218                additionalFacs.add(facilities) ;
219        }
220        
221        public void setConsideredActivityType(String activityType) {
222                // yyyy could be done via config (list of activities)
223                this.activityType = activityType ;
224        }
