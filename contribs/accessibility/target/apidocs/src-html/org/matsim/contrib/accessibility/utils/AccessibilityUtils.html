059        /**
060         * Collects all facilities of a given type that have been loaded to the sceanrio.
061         */
062        public static ActivityFacilities collectActivityFacilitiesWithOptionOfType(Scenario scenario, String activityOptionType) {
063                ActivityFacilities activityFacilities = FacilitiesUtils.createActivityFacilities(activityOptionType) ;
064                for (ActivityFacility facility : scenario.getActivityFacilities().getFacilities().values()) {
065                        if (activityOptionType == null) { // no activity option type for facility given, use all of them
066                                activityFacilities.addActivityFacility(facility);
067                        } else {
068                                for (ActivityOption option : facility.getActivityOptions().values()) {
069                                        if (option.getType().equals(activityOptionType)) {
070                                                activityFacilities.addActivityFacility(facility);
071                                        }
072                                }
073                        }
074                }
075                return activityFacilities;
076        }

078        /**
079         * Collects the types of all facilities that have been loaded to the scenario.
080         */
081        public static List<String> collectAllFacilityOptionTypes(Scenario scenario) {
082                List<String> activityOptionTypes = new ArrayList<>() ;
083                for (ActivityFacility facility : scenario.getActivityFacilities().getFacilities().values()) {
084                        for (ActivityOption option : facility.getActivityOptions().values()) {
085                                // collect all activity types that are contained within the provided facilities file
086                                if (!activityOptionTypes.contains(option.getType())) {
087                                        activityOptionTypes.add(option.getType()) ;
088                                }
089                        }
090                }
091                LOG.warn("The following activity option types where found within the activity facilities: " + activityOptionTypes);
092                return activityOptionTypes;
093        }

095        public static void combineDifferentActivityOptionTypes(final Scenario scenario, String combinedType, final List<String> activityOptionsToBeIncluded) {
096                ActivityOption markerOption = new ActivityOptionImpl(combinedType); 
097                
098                // Memorize all facilities that have certain activity options in a activity facilities container
099                final ActivityFacilities consideredFacilities = FacilitiesUtils.createActivityFacilities();
100                for (ActivityFacility facility : scenario.getActivityFacilities().getFacilities().values()) {
101                        for (ActivityOption option : facility.getActivityOptions().values()) {
102                                if (activityOptionsToBeIncluded.contains(option.getType())) {
103                                        // if (!option.getType().equals(FacilityTypes.HOME) && !option.getType().equals(FacilityTypes.WORK) && !option.getType().equals("minor")) {
104                                        if (!consideredFacilities.getFacilities().containsKey(facility.getId())) {
105                                                consideredFacilities.addActivityFacility(facility);
106                                        }
107                                }
108                        }
109                }
110                
111                // Add  marker option to facilities to be considered
112                for (ActivityFacility facility : consideredFacilities.getFacilities().values()) {
113                        facility.addActivityOption(markerOption);
114                }
115        }

117        public static final ActivityFacilities createFacilityForEachLink( Network network ) {
118                ActivityFacilities facilities = FacilitiesUtils.createActivityFacilities("LinkFacilities") ;
119                ActivityFacilitiesFactory ff = facilities.getFactory() ;
120                for ( Link link : network.getLinks().values() ) {
121                        ActivityFacility facility = ff.createActivityFacility( Id.create(link.getId(),ActivityFacility.class) , link.getCoord(), link.getId() ) ;
122                        facilities.addActivityFacility(facility);
123                }
124                return facilities ;
125        }

127        public static final ActivityFacilities createFacilityFromBuildingShapefile(String shapeFileName, String identifierCaption, String numberOfHouseholdsCaption) {
128                ShapeFileReader shapeFileReader = new ShapeFileReader();
129                Collection<SimpleFeature> features = shapeFileReader.readFileAndInitialize(shapeFileName);
130                
131                ActivityFacilities facilities = FacilitiesUtils.createActivityFacilities("DensitiyFacilities");
132                ActivityFacilitiesFactory aff = facilities.getFactory();
133                
134                for (SimpleFeature feature : features) {
135                        String featureId = (String) feature.getAttribute(identifierCaption);
136                        Integer numberOfHouseholds = Integer.parseInt((String) feature.getAttribute(numberOfHouseholdsCaption));
137                        Geometry geometry = (Geometry) feature.getDefaultGeometry();
138                        Coord coord = CoordUtils.createCoord(geometry.getCentroid().getX(), geometry.getCentroid().getY());
139                        
140                        for (int i = 0; i < numberOfHouseholds; i++) {
141                                ActivityFacility facility = aff.createActivityFacility(Id.create(featureId + "_" + i, ActivityFacility.class), coord);
142                                facilities.addActivityFacility(facility);
143                        }
144                }
145                return facilities ;
146        }

193        /**
194         * Calculates the sum of the values of a given list.
195         * 
196         * @param valueList
197         * @return sum
198         */
199        public static double calculateSum(List<Double> valueList) {
200                double sum = 0.;
201                for (double i : valueList) {
202                        sum = sum + i;
203                }
204                return sum;
205        }

207        /**
208         * Calculates Gini coefficient of the values of a given values. The Gini Coefficient is equals to the half of
209         * the relative mean absolute difference (RMD).
210         * 
211         * @see <a href="https://en.wikipedia.org/wiki/Gini_coefficient">
212         * @see <a href="https://en.wikipedia.org/wiki/Mean_absolute_difference#Relative_mean_absolute_difference">
213         * @param valueList
214         * @return giniCoefficient
215         */
216        public static double calculateGiniCoefficient(List<Double> valueList) {
217                int numberOfValues = valueList.size();
218                double sumOfValues = calculateSum(valueList);
219                double arithmeticMean = sumOfValues / numberOfValues;
220                
221                double sumOfAbsoluteDifferences = 0.;
222                for (double i : valueList) {
223                        for (double j : valueList) {
224                                double absoulteDifference = Math.abs( i - j );
225                                sumOfAbsoluteDifferences = sumOfAbsoluteDifferences + absoulteDifference;
226                        }
227                }
228                double giniCoefficient = sumOfAbsoluteDifferences / (2 * Math.pow(numberOfValues, 2) * arithmeticMean);
229                return giniCoefficient;
230        }

232        /**
233         * Creates facilities from plans. Note that a new additional facility is created for each activity.
234         * @param population
235         * @return
236         */
237        public static ActivityFacilities createFacilitiesFromPlans(Population population) {
238                ActivityFacilitiesFactory aff = new ActivityFacilitiesFactoryImpl();
239                ActivityFacilities facilities = FacilitiesUtils.createActivityFacilities();
240                
241                for(Person person : population.getPersons().values()) {
242                        for(Plan plan : person.getPlans()) {
243                                Id <Person> personId = person.getId();
244                                
245                                for (PlanElement planElement : plan.getPlanElements()) {
246                                        if (planElement instanceof Activity) {
247                                                Activity activity = (Activity) planElement;
248                                                
249                                                Coord coord= activity.getCoord();
250                                                if (coord == null) {
251                                                        throw new NullPointerException("Activity does not have any coordinates.");
252                                                }
253                                                
254                                                String activityType = activity.getType();
255                                                
256                                                // In case an agent visits the same activity location twice, create another activity facility with a modified ID
257                                                Integer i = 1;                                  
258                                                Id<ActivityFacility> facilityId = Id.create(activityType + "_" + personId.toString() + "_" + i.toString(), ActivityFacility.class);
259                                                while (facilities.getFacilities().containsKey(facilityId)) {
260                                                        i++;
261                                                        facilityId = Id.create(activityType + "_" + personId.toString() + "_" + i.toString(), ActivityFacility.class);
262                                                }
263
264                                                ActivityFacility facility = aff.createActivityFacility(facilityId, activity.getCoord());
265                                                
266                                                facility.addActivityOption(aff.createActivityOption(activityType));
267                                                facilities.addActivityFacility(facility);
268//                                              log.info("Created activity with option of type " + activityType + " and ID " + facilityId + ".");
269                                        }
270                                }
271                        }
272                }
273//              FacilitiesWriter facilitiesWriter = new FacilitiesWriter(facilities);
274//              facilitiesWriter.write("../../../public-svn/matsim/specifiy_location/...xml.gz");
275                return facilities;
276        }

278        public static String getDate() {
279                Calendar cal = Calendar.getInstance ();
280                int month = cal.get(Calendar.MONTH) + 1;
281                String monthStr = month + "";
282                if (month < 10)
283                        monthStr = "0" + month;
284                String date = cal.get(Calendar.YEAR) + "-" 
285                                + monthStr + "-" + cal.get(Calendar.DAY_OF_MONTH);
286                return date;
287        }
