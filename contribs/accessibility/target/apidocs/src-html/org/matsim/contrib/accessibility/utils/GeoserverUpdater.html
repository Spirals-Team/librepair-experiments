050public class GeoserverUpdater implements FacilityDataExchangeInterface {
051
052        static Logger LOG = Logger.getLogger(GeoserverUpdater.class);
053
054        private String crs;
055        private String name;
056        private long cellSize;
057
058        public GeoserverUpdater (String crs, String name, long cellSize) {
059                this.crs = crs;
060                this.name = name;
061                this.cellSize = cellSize;
062        }
063        
064        private Map<Tuple<ActivityFacility, Double>, Map<String,Double>> accessibilitiesMap = new HashMap<>() ;
065
066        @Override
067        public void setFacilityAccessibilities(ActivityFacility measurePoint, Double timeOfDay, Map<String, Double> accessibilities) {
068                accessibilitiesMap.put( new Tuple<>(measurePoint, timeOfDay), accessibilities ) ;
069        }
070
071        @Override
072        public void finish() {
073                GeometryFactory geometryFactory = new GeometryFactory();
074                SimpleFeatureTypeBuilder featureTypeBuilder = createFeatureTypeBuilder();
075                SimpleFeatureType featureType = featureTypeBuilder.buildFeatureType();
076                DefaultFeatureCollection featureCollection = createFeatureCollection(geometryFactory, featureType);
077                updateOnGeoserver(featureType, featureCollection);
078        }
079
080        private SimpleFeatureTypeBuilder createFeatureTypeBuilder() {
081                        SimpleFeatureTypeBuilder featureTypeBuilder = new SimpleFeatureTypeBuilder();
082                        featureTypeBuilder.setName(name);
083                        featureTypeBuilder.setCRS(MGC.getCRS(TransformationFactory.WGS84));
084                        featureTypeBuilder.add("the_geom", Polygon.class);
085                        featureTypeBuilder.add("id", Integer.class);
086                        featureTypeBuilder.add("time", Double.class);
087                        for (Modes4Accessibility mode : Modes4Accessibility.values()) {
088                                featureTypeBuilder.add(mode.toString(), Double.class);
089                        }
090        //              for (ActivityFacilities facilities : additionalFacilityData) {
091        //                      b.add(facilities.getName(), Double.class);
092        //              }
093                        // yyyyyy add population here
094                        return featureTypeBuilder;
095                }
096
097        private DefaultFeatureCollection createFeatureCollection(GeometryFactory geometryFactory, SimpleFeatureType featureType) {
098                LOG.info("Start creating features from accessibility data.");
099                DefaultFeatureCollection featureCollection = new DefaultFeatureCollection("internal", featureType);
100                SimpleFeatureBuilder featureBuilder = new SimpleFeatureBuilder(featureType);
101                
102                CoordinateTransformation transformation = TransformationFactory.getCoordinateTransformation(this.crs, TransformationFactory.WGS84);
103                int id = 0;
104                
105                for (Entry<Tuple<ActivityFacility, Double>, Map<String, Double>> entry : accessibilitiesMap.entrySet()) {
106                        id++;
107                        Coord coord = entry.getKey().getFirst().getCoord() ;
108                        Coordinate coord1 = MGC.coord2Coordinate(transformation.transform(CoordUtils.createCoord(coord.getX() - (cellSize / 2), coord.getY() - (cellSize / 2))));
109                        Coordinate coord2 = MGC.coord2Coordinate(transformation.transform(CoordUtils.createCoord(coord.getX() + (cellSize / 2), coord.getY() - (cellSize / 2))));
110                        Coordinate coord3 = MGC.coord2Coordinate(transformation.transform(CoordUtils.createCoord(coord.getX() + (cellSize / 2), coord.getY() + (cellSize / 2))));
111                        Coordinate coord4 = MGC.coord2Coordinate(transformation.transform(CoordUtils.createCoord(coord.getX() - (cellSize / 2), coord.getY() + (cellSize / 2))));
112                        
113                        featureBuilder.add(geometryFactory.createPolygon(new Coordinate[]{coord1, coord2, coord3, coord4, coord1}));
114                        featureBuilder.add(id);
115                        featureBuilder.add(entry.getKey().getSecond());
116                        
117                        for (Modes4Accessibility modeEnum : Modes4Accessibility.values()) {
118                                String mode = modeEnum.toString();
119                                Double accessibility = entry.getValue().get(mode);
120                                if (accessibility != null && !Double.isNaN(accessibility)) {
121                                        featureBuilder.add(accessibility);
122                                } else {
123                                        featureBuilder.add(null);
124                                }
125                        }
126                        // yyyyyy write population density here. Probably not aggregated to grid.
127
128                        SimpleFeature feature = featureBuilder.buildFeature(null);
129                        featureCollection.add(feature);
130                }
131                LOG.info("Finished creating features from accessibility data.");
132                return featureCollection;
133        }
134
135        private void updateOnGeoserver(SimpleFeatureType featureType, DefaultFeatureCollection featureCollection) {
136                LOG.info("Start pushing accessibility data into PostGIS database.");
137                try {
138                        Map<String,Object> params = new HashMap<>();
139                        params.put( "dbtype", "postgis");
140                        params.put( "host", "geo.vsp.tu-berlin.de");
141                        params.put( "port", 5432);
142                        params.put( "schema", "public");
143                        params.put( "database", "vspgeodb");
144                        params.put( "user", "vsppostgres");
145                        params.put( "passwd", "jafs30_A");
146                        DataStore dataStore = DataStoreFinder.getDataStore(params);
147                        System.out.println("dataStore = " + dataStore);
148                        
149                        // Remove schema in case it already exists
150                        try {
151                                dataStore.removeSchema(name);
152                        } catch (IllegalArgumentException e) {
153                                LOG.warn("Could not remove schema. Probably, it has not existed yet.");
154                        }
155                        
156                        dataStore.createSchema(featureType);
157                        SimpleFeatureStore featureStore = (SimpleFeatureStore) dataStore.getFeatureSource(name);
158                        featureStore.addFeatures(featureCollection);
159                        dataStore.dispose() ;
160                } catch (IOException e) {
161                        throw new RuntimeException(e);
162                }
163                LOG.info("Finished pushing accessibility data into PostGIS database.");
164                // Re-publish layer using the REST api (of geoserver; the above is the postgis db) if we want to automatically recompute the bounding box.  mz & kai, nov'15
165        }
