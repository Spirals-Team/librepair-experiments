031         * @param shapeFileName
032         * @return Geometry determines the scenario boundary for the accessibility measure
033         */
034        public static Geometry getBoundary(String shapeFileName){
035
036                try{
037                        // get boundaries of study area
038                        Set<SimpleFeature> featureSet = FeatureSHP.readFeatures(shapeFileName);
039                        // yyyy I find this quite terrible to have the reader hidden in here.  Now I cannot pass a shape file which
040                        // I may have gotten in some way, but need to first write it to file. kai, mar'14
041
042
043                        log.info("Extracting boundary of the shape file ...");
044                        Geometry boundary = (Geometry) featureSet.iterator().next().getDefaultGeometry();
045                        log.info("Done extracting boundary ...");
046
047                        if(featureSet.size() > 1){
048                                log.warn("The given shape file is not suitable for accessibility calculations.");
049                                log.warn("This means you have to provide a shape file that only contains the border of the study area without any further features, i.e. zones or fazes, are allowed!");
050                                log.warn("Replace the shape file in your UrbanSim configuration provided at \"travel_model_configuration/matsim4urbansim/controler_parameter/shape_file\"");
051                                log.warn("If the shape file contains features accessibilities will be computes for only feature, i.e. for one zone or faz.");
052                        }
053                        return boundary;
054                } catch (NullPointerException npe){
055                        npe.printStackTrace();
056                } catch (IOException io){
057                        io.printStackTrace();
058                        log.error("Geometry object containing the study area boundary shape is null !");
059                        System.exit(-1);
060                }
061                return null;
062        }

065        /**
066         * creates measuring points for accessibility computation
067         * @param boundary
068         * @param gridSize
069         * 
070         * @return ActivityFacilitiesImpl containing the coordinates for the measuring points 
071         */
072        public static ActivityFacilitiesImpl createGridLayerByGridSizeByShapeFileV2(Geometry boundary, double gridSize) {
073
074                log.info("Setting statring points for accessibility measure ...");
075
076                int skippedPoints = 0;
077                int setPoints = 0;
078
079                GeometryFactory factory = new GeometryFactory();
080
081                ActivityFacilitiesImpl measuringPoints = (ActivityFacilitiesImpl) FacilitiesUtils.createActivityFacilities("Accessibility measuring points");
082                Envelope env = boundary.getEnvelopeInternal();
083
084                // goes step by step from the min x and y coordinate to max x and y coordinate
085                for(double x = env.getMinX(); x < env.getMaxX(); x += gridSize) {
086
087                        for(double y = env.getMinY(); y < env.getMaxY(); y += gridSize) {
088
089                                // check first if cell centroid is within study area
090                                double centerX = x + (gridSize/2);
091                                double centerY = y + (gridSize/2);
092                                Point centroid = factory.createPoint(new Coordinate(centerX, centerY));
093
094                                if(boundary.contains(centroid)) {
095                                        Coord center = new Coord(centerX, centerY);
096                                        measuringPoints.createAndAddFacility(Id.create( setPoints , ActivityFacility.class), center);
097                                        setPoints++;
098                                }
099                                else skippedPoints++;
100                        }
101                }
102
103                log.info("Having " + setPoints + " inside the shape file boundary (and " + skippedPoints + " outside).");
104                log.info("Done with setting starting points!");
105                return measuringPoints;
106        }

112        /**
113         * creates measuring points for accessibility computation
114         * @param minX The smallest x coordinate (easting, longitude) expected
115         * @param minY The smallest y coordinate (northing, latitude) expected
116         * @param maxX The largest x coordinate (easting, longitude) expected
117         * @param maxY The largest y coordinate (northing, latitude) expected
118         * @param gridSize
119         * 
120         * @return ActivityFacilitiesImpl containing the coordinates for the measuring points 
121         */
122        public static ActivityFacilitiesImpl createGridLayerByGridSizeByBoundingBoxV2(double minX, double minY, double maxX, double maxY, double gridSize) {
123
124                log.info("Setting statring points for accessibility measure ...");
125
126                int skippedPoints = 0;
127                int setPoints = 0;
128
129                //              ActivityFacilitiesImpl measuringPoints = new ActivityFacilitiesImpl("accessibility measuring points");
130                ActivityFacilitiesImpl measuringPoints = (ActivityFacilitiesImpl) FacilitiesUtils.createActivityFacilities("accessibility measuring points");
131
132                // goes step by step from the min x and y coordinate to max x and y coordinate
133                for(double x = minX; x <maxX; x += gridSize) {
134
135                        for(double y = minY; y < maxY; y += gridSize) {
136
137                                // check first if cell centroid is within study area
138                                double centerX = x + (gridSize/2);
139                                double centerY = y + (gridSize/2);
140
141                                // check if x, y is within network boundary
142                                if (centerX <= maxX && centerX >= minX && 
143                                                centerY <= maxY && centerY >= minY) {
144
145                                        Coord center = new Coord(centerX, centerY);
146                                        measuringPoints.createAndAddFacility(Id.create( setPoints , ActivityFacility.class), center);
147                                        setPoints++;
148                                }
149                                else skippedPoints++;
150                        }
151                }
152
153                log.info("Having " + setPoints + " inside the shape file boundary (and " + skippedPoints + " outside).");
154                log.info("Done with setting starting points!");
155                return measuringPoints;
156        }

159        /**
160         * returns a spatial grid for a given geometry (e.g. shape file) with a given grid size
161         * @param boundary a boundary, e.g. from a shape file
162         * @param gridSize side length of the grid
163         * 
164         * @return SpatialGrid storing accessibility values
165         */
166        public static SpatialGrid createSpatialGridByShapeBoundary(Geometry boundary, double gridSize) {
167                Envelope env = boundary.getEnvelopeInternal();
168                double xMin = env.getMinX();
169                double xMax = env.getMaxX();
170                double yMin = env.getMinY();
171                double yMax = env.getMaxY();
172
173                return new SpatialGrid(xMin, yMin, xMax, yMax, gridSize, Double.NaN);
174        }

196        public static final void writeSpatialGrids( List<SpatialGrid> spatialGrids, String path ) {
197                // seems that we cannot re-use the accessibility write method since it plays games with the modes. kai, mar'14
198
199                final CSVWriter writer = new CSVWriter( path ) ;
200
201                writer.writeField("x");
202                writer.writeField("y");
203                for ( SpatialGrid spatialGrid : spatialGrids ) {
204                        writer.writeField( spatialGrid.getLabel() ) ; 
205                }
206                writer.writeNewLine();
207
208                final SpatialGrid spatialGrid = spatialGrids.get(0) ;
209                for ( double y = spatialGrid.getYmin() ; y <= spatialGrid.getYmax() ; y += spatialGrid.getResolution() ) {
210                        for ( double x = spatialGrid.getXmin() ; x <= spatialGrid.getXmax(); x+= spatialGrid.getResolution() ) {
211                                writer.writeField( x ) ;
212                                writer.writeField( y ) ;
213                                for ( SpatialGrid theSpatialGrid : spatialGrids ) {
214                                        writer.writeField( theSpatialGrid.getValue(x,y) ) ;
215                                }
216                                writer.writeNewLine();
217                        }
218                        writer.writeNewLine() ; // gnuplot pm3d scanlines
219                }
220                writer.close();
221
222        }

225        public static void aggregateFacilitiesIntoSpatialGrid(ActivityFacilities facilities, SpatialGrid spatialGridSum, SpatialGrid spatialGridCnt,
226                        SpatialGrid spatialGridAv ) {
227                for ( ActivityFacility fac : facilities.getFacilities().values() ) {
228                        Coord coord = fac.getCoord() ;
229
230                        final Object weight = fac.getCustomAttributes().get(WEIGHT);
231                        double value = 1 ;
232                        if ( weight != null ) {
233                                value = (Double) weight ;
234                        }
235                        //                              double value = fac.getActivityOptions().get("h").getCapacity() ; // infinity if undefined!!!
236
237                        spatialGridSum.addToValue(value, coord) ;
238                        spatialGridCnt.addToValue(1., coord) ;
239                }
240                if ( spatialGridAv!=null ) {
241                        double[][] cntMatrix = spatialGridCnt.getMatrix();
242                        for ( int ii=0 ; ii<cntMatrix.length ; ii++ ) {
243                                //                              log.warn("ii=" + ii );
244                                for ( int jj=0 ; jj<cntMatrix[ii].length ; jj++ ) {
245                                        double cnt = cntMatrix[ii][jj];
246                                        if ( cnt > 0. ) {
247                                                //                                              log.warn("jj=" + jj );
248                                                double sum = spatialGridSum.getMatrix()[ii][jj];
249                                                spatialGridAv.getMatrix()[ii][jj] = sum/cnt ;
250                                                log.warn("sum=" + sum + "; cnt=" + cnt + "; av=" + spatialGridAv.getMatrix()[ii][jj] ) ;
251                                        }
252                                }
253                        }
254                }
255        }
