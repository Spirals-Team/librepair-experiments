024package org.matsim.contrib.accessibility.osm;
025
026import java.io.BufferedWriter;
027import java.io.File;
028import java.io.FileNotFoundException;
029import java.io.IOException;
030import java.util.Map;
031
032import org.apache.log4j.Logger;
033import org.matsim.api.core.v01.Id;
034import org.matsim.core.utils.collections.QuadTree;
035import org.matsim.core.utils.geometry.CoordinateTransformation;
036import org.matsim.core.utils.io.IOUtils;
037import org.matsim.facilities.ActivityFacilities;
038import org.matsim.facilities.ActivityFacility;
039import org.matsim.facilities.FacilitiesUtils;
040import org.matsim.facilities.FacilitiesWriter;
041import org.matsim.facilities.Facility;
042import org.matsim.utils.objectattributes.ObjectAttributes;
043import org.matsim.utils.objectattributes.ObjectAttributesXmlWriter;
044import org.openstreetmap.osmosis.xml.common.CompressionMethod;
045import org.openstreetmap.osmosis.xml.v0_6.XmlReader;
046
047
048/**
049 * Parses all landuse and building tags from an OpenStreetMap file.
050 * 
051 * @author dziemke
052 * @see <a href="http://wiki.openstreetmap.org/wiki/Key:landuse">OpenStreetMap: Land Use</a>
053 */
054public class LandUseBuildingReader {
055        private final static Logger log = Logger.getLogger(LandUseBuildingReader.class);
056        private QuadTree<Id<ActivityFacility>> linkQT;
057        private ActivityFacilities landuse;
058        private ObjectAttributes amenityAttributes;
059        private final CoordinateTransformation ct;
060        private Map<String, String> osmLandUseToMatsimTypeMap;
061        private Map<String, String> osmBuildingToMatsimTypeMap;
062        
063        private double buildingTypeFromVicinityRange;
064        private String[] tagsToIgnoreBuildings;
065        
066
067        /**
068         * Constructing a LandUseBuildingReader to parse the OpenStreetMap landuse and Buildings.
069         * 
070         * @param file the path to the *.osm OpenStreetMap file;
071         * @param ct the (projected) coordinate reference system to which the 
072         *                WGS84 coordinates of OpenStreetMap will be converted to; and
073         * @param osmToMatsimTypeMap a mapping of OpenStreetMap
074         *                <a href="http://wiki.openstreetmap.org/wiki/Key:landuse">Land Use</a>
075         *                to MATSim activity types.
076         */
077        public LandUseBuildingReader(CoordinateTransformation ct, Map<String, String> osmLandUseToMatsimTypeMap,
078                        Map<String, String> osmBuildingToMatsimTypeMap, double buildingTypeFromVicinityRange,
079                        String[] tagsToIgnoreBuildings) {
080                log.info("Creating LandUseAndBuildingReader");
081                
082                this.ct = ct;
083                this.osmLandUseToMatsimTypeMap = osmLandUseToMatsimTypeMap;
084                this.osmBuildingToMatsimTypeMap = osmBuildingToMatsimTypeMap;
085                this.landuse = FacilitiesUtils.createActivityFacilities("OpenStreetMap landuse ???");
086                
087                this.buildingTypeFromVicinityRange = buildingTypeFromVicinityRange;
088                this.tagsToIgnoreBuildings = tagsToIgnoreBuildings;
089        }
090        
091        
092        public QuadTree<Id<ActivityFacility>> getQuadTree(){
093                return this.linkQT;
094        }
095        
096        
097        /**
098         * Parses a given <i>OpenStreetMap</i> file for land use.
099         * @param file the {@code *.osm} file to parse for land use
100         * @throws FileNotFoundException 
101         */
102        public void parseLandUseAndBuildings(String file) throws FileNotFoundException{
103                File f = new File(file);
104                if(!f.exists()){
105                        throw new FileNotFoundException("Could not find " + file);
106                }
107                LandUseBuildingSink landUseBuildingSink = new LandUseBuildingSink(this.ct,
108                                this.osmLandUseToMatsimTypeMap, this.osmBuildingToMatsimTypeMap,
109                                this.buildingTypeFromVicinityRange, this.tagsToIgnoreBuildings);
110                XmlReader xmlReader = new XmlReader(f, false, CompressionMethod.None);
111                xmlReader.setSink(landUseBuildingSink);
112                xmlReader.run();                
113                
114                this.landuse = landUseBuildingSink.getFacilities();
115                this.amenityAttributes = landUseBuildingSink.getFacilityAttributes();           
116        }
117
118        
119        /**
120         * Writes the facility coordinates so that it can be imported into QGis.
121         */
122        public void writeFacilityCoordinates(String file){
123                log.info("Writing facility coordinates to " + file);
124                BufferedWriter bw = IOUtils.getBufferedWriter(file);
125                try{
126                        bw.write("FacilityId,Long,Lat,Type");
127                        bw.newLine();
128                        for(Id<ActivityFacility> id : this.landuse.getFacilities().keySet()){
129                                ActivityFacility facility = this.landuse.getFacilities().get(id);
130                                bw.write(id.toString());
131                                bw.write(",");
132                                bw.write(String.format("%.0f,%.0f\n", facility.getCoord().getX(), facility.getCoord().getY()));
133                        }
134                } catch (IOException e) {
135                        throw new RuntimeException("Could not write to BufferedWriter " + file);
136                } finally{
137                        try {
138                                bw.close();
139                        } catch (IOException e) {
140                                throw new RuntimeException("Could not close BufferedWriter " + file);
141                        }
142                }
143                log.info("Done writing coordinates to file.");
144        }
145        
146        
147        /**
148         * Writes the amenities {@link Facility}s to file.
149         * @param file
150         */
151        public void writeFacilities(String file){
152                FacilitiesWriter fw = new FacilitiesWriter(this.landuse);
153                fw.write(file);
154        }
155        
156        
157        /**
158         * Writes the facility attributes to file.
159         * @param file
160         */
161        public void writeFacilityAttributes(String file){
162                ObjectAttributesXmlWriter ow = new ObjectAttributesXmlWriter(this.amenityAttributes);
163                ow.writeFile(file);
164        }
165        
166        
167        public ActivityFacilities getActivityFacilities(){
168                return this.landuse;
169        }
170        
171
172}
