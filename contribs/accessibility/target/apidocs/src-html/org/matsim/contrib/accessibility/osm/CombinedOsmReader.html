062/**
063 * Parses all landuse and building tags from an OpenStreetMap file.
064 * 
065 * @author dziemke
066 * @see <a href="http://wiki.openstreetmap.org/wiki/Key:landuse">OpenStreetMap: Land Use</a>
067 */
068public class CombinedOsmReader {
069        private final static Logger log = Logger.getLogger(CombinedOsmReader.class);
070        private QuadTree<Id<ActivityFacility>> linkQT;
071        private ActivityFacilities facilities;
072        private ObjectAttributes facilityAttributes;
073        private String idPrefix = "";
074        
075        private final String outputCRS;
076        
077        private Map<String, String> osmLandUseToMatsimTypeMap;
078        private Map<String, String> osmBuildingToMatsimTypeMap;
079        private Map<String, String> osmAmenityToMatsimTypeMap;
080        private Map<String, String> osmLeisureToMatsimTypeMap;
081        private Map<String, String> osmTourismToMatsimTypeMap;
082        private List<String> unmannedEntitiesList;
083        
084        private double buildingTypeFromVicinityRange;
085        // private String[] tagsToIgnoreBuildings;
086        
087
088        /**
089         * Constructing a LandUseBuildingReader to parse the OpenStreetMap landuse and Buildings.
090         * 
091         * @param file the path to the *.osm OpenStreetMap file;
092         * @param ct the (projected) coordinate reference system to which the 
093         *                WGS84 coordinates of OpenStreetMap will be converted to; and
094         * @param osmToMatsimTypeMap a mapping of OpenStreetMap
095         *                <a href="http://wiki.openstreetmap.org/wiki/Key:landuse">Land Use</a>
096         *                to MATSim activity types.
097         */
098        public CombinedOsmReader(String outputCRS, Map<String, String> osmLandUseToMatsimTypeMap,
099                        Map<String, String> osmBuildingToMatsimTypeMap, Map<String, String> osmAmenityToMatsimTypeMap,
100                        Map<String, String> osmLeisureToMatsimTypeMap, Map<String, String> osmTourismToMatsimTypeMap,
101                        List<String> unmannedEntitiesList, double buildingTypeFromVicinityRange)
102                        {
103                log.info("Creating CombinedOsmReader");
104                
105                this.outputCRS = outputCRS;
106                
107                this.osmLandUseToMatsimTypeMap = osmLandUseToMatsimTypeMap;
108                this.osmBuildingToMatsimTypeMap = osmBuildingToMatsimTypeMap;
109                this.osmAmenityToMatsimTypeMap = osmAmenityToMatsimTypeMap;
110                this.osmLeisureToMatsimTypeMap = osmLeisureToMatsimTypeMap;
111                this.osmTourismToMatsimTypeMap = osmTourismToMatsimTypeMap;
112                this.unmannedEntitiesList = unmannedEntitiesList;
113                
114                this.facilities = FacilitiesUtils.createActivityFacilities("OpenStreetMap landuse ???");
115                
116                this.buildingTypeFromVicinityRange = buildingTypeFromVicinityRange;
117        }
118        
119        
120        public QuadTree<Id<ActivityFacility>> getQuadTree(){
121                return this.linkQT;
122        }
123        
124        
125        /**
126         * Parses a given <i>OpenStreetMap</i> file for land use.
127         * @param osmFile the {@code *.osm} file to parse for land use
128         * @throws FileNotFoundException 
129         */
130        public void parseFile(String osmFile) throws FileNotFoundException{
131                File file = new File(osmFile);
132                if(!file.exists()){
133                        throw new FileNotFoundException("Could not find OSM file " + osmFile);
134                }
135                CombinedOsmSink combinedOsmSink = new CombinedOsmSink(this.outputCRS,
136                                this.osmLandUseToMatsimTypeMap, this.osmBuildingToMatsimTypeMap,
137                                this.osmAmenityToMatsimTypeMap, this.osmLeisureToMatsimTypeMap,
138                                this.osmTourismToMatsimTypeMap, this.unmannedEntitiesList,
139                                this.buildingTypeFromVicinityRange);
140                combinedOsmSink.setIdPrefix(idPrefix);
141                XmlReader xmlReader = new XmlReader(file, false, CompressionMethod.None);
142                xmlReader.setSink(combinedOsmSink);
143                xmlReader.run();                
144                
145                this.facilities = combinedOsmSink.getFacilities();
146                this.facilityAttributes = combinedOsmSink.getFacilityAttributes();              
147        }
148        
149        
150        public void parseFile(InputStream osmInputStream) {
151                CombinedOsmSink combinedOsmSink = new CombinedOsmSink(this.outputCRS,
152                                this.osmLandUseToMatsimTypeMap, this.osmBuildingToMatsimTypeMap,
153                                this.osmAmenityToMatsimTypeMap, this.osmLeisureToMatsimTypeMap,
154                                this.osmTourismToMatsimTypeMap, this.unmannedEntitiesList,
155                                this.buildingTypeFromVicinityRange);
156                combinedOsmSink.setIdPrefix(idPrefix);
157                org.matsim.contrib.accessibility.osm.XmlReader xmlReader = new org.matsim.contrib.accessibility.osm.XmlReader(osmInputStream, false, CompressionMethod.None);
158                xmlReader.setSink(combinedOsmSink);
159                xmlReader.run();
160                this.facilities = combinedOsmSink.getFacilities();
161                this.facilityAttributes = combinedOsmSink.getFacilityAttributes();              
162        }
163
164        
165        /**
166         * Writes the facility coordinates so that it can be imported into QGis.
167         */
168        public void writeFacilityCoordinates(String file){
169                log.info("Writing facility coordinates to " + file);
170                BufferedWriter bw = IOUtils.getBufferedWriter(file);
171                try{
172                        bw.write("FacilityId,Long,Lat,Type");
173                        bw.newLine();
174                        for(Id<ActivityFacility> id : this.facilities.getFacilities().keySet()){
175                                ActivityFacility facility = this.facilities.getFacilities().get(id);
176                                bw.write(id.toString());
177                                bw.write(",");
178                                bw.write(String.format("%.0f,%.0f\n", facility.getCoord().getX(), facility.getCoord().getY()));
179                        }
180                } catch (IOException e) {
181                        throw new RuntimeException("Could not write to BufferedWriter " + file);
182                } finally{
183                        try {
184                                bw.close();
185                        } catch (IOException e) {
186                                throw new RuntimeException("Could not close BufferedWriter " + file);
187                        }
188                }
189                log.info("Done writing coordinates to file.");
190        }

194         * Writes the facilities {@link Facility}s to file.
195         * @param facilitiesFile
196         */
197        public void writeFacilities(String facilitiesFile){
198                FacilitiesWriter fw = new FacilitiesWriter(this.facilities);
199                fw.write(facilitiesFile);
200        }
201        
202        
203        /**
204         * Writes the facility attributes to file.
205         * @param facilityAttributeFile
206         */
207        public void writeFacilityAttributes(String facilityAttributeFile){
208                ObjectAttributesXmlWriter ow = new ObjectAttributesXmlWriter(this.facilityAttributes);
209                ow.writeFile(facilityAttributeFile);
210        }
211        
212        
213        public ActivityFacilities getActivityFacilities(){
214                return this.facilities;
215        }

218         * Sets a prefix that will be used for each id. For example, useful when
219         * parsing from osm (as one of many sources) to use the prefix "osm_" so 
220         * that the resulting facilities can be identified accordingly.
221         * 
222         * @param prefix
223         */
224        public void setIdPrefix(String prefix){
225                this.idPrefix = prefix;
226        }
