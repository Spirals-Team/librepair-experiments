001package org.matsim.contrib.accessibility;
002
003import java.io.BufferedWriter;
004import java.io.IOException;
005
006import org.apache.log4j.Logger;
007import org.matsim.core.utils.io.IOUtils;
008
009public final class CSVWriter {
010        private static final Logger log = Logger.getLogger(CSVWriter.class);
011
012        public static final String FILE_NAME= "accessibilities.csv";
013        
014        private static final String SEPARATOR = ",";
015        // comma is the correct choice for excel.  But gnuplot cannot deal with comma, needs "\t"!
016
017        private BufferedWriter writer ;
018
019        private boolean first = true ;
020
021        /**
022         * writes the header of accessibility data csv file
023         */
024        public CSVWriter(String path){
025                log.info("Initializing  ...");
026                try {
027                writer = IOUtils.getBufferedWriter( path );
028                } catch ( Exception ee ) {
029                        ee.printStackTrace();
030                        throw new RuntimeException("writer could not be instantiated") ;
031                }
032
033                if ( writer==null ) {
034                        throw new RuntimeException( "writer is null") ;
035                }
036
037                log.info("... done!");
038        }
039
040        public final void writeField( double val ) {
041                try {
042                        if ( first ) {
043                                writer.write( val + "" ) ;
044                                first = false ;
045                        } else {
046                                writer.write( SEPARATOR + val ) ;
047                        }
048                } catch (IOException e) {
049                        e.printStackTrace();
050                        throw new RuntimeException("could not write");
051                }
052        }
053        
054        public final void writeField( String val ) {
055                try {
056                        if ( first ) {
057                                writer.write( val ) ;
058                                first = false ;
059                        } else {
060                                writer.write( SEPARATOR + val ) ;
061                        }
062                } catch (IOException e) {
063                        e.printStackTrace();
064                        throw new RuntimeException("could not write");
065                }
066        }
067
068        public final void writeNewLine() {
069                try {
070                        writer.newLine() ;
071                } catch (IOException e) {
072                        e.printStackTrace();
073                        throw new RuntimeException("i/o failure") ;
074                }
075                first = true ;
076        }
077
078        /**
079         * finalize and close csv file
080         */
081        public final void close(){
082                try {
083                        log.info("Closing ...");
084                        assert(writer != null);
085                        writer.flush();
086                        writer.close();
087                        log.info("... done!");
088                } catch (IOException e) {
089                        e.printStackTrace();
090                }       
091        }
092        
093}
