001package org.matsim.contrib.accessibility.utils;
002
003import org.apache.log4j.Logger;
004import org.matsim.api.core.v01.Coord;
005import org.matsim.api.core.v01.network.Link;
006import org.matsim.api.core.v01.network.Node;
007import org.matsim.core.network.NetworkUtils;
008import org.matsim.core.utils.geometry.CoordUtils;
009
010public final class NetworkUtil {
011        private NetworkUtil(){} // do not instantiate
012
013        @SuppressWarnings("unused")
014        private static final Logger log = Logger.getLogger(NetworkUtil.class);
015        
016        /**
017         * This method calculates the distance between a point and a node via a link (which should contains
018         * the node) as the sum of
019         * <ul>
020         * <li> the orthogonal distance between the point and the link and
021         * <li> the distance between the intersection point of the orthogonal
022         * projection (from the point to the link) to the node.
023         * </ul>
024         * If the orthogonal projection of the point to the line does not intersects the link, the method
025         * returns the Euclidean distance between the point and the node.
026         */
027        public static Distances getDistances2NodeViaGivenLink(Coord coord, Link link, Node destinationNode){
028                Coord intersection = CoordUtils.orthogonalProjectionOnLineSegment(link.getFromNode().getCoord(), link.getToNode().getCoord(), coord);
029                
030                Distances distances = new Distances();
031                
032                double distanceCoord2Intersection = NetworkUtils.getEuclideanDistance(coord, intersection);
033                double distanceIntersection2Node = NetworkUtils.getEuclideanDistance(intersection, destinationNode.getCoord());
034
035                distances.setDistanceCoord2Intersection(distanceCoord2Intersection);
036                distances.setDistanceIntersetion2Node(distanceIntersection2Node);
037                return distances;
038        }
039}
