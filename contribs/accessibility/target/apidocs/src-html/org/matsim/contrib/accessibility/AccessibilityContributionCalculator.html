001package org.matsim.contrib.accessibility;
002
003import org.matsim.api.core.v01.network.Node;
004import org.matsim.contrib.accessibility.utils.AggregationObject;
005import org.matsim.facilities.ActivityFacility;
006
007/**
008 * Interface to provide a means to compute the utility of a given
009 * origin destination, for a given mode.
010 * This includes travel time estimation and utility formulation.
011 *
012 * @author thibautd
013 */
014// It might make sense to split this further, into travel time estimation
015// and utility estimation. Ideally, one would actually not need any additional
016// interface compared to what MATSim provides. Not there yet [td, june 15]
017public interface AccessibilityContributionCalculator {
018        /**
019         * Provided for performance purpose.
020         * The accessibility listener(s) aggregate first all "measuring points"
021         * according to the node they are associated to on the network,
022         * allowing to perform some preprocessing (for instance pre-computing
023         * a least-cost path tree).
024         *
025         * @param fromNode the Node on the network to which will be associated the next
026         *                 examined opportunities.
027         * @param departureTime TODO
028         */
029        void notifyNewOriginNode( Node fromNode, Double departureTime );
030
031        /**
032         * estimates the contribution of a given opportunity to the accessibility metric,
033         * defined as an origin point (provided as a facility) and a group of facilities.
034         * "Contribution" is understood here as what is summed, that is, for the logsum,
035         * the exponential of the utility, scaled by the logit scale parameter.
036         *
037         * @param origin the origin point
038         * @param destination the opportunities at the destination
039         * @param departureTime TODO
040         * @return the utility of the OD pair, to be included in the logsum
041         */
042        double computeContributionOfOpportunity(ActivityFacility origin, AggregationObject destination, Double departureTime);
043        // yyyy I am somewhat sceptic if we tryly need both "fromNode" (above) and origin.
044        // yyyy And I am quite confident that we do not need the departure time twice.
045}
