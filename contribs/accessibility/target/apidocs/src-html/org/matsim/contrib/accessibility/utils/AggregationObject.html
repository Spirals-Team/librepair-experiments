001package org.matsim.contrib.accessibility.utils;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import org.matsim.api.core.v01.Id;
007import org.matsim.api.core.v01.network.Node;
008
009public final class AggregationObject {
010        
011        private List<Id> objectIdList = null; // either job or person id
012        private Id zoneID;
013        private Id parcelID;
014        private Node nearestNode;
015        
016        private double sum = 0. ;
017        private double cnt = 0. ;
018        
019        // not used anywhere, dz, jul'17
020//      public AggregationObject(Id objectID, Id parcelId, Id zoneId, double value){
021//              this(objectID, parcelId, zoneId, null, value);
022//      }
023        
024        public AggregationObject(Id objectID, Id parcelId, Id zoneId, Node nearestNode, double value){
025                if(this.objectIdList == null)
026                        this.objectIdList = new ArrayList<Id>();
027                this.objectIdList.add( objectID );
028                this.parcelID = parcelId;
029                this.zoneID = zoneId;
030                this.nearestNode = nearestNode;
031                
032                this.sum = value;
033        }
034        
035        public void setNearestNode(Node nearestNode){
036                this.nearestNode = nearestNode;
037        }
038        
039        public void addObject(Id objectID, double value){
040                this.objectIdList.add( objectID );
041                this.sum += value;
042                this.cnt ++ ; // could be generalized into a weight.  kai, mar'14
043        }
044        
045        public Node getNearestNode(){
046                return this.nearestNode;
047        }
048        
049        public int getNumberOfObjects(){
050                return this.objectIdList.size();
051        }
052        
053        public List<Id> getObjectIds(){
054                return this.objectIdList;
055        }
056        
057        public Id getParcelID(){
058                return this.parcelID;
059        }
060        
061        public Id getZoneID(){
062                return this.zoneID;
063        }
064        
065        public double getSum(){
066                return this.sum;
067        }
068        public double getCnt() {
069                return this.cnt ;
070        }
071}
