019package org.matsim.contrib.accessibility.utils;
020
021import java.io.BufferedWriter;
022import java.util.List;
023
024import org.apache.log4j.Logger;
025import org.matsim.contrib.accessibility.Modes4Accessibility;
026import org.matsim.core.config.Config;
027import org.matsim.core.utils.io.IOUtils;
028import org.matsim.core.utils.misc.ExeRunner;
029
030/**
031 * @author nagel
032 *
033 */
034public final class GnuplotScriptWriter {
035        private static final Logger log = Logger.getLogger( GnuplotScriptWriter.class ) ;
036        
037        private GnuplotScriptWriter() {} // do not instantiate
038
039        public static void createGnuplotScript(Config config, List<String> activityTypes) {
040                        // two loops over activity types and modes to produce one gnuplot script for each combination
041                        for (String actType : activityTypes) {
042                                for ( Modes4Accessibility mode : Modes4Accessibility.values()) {
043        //                              if ( !actType.equals("w") ) {
044        //                                      log.error("skipping everything except work for debugging purposes; remove in production code. kai, feb'14") ;
045        //                                      continue ;
046        //                              }
047                                        if ( !mode.equals(Modes4Accessibility.freespeed) ) {
048                                                log.error("skipping everything except freespeed for debugging purposes; remove in production code. dz, nov'14") ;
049                                                continue ;
050                                        }
051                                        
052                                        // the following might be made configurable if this is ever re-used:
053                                        Boolean doPopulationWeightedPlot = true ;
054                                        Boolean doNonPopulationWeightedPlot = true ;
055        
056                                        
057                                        if (doPopulationWeightedPlot == false && doNonPopulationWeightedPlot == false) {
058                                                throw new RuntimeException("At least one plot (pop-weighted or non-pop-weighted) needs to be activated.");
059                                        }
060                                        
061                                        // produce gnuplot scripts
062                                        try {
063                                                BufferedWriter writer = IOUtils.getBufferedWriter( config.controler().getOutputDirectory() + "/" + actType 
064                                                                + "/" + mode + "/t.gpl" ) ;
065                                                // pm3d is an splot style for drawing palette-mapped 3d and 4d data as color/gray maps and surfaces (docu p.134)
066                                                // To use pm3d coloring to generate a two-dimensional plot rather than a 3D surface, use set view map
067                                                // or set pm3d map (p.135)
068                                                writer.write("set pm3d map\n");
069                                                
070                                                // flush { begin | center | end }
071                                                writer.write("set pm3d flush begin\n") ;
072                                                
073                                                // corners2color { mean|geomean|median|min|max|c1|c2|c3|c4 }
074                                                writer.write("set pm3d corners2color c1\n");
075                                                
076                                                // The set style data command changes the default plotting style for data plots (docu p.148)
077                                                // There are many plotting styles available in gnuplot. They are listed alphabetically below. The commands
078                                                // set style data and set style function change the default plotting style for subsequent plot and splot
079                                                // commands. (docu p.42)
080                                                writer.write("set style data pm3d\n");
081                                                
082                                                // Palette is a color storage for use by pm3d, filled color contours or polygons, color histograms, color gradient
083                                                // background, and whatever it is or it will be implemented (docu p.137)
084                                                // Gray-to-rgb mapping can be manually set by use of palette defined: A color gradient is defined and used
085                                                // to give the rgb values. Such a gradient is a piecewise linear mapping from gray values in [0,1] to the RGB
086                                                // space [0,1]x[0,1]x[0,1]. You must specify the gray values and the corresponding RGB values between which
087                                                // linear interpolation will be done (docu p.139)
088                                                writer.write("set palette defined ( 0. '#ff0000', 0.82 '#ff0000', 0.86 '#00ff00', 0.9 '#0000ff', 1.0 '#0000ff' )\n");
089                                                
090                                                // The set zrange command sets the range that will be displayed on the z axis. The zrange is used only by
091                                                // splot and is ignored by plot (docu p.166)
092                                                writer.write("#set zrange [-40:10]\n");
093                                                
094                                                // The set cbrange command sets the range of values which are colored using the current palette by styles
095                                                // with pm3d, with image and with palette. Values outside of the color range use color of the nearest
096                                                // extreme (docu p.167)
097                                                writer.write("#set cbrange [-0:10]\n");
098                                                
099                                                // gnuplot supports many different graphics devices. Use set terminal to tell gnuplot what kind of output
100                                                // to generate (docu p.152)
101                                                // This terminal produces files in the Adobe Portable Document Format (PDF), useable for printing or display
102                                                // with tools like Acrobat Reader (docu p.206)
103                                                //writer.write("set term pdf size 25cm,20cm\n");
104                                                writer.write("set term pdf font 'Helvetica,6' size 25cm,20cm\n");
105                                                
106                                                // The set view command sets the viewing angle for splots. It controls how the 3D coordinates of the plot are
107                                                // mapped into the 2D screen space. It provides controls for both rotation and scaling of the plotted data, but
108                                                // supports orthographic projections only (docu p.156)
109                                                writer.write("#set view 45,30;\n");
110                                                writer.write("\n") ;
111                                                
112                                                // see docu p.137
113                                                writer.write("# set palette model HSV functions gray, 1, 1\n");
114                                                writer.write("\n");
115                                                                                
116                                                // New user-defined variables and functions of one through twelve variables may be declared and used anywhere,
117                                                // including on the plot command itself (docu p.30)
118                                                // define minimum and maximum functions
119                                                writer.write("min(a,b) = (a < b) ? a : b\n");
120                                                writer.write("max(a,b) = (a < b) ? b : a\n");
121                                                
122                                                // define two variables
123                                                writer.write("accmin=3 ; # accessibilities below this are red\n");
124                                                writer.write("accmax=9 ; # accessibilities above this are blue. max is around 12\n");
125                                                
126                                                // define a function to determine the shade of gray. gray(acc) will have a value from 0 through 1
127                                                // Acc values equal to or below accmin lead to 0; acc values equal to or above accmax lead to 1
128                                                writer.write("gray(acc) = 2.*min( 1, max(0 , (acc-accmin)/(accmax-accmin) ) ) ;\n") ;
129                                                writer.write("# I have no idea why this needs to be multiplied by 2. kai, feb'14\n") ;
130                                                writer.write("\n") ;
131                                                
132                                                // consider population density
133                                                // define two variables
134                                                writer.write("densmax=1000 ; # 2726 is the maximum value in NMB\n") ;
135                                                writer.write("maxwhite = 240 ; # 255 means that we go all the way to full white\n") ;
136                                                
137                                                while (doPopulationWeightedPlot == true || doNonPopulationWeightedPlot == true) {
138                                                        // define a function that gets the higher the smaller the population density
139                                                        // maxwhite*1 for pop dens = 0; maxwhite*0 for pop dens higher than densmax
140                                                        if (doPopulationWeightedPlot == true) {
141                                                                writer.write("val(dens) = max(0,maxwhite*(densmax-dens)/densmax) ;\n") ;
142                                                                
143                                                                // By default, screens are displayed to the standard output. The set output command redirects the display
144                                                                // to the specified file or device (docu p.132)
145                                                                writer.write("set out 'accessibility-pop-weighted.pdf'\n");
146                                                                
147                                                                // The set title command produces a plot title that is centered at the top of the plot. set title is a special
148                                                                // case of set label (docu p.155)
149                                                                //writer.write("set title 'accessibility to " + actType + " by " + mode + " (population-weighted)'\n") ;
150                                                                
151                                                                doPopulationWeightedPlot = false;
152                                                                
153                                                        } else if (doNonPopulationWeightedPlot == true) {
154                                                                writer.write("val(dens) = 0. ; # unset this comment to get the non-pop-weighted version (for paper)\n");
155                                                                writer.write("set out 'accessibility.pdf'\n");
156                                                                //writer.write("set title 'accessibility to " + actType + " by " + mode + " (non-population-weighted)'\n") ;
157                                                                
158                                                                doNonPopulationWeightedPlot = false;
159                                                        }
160                                                        writer.write("\n") ;
161                                                        
162                                                        // define three color functions
163                                                        // so far not clear to me where the functions come from, dz mai14
164                                                        writer.write("blue(acc,dens) = min(255,  val(dens)+255*max(0,1.-2.*gray(acc))  ) ;\n") ;
165                                                        writer.write("green(acc,dens) = min(255,  val(dens)+255*max(0,min(2.*gray(acc),2.-2.*gray(acc)))  ) ;\n") ;
166                                                        writer.write("red(acc,dens) = min(255,  val(dens)+255*max(0,2.*gray(acc)-1)  ) ;\n") ;
167                                                        writer.write("\n") ;
168                                                        
169                                                        // define color function based on accessibility and population density
170                                                        // int(x) = integer part of x, truncated toward zero (docu p.26)
171                                                        // Example: rgb(r,g,b) = 65536 * int(r) + 256 * int(g) + int(b) (docu p.35)
172                                                        // Why are blue and red swapped here???
173                                                        writer.write("rgb(acc,dens) = 256*256*int(blue(acc,dens))+256*int(green(acc,dens))+int(red(acc,dens)) ;\n") ;
174                                                        writer.write("\n") ;
175                                                        writer.write("unset colorbox ; # useless with lc rgb variable\n") ;
176                                                        writer.write("\n") ;  // end new
177                                                        
178                                                        // plot csv file based on three values; first two are coordinates; third takes into account accessibility and 
179                                                        // population density and is calculated based on above-defined rgb formula
180                                                        // The "lc rgbcolor variable" tells the program to read RGB color information for each line in the data file.
181                                                        // This requires a corresponding additional column in the using specifier. The extra column is interpreted as a
182                                                        // 24-bit packed RGB triple (docu p.35)
183                                                        // writer.write("splot \"<awk '!/access/{print $0}' doAccessibilityTest/w/accessibilities.csv\" u 1:2:(rgb($3,$8)) lc rgb variable\n") ;
184                                                        writer.write("splot \"<awk '!/access/{print $0}' accessibilities.csv\" u 1:2:(rgb($3,$8)) lc rgb variable\n");
185                                                        // writer.write("splot \"accessibilities.csv\" u 1:2:(rgb($3,$8)) lc rgb variable\n");
186                                                        // writer.write("splot \"accessibilities.csv\" u 1:2:(rgb($3,$8)) notitle lc rgb variable\n");
187                                                }
188                                                        
189                                                writer.close();
190                                        } catch (Exception ee ) {
191                                                ee.printStackTrace(); 
192                                                throw new RuntimeException( "writing t.gpl did not work") ;
193                                        }
194                                        
195                                        
196                                        // start running gnuplot with above-created script
197                                        
198                                        // If working on a Windows system, it is important that the environment variable PATH includes the gnuplot folder
199                                        // so that gnuplot can be run from the folder where the data is stored (otherwise the relative paths won't work)
200                                        String cmd = "gnuplot t.gpl";
201                                                                
202                                        // Doesn't work if root of directory is not passed
203                                        String stdoutFileName = config.controler().getOutputDirectory() + actType + "/" + mode + "/gnuplot.log";
204                                        // String stdoutFileName = config.controler().getOutputDirectory() + "gnuplot.log";
205                                        int timeout = 99999;
206                                        
207                                        // 4th argument = workingDirectory. Since we are working with relative paths, workingDirectory needs to be passed.
208                                        ExeRunner.run(cmd, stdoutFileName, timeout, config.controler().getOutputDirectory() + actType + "/" + mode);
209                                }
210                        }
211                }
