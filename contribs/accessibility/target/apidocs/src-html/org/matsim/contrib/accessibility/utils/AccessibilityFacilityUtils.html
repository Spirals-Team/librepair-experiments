040        public static ActivityFacilities createFacilites(InputStream osmInputStream, String outputCRS, double buildingTypeFromVicinityRange) {
041                LOG.info("Parsing facility information from OpenStreetMap input stream.");
042
043                CombinedOsmReader combinedOsmReader = new CombinedOsmReader(
044                        outputCRS,
045                        AccessibilityFacilityUtils.buildOsmLandUseToMatsimTypeMap(),
046                        AccessibilityFacilityUtils.buildOsmBuildingToMatsimTypeMap(),
047                        buildOsmAmenityToMatsimTypeMap(), //local
048                        AccessibilityFacilityUtils.buildOsmLeisureToMatsimTypeMap(),
049                        AccessibilityFacilityUtils.buildOsmTourismToMatsimTypeMap(),
050                        AccessibilityFacilityUtils.buildUnmannedEntitiesList(),
051                        buildingTypeFromVicinityRange);
052                combinedOsmReader.parseFile(osmInputStream);
053                ActivityFacilities facilities = combinedOsmReader.getActivityFacilities();
054
055                return facilities;
056        }
