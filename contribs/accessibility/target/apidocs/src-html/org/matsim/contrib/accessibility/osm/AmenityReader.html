024package org.matsim.contrib.accessibility.osm;
025
026import java.io.BufferedWriter;
027import java.io.File;
028import java.io.FileNotFoundException;
029import java.io.IOException;
030import java.util.Map;
031
032import org.apache.log4j.Logger;
033import org.matsim.api.core.v01.Id;
034import org.matsim.core.utils.collections.QuadTree;
035import org.matsim.core.utils.geometry.CoordinateTransformation;
036import org.matsim.core.utils.io.IOUtils;
037import org.matsim.facilities.ActivityFacilities;
038import org.matsim.facilities.ActivityFacility;
039import org.matsim.facilities.FacilitiesUtils;
040import org.matsim.facilities.FacilitiesWriter;
041import org.matsim.facilities.Facility;
042import org.matsim.utils.objectattributes.ObjectAttributes;
043import org.matsim.utils.objectattributes.ObjectAttributesXmlWriter;
044import org.openstreetmap.osmosis.xml.common.CompressionMethod;
045import org.openstreetmap.osmosis.xml.v0_6.XmlReader;

048/**
049 * Parses all amenities from an OpenStreetMap file.
050 * 
051 * @author jwjoubert
052 * @see <a href="http://wiki.openstreetmap.org/wiki/Key:amenity">OpenStreetMap: Amenity</a>
053 */
054public class AmenityReader {
055        private final static Logger log = Logger.getLogger(AmenityReader.class);
056        private QuadTree<Id<ActivityFacility>> linkQT;
057        private ActivityFacilities amenities;
058        private ObjectAttributes amenityAttributes;
059        private final CoordinateTransformation ct;
060        private Map<String, String> osmToMatsimTypeMap;
061        
062
063        /**
064         * Constructing an amenity reader to parse the OpenStreetMap amenities.
065         * 
066         * @param file the path to the *.osm OpenStreetMap file;
067         * @param ct the (projected) coordinate reference system to which the 
068         *                WGS84 coordinates of OpenStreetMap will be converted to; and
069         * @param osmToMatsimTypeMap a mapping of OpenStreetMap
070         *                <a href="http://wiki.openstreetmap.org/wiki/Key:amenity">Amenity values</a>
071         *                to MATSim activity types.
072         */
073        public AmenityReader(String file, CoordinateTransformation ct, 
074                        Map<String, String> osmToMatsimTypeMap) {
075                log.info("Creating amenity reader");
076                
077                this.ct = ct;
078                this.osmToMatsimTypeMap = osmToMatsimTypeMap;
079                this.amenities = FacilitiesUtils.createActivityFacilities("OpenStreetMap amenities");
080                this.amenityAttributes = new ObjectAttributes();
081        }
082        
083        
084        public QuadTree<Id<ActivityFacility>> getQuadTree(){
085                return this.linkQT;
086        }
087        
088        
089        /**
090         * Parses a given <i>OpenStreetMap</i> file for amenity facilities.
091         * @param file the {@code *.osm} file to parse for amenity facilities.
092         * @throws FileNotFoundException 
093         */
094        public void parseAmenity(String file) throws FileNotFoundException{
095                File f = new File(file);
096                if(!f.exists()){
097                        throw new FileNotFoundException("Could not find " + file);
098                }
099                AmenitySink mes = new AmenitySink(this.ct, this.osmToMatsimTypeMap);
100                XmlReader xr = new XmlReader(f, false, CompressionMethod.None);
101                xr.setSink(mes);
102                xr.run();               
103                
104                this.amenities = mes.getFacilities();
105                this.amenityAttributes = mes.getFacilityAttributes();           
106        }
107
108
109        /**
110         * Writes the facility coordinates so that it can be imported into QGis.
111         */
112        public void writeFacilityCoordinates(String file){
113                log.info("Writing facility coordinates to " + file);
114                BufferedWriter bw = IOUtils.getBufferedWriter(file);
115                try{
116                        bw.write("FacilityId,Long,Lat,Type");
117                        bw.newLine();
118                        for(Id<ActivityFacility> id : this.amenities.getFacilities().keySet()){
119                                ActivityFacility facility = this.amenities.getFacilities().get(id);
120                                bw.write(id.toString());
121                                bw.write(",");
122                                bw.write(String.format("%.0f,%.0f\n", facility.getCoord().getX(), facility.getCoord().getY()));
123                        }
124                } catch (IOException e) {
125                        throw new RuntimeException("Could not write to BufferedWriter " + file);
126                } finally{
127                        try {
128                                bw.close();
129                        } catch (IOException e) {
130                                throw new RuntimeException("Could not close BufferedWriter " + file);
131                        }
132                }
133                log.info("Done writing coordinates to file.");
134        }
135        
136        
137        /**
138         * Writes the amenities {@link Facility}s to file.
139         * @param file
140         */
141        public void writeFacilities(String file){
142                FacilitiesWriter fw = new FacilitiesWriter(this.amenities);
143                fw.write(file);
144        }
145        
146        
147        /**
148         * Writes the facility attributes to file.
149         * @param file
150         */
151        public void writeFacilityAttributes(String file){
152                ObjectAttributesXmlWriter ow = new ObjectAttributesXmlWriter(this.amenityAttributes);
153                ow.writeFile(file);
154        }
155        
156        
157        public ActivityFacilities getActivityFacilities(){
158                return this.amenities;
159        }
160        
161
162}
