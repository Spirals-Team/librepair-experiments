106        @Override
107        public Map<String,String> getComments() {
108                Map<String,String> map = new TreeMap<>() ;
109                
110                map.put(TIME_OF_DAY, "time of day at which trips for accessibility computations are assumed to start");
111                
112                map.put(ACCESSIBILITY_DESTINATION_SAMPLING_RATE, "if only a sample of destinations should be used " +
113                                "(reduces accuracy -- not recommended except when necessary for computational speed reasons)");
114                
115                map.put(ACCESSIBILITY_MEASURE_TYPE, "defines type of measure for accessibility computation.");
116                
117                map.put(USING_CUSTOM_BOUNDING_BOX, "true if custom bounding box should be used for accessibility computation (otherwise e.g. extent of network will be used)");
118                map.put(BOUNDING_BOX_BOTTOM,"custom bounding box parameters for accessibility computation (if enabled)");
119                
120                StringBuilder stb = new StringBuilder() ;
121                for (AreaOfAccesssibilityComputation val : AreaOfAccesssibilityComputation.values()) {
122                        stb.append(val.toString() ) ;
123                        stb.append( " " ) ;
124                }
125                map.put(AREA_OF_ACC_COMP, "method to determine the area for which the accessibility will be computed; possible values: " + stb);
126                
127                map.put(MEASURING_POINTS_FILE, "if the accibility is computed using the `fromFile` option, " +
128                                "the this must be the file containing the measuring points' coordinates. ");
129                return map ;
130        }

132        public void setComputingAccessibilityForMode(Modes4Accessibility mode, boolean val) {
133                if (val) {
134                        this.isComputingMode.add(mode);
135                } else {
136                        this.isComputingMode.remove(mode);
137                }
138        }

152        @StringSetter("cellSizeForCellBasedAccessibility")  // size in meters (whatever that is since the coord system does not know about meters)
153    public void setCellSizeCellBasedAccessibility(long value) {
154                if (value <= 0) {
155                        throw new IllegalArgumentException("Cell size must be greater than zero.");
156                }
157                if (value < 100) {
158                        log.warn("Cell size is set to " + value + ". This is a comparatively small" +
159                                        " value, which may lead to computational problems.");
160                }
161                this.cellSizeCellBasedAccessibility = value;
162    }
