023import java.io.InputStream;
024
025import org.apache.log4j.Logger;
026import org.matsim.api.core.v01.network.Network;
027import org.matsim.core.config.ConfigUtils;

036/**
037 * @author dziemke
038 */
039public class AccessibilityOsmNetworkReader {
040        final private static Logger LOG = Logger.getLogger(AccessibilityOsmNetworkReader.class);
041        
042        private String osmFileName;
043        private InputStream osmInputStream;
044        private final String outputCRS;
045        private String inputCRS = "EPSG:4326"; // EPSG:4326 = WGS84, OSM input is always in this CRS
046        private boolean keepPaths = false; 
047        private boolean includeLowHierarchyWays = false;
048        private boolean onlyMotorwayToTertiary = false; // "Thinner" network down to tertiary; do not use this together with "includeLowHierarchyWays"
049        private boolean onlyMotorwayToSecondary = false; // "Thinner" network down to tertiary; do not use this together with "includeLowHierarchyWays"
050        
051        private Network network;
052
053        public static void main(String[] args) {
054                String osmFileName = "../../upretoria/data/capetown/osm/2017-10-03";
055                String outputRoot = "../../upretoria/data/capetown/network/";
056                String networkFileName = outputRoot + "2017-10-03_network.xml.gz";
057
058                // EPSG:4326 = WGS84
059                // EPSG:31468 = DHDN GK4, for Berlin; DE
060                // EPSG:26918 = NAD83 / UTM zone 18N, for Maryland, US
061                // EPSG:25832 = ETRS89 / UTM zone 32N, for Nordrhein-Westfalen
062                // EPSG:22235 = Cape / UTM zone 35S, for South Africa
063                String outputCRS = "EPSG:22235";
064                
065                AccessibilityOsmNetworkReader osmNetworkCreatorDZ = new AccessibilityOsmNetworkReader(osmFileName, outputCRS);
066                osmNetworkCreatorDZ.createNetwork();
067                osmNetworkCreatorDZ.writeNetwork(outputRoot, networkFileName);
068        }
069
070        public AccessibilityOsmNetworkReader(String osmFileName, String outputCRS) {
071                this.osmFileName = osmFileName;
072                this.outputCRS = outputCRS;
073        }
074        
075        public AccessibilityOsmNetworkReader(InputStream osmInputStream, String outputCRS) {
076                this.osmInputStream = osmInputStream;
077                this.outputCRS = outputCRS;
078        }
079
080        public void createNetwork() {
081                LOG.info("Input CRS is " + inputCRS + "; output CRS is " + outputCRS);
082                LOG.info("Setting \"keepPaths\" = " + keepPaths);
083                LOG.info("Setting \"includeLowHierarchyWays\" = " + includeLowHierarchyWays);
084                LOG.info("Setting \"onlyMotorwayToTertiary\" = " + onlyMotorwayToTertiary);
085                LOG.info("Setting \"onlyMotorwayToSecondary\" = " + onlyMotorwayToSecondary);
086
087                // Infrastructure
088                network = ScenarioUtils.createScenario(ConfigUtils.createConfig()).getNetwork();
089                CoordinateTransformation coordinateTransformation = TransformationFactory.getCoordinateTransformation(inputCRS, outputCRS);
090                OsmNetworkReader osmNetworkReader;
091                if (onlyMotorwayToTertiary == true || onlyMotorwayToSecondary == true) {
092                        osmNetworkReader = new OsmNetworkReader(network, coordinateTransformation, false);
093                } else {
094                        osmNetworkReader = new OsmNetworkReader(network, coordinateTransformation, true);
095                }
096                
097                // Setting this to true means that pure geometry-describing nodes (i.e. nodes which are not intersections etc.) are kept.
098                // This makes the file (for the Nairobi case) three times as big (22.4MB vs. 8.7MB)
099                if (keepPaths == true) {
100                        LOG.info("Detailed geometry of paths is kept.");
101                        osmNetworkReader.setKeepPaths(true);
102                }
103                                
104                // This block is for the low hierarchy roads
105                if (includeLowHierarchyWays == true) {
106                        LOG.info("Low hierarchy ways are included.");
107                        // By defaults set for motorway, motorway_link, trunk, trunk_link, primary, primary_link, secondary, secondary_link,
108                        // tertiary, tertiary_link, minor, unclassified, residential, living_street; minor does not exist on the website anymore
109                        // Parameters for living_street: (6, "living_street", 1,  15.0/3.6, 1.0,  300);
110                        // (hierarchy, highwayType, lanes, freespeed, freespeedFactor, laneCapacity_vehPerHour)
111                        //
112                        // Other types on OSM, see: http://wiki.openstreetmap.org/wiki/Key:highway
113                        osmNetworkReader.setHighwayDefaults(7, "cycleway", 1, 30/3.6, 1.0, 0);
114                        osmNetworkReader.setHighwayDefaults(7, "pedestrian", 1, 15/3.6, 1.0, 0);
115                        osmNetworkReader.setHighwayDefaults(7, "road", 1, 15/3.6, 1.0, 300); // like "living_street"
116                        osmNetworkReader.setHighwayDefaults(8, "track", 1, 15/3.6, 1.0, 0);
117                        osmNetworkReader.setHighwayDefaults(8, "footway", 1, 15/3.6, 1.0, 0);
118                        osmNetworkReader.setHighwayDefaults(8, "steps", 1, 5/3.6, 1.0, 0);
119                        osmNetworkReader.setHighwayDefaults(8, "path", 1, 15/3.6, 1.0, 0);
120                }               
121                                
122                // For a coarser network ("onlyMotorwayToTertiaryMotorways" leads to a network only a 14th as big (77.8MB vs. 1.04GB) for Maryland)
123                if (onlyMotorwayToSecondary == true || onlyMotorwayToTertiary == true) {
124                        LOG.info("Only bigger roads are included.");
125                        if (includeLowHierarchyWays == true) {
126                                throw new RuntimeException("It does not make sense to set both \"includeLowHierarchyWays\""
127                                                + " and \"onlyMotorwayToSecondary\" or \"onlyMotorwayToTertiary\" to true");
128                        }
129                        osmNetworkReader.setHighwayDefaults(1, "motorway",      2, 120.0/3.6, 1.0, 2000, true);
130                        osmNetworkReader.setHighwayDefaults(1, "motorway_link", 1,  80.0/3.6, 1.0, 1500, true);
131                        osmNetworkReader.setHighwayDefaults(2, "trunk",         1,  80.0/3.6, 1.0, 2000);
132                        osmNetworkReader.setHighwayDefaults(2, "trunk_link",    1,  50.0/3.6, 1.0, 1500);
133                        osmNetworkReader.setHighwayDefaults(3, "primary",       1,  80.0/3.6, 1.0, 1500);
134                        osmNetworkReader.setHighwayDefaults(3, "primary_link",  1,  60.0/3.6, 1.0, 1500);
135                        osmNetworkReader.setHighwayDefaults(4, "secondary",     1,  30.0/3.6, 1.0, 1000);
136                        osmNetworkReader.setHighwayDefaults(4, "secondary_link",     1,  30.0/3.6, 1.0, 1000);
137                        if (onlyMotorwayToTertiary == true) {
138                                osmNetworkReader.setHighwayDefaults(5, "tertiary",      1,  25.0/3.6, 1.0,  600);
139                                osmNetworkReader.setHighwayDefaults(5, "tertiary_link",      1,  25.0/3.6, 1.0,  600);
140                        }
141                }
142
143                if (osmFileName != null && osmInputStream == null) {
144                        osmNetworkReader.parse(osmFileName);
145                } else if (osmFileName == null && osmInputStream != null) {
146                        osmNetworkReader.parse(osmInputStream); 
147                } else {
148                        throw new IllegalArgumentException("Either the input stream OR the input file should be defined, but not both.");
149                }
150                
151                new NetworkCleaner().run(network);
152        }
153        
154        public void writeNetwork(String outputRoot, String networkFileName) {
155                try {
156                        OutputDirectoryLogging.initLoggingWithOutputDirectory(outputRoot);
157                } catch (IOException e) {
158                        e.printStackTrace();
159                }
160                NetworkWriter networkWriter = new NetworkWriter(network);
161                networkWriter.write(networkFileName);
162                LOG.info("Network file written to " + networkFileName);
163        }
164        
165        /**
166         * The input CRS for data from OSM should in almost all cases be EPSG:4326 = WGS84, which is used here as default.
167         */
168        public void setInputCRS(String inputCRS) {
169                this.inputCRS = inputCRS;
170        }
171
172        public void setKeepPaths(boolean keepPaths) {
173                this.keepPaths = keepPaths;
174        }
175        
176        public void setIincludeLowHierarchyWays(boolean includeLowHierarchyWays) {
177                this.includeLowHierarchyWays = includeLowHierarchyWays;
178        }
179
180        public void setOnlyMotorwayToTertiary(boolean onlyMotorwayToTertiary) {
181                this.onlyMotorwayToTertiary = onlyMotorwayToTertiary;
182        }
183
184        public void setOnlyMotorwayToSecondary(boolean onlyMotorwayToSecondary) {
185                this.onlyMotorwayToSecondary = onlyMotorwayToSecondary;
186        }
187        
188        public Network getNetwork() {
189                return this.network;
190        }
