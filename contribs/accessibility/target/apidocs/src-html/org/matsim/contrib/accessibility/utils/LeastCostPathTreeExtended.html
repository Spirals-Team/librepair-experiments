050/**
051 * This runs the normal LeastCostPathTree, with whatever TravelDisutility object it is using (e.g. including distance and/or toll).
052 * However, since the normal LeastCostPathTree does not return its contributing cost dimensions, this class here is used to also
053 * accumulate distance and toll separately along the route.
054 * <p></p>
055 * To re-iterate: This class does <i>not</i> lead to a different tree than the base class.
056 * 
057 * @author thomas
058 * 
059 */
060public final class LeastCostPathTreeExtended extends LeastCostPathTree{
061        
062        protected static final Logger log = Logger.getLogger(LeastCostPathTreeExtended.class);
063        
064        private Map<Id<Node>, NodeDataExtended> nodeDataExt = null;
065        private RoadPricingScheme scheme = null;
066        
067        /**
068         * constructor
069         * @param controler Controler, to get the RoadPricingScheme if available
070         */
071        public LeastCostPathTreeExtended(final TravelTime tt, final TravelDisutility td, final RoadPricingScheme scheme) {
072                super(tt, td);
073                this.scheme = scheme;
074        }
075
076        /**
077         * 
078         * @param network
079         * @param origin
080         * @param time
081         */
082        public final void calculateExtended(final Network network, final Node origin, final double time) {
083                
084                this.nodeDataExt = new ConcurrentHashMap<Id<Node>, NodeDataExtended>((int) (network.getNodes().size() * 1.1), 0.95f);
085                if(this.nodeDataExt.get(origin.getId()) == null){
086                        NodeDataExtended nde = new NodeDataExtended();
087                        nde.distance = 0.;
088                        nde.toll         = 0.;
089                        this.nodeDataExt.put(origin.getId(), nde);
090                }
091                
092                calculate(network, origin, time);
093        }
094        
095        /**
096         * @param link  
097         * @param currTime 
098         */
099        @Override
100        protected final void additionalComputationsHook( final Link link, final double currTime ) {
101                
102                Node fromNode = link.getFromNode();
103                // get current distance and toll so far
104                NodeDataExtended nde = nodeDataExt.get( fromNode.getId() );
105                double currDistance = nde.getDistance();
106                double currToll = nde.getToll();
107
108                // query toll
109                double toll = 0.;
110                if(this.scheme != null){
111                        Cost cost = scheme.getLinkCostInfo(link.getId(), currTime, null, null);
112                        if(cost != null)
113                                toll = cost.amount;
114                }
115                
116                double visitDistance = currDistance + link.getLength();
117                double visitToll = currToll + toll;
118                
119                // put new nodes into nodeDataExtended
120                Node toNode = link.getToNode();
121                NodeDataExtended ndeNew = this.nodeDataExt.get( toNode.getId() );
122                if(ndeNew == null){
123                        ndeNew = new NodeDataExtended();
124                        this.nodeDataExt.put( toNode.getId(), ndeNew);
125                }
126                ndeNew.visit(visitDistance, visitToll);
127        }
128        
129        // ////////////////////////////////////////////////////////////////////
130        // get methods
131        // ////////////////////////////////////////////////////////////////////
132        
133        public final Map<Id<Node>, NodeDataExtended> getTreeExtended() {
134                return this.nodeDataExt;
135        }
136        
137        // ////////////////////////////////////////////////////////////////////
138        // inner classes
139        // ////////////////////////////////////////////////////////////////////
140        
141        public static class NodeDataExtended {
142                private double distance = 0.;   // meter
143                private double toll     = 0.;   // money
144
145                /*package*/ final void reset() {
146                        this.distance   = 0.;
147                        this.toll               = 0.;
148                }
149
150                final void visit(final double distance, final double toll) {
151                        this.distance   = distance;
152                        this.toll               = toll;
153                }
154
155                public final double getDistance() {
156                        return this.distance;
157                }
158
159                public final double getToll() {
160                        return this.toll;
161                }
162        }
163        
164        // ////////////////////////////////////////////////////////////////////
165        // testing 
166        // ////////////////////////////////////////////////////////////////////
167        
168        /**
169         * for testing
170         * @param args
171         */
172        public static void main(String args[]){
173                // create temp output dir
174                String tmpOutputLocation = TempDirectoryUtil.createCustomTempDirectory("test");
175                
176                // create network
177                Network network = LeastCostPathTreeExtended.createTriangularNetwork();
178                // create scenario
179                Config config = ConfigUtils.createConfig();
180                // set last iteration and output
181                ControlerConfigGroup controlerCG = (ControlerConfigGroup) config.getModule(ControlerConfigGroup.GROUP_NAME);
182                controlerCG.setLastIteration( 1 );
183                controlerCG.setOutputDirectory( tmpOutputLocation );
184                // set scenario
185                ScenarioUtils.ScenarioBuilder builder = new ScenarioUtils.ScenarioBuilder(config) ;
186                builder.setNetwork( createTriangularNetwork() ) ;
187                Scenario scenario = builder.build() ;
188                Controler controler = new Controler(scenario);
189                controler.run();
190                // init lcpte
191                LeastCostPathTreeExtended lcpte = new LeastCostPathTreeExtended(controler.getLinkTravelTimes(), controler.createTravelDisutilityCalculator(), (RoadPricingSchemeImpl) controler.getScenario().getScenarioElement(RoadPricingScheme.ELEMENT_NAME));
192                
193                // contains all network nodes
194                Map<Id<Node>, ? extends Node> networkNodesMap = network.getNodes();
195                Id<Node> originNodeID = Id.create(1, Node.class);
196                Id<Node> destinationNodeId = Id.create(4, Node.class);
197                // run lcpte
198                lcpte.calculateExtended(network, networkNodesMap.get( originNodeID ), 3600.);
199                double time = lcpte.getTree().get( destinationNodeId ).getTime();
200                double disutility = lcpte.getTree().get( destinationNodeId ).getCost();
201                double distance = lcpte.getTreeExtended().get( destinationNodeId ).getDistance();
202                double toll = lcpte.getTreeExtended().get( destinationNodeId ).getToll();
203                
204                log.info("Time = " + time);
205                log.info("Disutility = " + disutility);         
206                log.info("Distance = " + distance );
207                log.info("Toll = " + toll);
208                
209                TempDirectoryUtil.cleanUpCustomTempDirectories();
210        }
211        
212        /**
213         * creating a test network
214         * the path 1,2,4 has a total length of 1000m with a free speed travel time of 10m/s
215         * the second path 1,3,4 has a total length of 100m but only a free speed travel time of 0.1m/s
216         */
217        private static Network createTriangularNetwork() {
218                /*
219                 *                      (2)
220                 *         /   \
221                 *        /     \
222                 *(10m/s)/       \(10m/s)
223                 *(500m)/             \(500m)
224                 *     /           \
225                 *    /             \
226                 *       /               \
227                 *(1)-------(3)-------(4)
228                 *(50m,0.1m/s)(50m,0.1m/s)                      
229                 */
230                MutableScenario scenario = (MutableScenario) ScenarioUtils.createScenario(ConfigUtils.createConfig());
231
232                Network network = (Network) scenario.getNetwork();
233                
234                // add nodes
235                Node node1 = NetworkUtils.createAndAddNode(network, Id.create(1, Node.class), new Coord(0, 0));
236                Node node2 = NetworkUtils.createAndAddNode(network, Id.create(2, Node.class), new Coord(50, 100));
237                Node node3 = NetworkUtils.createAndAddNode(network, Id.create(3, Node.class), new Coord(50, 0));
238                Node node4 = NetworkUtils.createAndAddNode(network, Id.create(4, Node.class), new Coord(100, 0));
239                final Node fromNode = node1;
240                final Node toNode = node2;
241
242                // add links
243                NetworkUtils.createAndAddLink(network,Id.create(1, Link.class), fromNode, toNode, 500.0, 10.0, 3600.0, (double) 1 );
244                final Node fromNode1 = node2;
245                final Node toNode1 = node4;
246                NetworkUtils.createAndAddLink(network,Id.create(2, Link.class), fromNode1, toNode1, 500.0, 10.0, 3600.0, (double) 1 );
247                final Node fromNode2 = node1;
248                final Node toNode2 = node3;
249                NetworkUtils.createAndAddLink(network,Id.create(3, Link.class), fromNode2, toNode2, 50.0, 0.1, 3600.0, (double) 1 );
250                final Node fromNode3 = node3;
251                final Node toNode3 = node4;
252                NetworkUtils.createAndAddLink(network,Id.create(4, Link.class), fromNode3, toNode3, 50.0, 0.1, 3600.0, (double) 1 );
253                
254                return network;
255        }
