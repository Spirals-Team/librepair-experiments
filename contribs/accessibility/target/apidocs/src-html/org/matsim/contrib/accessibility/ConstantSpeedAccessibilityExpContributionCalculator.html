037/**
038 * @author thibautd, dziemke
039 */
040public final class ConstantSpeedAccessibilityExpContributionCalculator implements AccessibilityContributionCalculator {
041        private static final Logger LOG = Logger.getLogger(ConstantSpeedAccessibilityExpContributionCalculator.class);
042
043        // Estimates travel time by a constant speed along network, considering all links (including highways, which seems
044        // to be realistic in South Africa, but less elsewhere)
045        private final LeastCostPathTree lcptTravelDistance = new LeastCostPathTree(new FreeSpeedTravelTime(), new LinkLengthTravelDisutility());
046
047        private final Network network;
048        
049        private double logitScaleParameter;
050        
051        private double betaModeTT;      // in MATSim this is [utils/h]: cnScoringGroup.getTravelingBike_utils_hr() - cnScoringGroup.getPerforming_utils_hr()
052        private double betaModeTD;      // in MATSim this is 0 !!! since getMonetaryDistanceCostRateBike doesn't exist:
053        private double constMode;
054        private double modeSpeed_m_h = -1;
055        
056        private final double betaWalkTT;
057        private final double betaWalkTD;
058        private final double walkSpeed_m_h;
059
060        private Node fromNode = null;
061
062
063        public ConstantSpeedAccessibilityExpContributionCalculator(final String mode, Config config, Network network) {
064                this.network = network;
065                final PlanCalcScoreConfigGroup planCalcScoreConfigGroup = config.planCalcScore() ;
066
067                if (planCalcScoreConfigGroup.getOrCreateModeParams(mode).getMonetaryDistanceRate() != 0.) {
068                        LOG.error("Monetary distance cost rate for " + mode + " different from zero, but not used in accessibility computations");
069                }
070
071                logitScaleParameter = planCalcScoreConfigGroup.getBrainExpBeta();
072                
073                if (config.plansCalcRoute().getTeleportedModeSpeeds().get(mode) == null) {
074                        LOG.error("No teleported mode speed for mode " + mode + " set.");
075                }
076                this.modeSpeed_m_h = config.plansCalcRoute().getTeleportedModeSpeeds().get(mode) * 3600.;
077                
078                final PlanCalcScoreConfigGroup.ModeParams modeParams = planCalcScoreConfigGroup.getOrCreateModeParams(mode);
079                betaModeTT = modeParams.getMarginalUtilityOfTraveling() - planCalcScoreConfigGroup.getPerforming_utils_hr();
080                betaModeTD = modeParams.getMarginalUtilityOfDistance();
081                constMode = modeParams.getConstant();
082
083                betaWalkTT = planCalcScoreConfigGroup.getModes().get(TransportMode.walk).getMarginalUtilityOfTraveling() - planCalcScoreConfigGroup.getPerforming_utils_hr();
084                betaWalkTD = planCalcScoreConfigGroup.getModes().get(TransportMode.walk).getMarginalUtilityOfDistance();
085                this.walkSpeed_m_h = config.plansCalcRoute().getTeleportedModeSpeeds().get(TransportMode.walk) * 3600;
086        }
087
088        
089        @Override
090        public void notifyNewOriginNode(Node fromNode, Double departureTime) {
091                this.fromNode = fromNode;
092                this.lcptTravelDistance.calculate(network, fromNode, departureTime);
093        }
094
095        
096        @Override
097        public double computeContributionOfOpportunity(ActivityFacility origin, AggregationObject destination, Double departureTime) {
098                // TODO departure time is not used, dz, apr'17
099                Link nearestLinkToOrigin = NetworkUtils.getNearestLinkExactly(network, origin.getCoord());
100
101                // Captures the distance between the origin via the link to the node:
102                Distances distances = NetworkUtil.getDistances2NodeViaGivenLink(origin.getCoord(), nearestLinkToOrigin, fromNode);
103
104                // TODO: extract this walk part?
105                // In the state found before modularization (june 15), this was anyway not consistent accross modes
106                // (different for PtMatrix), pointing to the fact that making this mode-specific might make sense.
107                // distance to road, and then to node:
108                
109                // Utility to get on the network by walking
110                double distancePoint2Intersection_m = distances.getDistancePoint2Intersection();
111                double utilityMeasuringPoint2Road = (distancePoint2Intersection_m / this.walkSpeed_m_h * betaWalkTT)    + (distancePoint2Intersection_m * betaWalkTD);
112                
113                // Utility on the network to first node
114                double distanceIntersection2Node_m = distances.getDistanceIntersection2Node();
115                double utilityRoad2Node = (distanceIntersection2Node_m / modeSpeed_m_h * betaModeTT)    + (distanceIntersection2Node_m * betaModeTD); // toll or money ???
116        
117                // Uutility on the network from first node to destination node
118                double travelDistance_m = lcptTravelDistance.getTree().get(destination.getNearestNode().getId()).getCost(); // travel link distances on road network for bicycle and walk
119                double utility = ((travelDistance_m / modeSpeed_m_h * betaModeTT) + (travelDistance_m * betaModeTD)); // toll or money ???
120
121                // Utility based on opportunities that are attached to destination node
122                double sumExpVjkWalk = destination.getSum();
123                
124                // exp(beta * a) * exp(beta * b) = exp(beta * (a+b))
125                return Math.exp(logitScaleParameter * (constMode + utilityMeasuringPoint2Road + utilityRoad2Node + utility)) * sumExpVjkWalk;
126        }
127}
