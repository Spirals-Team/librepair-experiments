001package org.matsim.contrib.accessibility;
002
003import org.matsim.contrib.accessibility.gis.SpatialGrid;
004import org.matsim.contrib.accessibility.interfaces.FacilityDataExchangeInterface;
005import org.matsim.facilities.ActivityFacility;
006
007import java.util.HashMap;
008import java.util.Map;
009
010public class SpatialGridAggregator implements FacilityDataExchangeInterface {
011
012        private Map<String, SpatialGrid> accessibilityGrids = new HashMap<>() ;
013
014        @Override
015        public void setFacilityAccessibilities(ActivityFacility origin, Double timeOfDay, Map<String, Double> accessibilities) {
016                for (Map.Entry<String, Double> modes4AccessibilityDoubleEntry : accessibilities.entrySet()) {
017                        accessibilityGrids.get(modes4AccessibilityDoubleEntry.getKey()).setValue(modes4AccessibilityDoubleEntry.getValue(), origin.getCoord().getX(), origin.getCoord().getY());
018                }
019        }
020
021        @Override
022        public void finish() {
023
024        }
025
026        public Map<String, SpatialGrid> getAccessibilityGrids() {
027                return accessibilityGrids;
028        }
029
030}
