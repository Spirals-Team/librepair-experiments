037        @Deprecated // yyyy should be possible to get this from car travel disutility

053        public NetworkModeAccessibilityExpContributionCalculator(
054                        final TravelTime travelTime,
055                        final TravelDisutilityFactory travelDisutilityFactory,
056                        final Scenario scenario){               
057                
058                this.scenario = scenario;
059
060                final PlanCalcScoreConfigGroup planCalcScoreConfigGroup = scenario.getConfig().planCalcScore();
061                this.scheme = (RoadPricingScheme) scenario.getScenarioElement( RoadPricingScheme.ELEMENT_NAME );
062                this.travelTime = travelTime;           
063
064                Gbl.assertNotNull(travelDisutilityFactory);
065                TravelDisutility travelDisutility = travelDisutilityFactory.createTravelDisutility(travelTime);
066
067                this.lcpt = new LeastCostPathTreeExtended(travelTime, travelDisutility, this.scheme);
068
069                if ( planCalcScoreConfigGroup.getOrCreateModeParams(TransportMode.car).getMarginalUtilityOfDistance() != 0. ) {
070                        log.error( "marginal utility of distance for car different from zero but not used in accessibility computations");
071                }
072
073                logitScaleParameter = planCalcScoreConfigGroup.getBrainExpBeta() ;
074
075                betaCarTT               = planCalcScoreConfigGroup.getModes().get(TransportMode.car).getMarginalUtilityOfTraveling() - planCalcScoreConfigGroup.getPerforming_utils_hr();
076                betaCarTD               = planCalcScoreConfigGroup.getMarginalUtilityOfMoney() * planCalcScoreConfigGroup.getModes().get(TransportMode.car).getMonetaryDistanceRate();
077                betaCarTMC              = - planCalcScoreConfigGroup.getMarginalUtilityOfMoney() ;
078
079                constCar                = planCalcScoreConfigGroup.getModes().get(TransportMode.car).getConstant();
080
081                betaWalkTT              = planCalcScoreConfigGroup.getModes().get(TransportMode.walk).getMarginalUtilityOfTraveling() - planCalcScoreConfigGroup.getPerforming_utils_hr();
082                betaWalkTD              = planCalcScoreConfigGroup.getModes().get(TransportMode.walk).getMarginalUtilityOfDistance();
083
084                this.walkSpeed_m_s = scenario.getConfig().plansCalcRoute().getTeleportedModeSpeeds().get(TransportMode.walk);
085        }

088        @Override
089        public void notifyNewOriginNode(Node fromNode, Double departureTime) {
090                this.fromNode = fromNode;
091                this.lcpt.calculateExtended(scenario.getNetwork(), fromNode, departureTime);
092        }

094        
095        @Override
096        public double computeContributionOfOpportunity(ActivityFacility origin, AggregationObject destination, Double departureTime) {
097
098                Link nearestLink = NetworkUtils.getNearestLinkExactly(scenario.getNetwork(), origin.getCoord());
099
100                // === (1) ORIGIN to LINK to NODE (captures the distance (as walk time) between the origin via the link to the node):
101                Distances distance = NetworkUtil.getDistances2NodeViaGivenLink(origin.getCoord(), nearestLink, fromNode);
102                
103                // TODO: extract this walk part?
104                // In the state found before modularization (june 15), this was anyway not consistent accross modes
105                // (different for PtMatrix), pointing to the fact that making this mode-specific might make sense. (comment by thibaut?)
106                double walkTravelTimeMeasuringPoint2Road_h      = distance.getDistancePoint2Intersection() / (this.walkSpeed_m_s * 3600);
107                
108                // (a) disutilities to get on or off the network
109                double walkDisutilityMeasuringPoint2Road = (walkTravelTimeMeasuringPoint2Road_h * betaWalkTT)
110                                + (distance.getDistancePoint2Intersection() * betaWalkTD);
111                
112                // (b) TRAVEL ON NETWORK to FIRST NODE:
113                double toll_money = getTollMoney(departureTime, nearestLink, distance);
114                double carSpeedOnNearestLink_m_s = nearestLink.getLength() / travelTime.getLinkTravelTime(nearestLink, departureTime, null, null);
115                double road2NodeCongestedCarTime_h = distance.getDistanceIntersection2Node() / (carSpeedOnNearestLink_m_s * 3600.);
116                
117                double congestedCarDisutilityRoad2Node = (road2NodeCongestedCarTime_h * betaCarTT) 
118                                + (distance.getDistanceIntersection2Node() * betaCarTD) + (toll_money * betaCarTMC);
119//              // yyyyyy dzdzdz: replace the above by link disutility multiplied by fraction of link that is used according to the entry point.  (toll should be in there automatically??)
120
121                // === (2) REMAINING TRAVEL ON NETWORK:
122                double congestedCarDisutility = - lcpt.getTree().get(destination.getNearestNode().getId()).getCost();
123                // travel disutility congested car on road network (including toll)
124                
125                // === (3) Pre-computed effect of all opportunities reachable from destination network node:
126                double sumExpVjkWalk = destination.getSum();
127                // works because something like exp(A+c1) + exp(A+c2) + ... = exp(A) * [ exp(c1) + exp(c2) + ...]  =: exp(A) * sumExpVjkWalk
128                
129                // === (4) Everything together:
130                // note that exp(a+b) = exp(a) * exp(b), so for b the exponentiation has already been done.
131                return Math.exp(this.logitScaleParameter * (walkDisutilityMeasuringPoint2Road + constCar + congestedCarDisutilityRoad2Node
132                                + congestedCarDisutility) ) * sumExpVjkWalk;
133        }

136        @Deprecated // yyyy should be possible to get this from car travel disutility
137        private double getTollMoney(Double departureTime, Link nearestLink, Distances distance) {
138                // yy there should be a way of doing this that is closer to the mobsim (and thus more general/automatic).  kai, jun'16
139                
140                double result = 0. ;
141                if(scheme != null){
142                        RoadPricingSchemeImpl.Cost cost = scheme.getLinkCostInfo(nearestLink.getId(), departureTime, null, null);
143                        if(cost != null) {
144                                result = cost.amount;
145                        }
146                        if(RoadPricingScheme.TOLL_TYPE_CORDON.equals(scheme.getType())) {
147                                // do nothing
148                        } else if( RoadPricingScheme.TOLL_TYPE_DISTANCE.equals(scheme.getType())) {
149                                result *= distance.getDistanceIntersection2Node();
150                        } else {
151                                throw new RuntimeException("accessibility not impelemented for requested toll scheme") ;
152                        }
153                }
154                return result ;
155        }
