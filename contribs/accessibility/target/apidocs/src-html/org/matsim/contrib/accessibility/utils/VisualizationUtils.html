048    public static void createQGisOutputGraduated(String actType, String mode, Envelope envelope,
049                String workingDirectory, String crs, boolean includeDensityLayer,
050                Double lowerBound, Double upperBound, Integer range,
051                ColorRangeUtils.ColorRange colorRange, int symbolSize, int populationThreshold) {
052
053        // Create Mapnik file that is needed for OSM layer
054                QGisMapnikFileCreator.writeMapnikFile(workingDirectory + "osm_mapnik.xml");
055
056                // Write QGis project file
057                QGisWriter writer = new QGisWriter(crs, workingDirectory);
058                String qGisProjectFile = "QGisProjectFile_" + mode + ".qgs";
059                writer.setEnvelope(envelope);
060
061                // Osm raster layer
062                writer.changeWorkingDirectory(workingDirectory); // has to be the storage location of the file
063
064                RasterLayer mapnikLayer = new RasterLayer("osm_mapnik_xml", workingDirectory + "/osm_mapnik.xml");
065                new AccessibilityXmlRenderer(mapnikLayer);
066                mapnikLayer.setSrs("WGS84_Pseudo_Mercator");
067                writer.addLayer(0,mapnikLayer);
068
069                String actSpecificWorkingDirectory =  workingDirectory + actType + "/";  // has to be the storage location of the file
070                writer.changeWorkingDirectory(actSpecificWorkingDirectory);
071
072                if (includeDensityLayer) {
073                        VectorLayer densityLayer = new VectorLayer(
074                                        "density", actSpecificWorkingDirectory + "accessibilities.csv", QGisConstants.geometryType.Point, true);
075                        densityLayer.setXField(Labels.X_COORDINATE);
076                        densityLayer.setYField(Labels.Y_COORDINATE);
077                        GraduatedSymbolRenderer dRenderer = RendererFactory.createDensitiesRenderer(densityLayer, populationThreshold, symbolSize);
078                        dRenderer.setRenderingAttribute(Labels.POPULATION_DENSITIY);
079                        writer.addLayer(densityLayer);
080                }
081
082                VectorLayer accessibilityLayer = new VectorLayer(
083                                "accessibility", actSpecificWorkingDirectory + "accessibilities.csv", QGisConstants.geometryType.Point, true);
084                // there are two ways to set x and y fields for csv geometry files
085                // 1) if there is a header, you can set the members xField and yField to the name of the column headers
086                // 2) if there is no header, you can write the column index into the member (e.g. field_1, field_2,...), but works also if there is a header
087                accessibilityLayer.setXField(Labels.X_COORDINATE);
088                accessibilityLayer.setYField(Labels.Y_COORDINATE);
089//              AccessibilityRenderer renderer = new AccessibilityRenderer(accessibilityLayer, lowerBound, upperBound,
090//                              range, symbolSize);
091                GraduatedSymbolRenderer renderer = new GraduatedSymbolRenderer(accessibilityLayer, lowerBound, upperBound,
092                                range, symbolSize, colorRange);
093                        renderer.setRenderingAttribute(mode.toString() + "_accessibility");
094                writer.addLayer(accessibilityLayer);
095
096                writer.write(qGisProjectFile);
097        }

152        /**
153         * This method creates a snapshot of the accessibility map that is held in the created QGis file.
154         * The syntax within the method is different dependent on the operating system.
155         * 
156         * @param workingDirectory The directory where the QGisProjectFile (the data source of the to-be-created snapshot) is stored
157         * @param mode
158         * @param osName
159         */
160        public static void createSnapshot(String workingDirectory, String mode, String osName) {
161                // if OS is Windows
162                // example (daniel r) // os.arch=amd64 // os.name=Windows 7 // os.version=6.1
163                if ( osName.contains("Win") || osName.contains("win")) {
164                        // On Windows, the PATH variables need to be set correctly to be able to call "qgis.bat" on the command line
165                        // This needs to be done manually. It does not seem to be set automatically when installing QGis
166                        String cmd = "qgis.bat " + workingDirectory + "QGisProjectFile_" + mode + ".qgs" +
167                                        " --snapshot " + workingDirectory + "snapshot_" + mode + ".png";
168
169                        String stdoutFileName = workingDirectory + "snapshot_" + mode + ".log";
170                        int timeout = 99999;
171
172                        ExeRunner.run(cmd, stdoutFileName, timeout);
173
174                        // if OS is Macintosh
175                        // example (dominik) // os.arch=x86_64 // os.name=Mac OS X // os.version=10.10.2
176                } else if ( osName.contains("Mac") || osName.contains("mac") ) {
177                        String cmd = "/Applications/QGIS.app/Contents/MacOS/QGIS " + workingDirectory + "QGisProjectFile_" + mode + ".qgs" +
178                                        " --snapshot " + workingDirectory + "snapshot_" + mode + ".png";
179
180                        String stdoutFileName = workingDirectory + "snapshot_" + mode + ".log";
181
182                        int timeout = 99999;
183
184                        ExeRunner.run(cmd, stdoutFileName, timeout);
185
186                        // if OS is Linux
187                        // example (benjamin) // os.arch=amd64 // os.name=Linux // os.version=3.13.0-45-generic
188                        //} else if ( osName.contains("Lin") || osName.contains("lin") ) {
189                        // TODO for linux
190
191                        // if OS is other
192                } else {
193                        log.warn("generating png files not implemented for os.arch=" + System.getProperty("os.arch") );
194                }
195        }
