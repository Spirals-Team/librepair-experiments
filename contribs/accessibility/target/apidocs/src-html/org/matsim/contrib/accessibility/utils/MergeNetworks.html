021package org.matsim.contrib.accessibility.utils;
022
023import org.matsim.api.core.v01.Id;
024import org.matsim.api.core.v01.network.Link;
025import org.matsim.api.core.v01.network.Network;
026import org.matsim.api.core.v01.network.NetworkFactory;
027import org.matsim.api.core.v01.network.Node;
028
029/**
030 * Merges two networks into one, by copying all nodes and links from the two
031 * given networks to a third one.
032 *
033 * @author mrieser
034 */
035public class MergeNetworks {
036        // TODO copied here like several others did before. potentially find a central place where this can go
037
038        /**
039         * Merges two networks into one, by copying all nodes and links from the addNetworks to the baseNetwork.
040         *
041         * @param baseNetwork
042         * @param addNetwork
043         */
044        public static void merge(final Network baseNetwork, final String addPrefix, final Network addNetwork) {
045                double capacityFactor = baseNetwork.getCapacityPeriod() / addNetwork.getCapacityPeriod();
046                NetworkFactory factory = baseNetwork.getFactory();
047                for (Node node : addNetwork.getNodes().values()) {
048                        Node node2 = (Node) factory.createNode(Id.create(addPrefix + node.getId().toString(), Node.class), node.getCoord());
049                        baseNetwork.addNode(node2);
050                }
051                for (Link link : addNetwork.getLinks().values()) {
052                        Id<Node> fromNodeId = Id.create(addPrefix + link.getFromNode().getId().toString(), Node.class);
053                        Id<Node> toNodeId = Id.create(addPrefix + link.getToNode().getId().toString(), Node.class);
054                        Node fromNode = baseNetwork.getNodes().get(fromNodeId);
055                        Node toNode = baseNetwork.getNodes().get(toNodeId);
056                        Link link2 = factory.createLink(Id.create(addPrefix + link.getId().toString(), Link.class),
057                                        fromNode, toNode);
058                        link2.setAllowedModes(link.getAllowedModes());
059                        link2.setCapacity(link.getCapacity() * capacityFactor);
060                        link2.setFreespeed(link.getFreespeed());
061                        link2.setLength(link.getLength());
062                        link2.setNumberOfLanes(link.getNumberOfLanes());
063                        baseNetwork.addLink(link2);
064                }
065        }
066
067        /**
068         * Merges two networks into one, by copying all nodes and links from the two
069         * given networks to a third one.
070         *
071         * @param networkA
072         * @param prefixA
073         * @param networkB
074         * @param prefixB
075         * @param mergedNetwork
076         */
077        public static void merge(final Network networkA, final String prefixA, final Network networkB, final String prefixB, final Network mergedNetwork) {
078                double capacityFactor = mergedNetwork.getCapacityPeriod() / networkA.getCapacityPeriod();
079                NetworkFactory factory = mergedNetwork.getFactory();
080                for (Node node : networkA.getNodes().values()) {
081                        Node node2 = (Node) factory.createNode(Id.create(prefixA + node.getId().toString(), Node.class), node.getCoord());
082                        mergedNetwork.addNode(  node2);
083                }
084                for (Link link : networkA.getLinks().values()) {
085                        Id<Node> fromNodeId = Id.create(prefixA + link.getFromNode().getId().toString(), Node.class);
086                        Id<Node> toNodeId = Id.create(prefixA + link.getToNode().getId().toString(), Node.class);
087                        Node fromNode = mergedNetwork.getNodes().get(fromNodeId);
088                        Node toNode = mergedNetwork.getNodes().get(toNodeId);
089                        Link link2 = factory.createLink(Id.create(prefixA + link.getId().toString(), Link.class),
090                                        fromNode, toNode);
091                        link2.setAllowedModes(link.getAllowedModes());
092                        link2.setCapacity(link.getCapacity() * capacityFactor);
093                        link2.setFreespeed(link.getFreespeed());
094                        link2.setLength(link.getLength());
095                        link2.setNumberOfLanes(link.getNumberOfLanes());
096                        mergedNetwork.addLink(link2);
097                        fromNode.addOutLink(link2);
098                        toNode.addInLink(link2);
099                }
100                capacityFactor = mergedNetwork.getCapacityPeriod() / networkB.getCapacityPeriod();
101                for (Node node : networkB.getNodes().values()) {
102                        Node node2 = (Node) factory.createNode(Id.create(prefixB + node.getId().toString(), Node.class), node.getCoord());
103                        mergedNetwork.addNode(  node2);
104                }
105                for (Link link : networkB.getLinks().values()) {
106                        Id<Node> fromNodeId = Id.create(prefixB + link.getFromNode().getId().toString(), Node.class);
107                        Id<Node> toNodeId = Id.create(prefixB + link.getToNode().getId().toString(), Node.class);
108                        Node fromNode = mergedNetwork.getNodes().get(fromNodeId);
109                        Node toNode = mergedNetwork.getNodes().get(toNodeId);
110                        Link link2 = factory.createLink(Id.create(prefixB + link.getId().toString(), Link.class),
111                                        fromNode, toNode);
112                        link2.setAllowedModes(link.getAllowedModes());
113                        link2.setCapacity(link.getCapacity() * capacityFactor);
114                        link2.setFreespeed(link.getFreespeed());
115                        link2.setLength(link.getLength());
116                        link2.setNumberOfLanes(link.getNumberOfLanes());
117                        mergedNetwork.addLink(link2);
118                        fromNode.addOutLink(link2);
119                        toNode.addInLink(link2);
120                }
121        }
