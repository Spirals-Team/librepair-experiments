090        @Override
091        public void complete() {
092                log.info("    nodes: " + nodeMap.size());
093                log.info("     ways: " + wayMap.size());
094                log.info("relations: " + relationMap.size());
095                
096                log.info("Creating facilities..");
097                
098                ActivityFacilitiesFactory aff = new ActivityFacilitiesFactoryImpl();
099
100                /* First check all the point features. */
101                processFacilities(aff, nodeMap);
102                
103                /* Second, check for way features. */
104                processFacilities(aff, wayMap);
105
106                /* Thirdly, check for relation. */
107                processFacilities(aff, relationMap);
108                
109                /*TODO Report the final counts of different amenity types. */
110        }

124        private void processFacilities(ActivityFacilitiesFactory aff,
125                        Map<Long,? extends EntityContainer> entityMap) {
126                for(long n : entityMap.keySet()){
127                        Entity entity = entityMap.get(n).getEntity();
128                        Map<String, String> tags = new TagCollectionImpl(entity.getTags()).buildMap();
129                        
130                        /* Check amenities */
131                        String amenity = tags.get("amenity");
132                        String matsimType = null;
133                        if(amenity != null) {
134                                matsimType = getActivityType(amenity);
135                        }
136                        if(matsimType != null){
137                                //String activityType = getActivityType(amenity);
138                                String name = tags.get("name");
139                                if(name != null){
140                                        /* Check education level. */
141                                        if(matsimType.equalsIgnoreCase("e")){
142                                                getEducationLevel(name);
143                                        }                                       
144                                } else{
145                                        log.warn("      ---> Amenity " + n + " without a name.");
146                                }
147
148                                /* Facility identified. Now get the centroid of all members. */ 
149                                //Coord coord = getCoord(entity);
150                                Coord coord = CoordUtils.getCentroidCoord(entity, this.ct, this.nodeMap, this.wayMap, this.relationMap);
151                                Id<ActivityFacility> newId = Id.create(entity.getId(), ActivityFacility.class);
152                                ActivityFacility af;
153                                if(!facilities.getFacilities().containsKey(newId)){
154                                        af = aff.createActivityFacility(newId, coord);
155                                        ((ActivityFacilityImpl)af).setDesc(name);
156                                        facilities.addActivityFacility(af);
157                                } else{
158                                        af = (ActivityFacilityImpl) facilities.getFacilities().get(newId);
159                                }
160                                ActivityOption ao = aff.createActivityOption(matsimType);
161                                af.addActivityOption(ao);
162//                              setFacilityDetails(ao);
163//                              nodeFacilities++;
164                        }
165                        
166                        /* Check shops */
167                        String shops = tags.get("shop");
168                        if(shops != null){
169                                String name = tags.get("name");
170                                if(name == null){
171                                        log.warn("      ---> Shop " + n + " without a name.");
172                                }
173
174                                /* Facility identified. Now get the centroid of all members. */ 
175//                              Coord coord = getCoord(entity);
176                                Coord coord = CoordUtils.getCentroidCoord(entity, this.ct, this.nodeMap, this.wayMap, this.relationMap);
177                                Id<ActivityFacility> newId = Id.create(entity.getId(), ActivityFacility.class);
178                                ActivityFacility af;
179                                if(!facilities.getFacilities().containsKey(newId)){
180                                        af = aff.createActivityFacility(newId, coord);                                  
181                                        ((ActivityFacilityImpl)af).setDesc(name);
182                                        facilities.addActivityFacility(af);
183                                } else{
184                                        af = (ActivityFacilityImpl) facilities.getFacilities().get(newId);
185                                }
186                                ActivityOption ao = aff.createActivityOption("s");
187                                af.addActivityOption(ao);
188//                              setFacilityDetails(ao);
189//                              shoppingCounter++;
190//                              nodeFacilities++;
191                        }
192                }
193        }

370        /**
371         * Adds each <i>OpenStreetMap</i> element to an internal container.
372         * <ul>
373         *      <li> <b>relation</b>s are add to a {@link SimpleObjectStore};
374         *      <li> <b>way</b>s are added to a {@link Map};
375         *      <li> <b>node</b>s are added to a {@link Map};
376         * </ul> 
377         */
378        @Override
379        public void process(EntityContainer entityContainer) {
380                entityContainer.process(new EntityProcessor() {
381                        
382                        @Override
383                        public void process(RelationContainer relationContainer) {
384                                relationMap.put(relationContainer.getEntity().getId(), relationContainer);                                      
385                        }
386                        
387                        @Override
388                        public void process(WayContainer wayContainer) {
389                                wayMap.put(wayContainer.getEntity().getId(), wayContainer);
390                        }
391                        
392                        @Override
393                        public void process(NodeContainer nodeContainer) {
394                                nodeMap.put(nodeContainer.getEntity().getId(), nodeContainer);
395                        }
396                        
397                        @Override
398                        public void process(BoundContainer boundContainer) {
399                                
400                        }
401                });
402        }

404        private String getActivityType(String amenity){
405                String type = typeMap.get(amenity);
406                if(type == null){
407                        log.warn("Do not have an activity type mapping for " + amenity + "! Returning NULL.");
408                } else{
409                }
410                MapUtils.addToInteger(type, typeCount, 0, 1);
411                return type;
412        }

415        private void getEducationLevel(String facilityName){
416                /* Try and figure out the type/level of school. */
417                String level = "unknown";
418                if(facilityName.contains("Primary") || facilityName.contains("Laerskool")){
419                        level="primary";
420                } else if(facilityName.contains("Secondary") || 
421                                  facilityName.contains("High") || 
422                                  facilityName.contains("Hoerskool") || 
423                                  facilityName.contains("Intermediate") || 
424                                  facilityName.contains("College")){
425                        level="secondary";
426                } else if(facilityName.contains("University")){
427                        level = "tertiary";
428                } 
429
430                educationLevelMap.put(level, educationLevelMap.get(level)+1);
431        }
