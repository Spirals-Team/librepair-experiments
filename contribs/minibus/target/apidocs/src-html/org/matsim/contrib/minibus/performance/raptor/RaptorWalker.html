035/**
036 * Transit route searching algorithm following the idea of the connection scan algorithm as described by e.g.
037 * Delling, D. and Pajor, T. and Werneck, R.F., "Round-Based Public Transit Routing" in "Sixth Annual Symposium on Combinatorial Search", 2013.<br>
038 * <br>
039 * The algorithms had been extended to allow for circle transit route, i.e. routes that serve the same transit route stop more than once.
040 * The search itself is based on earliest arrival time only. However, from the set of routes found in each round the one with the least costs is kept.
041 * That is, depending on the configuration of {@linkplain RaptorDisutility}, a longer route but with fewer transfers may be returned.<br>
042 * <br>
043 * Note that
044 * <ul>
045 *      <li>The cost (per boarding or per distance) do not respect the route used. In consequence, the algorithm will always return the fastest route although
046 *  a cheaper route with the same number of transfers may exist. For example, a high-speed train and a local stop train both offer a direct connection between two cities.
047 *  The algorithm returns always the high-speed connection.
048 *      <li>Changing the {@linkplain RaptorDisutility} may not be sufficient in case a) both connections serve the same target stops or b) the target stops are too
049 *  close to each other so that transferring from the high-speed stop to the local train stop allows for an earlier arrival time than taking the stop in the first place.
050 *      <li>The algorithm allows for transferring multiple times without having to board a vehicle. Thus, longer transfers are possible if this is indeed faster. In the end, consecutive transfers are merged.
051 * </ul>
052 * 
053 * The algorithm works in rounds. In detail
054 * <ol>
055 *      <li>Search for a all transit stop facilities within a given range from the coordinate of the start activity.
056 *      <li>Parse all transit routes serving these transit stop facilities, each starting from the transit route stop associated with the transit stop facility.
057 *      <li>Check all transit stop facilities that could be updated with a earlier arrival time for possible transfers.
058 *      <li>Start again with 2. and check the routes serving the transfer destination stops.
059 * </ol>
060 * Each route stop has a back pointer pointing to its predecessor (route stop).<br>
061 * <br>
062 * 
063 * In each round, starting with zero transfers, the best route is retrieved by
064 * <ol>
065 *      <li>Take all transit stop facilities within a given range of the coordinate of the target activity.
066 *      <li>Backtrace from each destination stop to the starting stops and create a sequence of route segments.
067 *      <li>Score each sequence of route segments.
068 *      <li>Return the sequence with the least costs.
069 * </ol>
070 * The best route of a round is compared with the best route of the next round. The better one is kept and eventually returned.  
071 *
072 * The algorithm returns null if no route is found after the given number of max transfers.
073 * Once a route is found the algorithm runs for an additional number of rounds specified by the number of grace transfers.
074 *  
075 * 
076 * @author aneumann
077 *
078 */

118        public TransitPassengerRoute calcLeastCostPath(Map<TransitStopFacility, InitialNode> fromTransitStops, Map<TransitStopFacility, InitialNode> toTransitStops) {
119                
120                TransitPassengerRoute bestRoute = null;
121
122                // init
123                Arrays.fill(this.earliestArrivalTimeAtRouteStop, Double.POSITIVE_INFINITY);
124                Arrays.fill(this.sourcePointerRouteStop, this.initialSourcePointer);
125                Arrays.fill(this.earliestArrivalTimeAtTransitStop, Double.POSITIVE_INFINITY);
126                Arrays.fill(this.sourcePointerTransitStops, this.initialSourcePointer);
127                Arrays.fill(this.routeStopsToCheck, false);
128                Arrays.fill(this.transferTransitStopsToCheck, false);
129
130                // init first stops
131                for (TransitStopFacility fromTransitStop : fromTransitStops.keySet()) {
132                        // (these are all the transit stops initialized by the Multi-Node router)
133
134                        double departureTime = fromTransitStops.get(fromTransitStop).initialTime;
135                        // (get dp time from node (since they vary by multi-node))
136
137                        int indexOfFromTransitStop = this.getIndexForTransitStop(fromTransitStop);
138                        TransitStopEntry transitStopEntry = this.raptorSearchData.stops[indexOfFromTransitStop];
139                        // (look up index and size of corresponding transit stop entry (fortran))
140
141                        for (int indexOfTransferToCheck = transitStopEntry.indexOfFirstTransfer; indexOfTransferToCheck < transitStopEntry.indexOfFirstTransfer + transitStopEntry.numberOfTransfers; indexOfTransferToCheck++) {
142                                // check all route stops that can be reached from this transit stop
143                                // ignore transfers that result in a different transit stop facility
144                                // otherwise we would extend the search radius
145                                TransferEntry transferEntry = this.raptorSearchData.transfers[indexOfTransferToCheck];
146                                RouteStopEntry transferRouteStopEntry = this.raptorSearchData.routeStops[transferEntry.indexOfRouteStop];
147                                if (indexOfFromTransitStop == transferRouteStopEntry.indexOfStopFacility) {
148                                        // this is a route stop of the transit stop facility
149                                        SourcePointer source = new SourcePointer(departureTime, transferEntry.indexOfRouteStop, null, false);
150                                        sourcePointerRouteStop[transferEntry.indexOfRouteStop] = source;
151                                        earliestArrivalTimeAtRouteStop[transferEntry.indexOfRouteStop] = departureTime;
152                                        earliestArrivalTimeAtTransitStop[indexOfFromTransitStop] = departureTime;
153                                        routeStopsToCheck[transferEntry.indexOfRouteStop] = true;
154                                }
155                        }
156                }
157
158                int graceTransfersLeft = this.graceTransfers;
159
160                // MAIN LOOP:
161                for (int nTransfers = 0; nTransfers <= this.maxTransfers; nTransfers++) {
162                        Arrays.fill(transferTransitStopsToCheck, false);
163
164                        this.checkRouteStops();
165
166                        TransitPassengerRoute bestRouteOfThisRound = this.getBestRouteFoundSoFar(fromTransitStops, toTransitStops, sourcePointerTransitStops);
167                        if (bestRouteOfThisRound != null) {
168                                if (bestRoute == null) {
169                                        // this is the first route found
170                                        bestRoute = bestRouteOfThisRound;
171                                } else {
172                                        // check if this one is better
173                                        if (bestRouteOfThisRound.getTravelCost() < bestRoute.getTravelCost()) {
174                                                bestRoute = bestRouteOfThisRound;
175                                        }
176                                }
177                        }
178
179
180                        if (bestRoute != null) {
181                                // one route found - check how many additional runs are allowed
182                                if (graceTransfersLeft == 0) {
183                                        // no more transfers left - abort and return best route found so far
184                                        return bestRoute;
185                                }
186                                graceTransfersLeft--;
187                        }
188
189                        Arrays.fill(routeStopsToCheck, false);
190                        // we have now explored every place without additional transfers
191
192                        // increase the transfers and proceed
193                        if(nTransfers < this.maxTransfers){
194//                              nTransfers++;
195
196                                // yyyy to me, this looks like in the end we are now going 0, 2, 4, 6, since we are incrementing both here and in
197                                // the for loop.  ????  kai, jun'16
198                                //right, there is no reason to increase the counter twice. Commenting one. Amit Jan'18
199
200                                this.checkTransferTransitStops();
201                        }
202                }
203
204                return bestRoute;
205        }

207        private void checkRouteStops() {
208
209                boolean atLeastOneRouteStopImproved = false;
210                int indexOfLastRouteProcessed = -1;
211
212                // process all start stops
213                for (int indexOfStartRouteStop = 0; indexOfStartRouteStop < routeStopsToCheck.length; indexOfStartRouteStop++) {
214                        // I think that these are all route stops, and the "check" refers to the boolean
215
216                        if (routeStopsToCheck[indexOfStartRouteStop] == true) {
217                                RouteStopEntry startStop = this.raptorSearchData.routeStops[indexOfStartRouteStop];
218                                RouteEntry routeToCheck = this.raptorSearchData.routes[startStop.indexOfRoute];
219                                
220                                // TODO do not proceed if this stop's best arrival time has just got updated in this round
221                                // updates at this part of the round can only occur by parsing the same route but from one of the upstream stops
222                                // thus all following downstream stops were already updated as well and have all a better arrival time as possibly this stop's departure would yield
223
224                                // this is not entirely true because if one of the intermediate stop get a better arrival time (by transfer).
225                                // This means, if any of the intermediate stop is not improved, let it run if the same route is encounterd again.
226                                // doing this by setting "atLeastOneRouteStopImproved = false;" and breaking the process of updating arrival times of rest of the stops of the route. Amit Sep'17
227
228                                if (indexOfLastRouteProcessed == startStop.indexOfRoute && atLeastOneRouteStopImproved) {
229                                        // we process the same route again
230                                        // since route stops have increasing indices we only need to process all downstream stops again if the last try could not improve the arrival times
231
232                                        // do nothing and proceed
233                                } else {
234                                        // this is a different route - process all stops
235                                        indexOfLastRouteProcessed = startStop.indexOfRoute;
236                                        atLeastOneRouteStopImproved = false;
237
238                                        double earliestArrivalTimeAtStartRouteStop = sourcePointerRouteStop[indexOfStartRouteStop].earliestArrivalTime;
239
240                                        int indexOfRouteStopWithinRouteSequence = routeToCheck.numberOfRouteStops - startStop.numberOfRemainingStopsInThisRoute - 1;
241                                        int indexOfEarliestDepartureTime = this.getIndexOfEarliestDepartureTime(earliestArrivalTimeAtStartRouteStop, routeToCheck, indexOfRouteStopWithinRouteSequence);
242
243                                        // It appears that if departure time is after midnight, it needs to be updated here
244                                        // so that an earliest connection for next day can be looked.
245                                        // However, the arrival time in the array is actual time. Amit Aug'17
246                                        double adjustedEarliestArrivalTimeAtStartRouteStop  = earliestArrivalTimeAtStartRouteStop;
247
248                                        // transit router may have departures after midnight, so if there is no available departure for rest of the day, take earliest departure next day.
249                                        if (indexOfEarliestDepartureTime <= -1 && adjustedEarliestArrivalTimeAtStartRouteStop >= RaptorDisutility.MIDNIGHT ) {
250                                                adjustedEarliestArrivalTimeAtStartRouteStop = adjustedEarliestArrivalTimeAtStartRouteStop % RaptorDisutility.MIDNIGHT;
251                                        }
252
253                                        indexOfEarliestDepartureTime = this.getIndexOfEarliestDepartureTime(adjustedEarliestArrivalTimeAtStartRouteStop, routeToCheck, indexOfRouteStopWithinRouteSequence);
254
255                                        if(indexOfEarliestDepartureTime > -1) {
256                                                // we have found a valid departure time - process all upcoming stops of the route
257                                                for (int indexOfRouteStopToCheck = indexOfStartRouteStop + 1; indexOfRouteStopToCheck < indexOfStartRouteStop + startStop.numberOfRemainingStopsInThisRoute + 1; indexOfRouteStopToCheck++) {
258
259                                                        // we go now one stop further
260                                                        indexOfEarliestDepartureTime += routeToCheck.numberOfDepartures;
261                                                        RouteStopEntry routeStopToCheck =  this.raptorSearchData.routeStops[indexOfRouteStopToCheck];
262
263                                                        double arrivalTimeAtTheFollowingRouteStop = this.raptorSearchData.arrivalTimes[indexOfEarliestDepartureTime];
264
265                                                        while (arrivalTimeAtTheFollowingRouteStop < earliestArrivalTimeAtStartRouteStop ) {
266                                                                arrivalTimeAtTheFollowingRouteStop += RaptorDisutility.MIDNIGHT;
267                                                                // (add enough "MIDNIGHT"s until we are _after_ the desired departure time)
268                                                        }
269
270                                                        if (arrivalTimeAtTheFollowingRouteStop < earliestArrivalTimeAtRouteStop[routeStopToCheck.indexOfRouteStop]) {
271                                                                // this really is better than anything before - set arrival and source and mark the stop to be checked for transfers
272                                                                atLeastOneRouteStopImproved = true;
273
274                                                                SourcePointer source = new SourcePointer(arrivalTimeAtTheFollowingRouteStop, routeStopToCheck.indexOfRouteStop, sourcePointerRouteStop[indexOfStartRouteStop], false);
275                                                                sourcePointerRouteStop[routeStopToCheck.indexOfRouteStop] = source;
276                                                                earliestArrivalTimeAtRouteStop[routeStopToCheck.indexOfRouteStop] = arrivalTimeAtTheFollowingRouteStop;
277
278                                                                // is it also an earlier arrival time at the transit stop
279                                                                if (arrivalTimeAtTheFollowingRouteStop < earliestArrivalTimeAtTransitStop[routeStopToCheck.indexOfStopFacility]) {
280                                                                        earliestArrivalTimeAtTransitStop[routeStopToCheck.indexOfStopFacility] = arrivalTimeAtTheFollowingRouteStop;
281                                                                        sourcePointerTransitStops[routeStopToCheck.indexOfStopFacility] = source;
282                                                                        transferTransitStopsToCheck[routeStopToCheck.indexOfStopFacility] = true;
283                                                                }
284                                                        } else {
285                                                                atLeastOneRouteStopImproved = false;
286                                                                break;
287                                                        }
288                                                }
289                                        } else {
290                                                // there is no further departure
291                                                // TODO search for the earliest departure and add 24h
292                                                // implemented above by adjusting the time; still keeping the comment until all tests are happy. Amit Aug'17
293
294                                                // another possibility, there is no futher departures on this stop for this route (routeToCheck), but let's say agent is here and now make another transfer.
295                                                // this will increase the effective "maxBeelineWalkConnectionDistance"... Amit Aug'17
296                                                transferTransitStopsToCheck[startStop.indexOfStopFacility] = true;
297                                        }
298                                }
299
300                        }
301
302                }
303        }

305        private void checkTransferTransitStops() {
306
307                // check all transfer stops
308                for (int indexOfTransitStopToCheck = 0; indexOfTransitStopToCheck < transferTransitStopsToCheck.length; indexOfTransitStopToCheck++) {
309                        if (transferTransitStopsToCheck[indexOfTransitStopToCheck]) {
310                                TransitStopEntry transitStopEntry = this.raptorSearchData.stops[indexOfTransitStopToCheck];
311
312                                for (int indexOfTransferToCheck = transitStopEntry.indexOfFirstTransfer; indexOfTransferToCheck < transitStopEntry.indexOfFirstTransfer + transitStopEntry.numberOfTransfers; indexOfTransferToCheck++) {
313                                        // process all transfers of the stop
314
315                                        TransferEntry transferDestination = this.raptorSearchData.transfers[indexOfTransferToCheck];
316                                        int indexOfTransferRouteStop = transferDestination.indexOfRouteStop;
317                                        double arrivalTimeAtTheTransferTargetRouteStop = earliestArrivalTimeAtTransitStop[indexOfTransitStopToCheck] + transferDestination.transferTime;
318
319                                        if (arrivalTimeAtTheTransferTargetRouteStop < earliestArrivalTimeAtRouteStop[indexOfTransferRouteStop]) {
320                                                // this really is better than anything before - set arrival and source and mark the stop to be checked in the next iteration
321                                                SourcePointer source = new SourcePointer(arrivalTimeAtTheTransferTargetRouteStop, indexOfTransferRouteStop, sourcePointerTransitStops[indexOfTransitStopToCheck], true);
322                                                sourcePointerRouteStop[indexOfTransferRouteStop] = source;
323                                                earliestArrivalTimeAtRouteStop[indexOfTransferRouteStop] = arrivalTimeAtTheTransferTargetRouteStop;
324                                                routeStopsToCheck[indexOfTransferRouteStop] = true;
325
326                                                int indexOfTargetTransitStop = this.raptorSearchData.routeStops[indexOfTransferRouteStop].indexOfStopFacility;
327                                                if (arrivalTimeAtTheTransferTargetRouteStop < earliestArrivalTimeAtTransitStop[indexOfTargetTransitStop]) {
328                                                        // this is also the best arrival time for the transit facility of the route stop
329                                                        earliestArrivalTimeAtTransitStop[indexOfTargetTransitStop] = arrivalTimeAtTheTransferTargetRouteStop;
330                                                        sourcePointerTransitStops[indexOfTargetTransitStop] = source;
331                                                }
332                                        }
333                                }
334
335                        }
336
337                }
338        }

340        private TransitPassengerRoute getBestRouteFoundSoFar(Map<TransitStopFacility, InitialNode> fromTransitStops, Map<TransitStopFacility, InitialNode> toTransitStops, SourcePointer[] sourcePointerTransitStops) {
341                // get routes for all n of transfers and return them
342                List<List<RouteSegment>> routesFound = new LinkedList<List<RouteSegment>>();
343                for (TransitStopFacility toTransitStop : toTransitStops.keySet()) {
344                        int indexOfToTransitStop = this.getIndexForTransitStop(toTransitStop);
345                                        
346                        SourcePointer source = sourcePointerTransitStops[indexOfToTransitStop];
347                        if (source.source != null) {
348                                // something found - backtrace
349                                List<RouteSegment> route = this.returnBacktracedRouteFromSourcePointer(source, fromTransitStops);
350                                if (!route.isEmpty()) {
351                                        // there is a route with this number of transfers to that stop
352                                        routesFound.add(route);
353                                }
354                        }
355                }
356                
357                if (!routesFound.isEmpty()) {
358                        // score and pick best
359                        return this.scoreRoutesAndReturnBest(routesFound, fromTransitStops, toTransitStops);
360                } else {
361                        return null;
362                }
363        }

418        private List<RouteSegment> returnBacktracedRouteFromSourcePointer(SourcePointer sourcePointer, Map<TransitStopFacility, InitialNode> fromTransitStops) {
419                List<RouteSegment> route = new LinkedList<>();
420                
421                SourcePointer currentSourcePointer = sourcePointer;
422                RouteSegment lastRouteSegment = null;
423                
424                while (currentSourcePointer.source != null) {
425
426                        RouteStopEntry toRouteStopEntry = this.raptorSearchData.routeStops[currentSourcePointer.indexOfTargetRouteStop];
427                        TransitStopEntry transitStopEntry = this.raptorSearchData.stops[toRouteStopEntry.indexOfStopFacility];
428                        TransitStopFacility toTransitStop = transitStopEntry.transitStopFacility;
429
430                        if (currentSourcePointer.source.indexOfTargetRouteStop == -1 ){
431                                if (currentSourcePointer.transfer) {
432                                        // access_stop and transfer
433                                        TransitStopFacility fromTransitStop = null;
434                                        double dist = Double.POSITIVE_INFINITY;
435                                        for (TransitStopFacility possibleFromStop : fromTransitStops.keySet()){ // an alternative would be to get the transfer stops at _toTransitStop_ and match with fromStops. Amit Jan'18
436                                                double tempDist = NetworkUtils.getEuclideanDistance(possibleFromStop.getCoord().getX(), possibleFromStop.getCoord().getY(), toTransitStop.getCoord().getX(), toTransitStop.getCoord().getY());
437                                                if (tempDist < dist) {
438                                                        dist = tempDist;
439                                                        fromTransitStop = possibleFromStop;
440                                                }
441                                        }
442
443                                        if (fromTransitStop!=null) {
444                                                RouteSegment routeSegment = new RouteSegment(fromTransitStop,
445                                                                toTransitStop,
446                                                                this.raptorSearchData.transfers[getIndexForTransitStop(fromTransitStop)].transferTime,
447                                                                null,
448                                                                null);
449                                                route.add(0,routeSegment);
450                                        }
451                                } else {
452                                        // this additional condition is required to exclude the case in which origin and destination are same transit stop (source.source.indexOfTargetRouteStop=-1).
453                                        // This can be verified by something like "this.raptorSearchData.routeStops[source.indexOfTargetRouteStop].indexOfStopFacility != indexOfToTransitStop"
454                                        // in getBestRouteFoundSoFar(...) method. However, if above condition is used in "getBestRouteFoundSoFar",
455                                        // it still throws exception here, so, excluding such situations here. Amit Jan'18
456                                }
457                        } else {
458                                RouteStopEntry fromRouteStopEntry = this.raptorSearchData.routeStops[currentSourcePointer.source.indexOfTargetRouteStop];
459                                TransitStopFacility fromTransitStop = this.raptorSearchData.stops[fromRouteStopEntry.indexOfStopFacility].transitStopFacility;
460
461                                double travelTime = currentSourcePointer.earliestArrivalTime - currentSourcePointer.source.earliestArrivalTime;
462
463                                RouteSegment routeSegment;
464                                if (currentSourcePointer.transfer) {
465                                        if (lastRouteSegment != null) {
466                                                if (lastRouteSegment.getRouteTaken() == null) {
467                                                        // had been a transfer, too - merge both
468                                                        toTransitStop = lastRouteSegment.getToStop();
469                                                        travelTime += lastRouteSegment.getTravelTime();
470                                                        route.remove(0);
471                                                }
472                                        }
473                                        routeSegment = new RouteSegment(fromTransitStop, toTransitStop, travelTime, null, null);
474                                } else {
475                                        RouteStopEntry routeStopEntry = this.raptorSearchData.routeStops[currentSourcePointer.indexOfTargetRouteStop];
476                                        RouteEntry routeEntry = this.raptorSearchData.routes[routeStopEntry.indexOfRoute];
477                                        routeSegment = new RouteSegment(fromTransitStop, toTransitStop, travelTime, routeEntry.lineId, routeEntry.routeId);
478                                }
479                                route.add(0, routeSegment);
480                                lastRouteSegment = routeSegment;
481                        }
482                        // set pointer to next alighting stop, skip transfers
483                        currentSourcePointer = currentSourcePointer.source;
484                }
485                
486                return route;
487        }

489        private TransitPassengerRoute scoreRoutesAndReturnBest(List<List<RouteSegment>> routesFound, Map<TransitStopFacility, InitialNode> fromTransitStops, Map<TransitStopFacility, InitialNode> toTransitStops) {
490                
491                TransitPassengerRoute bestRouteSoFar = null;
492                
493                for (List<RouteSegment> route : routesFound) {
494                        double cost = this.scoreRoute(route, fromTransitStops, toTransitStops);
495                        if (bestRouteSoFar == null) {
496                                bestRouteSoFar = new TransitPassengerRoute(cost, route);
497                        } else if (cost < bestRouteSoFar.getTravelCost()) {
498                                bestRouteSoFar = new TransitPassengerRoute(cost, route);
499                        }
500                }
501                
502                return bestRouteSoFar;
503        }
504
505        private double scoreRoute(List<RouteSegment> route, Map<TransitStopFacility, InitialNode> fromStops, Map<TransitStopFacility, InitialNode> toStops) {
506                
507                double cost = 0.0;
508                
509                for (RouteSegment routeSegment : route) {
510                        if (routeSegment.getRouteTaken() == null) {
511                                // handle transfer
512                                cost += this.raptorDisutility.getTransferCost(routeSegment.getFromStop().getCoord(), routeSegment.getToStop().getCoord());
513                        } else {
514                                // pt trip
515                                cost += this.raptorDisutility.getInVehicleTravelDisutility(routeSegment);
516                        }
517                }
518                
519                // add cost for getting to the first and last stop
520                cost += fromStops.get(route.get(0).getFromStop()).initialCost;
521                cost += toStops.get(route.get(route.size() - 1).getToStop()).initialCost;
522                
523                return cost;
524        }
