021
022import java.util.HashMap;
023
024import org.apache.log4j.Logger;
025import org.matsim.api.core.v01.Id;
026import org.matsim.api.core.v01.Scenario;

034/**
035 * Merges all routes of a transit line that have the same sequence of stops. Does not respect the time profile of the routes.
036 * 
037 * @author aneumann
038 *
039 */
040public class PTransitLineMerger {
041        
042        private static final Logger log = Logger.getLogger(PTransitLineMerger.class);
043        
044        /**
045         * Merges all routes of a transit line that have the same sequence of stops. Does not respect the time profile of the routes.
046         * 
047         * @param transitSchedule to be modified.
048         * @return A copy of the transit schedule with transit lines merged.
049         */
050        public static TransitSchedule mergeSimilarRoutes(TransitSchedule transitScheduleOrig){
051                printStatistic(transitScheduleOrig);
052                
053                TransitScheduleFactory transitScheduleFactory = transitScheduleOrig.getFactory();
054                TransitSchedule transitSchedule = transitScheduleFactory.createTransitSchedule();
055                
056                for (TransitStopFacility stop : transitScheduleOrig.getFacilities().values()) {
057                        transitSchedule.addStopFacility(stop);                  
058                }
059                
060                for (TransitLine oldLine : transitScheduleOrig.getTransitLines().values()) {
061                        TransitLine newLine = mergeTransitLine(transitScheduleFactory, oldLine);
062                        transitSchedule.addTransitLine(newLine);
063                }
064                
065                printStatistic(transitSchedule);
066                return transitSchedule;
067        }
068
069        /**
070         * Merges all routes of a transit line that have the same sequence of stops. Does not respect the time profile of the routes.
071         * 
072         * @param oldLine The transit line containing the routes to be merged.
073         * @return A copy of the transit line with its transit routes merged.
074         */
075        public static TransitLine mergeTransitLine(TransitLine oldLine) {
076                return mergeTransitLine(new TransitScheduleFactoryImpl(), oldLine);
077        }
078        
079        private static TransitLine mergeTransitLine(TransitScheduleFactory transitScheduleFactory, TransitLine oldLine) {
080                TransitLine newLine = transitScheduleFactory.createTransitLine(oldLine.getId());
081                
082                HashMap<String, TransitRoute> routeHash2TransitRoute = new HashMap<String, TransitRoute>();
083                
084                for (TransitRoute transitRoute : oldLine.getRoutes().values()) {
085                        String routeHash = getHashForRoute(transitRoute);
086                        
087                        if (routeHash2TransitRoute.keySet().contains(routeHash)) {
088                                // route with same stop sequence exists - merge departures and drop the route
089                                // Note that this ignores the exact time profile of this route.
090                                // No severe impact for minibuses since the time profile varies by seconds only
091                                
092                                TransitRoute mergeDestination = routeHash2TransitRoute.get(routeHash);
093                                
094                                for (Departure oldDeparture : transitRoute.getDepartures().values()) {
095                                        // increase departure id
096                                        Id<Departure> newDepartureId = Id.create("new_" + mergeDestination.getDepartures().values().size(), Departure.class);
097                                        Departure newDeparture = transitScheduleFactory.createDeparture(newDepartureId, oldDeparture.getDepartureTime());
098                                        newDeparture.setVehicleId(oldDeparture.getVehicleId());
099                                        mergeDestination.addDeparture(newDeparture);
100                                }
101                                
102                                // drop the route by not adding it to the new transit line
103                                
104                        } else {
105                                // new route - add it
106                                routeHash2TransitRoute.put(routeHash, transitRoute);
107                                newLine.addRoute(transitRoute);
108                        }
109                }
110                
111                return newLine;
112        }
113        
114        /**
115         * Create a simple hash unique for a certain sequence of stops.
116         * 
117         * @param transitRoute
118         * @return
119         */
120        private static String getHashForRoute(TransitRoute transitRoute){
121                StringBuffer strB = null;
122                for (TransitRouteStop transitStop : transitRoute.getStops()) {
123                        if (strB == null) {
124                                strB = new StringBuffer();
125                        } else {
126                                strB.append("-");
127                        }
128                        strB.append(transitStop.getStopFacility().getId().toString());
129                }
130                
131                return strB.toString();
132        }
133        
134        private static void printStatistic(TransitSchedule transitSchedule){
135                int nStops = 0;
136                int nLines = 0;
137                int nRoutes = 0;
138                int nDepartures = 0;
139                
140                nStops = transitSchedule.getFacilities().values().size();
141                
142                for (TransitLine transitLine : transitSchedule.getTransitLines().values()) {
143                        nLines++;
144                        
145                        for (TransitRoute transitRoute : transitLine.getRoutes().values()) {
146                                nRoutes++;
147                                nDepartures += transitRoute.getDepartures().size();
148                        }
149                }
150                
151                log.info("Transit schedule stats: " + nStops + " stops, " + nLines + " lines, " + nRoutes + " routes, " + nDepartures + " departures.");
152        }
153
154        public static void main(String[] args) {
155                String folder = "e:/transitScheduleTest/";
156                String networkFile = folder + "network.final.xml.gz";
157                String transitScheduleInFile = folder + "bvg6_b_0.1250.transitSchedule.xml.gz";
158                String transitScheduleOutFile = folder + "bvg6_b_0.1250.transitSchedule_merged.xml.gz";
159                String vehicleFile = folder + "bvg6_b_0.1250.vehicles.xml.gz";
160                
161                Scenario scenario =  ScenarioUtils.createScenario(ConfigUtils.createConfig());
162                scenario.getConfig().transit().setUseTransit(true);
163                
164                MatsimNetworkReader networkReader = new MatsimNetworkReader(scenario.getNetwork());
165                networkReader.readFile(networkFile);
166                
167                VehicleReaderV1 vehicleReader = new VehicleReaderV1(scenario.getTransitVehicles());
168                vehicleReader.readFile(vehicleFile);
169                TransitScheduleReader scheduleReader = new TransitScheduleReader(scenario);
170                scheduleReader.readFile(transitScheduleInFile);
171                
172                PTransitLineMerger.mergeSimilarRoutes(scenario.getTransitSchedule());
173                
174                TransitScheduleWriter writer = new TransitScheduleWriter(scenario.getTransitSchedule());
175                writer.writeFile(transitScheduleOutFile);
176        }
