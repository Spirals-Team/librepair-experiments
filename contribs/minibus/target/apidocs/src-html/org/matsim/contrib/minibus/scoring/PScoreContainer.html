020package org.matsim.contrib.minibus.scoring;
021
022import org.apache.log4j.Logger;
023import org.matsim.api.core.v01.Id;
024import org.matsim.contrib.minibus.fare.StageContainer;
025import org.matsim.contrib.minibus.fare.TicketMachineI;
026import org.matsim.vehicles.Vehicle;
027
028/**
029 * Simple container class collecting all incomes and expenses for one single vehicle.
030 * 
031 * @author aneumann
032 *
033 */
034public final class PScoreContainer {
035        
036        @SuppressWarnings("unused")
037        private final static Logger log = Logger.getLogger(PScoreContainer.class);
038
039        private final Id<Vehicle> vehicleId;
040        private final TicketMachineI ticketMachine;
041        private boolean isFirstTour = true;
042        
043        private int servedTrips = 0;
044        private double costs = 0;
045        private double earnings = 0;
046        
047        public PScoreContainer(Id<Vehicle> vehicleId, TicketMachineI ticketMachine) {
048                this.vehicleId = vehicleId;
049                this.ticketMachine = ticketMachine;
050        }
051
052        public void handleStageContainer(StageContainer stageContainer) {
053                this.servedTrips++;
054                this.earnings += this.ticketMachine.getFare(stageContainer);
055        }
056
057        public void handleOperatorCostContainer(OperatorCostContainer operatorCostContainer) {
058                if (this.isFirstTour) {
059                        this.costs += operatorCostContainer.getFixedCostPerDay();
060                        this.isFirstTour = false;
061                }
062                this.costs += operatorCostContainer.getRunningCostDistance();
063                this.costs += operatorCostContainer.getRunningCostTime();
064        }
065
066        public double getTotalRevenue(){
067                return this.earnings - this.costs;
068        }
069        
070        public double getTotalRevenuePerPassenger(){
071                if(this.servedTrips == 0){
072                        return Double.NaN;
073                } else {
074                        return (this.earnings - this.costs) / this.servedTrips;
075                }
076        }
077        
078        public int getTripsServed(){
079                return this.servedTrips;
080        }
081        
082        @Override
083        public String toString() {
084                return "Paratransit vehicle " + this.vehicleId.toString() + " served " + this.servedTrips + " trips spending a total of " + this.costs + " vs. " + this.earnings + " earnings";
085        }
086}
