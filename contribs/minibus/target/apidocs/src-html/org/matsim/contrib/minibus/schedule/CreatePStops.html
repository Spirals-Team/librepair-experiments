045 * Create one TransitStopFacility for each car mode link of the network

054        private final Network net;

068        public static TransitSchedule createPStops(Network network, PConfigGroup pConfigGroup){
069                return createPStops(network, pConfigGroup, null);

072        public static TransitSchedule createPStops(Network network, PConfigGroup pConfigGroup, TransitSchedule realTransitSchedule) {
073                CreatePStops cS = new CreatePStops(network, pConfigGroup, realTransitSchedule);
074                cS.run();
075                return cS.getTransitSchedule();
076        }

078        /**
079         * Creates PStops in two ways. First, if a serviceAreaFile is defined in the config and this file exists, the file is used.
080         * Second, the (default) min/max-x/y-values are used.
081         * 
082         * Following FileTypes are supported:
083         * <ul>
084         *      <li>Shapefiles with polygons. If one ore more attributes are defined, the last one is parsed 
085         *              to Boolean and used to get include- and exclude-areas.</li>
086         *      <li>Textfile, containing a List of x/y-pairs per row, divided by semicolon. The first and the last coordinate should be equal
087         *              to get a closed and well defined Geometry.</li>
088         * </ul>
089         * @param net
090         * @param pConfigGroup
091         * @param realTransitSchedule
092         */
093    private CreatePStops(Network net, PConfigGroup pConfigGroup, TransitSchedule realTransitSchedule) {
094                this.net = net;
095                this.pConfigGroup = pConfigGroup;
096                this.factory = new GeometryFactory();
097                
098                this.linkId2StopFacilityMap = new LinkedHashMap<>();
099                
100                Set<Id<TransitStopFacility>> stopsWithoutLinkIds = new TreeSet<>();
101
102                int warnCounter = 10;
103                
104                if (realTransitSchedule != null) {
105                        for (TransitStopFacility stopFacility : realTransitSchedule.getFacilities().values()) {
106                                if (stopFacility.getLinkId() != null) {
107                                        if (this.linkId2StopFacilityMap.get(stopFacility.getLinkId()) != null) {
108                                                if (warnCounter > 0) {
109                                                        log.warn("There is more than one stop registered on link " + stopFacility.getLinkId() + ". "
110                                                                        + this.linkId2StopFacilityMap.get(stopFacility.getLinkId()).getId() + " stays registered as paratransit stop. Will ignore stop " + stopFacility.getId());
111                                                        warnCounter--;
112                                                } if (warnCounter == 0) {
113                                                        log.warn("Future occurences of this logging statement are suppressed.");
114                                                        warnCounter--;
115                                                }
116                                        } else {
117                                                this.linkId2StopFacilityMap.put(stopFacility.getLinkId(), stopFacility);
118                                        }
119                                } else {
120                                        stopsWithoutLinkIds.add(stopFacility.getId());
121                                }
122                        }
123                }
124                
125                this.exclude = this.factory.buildGeometry(new ArrayList<Geometry>());
126                if(!new File(pConfigGroup.getServiceAreaFile()).exists()){
127                        log.warn("file " + this.pConfigGroup.getServiceAreaFile() + " not found. Falling back to min/max serviceArea parameters.");
128                        createServiceArea(pConfigGroup.getMinX(), pConfigGroup.getMaxX(), pConfigGroup.getMinY(), pConfigGroup.getMaxY());
129                }else{
130                        log.warn("using " + this.pConfigGroup.getServiceAreaFile() + " for servicearea. x/y-values defined in the config are not used.");
131                        createServiceArea(pConfigGroup.getServiceAreaFile());
132                }
133                
134                if (stopsWithoutLinkIds.size() > 0) {
135                        log.warn("There are " + stopsWithoutLinkIds.size() + " stop facilities without a link id, namely: " + stopsWithoutLinkIds.toString());
136                }
137                this.topoTypesForStops = this.pConfigGroup.getTopoTypesForStops();
138                if(!(this.topoTypesForStops == null)){
139                        this.networkCalcTopoType = new NetworkCalcTopoType();
140                        this.networkCalcTopoType.run(net);
141                }
142        }
143
144        /**
145         * @param minX
146         * @param maxX
147         * @param minY
148         * @param maxY
149         */
150        private void createServiceArea(double minX, double maxX, double minY, double maxY) {
151                Coordinate[] c = new Coordinate[4];
152                c[0] = new Coordinate(minX, minY);
153                c[1] = new Coordinate(minX, maxY);
154                c[2] = new Coordinate(maxX, minY);
155                c[3] = new Coordinate(maxX, maxY);
156                this.include = this.factory.createMultiPoint(c).convexHull();
157        }
158
159        /**
160         * @param serviceAreaFile
161         */
162        private void createServiceArea(String serviceAreaFile) {
163                if(serviceAreaFile.endsWith(".txt")){
164                        createServiceAreaTxt(serviceAreaFile);
165                }else if (serviceAreaFile.endsWith(".shp")){
166                        createServiceAreaShp(serviceAreaFile);
167                }else{
168                        log.warn(serviceAreaFile + ". unknown filetype. Falling back to simple x/y-values...");
169                        this.createServiceArea(pConfigGroup.getMinX(), pConfigGroup.getMaxX(), pConfigGroup.getMinY(), pConfigGroup.getMaxY());
170                }
171        }
172
173        /**
174         * @param serviceAreaFile
175         */
176        private void createServiceAreaTxt(String serviceAreaFile) {
177                
178                List<String> lines = new ArrayList<>();
179                String line;
180                try {
181                        BufferedReader reader = IOUtils.getBufferedReader(serviceAreaFile);
182                        line = reader.readLine();
183                        do{
184                                if(!(line == null)){
185                                        if(line.contains(";")){
186                                                lines.add(line);
187                                        }
188                                        line = reader.readLine();
189                                }
190                        }while(!(line == null));
191                        reader.close();
192                } catch (IOException e) {
193                        e.printStackTrace();
194                }
195                
196                if(lines.size() < 3){
197                        log.warn("an area needs at least 3 points, to be defined. Falling back to simple (default) x/y-values...");
198                        this.createServiceArea(pConfigGroup.getMinX(), pConfigGroup.getMaxX(), pConfigGroup.getMinY(), pConfigGroup.getMaxY());
199                        return; 
200                }
201                
202                Coordinate[] c = new Coordinate[lines.size() + 1];
203                        
204                double x,y;
205                for(int i = 0; i < lines.size(); i++){
206                        x = Double.parseDouble(lines.get(i).split(";")[0]);
207                        y = Double.parseDouble(lines.get(i).split(";")[1]);
208                        c[i] = new Coordinate(x, y);
209                }
210                // a linear ring has to be closed, so add the first coordinate again at the end
211                c[lines.size()] = c[0];
212                this.include = this.factory.createPolygon(this.factory.createLinearRing(c), null);
213        }
214
215        /**
216         * @param serviceAreaFile
217         */
218        private void createServiceAreaShp(String serviceAreaFile) {
219                Collection<SimpleFeature> features = new ShapeFileReader().readFileAndInitialize(serviceAreaFile);
220                Collection<Geometry> include = new ArrayList<>();
221                Collection<Geometry> exclude = new ArrayList<>();
222                
223                for(SimpleFeature f: features){
224                        boolean incl = true;
225                        Geometry g = null;
226                        for(Object o: f.getAttributes()){
227                                if(o instanceof Polygon){
228                                        g = (Geometry) o;
229                                }else if (o instanceof MultiPolygon){
230                                        g = (Geometry) o;
231                                }
232                                // TODO use a better way to get the attributes, maybe directly per index.
233                                // Now the last attribute is used per default... 
234                                else if (o instanceof String){
235                                        incl = Boolean.parseBoolean((String) o);
236                                }
237                        }
238                        if(! (g == null)){
239                                if(incl){
240                                        include.add(g);
241                                }else{
242                                        exclude.add(g);
243                                }
244                        }
245                }
246                this.include = this.factory.createGeometryCollection( 
247                                include.toArray(new Geometry[include.size()])).buffer(0);
248                this.exclude = this.factory.createGeometryCollection( 
249                                exclude.toArray(new Geometry[exclude.size()])).buffer(0);
250        }
251
252        private void run(){
253                this.transitSchedule = new TransitScheduleFactoryImpl().createTransitSchedule();
254                int stopsAdded = 0;
255                
256                for (Link link : this.net.getLinks().values()) {
257                        if(link.getAllowedModes().contains(TransportMode.car)){
258                                stopsAdded += addStopOnLink(link);
259                        }
260                }
261                
262                log.info("Added " + stopsAdded + " additional stops for paratransit services");
263        }
264        
265        private int addStopOnLink(Link link) {
266                if(link == null){
267                        return 0;
268                }
269                
270                if(!linkToNodeInServiceArea(link)){
271                        return 0;
272                }
273                
274                if (linkHasAlreadyAFormalPTStopFromTheGivenSchedule(link)) {
275                        return 0;
276                }
277                
278                if(!topoTypeAllowed(link)){
279                        return 0;
280                }
281                
282                if (link.getFreespeed() >= this.pConfigGroup.getSpeedLimitForStops()) {
283                        return 0;
284                }
285                
286                if (link.getCapacity() < this.pConfigGroup.getMinCapacityForStops()) {
287                        return 0;
288                }
289
290                if (this.linkId2StopFacilityMap.get(link.getId()) != null) {
291                        log.warn("Link " + link.getId() + " has already a stop. This should not happen. Check code.");
292                        return 0;
293                }
294                
295                Id<TransitStopFacility> stopId = Id.create(this.pConfigGroup.getPIdentifier() + link.getId(), TransitStopFacility.class);
296                TransitStopFacility stop = this.transitSchedule.getFactory().createTransitStopFacility(stopId, link.getToNode().getCoord(), false);
297                stop.setLinkId(link.getId());
298                this.transitSchedule.addStopFacility(stop);
299                return 1;               
300        }
301
302        private boolean topoTypeAllowed(Link link) {
303                if(this.topoTypesForStops == null){
304                        // flag not set or null in config
305                        return true;
306                }
307                Integer topoType = this.networkCalcTopoType.getTopoType(link.getToNode());
308                return this.topoTypesForStops.contains(topoType);
309        }       
310
311        private boolean linkToNodeInServiceArea(Link link) {
312                Point p = factory.createPoint(MGC.coord2Coordinate(link.getToNode().getCoord()));
313                if(this.include.contains(p)){
314                        if(exclude.contains(p)){
315                                return false;
316                        }
317                        return true;
318                }
319                return false;
320        }
321
322        private boolean linkHasAlreadyAFormalPTStopFromTheGivenSchedule(Link link) {
323                if (this.linkId2StopFacilityMap.containsKey(link.getId())) {
324                        // There is already a stop at this link, used by formal public transport - Use this one instead
325                        this.transitSchedule.addStopFacility(this.linkId2StopFacilityMap.get(link.getId()));
326                        return true;
327                } else {
328                        return false;
329                }
330        }
331
332        private TransitSchedule getTransitSchedule() {
333                return this.transitSchedule;
334        }
