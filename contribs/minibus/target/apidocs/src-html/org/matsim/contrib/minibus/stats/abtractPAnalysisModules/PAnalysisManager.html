020package org.matsim.contrib.minibus.stats.abtractPAnalysisModules;
021
022import java.io.BufferedWriter;
023import java.io.File;
024import java.io.IOException;
025import java.util.HashMap;
026import java.util.LinkedList;
027import java.util.List;
028
029import javax.inject.Inject;
030
031import org.apache.log4j.Logger;
032import org.matsim.api.core.v01.Id;
033import org.matsim.contrib.minibus.PConfigGroup;
034import org.matsim.contrib.minibus.PConstants;
035import org.matsim.core.controler.events.IterationEndsEvent;
036import org.matsim.core.controler.events.IterationStartsEvent;
037import org.matsim.core.controler.events.StartupEvent;
038import org.matsim.core.controler.listener.IterationEndsListener;
039import org.matsim.core.controler.listener.IterationStartsListener;
040import org.matsim.core.controler.listener.StartupListener;
041import org.matsim.core.utils.io.IOUtils;
042import org.matsim.pt.transitSchedule.api.TransitLine;
043import org.matsim.pt.transitSchedule.api.TransitSchedule;
044import org.matsim.vehicles.Vehicles;
045
046/**
047 * Plugs in all analysis.
048 * 
049 * @author aneumann
050 *
051 */
052public final class PAnalysisManager implements StartupListener, IterationStartsListener, IterationEndsListener{
053        private final static Logger log = Logger.getLogger(PAnalysisManager.class);
054        
055        private final String pIdentifier;
056        private final List<PAnalysisModule> pAnalyzesList = new LinkedList<>();
057        private final HashMap<String, BufferedWriter> pAnalyis2Writer = new HashMap<>();
058        private boolean firstIteration = true;
059        @Inject private LineId2PtMode lineSetter;
060
061        public PAnalysisManager(PConfigGroup pConfig){
062                log.info("enabled");
063                this.pIdentifier = pConfig.getPIdentifier();
064        }
065        @Override
066        public void notifyStartup(StartupEvent event) {
067                // create all analyzes
068                this.pAnalyzesList.add(new CountTripsPerMode());
069                this.pAnalyzesList.add(new CountVehPerMode());
070        this.pAnalyzesList.add(new CountVehicleMeterPerMode(event.getServices().getScenario().getNetwork()));
071        this.pAnalyzesList.add(new AverageTripDistanceMeterPerMode(event.getServices().getScenario().getNetwork()));
072                this.pAnalyzesList.add(new AverageInVehicleTripTravelTimeSecondsPerMode());
073                this.pAnalyzesList.add(new AverageWaitingTimeSecondsPerMode());
074                this.pAnalyzesList.add(new AverageNumberOfStopsPerMode());
075                this.pAnalyzesList.add(new CountTransfersPerModeModeCombination());
076                this.pAnalyzesList.add(new CountTripsPerPtModeCombination());
077                this.pAnalyzesList.add(new AverageLoadPerDeparturePerMode());
078                this.pAnalyzesList.add(new CountDeparturesWithNoCapacityLeftPerMode());
079                this.pAnalyzesList.add(new CountDeparturesPerMode());
080
081        CountPassengerMeterPerMode countPassengerMeterPerMode = new CountPassengerMeterPerMode(event.getServices().getScenario().getNetwork());
082                this.pAnalyzesList.add(countPassengerMeterPerMode);
083        CountCapacityMeterPerMode countCapacityMeterPerMode = new CountCapacityMeterPerMode(event.getServices().getScenario().getNetwork());
084                this.pAnalyzesList.add(countCapacityMeterPerMode);
085                this.pAnalyzesList.add(new AverageLoadPerDistancePerMode(countPassengerMeterPerMode, countCapacityMeterPerMode));
086                
087                // register all analyzes
088                for (PAnalysisModule ana : this.pAnalyzesList) {
089                        event.getServices().getEvents().addHandler(ana);
090                }
091        }
092
093        @Override
094        public void notifyIterationStarts(IterationStartsEvent event) {
095                // update pt mode for each line in schedule
096                updateLineId2ptModeMap(event.getServices().getScenario().getTransitSchedule());
097                updateVehicleTypes(event.getServices().getScenario().getTransitVehicles());
098        }
099
100        @Override
101        public void notifyIterationEnds(IterationEndsEvent event) {
102                if (this.firstIteration) {
103                        // create the output folder for this module
104                        String outFilename = event.getServices().getControlerIO().getOutputPath() + PConstants.statsOutputFolder + PAnalysisManager.class.getSimpleName() + "/";
105                        new File(outFilename).mkdir();
106                        
107                        // create one output stream for each analysis
108                        for (PAnalysisModule ana : this.pAnalyzesList) {
109                                try {
110                                        String moduleOutFilename = outFilename + ana.getName() + ".txt";
111                                        BufferedWriter writer = IOUtils.getBufferedWriter(moduleOutFilename);
112                                        writer.write("# iteration" + ana.getHeader());
113                                        writer.newLine();
114                                        this.pAnalyis2Writer.put(ana.getName(), writer);
115                                } catch (IOException e) {
116                                        // TODO Auto-generated catch block
117                                        e.printStackTrace();
118                                }
119                        }
120                        this.firstIteration = false;
121                }
122                
123                // write results to corresponding files
124                for (PAnalysisModule ana : this.pAnalyzesList) {
125                        BufferedWriter writer = this.pAnalyis2Writer.get(ana.getName());
126                        try {
127                                writer.write(event.getIteration() + ana.getResult());
128                                writer.newLine();
129                                writer.flush();
130                        } catch (IOException e) {
131                                // TODO Auto-generated catch block
132                                e.printStackTrace();
133                        }
134                }
135        }
136
137        private void updateLineId2ptModeMap(TransitSchedule transitSchedule) {
138                this.lineSetter.setPtModesForEachLine(transitSchedule, this.pIdentifier);
139                HashMap<Id<TransitLine>, String> lineIds2ptModeMap = this.lineSetter.getLineId2ptModeMap();
140                
141                for (PAnalysisModule ana : this.pAnalyzesList) {
142                        ana.setLineId2ptModeMap(lineIds2ptModeMap);
143                }
144        }
145
146        private void updateVehicleTypes(Vehicles vehicles) {
147                for (PAnalysisModule ana : this.pAnalyzesList) {
148                        ana.updateVehicles(vehicles);
149                }               
150        }
151}
