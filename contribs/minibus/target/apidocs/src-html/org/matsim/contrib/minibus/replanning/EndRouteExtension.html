042/**
043 * Takes the route transformed into a lineString to calculate a buffer around it.
044 * Excludes the part of the buffer going parallel/along the actual route, so only the end caps remain.
045 * Chooses then randomly a new stop within the buffer and inserts it after the nearest existing stop.
046 * 
047 * @author aneumann
048 *
049 */
050public final class EndRouteExtension extends AbstractPStrategyModule {
051        
052        private final static Logger log = Logger.getLogger(EndRouteExtension.class);
053        public static final String STRATEGY_NAME = "EndRouteExtension";
054        private final double bufferSize;
055        private final double ratio;
056        
057        public EndRouteExtension(ArrayList<String> parameter) {
058                super();
059                if(parameter.size() != 2){
060                        log.error("Parameter 1: Buffer size in meter");
061                        log.error("Parameter 2: Ratio bufferSize to route's beeline length. If set to something very small, e.g. 0.01, the calculated buffer size may be smaller than the one specified in parameter 1. Parameter 1 will then be taken as minimal buffer size.");
062                }
063                this.bufferSize = Double.parseDouble(parameter.get(0));
064                this.ratio = Double.parseDouble(parameter.get(1));
065        }
066
067        @Override
068        public PPlan run(Operator operator) {
069
070                PPlan oldPlan = operator.getBestPlan();
071                ArrayList<TransitStopFacility> currentStopsToBeServed = oldPlan.getStopsToBeServed();
072                
073                TransitStopFacility baseStop = currentStopsToBeServed.get(0);
074                /* 
075                 * Paratransit TransitRoutes are always circular instead of having one TransitRoute each for each direction between two 
076                 * terminus stops. So we only know one terminus stop, the base stop, and try to find the other terminus, the other end 
077                 * of the route, by scanning the stops on the TransitRoute for the stop which has the largest distance from the base stop.
078                 */
079                TransitStopFacility remoteStop = this.findStopWithLargestDistance(currentStopsToBeServed);
080                
081                if (baseStop.equals(remoteStop)) {
082                        /*
083                         * TODO:
084                         * findStopWithLargestDistance() should never return a remote stop which equals the base stop unless the input 
085                         * stop sequence of the oldPlan has only stops at exactly the same coordinate (which should be prohibited somewhere
086                         * else). If baseStop and remoteStop are equal createGeometryFromStops() fails because it asks for a LineString of 
087                         * just one point (the base stop = remote stop).
088                         */
089                        log.debug(
090                                        "EndRouteExtension replanning is skipped for TransitLine "
091                                                        + operator.getCurrentTransitLine().getId()
092                                                        + ", because base stop and remote stop returned by findStopWithLargestDistance() are equal. This"
093                                                        + " should not happen. Base stop: " + baseStop.getId() + ". Remote stop: " + remoteStop.getId());
094                        return null;
095                }
096
097                double bufferSizeBasedOnRatio = CoordUtils.calcEuclideanDistance(baseStop.getCoord(), remoteStop.getCoord()) * this.ratio;
098                
099                List<Geometry> lineStrings = this.createGeometryFromStops(currentStopsToBeServed, remoteStop);
100                Geometry bufferWithoutEndCaps = this.createBuffer(lineStrings, Math.max(this.bufferSize, bufferSizeBasedOnRatio), true);
101                Geometry bufferWithEndCaps = this.createBuffer(lineStrings, Math.max(this.bufferSize, bufferSizeBasedOnRatio), false);
102                Geometry buffer = bufferWithEndCaps.difference(bufferWithoutEndCaps);
103                
104                Set<Id<TransitStopFacility>> stopsUsed = this.getStopsUsed(oldPlan.getLine().getRoutes().values());
105                TransitStopFacility newStop = this.drawRandomStop(buffer, operator.getRouteProvider(), stopsUsed);
106                
107                if (newStop == null) {
108                        return null;
109                }
110                
111                ArrayList<TransitStopFacility> newStopsToBeServed = this.addStopToExistingStops(baseStop, remoteStop, currentStopsToBeServed, newStop);
112                
113                // create new plan
114                PPlan newPlan = new PPlan(operator.getNewPlanId(), this.getStrategyName(), oldPlan.getId());
115                newPlan.setNVehicles(1);
116                newPlan.setStartTime(oldPlan.getStartTime());
117                newPlan.setEndTime(oldPlan.getEndTime());
118                newPlan.setStopsToBeServed(newStopsToBeServed);
119                
120                newPlan.setLine(operator.getRouteProvider().createTransitLineFromOperatorPlan(operator.getId(), newPlan));
121                
122                return newPlan;
123        }
124
125
126        private ArrayList<TransitStopFacility> addStopToExistingStops(TransitStopFacility baseStop, TransitStopFacility remoteStop, ArrayList<TransitStopFacility> currentStopsToBeServed, TransitStopFacility newStop) {
127                ArrayList<TransitStopFacility> newStopsToBeServed = new ArrayList<>(currentStopsToBeServed);
128                
129                // decide which stop is closer
130                if (CoordUtils.calcEuclideanDistance(baseStop.getCoord(), newStop.getCoord()) < CoordUtils.calcEuclideanDistance(remoteStop.getCoord(), newStop.getCoord())) {
131                        // baseStop is closer - insert before baseStop
132                        newStopsToBeServed.add(0, newStop);
133                } else {
134                        // remote stop is closer or both have the same distance - add after remote stop
135                        newStopsToBeServed.add(newStopsToBeServed.indexOf(remoteStop) + 1, newStop);
136                }
137                
138                return newStopsToBeServed;
139        }
140
141        private TransitStopFacility findStopWithLargestDistance(ArrayList<TransitStopFacility> stops) {
142                Coord startCoord = stops.get(0).getCoord();
143                double largestDistance = 0;
144                TransitStopFacility stopWithLargestDistance = stops.get(0);
145                for (TransitStopFacility transitStopFacility : stops) {
146                        double currentDistance = CoordUtils.calcEuclideanDistance(startCoord, transitStopFacility.getCoord());
147                        if (currentDistance > largestDistance) {
148                                largestDistance = currentDistance;
149                                stopWithLargestDistance = transitStopFacility;
150                        }
151                }
152                return stopWithLargestDistance;
153        }
154
155        private Set<Id<TransitStopFacility>> getStopsUsed(Collection<TransitRoute> routes) {
156                Set<Id<TransitStopFacility>> stopsUsed = new TreeSet<>();
157                for (TransitRoute route : routes) {
158                        for (TransitRouteStop stop : route.getStops()) {
159                                stopsUsed.add(stop.getStopFacility().getId());
160                        }
161                }
162                return stopsUsed;
163        }
164
165        private TransitStopFacility drawRandomStop(Geometry buffer, PRouteProvider pRouteProvider, Set<Id<TransitStopFacility>> stopsUsed) {
166                List<TransitStopFacility> choiceSet = new LinkedList<>();
167                
168                // find choice-set
169                for (TransitStopFacility stop : pRouteProvider.getAllPStops()) {
170                        if (!stopsUsed.contains(stop.getId())) {
171                                if (buffer.contains(MGC.coord2Point(stop.getCoord()))) {
172                                        choiceSet.add(stop);
173                                }
174                        }
175                }
176                
177                return pRouteProvider.drawRandomStopFromList(choiceSet);
178        }
179
180
181        private Geometry createBuffer(List<Geometry> lineStrings, double bufferSize, boolean excludeTermini) {
182                BufferParameters bufferParameters = new BufferParameters();
183                
184                if (excludeTermini) {
185                        bufferParameters.setEndCapStyle(BufferParameters.CAP_FLAT);
186                } else {
187                        bufferParameters.setEndCapStyle(BufferParameters.CAP_ROUND);
188                }
189                
190                Geometry union = null;
191                
192                for (Geometry lineString : lineStrings) {
193                        Geometry buffer = BufferOp.bufferOp(lineString, bufferSize, bufferParameters);
194                        if (union == null) {
195                                union = buffer;
196                        } else {
197                                union = union.union(buffer);
198                        }
199                }
200                
201                return union;
202        }
203
204        private List<Geometry> createGeometryFromStops(ArrayList<TransitStopFacility> stops, TransitStopFacility remoteStop) {
205                List<Geometry> geometries = new LinkedList<>();
206                
207                ArrayList<Coordinate> coords = new ArrayList<>();
208                for (TransitStopFacility stop : stops) {
209                        if (stop.equals(remoteStop)) {
210                                // terminate current line string
211                                coords.add(new Coordinate(stop.getCoord().getX(), stop.getCoord().getY(), 0.0));
212                                Coordinate[] coordinates = coords.toArray(new Coordinate[coords.size()]);
213                                Geometry lineString = new GeometryFactory().createLineString(coordinates);
214                                geometries.add(lineString);
215                                // create new line string
216                                coords = new ArrayList<>();
217                                coords.add(new Coordinate(stop.getCoord().getX(), stop.getCoord().getY(), 0.0));
218                        } else {
219                                coords.add(new Coordinate(stop.getCoord().getX(), stop.getCoord().getY(), 0.0));
220                        }
221                }
222                // add first stop to close the circle
223                coords.add(new Coordinate(stops.get(0).getCoord().getX(), stops.get(0).getCoord().getY(), 0.0));
224                
225                Coordinate[] coordinates = coords.toArray(new Coordinate[coords.size()]);
226                Geometry lineString = new GeometryFactory().createLineString(coordinates);
227                geometries.add(lineString);
228                return geometries;
229        }
230
231        public String getStrategyName() {
232                return EndRouteExtension.STRATEGY_NAME;
233        }
