083        public void initializeFromSchedule(final TransitSchedule transitSchedule, final double maxBeelineWalkConnectionDistance) {
084                
085                this.fillArrays(transitSchedule, maxBeelineWalkConnectionDistance);
086                
087                log.info("transit router network statistics:");
088                log.info(" # stops:           " + this.transitStops.length);
089                log.info(" # routes:          " + this.routes.length);
090                log.info(" # transfer links:  " + this.transfers.length);
091        }

093        private void fillArrays(TransitSchedule transitSchedule, double maxBeelineWalkConnectionDistance) {
094                
095                // arrays that need to be converted in the end
096                ArrayList<Double> departureTimesList = new ArrayList<Double>();
097                ArrayList<Double> arrivalTimesList = new ArrayList<Double>();
098                ArrayList<RouteEntry> routesList = new ArrayList<RouteEntry>();
099                ArrayList<RouteStopEntry> routeStopList = new ArrayList<RouteStopEntry>();
100                ArrayList<TransferEntry> transfersList = new ArrayList<TransferEntry>();
101                ArrayList<TransitStopEntry> stopsList = new ArrayList<TransitStopEntry>();
102                
103                // temporary storage
104                Map<WrappedTransitRouteStop, Id<TransitRoute>> routeStop2routeId = new HashMap<WrappedTransitRouteStop, Id<TransitRoute>>();
105                Map<TransitStopFacility, Set<Id<TransitRoute>>> transitStopFacilities2RouteIdsServed = new HashMap<TransitStopFacility, Set<Id<TransitRoute>>>();
106                ArrayList<TransitStopFacility> transitStopsInRightOrder = new ArrayList<TransitStopFacility>();
107                HashMap<String, WrappedTransitRouteStop> hash2routeStop = new HashMap<String, WrappedTransitRouteStop>();
108                
109                for (TransitLine line : transitSchedule.getTransitLines().values()) {
110                        for (TransitRoute route : line.getRoutes().values()) {
111                                int position = 0;
112                                for (TransitRouteStop routeStop : route.getStops()) {
113                                        
114                                        String routeStopHash = this.getHash(line, route, routeStop, position);
115                                        WrappedTransitRouteStop wrappedRouteStop = new WrappedTransitRouteStop(routeStop);
116                                        hash2routeStop.put(routeStopHash, wrappedRouteStop);
117                                        
118                                        position++;
119                                        
120                                        routeStop2routeId.put(wrappedRouteStop, route.getId());
121                                        
122                                        if (transitStopFacilities2RouteIdsServed.get(routeStop.getStopFacility()) == null) {
123                                                transitStopFacilities2RouteIdsServed.put(routeStop.getStopFacility(), new TreeSet<Id<TransitRoute>>());
124                                        }
125                                        transitStopFacilities2RouteIdsServed.get(routeStop.getStopFacility()).add(route.getId());
126                                }
127                        }
128                }
129                
130                this.quadTree = this.createTransitStopFacilityQuadTree(transitStopFacilities2RouteIdsServed.keySet());
131                Map<TransitStopFacility, HashMap<WrappedTransitRouteStop, TransferEntryPointer>> stop2TransitRouteStop2Transfers = this.createTransfers(transitStopFacilities2RouteIdsServed, routeStop2routeId, maxBeelineWalkConnectionDistance);
132                stop2TransitRouteStop2Transfers = filterTransfers(transitSchedule, stop2TransitRouteStop2Transfers, routeStop2routeId);
133
134                for (TransitStopFacility transitStopFacility : stop2TransitRouteStop2Transfers.keySet()) {
135                        this.transitStopFacility2Index.put(transitStopFacility, transitStopsInRightOrder.size());
136                        transitStopsInRightOrder.add(transitStopFacility);
137                }
138                
139                Map<WrappedTransitRouteStop, Integer> routeStopEntry2Index = new HashMap<>();
140                
141                for (TransitLine line : transitSchedule.getTransitLines().values()) {
142                        for (TransitRoute route : line.getRoutes().values()) {
143                                
144                                // fill departures
145                                final int indexOfFirstDeparture = departureTimesList.size();
146                                final int numberOfDepartures = route.getDepartures().size();
147                                double[] departureTimes = new double[numberOfDepartures];
148
149                                int i = 0;
150                                for (Departure dep : route.getDepartures().values()) {
151                                        departureTimes[i++] = dep.getDepartureTime();
152                                }
153                                Arrays.sort(departureTimes);
154
155                                for (TransitRouteStop stop : route.getStops()) {
156                                        for (int j = 0; j < departureTimes.length; j++) {
157                                                if (stop.getArrivalOffset() == Double.NEGATIVE_INFINITY) {
158                                                        // There should always be a valid number set OR the field should not be present - it's optional
159                                                        // Take the departure offset as fallback
160                                                        arrivalTimesList.add(departureTimes[j] + stop.getDepartureOffset());
161                                                } else {
162                                                arrivalTimesList.add(departureTimes[j] + stop.getArrivalOffset());
163                                                }
164                                                
165                                                departureTimesList.add(departureTimes[j] + stop.getDepartureOffset());
166                                        }
167                                }
168                                // So the arrivalTimesList contains | stop1ar1 stop1ar2 ... stop1arLast stop2ar1 stop2ar2 stop2arLast ... |,
169                                // and this in addition flattened for all routes.
170                                // It seems that it can reconstruct everything from "indexOfFirstDeparture", "numberOfDepartures", "numberOfRouteSteps".
171                                
172                                final int indexOfFirstStop = routeStopList.size();
173                                final int numberOfRouteStops = route.getStops().size();
174                                
175                                final int indexOfRoute = routesList.size();
176                                
177                                // fill route
178                                RouteEntry routeEntry = new RouteEntry(line.getId(), route.getId(), indexOfFirstDeparture, numberOfDepartures, indexOfFirstStop, numberOfRouteStops);
179                                routesList.add(routeEntry);
180                                
181                                
182                                // fill route stops
183                                int placeOfCurrentStop = 0;
184                                for (TransitRouteStop routeStop : route.getStops()) {
185                                        String routeStopHash = getHash(line, route, routeStop, placeOfCurrentStop);
186                                        WrappedTransitRouteStop wrappedRouteStop = hash2routeStop.get(routeStopHash);
187                                        
188                                        placeOfCurrentStop++;
189
190                                        final int indexOfRouteStop = routeStopList.size();
191                                        final int numberOfRemainingStopsInThisRoute = route.getStops().size() - placeOfCurrentStop;
192                                        
193                                        routeStopList.add(new RouteStopEntry(indexOfRoute, this.transitStopFacility2Index.get(routeStop.getStopFacility()).intValue(), indexOfRouteStop, numberOfRemainingStopsInThisRoute));
194                                        routeStopEntry2Index.put(wrappedRouteStop, indexOfRouteStop);
195                                }
196                                
197
198                        }
199                }
200                
201                // complete stops and their transfers
202                for (TransitStopFacility transitStopFacility : transitStopsInRightOrder) {
203                        int indexOfFirstTransfer = transfersList.size();
204                        int numberOfTransfers = 0;
205                        
206                        for (TransferEntryPointer transferEntryPointer : stop2TransitRouteStop2Transfers.get(transitStopFacility).values()) {
207                                int indexOfRouteStop = routeStopEntry2Index.get(transferEntryPointer.transferDestination).intValue();
208                                transfersList.add(new TransferEntry(indexOfRouteStop, transferEntryPointer.transferTime));
209                                numberOfTransfers++;
210                        }
211                        
212                        stopsList.add(new TransitStopEntry(transitStopFacility, numberOfTransfers, indexOfFirstTransfer));
213                }
214                
215                // create the final arrays
216                this.arrivalTimes = getPrimitiveArrayFromList(arrivalTimesList);
217                
218                this.departureTimes = getPrimitiveArrayFromList(departureTimesList);
219
220                this.routes = new RouteEntry[routesList.size()];
221                routesList.toArray(this.routes);
222                
223                this.routeStops = new RouteStopEntry[routeStopList.size()];
224                routeStopList.toArray(this.routeStops);
225                
226                this.transfers = new TransferEntry[transfersList.size()];
227                transfersList.toArray(this.transfers);
228
229                this.transitStops = new TransitStopEntry[stopsList.size()];
230                stopsList.toArray(this.transitStops);
231        }

233        private Map<TransitStopFacility, HashMap<WrappedTransitRouteStop, TransferEntryPointer>> createTransfers(Map<TransitStopFacility, Set<Id<TransitRoute>>> transitStopFacilities2RouteIdsServed, Map<WrappedTransitRouteStop, Id<TransitRoute>> routeStop2routeId, double maxBeelineWalkConnectionDistance) {
234                Map<TransitStopFacility, HashMap<WrappedTransitRouteStop, TransferEntryPointer>> stop2TransitRouteStop2Transfers = new HashMap<>();
235                
236                QuadTree<WrappedTransitRouteStop> transitRouteStopQuadTree = this.createTransitRouteStopQuadTree(routeStop2routeId.keySet());
237                
238                // create transfers for all transit route stops if they're located less than beelineWalkConnectionDistance from each other
239                for (TransitStopFacility sourceStop : transitStopFacilities2RouteIdsServed.keySet()) {
240                        // (just goes through all stops)
241
242                        HashMap<WrappedTransitRouteStop, TransferEntryPointer> transfersFromThis = new HashMap<WrappedTransitRouteStop, TransferEntryPointer>();
243                        
244                        for (WrappedTransitRouteStop destinationStop : transitRouteStopQuadTree.getDisk(sourceStop.getCoord().getX(), sourceStop.getCoord().getY(), maxBeelineWalkConnectionDistance)) {
245                                // (goes through all stops within walkConnectionDistance)
246
247                                double transferTime = this.raptorDisutility.getTransferTime(sourceStop.getCoord(), destinationStop.transitRouteStop.getStopFacility().getCoord());
248                                transfersFromThis.put(destinationStop, new TransferEntryPointer(-1, transferTime, destinationStop, routeStop2routeId.get(destinationStop)));
249                                // (memorize: (1) which stop the walk goes to; (2) route id associated with that stop (presumably one stop per route even at same location)
250                        }
251                        
252                        stop2TransitRouteStop2Transfers.put(sourceStop, transfersFromThis);
253                }
254                
255                int transfersAdded = 0;
256                for (HashMap<WrappedTransitRouteStop, TransferEntryPointer> routeStop2Transfers : stop2TransitRouteStop2Transfers.values()) {
257                        transfersAdded += routeStop2Transfers.size();
258                }
259                log.info("Added " + transfersAdded + " transfers (from each transit stop facility to each other transit route stop within " + maxBeelineWalkConnectionDistance + "m beeline distance.");
260                
261                return stop2TransitRouteStop2Transfers;
262        }

264        private Map<TransitStopFacility, HashMap<WrappedTransitRouteStop, TransferEntryPointer>> filterTransfers(TransitSchedule transitSchedule, Map<TransitStopFacility, HashMap<WrappedTransitRouteStop, TransferEntryPointer>> stop2TransitRouteStop2Transfers, Map<WrappedTransitRouteStop, Id<TransitRoute>> routeStop2routeId) {
265                Map<TransitStopFacility, Set<Id<TransitRoute>>> transitStopFacilities2RouteIdsThatCanBeTransferedTo = new HashMap<TransitStopFacility, Set<Id<TransitRoute>>>();
266                
267                for (TransitStopFacility transitStopFacility : stop2TransitRouteStop2Transfers.keySet()) {
268                        for (WrappedTransitRouteStop routeStopEntry : stop2TransitRouteStop2Transfers.get(transitStopFacility).keySet()) {
269                                if (transitStopFacilities2RouteIdsThatCanBeTransferedTo.get(transitStopFacility) == null) {
270                                        transitStopFacilities2RouteIdsThatCanBeTransferedTo.put(transitStopFacility, new TreeSet<Id<TransitRoute>>());
271                                }
272                                transitStopFacilities2RouteIdsThatCanBeTransferedTo.get(transitStopFacility).add(routeStop2routeId.get(routeStopEntry));
273                        }
274                }
275                
276                for (TransitLine line : transitSchedule.getTransitLines().values()) {
277                        for (TransitRoute route : line.getRoutes().values()) {
278                                Set<Id<TransitRoute>> secondLastRouteIdsServed = null;
279                                Set<Id<TransitRoute>> lastRouteIdsServed = null;
280                                Set<Id<TransitRoute>> currentRouteIdsServed = null;
281                                
282                                TransitStopFacility lastTransitStopFacility = null;
283                                
284                                for (TransitRouteStop stop : route.getStops()) {
285                                        currentRouteIdsServed = transitStopFacilities2RouteIdsThatCanBeTransferedTo.get(stop.getStopFacility());
286        
287                                        if (secondLastRouteIdsServed != null) {
288                                                // we now have three stops that can be compared
289                                                // remove all transfers from the last (the middle stop) if all three stops serve at least the same routes as well
290                                                
291                                                boolean atLeastOneStopDoesNotServeAllRoutes = false;
292                                                for (Id<TransitRoute> routeId : lastRouteIdsServed) {
293                                                        // check for all routes served by the middle stop if those are also served by its predecessor and the following stop
294                                                        if (!secondLastRouteIdsServed.contains(routeId) || !currentRouteIdsServed.contains(routeId)) {
295                                                                // one of those two stops does not serve the route
296                                                                atLeastOneStopDoesNotServeAllRoutes = true;
297                                                                break;
298                                                        }
299                                                }
300                                                if (!atLeastOneStopDoesNotServeAllRoutes) {
301                                                        // ok, all stops serve the routes
302                                                        // remove all transfers for this stop
303                                                        stop2TransitRouteStop2Transfers.put(lastTransitStopFacility, new HashMap<WrappedTransitRouteStop, TransferEntryPointer>());
304                                                }
305                                        }
306                                        secondLastRouteIdsServed = lastRouteIdsServed;
307                                        lastRouteIdsServed = currentRouteIdsServed;
308                                        lastTransitStopFacility = stop.getStopFacility();
309                                }
310                        }
311                }
312                
313                // count again
314                int transfersAdded = 0;
315                for (HashMap<WrappedTransitRouteStop, TransferEntryPointer> routeStop2Transfers : stop2TransitRouteStop2Transfers.values()) {
316                        transfersAdded += routeStop2Transfers.size();
317                }
318                log.info(transfersAdded + " transfers remain after filtering.");
319        
320                return stop2TransitRouteStop2Transfers;
321        }

323        private QuadTree<TransitStopFacility> createTransitStopFacilityQuadTree(Set<TransitStopFacility> stops) {
324                double minX = Double.POSITIVE_INFINITY;
325                double minY = Double.POSITIVE_INFINITY;
326                double maxX = Double.NEGATIVE_INFINITY;
327                double maxY = Double.NEGATIVE_INFINITY;
328        
329                for (TransitStopFacility stop : stops) {
330                        Coord c = stop.getCoord();
331                        if (c.getX() < minX) {
332                                minX = c.getX();
333                        }
334                        if (c.getY() < minY) {
335                                minY = c.getY();
336                        }
337                        if (c.getX() > maxX) {
338                                maxX = c.getX();
339                        }
340                        if (c.getY() > maxY) {
341                                maxY = c.getY();
342                        }
343                }
344        
345                QuadTree<TransitStopFacility> quadTree = new QuadTree<TransitStopFacility>(minX, minY, maxX, maxY);
346                for (TransitStopFacility stop : stops) {
347                        Coord c = stop.getCoord();
348                        quadTree.put(c.getX(), c.getY(), stop);
349                }
350                
351                return quadTree;
352        }

354        private QuadTree<WrappedTransitRouteStop> createTransitRouteStopQuadTree(Set<WrappedTransitRouteStop> routeStops) {
355                double minX = Double.POSITIVE_INFINITY;
356                double minY = Double.POSITIVE_INFINITY;
357                double maxX = Double.NEGATIVE_INFINITY;
358                double maxY = Double.NEGATIVE_INFINITY;
359        
360                for (WrappedTransitRouteStop stop : routeStops) {
361                        Coord c = stop.transitRouteStop.getStopFacility().getCoord();
362                        if (c.getX() < minX) {
363                                minX = c.getX();
364                        }
365                        if (c.getY() < minY) {
366                                minY = c.getY();
367                        }
368                        if (c.getX() > maxX) {
369                                maxX = c.getX();
370                        }
371                        if (c.getY() > maxY) {
372                                maxY = c.getY();
373                        }
374                }
375        
376                QuadTree<WrappedTransitRouteStop> quadTree = new QuadTree<WrappedTransitRouteStop>(minX, minY, maxX, maxY);
377                for (WrappedTransitRouteStop stop : routeStops) {
378                        Coord c = stop.transitRouteStop.getStopFacility().getCoord();
379                        quadTree.put(c.getX(), c.getY(), stop);
380                }
381                
382                return quadTree;
383        }

385        private static double[] getPrimitiveArrayFromList(List<Double> list) {
386            double[] array = new double[list.size()];
387            int i = 0;
388            for (Double entry : list) {
389                array[i++] = entry.doubleValue();
390            }
391            return array;
392        }

394        private static String getHash(TransitLine transitLine, TransitRoute transitRoute, TransitRouteStop transitRouteStop, int position){
395                return transitLine.getId().toString() + "-" + transitRoute.getId().toString() + "-" + transitRouteStop.getStopFacility().getId().toString() + "-" + position;
