020package org.matsim.contrib.minibus.performance.raptor;
021
022import java.util.Map;
023
024import org.matsim.pt.transitSchedule.api.TransitStopFacility;
025
026
027/**
028 * 
029 * @author aneumann
030 *
031 */
032public class RaptorSearchData {
033        // yy It is not clear to me what a "block" is in the following description.  Possibly, if we have routes a and b, and stops a1, a2, ..., b1, b2, ...,
034        // then they are just consecutively put into an array, so that the next entry is either the next bus stop downstream, or the
035        // starting point of the next route.  ??  kai, jun'16
036        
037        // Each block contains data for one route. Each route block holds sub-blocks for each departure. Each sub-block holds arrival and departure times sorted in increasing order.  
038        final double[] arrivalTimes;
039        final double[] departureTimes;
040
041        // All routes each linking to its departures (stopTimes) and served stops (routeStops).
042        final RouteEntry[] routes;
043
044        // Each block holds the stops served by a route in the sequence of serving it.
045        final RouteStopEntry[] routeStops;
046
047        // Each block holds the potential transfers for each stop.
048        final TransferEntry[] transfers;
049
050        // All stops serving at least one route with links to their transfers (transfers) and routes served (stopRoutes).
051        final TransitStopEntry[] stops;
052
053        // Simple list of transit stop to index - just for speedup 
054        final Map<TransitStopFacility, Integer> transitStopFacility2Index;
055
056        public RaptorSearchData(double[] arrivalTimes, double[] departureTimes, RouteEntry[] routes, RouteStopEntry[] routeStops, TransferEntry[] transfers, TransitStopEntry[] stops, Map<TransitStopFacility, Integer> transitStopFacility2Index) {
057                this.arrivalTimes = arrivalTimes;
058                this.departureTimes = departureTimes;
059                this.routes = routes;
060                this.routeStops = routeStops;
061                this.transfers = transfers;
062                this.stops = stops;
063                this.transitStopFacility2Index = transitStopFacility2Index;
064        }
065
066}
