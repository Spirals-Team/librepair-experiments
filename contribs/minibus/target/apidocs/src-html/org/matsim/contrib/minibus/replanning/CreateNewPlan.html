020package org.matsim.contrib.minibus.replanning;
021
022import org.apache.log4j.Logger;
023import org.matsim.contrib.minibus.PConstants;
024import org.matsim.contrib.minibus.operator.Operator;
025import org.matsim.contrib.minibus.operator.PPlan;
026import org.matsim.contrib.minibus.operator.TimeProvider;
027import org.matsim.core.utils.geometry.CoordUtils;
028import org.matsim.pt.transitSchedule.api.TransitStopFacility;
029
030import java.util.ArrayList;
031
032/**
033 * 
034 * Creates a new plan from scratch. New stops have a minimal distance of twice the grid size. 
035 * 
036 * @author aneumann
037 *
038 */
039public final class CreateNewPlan extends AbstractPStrategyModule {
040        
041        private final static Logger log = Logger.getLogger(CreateNewPlan.class);
042        public static final String STRATEGY_NAME = "CreateNewPlan";
043
044    private final double timeSlotSize;
045        private final double minInitialStopDistance;
046        private TimeProvider timeProvider;
047
048        public CreateNewPlan(ArrayList<String> parameter) {
049                super();
050                if(parameter.size() != 2){
051                        log.error("Wrong number of parameters. Will ignore: " + parameter);
052                        log.error("Parameter 1: Time slot size for new start and end times");
053                        log.error("Parameter 2: Min distance of the two initial stops");
054                }
055                
056                this.timeSlotSize = Double.parseDouble(parameter.get(0));
057                this.minInitialStopDistance = Double.parseDouble(parameter.get(1));
058        }
059        
060        public void setTimeProvider(TimeProvider timeProvider){
061                this.timeProvider = timeProvider;
062        }
063        
064        @Override
065        public PPlan run(Operator operator) {
066                PPlan newPlan;          
067                
068                int triesPerformed = 0;
069
070        int maxNumberOfTries = 100;
071        do {
072                        triesPerformed++;
073                        
074                        // get a valid start time
075                        double startTime = this.timeProvider.getRandomTimeInInterval(0 * 3600.0, 24 * 3600.0 - operator.getMinOperationTime());
076                        double endTime = this.timeProvider.getRandomTimeInInterval(startTime + operator.getMinOperationTime(), 24 * 3600.0);
077                        
078                        if (startTime == endTime) {
079                                endTime += this.timeSlotSize;
080                        }
081                        
082                        while (startTime + operator.getMinOperationTime() > endTime) {
083                                endTime += this.timeSlotSize;
084                        }
085                        
086                        if (startTime + operator.getMinOperationTime() > endTime) {
087                                log.warn("Already increased the time of operation by one time slot in order to meet the minimum time of operation criteria of " + operator.getMinOperationTime());
088                                log.warn("Start time is: " + startTime);
089                                log.warn("End time is: " + endTime);
090                                log.warn("Will continue anyway...");
091                        }
092                        
093                        TransitStopFacility stop1 = operator.getRouteProvider().getRandomTransitStop(operator.getCurrentIteration());
094                        TransitStopFacility stop2 = operator.getRouteProvider().getRandomTransitStop(operator.getCurrentIteration());
095
096                        while (CoordUtils.calcEuclideanDistance(stop1.getCoord(), stop2.getCoord()) < this.minInitialStopDistance) {
097                                stop2 = operator.getRouteProvider().getRandomTransitStop(operator.getCurrentIteration());
098                        }
099                        
100                        ArrayList<TransitStopFacility> stopsToBeServed = new ArrayList<>();
101                        stopsToBeServed.add(stop1);
102                        stopsToBeServed.add(stop2);
103                        
104                        newPlan = new PPlan(operator.getNewPlanId(), this.getStrategyName(), PConstants.founderPlanId);
105                        newPlan.setStopsToBeServed(stopsToBeServed);
106                        newPlan.setStartTime(startTime);
107                        newPlan.setEndTime(endTime);
108                        newPlan.setNVehicles(1);
109                        newPlan.setStopsToBeServed(stopsToBeServed);
110                        
111                        newPlan.setLine(operator.getRouteProvider().createTransitLineFromOperatorPlan(operator.getId(), newPlan));
112
113                } while (operator.getFranchise().planRejected(newPlan) && triesPerformed < maxNumberOfTries);
114
115                if(!(triesPerformed < maxNumberOfTries)){
116                        log.warn("Exceeded the maximum number of tries (" + maxNumberOfTries + ") to find a new plan for operator " + operator.getId() + ". Returning null");
117                        return null;
118                }
119                
120                return newPlan;
121        }
122
123        @Override
124        public String getStrategyName() {
125                return CreateNewPlan.STRATEGY_NAME;
126        }
127
128}
