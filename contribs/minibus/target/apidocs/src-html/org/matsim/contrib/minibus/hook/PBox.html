066        private final static Logger log = Logger.getLogger(PBox.class);

077        private final PScorePlansHandler scorePlansHandler;
078        private final StageContainerCreator stageCollectorHandler;
079        private final OperatorCostCollectorHandler operatorCostCollectorHandler;
080        private final PStrategyManager strategyManager = new PStrategyManager();

082        private final TicketMachineI ticketMachine;

084        @Inject(optional=true) private SubsidyI subsidy;
085        // yy my intuition would be to pass an empty subsidy rather than making it optional.

087        /**
088         * Constructor that allows to set the ticketMachine.  Deliberately in constructor and not as setter to keep the variable final.  Might be
089         * replaced by a builder and/or guice at some later point in time.  But stay with "direct" injection for the time being.  kai, jan'17
090         */
091        @Inject PBox(PConfigGroup pConfig, TicketMachineI ticketMachine) {
092                this.pConfig = pConfig;
093                this.ticketMachine = ticketMachine ;
094                this.scorePlansHandler = new PScorePlansHandler(this.ticketMachine);
095                this.stageCollectorHandler = new StageContainerCreator(this.pConfig.getPIdentifier());
096                this.operatorCostCollectorHandler = new OperatorCostCollectorHandler(this.pConfig.getPIdentifier(), this.pConfig.getCostPerVehicleAndDay(), this.pConfig.getCostPerKilometer() / 1000.0, this.pConfig.getCostPerHour() / 3600.0);
097                this.franchise = new PFranchise(this.pConfig.getUseFranchise(), pConfig.getGridSize()); 
098        }
099
100        void notifyStartup(StartupEvent event) {
101                // This is the first iteration
102
103                TimeProvider timeProvider = new TimeProvider(this.pConfig, event.getServices().getControlerIO().getOutputPath());
104                event.getServices().getEvents().addHandler(timeProvider);
105
106                // initialize strategy manager
107                this.strategyManager.init(this.pConfig, this.stageCollectorHandler, this.ticketMachine, timeProvider);
108
109                // init fare collector
110                this.stageCollectorHandler.init(event.getServices().getScenario().getNetwork());
111                event.getServices().getEvents().addHandler(this.stageCollectorHandler);
112                event.getServices().addControlerListener(this.stageCollectorHandler);
113                this.stageCollectorHandler.addStageContainerHandler(this.scorePlansHandler);
114
115                // init operator cost collector
116                this.operatorCostCollectorHandler.init(event.getServices().getScenario().getNetwork());
117                event.getServices().getEvents().addHandler(this.operatorCostCollectorHandler);
118                event.getServices().addControlerListener(this.operatorCostCollectorHandler);
119                this.operatorCostCollectorHandler.addOperatorCostContainerHandler(this.scorePlansHandler);
120
121                // init fare2moneyEvent
122                StageContainer2AgentMoneyEvent fare2AgentMoney = new StageContainer2AgentMoneyEvent(event.getServices(), this.ticketMachine);
123                this.stageCollectorHandler.addStageContainerHandler(fare2AgentMoney);
124
125                // init possible paratransit stops
126                this.pStopsOnly = PStopsFactory.createPStops(event.getServices().getScenario().getNetwork(), this.pConfig, event.getServices().getScenario().getTransitSchedule());
127
128                this.operators = new LinkedList<>();
129                this.operatorInitializer = new OperatorInitializer(this.pConfig, this.franchise, this.pStopsOnly, event.getServices(), timeProvider);
130
131                // init additional operators from a given transit schedule file
132                LinkedList<Operator> operatorsFromSchedule = this.operatorInitializer.createOperatorsFromSchedule(event.getServices().getScenario().getTransitSchedule());
133                this.operators.addAll(operatorsFromSchedule);
134
135                // init initial set of operators - reduced by the number of preset operators
136                LinkedList<Operator> initialOperators = this.operatorInitializer.createAdditionalOperators(this.strategyManager, event.getServices().getConfig().controler().getFirstIteration(), (this.pConfig.getNumberOfOperators() - operatorsFromSchedule.size()));
137                this.operators.addAll(initialOperators);
138
139                // collect the transit schedules from all operators
140                this.pTransitSchedule = new TransitScheduleFactoryImpl().createTransitSchedule();
141                for (TransitStopFacility stop : this.pStopsOnly.getFacilities().values()) {
142                        this.pTransitSchedule.addStopFacility(stop);
143                }
144                for (Operator operator : this.operators) {
145                        this.pTransitSchedule.addTransitLine(operator.getCurrentTransitLine());
146                }
147
148                // Reset the franchise system - TODO necessary?
149                this.franchise.reset(this.operators);
150        }
151
152        void notifyIterationStarts(IterationStartsEvent event) {
153
154                this.strategyManager.updateStrategies(event.getIteration());
155
156                // Adapt number of operators
157                this.handleBankruptOperators(event.getIteration());
158
159                // Replan all operators
160                for (Operator operator : this.operators) {
161                        operator.replan(this.strategyManager, event.getIteration());
162                }
163
164                // Collect current lines offered
165                // why is the following done twice (see notifyScoring)?
166                this.pTransitSchedule = new TransitScheduleFactoryImpl().createTransitSchedule();
167                for (TransitStopFacility stop : this.pStopsOnly.getFacilities().values()) {
168                        this.pTransitSchedule.addStopFacility(stop);
169                }
170                for (Operator operator : this.operators) {
171                        this.pTransitSchedule.addTransitLine(operator.getCurrentTransitLine());
172                }
173
174                // Reset the franchise system
175                this.franchise.reset(this.operators);
176        }
177
178        void notifyScoring(ScoringEvent event) {
179
180                if (this.subsidy != null) {
181                        subsidy.computeSubsidy();
182                }
183
184                Map<Id<Vehicle>, PScoreContainer> driverId2ScoreMap = this.scorePlansHandler.getDriverId2ScoreMap();
185                for (Operator operator : this.operators) {
186                        operator.score(driverId2ScoreMap, subsidy);
187                }
188
189                // why is the following done twice (see notifyIterationstarts)?
190                this.pTransitSchedule = new TransitScheduleFactoryImpl().createTransitSchedule();
191                for (TransitStopFacility stop : this.pStopsOnly.getFacilities().values()) {
192                        this.pTransitSchedule.addStopFacility(stop);
193                }
194                for (Operator operator : this.operators) {
195                        this.pTransitSchedule.addTransitLine(operator.getCurrentTransitLine());
196                }
197
198                writeScheduleToFile(this.pTransitSchedule, event.getServices().getControlerIO().getIterationFilename(event.getIteration(), "transitScheduleScored.xml.gz"));
199        }

201        private void handleBankruptOperators(int iteration) {
202
203                LinkedList<Operator> operatorsToKeep = new LinkedList<>();
204                int operatorsProspecting = 0;
205                int operatorsInBusiness = 0;
206                int operatorsBankrupt = 0;
207
208                // Get operators with positive budget
209                for (Operator operator : this.operators) {
210                        if(operator.getOperatorState().equals(OperatorState.PROSPECTING)){
211                                operatorsToKeep.add(operator);
212                                operatorsProspecting++;
213                        }
214
215                        if(operator.getOperatorState().equals(OperatorState.INBUSINESS)){
216                                operatorsToKeep.add(operator);
217                                operatorsInBusiness++;
218                        }
219
220                        if(operator.getOperatorState().equals(OperatorState.BANKRUPT)){
221                                operatorsBankrupt++;
222                        }
223                }
224
225                // get the number of new operators
226                int numberOfNewOperators = operatorsBankrupt;
227
228                if(this.pConfig.getUseAdaptiveNumberOfOperators()){
229                        // adapt the number of operators by calculating the exact number necessary
230                        numberOfNewOperators = (int) (operatorsInBusiness * (1.0/this.pConfig.getShareOfOperatorsWithProfit() - 1.0) + 0.0000000000001) - operatorsProspecting;
231                }
232
233                // delete bankrupt ones
234                this.operators = operatorsToKeep;
235
236                if (this.pConfig.getDisableCreationOfNewOperatorsInIteration() > iteration) {
237                        // recreate all other
238                        LinkedList<Operator> newOperators1 = this.operatorInitializer.createAdditionalOperators(this.strategyManager, iteration, numberOfNewOperators);
239                        this.operators.addAll(newOperators1);
240
241                        // too few operators in play, increase to the minimum specified in the config
242                        LinkedList<Operator> newOperators2 = this.operatorInitializer.createAdditionalOperators(this.strategyManager, iteration, (this.pConfig.getNumberOfOperators() - this.operators.size()));
243                        this.operators.addAll(newOperators2);
244
245                        // all operators are in business, increase by one to ensure minimal mutation
246                        if (this.operators.size() == operatorsInBusiness) {
247                                LinkedList<Operator> newOperators3 = this.operatorInitializer.createAdditionalOperators(this.strategyManager, iteration, 1);
248                                this.operators.addAll(newOperators3);
249                        }
250                }
251        }
252
253        TransitSchedule getpTransitSchedule() {
254                return this.pTransitSchedule;
255        }

257        public List<Operator> getOperators() {
258                return operators;
259        }

261        private void writeScheduleToFile(TransitSchedule schedule, String iterationFilename) {
262                TransitScheduleWriter writer = new TransitScheduleWriter(schedule);
263                writer.writeFile(iterationFilename);
264        }
