020package org.matsim.contrib.minibus.scoring;
021
022import org.apache.log4j.Logger;
023import org.matsim.api.core.v01.Id;
024import org.matsim.api.core.v01.events.LinkEnterEvent;
025import org.matsim.api.core.v01.events.PersonLeavesVehicleEvent;
026import org.matsim.api.core.v01.events.TransitDriverStartsEvent;
027import org.matsim.api.core.v01.events.VehicleAbortsEvent;
028import org.matsim.api.core.v01.events.handler.LinkEnterEventHandler;
029import org.matsim.api.core.v01.events.handler.PersonLeavesVehicleEventHandler;
030import org.matsim.api.core.v01.events.handler.TransitDriverStartsEventHandler;
031import org.matsim.api.core.v01.events.handler.VehicleAbortsEventHandler;
032import org.matsim.api.core.v01.network.Network;
033import org.matsim.core.controler.events.AfterMobsimEvent;
034import org.matsim.core.controler.listener.AfterMobsimListener;
035import org.matsim.core.utils.misc.Time;
036import org.matsim.vehicles.Vehicle;
037
038import java.util.HashMap;
039import java.util.LinkedList;
040import java.util.List;
041
042/**
043 * 
044 * Collects all information needed to calculate the fare.
045 * 
046 * @author aneumann
047 *
048 */
049public final class OperatorCostCollectorHandler implements TransitDriverStartsEventHandler, LinkEnterEventHandler, PersonLeavesVehicleEventHandler, AfterMobsimListener, VehicleAbortsEventHandler {
050        
051        private final static Logger log = Logger.getLogger(OperatorCostCollectorHandler.class);
052        
053        private Network network;
054        private final String pIdentifier;
055        private final double costPerVehicleAndDay;
056        private final double expensesPerMeter;
057        private final double expensesPerSecond;
058        
059        private final List<OperatorCostContainerHandler> operatorCostContainerHandlerList = new LinkedList<>();
060        private HashMap<Id<Vehicle>, OperatorCostContainer> vehId2OperatorCostContainer = new HashMap<>();
061        
062        public OperatorCostCollectorHandler(String pIdentifier, double costPerVehicleAndDay, double expensesPerMeter, double expensesPerSecond){
063                this.pIdentifier = pIdentifier;
064                this.costPerVehicleAndDay = costPerVehicleAndDay;
065                this.expensesPerMeter = expensesPerMeter;
066                this.expensesPerSecond = expensesPerSecond;
067                log.info("enabled");
068        }
069        
070        public void init(Network network){
071                this.network = network;
072        }
073        
074        public void addOperatorCostContainerHandler(OperatorCostContainerHandler operatorCostContainerHandler){
075                this.operatorCostContainerHandlerList.add(operatorCostContainerHandler);
076        }
077        
078        @Override
079        public void reset(int iteration) {
080                this.vehId2OperatorCostContainer = new HashMap<>();
081                
082                for (OperatorCostContainerHandler operatorCostContainerHandler : this.operatorCostContainerHandlerList) {
083                        operatorCostContainerHandler.reset();
084                }
085        }
086        
087        @Override
088        public void handleEvent(TransitDriverStartsEvent event) {
089                this.vehId2OperatorCostContainer.put(event.getVehicleId(), new OperatorCostContainer(this.costPerVehicleAndDay, this.expensesPerMeter, this.expensesPerSecond));
090                this.vehId2OperatorCostContainer.get(event.getVehicleId()).handleTransitDriverStarts(event);
091        }
092
093        @Override
094        public void handleEvent(LinkEnterEvent event) {
095                if (this.vehId2OperatorCostContainer.containsKey(event.getVehicleId())) {
096                        // It's a transit driver
097                        double linkLength = this.network.getLinks().get(event.getLinkId()).getLength();
098                        this.vehId2OperatorCostContainer.get(event.getVehicleId()).addDistanceTravelled(linkLength);
099                }
100        }
101
102        @Override
103        public void handleEvent(PersonLeavesVehicleEvent event) {
104                if(event.getVehicleId().toString().startsWith(this.pIdentifier)){
105                        // it's a paratransit vehicle
106                        if(event.getPersonId().toString().contains(this.pIdentifier)){
107                                // it's the driver
108                                OperatorCostContainer operatorCostContainer = this.vehId2OperatorCostContainer.remove(event.getVehicleId());
109                                operatorCostContainer.handleTransitDriverAlights(event);
110                                
111                                // call all OperatorCostContainerHandler
112                                for (OperatorCostContainerHandler operatorCostContainerHandler : this.operatorCostContainerHandlerList) {
113                                        operatorCostContainerHandler.handleOperatorCostContainer(operatorCostContainer);
114                                }
115                                
116                                // Note the operatorCostContainer is dropped at this point.
117                        }
118                }
119        }
120        
121        /**
122         * If the driver agent stucks, no PersonLeavesVehicleEvent will be thrown. Use the 
123         * PersonStuckEvent instead as a replacement to avoid NullPointerExceptions and
124         * calculate an approximate running time cost.
125         */
126        @Override
127        public void handleEvent(VehicleAbortsEvent event) {
128                if (event.getVehicleId().toString().startsWith(this.pIdentifier)) {
129                        // it's a paratransit vehicle
130                        log.info("Paratransit vehicle " + event.getVehicleId() + " got stuck at "
131                                        + Time.writeTime(event.getTime())
132                                        + ". Operation cost is calculated until the stuck time.");
133                        OperatorCostContainer operatorCostContainer = this.vehId2OperatorCostContainer.remove(event.getVehicleId());
134                        operatorCostContainer.handleVehicleAborts(event);
135
136                        // call all OperatorCostContainerHandler
137                        for (OperatorCostContainerHandler operatorCostContainerHandler : this.operatorCostContainerHandlerList) {
138                                operatorCostContainerHandler.handleOperatorCostContainer(operatorCostContainer);
139                        }
140
141                        // Note the operatorCostContainer is dropped at this point.
142                }
143        }
144
145        @Override
146        public void notifyAfterMobsim(AfterMobsimEvent event) {
147                // ok, mobsim is done - finish incomplete entries
148                
149                // remove non paratransit entries
150                List<OperatorCostContainer> entriesToProcess = new LinkedList<>();
151                for (OperatorCostContainer operatorCostContainer : this.vehId2OperatorCostContainer.values()) {
152                        if(operatorCostContainer.getVehicleId().toString().startsWith(this.pIdentifier)){
153                                // it's a paratransit vehicle
154                                entriesToProcess.add(operatorCostContainer);
155                        }
156                }
157                
158                if (entriesToProcess.size() > 0) {
159                        log.warn("There are " + entriesToProcess.size() + " incomplete entries. Will forward them anyway...");
160                        for (OperatorCostContainer operatorCostContainer : entriesToProcess) {
161                                // call all OperatorCostContainerHandler
162                                for (OperatorCostContainerHandler operatorCostContainerHandler : this.operatorCostContainerHandlerList) {
163                                        operatorCostContainerHandler.handleOperatorCostContainer(operatorCostContainer);
164                                }
165                        }
166                }
167        }
168}
