020package org.matsim.contrib.minibus.stats.operatorLogger;
021
022import java.util.ArrayList;
023
024import org.apache.log4j.Logger;
025import org.matsim.api.core.v01.Id;
026import org.matsim.api.core.v01.network.Link;
027import org.matsim.contrib.minibus.PConstants.OperatorState;
028import org.matsim.contrib.minibus.operator.Operator;
029import org.matsim.contrib.minibus.operator.PPlan;
030import org.matsim.core.utils.misc.Time;
031import org.matsim.pt.transitSchedule.api.TransitStopFacility;
032
033/**
034 * Simple storage class for one log entry
035 * 
036 * @author aneumann
037 *
038 */
039public final class LogElement {
040        
041        @SuppressWarnings("unused")
042        private static final Logger log = Logger.getLogger(LogElement.class);
043
044        final static String COMMENTTAG = "#";
045        final static String DELIMITER = "\t";
046        final static String NOVALUE = "=====";
047        
048        private int iteration;
049        private Id<Operator> operatorId;
050        private OperatorState status;
051        private Id<PPlan> planId;
052        private String creatorId;
053        private Id<PPlan> parentId;
054        private int nVeh;
055        private int nPax;
056        private double score;
057        private double budget;
058        private double startTime;
059        private double endTime;
060        private ArrayList<Id<TransitStopFacility>> stopsToBeServed;
061        private ArrayList<Id<Link>> linksServed;
062
063        
064        public String getUniquePlanIdentifier() {
065                return this.operatorId + "_" + this.planId;
066        }
067        
068        public int getIteration() {
069                return iteration;
070        }
071        public void setIteration(int iteration) {
072                this.iteration = iteration;
073        }
074        public Id<Operator> getOperatorId() {
075                return operatorId;
076        }
077        public void setOperatorId(Id<Operator> operatorId) {
078                this.operatorId = operatorId;
079        }
080        public OperatorState getStatus() {
081                return status;
082        }
083        public void setStatus(OperatorState operatorState) {
084                this.status = operatorState;
085        }
086        public Id<PPlan> getPlanId() {
087                return planId;
088        }
089        public void setPlanId(Id<PPlan> planId) {
090                this.planId = planId;
091        }
092        public String getCreatorId() {
093                return creatorId;
094        }
095        public void setCreatorId(String creatorId) {
096                this.creatorId = creatorId;
097        }
098        public Id<PPlan> getParentId() {
099                return parentId;
100        }
101        public void setParentId(Id<PPlan> parentId) {
102                this.parentId = parentId;
103        }
104        public int getnVeh() {
105                return nVeh;
106        }
107        public void setnVeh(int nVeh) {
108                this.nVeh = nVeh;
109        }
110        public int getnPax() {
111                return nPax;
112        }
113        public void setnPax(int nPax) {
114                this.nPax = nPax;
115        }
116        public double getScore() {
117                return score;
118        }
119        public void setScore(double score) {
120                this.score = score;
121        }
122        public double getBudget() {
123                return budget;
124        }
125        public void setBudget(double budget) {
126                this.budget = budget;
127        }
128        public double getStartTime() {
129                return startTime;
130        }
131        public void setStartTime(double startTime) {
132                this.startTime = startTime;
133        }
134        public double getEndTime() {
135                return endTime;
136        }
137        public void setEndTime(double endTime) {
138                this.endTime = endTime;
139        }
140        public ArrayList<Id<TransitStopFacility>> getStopsToBeServed() {
141                return stopsToBeServed;
142        }
143        public void setStopsToBeServed(ArrayList<Id<TransitStopFacility>> stopsServed) {
144                this.stopsToBeServed = stopsServed;
145        }
146        public ArrayList<Id<Link>> getLinksServed() {
147                return linksServed;
148        }
149        public void setLinksServed(ArrayList<Id<Link>> linksServed) {
150                this.linksServed = linksServed;
151        }
152        
153        static String getHeaderLine(){
154                StringBuffer strB = new StringBuffer();
155                strB.append(COMMENTTAG + " ");
156                strB.append("iteration");
157                strB.append(DELIMITER).append("operator");
158                strB.append(DELIMITER).append("status");
159                strB.append(DELIMITER).append("plan id");
160                strB.append(DELIMITER).append("creator");
161                strB.append(DELIMITER).append("parent");
162                strB.append(DELIMITER).append("vehicle");
163                strB.append(DELIMITER).append("pax");
164                strB.append(DELIMITER).append("score");
165                strB.append(DELIMITER).append("budget");
166                strB.append(DELIMITER).append("start time");
167                strB.append(DELIMITER).append("end time");
168                strB.append(DELIMITER).append("important stops");
169                strB.append(DELIMITER).append("links");
170                return strB.toString();
171        }
172        
173        public String toString(){
174                StringBuffer strB = new StringBuffer();
175                strB.append(this.iteration);
176                strB.append(DELIMITER).append(this.operatorId.toString());
177                strB.append(DELIMITER).append(this.status);
178                strB.append(DELIMITER).append(this.planId);
179                strB.append(DELIMITER).append(this.creatorId);
180                strB.append(DELIMITER).append(this.parentId);
181                strB.append(DELIMITER).append(this.nVeh);
182                strB.append(DELIMITER).append(this.nPax);
183                strB.append(DELIMITER).append(this.score);
184                strB.append(DELIMITER).append(this.budget);
185                strB.append(DELIMITER).append(Time.writeTime(this.startTime));
186                strB.append(DELIMITER).append(Time.writeTime(this.endTime));
187                strB.append(DELIMITER).append(this.stopsToBeServed);
188                strB.append(DELIMITER).append(this.linksServed);
189                return strB.toString();
190        }
191        
192        String getTotalString(){
193                StringBuffer strB = new StringBuffer();
194                strB.append(this.iteration);
195                strB.append(DELIMITER).append(this.operatorId.toString());
196                strB.append(DELIMITER).append(this.status);
197                strB.append(DELIMITER).append(NOVALUE);
198                strB.append(DELIMITER).append(NOVALUE);
199                strB.append(DELIMITER).append(NOVALUE);
200                strB.append(DELIMITER).append(nVeh);
201                strB.append(DELIMITER).append(nPax);
202                strB.append(DELIMITER).append(score);
203                strB.append(DELIMITER).append(budget);
204                strB.append(DELIMITER).append(NOVALUE);
205                strB.append(DELIMITER).append(NOVALUE);
206                strB.append(DELIMITER).append(NOVALUE);
207                strB.append(DELIMITER).append(NOVALUE);
208                return strB.toString();
209        }
210}
