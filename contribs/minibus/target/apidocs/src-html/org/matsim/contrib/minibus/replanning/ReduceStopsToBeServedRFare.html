020package org.matsim.contrib.minibus.replanning;
021
022import org.apache.log4j.Logger;
023import org.matsim.api.core.v01.Id;
024import org.matsim.contrib.minibus.fare.StageContainer;
025import org.matsim.contrib.minibus.fare.StageContainerHandler;
026import org.matsim.contrib.minibus.fare.TicketMachineI;
027import org.matsim.contrib.minibus.genericUtils.RecursiveStatsContainer;
028import org.matsim.contrib.minibus.operator.Operator;
029import org.matsim.contrib.minibus.operator.PPlan;
030import org.matsim.pt.transitSchedule.api.TransitRoute;
031import org.matsim.pt.transitSchedule.api.TransitRouteStop;
032import org.matsim.pt.transitSchedule.api.TransitStopFacility;
033
034import java.util.ArrayList;
035import java.util.LinkedHashMap;
036import java.util.Map.Entry;
037import java.util.Set;
038import java.util.TreeSet;
039
040/**
041 * 
042 * Removes all stops belonging to a demand relation with trips below a certain threshold.
043 * Threshold is standard deviation of number of trips or collected fare of all relations twice a scaling factor.
044 * 
045 * @author aneumann
046 *
047 */
048public final class ReduceStopsToBeServedRFare extends AbstractPStrategyModule implements StageContainerHandler{
049        
050        private final static Logger log = Logger.getLogger(ReduceStopsToBeServedRFare.class);
051        
052        public static final String STRATEGY_NAME = "ReduceStopsToBeServedRFare";
053        
054        private final double sigmaScale;
055        private final boolean useFareAsWeight;
056
057        private TicketMachineI ticketMachine;
058        private LinkedHashMap<Id<TransitRoute>, LinkedHashMap<Id<TransitStopFacility>, LinkedHashMap<Id<TransitStopFacility>, Double>>> route2StartStop2EndStop2WeightMap = new LinkedHashMap<>();
059        
060        public ReduceStopsToBeServedRFare(ArrayList<String> parameter) {
061                super();
062                if(parameter.size() != 2){
063                        log.error("Too many parameter. Will ignore: " + parameter);
064                        log.error("Parameter 1: Scaling factor for sigma");
065                        log.error("Parameter 2: true=use the fare as weight, false=use number of trips as weight");
066                }
067                this.sigmaScale = Double.parseDouble(parameter.get(0));
068                this.useFareAsWeight = Boolean.parseBoolean(parameter.get(1));
069                log.info("enabled");
070        }
071        
072        @Override
073        public PPlan run(Operator operator) {
074                // get best plans route id
075                TransitRoute routeToOptimize = null;
076                if (operator.getBestPlan().getLine().getRoutes().size() != 1) {
077                        log.error("There should be only one route at this time - Please check");
078                }
079                for (TransitRoute route : operator.getBestPlan().getLine().getRoutes().values()) {
080                        routeToOptimize = route;
081                }
082                
083                ArrayList<TransitStopFacility> stopsToBeServed = getStopsToBeServed(this.route2StartStop2EndStop2WeightMap.get(routeToOptimize.getId()), routeToOptimize);
084                
085                if (stopsToBeServed.size() < 2) {
086                        // too few stops left - cannot return a new plan
087                        return null;
088                }
089                
090                // profitable route, change startTime
091                PPlan newPlan = new PPlan(operator.getNewPlanId(), this.getStrategyName(), operator.getBestPlan().getId());
092                newPlan.setNVehicles(1);
093                newPlan.setStopsToBeServed(stopsToBeServed);
094                newPlan.setStartTime(operator.getBestPlan().getStartTime());
095                newPlan.setEndTime(operator.getBestPlan().getEndTime());
096                
097                newPlan.setLine(operator.getRouteProvider().createTransitLineFromOperatorPlan(operator.getId(), newPlan));
098                
099                return newPlan;
100        }
101
102        private ArrayList<TransitStopFacility> getStopsToBeServed(LinkedHashMap<Id<TransitStopFacility>, LinkedHashMap<Id<TransitStopFacility>, Double>> startStop2EndStop2WeightMap, TransitRoute routeToOptimize) {
103                ArrayList<TransitStopFacility> tempStopsToBeServed = new ArrayList<>();
104                RecursiveStatsContainer stats = new RecursiveStatsContainer();
105                
106                if (startStop2EndStop2WeightMap == null) {
107                        // There is no entry for that particular line - possibly no demand - returning empty line
108                        return tempStopsToBeServed;
109                }
110                
111                // calculate standard deviation
112                for (LinkedHashMap<Id<TransitStopFacility>, Double> endStop2TripsMap : startStop2EndStop2WeightMap.values()) {
113                        for (Double trips : endStop2TripsMap.values()) {
114                                stats.handleNewEntry(trips);
115                        }
116                }
117                
118                if (stats.getNumberOfEntries() == 1) {
119                        // We use circular routes. There is always a way back (with no demand). Add a second entry.
120                        stats.handleNewEntry(0.0);                      
121                }
122                
123                double sigmaTreshold = stats.getStdDev() * this.sigmaScale;
124                Set<Id<TransitStopFacility>> stopIdsAboveTreshold = new TreeSet<>();
125                
126                // Get all stops serving a demand above threshold
127                for (Entry<Id<TransitStopFacility>, LinkedHashMap<Id<TransitStopFacility>, Double>> endStop2TripsMapEntry : startStop2EndStop2WeightMap.entrySet()) {
128                        for (Entry<Id<TransitStopFacility>, Double> tripEntry : endStop2TripsMapEntry.getValue().entrySet()) {
129                                if (tripEntry.getValue() > sigmaTreshold) {
130                                        // ok - add the corresponding stops to the set
131                                        stopIdsAboveTreshold.add(endStop2TripsMapEntry.getKey());
132                                        stopIdsAboveTreshold.add(tripEntry.getKey());
133                                }
134                        }
135                }
136                
137                // Get new stops to be served
138                for (TransitRouteStop stop : routeToOptimize.getStops()) {
139                        if (stopIdsAboveTreshold.contains(stop.getStopFacility().getId())) {
140                                tempStopsToBeServed.add(stop.getStopFacility());
141                        }
142                }
143                
144                ArrayList<TransitStopFacility> stopsToBeServed = new ArrayList<>();
145                
146                // avoid using a stop twice in a row
147                for (TransitStopFacility stop : tempStopsToBeServed) {
148                        if (stopsToBeServed.size() > 0) {
149                                if (stopsToBeServed.get(stopsToBeServed.size() - 1).getId() != stop.getId()) {
150                                        stopsToBeServed.add(stop);
151                                }
152                        } else {
153                                stopsToBeServed.add(stop);
154                        }                       
155                }
156                
157                // delete last stop, if it is the same as the first one
158                if (stopsToBeServed.size() > 1) {
159                        if (stopsToBeServed.get(0).getId() == stopsToBeServed.get(stopsToBeServed.size() - 1).getId()) {
160                                stopsToBeServed.remove(stopsToBeServed.size() - 1);
161                        }
162                }
163                
164                return stopsToBeServed;
165        }
166
167        @Override
168        public String getStrategyName() {
169                return ReduceStopsToBeServedRFare.STRATEGY_NAME;
170        }
171        
172        @Override
173        public void reset() {
174                this.route2StartStop2EndStop2WeightMap = new LinkedHashMap<>();
175        }
176
177        @Override
178        public void handleFareContainer(StageContainer stageContainer) {
179                Id<TransitRoute> routeId = stageContainer.getRouteId();
180                Id<TransitStopFacility> startStopId = stageContainer.getStopEntered();
181                Id<TransitStopFacility> endStopId = stageContainer.getStopLeft();
182                
183                if (this.route2StartStop2EndStop2WeightMap.get(routeId) == null) {
184                        this.route2StartStop2EndStop2WeightMap.put(routeId, new LinkedHashMap<Id<TransitStopFacility>, LinkedHashMap<Id<TransitStopFacility>, Double>>());
185                }
186
187                if (this.route2StartStop2EndStop2WeightMap.get(routeId).get(startStopId) == null) {
188                        this.route2StartStop2EndStop2WeightMap.get(routeId).put(startStopId, new LinkedHashMap<Id<TransitStopFacility>, Double>());
189                }
190
191                if (this.route2StartStop2EndStop2WeightMap.get(routeId).get(startStopId).get(endStopId) == null) {
192                        this.route2StartStop2EndStop2WeightMap.get(routeId).get(startStopId).put(endStopId, 0.0);
193                }
194
195                double oldWeight = this.route2StartStop2EndStop2WeightMap.get(routeId).get(startStopId).get(endStopId);
196                double additionalWeight = 1.0;
197                if (this.useFareAsWeight) {
198                        additionalWeight = this.ticketMachine.getFare(stageContainer);
199                }
200                this.route2StartStop2EndStop2WeightMap.get(routeId).get(startStopId).put(endStopId, oldWeight + additionalWeight);
201        }
202
203        public void setTicketMachine(TicketMachineI ticketMachine) {
204                this.ticketMachine = ticketMachine;
205        }
