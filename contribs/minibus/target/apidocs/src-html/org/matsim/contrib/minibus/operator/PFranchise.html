020package org.matsim.contrib.minibus.operator;
021
022import org.apache.log4j.Logger;
023import org.matsim.contrib.minibus.genericUtils.GridNode;
024import org.matsim.pt.transitSchedule.api.TransitStopFacility;
025
026import java.util.ArrayList;
027import java.util.LinkedList;
028import java.util.Set;
029import java.util.TreeSet;
030
031/**
032 * Simple Franchise system rejecting all routes already operated with respect to stops served and time of operation
033 * 
034 * @author aneumann
035 *
036 */
037public final class PFranchise {
038
039        private final static Logger log = Logger.getLogger(PFranchise.class);
040        
041        private final boolean activated;
042        private final double gridSize;
043        private TreeSet<String> routeHashes = new TreeSet<>();
044
045        
046        public PFranchise(boolean useFranchise, double gridSize) {
047                this.activated = useFranchise;
048                this.gridSize = gridSize;
049                if(this.activated){
050                        log.info("Franchise system activated");
051                } else{
052                        log.info("Franchise system NOT activated");
053                }
054        }
055
056        public boolean planRejected(PPlan plan) {
057                
058                if(!this.activated){
059                        return false;
060                }
061                
062                String routeHash = generateRouteHash(plan);
063                boolean reject = this.routeHashes.contains(routeHash);
064                this.routeHashes.add(routeHash);
065                
066                return reject;
067        }
068
069        /**
070         * Reset all route hashes to the routes currently in use
071         * 
072         * @param operators
073         */
074        public void reset(LinkedList<Operator> operators) {
075                
076                if(this.activated){
077                        this.routeHashes = new TreeSet<>();
078
079                        for (Operator operator : operators) {
080                                Set<String> routesHashesOfOperator = new TreeSet<>();
081                                for (PPlan plan : operator.getAllPlans()) {
082                                        String routeHash = generateRouteHash(plan);                             
083                                        if (this.routeHashes.contains(routeHash)) {
084                                                if (routesHashesOfOperator.contains(routeHash)) {
085                                                        // This route is already served by the same operator
086                                                        log.warn("Operator " + operator.getId() + " offers the same plan twice. Plan: " + plan.getId() + " managed to circumvent the franchise system with route " + routeHash);
087                                                } else {
088                                                        // This route is already served by another operator
089                                                        log.warn("Operator " + operator.getId() + " with plan " + plan.getId() + " managed to circumvent the franchise system with route " + routeHash);
090                                                }
091                                        }
092                                        this.routeHashes.add(routeHash);
093                                        routesHashesOfOperator.add(routeHash);
094                                }
095                        }
096                }
097        }
098        
099        private String generateRouteHash(PPlan plan) {
100        //      return generateRouteHash(plan.getStopsToBeServed());
101        //      return generateRouteHash(plan.getStartTime(), plan.getEndTime(), plan.getStopsToBeServed());
102                return generateRouteHashWithGridNodes(plan.getStartTime(), plan.getEndTime(), plan.getStopsToBeServed());
103        }
104
105        /**
106         * Generates a unique String from the stops given
107         * 
108         * @param stopsToBeServed
109         * @return
110         */
111        private String generateRouteHash(ArrayList<TransitStopFacility> stopsToBeServed){
112                StringBuffer sB = null;
113                for (TransitStopFacility transitStopFacility : stopsToBeServed) {
114                        if (sB == null) {
115                                sB = new StringBuffer();
116                                sB.append(transitStopFacility.getId().toString());
117                        } else {
118                                sB.append("-");
119                                sB.append(transitStopFacility.getId().toString()); 
120                        }                       
121                }
122                return sB.toString();
123        }
124
125        private String generateRouteHash(double startTime, double endTime, ArrayList<TransitStopFacility> stopsToBeServed) {
126                StringBuffer sB = new StringBuffer();
127                sB.append(startTime);
128                sB.append("-" + endTime);
129                
130                for (TransitStopFacility transitStopFacility : stopsToBeServed) {
131                        sB.append("-");
132                        sB.append(transitStopFacility.getId().toString()); 
133                }
134
135                return sB.toString();
136        }
137        
138        private String generateRouteHashWithGridNodes(double startTime, double endTime, ArrayList<TransitStopFacility> stopsToBeServed) {
139                StringBuffer sB = new StringBuffer();
140                sB.append(startTime);
141                sB.append("-" + endTime);
142                
143                String lastGridNodeId = null;
144                for (TransitStopFacility transitStopFacility : stopsToBeServed) {
145                        String gridNodeId = GridNode.getGridNodeIdForCoord(transitStopFacility.getCoord(), this.gridSize);
146                        
147                        if (lastGridNodeId == null) {
148                                lastGridNodeId = gridNodeId;
149                        } else {
150                                if (gridNodeId.equalsIgnoreCase(lastGridNodeId)) {
151                                        // still in same gridSquare
152                                        continue;
153                                }
154                                lastGridNodeId = gridNodeId;
155                        }
156
157                        sB.append("-");
158                        sB.append(lastGridNodeId); 
159                }
160
161                return sB.toString();
162        }
163}
