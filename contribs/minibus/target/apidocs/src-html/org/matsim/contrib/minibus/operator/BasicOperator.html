020package org.matsim.contrib.minibus.operator;
021
022import org.matsim.api.core.v01.Id;
023import org.matsim.contrib.minibus.PConfigGroup;
024import org.matsim.contrib.minibus.PConstants.OperatorState;
025import org.matsim.contrib.minibus.replanning.PStrategy;
026import org.matsim.contrib.minibus.replanning.PStrategyManager;
027
028/**
029 * Operator that manages one paratransit line
030 * 
031 * @author aneumann
032 *
033 */
034public final class BasicOperator extends AbstractOperator{
035        
036        public static final String OPERATOR_NAME = "BasicOperator"; 
037
038        public BasicOperator(Id<Operator> id, PConfigGroup pConfig, PFranchise franchise){
039                super(id, pConfig, franchise);
040        }
041
042        @Override
043        public void replan(PStrategyManager pStrategyManager, int iteration) {
044                this.currentIteration = iteration;
045                
046                if(this.testPlan != null){
047                        // compare scores
048                        if (this.getScore() > this.getScoreLastIteration()){
049                                // testPlan improves the plan, apply its modification to bestPlan, transfer the vehicle from the testPlan to the bestPlan
050                                // changed to get a more useful output in the pOperatorLogger /dr
051                                this.testPlan.setNVehicles(this.testPlan.getNVehicles() + this.bestPlan.getNVehicles());
052                                this.bestPlan = this.testPlan;
053                        }else{
054                                this.bestPlan.setNVehicles(this.bestPlan.getNVehicles() + this.testPlan.getNVehicles());
055                        }
056                        this.testPlan = null;
057                }
058                
059                // balance the budget
060                if(this.budget < 0){
061                        // insufficient, sell vehicles
062                        int numberOfVehiclesToSell = -1 * Math.min(-1, (int) Math.floor(this.budget / this.costPerVehicleSell));
063                        
064                        if(this.bestPlan.getNVehicles() - numberOfVehiclesToSell < 1){
065                                // can not balance the budget by selling vehicles, bankrupt
066                                log.error("This should not be possible at this time.");
067                                this.operatorState = OperatorState.BANKRUPT;
068                                return;
069                        }
070
071                        // can balance the budget, so sell vehicles
072                        this.bestPlan.setNVehicles(this.bestPlan.getNVehicles() - numberOfVehiclesToSell);
073                        this.budget += this.costPerVehicleSell * numberOfVehiclesToSell;
074//                      log.info("Sold " + numberOfVehiclesToSell + " vehicle from line " + this.id + " - new budget is " + this.budget);
075                }
076
077                // THIS IS A COPY FROM org.matsim.contrib.minibus.replanning.modules.deprecated.AggressiveIncreaseNumberOfVehicles;
078                // First buy vehicles
079                int vehicleBought = 0;
080                
081                while (this.getBudget() > this.getCostPerVehicleBuy()) {
082                        // budget ok, buy one
083                        this.setBudget(this.getBudget() - this.getCostPerVehicleBuy());
084                        vehicleBought++;
085                }                                       
086                
087                if (vehicleBought == 0) {
088                        this.testPlan = null;
089                }
090                        
091                // vehicles were bought - create plan
092                PPlan oldPlan = this.getBestPlan();
093                PPlan plan = new PPlan(this.getNewPlanId(), "Copied code from AggressiveIncreaseNumberOfVehicles", oldPlan.getId());
094                plan.setStopsToBeServed(oldPlan.getStopsToBeServed());
095                plan.setStartTime(oldPlan.getStartTime());
096                plan.setEndTime(oldPlan.getEndTime());
097                plan.setScore(this.getBestPlan().getScore());
098                
099                plan.setNVehicles(vehicleBought);
100                
101                plan.setLine(this.getRouteProvider().createTransitLineFromOperatorPlan(this.getId(), plan));
102                
103                this.testPlan = plan;
104                // END OF COPY
105                
106                
107                // Second replan, if testplan null
108                if (this.testPlan == null) {
109                        PStrategy strategy = pStrategyManager.chooseStrategy();
110                        if (strategy != null) {
111                                this.testPlan = strategy.run(this);
112                                if (this.testPlan != null) {
113                                        this.bestPlan.setNVehicles(this.bestPlan.getNVehicles() - 1);
114                                }
115                        }
116                }
117                
118                // reinitialize the plan
119                this.bestPlan.setLine(this.routeProvider.createTransitLineFromOperatorPlan(this.id, this.bestPlan));
120                
121                this.updateCurrentTransitLine();
122        }
123        
124}
