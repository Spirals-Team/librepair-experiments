037/**
038 * Wrapper for {@linkplain RaptorWalker}.
039 *
040 * @author aneumann
041 */
042public class Raptor extends AbstractTransitRouter implements TransitRouter {
043
044        private final TransitRouterQuadTree transitRouterQuadTree;
045
046        private final RaptorWalker raptorWalker;
047        private final TransitRouterConfig config;
048        private RaptorDisutility raptorDisutility ;
049
050        // MAGIC NUMBERS BELOW THIS LINE
051        int maxTransfers = 10;
052        int graceRuns = 1;
053        // END MAGIC NUMBERS
054
055        public Raptor(TransitRouterConfig transitRouterConfig, TransitSchedule transitSchedule, TransitTravelDisutility raptorDisutility) {
056                super (transitRouterConfig, raptorDisutility);
057                this.config = transitRouterConfig;
058                // casting to raptorDisutility is necessary here. At the moment, I dont know if there is a better way. Amit Oct'17
059                this.transitRouterQuadTree = new TransitRouterQuadTree( (RaptorDisutility) raptorDisutility);
060                this.transitRouterQuadTree.initializeFromSchedule(transitSchedule, config.getBeelineWalkConnectionDistance());
061                this.raptorDisutility = (RaptorDisutility) raptorDisutility;
062                this.raptorWalker = new RaptorWalker(this.transitRouterQuadTree.getSearchData(), raptorDisutility, this.maxTransfers, this.graceRuns);
063        }
064
065        public Raptor(TransitRouterConfig transitRouterConfig, TransitSchedule transitSchedule) {
066                // do something about these two magic numbers for raptor disutility. Amit Oct'17
067                this(transitRouterConfig, transitSchedule, new RaptorDisutility(transitRouterConfig, 0., 0.));
068        }
069
070        private Map<TransitStopFacility, InitialNode> locateWrappedNearestTransitStops(Person person, Coord coord, double departureTime) {
071                Collection<TransitStopFacility> nearestTransitStops = this.transitRouterQuadTree.getNearestTransitStopFacilities(coord, this.config.getSearchRadius());
072                if (nearestTransitStops.size() < 2) {
073                        // also enlarge search area if only one stop found, maybe a second one is near the border of the search area
074                        TransitStopFacility nearestTransitStop = this.transitRouterQuadTree.getNearestTransitStopFacility(coord);
075                        double distance = CoordUtils.calcEuclideanDistance(coord, nearestTransitStop.getCoord());
076                        nearestTransitStops = this.transitRouterQuadTree.getNearestTransitStopFacilities(coord, distance + this.config.getExtensionRadius());
077                }
078                Map<TransitStopFacility, InitialNode> wrappedNearestTransitStops2AccessCost = new LinkedHashMap<>();
079                for (TransitStopFacility node : nearestTransitStops) {
080                        Coord toCoord = node.getCoord();
081                        double initialTime = getWalkTime(null, coord, toCoord);
082                        double initialCost = getWalkDisutility(coord, toCoord);
083                        wrappedNearestTransitStops2AccessCost.put(node, new InitialNode(initialCost, initialTime + departureTime));
084                }
085                return wrappedNearestTransitStops2AccessCost;
086        }
087
088        private double getWalkDisutility(Coord coord, Coord toCoord) {
089                return this.raptorDisutility.getWalkTravelDisutility(null, coord, toCoord);
090        }
091
092        @Override
093        public List<Leg> calcRoute(final Facility<?> fromFacility, final Facility<?> toFacility, final double departureTime, final Person person) {
094                // find possible start stops
095                Map<TransitStopFacility, InitialNode> fromStops = this.locateWrappedNearestTransitStops(person, fromFacility.getCoord(), departureTime);
096                // find possible end stops
097                Map<TransitStopFacility, InitialNode> toStops = this.locateWrappedNearestTransitStops(person, toFacility.getCoord(), departureTime);
098
099                // find routes between start and end stops
100                TransitPassengerRoute p = this.raptorWalker.calcLeastCostPath(fromStops, toStops);
101
102                if (p == null) {
103                        // return null;
104                        // returning at least walk legs if no PT route is found. Amit Aug'17
105                        return this.createDirectWalkLegList(null, fromFacility.getCoord(), toFacility.getCoord());
106                }
107
108                double directWalkCost = getWalkDisutility(fromFacility.getCoord(), toFacility.getCoord());
109                double pathCost = p.getTravelCost();
110
111                if (directWalkCost * getConfig().getDirectWalkFactor() < pathCost) {
112                        return this.createDirectWalkLegList(null, fromFacility.getCoord(), toFacility.getCoord());
113                }
114                return convertPassengerRouteToLegList(departureTime, p, fromFacility.getCoord(), toFacility.getCoord(), person);
115        }

117}
