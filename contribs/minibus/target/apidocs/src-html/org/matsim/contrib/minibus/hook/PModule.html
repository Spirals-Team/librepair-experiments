023package org.matsim.contrib.minibus.hook;
024
025import org.apache.log4j.Logger;
026import org.matsim.contrib.minibus.PConfigGroup;
027import org.matsim.contrib.minibus.fare.TicketMachineDefaultImpl;
028import org.matsim.contrib.minibus.fare.TicketMachineI;
029import org.matsim.contrib.minibus.operator.POperators;
030import org.matsim.contrib.minibus.operator.PerPassengerSubsidy;
031import org.matsim.contrib.minibus.operator.SubsidyI;
032import org.matsim.contrib.minibus.stats.PStatsModule;
033import org.matsim.contrib.minibus.stats.abtractPAnalysisModules.BVGLines2PtModes;
034import org.matsim.contrib.minibus.stats.abtractPAnalysisModules.LineId2PtMode;
035import org.matsim.core.config.ConfigUtils;
036import org.matsim.core.controler.AbstractModule;
037import org.matsim.pt.router.TransitRouter;
038
039public final class PModule extends AbstractModule {
040        private final static Logger log = Logger.getLogger(PModule.class);
041
042        @Override public void install() {
043                final PTransitRouterFactory pTransitRouterFactory = new PTransitRouterFactory(this.getConfig());
044                bind(TransitRouter.class).toProvider(pTransitRouterFactory);
045                bind(PTransitRouterFactory.class).toInstance(pTransitRouterFactory);
046
047                addControlerListenerBinding().to(PControlerListener.class) ;
048//              addControlerListenerBinding().toInstance(pTransitRouterFactory);
049                // (needs to be injected _after_ PControlerListener, so that it is executed _before_ PControlerListener.
050                // yyyy injecting the TransitRouterFactory besides the TransitRouter is a fix to re-configure the factory in every iteration.
051                // A more general solution suggested by MZ would be to define an iteration scope.  Then the factory could be forced
052                // to reconstruct itself in every iteration, thus pulling new information (in this case the updated transit schedule)
053                // by itself.  Is on the "list", has not been done yet, will be done eventually, until then this remains the way it is.
054                // kai, jan'17)
055                // The iteration scope of PControlerListener should be executed before iteration scope of PTransitRouterFactory.
056                // It is not clear, how to control over the order of the execution. If transit schedule and thus transit router needs to be updated at every iteration,
057                // it needs to be explicitly triggered in PControlerListener which is mainly same as before. See also MATSim-768. GL, AA, AN. Jan'18
058
059                bind(TicketMachineI.class).to(TicketMachineDefaultImpl.class);
060                
061                if ( ConfigUtils.addOrGetModule(getConfig(), PConfigGroup.class ).getSubsidyApproach() == null ) {
062                        log.info("There is no subsidy added to the operators' score.");
063                        
064                } else if ( ConfigUtils.addOrGetModule(getConfig(), PConfigGroup.class ).getSubsidyApproach().equals("perPassenger") )  {
065                        log.warn("There is a subsidy added to the operators' score. Approach: 'perPassenger'."
066                                        + " This approach is rather an example how to implement a subsidy computation approach and should not be used"
067                                        + " for real studies...");
068                        bind(SubsidyI.class).to(PerPassengerSubsidy.class);
069                
070                } else {
071                        log.warn("Unknown subsidy approach: " + ConfigUtils.addOrGetModule(getConfig(), PConfigGroup.class ).getSubsidyApproach());
072                        log.warn("Add the following lines of code to your run class:");
073                        log.warn("");
074                        log.warn("controler.addOverridingModule(new AbstractModule() {");
075                        log.warn("");
076                        log.warn("      @Override");
077                        log.warn("      public void install() {");
078                        log.warn("              this.bind(SubsidyI.class).to(<YourSubsidyApproach>.class)");
079                        log.warn("      }");
080                        log.warn("});");
081                        log.warn("");
082                        throw new RuntimeException("Aborting...");
083                }
084
085                bind(POperators.class).to(PBox.class).asEagerSingleton();
086                // (needs to be a singleton since it is a data container, and all clients should use the same data container. kai, jan'17)
087
088                bindMobsim().toProvider(PQSimProvider.class) ;
089                bind(LineId2PtMode.class).to( BVGLines2PtModes.class ) ;
090
091                install( new PStatsModule() ) ;
092
093                if ( ConfigUtils.addOrGetModule(getConfig(), PConfigGroup.class ).getReRouteAgentsStuck() ) {
094                        bind(PersonReRouteStuckFactory.class).to( PersonReRouteStuckFactoryImpl.class );
095                        bind(AgentsStuckHandlerImpl.class) ;
096                }
097
098        }
099
100}
