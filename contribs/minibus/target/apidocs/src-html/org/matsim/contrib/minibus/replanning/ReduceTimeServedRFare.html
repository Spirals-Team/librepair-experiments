037/**
038 * 
039 * Restricts the time of operation to temporal relations higher than a certain threshold.
040 * Threshold is standard deviation of number of trips or collected fare of all relations twice a scaling factor.
041 * 
042 * @author aneumann
043 *
044 */
045public final class ReduceTimeServedRFare extends AbstractPStrategyModule implements StageContainerHandler{
046        
047        private final static Logger log = Logger.getLogger(ReduceTimeServedRFare.class);
048        
049        public static final String STRATEGY_NAME = "ReduceTimeServedRFare";
050
051        private final double sigmaScale;
052        private final int timeBinSize;
053        private final boolean useFareAsWeight;
054        private final boolean allowForSplitting;
055        
056        private LinkedHashMap<Id<TransitRoute>,LinkedHashMap<Integer,LinkedHashMap<Integer,Double>>> route2StartTimeSlot2EndTimeSlot2WeightMap = new LinkedHashMap<>();
057        private TicketMachineI ticketMachine;
058
059
060        public ReduceTimeServedRFare(ArrayList<String> parameter) {
061                super();
062                if(parameter.size() != 4){
063                        log.error("Too many parameter. Will ignore: " + parameter);
064                        log.error("Parameter 1: Scaling factor for sigma");
065                        log.error("Parameter 2: Time bin size in seconds");
066                        log.error("Parameter 3: true=use the fare as weight, false=use number of trips as weight");
067                        log.error("Parameter 4: true=allow for picking one of the demand segments, false=enforce coverage of all demand segments");
068                }
069                this.sigmaScale = Double.parseDouble(parameter.get(0));
070                this.timeBinSize = Integer.parseInt(parameter.get(1));
071                this.useFareAsWeight = Boolean.parseBoolean(parameter.get(2));
072                this.allowForSplitting = Boolean.parseBoolean(parameter.get(3));
073                log.info("enabled");
074        }
075        
076        @Override
077        public PPlan run(Operator operator) {
078                // get best plans route id
079                TransitRoute routeToOptimize = null;
080                
081                if (operator.getBestPlan().getLine().getRoutes().size() != 1) {
082                        log.error("There should be only one route at this time - Please check");
083                }
084                for (TransitRoute route : operator.getBestPlan().getLine().getRoutes().values()) {
085                        routeToOptimize = route;
086                }
087                
088                Tuple<Double,Double> timeToBeServed = getTimeToBeServed(this.route2StartTimeSlot2EndTimeSlot2WeightMap.get(routeToOptimize.getId()));
089                
090                if (timeToBeServed == null) {
091                        // Could not modify plan
092                        return null;
093                }
094                
095                // profitable route, change startTime
096                PPlan newPlan = new PPlan(operator.getNewPlanId(), this.getStrategyName(), operator.getBestPlan().getId());
097                newPlan.setNVehicles(1);
098                newPlan.setStopsToBeServed(operator.getBestPlan().getStopsToBeServed());
099                
100                newPlan.setStartTime(timeToBeServed.getFirst());
101                newPlan.setEndTime(timeToBeServed.getSecond());
102                
103                newPlan.setLine(operator.getRouteProvider().createTransitLineFromOperatorPlan(operator.getId(), newPlan));
104                
105                return newPlan;
106        }

109        private Tuple<Double,Double> getTimeToBeServed(LinkedHashMap<Integer,LinkedHashMap<Integer,Double>> startSlot2EndSlot2TripsMap) {
110                RecursiveStatsContainer stats = new RecursiveStatsContainer();
111                
112                if (startSlot2EndSlot2TripsMap == null) {
113                        // There is no entry for that particular line - possibly no demand - returning empty line
114                        return null;
115                }
116                
117                // calculate standard deviation
118                for (LinkedHashMap<Integer, Double> EndSlot2TripsMap : startSlot2EndSlot2TripsMap.values()) {
119                        for (Double trips : EndSlot2TripsMap.values()) {
120                                stats.handleNewEntry(trips);
121                        }
122                }
123                
124                if (stats.getNumberOfEntries() == 1) {
125                        // We use circular routes. There is always a way back (with no demand). Add a second entry.
126                        stats.handleNewEntry(0.0);                      
127                }
128                
129                double sigmaTreshold = stats.getStdDev() * this.sigmaScale;
130                Set<Integer> slotsAboveTreshold = new TreeSet<>();
131                
132                // Get all slots serving a demand above threshold
133                for (Entry<Integer, LinkedHashMap<Integer, Double>> endSlot2TripsMapEntry : startSlot2EndSlot2TripsMap.entrySet()) {
134                        for (Entry<Integer, Double> tripEntry : endSlot2TripsMapEntry.getValue().entrySet()) {
135                                if (tripEntry.getValue() > sigmaTreshold) {
136                                        // ok - add the corresponding slots to the set
137                                        slotsAboveTreshold.add(endSlot2TripsMapEntry.getKey());
138                                        slotsAboveTreshold.add(tripEntry.getKey());
139                                }
140                        }
141                }
142                
143                // Get new slots to be served
144                if (allowForSplitting) {
145                        return this.getSplittedTimeOfOperation(slotsAboveTreshold);
146                } else {
147                        return this.getMaxTimeOfOperation(slotsAboveTreshold);
148                }
149        }
150
151        private Tuple<Double, Double> getSplittedTimeOfOperation(Set<Integer> slotsAboveTreshold) {
152                
153                // Sort the time slots
154                LinkedList<Integer> sortedSlotsAboveTreshold = new LinkedList<>();
155                sortedSlotsAboveTreshold.addAll(slotsAboveTreshold);
156                Collections.sort(sortedSlotsAboveTreshold);
157                
158                // Generate seamlessly connected time slot tuples
159                List<Tuple<Integer, Integer>> timeSlotTuples = new LinkedList<>();
160                
161                int currentStart = -1;
162                int currentEnd = -1;
163                for (Integer slot : sortedSlotsAboveTreshold) {
164                        if (currentStart == -1) {
165                                // a new slot begins
166                                currentStart = slot;
167                                currentEnd = currentStart;
168                        } else {
169                                // continue with the current slot
170                                // it's an old time slot which needs to be enlarged or completed
171                                if (slot == currentEnd + 1) {
172                                        // this timeslot can be enlarged
173                                        currentEnd = slot;
174                                } else {
175                                        // need to finish this one and start a new one
176                                        if (currentEnd == currentStart) {
177                                                // end time is not allowed to be the same as start time
178                                                currentEnd++;
179                                        }
180                                        timeSlotTuples.add(new Tuple<>(currentStart, currentEnd));
181                                        currentStart = slot;
182                                        currentEnd = currentStart;
183                                }
184                        }
185                }
186
187                // complete the last one
188                if (currentEnd == currentStart) {
189                        // end time is not allowed to be the same as start time
190                        currentEnd++;
191                }
192                timeSlotTuples.add(new Tuple<>(currentStart, currentEnd));
193                
194                // Get total weight of all time slot tuples
195                double totalWeight = 0.0;
196                for (Tuple<Integer, Integer> tuple : timeSlotTuples) {
197                        int weight = tuple.getSecond() - tuple.getFirst();
198                        totalWeight += weight;
199                }
200                
201                // Weighted random draw
202                double accumulatedWeight = 0.0;
203                double rndTreshold = MatsimRandom.getRandom().nextDouble() * totalWeight;
204                for (Tuple<Integer, Integer> tuple : timeSlotTuples) {
205                        accumulatedWeight += tuple.getSecond() - tuple.getFirst();
206                        if (rndTreshold <= accumulatedWeight) {
207                                // ok, take this one
208                                return getTimeOfOperationFromTimeSlots(tuple.getFirst(), tuple.getSecond());
209                        }
210                }               
211                
212                log.error("Should never be able to get here");
213                return null;
214        }
215        
216        private Tuple<Double, Double> getMaxTimeOfOperation(Set<Integer> slotsAboveTreshold) {
217                int min = Integer.MAX_VALUE;
218                int max = Integer.MIN_VALUE;
219                
220                for (Integer slot : slotsAboveTreshold) {
221                        if (slot.doubleValue() < min) {
222                                min = slot;
223                        }
224                        if (slot.doubleValue() > max) {
225                                max = slot;
226                        }
227                }
228                
229                return getTimeOfOperationFromTimeSlots(min, max);
230        }
231
232        @Override
233        public String getStrategyName() {
234                return ReduceTimeServedRFare.STRATEGY_NAME;
235        }
236        
237        @Override
238        public void reset() {
239                this.route2StartTimeSlot2EndTimeSlot2WeightMap = new LinkedHashMap<>();
240        }
241
242        @Override
243        public void handleFareContainer(StageContainer stageContainer) {
244                Id<TransitRoute> routeId = stageContainer.getRouteId();
245                Integer startTimeSlot = this.getTimeSlotForTime(stageContainer.getTimeEntered());
246                Integer endTimeSlot = this.getTimeSlotForTime(stageContainer.getTimeLeft());
247                
248                if (this.route2StartTimeSlot2EndTimeSlot2WeightMap.get(routeId) == null) {
249                        this.route2StartTimeSlot2EndTimeSlot2WeightMap.put(routeId, new LinkedHashMap<Integer, LinkedHashMap<Integer,Double>>());
250                }
251        
252                if (this.route2StartTimeSlot2EndTimeSlot2WeightMap.get(routeId).get(startTimeSlot) == null) {
253                        this.route2StartTimeSlot2EndTimeSlot2WeightMap.get(routeId).put(startTimeSlot, new LinkedHashMap<Integer,Double>());
254                }
255        
256                if (this.route2StartTimeSlot2EndTimeSlot2WeightMap.get(routeId).get(startTimeSlot).get(endTimeSlot) == null) {
257                        this.route2StartTimeSlot2EndTimeSlot2WeightMap.get(routeId).get(startTimeSlot).put(endTimeSlot, (double) 0);
258                }
259        
260                double oldWeight = this.route2StartTimeSlot2EndTimeSlot2WeightMap.get(routeId).get(startTimeSlot).get(endTimeSlot);
261                double additionalWeight = 1.0;
262                if (this.useFareAsWeight) {
263                        additionalWeight = this.ticketMachine.getFare(stageContainer);
264                }
265                this.route2StartTimeSlot2EndTimeSlot2WeightMap.get(routeId).get(startTimeSlot).put(endTimeSlot, oldWeight + additionalWeight);
266        }
267
268        public void setTicketMachine(TicketMachineI ticketMachine) {
269                this.ticketMachine = ticketMachine;
270        }
271
272        private int getTimeSlotForTime(double time){
273                return ((int) time / this.timeBinSize);
274        }
275
276        private Tuple<Double, Double> getTimeOfOperationFromTimeSlots(int startSlot, int endSlot){
277                // convert slots to time
278                double startTime = ((double) startSlot) * this.timeBinSize;
279                double endTime = ((double) endSlot + 1) * this.timeBinSize;
280                
281                return new Tuple<>(startTime, endTime);
282        }
