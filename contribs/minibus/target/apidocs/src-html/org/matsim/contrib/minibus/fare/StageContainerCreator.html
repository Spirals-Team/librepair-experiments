020package org.matsim.contrib.minibus.fare;
021
022import org.apache.log4j.Logger;
023import org.matsim.api.core.v01.Id;
024import org.matsim.api.core.v01.events.LinkEnterEvent;
025import org.matsim.api.core.v01.events.PersonEntersVehicleEvent;
026import org.matsim.api.core.v01.events.PersonLeavesVehicleEvent;
027import org.matsim.api.core.v01.events.TransitDriverStartsEvent;
028import org.matsim.api.core.v01.events.handler.LinkEnterEventHandler;
029import org.matsim.api.core.v01.events.handler.PersonEntersVehicleEventHandler;
030import org.matsim.api.core.v01.events.handler.PersonLeavesVehicleEventHandler;
031import org.matsim.api.core.v01.events.handler.TransitDriverStartsEventHandler;
032import org.matsim.api.core.v01.network.Network;
033import org.matsim.api.core.v01.population.Person;
034import org.matsim.core.api.experimental.events.VehicleArrivesAtFacilityEvent;
035import org.matsim.core.api.experimental.events.handler.VehicleArrivesAtFacilityEventHandler;
036import org.matsim.core.controler.events.AfterMobsimEvent;
037import org.matsim.core.controler.listener.AfterMobsimListener;
038import org.matsim.vehicles.Vehicle;
039
040import java.util.HashMap;
041import java.util.LinkedList;
042import java.util.List;
043
044/**
045 * 
046 * Collects all information needed to calculate create a {@link StageContainer}
047 * and pushes them to all registered {@link StageContainerHandler}.
048 * 
049 * @author aneumann
050 *
051 */
052public final class StageContainerCreator implements TransitDriverStartsEventHandler, VehicleArrivesAtFacilityEventHandler, LinkEnterEventHandler, PersonEntersVehicleEventHandler, PersonLeavesVehicleEventHandler, AfterMobsimListener{
053        
054        private final static Logger log = Logger.getLogger(StageContainerCreator.class);
055        
056        private Network network;
057        private final String pIdentifier;
058        
059        private final List<StageContainerHandler> stageContainerHandlerList = new LinkedList<>();
060        private HashMap<Id<Vehicle>, TransitDriverStartsEvent> vehId2TransitDriverStartsE = new HashMap<>();
061        private HashMap<Id<Vehicle>, VehicleArrivesAtFacilityEvent> vehId2VehArrivesAtFacilityE = new HashMap<>();
062        private HashMap<Id<Vehicle>, LinkedList<StageContainer>> vehId2StageContainerListMap = new HashMap<>();
063        private HashMap<Id<Person>, StageContainer> personId2StageContainer = new HashMap<>();
064
065        public StageContainerCreator(String pIdentifier){
066                this.pIdentifier = pIdentifier;
067                log.info("enabled");
068        }
069        
070        public void init(Network network){
071                this.network = network;
072        }
073        
074        public void addStageContainerHandler(StageContainerHandler stageContainerHandler){
075                this.stageContainerHandlerList.add(stageContainerHandler);
076        }
077        
078        @Override
079        public void reset(int iteration) {
080                this.vehId2TransitDriverStartsE = new HashMap<>();
081                this.vehId2VehArrivesAtFacilityE = new HashMap<>();
082                this.vehId2StageContainerListMap = new HashMap<>();
083                this.personId2StageContainer = new HashMap<>();
084                
085                for (StageContainerHandler stageContainerHandler : this.stageContainerHandlerList) {
086                        stageContainerHandler.reset();
087                }
088        }
089        
090        @Override
091        public void handleEvent(TransitDriverStartsEvent event) {
092                this.vehId2TransitDriverStartsE.put(event.getVehicleId(), event);
093                if (this.vehId2StageContainerListMap.get(event.getVehicleId()) == null) {
094                        this.vehId2StageContainerListMap.put(event.getVehicleId(), new LinkedList<StageContainer>());
095                }
096        }
097
098        @Override
099        public void handleEvent(VehicleArrivesAtFacilityEvent event) {
100                this.vehId2VehArrivesAtFacilityE.put(event.getVehicleId(), event);
101        }
102
103        @Override
104        public void handleEvent(LinkEnterEvent event) {
105                if (this.vehId2TransitDriverStartsE.containsKey(event.getVehicleId())) {
106                        // It's a transit driver
107                        double linkLength = this.network.getLinks().get(event.getLinkId()).getLength();
108                        for (StageContainer stageContainer : this.vehId2StageContainerListMap.get(event.getVehicleId())) {
109                                stageContainer.addDistanceTravelled(linkLength);
110                        }
111                }
112        }
113
114        @Override
115        public void handleEvent(PersonEntersVehicleEvent event) {
116                if(event.getVehicleId().toString().startsWith(this.pIdentifier)){
117                        // it's a paratransit vehicle
118                        if(!event.getPersonId().toString().contains(this.pIdentifier)){
119                                // it's not the driver
120                                StageContainer stageContainer = new StageContainer();
121                                stageContainer.handlePersonEnters(event, this.vehId2VehArrivesAtFacilityE.get(event.getVehicleId()), this.vehId2TransitDriverStartsE.get(event.getVehicleId()));
122                                this.vehId2StageContainerListMap.get(event.getVehicleId()).add(stageContainer);
123                                this.personId2StageContainer.put(event.getPersonId(), stageContainer);
124                        }
125                }
126        }
127
128        @Override
129        public void handleEvent(PersonLeavesVehicleEvent event) {
130                if(event.getVehicleId().toString().startsWith(this.pIdentifier)){
131                        // it's a paratransit vehicle
132                        if(!event.getPersonId().toString().contains(this.pIdentifier)){
133                                // it's not the driver
134                                StageContainer stageContainer = this.personId2StageContainer.remove(event.getPersonId());
135                                this.vehId2StageContainerListMap.get(event.getVehicleId()).remove(stageContainer);
136                                stageContainer.handlePersonLeaves(event, this.vehId2VehArrivesAtFacilityE.get(event.getVehicleId()));
137                                
138                                // call all StageContainerHandler
139                                for (StageContainerHandler stageContainerHandler : this.stageContainerHandlerList) {
140                                        stageContainerHandler.handleFareContainer(stageContainer);
141                                }
142                                
143                                // Note the stageContainer is dropped at this point.
144                        }
145                }
146        }
147
148        @Override
149        public void notifyAfterMobsim(AfterMobsimEvent event) {
150                // ok, mobsim is done - finish incomplete entries
151                if (this.personId2StageContainer.size() > 0) {
152                        log.warn("There are " + this.personId2StageContainer.size() + " passengers with incomplete trips. Cannot finish them. Will not forward those entries");
153                }
154        }
155}
