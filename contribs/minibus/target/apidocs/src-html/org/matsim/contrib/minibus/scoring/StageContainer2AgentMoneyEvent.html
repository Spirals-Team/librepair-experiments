020package org.matsim.contrib.minibus.scoring;
021
022import java.util.HashMap;
023import java.util.LinkedList;
024import java.util.List;
025import java.util.Map.Entry;
026
027import org.matsim.api.core.v01.Id;
028import org.matsim.api.core.v01.events.PersonMoneyEvent;
029import org.matsim.api.core.v01.population.Person;
030import org.matsim.contrib.minibus.fare.StageContainer;
031import org.matsim.contrib.minibus.fare.StageContainerHandler;
032import org.matsim.contrib.minibus.fare.TicketMachineI;
033import org.matsim.core.api.experimental.events.EventsManager;
034import org.matsim.core.controler.MatsimServices;
035import org.matsim.core.controler.events.AfterMobsimEvent;
036import org.matsim.core.controler.listener.AfterMobsimListener;
037
038/**
039 * Collects {@link StageContainer} and creates {@link PersonMoneyEvent}.
040 * 
041 * @author aneumann
042 *
043 */
044public final class StageContainer2AgentMoneyEvent implements StageContainerHandler, AfterMobsimListener{
045
046        private final EventsManager eventsManager;
047        private final double mobsimShutdownTime;
048        private HashMap<Id<Person>, List<StageContainer>> agentId2stageContainerListMap = new HashMap<>();
049        private final TicketMachineI ticketMachine;
050
051        public StageContainer2AgentMoneyEvent(MatsimServices controler, TicketMachineI ticketMachine) {
052                controler.addControlerListener(this);
053                this.eventsManager = controler.getEvents();
054                this.mobsimShutdownTime = controler.getConfig().qsim().getEndTime();
055                this.ticketMachine = ticketMachine;
056        }
057
058        @Override
059        public void notifyAfterMobsim(AfterMobsimEvent event) {
060                // TODO[AN] This can be used to hook in a Farebox
061                for (Entry<Id<Person>, List<StageContainer>> agentId2stageContainersEntry : this.agentId2stageContainerListMap.entrySet()) {
062                        double totalFareOfAgent = 0.0;
063                        for (StageContainer stageContainer : agentId2stageContainersEntry.getValue()) {
064                                totalFareOfAgent += this.ticketMachine.getFare(stageContainer);
065                        }
066                        this.eventsManager.processEvent(new PersonMoneyEvent(this.mobsimShutdownTime, agentId2stageContainersEntry.getKey(), -totalFareOfAgent));
067                }
068        }
069
070        @Override
071        public void handleFareContainer(StageContainer stageContainer) {
072                if (this.agentId2stageContainerListMap.get(stageContainer.getAgentId()) == null) {
073                        this.agentId2stageContainerListMap.put(stageContainer.getAgentId(), new LinkedList<StageContainer>());
074                }
075                
076                this.agentId2stageContainerListMap.get(stageContainer.getAgentId()).add(stageContainer);
077        }
078
079        @Override
080        public void reset() {
081                this.agentId2stageContainerListMap = new HashMap<>();
082        }
083}
