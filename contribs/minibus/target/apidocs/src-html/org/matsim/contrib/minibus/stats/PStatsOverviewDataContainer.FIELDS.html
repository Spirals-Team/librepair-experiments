031 * 
032 * @author aneumann
033 *
034 */
035public class PStatsOverviewDataContainer {
036        
037        @SuppressWarnings("unused")
038        private static final Logger log = Logger.getLogger(PStatsOverviewDataContainer.class);
039        
040        final static String COMMENTTAG = "#";
041        final static String DELIMITER = "\t";
042        
043        public enum FIELDS {
044                iteration("Iteration [ ]", "Iteration [ ]", new DecimalFormat( "#########0", new DecimalFormatSymbols(Locale.US))),
045                nOperators("Operators [ ]", "Operators [ ]", new DecimalFormat( "#########0", new DecimalFormatSymbols(Locale.US))),
046                nOperatorsInBusiness("Operators IB [ ]", "Operators IB [ ]", new DecimalFormat( "#########0", new DecimalFormatSymbols(Locale.US))),
047                nRoutes("Routes [ ]", "Routes [ ]", new DecimalFormat( "#########0", new DecimalFormatSymbols(Locale.US))),
048                nRoutesOfInBusiness("Routes IB [ ]", "Routes IB [ ]", new DecimalFormat( "#########0", new DecimalFormatSymbols(Locale.US))),
049                nPax("Pax served [ ]", "Pax served [ ]", new DecimalFormat( "#########0", new DecimalFormatSymbols(Locale.US))),

054                avgBudgetPerOperator("Avg. budget per operator [$]", "Avg. budget per operator [$]", new DecimalFormat( "#########0.0000000000", new DecimalFormatSymbols(Locale.US))),

056                avgCashflowPerRoute("Avg. cash flow per route [$]", "Avg. cash flow per route [$]", new DecimalFormat( "#########0.0000000000", new DecimalFormatSymbols(Locale.US))),
057                avgCashflowPerRouteOfInBusiness("Avg. cash flow per route of IB [$]", "Avg. cash flow per route of IB [$]", new DecimalFormat( "#########0.0000000000", new DecimalFormatSymbols(Locale.US))),
058                
059                shareOfInBusinessOperators("Share of IB operators [%]", "Share of IB operators [%]", new DecimalFormat( "#########0.0000000000", new DecimalFormatSymbols(Locale.US))),
060                shareOfInBusinessRoutes("Share of IB routes [%]", "Share of IB routes [%]", new DecimalFormat( "#########0.0000000000", new DecimalFormatSymbols(Locale.US))),
061                shareOfPaxServedByInBusiness("Share of pax served by IB [%]", "Share of pax served by IB [%]", new DecimalFormat( "#########0.0000000000", new DecimalFormatSymbols(Locale.US))),
062                shareOfVehOfInBusiness("Share of veh served by IB [%]", "Share of veh served by IB [%]", new DecimalFormat( "#########0.0000000000", new DecimalFormatSymbols(Locale.US))),
063                
064                estimatedMeanOperatorsInBusiness("Estimated mean of IB operators [ ]", "Estimated mean of IB operators [ ]", new DecimalFormat( "#########0.0000000000", new DecimalFormatSymbols(Locale.US))),
065                estimatedSDOperatorsInBusiness("Estimated SD of IB operators [ ]", "Estimated SD of IB operators [ ]", new DecimalFormat( "#########0.0000000000", new DecimalFormatSymbols(Locale.US))),
066                estimatedMeanRouteOfInBusiness("Estimated mean of IB routes [ ]", "Estimated mean of IB routes [ ]", new DecimalFormat( "#########0.0000000000", new DecimalFormatSymbols(Locale.US))),

071                estimatedSDVehicleOfInBusiness("Estimated SD of IB veh [ ]", "Estimated SD of IB veh [ ]", new DecimalFormat( "#########0.0000000000", new DecimalFormatSymbols(Locale.US))),
072                
073                exactMeanOperatorsInBusiness("Exact mean of IB operators [ ]", "Exact mean of IB operators [ ]", new DecimalFormat( "#########0.0000000000", new DecimalFormatSymbols(Locale.US))),

076                exactSDRouteOfInBusiness("Exact SD of IB routes [ ]", "Exact SD of IB routes [ ]", new DecimalFormat( "#########0.0000000000", new DecimalFormatSymbols(Locale.US))),

081                
082                private String enName;
083                private String deName;
084                private DecimalFormat decimalFormat;
085
086                private FIELDS(String enName, String deName, DecimalFormat decimalFormat){
087                        this.enName = enName;
088                        this.deName = deName;
089                        this.decimalFormat = decimalFormat;
090                }
091                
092                public String getEnName(){
093                        return this.enName;
094                }
095                
096                public String getDeName(){
097                        return this.deName;
098                }
099                
100                public DecimalFormat getDecimalFormat(){
101                        return this.decimalFormat;
102                }
103        }
104        
105        public final static int nColumns = FIELDS.values().length;
106        
107        private double[] datafields = new double[PStatsOverviewDataContainer.nColumns];
108        
109        public void addData(int column, double data){
110                        datafields[column] = data;
111        }
112        
113        public double getData(int column){
114                return datafields[column];
115        }
116        
117        public static String getHeaderLine(){
118                StringBuffer strB = new StringBuffer();
119                boolean initialized = false;
120                for (FIELDS field : FIELDS.values()) {
121                        if (!initialized) {
122                                strB.append(COMMENTTAG + " ");
123                                initialized = true;
124                        } else {
125                                strB.append(DELIMITER);
126                        }
127                        strB.append(field.getEnName());
128                }
129                return strB.toString();
130        }
131}
