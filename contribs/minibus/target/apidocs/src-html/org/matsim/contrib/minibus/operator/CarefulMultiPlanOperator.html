020package org.matsim.contrib.minibus.operator;
021
022import org.matsim.api.core.v01.Id;
023import org.matsim.contrib.minibus.PConfigGroup;
024import org.matsim.contrib.minibus.replanning.PStrategy;
025import org.matsim.contrib.minibus.replanning.PStrategyManager;
026import org.matsim.contrib.minibus.routeProvider.PRouteProvider;
027import org.matsim.core.gbl.MatsimRandom;
028
029import java.util.Collections;
030import java.util.LinkedList;
031import java.util.List;
032
033/**
034 * This operator has multiple plans. Each is weighted by the number of vehicles associated with.
035 * The number of vehicles depends on the score per vehicle and plan. In the end, each plan should have approximately the same score.
036 * In contrast to {@link MultiPlanOperator}, this operator accepts all plans as blueprints for replanning, again picking one is a weighted random draw.
037 * Vehicle are shifted away from plans instantly, i.e. if a plan scored negative at least one vehicle is removed.
038 * Vehicles are distributed to positive plans and new plans.
039 * 
040 * @author aneumann
041 *
042 */
043public final class CarefulMultiPlanOperator extends AbstractOperator{
044        
045        public static final String OPERATOR_NAME = "CarefulMultiPlanOperator";
046
047    private List<PPlan> plans;
048
049        public CarefulMultiPlanOperator(Id<Operator> id, PConfigGroup pConfig, PFranchise franchise){
050                super(id, pConfig, franchise);
051                this.plans = new LinkedList<>();
052        }
053        
054        public boolean init(PRouteProvider pRouteProvider, PStrategy initialStrategy, int iteration, double initialBudget) {
055                boolean initializedSuccessfully = super.init(pRouteProvider, initialStrategy, iteration, initialBudget);
056                if (initializedSuccessfully) {
057                        this.buyAsManyVehiclesAsPossible();
058                        this.plans.add(this.bestPlan);
059                        this.bestPlan.setNVehicles(this.bestPlan.getNVehicles() + this.numberOfVehiclesInReserve);
060                        this.numberOfVehiclesInReserve = 0;
061                        this.bestPlan = null;
062                }
063                return initializedSuccessfully;
064        }
065        
066        @Override
067        public List<PPlan> getAllPlans(){
068                return this.plans;              
069        }
070        
071        @Override
072        public PPlan getBestPlan() {
073                if (this.bestPlan == null) {
074                        
075                        // will not return the best plan, but one random plan selected from all plans weighted by the number of vehicles of each plan
076                        int numberOfVehicles = 0;
077                        for (PPlan pPlan : this.plans) {
078                                numberOfVehicles += pPlan.getNVehicles();
079                        }
080
081                        double accumulatedWeight = 0.0;
082                        double rndTreshold = MatsimRandom.getRandom().nextDouble() * numberOfVehicles;
083                        for (PPlan pPlan : this.plans) {
084                                accumulatedWeight += pPlan.getNVehicles();
085                                if (rndTreshold <= accumulatedWeight) {
086                                        this.bestPlan = pPlan;
087                                        return this.bestPlan;
088                                }
089                        }
090                }
091                
092                return this.bestPlan;
093        }
094
095        public void replan(PStrategyManager pStrategyManager, int iteration) {
096                this.currentIteration = iteration;
097                
098                // First remove vehicles from all plans scored negative and add them to the reserve
099                this.numberOfVehiclesInReserve += this.removeVehiclesFromAllPlansWithNegativeScore(this.plans);
100                
101                // Second, balance the budget
102                if(this.budget < 0){
103                        // insufficient, try to balance from reserve vehicles
104                        int numberOfVehiclesToSell = -1 * Math.min(-1, (int) Math.floor(this.budget / this.costPerVehicleSell));
105                        
106                        while (this.numberOfVehiclesInReserve > 0 && numberOfVehiclesToSell > 0) {
107                                // sell one vehicle from the reserve
108                                this.numberOfVehiclesInReserve--;
109                                numberOfVehiclesToSell--;
110                                this.budget += this.costPerVehicleSell;
111                        }
112                        
113                        while (numberOfVehiclesToSell > 0) {
114                                this.findWorstPlanAndRemoveOneVehicle();
115                                this.budget += this.costPerVehicleSell;
116                                numberOfVehiclesToSell--;
117                        }
118                }
119
120                // Third, buy vehicles
121                buyAsManyVehiclesAsPossible();
122                
123                // Fourth, increase the number of vehicles by one for all positive plans
124                addVehiclesToPlansWithAPositiveScore();
125                
126                // Fifth, add new plans for all vehicles in reserve, but half the number of vehicles already in service as max
127                int maxVehiclesToDistribute = 0;
128                for (PPlan plan : this.plans) {
129                        maxVehiclesToDistribute += plan.getNVehicles();
130                }
131                maxVehiclesToDistribute = Math.max(1, maxVehiclesToDistribute / 2);
132
133        int tryouts = 10;
134        int remainingNumberOfTryouts = tryouts;
135                
136                while (this.numberOfVehiclesInReserve > 0 && maxVehiclesToDistribute > 0 && remainingNumberOfTryouts > 0) {
137                        
138                        remainingNumberOfTryouts--;
139                        
140                        // find a random plan
141                        this.getBestPlan();
142                        
143                        PStrategy strategy = pStrategyManager.chooseStrategy();
144                        if (strategy != null) {
145                                PPlan newPlan = strategy.run(this);
146                                if (newPlan != null) {
147                                        if(this.getFranchise().planRejected(newPlan)){
148                                                // plan is rejected by franchise system
149                                                newPlan = null;
150                                        }       
151                                        
152                                        if (newPlan != null) {
153                                                // get one vehicle from the reserve
154                                                this.numberOfVehiclesInReserve--;
155                                                maxVehiclesToDistribute--;
156                                                this.plans.add(newPlan);
157                                        }
158                                }
159                                
160                                this.bestPlan = null;
161                        }
162                }
163                
164                removeAllPlansWithZeroVehicles();
165                
166                // Fifth, reinitialize all plans
167                for (PPlan plan : this.plans) {
168                        plan.setLine(this.routeProvider.createTransitLineFromOperatorPlan(this.id, plan));
169                }
170                
171                this.updateCurrentTransitLine();
172        }
173
174        private void addVehiclesToPlansWithAPositiveScore() {
175                List<PPlan> plansWithAPositiveScore = new LinkedList<>();
176                for (PPlan plan : this.plans) {
177                        if (plan.getScore() > 0) {
178                                // positive
179                                plansWithAPositiveScore.add(plan);
180                        }
181                }
182                
183                Collections.sort(plansWithAPositiveScore);
184                
185                for (PPlan plan : plansWithAPositiveScore) {
186                        if (this.numberOfVehiclesInReserve > 0) {
187                                // increase by one vehicle
188                                plan.setNVehicles(plan.getNVehicles() + 1);
189                                this.numberOfVehiclesInReserve--;
190                        }
191                }
192        }
193
194        private void removeAllPlansWithZeroVehicles() {
195                // remove all plans with no vehicles
196                List<PPlan> plansToKeep = new LinkedList<>();
197                for (PPlan plan : this.plans) {
198                        if (plan.getNVehicles() > 0) {
199                                plansToKeep.add(plan);
200                        }
201                }
202                if (plansToKeep.size() == 0) {
203                        log.info("All plans have no vehicles. This may happen, if replanning fails to return a valid new plan AND all remaining plans scored negative while having only one vehicle.");
204                        log.info(this.plans.toString());
205                        for (PPlan pPlan : this.plans) {
206                                if (this.numberOfVehiclesInReserve > 0) {
207                                        log.info("Anyway, adding one vehicle to plan " + pPlan);
208                                        pPlan.setNVehicles(pPlan.getNVehicles() + 1);
209                                        this.numberOfVehiclesInReserve--;
210                                }
211                        }
212                        // Rerun removing plans - not all plans may got a vehicle
213                        this.removeAllPlansWithZeroVehicles();
214                } else {
215                        this.plans = plansToKeep;
216                }
217        }
218
219        private void buyAsManyVehiclesAsPossible() {
220                while (this.getBudget() > this.getCostPerVehicleBuy()) {
221                        // budget ok, buy one
222                        this.setBudget(this.getBudget() - this.getCostPerVehicleBuy());
223                        this.numberOfVehiclesInReserve++;
224                }
225        }
226        
227        /**
228         * 
229         * @param plansToCheck
230         * @return The number of vehicles removed
231         */
232        private int removeVehiclesFromAllPlansWithNegativeScore(List<PPlan> plansToCheck) {
233                int numberOfRemovedVehicles = 0;
234                
235                for (PPlan pPlan : plansToCheck) {
236                        if (pPlan.getScore() < 0.0) {
237                                // okay plan scored negative - tackle it
238                                double score = Math.abs(pPlan.getScore());
239                                // remove as many vehicles as necessary to compensate the impact on the budget plus one to hopefully get that plan positively scored in the next iteration
240                                int vehiclesToRemove = (int) (score / this.costPerVehicleSell) + 1;
241                                if(pPlan.getNVehicles() < vehiclesToRemove) {
242                                        // this plan cannot compensate - remove all vehicles;
243                                        vehiclesToRemove = pPlan.getNVehicles();
244                                }
245                                
246                                pPlan.setNVehicles(pPlan.getNVehicles() - vehiclesToRemove);
247                                numberOfRemovedVehicles += vehiclesToRemove;
248                        }
249                }
250                return numberOfRemovedVehicles;
251        }
252
253                
254        /**
255         * Find plan with the worst score per vehicle. Removes one vehicle. Removes the whole plan, if no vehicle is left.
256         * 
257         * @return
258         */
259        private void findWorstPlanAndRemoveOneVehicle(){
260                PPlan worstPlan = null;
261                for (PPlan plan : this.plans) {
262                        if (plan.getNVehicles() > 0) {
263                                if (worstPlan == null) {
264                                        worstPlan = plan;
265                                } else {
266                                        if (plan.getScorePerVehicle() < worstPlan.getScorePerVehicle()) {
267                                                worstPlan = plan;
268                                        }
269                                }
270                        }
271                }
272                
273                // remove one vehicle
274                worstPlan.setNVehicles(worstPlan.getNVehicles() - 1);
275                // remove plan, if not served anymore
276                if (worstPlan.getNVehicles() == 0) {
277                        this.plans.remove(worstPlan);
278                }
279        }
