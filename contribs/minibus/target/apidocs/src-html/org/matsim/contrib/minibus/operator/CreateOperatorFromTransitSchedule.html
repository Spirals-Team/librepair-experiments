037/**
038 * 
039 * Creates operators from a given transit schedule
040 * 
041 * @author aneumann
042 *
043 */
044public final class CreateOperatorFromTransitSchedule implements PStrategy {
045        
046        private final static Logger log = Logger.getLogger(CreateOperatorFromTransitSchedule.class);
047        private static final String STRATEGY_NAME = "CreateOperatorFromSchedule";
048
049        private final OperatorFactory operatorFactory;
050        private final PRouteProvider routeProvider;
051        private final PConfigGroup pConfig;
052        private final LinkedHashMap<String, TransitStopFacility> originalStops;
053        
054        private LinkedHashMap<Id<Operator>, PPlan> operatorId2PlanMap = new LinkedHashMap<>();
055
056        public CreateOperatorFromTransitSchedule(OperatorFactory operatorFactory, PRouteProvider routeProvider, PConfigGroup pConfig, TransitSchedule originalSchedule) {
057                this.operatorFactory = operatorFactory;
058                this.routeProvider = routeProvider;
059                this.pConfig = pConfig;
060                
061                this.originalStops = new LinkedHashMap<>();
062                for (TransitStopFacility stop : originalSchedule.getFacilities().values()) {
063                        this.originalStops.put(stop.getId().toString(), stop);
064                }
065        }
066
067        public LinkedList<Operator> run() {
068                LinkedList<Operator> operatorsToReturn = new LinkedList<>();
069                
070                if (pConfig.getTransitScheduleToStartWith() != null) {
071                        TransitSchedule tS = readTransitSchedule(pConfig.getTransitScheduleToStartWith());
072
073                        for (Entry<Id<TransitLine>, TransitLine> lineEntry : tS.getTransitLines().entrySet()) {
074                                Operator operator = this.operatorFactory.createNewOperator(Id.create(this.pConfig.getPIdentifier() + lineEntry.getKey(), Operator.class));
075
076                                PPlan plan = createPlan(lineEntry.getValue());
077                                this.operatorId2PlanMap.put(operator.getId(), plan);                    
078
079                                double initialBudget = this.pConfig.getInitialBudget() % pConfig.getPricePerVehicleBought();                    
080                                operator.init(this.routeProvider, this, 0, initialBudget);
081                                operatorsToReturn.add(operator);
082                        }
083                }
084                
085                this.operatorId2PlanMap = null;
086                log.info("Returning " + operatorsToReturn.size() + " operators created from transit schedule file " + pConfig.getTransitScheduleToStartWith());
087                return operatorsToReturn;
088        }
089
090        private PPlan createPlan(TransitLine line) {
091                Id<PPlan> id = Id.create(0, PPlan.class);
092                TransitRoute longestRouteH = null;
093                TransitRoute longestRouteR = null;
094                
095                double startTime = Double.MAX_VALUE;
096                double endTime = Double.MIN_VALUE;
097                Set<Id<Vehicle>> vehicleIds = new TreeSet<>();
098                
099                for (TransitRoute route : line.getRoutes().values()) {
100                        if(route.getId().toString().endsWith(".H")){
101                                if (longestRouteH == null) {
102                                        longestRouteH = route;
103                                } else if (longestRouteH.getStops().size() < route.getStops().size()) {
104                                        longestRouteH = route;
105                                }                               
106                        } else if (route.getId().toString().endsWith(".R")) {
107                                if (longestRouteR == null) {
108                                        longestRouteR = route;
109                                } else if (longestRouteR.getStops().size() < route.getStops().size()) {
110                                        longestRouteR = route;
111                                }
112                        }                       
113                        
114                        for (Departure departure : route.getDepartures().values()) {
115                                if (startTime > departure.getDepartureTime()) {
116                                        startTime = departure.getDepartureTime();
117                                }
118                                
119                                if (endTime < departure.getDepartureTime()) {
120                                        endTime = departure.getDepartureTime();
121                                }
122                                
123                                vehicleIds.add(departure.getVehicleId());
124                        }
125                }
126                
127                // limit parameters to reasonable ones          
128                startTime = Math.max(startTime, 0.0);
129                endTime = Math.min(endTime, 24 * 3600.0);
130                
131                // current planned headway in case number of vehicles is not changed
132                double departureOffset = 0.0;
133                if (longestRouteH != null) {
134                        departureOffset += longestRouteH.getStops().get(longestRouteH.getStops().size() - 1).getDepartureOffset();
135                }
136                
137                if (longestRouteR != null) {
138                        departureOffset += longestRouteR.getStops().get(longestRouteR.getStops().size() - 1).getDepartureOffset();
139                }
140                
141                double headway = departureOffset / vehicleIds.size(); 
142                // set headway to a minimum of 3min
143                headway = Math.max(3*60, headway);
144                // resulting number of vehicles with the new headway
145                int nVehicles = (int) (departureOffset / headway);
146                
147                ArrayList<TransitStopFacility> stopsToBeServed = new ArrayList<>();
148                if (longestRouteH != null) {
149                        for (TransitRouteStop routeStop : longestRouteH.getStops()) {
150                                if (this.checkStopInServiceArea(routeStop.getStopFacility(), this.pConfig)) {
151                                        TransitStopFacility originalStopFacility = this.originalStops.get(routeStop.getStopFacility().getId().toString());
152                                        stopsToBeServed.add(originalStopFacility);
153                                }
154                        }
155                }
156                if (longestRouteR != null) {
157                        for (TransitRouteStop routeStop : longestRouteR.getStops()) {
158                                if (this.checkStopInServiceArea(routeStop.getStopFacility(), this.pConfig)) {
159                                        TransitStopFacility originalStopFacility = this.originalStops.get(routeStop.getStopFacility().getId().toString());
160                                        stopsToBeServed.add(originalStopFacility);
161                                }
162                        }
163                }
164                
165                PPlan plan = new PPlan(id, this.getStrategyName(), PConstants.founderPlanId);
166                plan.setStopsToBeServed(stopsToBeServed);
167                plan.setStartTime(startTime);
168                plan.setEndTime(endTime);
169                plan.setNVehicles(nVehicles);
170                return plan;
171        }
172
173
174        private TransitSchedule readTransitSchedule(String transitScheduleToStartWith) {
175                MutableScenario sc = (MutableScenario) ScenarioUtils.createScenario(ConfigUtils.createConfig());
176                sc.getConfig().transit().setUseTransit(true);
177                log.info("Reading " + transitScheduleToStartWith);
178                new TransitScheduleReader(sc).readFile(transitScheduleToStartWith);
179                return sc.getTransitSchedule();
180        }
181
182        private boolean checkStopInServiceArea(TransitStopFacility stop, PConfigGroup pC){
183                boolean isInServiceArea = true;
184                
185                if(stop.getCoord().getX() < pC.getMinX()){isInServiceArea = false;}
186                if(stop.getCoord().getX() > pC.getMaxX()){isInServiceArea = false;}
187                
188                if(stop.getCoord().getY() < pC.getMinY()){isInServiceArea = false;}
189                if(stop.getCoord().getY() > pC.getMaxY()){isInServiceArea = false;}
190                
191                return isInServiceArea;
192        }
193
194
195        @Override
196        public PPlan run(Operator operator) {
197                PPlan newPlan = this.operatorId2PlanMap.get(operator.getId());          
198                newPlan.setLine(operator.getRouteProvider().createTransitLineFromOperatorPlan(operator.getId(), newPlan));
199                return newPlan;
200        }
201
202        @Override
203        public String getStrategyName() {
204                return CreateOperatorFromTransitSchedule.STRATEGY_NAME;
205        }
206}
