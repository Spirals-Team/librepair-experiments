020package org.matsim.contrib.minibus.genericUtils;
021
022import org.apache.log4j.Logger;
023import org.matsim.api.core.v01.Coord;
024
025import java.util.HashMap;
026import java.util.Map.Entry;
027
028/**
029 * Stores Type-Count pairs clustered to one node. Optional: shifts the node's centroid to the average of all given data points.
030 * 
031 * @author aneumann
032 *
033 */
034public final class GridNode {
035        
036        @SuppressWarnings("unused")
037        private static final Logger log = Logger.getLogger(GridNode.class);
038        private final String id;
039        private boolean isShifting = true;
040        private double xMean;
041        private double yMean;
042        private double nEntries = 0;
043        
044        private final HashMap<String, Integer> type2countMap = new HashMap<>();
045
046        /**
047         * Creates a unique id from two different slot numbers
048         * 
049         * @param x - number of xSlot
050         * @param y - number of ySlot
051         * @return Unique combination of x and y
052         */
053        private static String createGridNodeId(int x, int y){
054                return (x + "_" + y);
055        }
056
057        /**
058         * Calculates the slot number for a given one-dimensional coordinate 
059         * 
060         * @param coord - the coordinate, e.g. x-coordinate
061         * @param gridSize - distance between two slots
062         * @return the slot number corresponding to the given coordinate
063         */
064        private static int getSlotForCoord(double coord, double gridSize){
065                return (int) (coord / gridSize);
066        }
067
068        /**
069         * Returns the unique node id for a given x,y-coodinate and the gridsize
070         * 
071         * @param coord - the coordinate for which the node is searched for
072         * @param gridSize - distance between two nodes/slots
073         * @return The unique id of the corresponding node
074         */
075        public static String getGridNodeIdForCoord(Coord coord, double gridSize){
076                int xSlot = GridNode.getSlotForCoord(coord.getX(), gridSize);
077                int ySlot = GridNode.getSlotForCoord(coord.getY(), gridSize);
078                
079                return GridNode.createGridNodeId(xSlot, ySlot);
080        }
081
082        /**
083         * Creates a node with shifting centroid
084         * 
085         * @param id
086         */
087        public GridNode(String id){
088                this.id = id;
089        }
090        
091        /**
092         * Creates a node with fixed coord
093         * 
094         * @param id
095         * @param coord
096         */
097    private GridNode(String id, Coord coord){
098                this(id);
099                this.isShifting = false;
100                this.xMean = coord.getX();
101                this.yMean = coord.getY();
102        }
103        
104        public void addPoint(String type, Coord coord){
105                if(isShifting){
106                        // register coord
107                        if (nEntries == 0) {
108                                xMean = coord.getX();
109                                yMean = coord.getY();                   
110                        } else {
111                                xMean =  (this.nEntries * this.xMean + coord.getX()) / (this.nEntries + 1);
112                                yMean =  (this.nEntries * this.yMean + coord.getY()) / (this.nEntries + 1);
113                        }
114                }
115                nEntries++;
116                
117                // register actType
118                if (this.type2countMap.get(type) == null) {
119                        this.type2countMap.put(type, 0);
120                }
121                
122                this.type2countMap.put(type, this.type2countMap.get(type) + 1);
123        }
124
125        public String getId() {
126                return id;
127        }
128        
129        public double getX() {
130                return xMean;
131        }
132
133        public double getY() {
134                return yMean;
135        }
136        
137        public int getCountForType(String type) {
138                if (this.type2countMap.get(type) == null) {
139                        return 0;
140                } else {
141                        return this.type2countMap.get(type);
142                }
143        }
144
145        @Override
146        public String toString() {
147                StringBuffer strB = new StringBuffer();
148                strB.append("Type Count");
149                for (Entry<String, Integer> typeEntry : this.type2countMap.entrySet()) {
150                        strB.append(" | " + typeEntry.getKey() + " " + typeEntry.getValue().toString());
151                }
152                return strB.toString();
153        }
154}
