020package org.matsim.contrib.minibus.stats.operatorLogger;
021
022import java.util.ArrayList;
023import java.util.Arrays;
024
025import org.apache.log4j.Logger;
026import org.matsim.api.core.v01.Id;
027import org.matsim.api.core.v01.network.Link;
028import org.matsim.contrib.minibus.PConstants;
029import org.matsim.contrib.minibus.operator.Operator;
030import org.matsim.contrib.minibus.operator.PPlan;
031import org.matsim.core.utils.io.tabularFileParser.TabularFileHandler;
032import org.matsim.core.utils.io.tabularFileParser.TabularFileParser;
033import org.matsim.core.utils.io.tabularFileParser.TabularFileParserConfig;
034import org.matsim.core.utils.misc.Time;
035import org.matsim.pt.transitSchedule.api.TransitStopFacility;
036
037/**
038 * Reads a pLogger file omitting the header and all totals.
039 * 
040 * @author aneumann
041 *
042 */
043public final class LogReader implements TabularFileHandler {
044        
045        private static final Logger log = Logger.getLogger(LogReader.class);
046
047        private final TabularFileParserConfig tabFileParserConfig;
048        private final ArrayList<LogElement> processedLines = new ArrayList<>();
049        private final LogElementSinkImpl sink = new LogElementSinkImpl();
050        private int linesRejected = 0;
051
052        static interface LogElementSink {
053                void process(LogElement logElement);
054        }
055
056        class LogElementSinkImpl implements LogElementSink {
057                @Override
058                public void process(LogElement logElement) {
059                        processedLines.add(logElement);
060                }
061        }
062
063        private LogReader(String filename) {
064                tabFileParserConfig = new TabularFileParserConfig();
065                tabFileParserConfig.setFileName(filename);
066                tabFileParserConfig.setDelimiterTags(new String[] {LogElement.DELIMITER});
067        }
068
069        public static ArrayList<LogElement> readFile(String filename){
070                LogReader reader = new LogReader(filename);
071                log.info("Start parsing " + filename);
072                reader.parse();
073                log.info("Finished parsing " + filename);
074                log.info("Rejected " + reader.linesRejected + " lines");
075                log.info("Imported " + reader.processedLines.size() + " lines");
076                return reader.processedLines;           
077        }       
078
079        private void parse() {
080        new TabularFileParser().parse(tabFileParserConfig, this);
081    }
082        
083        @Override
084        public void startRow(String[] row) {
085                if(!row[0].trim().startsWith(LogElement.COMMENTTAG) || !row[0].trim().startsWith("iter")){
086                        if(!row[3].trim().startsWith(LogElement.NOVALUE) || !row[3].trim().startsWith("===")){
087                                try {
088                                        LogElement logElement = new LogElement();
089                                        
090                                        logElement.setIteration(Integer.parseInt(row[0]));
091                                        logElement.setOperatorId(Id.create(row[1], Operator.class));
092                                        
093                                        String state = row[2];
094                                        if (state.equalsIgnoreCase(PConstants.OperatorState.PROSPECTING.name())) {
095                                                logElement.setStatus(PConstants.OperatorState.PROSPECTING);
096                                        } else if (state.equalsIgnoreCase(PConstants.OperatorState.INBUSINESS.name())) {
097                                                logElement.setStatus(PConstants.OperatorState.INBUSINESS);
098                                        } else if (state.equalsIgnoreCase(PConstants.OperatorState.BANKRUPT.name())) {
099                                                logElement.setStatus(PConstants.OperatorState.BANKRUPT);
100                                        } else {
101                                                log.error("Couldn't determine the operator state " + state);
102                                        }
103                                        
104                                        logElement.setPlanId(Id.create(row[3], PPlan.class));
105                                        logElement.setCreatorId(row[4]);
106                                        logElement.setParentId(Id.create(row[5], PPlan.class));
107                                        logElement.setnVeh(Integer.parseInt(row[6]));
108                                        logElement.setnPax(Integer.parseInt(row[7]));
109                                        logElement.setScore(Double.parseDouble(row[8]));
110                                        logElement.setBudget(Double.parseDouble(row[9]));
111                                        logElement.setStartTime(Time.parseTime(row[10]));
112                                        logElement.setEndTime(Time.parseTime(row[11]));
113                                        
114                                        String stops = row[12];
115                                        stops = stops.substring(1, stops.length() - 1); // remove brackets
116                                        String[] stopArray = stops.split(","); // split the array
117                                        ArrayList<Id<TransitStopFacility>> stopIds = new ArrayList<Id<TransitStopFacility>>();
118                                        for (String stop : stopArray) {
119                                                stopIds.add(Id.create(stop.trim(), TransitStopFacility.class));
120                                        }
121                                        logElement.setStopsToBeServed(stopIds);
122                                        
123                                        String links = row[13];
124                                        links = links.substring(1, links.length() - 1); // remove brackets
125                                        String[] linkArray = links.split(","); // split the array
126                                        ArrayList<Id<Link>> linkIds = new ArrayList<Id<Link>>();
127                                        for (String link : linkArray) {
128                                                linkIds.add(Id.create(link.trim(), Link.class));
129                                        }
130                                        logElement.setLinksServed(linkIds);
131                                        
132                                        sink.process(logElement);
133                                        
134                                } catch (NumberFormatException e) {
135                                        this.linesRejected++;
136                                        log.info("Ignoring line : " + Arrays.asList(row));
137                                }
138
139                        } else {
140                                StringBuffer tempBuffer = new StringBuffer();
141                                for (String string : row) {
142                                        tempBuffer.append(string);
143                                        tempBuffer.append(", ");
144                                }
145                                this.linesRejected++;
146                                log.info("Ignoring: " + tempBuffer);
147                        }
148                }
149        }
150
151}
