020package org.matsim.contrib.minibus.stats.operatorLogger;
021
022import java.util.ArrayList;
023
024import org.apache.log4j.Logger;
025import org.matsim.api.core.v01.Id;
026import org.matsim.api.core.v01.network.Link;
027import org.matsim.contrib.minibus.PConstants.OperatorState;
028import org.matsim.contrib.minibus.operator.Operator;
029import org.matsim.contrib.minibus.operator.PPlan;
030import org.matsim.core.utils.collections.Tuple;
031import org.matsim.pt.transitSchedule.api.TransitStopFacility;
032
033public final class PlanElement {
034
035        @SuppressWarnings("unused")
036        private static final Logger log = Logger.getLogger(PlanElement.class);
037        
038        private final int iterationFounded;
039        private final Id<Operator> operatorId;
040        private final ArrayList<Tuple<Integer, OperatorState>> status;
041        private final Id<PPlan> planId;
042        private final String creatorId;
043        private final Id<PPlan> parentId;
044        private final ArrayList<Tuple<Integer, Integer>> nVeh;
045        private final ArrayList<Tuple<Integer, Integer>> nPax;
046        private final ArrayList<Tuple<Integer, Double>> score;
047        private final ArrayList<Tuple<Integer, Double>> budget;
048        private final double startTime;
049        private final double endTime;
050        private final ArrayList<Id<TransitStopFacility>> stopsToBeServed;
051        private final ArrayList<Id<Link>> linksServed;
052        
053        private PlanElement parentPlan;
054        
055        private int iterationCeased = Integer.MAX_VALUE;
056
057
058        public PlanElement(LogElement logElement) {
059                this.iterationFounded = logElement.getIteration();
060                this.operatorId = logElement.getOperatorId();
061                
062                this.status = new ArrayList<>();
063                this.status.add(new Tuple<>(this.iterationFounded, logElement.getStatus()));
064                
065                this.planId = logElement.getPlanId();
066                this.creatorId = logElement.getCreatorId();
067                this.parentId = logElement.getParentId();
068                
069                this.nVeh = new ArrayList<>();
070                this.nVeh.add(new Tuple<>(this.iterationFounded, logElement.getnVeh()));
071                
072                this.nPax = new ArrayList<>();
073                this.nPax.add(new Tuple<>(this.iterationFounded, logElement.getnPax()));
074                
075                this.score = new ArrayList<>();
076                this.score.add(new Tuple<>(this.iterationFounded, logElement.getScore()));
077                
078                this.budget = new ArrayList<>();
079                this.budget.add(new Tuple<>(this.iterationFounded, logElement.getBudget()));
080                
081                this.startTime = logElement.getStartTime();
082                this.endTime = logElement.getEndTime();
083                this.stopsToBeServed = logElement.getStopsToBeServed();
084                this.linksServed = logElement.getLinksServed();
085        }
086        
087        public String getUniquePlanIdentifier() {
088                return this.operatorId + "_" + this.planId;
089        }
090        
091        public int getIterationFounded() {
092                return iterationFounded;
093        }
094
095        public Id<Operator> getOperatorId() {
096                return operatorId;
097        }
098
099        public ArrayList<Tuple<Integer, OperatorState>> getStatus() {
100                return status;
101        }
102
103        public Id<PPlan> getPlanId() {
104                return planId;
105        }
106
107        public String getCreatorId() {
108                return creatorId;
109        }
110        
111        public Id<PPlan> getParentId() {
112                return parentId;
113        }
114
115        public ArrayList<Tuple<Integer, Integer>> getnVeh() {
116                return nVeh;
117        }
118
119        public ArrayList<Tuple<Integer, Integer>> getnPax() {
120                return nPax;
121        }
122
123        public ArrayList<Tuple<Integer, Double>> getScore() {
124                return score;
125        }
126
127        public ArrayList<Tuple<Integer, Double>> getBudget() {
128                return budget;
129        }
130
131        public double getStartTime() {
132                return startTime;
133        }
134
135        public double getEndTime() {
136                return endTime;
137        }
138
139        public ArrayList<Id<TransitStopFacility>> getStopsToBeServed() {
140                return stopsToBeServed;
141        }
142        
143        public ArrayList<Id<Link>> getLinksServed() {
144                return linksServed;
145        }
146        
147        public int getIterationCeased() {
148                return this.iterationCeased;
149        }
150        
151        public void setIterationCeased(int iterationCeased) {
152                this.iterationCeased = iterationCeased;
153        }
154
155        public PlanElement getParentPlan() {
156                return this.parentPlan;
157        }
158        
159        public void setParentPlan(PlanElement parent) {
160                this.parentPlan = parent;
161        }
162        
163        public void update(LogElement logElement) {
164                this.status.add(new Tuple<>(logElement.getIteration(), logElement.getStatus()));
165                this.nVeh.add(new Tuple<>(logElement.getIteration(), logElement.getnVeh()));
166                this.nPax.add(new Tuple<>(logElement.getIteration(), logElement.getnPax()));
167                this.score.add(new Tuple<>(logElement.getIteration(), logElement.getScore()));
168                this.budget.add(new Tuple<>(logElement.getIteration(), logElement.getBudget()));
169        }
170
171}
