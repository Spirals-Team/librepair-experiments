020package org.matsim.contrib.minibus.replanning;
021
022import org.apache.log4j.Logger;
023import org.matsim.contrib.minibus.PConfigGroup;
024import org.matsim.contrib.minibus.PConfigGroup.PStrategySettings;
025import org.matsim.contrib.minibus.fare.StageContainerCreator;
026import org.matsim.contrib.minibus.fare.TicketMachineI;
027import org.matsim.contrib.minibus.operator.TimeProvider;
028import org.matsim.core.gbl.MatsimRandom;
029
030import java.util.ArrayList;
031
032/**
033 * Loads strategies from config and chooses strategies according to their weights.
034 * 
035 * @author aneumann
036 *
037 */
038public final class PStrategyManager {
039        
040        private final static Logger log = Logger.getLogger(PStrategyManager.class);
041        
042        private final ArrayList<PStrategy> strategies = new ArrayList<>();
043        private final ArrayList<Double> weights = new ArrayList<>();
044        private final ArrayList<Integer> disableInIteration = new ArrayList<>();
045        private double totalWeights = 0.0;
046        private boolean allStrategiesAreDisabled = false;
047
048        public void init(PConfigGroup pConfig, StageContainerCreator stageContainerCreator, TicketMachineI ticketMachine, TimeProvider timeProvider) {
049                for (PStrategySettings settings : pConfig.getStrategySettings()) {
050                        String classname = settings.getModuleName();
051                        double rate = settings.getProbability();
052                        if (rate == 0.0) {
053                                log.info("The following strategy has a weight set to zero. Will drop it. " + classname);
054                                continue;
055                        }
056                        PStrategy strategy = loadStrategy(classname, settings, stageContainerCreator, ticketMachine, timeProvider);
057                        this.addStrategy(strategy, rate, settings.getDisableInIteration());
058                }
059                
060                log.info("enabled with " + this.strategies.size()  + " strategies");
061        }
062
063        private PStrategy loadStrategy(final String name, final PStrategySettings settings, StageContainerCreator stageContainerCreator, TicketMachineI ticketMachine, TimeProvider timeProvider) {
064                PStrategy strategy = null;
065                
066                if (name.equals(MaxRandomStartTimeAllocator.STRATEGY_NAME)) {
067                        strategy = new MaxRandomStartTimeAllocator(settings.getParametersAsArrayList());
068                } else if (name.equals(MaxRandomEndTimeAllocator.STRATEGY_NAME)) {
069                        strategy = new MaxRandomEndTimeAllocator(settings.getParametersAsArrayList());
070                } else if(name.equals(SidewaysRouteExtension.STRATEGY_NAME)){
071                        strategy = new SidewaysRouteExtension(settings.getParametersAsArrayList());
072                } else if(name.equals(EndRouteExtension.STRATEGY_NAME)){
073                        strategy = new EndRouteExtension(settings.getParametersAsArrayList());
074                } else if (name.equals(ReduceTimeServedRFare.STRATEGY_NAME)) {
075                        ReduceTimeServedRFare strat = new ReduceTimeServedRFare(settings.getParametersAsArrayList());
076                        strat.setTicketMachine(ticketMachine);
077                        stageContainerCreator.addStageContainerHandler(strat);
078                        strategy = strat;
079                } else if (name.equals(ReduceStopsToBeServedRFare.STRATEGY_NAME)) {
080                        ReduceStopsToBeServedRFare strat = new ReduceStopsToBeServedRFare(settings.getParametersAsArrayList());
081                        strat.setTicketMachine(ticketMachine);
082                        stageContainerCreator.addStageContainerHandler(strat);
083                        strategy = strat;
084                } else if (name.equals(WeightedStartTimeExtension.STRATEGY_NAME)) {
085                        WeightedStartTimeExtension strat = new WeightedStartTimeExtension(settings.getParametersAsArrayList());
086                        strat.setTimeProvider(timeProvider);
087                        strategy = strat;
088                } else if (name.equals(WeightedEndTimeExtension.STRATEGY_NAME)) {
089                        WeightedEndTimeExtension strat = new WeightedEndTimeExtension(settings.getParametersAsArrayList());
090                        strat.setTimeProvider(timeProvider);
091                        strategy = strat;
092                }
093                
094                if (strategy == null) {
095                        log.error("Could not initialize strategy named " + name);
096                }
097                
098                return strategy;
099        }
100
101        private void addStrategy(final PStrategy strategy, final double weight, int disableInIteration) {
102                this.strategies.add(strategy);
103                this.weights.add(weight);
104                this.disableInIteration.add(disableInIteration);
105                this.totalWeights += weight;
106        }
107
108        /**
109         * Changes the weights of each strategy to zero and removes it from the choice set if it needs to be disabled
110         * 
111         * @param iteration
112         */
113        public void updateStrategies(int iteration) {
114                for (int i = 0; i < this.disableInIteration.size(); i++) {
115                        if (this.disableInIteration.get(i) == iteration) {
116                                double weight = this.weights.get(i);
117                                this.weights.set(i, 0.0);
118                                this.strategies.set(i, null);
119                                this.totalWeights -= weight;
120                        }
121                }
122        }
123
124        /**
125         * Picks randomly one strategy from the set of available strategies. Strategies with a higher weight have a higher probability of being picked.
126         *  
127         * @return The strategy picked or null if all strategies were disabled.
128         */
129        public PStrategy chooseStrategy() {
130                if (!allStrategiesAreDisabled) {
131                        double rnd = MatsimRandom.getRandom().nextDouble() * this.totalWeights;
132                        double accumulatedWeight = 0.0;
133                        for (int i = 0; i < this.weights.size(); i++) {
134                                accumulatedWeight += this.weights.get(i);
135                                if (accumulatedWeight >= rnd) {
136                                        return this.strategies.get(i);
137                                }
138                        }
139                        log.info("Total weight is " + this.totalWeights + ". All strategies seem to be disabled.");
140                        log.info("Will start to return null only. This message is given only once.");
141                        this.allStrategiesAreDisabled = true;
142                }
143                return null;
144        }
145
146        @Override
147        public String toString() {
148                StringBuffer strBuffer = new StringBuffer();
149                strBuffer.append("Strategies: ");
150                strBuffer.append(this.strategies.get(0).getStrategyName()); strBuffer.append(" ("); strBuffer.append(this.weights.get(0)); strBuffer.append(")");
151                
152                for (int i = 1; i < this.strategies.size(); i++) {
153                        strBuffer.append(", "); strBuffer.append(this.strategies.get(i).getStrategyName()); strBuffer.append(" ("); strBuffer.append(this.weights.get(i)); strBuffer.append(")");
154                }
155                return strBuffer.toString();
156        }
157}
