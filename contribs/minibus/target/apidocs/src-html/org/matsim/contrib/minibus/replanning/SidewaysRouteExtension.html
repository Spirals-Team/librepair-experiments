042/**
043 * Takes the route transformed into a lineString to calculate a buffer around it. 
044 * Chooses then randomly a new stop within the buffer and inserts it in both directions of the line.
045 * This is done assuming the most distant stops to be the termini.
046 * 
047 * @author aneumann
048 *
049 */
050public final class SidewaysRouteExtension extends AbstractPStrategyModule {
051        
052        private final static Logger log = Logger.getLogger(SidewaysRouteExtension.class);
053        public static final String STRATEGY_NAME = "SidewaysRouteExtension";
054        private final double bufferSize;
055        private final double ratio;
056        private final boolean excludeTermini;
057        
058        public SidewaysRouteExtension(ArrayList<String> parameter) {
059                super();
060                if(parameter.size() != 3){
061                        log.error("Parameter 1: Buffer size in meter");
062                        log.error("Parameter 2: Ratio bufferSize to route's beeline length. If set to something very small, e.g. 0.01, the calculated buffer size may be smaller than the one specified in parameter 1. Parameter 1 will then be taken as minimal buffer size.");
063                        log.error("Parameter 3: Remove buffer from termini - true/false");
064                }
065                this.bufferSize = Double.parseDouble(parameter.get(0));
066                this.ratio = Double.parseDouble(parameter.get(1));
067                this.excludeTermini = Boolean.parseBoolean(parameter.get(2));
068        }
069
070        @Override
071        public PPlan run(Operator operator) {
072
073                PPlan oldPlan = operator.getBestPlan();
074                ArrayList<TransitStopFacility> currentStopsToBeServed = oldPlan.getStopsToBeServed();
075                
076                TransitStopFacility baseStop = currentStopsToBeServed.get(0);
077                TransitStopFacility remoteStop = this.findStopWithLargestDistance(currentStopsToBeServed);
078                double bufferSizeBasedOnRatio = CoordUtils.calcEuclideanDistance(baseStop.getCoord(), remoteStop.getCoord()) * this.ratio;
079                
080                List<Geometry> lineStrings = this.createGeometryFromStops(currentStopsToBeServed, remoteStop);
081                Geometry buffer = this.createBuffer(lineStrings, Math.max(this.bufferSize, bufferSizeBasedOnRatio), this.excludeTermini);
082                
083                Set<Id<TransitStopFacility>> stopsUsed = this.getStopsUsed(oldPlan.getLine().getRoutes().values());
084                TransitStopFacility newStop = this.drawRandomStop(buffer, operator.getRouteProvider(), stopsUsed);
085                
086                if (newStop == null) {
087                        return null;
088                }
089                
090                ArrayList<TransitStopFacility> newStopsToBeServed = this.addStopToExistingStops(baseStop, remoteStop, currentStopsToBeServed, newStop);
091                
092                // create new plan
093                PPlan newPlan = new PPlan(operator.getNewPlanId(), this.getStrategyName(), oldPlan.getId());
094                newPlan.setNVehicles(1);
095                newPlan.setStartTime(oldPlan.getStartTime());
096                newPlan.setEndTime(oldPlan.getEndTime());
097                newPlan.setStopsToBeServed(newStopsToBeServed);
098                
099                newPlan.setLine(operator.getRouteProvider().createTransitLineFromOperatorPlan(operator.getId(), newPlan));
100                
101                return newPlan;
102        }

105        private ArrayList<TransitStopFacility> addStopToExistingStops(TransitStopFacility baseStop, TransitStopFacility remoteStop, ArrayList<TransitStopFacility> currentStopsToBeServed, TransitStopFacility newStop) {
106                List<TransitStopFacility> markerStops = new LinkedList<>();
107                markerStops.add(baseStop);
108                markerStops.add(remoteStop);
109                
110                List<ArrayList<TransitStopFacility>> subRoutes = this.cutRouteIntoSubRoutes(markerStops, currentStopsToBeServed);
111                for (ArrayList<TransitStopFacility> subRoute : subRoutes) {
112                        this.addNewStopBetweenTwoNearestExistingStops(subRoute, newStop);
113                }
114                
115                // merger sub routes
116                ArrayList<TransitStopFacility> newStopsToBeServed = new ArrayList<>();
117                for (ArrayList<TransitStopFacility> subRoute : subRoutes) {
118                        for (TransitStopFacility stop : subRoute) {
119                                newStopsToBeServed.add(stop);
120                        }
121                }
122                return newStopsToBeServed;
123        }
124
125
126        private void addNewStopBetweenTwoNearestExistingStops(ArrayList<TransitStopFacility> stops, TransitStopFacility stop) {
127                // find nearest stop
128                double smallestDistance = Double.MAX_VALUE;
129                TransitStopFacility stopWithSmallestDistance = stops.get(0);
130                TransitStopFacility stopWithSecondSmallestDistance = stops.get(0);
131                for (TransitStopFacility transitStopFacility : stops) {
132                        double currentDistance = CoordUtils.calcEuclideanDistance(stop.getCoord(), transitStopFacility.getCoord());
133                        if (currentDistance < smallestDistance) {
134                                smallestDistance = currentDistance;
135                                stopWithSecondSmallestDistance = stopWithSmallestDistance;
136                                stopWithSmallestDistance = transitStopFacility;
137                        }
138                }
139                
140                // find index to insert
141                int index = Math.min(stops.indexOf(stopWithSecondSmallestDistance), stops.indexOf(stopWithSmallestDistance));
142                
143                // insert
144                stops.add(index + 1, stop);
145        }
146
147        private List<ArrayList<TransitStopFacility>> cutRouteIntoSubRoutes(List<TransitStopFacility> markerStops, ArrayList<TransitStopFacility> stops) {
148                List<ArrayList<TransitStopFacility>> subRoutes = null;
149                ArrayList<TransitStopFacility> subRoute = null;
150                
151                for (TransitStopFacility stop : stops) {
152                        if (markerStops.contains(stop)) {
153                                if (subRoutes == null) {
154                                        // very first sub route
155                                        subRoutes = new LinkedList<>();
156                                        subRoute = new ArrayList<>();
157                                        subRoute.add(stop);
158                                        subRoutes.add(subRoute);
159                                } else {
160                                        // create new one
161                                        subRoute = new ArrayList<>();
162                                        subRoutes.add(subRoute);
163                                        subRoute.add(stop);
164                                }
165                        } else {
166                                subRoute.add(stop);
167                        }
168                }
169                
170                return subRoutes;
171        }
172
173        private TransitStopFacility findStopWithLargestDistance(ArrayList<TransitStopFacility> stops) {
174                Coord startCoord = stops.get(0).getCoord();
175                double largestDistance = 0;
176                TransitStopFacility stopWithLargestDistance = stops.get(0);
177                for (TransitStopFacility transitStopFacility : stops) {
178                        double currentDistance = CoordUtils.calcEuclideanDistance(startCoord, transitStopFacility.getCoord());
179                        if (currentDistance > largestDistance) {
180                                largestDistance = currentDistance;
181                                stopWithLargestDistance = transitStopFacility;
182                        }
183                }
184                return stopWithLargestDistance;
185        }
186
187        private Set<Id<TransitStopFacility>> getStopsUsed(Collection<TransitRoute> routes) {
188                Set<Id<TransitStopFacility>> stopsUsed = new TreeSet<>();
189                for (TransitRoute route : routes) {
190                        for (TransitRouteStop stop : route.getStops()) {
191                                stopsUsed.add(stop.getStopFacility().getId());
192                        }
193                }
194                return stopsUsed;
195        }
196
197        private TransitStopFacility drawRandomStop(Geometry buffer, PRouteProvider pRouteProvider, Set<Id<TransitStopFacility>> stopsUsed) {
198                List<TransitStopFacility> choiceSet = new LinkedList<>();
199                
200                // find choice-set
201                for (TransitStopFacility stop : pRouteProvider.getAllPStops()) {
202                        if (!stopsUsed.contains(stop.getId())) {
203                                if (buffer.contains(MGC.coord2Point(stop.getCoord()))) {
204                                        choiceSet.add(stop);
205                                }
206                        }
207                }
208                
209                return pRouteProvider.drawRandomStopFromList(choiceSet);
210        }

213        private Geometry createBuffer(List<Geometry> lineStrings, double bufferSize, boolean excludeTermini) {
214                BufferParameters bufferParameters = new BufferParameters();
215                
216                if (excludeTermini) {
217                        bufferParameters.setEndCapStyle(BufferParameters.CAP_FLAT);
218                } else {
219                        bufferParameters.setEndCapStyle(BufferParameters.CAP_ROUND);
220                }
221                
222                Geometry union = null;
223                
224                for (Geometry lineString : lineStrings) {
225                        Geometry buffer = BufferOp.bufferOp(lineString, bufferSize, bufferParameters);
226                        if (union == null) {
227                                union = buffer;
228                        } else {
229                                union = union.union(buffer);
230                        }
231                }
232                
233                return union;
234        }

237        private List<Geometry> createGeometryFromStops(ArrayList<TransitStopFacility> stops, TransitStopFacility remoteStop) {
238                List<Geometry> geometries = new LinkedList<>();
239                
240                ArrayList<Coordinate> coords = new ArrayList<>();
241                for (TransitStopFacility stop : stops) {
242                        if (stop.equals(remoteStop)) {
243                                // terminate current line string
244                                coords.add(new Coordinate(stop.getCoord().getX(), stop.getCoord().getY(), 0.0));
245                                Coordinate[] coordinates = coords.toArray(new Coordinate[coords.size()]);
246                                Geometry lineString = new GeometryFactory().createLineString(coordinates);
247                                geometries.add(lineString);
248                                // create new line string
249                                coords = new ArrayList<>();
250                                coords.add(new Coordinate(stop.getCoord().getX(), stop.getCoord().getY(), 0.0));
251                        } else {
252                                coords.add(new Coordinate(stop.getCoord().getX(), stop.getCoord().getY(), 0.0));
253                        }
254                }
255                // add first stop to close the circle
256                coords.add(new Coordinate(stops.get(0).getCoord().getX(), stops.get(0).getCoord().getY(), 0.0));
257                
258                Coordinate[] coordinates = coords.toArray(new Coordinate[coords.size()]);
259                Geometry lineString = new GeometryFactory().createLineString(coordinates);
260                geometries.add(lineString);
261                return geometries;
262        }
263
264
265        @Override
266        public String getStrategyName() {
267                return SidewaysRouteExtension.STRATEGY_NAME;
268        }
