020package org.matsim.contrib.minibus.replanning;
021
022import org.apache.log4j.Logger;
023import org.matsim.contrib.minibus.operator.Operator;
024import org.matsim.contrib.minibus.operator.PPlan;
025import org.matsim.contrib.minibus.operator.TimeProvider;
026import org.matsim.core.gbl.MatsimRandom;
027
028import java.util.ArrayList;
029
030/**
031 * 
032 * Sets a new endTime between the old endTime and midnight.
033 * 
034 * @author aneumann
035 *
036 */
037public final class MaxRandomEndTimeAllocator extends AbstractPStrategyModule {
038        
039        private final static Logger log = Logger.getLogger(MaxRandomEndTimeAllocator.class);    
040        public static final String STRATEGY_NAME = "MaxRandomEndTimeAllocator";
041        
042        private final int mutationRange;
043        private final int timeBinSize;
044        private final boolean searchInBothDirections;
045        
046        public MaxRandomEndTimeAllocator(ArrayList<String> parameter) {
047                super();
048                if(parameter.size() != 3){
049                        log.error("Parameter 1: Mutation range in seconds");
050                        log.error("Parameter 2: Time bin size in seconds");
051                        log.error("Parameter 3: Search in both directions true/false. Setting it true will double the total mutation range.");
052                }
053                this.mutationRange = Integer.parseInt(parameter.get(0));
054                this.timeBinSize = Integer.parseInt(parameter.get(1));
055                this.searchInBothDirections = Boolean.parseBoolean(parameter.get(2));
056        }
057        
058        @Override
059        public PPlan run(Operator operator) {
060                // change endTime
061                PPlan newPlan = new PPlan(operator.getNewPlanId(), this.getStrategyName(), operator.getBestPlan().getId());
062                newPlan.setNVehicles(1);
063                newPlan.setStopsToBeServed(operator.getBestPlan().getStopsToBeServed());
064                newPlan.setStartTime(operator.getBestPlan().getStartTime());
065                
066                // get a valid new end time
067                double timeMutation;
068                if (searchInBothDirections) {
069                        timeMutation = (MatsimRandom.getRandom().nextDouble() - 0.5) * 2.0 * this.mutationRange;
070                } else {
071                        timeMutation = MatsimRandom.getRandom().nextDouble() * this.mutationRange;
072                }
073                
074                double newEndTime = Math.min(24 * 3600.0, operator.getBestPlan().getEndTime() + timeMutation);
075                                
076                // cast time to time bin size
077                newEndTime = TimeProvider.getSlotForTime(newEndTime, this.timeBinSize) * this.timeBinSize;
078                
079                // increase to match min operation time
080                while (newEndTime < operator.getBestPlan().getStartTime() + operator.getMinOperationTime()) {
081                        newEndTime += this.timeBinSize;
082                }
083                
084                newPlan.setEndTime(newEndTime);
085                
086                if(newPlan.getEndTime() <= newPlan.getStartTime()){
087                        // Could not find a valid new plan
088                        return null;
089                }
090                
091                newPlan.setLine(operator.getRouteProvider().createTransitLineFromOperatorPlan(operator.getId(), newPlan));
092                
093                return newPlan;
094        }
095
096        public String getStrategyName() {
097                return MaxRandomEndTimeAllocator.STRATEGY_NAME;
098        }
099}
