021package org.matsim.contrib.locationchoice.bestresponse.preprocess;
022
023import org.apache.log4j.Logger;
024import org.matsim.api.core.v01.Scenario;
025import org.matsim.api.core.v01.population.Person;
026import org.matsim.contrib.locationchoice.DestinationChoiceConfigGroup;
027import org.matsim.contrib.locationchoice.utils.RandomFromVarDistr;
028import org.matsim.core.config.Config;
029import org.matsim.core.utils.io.UncheckedIOException;
030import org.matsim.facilities.ActivityFacility;
031import org.matsim.utils.objectattributes.ObjectAttributes;
032import org.matsim.utils.objectattributes.ObjectAttributesXmlReader;
033import org.matsim.utils.objectattributes.ObjectAttributesXmlWriter;
034
035public class ReadOrCreateKVals {
036        
037        private static final Logger log = Logger.getLogger(ReadOrCreateKVals.class);
038        
039        public static String fkValuesFile = "facilitiesKValues.xml";
040        public static String pkValuesFile = "personsKValues.xml";
041        
042        private Scenario scenario;      
043        private ObjectAttributes facilitiesKValues = new ObjectAttributes();
044        private ObjectAttributes personsKValues = new ObjectAttributes();
045        private Config config;  
046        private RandomFromVarDistr rnd;
047        
048        public ReadOrCreateKVals(long seed, Scenario scenario) {
049                this.scenario = scenario;
050                this.config = scenario.getConfig();
051                this.rnd = new RandomFromVarDistr();
052                this.rnd.setSeed(seed);
053        }
054        
055        /*
056         * return 0 if files are read and 1 if k values are created. 
057         * This is important to know for reading (case 0) or computation of maxDCScore (case 1)
058         */
059        public int run() {
060                DestinationChoiceConfigGroup dccg = (DestinationChoiceConfigGroup) scenario.getConfig().getModule(DestinationChoiceConfigGroup.GROUP_NAME);
061                String pkValuesFileName = dccg.getpkValuesFile();
062                String fkValuesFileName = dccg.getfkValuesFile();
063                String maxEpsValuesFileName = dccg.getMaxEpsFile();
064                if (pkValuesFileName != null && fkValuesFileName != null && maxEpsValuesFileName != null) {                     
065                        ObjectAttributesXmlReader persKValuesReader = new ObjectAttributesXmlReader(this.personsKValues);
066                        ObjectAttributesXmlReader facKValuesReader = new ObjectAttributesXmlReader(this.facilitiesKValues);
067                        try {
068                                persKValuesReader.readFile(pkValuesFileName);
069                                facKValuesReader.readFile(fkValuesFileName);
070                                log.info("reading kvals from files:\n"+ pkValuesFileName + "\n" + fkValuesFileName);
071                                return 0;
072                        } catch  (UncheckedIOException e) {
073                                // reading was not successful
074                                log.error("unsuccessful reading kvals from files!\nThe values are now computed" +
075                                " and following files are not considered!:\n" + pkValuesFileName + "\n" + fkValuesFileName);
076                                this.assignKValues();
077                                return 1;
078                        }
079                }
080                else {
081                        this.assignKValues();
082                        return 1;
083                }
084        }
085                
086        public void assignKValues() {
087                log.info("generating kVals");
088                this.assignKValuesPersons();
089                this.assignKValuesAlternatives();
090        }
091                
092        // does not matter which distribution is chosen here
093        private void assignKValuesPersons() {
094                for (Person p : this.scenario.getPopulation().getPersons().values()) {
095                        this.personsKValues.putAttribute(p.getId().toString(), "k", rnd.getUniform(1.0));
096                }
097                // write person k values
098                ObjectAttributesXmlWriter attributesWriter = new ObjectAttributesXmlWriter(this.personsKValues);
099                attributesWriter.writeFile(config.controler().getOutputDirectory() + pkValuesFile);
100        }       
101        private void assignKValuesAlternatives() {
102                for (ActivityFacility facility : this.scenario.getActivityFacilities().getFacilities().values()) {
103                        this.facilitiesKValues.putAttribute(facility.getId().toString(), "k", rnd.getUniform(1.0));
104                }
105                ObjectAttributesXmlWriter attributesWriter = new ObjectAttributesXmlWriter(this.facilitiesKValues);
106                attributesWriter.writeFile(config.controler().getOutputDirectory() + fkValuesFile);
107        }
108        
109        public ObjectAttributes getFacilitiesKValues() {
110                return this.facilitiesKValues;
111        }
112        
113        public ObjectAttributes getPersonsKValues() {
114                return this.personsKValues;
115        }
116}
