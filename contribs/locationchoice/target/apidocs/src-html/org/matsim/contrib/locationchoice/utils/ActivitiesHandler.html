020package org.matsim.contrib.locationchoice.utils;
021
022import java.util.HashSet;
023import java.util.List;
024import java.util.Vector;
025
026import org.apache.log4j.Logger;
027import org.matsim.api.core.v01.population.Activity;
028import org.matsim.api.core.v01.population.Plan;
029import org.matsim.contrib.locationchoice.DestinationChoiceConfigGroup;
030import org.matsim.contrib.locationchoice.bestresponse.scoring.ScaleEpsilon;
031
032public class ActivitiesHandler {
033        private HashSet<String> flexibleTypes = new HashSet<String>();
034        private DestinationChoiceConfigGroup dcconfig;
035        private ActTypeConverter converter = null;
036
037        public ActivitiesHandler(final DestinationChoiceConfigGroup dcconfig) {
038                this.dcconfig = dcconfig;
039                this.createActivityTypeConverter();
040                this.initFlexibleTypes(dcconfig);
041        }
042        
043        public ScaleEpsilon createScaleEpsilon() {
044                ScaleEpsilon scaleEpsilon = new ScaleEpsilon();
045                String factors = this.dcconfig.getEpsilonScaleFactors();
046                String types = this.dcconfig.getFlexibleTypes();
047                
048                if (!types.equals("null")) {                    
049                        String[] fentries = factors.split(",", -1);
050                        String[] tentries = types.split(",", -1);
051                        
052                        // check if demand is v1 with types = s0.5, ...
053                        if (tentries[0].length() == 1) {
054                                scaleEpsilon.setUseSimpleTypes(true);
055                        }
056                        
057                        int fentriesLength = fentries.length;
058                        if (fentries[0].equals("null")) fentriesLength = 0;
059                        
060                        int tentriesLength = tentries.length;
061                        if (tentries[0].equals("null")) tentriesLength = 0;
062                        
063                        if (fentriesLength != tentriesLength || factors.equals("null")) {
064                                throw new RuntimeException("Specify an epsilon (now: " + fentriesLength + " specified) " +
065                                                "for every flexible activity type (now: " + tentriesLength + " specified)!");
066                        }                       
067                        for (int i = 0; i < fentries.length; i++) {
068                                        scaleEpsilon.setEpsilonFactor(tentries[i].trim(), Double.parseDouble(fentries[i].trim()));
069                        }
070                }       
071                return scaleEpsilon;
072        }
073        
074        private void createActivityTypeConverter() {
075                String types = this.dcconfig.getFlexibleTypes();
076                String[] tentries = types.split(",", -1);
077                
078                // check if demand = v1
079                if (tentries[0].length() == 1) {
080                        this.converter = new ActTypeConverter(true);
081                        Logger.getLogger(this.getClass()).warn("LocationChoice uses an activity type converter that only uses the first letter of every activity type. " +
082                                        "For example, `shopping' and `school' may be treated as the same activity.  Make sure this is what you want.") ; // kai, jan'13
083                }
084                else {
085                        this.converter = new ActTypeConverter(false);
086                }
087                // yyyy If I understand this correctly, this means: If the first activity in the list of flexible activities is of length
088                // one, then "true" is used (i.e. all activities are reduced to length one).  Otherwise "false".  
089                // I find this "automagic" a bit dangerous.  kai, jan'13
090        }
091
092        // only used by TGSimple
093        public List<Activity> getFlexibleActivities(final Plan plan) {
094                List<Activity> flexibleActivities = new Vector<Activity>();
095                this.getFlexibleActs(plan, flexibleActivities);
096                return flexibleActivities;
097        }
098
099        private void getFlexibleActs(Plan plan, List<Activity> flexibleActivities) {
100                for (int i = 0; i < plan.getPlanElements().size(); i = i + 2) {
101                        Activity act = (Activity)plan.getPlanElements().get(i);
102                        if (this.flexibleTypes.contains(this.converter.convertType(act.getType()))) {
103                                flexibleActivities.add(act);
104                        }
105                }
106        }
107
108        private void initFlexibleTypes(DestinationChoiceConfigGroup config) {
109                String types = config.getFlexibleTypes();
110                if (!types.equals("null")) {
111                        String[] entries = types.split(",", -1);
112                        for (int i = 0; i < entries.length; i++) {
113                                this.flexibleTypes.add(this.converter.convertType(entries[i].trim()));
114                        }
115                }
116        }
117
118        public HashSet<String> getFlexibleTypes() {
119                return flexibleTypes;
120        }
121
122        public void setFlexibleTypes(HashSet<String> flexibleTypes) {
123                this.flexibleTypes = flexibleTypes;
124        }
125
126        public ActTypeConverter getConverter() {
127                return converter;
128        }
129
130        public void setConverter(ActTypeConverter converter) {
131                this.converter = converter;
132        }
133
134}
