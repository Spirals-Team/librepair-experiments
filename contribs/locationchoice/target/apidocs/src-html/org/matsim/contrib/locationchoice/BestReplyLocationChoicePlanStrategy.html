020package org.matsim.contrib.locationchoice;
021
022import org.matsim.api.core.v01.Scenario;
023import org.matsim.api.core.v01.population.HasPlansAndId;
024import org.matsim.api.core.v01.population.Person;
025import org.matsim.api.core.v01.population.Plan;
026import org.matsim.contrib.locationchoice.bestresponse.DestinationChoiceBestResponseContext;
027import org.matsim.contrib.locationchoice.bestresponse.preprocess.MaxDCScoreWrapper;
028import org.matsim.core.config.Config;
029import org.matsim.core.replanning.PlanStrategy;
030import org.matsim.core.replanning.PlanStrategyImpl;
031import org.matsim.core.replanning.ReplanningContext;
032import org.matsim.core.replanning.modules.ReRoute;
033import org.matsim.core.replanning.modules.TripsToLegsModule;
034import org.matsim.core.replanning.selectors.BestPlanSelector;
035import org.matsim.core.replanning.selectors.ExpBetaPlanChanger;
036import org.matsim.core.replanning.selectors.ExpBetaPlanSelector;
037import org.matsim.core.replanning.selectors.RandomPlanSelector;
038import org.matsim.core.router.TripRouter;
039import org.matsim.core.router.costcalculators.TravelDisutilityFactory;
040import org.matsim.core.router.util.TravelTime;
041import org.matsim.core.scoring.ScoringFunctionFactory;
042
043import javax.inject.Inject;
044import javax.inject.Provider;
045import java.util.Map;
046
047public class BestReplyLocationChoicePlanStrategy implements PlanStrategy {
048
049        private PlanStrategyImpl delegate;
050        private Scenario scenario;
051        private final Provider<TripRouter> tripRouterProvider;
052        private ScoringFunctionFactory scoringFunctionFactory;
053        private Map<String, TravelTime> travelTimes;
054        private Map<String, TravelDisutilityFactory> travelDisutilities;
055
056        @Inject
057        BestReplyLocationChoicePlanStrategy(Scenario scenario, Provider<TripRouter> tripRouterProvider, ScoringFunctionFactory scoringFunctionFactory, Map<String, TravelTime> travelTimes, Map<String, TravelDisutilityFactory> travelDisutilities) {
058                this.scenario = scenario;
059                this.tripRouterProvider = tripRouterProvider;
060                this.scoringFunctionFactory = scoringFunctionFactory;
061                this.travelTimes = travelTimes;
062                this.travelDisutilities = travelDisutilities;
063        }
064                
065        @Override
066        public void run(HasPlansAndId<Plan, Person> person) {
067                delegate.run(person);
068        }
069
070        @Override
071        public void init(ReplanningContext replanningContext) {
072                /*
073                 * Somehow this is ugly. Should be initialized in the constructor. But I do not know, how to initialize the lc scenario elements
074                 * such that they are already available at the time of constructing this object. ah feb'13
075                 */
076                DestinationChoiceBestResponseContext lcContext = (DestinationChoiceBestResponseContext) scenario.getScenarioElement(DestinationChoiceBestResponseContext.ELEMENT_NAME);
077                Config config = lcContext.getScenario().getConfig();
078                DestinationChoiceConfigGroup dccg = (DestinationChoiceConfigGroup) config.getModule(DestinationChoiceConfigGroup.GROUP_NAME);
079                MaxDCScoreWrapper maxDcScoreWrapper = (MaxDCScoreWrapper)scenario.getScenarioElement(MaxDCScoreWrapper.ELEMENT_NAME);
080                if ( !DestinationChoiceConfigGroup.Algotype.bestResponse.equals(dccg.getAlgorithm())) {
081                        throw new RuntimeException("wrong class for selected location choice algorithm type; aborting ...") ;
082                }               
083                String planSelector = dccg.getPlanSelector();
084                if (planSelector.equals("BestScore")) {
085                        delegate = new PlanStrategyImpl(new BestPlanSelector<Plan, Person>());
086                } else if (planSelector.equals("ChangeExpBeta")) {
087                        delegate = new PlanStrategyImpl(new ExpBetaPlanChanger(config.planCalcScore().getBrainExpBeta()));
088                } else if (planSelector.equals("SelectRandom")) {
089                        delegate = new PlanStrategyImpl(new RandomPlanSelector());
090                } else {
091                        delegate = new PlanStrategyImpl(new ExpBetaPlanSelector(config.planCalcScore()));
092                }
093                delegate.addStrategyModule(new TripsToLegsModule(tripRouterProvider, config.global()));
094                delegate.addStrategyModule(new BestReplyDestinationChoice(tripRouterProvider, lcContext, maxDcScoreWrapper.getPersonsMaxDCScoreUnscaled(), scoringFunctionFactory, travelTimes, travelDisutilities));
095                delegate.addStrategyModule(new ReRoute(lcContext.getScenario(), tripRouterProvider));
096                
097                delegate.init(replanningContext);
098        }
099
100        @Override
101        public void finish() {
102                delegate.finish();
103        }
104}
