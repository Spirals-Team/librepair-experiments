021package org.matsim.contrib.locationchoice.router;
022
023import org.matsim.api.core.v01.network.Network;
024import org.matsim.core.router.ArrayFastRouterDelegateFactory;
025import org.matsim.core.router.FastRouterDelegateFactory;
026import org.matsim.core.router.util.*;
027
028import java.util.HashMap;
029import java.util.Map;
030
031/**
032 * Creates a MultiNodeDijkstra which is routing backwards. To do so,
033 * The network is converted into an inverse routing network, i.e. in- and
034 * outgoing links are exchanged.
035 * 
036 * @author cdobler
037 */
038public class BackwardFastMultiNodeDijkstraFactory implements LeastCostPathCalculatorFactory {
039        
040        private final boolean searchAllEndNodes;
041        private final boolean usePreProcessData;
042        private final RoutingNetworkFactory routingNetworkFactory;
043        private final Map<Network, RoutingNetwork> routingNetworks = new HashMap<>();
044        private final Map<Network, PreProcessDijkstra> preProcessData = new HashMap<>();
045
046        public BackwardFastMultiNodeDijkstraFactory(){
047                this(false);
048        }
049        
050    public BackwardFastMultiNodeDijkstraFactory(final boolean searchAllEndNodes) {
051                this(false, searchAllEndNodes);
052        }
053                
054        public BackwardFastMultiNodeDijkstraFactory(final boolean usePreProcessData, final boolean searchAllEndNodes) {
055                this.usePreProcessData = usePreProcessData;
056                this.searchAllEndNodes = searchAllEndNodes;
057                this.routingNetworkFactory = new InverseArrayRoutingNetworkFactory();
058        }
059
060        @Override
061        public LeastCostPathCalculator createPathCalculator(final Network network, final TravelDisutility travelCosts, final TravelTime travelTimes) {
062                
063                RoutingNetwork routingNetwork = this.routingNetworks.get(network);
064                PreProcessDijkstra preProcessDijkstra = this.preProcessData.get(network);
065        
066                if (routingNetwork == null) {
067                        routingNetwork = this.routingNetworkFactory.createRoutingNetwork(network);
068                        
069                        if (this.usePreProcessData) {
070                                if (preProcessDijkstra == null) {
071                                        preProcessDijkstra = new PreProcessDijkstra();
072                                        preProcessDijkstra.run(network);
073                                        this.preProcessData.put(network, preProcessDijkstra);
074                                }
075                                if (preProcessDijkstra.containsData()) {
076                                        for (RoutingNetworkNode node : routingNetwork.getNodes().values()) {
077                                                node.setDeadEndData(preProcessDijkstra.getNodeData(node.getNode()));
078                                        }
079                                }
080                        }
081                        
082                        this.routingNetworks.put(network, routingNetwork);
083                }
084                FastRouterDelegateFactory fastRouterFactory = new ArrayFastRouterDelegateFactory();
085                
086                return new BackwardFastMultiNodeDijkstra(routingNetwork, travelCosts, travelTimes, 
087                                preProcessDijkstra, fastRouterFactory, this.searchAllEndNodes);
088        }
089}
