001package org.matsim.contrib.locationchoice;
002
003import org.matsim.api.core.v01.Scenario;
004import org.matsim.api.core.v01.population.HasPlansAndId;
005import org.matsim.api.core.v01.population.Person;
006import org.matsim.api.core.v01.population.Plan;
007import org.matsim.core.replanning.PlanStrategy;
008import org.matsim.core.replanning.PlanStrategyImpl;
009import org.matsim.core.replanning.ReplanningContext;
010import org.matsim.core.replanning.modules.ReRoute;
011import org.matsim.core.replanning.selectors.BestPlanSelector;
012import org.matsim.core.replanning.selectors.ExpBetaPlanChanger;
013import org.matsim.core.replanning.selectors.ExpBetaPlanSelector;
014import org.matsim.core.replanning.selectors.RandomPlanSelector;
015import org.matsim.core.router.TripRouter;
016
017import javax.inject.Provider;
018
019public class LocationChoicePlanStrategy implements PlanStrategy {
020
021        private final Provider<TripRouter> tripRouterProvider;
022        private PlanStrategyImpl delegate;
023        
024        public LocationChoicePlanStrategy(Scenario scenario, Provider<TripRouter> tripRouterProvider) {
025                this.tripRouterProvider = tripRouterProvider;
026                if ( DestinationChoiceConfigGroup.Algotype.bestResponse==((DestinationChoiceConfigGroup)scenario.getConfig().getModule("locationchoice")).getAlgorithm() ) {
027                        throw new RuntimeException("best response location choice not supported as part of LocationChoicePlanStrategy. " +
028                                        "Use BestReplyLocationChoicePlanStrategy instead, but be aware that as of now some Java coding is necessary to do that. kai, feb'13") ;
029                }
030                DestinationChoiceConfigGroup dccg = (DestinationChoiceConfigGroup) scenario.getConfig().getModule(DestinationChoiceConfigGroup.GROUP_NAME);
031                String planSelector = dccg.getPlanSelector();
032                if (planSelector.equals("BestScore")) {
033                        delegate = new PlanStrategyImpl(new BestPlanSelector());
034                } else if (planSelector.equals("ChangeExpBeta")) {
035                        delegate = new PlanStrategyImpl(new ExpBetaPlanChanger(scenario.getConfig().planCalcScore().getBrainExpBeta()));
036                } else if (planSelector.equals("SelectRandom")) {
037                        delegate = new PlanStrategyImpl(new RandomPlanSelector());
038                } else {
039                        delegate = new PlanStrategyImpl(new ExpBetaPlanSelector(scenario.getConfig().planCalcScore()));
040                }
041                delegate.addStrategyModule(new DestinationChoice(this.tripRouterProvider, scenario));
042                delegate.addStrategyModule(new ReRoute(scenario, tripRouterProvider));
043        }
044        
045        @Override
046        public void run(HasPlansAndId<Plan, Person> person) {
047                delegate.run(person);
048        }
049
050        @Override
051        public void init(ReplanningContext replanningContext) {
052                delegate.init(replanningContext);
053        }
054
055        @Override
056        public void finish() {
057                delegate.finish();
058        }
059        
060}
