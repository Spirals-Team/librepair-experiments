020package org.matsim.contrib.locationchoice.facilityload;
021
022
023//import org.apache.log4j.Logger;
024
025
026public class ScoringPenalty {
027
028        private double startTime = 0;
029        private double endTime = 0;
030        private FacilityPenalty facilityPenalty = null;
031        private double score = 0.0;
032
033        //private static final Logger log = Logger.getLogger(Penalty.class);
034
035        public ScoringPenalty(double startTime, double endTime, FacilityPenalty facilityPenalty, double score) {
036                this.startTime = startTime;
037                this.endTime = endTime;
038                this.facilityPenalty = facilityPenalty;
039                this.score = score;
040        }
041
042
043        public double getScore() {
044                return score;
045        }
046        public void setScore(double score) {
047                this.score = score;
048        }
049
050        public double getPenalty() {
051                
052                double penaltyFactor = 0.0;
053                if (this.facilityPenalty != null) {
054                        this.facilityPenalty.finish(); // is this still needed? we have a call in EventsToFacilityLoad
055                        penaltyFactor = this.facilityPenalty.getCapacityPenaltyFactor(startTime, endTime);
056                }
057                return this.score * penaltyFactor;
058        }
059}
