042                        final TravelTime timeFunction, final PreProcessDijkstra preProcessData, 
043                        final FastRouterDelegateFactory fastRouterFactory, boolean searchAllEndNodes) {
044                super(routingNetwork, costFunction, timeFunction, preProcessData, fastRouterFactory, searchAllEndNodes);

047        @Override
048        protected boolean addToPendingNodes(final Link l, final Node n,
049                        final RouterPriorityQueue<Node> pendingNodes, final double currTime,
050                        final double currCost, final Node toNode) {
051                /*
052                 * This part is exchanging the two commented out lines. Maybe this could be moved
053                 * to a separate method in the main class?
054                 */
055                // bw: travel time has to be negative while costs are still positive
056                // But we have a problem if current time is negative. Use previous day instead! 
057                double travelTime;
058                double travelCost;
059                if (currTime < 0) {
060                        double timeMod = 24.0 * 3600.0 - Math.abs(currTime % (24.0 * 3600.0));
061                        travelTime = -1.0 * this.timeFunction.getLinkTravelTime(l, timeMod, getPerson(), getVehicle());
062                        travelCost = this.costFunction.getLinkTravelDisutility(l, timeMod, getPerson(), getVehicle());                  
063                } else {
064                        travelTime = -1.0 * this.timeFunction.getLinkTravelTime(l, currTime, getPerson(), getVehicle());
065                        travelCost = this.costFunction.getLinkTravelDisutility(l, currTime,  getPerson(), getVehicle());
066                }       
067//              double travelTime = this.timeFunction.getLinkTravelTime(l, currTime, person, vehicle);
068//              double travelCost = this.costFunction.getLinkTravelDisutility(l, currTime, this.person, this.vehicle);
069
070                /*
071                 * This is copied from the default Dijkstra implementation.
072                 */
073                DijkstraNodeData data = getData(n);
074                double nCost = data.getCost();
075                if (!data.isVisited(getIterationId())) {
076                        visitNode(n, data, pendingNodes, currTime + travelTime, currCost + travelCost, l);
077                        return true;
078                }
079                double totalCost = currCost + travelCost;
080                if (totalCost < nCost) {
081                        revisitNode(n, data, pendingNodes, currTime + travelTime, totalCost, l);
082                        return true;
083                }
084
085                return false;
086        }
087        
088        @Override
089        protected Path constructPath(Node fromNode, Node toNode, double startTime, double arrivalTime) {
090                
091                ArrayList<Node> nodes = new ArrayList<>();
092                ArrayList<Link> links = new ArrayList<>();
093
094                nodes.add(((RoutingNetworkNode) toNode).getNode());
095                Link tmpLink = getData(toNode).getPrevLink();
096                if (tmpLink != null) {
097                        while (tmpLink.getFromNode() != fromNode) {
098                                links.add(((RoutingNetworkLink) tmpLink).getLink());
099                                nodes.add(((RoutingNetworkLink) tmpLink).getLink().getToNode());
100                                tmpLink = getData(tmpLink.getFromNode()).getPrevLink();
101                        }
102                        links.add(((RoutingNetworkLink) tmpLink).getLink());
103                        nodes.add(((RoutingNetworkNode) tmpLink.getFromNode()).getNode());
104                }
105                
106                NodeData toNodeData = getData(toNode);
107                Path path = new Path(nodes, links, startTime - arrivalTime, toNodeData.getCost());
108
109                return path;
