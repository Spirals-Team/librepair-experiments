020package org.matsim.contrib.locationchoice.bestresponse.preprocess;
021
022import java.util.HashSet;
023
024import org.apache.log4j.Logger;
025import org.matsim.api.core.v01.Scenario;
026import org.matsim.api.core.v01.population.Person;
027import org.matsim.contrib.locationchoice.DestinationChoiceConfigGroup;
028import org.matsim.contrib.locationchoice.bestresponse.DestinationChoiceContext;
029import org.matsim.contrib.locationchoice.bestresponse.DestinationSampler;
030import org.matsim.contrib.locationchoice.bestresponse.scoring.ScaleEpsilon;
031import org.matsim.core.config.Config;
032import org.matsim.core.utils.io.UncheckedIOException;
033import org.matsim.utils.objectattributes.ObjectAttributes;
034import org.matsim.utils.objectattributes.ObjectAttributesXmlReader;
035import org.matsim.utils.objectattributes.ObjectAttributesXmlWriter;
036
037public class ReadOrComputeMaxDCScore {  
038        
039        private final static Logger log = Logger.getLogger(ReadOrComputeMaxDCScore.class);
040
041        public static String maxEpsFile = "personsMaxDCScoreUnscaled.xml";
042
043        private Config config;
044        private Scenario scenario; 
045        private DestinationChoiceConfigGroup dccg;
046        private DestinationChoiceContext lcContext;
047        private ObjectAttributes personsMaxDCScoreUnscaled = new ObjectAttributes();    
048        private ScaleEpsilon scaleEpsilon;      
049        private HashSet<String> flexibleTypes;
050        
051        public ReadOrComputeMaxDCScore(DestinationChoiceContext lcContext) {
052                this.scenario = lcContext.getScenario();
053                this.config = this.scenario.getConfig();
054                this.dccg = (DestinationChoiceConfigGroup) scenario.getConfig().getModule(DestinationChoiceConfigGroup.GROUP_NAME);
055                this.scaleEpsilon = lcContext.getScaleEpsilon();
056                this.flexibleTypes = lcContext.getFlexibleTypes();
057                this.lcContext = lcContext;
058        }
059
060    public void readOrCreateMaxDCScore(Config config, boolean arekValsRead) {
061                String maxEpsValuesFileName = this.dccg.getMaxEpsFile();
062                if (maxEpsValuesFileName != null && arekValsRead) {
063                        ObjectAttributesXmlReader maxEpsReader = new ObjectAttributesXmlReader(this.personsMaxDCScoreUnscaled);
064                        try {
065                                maxEpsReader.readFile(maxEpsValuesFileName);
066                                log.info("reading maxEpsilons from file:\n"+ maxEpsValuesFileName);
067                        } catch  (UncheckedIOException e) {
068                                // reading was not successful
069                                log.error("unsuccessful reading of maxDCScore from file!\nThe values are now computed" +
070                                " and following files are not considered!:\n" + maxEpsValuesFileName);
071                                this.computeMaxDCScore();
072                        }
073                }
074                else {
075                        log.info("computing maxDCScore");
076                        this.computeMaxDCScore();
077                }
078        }
079        
080        private void computeMaxDCScore() {
081                DestinationSampler sampler = new DestinationSampler(this.lcContext.getPersonsKValuesArray(), 
082                                this.lcContext.getFacilitiesKValuesArray(), 
083                                this.dccg);
084                                
085                log.info("Computing max epsilon ... for " + this.scenario.getPopulation().getPersons().size() + " persons");
086                for (String actType : this.scaleEpsilon.getFlexibleTypes()) {
087                        log.info("Computing max epsilon for activity type " + actType);
088                        ComputeMaxDCScoreMultiThreatedModule maxEpsilonComputer = new ComputeMaxDCScoreMultiThreatedModule(
089                                        actType, this.lcContext, sampler);
090                        maxEpsilonComputer.prepareReplanning(null);
091                        for (Person p : this.scenario.getPopulation().getPersons().values()) {
092                                maxEpsilonComputer.handlePlan(p.getSelectedPlan());
093                        }
094                        maxEpsilonComputer.finishReplanning();
095                }               
096                this.writeMaxEps();
097        }
098        
099        private void writeMaxEps() {
100                for (Person person : this.scenario.getPopulation().getPersons().values()) {
101                        for (String flexibleType : this.flexibleTypes) {
102                                double maxType = (Double)person.getCustomAttributes().get(flexibleType);
103                                this.personsMaxDCScoreUnscaled.putAttribute(person.getId().toString(), flexibleType, maxType);
104                        }       
105                }
106                ObjectAttributesXmlWriter attributesWriter = new ObjectAttributesXmlWriter(this.personsMaxDCScoreUnscaled);
107                attributesWriter.writeFile(this.config.controler().getOutputDirectory() + maxEpsFile);
108        }
109        
110        public ObjectAttributes getPersonsMaxEpsUnscaled() {
111                return personsMaxDCScoreUnscaled;
112        }
113}
