021package org.matsim.contrib.locationchoice.facilityload;
022
023import java.util.Iterator;
024import java.util.TreeMap;
025
026import org.apache.log4j.Logger;
027import org.matsim.api.core.v01.Id;
028import org.matsim.api.core.v01.events.ActivityEndEvent;
029import org.matsim.api.core.v01.events.ActivityStartEvent;
030import org.matsim.api.core.v01.events.handler.ActivityEndEventHandler;
031import org.matsim.api.core.v01.events.handler.ActivityStartEventHandler;
032import org.matsim.contrib.locationchoice.DestinationChoiceConfigGroup;
033import org.matsim.facilities.ActivityFacilities;
034import org.matsim.facilities.ActivityFacility;
035import org.matsim.facilities.ActivityOption;
036
037/**
038 * @author anhorni
039 * Uses FacilityPenalty to manage the facililities' loads by taking care of activity start and end events.
040 */
041public class EventsToFacilityLoad implements ActivityStartEventHandler, ActivityEndEventHandler {
042
043        private TreeMap<Id, FacilityPenalty> facilityPenalties;
044        private final static Logger log = Logger.getLogger(EventsToFacilityLoad.class);
045
046        public EventsToFacilityLoad(final ActivityFacilities facilities, double scaleNumberOfPersons,
047                        TreeMap<Id, FacilityPenalty> facilityPenalties, DestinationChoiceConfigGroup config) {
048                super();
049
050                this.facilityPenalties = facilityPenalties;
051
052                log.info("facilities size: " + facilities.getFacilities().values().size());
053                                
054                int counter = 0;
055                int nextMsg = 1;
056                for (ActivityFacility f : facilities.getFacilities().values()) {
057                        counter++;
058                        if (counter % nextMsg == 0) {
059                                nextMsg *= 2;
060                                log.info(" facility # " + counter);
061                        }
062                        double capacity = Double.MAX_VALUE;
063                        Iterator<? extends ActivityOption> iter_act = f.getActivityOptions().values().iterator();
064                        while (iter_act.hasNext()){
065                                ActivityOption act = iter_act.next();
066                                if (act.getCapacity() < capacity) {
067                                        capacity = act.getCapacity();
068                                }
069                        }
070                        this.facilityPenalties.put(f.getId(), new FacilityPenalty(capacity, scaleNumberOfPersons, config));
071                }
072                log.info("finished init");
073        }
074
075        /**
076         * Add an arrival event in "FacilityLoad" for every start of an activity
077         * Home activities are excluded.
078         */
079        @Override
080        public void handleEvent(final ActivityStartEvent event) {
081                if (!(event.getActType().startsWith("h") || event.getActType().startsWith("tta"))) {
082                        Id facilityId = event.getFacilityId();
083                        this.facilityPenalties.get(facilityId).getFacilityLoad().addArrival(event.getTime());
084                }
085        }
086
087        /**
088         * Add a departure event in "FacilityLoad" for every ending of an activity
089         * Home activities are excluded
090         */
091        @Override
092        public void handleEvent(final ActivityEndEvent event) {
093                if (!(event.getActType().startsWith("h") || event.getActType().startsWith("tta"))) {
094                        Id facilityId = event.getFacilityId();
095                        this.facilityPenalties.get(facilityId).getFacilityLoad().addDeparture(event.getTime());
096                }
097        }
098
099        public void finish() {
100                log.info("EventsToFacilityLoad start finish() method");
101                Iterator<? extends FacilityPenalty> iter_fp = this.facilityPenalties.values().iterator();
102                while (iter_fp.hasNext()){
103                        FacilityPenalty fp = iter_fp.next();
104                        fp.finish();
105                }
106                log.info("EventsToFacilityLoad end finish() method");
107        }
108
109        @Override
110        public void reset(final int iteration) {
111                log.info("Not really resetting anything here.");
112        }
113
114        public void resetAll(final int iteration) {
115                Iterator<? extends FacilityPenalty> iter_fp = this.facilityPenalties.values().iterator();
116                while (iter_fp.hasNext()){
117                        FacilityPenalty fp = iter_fp.next();
118                        fp.reset();
119                }
120                log.info("EventsToFacilityLoad resetted");
121        }
122
123        public TreeMap<Id, FacilityPenalty> getFacilityPenalties() {
124                return facilityPenalties;
125        }
