021package org.matsim.contrib.locationchoice.timegeography;
022
023import java.util.ArrayList;
024import java.util.Collections;
025import java.util.List;
026import java.util.Random;
027import java.util.TreeMap;
028
029import org.matsim.api.core.v01.Coord;
030import org.matsim.api.core.v01.Scenario;
031import org.matsim.api.core.v01.TransportMode;
032import org.matsim.api.core.v01.network.Network;
033import org.matsim.api.core.v01.population.Activity;
034import org.matsim.api.core.v01.population.Activity;
035import org.matsim.api.core.v01.population.Leg;
036import org.matsim.api.core.v01.population.Plan;
037import org.matsim.api.core.v01.population.PlanElement;
038import org.matsim.contrib.locationchoice.DestinationChoiceConfigGroup;
039import org.matsim.contrib.locationchoice.LocationMutator;
040import org.matsim.contrib.locationchoice.utils.ActivitiesHandler;
041import org.matsim.contrib.locationchoice.utils.PlanUtils;
042import org.matsim.core.network.NetworkUtils;
043import org.matsim.core.population.routes.NetworkRoute;
044import org.matsim.core.population.routes.RouteUtils;
045import org.matsim.core.utils.collections.QuadTree;
046import org.matsim.core.utils.geometry.CoordUtils;
047import org.matsim.facilities.ActivityFacility;
048import org.matsim.facilities.ActivityFacilityImpl;
049
050public class SingleActLocationMutator extends LocationMutator {
051
052        protected int unsuccessfullLC = 0;
053        private final ActivitiesHandler defineFlexibleActivities;
054
055        public SingleActLocationMutator(final Scenario scenario, TreeMap<String, ? extends QuadTree<ActivityFacility>> quad_trees, 
056                        TreeMap<String, ActivityFacilityImpl []> facilities_of_type,
057                        Random random) {
058
059                super(scenario, quad_trees, facilities_of_type, random);
060                this.defineFlexibleActivities = new ActivitiesHandler((DestinationChoiceConfigGroup) scenario.getConfig().getModule("locationchoice"));
061        }
062
063        @Override
064        public final void run(final Plan plan){
065
066                List<Activity> flexibleActivities = this.getFlexibleActivities(plan);
067
068                if (flexibleActivities.size() == 0) {
069                        this.unsuccessfullLC++;
070                        return;
071                }
072                Collections.shuffle(flexibleActivities);
073                Activity actToMove = flexibleActivities.get(0);
074                List<PlanElement> actslegs = plan.getPlanElements();
075                int indexOfActToMove = actslegs.indexOf(actToMove);
076
077                // starting home and ending home are never flexible
078                final Leg legPre = (Leg)actslegs.get(indexOfActToMove -1);
079                final Leg legPost = (Leg)actslegs.get(indexOfActToMove + 1);
080                final Activity actPre = (Activity)actslegs.get(indexOfActToMove - 2);
081                final Activity actPost = (Activity)actslegs.get(indexOfActToMove + 2);
082
083                double travelDistancePre = 0.0;
084                double travelDistancePost = 0.0;
085
086                if (legPre.getMode().compareTo(TransportMode.car) == 0) {
087                        travelDistancePre = RouteUtils.calcDistanceExcludingStartEndLink((NetworkRoute) legPre.getRoute(), this.scenario.getNetwork());
088                }
089                else {
090                        travelDistancePre = CoordUtils.calcEuclideanDistance(actPre.getCoord(), actToMove.getCoord());
091                }
092                if (legPost.getMode().compareTo(TransportMode.car) == 0) {
093                        travelDistancePost = RouteUtils.calcDistanceExcludingStartEndLink((NetworkRoute) legPost.getRoute(), this.scenario.getNetwork());
094                }
095                else {
096                        travelDistancePost = CoordUtils.calcEuclideanDistance(actToMove.getCoord(), actPost.getCoord());
097                }
098                double radius =  0.5 * (travelDistancePre + travelDistancePost);
099
100                if (Double.isNaN(radius)) {
101                        this.unsuccessfullLC++;
102                        return;
103                }
104
105                if (!this.modifyLocation((Activity) actToMove, actPre.getCoord(), actPost.getCoord(), radius)) {
106                        this.unsuccessfullLC++;
107                        return;
108                }
109                PlanUtils.resetRoutes(plan);
110        }
111
112        private List<Activity> getFlexibleActivities(final Plan plan) {
113                List<Activity> flexibleActivities;
114                flexibleActivities = this.defineFlexibleActivities.getFlexibleActivities(plan);
115                return flexibleActivities;
116        }
117
118        protected final boolean modifyLocation(Activity act, Coord startCoord, Coord endCoord, double radius) {
119                double midPointX = (startCoord.getX() + endCoord.getX()) / 2.0;
120                double midPointY = (startCoord.getY() + endCoord.getY()) / 2.0;
121                ArrayList<ActivityFacility> facilitySet =
122                                (ArrayList<ActivityFacility>) this.quadTreesOfType.get(this.defineFlexibleActivities.getConverter().convertType(act.getType())).
123                                                getDisk(midPointX, midPointY, radius);
124
125                ActivityFacility facility = null;
126                if (facilitySet.size() > 1) {
127                        facility = facilitySet.get(super.random.nextInt(facilitySet.size()));
128                }
129                else {
130                        return false;
131                }
132                act.setFacilityId(facility.getId());
133                act.setLinkId(NetworkUtils.getNearestLink(((Network) this.scenario.getNetwork()), facility.getCoord()).getId());
134                act.setCoord(facility.getCoord());
135
136                return true;
137        }
138
139        public final int getNumberOfUnsuccessfull() {
140                return this.unsuccessfullLC;
141        }
142
143        public final void resetUnsuccsessfull() {
144                this.unsuccessfullLC = 0;
145        }
146}
