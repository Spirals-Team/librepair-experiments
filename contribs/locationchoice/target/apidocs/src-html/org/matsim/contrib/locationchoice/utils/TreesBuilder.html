041public class TreesBuilder {
042
043        private Network network = null;
044        private static final Logger log = Logger.getLogger(TreesBuilder.class);
045        private Set<String> flexibleTypes = new HashSet<String>();
046        private final DestinationChoiceConfigGroup config;
047        
048        protected TreeMap<String, QuadTree<ActivityFacility>> quadTreesOfType = new TreeMap<String, QuadTree<ActivityFacility>>();
049        protected TreeMap<String, ActivityFacilityImpl []> facilitiesOfType = new TreeMap<String, ActivityFacilityImpl []>();

051        private ActTypeConverter converter = new ActTypeConverter(true);

055                this.flexibleTypes = flexibleTypes;
056                this.network = network;
057                this.config = config;
058        }
059
060        public TreesBuilder(Network network, DestinationChoiceConfigGroup config) {
061                this.network = network;
062                this.config = config;
063                this.initFlexibleTypes();
064        }
065
066        private void initFlexibleTypes() {
067                String types = config.getFlexibleTypes();
068
069                if (!types.equals("null")) {
070                        log.info("Doing location choice for activity types: " + types);
071                        String[] entries = types.split(",", -1);
072                        for (int i = 0; i < entries.length; i++) {
073                                if (!entries[i].trim().equals("null")) {
074                                        this.flexibleTypes.add(this.converter.convertType(entries[i].trim()));
075                                }
076                        }
077                }
078        }
079
080        public void createTrees(ActivityFacilities facilities) {
081                TreeMap<String, TreeMap<Id<ActivityFacility>, ActivityFacility>> treesForTypes = this.createTreesForTypes(facilities);
082                this.createQuadTreesAndArrays(treesForTypes);
083        }
084
085        private TreeMap<String, TreeMap<Id<ActivityFacility>, ActivityFacility>> createTreesForTypes(ActivityFacilities facilities) {
086
087                boolean regionalScenario = false;
088                double radius = 0.0;
089                Node centerNode = null;
090
091                if (this.config.getCenterNode() != null && this.config.getRadius() != null) {
092                        regionalScenario = true;
093                        centerNode = this.network.getNodes().get(Id.create(config.getCenterNode(), Node.class));
094                        radius = config.getRadius();
095                        log.info("Building trees regional scenario");
096                }
097                else {
098                        log.info("Building trees complete scenario");
099                }
100
101                TreeMap<String, TreeMap<Id<ActivityFacility>, ActivityFacility>> trees = new TreeMap<>();
102                // get all types of activities
103                for (ActivityFacility f : facilities.getFacilities().values()) {
104                        Map<String, ? extends ActivityOption> facilityActOpts = f.getActivityOptions();
105
106                        // do not add facility if it is not in region of interest ------------------------
107                        if (regionalScenario && (CoordUtils.calcEuclideanDistance(f.getCoord(), centerNode.getCoord()) > radius)) {
108                                continue;
109                        }
110                        // -------------------------------------------------------------------------------
111
112                        Iterator<? extends ActivityOption> actOpt_it = facilityActOpts.values().iterator();
113                        while (actOpt_it.hasNext()) {
114                                ActivityOption actOpt = actOpt_it.next();
115
116                                // if flexibleTypes is empty we add all types to trees as potentially all types can be relocated
117                                // otherwise we add all types given by flexibleTypes
118                                if (this.flexibleTypes.size() == 0 ||  this.flexibleTypes.contains(this.converter.convertType(actOpt.getType()))) {
119                                        if (!trees.containsKey(this.converter.convertType(actOpt.getType()))) {
120                                                trees.put(this.converter.convertType(actOpt.getType()), new TreeMap<Id<ActivityFacility>, ActivityFacility>());
121                                        }
122                                        trees.get(this.converter.convertType(actOpt.getType())).put(f.getId(), f);
123                                }
124                        }
125                }
126                return trees;
127        }
128
129        private void createQuadTreesAndArrays(TreeMap<String, TreeMap<Id<ActivityFacility>, ActivityFacility>> trees) {
130                Iterator<TreeMap<Id<ActivityFacility>, ActivityFacility>> tree_it = trees.values().iterator();
131                Iterator<String> type_it = trees.keySet().iterator();
132
133                while (tree_it.hasNext()) {
134                        TreeMap<Id<ActivityFacility>, ActivityFacility> tree_of_type = tree_it.next();
135                        String type = type_it.next();
136
137                        // do not construct tree for home and tta act
138//                      if (type.startsWith("h") || type.startsWith("tta")) continue;   // startsWith("h") also removed oder activity types such as "hotel". cdobler, nov'14
139                        if (type.equals("h") || type.equals("home") || type.startsWith("tta")) continue;
140
141                        this.quadTreesOfType.put(this.converter.convertType(type), this.buildFacQuadTree(this.converter.convertType(type), tree_of_type));
142                        this.facilitiesOfType.put(this.converter.convertType(type), tree_of_type.values().toArray(new ActivityFacilityImpl[tree_of_type.size()]));
143                }
144        }
145
146        private QuadTree<ActivityFacility> buildFacQuadTree(String type, TreeMap<Id<ActivityFacility>,ActivityFacility> facilities_of_type) {
147                Gbl.startMeasurement();
148                log.info(" building " + type + " facility quad tree");
149                double minx = Double.POSITIVE_INFINITY;
150                double miny = Double.POSITIVE_INFINITY;
151                double maxx = Double.NEGATIVE_INFINITY;
152                double maxy = Double.NEGATIVE_INFINITY;
153
154                for (final ActivityFacility f : facilities_of_type.values()) {
155                        if (f.getCoord().getX() < minx) { minx = f.getCoord().getX(); }
156                        if (f.getCoord().getY() < miny) { miny = f.getCoord().getY(); }
157                        if (f.getCoord().getX() > maxx) { maxx = f.getCoord().getX(); }
158                        if (f.getCoord().getY() > maxy) { maxy = f.getCoord().getY(); }
159                }
160                minx -= 1.0;
161                miny -= 1.0;
162                maxx += 1.0;
163                maxy += 1.0;
164                System.out.println("        xrange(" + minx + "," + maxx + "); yrange(" + miny + "," + maxy + ")");
165                QuadTree<ActivityFacility> quadtree = new QuadTree<ActivityFacility>(minx, miny, maxx, maxy);
166                for (final ActivityFacility f : facilities_of_type.values()) {
167                        quadtree.put(f.getCoord().getX(),f.getCoord().getY(),f);
168                }
169                log.info("    done");
170                Gbl.printRoundTime();
171                Gbl.printMemoryUsage();
172                return quadtree;
173        }

176                return quadTreesOfType;
177        }

179                return facilitiesOfType;
180        }

183                return converter;
