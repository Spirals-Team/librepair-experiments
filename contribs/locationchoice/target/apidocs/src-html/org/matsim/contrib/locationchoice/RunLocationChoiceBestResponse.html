023package org.matsim.contrib.locationchoice;
024
025import org.matsim.api.core.v01.Scenario;
026import org.matsim.contrib.locationchoice.bestresponse.DestinationChoiceBestResponseContext;
027import org.matsim.contrib.locationchoice.bestresponse.DestinationChoiceInitializer;
028import org.matsim.contrib.locationchoice.bestresponse.scoring.DCScoringFunctionFactory;
029import org.matsim.contrib.locationchoice.facilityload.FacilitiesLoadCalculator;
030import org.matsim.core.config.Config;
031import org.matsim.core.config.ConfigUtils;
032import org.matsim.core.controler.Controler;
033import org.matsim.core.scenario.ScenarioUtils;
034
035public class RunLocationChoiceBestResponse {
036
037    public static void main(String[] args) {
038        Config config = ConfigUtils.loadConfig(args[0], new DestinationChoiceConfigGroup());
039        final Scenario scenario = ScenarioUtils.loadScenario(config);
040        run(scenario);
041    }
042
043    public static void run(Scenario scenario) {
044        DestinationChoiceBestResponseContext dcContext = new DestinationChoiceBestResponseContext(scenario);
045        dcContext.init();
046        DCScoringFunctionFactory dcScoringFunctionFactory = new DCScoringFunctionFactory(scenario, dcContext);
047        DestinationChoiceConfigGroup dccg = (DestinationChoiceConfigGroup) dcContext.getScenario().getConfig().getModule(DestinationChoiceConfigGroup.GROUP_NAME);
048        if (dccg.getPrefsFile() == null && !scenario.getConfig().facilities().getInputFile().equals("null")) {
049            dcScoringFunctionFactory.setUsingConfigParamsForScoring(false);
050        } else {
051            dcScoringFunctionFactory.setUsingConfigParamsForScoring(true);
052        }
053
054        Controler controler = new Controler(scenario);
055        controler.addControlerListener(new DestinationChoiceInitializer(dcContext));
056        if (dccg.getRestraintFcnExp() > 0.0 && dccg.getRestraintFcnFactor() > 0.0) {
057            controler.addControlerListener(new FacilitiesLoadCalculator(dcContext.getFacilityPenalties()));
058        }
059        controler.setScoringFunctionFactory(dcScoringFunctionFactory);
060        controler.run();
061    }
062
063}
