058        public DCActivityScoringFunction(Plan plan, DestinationChoiceContext dcContext) {
059                super(dcContext.getParams());

061                this.facilities = dcContext.getScenario().getActivityFacilities();

066                this.dcContext = dcContext;

068                this.penalty = new Vector<ScoringPenalty>();

071        @Override
072        @Deprecated // needs to be re-designed with delegation instead of inheritance. kai, oct'14
073        public void finish() {
074                super.finish();
075                int activityIndex = -1 ;
076                for (PlanElement pe : this.plan.getPlanElements()) {
077                        if (pe instanceof Activity) {
078                                activityIndex++ ;
079                                this.score += destinationChoiceScoring.getDestinationScore(  (Activity)pe, 
080                                                BestReplyDestinationChoice.useScaleEpsilonFromConfig, activityIndex, this.plan.getPerson().getId() );
081                        }
082                }
083                // reduce score by penalty from capacity restraints
084                Iterator<ScoringPenalty> pen_it = this.penalty.iterator();
085                while (pen_it.hasNext()){
086                        ScoringPenalty penalty = pen_it.next();
087                        this.score -=penalty.getPenalty();
088                }
089                this.penalty.clear();
090        }

092        @Deprecated // needs to be re-designed with delegation instead of inheritance. kai, oct'14
093        protected final double calcActScore(final double arrivalTime, final double departureTime, final Activity act) {
094                
095                if (act.getType().equals(PtConstants.TRANSIT_ACTIVITY_TYPE)) return 0.0;
096
097                double tmpScore = 0.0;
098
099                        /* Calculate the times the agent actually performs the
100                         * activity.  The facility must be open for the agent to
101                         * perform the activity.  If it's closed, but the agent is
102                         * there, the agent must wait instead of performing the
103                         * activity (until it opens).
104                         *
105                         *                                             Interval during which
106                         * Relationship between times:                 activity is performed:
107                         *
108                         *      O________C A~~D  ( 0 <= C <= A <= D )   D...D (not performed)
109                         * A~~D O________C       ( A <= D <= O <= C )   D...D (not performed)
110                         *      O__A+++++C~~D    ( O <= A <= C <= D )   A...C
111                         *      O__A++D__C       ( O <= A <= D <= C )   A...D
112                         *   A~~O++++++++C~~D    ( A <= O <= C <= D )   O...C
113                         *   A~~O+++++D__C       ( A <= O <= D <= C )   O...D
114                         *
115                         * Legend:
116                         *  A = arrivalTime    (when agent gets to the facility)
117                         *  D = departureTime  (when agent leaves the facility)
118                         *  O = openingTime    (when facility opens)
119                         *  C = closingTime    (when facility closes)
120                         *  + = agent performs activity
121                         *  ~ = agent waits (agent at facility, but not performing activity)
122                         *  _ = facility open, but agent not there
123                         *
124                         * assume O <= C
125                         * assume A <= D
126                         */
127
128                        double[] openingInterval = this.getOpeningInterval(act);
129                        double openingTime = openingInterval[0];
130                        double closingTime = openingInterval[1];
131
132                        double activityStart = arrivalTime;
133                        double activityEnd = departureTime;
134
135                        if ((openingTime >=  0) && (arrivalTime < openingTime)) {
136                                activityStart = openingTime;
137                        }
138                        if ((closingTime >= 0) && (closingTime < departureTime)) {
139                                activityEnd = closingTime;
140                        }
141                        if ((openingTime >= 0) && (closingTime >= 0)
142                                        && ((openingTime > departureTime) || (closingTime < arrivalTime))) {
143                                // agent could not perform action
144                                activityStart = departureTime;
145                                activityEnd = departureTime;
146                        }
147                        double duration = activityEnd - activityStart;
148
149                        // disutility if too early
150                        if (arrivalTime < activityStart) {
151                                // agent arrives to early, has to wait
152                                tmpScore += this.params.marginalUtilityOfWaiting_s * (activityStart - arrivalTime);
153                        }
154
155                        // disutility if too late
156
157                        double latestStartTime = (Double)this.prefs.getAttribute(this.plan.getPerson().getId().toString(), "latestStartTime_" + act.getType()); 
158                        if ((latestStartTime >= 0) && (activityStart > latestStartTime)) {
159                                tmpScore += this.params.marginalUtilityOfLateArrival_s * (activityStart - latestStartTime);
160                        }
161
162                        // utility of performing an action, duration is >= 1, thus log is no problem
163                        double typicalDuration = (Double)this.prefs.getAttribute(this.plan.getPerson().getId().toString(), "typicalDuration_" + act.getType());                 
164                        // initialize zero utility durations here for better code readability, because we only need them here
165                        double zeroUtilityDuration;
166                        if (this.zeroUtilityDurations.containsKey(act.getType())) {
167                                zeroUtilityDuration = this.zeroUtilityDurations.get(act.getType());
168                        } else {
169                                zeroUtilityDuration = (typicalDuration / 3600.0) * Math.exp( -10.0 / (typicalDuration / 3600.0) / DCActivityScoringFunction.DEFAULT_PRIORITY);
170                                this.zeroUtilityDurations.put(act.getType(), zeroUtilityDuration);
171                        }
172                        
173                        // disutility if stopping too early
174                        double earliestEndTime = (Double)this.prefs.getAttribute(this.plan.getPerson().getId().toString(), "earliestEndTime_" + act.getType()); 
175                        if ((earliestEndTime >= 0) && (activityEnd < earliestEndTime)) {
176                                tmpScore += this.params.marginalUtilityOfEarlyDeparture_s * (earliestEndTime - activityEnd);
177                        }
178
179                        // disutility if going to away to late
180                        if (activityEnd < departureTime) {
181                                tmpScore += this.params.marginalUtilityOfWaiting_s * (departureTime - activityEnd);
182                        }
183
184                        // disutility if duration was too short
185                        double minimalDuration = (Double)this.prefs.getAttribute(this.plan.getPerson().getId().toString(), "minimalDuration_" + act.getType()); 
186                        if ((minimalDuration >= 0) && (duration < minimalDuration)) {
187                                tmpScore += this.params.marginalUtilityOfEarlyDeparture_s * (minimalDuration - duration);
188                        }
189
190                        if (duration > 0) {
191                                
192                                double utilPerf = this.params.marginalUtilityOfPerforming_s * typicalDuration
193                                                * Math.log((duration / 3600.0) / zeroUtilityDuration);
194                                
195                                double utilWait = this.params.marginalUtilityOfWaiting_s * duration;
196                                tmpScore += Math.max(0, Math.max(utilPerf, utilWait));
197                                
198                                if (this.dcContext.getScaleEpsilon().isFlexibleType(this.converter.convertType(act.getType())) &&
199                                                this.dccg.getRestraintFcnExp() > 0.0 &&
200                                                this.dccg.getRestraintFcnFactor() > 0.0) {
201                                        
202                                                /* Penalty due to facility load: --------------------------------------------
203                                                 * Store the temporary score to reduce it in finish() proportionally
204                                                 * to score and dep. on facility load.
205                                                 * TODO: maybe checking if activity is movable for this person (discussion)
206                                                 */
207                                                this.penalty.add(new ScoringPenalty(activityStart, activityEnd,
208                                                                        this.dcContext.getFacilityPenalties().get(act.getFacilityId()), tmpScore));
209                                                //---------------------------------------------------------------------------
210                                }
211                        } else {
212                                tmpScore += 2*this.params.marginalUtilityOfLateArrival_s*Math.abs(duration);
213                        }
214                return tmpScore;
215        }

218        @Override
219        @Deprecated // needs to be re-designed with delegation instead of inheritance. kai, oct'14
220        protected double[] getOpeningInterval(Activity act) {
221                // openInterval has two values
222                // openInterval[0] will be the opening time
223                // openInterval[1] will be the closing time
224                double[] openInterval = new double[]{Time.UNDEFINED_TIME, Time.UNDEFINED_TIME};
225                boolean foundAct = false;
226                
227                if (act.getType().contains("interaction") || // yyyy might be too loose. kai, feb'16 
228                                (this.converter.convertType(act.getType()).startsWith("h") && this.converter.isV1()) || 
229                                act.getType().equals("home")) {
230                        return openInterval;
231                } // pt interaction and home always open
232                
233                ActivityFacility facility = this.facilities.getFacilities().get(act.getFacilityId());           
234                Iterator<String> facilityActTypeIterator = facility.getActivityOptions().keySet().iterator();
235                String facilityActType = null;
236                Set<OpeningTime> opentimes = null;
237
238                while (facilityActTypeIterator.hasNext() && !foundAct) {
239                        facilityActType = facilityActTypeIterator.next();
240                        if (this.converter.convertType(act.getType()).equals(this.converter.convertType(facilityActType))) { // TODO: check here actType conversions
241                                foundAct = true;
242                                // choose appropriate opentime: either wed, wkday or wk
243                                // if none is given, use undefined opentimes
244                                opentimes = ((ActivityFacilityImpl) facility).getActivityOptions().get(facilityActType).getOpeningTimes();
245                                if (opentimes != null) {
246                                        // ignoring lunch breaks with the following procedure:
247                                        // if there is only one wed/wkday/wk open time interval, use it
248                                        // if there are two or more, use the earliest start time and the latest end time
249                                        openInterval[0] = Double.MAX_VALUE;
250                                        openInterval[1] = Double.MIN_VALUE;
251                                        for (OpeningTime opentime : opentimes) {
252                                                openInterval[0] = Math.min(openInterval[0], opentime.getStartTime());
253                                                openInterval[1] = Math.max(openInterval[1], opentime.getEndTime());
254                                        }
255                                }
256                        }
257                }
258                if (!foundAct) {
259                        throw new RuntimeException("No suitable facility activity type found for activity " + act.getType() + 
260                        " and facility " + facility.getId() + ". Aborting...");
261                }
262                return openInterval;
263        }
