107        public void init() {
108                this.params = new ScoringParameters.Builder(scenario.getConfig().planCalcScore(), scenario.getConfig().planCalcScore().getScoringParameters(null), scenario.getConfig().scenario()).build();
109                this.dccg = ConfigUtils.addOrGetModule( this.scenario.getConfig(), DestinationChoiceConfigGroup.class );
110                ActivitiesHandler defineFlexibleActivities = new ActivitiesHandler(this.dccg);
111                this.scaleEpsilon = defineFlexibleActivities.createScaleEpsilon();
112                this.actTypeConverter = defineFlexibleActivities.getConverter();
113                this.flexibleTypes = defineFlexibleActivities.getFlexibleTypes();
114                
115                this.readOrCreateKVals(dccg.getRandomSeed());
116                this.readFacilitesAttributesAndBetas();
117                this.readPrefs();
118                
119                log.info("dc context initialized");
120        }

122        private void readOrCreateKVals(long seed) {
123                ReadOrCreateKVals computer = new ReadOrCreateKVals(seed, this.scenario);
124                this.arekValsRead = computer.run();
125                ObjectAttributes personsKValues = computer.getPersonsKValues();
126                ObjectAttributes facilitiesKValues = computer.getFacilitiesKValues();
127                
128                this.personIndices = new TObjectIntHashMap<>();
129                this.personsKValuesArray = new double[this.scenario.getPopulation().getPersons().size()];
130                int personIndex = 0;
131                for (Id<Person> personId : this.scenario.getPopulation().getPersons().keySet()) {
132                        this.personIndices.put(personId, personIndex);
133                        this.personsKValuesArray[personIndex] = (Double) personsKValues.getAttribute(personId.toString(), "k");
134                        personIndex++;
135                }               
136                
137                this.facilityIndices = new TObjectIntHashMap<>();
138                this.faciliesWithIndexMap = new HashMap<>();
139                this.facilitiesKValuesArray = new double[this.scenario.getActivityFacilities().getFacilities().size()];
140                int facilityIndex = 0;
141                for (ActivityFacility facility : this.scenario.getActivityFacilities().getFacilities().values()) {
142                        Id<ActivityFacility> facilityId = facility.getId();
143                        
144                        this.facilityIndices.put(facilityId, facilityIndex);
145                        this.facilitiesKValuesArray[facilityIndex] = (Double) facilitiesKValues.getAttribute(facilityId.toString(), "k");
146                        this.faciliesWithIndexMap.put(facilityId, new ActivityFacilityWithIndex(facility, facilityIndex));
147                        
148                        facilityIndex++;
149                }
150        }

152        private void readFacilitesAttributesAndBetas() {
153                String pBetasFileName = this.dccg.getpBetasFile();
154                String fAttributesFileName = this.dccg.getfAttributesFile();
155                if (pBetasFileName != null && fAttributesFileName!= null) {                     
156                        ObjectAttributesXmlReader personsBetasReader = new ObjectAttributesXmlReader(this.personsBetas);
157                        ObjectAttributesXmlReader facilitiesAttributesReader = new ObjectAttributesXmlReader(this.facilitiesAttributes);
158                        try {
159                                personsBetasReader.readFile(pBetasFileName);
160                                facilitiesAttributesReader.readFile(fAttributesFileName);
161                                log.info("reading betas and facilities attributes from: \n"+ pBetasFileName + "\n" + fAttributesFileName);
162                        } catch  (UncheckedIOException e) {
163                                // reading was not successful
164                                log.error("unsuccessful betas and facilities attributes from files!\n" + pBetasFileName + "\n" + fAttributesFileName);
165                        }
166                }
167        }
168        
169        private void readPrefs() {
170                String prefsFileName = this.dccg.getPrefsFile();
171                if (prefsFileName != null) {                    
172                        ObjectAttributesXmlReader prefsReader = new ObjectAttributesXmlReader(this.prefsAttributes);
173                        try {
174                                prefsReader.readFile(prefsFileName);
175                                log.info("reading prefs attributes from: \n"+ prefsFileName);
176                        } catch  (UncheckedIOException e) {
177                                // reading was not successful
178                                log.error("unsuccessful prefs reading from files!\n" + prefsFileName);
179                        }
180                } else {
181                        log.warn("prefs are taken from the config and if available from the desires as there is no preferences file specified \n");
182                        for (ActivityParams activityParams : this.scenario.getConfig().planCalcScore().getActivityParams()) {                           
183                                for (Person p : this.scenario.getPopulation().getPersons().values()) {
184                    prefsAttributes.putAttribute(p.getId().toString(), "typicalDuration_" + activityParams.getActivityType(), activityParams.getTypicalDuration());
185                                        prefsAttributes.putAttribute(p.getId().toString(), "latestStartTime_" + activityParams.getActivityType(), activityParams.getLatestStartTime());
186                                        prefsAttributes.putAttribute(p.getId().toString(), "earliestEndTime_" + activityParams.getActivityType(), activityParams.getEarliestEndTime());
187                                        prefsAttributes.putAttribute(p.getId().toString(), "minimalDuration_" + activityParams.getActivityType(), activityParams.getMinimalDuration());
188                                }
189                        }
190                }
191        }

197        public void cacheQuadTrees(boolean cacheQuadTrees) {
198                this.cacheQuadTrees = cacheQuadTrees;
199                if (!cacheQuadTrees) {
200                        this.quadTreesOfType.clear();
201                        this.facilitiesOfType.clear();
202                }
203        }

205        public Tuple<QuadTree<ActivityFacilityWithIndex>, ActivityFacilityImpl[]> getQuadTreeAndFacilities(String activityType) {
206                if (this.cacheQuadTrees) {
207                        QuadTree<ActivityFacilityWithIndex> quadTree = this.quadTreesOfType.get(activityType);
208                        ActivityFacilityImpl[] facilities = this.facilitiesOfType.get(activityType);
209                        if (quadTree == null || facilities == null) {                           
210                                Tuple<QuadTree<ActivityFacilityWithIndex>, ActivityFacilityImpl[]> tuple = getTuple(activityType);
211                                this.quadTreesOfType.put(activityType, tuple.getFirst());
212                                this.facilitiesOfType.put(activityType, tuple.getSecond());
213                                
214                                return tuple;
215                        } else return new Tuple<QuadTree<ActivityFacilityWithIndex>, ActivityFacilityImpl[]>(quadTree, facilities);
216                } else return getTuple(activityType);
217        }
218        
219        private Tuple<QuadTree<ActivityFacilityWithIndex>, ActivityFacilityImpl[]> getTuple(String activityType) {
220
221                TreesBuilder treesBuilder = new TreesBuilder(CollectionUtils.stringToSet(activityType), this.scenario.getNetwork(), this.dccg);
222                treesBuilder.setActTypeConverter(this.getConverter());
223                treesBuilder.createTrees(scenario.getActivityFacilities());
224                
225                ActivityFacilityImpl[] facilities = treesBuilder.getFacilitiesOfType().get(activityType);
226                
227                /*
228                 * Create a copy of the treesBuilder.getQuadTreesOfType() outcome where the
229                 * ActivityFacility objects are replaced by ActivityFacilityWithIndex objects.
230                 * TODO: let the TreeBuilder use ActivityFacilityWithIndex objects directly?
231                 */
232                QuadTree<ActivityFacilityWithIndex> quadTree = null;
233                
234                QuadTree<ActivityFacility> qt = treesBuilder.getQuadTreesOfType().get(activityType);
235                if (qt != null) {
236                        double minX = qt.getMinEasting();
237                        double maxX = qt.getMaxEasting();
238                        double minY = qt.getMinNorthing();
239                        double maxY = qt.getMaxNorthing();
240                        quadTree = new QuadTree<ActivityFacilityWithIndex>(minX, minY, maxX, maxY);
241                        for (ActivityFacility activityFacility : qt.values()) {
242                                quadTree.put(activityFacility.getCoord().getX(), activityFacility.getCoord().getY(), this.faciliesWithIndexMap.get(activityFacility.getId()));
243                        }                       
244                }
245                
246                return new Tuple<QuadTree<ActivityFacilityWithIndex>, ActivityFacilityImpl[]>(quadTree, facilities);
247        }
