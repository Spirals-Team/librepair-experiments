036
037public class ComputeMaxDCScorePlanAlgo implements PlanAlgorithm {
038        
039        private final String type;
040        private final ActivityFacilityWithIndex[] typedFacilities;
041        private final DestinationScoring scorer;
042        private final DestinationSampler sampler;
043        private final DestinationChoiceContext lcContext;
044        private final Id<Link> dummyLinkId = Id.createLinkId(1);
045        private final DummyActivity dummyActivity = new DummyActivity(this.dummyLinkId); 
046        
047        public ComputeMaxDCScorePlanAlgo(final String type, final ActivityFacilityWithIndex[] typedFacilities,
048                        final DestinationScoring scorer, final DestinationSampler sampler, final DestinationChoiceContext lcContext) {
049                this.type = type;
050                this.typedFacilities = typedFacilities;
051                this.scorer = scorer;
052                this.sampler = sampler;
053                this.lcContext = lcContext;
054        }
055        
056        @Override
057        public void run(Plan plan) {
058                Person p = plan.getPerson();
059                int personIndex = this.lcContext.getPersonIndex(p.getId());
060                double maxDCScore = 0.0;                
061                /*
062                 * Find the max dc score of all activities of this.type.
063                 * Different activities in a plan have now different epsilons!
064                 * TODO: Future improvement: store max dc score *per* activity -> computational gain 
065                 */                     
066                int activityIndex = -1 ;
067                for (PlanElement pe : plan.getPlanElements()) {
068                        if (pe instanceof Activity) {
069                                activityIndex++ ;
070                                if (this.lcContext.getConverter().convertType(((Activity) pe).getType()).equals(type)) {
071                        
072                                        for (ActivityFacilityWithIndex f : this.typedFacilities) {
073                                                //check if facility is sampled
074                                                int facilityIndex = f.getArrayIndex();
075                                                if (!this.sampler.sample(facilityIndex, personIndex)) continue;
076                                                
077//                                              ActivityImpl act = new ActivityImpl(type, this.dummyLinkId);
078//                                              act.setFacilityId(f.getId());
079                                                this.dummyActivity.setType(type);
080                                                this.dummyActivity.setFacilityId(f.getId());
081                                                
082                                                double epsilonScaleFactor = 1.0; // no scaling back needed here anymore
083//                                              double dcScore = scorer.getDestinationScore(act, epsilonScaleFactor, activityIndex, p.getId());
084                                                double dcScore = scorer.getDestinationScore(this.dummyActivity, epsilonScaleFactor, activityIndex, p.getId());
085                                                                                
086                                                if (dcScore > maxDCScore) {
087                                                        maxDCScore = dcScore;
088                                                }
089                                        }
090                                }
091                        }
092                }
093                p.getCustomAttributes().put(this.type, maxDCScore);     
094        }
095        
096        /*
097         * Creating new ActivityImpl objects over and over again in the run(...) method is very expensive compared to the
098         * other stuff performed in that method (seems that "type.intern()", which is called in the constructor, is the problem). 
099         * The object is passed over to scorer.getDestinationScore(...) where only its type and facilityId are required. 
100         * Therefore, we create a DummyActivity object and re-use the existing string object from the input activity.
101         * 
102         * Tested this with a sample of 6.9k agents and 2.3m facilities and found a computation time reduction of 30%.
103         * 
104         * cdobler, oct'15
105         */
106        private static final class DummyActivity implements Activity {
107
108                private String type = null;
109                private Id<ActivityFacility> facilityId = null;
110                private final Id<Link> linkId;
111                
112                public DummyActivity(Id<Link> linkId) { this.linkId = linkId; }
113                
114                @Override
115                public double getEndTime() { return 0; }
116
117                @Override
118                public void setEndTime(double seconds) { }
119
120                @Override
121                public String getType() { return this.type; }
122
123                @Override
124                public void setType(String type) { this.type = type; }
125
126                @Override
127                public Coord getCoord() { return null; }
128
129                @Override
130                public double getStartTime() { return 0; }
131
132                @Override
133                public void setStartTime(double seconds) { }
134
135                @Override
136                public double getMaximumDuration() { return 0; }
137
138                @Override
139                public void setMaximumDuration(double seconds) { }
140
141                @Override
142                public Id<Link> getLinkId() { return this.linkId; }
143
144                @Override
145                public Id<ActivityFacility> getFacilityId() { return this.facilityId; }
146                
147                public void setFacilityId(Id<ActivityFacility> facilityId) { this.facilityId = facilityId; }
148
149                @Override
150                public void setLinkId(Id<Link> id) {
151                        // TODO Auto-generated method stub
152                        throw new RuntimeException("not implemented") ;
153                }
154
155                @Override
156                public void setCoord(Coord coord) {
157                        // TODO Auto-generated method stub
158                        throw new RuntimeException("not implemented") ;
159                }
160
161                @Override
162                public Attributes getAttributes() {
163                        throw new RuntimeException("not implemented") ;
164                }
165        }
