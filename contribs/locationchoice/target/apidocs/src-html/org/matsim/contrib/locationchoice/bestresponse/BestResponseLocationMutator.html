157        private void handleActivities(final Plan plan, final Plan bestPlan, final int personIndex) {
158                ApproximationLevel travelTimeApproximationLevel = this.dccg.getTravelTimeApproximationLevel();
159
160                int actlegIndex = -1;
161                for (PlanElement pe : plan.getPlanElements()) {
162                        actlegIndex++;
163                        if (pe instanceof Activity) {
164                                String actType = ((Activity) plan.getPlanElements().get(actlegIndex)).getType();
165//                              System.err.println("looking at act of type: " + actType ) ;
166                                if (actlegIndex > 0 && this.scaleEpsilon.isFlexibleType(actType)) {
167
168                                        List<? extends PlanElement> actslegs = plan.getPlanElements();
169                                        final Activity actToMove = (Activity) pe;
170                                        final Activity actPre = (Activity) actslegs.get(actlegIndex - 2);
171
172                                        final Activity actPost = (Activity) actslegs.get(actlegIndex + 2);                                      
173                                        double distanceDirect = CoordUtils.calcEuclideanDistance(actPre.getCoord(), actPost.getCoord());
174                                        double maximumDistance = this.convertEpsilonIntoDistance(plan.getPerson(), 
175                                                        this.actTypeConverter.convertType(actToMove.getType()));
176
177                                        double maxRadius = (distanceDirect +  maximumDistance) / 2.0;
178
179                                        double x = (actPre.getCoord().getX() + actPost.getCoord().getX()) / 2.0;
180                                        double y = (actPre.getCoord().getY() + actPost.getCoord().getY()) / 2.0;
181                                        Coord center = new Coord(x, y);
182
183                                        ChoiceSet cs = createChoiceSetFromCircle(plan, personIndex, travelTimeApproximationLevel, actToMove, maxRadius, center);
184                                        
185//                                      System.err.println("ChoiceSet cs:\n" + cs.toString() ) ;
186
187                                        // **************************************************
188                                        // maybe repeat this a couple of times
189                                        // yy why? kai, feb'13
190
191                                        final Id<ActivityFacility> choice = cs.getWeightedRandomChoice(
192                                                        actlegIndex, this.scoringFunctionFactory, plan, this.getTripRouter(), this.lcContext.getPersonsKValuesArray()[personIndex],
193                                                        this.forwardMultiNodeDijkstra, this.backwardMultiNodeDijkstra, this.iteration);
194
195                                        this.setLocation(actToMove, choice);
196                                        
197//                                      printTentativePlanToConsole(plan);
198
199                                        // the change was done to "plan".  Now check if we want to copy this to bestPlan:
200                                        this.evaluateAndAdaptPlans(plan, bestPlan, cs, this.scoringFunctionFactory);
201
202                                        // yyyy if I understand this correctly, this means that, if the location change is accepted, all subsequent locachoice 
203                                        // optimization attempts will be run with that new location.  kai, jan'13
204                                        
205                                        // **************************************************
206                                }
207                        }
208                }               
209        }

211        private ChoiceSet createChoiceSetFromCircle(Plan plan, int personIndex,
212                        final ApproximationLevel travelTimeApproximationLevel,
213                        final Activity actToMove, double maxRadius, Coord center) {
214
215                ChoiceSet cs = new ChoiceSet(travelTimeApproximationLevel, this.scenario, this.nearestLinks, this.teleportedModeSpeeds, this.beelineDistanceFactors);
216
217                final String convertedType = this.actTypeConverter.convertType(actToMove.getType());
218                Collection<ActivityFacilityWithIndex> list = this.quadTreesOfType.get(convertedType).getDisk(center.getX(), center.getY(), maxRadius);
219                
220                for (ActivityFacilityWithIndex facility : list) {
221//                      int facilityIndex = this.lcContext.getFacilityIndex(facility.getId());
222                        int facilityIndex = facility.getArrayIndex();
223                        if (this.sampler.sample(facilityIndex, personIndex)) { 
224                                
225                                // only add destination if it can be reached with the chosen mode
226                                Leg previousLeg = PlanUtils.getPreviousLeg(plan, actToMove);
227                                String mode = previousLeg.getMode();    
228                                
229                                Id<Link> linkId = null;
230                                // try to get linkId from facility, else get it from act. other options not allowed!
231                                if (facility.getLinkId() != null) {
232                                        linkId = facility.getLinkId();
233                                }
234                                else {
235                                        linkId = actToMove.getLinkId();
236                                }
237                                // TODO: solve this generic. for that we need info from the config, which modes are actually teleported.
238                                if (this.lcContext.getScenario().getNetwork().getLinks().get(linkId).getAllowedModes().contains(mode) || 
239                                                mode.equals(TransportMode.bike) || 
240                                                mode.equals(TransportMode.walk) ||
241                                                mode.equals(TransportMode.transit_walk) ||
242                                                mode.equals(TransportMode.other)) {
243                                        cs.addDestination(facility.getId());
244                                }       
245                        }
246                }
247                
248                return cs;
249        }

273//              System.err.println("expected score of new plan is: " + score ) ;
274//              System.err.println("existing score of old plan is: " + bestPlanSoFar.getScore() ) ;
275                if (score > bestPlanSoFar.getScore() + 0.0000000000001) {
276//                      plan.setScore(score);
277//                      bestPlanSoFar.getPlanElements().clear();
278//                      ((PlanImpl) bestPlanSoFar).copyFrom(plan);
279                        Plan source = plan;
280                        Plan destination = bestPlanSoFar;
281                        PlanUtils.copyFrom(source, destination);
282                }
283        }

285        private double computeScoreAndAdaptPlan(Plan plan, ChoiceSet cs, ScoringFunctionFactory scoringFunction) {
286                // yyyy why is all this plans copying necessary?  kai, jan'13
287                // looked into it but could not find a reason. Removed it and tests are still fine. cdobler, oct'15
288
289//              Plan planTmp = plan;
290
291                Plan planTmp = null;
292                if (this.dccg.getInternalPlanDataStructure() == InternalPlanDataStructure.planImpl) {
293                        planTmp = PopulationUtils.createPlan(plan.getPerson());
294                        PlanUtils.copyFrom(plan, planTmp);
295                } else if (this.dccg.getInternalPlanDataStructure() == InternalPlanDataStructure.lcPlan) {
296                        planTmp = new LCPlan(plan);
297                }
298
299                final double score =
300                                cs.adaptAndScoreTimes(
301                                                plan,
302                                                planTmp,
303                                                scoringFunction,
304                                                this.getTripRouter(),
305                                                DestinationChoiceConfigGroup.ApproximationLevel.completeRouting );
306
307                PlanUtils.copyPlanFieldsToFrom(plan, planTmp);  // copy( to, from )
308                return score;
309        }

311        /**
312         * Conversion of the "frozen" logit model epsilon into a distance.
313         */
314        private double convertEpsilonIntoDistance(Person person, String type) {
315                double maxDCScore = 0.0;
316                double scale = this.scaleEpsilon.getEpsilonFactor(type);                
317                maxDCScore = (Double) this.personsMaxDCScoreUnscaled.getAttribute(person.getId().toString(), type);
318                maxDCScore *= scale; // apply the scale factors given in the config file
319
320                /* 
321                 * here one could do a much more sophisticated calculation including time use and travel speed estimations (from previous iteration)
322                 */
323                double travelSpeedCrowFly = this.dccg.getTravelSpeed_car();
324                double betaTime = this.scenario.getConfig().planCalcScore().getModes().get(TransportMode.car).getMarginalUtilityOfTraveling();
325//              if ( Boolean.getBoolean(this.scenario.getConfig().vspExperimental().getValue(VspExperimentalConfigKey.isUsingOpportunityCostOfTimeForLocationChoice)) ) {
326                if ( this.scenario.getConfig().vspExperimental().isUsingOpportunityCostOfTimeForLocationChoice() ) {
327                        betaTime -= this.scenario.getConfig().planCalcScore().getPerforming_utils_hr() ;
328                        // needs to be negative (I think) since AH uses this as a cost parameter. kai, jan'13
329                }
330                double maxTravelTime = Double.MAX_VALUE;
331                if (betaTime != 0.0) {
332                        if ( betaTime >= 0. ) {
333                                throw new RuntimeException("betaTime >= 0 in location choice; method not designed for this; aborting ...") ;
334                        }
335                        maxTravelTime = Math.abs(maxDCScore / (-1.0 * betaTime) * 3600.0); //[s] // abs used for the case when somebody defines beta > 0
336                        // yy maybe one can still used it with betaTime>0 (i.e. traveling brings utility), but taking the "abs" of it is almost certainly not the
337                        // correct way.  kai, jan'13
338                }
339                // distance linear
340                double maxDistance = travelSpeedCrowFly * maxTravelTime; 
341
342                // define a maximum distance choice set manually
343                if (this.dccg.getMaxDistanceDCScore() > 0.0) {
344                        maxDistance = this.dccg.getMaxDistanceDCScore();
345                }
346                return maxDistance;
347        }
