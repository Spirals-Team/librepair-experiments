021package org.matsim.contrib.locationchoice.timegeography;
022
023import java.util.List;
024import java.util.Vector;
025
026import org.matsim.api.core.v01.population.Activity;
027import org.matsim.api.core.v01.population.Leg;
028
029public class ManageSubchains {
030
031        private List<SubChain> subChains = new Vector<SubChain>();
032        private int subChainIndex = -1;
033        boolean chainStarted = false;
034        boolean secondaryActFound = false;
035
036        private double ttBudget = 0.0;
037        private double totalTravelDistance = 0.0;
038
039        public void secondaryActivityFound(Activity act, Leg leg) {
040                /*
041                 * No plan starts with secondary activity!
042                 */
043                this.subChains.get(subChainIndex).defineMode(leg.getMode());
044                this.subChains.get(subChainIndex).addAct(act);
045                this.secondaryActFound = true;
046                this.ttBudget += leg.getTravelTime();
047                this.totalTravelDistance += leg.getRoute().getDistance();
048        }
049
050        public void primaryActivityFound(Activity act, Leg leg) {
051                /*
052                 * close chain
053                 */
054                if (chainStarted) {
055                        if (secondaryActFound) {
056                                this.subChains.get(subChainIndex).setTotalTravelDistance(this.totalTravelDistance);
057                                this.subChains.get(subChainIndex).setTtBudget(this.ttBudget);
058                                this.subChains.get(subChainIndex).setEndCoord(act.getCoord());
059                                this.subChains.get(subChainIndex).setLastPrimAct(act);
060                        }
061                        else {
062                                this.subChains.remove(subChainIndex);
063                                this.subChainIndex--;
064                        }
065                }
066
067                // it is not the second home act
068                if (!(leg == null)) {
069                        //open chain
070                        this.subChains.add(new SubChain());
071                        this.subChainIndex++;
072                        this.subChains.get(subChainIndex).setFirstPrimAct(act);
073                        this.subChains.get(subChainIndex).setStartCoord(act.getCoord());
074                        this.chainStarted = true;
075                        this.secondaryActFound = false;
076                        this.ttBudget = leg.getTravelTime();
077                        this.totalTravelDistance = leg.getRoute().getDistance();
078                        this.subChains.get(subChainIndex).defineMode(leg.getMode());
079                }
080        }
081
082        public List<SubChain> getSubChains() {
083                return subChains;
084        }
085
086        public void setSubChains(List<SubChain> subChains) {
087                this.subChains = subChains;
088        }
089}
