021package org.matsim.contrib.locationchoice;
022
023import java.util.Random;
024import java.util.TreeMap;
025
026import org.matsim.api.core.v01.Scenario;
027import org.matsim.api.core.v01.network.Network;
028import org.matsim.contrib.locationchoice.utils.ActivitiesHandler;
029import org.matsim.contrib.locationchoice.utils.TreesBuilder;
030import org.matsim.core.population.algorithms.PlanAlgorithm;
031import org.matsim.core.utils.collections.QuadTree;
032import org.matsim.facilities.ActivityFacilities;
033import org.matsim.facilities.ActivityFacility;
034import org.matsim.facilities.ActivityFacilityImpl;
035
036public abstract class LocationMutator implements PlanAlgorithm {
037
038        protected TreeMap<String, ? extends QuadTree<ActivityFacility>> quadTreesOfType;
039
040        // avoid costly call of .toArray() within handlePlan() (System.arraycopy()!)
041        protected TreeMap<String, ActivityFacilityImpl []> facilitiesOfType;
042        protected ActivitiesHandler defineFlexibleActivities;
043        protected boolean locationChoiceBasedOnKnowledge = true;
044        protected final Random random;
045
046        protected final Scenario scenario;
047        protected final DestinationChoiceConfigGroup dccg;
048
049        // ----------------------------------------------------------
050
051        public LocationMutator(final Scenario scenario, Random random) {
052                this.dccg = (DestinationChoiceConfigGroup) scenario.getConfig().getModule(DestinationChoiceConfigGroup.GROUP_NAME);
053                this.defineFlexibleActivities = new ActivitiesHandler(this.dccg);
054                this.quadTreesOfType = new TreeMap<String, QuadTree<ActivityFacility>>();
055                this.facilitiesOfType = new TreeMap<String, ActivityFacilityImpl []>();
056                this.scenario = scenario;
057                this.random = random;
058                this.initLocal(scenario.getNetwork());
059        }
060
061        public LocationMutator(Scenario scenario, TreeMap<String, ? extends QuadTree<ActivityFacility>> quad_trees,
062                        TreeMap<String, ActivityFacilityImpl []> facilities_of_type, Random random) {
063                this.dccg = (DestinationChoiceConfigGroup) scenario.getConfig().getModule(DestinationChoiceConfigGroup.GROUP_NAME);
064                this.defineFlexibleActivities = new ActivitiesHandler(this.dccg);
065                this.quadTreesOfType = quad_trees;
066                this.facilitiesOfType = facilities_of_type;
067                if (this.defineFlexibleActivities.getFlexibleTypes().size() > 0) {
068                        locationChoiceBasedOnKnowledge = false;
069                }
070                this.scenario = scenario;
071                this.random = random;
072        }
073        
074        private void initLocal(final Network network) {
075
076                if (this.defineFlexibleActivities.getFlexibleTypes().size() > 0) {
077                        locationChoiceBasedOnKnowledge = false;
078                }
079                this.initTrees(scenario.getActivityFacilities());
080        }
081
082        /**
083         * Initialize the quadtrees of all available activity types
084         */
085        private void initTrees(ActivityFacilities facilities) {
086                TreesBuilder treesBuilder = new TreesBuilder(this.scenario.getNetwork(), (DestinationChoiceConfigGroup) this.scenario.getConfig().getModule(DestinationChoiceConfigGroup.GROUP_NAME));
087                treesBuilder.createTrees(facilities);
088                this.facilitiesOfType = treesBuilder.getFacilitiesOfType();
089                this.quadTreesOfType = treesBuilder.getQuadTreesOfType();
090        }
091
092}
