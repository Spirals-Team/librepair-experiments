020package org.matsim.contrib.locationchoice.bestresponse;
021
022import org.matsim.contrib.locationchoice.DestinationChoiceConfigGroup;
023
024/**
025 * Chooses a stable sample of facilities for every person 
026 * 
027 * At the moment it is only working for powers of 10 in the range [0..100]
028 * The sample must remain stable over the course of iterations. Thus, person as well as facility k values
029 * are used. 
030 *              
031 * (Conditional) probability (in %) of a facility k-value matching the given person k-value is 
032 * this.samplePercent   
033 * example: this.samplePercent = 10;
034 * person i has a personValue k between 0 and 9
035 * probability that facilityValue is k is expected 0.1 (or 10%)
036 * 
037 */
038public class DestinationSampler {
039        
040        private double[] facilitiesKValues;
041        private double[] personsKValues;
042        double samplePercent = 100.0;
043        
044        public DestinationSampler(double[] personsKValues, double[] facilitiesKValues, DestinationChoiceConfigGroup module) {
045                this.facilitiesKValues = facilitiesKValues;
046                this.personsKValues = personsKValues;
047                this.samplePercent = module.getDestinationSamplePercent();
048        }
049        
050        public boolean sample(int facilityIndex, int personIndex) { 
051                
052                if (Math.ceil(this.samplePercent) == 100) return true;          
053                                
054                // assign the person randomly but frozen a number between 0 and 100/samplePercent:
055                int personValue = (int)Math.floor(100.0 / samplePercent * this.personsKValues[personIndex]);
056                
057                // assign the facility randomly but frozen a number between 0 and 100/samplePercent:
058                int facilityValue = (int)Math.floor(100.0 / samplePercent * this.facilitiesKValues[facilityIndex]);
059                
060                // return true if the facility has the same number as the person. This will happen for samplePercent of all facilities:
061                return (facilityValue == personValue);          
062        }
063}
