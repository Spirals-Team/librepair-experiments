023

029import org.matsim.contrib.locationchoice.bestresponse.BestResponseLocationMutator;
030import org.matsim.contrib.locationchoice.bestresponse.DestinationChoiceContext;

051import org.matsim.facilities.ActivityFacilityImpl;
052import org.matsim.utils.objectattributes.ObjectAttributes;
053
054import javax.inject.Provider;
055
056public class BestReplyDestinationChoice extends AbstractMultithreadedModule {
057
058    private static final Logger log = Logger.getLogger(BestReplyDestinationChoice.class);
059        private final Provider<TripRouter> tripRouterProvider;
060
061    private DestinationChoiceConfigGroup dccg;
062        private ObjectAttributes personsMaxEpsUnscaled;
063        private DestinationSampler sampler;
064        protected TreeMap<String, QuadTree<ActivityFacilityWithIndex>> quadTreesOfType = new TreeMap<String, QuadTree<ActivityFacilityWithIndex>>();
065        protected TreeMap<String, ActivityFacilityImpl []> facilitiesOfType = new TreeMap<String, ActivityFacilityImpl []>();
066        private final Scenario scenario;
067        private DestinationChoiceContext lcContext;
068        private HashSet<String> flexibleTypes;
069        private final LeastCostPathCalculatorFactory forwardMultiNodeDijsktaFactory;
070        private final LeastCostPathCalculatorFactory backwardMultiNodeDijsktaFactory;
071        private final Map<Id<ActivityFacility>, Id<Link>> nearestLinks;
072
073        public static double useScaleEpsilonFromConfig = -99.0;
074        private ScoringFunctionFactory scoringFunctionFactory;
075        private Map<String, TravelTime> travelTimes;
076        private Map<String, TravelDisutilityFactory> travelDisutilities;
077
078        public BestReplyDestinationChoice(Provider<TripRouter> tripRouterProvider, DestinationChoiceContext lcContext, ObjectAttributes personsMaxDCScoreUnscaled, ScoringFunctionFactory scoringFunctionFactory, Map<String, TravelTime> travelTimes, Map<String, TravelDisutilityFactory> travelDisutilities) {
079                super(lcContext.getScenario().getConfig().global());
080                this.tripRouterProvider = tripRouterProvider;
081                this.scoringFunctionFactory = scoringFunctionFactory;
082                this.travelTimes = travelTimes;
083                this.travelDisutilities = travelDisutilities;
084
085                this.dccg = (DestinationChoiceConfigGroup) lcContext.getScenario().getConfig().getModule(DestinationChoiceConfigGroup.GROUP_NAME);
086                if (!DestinationChoiceConfigGroup.Algotype.bestResponse.equals(this.dccg.getAlgorithm())) {
087                        throw new RuntimeException("wrong class for selected location choice algorithm type; aborting ...") ;
088                }
089                this.lcContext = lcContext;
090                this.scenario = lcContext.getScenario();
091                this.personsMaxEpsUnscaled = personsMaxDCScoreUnscaled;
092                this.forwardMultiNodeDijsktaFactory = new FastMultiNodeDijkstraFactory(true);
093                this.backwardMultiNodeDijsktaFactory = new BackwardFastMultiNodeDijkstraFactory(true);
094                
095                // create cache which is used in ChoiceSet
096                // instead of just the nearest link we probably should check whether the facility is attached to a link? cdobler, oct'14
097                this.nearestLinks = new HashMap<>();
098                for (ActivityFacility facility : this.scenario.getActivityFacilities().getFacilities().values()) {
099                        if (facility.getLinkId() != null)
100                                this.nearestLinks.put(facility.getId(), facility.getLinkId());
101                        else
102                                this.nearestLinks.put(facility.getId(), NetworkUtils.getNearestLink(scenario.getNetwork(),
103                                                facility.getCoord()).getId());
104                }
105                
106                initLocal();
107        }
108
109        private void initLocal() {
110                this.flexibleTypes = this.lcContext.getFlexibleTypes();         
111//              ((NetworkImpl) this.scenario.getNetwork()).connect();
112                this.initTrees(this.scenario.getActivityFacilities(), this.dccg);
113                this.sampler = new DestinationSampler(
114                                this.lcContext.getPersonsKValuesArray(), 
115                                this.lcContext.getFacilitiesKValuesArray(), 
116                                this.dccg);
117        }
118
119        /**
120         * Initialize the quadtrees of all available activity types
121         */
122        private void initTrees(ActivityFacilities facilities, DestinationChoiceConfigGroup config) {
123                log.info("Doing location choice for activities: " + this.flexibleTypes.toString());
124                
125                for (String flexibleType : this.flexibleTypes) {
126                        Tuple<QuadTree<ActivityFacilityWithIndex>, ActivityFacilityImpl[]> tuple = this.lcContext.getQuadTreeAndFacilities(flexibleType);
127                        this.quadTreesOfType.put(flexibleType, tuple.getFirst());
128                        this.facilitiesOfType.put(flexibleType, tuple.getSecond());
129                }
130        }
131
132        @Override
133        protected final void beforeFinishReplanningHook() {
134                Gbl.printMemoryUsage();
135        }
136
137        @Override
138        public final PlanAlgorithm getPlanAlgoInstance() {
139                
140                ReplanningContext replanningContext = this.getReplanningContext();
141                
142                MultiNodeDijkstra forwardMultiNodeDijkstra = (MultiNodeDijkstra) this.forwardMultiNodeDijsktaFactory.createPathCalculator(this.scenario.getNetwork(), 
143                                travelDisutilities.get(TransportMode.car).createTravelDisutility(travelTimes.get(TransportMode.car)), travelTimes.get(TransportMode.car));
144
145                BackwardFastMultiNodeDijkstra backwardMultiNodeDijkstra = (BackwardFastMultiNodeDijkstra) this.backwardMultiNodeDijsktaFactory.createPathCalculator(
146                                this.scenario.getNetwork(), travelDisutilities.get(TransportMode.car).createTravelDisutility(travelTimes.get(TransportMode.car)), travelTimes.get(TransportMode.car));
147                
148                // this one corresponds to the "frozen epsilon" paper(s)
149                // the random number generators are re-seeded anyway in the dc module. So we do not need a MatsimRandom instance here
150
151                TripRouter tripRouter = tripRouterProvider.get();
152                int iteration = replanningContext.getIteration();
153                
154                return new BestResponseLocationMutator(this.quadTreesOfType, this.facilitiesOfType, this.personsMaxEpsUnscaled, 
155                                this.lcContext, this.sampler, tripRouter, forwardMultiNodeDijkstra, backwardMultiNodeDijkstra, scoringFunctionFactory, iteration, this.nearestLinks);
156        }
157}
