057import javax.inject.Provider;
058
059public class BestReplyDestinationChoice extends AbstractMultithreadedModule {
060
061    private static final Logger log = Logger.getLogger(BestReplyDestinationChoice.class);
062        private final Provider<TripRouter> tripRouterProvider;
063
064    private DestinationChoiceConfigGroup dccg;
065        private ObjectAttributes personsMaxEpsUnscaled;
066        private DestinationSampler sampler;
067        protected TreeMap<String, QuadTree<ActivityFacilityWithIndex>> quadTreesOfType = new TreeMap<String, QuadTree<ActivityFacilityWithIndex>>();
068        protected TreeMap<String, ActivityFacilityImpl []> facilitiesOfType = new TreeMap<String, ActivityFacilityImpl []>();
069        private final Scenario scenario;
070        private DestinationChoiceBestResponseContext lcContext;
071        private HashSet<String> flexibleTypes;
072        private final LeastCostPathCalculatorFactory forwardMultiNodeDijsktaFactory;
073        private final LeastCostPathCalculatorFactory backwardMultiNodeDijsktaFactory;
074        private final Map<Id<ActivityFacility>, Id<Link>> nearestLinks;
075
076        public static double useScaleEpsilonFromConfig = -99.0;
077        private ScoringFunctionFactory scoringFunctionFactory;
078        private Map<String, TravelTime> travelTimes;
079        private Map<String, TravelDisutilityFactory> travelDisutilities;
080
081        public BestReplyDestinationChoice(Provider<TripRouter> tripRouterProvider, DestinationChoiceBestResponseContext lcContext, ObjectAttributes personsMaxDCScoreUnscaled, ScoringFunctionFactory scoringFunctionFactory, Map<String, TravelTime> travelTimes, Map<String, TravelDisutilityFactory> travelDisutilities) {
082                super(lcContext.getScenario().getConfig().global());
083                this.tripRouterProvider = tripRouterProvider;
084                this.scoringFunctionFactory = scoringFunctionFactory;
085                this.travelTimes = travelTimes;
086                this.travelDisutilities = travelDisutilities;
087
088                this.dccg = (DestinationChoiceConfigGroup) lcContext.getScenario().getConfig().getModule(DestinationChoiceConfigGroup.GROUP_NAME);
089                if (!DestinationChoiceConfigGroup.Algotype.bestResponse.equals(this.dccg.getAlgorithm())) {
090                        throw new RuntimeException("wrong class for selected location choice algorithm type; aborting ...") ;
091                }
092                this.lcContext = lcContext;
093                this.scenario = lcContext.getScenario();
094                this.personsMaxEpsUnscaled = personsMaxDCScoreUnscaled;
095                this.forwardMultiNodeDijsktaFactory = new FastMultiNodeDijkstraFactory(true);
096                this.backwardMultiNodeDijsktaFactory = new BackwardFastMultiNodeDijkstraFactory(true);
097                
098                // create cache which is used in ChoiceSet
099                // instead of just the nearest link we probably should check whether the facility is attached to a link? cdobler, oct'14
100                this.nearestLinks = new HashMap<>();
101                for (ActivityFacility facility : this.scenario.getActivityFacilities().getFacilities().values()) {
102                        if (facility.getLinkId() != null)
103                                this.nearestLinks.put(facility.getId(), facility.getLinkId());
104                        else
105                                this.nearestLinks.put(facility.getId(), NetworkUtils.getNearestLink(scenario.getNetwork(),
106                                                facility.getCoord()).getId());
107                }
108                
109                initLocal();
110        }
111
112        private void initLocal() {
113                this.flexibleTypes = this.lcContext.getFlexibleTypes();         
114//              ((NetworkImpl) this.scenario.getNetwork()).connect();
115                this.initTrees(this.scenario.getActivityFacilities(), this.dccg);
116                this.sampler = new DestinationSampler(
117                                this.lcContext.getPersonsKValuesArray(), 
118                                this.lcContext.getFacilitiesKValuesArray(), 
119                                this.dccg);
120        }
121
122        /**
123         * Initialize the quadtrees of all available activity types
124         */
125        private void initTrees(ActivityFacilities facilities, DestinationChoiceConfigGroup config) {
126                log.info("Doing location choice for activities: " + this.flexibleTypes.toString());
127                
128                for (String flexibleType : this.flexibleTypes) {
129                        Tuple<QuadTree<ActivityFacilityWithIndex>, ActivityFacilityImpl[]> tuple = this.lcContext.getQuadTreeAndFacilities(flexibleType);
130                        this.quadTreesOfType.put(flexibleType, tuple.getFirst());
131                        this.facilitiesOfType.put(flexibleType, tuple.getSecond());
132                }
133        }
134
135        @Override
136        protected final void beforeFinishReplanningHook() {
137                Gbl.printMemoryUsage();
138        }
139
140        @Override
141        public final PlanAlgorithm getPlanAlgoInstance() {
142                
143                ReplanningContext replanningContext = this.getReplanningContext();
144                
145                MultiNodeDijkstra forwardMultiNodeDijkstra = (MultiNodeDijkstra) this.forwardMultiNodeDijsktaFactory.createPathCalculator(this.scenario.getNetwork(), 
146                                travelDisutilities.get(TransportMode.car).createTravelDisutility(travelTimes.get(TransportMode.car)), travelTimes.get(TransportMode.car));
147
148                BackwardFastMultiNodeDijkstra backwardMultiNodeDijkstra = (BackwardFastMultiNodeDijkstra) this.backwardMultiNodeDijsktaFactory.createPathCalculator(
149                                this.scenario.getNetwork(), travelDisutilities.get(TransportMode.car).createTravelDisutility(travelTimes.get(TransportMode.car)), travelTimes.get(TransportMode.car));
150                
151                // this one corresponds to the "frozen epsilon" paper(s)
152                // the random number generators are re-seeded anyway in the dc module. So we do not need a MatsimRandom instance here
153
154                TripRouter tripRouter = tripRouterProvider.get();
155                int iteration = replanningContext.getIteration();
156                
157                return new BestResponseLocationMutator(this.quadTreesOfType, this.facilitiesOfType, this.personsMaxEpsUnscaled, 
158                                this.lcContext, this.sampler, tripRouter, forwardMultiNodeDijkstra, backwardMultiNodeDijkstra, scoringFunctionFactory, iteration, this.nearestLinks);
159        }
