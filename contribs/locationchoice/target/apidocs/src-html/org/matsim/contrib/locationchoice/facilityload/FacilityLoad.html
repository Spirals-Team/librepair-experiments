021package org.matsim.contrib.locationchoice.facilityload;
022
023//import org.apache.log4j.Logger;
024
025public class FacilityLoad {
026        
027        //private final static Logger log = Logger.getLogger(FacilityLoad.class);
028        
029        private int [] arrivals = null;
030        private int [] departures = null;
031        private double [] load = null;
032        
033        private int numberOfTimeBins = 0;
034        
035        // visitors which are included in the penalty calculation (arrival before 24:00)
036        private double numberOfVisitorsPerDay = 0.0;
037        
038        // including also visitors which arrive after 24:00
039        private double allVisitors = 0.0;
040        
041        private double scaleNumberOfPersons = 1.0;
042        
043        // ----------------------------------------------------------------------
044                
045        FacilityLoad(int numberOfTimeBins, double scaleNumberOfPersons) {
046                this.numberOfTimeBins = numberOfTimeBins;
047                this.arrivals = new int [numberOfTimeBins];
048                this.departures = new int [numberOfTimeBins];
049                this.load = new double [numberOfTimeBins];
050                this.scaleNumberOfPersons = scaleNumberOfPersons;
051
052                for (int i = 0; i < numberOfTimeBins; i++){
053                        this.arrivals[i] = 0;
054                        this.departures[i] = 0;
055                        this.load[i] = 0.0;
056                }
057        }
058        
059        private void calculateFacilityLoad24() {
060                //log.info("calculateFacilityLoad24");
061                int numberOfVisitors = 0;
062                for (int i = 0; i < this.numberOfTimeBins; i++) {
063                        numberOfVisitors += this.arrivals[i];
064                        this.load[i] = numberOfVisitors * this.scaleNumberOfPersons;
065                        numberOfVisitors -= this.departures[i];                 
066                }
067        }
068        
069        public void addArrival(double time) {           
070                this.addToAllVisitors(this.scaleNumberOfPersons);
071                /* We do not handle times > 24h
072                 * We do not care about #arrivals==#departures after the last time bin
073                 */
074                if (time > 24.0*3600.0) {               
075                        return;
076                }
077                int timeBinIndex = this.timeBinIndex(time);             
078                this.arrivals[timeBinIndex] += 1;
079                //log.info("arrival at: " + time + " bin: " + timeBinIndex);
080                this.addToVisitorsPerDay(this.scaleNumberOfPersons);
081        }
082        
083        public void addDeparture(double time) {
084                /* We do not handle times > 24h
085                 * We do not care about #arrivals==#departures after the last time bin
086                 */
087                if (time > 24.0*3600.0) {
088                        return;
089                }
090                int timeBinIndex = this.timeBinIndex(time);
091                this.departures[timeBinIndex]+=1;
092                //log.info("departure at: " + time + " bin: " + timeBinIndex);
093        }
094        
095        public void addToAllVisitors(double scaleNumberOfPersons) {
096                this.allVisitors += scaleNumberOfPersons;
097        }
098        
099        public void addToVisitorsPerDay(double scaleNumberOfPersons) {
100                this.numberOfVisitorsPerDay += scaleNumberOfPersons;
101        }
102
103        public double [] getLoad() {
104                return load;
105        }
106        
107        public double getLoadPerHour(int hour) {
108                double hourlyLoad = 0.0;
109
110                for (int i = 0; i < 4 ; i++) {
111                        int index = hour*4 + i;
112                        hourlyLoad += this.load[index];
113                }
114                return hourlyLoad/4;
115        }
116        
117        public double getNumberOfVisitorsPerDay() {
118                return numberOfVisitorsPerDay;
119        }
120
121        public double getAllVisitors() {
122                return this.allVisitors;
123        }
124
125        public void finish() {
126                //log.info("FacilityLoad finished");
127                this.calculateFacilityLoad24();
128        }
129        
130        public void reset() {
131                for (int i=0; i<this.numberOfTimeBins; i++) {
132                        this.arrivals[i] = 0;
133                        this.departures[i] = 0;
134                        this.load[i] = 0.0;
135                }
136                this.numberOfVisitorsPerDay = 0.0;
137                this.allVisitors = 0.0;
138        }
139        
140        /* 
141         * All values >= 86400s (24h) are merged into the last time bin
142         */
143        public int timeBinIndex(double time) {
144                int lastBinIndex = this.numberOfTimeBins-1;
145                int numberOfBinsPerHour = this.numberOfTimeBins/24;
146                int secondsPerBin = 3600/numberOfBinsPerHour;
147                return Math.min(lastBinIndex, (int)(time/secondsPerBin));
148        }
149}
