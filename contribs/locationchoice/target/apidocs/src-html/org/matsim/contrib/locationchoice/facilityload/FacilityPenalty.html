021package org.matsim.contrib.locationchoice.facilityload;
022
023import org.matsim.contrib.locationchoice.DestinationChoiceConfigGroup;
024
025/*
026 * TODO: (discussion)
027 * 1.   At the moment a facility has activities and an activity has a capacity.
028 *              We have to define this more precise:
029 *              For work and home in the same facility we need two independent capacities
030 *              (-> could be done in Activity)
031 *              But shopping and leisure in a shopping mall with movie theaters additionally needs a
032 *              shared capacity (e.g. parking)
033 *
034 *              At the moment I need only shopping (and leisure) thus I only use one cap.
035 *              (The smallest of all shopping (and leisure) activities of the facility).
036 *
037 * 2.   The mobsim handles times > 24 h
038 *              Facility load has to be handled for hour 0..24 only (according to M.B.)
039 */
040
041public class FacilityPenalty {
042
043        private FacilityLoad facilityLoad;
044        private double capacity = 0.0;
045        private static int numberOfTimeBins = 4*24;
046        private double scaleNumberOfPersons = 1;
047        private double sumCapacityPenaltyFactor = 0.0;
048        private double restraintFcnFactor = 0.0;
049        private double restraintFcnExp = 0.0;
050
051        FacilityPenalty(double minCapacity, double scaleNumberOfPersons, DestinationChoiceConfigGroup config) {
052                this.capacity = minCapacity;
053                this.facilityLoad = new FacilityLoad(FacilityPenalty.numberOfTimeBins, scaleNumberOfPersons);
054                this.scaleNumberOfPersons = scaleNumberOfPersons;
055                this.restraintFcnFactor = config.getRestraintFcnFactor();
056                this.restraintFcnExp = config.getRestraintFcnExp();
057        }
058
059        private double calculateCapPenaltyFactor(int startTimeBinIndex, int endTimeBinIndex) {
060
061                double [] facilityload = this.facilityLoad.getLoad();
062                double capPenaltyFactor = 0.0;
063
064                for (int i=startTimeBinIndex; i<endTimeBinIndex+1; i++) {
065                        if (this.capacity > 0) {
066                        capPenaltyFactor += restraintFcnFactor*Math.pow(
067                                        (facilityload[i]-scaleNumberOfPersons)/(this.capacity), restraintFcnExp);
068                        }
069
070                        /*
071                         * facilityload[i]-scaleNumberOfPersons: being alone in a facility does not
072                         * reduce the utility.
073                         *
074                         * } else: do nothing: is penalized by costs for waiting time
075                         */
076                }
077                capPenaltyFactor /= (endTimeBinIndex-startTimeBinIndex+1);
078                capPenaltyFactor = Math.min(0.5, capPenaltyFactor);
079                this.sumCapacityPenaltyFactor += capPenaltyFactor * this.scaleNumberOfPersons;
080                return capPenaltyFactor;
081        }
082
083        public double getCapacityPenaltyFactor(double startTime, double endTime) {
084
085                if (startTime>24.0*3600.0 && endTime>24.0*3600.0) {
086                        return 0.0;
087                }
088                else if (endTime>24.0*3600.0) {
089                        endTime=24.0*3600.0;
090                }
091
092                int startTimeBinIndex = this.facilityLoad.timeBinIndex(startTime);
093                int endTimeBinIndex = this.facilityLoad.timeBinIndex(endTime);
094                return calculateCapPenaltyFactor(startTimeBinIndex, endTimeBinIndex);
095        }
096
097        public FacilityLoad getFacilityLoad() {
098                return facilityLoad;
099        }
100
101//      public void setFacilityLoad(FacilityLoad facilityLoad) {
102//              this.facilityLoad = facilityLoad;
103//      }
104
105        public double getCapacity() {
106                return capacity;
107        }
108
109
110        public void setCapacity(double capacity) {
111                this.capacity = capacity;
112        }
113
114
115        public double getSumCapacityPenaltyFactor() {
116                return sumCapacityPenaltyFactor;
117        }
118
119
120        public void setSumCapacityPenaltyFactor(double sumCapacityPenaltyFactor) {
121                this.sumCapacityPenaltyFactor = sumCapacityPenaltyFactor;
122        }
123
124        public void finish() {
125                this.facilityLoad.finish();
126        }
127
128        public void reset() {
129                this.sumCapacityPenaltyFactor = 0.0;
130                this.facilityLoad.reset();
131        }
132
133}
