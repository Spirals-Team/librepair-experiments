020package org.matsim.contrib.locationchoice.bestresponse.scoring;
021
022import java.util.Collection;
023import java.util.Random;
024
025import org.matsim.api.core.v01.Id;
026import org.matsim.api.core.v01.population.Activity;
027import org.matsim.api.core.v01.population.Person;
028import org.matsim.contrib.locationchoice.DestinationChoiceConfigGroup;
029import org.matsim.contrib.locationchoice.DestinationChoiceConfigGroup.EpsilonDistributionTypes;
030import org.matsim.contrib.locationchoice.bestresponse.DestinationChoiceContext;
031import org.matsim.core.config.Config;
032import org.matsim.facilities.ActivityFacility;
033import org.matsim.utils.objectattributes.ObjectAttributesUtils;
034
035public class DestinationScoring { 
036        //As the random number generator is re-seeded here anyway, we do not need a rng given from outside!
037        private Random rnd = new Random();
038        private Config config;
039        private DestinationChoiceConfigGroup dccg;
040        private double[] facilitiesKValuesArray;
041        private double[] personsKValuesArray;
042        private ScaleEpsilon scaleEpsilon;
043        private DestinationChoiceContext lcContext;
044                
045        public DestinationScoring(DestinationChoiceContext lcContext) {
046                this.config = lcContext.getScenario().getConfig();
047                this.dccg = (DestinationChoiceConfigGroup) this.config.getModule(DestinationChoiceConfigGroup.GROUP_NAME);
048                this.facilitiesKValuesArray = lcContext.getFacilitiesKValuesArray();
049                this.personsKValuesArray = lcContext.getPersonsKValuesArray();
050                this.scaleEpsilon = lcContext.getScaleEpsilon();
051                this.lcContext = lcContext;
052        }
053        
054        public double getDestinationScore(Activity act, double fVar, int activityIndex, Id<Person> personId) {
055                double score = 0.0;
056                if (this.scaleEpsilon.isFlexibleType(act.getType())) {
057                        /*
058                         * The following will not work if activities are inserted or removed during replanning (e.g. ptInteractionActivities).  kai/mz, oct'14
059                         */
060                        int actIndex =  2*activityIndex ; // "2": retrofitting; this used to be the PlanElementIndex.  One could probably remove the "2",
061                        // but then the test fail because this is equivalent to a different random seed.  So more diligent checking would be needed.
062                        
063                        if (fVar < 0.0) fVar = this.scaleEpsilon.getEpsilonFactor(act.getType());
064                        score += (fVar * this.getEpsilonAlternative(act.getFacilityId(), personId, actIndex));
065                        score += this.getAttributesScore(act.getFacilityId(), personId );
066                }
067                return score;
068        }
069        
070        /*
071         * linear at the moment
072         */
073        private double getAttributesScore(Id<ActivityFacility> facilityId, Id<Person> personId) {
074                double accumulatedScore = 0.0;
075                
076                if (this.lcContext.getPersonsBetas() != null && this.lcContext.getFacilitiesAttributes() != null) {
077                        
078                        // Maybe this is too dangerous and we should specify the available attributes and corresponding betas in the config as well
079                        // let us see!
080                        Collection<String> betas = ObjectAttributesUtils.getAllAttributeNames(this.lcContext.getPersonsBetas(), personId.toString());
081                        for (String name:betas) {
082                                double beta = (Double) this.lcContext.getPersonsBetas().getAttribute(personId.toString(), name);
083                                double attribute = (Double) this.lcContext.getFacilitiesAttributes().getAttribute(facilityId.toString(), name);
084                                accumulatedScore += beta * attribute;
085                        }
086                }
087                
088                return accumulatedScore;
089        }
090        
091        private double getEpsilonAlternative(Id<ActivityFacility> facilityId, Id<Person> personId, int actIndex) {
092                /*
093                 * k values are uniform in [0..1[, see class ReadOrCreateKVals.
094                 */             
095                double kf = this.facilitiesKValuesArray[this.lcContext.getFacilityIndex(facilityId)];
096                double kp = this.personsKValuesArray[this.lcContext.getPersonIndex(personId)]; 
097                
098                /* generate another stable random number for the activity
099                 * TODO: check if there is enough randomness with this seed
100                 */
101                rnd.setSeed(actIndex);
102                double ka = rnd.nextDouble();
103                
104                /*
105                 * generates a uniform rnd seed in [0,1[ 
106                 */
107                long seed = (long) (((kp + kf + ka) % 1.0) * Long.MAX_VALUE);
108                rnd.setSeed(seed);
109                
110                /*
111                 * generate the epsilons according to standard Gumbel or standard Gaussian distribution
112                 */
113                if (this.dccg.getEpsilonDistribution() == EpsilonDistributionTypes.gumbel) {
114                        // take a few draws to come to the "chaotic region"
115                        for (int i = 0; i < 5; i++) {
116                                rnd.nextDouble();
117                        }
118                        double uniform = rnd.nextDouble();
119                        // interval MUST be ]0,1[
120                        while (uniform == 0.0 || uniform == 1.0) {
121                                uniform = rnd.nextDouble();
122                        }
123                        double r = 0.0 - 1.0 * Math.log(-Math.log(1.0 * uniform));
124                        //scale to sigma^2 = 1.0: sigma_gumbel = PI / sqrt(6.0)
125                        return (r * Math.sqrt(6.0) / Math.PI);  
126                }
127                else {
128                        // take a few draws to come to the "chaotic region"
129                        for (int i = 0; i < 5; i++) {
130                                rnd.nextGaussian();
131                        }
132                        return rnd.nextGaussian();      
133                }
134        }
135}
