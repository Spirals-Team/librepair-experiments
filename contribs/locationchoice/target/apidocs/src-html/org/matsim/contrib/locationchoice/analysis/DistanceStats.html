020package org.matsim.contrib.locationchoice.analysis;
021
022import org.matsim.analysis.Bins;
023import org.matsim.api.core.v01.population.Activity;
024import org.matsim.api.core.v01.population.Person;
025import org.matsim.api.core.v01.population.Plan;
026import org.matsim.api.core.v01.population.PlanElement;
027import org.matsim.api.core.v01.population.Plan;
028import org.matsim.contrib.locationchoice.DestinationChoiceConfigGroup;
029import org.matsim.contrib.locationchoice.utils.ActTypeConverter;
030import org.matsim.core.config.Config;
031import org.matsim.core.controler.events.IterationEndsEvent;
032import org.matsim.core.controler.listener.IterationEndsListener;
033import org.matsim.core.population.PopulationUtils;
034import org.matsim.core.utils.geometry.CoordUtils;
035
036public class DistanceStats implements IterationEndsListener {   
037
038        private DestinationChoiceConfigGroup dccg;
039        private double analysisBoundary;
040        private Bins bins;
041        private String bestOrSelected = "selected";
042        private String type = null;
043        private ActTypeConverter actTypeConverter;
044        private String mode;
045        
046        public DistanceStats(Config config, String bestOrSelected, String type, ActTypeConverter actTypeConverter, String mode) {
047                
048                this.dccg = (DestinationChoiceConfigGroup) config.getModule(DestinationChoiceConfigGroup.GROUP_NAME);
049                this.analysisBoundary = this.dccg.getAnalysisBoundary(); 
050                this.bestOrSelected = bestOrSelected;
051                this.type = type;
052                this.actTypeConverter = actTypeConverter;
053                this.mode = mode;
054                this.bins = new Bins(this.dccg.getAnalysisBinSize(), analysisBoundary, type + "_" + mode + "_distance");
055        }
056
057        public void notifyIterationEnds(final IterationEndsEvent event) {       
058                this.bins.clear();
059
060        for (Person p : event.getServices().getScenario().getPopulation().getPersons().values()) {
061                        
062                        // continue if person is in the analysis population or if the id is not numeric
063                        if (this.dccg.getIdExclusion() == null || !this.isLong(p.getId().toString()) ||
064                                        Long.parseLong(p.getId().toString()) > this.dccg.getIdExclusion()) continue;
065                                        
066                        Plan plan = (Plan) p.getSelectedPlan();
067                        
068                        if (bestOrSelected.equals("best")) {
069                                double bestPlanScore = -999.0;
070                                int bestIndex = 0;
071                                int cnt = 0;
072                                for (Plan planTmp : p.getPlans()) {
073                                        if (planTmp.getScore() > bestPlanScore){
074                                                bestPlanScore = planTmp.getScore();
075                                                bestIndex = cnt;
076                                        }
077                                        cnt++;
078                                }
079                                plan = (Plan) p.getPlans().get(bestIndex);      
080                                //log.info(plan.getScore() + "\t" + bestIndex + "\tperson " + p.getId());
081                        }               
082                        for (PlanElement pe : plan.getPlanElements()) {
083                                if (pe instanceof Activity) {
084                                        if (this.actTypeConverter.convertType(((Activity) pe).getType()).equals(this.actTypeConverter.convertType(type)) &&
085                                                        PopulationUtils.getPreviousLeg(plan, (Activity)pe).getMode().equals(this.mode)) {
086                                                double distance = CoordUtils.calcEuclideanDistance(((Activity) pe).getCoord(), PopulationUtils.getPreviousActivity(plan, PopulationUtils.getPreviousLeg(plan, (Activity)pe)).getCoord()); 
087                                                this.bins.addVal(distance, 1.0);
088                                        }       
089                                }
090                        }
091                }
092                
093                // Actually, path is not the full file name - inside the plotBinnedDistribution some other stuff is added. 
094                String path = event.getServices().getControlerIO().getIterationFilename(event.getIteration(), "plan=" + this.bestOrSelected + "_");
095                this.bins.plotBinnedDistribution(path, "#", "m");
096        }
097        
098        private boolean isLong(String str) {
099            try {
100                Long.parseLong(str);
101                return true;
102            } catch (NumberFormatException nfe) {}
103            return false;
104        }
105}
