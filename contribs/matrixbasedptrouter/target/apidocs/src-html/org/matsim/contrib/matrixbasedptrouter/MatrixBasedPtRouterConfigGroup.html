023package org.matsim.contrib.matrixbasedptrouter;
024
025import java.io.File;
026
027import org.apache.log4j.Logger;
028import org.matsim.core.config.Config;
029import org.matsim.core.config.ReflectiveConfigGroup;
030
031/**
032 * @author nagel
033 *
034 */
035public final class MatrixBasedPtRouterConfigGroup extends ReflectiveConfigGroup {
036        static final Logger log = Logger.getLogger(MatrixBasedPtRouterConfigGroup.class) ;
037
038        public static final String GROUP_NAME="matrixBasedPtRouter" ;
039
040        public static final String PT_STOPS = "ptStopsFile"; // tnicolai: originally, this was named "ptStops" in the matsim config. old config files need to be adjusted
041        public static final String USING_PT_STOPS = "usingPtStops";
042        public static final String PT_TRAVEL_TIMES = "ptTravelTimesFile"; // tnicolai: originally, this was named "ptTravelTimes" in the matsim config. old config files need to be adjusted
043        public static final String PT_TRAVEL_DISTANCES = "ptTravelDistancesFile"; // tnicolai: originally, this was named "ptTravelDistances" in the matsim config. old config files need to be adjusted
044        public static final String PT_TRAVEL_TIMES_AND_DISTANCES_SWITCH = "usingTravelTimesAndDistances"; // tnicolai: originally, this was named "useTravelTimesAndDistances" in the matsim config. old config files need to be adjusted
045        private String ptStopsInputFile;
046        private String ptTravelTimesInputFile;
047        private String ptTravelDistancesInputFile;
048        private boolean usingTravelTimesAndDistances = false ;
049        private boolean usingPtStops = false ;
050
051        public MatrixBasedPtRouterConfigGroup() {
052                super(GROUP_NAME);
053        }
054
055        @StringSetter(PT_STOPS)
056        public void setPtStopsInputFile(String ptStops){
057                this.ptStopsInputFile = ptStops;
058        }
059        @StringGetter(PT_STOPS)
060        public String getPtStopsInputFile(){
061                return this.ptStopsInputFile;
062        }
063        @StringSetter(PT_TRAVEL_TIMES)
064        public void setPtTravelTimesInputFile(String ptTravelTimes){
065                this.ptTravelTimesInputFile = ptTravelTimes;
066        }
067        @StringGetter(PT_TRAVEL_TIMES)
068        public String getPtTravelTimesInputFile(){
069                return this.ptTravelTimesInputFile;
070        }
071        @StringSetter(PT_TRAVEL_DISTANCES )
072        public void setPtTravelDistancesInputFile(String ptTravelDistances){
073                this.ptTravelDistancesInputFile = ptTravelDistances;
074        }
075        @StringGetter(PT_TRAVEL_DISTANCES )
076        public String getPtTravelDistancesInputFile(){
077                return this.ptTravelDistancesInputFile;
078        }
079        @StringSetter(PT_TRAVEL_TIMES_AND_DISTANCES_SWITCH)
080        public void setUsingTravelTimesAndDistances( boolean val ) {
081                this.usingTravelTimesAndDistances = val ;
082        }
083        @StringGetter(PT_TRAVEL_TIMES_AND_DISTANCES_SWITCH)
084        public boolean isUsingTravelTimesAndDistances() {
085                return this.usingTravelTimesAndDistances ;
086        }
087        @StringGetter(USING_PT_STOPS)
088        public boolean isUsingPtStops() {
089                return usingPtStops;
090        }
091        @StringSetter(USING_PT_STOPS)
092        public void setUsingPtStops(boolean usingPtStops) {
093                this.usingPtStops = usingPtStops;
094        }
095
096        @Override
097        protected void checkConsistency(Config config) {
098                boolean problem = false ;
099                if( isUsingPtStops() ) {
100                        log.info(MatrixBasedPtRouterConfigGroup.USING_PT_STOPS + " switch is set to true. Trying to find pt stops file ...");
101                        // checking for pt stops
102                        if( getPtStopsInputFile() != null){
103                                File ptStopsFile = new File(getPtStopsInputFile());
104                                if(ptStopsFile.exists()){
105                                        log.info("Found pt stops file " + getPtStopsInputFile());
106                                } else {
107                                        problem = true ;
108                                        log.error("Pt stops file " + getPtStopsInputFile() + " not found although switch is set to true! Will abort ...");
109
110                                }
111
112                                // checking for other input files
113                                if( isUsingTravelTimesAndDistances() ) {
114                                        log.info(MatrixBasedPtRouterConfigGroup.PT_TRAVEL_TIMES_AND_DISTANCES_SWITCH + " switch is set to true. Trying to find travel times and distances files ...");
115
116                                        if(new File( getPtTravelTimesInputFile() ).exists() && new File( getPtTravelDistancesInputFile() ).exists()){
117                                                log.info("Found travel times and travel distances input files:");
118                                                log.info("Travel times input file: " + getPtTravelTimesInputFile());
119                                                log.info("Travel distances input file: " + getPtTravelDistancesInputFile() );
120                                        } else {
121                                                problem = true ;
122                                                log.error("Travel times and/or travel distances input files not found!  Will abort ...");
123                                                log.warn("Travel times input file name: " + getPtTravelTimesInputFile());
124                                                log.warn("Travel distances input file name: " + getPtTravelDistancesInputFile());
125                                        }
126                                } else {
127                                        log.info(MatrixBasedPtRouterConfigGroup.PT_TRAVEL_TIMES_AND_DISTANCES_SWITCH + " switch is set to false. Additional travel times and distances files will not be read!");
128                                }
129
130                        } else {
131                                problem = true ;
132                                log.warn("No pt stops file given although switch is set to true!  Will abort ...");
133                        }
134                } else {
135                        log.info(MatrixBasedPtRouterConfigGroup.USING_PT_STOPS + " switch is set to false. Matrix based pt router will not be initialized.");
136                }
137                if ( problem ) {
138                        throw new RuntimeException("found fatal problem in matrix based pt router initialization; aborting ...") ;
139                }
140
141        }
142
143}
