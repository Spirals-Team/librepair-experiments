001package org.matsim.contrib.matrixbasedptrouter;
002
003import org.matsim.api.core.v01.TransportMode;
004import org.matsim.api.core.v01.network.Network;
005import org.matsim.contrib.matrixbasedptrouter.utils.BoundingBox;
006import org.matsim.core.config.groups.PlansCalcRouteConfigGroup;
007import org.matsim.core.controler.AbstractModule;
008
009import com.google.inject.Provides;
010
011public class MatrixBasedPtModule extends AbstractModule {
012
013        @Override
014        public void install() {
015                if (getConfig().transit().isUseTransit()) {
016                        System.out.println("You are trying to use MatrixBasedPtModule and physical transit simulation "
017                                        + "at the same time. This probably will not work!");
018                }
019                addRoutingModuleBinding(TransportMode.pt).to(MatrixBasedPtRoutingModule.class);
020        }
021        
022        @Provides
023        PtMatrix createPtMatrix(PlansCalcRouteConfigGroup plansCalcRoute, MatrixBasedPtRouterConfigGroup ippcm, Network network) {
024                return PtMatrix.createPtMatrix(plansCalcRoute, BoundingBox.createBoundingBox(network), ippcm);
025        }
026}
