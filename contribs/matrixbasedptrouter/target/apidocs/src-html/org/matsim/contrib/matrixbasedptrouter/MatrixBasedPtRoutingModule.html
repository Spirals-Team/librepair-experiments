043/**
044 * Design thoughts:<ul>
045 * <li>yyyy The module includes walk times and walk distances.  But rather than putting them into a separate leg, they are
046 * included "silently" in the pt leg, which thus pretends to go door to door. kai, jul'13
047 * </ul>
048 * 
049 * @author thomas
050 *
051 */
052public final class MatrixBasedPtRoutingModule implements RoutingModule {
053        
054        // 1) read a file with every transit stop
055        // 2) read a file with travel times and general costs from every stop to every other stop ("matrix")
056        // 3) route:
057        //    - find the nearest transit stop to the coordinates of the previous activity (via QuadTree)
058        //    - find the nearest transit stop to the coordinates of the next activity (via QuadTree)
059        //    - determine the time and generalized cost of the trip (walk + "matrix" entry + walk)
060        // 4) return that route as pseudo transit, i.e. such that matsim executes this as teleportation
061
062        private final GenericRouteFactory genericRouteFactory;
063        private final PtMatrix ptMatrix;
064        private final Scenario scenario;
065
066        @Inject
067        public MatrixBasedPtRoutingModule(Scenario scenario, PtMatrix ptMatrix) {
068                this.scenario = scenario;
069                this.genericRouteFactory = new GenericRouteFactory();
070                this.ptMatrix = ptMatrix;
071        }
072        
073        @Override
074        public List<? extends PlanElement> calcRoute(Facility fromFacility, Facility toFacility, double departureTime, Person person) {
075                Leg newLeg = scenario.getPopulation().getFactory().createLeg( TransportMode.pt );
076                Id<Link> startLinkId = fromFacility.getLinkId();
077                Id<Link> endLinkId = toFacility.getLinkId();
078                newLeg.setDepartureTime( departureTime );
079                double travelTime = this.ptMatrix.getTotalTravelTime_seconds(fromFacility.getCoord(), toFacility.getCoord());
080                newLeg.setTravelTime( travelTime );
081                
082                final Route route = genericRouteFactory.createRoute(startLinkId, endLinkId);
083                double distance = this.ptMatrix.getTotalTravelDistance_meter(fromFacility.getCoord(), toFacility.getCoord()) ;
084                route.setDistance(distance) ;
085                newLeg.setRoute(route);
086                
087                return Arrays.asList( newLeg );
088        }
089        
090        @Override
091        public StageActivityTypes getStageActivityTypes(){
092                return EmptyStageActivityTypes.INSTANCE;
093        }
094}
