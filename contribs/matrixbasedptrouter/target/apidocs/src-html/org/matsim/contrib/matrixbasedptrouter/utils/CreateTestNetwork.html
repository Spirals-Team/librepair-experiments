188        /**
189         * This method creates 4 pt stops for the test network from createTestNetwork().
190         * The information about the coordinates will be written to a csv file.
191         * The 4 pt stops are located as a square in the coordinate plane with a side length of 180 meter (see the sketch below).
192         *
193         * @return the location of the written csv file
194         */
195        public static String createTestPtStationCSVFile(File file){
196
197                /*
198                 * (2)      (5)------(8)
199                 *      |(pt2)   |   (pt3)
200                 *      |                |
201                 * (1)------(4)------(7)
202                 *      |                |
203                 *      |(pt1)   |   (pt4)
204                 * (3)      (6)------(9)
205                 */
206
207                try (BufferedWriter bw = new BufferedWriter(new FileWriter(file))) {
208                        bw.write("id,x,y");     // header
209                        bw.newLine();
210                        bw.write("1,10,10");    // pt stop next to node (3)
211                        bw.newLine();
212                        bw.write("2,10, 190"); // pt stop next to node (2)
213                        bw.newLine();
214                        bw.write("3,190,190"); // pt stop next to node (8)
215                        bw.newLine();
216                        bw.write("4,190,10");  // pt stop next to node (9)
217                        bw.newLine();
218                        return file.getCanonicalPath();
219                } catch (IOException e) {
220                        throw new RuntimeException(e);
221                }
222        }

224        /**
225         * This methods creates a csv file with informations about pt travel times and pt distances for the test network from createTestNetwork().
226         * We set the pt travel time between all pairs of pt stops to 100 seconds, except pairs of same pt stops where the travel time is 0 seconds.
227         * We set the pt distance between all pairs of pt stops to 100 meter, except pairs of same pt stops where the distance is 0 meter.
228         * Because the data in the csv file does not need an entity, you can use the same csv file for both informations.
229         * 
230         * @return the location of the written file
231         */
232        public static String createTestPtTravelTimesAndDistancesCSVFile(File file){
233                // set dummy travel times or distances to all possible pairs of pt stops
234
235                try (BufferedWriter bw = IOUtils.getBufferedWriter(file.getCanonicalPath())) {
236                        for (int origin = 1; origin <= 4; origin++){
237                                for (int destination = 1; destination <= 4; destination++){
238                                        if (origin == destination) {
239                                                // set a travel time/distance of 0s or 0m between same origin and destination pt stops
240                                                bw.write(origin + " " + destination + " 0");
241                                                bw.newLine();
242                                        } else {
243                                                // set a dummy travel time/distance of 100s or 100m between different origin and destination pt stops
244                                                bw.write(origin + " " + destination + " 100");
245                                                bw.newLine();
246                                        }
247
248                                }
249                        }
250                        bw.flush();
251                        return file.getCanonicalPath();
252                } catch (IOException e) {
253                        throw new RuntimeException(e);
254                }
255        }

257        /**
258         * This method creates 4 facilities for the test network from createTestNetwork().
259         * The distance between each facility and the nearest pt stop is 50 meter (see the sketch below).
260         * 
261         * @return the facility list
262         */
263        public static List<Coord> getTestFacilityLocations(){
264
265                /*    B             C
266                 * (2)          (5)------(8)
267                 *      |                |
268                 *      |                |    
269                 * (1)------(4)------(7)
270                 *      |                |           
271                 *      |                |
272                 * (3)          (6)------(9)
273                 *    A             D
274                 */   
275
276                List<Coord> facilityList = new ArrayList<Coord>();
277                final double y1 = -40;
278                facilityList.add(new Coord((double) 10, y1));  // 50m to pt station 1
279                facilityList.add(new Coord((double) 10, (double) 240));  // 50m to pt station 2
280                facilityList.add(new Coord((double) 190, (double) 240)); // 50m to pt station 3
281                final double y = -40;
282                facilityList.add(new Coord((double) 190, y)); // 50m to pt station 4
283                return facilityList;
284        }

286        /**
287         * creating a test network
288         * the path 1,2,4 has a total length of 1000m with a free speed travel time of 10m/s
289         * the second path 1,3,4 has a total length of 100m but only a free speed travel time of 0.1m/s
290         */
291        public static Network createTriangularNetwork() {
292                /*
293                 *                      (2)
294                 *         /   \
295                 *        /     \
296                 *(10m/s)/       \(10m/s)
297                 *(500m)/             \(500m)
298                 *     /           \
299                 *    /             \
300                 *       /               \
301                 *(1)-------(3)-------(4)
302                 *(50m,0.1m/s)(50m,0.1m/s)                      
303                 */
304                MutableScenario scenario = (MutableScenario) ScenarioUtils.createScenario(ConfigUtils.createConfig());
305
306                Network network = (Network) scenario.getNetwork();
307
308                // add nodes
309                Node node1 = NetworkUtils.createAndAddNode(network, Id.create(1, Node.class), new Coord((double) 0, (double) 0));
310                Node node2 = NetworkUtils.createAndAddNode(network, Id.create(2, Node.class), new Coord((double) 50, (double) 100));
311                Node node3 = NetworkUtils.createAndAddNode(network, Id.create(3, Node.class), new Coord((double) 50, (double) 0));
312                Node node4 = NetworkUtils.createAndAddNode(network, Id.create(4, Node.class), new Coord((double) 100, (double) 0));
313                final Node fromNode = node1;
314                final Node toNode = node2;
315
316                // add links
317                NetworkUtils.createAndAddLink(network,Id.create(1, Link.class), fromNode, toNode, 500.0, 10.0, 3600.0, (double) 1 );
318                final Node fromNode1 = node2;
319                final Node toNode1 = node4;
320                NetworkUtils.createAndAddLink(network,Id.create(2, Link.class), fromNode1, toNode1, 500.0, 10.0, 3600.0, (double) 1 );
321                final Node fromNode2 = node1;
322                final Node toNode2 = node3;
323                NetworkUtils.createAndAddLink(network,Id.create(3, Link.class), fromNode2, toNode2, 50.0, 0.1, 3600.0, (double) 1 );
324                final Node fromNode3 = node3;
325                final Node toNode3 = node4;
326                NetworkUtils.createAndAddLink(network,Id.create(4, Link.class), fromNode3, toNode3, 50.0, 0.1, 3600.0, (double) 1 );
327
328                return network;
329        }
