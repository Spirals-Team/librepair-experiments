001package org.matsim.contrib.matrixbasedptrouter.utils;
002
003import org.matsim.api.core.v01.Coord;
004import org.matsim.api.core.v01.Id;
005import org.matsim.api.core.v01.Scenario;
006import org.matsim.api.core.v01.TransportMode;
007import org.matsim.api.core.v01.population.Activity;
008import org.matsim.api.core.v01.population.Leg;
009import org.matsim.api.core.v01.population.Person;
010import org.matsim.api.core.v01.population.Plan;
011import org.matsim.api.core.v01.population.Population;
012import org.matsim.core.config.ConfigUtils;
013import org.matsim.core.population.PersonUtils;
014import org.matsim.core.scenario.ScenarioUtils;
015
016/**
017 * 
018 * This class creates a test population for junit tests.
019 * 
020 * @author dhosse
021 *
022 */
023
024public final class CreateTestPopulation {
025        
026        
027        /**
028         * 
029         * This method creates n persons for a test population. NOTE: It was written for the
030         * <code>AccessibilityTest</code>, all home locations are at (0,100) and all work
031         * locations are at (200,100). All legs are car legs.
032         * 
033         * @param nPersons the number of persons created for the population
034         * @return test population
035         */
036        public static Population createTestPopulation(int nPersons){
037                
038                Scenario scenario = ScenarioUtils.createScenario(ConfigUtils.createConfig());
039                Population population = scenario.getPopulation();
040                
041                //create persons and add them to the population
042                for(int i=0;i<nPersons;i++){
043                        Person person = population.getFactory().createPerson(Id.create(i, Person.class));
044                        PersonUtils.setAge(person, 30);
045                        Plan plan = population.getFactory().createPlan();
046                        
047                        //create home activities at (0,100) and work activities at (200,100), all modes are by car
048                        Activity home = population.getFactory().createActivityFromCoord("home", new Coord((double) 0, (double) 100));
049                        home.setEndTime(8.*3600);
050                        Activity work = population.getFactory().createActivityFromCoord("work", new Coord((double) 200, (double) 100));
051                        work.setEndTime(17.*3600);
052                        Activity home2 = population.getFactory().createActivityFromCoord("home", new Coord((double) 0, (double) 100));
053                        home2.setEndTime(24.*3600);
054                        Leg leg = population.getFactory().createLeg(TransportMode.car);
055                        
056                        plan.addActivity(home);
057                        plan.addLeg(leg);
058                        plan.addActivity(work);
059                        plan.addLeg(leg);
060                        plan.addActivity(home2);
061                        
062                        person.addPlan(plan);
063                        population.addPerson(person);
064                }
065                
066                return population;
067        }
068
069        /**
070         * 
071         * This method creates n persons for a test population to test pseudo pt.
072         * ALL home activites are at <code>homeCoord</code> and ALL work activities are
073         * at <code>workCoord</code>. All legs are pt legs.
074         * 
075         * @param nPersons the number of persons to be created
076         * @param homeCoord coordinate for ALL home activities
077         * @param workCoord coordinate for ALL work activities
078         * @return
079         */
080        public static Population createTestPtPopulation(int nPersons, Coord homeCoord, Coord workCoord){
081                
082                Scenario scenario = ScenarioUtils.createScenario(ConfigUtils.createConfig());
083                Population population = scenario.getPopulation();
084                
085                //create persons and add them to the population
086                for(int i=0;i<nPersons;i++){
087                        Person person = population.getFactory().createPerson(Id.create(i, Person.class));
088                        PersonUtils.setAge(person, 30);
089                        Plan plan = population.getFactory().createPlan();
090                        
091                        //create home activities at homeCoord and work activities at workCoord, all modes are by car                    
092                        Activity home = population.getFactory().createActivityFromCoord("home", homeCoord);
093                        home.setEndTime(8.*3600);
094                        Activity work = population.getFactory().createActivityFromCoord("work", workCoord);
095                        work.setEndTime(17.*3600);
096                        Activity home2 = population.getFactory().createActivityFromCoord("home", homeCoord);
097                        home2.setEndTime(24.*3600);
098                        Leg leg = population.getFactory().createLeg(TransportMode.pt);
099                        
100                        plan.addActivity(home);
101                        plan.addLeg(leg);
102                        plan.addActivity(work);
103                        plan.addLeg(leg);
104                        plan.addActivity(home2);
105                        
106                        person.addPlan(plan);
107                        population.addPerson(person);
108                }
109                
110                return population;
111        }
