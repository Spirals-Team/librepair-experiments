001package org.matsim.contrib.matrixbasedptrouter.utils;
002
003import java.util.Map;
004import java.util.concurrent.ConcurrentHashMap;
005
006public final class HeaderParser {
007        
008        /**
009         * This is used to parse a header line from a tab-delimited urbansim header and generate a Map that allows to look up column
010         * numbers (starting from 0) by giving the header line.
011         *
012         * I.e. if you have a header "from_id <tab> to_id <tab> travel_time", then idxFromKey.get("to_id") will return "1".
013         *
014         * This makes the reading of column-oriented files independent from the sequence of the columns.
015         *
016         * @param line
017         * @return idxFromKey as described above (mapping from column headers into column numbers)
018         *
019         * @author nagel
020         */
021        public static Map<String,Integer> createIdxFromKey( String line, String seperator ) {
022                String[] keys = line.split( seperator ) ;
023
024                Map<String,Integer> idxFromKey = new ConcurrentHashMap<String, Integer>() ;
025                for ( int i=0 ; i<keys.length ; i++ ) {
026                        idxFromKey.put(keys[i], i ) ;
027                }
028                return idxFromKey ;
029        }
030        
031        /**
032         * See describtion from createIdxFromKey( String line, String seperator )
033         * 
034         * @param keys
035         * @return idxFromKey
036         */
037        public static Map<Integer,Integer> createIdxFromKey( int keys[]) {
038
039                Map<Integer,Integer> idxFromKey = new ConcurrentHashMap<Integer, Integer>() ;
040                for ( int i=0 ; i<keys.length ; i++ ) {
041                        idxFromKey.put(keys[i], i ) ;
042                }
043                return idxFromKey ;
044        }
045
046}
