040/**
041 * The idea, as the name says, is to not remove the worst plan, but remove plans such that diversity is maintained.
042 * The design idea stems from path size logit, which reduces the probability to select alternatives which are similar
043 * to other alternatives by adding a penalty to their utility.  
044 * <p></p>
045 * So if you have, for example, 4 similar plans with high scores and one other plan with a lower score, then the scores of 
046 * the similar plans are artificially reduced, and one of them will in consequence be removed. 
047 * In the present incarnation, we make sure that not the best of those 4 will be removed.
048 * <p></p>
049 * Note that once all similar plans are removed, the remaining best plan will not be similar to any other plan any more, and
050 * thus no longer incur a similarity penalty.  So it will never be removed.
051 * <p></p>
052 * This class has <i>not</i> yet been extensively tested and so it is not clear if it contains bugs, how well it works, or if parameters
053 * should be set differently.  If someone wants to experiment, the class presumably should be made configurable (or copied before 
054 * modification). 
055 * <p></p>
056 * There is also material in playground.vsp .
057 * <p></p>
058 * There are also some hints to literature at {@link PopulationUtils#calculateSimilarity}
059 * 
060 * @author nagel, ikaddoura
061 */

146        @Override
147        protected final Map<Plan, Double> calcWeights(List<? extends Plan> plans) {
148                if ( plans.isEmpty() ) {
149                        throw new RuntimeException("empty plans set; this will not work ...") ;
150                }
151
152                Map<Plan,Double> map = new HashMap<Plan,Double>() ;
153
154                double[] utils = new double[plans.size()] ;
155
156                // --- initialize utils: ---
157                int pp=0 ;
158                for ( Plan plan : plans ) {
159                        utils[pp] = plan.getScore() ;
160                        if ( Double.isNaN(utils[pp]) ) {
161                                log.warn( "utils is NaN; id: "  + plan.getPerson().getId());
162                        }
163                        pp++ ;
164                }
165
166                int rr=0 ;
167                for ( Plan plan1 : plans ) {
168                        for ( Plan plan2 : plans ) {
169                                // yyyy there is really no need to compare the plan with itself.  kai/johan, mar'14, should not happen anymore... ihab, may'14
170                                if (plan1 == plan2) {
171                                        // same plan
172                                } else {
173                                        utils[rr] -= similarity( plan1, plan2 ) ;
174
175                                        if ( Double.isNaN(utils[rr]) ) {
176                                                log.warn( "utils is NaN; id: " + plan1.getPerson().getId() ) ;
177                                        }
178                                }       
179                        }
180                        rr++ ;
181                }
182
183                //              // --- calculate expSum: ---
184                //              double expSum = 0. ;
185                //              for ( int ii=0 ; ii<utils.length ; ii++ ) {
186                //                      expSum += Math.exp( utils[ii] ) ;
187                //              }
188                //
189                //              // --- calculate weights
190                //              int qq=0 ;
191                //              for ( Plan plan : plans ) {
192                //                      double weight = Math.exp( utils[qq] ) / expSum ;
193                //                      map.put( plan,  weight ) ;
194                //              }
195
196                // start with an exact version: for the time being, we do not want that the best plan vanishes.
197                // The worst plan (taking into account the penalty for similarity!) will be removed.
198                // Alternative (Ihab): Remove the best plan from the evaluation; apply algo only to other plans. may'14
199                double minUtil = Double.POSITIVE_INFINITY ;
200                Integer minIdx = null ;
201                for ( int kk = 0 ; kk<utils.length ; kk++ ) {
202                        if ( utils[kk] < minUtil ) {
203                                minUtil = utils[kk] ;
204                                minIdx = kk ;
205                        }
206                }
207                if ( minIdx==null ) {
208                        log.warn("minIdx is still null; there is a problem somehwere.") ;
209                        for ( int kk=0 ; kk<utils.length; kk++ ) {
210                                log.warn( "kk: " + kk + "; utils: " + utils[kk] ) ;
211                        }
212                }
213
214                int ab = 0 ;
215                for ( Plan plan : plans ) {
216                        if ( ab==minIdx){
217                                map.put( plan, 1. ) ;
218                        } else {
219                                map.put( plan, 0. ) ;
220                        }
221                        ab++ ;
222                }
223
224
225                return map ;
226        }
227
228        private double similarity( Plan plan1, Plan plan2 ) {
229                double simil = 0. ;
230                {
231                        List<Activity> activities1 = TripStructureUtils.getActivities(plan1, stageActivities) ;
232                        List<Activity> activities2 = TripStructureUtils.getActivities(plan2, stageActivities) ;
233                        simil += PopulationUtils.calculateSimilarity(activities1, activities2, actTypeWeight, locationWeight, actTimeWeight ) ;
234                        if ( Double.isNaN(simil) ) {
235                                log.warn("simil is NaN; id: " + plan1.getPerson().getId() ) ;
236                        }
237                }
238                {
239                        List<Leg> legs1 = TripStructureUtils.getLegs(plan1 ) ;
240                        List<Leg> legs2 = TripStructureUtils.getLegs(plan2 ) ;
241                        simil += PopulationUtils.calculateSimilarity(legs1, legs2, network, this.sameModePenalty, this.sameRoutePenalty ) ;
242                        if ( Double.isNaN(simil) ) {
243                                log.warn("simil is NaN; id: " + plan1.getPerson().getId() ) ;
244                        }
245                }               
246
247                return simil ;
248        }
