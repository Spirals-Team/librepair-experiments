020package org.matsim.contrib.common.gis;
021
022import org.geotools.data.DataStore;
023import org.geotools.data.DataStoreFinder;
024import org.geotools.data.simple.SimpleFeatureIterator;
025import org.geotools.data.simple.SimpleFeatureSource;
026import org.opengis.feature.simple.SimpleFeature;
027
028import java.io.File;
029import java.io.IOException;
030import java.net.URL;
031import java.util.HashMap;
032import java.util.HashSet;
033import java.util.Map;
034import java.util.Set;
035
036/**
037 * Utility class for reading shape (.shp) files.
038 * 
039 * @author illenberger
040 * 
041 */
042public class EsriShapeIO {
043
044        /**
045         * Calls {@link EsriShapeIO#readFeatures(String, String)} with <tt>null</tt>
046         * as feature type indicating to read all features.
047         * 
048         * @param filename
049         *            the path to the shape file.
050         * @return a set of features.
051         * @throws IOException
052         */
053        public static Set<SimpleFeature> readFeatures(String filename) throws IOException {
054                return EsriShapeIO.readFeatures(filename, null);
055        }
056
057        /**
058         * Reads a shape file and returns a set of features.
059         * 
060         * @param filename
061         *            the path to the shape file.
062         * @param type
063         *            the feature type to read in, or <tt>null</tt> if all available
064         *            feature types should be read. (I have the feeling that this
065         *            does not work reliable. joh15/01/10)
066         * @return a set of features
067         * @throws IOException
068         */
069        public static Set<SimpleFeature> readFeatures(String filename, String type) throws IOException {
070                Map<String, URL> params = new HashMap<String, URL>();
071                params.put("url", new File(filename).toURI().toURL());
072
073                DataStore dataStore = DataStoreFinder.getDataStore(params);
074                
075                Set<SimpleFeature> features = new HashSet<SimpleFeature>();
076                if(type == null) {
077                        for(String fType : dataStore.getTypeNames()) {
078                                addFeatures(dataStore.getFeatureSource(fType), features);
079                        }
080                } else {
081                        addFeatures(dataStore.getFeatureSource(type), features);
082                }
083                
084                return features;
085        }
086        
087        private static void addFeatures(SimpleFeatureSource source, Set<SimpleFeature> features) throws IOException {
088                SimpleFeatureIterator fIt = source.getFeatures().features();
089                while (fIt.hasNext()) {
090                        features.add(fIt.next());
091                }
092                fIt.close();
093        }
094}
