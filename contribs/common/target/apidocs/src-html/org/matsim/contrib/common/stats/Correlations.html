024package org.matsim.contrib.common.stats;
025
026import gnu.trove.iterator.TDoubleDoubleIterator;
027import gnu.trove.map.hash.TDoubleDoubleHashMap;
028import gnu.trove.map.hash.TDoubleIntHashMap;
029import gnu.trove.map.hash.TDoubleObjectHashMap;
030
031import java.io.FileNotFoundException;
032import java.io.IOException;
033
034import org.apache.commons.math.stat.descriptive.DescriptiveStatistics;
035
036/**
037 * @author illenberger
038 * 
039 */
040public class Correlations {
041
042        /**
043         * @param valuesX
044         *            a set of values
045         * @param valuesY
046         *            a set of values
047         * @return a map where the keys are the elements of <tt>valuesX</tt> and
048         *         where the values are the mean of a subset of <tt>valuesY</tt> the
049         *         index of which is equal to elements of <tt>valuesX</tt> with
050         *         equal value.
051         */
052        public static TDoubleDoubleHashMap mean(double[] valuesX, double[] valuesY) {
053                return mean(valuesX, valuesY, new DummyDiscretizer());
054        }
055
056        /**
057         * 
058         * @param valuesX
059         *            a set of values.
060         * @param valuesY
061         *            a set of values.
062         * @param binwidth
063         *            the bin width for a {@link LinearDiscretizer} applied to
064         *            elements of <tt>valuesX</tt>.
065         * @return a map where the keys are the elements of <tt>valuesX</tt> and
066         *         where the values are the mean of a subset of <tt>valuesY</tt> the
067         *         index of which is equal to elements of <tt>valuesX</tt> with
068         *         equal value. The elements of <tt>valuesX</tt> are discretized
069         *         with <tt>binwidth</tt>.
070         */
071        public static TDoubleDoubleHashMap mean(double[] valuesX, double[] valuesY, double binwidth) {
072                return mean(valuesX, valuesY, new LinearDiscretizer(binwidth));
073        }
074
075        /**
076         * 
077         * @param valuesX
078         *            a set of values.
079         * @param valuesY
080         *            a set of values.
081         * @param discretizer
082         *            a discretizer applied to the elements of <tt>valuesX</tt>.
083         * @return a map where the keys are the elements of <tt>valuesX</tt> and
084         *         where the values are the mean of a subset of <tt>valuesY</tt> the
085         *         index of which is equal to elements of <tt>valuesX</tt> with
086         *         equal value. The elements of <tt>valuesX</tt> are discretized
087         *         with <tt>discretizer</tt>.
088         */
089        public static TDoubleDoubleHashMap mean(double[] valuesX, double[] valuesY, Discretizer discretizer) {
090                if (valuesX.length != valuesY.length)
091                        throw new IllegalArgumentException("Both arrays must not differ in size!");
092
093                TDoubleDoubleHashMap sums = new TDoubleDoubleHashMap();
094                TDoubleIntHashMap counts = new TDoubleIntHashMap();
095
096                for (int i = 0; i < valuesX.length; i++) {
097                        double key = valuesX[i];
098                        key = discretizer.discretize(key);
099                        sums.adjustOrPutValue(key, valuesY[i], valuesY[i]);
100                        counts.adjustOrPutValue(key, 1, 1);
101                }
102
103                TDoubleDoubleIterator it = sums.iterator();
104                for (int i = 0; i < sums.size(); i++) {
105                        it.advance();
106                        it.setValue(it.value() / (double) counts.get(it.key()));
107                }
108
109                return sums;
110        }
111
112        /**
113         * 
114         * @param valuesX
115         *            a set of values.
116         * @param valuesY
117         *            a set of values.
118         * @param discretizer
119         *            a discretizer applied to the elements of <tt>valuesX</tt>.
120         * @return a map where the keys are the elements of <tt>valuesX</tt> and
121         *         where the values are descriptive statistics objects of a subset
122         *         of <tt>valuesY</tt> the index of which is equal to elements of
123         *         <tt>valuesX</tt> with equal value. The elements of
124         *         <tt>valuesX</tt> are discretized with <tt>discretizer</tt>.
125         */
126        public static TDoubleObjectHashMap<DescriptiveStatistics> statistics(double[] valuesX, double[] valuesY,
127                        Discretizer discretizer) {
128                if (valuesX.length != valuesY.length)
129                        throw new IllegalArgumentException("Both arrays must not differ in size!");
130
131                TDoubleObjectHashMap<DescriptiveStatistics> map = new TDoubleObjectHashMap<DescriptiveStatistics>();
132
133                for (int i = 0; i < valuesX.length; i++) {
134                        double x = discretizer.discretize(valuesX[i]);
135                        DescriptiveStatistics stats = map.get(x);
136                        if (stats == null) {
137                                stats = new DescriptiveStatistics();
138                                map.put(x, stats);
139                        }
140                        stats.addValue(valuesY[i]);
141                }
142
143                return map;
144        }
145
146        /**
147         * Writes a tab-separated file with the keys of <tt>values</tt> in the first
148         * column and the values of <tt>values</tt> in the second column.
149         * 
150         * @param values
151         *            a map of values.
152         * @param filename
153         *            the file name.
154         * @param xLabel
155         *            the header for the values of the first row.
156         * @param yLabel
157         *            the header for the values of the second row.
158         * @throws FileNotFoundException
159         * @throws IOException
160         * @deprecated
161         */
162        public static void writeToFile(TDoubleDoubleHashMap values, String filename, String xLabel, String yLabel)
163                        throws IOException {
164                StatsWriter.writeHistogram(values, xLabel, yLabel, filename);
165        }
