020package org.matsim.contrib.common.stats;
021
022import org.apache.commons.math.stat.descriptive.DescriptiveStatistics;
023import org.apache.commons.math.stat.descriptive.UnivariateStatistic;
024import org.apache.commons.math.stat.descriptive.rank.Max;
025import org.apache.commons.math.stat.descriptive.rank.Min;
026import org.apache.commons.math.stat.descriptive.rank.Percentile;
027import org.apache.commons.math.util.ResizableDoubleArray;
028
029/**
030 * Descriptive statistics object where each sample has a pi-value (1/weight) assigned.
031 * 
032 * @author illenberger
033 * 
034 */
035public class DescriptivePiStatistics extends DescriptiveStatistics {
036
037        private ResizableDoubleArray piValues = new ResizableDoubleArray();
038
039        private static final long serialVersionUID = 552953838961616624L;
040
041        /**
042         * Creates a new descriptive statistics object initialized with dummy
043         * implementations that return {@link Double#NaN} (except
044         * min/max-implementations).
045         */
046        public DescriptivePiStatistics() {
047                DummyPiStatistics dummyStats = new DummyPiStatistics();
048                setMeanImpl(dummyStats);
049                setGeometricMeanImpl(dummyStats);
050                setKurtosisImpl(dummyStats);
051                setMaxImpl(new StatisticsWrapper(new Max()));
052                setMinImpl(new StatisticsWrapper(new Min()));
053                setPercentileImpl(new DummyPiPercentile());
054                setSkewnessImpl(dummyStats);
055                setVarianceImpl(dummyStats);
056                setSumsqImpl(dummyStats);
057                setSumImpl(dummyStats);
058        }
059
060        /**
061         * Adds a sample with pi=1.0.
062         */
063        @Override
064        public void addValue(double v) {
065                addValue(v, 1.0);
066        }
067
068        /**
069         * Adds a sample with a pi-value.
070         * 
071         * @param v
072         *            a sample
073         * @param pi
074         *            the sample's pi-value
075         */
076        public void addValue(double v, double pi) {
077                if (windowSize != INFINITE_WINDOW) {
078                        if (getN() == windowSize) {
079                                piValues.addElementRolling(pi);
080                        } else if (getN() < windowSize) {
081                                piValues.addElement(pi);
082                        }
083                } else {
084                        piValues.addElement(pi);
085                }
086
087                super.addValue(v);
088        }
089
090        /**
091         * Returns the pi-values (insertion-ordered).
092         * 
093         * @return the pi-values.
094         */
095        public double[] getPiValues() {
096                return piValues.getElements();
097        }
098
099        /**
100         * @see {@link DescriptiveStatistics#apply(UnivariateStatistic)}.
101         */
102        @Override
103        public double apply(UnivariateStatistic stat) {
104                ((UnivariatePiStatistic) stat).setPiValues(piValues.getInternalValues());
105                return super.apply(stat);
106        }
107
108        /**
109         * @see {@link DescriptiveStatistics#clear()}.
110         */
111        @Override
112        public void clear() {
113                piValues.clear();
114                super.clear();
115        }
116
117        /**
118         * @see {@link DescriptiveStatistics#removeMostRecentValue()}.
119         */
120        @Override
121        public void removeMostRecentValue() {
122                piValues.discardMostRecentElements(1);
123                super.removeMostRecentValue();
124        }
125
126        private static class DummyPiStatistics implements UnivariatePiStatistic {
127
128                @Override
129                public void setPiValues(double[] piValues) {
130                        // does nothing
131                }
132
133                @Override
134                public UnivariateStatistic copy() {
135                        return new DummyPiStatistics();
136                }
137
138                @Override
139                public double evaluate(double[] values) {
140                        return Double.NaN;
141                }
142
143                @Override
144                public double evaluate(double[] values, int begin, int length) {
145                        return Double.NaN;
146                }
147        }
148
149        private static class StatisticsWrapper implements UnivariatePiStatistic {
150
151                private UnivariateStatistic delegate;
152
153                public StatisticsWrapper(UnivariateStatistic delegate) {
154                        this.delegate = delegate;
155                }
156
157                @Override
158                public void setPiValues(double[] piValues) {
159                        // does nothing
160                }
161
162                @Override
163                public UnivariateStatistic copy() {
164                        return new StatisticsWrapper(delegate);
165                }
166
167                @Override
168                public double evaluate(double[] values) {
169                        return delegate.evaluate(values);
170                }
171
172                @Override
173                public double evaluate(double[] values, int begin, int length) {
174                        return delegate.evaluate(values, begin, length);
175                }
176        }
177
178        private static class DummyPiPercentile extends Percentile implements UnivariatePiStatistic {
179
180                private static final long serialVersionUID = 3778314678555629077L;
181
182                @Override
183                public void setPiValues(double[] piValues) {
184                }
185
186                @Override
187                public Percentile copy() {
188                        return new DummyPiPercentile();
189                }
190
191                @Override
192                public double evaluate(double[] values, double p) {
193                        return Double.NaN;
194                }
195
196                @Override
197                public double evaluate(double[] values, int begin, int length, double p) {
198                        return Double.NaN;
199                }
200
201                @Override
202                public double evaluate(double[] values, int start, int length) {
203                        return Double.NaN;
204                }
205
206                @Override
207                public double getQuantile() {
208                        return Double.NaN;
209                }
210
211                @Override
212                public double evaluate(double[] values) {
213                        return Double.NaN;
214                }
215
216        }
217}
