020package org.matsim.contrib.common.stats;
021
022/**
023 * A composition of two discretizers. Values are first discretized with the first discretizer, the bin indicies are the
024 * discretized with the second discretizer, e.g., meters to kilometers and then kilometers in log2 bins.
025 *
026 * @author illenberger
027 */
028public class DiscretizerComposite implements Discretizer {
029
030    private Discretizer first;
031
032    private Discretizer second;
033
034    /**
035     * Creates a new discretizer composite.
036     *
037     * @param first  the first discretizer.
038     * @param second the second discretizer.
039     */
040    public DiscretizerComposite(Discretizer first, Discretizer second) {
041        this.first = first;
042        this.second = second;
043    }
044
045    /**
046     * @see {@link Discretizer#binWidth(double)}
047     */
048    @Override
049    public double binWidth(double value) {
050        double bin = first.index(value);
051        return second.binWidth(bin) * first.binWidth(value);
052    }
053
054    /**
055     * @see {@link Discretizer#discretize(double)}
056     */
057    @Override
058    public double discretize(double value) {
059        double bin = first.index(value);
060        bin = second.discretize(bin);
061        return bin * first.binWidth(value);
062    }
063
064    /**
065     * @see {@link Discretizer#index(double)}
066     */
067    @Override
068    public int index(double value) {
069        double bin = first.index(value);
070        return second.index(bin);
071    }
072
073}
