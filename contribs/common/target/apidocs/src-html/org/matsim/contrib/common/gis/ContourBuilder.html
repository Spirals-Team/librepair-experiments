014package org.matsim.contrib.common.gis;
015
016import com.vividsolutions.jts.algorithm.CGAlgorithms;
017import com.vividsolutions.jts.geom.*;
018import com.vividsolutions.jts.triangulate.quadedge.QuadEdge;
019import com.vividsolutions.jts.triangulate.quadedge.QuadEdgeSubdivision;
020
021import java.util.*;
022
023/**
024 *
025 * Adapted from org.opentripplanner.common.geometry.DelaunayIsolineBuilder,
026 * which is under LGPL.
027 *
028 * @author laurent
029 * @author michaz
030 *
031 */
032public class ContourBuilder {
033
034    private static final double EPSILON = 0.000001;
035    private QuadEdgeSubdivision triangulation;
036
037    private GeometryFactory geometryFactory = new GeometryFactory();
038
039    public ContourBuilder(QuadEdgeSubdivision triangulation) {
040        this.triangulation = triangulation;
041    }
042
043    public Geometry computeIsoline(double z0) {
044        Set<QuadEdge> processed = new HashSet<QuadEdge>();
045        Queue<QuadEdge> processQ = new ArrayDeque<QuadEdge>();
046        List<LinearRing> rings = new ArrayList<LinearRing>();
047
048        for (QuadEdge e : ((Collection<QuadEdge>) triangulation.getPrimaryEdges(true))) {
049            processQ.add(e);
050        }
051        while (!processQ.isEmpty()) {
052            QuadEdge e = processQ.remove();
053            if (processed.contains(e))
054                continue;
055            processed.add(e);
056            int cut = cut(e.orig().getZ(), e.dest().getZ(), z0);
057            if (cut == 0) {
058                continue; // While, next edge
059            }
060            List<Coordinate> polyPoints = new ArrayList<Coordinate>();
061            boolean ccw = cut > 0;
062            while (true) {
063                // Add a point to polyline
064                Coordinate cC;
065                if (triangulation.isFrameVertex(e.orig())) {
066                    cC = moveEpsilonTowards(e.dest().getCoordinate(), e.orig().getCoordinate());
067                } else if (triangulation.isFrameVertex(e.dest())) {
068                    cC = moveEpsilonTowards(e.orig().getCoordinate(), e.dest().getCoordinate());
069                } else {
070                    cC = e.orig().midPoint(e.dest()).getCoordinate();
071                }
072                polyPoints.add(cC);
073                processed.add(e);
074                QuadEdge E1 = ccw ? e.oNext().getPrimary() : e.oPrev().getPrimary();
075                QuadEdge E2 = ccw ? e.dPrev().getPrimary() : e.dNext().getPrimary();
076                int cut1 = E1 == null ? 0 : cut(E1.orig().getZ(), E1.dest().getZ(), z0);
077                int cut2 = E2 == null ? 0 : cut(E2.orig().getZ(), E2.dest().getZ(), z0);
078                boolean ok1 = cut1 != 0 && !processed.contains(E1);
079                boolean ok2 = cut2 != 0 && !processed.contains(E2);
080                if (ok1) {
081                    e = E1;
082                    ccw = cut1 > 0;
083                } else if (ok2) {
084                    e = E2;
085                    ccw = cut2 > 0;
086                } else {
087                    // This must be the end of the polyline...
088                    break;
089                }
090            }
091            // Close the polyline
092            polyPoints.add(polyPoints.get(0));
093            if (polyPoints.size() >= 4) {
094                LinearRing ring = geometryFactory.createLinearRing(polyPoints
095                        .toArray(new Coordinate[polyPoints.size()]));
096                rings.add(ring);
097            }
098        }
099        List<Polygon> retval = punchHoles(rings);
100        return geometryFactory.createMultiPolygon(retval.toArray(new Polygon[retval.size()]));
101    }
102
103    private Coordinate moveEpsilonTowards(Coordinate coordinate, Coordinate distantFrameCoordinate) {
104        return new Coordinate(coordinate.x + EPSILON * (distantFrameCoordinate.x - coordinate.x), coordinate.y + EPSILON * (distantFrameCoordinate.y - coordinate.y));
105    }
106
107    private int cut(double za, double zb, double z0) {
108        if (za < z0 && zb > z0) return 1;
109        if (za > z0 && zb < z0) return -1;
110        return 0;
111    }
112
113    @SuppressWarnings("unchecked")
114    private List<Polygon> punchHoles(List<LinearRing> rings) {
115        List<Polygon> shells = new ArrayList<Polygon>(rings.size());
116        List<LinearRing> holes = new ArrayList<LinearRing>(rings.size() / 2);
117        // 1. Split the polygon list in two: shells and holes (CCW and CW)
118        for (LinearRing ring : rings) {
119            if (CGAlgorithms.signedArea(ring.getCoordinateSequence()) > 0.0)
120                holes.add(ring);
121            else
122                shells.add(geometryFactory.createPolygon(ring));
123        }
124        // 2. Sort the shells based on number of points to optimize step 3.
125        Collections.sort(shells, new Comparator<Polygon>() {
126            @Override
127            public int compare(Polygon o1, Polygon o2) {
128                return o2.getNumPoints() - o1.getNumPoints();
129            }
130        });
131        for (Polygon shell : shells) {
132            shell.setUserData(new ArrayList<LinearRing>());
133        }
134        // 3. For each hole, determine which shell it fits in.
135        int nHolesFailed = 0;
136        for (LinearRing hole : holes) {
137            outer: {
138                // Probably most of the time, the first shell will be the one
139                for (Polygon shell : shells) {
140                    if (shell.contains(hole)) {
141                        ((List<LinearRing>) shell.getUserData()).add(hole);
142                        break outer;
143                    }
144                }
145                throw new RuntimeException("Found a hole without a shell.");
146            }
147        }
148        // 4. Build the list of punched polygons
149        List<Polygon> punched = new ArrayList<Polygon>(shells.size());
150        for (Polygon shell : shells) {
151            List<LinearRing> shellHoles = ((List<LinearRing>) shell.getUserData());
152            punched.add(geometryFactory.createPolygon((LinearRing) (shell.getExteriorRing()),
153                    shellHoles.toArray(new LinearRing[shellHoles.size()])));
154        }
155        return punched;
156    }
157}
