020package org.matsim.contrib.common.stats;
021
022import gnu.trove.list.array.TDoubleArrayList;
023
024import java.util.Arrays;
025
026/**
027 * An <tt>InterpolatingDiscretizer</tt> is a discretizer with fixed borders so that a border lies in the middle between
028 * two neighbouring values. For instance, for a set of samples <tt>2, 2, 2, 4, 6, 6</tt> the borders would be <tt>3</tt>
029 * and <tt>5</tt> and the bin values would be <tt>2, 4</tt> and <tt>6</tt>. Thas is, {@code
030 * InterpolatingDiscretizer#discretize(4.2)} returns <tt>4.0</tt> and {@code InterpolatingDiscretizer#discretize(10)}
031 * returns <tt>6</tt>.
032 *
033 * @author johannes
034 */
035public class InterpolatingDiscretizer implements Discretizer {
036
037    private double[] binValues;
038
039    private Discretizer borders;
040
041    /**
042     * Creates a new discretizer with borders and bin values set according to <tt>values</tt>.
043     *
044     * @param values a list of samples
045     */
046    public InterpolatingDiscretizer(double[] values) {
047        Arrays.sort(values);
048        TDoubleArrayList tmpBorders = new TDoubleArrayList();
049        TDoubleArrayList tmpValues = new TDoubleArrayList();
050        double low = values[0];
051        double high;
052        for (int i = 1; i < values.length; i++) {
053            high = values[i];
054            if (low < high) {
055                tmpBorders.add(low + (high - low) / 2.0);
056                tmpValues.add(low);
057            }
058            low = high;
059        }
060        tmpValues.add(values[values.length - 1]);
061
062        borders = new FixedBordersDiscretizer(tmpBorders.toArray());
063        binValues = tmpValues.toArray();
064    }
065
066    /**
067     * Discretizes <tt>value</tt> and returns the middle of the corresponding bin.
068     *
069     * @param value the value to discretize
070     * @return the discretized value
071     */
072    @Override
073    public double discretize(double value) {
074        int idx = (int) index(value);
075        return binValues[idx];
076    }
077
078    /**
079     * Returns the bin's index in which <tt>value</tt> is associated to.
080     *
081     * @param value the value to discretize
082     * @return the bin's index
083     */
084    @Override
085    public int index(double value) {
086        return borders.index(value);
087    }
088
089    /**
090     * Returns the bin's width in which <tt>value</tt> is associated to.
091     *
092     * @param value the value to discretize
093     * @return the bin's width
094     * @see FixedBordersDiscretizer#binWidth(double)
095     */
096    @Override
097    public double binWidth(double value) {
098        return borders.binWidth(value);
099    }
100
101}
