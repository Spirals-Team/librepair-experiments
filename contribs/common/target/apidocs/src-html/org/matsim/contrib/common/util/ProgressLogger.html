020package org.matsim.contrib.common.util;
021
022import java.text.NumberFormat;
023import java.util.concurrent.atomic.AtomicLong;
024
025
026/**
027 * Prints a simple progress bar to stdout:
028 * <br>
029 * <tt>0%....50%....100%</tt>
030 *
031 * @author jillenberger
032 *
033 */
034public class ProgressLogger {
035        
036        private static final String DOT = ".";
037
038        private static long maxVal;
039        
040        private static long minorTickVal;
041        
042        private static long majorTickVal;
043        
044        private static AtomicLong counter;
045
046        /**
047         * Initializes the progress logger.
048         * @param max the total number of steps
049         * @param minorTick the number of steps after that a dot is printed
050         * @param majorTick the number of steps after that the percentage is printed
051         */
052        public static void init(long max, long minorTick, long majorTick) {
053                maxVal = max;
054                minorTickVal = (long) Math.ceil(max/100.0 * minorTick);
055                majorTickVal = (long) Math.ceil(max/100.0 * majorTick);
056                counter = new AtomicLong(0);
057                System.out.print("\tProgress: 0%");
058        }
059
060        /**
061         * Moves the internal counter one step forward and prints a dot or the percentage respectively.
062         */
063        public static void step() {
064                counter.incrementAndGet();
065                if(counter.get() == maxVal) {
066                        System.out.print(NumberFormat.getPercentInstance().format(counter.get()/(double)maxVal));
067                        System.out.print("\n");
068                } else if(counter.get() % majorTickVal == 0) {
069                        System.out.print(NumberFormat.getPercentInstance().format(counter.get()/(double)maxVal));
070                } else if(counter.get() % minorTickVal == 0) {
071                        System.out.print(DOT);
072                }
073                
074        }
075
076        /**
077         * Terminates the logger. Use this function to abort logging before the number of {@code #step()} calls has
078         * reached the total number of steps.
079         */
080        public static void terminate() {
081                if(counter.get() != maxVal) {
082                        System.out.println(NumberFormat.getPercentInstance().format(counter.get()/(double)maxVal));
083                }
084        }
085}
