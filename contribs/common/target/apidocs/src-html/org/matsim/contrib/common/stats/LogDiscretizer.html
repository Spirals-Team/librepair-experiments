020package org.matsim.contrib.common.stats;
021
022/**
023 * A discretizer with logarithmically scaled bins.
024 *
025 * @author illenberger
026 */
027public class LogDiscretizer implements Discretizer {
028
029    private double base;
030
031    private double upperBound;
032
033    private double lowerBound;
034
035    /**
036     * Creates a new discretizer. Values less than 1 are rounded up to 1.
037     *
038     * @param base the base of the logarithm
039     */
040    public LogDiscretizer(double base) {
041        this(base, Double.POSITIVE_INFINITY, 1.0);
042    }
043
044    /**
045     * Creates a new discretizer where values greater than <tt>upperBound</tt> are rounded down to <tt>upperBound</tt>.
046     * Values less than 1 are rounded up to 1.
047     *
048     * @param base       the base of the logarithm
049     * @param upperBound the upper bound
050     */
051    public LogDiscretizer(double base, double upperBound) {
052        this(base, upperBound, 1.0);
053    }
054
055    /**
056     * Creates a new discretizer where values greater than <tt>upperBound</tt> are rounded down to <tt>upperBound</tt>
057     * and values less than <tt>lowerBound</tt> are rounded up to <tt>lowerBound</tt>.
058     *
059     * @param base       the base of the logarithm
060     * @param upperBound the upper bound
061     * @param lowerBound the lower bound
062     */
063    public LogDiscretizer(double base, double upperBound, double lowerBound) {
064        this.base = base;
065        this.upperBound = upperBound;
066        this.lowerBound = lowerBound;
067    }
068
069    /**
070     * @param value a value
071     * @returns rounds <tt>value</tt> up to the upper bin border.
072     */
073    @Override
074    public double discretize(double value) {
075        value = checkValue(value);
076        double bin = index(value);
077        return Math.pow(base, bin);
078    }
079
080    /**
081     * @param value a value
082     * @return the width of the bin of <tt>value</tt>. The width of the 0-th bin is 1.
083     */
084    @Override
085    public double binWidth(double value) {
086        double bin = index(value);
087        if (bin == 0)
088            return Math.pow(base, bin);
089        else
090            return Math.pow(base, bin) - Math.pow(base, bin - 1);
091    }
092
093    /**
094     * @see {@link Discretizer#index(double)}
095     */
096    @Override
097    public int index(double value) {
098        value = checkValue(value);
099        return (int) Math.ceil(Math.log(value) / Math.log(base));
100    }
101
102    private double checkValue(double value) {
103        value = Math.max(lowerBound, value);
104        value = Math.min(upperBound, value);
105        return value;
106
107    }
108}
