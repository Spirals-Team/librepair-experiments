020package org.matsim.contrib.common.gis;
021
022import com.vividsolutions.jts.geom.Coordinate;
023import com.vividsolutions.jts.geom.Geometry;
024import com.vividsolutions.jts.geom.GeometryFactory;
025import com.vividsolutions.jts.geom.Point;
026import org.apache.log4j.Logger;
027import org.geotools.referencing.CRS;
028import org.opengis.metadata.Identifier;
029import org.opengis.referencing.FactoryException;
030import org.opengis.referencing.NoSuchAuthorityCodeException;
031import org.opengis.referencing.crs.CRSAuthorityFactory;
032import org.opengis.referencing.crs.CoordinateReferenceSystem;
033import org.opengis.referencing.operation.MathTransform;
034import org.opengis.referencing.operation.TransformException;
035
036import java.util.HashMap;
037import java.util.Map;
038
039/**
040 * Utility-class providing functionality related to coordinate reference
041 * systems.
042 * 
043 * @author illenberger
044 * 
045 */
046public class CRSUtils {
047
048        private static Logger logger = Logger.getLogger(CRSUtils.class);
049
050        private static final Map<Integer, CoordinateReferenceSystem> crsMappings = new HashMap<Integer, CoordinateReferenceSystem>();
051
052        private static GeometryFactory geoFactory;
053
054        /**
055         * Retrieves the coordinate reference system from the EPSG database.
056         * 
057         * @param srid
058         *            the spatial reference id.
059         * 
060         * @return a coordinate reference system.
061         */
062        public static CoordinateReferenceSystem getCRS(int srid) {
063                CoordinateReferenceSystem crs = crsMappings.get(srid);
064                if (crs == null) {
065                        /*
066                         * Force longitude/latitude order.
067                         */
068                        CRSAuthorityFactory factory = CRS.getAuthorityFactory(true);
069                        try {
070                                crs = factory.createCoordinateReferenceSystem("EPSG:" + srid);
071                        } catch (NoSuchAuthorityCodeException e) {
072                                logger.warn(e.getLocalizedMessage());
073                        } catch (FactoryException e) {
074                                e.printStackTrace();
075                        }
076                }
077
078                return crs;
079        }
080
081        /**
082         * Returns the spatial reference id for a given coordinate reference system.
083         * If the coordinate reference system has multiple identifiers one is
084         * randomly selected.
085         * 
086         * @param crs
087         *            a coordinate reference system.
088         * 
089         * @return the spatial reference id for the coordinate reference system or
090         *         <tt>0</tt> if the coordinate reference system has no identifiers.
091         */
092        public static int getSRID(CoordinateReferenceSystem crs) {
093                /*
094                 * Randomly get one identifier.
095                 */
096                Identifier identifier = (Identifier) (crs.getIdentifiers().iterator().next());
097                if (identifier == null) {
098                        return 0;
099                } else {
100                        return Integer.parseInt(identifier.getCode());
101                }
102        }
103
104        /**
105         * Determines the transformation from the coordinate reference system of
106         * <tt>source</tt> to the one of <tt>target</tt>.
107         * 
108         * @param source
109         *            a geometry.
110         * @param target
111         *            a geometry.
112         * @return a transformation or <tt>null</tt> if the transformation could not
113         *         be determined.
114         */
115        public static MathTransform findTransform(Geometry source, Geometry target) {
116                CoordinateReferenceSystem sourceCRS = getCRS(source.getSRID());
117                CoordinateReferenceSystem targetCRS = getCRS(target.getSRID());
118
119                try {
120                        return CRS.findMathTransform(sourceCRS, targetCRS);
121                } catch (FactoryException e) {
122                        e.printStackTrace();
123                        return null;
124                }
125        }
126
127        /**
128         * Creates a new point that is a transformed copy of the original point.
129         * 
130         * @param point
131         *            a the original point.
132         * @param transform
133         *            the transformation to be applied.
134         * @return a new transformed point.
135         */
136        public static Point transformPoint(Point point, MathTransform transform) {
137                if (geoFactory == null)
138                        geoFactory = new GeometryFactory();
139
140                double[] points = new double[] { point.getCoordinate().x, point.getCoordinate().y };
141                try {
142                        transform.transform(points, 0, points, 0, 1);
143                } catch (TransformException e) {
144                        e.printStackTrace();
145                }
146                Point p = geoFactory.createPoint(new Coordinate(points[0], points[1]));
147                return p;
148        }
149
150        public static void transformCoordinate(Coordinate coord, MathTransform transform) {
151                double[] points = new double[] { coord.x, coord.y };
152                try {
153                        transform.transform(points, 0, points, 0, 1);
154                } catch (TransformException e) {
155                        e.printStackTrace();
156                }
157                coord.x = points[0];
158                coord.y = points[1];
159        }
160}
