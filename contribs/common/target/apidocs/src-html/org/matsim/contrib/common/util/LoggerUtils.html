020package org.matsim.contrib.common.util;
021
022import org.apache.log4j.Appender;
023import org.apache.log4j.Level;
024import org.apache.log4j.Logger;
025import org.apache.log4j.PatternLayout;
026
027/**
028 * Utility functions for logging.
029 *
030 * @author illenberger
031 *
032 */
033public class LoggerUtils {
034
035        private static boolean disallowVerbose = true;
036
037        private static PatternLayout defaultLayout;
038
039        /**
040         * Debugging purpose: Allows to disable the effect of {@link #setVerbose(boolean)}. Use this function to ensure
041         * that all messages are logged.
042         *
043         * @param disable if <tt>true</tt> {@code #setVerbose(false)} will have no effect
044         */
045        public static void setDisableVerbose(boolean disable) {
046                disallowVerbose = disable;
047        }
048
049        /**
050         * Allows to quickly switch between logging at all levels and logging only from warn-level. Use this function to
051         * temporary suppress extensive logging.
052         *
053         * @param verbose if <tt>true</tt> logger level is set to {@link Level#ALL}, otherwise level is set to {@link
054         * Level#WARN}.
055         */
056        public static void setVerbose(boolean verbose) {
057                if(verbose)
058                        Logger.getRootLogger().setLevel(Level.ALL);
059                else if(!verbose && !disallowVerbose){
060                        Logger.getRootLogger().setLevel(Level.WARN);
061                }
062                        
063        }
064
065        /**
066         * Remove the line break form the logger pattern. Subsequent writes to stdout are printed to the same line.
067         */
068        public static void disableNewLine() {
069                Appender appender = Logger.getRootLogger().getAppender("stdout");
070                defaultLayout = (PatternLayout)appender.getLayout();
071                PatternLayout newLayout = new PatternLayout(defaultLayout.getConversionPattern().replace("%n",""));
072                appender.setLayout(newLayout);
073        }
074
075        /**
076         * Appends a line break to the logger pattern (default behavior).
077         */
078        public static void enableNewLine() {
079                Appender appender = Logger.getRootLogger().getAppender("stdout");
080                appender.setLayout(defaultLayout);
081        }
082}
