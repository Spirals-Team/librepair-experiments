020package org.matsim.contrib.common.stats;
021
022import gnu.trove.function.TDoubleFunction;
023import gnu.trove.iterator.TDoubleDoubleIterator;
024import gnu.trove.map.hash.TDoubleDoubleHashMap;
025
026import java.util.Arrays;
027
028import org.apache.commons.math.stat.StatUtils;
029import org.apache.commons.math.stat.descriptive.DescriptiveStatistics;
030
031/**
032 * Utility functions to generate histograms.
033 *
034 * @author illenberger
035 */
036public class Histogram {
037
038    /**
039     * Creates a histogram from the values in <tt>stats</tt> discretized with <tt>discretizer</tt>. Correctly handles
040     * {@link DescriptivePiStatistics} objects.
041     *
042     * @param stats       a descriptive statistics object
043     * @param discretizer a discretizer
044     * @return a double-double map where the key denotes the bin and the value the bin height.
045     */
046    public static TDoubleDoubleHashMap createHistogram(DescriptiveStatistics stats, Discretizer discretizer, boolean reweight) {
047        if (stats instanceof DescriptivePiStatistics)
048            return createHistogram((DescriptivePiStatistics) stats, discretizer, reweight);
049        else
050            return createHistogram(stats.getValues(), discretizer, reweight);
051    }
052
053    /**
054     * Creates a histogram from the values in <tt>stats</tt> discretized with <tt>discretizer</tt> and weighted with the
055     * inverse of the samples' pi-values.
056     *
057     * @param stats       a descriptive statistics object
058     * @param discretizer a discretizer
059     * @param reweight    if <tt>true</tt> the bin height is reweighted with (divided by) the bin's width. Use this for
060     *                    discretizers with variable bin width
061     * @return a double-double map where the key denotes the bin and the value the bin height.
062     */
063    public static TDoubleDoubleHashMap createHistogram(DescriptivePiStatistics stats, Discretizer discretizer, boolean reweight) {
064        double[] piValues = stats.getPiValues();
065        double[] weights = new double[piValues.length];
066        for (int i = 0; i < piValues.length; i++) {
067            weights[i] = 1 / piValues[i];
068        }
069
070        return createHistogram(stats.getValues(), weights, discretizer, reweight);
071    }
072
073    /**
074     * Creates a histogram out of <tt>values</tt> discretized with <tt>discretizer</tt>.
075     *
076     * @param values      the samples
077     * @param discretizer a discretizer
078     * @param reweight    if <tt>true</tt> the bin height is reweighted with (divided by) the bin's width. Use this for
079     *                    discretizers with variable bin width
080     * @return a double-double map where the key denotes the bin and the value the bin height.
081     */
082    public static TDoubleDoubleHashMap createHistogram(double[] values, Discretizer discretizer, boolean reweight) {
083        double[] weights = new double[values.length];
084        Arrays.fill(weights, 1.0);
085        return createHistogram(values, weights, discretizer, reweight);
086    }
087
088    /**
089     * Creates a histogram out of <tt>values</tt> discretized with <tt>discretizer</tt> and weighted with the values in
090     * <tt>weights</tt>.
091     *
092     * @param values      the samples
093     * @param weights     the weights
094     * @param discretizer a discretizer
095     * @param reweight    if <tt>true</tt> the bin height is reweighted with (divided by) the bin's width. Use this for
096     *                    discretizers with variable bin width
097     * @return a double-double map where the key denotes the bin and the value the bin height.
098     */
099    public static TDoubleDoubleHashMap createHistogram(double[] values, double[] weights, Discretizer discretizer, boolean reweight) {
100        TDoubleDoubleHashMap histogram = new TDoubleDoubleHashMap();
101        for (int i = 0; i < values.length; i++) {
102            double bin = discretizer.discretize(values[i]);
103            double weight = weights[i];
104            if (reweight) {
105                weight = weights[i] / discretizer.binWidth(values[i]);
106            }
107
108            histogram.adjustOrPutValue(bin, weight, weight);
109        }
110
111        return histogram;
112    }
113
114    /**
115     * Normalizes a histogram so that the sum of all bin heights equals 1.
116     *
117     * @param histogram a histogram
118     * @return a normalized histogram.
119     */
120    public static TDoubleDoubleHashMap normalize(TDoubleDoubleHashMap histogram) {
121        double sum = 0;
122        double[] values = histogram.values();
123
124        for (int i = 0; i < values.length; i++) {
125            sum += values[i];
126        }
127
128        return normalize(histogram, sum);
129    }
130
131    /**
132     * Normalizes a histogram so that the sum of all bin heights equals <tt>sum</tt>.
133     *
134     * @param histogram a histogram
135     * @param sum       the sum of all bin heights
136     * @return a normalized histogram
137     */
138    public static TDoubleDoubleHashMap normalize(TDoubleDoubleHashMap histogram, double sum) {
139        final double norm = 1 / sum;
140
141        TDoubleFunction fct = new TDoubleFunction() {
142            public double execute(double value) {
143                return value * norm;
144            }
145
146        };
147
148        histogram.transformValues(fct);
149
150        return histogram;
151
152    }
153
154    /**
155     * Calculates the sum of all bin heights.
156     *
157     * @param histogram a histogram
158     * @return the sum of all bin heights
159     */
160    public static double sum(TDoubleDoubleHashMap histogram) {
161        TDoubleDoubleIterator it = histogram.iterator();
162        double sum = 0;
163        for (int i = 0; i < histogram.size(); i++) {
164            it.advance();
165            sum += it.value();
166        }
167        return sum;
168    }
169
170    /**
171     * Creates a cumulative histogram where each bin height is the sum of the previous bin's height and the current
172     * bin's height.
173     *
174     * @param histogram a histogram
175     * @return a cumulative histogram
176     */
177    public static TDoubleDoubleHashMap createCumulativeHistogram(TDoubleDoubleHashMap histogram) {
178        double[] keys = histogram.keys();
179        Arrays.sort(keys);
180        double sum = 0;
181        for (double key : keys) {
182            double val = histogram.get(key);
183            sum += val;
184            histogram.put(key, sum);
185        }
186
187        return histogram;
188    }
189
190    /**
191     * Normalizes a (cumulative) histogram so that the last bin's height equals 1.
192     *
193     * @param histogram a (cumulative) histogram
194     * @return a normalized (cumulative) histogram
195     */
196    public static TDoubleDoubleHashMap normalizeCumulative(TDoubleDoubleHashMap histogram) {
197        return normalizeCumulative(histogram, 1.0);
198    }
199
200    /**
201     * Normalizes a (cumulative) histogram so that the last bin's height equals <tt>sum</tt>.
202     *
203     * @param histogram a (cumulative) histogram
204     * @param sum       the value if the last bin's height
205     * @return a normalized (cumulative) histogram
206     */
207    public static TDoubleDoubleHashMap normalizeCumulative(TDoubleDoubleHashMap histogram, double sum) {
208        double[] keys = histogram.keys();
209        double max = StatUtils.max(keys);
210
211        final double norm = sum / histogram.get(max);
212
213        TDoubleFunction fct = new TDoubleFunction() {
214            public double execute(double value) {
215                return value * norm;
216            }
217
218        };
219
220        histogram.transformValues(fct);
221
222        return histogram;
223    }
224
225    /**
226     * Calculates the complementary (1 - bin height) of a (normalized) histogram.
227     *
228     * @param histogram a (normalized) histogram
229     * @return the complementary histogram
230     */
231    public static TDoubleDoubleHashMap complementary(TDoubleDoubleHashMap histogram) {
232        TDoubleFunction fct = new TDoubleFunction() {
233            public double execute(double value) {
234                return 1 - value;
235            }
236
237        };
238
239        histogram.transformValues(fct);
240
241        return histogram;
242    }
243}
