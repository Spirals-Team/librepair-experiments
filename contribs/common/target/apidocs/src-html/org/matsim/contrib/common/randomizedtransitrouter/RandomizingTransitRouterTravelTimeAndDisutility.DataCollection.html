037/**
038 * When plugged into the transit router, will switch to a different, randomly generated combination of
039 * marginal utilities every time it is called for a new agent.
040 * <p></p>
041 * Comments:<ul>
042 * <li> In the literature, they seem to use log-normal distributions for the tastes.  (Makes sense
043 * since it addresse the symmetry problem which I am addressing ad-hoc.) [M. K. Anderson, hEART'13]
044 * kai, sep'13
045 * <li> People seem to use "gamma distributions for the error term" ("double stochastic assignment").
046 * Not sure what that does, if we need it, etc.  [M. K. Anderson, hEART'13] kai, sep'13
047 * </ul>
048 * @author kai
049 * @author dgrether
050 *
051 */
052public class RandomizingTransitRouterTravelTimeAndDisutility  extends TransitRouterNetworkTravelTimeAndDisutility {
053
054        public enum DataCollection {randomizedParameters, additionalInformation}
055        private Id<Person> cachedPersonId = null ;
056        private final TransitRouterConfig originalTransitRouterConfig ;
057
058        private double localMarginalUtilityOfTravelTimeWalk_utl_s = Double.NaN ;
059        private double localMarginalUtilityOfWaitingPt_utl_s = Double.NaN ;
060        private double localUtilityOfLineSwitch_utl = Double.NaN ;
061        private double localMarginalUtilityOfTravelTimePt_utl_s = Double.NaN ;
062        private double localMarginalUtilityOfTravelDistancePt_utl_m = Double.NaN ;
063
064        private Map<DataCollection,Boolean> dataCollectionConfig = new HashMap<DataCollection,Boolean>() ;
065        private Map<DataCollection,StringBuffer> dataCollectionStrings = new HashMap<DataCollection,StringBuffer>() ;
066        
067        public RandomizingTransitRouterTravelTimeAndDisutility(TransitRouterConfig routerConfig) {
068                super(routerConfig);
069                
070                prepareDataCollection();
071                
072                // make sure that some parameters are not zero since otherwise the randomization will not work:
073
074                // marg utl time wlk should be around -3/h or -(3/3600)/sec.  Give warning if not at least 1/3600:
075                if ( -routerConfig.getMarginalUtilityOfTravelTimeWalk_utl_s() < 1./3600. ) {
076                        Logger.getLogger(this.getClass()).warn( "marg utl of walk rather close to zero; randomization may not work") ;
077                }
078                // utl of line switch should be around -300sec or -0.5u.  Give warning if not at least 0.1u:
079                if ( -routerConfig.getUtilityOfLineSwitch_utl() < 0.1 ) {
080                        Logger.getLogger(this.getClass()).warn( "utl of line switch rather close to zero; randomization may not work") ;
081                }
082                        
083                this.originalTransitRouterConfig = routerConfig ;
084
085                this.localMarginalUtilityOfTravelDistancePt_utl_m = routerConfig.getMarginalUtilityOfTravelDistancePt_utl_m();
086                this.localMarginalUtilityOfTravelTimePt_utl_s = routerConfig.getMarginalUtilityOfTravelTimePt_utl_s() ;
087                this.localMarginalUtilityOfTravelTimeWalk_utl_s = routerConfig.getMarginalUtilityOfTravelTimeWalk_utl_s() ;
088//              this.localMarginalUtilityOfWaitingPt_utl_s = routerConfig.getMarginalUtilityOfTravelTimePt_utl_s() ;
089                this.localMarginalUtilityOfWaitingPt_utl_s = routerConfig.getMarginalUtilityOfWaitingPt_utl_s() ;
090                this.localUtilityOfLineSwitch_utl = routerConfig.getUtilityOfLineSwitch_utl() ;
091        }
092        public final String getDataCollectionString( DataCollection item ) {
093                return dataCollectionStrings.get(item).toString() ;
094        }
095        
096        @Override
097        public double getLinkTravelDisutility(final Link link, final double time, final Person person, final Vehicle vehicle, 
098                        final CustomDataManager dataManager) {
099                
100                regenerateUtilityParametersIfPersonHasChanged(person);
101                                
102                double disutl;
103                if (((TransitRouterNetworkLink) link).getRoute() == null) {
104                        // (this means that it is a transfer link (walk))
105
106                        double transfertime = getLinkTravelTime(link, time, person, vehicle);
107                        double waittime = this.originalTransitRouterConfig.getAdditionalTransferTime();
108                        
109                        // say that the effective walk time is the transfer time minus some "buffer"
110                        double walktime = transfertime - waittime;
111                        
112                        disutl = - walktime * localMarginalUtilityOfTravelTimeWalk_utl_s
113                               - waittime * localMarginalUtilityOfWaitingPt_utl_s
114                               - localUtilityOfLineSwitch_utl;
115                        
116                } else {
117                        
118                        double offVehWaitTime = offVehicleWaitTime(link, time);
119                        
120                        double inVehTime = getLinkTravelTime(link,time, person, vehicle) - offVehWaitTime ;
121
122                        disutl = -inVehTime * this.localMarginalUtilityOfTravelTimePt_utl_s 
123                                        - offVehWaitTime * this.localMarginalUtilityOfWaitingPt_utl_s
124                                        - link.getLength() * this.localMarginalUtilityOfTravelDistancePt_utl_m;
125                }
126
127                if ( this.dataCollectionConfig.get(DataCollection.additionalInformation )) {
128                        StringBuffer strb = this.dataCollectionStrings.get(DataCollection.additionalInformation ) ;
129                        strb.append("also collecting additional information") ;
130                }
131
132                return disutl;
133        }
134        
135        @Override
136        public double getWalkTravelDisutility(Person person, Coord coord, Coord toCoord) {
137                regenerateUtilityParametersIfPersonHasChanged(person);
138                return - getWalkTravelTime(person, coord, toCoord) * localMarginalUtilityOfTravelTimeWalk_utl_s ;
139        }
140        
141        public final void setDataCollection( DataCollection item, Boolean bbb ) {
142                Logger.getLogger(this.getClass()).info( " settin data collection of " + item.toString() + " to " + bbb.toString() ) ;
143                dataCollectionConfig.put( item, bbb ) ;
144        }
145
146        private void prepareDataCollection() {
147                for ( DataCollection dataCollection : DataCollection.values() ) {
148                        switch ( dataCollection ) {
149                        case randomizedParameters:
150                                dataCollectionConfig.put( dataCollection, false ) ;
151                                dataCollectionStrings.put( dataCollection, new StringBuffer() ) ;
152                                break;
153                        case additionalInformation:
154                                dataCollectionConfig.put( dataCollection, false ) ;
155                                dataCollectionStrings.put( dataCollection, new StringBuffer() ) ;
156                                break;
157                        }
158                }
159        }
160
161        private void regenerateUtilityParametersIfPersonHasChanged(final Person person) {
162                if ( !person.getId().equals(this.cachedPersonId)) {
163                        // yyyyyy probably not thread safe (?!?!)
164                        
165                        // person has changed, so ...
166                        
167                        // ... memorize new person id:
168                        this.cachedPersonId = person.getId() ;
169                        
170                        // ... generate new random parameters:
171                        {
172                                double tmp = this.originalTransitRouterConfig.getMarginalUtilityOfTravelTimeWalk_utl_s() ;
173                                tmp *= 5. * MatsimRandom.getRandom().nextDouble() ;
174                                localMarginalUtilityOfTravelTimeWalk_utl_s = tmp ;
175                                // yy if this becomes too small, they may walk the whole way (is it really clear why this can happen?)
176                        }
177                        {
178                                double tmp = this.originalTransitRouterConfig.getUtilityOfLineSwitch_utl() ;
179                                tmp *= 5. * MatsimRandom.getRandom().nextDouble() ;
180                                localUtilityOfLineSwitch_utl = tmp ;
181                        }
182                        {
183                                double tmp = this.originalTransitRouterConfig.getMarginalUtilityOfWaitingPt_utl_s();
184                                tmp *= 5. * MatsimRandom.getRandom().nextDouble();
185                                localMarginalUtilityOfWaitingPt_utl_s = tmp;
186                        }
187                        {
188                                // (Conceptually, the following is not necessary, but empirically, it seems to help. kai, jan'13)
189                                double tmp = this.originalTransitRouterConfig.getMarginalUtilityOfTravelTimePt_utl_s() ;
190                                tmp *= 5. * MatsimRandom.getRandom().nextDouble();
191                                localMarginalUtilityOfTravelTimePt_utl_s = tmp;
192                        }
193                        
194                        if ( this.dataCollectionConfig.get(DataCollection.randomizedParameters) ) {
195//                              StringBuffer strb = this.dataCollectionStrings.get(DataCollection.randomizedParameters) ;
196//                              strb.append
197                                System.out.println("personId: " + person.getId() +
198                                                "; margUtlOfTimeWlk_h: " + this.localMarginalUtilityOfTravelTimeWalk_utl_s*3600. +
199                                                "; utlOfLineSwitch: " + this.localUtilityOfLineSwitch_utl +
200                                                "; margUtlOfWait_h: " + this.localMarginalUtilityOfWaitingPt_utl_s*3600. + 
201                                                "; margUtlOfTimePt_h: " + this.localMarginalUtilityOfTravelTimePt_utl_s*3600. ) ;
202                        }
203                }
204        }
