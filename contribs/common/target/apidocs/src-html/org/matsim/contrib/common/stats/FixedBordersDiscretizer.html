020package org.matsim.contrib.common.stats;
021
022import org.apache.commons.math.stat.StatUtils;
023
024import java.util.Arrays;
025
026/**
027 * A discretizer with predefined bin borders. Values are rounded up to the upper
028 * bin border. A bin starts at the lower bin border (exclusive) and ends at the
029 * upper bin border (inclusive).
030 * 
031 * @author illenberger
032 * 
033 */
034public class FixedBordersDiscretizer implements Discretizer {
035
036        private double[] borders;
037
038        /**
039         * Creates a new discretizer with the borders specified in <tt>borders</tt>.
040         * 
041         * @param borders
042         *            an array with borders.
043         */
044        public FixedBordersDiscretizer(double[] borders) {
045                this.borders = borders;
046                Arrays.sort(this.borders);
047        }
048
049        /**
050         * Creates a new discretizer with the borders specified in <tt>borders</tt>
051         * but additionally adds a lower border <tt>min[values] - 1E-10</tt> and an
052         * upper border <tt>max[values]</tt>, so that if only values contained in
053         * <tt>values</tt> are passed to {@link #discretize(double)} or
054         * {@link #binWidth(double)} both methods never return
055         * {@link Double#POSITIVE_INFINITY}.
056         * 
057         * @param values
058         *            an array of values.
059         * @param borders
060         *            an array with borders.
061         */
062        public FixedBordersDiscretizer(double[] values, double[] borders) {
063                double min = StatUtils.min(values);
064                double max = StatUtils.max(values);
065
066                min = min - 1E-10;
067
068                double[] newBorders = new double[borders.length + 2];
069                newBorders[0] = min;
070                for (int i = 0; i < borders.length; i++)
071                        newBorders[i + 1] = borders[i];
072
073                newBorders[newBorders.length - 1] = max;
074
075                this.borders = newBorders;
076                Arrays.sort(this.borders);
077        }
078
079        /**
080         * @param value
081         *            a value.
082         * @return the upper bin border of <tt>value</tt> or
083         *         {@link Double#POSITIVE_INFINITY} if <tt>value</tt> is greater
084         *         than the greatest border.
085         */
086        @Override
087        public double discretize(double value) {
088                int idx = index(value);
089                if (idx >= borders.length)
090                        return Double.POSITIVE_INFINITY;
091                else
092                        return borders[idx];
093        }
094
095        /**
096         * @param value
097         *            a value.
098         * @returns the width of the bin of <tt>value</tt> or
099         *          {@link Double#POSITIVE_INFINITY} if <tt>value</tt> is less than
100         *          the smallest border or greater than the greatest border.
101         */
102        @Override
103        public double binWidth(double value) {
104                int idx = index(value);
105                if (idx == 0 || idx >= borders.length)
106                        return Double.POSITIVE_INFINITY;
107                else
108                        return borders[idx] - borders[idx - 1];
109        }
110
111        /**
112         * @see {@link Discretizer#index(double)}
113         */
114        @Override
115        public int index(double value) {
116                int idx = Arrays.binarySearch(borders, value);
117                if (idx > -1) {
118                        return idx;
119                } else {
120                        return -idx - 1;
121                }
122        }
123}
