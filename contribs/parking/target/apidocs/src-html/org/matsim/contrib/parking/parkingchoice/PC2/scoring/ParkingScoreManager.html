019package org.matsim.contrib.parking.parkingchoice.PC2.scoring;
020
021import java.util.Map;
022
023import org.matsim.api.core.v01.Coord;
024import org.matsim.api.core.v01.Id;
025import org.matsim.api.core.v01.Scenario;
026import org.matsim.api.core.v01.network.Link;
027import org.matsim.api.core.v01.population.Person;
028import org.matsim.contrib.multimodal.router.util.WalkTravelTime;
029import org.matsim.contrib.parking.parkingchoice.PC2.infrastructure.PC2Parking;
030import org.matsim.contrib.parking.parkingchoice.lib.DebugLib;
031import org.matsim.contrib.parking.parkingchoice.lib.obj.DoubleValueHashMap;
032import org.matsim.core.network.NetworkUtils;
033import org.matsim.core.utils.geometry.CoordUtils;
034
035public final class ParkingScoreManager implements ParkingScore {
036        
037        private Double beelineDistanceFactor = 1.3 ;
038
039        private AbstractParkingBetas parkingBetas;
040        private double parkingScoreScalingFactor;
041        private double randomErrorTermScalingFactor;
042        private DoubleValueHashMap<Id<Person>> scores;
043        private final Scenario scenario;
044        private final WalkTravelTime walkTravelTime;
045        private RandomErrorTermManager randomErrorTermManager;
046
047        public ParkingScoreManager(WalkTravelTime walkTravelTime, Scenario scenario) {
048                this.walkTravelTime = walkTravelTime;
049                this.scenario = scenario;
050        }
051
052        @Override
053        public double calcWalkScore(Coord destCoord, PC2Parking parking, Id<Person> personId, double parkingDurationInSeconds) {
054                Map<Id<Person>, ? extends Person> persons = scenario.getPopulation().getPersons();
055                Person person = persons.get(personId);
056
057                double parkingWalkBeta = getParkingBetas().getParkingWalkBeta(person, parkingDurationInSeconds);
058
059                Link link = NetworkUtils.getNearestLink((scenario.getNetwork()), destCoord);
060                
061                double linkLength = link.getLength();           
062                
063                double walkDistance = CoordUtils.calcEuclideanDistance(destCoord, parking.getCoordinate()) 
064                                * scenario.getConfig().plansCalcRoute().getBeelineDistanceFactors().get("walk")* beelineDistanceFactor;
065                
066                double walkSpeed = linkLength / this.walkTravelTime.getLinkTravelTime(link, 0, person, null);
067
068                double walkDurationInSeconds = walkDistance / walkSpeed;
069
070                double walkingTimeTotalInMinutes = walkDurationInSeconds / 60;
071
072                if (parking.getId().toString().contains("stp")){
073                        DebugLib.emptyFunctionForSettingBreakPoint();
074                }
075
076                return (parkingWalkBeta * walkingTimeTotalInMinutes) * parkingScoreScalingFactor;
077        }
078
079        @Override
080        public double calcCostScore(double arrivalTime, double parkingDurationInSeconds, PC2Parking parking, Id<Person> personId) {
081                Person person = scenario.getPopulation().getPersons().get(personId);
082                double parkingCostBeta = getParkingBetas().getParkingCostBeta(person);
083                double parkingCost = parking.getCost(personId, arrivalTime, parkingDurationInSeconds);
084                return (parkingCostBeta * parkingCost) * parkingScoreScalingFactor;
085        }
086        
087        
088        @Override
089        public double calcScore(Coord destCoord, double arrivalTime, double parkingDurationInSeconds, PC2Parking parking, Id<Person> personId, int legIndex, boolean setCostToZero) {
090                double walkScore = calcWalkScore(destCoord, parking, personId, parkingDurationInSeconds);
091                double costScore = calcCostScore(arrivalTime, parkingDurationInSeconds, parking, personId);
092                
093                if (setCostToZero){
094                        costScore=0;
095                }
096                
097                double randomError=0;
098
099                if (randomErrorTermManager!=null){
100                        randomError= randomErrorTermManager.getEpsilonAlternative(parking.getId(),personId,legIndex)*randomErrorTermScalingFactor*parkingScoreScalingFactor;
101                }
102                return costScore + walkScore + randomError;
103        }
104
105        @Override
106        public double getScore(Id<Person> id) {
107                return scores.get(id);
108        }
109
110        @Override
111        public synchronized void addScore(Id<Person> id, double incValue) {
112                scores.incrementBy(id, incValue);
113        }
114
115        @Override
116        public synchronized void prepareForNewIteration() {
117                scores = new DoubleValueHashMap<>();
118        }
119
120        @Override
121        public double getParkingScoreScalingFactor() {
122                return parkingScoreScalingFactor;
123        }
124
125        @Override
126        public void setParkingScoreScalingFactor(double parkingScoreScalingFactor) {
127                this.parkingScoreScalingFactor = parkingScoreScalingFactor;
128        }
129
130        @Override
131        public double getRandomErrorTermScalingFactor() {
132                return randomErrorTermScalingFactor;
133        }
134
135        @Override
136        public void setRandomErrorTermScalingFactor(double randomErrorTermScalingFactor) {
137                this.randomErrorTermScalingFactor = randomErrorTermScalingFactor;
138        }
139
140        @Override
141        public AbstractParkingBetas getParkingBetas() {
142                return parkingBetas;
143        }
144
145        @Override
146        public void setParkingBetas(AbstractParkingBetas parkingBetas) {
147                this.parkingBetas = parkingBetas;
148        }
149
150        @Override
151        public void setRandomErrorTermManger(RandomErrorTermManager randomErrorTermManager) {
152                this.randomErrorTermManager = randomErrorTermManager;
153        }
154
155}
