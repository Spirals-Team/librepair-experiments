048/**
049 * @author  jbischoff
050 *
051 */
052
053public class SetupParking {
054        // TODO: create config group and make this all configurable?
055
056        static public void installParkingModules(Controler controler) {
057                // No need to route car routes in Routing module in advance, as they are
058                // calculated on the fly
059                if (!controler.getConfig().getModules().containsKey(DvrpConfigGroup.GROUP_NAME)){
060                controler.getConfig().addModule(new DvrpConfigGroup());
061                }
062                final DynRoutingModule routingModuleCar = new DynRoutingModule(TransportMode.car);
063                StageActivityTypes stageActivityTypesCar = new StageActivityTypes() {
064                        @Override
065                        public boolean isStageActivity(String activityType) {
066
067                                return (activityType.equals(ParkingUtils.PARKACTIVITYTYPE));
068                        }
069                };
070                routingModuleCar.setStageActivityTypes(stageActivityTypesCar);
071                controler.addOverridingModule(new DvrpTravelTimeModule());
072                controler.addOverridingModule(new AbstractModule() {
073                        @Override
074                        public void install() {
075                                addRoutingModuleBinding(TransportMode.car).toInstance(routingModuleCar);
076                                bind(Network.class).annotatedWith(Names.named(DvrpRoutingNetworkProvider.DVRP_ROUTING)).to(Network.class).asEagerSingleton();
077                                bind(ParkingSearchManager.class).to(FacilityBasedParkingManager.class).asEagerSingleton();
078                                bind(WalkLegFactory.class).asEagerSingleton();
079                                bind(PrepareForSim.class).to(ParkingSearchPrepareForSimImpl.class);
080                                this.install(new ParkingSearchQSimModule());
081                                addControlerListenerBinding().to(ParkingListener.class);
082                                bind(ParkingRouter.class).to(WithinDayParkingRouter.class);
083                                bind(VehicleTeleportationLogic.class).to(VehicleTeleportationToNearbyParking.class);
084                        }
085                });
086
087        }
088
089}
