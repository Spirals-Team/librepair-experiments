057 */
058public class ParkingAgentLogic implements DynAgentLogic {
059
060        public enum LastParkActionState  {
061                        
062                        // we have the following cases of ending dynacts:
063                        NONCARTRIP,     // non-car trip arrival: start Activity
064                        CARTRIP, // car-trip arrival: add park-car activity 
065                        PARKACTIVITY, // park-car activity: get next PlanElement & add walk leg to activity location
066                        WALKFROMPARK ,// walk-leg to act: start next PlanElement Activity
067                        ACTIVITY, // ordinary activity: get next Leg, if car: go to car, otherwise add ordinary leg by other mode
068                        WALKTOPARK, // walk-leg to car: add unpark activity
069                        UNPARKACTIVITY // unpark activity: find the way to the next route & start leg
070        }
071        protected LastParkActionState lastParkActionState;
072        protected DynAgent agent;
073        protected Iterator<PlanElement> planElemIter;
074        protected PlanElement currentPlanElement;
075        protected ParkingSearchManager parkingManager;
076        protected WalkLegFactory walkLegFactory;
077        protected ParkingRouter parkingRouter;
078        protected MobsimTimer timer;
079        protected EventsManager events;
080        protected ParkingSearchLogic parkingLogic;
081        protected VehicleTeleportationLogic teleportationLogic;
082        protected boolean isinitialLocation = true;
083        protected Id<Vehicle> currentlyAssignedVehicleId = null;
084        protected String stageInteractionType = null;
085        private ParkingSearchConfigGroup configGroup;
086
087        /**
088         * @param plan
089         *            (always starts with Activity)
090         */
091        public ParkingAgentLogic(Plan plan, ParkingSearchManager parkingManager, WalkLegFactory walkLegFactory, ParkingRouter parkingRouter, EventsManager events, ParkingSearchLogic parkingLogic, MobsimTimer timer, VehicleTeleportationLogic teleportationLogic, ParkingSearchConfigGroup configGroup) {
092                planElemIter = plan.getPlanElements().iterator();
093                this.parkingManager = parkingManager;
094                this.walkLegFactory = walkLegFactory;
095                this.parkingRouter = parkingRouter;
096                this.timer = timer;
097                this.events = events;
098                this.parkingLogic = parkingLogic;
099                this.teleportationLogic = teleportationLogic;
100                this.configGroup = configGroup;
101                
102                
103        }
104
105        @Override
106        public DynActivity computeInitialActivity(DynAgent adapterAgent) {
107                this.agent = adapterAgent;
108                this.lastParkActionState = LastParkActionState.ACTIVITY;
109                this.currentPlanElement = planElemIter.next();
110                Activity act = (Activity) currentPlanElement;
111                //TODO: assume something different regarding initial parking location
112                
113                return new StaticDynActivity(act.getType(), act.getEndTime());
114        }
115
116        @Override
117        public DynAgent getDynAgent() {
118                return agent;
119        }
120
121        @Override
122        public DynAction computeNextAction(DynAction oldAction, double now) {
123                // we have the following cases of ending dynacts:
124                // non-car trip arrival: start Activity
125                // car-trip arrival: add park-car activity 
126                // park-car activity: get next PlanElement & add walk leg to activity location
127                // walk-leg to act: start next PlanElement Activity
128                // ordinary activity: get next Leg, if car: go to car, otherwise add ordinary leg by other mode
129                // walk-leg to car: add unpark activity
130                // unpark activity: find the way to the next route & start leg
131                switch (lastParkActionState){
132                case ACTIVITY:
133                        return nextStateAfterActivity(oldAction, now);
134                                
135                case CARTRIP:
136                        return nextStateAfterCarTrip(oldAction,now);
137                        
138                case NONCARTRIP:
139                        return nextStateAfterNonCarTrip(oldAction,now);
140                        
141                case PARKACTIVITY:
142                        return nextStateAfterParkActivity(oldAction,now);
143                
144                case UNPARKACTIVITY:
145                        return nextStateAfterUnParkActivity(oldAction,now);
146                        
147                case WALKFROMPARK:
148                        return nextStateAfterWalkFromPark(oldAction,now);
149                        
150                case WALKTOPARK:
151                        return nextStateAfterWalkToPark(oldAction,now);
152                
153                }
154                throw new RuntimeException("unreachable code");
155
156        }
157
158        protected DynAction nextStateAfterUnParkActivity(DynAction oldAction, double now) {
159                // we have unparked, now we need to get going by car again.
160                
161                Leg currentPlannedLeg = (Leg) currentPlanElement;
162                Route plannedRoute = currentPlannedLeg.getRoute();
163                NetworkRoute actualRoute = this.parkingRouter.getRouteFromParkingToDestination(plannedRoute.getEndLinkId(), now, agent.getCurrentLinkId());
164                if ((this.parkingManager.unParkVehicleHere(currentlyAssignedVehicleId, agent.getCurrentLinkId(), now))||(isinitialLocation)){
165                        this.lastParkActionState = LastParkActionState.CARTRIP;
166                        isinitialLocation = false;
167                        Leg currentLeg = (Leg) this.currentPlanElement;
168                        //this could be Car, Carsharing, Motorcylce, or whatever else mode we have, so we want our leg to reflect this.
169                        return new ParkingDynLeg(currentLeg.getMode(), actualRoute, parkingLogic, parkingManager, currentlyAssignedVehicleId, timer, events);
170                
171                }
172                else throw new RuntimeException("parking location mismatch");
173                
174        }
175
176        protected DynAction nextStateAfterWalkToPark(DynAction oldAction, double now) {
177                //walk2park is complete, we can unpark.
178                this.lastParkActionState = LastParkActionState.UNPARKACTIVITY;
179                return new StaticDynActivity(this.stageInteractionType, now + configGroup.getUnparkduration());
180        }
181
182        protected DynAction nextStateAfterWalkFromPark(DynAction oldAction, double now) {
183                //walkleg complete, time to get the next activity from the plan Elements and start it, this is basically the same as arriving on any other mode
184                return nextStateAfterNonCarTrip(oldAction, now);
185        }
186
187        protected DynAction nextStateAfterParkActivity(DynAction oldAction, double now) {
188                // add a walk leg after parking
189                Leg currentPlannedLeg = (Leg) currentPlanElement;
190                Id<Link> walkDestination = currentPlannedLeg.getRoute().getEndLinkId();
191                Leg walkLeg = walkLegFactory.createWalkLeg(agent.getCurrentLinkId(), walkDestination, now, TransportMode.egress_walk);
192                this.lastParkActionState = LastParkActionState.WALKFROMPARK;
193                this.stageInteractionType = null;
194                return new StaticPassengerDynLeg(walkLeg.getRoute(), walkLeg.getMode());
195        }
196
197        protected DynAction nextStateAfterNonCarTrip(DynAction oldAction, double now) {
198                // switch back to activity
199                this.currentPlanElement = planElemIter.next();
200                Activity nextPlannedActivity = (Activity) this.currentPlanElement;
201                this.lastParkActionState = LastParkActionState.ACTIVITY;
202                double endTime =nextPlannedActivity.getEndTime() ; 
203                if (endTime == Time.UNDEFINED_TIME){
204                        endTime = Double.POSITIVE_INFINITY;
205                }
206                         
207                return new StaticDynActivity(nextPlannedActivity.getType(),endTime);
208                
209        }
210
211        protected DynAction nextStateAfterCarTrip(DynAction oldAction, double now) {
212                // car trip is complete, we have found a parking space (not part of the logic), block it and start to park
213                if (this.parkingManager.parkVehicleHere(Id.create(this.agent.getId(), Vehicle.class), agent.getCurrentLinkId(), now)){
214                this.lastParkActionState = LastParkActionState.PARKACTIVITY;
215                this.currentlyAssignedVehicleId = null;
216                this.parkingLogic.reset();
217                return new StaticDynActivity(this.stageInteractionType,now + configGroup.getParkduration());}
218                else throw new RuntimeException ("No parking possible");
219        }
220
221        protected DynAction nextStateAfterActivity(DynAction oldAction, double now) {
222                // we could either depart by car or not next
223                if (planElemIter.hasNext()){
224                this.currentPlanElement = planElemIter.next();
225                Leg currentLeg = (Leg) currentPlanElement;
226                if (currentLeg.getMode().equals(TransportMode.car)){
227                        Id<Vehicle> vehicleId = Id.create(this.agent.getId(), Vehicle.class);
228                        Id<Link> parkLink = this.parkingManager.getVehicleParkingLocation(vehicleId);
229                        
230                        if (parkLink == null){
231                                //this is the first activity of a day and our parking manager does not provide informations about initial stages. We suppose the car is parked where we are
232                                parkLink = agent.getCurrentLinkId();
233                        }
234                        
235                        Id<Link> telePortedParkLink = this.teleportationLogic.getVehicleLocation(agent.getCurrentLinkId(), vehicleId, parkLink, now);
236                        Leg walkleg = walkLegFactory.createWalkLeg(agent.getCurrentLinkId(), telePortedParkLink, now, TransportMode.access_walk);
237                        this.lastParkActionState = LastParkActionState.WALKTOPARK;
238                        this.currentlyAssignedVehicleId = vehicleId;
239                        this.stageInteractionType = ParkingUtils.PARKACTIVITYTYPE;
240                        return new StaticPassengerDynLeg(walkleg.getRoute(), walkleg.getMode());
241                }
242                else if (currentLeg.getMode().equals(TransportMode.pt)) {
243                        if (currentLeg.getRoute() instanceof ExperimentalTransitRoute){
244                                throw new IllegalStateException ("not yet implemented");
245                        }
246                        else {
247                                this.lastParkActionState = LastParkActionState.NONCARTRIP;
248                                return new StaticPassengerDynLeg(currentLeg.getRoute(), currentLeg.getMode());
249                        }
250                //teleport or pt route  
251                } 
252                else {
253                //teleport      
254                        this.lastParkActionState = LastParkActionState.NONCARTRIP;
255                        return new StaticPassengerDynLeg(currentLeg.getRoute(), currentLeg.getMode());
256                }
257                
258        }else throw new RuntimeException("no more leg to follow but activity is ending\nLastPlanElement: "+currentPlanElement.toString()+"\n Agent "+this.agent.getId()+"\nTime: "+Time.writeTime(now));
259        }
