004package org.matsim.contrib.parking.parkingsearch.evaluation;
005

007/**
008 * @author schlenther
009 *
010 */
011public final class ParkingTuple implements Comparable{
012        
013        private double time;
014        private double occupancy;
015        
016        public ParkingTuple(double time, double occupancy){
017                this.time = time;
018                this.occupancy = occupancy;
019        }
020        
021        public double getTime(){
022                return this.time;
023        }
024        
025        public double getOccupancy(){
026                return this.occupancy;
027        }
028
029        @Override
030        public int compareTo(Object o) {
031                if (!(o instanceof ParkingTuple)){
032                        throw new RuntimeException("cannot compare ParkingTuple with another object");
033                }
034                else{
035                        ParkingTuple other = (ParkingTuple) o;
036                        return (other.getTime() < this.getTime()) ? 1 : (other.getTime() == this.getTime() ? 0 : -1);
037                }
038        }
039        
040        @Override
041        public String toString(){
042                return "" + this.time + ":" + this.occupancy;
043        }
044        
045        
046}
