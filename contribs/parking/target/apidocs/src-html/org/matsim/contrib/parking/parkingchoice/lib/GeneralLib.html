200        public static void writeList(ArrayList<String> list, String fileName) {
201                Writer writer=null;
202                if (fileName.toLowerCase(Locale.ROOT).endsWith(".gz")) {
203                        writer = IOUtils.getBufferedWriter(fileName);
204                }
205                
206                try {
207                        if (writer==null){

209                                writer = new OutputStreamWriter(fos);

211                        char[] charArray = Lists
212                                        .getCharsOfAllArrayItemsWithNewLineCharacterInbetween(list);
213                        writer.write(charArray);
214
215                        writer.flush();
216                        writer.close();
217                } catch (Exception e) {
218                        e.printStackTrace();

246         * if headerLine=null, then add no line at top of file. "\n" is added at end
247         * of first line by this method.
248         * 
249         * matrix[numberOfRows][numberOfColumns]
250         * 
251         * @param matrix
252         * @param fileName
253         * @param headerLine
254         */
255        public static void writeMatrix(double[][] matrix, String fileName,
256                        String headerLine) {
257                ArrayList<String> list = new ArrayList<String>();
258
259                if (headerLine != null) {
260                        list.add(headerLine);
261                }
262
263                for (int i = 0; i < matrix.length; i++) {
264                        String line = "";
265                        for (int j = 0; j < matrix[0].length - 1; j++) {
266                                line += matrix[i][j];
267                                line += "\t";
268                        }
269                        line += matrix[i][matrix[0].length - 1];
270                        list.add(line);
271                }
272
273                writeList(list, fileName);
274        }

276        /**
277         * reads in data from a file.
278         * 
279         * 
280         * @param numberOfRows
281         * @param numberOfColumns
282         * @param ignoreFirstLine
283         * @return
284         */
285        public static double[][] readMatrix(int numberOfRows, int numberOfColumns,
286                        boolean ignoreFirstLine, String fileName) {
287
288                double[][] matrix = new double[numberOfRows][numberOfColumns];
289
290                try {
291
292                        FileReader fr = new FileReader(fileName);
293
294                        BufferedReader br = new BufferedReader(fr);
295                        String line;
296                        StringTokenizer tokenizer;
297                        String token;
298
299                        if (ignoreFirstLine) {
300                                br.readLine();
301                        }
302
303                        line = br.readLine();
304                        int rowId = 0;
305                        while (line != null) {
306                                tokenizer = new StringTokenizer(line);
307
308                                for (int i = 0; i < numberOfColumns; i++) {
309                                        token = tokenizer.nextToken();
310                                        double parsedNumber = Double.parseDouble(token);
311                                        matrix[rowId][i] = parsedNumber;
312                                }
313
314                                if (tokenizer.hasMoreTokens()) {
315                                        // if there are more columns than expected, throw an
316                                        // exception
317                                        throw new RuntimeException("the number of columns is wrong");
318                                }
319
320                                line = br.readLine();
321                                rowId++;
322                        }
323                        if (rowId != numberOfRows) {
324                                throw new RuntimeException("the number of rows is wrong");
325                        }
326
327                } catch (RuntimeException e) {
328                        // just forward the runtime exception
329                        throw e;
330                } catch (Exception e) {
331                        e.printStackTrace();
332
333                        throw new RuntimeException("Error reading the matrix from the file");
334                }
335
336                return matrix;
337        }

368
369                File sourceDirectory = new File(sourceDirectoryPath);
370                File targetDirectory = new File(targetDirectoryPath);
371
372                try {
373                        if (sourceDirectory.isDirectory()) {
374                                // do recursion
375                                if (!targetDirectory.exists()) {
376                                        targetDirectory.mkdir();
377                                }
378
379                                String[] children = sourceDirectory.list();
380                                for (int i = 0; i < children.length; i++) {
381                                        copyDirectory(sourceDirectoryPath + "\\" + children[i],
382                                                        targetDirectory + "\\" + children[i]);
383                                }
384                        } else {
385
386                                // copy the file
387                                copyFile(sourceDirectory.getAbsolutePath(),
388                                                targetDirectory.getAbsolutePath());
389                        }
390                } catch (Exception e) {
391                        e.printStackTrace();
392                }

617        /**
618         * If time is > 60*60*24 [seconds], it will be projected into next day, e.g.
619         * time=60*60*24+1=1
620         * 
621         * even if time is negative, it is turned into a positive time by adding
622         * number of seconds of day into it consecutively
623         * 
624         * @param time
625         * @return
626         */
627        public static double projectTimeWithin24Hours(double time) {
628                double secondsInOneDay = 60 * 60 * 24;
629
630                if (time == Double.NEGATIVE_INFINITY
631                                || time == Double.POSITIVE_INFINITY) {
632                        DebugLib.stopSystemAndReportInconsistency("time is not allowed to be minus or plus infinity");
633                }
634
635                while (time < 0) {
636                        time += secondsInOneDay;
637                }
638
639                if (time < secondsInOneDay) {
640                        return time;
641                } else {
642                        return ((time / secondsInOneDay) - (Math.floor(time
643                                        / secondsInOneDay)))
644                                        * secondsInOneDay;
645                }
646        }

648        /**
649         * 24 hour check is performed (no projection required as pre-requisite).
650         * 
651         * @param startIntervalTime
652         * @param endIntervalTime
653         * @return
654         */
655        public static double getIntervalDuration(double startIntervalTime,
656                        double endIntervalTime) {
657                double secondsInOneDay = 60 * 60 * 24;
658                startIntervalTime = projectTimeWithin24Hours(startIntervalTime);
659                endIntervalTime = projectTimeWithin24Hours(endIntervalTime);
660
661                if (startIntervalTime==endIntervalTime){
662                        return 0;
663                }
664                
665                if (startIntervalTime < endIntervalTime) {
666                        return endIntervalTime - startIntervalTime;
667                } else {
668                        return endIntervalTime + (secondsInOneDay - startIntervalTime);
669                }
670        }

672        /**
673         * Interval start and end are inclusive.
674         * 
675         * @param startIntervalTime
676         * @param endIntervalTime
677         * @param timeToCheck
678         * @return
679         */
680        public static boolean isIn24HourInterval(double startIntervalTime,
681                        double endIntervalTime, double timeToCheck) {
682                errorIfNot24HourProjectedTime(startIntervalTime);
683                errorIfNot24HourProjectedTime(endIntervalTime);
684                errorIfNot24HourProjectedTime(timeToCheck);
685
686                if (startIntervalTime < endIntervalTime
687                                && timeToCheck >= startIntervalTime
688                                && timeToCheck <= endIntervalTime) {
689                        return true;
690                }
691
692                if (startIntervalTime > endIntervalTime
693                                && (timeToCheck >= startIntervalTime || timeToCheck <= endIntervalTime)) {
694                        return true;
695                }
696
697                return false;
698        }

702
703                if (time >= secondsInOneDay) {
704                        throw new Error("time not projected within 24 hours!");
705                }

713                double xDiff = coordA.getX() - coordB.getX();

846        public static Matrix<String> readStringMatrix(String fileName, String delim,
847                        StringMatrixFilter filter) {
848                Matrix<String> matrix = new Matrix<String>();
849
850                try {
851
852                        // FileReader fr = new FileReader(fileName);
853
854                        FileInputStream fis = new FileInputStream(fileName);
855                        InputStreamReader isr = new InputStreamReader(fis, "ISO-8859-1");
856
857                        BufferedReader br=null;
858                        if (fileName.toLowerCase(Locale.ROOT).endsWith(".gz")) {
859                                br = IOUtils.getBufferedReader(fileName);
860                        } else {
861                                br = new BufferedReader(isr);
862                        }
863                        
864                        String line;
865                        StringTokenizer tokenizer;
866                        line = br.readLine();
867                        while (line != null) {
868                                ArrayList<String> row = new ArrayList<String>();
869
870                                if (delim == null) {
871                                        tokenizer = new StringTokenizer(line);
872                                        
873                                        while (tokenizer.hasMoreTokens()) {
874                                                row.add(tokenizer.nextToken());
875                                        }
876                                } else {
877                                        String[] split = line.split(delim,-1);
878                                        
879                                        for (String s:split){
880                                                row.add(s);
881                                        }
882                                }
883
884                                if (filter != null && filter.removeLine(line)) {
885
886                                } else {
887                                        matrix.addRow(row);
888                                }
889
890                                line = br.readLine();
891                        }
892                } catch (Exception e) {
893                        e.printStackTrace();
894
895                        throw new Error("Error reading the file: " + fileName);
896                }
897
898                return matrix;
899        }
