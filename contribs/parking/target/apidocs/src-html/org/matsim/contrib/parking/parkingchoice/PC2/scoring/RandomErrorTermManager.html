019package org.matsim.contrib.parking.parkingchoice.PC2.scoring;
020
021import java.util.Collection;
022import java.util.HashMap;
023import java.util.LinkedList;
024import java.util.Random;
025
026import org.matsim.api.core.v01.Id;
027import org.matsim.api.core.v01.population.Person;
028import org.matsim.contrib.parking.parkingchoice.PC2.infrastructure.PC2Parking;
029import org.matsim.contrib.parking.parkingchoice.lib.DebugLib;
030import org.matsim.contrib.parking.parkingchoice.lib.obj.DoubleValueHashMap;
031import org.matsim.api.core.v01.population.Person;
032import org.matsim.core.gbl.MatsimRandom;
033/**
034 * This class is based on work by Andreas Horni related to location choice, located at:
035 * org.matsim.contrib.locationchoice.bestresponse.scoring.DestinationScoring
036 * 
037 * @author wrashid, ahorni
038 *
039 */
040public class RandomErrorTermManager {
041        
042        private Random rnd = new Random();
043        HashMap<Id,Double> parkingKValue=new HashMap<Id,Double>();
044        HashMap<Id,Double> personKValue=new HashMap<Id,Double>();
045        String epsilonDistribution;
046        
047        public RandomErrorTermManager(String epsilonDistribution,
048                        LinkedList<Id> parkingIds, Collection<? extends Person> persons, int seed) {
049                                this.epsilonDistribution = epsilonDistribution;
050                
051                Random random = new Random();
052                random.setSeed(seed);
053                for (Id parkingId:parkingIds){
054                        parkingKValue.put(parkingId, random.nextDouble());
055                }
056                
057                for (Person person: persons){
058                        personKValue.put(person.getId(), random.nextDouble());
059                }
060        }
061
062        public double getEpsilonAlternative(Id parkingId, Id personId, int actIndex) {
063                /*
064                 * k values are uniform in [0..1[, see class ReadOrCreateKVals.
065                 */             
066                double kparking = parkingKValue.get(parkingId);
067                double kperson = personKValue.get(personId); 
068                
069                /* generate another stable random number for the activity
070                 */
071                rnd.setSeed(actIndex);
072                double kactIndex = rnd.nextDouble();
073                
074                /*
075                 * generates a uniform rnd seed in [0,1[ 
076                 */
077                long seed = (long) (((kperson + kparking + kactIndex) % 1.0) * Long.MAX_VALUE);
078                rnd.setSeed(seed);
079                
080                /*
081                 * generate the epsilons according to standard Gumbel or standard Gaussian distribution
082                 */
083                if (epsilonDistribution.equalsIgnoreCase("gumbel")) {
084                        // take a few draws to come to the "chaotic region"
085                        for (int i = 0; i < 5; i++) {
086                                rnd.nextDouble();
087                        }
088                        double uniform = rnd.nextDouble();
089                        // interval MUST be ]0,1[
090                        while (uniform == 0.0 || uniform == 1.0) {
091                                uniform = rnd.nextDouble();
092                        }
093                        double r = 0.0 - 1.0 * Math.log(-Math.log(1.0 * uniform));
094                        //scale to sigma^2 = 1.0: sigma_gumbel = PI / sqrt(6.0)
095                        return (r * Math.sqrt(6.0) / Math.PI);  
096                }
097                else if (epsilonDistribution.equalsIgnoreCase("gaussian")) {
098                        // take a few draws to come to the "chaotic region"
099                        for (int i = 0; i < 5; i++) {
100                                rnd.nextGaussian();
101                        }
102                        return rnd.nextGaussian();      
103                } else {
104                        DebugLib.stopSystemAndReportInconsistency("unknown epsilonDistribution: " + epsilonDistribution);
105                        return 0;
106                }
107        }
108
109}
