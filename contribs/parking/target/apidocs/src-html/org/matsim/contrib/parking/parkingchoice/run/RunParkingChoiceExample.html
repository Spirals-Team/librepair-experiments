024

037import org.matsim.contrib.parking.parkingchoice.example.ParkingCostCalculatorExample;

039import org.matsim.core.config.ConfigUtils;

041import org.matsim.core.controler.OutputDirectoryHierarchy;

044/**
045 * @author jbischoff
046 * Sample main file for setting up an Matsim run with parking. Check the output events for Parking events and the scores.
047 * 
048 * 
049 */
050public class RunParkingChoiceExample {
051
052        public static void main(String[] args) {
053                Config config = ConfigUtils.loadConfig("parkingchoice/config.xml");
054                run(config);
055        }
056
057        public static void run(Config config) {
058                Scenario scenario = ScenarioUtils.loadScenario(config);
059
060                Controler controler = new Controler(scenario);
061                controler.getConfig().controler().setOverwriteFileSetting( OutputDirectoryHierarchy.OverwriteFileSetting.overwriteExistingFiles );
062
063                // ---
064
065                // we need some settings to walk from parking to destination:
066                ParkingScore parkingScoreManager = new ParkingScoreManager(new WalkTravelTime(controler.getConfig().plansCalcRoute()), scenario);
067                parkingScoreManager.setParkingScoreScalingFactor(1);
068                parkingScoreManager.setParkingBetas(new ParkingBetaExample());
069
070                // ---
071
072                ParkingInfrastructure parkingInfrastructureManager = new ParkingInfrastructureManager(parkingScoreManager, controler.getEvents());
073                {
074                        LinkedList<PublicParking> publicParkings = new LinkedList<PublicParking>();
075                        //parking 1: we place this near the workplace
076                        publicParkings.add(new PublicParking(Id.create("workPark", PC2Parking.class), 98, new Coord((double) 10000, (double) 0),
077                                        new ParkingCostCalculatorExample(1), "park"));
078                        //parking 2: we place this at home
079                        final double x = -25000;
080                        publicParkings.add(new PublicParking(Id.create("homePark", PC2Parking.class), 98, new Coord(x, (double) 0),
081                                        new ParkingCostCalculatorExample(0), "park"));
082                        parkingInfrastructureManager.setPublicParkings(publicParkings);
083                }
084
085
086                //setting up the Parking Module
087                GeneralParkingModule generalParkingModule = new GeneralParkingModule(controler);
088                generalParkingModule.setParkingScoreManager(parkingScoreManager);
089                generalParkingModule.setParkingInfrastructurManager(parkingInfrastructureManager);
090
091                controler.run();
092        }
093
094}
