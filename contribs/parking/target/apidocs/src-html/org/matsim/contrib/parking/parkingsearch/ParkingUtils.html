047        public static Coord getRandomPointAlongLink(Random rnd, Link link){
048                Coord fromNodeCoord = link.getFromNode().getCoord();
049                Coord toNodeCoord = link.getToNode().getCoord();
050                double r = rnd.nextDouble();
051                
052                double x = (fromNodeCoord.getX()*r)+(toNodeCoord.getX()*(1-r));
053                double y = (fromNodeCoord.getY()*r)+(toNodeCoord.getY()*(1-r));
054                
055                return new Coord(x,y);
056        }
057
058        public static List<Coord> getEvenlyDistributedCoordsAlongLink(Link link, int numberOfCoords){
059                Coord fromNodeCoord = link.getFromNode().getCoord();
060                Coord toNodeCoord = link.getToNode().getCoord();
061                double fX = fromNodeCoord.getX();
062                double fY = fromNodeCoord.getY();
063                double tX = toNodeCoord.getX();
064                double tY = toNodeCoord.getY();
065                double nrSlots = (double) numberOfCoords;
066                
067                if(fX == tX){
068                        double x;
069                        double yDistance = tY-fY;
070                        
071                        if(tY > fY) x = tX+10;
072                        else x = tX -10;
073                        
074                        List<Coord> points = new ArrayList<Coord>();
075                        if(numberOfCoords == 1){
076                                points.add(new Coord( x , (fY + 0.5*yDistance) ));
077                                return points;
078                        }
079                        
080                        if(Math.abs(yDistance) > (nrSlots+10) ){
081                                if(yDistance < -5 ) yDistance += 10;
082                                if(yDistance > 0) yDistance -= 10;
083                        }
084                        
085                        for(int i = 1; i <= numberOfCoords ; i ++){
086                                double y = (fY + (i*(1/nrSlots)*yDistance));
087                                points.add(new Coord(x,y));
088                        }
089                                return points;
090                }
091                else{
092                        // f(x) = mx + b
093                        
094                        // m = y2-y1/x2-x1
095                        double m = (tY-fY)/(tX-fX);
096                        
097                        //b = y-mx
098                        double b = fY - m*(fX);
099                        
100                        //displace
101                        if(m>0){
102                                if(tY>fY)       b -= 10;
103                                if(tY<fY)       b += 10;
104                        }
105                        if(m<0){
106                                if(tY>fY)       b += 10;
107                                if(tY<fY)       b -= 10;
108                        }
109                        if(m == 0){
110                                if(tX > fX) b -= 10;
111                                if(tX < fX) b += 10;
112                        }
113                        
114                        double xDistance = tX-fX;
115                        
116                        //calc Coords
117                        List<Coord> points = new ArrayList<Coord>();
118                        if(numberOfCoords == 1){
119                                double x = (fX + 0.5*xDistance);
120                                points.add(new Coord( x , (m*x+b) ));
121                                return points;
122                        }
123                        
124                        //distance to nodes
125                        if(Math.abs(xDistance) > (nrSlots+10) ){
126                                if(xDistance < -5 ) xDistance += 10;
127                                if(xDistance > 5) xDistance -= 10;
128                        }
129                        
130                        for(int i = 1; i <= numberOfCoords ; i ++){
131                                double x = (fX + (i*(1/nrSlots)*xDistance));
132                                double y = m*x+b;
133                                points.add(new Coord(x,y));
134                        }
135                                return points;
136                }
137        }
