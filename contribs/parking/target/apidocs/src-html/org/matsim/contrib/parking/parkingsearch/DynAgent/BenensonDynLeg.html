021public class BenensonDynLeg extends ParkingDynLeg{
022        
023        private static final Logger logger = Logger.getLogger(BenensonDynLeg.class);
024        private static final boolean logForDebug = false;
025        
026        private double totalObservedParkingSpaces = 0.0;
027        private double observedFreeParkingSpaces = 0.0;
028        private double firstDestLinkEnterTime = 0;
029        private ParkingMode legStage = ParkingMode.DRIVING;
030        
031        public BenensonDynLeg(String mode, NetworkRoute route, ParkingSearchLogic logic,
032                        ParkingSearchManager parkingManager, Id<Vehicle> vehicleId, MobsimTimer timer, EventsManager events) {
033                super(mode, route, logic, parkingManager, vehicleId, timer, events);
034                if (!(logic instanceof BenensonParkingSearchLogic)){
035                        throw new RuntimeException();
036                }
037        }
038
039        @Override
040        public void movedOverNode(Id<Link> newLinkId) {
041                currentLinkIdx++;
042                currentLinkId = newLinkId;
043                if (this.legStage == ParkingMode.DRIVING) {
044                        
045                        if(((BenensonParkingSearchLogic)this.logic).goIntoObserving(currentLinkId, this.route.getEndLinkId())){
046                                this.legStage = ParkingMode.OBSERVING;
047                                this.events.processEvent(new StartParkingSearchEvent(timer.getTimeOfDay(), vehicleId, currentLinkId));
048                                if(logForDebug)logger.error("vehicle " + this.vehicleId + " goes into observing on link " + this.currentLinkId);
049                        }
050                }
051                if(this.legStage == ParkingMode.OBSERVING ){
052                        memorizeParkingSituationAndIsSomethingFree();
053                        
054                        if(((BenensonParkingSearchLogic)this.logic).goIntoParking(currentLinkId, this.route.getEndLinkId())){
055                                this.legStage = ParkingMode.SEARCH_WHILE_APPROACH;
056                                if(logForDebug)logger.error("vehicle " + this.vehicleId + " goes into parking on link " + this.currentLinkId);
057                        }
058                }
059                if(this.legStage == ParkingMode.SEARCH_WHILE_APPROACH){
060                        if(currentLinkId.equals(route.getEndLinkId())){
061                                this.legStage = ParkingMode.SEARCH_FOR_NEXT;
062                                this.firstDestLinkEnterTime = timer.getTimeOfDay();
063                        }
064                        else{
065                                if(memorizeParkingSituationAndIsSomethingFree()){
066                                        double pUnoccupied = 0;
067                                        if(this.totalObservedParkingSpaces > 0){
068                                                pUnoccupied = this.observedFreeParkingSpaces / this.totalObservedParkingSpaces;
069                                        }
070                                        if ( ((BenensonParkingSearchLogic)this.logic).wantToParkHere(pUnoccupied, currentLinkId, route.getEndLinkId())){
071                                                if(logForDebug)logger.error("vehicle " + this.vehicleId + " w√ºrde gerne auf Link " + currentLinkId + " parken.\n "
072                                                                + "\t pUnoccupied = " + pUnoccupied + "\n\t totalObservedParkingSpaces = " + totalObservedParkingSpaces + "\n\t observedFreeSpaces = " + this.observedFreeParkingSpaces );
073                                                hasFoundParking = parkingManager.reserveSpaceIfVehicleCanParkHere(vehicleId, currentLinkId);    
074                                        }
075                                }
076                                else{
077                                        if(logForDebug)logger.error("nothing free for vehicle " + vehicleId + " on link " + currentLinkId);
078                                }
079                        }
080                }
081                if (this.legStage == ParkingMode.SEARCH_FOR_NEXT){
082                        if(logForDebug)logger.error("vehicle " + this.vehicleId + " in PHASE3 auf link " + this.currentLinkId);
083                        //if( ((BenensonParkingSearchLogic)this.logic).isDriverInAcceptableDistance(currentLinkId, route.getEndLinkId(), this.firstDestLinkEnterTimer, timer.getTimeOfDay()) ){
084                        
085                                hasFoundParking = parkingManager.reserveSpaceIfVehicleCanParkHere(vehicleId, currentLinkId);
086                                
087                                if(logForDebug)logger.error("vehicle " + this.vehicleId + " probiert in Phase 3 auf Link " + this.currentLinkId + ", " +
088                                                (int)(timer.getTimeOfDay() - this.firstDestLinkEnterTime)/60 + ":" + (int)(timer.getTimeOfDay() - this.firstDestLinkEnterTime)%60
089                                                + " min nach Erreichen des Ziels zu parken. Resultat: " + hasFoundParking);
090                        //}
091                }
092        }       
093        /**
094         * returns true if there is at least one empty slot on the current link
095         */
096        private boolean memorizeParkingSituationAndIsSomethingFree() {
097                this.totalObservedParkingSpaces += ((FacilityBasedParkingManager) this.parkingManager).getNrOfAllParkingSpacesOnLink(currentLinkId);
098                double freespaces = ((FacilityBasedParkingManager) this.parkingManager).getNrOfFreeParkingSpacesOnLink(currentLinkId);
099                this.observedFreeParkingSpaces += freespaces;
100                return !(freespaces == 0);
101        }
102
103
104        @Override
105        public Id<Link> getNextLinkId() {
106                if (this.legStage == ParkingMode.DRIVING) {
107                        List<Id<Link>> linkIds = route.getLinkIds();
108
109                        if (currentLinkIdx == linkIds.size() - 1) {
110                                return route.getEndLinkId();
111                        }
112                        return linkIds.get(currentLinkIdx + 1);
113                }
114                else {
115                        if (hasFoundParking) {
116                                if(logForDebug)logger.error("vehicle " + this.vehicleId + " has found a parking on link " + this.currentLinkId + " after passing " + Math.abs((this.route.getLinkIds().size() - this.currentLinkIdx - 3)) + " links");
117                                return null;
118                        }
119                        else {
120                                if (this.currentAndNextParkLink != null) {
121                                        if (currentAndNextParkLink.getFirst().equals(currentLinkId)) {
122                                                // we already calculated this
123                                                return currentAndNextParkLink.getSecond();
124                                        }
125                                }
126
127                                Id<Link> nextLinkId;
128                                if(this.legStage == ParkingMode.SEARCH_FOR_NEXT){
129                                        nextLinkId = ((BenensonParkingSearchLogic) this.logic).getNextLinkRandomInAcceptableDistance(currentLinkId, this.route.getEndLinkId(), vehicleId, firstDestLinkEnterTime, this.timer.getTimeOfDay());
130                                }
131                                else{
132                                        nextLinkId = ((BenensonParkingSearchLogic) (this.logic)).getNextLinkBenensonRouting(currentLinkId, route.getEndLinkId(), vehicleId);
133                                }
134//                              if(nextLinkId.equals(route.getEndLinkId()) && this.legStage == ParkingMode.SEARCH_WHILE_APPROACH){
135//                                              this.legStage = ParkingMode.SEARCH_FOR_NEXT;
136//                                              this.firstDestLinkEnterTime = timer.getTimeOfDay();
137//                              }
138                                currentAndNextParkLink = new Tuple<Id<Link>, Id<Link>>(currentLinkId, nextLinkId);
139                                return nextLinkId;
140                        }
141                }
142        }
143
144}
145/**
146 * 
147 * @author schlenther
148 *
149 *Benenson et al defined 3 phases of parking search
150 *OBSERVING:                            observation of parking situation while driving towards destination
151 *SEARCH_WHILE_APPROACH:        estimating the amount of free parking lots on the way to destination
152 *                                                      and if applicable parking before arriving
153 *SEARCH_FOR_NEXT:                      taking the next free parking space if it isn't too far away from destination
154 */
155enum ParkingMode{
156        DRIVING, OBSERVING, SEARCH_WHILE_APPROACH, SEARCH_FOR_NEXT;
157}
