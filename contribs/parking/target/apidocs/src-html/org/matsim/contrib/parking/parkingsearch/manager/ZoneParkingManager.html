006import java.util.HashMap;
007import java.util.HashSet;
008import java.util.List;

011import org.matsim.api.core.v01.Id;
012import org.matsim.api.core.v01.Scenario;

024/**
025 * @author tschlenther
026 */
027public class ZoneParkingManager extends FacilityBasedParkingManager {
028
029        private HashMap<String,HashSet<Id<Link>>> linksOfZone;
030        private HashMap<String,Double> totalCapOfZone;
031        private HashMap<String,Double> occupationOfZone;
032        
033        /**
034         * @param scenario
035         */
036        @Inject
037        public ZoneParkingManager(Scenario scenario, String[] pathToZoneTxtFiles) {
038                super(scenario);
039                
040                this.linksOfZone = new HashMap<String,HashSet<Id<Link>>>();
041                this.totalCapOfZone = new HashMap<String,Double>();
042                this.occupationOfZone = new HashMap<String,Double>();
043                
044                for(String zone: pathToZoneTxtFiles){
045                        readZone(zone);
046                }
047                
048                for(String zone: this.linksOfZone.keySet()){
049                        calculateTotalZoneParkCapacity(zone);
050                        this.occupationOfZone.put(zone,0.0);
051                }
052        }
053
054        
055        /**
056         * reads in a tabular file that declares which link id's are in the monitored zone
057         * the part between the last '/' and ".txt" of the given path is considered to be the zone name
058         * @param pathToZoneFile
059         */
060        void readZone(String pathToZoneFile){
061                String zone = pathToZoneFile.substring(pathToZoneFile.lastIndexOf("/")+1, pathToZoneFile.lastIndexOf("."));
062                
063                HashSet<Id<Link>> links = new HashSet<Id<Link>>();
064                
065                TabularFileParserConfig config = new TabularFileParserConfig();
066        config.setDelimiterTags(new String[] {"\t"});
067        config.setFileName(pathToZoneFile);
068        config.setCommentTags(new String[] { "#" });
069        new TabularFileParser().parse(config, new TabularFileHandler() {
070                        @Override
071                        public void startRow(String[] row) {
072                                Id<Link> linkId = Id.createLinkId(row[0]);
073                                links.add(linkId);
074                        }
075                
076        });
077                
078        this.linksOfZone.put(zone, links);       
079        }
080        
081        private void calculateTotalZoneParkCapacity(String zoneName){
082                double cap = 0.0;
083                for(Id<Link> link : this.linksOfZone.get(zoneName)){
084                        cap += getNrOfAllParkingSpacesOnLink(link);
085                } 
086                this.totalCapOfZone.put(zoneName, cap);
087        }
088
089        
090        @Override
091        public boolean parkVehicleHere(Id<Vehicle> vehicleId, Id<Link> linkId, double time) {
092                if (parkVehicleAtLink(vehicleId, linkId, time)) {
093                        for(String zone : this.linksOfZone.keySet()){
094                                if(linksOfZone.get(zone).contains(linkId) && this.facilitiesPerLink.containsKey(linkId)){
095                                        double newOcc = this.occupationOfZone.get(zone) + 1;
096                                        if(this.totalCapOfZone.get(zone)<newOcc){
097                                                String s = "FacilityID: " + this.parkingLocations.get(vehicleId);
098                                                String t = "Occupied: " + this.occupation.get(this.parkingLocations.get(vehicleId));
099                                                String u = "Capacity: " + this.parkingFacilities.get(this.parkingLocations.get(vehicleId)).getActivityOptions().get(ParkingUtils.PARKACTIVITYTYPE).getCapacity();
100                                                String v = "TotalCapacityOnLink: " + getNrOfAllParkingSpacesOnLink(linkId);
101                                                throw new RuntimeException("occupancy of zone " + zone + " is higher than 100%. Capacity= " + this.totalCapOfZone.get(zone) + "  occupancy=" + newOcc + "time = " + time
102                                                                + "\n" + s + "\n" + t + "\n" + u + "\n" + v);
103                                        }
104                                        this.occupationOfZone.put(zone,newOcc);
105                                        return true;                                                            // assumes: link is only part of exactly 1 zone
106                                }
107                        }
108                        return true;
109                } else
110                        return false;
111        }
112        
113        @Override
114        public boolean unParkVehicleHere(Id<Vehicle> vehicleId, Id<Link> linkId, double time) {
115                if (!this.parkingLocations.containsKey(vehicleId)) {
116                        return true;
117                        // we assume the person parks somewhere else
118                } else {
119                        Id<ActivityFacility> fac = this.parkingLocations.remove(vehicleId);
120                        this.occupation.get(fac).decrement();
121        
122                        Id<Link> parkingLink = this.parkingFacilities.get(fac).getLinkId();
123                        for(String zone : this.linksOfZone.keySet()){
124                                if(linksOfZone.get(zone).contains(parkingLink)){
125                                        double newOcc = this.occupationOfZone.get(zone) - 1;
126                                        if(newOcc < 0 ){
127                                                //in iteration 0 agents can "leave parking spaces" (get into traffic), but the manager didn't record them to be parked  
128                                                newOcc = 0;
129                                        }
130                                        this.occupationOfZone.put(zone,newOcc);
131                                }
132                        }
133                                return true;
134                }
135        }
136        
137        
138        public double getOccupancyRatioOfZone(String zone){
139                if(!(this.linksOfZone.keySet().contains(zone))) throw new RuntimeException("zone " + zone + " was not defined. thus, could'nt calculate occupancy ratio.");
140
141                return (this.occupationOfZone.get(zone) / this.totalCapOfZone.get(zone));
142        }
143        
144        public Set<String> getZones(){
145                return this.linksOfZone.keySet();
146        }
147        
148        public double getTotalCapacityOfZone(String zone){
149                return this.totalCapOfZone.get(zone);
150        }
151        
152}
