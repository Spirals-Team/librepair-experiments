021/**
022 * @author Work
023 *
024 */
025public class DistanceMemoryDynLeg extends ParkingDynLeg {
026
027        /**
028         * @param mode
029         * @param route
030         * @param logic
031         * @param parkingManager
032         * @param vehicleId
033         * @param timer
034         * @param events
035         */
036        public DistanceMemoryDynLeg(String mode, NetworkRoute route, ParkingSearchLogic logic,
037                        ParkingSearchManager parkingManager, Id<Vehicle> vehicleId, MobsimTimer timer, EventsManager events) {
038                super(mode, route, logic, parkingManager, vehicleId, timer, events);
039                // TODO Auto-generated constructor stub
040        }
041        
042        @Override
043        public void movedOverNode(Id<Link> newLinkId) {
044                currentLinkIdx++;
045                currentLinkId = newLinkId;
046                if (!parkingMode) {
047                        if (currentLinkId.equals(this.getDestinationLinkId())) {
048                                this.parkingMode = true;
049                                this.events.processEvent(new StartParkingSearchEvent(timer.getTimeOfDay(), vehicleId, currentLinkId));
050                                hasFoundParking = parkingManager.reserveSpaceIfVehicleCanParkHere(vehicleId, currentLinkId);
051                                ((DistanceMemoryParkingSearchLogic)this.logic).addToKnownLinks(currentLinkId);
052                        }
053                } else {
054                        hasFoundParking = parkingManager.reserveSpaceIfVehicleCanParkHere(vehicleId, currentLinkId);
055
056                }
057        }
058
059        @Override
060        public Id<Link> getNextLinkId() {
061                if (!parkingMode) {
062                        List<Id<Link>> linkIds = route.getLinkIds();
063
064                        if (currentLinkIdx == linkIds.size() - 1) {
065                                return route.getEndLinkId();
066                        }
067
068                        return linkIds.get(currentLinkIdx + 1);
069
070                } else {
071                        if (hasFoundParking) {
072                                // easy, we can just park where at our destination link
073                                return null;
074                        } else {
075                                if (this.currentAndNextParkLink != null) {
076                                        if (currentAndNextParkLink.getFirst().equals(currentLinkId)) {
077                                                // we already calculated this
078                                                return currentAndNextParkLink.getSecond();
079                                        }
080                                }
081                                // need to find the next link
082                                Id<Link> nextLinkId = ((DistanceMemoryParkingSearchLogic)this.logic).getNextLink(currentLinkId,route.getEndLinkId(), vehicleId);
083                                currentAndNextParkLink = new Tuple<Id<Link>, Id<Link>>(currentLinkId, nextLinkId);
084                                return nextLinkId;
085                        }
086                }
087        }
088        
089}
