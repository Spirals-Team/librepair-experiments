019package org.matsim.contrib.parking.parkingchoice.PC2.simulation;
020
021import java.util.Map;
022
023import org.matsim.api.core.v01.Coord;
024import org.matsim.api.core.v01.Id;
025import org.matsim.api.core.v01.events.Event;
026import org.matsim.api.core.v01.population.Person;
027import org.matsim.contrib.parking.parkingchoice.PC2.infrastructure.PC2Parking;
028import org.matsim.contrib.parking.parkingchoice.lib.DebugLib;
029
030public class ParkingArrivalEvent extends Event {
031
032        private Id parkingId;
033        private Id personId;
034        public final static String ATTRIBUTE_PARKING_ID = "parkingId";
035        public final static String ATTRIBUTE_PERSON_ID = "personId";
036        public final static String EVENT_TYPE = "parkingArrivalEvent";
037        public final static String DEST_COORD_X = "destinationCoordinateX";
038        public final static String DEST_COORD_Y = "destinationCoordinateY";
039        public final static String PARKING_SCORE = "parkingScore";
040        private Coord destCoordinate;
041        private double score;
042        
043        // yyyyyy the parking arrival event returns person id, not vehicle id.  Do we change to vehicle id, or add the vehicle
044        // id to the event?  It is used for the walk distance; for that we need the person id. kai, jul'15
045
046        public ParkingArrivalEvent(double time, Id parkingId, Id personId, Coord destCoordinate, double score) {
047                super(time);
048                this.destCoordinate = destCoordinate;
049                this.score = score;
050                
051                if (time>110000){
052                        DebugLib.emptyFunctionForSettingBreakPoint();
053                }
054                
055                this.parkingId = parkingId;
056                this.personId = personId;
057        }
058
059        @Override
060        public String getEventType() {
061                return EVENT_TYPE;
062        }
063        
064        @Override
065        public Map<String, String> getAttributes() {
066                final Map<String, String> attributes = super.getAttributes();
067                attributes.put(ATTRIBUTE_PARKING_ID, parkingId.toString());
068                attributes.put(ATTRIBUTE_PERSON_ID, personId!=null?personId.toString():null);
069                if (destCoordinate==null){
070                        attributes.put(DEST_COORD_X, null);
071                        attributes.put(DEST_COORD_Y, null);
072                } else {
073                        attributes.put(DEST_COORD_X, Double.toString(destCoordinate.getX()));
074                        attributes.put(DEST_COORD_Y, Double.toString(destCoordinate.getY()));
075                }
076                
077                attributes.put(PARKING_SCORE, Double.toString(score));
078                return attributes;
079        }
080        
081        public static Coord getDestCoord(Map<String, String> attributes){
082                String destCoordXString = attributes.get(ParkingArrivalEvent.DEST_COORD_X);
083                String destCoordYString = attributes.get(ParkingArrivalEvent.DEST_COORD_Y);
084                
085                if (destCoordXString==null || destCoordYString==null){
086                        return null;
087                } else {
088                        return new Coord(Double.parseDouble(destCoordXString), Double.parseDouble(destCoordYString));
089                }
090        }
091        
092        public static double getScore(Map<String, String> attributes){
093                String scoreString = attributes.get(ParkingArrivalEvent.PARKING_SCORE);
094                return Double.parseDouble(scoreString);
095        }
096        
097        public static Id<Person> getPersonId(Map<String, String> attributes){
098                String personIdString = attributes.get(ParkingArrivalEvent.ATTRIBUTE_PERSON_ID);
099                if (personIdString==null){
100                        return null;
101                } else {
102                        return Id.create(personIdString, Person.class);
103                }
104        }
105        
106        public static Id<PC2Parking> getParkingId(Map<String, String> attributes){
107                String parkingIdString = attributes.get(ParkingArrivalEvent.ATTRIBUTE_PARKING_ID);
108                if (parkingIdString==null){
109                        return null;
110                } else {
111                        return Id.create(parkingIdString, PC2Parking.class);
112                }
113        }
114
115}
