001package org.matsim.contrib.parking.parkingchoice.lib.obj;
002
003public class SortableMapObject<Key> implements Comparable<SortableMapObject> {
004
005        private final Key key;
006        private final double weight;
007
008        /**
009         * attention: it gives you the smaller number. If you are using negative
010         * weights and need the best score, use multiply by -1.0, before insertion
011         * 
012         * @param key
013         * @param weight
014         */
015        public SortableMapObject(Key key, double weight) {
016                this.key = key;
017                this.weight = weight;
018        }
019
020        @Override
021        public int compareTo(SortableMapObject other) {
022                if (getWeight() > other.getWeight())
023                        return 1;
024
025                if (getWeight() < other.getWeight())
026                        return -1;
027
028                return 0;
029        }
030
031        public double getWeight() {
032                return weight;
033        }
034
035        public Key getKey() {
036                return key;
037        }
038
039}
