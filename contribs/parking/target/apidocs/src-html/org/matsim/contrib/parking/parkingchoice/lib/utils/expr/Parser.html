029/**
030 * Parses strings representing mathematical formulas with variables. The
031 * following operators, in descending order of precedence, are defined:
032 * 
033 * <UL>
034 * <LI>^ (raise to a power)
035 * <LI>/
036 * <LI>Unary minus (-x)
037 * <LI>+ -
038 * <LI>&lt; &lt;= = &lt;&gt; &gt;= &gt;
039 * <LI>and
040 * <LI>or
041 * </UL>
042 * 
043 * ^ associates right-to-left; other operators associate left-to-right.
044 * 
045 * <P>
046 * These unary functions are defined: abs, acos, asin, atan, ceil, cos, exp,
047 * floor, log, round, sin, sqrt, tan. Each requires one argument enclosed in
048 * parentheses.
049 * 
050 * <P>
051 * There are also binary functions: atan2, min, max; and a ternary conditional
052 * function: if(test, then, else).
053 * 
054 * <P>
055 * Whitespace outside identifiers is ignored.
056 * 
057 * <P>
058 * Examples:
059 * <UL>
060 * <LI>42
061 * <LI>2-3
062 * <LI>cos(x^2) + sin(x^2)
063 * <UL>
064 */
065public class Parser {
066
067        private final String input;
068
069
070        /**
071         * Return the expression denoted by the input string.
072         * 
073         * @param input
074         *            the unparsed expression
075         * @exception SyntaxException
076         *                if the input is unparsable
077         */
078        private Expr parse(String input) throws SyntaxException {
079                return parseString(input);
080        }
081
082        public double parse() throws SyntaxException {
083                return parse(this.input).value();
084        }
085
086        public Parser(String input) {
087                this.input = input;
088                
089                //built-in constants
090                initializeBuiltinConstants();
091        }
092
093        private void initializeBuiltinConstants() {
094                setVariable("pi", Math.PI);
095        }
096
097        /**
098         * Set of Variables and values used in expression.
099         */
100        private Hashtable<String, Variable> variables = new Hashtable<String, Variable>();
101
102        /**
103         * Set of Variable's that are allowed to appear in input expressions. If
104         * null, any variable is allowed.
105         */
106        private Hashtable allowedVariables = null;
107
108        /**
109         * Adjust the set of allowed variables: create it (if not yet existent) and
110         * add optVariable (if it's nonnull). If the allowed-variable set exists,
111         * the parser will reject input strings that use any other variables.
112         * 
113         * @param optVariable
114         *            the variable to be allowed, or null
115         */
116        public void allow(Variable optVariable) {
117                if (null == allowedVariables) {
118                        allowedVariables = new Hashtable();
119                        // allowedVariables.put(pi, pi);
120                }
121                if (null != optVariable)
122                        allowedVariables.put(optVariable, optVariable);
123        }
124
125        Scanner tokens = null;
126        private Token token = null;
127
128        public void setVariable(String name, double value) {
129                if (!variables.containsKey(name))
130                        variables.put(name, new Variable(name));
131                Variable result = (Variable) variables.get(name);
132                result.setValue(value);
133        }
134
135        /**
136         * Return the expression denoted by the input string.
137         * 
138         * @param input
139         *            the unparsed expression
140         * @exception SyntaxException
141         *                if the input is unparsable
142         */
143        public Expr parseString(String input) throws SyntaxException {
144                tokens = new Scanner(input, operatorChars);
145                return reparse();
146        }
147
148        static private final String operatorChars = "*/+-^<>=,()";
149
150        private Expr reparse() throws SyntaxException {
151                tokens.index = -1;
152                nextToken();
153                Expr expr = parseExpr(0);
154                if (token.ttype != Token.TT_EOF)
155                        throw error("Incomplete expression", SyntaxException.INCOMPLETE,
156                                        null);
157                return expr;
158        }
159
160        private void nextToken() {
161                token = tokens.nextToken();
162        }
163
164        private Expr parseExpr(int precedence) throws SyntaxException {
165                Expr expr = parseFactor();
166                loop: for (;;) {
167                        int l, r, rator;
168
169                        // The operator precedence table.
170                        // l = left precedence, r = right precedence, rator = operator.
171                        // Higher precedence values mean tighter binding of arguments.
172                        // To associate left-to-right, let r = l+1;
173                        // to associate right-to-left, let r = l.
174
175                        switch (token.ttype) {
176
177                        case '<':
178                                l = 20;
179                                r = 21;
180                                rator = Expr.LT;
181                                break;
182                        case Token.TT_LE:
183                                l = 20;
184                                r = 21;
185                                rator = Expr.LE;
186                                break;
187                        case '=':
188                                l = 20;
189                                r = 21;
190                                rator = Expr.EQ;
191                                break;
192                        case Token.TT_NE:
193                                l = 20;
194                                r = 21;
195                                rator = Expr.NE;
196                                break;
197                        case Token.TT_GE:
198                                l = 20;
199                                r = 21;
200                                rator = Expr.GE;
201                                break;
202                        case '>':
203                                l = 20;
204                                r = 21;
205                                rator = Expr.GT;
206                                break;
207
208                        case '+':
209                                l = 30;
210                                r = 31;
211                                rator = Expr.ADD;
212                                break;
213                        case '-':
214                                l = 30;
215                                r = 31;
216                                rator = Expr.SUB;
217                                break;
218
219                        case '/':
220                                l = 40;
221                                r = 41;
222                                rator = Expr.DIV;
223                                break;
224                        case '*':
225                                l = 40;
226                                r = 41;
227                                rator = Expr.MUL;
228                                break;
229
230                        case '^':
231                                l = 50;
232                                r = 50;
233                                rator = Expr.POW;
234                                break;
235
236                        default:
237                                if (token.ttype == Token.TT_WORD && token.sval.equals("and")) {
238                                        l = 5;
239                                        r = 6;
240                                        rator = Expr.AND;
241                                        break;
242                                }
243                                if (token.ttype == Token.TT_WORD && token.sval.equals("or")) {
244                                        l = 10;
245                                        r = 11;
246                                        rator = Expr.OR;
247                                        break;
248                                }
249                                break loop;
250                        }
251
252                        if (l < precedence)
253                                break loop;
254
255                        nextToken();
256                        expr = Expr.makeApp2(rator, expr, parseExpr(r));
257                }
258                return expr;
259        }
260
261        static private final String[] procs1 = { "abs", "acos", "asin", "atan",
262                        "ceil", "cos", "exp", "floor", "log", "round", "sin", "sqrt", "tan" };
263        static private final int[] rators1 = { Expr.ABS, Expr.ACOS, Expr.ASIN,
264                        Expr.ATAN, Expr.CEIL, Expr.COS, Expr.EXP, Expr.FLOOR, Expr.LOG,
265                        Expr.ROUND, Expr.SIN, Expr.SQRT, Expr.TAN };
266
267        static private final String[] procs2 = { "atan2", "max", "min" };
268        static private final int[] rators2 = { Expr.ATAN2, Expr.MAX, Expr.MIN };
269
270        private Expr parseFactor() throws SyntaxException {
271                switch (token.ttype) {
272                case Token.TT_NUMBER: {
273                        Expr lit = Expr.makeLiteral(token.nval);
274                        nextToken();
275                        return lit;
276                }
277                case Token.TT_WORD: {
278                        for (int i = 0; i < procs1.length; ++i)
279                                if (procs1[i].equals(token.sval)) {
280                                        nextToken();
281                                        expect('(');
282                                        Expr rand = parseExpr(0);
283                                        expect(')');
284                                        return Expr.makeApp1(rators1[i], rand);
285                                }
286
287                        for (int i = 0; i < procs2.length; ++i)
288                                if (procs2[i].equals(token.sval)) {
289                                        nextToken();
290                                        expect('(');
291                                        Expr rand1 = parseExpr(0);
292                                        expect(',');
293                                        Expr rand2 = parseExpr(0);
294                                        expect(')');
295                                        return Expr.makeApp2(rators2[i], rand1, rand2);
296                                }
297
298                        if (token.sval.equals("if")) {
299                                nextToken();
300                                expect('(');
301                                Expr test = parseExpr(0);
302                                expect(',');
303                                Expr consequent = parseExpr(0);
304                                expect(',');
305                                Expr alternative = parseExpr(0);
306                                expect(')');
307                                return Expr.makeIfThenElse(test, consequent, alternative);
308                        }
309
310                        // Expr var = Variable.make(token.sval);
311                        Expr var = variables.get(token.sval);
312                        if (null != allowedVariables && null == allowedVariables.get(var))
313                                throw error("Unknown variable",
314                                                SyntaxException.UNKNOWN_VARIABLE, null);
315                        nextToken();
316                        return var;
317                }
318                case '(': {
319                        nextToken();
320                        Expr enclosed = parseExpr(0);
321                        expect(')');
322                        return enclosed;
323                }
324                case '-':
325                        nextToken();
326                        return Expr.makeApp1(Expr.NEG, parseExpr(35));
327                case Token.TT_EOF:
328                        throw error("Expected a factor", SyntaxException.PREMATURE_EOF,
329                                        null);
330                default:
331                        throw error("Expected a factor", SyntaxException.BAD_FACTOR, null);
332                }
333        }
334
335        private SyntaxException error(String complaint, int reason, String expected) {
336                return new SyntaxException(complaint, this, reason, expected);
337        }
338
339        private void expect(int ttype) throws SyntaxException {
340                if (token.ttype != ttype)
341                        throw error("'" + (char) ttype + "' expected",
342                                        SyntaxException.EXPECTED, "" + (char) ttype);
343                nextToken();
344        }
345
346        // Error correction
347
348        boolean tryCorrections() {
349                return tryInsertions() || tryDeletions() || trySubstitutions();
350        }
351
352        private boolean tryInsertions() {
353                Vector v = tokens.tokens;
354                for (int i = tokens.index; 0 <= i; --i) {
355                        Token t;
356                        if (i < v.size()) {
357                                t = (Token) v.elementAt(i);
358                        } else {
359                                String s = tokens.getInput();
360                                t = new Token(Token.TT_EOF, 0, s, s.length(), s.length());
361                        }
362                        Token[] candidates = possibleInsertions(t);
363                        for (int j = 0; j < candidates.length; ++j) {
364                                v.insertElementAt(candidates[j], i);
365                                try {
366                                        reparse();
367                                        return true;
368                                } catch (SyntaxException se) {
369                                        v.removeElementAt(i);
370                                }
371                        }
372                }
373                return false;
374        }
375
376        private boolean tryDeletions() {
377                Vector v = tokens.tokens;
378                for (int i = tokens.index; 0 <= i; --i) {
379                        if (v.size() <= i)
380                                continue;
381                        Object t = v.elementAt(i);
382                        v.remove(i);
383                        try {
384                                reparse();
385                                return true;
386                        } catch (SyntaxException se) {
387                                v.insertElementAt(t, i);
388                        }
389                }
390                return false;
391        }
392
393        private boolean trySubstitutions() {
394                Vector v = tokens.tokens;
395                for (int i = tokens.index; 0 <= i; --i) {
396                        if (v.size() <= i)
397                                continue;
398                        Token t = (Token) v.elementAt(i);
399                        Token[] candidates = possibleSubstitutions(t);
400                        for (int j = 0; j < candidates.length; ++j) {
401                                v.setElementAt(candidates[j], i);
402                                try {
403                                        reparse();
404                                        return true;
405                                } catch (SyntaxException se) {
406                                }
407                        }
408                        v.setElementAt(t, i);
409                }
410                return false;
411        }
412
413        private Token[] possibleInsertions(Token t) {
414                Token[] ts = new Token[operatorChars.length() + 6 + procs1.length
415                                + procs2.length];
416                int i = 0;
417
418                Token one = new Token(Token.TT_NUMBER, 1, "1", t);
419                ts[i++] = one;
420
421                for (int j = 0; j < operatorChars.length(); ++j) {
422                        char c = operatorChars.charAt(j);
423                        ts[i++] = new Token(c, 0, Character.toString(c), t);
424                }
425
426                ts[i++] = new Token(Token.TT_WORD, 0, "x", t);
427
428                for (int k = 0; k < procs1.length; ++k)
429                        ts[i++] = new Token(Token.TT_WORD, 0, procs1[k], t);
430
431                for (int m = 0; m < procs2.length; ++m)
432                        ts[i++] = new Token(Token.TT_WORD, 0, procs2[m], t);
433
434                ts[i++] = new Token(Token.TT_LE, 0, "<=", t);
435                ts[i++] = new Token(Token.TT_NE, 0, "<>", t);
436                ts[i++] = new Token(Token.TT_GE, 0, ">=", t);
437                ts[i++] = new Token(Token.TT_WORD, 0, "if", t);
438
439                return ts;
440        }
441
442        private Token[] possibleSubstitutions(Token t) {
443                return possibleInsertions(t);
444        }
