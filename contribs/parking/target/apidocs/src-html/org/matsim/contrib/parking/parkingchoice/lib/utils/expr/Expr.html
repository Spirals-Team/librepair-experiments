022// Mathematical expressions.
023// Copyright 1996 by Darius Bacon; see the file COPYING.
024
025
026/**
027 * A mathematical expression, built out of literal numbers, variables,
028 * arithmetic and relational operators, and elementary functions.  It
029 * can be evaluated to get its value given its variables' current
030 * values.  The operator names are from java.lang.Math where possible.
031 */
032public abstract class Expr {
033
034    /** Calculate the expression's value.
035     * @return the value given the current variable values */
036    public abstract double value();
037
038    /** Binary operator: addition        */  public static final int ADD =  0;  
039    /** Binary operator: subtraction     */  public static final int SUB =  1;
040    /** Binary operator: multiplication  */  public static final int MUL =  2;
041    /** Binary operator: division        */  public static final int DIV =  3;
042    /** Binary operator: exponentiation  */  public static final int POW =  4;
043    /** Binary operator: arctangent      */  public static final int ATAN2 = 5;
044    /** Binary operator: maximum         */  public static final int MAX =  6;
045    /** Binary operator: minimum         */  public static final int MIN =  7;
046    /** Binary operator: less than       */  public static final int LT  =  8;
047    /** Binary operator: less or equal   */  public static final int LE  =  9;
048    /** Binary operator: equality        */  public static final int EQ  = 10;
049    /** Binary operator: inequality      */  public static final int NE  = 11;
050    /** Binary operator: greater or equal*/  public static final int GE  = 12;
051    /** Binary operator: greater than    */  public static final int GT  = 13;
052    /** Binary operator: logical and     */  public static final int AND = 14;
053    /** Binary operator: logical or      */  public static final int OR  = 15;
054  
055    /** Unary operator: absolute value*/   public static final int ABS   = 100;
056    /** Unary operator: arccosine */       public static final int ACOS  = 101;
057    /** Unary operator: arcsine   */       public static final int ASIN  = 102;
058    /** Unary operator: arctangent*/       public static final int ATAN  = 103;
059    /** Unary operator: ceiling   */       public static final int CEIL  = 104;
060    /** Unary operator: cosine    */       public static final int COS   = 105;
061    /** Unary operator: e to the x*/       public static final int EXP   = 106;
062    /** Unary operator: floor     */       public static final int FLOOR = 107;
063    /** Unary operator: natural log*/      public static final int LOG   = 108;
064    /** Unary operator: negation        */ public static final int NEG   = 109;
065    /** Unary operator: rounding  */       public static final int ROUND = 110;
066    /** Unary operator: sine      */       public static final int SIN   = 111;
067    /** Unary operator: square root */     public static final int SQRT  = 112;
068    /** Unary operator: tangent */         public static final int TAN   = 113;
069
070    /** Make a literal expression.
071     * @param v the constant value of the expression
072     * @return an expression whose value is always v */
073    public static Expr makeLiteral(double v) { 
074        return new LiteralExpr(v); 
075    }
076    /** Make an expression that applies a unary operator to an operand.
077     * @param rator a code for a unary operator
078     * @param rand operand
079     * @return an expression meaning rator(rand)
080     */
081    public static Expr makeApp1(int rator, Expr rand) {
082        Expr app = new UnaryExpr(rator, rand);
083        return rand instanceof LiteralExpr
084            ? new LiteralExpr(app.value()) 
085            : app;
086    }
087    /** Make an expression that applies a binary operator to two operands.
088     * @param rator a code for a binary operator
089     * @param rand0 left operand
090     * @param rand1 right operand
091     * @return an expression meaning rator(rand0, rand1)
092     */
093    public static Expr makeApp2(int rator, Expr rand0, Expr rand1) {
094        Expr app = new BinaryExpr(rator, rand0, rand1);
095        return rand0 instanceof LiteralExpr && rand1 instanceof LiteralExpr
096            ? new LiteralExpr(app.value()) 
097            : app;
098    }
099    /** Make a conditional expression.
100     * @param test `if' part
101     * @param consequent `then' part
102     * @param alternative `else' part
103     * @return an expression meaning `if test, then consequent, else
104     *         alternative' 
105     */
106    public static Expr makeIfThenElse(Expr test,
107                                      Expr consequent,
108                                      Expr alternative) {
109        Expr cond = new ConditionalExpr(test, consequent, alternative);
110        if (test instanceof LiteralExpr)
111            return test.value() != 0 ? consequent : alternative;
112        else
113            return cond;
114    }
115}
116
117// These classes are all private to this module because we could
118// plausibly want to do it in a completely different way, such as a
119// stack machine.
120
121class LiteralExpr extends Expr {
122    double v;
123    LiteralExpr(double v) { this.v = v; }
124    public double value() { return v; }
125}
126
127class UnaryExpr extends Expr {
128    int rator;
129    Expr rand;
130
131    UnaryExpr(int rator, Expr rand) { 
132        this.rator = rator;
133        this.rand = rand;
134    }
135
136    public double value() {
137        double arg = rand.value();
138        switch (rator) {
139        case ABS:   return Math.abs(arg);
140        case ACOS:  return Math.acos(arg);
141        case ASIN:  return Math.asin(arg);
142        case ATAN:  return Math.atan(arg);
143        case CEIL:  return Math.ceil(arg);
144        case COS:   return Math.cos(arg);
145        case EXP:   return Math.exp(arg);
146        case FLOOR: return Math.floor(arg);
147        case LOG:   return Math.log(arg);
148        case NEG:   return -arg;
149        case ROUND: return Math.rint(arg);
150        case SIN:   return Math.sin(arg);
151        case SQRT:  return Math.sqrt(arg);
152        case TAN:   return Math.tan(arg);
153        default: throw new RuntimeException("BUG: bad rator");
154        }
155    }
156}
157
158class BinaryExpr extends Expr {
159    int rator;
160    Expr rand0, rand1;
161
162    BinaryExpr(int rator, Expr rand0, Expr rand1) {
163        this.rator = rator;
164        this.rand0 = rand0;
165        this.rand1 = rand1;
166    }
167    public double value() {
168        double arg0 = rand0.value();
169        double arg1 = rand1.value();
170        switch (rator) {
171        case ADD:   return arg0 + arg1;
172        case SUB:   return arg0 - arg1;
173        case MUL:   return arg0 * arg1;
174        case DIV:   return arg0 / arg1; // division by 0 has IEEE 754 behavior
175        case POW:   return Math.pow(arg0, arg1);
176        case ATAN2: return Math.atan2(arg0, arg1);
177        case MAX:   return arg0 < arg1 ? arg1 : arg0;
178        case MIN:   return arg0 < arg1 ? arg0 : arg1;
179        case LT:    return arg0 <  arg1 ? 1.0 : 0.0;
180        case LE:    return arg0 <= arg1 ? 1.0 : 0.0;
181        case EQ:    return arg0 == arg1 ? 1.0 : 0.0;
182        case NE:    return arg0 != arg1 ? 1.0 : 0.0;
183        case GE:    return arg0 >= arg1 ? 1.0 : 0.0;
184        case GT:    return arg0  > arg1 ? 1.0 : 0.0;
185        case AND:   return arg0 != 0 && arg1 != 0 ? 1.0 : 0.0;
186        case OR:    return arg0 != 0 || arg1 != 0 ? 1.0 : 0.0; 
187        default: throw new RuntimeException("BUG: bad rator");
188        }
189    }
190}
191
192class ConditionalExpr extends Expr {
193    Expr test, consequent, alternative;
194
195    ConditionalExpr(Expr test, Expr consequent, Expr alternative) {
196        this.test = test;
197        this.consequent = consequent;
198        this.alternative = alternative;
199    }
200
201    public double value() {
202        return test.value() != 0 ? consequent.value() : alternative.value();
203    }
204}
