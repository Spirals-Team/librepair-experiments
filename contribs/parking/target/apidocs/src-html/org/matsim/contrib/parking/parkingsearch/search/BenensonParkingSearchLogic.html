023/**
024 * @author schlenther
025 *
026 *the matsim version of the parking strategy used in PARKAGENT. see the following paper for more information:
027 *doi: 10.1016/j.compenvurbsys.2008.09.011
028 *
029 */
030public class BenensonParkingSearchLogic implements ParkingSearchLogic {
031        private static final Logger logger = Logger.getLogger(BenensonDynLeg.class);
032        private static final boolean logForDebug = false;
033        
034        private Network network;
035        private static final double MIN_THRESHOLD_PROB_FUNCTION = 2;
036        private static final double MAX_THRESHOLD_PROB_FUNCTION = 4;
037        
038        //border lines for phase transitions: 1 -> 2 and 2->3
039        private static final double THRESHOLD_OBSERVING_METER = 1000;
040        private static final double THRESHOLD_PARKING_METER = 500;
041        
042        private static final double ACCEPTED_DISTANCE_START = 100;
043        private static final double ACCEPTED_DISTANCE_INCREASING_RATE_PER_MIN = 100;
044        private static final double ACCEPTED_DISTANCE_MAX = 600;
045        
046        private final Random random = MatsimRandom.getLocalInstance();
047        
048        private ParkingSearchConfigGroup configGroup;
049        
050        public BenensonParkingSearchLogic(Network network, ParkingSearchConfigGroup cfgGroup) {
051                this.network = network;
052                this.configGroup = cfgGroup;
053        }
054        
055        @Override
056        public void reset() {
057
058        }
059        //----------------------------------------------------phase transitions----------------------------------------------------------------------------
060        
061        public boolean goIntoObserving (Id<Link> currLinkId, Id<Link> endLinkId){
062                double distToDest = NetworkUtils.getEuclideanDistance(
063                                network.getLinks().get(currLinkId).getCoord(), network.getLinks().get(endLinkId).getCoord());
064                return distToDest < THRESHOLD_OBSERVING_METER ;
065        }
066        
067        public boolean goIntoParking (Id<Link> currLinkId, Id<Link> endLinkId){
068                double distToDest = NetworkUtils.getEuclideanDistance(
069                                network.getLinks().get(currLinkId).getCoord(), network.getLinks().get(endLinkId).getCoord());
070                return distToDest < THRESHOLD_PARKING_METER ;
071        }
072        
073        //-------------------------------------------------------routing---------------------------------------------------------------------------
074        
075        /**
076         * 
077         * @param currentLinkId
078         * @param destinationLinkId
079         * @param vehicleId
080         * @param hasTriedDestLinkBefore
081         * @return
082         */
083        public Id<Link> getNextLinkBenensonRouting(Id<Link> currentLinkId, Id<Link> destinationLinkId, Id<Vehicle> vehicleId) {
084                Link currentLink = network.getLinks().get(currentLinkId);
085                //List<Id<Link>> nextNodes = new ArrayList<>();
086                
087                //es wird nicht die Distanz zur Aktivität, sondern zum fromNode des Aktivitätenlinks berechnet
088                Node destination = network.getLinks().get(destinationLinkId).getFromNode();
089                
090                double distanceToDest = Double.MAX_VALUE;
091                Node nextNode;
092                Id<Link> nextLinkId = null;
093
094                for (Id<Link> outlinkId : currentLink.getToNode().getOutLinks().keySet()){
095                        if(outlinkId.equals(destinationLinkId)){
096                                return outlinkId;
097                        }
098                        nextNode = network.getLinks().get(outlinkId).getToNode();
099                        double dd = NetworkUtils.getEuclideanDistance(destination.getCoord(),nextNode.getCoord());
100                        if( dd < distanceToDest){
101                                nextLinkId = outlinkId;
102                                distanceToDest = dd;
103                        }
104                        else if(dd == distanceToDest){
105                                if (Math.random() > 0.5){
106                                        nextLinkId = outlinkId;
107                                }
108                        }
109                }
110                return nextLinkId;
111        }
112
113        public Id<Link> getNextLinkRandomInAcceptableDistance(Id<Link> currentLinkId, Id<Link> endLinkId, Id<Vehicle> vehicleId, double firstDestLinkEnterTime, double timeOfDay) {
114
115                Id<Link> nextLink = null;
116                Link currentLink = network.getLinks().get(currentLinkId);
117                List<Id<Link>> keys = new ArrayList<>(currentLink.getToNode().getOutLinks().keySet());
118                do{
119                        if(!(nextLink == null)) keys.remove(keys.indexOf(nextLink));
120                        
121                        if(keys.size() == 0){   //no outlink in acceptable Distance
122                                keys = new ArrayList<>(currentLink.getToNode().getOutLinks().keySet());
123                                logger.error("vehicle " + vehicleId + " finds no outlink in acceptable distance going out from link " + currentLinkId + ". it just takes a random next link");
124                                return keys.get(random.nextInt(keys.size()));
125                        }
126                        nextLink= keys.get(random.nextInt(keys.size()));        
127                }
128                while(!isDriverInAcceptableDistance(nextLink, endLinkId, firstDestLinkEnterTime, timeOfDay));
129                return nextLink;
130        }
131
132        //---------------------------------------------------park decision-----------------------------------------------------------------------
133
134        /**
135         * 
136         * estimate amount of free parking spaces on the way to destination Link and decide whether to park on currentLink
137         * 
138         * @param pUnoccupied
139         * @param currentLinkId
140         * @param endLinkId
141         * @return
142         */
143        public boolean wantToParkHere (double pUnoccupied, Id<Link> currentLinkId, Id<Link> endLinkId) {
144                
145                //if pUnoccupied = 0, no free slot has been detected, so it is realistic to accept the very next free slot   
146                double distToDest = NetworkUtils.getEuclideanDistance(
147                                network.getLinks().get(currentLinkId).getToNode().getCoord(), network.getLinks().get(endLinkId).getToNode().getCoord());
148                double expectedFreeSlots = (pUnoccupied*distToDest/configGroup.getAvgparkingslotlength());
149                double rnd = Math.random();
150                if(logForDebug)logger.error("\n current link: "+ currentLinkId + "\n expected slots: " + expectedFreeSlots + "\n probabilty to continue driving: " + getProbabilityOfContinueDriving(expectedFreeSlots) + "\n rnd: " + rnd);
151                if (rnd < getProbabilityOfContinueDriving(expectedFreeSlots)) return false;
152                else return true;
153        }
154        
155        /**
156         * linear probability function, depending on maximum and minimium threshold 
157         */
158        private double getProbabilityOfContinueDriving (double expectedFreeSlots){
159                
160                if (expectedFreeSlots < MIN_THRESHOLD_PROB_FUNCTION) return 0.0;
161                else if(expectedFreeSlots > MAX_THRESHOLD_PROB_FUNCTION) return 1.0;
162                
163                return (expectedFreeSlots-MIN_THRESHOLD_PROB_FUNCTION)/(MAX_THRESHOLD_PROB_FUNCTION-MIN_THRESHOLD_PROB_FUNCTION);
164        }
165
166        /**
167         * 
168         * @param currentLinkId
169         * @param endLinkId
170         * @param firstDestLinkEnterTimer
171         * @param timeOfDay
172         * @return
173         */
174        private boolean isDriverInAcceptableDistance(Id<Link> currentLinkId, Id<Link> endLinkId,        double firstDestLinkEnterTime, double timeOfDay) {
175
176                // if we're on the destinationLink, we always want to park
177                if(currentLinkId.equals(endLinkId)) return true;
178                
179                double distToDest = NetworkUtils.getEuclideanDistance(network.getLinks().get(currentLinkId).getCoord(), network.getLinks().get(endLinkId).getCoord());
180                double timeSpent = timeOfDay - firstDestLinkEnterTime;
181                double acceptedDistance = ACCEPTED_DISTANCE_START + ACCEPTED_DISTANCE_INCREASING_RATE_PER_MIN * (timeSpent / 60);
182                
183                if (acceptedDistance > ACCEPTED_DISTANCE_MAX) acceptedDistance = ACCEPTED_DISTANCE_MAX;
184                
185                if (distToDest <= acceptedDistance){
186                        if(logForDebug){
187                                logger.error(" distance between link " + currentLinkId + " and destLink " + endLinkId + ": " + distToDest);
188                                logger.error("accepted : " + acceptedDistance);
189                                logger.error("time spent: " + timeSpent);
190                        }
191                        return true;            
192                }
193                return false;
194        }
