083        @Override
084        public synchronized void setPublicParkings(LinkedList<PublicParking> publicParkings) {
085                publicParkingGroupQuadTrees = new HashMap<String, QuadTree<PC2Parking>>();
086                EnclosingRectangle allPublicParkingRect = new EnclosingRectangle();
087                HashMap<String, EnclosingRectangle> groupRects = new HashMap<String, EnclosingRectangle>();
088
089                for (PublicParking parking : publicParkings) {
090                        if (parking.getAvailableParkingCapacity() <= 0) {
091                                DebugLib.stopSystemAndReportInconsistency("parking capacity non-positive: " + parking.getId());
092                        }
093
094                        allPublicParkingRect.registerCoord(parking.getCoordinate());
095                        getAllParkings().put(parking.getId(), parking);
096
097                        if (!groupRects.containsKey(parking.getGroupName())) {
098                                groupRects.put(parking.getGroupName(), new EnclosingRectangle());
099                        }
100                        EnclosingRectangle groupRect = groupRects.get(parking.getGroupName());
101                        groupRect.registerCoord(parking.getCoordinate());
102                }
103                this.publicParkingsQuadTree = (new QuadTreeInitializer<PC2Parking>()).getQuadTree(allPublicParkingRect);
104
105                for (String groupNames : groupRects.keySet()) {
106                        publicParkingGroupQuadTrees.put(groupNames,
107                                        (new QuadTreeInitializer<PC2Parking>()).getQuadTree(groupRects.get(groupNames)));
108                }
109
110                for (PublicParking parking : publicParkings) {
111                        addParkingToQuadTree(publicParkingsQuadTree, parking);
112                        addParkingToQuadTree(publicParkingGroupQuadTrees.get(parking.getGroupName()), parking);
113                }
114        }

116        public static void addParkingToQuadTree(QuadTree<PC2Parking> quadTree, PC2Parking parking) {
117                quadTree.put(parking.getCoordinate().getX(), parking.getCoordinate().getY(), parking);
118        }

175        @Override
176        public synchronized PC2Parking parkAtClosestPublicParkingNonPersonalVehicle(Coord destCoordinate,
177                        String groupName) {
178                PC2Parking parking = null;
179                if (groupName == null) {
180                        parking = publicParkingsQuadTree.getClosest(destCoordinate.getX(), destCoordinate.getY());
181                } else {
182                        QuadTree<PC2Parking> quadTree = publicParkingGroupQuadTrees.get(groupName);
183                        parking = quadTree.getClosest(destCoordinate.getX(), destCoordinate.getY());
184
185                        if (parking == null) {
186                                DebugLib.stopSystemAndReportInconsistency(
187                                                "not enough parking available for parkingGroupName:" + groupName);
188                        }
189                }
190                parkVehicle(parking);
191
192                return parking;
193        }

200        @Override
201        public synchronized PC2Parking parkAtClosestPublicParkingNonPersonalVehicle(Coord destCoordinate, String groupName,
202                        Id<Person> personId, double parkingDurationInSeconds, double arrivalTime) {
203                PC2Parking parking = parkAtClosestPublicParkingNonPersonalVehicle(destCoordinate, groupName);
204
205                double walkScore = parkingScoreManager.calcWalkScore(destCoordinate, parking, personId,
206                                parkingDurationInSeconds);
207                parkingScoreManager.addScore(personId, walkScore);
208
209                eventsManager.processEvent(
210                                new ParkingArrivalEvent(arrivalTime, parking.getId(), personId, destCoordinate, walkScore));
211
212                return parking;
213        }

220                // availablePublicParkingAtCityCentre();
221
222                double finalScore = 0;
223
224                PC2Parking selectedParking = null;
225                boolean parkingFound = false;
226
227                // first search for parking at selected facility:
228                for (PPRestrictedToFacilities pp : privateParkingsRestrictedToFacilities
229                                .get(parkingOperationRequestAttributes.facilityId)) {
230                        if (pp.getAvailableParkingCapacity() > 0) {
231
232                                // this tells the parking lot to decrease the number of
233                                // available spaces:
234                                pp.parkVehicle();
235
236                                // this puts the personId (!!!! yyyyyy) at the parking location:
237                                parkedVehicles.put(parkingOperationRequestAttributes.personId, pp.getId());
238
239                                parkingFound = true;
240                                selectedParking = pp;
241                        }
242                }
243
244                // if not found, search within distance of 300m:
245                double distance = 300;
246                if (!parkingFound) {
247                        Collection<PC2Parking> collection = getFilteredCollection(parkingOperationRequestAttributes, distance);
248
249                        while (collection.size() == 0) {
250                                distance *= 2;
251                                collection = getFilteredCollection(parkingOperationRequestAttributes, distance);
252
253                                if (distance > 100000000) {
254                                        // stop infinite loop
255                                        DebugLib.stopSystemAndReportInconsistency(
256                                                        "not enough public parking in scenario - introduce dummy parking to solve problem");
257                                }
258
259                        }
260
261                        // put parking that was found into a sorted queue:
262                        PriorityQueue<SortableMapObject<PC2Parking>> queue = new PriorityQueue<SortableMapObject<PC2Parking>>();
263                        for (PC2Parking parking : collection) {
264                                double score = parkingScoreManager.calcScore(parkingOperationRequestAttributes.destCoordinate,
265                                                parkingOperationRequestAttributes.arrivalTime,
266                                                parkingOperationRequestAttributes.parkingDurationInSeconds, parking,
267                                                parkingOperationRequestAttributes.personId, parkingOperationRequestAttributes.legIndex, false);
268                                queue.add(new SortableMapObject<PC2Parking>(parking, -1.0 * score)); // score made positive, so that priority queue works properly
269                        }
270
271                        // TODO: should I make MNL only on top 5 here?
272
273                        // select the best one from the queue:
274                        SortableMapObject<PC2Parking> poll = queue.peek();
275                        finalScore = poll.getWeight();
276                        selectedParking = poll.getKey();
277
278                        if (selectedParking instanceof RentableParking) {
279                                DebugLib.emptyFunctionForSettingBreakPoint();
280                        }
281
282                        if (rentablePrivateParking.containsKey(parkingOperationRequestAttributes.personId)) {
283                                RentableParking rentableParking = rentablePrivateParking
284                                                .get(parkingOperationRequestAttributes.personId);

287

304

307                        }
308
309                        // this puts the personId (!!!! yyyyyy) at the parking location:
310                        parkedVehicles.put(parkingOperationRequestAttributes.personId, selectedParking.getId());
311
312                        // this tells the parking lot to decrease the number of
313                        // available spaces:
314                        parkVehicle(selectedParking);
315
316                        // PC2Parking closestParking =
317                        // getPublicParkingQuadTree().get(parkingOperationRequestAttributes.destCoordinate.getX(),

386        private Collection<PC2Parking> getFilteredCollection(
387                        ParkingOperationRequestAttributes parkingOperationRequestAttributes, double distance) {
388                Collection<PC2Parking> collection = getPublicParkingQuadTree().getDisk(
389                                parkingOperationRequestAttributes.destCoordinate.getX(),
390                                parkingOperationRequestAttributes.destCoordinate.getY(), distance);
391
392                LinkedList<PC2Parking> deleteList = new LinkedList<>();
393
394                for (PC2Parking parking : collection) {
395                        if (parking instanceof RentableParking) {
396                                RentableParking rp = (RentableParking) parking;
397
398                                if (!rp.isRentable(parkingOperationRequestAttributes.arrivalTime)) {
399                                        deleteList.add(rp);
400                                } else {
401                                        DebugLib.emptyFunctionForSettingBreakPoint();
402                                }
403                        }
404                }
405                collection.removeAll(deleteList);
406                return collection;
407        }

429        private synchronized void parkVehicle(PC2Parking parking) {
430                int startAvailability = parking.getAvailableParkingCapacity();
431
432                parking.parkVehicle();
433                if (parking.getAvailableParkingCapacity() == 0) {
434                        boolean wasRemoved = false;
435                        wasRemoved = publicParkingsQuadTree.remove(parking.getCoordinate().getX(), parking.getCoordinate().getY(),
436                                        parking);
437
438                        if (!wasRemoved) {
439                                DebugLib.stopSystemAndReportInconsistency(parking.getId().toString());
440                        }
441
442                        wasRemoved = publicParkingGroupQuadTrees.get(parking.getGroupName()).remove(parking.getCoordinate().getX(),
443                                        parking.getCoordinate().getY(), parking);
444
445                        if (!wasRemoved) {
446                                DebugLib.stopSystemAndReportInconsistency(parking.getId().toString());
447                        }
448
449                        Collection<PC2Parking> collection = publicParkingsQuadTree.getDisk(parking.getCoordinate().getX(),
450                                        parking.getCoordinate().getY(), 1);
451                        if (collection.size() == 1) {
452                                for (PC2Parking p : collection) {
453                                        // yyyyyy no idea why we first test for "size of collection
454                                        // == 1" and then iterate over it. kai, jul'15
455
456                                        if (p.getId().toString().equalsIgnoreCase(parking.getId().toString())) {
457                                                // (i.e. the found parking is not the same as the one
458                                                // that should be at the coordinate; two parking
459                                                // locations are
460                                                // on top of each other)
461
462                                                DebugLib.stopSystemAndReportInconsistency();
463                                        }
464                                }
465                        }
466                }
467
468                int endAvailability = parking.getAvailableParkingCapacity();
469
470                DebugLib.assertTrue(startAvailability - 1 == endAvailability, "not equal");
471
472        }

487        @Override
488        public synchronized PC2Parking personCarDepartureEvent(
489                        ParkingOperationRequestAttributes parkingOperationRequestAttributes) {
490                final Id<Person> personId = parkingOperationRequestAttributes.personId;
491                Id<PC2Parking> parkingFacilityId = parkedVehicles.get(personId);
492                PC2Parking parking = getAllParkings().get(parkingFacilityId);
493
494                parkedVehicles.remove(personId);
495
496                unParkVehicle(parking, parkingOperationRequestAttributes.arrivalTime
497                                + parkingOperationRequestAttributes.parkingDurationInSeconds, personId);
498                return parking;
499        }

511        @Override
512        public synchronized void unParkVehicle(PC2Parking parking, double departureTime, Id<Person> personId) {
513                // yyyy I would prefer if this could not be called publicly ... since it
514                // allows to bypass parkedVehicles.remove(personId);
515                // in personCarDepartureEvent(...). kai, jul'15
516
517                if (parking == null) {
518                        DebugLib.emptyFunctionForSettingBreakPoint();
519                }
520                int startAvailability = parking.getAvailableParkingCapacity();
521
522                parking.unparkVehicle();
523
524                if (parking.getAvailableParkingCapacity() == 1) {
525                        if (!(parking instanceof PrivateParking)) {
526                                addParkingToQuadTree(publicParkingsQuadTree, parking);
527                                addParkingToQuadTree(publicParkingGroupQuadTrees.get(parking.getGroupName()), parking);
528                                // I could speculate that full parking lots are removed from the
529                                // quad tree, and it is re-added as soon
530                                // as space becomes available. But this is not commented, and it
531                                // also does not look like it would work in that way.
532                                // kai, jul'15
533                                // No, actually I now think that it works. When
534                                // remainingCapacity==0, then the parking is removed from the
535                                // list, and
536                                // when (again) ==1, it is re-inserted. In addition, it is
537                                // re-inserted in reset.
538                        }
539                }
540
541                int endAvailability = parking.getAvailableParkingCapacity();
542
543                DebugLib.assertTrue(startAvailability + 1 == endAvailability, "not equal");
544
545                eventsManager.processEvent(new ParkingDepartureEvent(departureTime, parking.getId(), personId));
546        }
