006import java.io.BufferedWriter;
007import java.io.IOException;
008import java.text.DecimalFormat;

010import java.util.HashMap;
011import java.util.TreeSet;
012
013import org.apache.log4j.Logger;
014import org.matsim.contrib.parking.parkingsearch.manager.ParkingSearchManager;
015import org.matsim.contrib.parking.parkingsearch.manager.ZoneParkingManager;
016import org.matsim.core.config.Config;

018import org.matsim.core.mobsim.framework.events.MobsimAfterSimStepEvent;

026/**
027 * @author tschlenther
028 *
029 */
030public class ZoneParkingOccupationListener implements MobsimBeforeCleanupListener, MobsimAfterSimStepListener{
031        
032        private final ZoneParkingManager zoneManager;
033        private Logger log = Logger.getLogger(ZoneParkingOccupationListener.class);
034        private HashMap<String,TreeSet<ParkingTuple>> zoneOccupationPerTime; 
035        private MatsimServices services;
036        int iteration;
037        
038        private TreeSet<Double> allMonitoredTimeStamps;
039        
040        /**
041         * 
042         */
043        @Inject
044        public ZoneParkingOccupationListener(ParkingSearchManager manager, MatsimServices services, Config config) {
045                this.zoneManager = (ZoneParkingManager) manager;
046                iteration = config.controler().getFirstIteration();
047                this.services = services;
048                this.zoneOccupationPerTime = new HashMap<String,TreeSet<ParkingTuple>>();
049                for(String zone : zoneManager.getZones()){
050                        this.zoneOccupationPerTime.put(zone, new TreeSet<ParkingTuple>());
051                }
052                this.allMonitoredTimeStamps = new TreeSet<Double>();
053        }
054
055        @Override
056        public void notifyMobsimAfterSimStep(MobsimAfterSimStepEvent e) {
057                if( (e.getSimulationTime()%900)  == 0 && !(e.getSimulationTime() == 0)){
058                        
059                        double  timeStamp = e.getSimulationTime();
060                        if (!(this.allMonitoredTimeStamps.contains(timeStamp)) ) this.allMonitoredTimeStamps.add(timeStamp);
061                        
062                        for(String zone : this.zoneOccupationPerTime.keySet()){
063                                double occ = this.zoneManager.getOccupancyRatioOfZone(zone);
064                                ParkingTuple tuple = new ParkingTuple(timeStamp, occ);
065                                this.zoneOccupationPerTime.get(zone).add(tuple);        
066                        }
067                }
068        }
069
070        @Override
071        public void notifyMobsimBeforeCleanup(MobsimBeforeCleanupEvent e) {
072                String fileName = services.getControlerIO().getIterationFilename(iteration, "OccupationStats.csv");
073                writeStatsVertical(fileName);
074                double lastOcc;
075                for(String zone : zoneManager.getZones()){
076                        lastOcc = this.zoneManager.getOccupancyRatioOfZone(zone);
077//                      lastOcc = this.zoneOccupationPerHour.get(zone)[this.zoneOccupationPerHour.get(zone).length-1];
078                        TreeSet<ParkingTuple> tSet = new TreeSet<ParkingTuple>();
079                        tSet.add(new ParkingTuple(-1.0,lastOcc));
080                        this.zoneOccupationPerTime.put(zone, tSet);
081                        this.allMonitoredTimeStamps.clear();
082                        this.allMonitoredTimeStamps.add(-1.0);
083                }               
084                iteration++;
085        }
086        
087        private void writeStats(String fileName) {
088                log.error("WRITING OCCUPANCY STATS TO " + fileName);
089
090                String head = "Zone;TotalCapacity;";
091                for(double d : this.allMonitoredTimeStamps){
092                        head += "" + d +";";
093                }
094                
095                BufferedWriter bw = IOUtils.getBufferedWriter(fileName);
096                try {
097                        bw.write(head);
098                        for (String zone : this.zoneOccupationPerTime.keySet()){
099                                bw.newLine();
100                                String line = zone + ";" + this.zoneManager.getTotalCapacityOfZone(zone) + ";";
101
102                                TreeSet<ParkingTuple> currentSet = this.zoneOccupationPerTime.get(zone);
103                                TreeSet<Double> timeSet = new TreeSet<Double>();
104                                timeSet.addAll(allMonitoredTimeStamps);
105                                double oldOcc = -5.0;
106                                
107                                int zoneSize = currentSet.size();
108                                int timeSize = timeSet.size();
109                                
110                                for(int i = 0; i < zoneSize; i++){
111                                        ParkingTuple ff = currentSet.pollFirst();
112                                        for(int zz = 0; zz < timeSize; zz ++ ){
113                                                double time = timeSet.pollFirst();
114                                                if(time < ff.getTime()){
115                                                        line += "" + oldOcc +";";
116                                                }
117                                                else if( time == ff.getTime()){
118                                                        line += "" + ff.getOccupancy() + ";";
119                                                        break;
120                                                }
121                                                else{
122                                                        String timeString = "This is what the TreeSet looks like:\n" + timeSet.toString() + "\n This is what the original looks like:\n" + this.allMonitoredTimeStamps.toString();
123                                                        throw new RuntimeException("timestamp of zone " + zone + " wasn't recorded.\n timeStamp of zone = " + ff.getTime() + "\t header column: " + time
124                                                                        + "\n\n" + timeString + "\n This is that the file looks like:\n" + head + "\n" + line);
125                                                }
126                                                oldOcc = ff.getOccupancy();
127                                        }
128                                }
129                                bw.write(line);
130                        }
131                        bw.flush();
132                        bw.close();
133                } catch (IOException e) {
134                        // TODO Auto-generated catch block
135                        e.printStackTrace();
136                }
137        
138        log.error("FINISHED WRITING OCCUPANCY STATS");
139        }
140
141        private void writeStatsVertical(String fileName) {
142                log.error("WRITING OCCUPANCY STATS TO " + fileName);
143
144                String head = ";;";
145                String capacityLine = "TotalCapacity;;";
146                String subheader = "SlotNr;SimTime;";
147                
148                ArrayList<double[]> zoneColumns = new ArrayList<double[]>();
149                int numberOfSlots = this.allMonitoredTimeStamps.size();
150                double[] slotTimes = new double[numberOfSlots];
151                int[] slotNumbers = new int[numberOfSlots];
152                TreeSet<Double> timeSet = new TreeSet<Double>();
153                timeSet.addAll(allMonitoredTimeStamps);
154                
155                for (int timeSlot = 0; timeSlot < numberOfSlots; timeSlot ++ ){
156                        double tt = timeSet.pollFirst();
157                        slotTimes[timeSlot] = tt;
158                        slotNumbers[timeSlot] = (int) (tt/900); 
159                }
160                
161                for(String zone : this.zoneManager.getZones()){
162                        head += zone +";";
163                        capacityLine += this.zoneManager.getTotalCapacityOfZone(zone) + ";";
164                        subheader += "OccRatio;";
165                        
166                        
167                        double[] ratios = new double[numberOfSlots];
168                        
169                        TreeSet<ParkingTuple> ratioSet = this.zoneOccupationPerTime.get(zone); 
170                        
171                        double oldRatio = 0.0;
172                        for(int i = 0; i < numberOfSlots; i++){
173                                
174                                ParkingTuple tuple = ratioSet.pollFirst(); 
175                                if(slotTimes[i] == tuple.getTime()){
176                                        ratios[i] = tuple.getOccupancy();
177                                        oldRatio = tuple.getOccupancy();
178                                }
179                                else if(slotTimes[i] < tuple.getTime()){
180                                        ratios[i] = oldRatio;
181                                }
182                                else{
183                                        throw new RuntimeException("timeSlot number is higher than recorded time. timeSlot: " + slotTimes[i] + "\t time: " + tuple.getTime());
184                                }
185                        }
186                        
187                        zoneColumns.add(ratios);                        
188                }
189                
190                                BufferedWriter bw = IOUtils.getBufferedWriter(fileName);
191                try {
192                        bw.write(head);
193                        bw.newLine();
194                        bw.write(capacityLine);
195                        bw.newLine();
196                        bw.write(subheader);
197                        
198                        DecimalFormat df = new DecimalFormat("##.##");
199                        
200                        for(int z = 0; z < numberOfSlots; z++){
201                                bw.newLine();
202                                
203                                String line = "" + slotNumbers[z] + ";" + df.format(slotTimes[z]) + ";";
204                                for(int p = 0; p < this.zoneManager.getZones().size(); p++){
205                                        line += "" + df.format(zoneColumns.get(p)[z]) + ";";
206                                }
207                                
208                                bw.write(line);
209                        }
210                                
211                        bw.flush();
212                        bw.close();
213                } catch (IOException e) {
214                        // TODO Auto-generated catch block
215                        e.printStackTrace();
216                }
217        
218        log.error("FINISHED WRITING OCCUPANCY STATS");
219        }

221}
