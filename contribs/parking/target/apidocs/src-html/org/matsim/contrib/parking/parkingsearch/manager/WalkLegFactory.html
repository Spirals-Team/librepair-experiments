020package org.matsim.contrib.parking.parkingsearch.manager;
021
022import javax.inject.Inject;
023
024import org.matsim.api.core.v01.Id;
025import org.matsim.api.core.v01.network.Link;
026import org.matsim.api.core.v01.network.Network;
027import org.matsim.api.core.v01.population.Leg;
028import org.matsim.api.core.v01.population.Route;
029import org.matsim.core.config.Config;
030import org.matsim.core.population.PopulationUtils;
031import org.matsim.core.population.routes.RouteUtils;
032import org.matsim.core.utils.geometry.CoordUtils;
033
034/**
035 * @author  jbischoff
036 *
037 */
038public class WalkLegFactory {
039        final Network network;
040        final double walkspeed;
041        final double beelinedistancefactor;
042        
043        @Inject
044        public WalkLegFactory(Network network, Config config) {
045                this.network = network;
046                this.beelinedistancefactor = config.plansCalcRoute().getBeelineDistanceFactors().get("walk");
047                this.walkspeed = config.plansCalcRoute().getTeleportedModeSpeeds().get("walk");
048                
049        }
050
051        public Leg createWalkLeg(Id<Link> from, Id<Link> to, double startTime, String walkMode){
052                Leg leg = PopulationUtils.createLeg(walkMode);
053                double walkDistance = CoordUtils.calcEuclideanDistance(network.getLinks().get(from).getCoord(), network.getLinks().get(to).getCoord())*beelinedistancefactor;
054                double walkTime = walkDistance / walkspeed;
055                Route route = RouteUtils.createGenericRouteImpl(from, to);
056                route.setDistance(walkDistance);
057                route.setTravelTime(walkTime);
058                leg.setRoute(route);
059                leg.setDepartureTime(startTime);
060                return leg;
061        }
062        
063        
064}
