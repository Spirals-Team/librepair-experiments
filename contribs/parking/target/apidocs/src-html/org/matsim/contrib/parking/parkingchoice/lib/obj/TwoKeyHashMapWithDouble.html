021package org.matsim.contrib.parking.parkingchoice.lib.obj;
022
023import java.util.HashMap;
024import java.util.Set;
025
026
027public class TwoKeyHashMapWithDouble<ClassKey1,ClassKey2> {
028
029        // TODO: write test
030private HashMap<ClassKey1, DoubleValueHashMap<ClassKey2>> hashMap=new HashMap<ClassKey1, DoubleValueHashMap<ClassKey2>>();
031        
032        public DoubleValueHashMap<ClassKey2> getDoubleValueHashMap(ClassKey1 key1){
033                return hashMap.get(key1);
034        }
035
036        public Set<ClassKey1> getKeySet1(){
037                return hashMap.keySet();
038        }
039        
040        public boolean containsKeyTwo(ClassKey1 key1, ClassKey2 key2){
041                checkHashMapAndInitializeIfNeeded(key1);
042                return hashMap.get(key1).containsKey(key2);
043        }
044        
045        public void removeValue(ClassKey1 key1, ClassKey2 key2){
046                checkHashMapAndInitializeIfNeeded(key1);
047                hashMap.get(key1).remove(key2);
048        }
049
050        public void put(ClassKey1 key1, ClassKey2 key2, double value){
051                checkHashMapAndInitializeIfNeeded(key1);
052                hashMap.get(key1).put(key2, value);
053        }
054        
055        public void incrementBy(ClassKey1 key1, ClassKey2 key2, double incValue){
056                checkHashMapAndInitializeIfNeeded(key1);
057                hashMap.get(key1).incrementBy(key2, incValue);
058        }
059        
060        public void increment(ClassKey1 key1, ClassKey2 key2){
061                incrementBy(key1,key2,1.0);
062        }
063        
064        public DoubleValueHashMap<ClassKey2> get(ClassKey1 key1){
065                return hashMap.get(key1);
066        }
067        
068        public double getAverage(){
069                double sum = 0;
070                for (ClassKey1 key : hashMap.keySet()) {
071                        DoubleValueHashMap<ClassKey2> curValue = hashMap.get(key);
072                        sum += curValue.getAverage();
073                }
074                return sum / hashMap.size();
075        }
076        
077        public double get(ClassKey1 key1, ClassKey2 key2){
078                checkHashMapAndInitializeIfNeeded(key1);
079                return hashMap.get(key1).get(key2);
080        }
081        
082        
083        
084        private void checkHashMapAndInitializeIfNeeded(ClassKey1 key1){
085                if (!hashMap.containsKey(key1)){
086                        hashMap.put(key1, new DoubleValueHashMap<ClassKey2>());
087                }
088        }
089        
090        public Set<ClassKey2> getKeySet2(ClassKey1 key1){
091                return hashMap.get(key1).keySet();
092        }
093}
