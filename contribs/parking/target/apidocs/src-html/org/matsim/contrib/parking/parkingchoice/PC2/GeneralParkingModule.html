001package org.matsim.contrib.parking.parkingchoice.PC2;
002
003import org.matsim.contrib.parking.parkingchoice.PC2.scoring.ParkingScore;
004import org.matsim.contrib.parking.parkingchoice.PC2.simulation.ParkingChoiceSimulation;
005import org.matsim.contrib.parking.parkingchoice.PC2.simulation.ParkingInfrastructure;
006import org.matsim.core.controler.Controler;
007import org.matsim.core.controler.events.BeforeMobsimEvent;
008import org.matsim.core.controler.events.StartupEvent;
009import org.matsim.core.controler.listener.BeforeMobsimListener;
010import org.matsim.core.controler.listener.StartupListener;
011
012public class GeneralParkingModule implements StartupListener, BeforeMobsimListener {
013
014        private Controler controler;
015        private ParkingScore parkingScoreManager;
016        public final ParkingScore getParkingScoreManager() {
017                return parkingScoreManager;
018        }
019
020        public final void setParkingScoreManager(ParkingScore parkingScoreManager) {
021                this.parkingScoreManager = parkingScoreManager;
022        }
023
024        private ParkingInfrastructure parkingInfrastructureManager;
025        private ParkingChoiceSimulation parkingSimulation;
026
027        public GeneralParkingModule(Controler controler){
028                this.controler = controler ;
029                
030                controler.addControlerListener(this);
031        }
032        
033        @Override
034        public void notifyStartup(StartupEvent event) {
035                parkingSimulation = new ParkingChoiceSimulation(controler.getScenario(), parkingInfrastructureManager);
036                controler.getEvents().addHandler(parkingSimulation);
037//              controler.addControlerListener(parkingSimulation);
038                // was not doing anything there. kai, jul'15
039        }
040
041        public final ParkingInfrastructure getParkingInfrastructure() {
042                return parkingInfrastructureManager;
043        }
044        
045        public final void setParkingInfrastructurManager(ParkingInfrastructure parkingInfrastructureManager) {
046                this.parkingInfrastructureManager = parkingInfrastructureManager;
047        }
048
049        @Deprecated
050        // lower level objects may keep back pointers to higher level objects if they have to, but we prefer that they do not provide them
051        // as a service. kai, apr'15
052        public final Controler getControler() {
053                return controler;
054        }
055
056        @Override
057        public void notifyBeforeMobsim(BeforeMobsimEvent event) {
058                parkingScoreManager.prepareForNewIteration();
059                parkingInfrastructureManager.reset();
060                parkingSimulation.prepareForNewIteration();
061        }
062
063        protected final ParkingInfrastructure getParkingInfrastructureManager() {
064                return parkingInfrastructureManager;
065        }
066        
067        protected final ParkingChoiceSimulation getParkingSimulation() {
068                return parkingSimulation;
069        }       
070
071}
