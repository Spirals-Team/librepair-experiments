004package org.matsim.contrib.parking.parkingsearch.search;
005
006import java.util.HashSet;
007import java.util.Iterator;
008import java.util.Set;
009
010import org.apache.log4j.Logger;
011import org.matsim.api.core.v01.Id;
012import org.matsim.api.core.v01.network.Link;
013import org.matsim.api.core.v01.network.Network;
014import org.matsim.api.core.v01.network.Node;
015import org.matsim.contrib.parking.parkingsearch.search.ParkingSearchLogic;
016import org.matsim.core.gbl.MatsimRandom;
017import org.matsim.core.network.NetworkUtils;
018import org.matsim.vehicles.Vehicle;
019
020/**
021 * @author tschlenther
022 *
023 *Agents drive to destination first. Knowledge about surrounding streets is assumed. If no parking slot is available, they always look
024 *for a slot on the one outgoing link that has the shortest distance to their destination and is unknown to them so far. If every outlink
025 *is known they choose the next link to search on randomly.
026 *
027 */
028public class DistanceMemoryParkingSearchLogic implements ParkingSearchLogic {
029
030        private static final Logger logger = Logger.getLogger(DistanceMemoryParkingSearchLogic.class);
031
032        private static final boolean doLogging = false;
033        
034        private Network network;
035        private HashSet<Id<Link>> knownLinks;   
036        
037        /**
038         * @param network 
039         * 
040         */
041        public DistanceMemoryParkingSearchLogic(Network network) {
042                this.network = network;
043                this.knownLinks = new HashSet<Id<Link>>();
044        }
045
046        public Id<Link> getNextLink(Id<Link> currentLinkId, Id<Link> destLinkId, Id<Vehicle> vehicleId) {
047                
048                Node toNode = network.getLinks().get(currentLinkId).getToNode();
049                
050                Set<Id<Link>> outLinks = toNode.getOutLinks().keySet();
051                double shortestDistance = Double.MAX_VALUE;
052                int nrKnownLinks = 0;
053                Id<Link> nextLink = null;
054                
055                if(doLogging) logger.info("number of outlinks of link " + currentLinkId + ": " + outLinks.size());
056                
057                for(Id<Link> outLink : outLinks){
058                        
059                        if(this.knownLinks.contains(outLink)) nrKnownLinks ++;
060                        else{
061                                double distToDest = NetworkUtils.getEuclideanDistance(network.getLinks().get(outLink).getCoord(), network.getLinks().get(destLinkId).getCoord());
062                                if (distToDest < shortestDistance){
063                                        nextLink = outLink;
064                                        shortestDistance = distToDest;
065                                        if(doLogging) logger.info("currently chosen link: " + nextLink + " distToDest: " + shortestDistance);
066                                }
067                                else if(distToDest == shortestDistance){
068                                        String message = "link " + nextLink + " and link " + outLink + " both are " + distToDest + "m away from destination.";
069                                                if (Math.random() > 0.5)
070                                                        nextLink = outLink;     
071                                                if(doLogging) logger.info(message + " link " + nextLink + " is chosen.");
072                                }
073                                else{
074                                        if (doLogging){
075                                                logger.info("link " + outLink + " was not chosen because it is " + distToDest + "m away whereas shortest distance is " + shortestDistance);
076                                        }
077                                }
078                        }
079                }
080                if(doLogging)logger.error("vehicle " + vehicleId + " knew " + nrKnownLinks + " out of " + outLinks.size() + " outlinks of link " + currentLinkId);
081                if(outLinks.size() == nrKnownLinks ){
082                        if(doLogging)logger.error("vehicle " + vehicleId + " knows all outlinks of link " + currentLinkId);
083                        
084                        //return random Link
085                        int index = MatsimRandom.getRandom().nextInt(outLinks.size());
086                        Iterator<Id<Link>> iter = outLinks.iterator();
087                        for (int i = 0; i < index; i++) {
088                            iter.next();
089                        }
090                        nextLink = iter.next();
091                }                               
092                
093                if(nextLink == null){
094                        throw new RuntimeException("the next Link Id for vehicle " + vehicleId + " on current link " + currentLinkId + " couldn't be calculated.");
095                }
096                if(doLogging)logger.error("vehicle " + vehicleId + " takes link " + nextLink + " as next link");
097                this.knownLinks.add(nextLink);
098                return nextLink;
099        }
100
101        @Override
102        public void reset() {
103        }
104
105        @Override
106        public Id<Link> getNextLink(Id<Link> currentLinkId, Id<Vehicle> vehicleId) {
107                throw new RuntimeException("shouldn't happen - method not implemented");
108        }
109
110        public void addToKnownLinks(Id<Link> linkId){
111                this.knownLinks.add(linkId);
112        }
113        
114}
