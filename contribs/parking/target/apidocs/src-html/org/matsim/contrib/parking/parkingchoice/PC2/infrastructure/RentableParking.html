001package org.matsim.contrib.parking.parkingchoice.PC2.infrastructure;
002
003import org.matsim.api.core.v01.Coord;
004import org.matsim.api.core.v01.Id;
005import org.matsim.api.core.v01.population.Person;
006import org.matsim.contrib.parking.parkingchoice.PC2.scoring.ParkingCostModel;
007import org.matsim.contrib.parking.parkingchoice.lib.GeneralLib;
008import org.matsim.facilities.ActivityFacility;
009
010public class RentableParking extends PublicParking  {
011
012        public RentableParking(Id<PC2Parking> id, int capacity, Coord coord, ParkingCostModel parkingCostModel,
013                        String groupName) {
014                super(id, capacity, coord, parkingCostModel, groupName);
015        }
016
017        
018        public Id<Person> getOwnerId() {
019                return ownerId;
020        }
021        public void setOwnerId(Id<Person> ownerId) {
022                this.ownerId = ownerId;
023        }
024
025public boolean isRentable(double time){
026        return GeneralLib.isIn24HourInterval(startRentableTime, endRentableTime, GeneralLib.projectTimeWithin24Hours(time)) && getAvailableParkingCapacity()>0;
027}       
028        
029        private Id<Person> ownerId;
030        double startRentableTime;
031        public double getStartRentableTime() {
032                return startRentableTime;
033        }
034
035
036        public void setStartRentableTime(double startRentableTime) {
037                this.startRentableTime = startRentableTime;
038        }
039
040
041        public double getEndRentableTime() {
042                return endRentableTime;
043        }
044
045
046        public void setEndRentableTime(double endRentableTime) {
047                this.endRentableTime = endRentableTime;
048        }
049
050
051        public double getRentingPricePerHourInCurrencyUnit() {
052                return rentingPricePerHourInCurrencyUnit;
053        }
054
055
056        public void setRentingPricePerHourInCurrencyUnit(double rentingPricePerHourInCurrencyUnit) {
057                this.rentingPricePerHourInCurrencyUnit = rentingPricePerHourInCurrencyUnit;
058        }
059
060
061        double endRentableTime;
062        double rentingPricePerHourInCurrencyUnit;
063        
064        @Override
065        public double getCost(Id<Person> personId, double arrivalTime, double parkingDurationInSecond){
066                return rentingPricePerHourInCurrencyUnit*parkingDurationInSecond/3600;
067        }
068        
069}
