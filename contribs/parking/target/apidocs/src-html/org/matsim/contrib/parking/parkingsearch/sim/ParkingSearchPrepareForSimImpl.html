020package org.matsim.contrib.parking.parkingsearch.sim;
021
022
023import java.util.HashSet;
024import javax.inject.Inject;
025import javax.inject.Provider;
026import org.apache.log4j.Logger;
027import org.matsim.api.core.v01.Scenario;
028import org.matsim.api.core.v01.TransportMode;
029import org.matsim.api.core.v01.network.Network;
030import org.matsim.api.core.v01.population.Population;
031import org.matsim.core.config.groups.GlobalConfigGroup;
032import org.matsim.core.controler.PrepareForSim;
033import org.matsim.core.network.NetworkUtils;
034import org.matsim.core.network.algorithms.TransportModeNetworkFilter;
035import org.matsim.core.population.algorithms.AbstractPersonAlgorithm;
036import org.matsim.core.population.algorithms.ParallelPersonAlgorithmUtils;
037import org.matsim.core.population.algorithms.PersonPrepareForSim;
038import org.matsim.core.router.PlanRouter;
039import org.matsim.core.router.TripRouter;
040import org.matsim.core.scenario.Lockable;
041import org.matsim.facilities.ActivityFacilities;
042
043public class ParkingSearchPrepareForSimImpl implements PrepareForSim {
044    private static Logger log = Logger.getLogger(PrepareForSim.class);
045
046    private final GlobalConfigGroup globalConfigGroup;
047    private final Scenario scenario;
048    private final Network network;
049    private final Population population;
050    private final ActivityFacilities activityFacilities;
051    private final Provider<TripRouter> tripRouterProvider;
052
053    @Inject
054    ParkingSearchPrepareForSimImpl(GlobalConfigGroup globalConfigGroup, Scenario scenario, Network network, Population population, ActivityFacilities activityFacilities, Provider<TripRouter> tripRouterProvider) {
055        this.globalConfigGroup = globalConfigGroup;
056        this.scenario = scenario;
057        this.network = network;
058        this.population = population;
059        this.activityFacilities = activityFacilities;
060        this.tripRouterProvider = tripRouterProvider;
061    }
062
063
064    @Override
065    public void run() {
066                /*
067                 * Create single-mode network here and hand it over to PersonPrepareForSim. Otherwise, each instance would create its
068                 * own single-mode network. However, this assumes that the main mode is car - which PersonPrepareForSim also does. Should
069                 * be probably adapted in a way that other main modes are possible as well. cdobler, oct'15.
070                 */
071        final Network net;
072        if (NetworkUtils.isMultimodal(network)) {
073            log.info("Network seems to be multimodal. Create car-only network which is handed over to PersonPrepareForSim.");
074            TransportModeNetworkFilter filter = new TransportModeNetworkFilter(network);
075            net = NetworkUtils.createNetwork();
076            HashSet<String> modes = new HashSet<>();
077            modes.add(TransportMode.car);
078            filter.filter(net, modes);
079        } else {
080            net = network;
081        }
082
083        // make sure all routes are calculated.
084        ParallelPersonAlgorithmUtils.run(population, globalConfigGroup.getNumberOfThreads(),
085                new ParallelPersonAlgorithmUtils.PersonAlgorithmProvider() {
086                    @Override
087                    public AbstractPersonAlgorithm getPersonAlgorithm() {
088                        return new PersonPrepareForSim(new PlanRouter(tripRouterProvider.get(), activityFacilities), scenario, net);
089                    }
090                });
091
092        if (scenario instanceof Lockable) {
093            ((Lockable)scenario).setLocked();
094            // see comment in ScenarioImpl. kai, sep'14
095        }
096
097        if (population instanceof Lockable) {
098            ((Lockable) population).setLocked();
099        }
100
101        if ( network instanceof Lockable ) {
102            ((Lockable) network).setLocked();
103        }
104
105        // (yyyy means that if someone replaces prepareForSim and does not add the above lines, the containers are not locked.  kai, nov'16)
106
107    }
108}
