001package org.matsim.contrib.parking.parkingchoice.lib.obj;
002
003import java.util.Collection;
004import java.util.HashMap;
005import java.util.LinkedList;
006// TODO: write tests.
007public class LinkedListValueHashMap<KeyClass,ValueClass> {
008
009        HashMap<KeyClass, LinkedList<ValueClass>> hashMap=new HashMap<KeyClass, LinkedList<ValueClass>>();
010        HashMap<ValueClass, KeyClass> hashMapBackPointer=new HashMap<ValueClass, KeyClass>();
011        
012        
013        public Collection<KeyClass> getKeySet(){
014                return hashMap.keySet();
015        }
016        
017        public void putAndSetBackPointer(KeyClass key,ValueClass value){
018                put(key,value);
019                
020                if (hashMapBackPointer.containsKey(value)){
021                        throw new Error("The backpointer concept only works for one-to-one mappings between key and value!");
022                }
023                
024                hashMapBackPointer.put(value, key);
025        }
026        
027        public KeyClass getKey(ValueClass value){
028                if (hashMapBackPointer.size()==0){
029                        throw new Error("use method putAndSetBackPointer(...) for insertion instead of put(...)!");
030                }
031                
032                return hashMapBackPointer.get(value);
033        }
034        
035        public void put(KeyClass key,ValueClass value){
036                initKey(key);
037                
038                LinkedList<ValueClass> list=hashMap.get(key);
039                list.add(value);
040                
041                hashMap.put(key, list);
042        }
043        
044        public LinkedList<ValueClass> get(KeyClass key){
045                initKey(key);
046                
047                return hashMap.get(key);
048        }
049        
050        public ValueClass getValue(KeyClass key){               
051                return get(key).get(0);
052        }
053        
054        private void initKey(KeyClass key){
055                if (!hashMap.containsKey(key)){
056                        hashMap.put(key, new LinkedList<ValueClass>());
057                }
058        }
059        
060        public int size(){
061                return hashMap.size();
062        }
063        
064        public boolean containsKey(KeyClass key){
065                return hashMap.containsKey(key);
066        }
067        
068        
069        public int getNumberOfEntriesInLongestList(){
070                int maxEntries=Integer.MIN_VALUE;
071                
072                for (KeyClass key:hashMap.keySet()){
073                        if (maxEntries<hashMap.get(key).size()){
074                                maxEntries=hashMap.get(key).size();
075                        }
076                }
077                
078                return maxEntries;
079        }
080        
081        
082        
083}
