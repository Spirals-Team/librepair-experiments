055public final class ParkingPopulationAgentSource implements AgentSource {
056        private static final Logger log = Logger.getLogger( ParkingPopulationAgentSource.class );
057
058        private final Population population;
059        private final AgentFactory agentFactory;
060        private final QSim qsim;
061        private Map<String, VehicleType> modeVehicleTypes;
062        private final Collection<String> mainModes;
063        private Map<Id<Vehicle>,Id<Link>> seenVehicleIds = new HashMap<>() ;
064
065        @Inject
066        public ParkingPopulationAgentSource(Population population, AgentFactory agentFactory, QSim qsim ) {
067                Vehicles vehicles = qsim.getScenario().getVehicles() ;
068                QSimConfigGroup qsimConfig = qsim.getScenario().getConfig().qsim() ;
069                
070                this.population = population;
071                this.agentFactory = agentFactory;
072                this.qsim = qsim;  
073                this.modeVehicleTypes = new HashMap<>();
074                this.mainModes = new HashSet<String>( qsim.getScenario().getConfig().qsim().getMainModes());
075                switch ( qsimConfig.getVehiclesSource() ) {
076                case defaultVehicle:
077                        for (String mode : mainModes) {
078                                // initialize each mode with default vehicle type:
079                                modeVehicleTypes.put(mode, VehicleUtils.getDefaultVehicleType());
080                        }
081                        break;
082                case fromVehiclesData:
083                        // don't do anything
084                        break;
085                case modeVehicleTypesFromVehiclesData:
086                        for (String mode : mainModes) {
087                                VehicleType vehicleType = vehicles.getVehicleTypes().get( Id.create(mode, VehicleType.class) ) ;
088                                modeVehicleTypes.put(mode, vehicleType );
089                        }
090                        break;
091                default:
092                        break;
093                
094                }
095                
096                
097                
098        }
099
100        @Override
101        public void insertAgentsIntoMobsim() {
102                for (Person p : population.getPersons().values()) {
103                        MobsimAgent agent = this.agentFactory.createMobsimAgentFromPerson(p);
104                        qsim.insertAgentIntoMobsim(agent);
105                }
106                for (Person p : population.getPersons().values()) {
107                        insertVehicles(p);
108                }
109        }
110
111        private void insertVehicles(Person p) {
112                Plan plan = p.getSelectedPlan();
113                Set<String> seenModes = new HashSet<>();
114                for (PlanElement planElement : plan.getPlanElements()) {
115                        if (planElement instanceof Leg) {
116                                Leg leg = (Leg) planElement;
117                                if (this.mainModes.contains(leg.getMode())) { // only simulated modes get vehicles
118                                        if (!seenModes.contains(leg.getMode())) { // create one vehicle per simulated mode, put it on the home location
119                                                Route route = leg.getRoute();
120                                                Id<Vehicle> vehicleId = null ;
121                                                if (route != null) {
122                                                        if (route instanceof NetworkRoute)      
123                                                        vehicleId = ((NetworkRoute) route).getVehicleId(); // may be null!
124                                                }
125                                                if (vehicleId == null) {
126                                                        if (qsim.getScenario().getConfig().qsim().getUsePersonIdForMissingVehicleId()) {
127                                                                vehicleId = Id.create(p.getId(), Vehicle.class);
128                                                        } else {
129                                                                throw new IllegalStateException("Found a network route without a vehicle id.");
130                                                        }
131                                                }
132                                                Vehicle vehicle = null ;
133                                                switch ( qsim.getScenario().getConfig().qsim().getVehiclesSource() ) {
134                                                case defaultVehicle:
135                                                case modeVehicleTypesFromVehiclesData:
136                                                        vehicle = VehicleUtils.getFactory().createVehicle(vehicleId, modeVehicleTypes.get(leg.getMode()));
137                                                        break;
138                                                case fromVehiclesData:
139                                                        vehicle = qsim.getScenario().getVehicles().getVehicles().get(vehicleId);
140                                                        if (vehicle == null) {
141                                                                throw new IllegalStateException("Expecting a vehicle id which is missing in the vehicles database: " + vehicleId);
142                                                        }
143                                                        break;
144                                                default:
145                                                        throw new RuntimeException("not implemented") ;
146                                                }
147                                                Id<Link> vehicleLinkId = findVehicleLink(p);
148                                                
149                                                // Checking if the vehicle has been seen before:
150                                                Id<Link> result = this.seenVehicleIds.get( vehicleId ) ;
151                                                if ( result != null ) {
152                                                        // if seen before, but placed on same link, then it is ok:
153                                                        log.info( "have seen vehicle with id " + vehicleId + " before; not placing it again." );
154                                                        if ( result != vehicleLinkId ) {
155                                                                throw new RuntimeException("vehicle placement error: vehicleId=" + vehicleId + 
156                                                                                "; previous placement link=" + vehicleLinkId + "; current placement link=" + result ) ; 
157                                                        }
158                                                } else {
159                                                        this.seenVehicleIds.put( vehicleId, vehicleLinkId ) ;
160                                                        qsim.createAndParkVehicleOnLink(vehicle, vehicleLinkId);
161                                                }
162                                                seenModes.add(leg.getMode());
163                                        }
164                                }
165                        }
166                }
167        }

169        /**
170         *      A more careful way to decide where this agent should have its vehicles created
171         *  than to ask agent.getCurrentLinkId() after creation.
172         * @param leg TODO
173         */
174        private Id<Link> findVehicleLink(Person p ) {
175                /* Cases that come to mind:
176                 * (1) multiple persons share car located at home, but possibly brought to different place by someone else.  
177                 *      This is treated by the following algo.
178                 * (2) person starts day with non-car leg and has car parked somewhere else.  This is NOT treated by the following algo.
179                 *      It could be treated by placing the vehicle at the beginning of the first link where it is needed, but this would not
180                 *      be compatible with variant (1).
181                 */
182                for (PlanElement planElement : p.getSelectedPlan().getPlanElements()) {
183                        if (planElement instanceof Activity) {
184                                Activity activity = (Activity) planElement;
185                                ActivityFacilities facilities = this.qsim.getScenario().getActivityFacilities() ;
186                                Config config = this.qsim.getScenario().getConfig() ;
187                                final Id<Link> activityLinkId = PopulationUtils.computeLinkIdFromActivity(activity, facilities, config ) ;
188                                if (activityLinkId != null) {
189                                        return activityLinkId;
190                                }
191                        } else if (planElement instanceof Leg) {
192                                Leg leg = (Leg) planElement;
193                                if (leg.getRoute().getStartLinkId() != null) {
194                                        return leg.getRoute().getStartLinkId();
195                                }
196                        }
197                }
198                throw new RuntimeException("Don't know where to put a vehicle for this agent.");
199        }

202                this.modeVehicleTypes = modeVehicleTypes;
203        }
