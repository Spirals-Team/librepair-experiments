047/**
048 * @author cdobler
049 */
050public class MultimodalTripRouterFactory implements Provider<TripRouter> {
051        
052        private static final Logger log = Logger.getLogger(MultimodalTripRouterFactory.class);
053        
054        private final Scenario scenario;
055        private final TravelDisutilityFactory travelDisutilityFactory;
056        private final Map<String, TravelTime> multimodalTravelTimes;
057        private final Provider<TripRouter> delegateFactory;
058        private final LeastCostPathCalculatorFactory leastCostPathCalculatorFactory;
059        
060        private final Map<String, Network> multimodalSubNetworks = new HashMap<>();
061        
062        @Deprecated
063        public MultimodalTripRouterFactory(Scenario scenario, Map<String, TravelTime> multimodalTravelTimes,
064                        TravelDisutilityFactory travelDisutilityFactory) {
065                this.scenario = scenario;
066                this.multimodalTravelTimes = multimodalTravelTimes;
067                this.travelDisutilityFactory = travelDisutilityFactory;
068
069        // commented this out because constructor is deprecated. hope it's ok. mz
070//              DefaultTripRouterFactoryImpl tripRouterFactory = DefaultTripRouterFactoryImpl.createDefaultTripRouterFactoryImpl(scenario);
071//              this.leastCostPathCalculatorFactory = tripRouterFactory.getLeastCostPathCalculatorFactory();
072//              this.delegateFactory = tripRouterFactory;
073        this.leastCostPathCalculatorFactory = null;
074        this.delegateFactory = null;
075        }
076        
077        public MultimodalTripRouterFactory(Scenario scenario, Map<String, TravelTime> multimodalTravelTimes,
078                        TravelDisutilityFactory travelDisutilityFactory, Provider<TripRouter> delegateFactory,
079                        LeastCostPathCalculatorFactory leastCostPathCalculatorFactory) {
080                this.scenario = scenario;
081                this.multimodalTravelTimes = multimodalTravelTimes;
082                this.travelDisutilityFactory = travelDisutilityFactory;
083                this.delegateFactory = delegateFactory;
084                this.leastCostPathCalculatorFactory = leastCostPathCalculatorFactory;
085        }
086        
087        @Override
088        public TripRouter get() {
089
090                TripRouter instance = this.delegateFactory.get();
091                
092                Network network = this.scenario.getNetwork();
093                PopulationFactory populationFactory = this.scenario.getPopulation().getFactory();
094                MultiModalConfigGroup multiModalConfigGroup = (MultiModalConfigGroup) scenario.getConfig().getModule(MultiModalConfigGroup.GROUP_NAME);
095        Set<String> simulatedModes = CollectionUtils.stringToSet(multiModalConfigGroup.getSimulatedModes());
096                for (String mode : simulatedModes) {
097
098                        if (instance.getRegisteredModes().contains(mode)) {
099                                log.warn("A routing algorithm for " + mode + " is already registered. It is replaced!");
100                        }
101                        
102                        TravelTime travelTime = this.multimodalTravelTimes.get(mode);
103                        if (travelTime == null) {
104                                throw new RuntimeException("No travel time object was found for mode " + mode + "! Aborting.");
105                        }
106                        
107                        Network subNetwork = multimodalSubNetworks.get(mode);
108                        if (subNetwork == null) {
109                                subNetwork = NetworkUtils.createNetwork();
110                                Set<String> restrictions = new HashSet<>();
111                                restrictions.add(mode);
112                                TransportModeNetworkFilter networkFilter = new TransportModeNetworkFilter(network);
113                                networkFilter.filter(subNetwork, restrictions);
114                                this.multimodalSubNetworks.put(mode, subNetwork);
115                        }
116                        
117                        /*
118                         * We cannot use the travel disutility object from the routingContext since it
119                         * has not been created for the modes used here.
120                         */
121                        TravelDisutility travelDisutility = this.travelDisutilityFactory.createTravelDisutility(travelTime);            
122                        LeastCostPathCalculator routeAlgo = this.leastCostPathCalculatorFactory.createPathCalculator(subNetwork, travelDisutility, travelTime);
123                        RoutingModule legRouterWrapper = DefaultRoutingModules.createPureNetworkRouter(mode, populationFactory, subNetwork, routeAlgo ); 
124                        instance.setRoutingModule(mode, legRouterWrapper);
125                }
126
127                return instance;
128        }
129}
