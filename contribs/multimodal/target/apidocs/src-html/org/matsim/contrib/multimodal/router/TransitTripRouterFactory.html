021package org.matsim.contrib.multimodal.router;
022
023import java.util.Set;
024
025import org.apache.log4j.Logger;
026import org.matsim.api.core.v01.Scenario;
027import org.matsim.api.core.v01.TransportMode;
028import org.matsim.api.core.v01.population.PopulationFactory;
029import org.matsim.api.core.v01.population.PopulationFactory;
030import org.matsim.contrib.multimodal.config.MultiModalConfigGroup;
031import org.matsim.core.config.groups.PlansCalcRouteConfigGroup;
032import org.matsim.core.population.routes.RouteFactories;
033import org.matsim.core.router.RoutingModule;
034import org.matsim.core.router.TransitRouterWrapper;
035import org.matsim.core.router.TripRouter;
036import org.matsim.core.router.DefaultRoutingModules;
037import org.matsim.core.utils.collections.CollectionUtils;
038import org.matsim.pt.router.TransitRouter;
039
040import javax.inject.Provider;
041
042/**
043 * @author cdobler
044 */
045public class TransitTripRouterFactory implements Provider<TripRouter> {
046        
047        private static final Logger log = Logger.getLogger(TransitTripRouterFactory.class);
048                
049        private final Provider<TripRouter> delegateFactory;
050        private final Provider<TransitRouter> transitRouterFactory;
051        private final Scenario scenario;
052                
053        public TransitTripRouterFactory(Scenario scenario, Provider<TripRouter> delegateFactory,
054                        Provider<TransitRouter> transitRouterFactory) {
055                this.scenario = scenario;
056                this.delegateFactory = delegateFactory;
057                this.transitRouterFactory = transitRouterFactory;
058        }
059        
060        @Override
061        public TripRouter get() {
062
063                TripRouter tripRouter = this.delegateFactory.get();
064                
065                if (this.scenario.getConfig().transit().isUseTransit()) {
066
067            PopulationFactory populationFactory = this.scenario.getPopulation().getFactory();
068                        RouteFactories modeRouteFactory = ((PopulationFactory) populationFactory).getRouteFactories();
069                        PlansCalcRouteConfigGroup routeConfigGroup = scenario.getConfig().plansCalcRoute();
070                        RoutingModule transitWalkRoutingModule;
071                        
072                        MultiModalConfigGroup multiModalConfigGroup = (MultiModalConfigGroup) scenario.getConfig().getModule(MultiModalConfigGroup.GROUP_NAME);
073                        Set<String> simulatedModes = CollectionUtils.stringToSet(multiModalConfigGroup.getSimulatedModes());
074
075                        // if transit_walk should also be simulated by the multi-modal simulation, re-use the RoutingModule created by the delegate
076                        if (simulatedModes.contains(TransportMode.transit_walk)) {
077                                transitWalkRoutingModule = tripRouter.getRoutingModule(TransportMode.walk);
078                        } else {
079                                transitWalkRoutingModule = DefaultRoutingModules.createTeleportationRouter(TransportMode.transit_walk, populationFactory, 
080                                                routeConfigGroup.getModeRoutingParams().get( TransportMode.walk )
081                                                );
082                        }
083                        
084                TransitRouterWrapper routingModule = new TransitRouterWrapper(transitRouterFactory.get(), scenario.getTransitSchedule(),
085                                scenario.getNetwork(), transitWalkRoutingModule);
086                
087            for (String mode : scenario.getConfig().transit().getTransitModes()) {
088                tripRouter.setRoutingModule(mode, routingModule);
089            }
090                }
091                
092                return tripRouter;
093        }
094}
