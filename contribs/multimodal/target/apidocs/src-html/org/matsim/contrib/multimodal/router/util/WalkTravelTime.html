137        @Override
138        public double getLinkTravelTime(Link link, double time, Person person, Vehicle vehicle) {               
139                setPerson(person);
140                double slope = getSlope(link);
141                double slopeFactor = getSlopeFactor(slope);
142//              double personWalkSpeed = this.referenceWalkSpeed * this.personFactorCache.get();
143                double personWalkSpeed = this.personWalkSpeedCache.get();
144                return link.getLength() / (personWalkSpeed * slopeFactor);
145        }
146
147        /*package*/ double getSlopeFactor(double slope) {
148                double slopeFactor;
149                
150                if (slope > 80.0) {
151                        if (slopeRangeWarnCount.get() < 10) {
152                                incSlopeRangeWarnCount("Slope is out of expected range (-40% .. -80%). Found slope of " + slope + ". Use 80.0 instead.");
153                        }
154                        slope = 80.0;
155                } else if (slope < -40) {
156                        if (slopeRangeWarnCount.get() < 10) {
157                                incSlopeRangeWarnCount("Slope is out of expected range (-40% .. -80%). Found slope of " + slope + ". Use -40.0 instead.");
158                        }
159                        slope = -40.0;
160                }
161                slopeFactor = slopeFactors[-(int)Math.round(slope) + 80];                       
162                return slopeFactor;
163        }

165        /*
166         * Returns the slope of a link in %.
167         */
168        /*package*/ final double getSlope(Link link) {
169                
170                // if no slope information is available
171                if (this.linkSlopes == null) return 0.0;
172                
173                Double slope = this.linkSlopes.get(link.getId());
174                if (slope == null) {
175                        incSlopeNotFoundWarnCount("No slope information for link " + link.getId().toString() +
176                                        " was found. Assuming slope of 0%.");
177                        return 0.0;
178                } else return slope;
179        }

218        /*package*/ void setPerson(Person person) {
219                /* 
220                 * Only recalculate the person's walk speed factor if
221                 * the person has changed and is not in the map.
222                 */
223                if (this.personCache.get() != null && person.getId().equals(this.personCache.get().getId())) return;
224                
225                /*
226                 * If the person's walk speed is already in the map, use that value.
227                 * Otherwise calculate it and add it to the map.
228                 */
229                Double value = this.personFactors.get(person.getId());
230                if (value != null) {
231                        double personFactor = value;
232                        this.personCache.set(person);
233                        this.personFactorCache.set(personFactor);
234                        this.personWalkSpeedCache.set(this.referenceWalkSpeed * personFactor);
235                        return;
236                }
237                
238                double scatterFactor;
239                double ageFactor = 1.0;
240                double genderFactor = 1.0;
241
242                // calculate scatter factor
243                Random random = new Random();
244                random.setSeed(person.getId().toString().hashCode());
245                for (int i = 0; i < 5; i++) random.nextDouble();
246                
247                /*
248                 * Limit scatter factor to +/- 4 times the standard deviation.
249                 * Therefore, the scatter factor is 0.224 ... 1.777
250                 */
251                double scatterLimit = 4 * weidmannScatterStandardDeviation;
252                double scatterSpeed = random.nextGaussian() * weidmannScatterStandardDeviation + weidmannReferenceWalkSpeed;
253                if (scatterSpeed < weidmannReferenceWalkSpeed - scatterLimit) {
254                        scatterSpeed = weidmannReferenceWalkSpeed - scatterLimit;
255                } else if (scatterSpeed > weidmannReferenceWalkSpeed + scatterLimit) {
256                        scatterSpeed = weidmannReferenceWalkSpeed + scatterLimit;
257                }
258                scatterFactor = scatterSpeed / weidmannReferenceWalkSpeed;
259                
260                if (person instanceof Person) {
261                        Person p = person;
262                        
263                        // get gender factor
264                        if (PersonUtils.getSex(p) == null) {
265                                if (genderWarnCount.get() < 10) {
266                                        incGenderWarnCount("Person's gender is not defined. Ignoring gender dependent walk speed factor.");
267                                }
268                        } else if (PersonUtils.getSex(p).equals("m")) genderFactor = maleScaleFactor;
269                        else if (PersonUtils.getSex(p).equals("f")) genderFactor = femaleScaleFactor;
270                        else {
271                                if (genderWarnCount.get() < 10) {
272                                        incGenderWarnCount("Person's gender is not defined. Ignoring gender dependent walk speed factor.");
273                                }
274                        }
275                        
276                        Integer age = PersonUtils.getAge(p);
277                        
278                        if (age == null) {
279                                if (ageWarnCount.get() < 10) {
280                                        incAgeWarnCount("Person's age is not defined. Ignoring age dependent walk speed factor.");
281                                }
282                        }
283                        else if (age < 0) {
284                                if (ageWarnCount.get() < 10) {
285                                        incAgeWarnCount("Person's age is out of expected range (0 .. 100). Founde age of " + age + ". Use 0 instead.");
286                                }
287                                ageFactor = ageFactors[0];
288                        } else if (age > 100) {
289                                if (ageWarnCount.get() < 10) {
290                                        incAgeWarnCount("Person's age is out of expected range (0 .. 100). Founde age of " + age + ". Use 100 instead.");
291                                }
292                                ageFactor = ageFactors[100];
293                        } else {
294                                ageFactor = ageFactors[PersonUtils.getAge(p)];
295                        }
296                }
297                
298                double personFactor = scatterFactor * ageFactor * genderFactor;
299                this.personCache.set(person);
300                this.personFactorCache.set(personFactor);
301                this.personWalkSpeedCache.set(this.referenceWalkSpeed * personFactor);
302                
303                this.personFactors.put(person.getId(), personFactor);
304        }
