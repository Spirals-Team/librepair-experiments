021package org.matsim.contrib.multimodal.router.util;
022
023import org.apache.log4j.Logger;
024import org.matsim.api.core.v01.Id;
025import org.matsim.api.core.v01.TransportMode;
026import org.matsim.api.core.v01.network.Link;
027import org.matsim.contrib.multimodal.config.MultiModalConfigGroup;
028import org.matsim.core.api.internal.MatsimFactory;
029import org.matsim.core.config.Config;
030import org.matsim.core.config.groups.PlansCalcRouteConfigGroup;
031import org.matsim.core.router.util.TravelTime;
032import org.matsim.core.utils.collections.CollectionUtils;
033
034import javax.inject.Provider;
035import java.util.HashMap;
036import java.util.LinkedHashMap;
037import java.util.Map;
038import java.util.Map.Entry;
039import java.util.Set;
040
041public class MultiModalTravelTimeFactory implements MatsimFactory {
042        
043        private static final Logger log = Logger.getLogger(MultiModalTravelTimeFactory.class);
044        
045        private final Map<String, Provider<TravelTime>> factories;
046        private final Map<String, Provider<TravelTime>> additionalFactories;
047        private final Map<Id<Link>, Double> linkSlopes;
048        
049        public MultiModalTravelTimeFactory(Config config) {
050                this(config, null, null);
051        }
052
053        public MultiModalTravelTimeFactory(Config config, Map<Id<Link>, Double> linkSlopes) {
054                this(config, linkSlopes, null);
055        }
056        
057        public MultiModalTravelTimeFactory(Config config, Map<Id<Link>, Double> linkSlopes, Map<String, Provider<TravelTime>> additionalFactories) {
058                this.linkSlopes = linkSlopes;
059                this.factories = new LinkedHashMap<>();
060                this.additionalFactories = additionalFactories;
061                
062                if (this.linkSlopes == null) {
063                        log.warn("No slope information for the links available - travel time will only take agents age and gender into account!");
064                }
065                
066                this.initMultiModalTravelTimeFactories(config);
067        }
068        
069        public Map<String, TravelTime> createTravelTimes() {
070                Map<String, TravelTime> travelTimes = new HashMap<>();
071                
072                for (Entry<String, Provider<TravelTime>> entry : factories.entrySet()) {
073                        travelTimes.put(entry.getKey(), entry.getValue().get());
074                }
075                
076                return travelTimes;
077        }
078        
079        private void initMultiModalTravelTimeFactories(Config config) {
080                
081                PlansCalcRouteConfigGroup plansCalcRouteConfigGroup = config.plansCalcRoute();
082        MultiModalConfigGroup multiModalConfigGroup = (MultiModalConfigGroup) config.getModule(MultiModalConfigGroup.GROUP_NAME);
083        Set<String> simulatedModes = CollectionUtils.stringToSet(multiModalConfigGroup.getSimulatedModes());
084                
085                for (String mode : simulatedModes) {            
086                        if (mode.equals(TransportMode.walk)) {
087                                Provider<TravelTime> factory = new WalkTravelTimeFactory(plansCalcRouteConfigGroup, linkSlopes);
088                                this.factories.put(mode, factory);
089                        } else if (mode.equals(TransportMode.transit_walk)) {
090                                Provider<TravelTime> factory = new TransitWalkTravelTimeFactory(plansCalcRouteConfigGroup, linkSlopes);
091                                this.factories.put(mode, factory);
092                        } else if (mode.equals(TransportMode.bike)) {
093                                Provider<TravelTime> factory = new BikeTravelTimeFactory(plansCalcRouteConfigGroup, linkSlopes);
094                                this.factories.put(mode, factory);
095                        } else {
096                                Provider<TravelTime> factory = getTravelTimeFactory(mode);
097                                
098                                if (factory == null) {
099                                        log.warn("Mode " + mode + " is not supported! " + 
100                                                        "Use a constructor where you provide the travel time objects. " +
101                                                        "Using a UnknownTravelTime calculator based on constant speed." +
102                                                        "Agent specific attributes are not taken into account!");
103                                        factory = new UnknownTravelTimeFactory(mode, plansCalcRouteConfigGroup);
104                                        this.factories.put(mode, factory);
105                                } else {
106                                        log.info("Found additional travel time factory from type " + factory.getClass().toString() +
107                                                        " for mode " + mode + ".");
108                                        this.factories.put(mode, factory);
109                                }
110                        }
111                }
112        }
113        
114        private Provider<TravelTime> getTravelTimeFactory(String mode) {
115                if (additionalFactories != null) return this.additionalFactories.get(mode);
116                else return null;
117        }
118}
