021package org.matsim.contrib.multimodal.router.util;
022
023import org.matsim.api.core.v01.Id;
024import org.matsim.api.core.v01.TransportMode;
025import org.matsim.api.core.v01.network.Link;
026import org.matsim.api.core.v01.population.Person;
027import org.matsim.core.config.groups.PlansCalcRouteConfigGroup;
028import org.matsim.vehicles.Vehicle;
029
030import java.util.Map;
031
032/**
033 * Data to calculate a person's travel time on a link is taken from:
034 * John Parkin and Jonathon Rotheram (2010): Design speeds and acceleration characteristics
035 * of bicycle traffic for use in planning and appraisal.
036 * 
037 * The default downhill speed is limited to 35 km/h. The value is scaled for each person
038 * based on age, gender and scatter.
039 * 
040 * On steep links, a person is faster by walking than by biking. For such links, the
041 * walk travel time is returned.
042 */
043public class BikeTravelTime extends WalkTravelTime {
044        
045        /*
046         * Cache variables for each thread accessing the object separately.
047         */
048        /*package*/ private final ThreadLocal<Double> personBikeSpeedCache;
049        /*package*/ private final ThreadLocal<Double> maxPersonBikeSpeedCache;
050        /*package*/ private final ThreadLocal<Double> personUphillFactorCache;
051        /*package*/ private final ThreadLocal<Double> personDownhillFactorCache;
052        
053        /*
054         * If the set reference speed does not match the default reference speed,
055         * the up- and downhill factors are scaled accordingly. 
056         */
057        private final double defaultReferenceSpeed = 6.01;      // [m/s]
058        
059        private final double referenceBikeSpeed;                // 6.01 according to Prakin and Rotheram
060        private final double maxBikeSpeed = 35.0 / 3.6; // according to Parkin and Rotheram
061        
062        private final double downhillFactor = 0.2379;   // 0%..-15%
063        private final double uphillFactor = -0.4002;    // 0%..12%
064        
065        public BikeTravelTime(PlansCalcRouteConfigGroup plansCalcGroup, Map<Id<Link>, Double> linkSlopes) {
066                super(plansCalcGroup, linkSlopes);
067                
068                this.referenceBikeSpeed = plansCalcGroup.getTeleportedModeSpeeds().get(TransportMode.bike);
069                
070                this.personBikeSpeedCache = new ThreadLocal<>();
071                this.maxPersonBikeSpeedCache = new ThreadLocal<>();
072                this.personUphillFactorCache = new ThreadLocal<>();
073                this.personDownhillFactorCache = new ThreadLocal<>();
074        }
075        
076        public BikeTravelTime(PlansCalcRouteConfigGroup plansCalcGroup) {
077                this(plansCalcGroup, null);
078        }
079
080
081        @Override
082        public double getLinkTravelTime(Link link, double time, Person person, Vehicle vehicle) {
083                setPerson(person);
084                double walkTravelTime = super.getLinkTravelTime(link, time, person, vehicle);
085                
086                double slope = super.getSlope(link);
087                double slopeShift = getSlopeShift(slope);
088                
089//              double linkSpeed = this.personBikeSpeed + slopeShift;
090                double linkSpeed = this.personBikeSpeedCache.get() + slopeShift;
091                
092                // limit min and max speed
093//              if (linkSpeed > maxPersonBikeSpeed) linkSpeed = maxPersonBikeSpeed;
094                if (linkSpeed > this.maxPersonBikeSpeedCache.get()) linkSpeed = this.maxPersonBikeSpeedCache.get();
095                else if (linkSpeed < 0.0) linkSpeed = Double.MIN_VALUE;
096                
097                double bikeTravelTime = link.getLength() / linkSpeed;
098                return Math.min(walkTravelTime, bikeTravelTime);
099        }
100        
101        /*
102         * It is assumed, that there is a linear relation between speed and slope.
103         * The returned values shifts the speed on a flat area.
104         * E.g. speed(10%) = speed(0%) + slopeShift(10%)
105         */
106        /*package*/ double getSlopeShift(double slope) {
107                if (slope > 0.0) return this.personUphillFactorCache.get() * slope;
108                else return this.personDownhillFactorCache.get() * slope;
109        }
110        
111        @Override
112    void setPerson(Person person) {
113                /* 
114                 * Only recalculate the person's speed factor if the person has 
115                 * changed. This check has to be performed before super.setPerson(...)
116                 * is called because there the personId is already updated!
117                 */
118                /* 
119                 * Only recalculate the person's walk speed factor if
120                 * the person has changed and is not in the map.
121                 */
122                if (this.personCache.get() != null && person.getId().equals(this.personCache.get().getId())) return;
123                
124                /*
125                 * If the person's walk speed is already in the map, use that value.
126                 * Otherwise calculate it and add it to the map.
127                 */
128                Double value = this.personFactors.get(person.getId());
129                if (value != null) {
130                        double personFactor = value;
131//                      this.personCache.set(person);   // set in the super-class
132//                      this.personFactorCache.set(personFactor);       // set in the super-class
133                        this.personBikeSpeedCache.set(this.referenceBikeSpeed * personFactor);
134                        this.maxPersonBikeSpeedCache.set(maxBikeSpeed * personFactor);
135                        this.personUphillFactorCache.set(uphillFactor * personFactor * referenceBikeSpeed / defaultReferenceSpeed);
136                        this.personDownhillFactorCache.set(downhillFactor * personFactor * referenceBikeSpeed / defaultReferenceSpeed);
137                        super.setPerson(person);
138                        return;
139                }
140                
141                super.setPerson(person);
142                
143//              this.personBikeSpeed = this.referenceBikeSpeed * this.personFactor;
144//              this.maxPersonBikeSpeed = maxBikeSpeed * personFactor;
145//              this.personUphillFactor = uphillFactor * personFactor * referenceBikeSpeed / defaultReferenceSpeed;
146//              this.personDownhillFactor = downhillFactor * personFactor * referenceBikeSpeed / defaultReferenceSpeed;
147
148                double personFactor = this.personFactorCache.get();
149                this.personBikeSpeedCache.set(this.referenceBikeSpeed * personFactor);
150                this.maxPersonBikeSpeedCache.set(maxBikeSpeed * personFactor);
151                this.personUphillFactorCache.set(uphillFactor * personFactor * referenceBikeSpeed / defaultReferenceSpeed);
152                this.personDownhillFactorCache.set(downhillFactor * personFactor * referenceBikeSpeed / defaultReferenceSpeed);
153        }
154}
