023package org.matsim.contrib.multimodal.simengine;
024
025import java.util.Map;
026
027import org.matsim.contrib.multimodal.config.MultiModalConfigGroup;
028import org.matsim.core.config.Config;
029import org.matsim.core.config.ConfigUtils;
030import org.matsim.core.mobsim.qsim.QSim;
031import org.matsim.core.router.util.TravelTime;
032
033public class MultiModalQSimModule {
034
035    private final Config config;
036    private final Map<String, TravelTime> multiModalTravelTimes;
037
038    public MultiModalQSimModule(Config config, Map<String, TravelTime> multiModalTravelTimes) {
039        this.config = config;
040        this.multiModalTravelTimes = multiModalTravelTimes;
041    }
042
043    public void configure(QSim qSim) {
044        MultiModalConfigGroup multiModalConfigGroup = ConfigUtils.addOrGetModule(this.config, MultiModalConfigGroup.GROUP_NAME, MultiModalConfigGroup.class);
045        MultiModalSimEngine multiModalEngine = new MultiModalSimEngine(this.multiModalTravelTimes, multiModalConfigGroup);
046        qSim.addMobsimEngine(multiModalEngine);
047        qSim.addDepartureHandler(new MultiModalDepartureHandler(multiModalEngine, multiModalConfigGroup));
048    }
049}
