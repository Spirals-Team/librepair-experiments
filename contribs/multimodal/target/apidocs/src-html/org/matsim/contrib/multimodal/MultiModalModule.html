023package org.matsim.contrib.multimodal;
024
025import com.google.inject.Key;
026import com.google.inject.name.Names;
027import org.apache.log4j.Logger;
028import org.matsim.api.core.v01.Id;
029import org.matsim.api.core.v01.TransportMode;
030import org.matsim.api.core.v01.network.Link;
031import org.matsim.contrib.multimodal.config.MultiModalConfigGroup;
032import org.matsim.contrib.multimodal.router.util.BikeTravelTimeFactory;
033import org.matsim.contrib.multimodal.router.util.UnknownTravelTimeFactory;
034import org.matsim.contrib.multimodal.router.util.WalkTravelTimeFactory;
035import org.matsim.core.config.groups.PlanCalcScoreConfigGroup;
036import org.matsim.core.config.groups.PlansCalcRouteConfigGroup;
037import org.matsim.core.controler.AbstractModule;
038import org.matsim.core.router.NetworkRoutingProvider;
039import org.matsim.core.router.RoutingModule;
040import org.matsim.core.router.costcalculators.RandomizingTimeDistanceTravelDisutilityFactory;
041import org.matsim.core.router.util.TravelTime;
042import org.matsim.core.utils.collections.CollectionUtils;
043
044import javax.inject.Provider;
045import java.util.LinkedHashMap;
046import java.util.Map;
047import java.util.Set;
048
049public class MultiModalModule extends AbstractModule {
050
051    private static final Logger log = Logger.getLogger(MultiModalModule.class);
052
053    private final Map<String, Provider<TravelTime>> additionalTravelTimeFactories = new LinkedHashMap<>();
054
055    private Map<Id<Link>, Double> linkSlopes;
056
057    @Override
058    public void install() {
059        PlansCalcRouteConfigGroup plansCalcRouteConfigGroup = getConfig().plansCalcRoute();
060        PlanCalcScoreConfigGroup cnScoringGroup = getConfig().planCalcScore();
061        MultiModalConfigGroup multiModalConfigGroup = (MultiModalConfigGroup) getConfig().getModule(MultiModalConfigGroup.GROUP_NAME);
062        Set<String> simulatedModes = CollectionUtils.stringToSet(multiModalConfigGroup.getSimulatedModes());
063        for (String mode : simulatedModes) {
064            if (mode.equals(TransportMode.walk)) {
065                Provider<TravelTime> factory = new WalkTravelTimeFactory(plansCalcRouteConfigGroup, linkSlopes);
066                addTravelTimeBinding(mode).toProvider(factory);
067                addTravelDisutilityFactoryBinding(mode).toInstance( new RandomizingTimeDistanceTravelDisutilityFactory( mode, cnScoringGroup ) );
068                addRoutingModuleBinding(mode).toProvider(new NetworkRoutingProvider(mode));
069            } else if (mode.equals(TransportMode.transit_walk)) {
070//                Provider<TravelTime> factory = new TransitWalkTravelTimeFactory(plansCalcRouteConfigGroup, linkSlopes);
071//                addTravelTimeBinding(mode).toProvider(factory);
072//                addTravelDisutilityFactoryBinding(mode).toInstance( new RandomizingTimeDistanceTravelDisutility.Builder( mode ) );
073                addRoutingModuleBinding(mode).to(Key.get(RoutingModule.class, Names.named(TransportMode.walk)));
074            } else if (mode.equals(TransportMode.bike)) {
075                Provider<TravelTime> factory = new BikeTravelTimeFactory(plansCalcRouteConfigGroup, linkSlopes);
076                addTravelTimeBinding(mode).toProvider(factory);
077                addTravelDisutilityFactoryBinding(mode).toInstance( new RandomizingTimeDistanceTravelDisutilityFactory( mode, cnScoringGroup ) );
078                addRoutingModuleBinding(mode).toProvider(new NetworkRoutingProvider(mode));
079            } else {
080                Provider<TravelTime> factory = additionalTravelTimeFactories.get(mode);
081                if (factory == null) {
082                    log.warn("Mode " + mode + " is not supported! " +
083                            "Use a constructor where you provide the travel time objects. " +
084                            "Using an UnknownTravelTime calculator based on constant speed." +
085                            "Agent specific attributes are not taken into account!");
086                    factory = new UnknownTravelTimeFactory(mode, plansCalcRouteConfigGroup);
087                } else {
088                    log.info("Found additional travel time factory from type " + factory.getClass().toString() +
089                            " for mode " + mode + ".");
090                }
091                addTravelTimeBinding(mode).toProvider(factory);
092                addTravelDisutilityFactoryBinding(mode).toInstance(new RandomizingTimeDistanceTravelDisutilityFactory( mode, cnScoringGroup ));
093                addRoutingModuleBinding(mode).toProvider(new NetworkRoutingProvider(mode));
094            }
095        }
096        addControlerListenerBinding().to(MultiModalControlerListener.class);
097        bindMobsim().toProvider(MultimodalQSimFactory.class);
098    }
099
100    public void setLinkSlopes(Map<Id<Link>, Double> linkSlopes) {
101        this.linkSlopes = linkSlopes;
102    }
103
104    public void addAdditionalTravelTimeFactory(String mode, Provider<TravelTime> travelTimeFactory) {
105        this.additionalTravelTimeFactories.put(mode, travelTimeFactory);
106    }
107
108
109}
