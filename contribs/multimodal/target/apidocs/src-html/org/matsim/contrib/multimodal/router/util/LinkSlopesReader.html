021package org.matsim.contrib.multimodal.router.util;
022
023import org.apache.log4j.Logger;
024import org.matsim.api.core.v01.Id;
025import org.matsim.api.core.v01.network.Link;
026import org.matsim.api.core.v01.network.Network;
027import org.matsim.contrib.multimodal.config.MultiModalConfigGroup;
028import org.matsim.utils.objectattributes.ObjectAttributes;
029import org.matsim.utils.objectattributes.ObjectAttributesXmlReader;
030
031import java.util.HashMap;
032import java.util.Map;
033
034public class LinkSlopesReader {
035
036        private static final Logger log = Logger.getLogger(LinkSlopesReader.class);
037        
038        private static final String ATTRIBUTE_NAME = "slope";
039        
040        /**
041         * If height information for a network is available, the links' slopes
042         * is returned in a Map<LinkId, Slope in %>.
043         *  
044         * It is assumed that the heightInformation ObjectAttributes contain a field
045         * "slope" for each link Id in the network.
046         * 
047         * @return Map<LinkId, Slope in %> or null, if no slopes file has been set in the config group.
048         */
049        public Map<Id<Link>, Double> getLinkSlopes(MultiModalConfigGroup configGroup, Network network) {
050                
051                ObjectAttributes slopeInformation = this.getSlopeInformation(configGroup);
052                
053                if (slopeInformation == null) return null;
054                
055                Map<Id<Link>, Double> linkSlopes = this.getLinkSlopes(network, slopeInformation);
056                
057                int found = linkSlopes.size();
058                int total = network.getLinks().size();
059                log.info("Found slope information for " + found + " of " + total + " links.");
060                
061                return linkSlopes;
062        }
063        
064        /**
065         * Reads height information from the file specified in the config group
066         * and returns them as ObjectAttributes.
067         */
068        private ObjectAttributes getSlopeInformation(MultiModalConfigGroup configGroup) {
069                
070                String slopeInformationFile = configGroup.getSlopeInformationFile();
071                if (slopeInformationFile != null) {
072                        ObjectAttributes objectAttributes = new ObjectAttributes();
073                        log.info("Loading slope information from " + slopeInformationFile);
074                        new ObjectAttributesXmlReader(objectAttributes).readFile(slopeInformationFile);
075                        return objectAttributes;
076                } else {
077                        log.warn("No slope information file specified in the multi modal config group!");
078                        return null;
079                }
080        }
081        
082        private Map<Id<Link>, Double> getLinkSlopes(Network network, ObjectAttributes slopeInformation) {
083                
084                Map<Id<Link>, Double> linkSlopes = new HashMap<>();
085                
086                for (Id<Link> linkId : network.getLinks().keySet()) {
087                        Object slope = slopeInformation.getAttribute(linkId.toString(), ATTRIBUTE_NAME);
088                        if (slope != null) linkSlopes.put(linkId, Double.valueOf(slope.toString()));
089                }
090                return linkSlopes;
091        }
092}
