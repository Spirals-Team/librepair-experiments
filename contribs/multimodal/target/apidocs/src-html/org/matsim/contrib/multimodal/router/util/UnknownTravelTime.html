021package org.matsim.contrib.multimodal.router.util;
022
023import org.matsim.api.core.v01.network.Link;
024import org.matsim.api.core.v01.population.Person;
025import org.matsim.core.config.groups.PlansCalcRouteConfigGroup;
026import org.matsim.core.router.util.TravelTime;
027import org.matsim.vehicles.Vehicle;
028
029/**
030 * Travel time calculator for unknown modes. Agents move with constant speed. No
031 * agent specific parameters are taken into account.
032 */
033public class UnknownTravelTime implements TravelTime {
034        
035        private final boolean speed;
036        private final boolean speedFactor;
037        private final double value;
038                
039        public UnknownTravelTime(String mode, PlansCalcRouteConfigGroup plansCalcRouteConfigGroup) {
040                
041                Double speed = plansCalcRouteConfigGroup.getTeleportedModeSpeeds().get(mode);
042                Double speedFactor = plansCalcRouteConfigGroup.getTeleportedModeFreespeedFactors().get(mode);
043                
044                if (speed != null && speedFactor != null) {
045                        throw new RuntimeException("Speed as well as speed factor was found for mode " + mode + 
046                                        "!  Don't know which should be used. Aborting.");
047                } else if (speed == null && speedFactor == null) {
048                        throw new RuntimeException("Neither speed nor speed factor was found for mode " + mode + "! Aborting.");
049                } else if (speed != null) {
050                        this.value = speed;
051                        this.speed = true;
052                        this.speedFactor = false;
053                } else {
054                        this.value = speedFactor;
055                        this.speed = false;
056                        this.speedFactor = true;
057                }
058        }
059        
060        @Override
061        public double getLinkTravelTime(Link link, double time, Person person, Vehicle vehicle) {
062                if (speed) return link.getLength() / this.value;                        
063                else if (speedFactor) return (link.getLength() / link.getFreespeed()) * this.value;
064                else throw new RuntimeException("Neither speed nor speed factor was found! Aborting.");
065        }
066}
