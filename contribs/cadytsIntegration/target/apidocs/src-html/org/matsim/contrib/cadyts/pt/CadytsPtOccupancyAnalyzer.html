050/**
051 * Collects occupancy data of transit-line stations
052 * <p></p>
053 * This is probably similar to code elsewhere.  However, it makes some sense to keep this here since the correct workings of cadyts 
054 * (obviously) depends on the fact that the counts are actually what it thinks, and so it makes sense to decouple this from the upstream
055 * counting method and leave it here. kai, sep'13 
056 */
057public class CadytsPtOccupancyAnalyzer implements CadytsPtOccupancyAnalyzerI {
058
059        private final int timeBinSize, maxSlotIndex;
060        private final double maxTime;
061        private Map<Id<TransitStopFacility>, int[]> occupancies; // Map< stopFacilityId,value[]>
062        private final Map<Id<Vehicle>, Id<TransitStopFacility>> vehStops = new HashMap<>(); // Map< vehId,stopFacilityId>
063        private final Map<Id<Vehicle>, Integer> vehPassengers = new HashMap<>(); // Map<vehId,passengersNo.in Veh>
064        private StringBuffer occupancyRecord = new StringBuffer("time\tvehId\tStopId\tno.ofPassengersInVeh\n");
065        private final Set<Id> analyzedTransitDrivers = new HashSet<>();
066        private final Set<Id> analyzedTransitVehicles = new HashSet<>();
067        private final Set<Id<TransitLine>> calibratedLines;
068
069        @Inject
070        public CadytsPtOccupancyAnalyzer( Config config ) {
071                CadytsConfigGroup ccc = ConfigUtils.addOrGetModule(config, CadytsConfigGroup.GROUP_NAME, CadytsConfigGroup.class ) ;
072
073                this.calibratedLines = toTransitLineIdSet( ccc.getCalibratedItems() ) ;
074                this.timeBinSize = ccc.getTimeBinSize() ;
075
076                this.maxTime = Time.MIDNIGHT-1; //24 * 3600 - 1;
077                // (yy not completely clear if it might be better to use 24*this.timeBimSize, but it is overall not so great
078                // to have this hardcoded.  kai/manuel, jul'12)
079
080                this.maxSlotIndex = ((int) this.maxTime) / this.timeBinSize + 1;
081                this.occupancies = new HashMap<>();
082        }
083
084        @Override
085        public void reset(final int iteration) {
086                this.occupancies.clear();
087                this.vehStops.clear();
088                this.vehPassengers.clear();
089                this.occupancyRecord = new StringBuffer("time\tvehId\tStopId\tno.ofPassengersInVeh\n");
090                this.analyzedTransitDrivers.clear();
091                this.analyzedTransitVehicles.clear();
092        }
093
094        @Override
095        public void handleEvent(final TransitDriverStartsEvent event) {
096                if (this.calibratedLines.contains(event.getTransitLineId())) {
097                        this.analyzedTransitDrivers.add(event.getDriverId());
098                        this.analyzedTransitVehicles.add(event.getVehicleId());
099                }
100        }
101
102        @Override
103        public void handleEvent(final PersonEntersVehicleEvent event) {
104                if (this.analyzedTransitDrivers.contains(event.getPersonId()) || !this.analyzedTransitVehicles.contains(event.getVehicleId())) {
105                        return; // ignore transit drivers or persons entering non-(analyzed-)transit vehicles
106                }
107
108                // ------------------veh_passenger- (for occupancy)-----------------
109                Id<Vehicle> vehId = event.getVehicleId();
110                Id<TransitStopFacility> stopId = this.vehStops.get(vehId);
111                double time = event.getTime();
112                Integer nPassengers = this.vehPassengers.get(vehId);
113                this.vehPassengers.put(vehId, (nPassengers != null) ? (nPassengers + 1) : 1);
114                this.occupancyRecord.append("time :\t").append(time).append(" veh :\t").append(vehId).append(" has Passenger\t").append(this.vehPassengers.get(vehId)).append(" \tat stop :\t").append(stopId).append(" ENTERING PERSON :\t").append(event.getPersonId()).append("\n");
115        }
116
117        @Override
118        public void handleEvent(final PersonLeavesVehicleEvent event) {
119                if (this.analyzedTransitDrivers.contains(event.getPersonId()) || !this.analyzedTransitVehicles.contains(event.getVehicleId())) {
120                        return; // ignore transit drivers or persons entering non-(analyzed-)transit vehicles
121                }
122
123                // ----------------veh_passenger-(for occupancy)--------------------------
124                Id<Vehicle> vehId = event.getVehicleId();
125                double time = event.getTime();
126                Integer nPassengers = this.vehPassengers.get(vehId);
127                if (nPassengers == null) {
128                        throw new RuntimeException("null passenger-No. in vehicle ?");
129                }
130                this.vehPassengers.put(vehId, nPassengers - 1);
131                if (this.vehPassengers.get(vehId) == 0) {
132                        this.vehPassengers.remove(vehId);
133                }
134                Integer passengers = this.vehPassengers.get(vehId);
135                this.occupancyRecord.append("time :\t").append(time).append(" veh :\t").append(vehId).append(" has Passenger\t").append((passengers != null) ? passengers : 0).append("\n");
136        }
137
138        @Override
139        public void handleEvent(final VehicleDepartsAtFacilityEvent event) {
140                Id<Vehicle> vehId = event.getVehicleId();
141                Id<TransitStopFacility> facId = event.getFacilityId();
142
143                // -----------------------occupancy--------------------------------
144                this.vehStops.remove(vehId);
145                int[] occupancyAtStop = this.occupancies.get(facId);
146                if (occupancyAtStop == null) { // no previous departure from this stop, therefore no occupancy
147                                                                                                                                                // record yet. Create this:
148                        occupancyAtStop = new int[this.maxSlotIndex + 1];
149                        this.occupancies.put(facId, occupancyAtStop);
150                }
151
152                Integer noPassengersInVeh = this.vehPassengers.get(vehId);
153
154                if (noPassengersInVeh != null) {
155                        occupancyAtStop[this.getTimeSlotIndex(event.getTime())] += noPassengersInVeh;
156                        this.occupancyRecord.append(event.getTime());
157                        this.occupancyRecord.append("\t");
158                        this.occupancyRecord.append(vehId);
159                        this.occupancyRecord.append("\t");
160                        this.occupancyRecord.append(facId);
161                        this.occupancyRecord.append("\t");
162                        this.occupancyRecord.append(noPassengersInVeh);
163                        this.occupancyRecord.append("\n");
164                }
165        }
166
167        @Override
168        public void handleEvent(final VehicleArrivesAtFacilityEvent event) {
169                Id<TransitStopFacility> stopId = event.getFacilityId();
170
171                this.vehStops.put(event.getVehicleId(), stopId);
172                // (constructing a table with vehId as key, and stopId as value; constructed when veh arrives at
173                // stop; necessary
174                // since personEnters/LeavesVehicle does not carry stop id)
175        }
176
177        private int getTimeSlotIndex(final double time) {
178                if (time > this.maxTime) {
179                        return this.maxSlotIndex;
180                }
181                return ((int) time / this.timeBinSize);
182        }
183
184        /**
185         * @param stopId
186         * @return Array containing the number of passengers in bus after the transfer at the stop
187         *         {@code stopId} per time bin, starting with time bin 0 from 0 seconds to
188         *         (timeBinSize-1)seconds.
189         */
190        @Override
191        public int[] getOccupancyVolumesForStop(final Id<TransitStopFacility> stopId) {
192                return this.occupancies.get(stopId);
193        }
194         /* (non-Javadoc)
195         * @see org.matsim.contrib.cadyts.pt.CadytsPtOccupancyAnalyzerI#getOccupancyVolumeForStopAndTime(org.matsim.api.core.v01.Id, int)
196         */
197        @Override
198        public int getOccupancyVolumeForStopAndTime(final Id<TransitStopFacility> stopId, final int time_s ) {
199                 if ( this.occupancies.get(stopId) != null ) {
200                         int timeBinIndex = getTimeSlotIndex( time_s ) ;
201                         return this.occupancies.get(stopId)[timeBinIndex] ;
202                 } else {
203                         return 0 ;
204                 }
205         }
206
207        public Set<Id<TransitStopFacility>> getOccupancyStopIds() {
208                return this.occupancies.keySet();
209        }
210
211        @Override
212        public void writeResultsForSelectedStopIds(final String filename, final Counts<Link> occupCounts, final Collection<Id<TransitStopFacility>> stopIds) {
213                SimpleWriter writer = new SimpleWriter(filename);
214
215                final String TAB = "\t";
216                final String NL = "\n";
217
218                // write header
219                writer.write("stopId\t");
220                for (int i = 0; i < 24; i++) {
221                        writer.write("oc" + i + "-" + (i + 1) + TAB);
222                }
223                for (int i = 0; i < 24; i++) {
224                        writer.write("scalSim" + i + "-" + (i + 1) + TAB);
225                }
226                writer.write("coordinate\tcsId\n");
227
228                // write content
229                for (Id<TransitStopFacility> stopId : stopIds) {
230                        // get count data
231                        Count count = occupCounts.getCounts().get(Id.create(stopId, Link.class));
232                        if (!occupCounts.getCounts().containsKey(Id.create(stopId, Link.class))) {
233                                continue;
234                        }
235
236                        // get sim-Values
237                        int[] ocuppancy = this.occupancies.get(stopId);
238                        writer.write(stopId.toString() + TAB);
239                        for (int i = 0; i < ocuppancy.length; i++) {
240                                Volume v = count.getVolume(i + 1);
241                                if (v != null) {
242                                        writer.write(v.getValue() + TAB);
243                                } else {
244                                        writer.write("n/a" + TAB);
245                                }
246                        }
247                        for (int anOcuppancy : ocuppancy) {
248                                writer.write((anOcuppancy) + TAB);
249                        }
250                        writer.write(count.getCoord().toString() + TAB + count.getCsLabel() + NL);
251                }
252                writer.write(this.occupancyRecord.toString());
253                writer.close();
254        }
255
256        @Override
257        public String toString() {
258                final StringBuilder stringBuffer2 = new StringBuilder();
259                final String STOPID = "stopId: ";
260                final String VALUES = "; values:";
261                final char TAB = '\t';
262                final char RETURN = '\n';
263
264                for (Id<TransitStopFacility> stopId : this.getOccupancyStopIds()) { // Only occupancy!
265                        StringBuilder stringBuffer = new StringBuilder();
266                        stringBuffer.append(STOPID);
267                        stringBuffer.append(stopId);
268                        stringBuffer.append(VALUES);
269
270                        boolean hasValues = false; // only prints stops with volumes > 0
271                        int[] values = this.getOccupancyVolumesForStop(stopId);
272
273                        for (int value : values) {
274                                hasValues = hasValues || (value > 0);
275
276                                stringBuffer.append(TAB);
277                                stringBuffer.append(value);
278                        }
279                        stringBuffer.append(RETURN);
280                        if (hasValues)
281                                stringBuffer2.append(stringBuffer.toString());
282
283                }
284                return stringBuffer2.toString();
285        }

287        public static Set<Id<TransitLine>> toTransitLineIdSet(Set<String> list) {
288                Set<Id<TransitLine>> converted = new LinkedHashSet<>();
289                
290                for ( String id : list) {
291                        converted.add(Id.create(id, TransitLine.class));
292                }
293                
294                return converted;
295        }
