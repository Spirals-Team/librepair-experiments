020package org.matsim.contrib.cadyts.general;
021
022import org.apache.log4j.Logger;
023import org.matsim.api.core.v01.population.Plan;
024import org.matsim.core.config.Config;
025import org.matsim.core.scoring.SumScoringFunction;
026
027import cadyts.calibrators.analytical.AnalyticalCalibrator;
028
029/**
030 * @author nagel
031 */
032public class CadytsScoring<T> implements SumScoringFunction.BasicScoring {
033        @SuppressWarnings("unused")
034        private static final Logger log = Logger.getLogger(CadytsScoring.class);
035
036        private double score = 0.;
037        private PlansTranslator<T> plansTranslator;
038        private AnalyticalCalibrator<T> matsimCalibrator;
039        private Plan plan;
040        private final double beta;
041        private double weightOfCadytsCorrection = 1.;
042
043        public CadytsScoring(final Plan plan, Config config, final CadytsContextI<T> context) {
044                this.plansTranslator = context.getPlansTranslator();
045                this.matsimCalibrator = context.getCalibrator();
046                this.plan = plan;
047                this.beta = config.planCalcScore().getBrainExpBeta();
048        }
049
050        @Override
051        public void finish() {
052                cadyts.demand.Plan<T> currentPlanSteps = this.plansTranslator.getCadytsPlan(plan);
053                double currentPlanCadytsCorrection = this.matsimCalibrator.calcLinearPlanEffect(currentPlanSteps) / this.beta;
054                this.score = weightOfCadytsCorrection * currentPlanCadytsCorrection;
055        }
056
057        @Override
058        public double getScore() {
059//              if ( score != 0. && score != -450. && score != 450. ) {
060//                      log.warn("cadyts correction=" + score );
061//              }
062                return score;
063        }
064
065        public void setWeightOfCadytsCorrection(double weightOfCadytsCorrection) {
066                this.weightOfCadytsCorrection = weightOfCadytsCorrection;
067        }
068}
