040        public static final String GROUP_NAME = "cadytsCar";
041
042        public static final String REGRESSION_INERTIA = "regressionInertia";
043        public static final String MIN_FLOW_STDDEV = "minFlowStddevVehH";
044        public static final String FREEZE_ITERATION = "freezeIteration";
045        public static final String PREPARATORY_ITERATIONS = "preparatoryIterations";
046        public static final String VARIANCE_SCALE = "varianceScale";
047        public static final String USE_BRUTE_FORCE = "useBruteForce";
048        public static final String START_TIME = "startTime";
049        public static final String END_TIME = "endTime";
050        public static final String WRITE_ANALYSIS_FILE = "writeAnalysisFile";
051        public static final String TIME_BIN_SIZE = "timeBinSize" ;
052
053        private static final String CALIBRATED_LINKS = "calibratedLinks";
054        public static final String CALIBRATED_LINES = "calibratedLines";
055        private static final String CALIBRATED_ITEMS = "calibratedItems";
056
057        private double regressionInertia = MATSimUtilityModificationCalibrator.DEFAULT_REGRESSION_INERTIA;
058        private double minFlowStddev = MATSimUtilityModificationCalibrator.DEFAULT_MIN_FLOW_STDDEV_VEH_H;
059        private int freezeIteration = MATSimUtilityModificationCalibrator.DEFAULT_FREEZE_ITERATION;
060        private int preparatoryIterations = MATSimUtilityModificationCalibrator.DEFAULT_PREPARATORY_ITERATIONS;
061        private double varianceScale = MATSimUtilityModificationCalibrator.DEFAULT_VARIANCE_SCALE;
062        private boolean bruteForce = MATSimUtilityModificationCalibrator.DEFAULT_BRUTE_FORCE;
063        private boolean writeAnalysisFile = false;
064        private int startTime = 0;
065        private int endTime = (int)Time.MIDNIGHT-1;
066        private int timeBinSize = 3600 ;
067
068        private final Set<String> calibratedItems = new HashSet<>();
069
070        public CadytsConfigGroup() {
071                super(GROUP_NAME);
072        }
073
074        @Override
075        public final void addParam(final String paramName, final String value) {
076                // emulate previous behavior of reader (ignore null values at reading). td Apr'15
077                if ( "null".equalsIgnoreCase( value ) ) return;
078
079                if (REGRESSION_INERTIA.equals(paramName)) {
080                        setRegressionInertia(Double.parseDouble(value));
081                } else if (MIN_FLOW_STDDEV.equals(paramName)) {
082                        setMinFlowStddev_vehPerHour(Double.parseDouble(value));
083                } else if (FREEZE_ITERATION.equals(paramName)) {
084                        setFreezeIteration(Integer.parseInt(value));
085                } else if (PREPARATORY_ITERATIONS.equals(paramName)) {
086                        setPreparatoryIterations(Integer.parseInt(value));
087                } else if (VARIANCE_SCALE.equals(paramName)) {
088                        setVarianceScale(Double.parseDouble(value));
089                } else if (USE_BRUTE_FORCE.equals(paramName)) {
090                        setUseBruteForce(Boolean.parseBoolean(value));
091                } else if (WRITE_ANALYSIS_FILE.equals(paramName)) {
092                        setWriteAnalysisFile(Boolean.parseBoolean(value));
093                } else if (START_TIME.equals(paramName)) {
094                        setStartTime((int)Time.parseTime(value));   //The "hh:mm:ss" format is converted here to seconds after midnight
095                } else if (END_TIME.equals(paramName)) {
096                        setEndTime((int)Time.parseTime(value));         //The "hh:mm:ss" format is converted here to seconds after midnight
097                } else if ( TIME_BIN_SIZE.equals(paramName)) {
098                        setTimeBinSize(Integer.parseInt(value)) ;
099                } else if (CALIBRATED_LINKS.equals(paramName) || CALIBRATED_LINES.equals(paramName) || CALIBRATED_ITEMS.equals(paramName) ) {
100                        this.calibratedItems.clear();
101                        for (String linkId : CollectionUtils.stringToArray(value)) {
102                                this.calibratedItems.add( linkId );
103                        }
104                } else {
105                        throw new IllegalArgumentException("Parameter '" + paramName + "' is not supported by config group '" + GROUP_NAME + "'.");
106                }
107        }
108
109        @Override
110        public final Map<String, String> getComments() {
111                Map<String, String> comments = super.getComments();
112
113                comments.put(CALIBRATED_ITEMS, "Comma-separated list of items with counts  to be calibrated.");
114                comments.put(START_TIME, "The first second of the day to be used for calibration.  hh:mm:ss format");
115                comments.put(END_TIME, "The last second of the day to be used for calibration.  hh:mm:ss format");
116                comments.put(TIME_BIN_SIZE, "Length of time bin for which counts are aggregated.  IN SECONDS!!!!  Default is 3600.") ;
117
118                return comments;
119        }
120
121        @Override
122        public final String getValue(final String param_name) {
123                throw new UnsupportedOperationException("Use getters for accessing values!");
124        }
125
126        @Override
127        public final Map<String, String> getParams() {
128                Map<String, String> params = super.getParams();
129
130                params.put(REGRESSION_INERTIA, Double.toString(getRegressionInertia()));
131                params.put(MIN_FLOW_STDDEV, Double.toString(getMinFlowStddev_vehPerHour()));
132                params.put(FREEZE_ITERATION, Integer.toString(getFreezeIteration()));
133                params.put(PREPARATORY_ITERATIONS, Integer.toString(getPreparatoryIterations()));
134                params.put(VARIANCE_SCALE, Double.toString(getVarianceScale()));
135                params.put(USE_BRUTE_FORCE, Boolean.toString(useBruteForce()));
136                params.put(WRITE_ANALYSIS_FILE, Boolean.toString(isWriteAnalysisFile()));
137                params.put(START_TIME, Integer.toString(getStartTime()));
138                params.put(END_TIME, Integer.toString(getEndTime()));
139                params.put(CALIBRATED_ITEMS, CollectionUtils.setToString(this.calibratedItems));
140                params.put(TIME_BIN_SIZE, Integer.toString(getTimeBinSize())) ;
141
142                return params;
143        }
