001package org.matsim.contrib.cadyts.car;
002
003
004import com.google.inject.Key;
005import com.google.inject.TypeLiteral;
006import com.google.inject.name.Names;
007
008import org.matsim.api.core.v01.network.Link;
009import org.matsim.core.config.Config;
010import org.matsim.core.config.groups.CountsConfigGroup;
011import org.matsim.core.controler.AbstractModule;
012import org.matsim.counts.Counts;
013import org.matsim.counts.MatsimCountsReader;
014
015import javax.inject.Inject;
016import javax.inject.Provider;
017import javax.inject.Singleton;
018
019public class CadytsCarModule extends AbstractModule {
020
021        private final Counts<Link> calibrationCounts;
022
023        public CadytsCarModule() {
024                this.calibrationCounts = null;
025        }
026
027        public CadytsCarModule(Counts<Link> calibrationCounts) {
028                this.calibrationCounts = calibrationCounts;
029        }
030
031        @Override
032        public void install() {
033                if (calibrationCounts != null) {
034                        bind(Key.get(new TypeLiteral<Counts<Link>>(){}, Names.named("calibration"))).toInstance(calibrationCounts);
035                } else {
036                        bind(Key.get(new TypeLiteral<Counts<Link>>(){}, Names.named("calibration"))).toProvider(CalibrationCountsProvider.class).in(Singleton.class);
037                }
038                // In principle this is bind(Counts<Link>).to...  But it wants to keep the option of multiple counts, under different names, open.
039                // I think.  kai, jan'16
040                
041                bind(CadytsContext.class).asEagerSingleton();
042                addControlerListenerBinding().to(CadytsContext.class);
043        }
044
045        private static class CalibrationCountsProvider implements Provider<Counts<Link>> {
046                @Inject CountsConfigGroup config;
047                @Inject Config matsimConfig;
048                @Override
049                public Counts<Link> get() {
050                        Counts<Link> calibrationCounts = new Counts<>();
051                        String occupancyCountsFilename = config.getCountsFileURL(matsimConfig.getContext()).getFile();
052                        new MatsimCountsReader(calibrationCounts).readFile(occupancyCountsFilename);
053                        return calibrationCounts;
054                }
055        }
056}
