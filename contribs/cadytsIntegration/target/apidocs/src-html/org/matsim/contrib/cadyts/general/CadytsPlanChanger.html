020package org.matsim.contrib.cadyts.general;
021
022import org.matsim.api.core.v01.Scenario;
023import org.matsim.api.core.v01.population.HasPlansAndId;
024import org.matsim.api.core.v01.population.Person;
025import org.matsim.api.core.v01.population.Plan;
026import org.matsim.core.gbl.MatsimRandom;
027import org.matsim.core.replanning.selectors.PlanSelector;
028import org.matsim.core.replanning.selectors.RandomPlanSelector;
029
030import java.util.Map;
031
032/**
033 * @author nagel
034 */
035public class CadytsPlanChanger<T> implements PlanSelector<Plan, Person> {
036
037        private final double beta ;
038        private double cadytsWeight = 1.0;
039
040        private CadytsContextI<T> cadytsContext;
041
042        public static final String CADYTS_CORRECTION = "cadytsCorrection";
043
044        public CadytsPlanChanger(Scenario scenario, CadytsContextI<T> cadytsContext) {
045                this.cadytsContext = cadytsContext;
046                this.beta = scenario.getConfig().planCalcScore().getBrainExpBeta() ;
047        }
048        
049        @Override
050        public Plan selectPlan(final HasPlansAndId<Plan, Person> person) {
051                final Plan currentPlan = person.getSelectedPlan();
052                if (person.getPlans().size() <= 1 || currentPlan.getScore() == null) {
053                        return currentPlan;
054                }
055
056                // random plan:
057                Plan otherPlan;
058                do {
059                        otherPlan = new RandomPlanSelector<Plan, Person>().selectPlan((person));
060                } while (otherPlan == currentPlan);
061
062                if (otherPlan.getScore() == null) {
063                        return otherPlan;
064                }
065
066                cadyts.demand.Plan<T> currentPlanSteps = this.cadytsContext.getPlansTranslator().getCadytsPlan(currentPlan);
067                double currentPlanCadytsCorrection = this.cadytsContext.getCalibrator().calcLinearPlanEffect(currentPlanSteps) / this.beta;
068                double currentScore = currentPlan.getScore() + cadytsWeight * currentPlanCadytsCorrection;
069
070                cadyts.demand.Plan<T> otherPlanSteps = this.cadytsContext.getPlansTranslator().getCadytsPlan(otherPlan);
071                double otherPlanCadytsCorrection = this.cadytsContext.getCalibrator().calcLinearPlanEffect(otherPlanSteps) / this.beta;
072                double otherScore = otherPlan.getScore() + cadytsWeight * otherPlanCadytsCorrection;
073
074                Map<String,Object> planAttributes = currentPlan.getCustomAttributes() ;
075                planAttributes.put(CadytsPlanChanger.CADYTS_CORRECTION,currentPlanCadytsCorrection) ;
076
077                Map<String,Object> planAttributesOther = otherPlan.getCustomAttributes() ;
078                planAttributesOther.put(CadytsPlanChanger.CADYTS_CORRECTION,otherPlanCadytsCorrection) ;
079
080                double weight = Math.exp(0.5 * this.beta * (otherScore - currentScore));
081
082                Plan selectedPlan = currentPlan;
083                if (MatsimRandom.getRandom().nextDouble() < 0.01 * weight) {
084                        // as of now, 0.01 is hardcoded (proba to change when both scores are the same)
085                        selectedPlan = otherPlan;
086                }
087                return selectedPlan;
088        }
089
090        public void setCadytsWeight(double cadytsWeight) {
091                this.cadytsWeight = cadytsWeight;
092        }
093}
