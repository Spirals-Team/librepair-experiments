020package org.matsim.contrib.cadyts.pt;
021
022import java.util.Map;
023
024import org.apache.log4j.Logger;
025import org.matsim.api.core.v01.Id;
026import org.matsim.contrib.cadyts.general.CadytsBuilderImpl;
027import org.matsim.contrib.cadyts.general.LookUpItemFromId;
028import org.matsim.core.config.Config;
029import org.matsim.counts.Count;
030import org.matsim.counts.Counts;
031import org.matsim.counts.Volume;
032
033import cadyts.calibrators.analytical.AnalyticalCalibrator;
034import cadyts.measurements.SingleLinkMeasurement;
035
036/**
037 * @author nagel
038 * @author mrieser
039 */
040public final class CadytsBuilderImplGT {
041        private static Logger log = Logger.getLogger( CadytsBuilderImplGT.class ) ;
042
043        public static <T> AnalyticalCalibrator<T> buildCalibratorAndAddMeasurements(final Config config, final Counts<T> occupCounts,
044                                                                                                                                                 LookUpItemFromId<T> lookUp, Class<T> idType ) {
045
046                if (occupCounts.getCounts().size() == 0) {
047                        log.warn("Counts container is empty.");
048                }
049
050                AnalyticalCalibrator<T> matsimCalibrator = CadytsBuilderImpl.buildCalibrator(config);
051
052                //add counts data into calibrator
053                int numberOfAddedMeasurements = 0 ;
054                for (Map.Entry<Id<T>, Count<T>> entry : occupCounts.getCounts().entrySet()) {
055                        // (loop over all counting "items" (usually locations/stations)
056
057                        T item = lookUp.getItem(Id.create(entry.getKey(), idType)) ;
058                        if ( item==null ) {
059                                throw new RuntimeException("item is null; entry=" + entry + " idType=" + idType ) ;
060                        }
061
062                        double sum = 0; 
063                        for (Volume volume : entry.getValue().getVolumes().values()){
064                                // (loop over the different time slots)
065                                sum += volume.getValue() ;
066                        }
067                        numberOfAddedMeasurements++ ;
068                        matsimCalibrator.addMeasurement(item, 0, 86400, sum, SingleLinkMeasurement.TYPE.COUNT_VEH );
069                }
070
071                if ( numberOfAddedMeasurements==0 ) {
072                        log.warn("No measurements were added.");
073                }
074
075                if ( matsimCalibrator.getProportionalAssignment() ) {
076                        throw new RuntimeException("Gunnar says that this may not work so do not set to true. kai, sep'14") ;
077                }
078                return matsimCalibrator;
079        }
080}
