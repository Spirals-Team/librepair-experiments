021package org.matsim.contrib.cadyts.car;
022
023import java.util.HashSet;
024import java.util.Set;
025
026import javax.inject.Inject;
027
028import org.apache.log4j.Logger;
029import org.matsim.api.core.v01.Id;
030import org.matsim.api.core.v01.Scenario;
031import org.matsim.api.core.v01.TransportMode;
032import org.matsim.api.core.v01.events.LinkLeaveEvent;
033import org.matsim.api.core.v01.events.VehicleLeavesTrafficEvent;
034import org.matsim.api.core.v01.events.VehicleEntersTrafficEvent;
035import org.matsim.api.core.v01.events.handler.LinkLeaveEventHandler;
036import org.matsim.api.core.v01.events.handler.VehicleLeavesTrafficEventHandler;
037import org.matsim.api.core.v01.network.Link;
038import org.matsim.api.core.v01.events.handler.VehicleEntersTrafficEventHandler;
039import org.matsim.api.core.v01.population.Person;
040import org.matsim.api.core.v01.population.Plan;
041import org.matsim.contrib.cadyts.general.CadytsConfigGroup;
042import org.matsim.contrib.cadyts.general.PlansTranslator;
043import org.matsim.core.config.ConfigUtils;
044import org.matsim.core.events.algorithms.Vehicle2DriverEventHandler;
045
046import cadyts.demand.PlanBuilder;
047
048public class PlansTranslatorBasedOnEvents implements PlansTranslator<Link>, LinkLeaveEventHandler, 
049VehicleEntersTrafficEventHandler, VehicleLeavesTrafficEventHandler {
050        
051        private static final Logger log = Logger.getLogger(PlansTranslatorBasedOnEvents.class);
052
053        private final Scenario scenario;
054
055        private Vehicle2DriverEventHandler delegate = new Vehicle2DriverEventHandler();
056        
057        private int iteration = -1;
058
059        // this is _only_ there for output:
060        Set<Plan> plansEverSeen = new HashSet<>();
061
062        private static final String STR_PLANSTEPFACTORY = "planStepFactory";
063        private static final String STR_ITERATION = "iteration";
064
065        private final Set<Id<Link>> calibratedLinks = new HashSet<>() ;
066
067        @Inject
068        PlansTranslatorBasedOnEvents(final Scenario scenario) {
069                this.scenario = scenario;
070                Set<String> abc = ConfigUtils.addOrGetModule(scenario.getConfig(), CadytsConfigGroup.GROUP_NAME, CadytsConfigGroup.class).getCalibratedItems();
071                for ( String str : abc ) {
072                        this.calibratedLinks.add( Id.createLinkId(str) ) ;
073                }
074        }
075
076        private long plansFound = 0;
077        private long plansNotFound = 0;
078
079        @Override
080        public final cadyts.demand.Plan<Link> getCadytsPlan(final Plan plan) {
081                @SuppressWarnings("unchecked")
082                PlanBuilder<Link> planStepFactory = (PlanBuilder<Link>) plan.getCustomAttributes().get(STR_PLANSTEPFACTORY);
083                if (planStepFactory == null) {
084                        this.plansNotFound++;
085                        return null;
086                }
087                this.plansFound++;
088                final cadyts.demand.Plan<Link> planSteps = planStepFactory.getResult();
089                return planSteps;
090        }
091
092        @Override
093        public void reset(final int iteration) {
094                this.iteration = iteration;
095
096                log.warn("found " + this.plansFound + " out of " + (this.plansFound + this.plansNotFound) + " ("
097                                + (100. * this.plansFound / (this.plansFound + this.plansNotFound)) + "%)");
098                log.warn("(above values may both be at zero for a couple of iterations if multiple plans per agent all have no score)");
099                
100                delegate.reset(iteration);
101        }
102
103        @Override
104        public void handleEvent(VehicleEntersTrafficEvent event) {
105                if (event.getNetworkMode().equals(TransportMode.car))
106                        delegate.handleEvent(event);
107        }
108        
109        @Override
110        public void handleEvent(VehicleLeavesTrafficEvent event) {
111                delegate.handleEvent(event);
112        }
113        
114        @Override
115        public void handleEvent(LinkLeaveEvent event) {
116                
117                Id<Person> driverId = delegate.getDriverOfVehicle(event.getVehicleId());
118                
119                // if it is not a car, ignore the event
120                if (driverId == null) 
121                        return;
122                
123                // if only a subset of links is calibrated but the link is not contained, ignore the event
124                if (!calibratedLinks.contains(event.getLinkId())) 
125                        return;
126                
127                // get the "Person" behind the id:
128                Person person = this.scenario.getPopulation().getPersons().get(driverId);
129                
130                // get the selected plan:
131                Plan selectedPlan = person.getSelectedPlan();
132                
133                // get the planStepFactory for the plan (or create one):
134                PlanBuilder<Link> tmpPlanStepFactory = getPlanStepFactoryForPlan(selectedPlan);
135                
136                if (tmpPlanStepFactory != null) {
137                                                
138                        Link link = this.scenario.getNetwork().getLinks().get(event.getLinkId());
139                                        
140                        // add the "turn" to the planStepfactory
141                        tmpPlanStepFactory.addTurn(link, (int) event.getTime());
142                }
143        }
144
145        // ###################################################################################
146        // only private functions below here (low level functionality)
147
148        private PlanBuilder<Link> getPlanStepFactoryForPlan(final Plan selectedPlan) {
149                PlanBuilder<Link> planStepFactory = null;
150
151                planStepFactory = (PlanBuilder<Link>) selectedPlan.getCustomAttributes().get(STR_PLANSTEPFACTORY);
152                Integer factoryIteration = (Integer) selectedPlan.getCustomAttributes().get(STR_ITERATION);
153                if (planStepFactory == null || factoryIteration == null || factoryIteration != this.iteration) {
154                        // attach the iteration number to the plan:
155                        selectedPlan.getCustomAttributes().put(STR_ITERATION, this.iteration);
156
157                        // construct a new PlanBulder and attach it to the plan:
158                        planStepFactory = new PlanBuilder<Link>();
159                        selectedPlan.getCustomAttributes().put(STR_PLANSTEPFACTORY, planStepFactory);
160
161                        // memorize the plan as being seen:
162                        this.plansEverSeen.add(selectedPlan);
163                }
164
165                return planStepFactory;
166        }
167
168}
