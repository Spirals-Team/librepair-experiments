020package org.matsim.contrib.cadyts.utils;
021
022import java.util.Map;
023import java.util.TreeMap;
024
025import org.apache.log4j.Logger;
026
027import cadyts.utilities.io.tabularFileParser.TabularFileHandler;
028
029/**Reader for the calibration-stat files produced after calibration.**/
030public class CalibrationStatReader implements TabularFileHandler {
031
032        private static final Logger log = Logger.getLogger(CalibrationStatReader.class);
033
034        private static final String[] HEADER = {"count-ll", "count-ll-pred-err", "p2p-ll", "total-ll", "link-lambda-avg", "link-lambda-stddev", "link-lambda-min", "link-lambda-max", "plan-lambda-avg", "plan-lambda-stddev", "plan-lambda-min", "plan-lambda-max", "replan-count"};
035        private int rowNum =0;
036        private final Map <Integer, StatisticsData> statDataMap = new TreeMap <Integer, StatisticsData>();
037
038        public CalibrationStatReader(){
039        }
040
041        @Override
042        public void startDocument() {
043                log.info("reading calibration-stats file...");
044        }
045
046        @Override
047        public void startRow(final String[] row)  {
048                String colArray[] = row[0].split("\t");
049                int col= 0;
050                if (this.rowNum > 0) {
051                        StatisticsData sData = new StatisticsData();
052                        for (String s : colArray) {
053                                switch (col) {
054                                case 0:  sData.setCount_ll(s); break;
055                            case 1:  sData.setCount_ll_pred_err(s); break;
056                            case 2:  sData.setP2p_ll(s); break;
057                            case 3:  sData.setTotal_ll(s); break;
058                            case 4:  sData.setLink_lambda_avg(s); break;
059                            case 5:  sData.setLink_lambda_stddev(s); break;
060                            case 6:  sData.setLink_lambda_min(s); break;
061                            case 7:  sData.setLink_lambda_max(s); break;
062                            case 8:  sData.setPlan_lambda_avg(s); break;
063                            case 9: sData.setPlan_lambda_stddev(s); break;
064                            case 10: sData.setPlan_lambda_min(s); break;
065                            case 11: sData.setPlan_lambda_max(s); break;
066                            case 12: sData.setReplan_count(s); break;
067                                }//switch
068                                col++;
069                        }
070                        this.statDataMap.put(this.rowNum, sData);
071                }else{
072                        boolean equalsHeader = true;
073                        for (String s : colArray) {
074                                if (!s.equalsIgnoreCase(HEADER[col])){
075                                        equalsHeader = false;
076                                        break;
077                                }
078                                col++;
079                        }
080                        if (!equalsHeader) {
081                                log.warn("the structure does not match. The header should be:  ");
082                                for (String g : HEADER) {
083                                        log.warn(g + "\t");
084                                }
085                        }
086                }
087                this.rowNum++;
088        }
089
090        @Override
091        public void endDocument() {
092                log.info("done.");
093        }
094
095        public static class StatisticsData {
096                private String count_ll;
097                private String count_ll_pred_err;
098                private String p2p_ll;
099                private String total_ll;
100                private String link_lambda_avg;
101                private String link_lambda_stddev;
102                private String link_lambda_min;
103                private String link_lambda_max;
104                private String plan_lambda_avg;
105                private String plan_lambda_stddev;
106                private String plan_lambda_min;
107                private String plan_lambda_max;
108                private String replan_count;
109
110                protected void setCount_ll(final String countLl) {
111                        this.count_ll = countLl;
112                }
113
114                protected void setCount_ll_pred_err(final String countLlPredErr) {
115                        this.count_ll_pred_err = countLlPredErr;
116                }
117
118                protected void setP2p_ll(final String p2pLl) {
119                        this.p2p_ll = p2pLl;
120                }
121
122                protected void setTotal_ll(final String totalLl) {
123                        this.total_ll = totalLl;
124                }
125
126                protected void setLink_lambda_avg(final String linkLambdaAvg) {
127                        this.link_lambda_avg = linkLambdaAvg;
128                }
129
130                protected void setLink_lambda_stddev(final String linkLambdaStddev) {
131                        this.link_lambda_stddev = linkLambdaStddev;
132                }
133
134                protected void setLink_lambda_min(final String linkLambdaMin) {
135                        this.link_lambda_min = linkLambdaMin;
136                }
137
138                protected void setLink_lambda_max(final String linkLambdaMax) {
139                        this.link_lambda_max = linkLambdaMax;
140                }
141
142                protected void setPlan_lambda_avg(final String planLambdaAvg) {
143                        this.plan_lambda_avg = planLambdaAvg;
144                }
145
146                protected void setPlan_lambda_stddev(final String planLambdaStddev) {
147                        this.plan_lambda_stddev = planLambdaStddev;
148                }
149
150                protected void setPlan_lambda_min(final String planLambdaMin) {
151                        this.plan_lambda_min = planLambdaMin;
152                }
153
154                protected void setPlan_lambda_max(final String planLambdaMax) {
155                        this.plan_lambda_max = planLambdaMax;
156                }
157
158                protected void setReplan_count(final String replanCount) {
159                        this.replan_count = replanCount;
160                }
161
162                public String getCount_ll() {
163                        return this.count_ll;
164                }
165
166                public String getCount_ll_pred_err() {
167                        return this.count_ll_pred_err;
168                }
169
170                public String getP2p_ll() {
171                        return this.p2p_ll;
172                }
173
174                public String getTotal_ll() {
175                        return this.total_ll;
176                }
177
178                public String getLink_lambda_avg() {
179                        return this.link_lambda_avg;
180                }
181
182                public String getLink_lambda_stddev() {
183                        return this.link_lambda_stddev;
184                }
185
186                public String getLink_lambda_min() {
187                        return this.link_lambda_min;
188                }
189
190                public String getLink_lambda_max() {
191                        return this.link_lambda_max;
192                }
193
194                public String getPlan_lambda_avg() {
195                        return this.plan_lambda_avg;
196                }
197
198                public String getPlan_lambda_stddev() {
199                        return this.plan_lambda_stddev;
200                }
201
202                public String getPlan_lambda_min() {
203                        return this.plan_lambda_min;
204                }
205
206                public String getPlan_lambda_max() {
207                        return this.plan_lambda_max;
208                }
209
210                public String getReplan_count() {
211                        return this.replan_count;
212                }
213        }
214
215        public Map <Integer, StatisticsData> getCalStatMap (){
216                return this.statDataMap;
217        }
218
219        public String preprocess(final String arg0) {
220                return arg0;
221        }
222

224}
