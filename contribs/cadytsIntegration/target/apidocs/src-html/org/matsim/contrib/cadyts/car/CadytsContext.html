052/**
053 * {@link PlanStrategy Plan Strategy} used for replanning in MATSim which uses Cadyts to
054 * select plans that better match to given occupancy counts.
055 */
056public class CadytsContext implements CadytsContextI<Link>, StartupListener, IterationEndsListener, BeforeMobsimListener {
057
058        private final static Logger log = Logger.getLogger(CadytsContext.class);
059
060        private final static String LINKOFFSET_FILENAME = "linkCostOffsets.xml";
061        private static final String FLOWANALYSIS_FILENAME = "flowAnalysis.txt";
062        
063        private final double countsScaleFactor;
064        private final Counts<Link> calibrationCounts;
065        private final boolean writeAnalysisFile;
066
067        private AnalyticalCalibrator<Link> calibrator;
068        private PlansTranslatorBasedOnEvents plansTranslator;
069        private SimResults<Link> simResults;
070        private Scenario scenario;
071        private EventsManager eventsManager;
072        private VolumesAnalyzer volumesAnalyzer;
073        private OutputDirectoryHierarchy controlerIO;
074
075        @Inject
076        CadytsContext(Config config, Scenario scenario, @Named("calibration") Counts<Link> calibrationCounts, EventsManager eventsManager, VolumesAnalyzer volumesAnalyzer, OutputDirectoryHierarchy controlerIO) {
077                this.scenario = scenario;
078                this.calibrationCounts = calibrationCounts;
079                this.eventsManager = eventsManager;
080                this.volumesAnalyzer = volumesAnalyzer;
081                this.controlerIO = controlerIO;
082                this.countsScaleFactor = config.counts().getCountsScaleFactor();
083
084                CadytsConfigGroup cadytsConfig = ConfigUtils.addOrGetModule(config, CadytsConfigGroup.GROUP_NAME, CadytsConfigGroup.class);
085                // addModule() also initializes the config group with the values read from the config file
086                cadytsConfig.setWriteAnalysisFile(true);
087
088
089                Set<String> countedLinks = new TreeSet<>();
090                for (Id<Link> id : this.calibrationCounts.getCounts().keySet()) {
091                        countedLinks.add(id.toString());
092                }
093
094                cadytsConfig.setCalibratedItems(countedLinks);
095
096                this.writeAnalysisFile = cadytsConfig.isWriteAnalysisFile();
097        }
098
099        @Override
100        public PlansTranslator<Link> getPlansTranslator() {
101                return this.plansTranslator;
102        }
103        
104        @Override
105        public void notifyStartup(StartupEvent event) {
106                this.simResults = new SimResultsContainerImpl(volumesAnalyzer, this.countsScaleFactor);
107                
108                // this collects events and generates cadyts plans from it
109                this.plansTranslator = new PlansTranslatorBasedOnEvents(scenario);
110                this.eventsManager.addHandler(plansTranslator);
111
112                this.calibrator = new CadytsBuilderImpl().buildCalibratorAndAddMeasurements(scenario.getConfig(), this.calibrationCounts , new LinkLookUp(scenario) /*, cadytsConfig.getTimeBinSize()*/, Link.class);
113        }
114
115        @Override
116        public void notifyBeforeMobsim(BeforeMobsimEvent event) {
117                // Register demand for this iteration with Cadyts.
118                // Note that planToPlanStep will return null for plans which have never been executed.
119                // This is fine, since the number of these plans will go to zero in normal simulations,
120                // and Cadyts can handle this "noise". Checked this with Gunnar.
121                // mz 2015
122                for (Person person : scenario.getPopulation().getPersons().values()) {
123                        this.calibrator.addToDemand(plansTranslator.getCadytsPlan(person.getSelectedPlan()));
124                }
125        }
126
127        @Override
128        public void notifyIterationEnds(final IterationEndsEvent event) {
129                if (this.writeAnalysisFile) {
130                        String analysisFilepath = null;
131                        if (isActiveInThisIteration(event.getIteration(), scenario.getConfig())) {
132                                analysisFilepath = controlerIO.getIterationFilename(event.getIteration(), FLOWANALYSIS_FILENAME);
133                        }
134                        this.calibrator.setFlowAnalysisFile(analysisFilepath);
135                }
136
137                this.calibrator.afterNetworkLoading(this.simResults);
138
139                // write some output
140                String filename = controlerIO.getIterationFilename(event.getIteration(), LINKOFFSET_FILENAME);
141                try {
142                        new CadytsCostOffsetsXMLFileIO<>(new LinkLookUp(scenario), Link.class)
143                           .write(filename, this.calibrator.getLinkCostOffsets());
144                } catch (IOException e) {
145                        log.error("Could not write link cost offsets!", e);
146                }
147        }
148
149        /**
150         * for testing purposes only
151         */
152        @Override
153        public AnalyticalCalibrator<Link> getCalibrator() {
154                return this.calibrator;
155        }
156
157        // ===========================================================================================================================
158        // private methods & pure delegate methods only below this line
159
160        private static boolean isActiveInThisIteration(final int iter, final Config config) {
161                return (iter > 0 && iter % config.counts().getWriteCountsInterval() == 0);
162        }
163}
