092        @Inject
093        CadytsPtContext(final Config config, EventsManager events, Scenario scenario, OutputDirectoryHierarchy controlerIO,
094                                        IterationStopWatch stopWatch, final CadytsPtOccupancyAnalyzerI cadytsPtOccupancyAnalyzer ) {
095                this.events = events;
096                this.scenario = scenario;
097                this.controlerIO = controlerIO;
098                this.stopWatch = stopWatch;
099                cadytsConfig = (CadytsConfigGroup) config.getModule(CadytsConfigGroup.GROUP_NAME);
100                this.cadytsPtOccupAnalyzer = cadytsPtOccupancyAnalyzer ;
101
102                // === prepare the structure which extracts the measurements from the simulation:
103                // since there is already some other method, we just need to write a wrapper.
104                
105//              this.cadytsPtOccupAnalyzer = new CadytsPtOccupancyAnalyzer(CadytsPtOccupancyAnalyzer.toTransitLineIdSet(cadytsConfig.getCalibratedItems()), cadytsConfig.getTimeBinSize() );
106                events.addHandler(this.cadytsPtOccupAnalyzer);
107
108                this.simResults = new SimResults<TransitStopFacility>() {
109                        private static final long serialVersionUID = 1L;
110                        @Override
111                        public double getSimValue(TransitStopFacility stop, int startTime_s, int endTime_s, TYPE type) {
112                                final int timeBinSize_s = cadytsConfig.getTimeBinSize() ;
113                                final double countsScaleFactor = config.ptCounts().getCountsScaleFactor() ;
114                                double retval = 0. ;
115                                switch ( type ) {
116                                case COUNT_VEH:
117                                        retval = cadytsPtOccupAnalyzer.getOccupancyVolumeForStopAndTime(stop.getId(), startTime_s) * countsScaleFactor ;
118                                        break;
119                                case FLOW_VEH_H:
120                                        int multiple = timeBinSize_s / 3600 ; // e.g. "3" when timeBinSize_s = 3*3600 = 10800
121                                        retval = cadytsPtOccupAnalyzer.getOccupancyVolumeForStopAndTime(stop.getId(), startTime_s) * countsScaleFactor / multiple ;
122                                        break;
123                                default:
124                                        throw new RuntimeException("not implemented ...") ;
125                                }
126//                              if ( retval != 0. ) {
127//                                      log.warn("retval=" + retval );
128//                              }
129                                return retval ;
130                        }
131                        @Override
132                        public String toString() {
133                                return cadytsPtOccupAnalyzer.toString() ;
134                        }
135                } ;
136                // === end wrapper ===
137
138        }

140        @Override
141        public void notifyStartup(StartupEvent event) {
142
143                // === prepare the calibrator by giving measurements to it:
144                String occupancyCountsFilename = scenario.getConfig().ptCounts().getOccupancyCountsFileName();
145                new MatsimCountsReader(this.occupCounts).readFile(occupancyCountsFilename);
146
147                // === build the calibrator:
148                this.calibrator = CadytsBuilderImpl.buildCalibratorAndAddMeasurements(scenario.getConfig(), this.occupCounts, new TransitStopFacilityLookUp(scenario) , TransitStopFacility.class);
149
150                // === find out which plan is contributing what to each measurement:
151                this.ptStep = new PtPlanToPlanStepBasedOnEvents<>(scenario, CadytsPtOccupancyAnalyzer.toTransitLineIdSet(cadytsConfig.getCalibratedItems()));
152                events.addHandler(ptStep);
153        }
154
155        @Override
156        public void notifyBeforeMobsim(final BeforeMobsimEvent event) {
157                this.cadytsPtOccupAnalyzer.reset(event.getIteration());
158                for (Person person : scenario.getPopulation().getPersons().values()) {
159                        this.calibrator.addToDemand(ptStep.getCadytsPlan(person.getSelectedPlan()));
160                }
161        }
162
163        @Override
164        public void notifyAfterMobsim(final AfterMobsimEvent event) {
165                int it = event.getIteration();
166
167                // Get all stations of all analyzed lines and invoke the method write to get all information of them
168                Set<Id<TransitStopFacility>> stopIds = new HashSet<>();
169                for ( String pseudoLineId : this.cadytsConfig.getCalibratedItems()) {
170                        Id<TransitLine> lineId = Id.create(pseudoLineId, TransitLine.class);
171                        TransitLine line = scenario.getTransitSchedule().getTransitLines().get(lineId);
172                        for (TransitRoute route : line.getRoutes().values()) {
173                                for (TransitRouteStop stop : route.getStops()) {
174                                        stopIds.add(stop.getStopFacility().getId());
175                                }
176                        }
177                }
178                String outFile = controlerIO.getIterationFilename(it, OCCUPANCYANALYSIS_FILENAME);
179                this.cadytsPtOccupAnalyzer.writeResultsForSelectedStopIds(outFile, this.occupCounts, stopIds);
180        }
181
182        @Override
183        public void notifyIterationEnds(final IterationEndsEvent event) {
184                if (cadytsConfig.isWriteAnalysisFile()) {
185                        String analysisFilepath = controlerIO.getIterationFilename(event.getIteration(), FLOWANALYSIS_FILENAME);
186                        this.calibrator.setFlowAnalysisFile(analysisFilepath);
187                }
188
189                this.calibrator.afterNetworkLoading(this.simResults);
190
191                // write some output
192                String filename = controlerIO.getIterationFilename(event.getIteration(), LINKOFFSET_FILENAME);
193                try {
194                        new CadytsCostOffsetsXMLFileIO<>(new TransitStopFacilityLookUp(scenario), TransitStopFacility.class)
195                        .write(filename, this.calibrator.getLinkCostOffsets());
196                } catch (IOException e) {
197                        log.error("Could not write link cost offsets!", e);
198                }
199
200                generateAndWriteCountsComparisons(event);
201        }

206        private void generateAndWriteCountsComparisons(final IterationEndsEvent event) {
207                if ( this.cadytsConfig.getTimeBinSize()!=3600 ) {
208                        log.warn("generateAndWriteCountsComparisons() does not work when time bin size != 3600.  See comments in code. Skipping the comparison ..." ) ;
209                        return ;
210                        // yyyy there are some conceptual problems behind this which are not resolved:
211                        // () There should reasonably be two methods: one describing what cadyts _thinks_ it is comparing, and one that just
212                        // compares the output.  There is one methods writing simCountCompare..., and then this one here 
213                        // writing cadytsSimCountCompare... .  It is not clarified which one is doing which.
214                        // () The method that just compares the output should not rely on cadyts but compute its own observations. --
215                        // Unfortunately, this collides with the fact that the time bin size is part of the cadyts configuration.  This is, in the end, a 
216                        // consequence of the fact that the Counts format assumes hourly counts (other than cadyts, which reasonably allows the
217                        // specify the time span for every observation separately).
218                        // kai, feb'13
219                }
220
221
222                PtCountsConfigGroup ptCountsConfig = scenario.getConfig().ptCounts();
223                if (ptCountsConfig.getOccupancyCountsFileName() == null) { // yyyy this check should reasonably also be done in isActiveInThisIteration. kai,oct'10
224                        log.warn("generateAndWriteCountsComparisons() does not work since occupancy counts file name not given ") ;
225                        return ;
226                }
227                int iter = event.getIteration();
228
229                stopWatch.beginOperation("compare with pt counts");
230
231        Network network = scenario.getNetwork();
232                CadytsPtCountsComparisonAlgorithm ccaOccupancy = new CadytsPtCountsComparisonAlgorithm(this.cadytsPtOccupAnalyzer,
233                                this.occupCounts, network, scenario.getConfig().ptCounts().getCountsScaleFactor());
234
235                Double distanceFilter = ptCountsConfig.getDistanceFilter();
236                String distanceFilterCenterNodeId  = ptCountsConfig.getDistanceFilterCenterNode();
237                if ((distanceFilter != null) && (distanceFilterCenterNodeId != null)) {
238                        ccaOccupancy.setDistanceFilter(distanceFilter, distanceFilterCenterNodeId);
239                }
240
241                ccaOccupancy.calculateComparison();
242
243                String outputFormat = ptCountsConfig.getOutputFormat();
244                if (outputFormat.contains("kml") || outputFormat.contains("all")) {
245                        String filename = controlerIO.getIterationFilename(iter, "cadytsPtCountscompare.kmz");
246                        final CoordinateTransformation coordTransform = TransformationFactory.getCoordinateTransformation(scenario.getConfig()
247                                        .global().getCoordinateSystem(), TransformationFactory.WGS84);
248//                      PtCountSimComparisonKMLWriter kmlWriter = new PtCountSimComparisonKMLWriter(null,
249//                                      null, ccaOccupancy.getComparison(), coordTransform, null, null, 
250//                                      this.occupCounts);
251
252                        CountSimComparisonKMLWriter kmlWriter = new CountSimComparisonKMLWriter(ccaOccupancy.getComparison(), this.occupCounts, coordTransform, "ptCountsOccup") ;
253
254                        kmlWriter.setIterationNumber(iter);
255                        kmlWriter.writeFile(filename);
256                }
257
258                if (outputFormat.contains("txt") || outputFormat.contains("all")) {
259                        //  As far as I can tell, this file is written twice, the other times without the "cadyts" part.  kai, feb'13
260                        //  yyyyyy As far as I can tell, the version here is wrong as soon as the time bin is different from 3600.--?? kai, feb'13
261                        //  See near beginning of method.  kai, feb'13 
262                        ccaOccupancy.write(controlerIO.getIterationFilename(iter, "cadytsSimCountCompareOccupancy.txt"));
263                }
264
265                stopWatch.endOperation("compare with pt counts");
266        }
