001package org.matsim.contrib.cadyts.measurement;
002
003import org.matsim.api.core.v01.Id;
004import org.matsim.api.core.v01.Identifiable;
005
006/**
007 * Cadyts measurements are attached to "facilities", such as links.  With distributions, the measurements are attached to nothing.
008 * To satisfy Cadyts, Measurement is used as facility to which cadyts can attach the measurement.
009 * 
010 * @author nagel
011 *
012 */
013public class Measurement implements Identifiable<Measurement>, Comparable<Measurement> {
014        // might be able to just feed Id<Measurement> to Cadyts.  On the other hand, Measurement is a good place
015        // to attach the lower bound to.
016
017        private Id<Measurement> id;
018        private double lowerBound;
019        
020        public Measurement( Id<Measurement> id, double lowerBound ) {
021                this.id = id ;
022                this.lowerBound = lowerBound ;
023        }
024
025        @Override
026        public Id<Measurement> getId() {
027                return this.id ;
028        }
029
030        public final double getLowerBound() {
031                return this.lowerBound;
032        }
033
034        @Override
035        public int compareTo(Measurement o) {
036                return this.id.compareTo( o.getId() ) ;
037        }
038
039}
