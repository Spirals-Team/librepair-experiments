001package org.matsim.contrib.cadyts.measurement;
002
003import java.util.Comparator;
004import java.util.Map;
005import java.util.SortedSet;
006import java.util.TreeMap;
007import java.util.TreeSet;
008
009import org.apache.log4j.Logger;
010import org.matsim.api.core.v01.Id;
011import org.matsim.contrib.cadyts.general.LookUpItemFromId;
012import org.matsim.counts.Count;
013
014public final class Measurements implements LookUpItemFromId<Measurement> {
015        private static final Logger log = Logger.getLogger( Measurements.class );
016        
017        private final Map< Id<Measurement>,Measurement > map = new TreeMap<>() ;
018
019        private final SortedSet<Measurement> set = new TreeSet<>( new Comparator<Measurement>(){
020                @Override
021                public int compare(Measurement o1, Measurement o2) {
022                        if ( o1.getLowerBound() < o2.getLowerBound() ) {
023                                return -1 ;
024                        } else if ( o1.getLowerBound() == o2.getLowerBound() ) {
025                                return 0 ;
026                        } else {
027                                return 1 ;
028                        }
029                }
030        } ) ;
031        
032        public void add( Count<Measurement> cnt, double lowerBound ) {
033                Measurement mea = new Measurement( cnt.getId(), lowerBound ) ;
034                map.put( mea.getId(), mea ) ;
035                set.add( mea ) ;
036        }
037        
038        @Override 
039        public Measurement getItem(Id<Measurement> id) {
040                return map.get( id ) ;
041        }
042        
043        Measurement getMeasurementFromTTimeInSeconds( double ttime ) {
044                return map.get(getMeasurementIdFromTTimeInSeconds(ttime)) ;
045        }
046
047        private Id<Measurement> getMeasurementIdFromTTimeInSeconds( double ttime ) {
048                log.warn("ttime=" + ttime  + " set=" + set );
049                Measurement prev = null ;
050                for ( Measurement mea : set ) {
051                        log.warn("lowerBound=" + mea.getLowerBound() ) ;
052                        if ( mea.getLowerBound() > ttime ) {
053                                if ( prev != null ) {
054                                        return prev.getId() ;
055                                } else {
056                                        log.warn( "entry below smallest category; adding to smallest category" ) ;
057                                        return mea.getId();
058                                }
059                        }
060                        prev = mea ;
061                }
062                return prev.getId() ;
063        }
064        
065
066}
