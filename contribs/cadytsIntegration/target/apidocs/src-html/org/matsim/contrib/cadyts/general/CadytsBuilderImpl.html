044        public static <T> AnalyticalCalibrator<T> buildCalibratorAndAddMeasurements(final Config config, final Counts<T> occupCounts,
045                                                                                                                                                                LookUpItemFromId<T> lookUp, Class<T> idType) {
046                
047                if (occupCounts.getCounts().size() == 0) {
048                        log.warn("Counts container is empty.");
049                }
050
051                CadytsConfigGroup cadytsConfig = ConfigUtils.addOrGetModule(config, CadytsConfigGroup.GROUP_NAME, CadytsConfigGroup.class);
052
053                AnalyticalCalibrator<T> matsimCalibrator = buildCalibrator(config);
054                
055                int multiple = cadytsConfig.getTimeBinSize() / 3600 ; // e.g. "3" when timeBinSize_s = 3*3600 = 10800
056
057                // If I remember correctly, the following is trying to get around the fact that the counts time bins are fixed at hourly, but we want to
058                // be able to be more flexible.  As a first step, time bins which are multiples from 1 hour are allowed, say "3".  In order to get a somewhat
059                // meaningful connection to the counts format, it will add up all entries from the corresponding 3 hours before giving this to cadyts, i.e.
060                // you may keep a higher resolution counts file but tell cadyts to work on longer time intervals. kai, dec'13
061
062                // yyyy However, it seems that some of this did not work: We are using "hourly" counts, and dividing the multi-hour values by
063                // the number of hours. ???????
064                
065                // yyyyyy I am currently of the opinion that the multi-hour version should be decoupled from the counts format.  There is a
066                // cadyts file format which allows setting mult-hour measurements, and that seems a lot more direct than trying to use a file
067                // format/data structure which is really not meant for this.  kai, dec'13
068                
069                //add counts data into calibrator
070                int numberOfAddedMeasurements = 0 ;
071                for (Map.Entry<Id<T>, Count<T>> entry : occupCounts.getCounts().entrySet()) {
072                        // (loop over all counting "items" (usually locations/stations)
073                        
074                        T item = lookUp.getItem(Id.create(entry.getKey(), idType)) ;
075                        if ( item==null ) {
076                                throw new RuntimeException("item is null; entry=" + entry + " idType=" + idType ) ;
077                        }
078                        int timeBinIndex = 0 ; // starting with zero which is different from the counts file!!!
079                        int startTimeOfBin_s = -1 ;
080                        double count = -1 ;
081                        for (Volume volume : entry.getValue().getVolumes().values()){
082                                // (loop over the different time slots)
083                                
084                                if ( timeBinIndex%multiple == 0 ) {
085                                        // (i.e. first timeBinIndex belonging to given bin)
086
087                                        startTimeOfBin_s = (volume.getHourOfDayStartingWithOne()-1)*3600 ;
088                                        count = 0 ;
089                                }
090                                count += volume.getValue() ;
091                                if ( ! ( (timeBinIndex%multiple) == (multiple-1) ) ) {
092                                        log.warn( " NOT adding measurement: timeBinIndex: " + timeBinIndex + "; multiple: " + multiple ) ;
093                                } else {
094                                        // (i.e. last timeBinIndex belonging to given bin)
095                                        
096                                        int endTimeOfBin_s   = volume.getHourOfDayStartingWithOne()*3600 - 1 ;
097                                        if ( !( cadytsConfig.getStartTime() <= startTimeOfBin_s && endTimeOfBin_s <= cadytsConfig.getEndTime()) ) {
098                                                log.warn( " NOT adding measurement: cadytsConfigStartTime: " + cadytsConfig.getStartTime() + "; startTimeOfBin_s: " + startTimeOfBin_s +
099                                                                "; endTimeOfBin_s: " + endTimeOfBin_s + "; cadytsConfigEndTime: " + cadytsConfig.getEndTime() );
100                                        } else { //add volumes for each bin to calibrator
101                                                numberOfAddedMeasurements++ ;
102//                                              matsimCalibrator.addMeasurement(item, startTimeOfBin_s, endTimeOfBin_s, count/multiple, SingleLinkMeasurement.TYPE.FLOW_VEH_H);
103                                                matsimCalibrator.addMeasurement(item, startTimeOfBin_s, endTimeOfBin_s, count, SingleLinkMeasurement.TYPE.COUNT_VEH );
104
105                                                // changed this from FLOW_VEH_H to COUNT_VEH on 30/jul/2012 since this is no longer "hourly".  
106                                                // kai/manuel, jul'12
107                                                // Despite the above comment, I am finding this with FLOW_VEH_H.  Why?  kai, feb'13
108                                                // yyyyyy For the test case, this seems to produce weird results.  The expected counts are 1 and 5, the expected result is 0 and 4.
109                                                // Possibly, it divides 1 and 5 by 2 and then has 0.5 and 2.5 for the given hour, in which case 1 and 3 would be best?????
110                                                // kai, feb'13
111                                        }
112                                }
113                                timeBinIndex++ ;
114                        }
115                }
116
117        if ( numberOfAddedMeasurements==0 ) {
118                        log.warn("No measurements were added.");
119        }
120                
121        if ( matsimCalibrator.getProportionalAssignment() ) {
122                throw new RuntimeException("Gunnar says that this may not work so do not set to true. kai, sep'14") ;
123        }
124                return matsimCalibrator;
125        }

127        public static <T> AnalyticalCalibrator<T> buildCalibrator(final Config config) {
128                CadytsConfigGroup cadytsConfig = ConfigUtils.addOrGetModule(config, CadytsConfigGroup.GROUP_NAME, CadytsConfigGroup.class ) ;
129                
130                //get timeBinSize_s and validate it
131                if ((Time.MIDNIGHT % cadytsConfig.getTimeBinSize())!= 0 ){
132                        throw new RuntimeException("Cadyts requires a divisor of 86400 as time bin size value .");
133                }
134                if ( (cadytsConfig.getTimeBinSize() % 3600) != 0 ) {
135                        throw new RuntimeException("At this point, time bin sizes need to be multiples of 3600.  This is not a restriction " +
136                                        "of Cadyts, but of the counts file format, which only allows for hourly inputs") ;
137                }
138                
139                
140                AnalyticalCalibrator<T> matsimCalibrator = new AnalyticalCalibrator<>(
141                                config.controler().getOutputDirectory() + "/cadyts.log",
142                                MatsimRandom.getLocalInstance().nextLong(),cadytsConfig.getTimeBinSize()
143                                 ) ;
144
145                matsimCalibrator.setRegressionInertia(cadytsConfig.getRegressionInertia()) ;
146                matsimCalibrator.setMinStddev(cadytsConfig.getMinFlowStddev_vehPerHour(), TYPE.FLOW_VEH_H);
147                matsimCalibrator.setMinStddev(cadytsConfig.getMinFlowStddev_vehPerHour(), TYPE.COUNT_VEH);
148                matsimCalibrator.setFreezeIteration(cadytsConfig.getFreezeIteration());
149                matsimCalibrator.setPreparatoryIterations(cadytsConfig.getPreparatoryIterations());
150                matsimCalibrator.setVarianceScale(cadytsConfig.getVarianceScale());
151
152                matsimCalibrator.setBruteForce(cadytsConfig.useBruteForce());
153                // I don't think this has an influence on any of the variants we are using. (Has an influence only when plan choice is left
154                // completely to cadyts, rather than just taking the score offsets.) kai, dec'13
155                // More formally, one would need to use the selectPlan() method of AnalyticalCalibrator which we are, however, not using. kai, mar'14
156                if ( matsimCalibrator.getBruteForce() ) {
157                        log.warn("setting bruteForce==true for calibrator, but this won't do anything in the way the cadyts matsim integration is set up. kai, mar'14") ;
158                }
159                
160                matsimCalibrator.setStatisticsFile(config.controler().getOutputDirectory() + "/calibration-stats.txt");
161                return matsimCalibrator;
162        }
