020@Deprecated // please test before using.  If it works, remove deprecated tag.  Optimally, write test case. kai, dec'13
021public final class PlanSelectionByCadyts<T> implements PlanSelector<Plan, Person> {
022
023        private final CadytsContextI<T> cContext;
024        private double beta;
025        
026        public PlanSelectionByCadyts(double beta, CadytsContextI<T> cContext ) {
027                this.cContext = cContext ;
028                this.beta = beta ;
029        }
030        
031
032        @Override
033        public Plan selectPlan(HasPlansAndId<Plan, Person> person) {
034                List<cadyts.demand.Plan<T>> plans = new ArrayList<cadyts.demand.Plan<T>>() ;
035                Vector choiceProbs = new Vector( person.getPlans().size() ) ;
036                int pos = 0 ;
037                for ( Plan plan : person.getPlans() ) {
038                        plans.add( cContext.getPlansTranslator().getCadytsPlan(plan) ) ;
039
040                        choiceProbs.add(pos, ExpBetaPlanSelector.getSelectionProbability(new ExpBetaPlanSelector(beta), plan.getPerson(), plan) );
041                        // I guess these are supposed to be the prior probabilities. If so, than the above shoudl be correct (albeit a bit expensive).
042                        // I would have expected that these are internally ignored when brute force is switched on, but they are not.
043                        // I think that what it does is to use them when plans are equal according to cadyts. kai, dec'13
044                        
045                        pos++ ;
046                }
047                int idx = cContext.getCalibrator().selectPlan(plans, choiceProbs) ;
048
049//              cContext.getCalibrator().addToDemand(cadytsPlan) ;
050                // I _think_ that's done inside calibrator.selectPlan. kai, dec'13
051
052                return person.getPlans().get(idx) ;
053        }
054        
055}
