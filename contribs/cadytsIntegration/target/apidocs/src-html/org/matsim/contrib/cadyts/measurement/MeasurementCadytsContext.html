021package org.matsim.contrib.cadyts.measurement;
022
023import cadyts.calibrators.analytical.AnalyticalCalibrator;
024import cadyts.demand.Plan;
025import org.apache.log4j.Logger;
026import org.matsim.api.core.v01.Id;
027import org.matsim.api.core.v01.Scenario;
028import org.matsim.api.core.v01.population.Person;
029import org.matsim.contrib.cadyts.general.*;
030import org.matsim.core.config.Config;
031import org.matsim.core.config.ConfigUtils;
032import org.matsim.core.controler.events.BeforeMobsimEvent;
033import org.matsim.core.controler.events.IterationEndsEvent;
034import org.matsim.core.controler.events.StartupEvent;
035import org.matsim.core.controler.listener.BeforeMobsimListener;
036import org.matsim.core.controler.listener.IterationEndsListener;
037import org.matsim.core.controler.listener.StartupListener;
038import org.matsim.core.utils.collections.Tuple;
039import org.matsim.counts.Counts;
040
041import java.io.IOException;
042import java.util.Set;
043import java.util.TreeSet;
044
045public class MeasurementCadytsContext implements CadytsContextI<Measurement>, StartupListener, IterationEndsListener, BeforeMobsimListener {
046
047        final static Logger log = Logger.getLogger(MeasurementCadytsContext.class);
048
049        private final static String COSTOFFSET_FILENAME = "costOffsets.xml";
050        private static final String ANALYSIS_FILENAME = "analysis.txt";
051
052        private final Counts<Measurement> counts;
053        private final boolean writeAnalysisFile;
054
055        private AnalyticalCalibrator<Measurement> calibrator;
056        private MeasurementListener measurementListener;
057
058        private final Measurements measurements ;
059
060        public MeasurementCadytsContext(Config config, Tuple<Counts<Measurement>, Measurements> tuple) {
061
062                CadytsConfigGroup cadytsConfig = ConfigUtils.addOrGetModule(config, CadytsConfigGroup.GROUP_NAME, CadytsConfigGroup.class ) ;
063                cadytsConfig.setWriteAnalysisFile(true);
064
065                this.counts = tuple.getFirst() ;
066                this.measurements = tuple.getSecond() ;
067
068                Set<String> measurementsSet = new TreeSet<>();
069                for (Id<Measurement> id : this.counts.getCounts().keySet()) {
070                        measurementsSet.add(id.toString());
071                }
072
073                cadytsConfig.setCalibratedItems(measurementsSet);
074
075                this.writeAnalysisFile = cadytsConfig.isWriteAnalysisFile();
076        }
077
078        @Override
079        public PlansTranslator<Measurement> getPlansTranslator() {
080                return this.measurementListener;
081        }
082
083        @Override
084        public void notifyStartup(StartupEvent event) {
085                // yy could probably move all of this method into the constructor, and then get rid of this method.  But should
086                // be debugged and a test case written before. kai, oct'15
087
088                Scenario scenario = event.getServices().getScenario();
089                Config config = scenario.getConfig();
090
091                // 1st major Cadyts method is "calibrator.addMesurement"
092                // in this implementation it is called by the "CadytsBuilderImpl", dz 09/15
093                this.calibrator = new CadytsBuilderImpl().buildCalibratorAndAddMeasurements(config, this.counts, measurements, Measurement.class) ;
094
095                this.measurementListener = new MeasurementListener(scenario, measurements );
096                event.getServices().getEvents().addHandler(measurementListener);
097        }
098
099        @Override
100        public void notifyBeforeMobsim(BeforeMobsimEvent event) {
101
102                // 2nd major Cadyts method is "analyzer.addToDemand", dz 09/15
103                
104                // Register demand for this iteration with Cadyts.
105                // Note that planToPlanStep will return null for plans which have never been executed.
106                // This is fine, since the number of these plans will go to zero in normal simulations,
107                // and Cadyts can handle this "noise". Checked this with Gunnar.
108                // mz 2015
109                for (Person person : event.getServices().getScenario().getPopulation().getPersons().values()) {
110                        Plan<Measurement> planSteps = this.measurementListener.getCadytsPlan(person.getSelectedPlan());
111                        this.calibrator.addToDemand(planSteps);
112                }
113        }
114
115        @Override
116        public void notifyIterationEnds(final IterationEndsEvent event) {
117                if (this.writeAnalysisFile) {
118                        String analysisFilepath = event.getServices().getControlerIO().getIterationFilename(event.getIteration(), ANALYSIS_FILENAME);
119                        this.calibrator.setFlowAnalysisFile(analysisFilepath);
120                }
121
122                // 3rd major Cadyts method "calibrator.afterNetworkLoading", dz 09/15
123                this.calibrator.afterNetworkLoading(this.measurementListener);
124
125                // write some output
126                String filename = event.getServices().getControlerIO().getIterationFilename(event.getIteration(), COSTOFFSET_FILENAME);
127                try {
128                        new CadytsCostOffsetsXMLFileIO<Measurement>( this.measurements, Measurement.class)
129                        .write(filename, this.calibrator.getLinkCostOffsets());
130                } catch (IOException e) {
131                        log.error("Could not write link cost offsets!", e);
132                }
133        }
134
135        // ===========================================================================================================================
136        // private methods & pure delegate methods only below this line
137
138        @Override
139        public AnalyticalCalibrator<Measurement> getCalibrator() {
140                return this.calibrator ;
141        }
142}
