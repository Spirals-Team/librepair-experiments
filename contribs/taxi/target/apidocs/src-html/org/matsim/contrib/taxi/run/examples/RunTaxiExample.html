035
036import com.google.inject.Provider;
037
038public class RunTaxiExample {
039        private static final String CONFIG_FILE = "mielec_2014_02/mielec_taxi_config.xml";
040
041        public static void run(boolean otfvis, int lastIteration) {
042                // load config
043                Config config = ConfigUtils.loadConfig(CONFIG_FILE, new TaxiConfigGroup(), new DvrpConfigGroup(),
044                                new OTFVisConfigGroup());
045                config.controler().setLastIteration(lastIteration);
046                config.addConfigConsistencyChecker(new TaxiConfigConsistencyChecker());
047                config.checkConsistency();
048
049                // load scenario
050                Scenario scenario = ScenarioUtils.loadScenario(config);
051
052                // setup controler
053                Controler controler = new Controler(scenario);
054
055                final boolean ownTaxiOptimizer = false;
056                if (!ownTaxiOptimizer) {
057                        controler.addOverridingModule(TaxiDvrpModules.create());
058                        // (default taxi optimizer)
059                } else {
060                        controler.addOverridingModule(TaxiDvrpModules.create(MyTaxiOptimizerProvider.class));
061                        // (implement your own taxi optimizer)
062                }
063
064                controler.addOverridingModule(new TaxiModule()); // taxi output (can be commented out)
065
066                if (otfvis) {
067                        controler.addOverridingModule(new OTFVisLiveModule()); // OTFVis visualisation
068                }
069
070                // run simulation
071                controler.run();
072        }
073
074        public static void main(String[] args) {
075                RunTaxiExample.run(false, 0); // switch to 'true' to turn on visualisation
076        }
077
078        /**
079         * See {@link DefaultTaxiOptimizerProvider} for examples.
080         */
081        private static class MyTaxiOptimizerProvider implements Provider<TaxiOptimizer> {
082                @Override
083                public TaxiOptimizer get() {
084                        return new TaxiOptimizer() {
085                                @Override
086                                public void vehicleEnteredNextLink(Vehicle vehicle, Link nextLink) {
087                                        // TODO Auto-generated method stub
088                                }
089
090                                @Override
091                                public void requestSubmitted(Request request) {
092                                        // TODO Auto-generated method stub
093                                }
094
095                                @Override
096                                public void nextTask(Vehicle vehicle) {
097                                        // TODO Auto-generated method stub
098                                }
099
100                                @Override
101                                public void notifyMobsimBeforeSimStep(@SuppressWarnings("rawtypes") MobsimBeforeSimStepEvent e) {
102                                        // TODO Auto-generated method stub
103                                }
104                        };
105                }
106        }
107}
