050/**
051 * @author michalm
052 */
053public class ZonalRequestInserter implements UnplannedRequestInserter {
054        private static final Comparator<Vehicle> LONGEST_WAITING_FIRST = (v1, v2) -> Double.compare(
055                        v1.getSchedule().getCurrentTask().getBeginTime(), v2.getSchedule().getCurrentTask().getBeginTime());
056
057        private final Fleet fleet;
058        private final TaxiScheduler scheduler;
059        private final BestDispatchFinder dispatchFinder;
060        private final RuleBasedRequestInserter requestInserter;
061
062        private final Map<Id<Zone>, Zone> zones;
063        private Map<Id<Zone>, PriorityQueue<Vehicle>> zoneToIdleVehicleQueue;
064        private final Map<Id<Link>, Zone> linkToZone;
065
066        public ZonalRequestInserter(Fleet fleet, TaxiScheduler scheduler, MobsimTimer timer, Network network,
067                        TravelTime travelTime, TravelDisutility travelDisutility, ZonalTaxiOptimizerParams params,
068                        IdleTaxiZonalRegistry idleTaxiRegistry, UnplannedRequestZonalRegistry unplannedRequestRegistry) {
069                this.fleet = fleet;
070                this.scheduler = scheduler;
071                this.dispatchFinder = new BestDispatchFinder(scheduler, network, timer, travelTime, travelDisutility);
072                this.requestInserter = new RuleBasedRequestInserter(scheduler, timer, dispatchFinder, params, idleTaxiRegistry,
073                                unplannedRequestRegistry);
074
075                zones = Zones.readZones(params.zonesXmlFile, params.zonesShpFile);
076                System.err.println("No conversion of SRS is done");
077
078                this.linkToZone = NetworkWithZonesUtils.createLinkToZoneMap(network,
079                                new ZoneFinderImpl(zones, params.expansionDistance));
080
081                // FIXME zonal system used in RuleBasedTaxiOptim (for registers) should be equivalent to
082                // the zones used in ZonalTaxiOptim (for dispatching)
083        }
084
085        @Override
086        public void scheduleUnplannedRequests(Collection<TaxiRequest> unplannedRequests) {
087                initIdleVehiclesInZones();
088                scheduleUnplannedRequestsWithinZones(unplannedRequests);
089
090                if (!unplannedRequests.isEmpty()) {
091                        requestInserter.scheduleUnplannedRequests(unplannedRequests);
092                }
093        }
094
095        private void initIdleVehiclesInZones() {
096                // TODO use idle vehicle register instead...
097
098                zoneToIdleVehicleQueue = new HashMap<>();
099                for (Id<Zone> zoneId : zones.keySet()) {
100                        zoneToIdleVehicleQueue.put(zoneId, new PriorityQueue<Vehicle>(10, LONGEST_WAITING_FIRST));
101                }
102
103                for (Vehicle veh : fleet.getVehicles().values()) {
104                        if (scheduler.isIdle(veh)) {
105                                Link link = ((StayTask)veh.getSchedule().getCurrentTask()).getLink();
106                                Zone zone = linkToZone.get(link.getId());
107                                if (zone != null) {
108                                        PriorityQueue<Vehicle> queue = zoneToIdleVehicleQueue.get(zone.getId());
109                                        queue.add(veh);
110                                }
111                        }
112                }
113        }
114
115        private void scheduleUnplannedRequestsWithinZones(Collection<TaxiRequest> unplannedRequests) {
116                Iterator<TaxiRequest> reqIter = unplannedRequests.iterator();
117                while (reqIter.hasNext()) {
118                        TaxiRequest req = reqIter.next();
119
120                        Zone zone = linkToZone.get(req.getFromLink().getId());
121                        if (zone == null) {
122                                continue;
123                        }
124
125                        PriorityQueue<Vehicle> idleVehsInZone = zoneToIdleVehicleQueue.get(zone.getId());
126                        if (idleVehsInZone.isEmpty()) {
127                                continue;
128                        }
129
130                        Stream<Vehicle> filteredVehs = Stream.of(idleVehsInZone.peek());
131                        BestDispatchFinder.Dispatch<TaxiRequest> best = dispatchFinder.findBestVehicleForRequest(req, filteredVehs);
132
133                        if (best != null) {
134                                scheduler.scheduleRequest(best.vehicle, best.destination, best.path);
135                                reqIter.remove();
136                                idleVehsInZone.remove(best.vehicle);
137                        }
138                }
139        }
140}
