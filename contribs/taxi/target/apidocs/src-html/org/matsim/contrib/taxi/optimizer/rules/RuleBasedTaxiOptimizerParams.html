020package org.matsim.contrib.taxi.optimizer.rules;
021
022import org.apache.commons.configuration.Configuration;
023import org.matsim.contrib.taxi.optimizer.DefaultTaxiOptimizerParams;
024import org.matsim.contrib.taxi.optimizer.rules.RuleBasedRequestInserter.Goal;
025
026public class RuleBasedTaxiOptimizerParams extends DefaultTaxiOptimizerParams {
027        public static final String GOAL = "goal";
028
029        public static final String NEAREST_REQUESTS_LIMIT = "nearestRequestsLimit";
030        public static final String NEAREST_VEHICLES_LIMIT = "nearestVehiclesLimit";
031
032        public static final String CELL_SIZE = "cellSize";
033
034        public final Goal goal;
035
036        public final int nearestRequestsLimit;
037        public final int nearestVehiclesLimit;
038
039        public final double cellSize;
040
041        public RuleBasedTaxiOptimizerParams(Configuration optimizerConfig) {
042                super(optimizerConfig, false, false);
043
044                goal = Goal.valueOf(optimizerConfig.getString(GOAL));
045
046                nearestRequestsLimit = optimizerConfig.getInt(NEAREST_REQUESTS_LIMIT);
047                nearestVehiclesLimit = optimizerConfig.getInt(NEAREST_VEHICLES_LIMIT);
048
049                cellSize = optimizerConfig.getDouble(CELL_SIZE);// 1000 m tested for Berlin
050        }
051}
