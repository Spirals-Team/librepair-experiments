020package org.matsim.contrib.taxi.util.stats;
021
022import java.util.*;
023
024import org.matsim.contrib.dvrp.data.Vehicle;
025
026public class TaxiStatsCalculators {
027        public static final String DAILY_STATS_ID = "daily";
028
029        public static int calcHourCount(Iterable<? extends Vehicle> vehicles) {
030                double maxEndTime = 0;
031                for (Vehicle v : vehicles) {
032                        double endTime = v.getSchedule().getEndTime();
033                        if (endTime > maxEndTime) {
034                                maxEndTime = endTime;
035                        }
036                }
037
038                return (int)Math.ceil(maxEndTime / 3600);
039        }
040
041        public static int getHour(double time) {
042                return (int)(time / 3600);
043        }
044
045        public static <T> List<T> createStatsList(T[] hourlyStats, T dailyStats) {
046                List<T> allStats = new ArrayList<>(hourlyStats.length + 1);
047                Collections.addAll(allStats, hourlyStats);
048                allStats.add(dailyStats);
049                return Collections.unmodifiableList(allStats);
050        }
051
052        public static int[] calcHourlyDurations(int from, int to) {
053                int firstHour = (int)from / 3600;
054                int lastHour = (int)to / 3600;
055
056                if (firstHour == lastHour) {
057                        return new int[] { to - from };
058                }
059
060                int[] hourlyDurations = new int[lastHour - firstHour + 1];
061                hourlyDurations[0] = 3600 - from % 3600;
062                hourlyDurations[hourlyDurations.length - 1] = to % 3600;
063                for (int i = 1; i < hourlyDurations.length - 1; i++) {
064                        hourlyDurations[i] = 3600;
065                }
066
067                return hourlyDurations;
068        }
069}
