035

043        public static final String DESTINATION_KNOWN = "destinationKnown";
044        public static final String VEHICLE_DIVERSION = "vehicleDiversion";
045        public static final String PICKUP_DURATION = "pickupDuration";
046        public static final String DROPOFF_DURATION = "dropoffDuration";
047        public static final String A_STAR_EUCLIDEAN_OVERDO_FACTOR = "AStarEuclideanOverdoFactor";
048        public static final String ONLINE_VEHICLE_TRACKER = "onlineVehicleTracker";
049        public static final String CHANGE_START_LINK_TO_LAST_LINK_IN_SCHEDULE = "changeStartLinkToLastLinkInSchedule";
050
051        // input
052        public static final String TAXIS_FILE = "taxisFile";
053
054        // output
055        public static final String TIME_PROFILES = "timeProfiles";
056        public static final String DETAILED_STATS = "detailedStats";
057
058        public static final String BREAK_IF_NOT_ALL_REQUESTS_SERVED = "breakIfNotAllRequestsServed";
059
060        public static final String OPTIMIZER_PARAMETER_SET = "optimizer";
061
062        private boolean destinationKnown = false;
063        private boolean vehicleDiversion = false;
064        
065        @PositiveOrZero
066        private double pickupDuration = Double.NaN;// seconds
067
068        @PositiveOrZero
069        private double dropoffDuration = Double.NaN;// seconds
070        
071        @Min(1)
072        private double AStarEuclideanOverdoFactor = 1.;
073        
074        private boolean onlineVehicleTracker = false;
075        private boolean changeStartLinkToLastLinkInSchedule = false;
076
077        @NotNull
078        private String taxisFile = null;
079
080        private boolean timeProfiles = false;
081        private boolean detailedStats = false;
082
083        private boolean breakSimulationIfNotAllRequestsServed = true;
084
085        public TaxiConfigGroup() {
086                super(GROUP_NAME);
087        }
088
089        @Override
090        public Map<String, String> getComments() {
091                Map<String, String> map = super.getComments();
092                map.put(DESTINATION_KNOWN, "If false, the drop-off location remains unknown to the optimizer and scheduler"
093                                + " until the end of pickup. False by default.");
094                map.put(VEHICLE_DIVERSION, "If true, vehicles can be diverted during empty trips. Requires online tracking."
095                                + " False by default.");
096                map.put(PICKUP_DURATION, "Typically, 120 seconds");
097                map.put(DROPOFF_DURATION, "Typically, 60 seconds");
098                map.put(A_STAR_EUCLIDEAN_OVERDO_FACTOR,
099                                "Used in AStarEuclidean for shortest path search for occupied drives. Default value is 1.0. "
100                                                + "Values above 1.0 (typically, 1.5 to 3.0) speed up search, "
101                                                + "but at the cost of obtaining longer paths");
102                map.put(ONLINE_VEHICLE_TRACKER,
103                                "If true, vehicles are (GPS-like) monitored while moving. This helps in getting more accurate "
104                                                + "estimates on the time of arrival. Online tracking is necessary for vehicle diversion. "
105                                                + "False by default.");
106                map.put(CHANGE_START_LINK_TO_LAST_LINK_IN_SCHEDULE,
107                                "If true, the startLink is changed to last link in the current schedule, so the taxi starts the next "
108                                                + "day at the link where it stopped operating the day before. False by default.");
109                map.put(TAXIS_FILE, "An XML file specifying the taxi fleet. The file format according to dvrp_vehicles_v1.dtd");
110                map.put(TIME_PROFILES,
111                                "If true, writes time profiles of vehicle statuses (i.e. current task type) and the number of unplanned "
112                                                + "requests are written to a text file (taxi_status_time_profiles) and saved as plots. "
113                                                + "False by default.");
114                map.put(DETAILED_STATS,
115                                "If true, detailed hourly taxi stats are dumped after each iteration. False by default.");
116                map.put(BREAK_IF_NOT_ALL_REQUESTS_SERVED,
117                                "Specifies whether the simulation should interrupt if not all requests were performed when"
118                                                + " an interation ends. Otherwise, a warning is given. True by default.");
119                map.put(OPTIMIZER_PARAMETER_SET, // currently, comments for parameter sets are not printed
120                                "Specifies the type and parameters of the TaxiOptimizer. See: TaxiOptimizerParams and classes"
121                                                + " implementing it, e.g. AbstractTaxiOptimizerParams.");
122                return map;
123        }

235        public ConfigGroup getOptimizerConfigGroup() {
236                return getParameterSets(OPTIMIZER_PARAMETER_SET).iterator().next();
237        }
238
239        public void setOptimizerConfigGroup(ConfigGroup optimizerCfg) {
240                clearParameterSetsForType(OPTIMIZER_PARAMETER_SET);
241                addParameterSet(optimizerCfg);
242        }
243
244        public URL getTaxisFileUrl(URL context) {
245                return ConfigGroup.getInputFileURL(context, this.taxisFile);
246        }
