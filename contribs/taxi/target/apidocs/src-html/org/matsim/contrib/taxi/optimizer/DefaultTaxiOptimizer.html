020package org.matsim.contrib.taxi.optimizer;
021
022import java.util.Collection;
023import java.util.List;
024import java.util.TreeSet;
025
026import org.matsim.api.core.v01.network.Link;
027import org.matsim.contrib.dvrp.data.Fleet;
028import org.matsim.contrib.dvrp.data.Request;
029import org.matsim.contrib.dvrp.data.Vehicle;
030import org.matsim.contrib.dvrp.passenger.PassengerRequests;
031import org.matsim.contrib.dvrp.schedule.Task;
032import org.matsim.contrib.taxi.data.TaxiRequest;
033import org.matsim.contrib.taxi.run.TaxiConfigGroup;
034import org.matsim.contrib.taxi.schedule.TaxiTask;
035import org.matsim.contrib.taxi.schedule.TaxiTask.TaxiTaskType;
036import org.matsim.contrib.taxi.scheduler.TaxiScheduler;
037import org.matsim.core.mobsim.framework.events.MobsimBeforeSimStepEvent;
038
039/**
040 * @author michalm
041 */
042public class DefaultTaxiOptimizer implements TaxiOptimizer {
043        private final Fleet fleet;
044        private final TaxiScheduler scheduler;
045
046        private final Collection<TaxiRequest> unplannedRequests = new TreeSet<TaxiRequest>(
047                        PassengerRequests.ABSOLUTE_COMPARATOR);
048        private final UnplannedRequestInserter requestInserter;
049
050        private final boolean destinationKnown;
051        private final boolean vehicleDiversion;
052        private final DefaultTaxiOptimizerParams params;
053
054        private boolean requiresReoptimization = false;
055
056        public DefaultTaxiOptimizer(TaxiConfigGroup taxiCfg, Fleet fleet, TaxiScheduler scheduler,
057                        DefaultTaxiOptimizerParams params, UnplannedRequestInserter requestInserter) {
058                this.fleet = fleet;
059                this.scheduler = scheduler;
060                this.requestInserter = requestInserter;
061                this.params = params;
062
063                destinationKnown = taxiCfg.isDestinationKnown();
064                vehicleDiversion = taxiCfg.isVehicleDiversion();
065        }
066
067        @Override
068        public void notifyMobsimBeforeSimStep(@SuppressWarnings("rawtypes") MobsimBeforeSimStepEvent e) {
069                if (requiresReoptimization && isNewDecisionEpoch(e, params.reoptimizationTimeStep)) {
070                        if (params.doUnscheduleAwaitingRequests) {
071                                unscheduleAwaitingRequests();
072                        }
073
074                        // TODO update timeline only if the algo really wants to reschedule in this time step,
075                        // perhaps by checking if there are any unplanned requests??
076                        if (params.doUpdateTimelines) {
077                                for (Vehicle v : fleet.getVehicles().values()) {
078                                        scheduler.updateTimeline(v);
079                                }
080                        }
081
082                        scheduleUnplannedRequests();
083
084                        if (params.doUnscheduleAwaitingRequests && vehicleDiversion) {
085                                handleAimlessDriveTasks();
086                        }
087
088                        requiresReoptimization = false;
089                }
090        }
091
092        public static boolean isNewDecisionEpoch(@SuppressWarnings("rawtypes") MobsimBeforeSimStepEvent e,
093                        int epochLength) {
094                return e.getSimulationTime() % epochLength == 0;
095        }
096
097        protected void unscheduleAwaitingRequests() {
098                List<TaxiRequest> removedRequests = scheduler.removeAwaitingRequestsFromAllSchedules();
099                unplannedRequests.addAll(removedRequests);
100        }
101
102        protected void scheduleUnplannedRequests() {
103                requestInserter.scheduleUnplannedRequests(unplannedRequests);
104        }
105
106        protected void handleAimlessDriveTasks() {
107                scheduler.stopAllAimlessDriveTasks();
108        }
109
110        @Override
111        public void requestSubmitted(Request request) {
112                unplannedRequests.add((TaxiRequest)request);
113                requiresReoptimization = true;
114        }
115
116        @Override
117        public void nextTask(Vehicle vehicle) {
118                scheduler.updateBeforeNextTask(vehicle);
119
120                Task newCurrentTask = vehicle.getSchedule().nextTask();
121
122                if (!requiresReoptimization && newCurrentTask != null) {// schedule != COMPLETED
123                        requiresReoptimization = doReoptimizeAfterNextTask((TaxiTask)newCurrentTask);
124                }
125        }
126
127        protected boolean doReoptimizeAfterNextTask(TaxiTask newCurrentTask) {
128                return !destinationKnown && newCurrentTask.getTaxiTaskType() == TaxiTaskType.OCCUPIED_DRIVE;
129        }
130
131        @Override
132        public void vehicleEnteredNextLink(Vehicle vehicle, Link nextLink) {
133                // TODO do we really need this??? timeline is updated always before reoptimisation
134                scheduler.updateTimeline(vehicle);// TODO comment this out...
135
136                // TODO we may here possibly decide whether or not to reoptimize
137                // if (delays/speedups encountered) {requiresReoptimization = true;}
138        }
139
140        protected void setRequiresReoptimization(boolean requiresReoptimization) {
141                this.requiresReoptimization = requiresReoptimization;
142        }
143}
