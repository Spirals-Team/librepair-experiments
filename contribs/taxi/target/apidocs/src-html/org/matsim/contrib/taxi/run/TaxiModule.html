020package org.matsim.contrib.taxi.run;
021
022import org.matsim.contrib.dvrp.data.Fleet;
023import org.matsim.contrib.dvrp.data.file.FleetProvider;
024import org.matsim.contrib.dvrp.router.TimeAsTravelDisutility;
025import org.matsim.contrib.taxi.optimizer.DefaultTaxiOptimizerProvider;
026import org.matsim.contrib.taxi.passenger.SubmittedTaxiRequestsCollector;
027import org.matsim.contrib.taxi.util.TaxiSimulationConsistencyChecker;
028import org.matsim.contrib.taxi.util.stats.TaxiStatsDumper;
029import org.matsim.contrib.taxi.util.stats.TaxiStatusTimeProfileCollectorProvider;
030import org.matsim.core.controler.AbstractModule;
031import org.matsim.core.router.costcalculators.TravelDisutilityFactory;
032
033import com.google.inject.name.Names;
034
035/**
036 * @author michalm
037 */
038public final class TaxiModule extends AbstractModule {
039        @Override
040        public void install() {
041                TaxiConfigGroup taxiCfg = TaxiConfigGroup.get(getConfig());
042                bind(Fleet.class).toProvider(new FleetProvider(taxiCfg.getTaxisFileUrl(getConfig().getContext())))
043                                .asEagerSingleton();
044                bind(TravelDisutilityFactory.class).annotatedWith(Names.named(DefaultTaxiOptimizerProvider.TAXI_OPTIMIZER))
045                                .toInstance(travelTime -> new TimeAsTravelDisutility(travelTime));
046                
047                bind(SubmittedTaxiRequestsCollector.class).toInstance(new SubmittedTaxiRequestsCollector());
048                addControlerListenerBinding().to(SubmittedTaxiRequestsCollector.class);
049
050                addControlerListenerBinding().to(TaxiSimulationConsistencyChecker.class);
051                addControlerListenerBinding().to(TaxiStatsDumper.class);
052
053                if (taxiCfg.getTimeProfiles()) {
054                        addMobsimListenerBinding().toProvider(TaxiStatusTimeProfileCollectorProvider.class);
055                        // add more time profiles if necessary
056                }
057        }
058}
