073        @Inject
074        public TaxiScheduler(TaxiConfigGroup taxiCfg, Fleet fleet,
075                        @Named(DvrpRoutingNetworkProvider.DVRP_ROUTING) Network network, MobsimTimer timer,
076                        @Named(DvrpTravelTimeModule.DVRP_ESTIMATED) TravelTime travelTime,
077                        @Named(DefaultTaxiOptimizerProvider.TAXI_OPTIMIZER) TravelDisutility travelDisutility) {
078                this.taxiCfg = taxiCfg;
079                this.fleet = fleet;
080                this.timer = timer;
081                this.travelTime = travelTime;
082
083                router = new FastAStarEuclideanFactory(taxiCfg.getAStarEuclideanOverdoFactor()).createPathCalculator(network,
084                                travelDisutility, travelTime);
085                initFleet(taxiCfg);
086        }

088        private void initFleet(TaxiConfigGroup taxiCfg) {
089                if (taxiCfg.isChangeStartLinkToLastLinkInSchedule()) {
090                        for (Vehicle veh : fleet.getVehicles().values()) {
091                                Vehicles.changeStartLinkToLastLinkInSchedule(veh);
092                        }
093                }
094
095                fleet.resetSchedules();
096                for (Vehicle veh : fleet.getVehicles().values()) {
097                        veh.getSchedule()
098                                        .addTask(new TaxiStayTask(veh.getServiceBeginTime(), veh.getServiceEndTime(), veh.getStartLink()));
099                }
100        }

102        @Override
103        public boolean isIdle(Vehicle vehicle) {
104                Schedule schedule = vehicle.getSchedule();
105                if (timer.getTimeOfDay() >= vehicle.getServiceEndTime() || schedule.getStatus() != ScheduleStatus.STARTED) {
106                        return false;
107                }
108
109                TaxiTask currentTask = (TaxiTask)schedule.getCurrentTask();
110                return currentTask.getTaskIdx() == schedule.getTaskCount() - 1 // last task
111                                && currentTask.getTaxiTaskType() == TaxiTaskType.STAY;
112        }

114        /**
115         * If the returned LinkTimePair is not null, then time is not smaller than the current time
116         */
117        @Override
118        public LinkTimePair getImmediateDiversionOrEarliestIdleness(Vehicle veh) {
119                if (taxiCfg.isVehicleDiversion()) {
120                        LinkTimePair diversion = getImmediateDiversion(veh);
121                        if (diversion != null) {
122                                return diversion;
123                        }
124                }
125
126                return getEarliestIdleness(veh);
127        }

129        /**
130         * If the returned LinkTimePair is not null, then time is not smaller than the current time
131         */
132        @Override
133        public LinkTimePair getEarliestIdleness(Vehicle veh) {
134                if (timer.getTimeOfDay() >= veh.getServiceEndTime()) {// time window exceeded
135                        return null;
136                }
137
138                Schedule schedule = veh.getSchedule();
139                Link link;
140                double time;
141
142                switch (schedule.getStatus()) {
143                        case PLANNED:
144                        case STARTED:
145                                TaxiTask lastTask = (TaxiTask)Schedules.getLastTask(schedule);
146
147                                switch (lastTask.getTaxiTaskType()) {
148                                        case STAY:
149                                                link = ((StayTask)lastTask).getLink();
150                                                time = Math.max(lastTask.getBeginTime(), timer.getTimeOfDay());// TODO very optimistic!!!
151                                                return createValidLinkTimePair(link, time, veh);
152
153                                        case PICKUP:
154                                                if (!taxiCfg.isDestinationKnown()) {
155                                                        return null;
156                                                }
157                                                // otherwise: IllegalStateException -- the schedule should end with STAY (or PICKUP if
158                                                // unfinished)
159
160                                        default:
161                                                throw new IllegalStateException(
162                                                                "Type of the last task is wrong: " + lastTask.getTaxiTaskType());
163                                }
164
165                        case COMPLETED:
166                                return null;
167
168                        case UNPLANNED:// there is always at least one STAY task in a schedule
169                        default:
170                                throw new IllegalStateException();
171                }
172        }
173
174        /**
175         * If the returned LinkTimePair is not null, then time is not smaller than the current time
176         */
177        @Override
178        public LinkTimePair getImmediateDiversion(Vehicle veh) {
179                if (!taxiCfg.isVehicleDiversion()) {
180                        throw new RuntimeException("Diversion must be on");
181                }
182
183                Schedule schedule = veh.getSchedule();
184                // timer.getTimeOfDay() >= veh.getServiceEndTime() is ALLOWED because we need to stop/divert delayed vehicles
185                // so do not return null
186                if (schedule.getStatus() != ScheduleStatus.STARTED) {
187                        return null;
188                }
189
190                TaxiTask currentTask = (TaxiTask)schedule.getCurrentTask();
191                // no prebooking ==> we can divert vehicle whose current task is an empty drive at the end of the schedule
192                if (currentTask.getTaskIdx() != schedule.getTaskCount() - 1 // not last task
193                                || currentTask.getTaxiTaskType() != TaxiTaskType.EMPTY_DRIVE) {
194                        return null;
195                }
196
197                OnlineDriveTaskTracker tracker = (OnlineDriveTaskTracker)currentTask.getTaskTracker();
198                return filterValidLinkTimePair(tracker.getDiversionPoint(), veh);
199        }
200
201        private static LinkTimePair filterValidLinkTimePair(LinkTimePair pair, Vehicle veh) {
202                return pair.time >= veh.getServiceEndTime() ? null : pair;
203        }
204
205        private static LinkTimePair createValidLinkTimePair(Link link, double time, Vehicle veh) {
206                return time >= veh.getServiceEndTime() ? null : new LinkTimePair(link, time);
207        }

211        public void scheduleRequest(Vehicle vehicle, TaxiRequest request, VrpPathWithTravelData vrpPath) {
212                if (request.getStatus() != TaxiRequestStatus.UNPLANNED) {
213                        throw new IllegalStateException();
214                }
215
216                Schedule schedule = vehicle.getSchedule();
217                divertOrAppendDrive(schedule, vrpPath);
218
219                double pickupEndTime = Math.max(vrpPath.getArrivalTime(), request.getEarliestStartTime())
220                                + taxiCfg.getPickupDuration();
221                schedule.addTask(new TaxiPickupTask(vrpPath.getArrivalTime(), pickupEndTime, request));
222
223                if (taxiCfg.isDestinationKnown()) {
224                        appendOccupiedDriveAndDropoff(schedule);
225                        appendTasksAfterDropoff(vehicle);
226                }
227        }
228
229        protected void divertOrAppendDrive(Schedule schedule, VrpPathWithTravelData vrpPath) {
230                TaxiTask lastTask = (TaxiTask)Schedules.getLastTask(schedule);
231                switch (lastTask.getTaxiTaskType()) {
232                        case EMPTY_DRIVE:
233                                divertDrive((TaxiEmptyDriveTask)lastTask, vrpPath);
234                                return;
235
236                        case STAY:
237                                scheduleDrive(schedule, (TaxiStayTask)lastTask, vrpPath);
238                                return;
239
240                        default:
241                                throw new IllegalStateException();
242                }
243        }
244
245        protected void divertDrive(TaxiEmptyDriveTask lastTask, VrpPathWithTravelData vrpPath) {
246                if (!taxiCfg.isVehicleDiversion()) {
247                        throw new IllegalStateException();
248                }
249
250                ((OnlineDriveTaskTracker)lastTask.getTaskTracker()).divertPath(vrpPath);
251        }
252
253        protected void scheduleDrive(Schedule schedule, TaxiStayTask lastTask, VrpPathWithTravelData vrpPath) {
254                switch (lastTask.getStatus()) {
255                        case PLANNED:
256                                if (lastTask.getBeginTime() == vrpPath.getDepartureTime()) { // waiting for 0 seconds!!!
257                                        schedule.removeLastTask();// remove WaitTask
258                                } else {
259                                        // actually this WAIT task will not be performed
260                                        lastTask.setEndTime(vrpPath.getDepartureTime());// shortening the WAIT task
261                                }
262                                break;
263
264                        case STARTED:
265                                lastTask.setEndTime(vrpPath.getDepartureTime());// shortening the WAIT task
266                                break;
267
268                        case PERFORMED:
269                        default:
270                                throw new IllegalStateException();
271                }
272
273                if (vrpPath.getLinkCount() > 1) {
274                        schedule.addTask(new TaxiEmptyDriveTask(vrpPath));
275                }
276        }

278        /**
279         * If diversion is enabled, this method must be called after scheduling in order to make sure that no vehicle is
280         * moving aimlessly.
281         * <p>
282         * </p>
283         * The reason: the destination/goal had been removed before scheduling (e.g. by calling the
284         * {@link #removeAwaitingRequestsFromAllSchedules()} method)
285         */
286        public void stopAllAimlessDriveTasks() {
287                for (Vehicle veh : fleet.getVehicles().values()) {
288                        if (getImmediateDiversion(veh) != null) {
289                                stopVehicle(veh);
290                        }
291                }
292        }

294        public void stopVehicle(Vehicle vehicle) {
295                if (!taxiCfg.isVehicleDiversion()) {
296                        throw new RuntimeException("Diversion must be on");
297                }
298
299                Schedule schedule = vehicle.getSchedule();
300                TaxiEmptyDriveTask driveTask = (TaxiEmptyDriveTask)Schedules.getLastTask(schedule);
301
302                OnlineDriveTaskTracker tracker = (OnlineDriveTaskTracker)driveTask.getTaskTracker();
303                LinkTimePair stopPoint = tracker.getDiversionPoint();
304                tracker.divertPath(
305                                new VrpPathWithTravelDataImpl(stopPoint.time, 0, new Link[] { stopPoint.link }, new double[] { 0 }));
306
307                appendStayTask(vehicle);
308        }

310        /**
311         * Check and decide if the schedule should be updated due to if vehicle is Update timings (i.e. beginTime and
312         * endTime) of all tasks in the schedule.
313         */
314        public void updateBeforeNextTask(Vehicle vehicle) {
315                Schedule schedule = vehicle.getSchedule();
316                // Assumption: there is no delay as long as the schedule has not been started (PLANNED)
317                if (schedule.getStatus() != ScheduleStatus.STARTED) {
318                        return;
319                }
320
321                updateTimelineImpl(vehicle, timer.getTimeOfDay());
322
323                if (!taxiCfg.isDestinationKnown()) {
324                        TaxiTask currentTask = (TaxiTask)schedule.getCurrentTask();
325                        if (currentTask.getTaxiTaskType() == TaxiTaskType.PICKUP) {
326                                appendOccupiedDriveAndDropoff(schedule);
327                                appendTasksAfterDropoff(vehicle);
328                        }
329                }
330        }
331
332        protected void appendOccupiedDriveAndDropoff(Schedule schedule) {
333                TaxiPickupTask pickupStayTask = (TaxiPickupTask)Schedules.getLastTask(schedule);
334
335                // add DELIVERY after SERVE
336                TaxiRequest req = ((TaxiPickupTask)pickupStayTask).getRequest();
337                Link reqFromLink = req.getFromLink();
338                Link reqToLink = req.getToLink();
339                double t3 = pickupStayTask.getEndTime();
340
341                VrpPathWithTravelData path = calcPath(reqFromLink, reqToLink, t3);
342                schedule.addTask(new TaxiOccupiedDriveTask(path, req));
343
344                double t4 = path.getArrivalTime();
345                double t5 = t4 + taxiCfg.getDropoffDuration();
346                schedule.addTask(new TaxiDropoffTask(t4, t5, req));
347        }
348
349        protected VrpPathWithTravelData calcPath(Link fromLink, Link toLink, double departureTime) {
350                return VrpPaths.calcAndCreatePath(fromLink, toLink, departureTime, router, travelTime);
351        }
352
353        protected void appendTasksAfterDropoff(Vehicle vehicle) {
354                appendStayTask(vehicle);
355        }
356
357        protected void appendStayTask(Vehicle vehicle) {
358                Schedule schedule = vehicle.getSchedule();
359                double tBegin = schedule.getEndTime();
360                double tEnd = Math.max(tBegin, vehicle.getServiceEndTime());// even 0-second WAIT
361                Link link = Schedules.getLastLinkInSchedule(vehicle);
362                schedule.addTask(new TaxiStayTask(tBegin, tEnd, link));
363        }

365        public void updateTimeline(Vehicle vehicle) {
366                Schedule schedule = vehicle.getSchedule();
367                if (schedule.getStatus() != ScheduleStatus.STARTED) {
368                        return;
369                }
370
371                double predictedEndTime = TaskTrackers.predictEndTime(schedule.getCurrentTask(), timer.getTimeOfDay());
372                updateTimelineImpl(vehicle, predictedEndTime);
373        }
374
375        private void updateTimelineImpl(Vehicle vehicle, double newEndTime) {
376                Schedule schedule = vehicle.getSchedule();
377                Task currentTask = schedule.getCurrentTask();
378                if (currentTask.getEndTime() == newEndTime) {
379                        return;
380                }
381
382                currentTask.setEndTime(newEndTime);
383
384                List<? extends Task> tasks = schedule.getTasks();
385                int startIdx = currentTask.getTaskIdx() + 1;
386                double newBeginTime = newEndTime;
387
388                for (int i = startIdx; i < tasks.size(); i++) {
389                        TaxiTask task = (TaxiTask)tasks.get(i);
390                        double calcEndTime = calcNewEndTime(vehicle, task, newBeginTime);
391
392                        if (calcEndTime == Time.UNDEFINED_TIME) {
393                                schedule.removeTask(task);
394                                i--;
395                        } else if (calcEndTime < newBeginTime) {// 0 s is fine (e.g. last 'wait')
396                                throw new IllegalStateException();
397                        } else {
398                                task.setBeginTime(newBeginTime);
399                                task.setEndTime(calcEndTime);
400                                newBeginTime = calcEndTime;
401                        }
402                }
403        }
404
405        protected double calcNewEndTime(Vehicle vehicle, TaxiTask task, double newBeginTime) {
406                switch (task.getTaxiTaskType()) {
407                        case STAY: {
408                                if (Schedules.getLastTask(vehicle.getSchedule()).equals(task)) {// last task
409                                        // even if endTime=beginTime, do not remove this task!!! A taxi schedule should end with WAIT
410                                        return Math.max(newBeginTime, vehicle.getServiceEndTime());
411                                } else {
412                                        // if this is not the last task then some other task (e.g. DRIVE or PICKUP)
413                                        // must have been added at time submissionTime <= t
414                                        double oldEndTime = task.getEndTime();
415                                        if (oldEndTime <= newBeginTime) {// may happen if the previous task is delayed
416                                                return Time.UNDEFINED_TIME;// remove the task
417                                        } else {
418                                                return oldEndTime;
419                                        }
420                                }
421                        }
422
423                        case EMPTY_DRIVE:
424                        case OCCUPIED_DRIVE: {
425                                // cannot be shortened/lengthen, therefore must be moved forward/backward
426                                VrpPathWithTravelData path = (VrpPathWithTravelData)((DriveTask)task).getPath();
427                                // TODO one may consider recalculation of SP!!!!
428                                return newBeginTime + path.getTravelTime();
429                        }
430
431                        case PICKUP: {
432                                double t0 = ((TaxiPickupTask)task).getRequest().getEarliestStartTime();
433                                // the actual pickup starts at max(t, t0)
434                                return Math.max(newBeginTime, t0) + taxiCfg.getPickupDuration();
435                        }
436                        case DROPOFF: {
437                                // cannot be shortened/lengthen, therefore must be moved forward/backward
438                                return newBeginTime + taxiCfg.getDropoffDuration();
439                        }
440
441                        default:
442                                throw new IllegalStateException();
443                }
444        }

448        private List<TaxiRequest> removedRequests;
449
450        /**
451         * Awaiting == unpicked-up, i.e. requests with status PLANNED or TAXI_DISPATCHED See {@link TaxiRequestStatus}
452         */
453        public List<TaxiRequest> removeAwaitingRequestsFromAllSchedules() {
454                removedRequests = new ArrayList<>();
455                for (Vehicle veh : fleet.getVehicles().values()) {
456                        removeAwaitingRequestsImpl(veh);
457                }
458
459                return removedRequests;
460        }
461
462        public List<TaxiRequest> removeAwaitingRequests(Vehicle vehicle) {
463                removedRequests = new ArrayList<>();
464                removeAwaitingRequestsImpl(vehicle);
465                return removedRequests;
466        }
467
468        private void removeAwaitingRequestsImpl(Vehicle vehicle) {
469                Schedule schedule = vehicle.getSchedule();
470                switch (schedule.getStatus()) {
471                        case STARTED:
472                                Integer unremovableTasksCount = countUnremovablePlannedTasks(schedule);
473                                if (unremovableTasksCount == null) {
474                                        return;
475                                }
476
477                                int newLastTaskIdx = schedule.getCurrentTask().getTaskIdx() + unremovableTasksCount;
478                                removePlannedTasks(vehicle, newLastTaskIdx);
479                                cleanupScheduleAfterTaskRemoval(vehicle);
480                                return;
481
482                        case PLANNED:
483                                removePlannedTasks(vehicle, -1);
484                                cleanupScheduleAfterTaskRemoval(vehicle);
485                                return;
486
487                        case COMPLETED:
488                                return;
489
490                        case UNPLANNED:
491                                throw new IllegalStateException();
492                }
493        }
494
495        protected Integer countUnremovablePlannedTasks(Schedule schedule) {
496                TaxiTask currentTask = (TaxiTask)schedule.getCurrentTask();
497                switch (currentTask.getTaxiTaskType()) {
498                        case PICKUP:
499                                return taxiCfg.isDestinationKnown() ? 2 : null;
500
501                        case OCCUPIED_DRIVE:
502                                return 1;
503
504                        case EMPTY_DRIVE:
505                                if (taxiCfg.isVehicleDiversion()) {
506                                        return 0;
507                                }
508
509                                if (((TaxiTask)Schedules.getNextTask(schedule)).getTaxiTaskType() == TaxiTaskType.PICKUP) {
510                                        // if no diversion and driving to pick up sb then serve that request
511                                        return taxiCfg.isDestinationKnown() ? 3 : null;
512                                }
513
514                                // potentially: driving back to the rank (e.g. to charge batteries)
515                                throw new RuntimeException("Currently won't happen");
516
517                        case DROPOFF:
518                        case STAY:
519                                return 0;
520
521                        default:
522                                throw new RuntimeException();
523                }
524        }
525
526        protected void removePlannedTasks(Vehicle vehicle, int newLastTaskIdx) {
527                Schedule schedule = vehicle.getSchedule();
528                List<? extends Task> tasks = schedule.getTasks();
529                for (int i = schedule.getTaskCount() - 1; i > newLastTaskIdx; i--) {
530                        TaxiTask task = (TaxiTask)tasks.get(i);
531                        schedule.removeTask(task);
532                        taskRemovedFromSchedule(vehicle, task);
533                }
534        }
535
536        protected void taskRemovedFromSchedule(Vehicle vehicle, TaxiTask task) {
537                if (task instanceof TaxiTaskWithRequest) {
538                        TaxiRequest request = ((TaxiTaskWithRequest)task).getRequest();
539
540                        if (task.getTaxiTaskType() == TaxiTaskType.PICKUP) {
541                                request.setPickupTask(null);
542                                removedRequests.add(request);
543                        } else if (task.getTaxiTaskType() == TaxiTaskType.DROPOFF) {
544                                request.setDropoffTask(null);
545                        }
546                }
547        }
548
549        // only for planned/started schedule
550        private void cleanupScheduleAfterTaskRemoval(Vehicle vehicle) {
551                Schedule schedule = vehicle.getSchedule();
552                if (schedule.getStatus() == ScheduleStatus.UNPLANNED) {
553                        schedule.addTask(new TaxiStayTask(vehicle.getServiceBeginTime(), vehicle.getServiceEndTime(),
554                                        vehicle.getStartLink()));
555                        return;
556                }
557                // else: PLANNED, STARTED
558
559                TaxiTask lastTask = (TaxiTask)Schedules.getLastTask(schedule);
560                double tBegin = schedule.getEndTime();
561                double tEnd = Math.max(tBegin, vehicle.getServiceEndTime());
562
563                switch (lastTask.getTaxiTaskType()) {
564                        case STAY:
565                                lastTask.setEndTime(tEnd);
566                                return;
567
568                        case DROPOFF:
569                                Link link = Schedules.getLastLinkInSchedule(vehicle);
570                                schedule.addTask(new TaxiStayTask(tBegin, tEnd, link));
571                                return;
572
573                        case EMPTY_DRIVE:
574                                if (!taxiCfg.isVehicleDiversion()) {
575                                        throw new RuntimeException("Currently won't happen");
576                                }
577
578                                // if diversion -- no STAY afterwards
579                                return;
580
581                        default:
582                                throw new RuntimeException();
583                }
584        }
