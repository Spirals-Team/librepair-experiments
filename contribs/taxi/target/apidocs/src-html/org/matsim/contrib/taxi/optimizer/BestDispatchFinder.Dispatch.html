028/**
029 * @author michalm
030 */
031public class BestDispatchFinder {
032        public static class Dispatch<D> {
033                public final Vehicle vehicle;
034                public final D destination;
035                public final VrpPathWithTravelData path;
036
037                public Dispatch(Vehicle vehicle, D destination, VrpPathWithTravelData path) {
038                        this.vehicle = vehicle;
039                        this.destination = destination;
040                        this.path = path;
041                }
042        }
043
044        // typically we search through the 20-40 nearest requests/vehicles
045        private static final int EXPECTED_NEIGHBOURHOOD_SIZE = 40;
046
047        private final MultiNodeDijkstra router;
048        private final TaxiScheduleInquiry scheduleInquiry;
049        private final MobsimTimer timer;
050        private final TravelTime travelTime;
051
052        public BestDispatchFinder(TaxiScheduleInquiry scheduleInquiry, Network network, MobsimTimer mobsimTimer,
053                        TravelTime travelTime, TravelDisutility travelDisutility) {
054                this.scheduleInquiry = scheduleInquiry;
055                this.timer = mobsimTimer;
056                this.travelTime = travelTime;
057
058                router = (MultiNodeDijkstra)new FastMultiNodeDijkstraFactory(false).createPathCalculator(network,
059                                travelDisutility, travelTime);
060        }
061
062        // for immediate requests only
063        // minimize TW
064        public Dispatch<TaxiRequest> findBestVehicleForRequest(TaxiRequest req, Stream<? extends Vehicle> vehicles) {
065                return findBestVehicle(req, vehicles, r -> r.getFromLink());
066        }
067
068        // We use many-to-one forward search. Therefore, we cannot assess all vehicles.
069        // However, that would be possible if one-to-many backward search were used instead.
070        // TODO intuitively, many-to-one is slower, some performance tests needed before switching to
071        // one-to-many
072        public <D> Dispatch<D> findBestVehicle(D destination, Stream<? extends Vehicle> vehicles,
073                        LinkProvider<D> destinationToLink) {
074                double currTime = timer.getTimeOfDay();
075                Link toLink = destinationToLink.apply(destination);
076                Node toNode = toLink.getFromNode();
077
078                Map<Id<Node>, Vehicle> nodeToVehicle = Maps.newHashMapWithExpectedSize(EXPECTED_NEIGHBOURHOOD_SIZE);
079                Map<Id<Node>, InitialNode> initialNodes = Maps.newHashMapWithExpectedSize(EXPECTED_NEIGHBOURHOOD_SIZE);
080                vehicles.forEach(veh -> {
081                        LinkTimePair departure = scheduleInquiry.getImmediateDiversionOrEarliestIdleness(veh);
082                        if (departure != null) {
083                                Node vehNode;
084                                double delay = departure.time - currTime;
085                                if (departure.link == toLink) {
086                                        // hack: we are basically there (on the same link), so let's pretend vehNode == toNode
087                                        vehNode = toNode;
088                                } else {
089                                        vehNode = departure.link.getToNode();
090
091                                        // simplified, but works for taxis, since pickup trips are short (about 5 mins)
092                                        delay += 1 + toLink.getFreespeed(departure.time);
093                                }
094
095                                InitialNode existingInitialNode = initialNodes.get(vehNode.getId());
096                                if (existingInitialNode == null || existingInitialNode.initialCost > delay) {
097                                        InitialNode newInitialNode = new InitialNode(vehNode, delay, delay);
098                                        initialNodes.put(vehNode.getId(), newInitialNode);
099                                        nodeToVehicle.put(vehNode.getId(), veh);
100                                }
101                        }
102                });
103
104                if (initialNodes.isEmpty()) {
105                        return null;
106                }
107
108                RoutingNetworkImaginaryNode fromNodes = new RoutingNetworkImaginaryNode(initialNodes.values());
109
110                Path path = router.calcLeastCostPath(fromNodes, toNode, currTime, null, null);
111                // the calculated path contains real nodes (no imaginary/initial nodes),
112                // the time and cost are of real travel (between the first and last real node)
113                // (no initial times/costs for imaginary<->initial are included)
114                Node fromNode = path.nodes.get(0);
115                Vehicle bestVehicle = nodeToVehicle.get(fromNode.getId());
116                LinkTimePair bestDeparture = scheduleInquiry.getImmediateDiversionOrEarliestIdleness(bestVehicle);
117
118                VrpPathWithTravelData vrpPath = VrpPaths.createPath(bestDeparture.link, toLink, bestDeparture.time, path,
119                                travelTime);
120                return new Dispatch<>(bestVehicle, destination, vrpPath);
121        }
122
123        // for immediate requests only
124        // minimize TP
125        public Dispatch<TaxiRequest> findBestRequestForVehicle(Vehicle veh, Stream<TaxiRequest> unplannedRequests) {
126                return findBestDestination(veh, unplannedRequests, r -> r.getFromLink());
127        }
128
129        public <D> Dispatch<D> findBestDestination(Vehicle veh, Stream<D> destinations,
130                        LinkProvider<D> destinationToLink) {
131                LinkTimePair departure = scheduleInquiry.getImmediateDiversionOrEarliestIdleness(veh);
132                Node fromNode = departure.link.getToNode();
133
134                Map<Id<Node>, D> nodeToDestination = Maps.newHashMapWithExpectedSize(EXPECTED_NEIGHBOURHOOD_SIZE);
135                Map<Id<Node>, InitialNode> initialNodes = Maps.newHashMapWithExpectedSize(EXPECTED_NEIGHBOURHOOD_SIZE);
136                for (D loc : (Iterable<D>)destinations::iterator) {
137                        Link link = destinationToLink.apply(loc);
138
139                        if (departure.link == link) {
140                                return new Dispatch<>(veh, loc, VrpPaths.createZeroLengthPath(departure.link, departure.time));
141                        }
142
143                        Id<Node> locNodeId = link.getFromNode().getId();
144
145                        if (!initialNodes.containsKey(locNodeId)) {
146                                // simplified, but works for taxis, since pickup trips are short (about 5 mins)
147                                double delayAtLastLink = link.getFreespeed(departure.time);
148
149                                // works most fair (FIFO) if unplannedRequests (=destinations) are sorted by T0 (ascending)
150                                InitialNode newInitialNode = new InitialNode(link.getFromNode(), delayAtLastLink, delayAtLastLink);
151                                initialNodes.put(locNodeId, newInitialNode);
152                                nodeToDestination.put(locNodeId, loc);
153                        }
154                }
155
156                RoutingNetworkImaginaryNode toNodes = new RoutingNetworkImaginaryNode(initialNodes.values());
157
158                // calc path for departure.time+1 (we need 1 second to move over the node)
159                Path path = router.calcLeastCostPath(fromNode, toNodes, departure.time + 1, null, null);
160
161                // the calculated path contains real nodes (no imaginary/initial nodes),
162                // the time and cost are of real travel (between the first and last real node)
163                // (no initial times/costs for imaginary<->initial are included)
164                Node toNode = path.nodes.get(path.nodes.size() - 1);
165                D bestDestination = nodeToDestination.get(toNode.getId());
166                VrpPathWithTravelData vrpPath = VrpPaths.createPath(departure.link, destinationToLink.apply(bestDestination),
167                                departure.time, path, travelTime);
168                return new Dispatch<>(veh, bestDestination, vrpPath);
169        }
