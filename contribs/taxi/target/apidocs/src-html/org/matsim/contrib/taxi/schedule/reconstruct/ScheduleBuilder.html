020package org.matsim.contrib.taxi.schedule.reconstruct;
021
022import org.matsim.api.core.v01.Id;
023import org.matsim.api.core.v01.network.Link;
024import org.matsim.api.core.v01.population.Person;
025import org.matsim.contrib.dvrp.data.FleetImpl;
026import org.matsim.contrib.dvrp.data.Vehicle;
027import org.matsim.contrib.dvrp.data.VehicleImpl;
028import org.matsim.contrib.dvrp.path.VrpPathWithTravelData;
029import org.matsim.contrib.taxi.data.TaxiRequest;
030import org.matsim.contrib.taxi.schedule.TaxiDropoffTask;
031import org.matsim.contrib.taxi.schedule.TaxiEmptyDriveTask;
032import org.matsim.contrib.taxi.schedule.TaxiOccupiedDriveTask;
033import org.matsim.contrib.taxi.schedule.TaxiPickupTask;
034import org.matsim.contrib.taxi.schedule.TaxiStayTask;
035import org.matsim.contrib.taxi.schedule.reconstruct.StayRecorder.Stay;
036import org.matsim.contrib.taxi.vrpagent.TaxiActionCreator;
037
038public class ScheduleBuilder {
039        private VehicleImpl vehicle;
040        private TaxiRequest currentRequest = null;
041
042        ScheduleBuilder(FleetImpl fleet, Id<Person> personId, Link link, double t0) {
043                vehicle = new VehicleImpl(Id.create(personId, Vehicle.class), link, 1, t0, Double.NaN);
044                fleet.addVehicle(vehicle);
045        }
046
047        void addDrive(VrpPathWithTravelData vrpPath) {
048                if (currentRequest != null) {
049                        vehicle.getSchedule().addTask(new TaxiOccupiedDriveTask(vrpPath, currentRequest));
050                } else {
051                        vehicle.getSchedule().addTask(new TaxiEmptyDriveTask(vrpPath));
052                }
053        }
054
055        void addStay(Stay stay) {
056                switch (stay.activityType) {
057                        case TaxiActionCreator.STAY_ACTIVITY_TYPE:
058                        case "Stay":// old naming (TODO to be removed soon)
059                                vehicle.getSchedule().addTask(new TaxiStayTask(stay.startTime, stay.endTime, stay.link));
060                                return;
061
062                        case TaxiActionCreator.PICKUP_ACTIVITY_TYPE:
063                        case "PassengerPickup":// old naming (TODO to be removed soon)
064                                vehicle.getSchedule().addTask(new TaxiPickupTask(stay.startTime, stay.endTime, currentRequest));
065                                return;
066
067                        case TaxiActionCreator.DROPOFF_ACTIVITY_TYPE:
068                        case "PassengerDropoff":// old naming (TODO to be removed soon)
069                                // TODO setting 'toLink' should be moved to RequestRecorder once the events are re-ordered
070                                ((TaxiRequestWithModifiableToLink)currentRequest).toLink = stay.link;
071                                vehicle.getSchedule().addTask(new TaxiDropoffTask(stay.startTime, stay.endTime, currentRequest));
072
073                                currentRequest = null;
074                                return;
075                }
076        }
077
078        void addRequest(TaxiRequest request) {
079                if (currentRequest != null) {
080                        throw new IllegalStateException("Currently only one passenger per vehicle");
081                }
082                currentRequest = request;
083        }
084
085        void endSchedule(double endTime) {
086                if (currentRequest != null) {
087                        throw new IllegalStateException();
088                }
089
090                vehicle.setServiceEndTime(endTime);
091                vehicle = null;// just to make sure no modifications will be made
092        }
093
094        boolean isScheduleBuilt() {
095                return vehicle == null;
096        }
097}
