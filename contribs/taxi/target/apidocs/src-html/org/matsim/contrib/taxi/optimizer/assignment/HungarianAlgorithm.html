045/**
046 * An implementation of the Hungarian algorithm for solving the assignment problem. An instance of the assignment
047 * problem consists of a number of workers along with a number of jobs and a cost matrix which gives the cost of
048 * assigning the i'th worker to the j'th job at position (i, j). The goal is to find an assignment of workers to jobs so
049 * that no job is assigned more than one worker and so that no worker is assigned to more than one job in such a manner
050 * so as to minimize the total cost of completing the jobs.
051 * <p>
052 * An assignment for a cost matrix that has more workers than jobs will necessarily include unassigned workers,
053 * indicated by an assignment value of -1; in no other circumstance will there be unassigned workers. Similarly, an
054 * assignment for a cost matrix that has more jobs than workers will necessarily include unassigned jobs; in no other
055 * circumstance will there be unassigned jobs. For completeness, an assignment for a square cost matrix will give
056 * exactly one unique worker to each job.
057 * <p>
058 * This version of the Hungarian algorithm runs in time O(n^3), where n is the maximum among the number of workers and
059 * the number of jobs.
060 * 
061 * @author Kevin L. Stern
062 */
063public class HungarianAlgorithm {
064        private final double[][] costMatrix;
065        private final int rows, cols, dim;
066        private final double[] labelByWorker, labelByJob;
067        private final int[] minSlackWorkerByJob;
068        private final double[] minSlackValueByJob;
069        private final int[] matchJobByWorker, matchWorkerByJob;
070        private final int[] parentWorkerByCommittedJob;
071        private final boolean[] committedWorkers;
072
073        /**
074         * Construct an instance of the algorithm.
075         * 
076         * @param costMatrix
077         *            the cost matrix, where matrix[i][j] holds the cost of assigning worker i to job j, for all i, j. The
078         *            cost matrix must not be irregular in the sense that all rows must be the same length.
079         */
080        public HungarianAlgorithm(double[][] costMatrix) {
081                this.dim = Math.max(costMatrix.length, costMatrix[0].length);
082                this.rows = costMatrix.length;
083                this.cols = costMatrix[0].length;
084                this.costMatrix = new double[this.dim][this.dim];
085
086                for (int w = 0; w < this.dim; w++) {
087                        if (w < costMatrix.length) {
088                                if (costMatrix[w].length != this.cols) {
089                                        throw new IllegalArgumentException("Irregular cost matrix");
090                                }
091                                this.costMatrix[w] = Arrays.copyOf(costMatrix[w], this.dim);
092                        } else {
093                                this.costMatrix[w] = new double[this.dim];
094                        }
095                }
096                labelByWorker = new double[this.dim];
097                labelByJob = new double[this.dim];
098                minSlackWorkerByJob = new int[this.dim];
099                minSlackValueByJob = new double[this.dim];
100                committedWorkers = new boolean[this.dim];
101                parentWorkerByCommittedJob = new int[this.dim];
102                matchJobByWorker = new int[this.dim];
103                Arrays.fill(matchJobByWorker, -1);
104                matchWorkerByJob = new int[this.dim];
105                Arrays.fill(matchWorkerByJob, -1);
106        }
107
108        /**
109         * Compute an initial feasible solution by assigning zero labels to the workers and by assigning to each job a label
110         * equal to the minimum cost among its incident edges.
111         */
112        protected void computeInitialFeasibleSolution() {
113                for (int j = 0; j < dim; j++) {
114                        labelByJob[j] = Double.POSITIVE_INFINITY;
115                }
116                for (int w = 0; w < dim; w++) {
117                        for (int j = 0; j < dim; j++) {
118                                if (costMatrix[w][j] < labelByJob[j]) {
119                                        labelByJob[j] = costMatrix[w][j];
120                                }
121                        }
122                }
123        }
124
125        /**
126         * Execute the algorithm.
127         * 
128         * @return the minimum cost matching of workers to jobs based upon the provided cost matrix. A matching value of -1
129         *         indicates that the corresponding worker is unassigned.
130         */
131        public int[] execute() {
132                /*
133                 * Heuristics to improve performance: Reduce rows and columns by their smallest element, compute an initial
134                 * non-zero dual feasible solution and create a greedy matching from workers to jobs of the cost matrix.
135                 */
136                reduce();
137                computeInitialFeasibleSolution();
138                greedyMatch();
139
140                int w = fetchUnmatchedWorker();
141                while (w < dim) {
142                        initializePhase(w);
143                        executePhase();
144                        w = fetchUnmatchedWorker();
145                }
146                int[] result = Arrays.copyOf(matchJobByWorker, rows);
147                for (w = 0; w < result.length; w++) {
148                        if (result[w] >= cols) {
149                                result[w] = -1;
150                        }
151                }
152                return result;
153        }
154
155        /**
156         * Execute a single phase of the algorithm. A phase of the Hungarian algorithm consists of building a set of
157         * committed workers and a set of committed jobs from a root unmatched worker by following alternating
158         * unmatched/matched zero-slack edges. If an unmatched job is encountered, then an augmenting path has been found
159         * and the matching is grown. If the connected zero-slack edges have been exhausted, the labels of committed workers
160         * are increased by the minimum slack among committed workers and non-committed jobs to create more zero-slack edges
161         * (the labels of committed jobs are simultaneously decreased by the same amount in order to maintain a feasible
162         * labeling).
163         * <p>
164         * The runtime of a single phase of the algorithm is O(n^2), where n is the dimension of the internal square cost
165         * matrix, since each edge is visited at most once and since increasing the labeling is accomplished in time O(n) by
166         * maintaining the minimum slack values among non-committed jobs. When a phase completes, the matching will have
167         * increased in size.
168         */
169        protected void executePhase() {
170                while (true) {
171                        int minSlackWorker = -1, minSlackJob = -1;
172                        double minSlackValue = Double.POSITIVE_INFINITY;
173                        for (int j = 0; j < dim; j++) {
174                                if (parentWorkerByCommittedJob[j] == -1) {
175                                        if (minSlackValueByJob[j] < minSlackValue) {
176                                                minSlackValue = minSlackValueByJob[j];
177                                                minSlackWorker = minSlackWorkerByJob[j];
178                                                minSlackJob = j;
179                                        }
180                                }
181                        }
182                        if (minSlackValue > 0) {
183                                updateLabeling(minSlackValue);
184                        }
185                        parentWorkerByCommittedJob[minSlackJob] = minSlackWorker;
186                        if (matchWorkerByJob[minSlackJob] == -1) {
187                                /*
188                                 * An augmenting path has been found.
189                                 */
190                                int committedJob = minSlackJob;
191                                int parentWorker = parentWorkerByCommittedJob[committedJob];
192                                while (true) {
193                                        int temp = matchJobByWorker[parentWorker];
194                                        match(parentWorker, committedJob);
195                                        committedJob = temp;
196                                        if (committedJob == -1) {
197                                                break;
198                                        }
199                                        parentWorker = parentWorkerByCommittedJob[committedJob];
200                                }
201                                return;
202                        } else {
203                                /*
204                                 * Update slack values since we increased the size of the committed workers set.
205                                 */
206                                int worker = matchWorkerByJob[minSlackJob];
207                                committedWorkers[worker] = true;
208                                for (int j = 0; j < dim; j++) {
209                                        if (parentWorkerByCommittedJob[j] == -1) {
210                                                double slack = costMatrix[worker][j] - labelByWorker[worker] - labelByJob[j];
211                                                if (minSlackValueByJob[j] > slack) {
212                                                        minSlackValueByJob[j] = slack;
213                                                        minSlackWorkerByJob[j] = worker;
214                                                }
215                                        }
216                                }
217                        }
218                }
219        }
220
221        /**
222         * @return the first unmatched worker or {@link #dim} if none.
223         */
224        protected int fetchUnmatchedWorker() {
225                int w;
226                for (w = 0; w < dim; w++) {
227                        if (matchJobByWorker[w] == -1) {
228                                break;
229                        }
230                }
231                return w;
232        }
233
234        /**
235         * Find a valid matching by greedily selecting among zero-cost matchings. This is a heuristic to jump-start the
236         * augmentation algorithm.
237         */
238        protected void greedyMatch() {
239                for (int w = 0; w < dim; w++) {
240                        for (int j = 0; j < dim; j++) {
241                                if (matchJobByWorker[w] == -1 && matchWorkerByJob[j] == -1
242                                                && costMatrix[w][j] - labelByWorker[w] - labelByJob[j] == 0) {
243                                        match(w, j);
244                                }
245                        }
246                }
247        }
248
249        /**
250         * Initialize the next phase of the algorithm by clearing the committed workers and jobs sets and by initializing
251         * the slack arrays to the values corresponding to the specified root worker.
252         * 
253         * @param w
254         *            the worker at which to root the next phase.
255         */
256        protected void initializePhase(int w) {
257                Arrays.fill(committedWorkers, false);
258                Arrays.fill(parentWorkerByCommittedJob, -1);
259                committedWorkers[w] = true;
260                for (int j = 0; j < dim; j++) {
261                        minSlackValueByJob[j] = costMatrix[w][j] - labelByWorker[w] - labelByJob[j];
262                        minSlackWorkerByJob[j] = w;
263                }
264        }
265
266        /**
267         * Helper method to record a matching between worker w and job j.
268         */
269        protected void match(int w, int j) {
270                matchJobByWorker[w] = j;
271                matchWorkerByJob[j] = w;
272        }
273
274        /**
275         * Reduce the cost matrix by subtracting the smallest element of each row from all elements of the row as well as
276         * the smallest element of each column from all elements of the column. Note that an optimal assignment for a
277         * reduced cost matrix is optimal for the original cost matrix.
278         */
279        protected void reduce() {
280                for (int w = 0; w < dim; w++) {
281                        double min = Double.POSITIVE_INFINITY;
282                        for (int j = 0; j < dim; j++) {
283                                if (costMatrix[w][j] < min) {
284                                        min = costMatrix[w][j];
285                                }
286                        }
287                        for (int j = 0; j < dim; j++) {
288                                costMatrix[w][j] -= min;
289                        }
290                }
291                double[] min = new double[dim];
292                for (int j = 0; j < dim; j++) {
293                        min[j] = Double.POSITIVE_INFINITY;
294                }
295                for (int w = 0; w < dim; w++) {
296                        for (int j = 0; j < dim; j++) {
297                                if (costMatrix[w][j] < min[j]) {
298                                        min[j] = costMatrix[w][j];
299                                }
300                        }
301                }
302                for (int w = 0; w < dim; w++) {
303                        for (int j = 0; j < dim; j++) {
304                                costMatrix[w][j] -= min[j];
305                        }
306                }
307        }
308
309        /**
310         * Update labels with the specified slack by adding the slack value for committed workers and by subtracting the
311         * slack value for committed jobs. In addition, update the minimum slack values appropriately.
312         */
313        protected void updateLabeling(double slack) {
314                for (int w = 0; w < dim; w++) {
315                        if (committedWorkers[w]) {
316                                labelByWorker[w] += slack;
317                        }
318                }
319                for (int j = 0; j < dim; j++) {
320                        if (parentWorkerByCommittedJob[j] != -1) {
321                                labelByJob[j] -= slack;
322                        } else {
323                                minSlackValueByJob[j] -= slack;
324                        }
325                }
326        }
