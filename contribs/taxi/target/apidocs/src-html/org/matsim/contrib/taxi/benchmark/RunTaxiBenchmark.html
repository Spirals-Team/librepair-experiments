042/**
043 * For a fair and consistent benchmarking of taxi dispatching algorithms we assume that link travel times are
044 * deterministic. To simulate this property, we remove (1) all other traffic, and (2) link capacity constraints (e.g. by
045 * increasing the capacities by 100+ times), as a result all vehicles move with the free-flow speed (which is the
046 * effective speed).
047 * <p>
048 * </p>
049 * To model the impact of traffic, we can use a time-variant network, where we specify different free-flow speeds for
050 * each link over time. The default approach is to specify free-flow speeds in each time interval (usually 15 minutes).
051 */
052public class RunTaxiBenchmark {
053        public static void run(String configFile, int runs) {
054                Config config = ConfigUtils.loadConfig(configFile, new TaxiConfigGroup(), new DvrpConfigGroup());
055                createControler(config, runs).run();
056        }
057
058        public static Controler createControler(Config config, int runs) {
059                config.controler().setLastIteration(runs - 1);
060                config.controler().setDumpDataAtEnd(false);
061                config.controler().setWriteEventsInterval(0);
062                config.controler().setWritePlansInterval(0);
063                config.controler().setCreateGraphs(false);
064
065                DvrpConfigGroup.get(config).setNetworkMode(null);// to switch off network filtering
066                config.addConfigConsistencyChecker(new TaxiBenchmarkConfigConsistencyChecker());
067                config.checkConsistency();
068
069                Scenario scenario = loadBenchmarkScenario(config, 15 * 60, 30 * 3600);
070
071                Controler controler = new Controler(scenario);
072                controler.setModules(new DvrpBenchmarkControlerModule());
073                controler.addOverridingModule(new DvrpBenchmarkModule(
074                                cfg -> TaxiDvrpModules.createModuleForQSimPlugin(DefaultTaxiOptimizerProvider.class),
075                                Collections.singleton(TaxiOptimizer.class)));
076
077                controler.addOverridingModule(new TaxiModule());
078                controler.addOverridingModule(new AbstractModule() {
079                        @Override
080                        public void install() {
081                                addControlerListenerBinding().to(TaxiBenchmarkStats.class).asEagerSingleton();
082                        };
083                });
084
085                return controler;
086        }
087
088        public static Scenario loadBenchmarkScenario(Config config, int interval, int maxTime) {
089                Scenario scenario = new ScenarioBuilder(config).build();
090
091                if (config.network().isTimeVariantNetwork()) {
092                        ((Network)scenario.getNetwork()).getFactory()
093                                        .setLinkFactory(new FixedIntervalTimeVariantLinkFactory(interval, maxTime));
094                }
095
096                ScenarioUtils.loadScenario(scenario);
097                return scenario;
098        }
099
100        public static void main(String[] args) {
101                run("./src/main/resources/one_taxi_benchmark/one_taxi_benchmark_config.xml", 20);
102        }
103}
