020package org.matsim.contrib.taxi.schedule.reconstruct;
021
022import java.util.*;
023
024import org.matsim.api.core.v01.Id;
025import org.matsim.api.core.v01.events.*;
026import org.matsim.api.core.v01.events.handler.*;
027import org.matsim.api.core.v01.network.Link;
028import org.matsim.api.core.v01.population.Person;
029import org.matsim.contrib.dvrp.path.*;
030import org.matsim.vehicles.Vehicle;
031
032public class DriveRecorder
033                implements PersonEntersVehicleEventHandler, LinkLeaveEventHandler, VehicleLeavesTrafficEventHandler {
034        private class Drive {
035                private final Id<Person> personId;
036                private final double departureTime;
037                private final List<Link> links = new ArrayList<>();
038                private final List<Double> linkLeaveTimes = new ArrayList<>();
039
040                private Drive(Id<Person> personId, double departureTime) {
041                        this.personId = personId;
042                        this.departureTime = departureTime;
043                }
044        }
045
046        private final Map<Id<Vehicle>, Drive> drives = new HashMap<>();
047        private final ScheduleReconstructor reconstructor;
048
049        DriveRecorder(ScheduleReconstructor reconstructor) {
050                this.reconstructor = reconstructor;
051        }
052
053        boolean hasOngoingDrives() {
054                return !drives.isEmpty();
055        }
056
057        @Override
058        public void handleEvent(PersonEntersVehicleEvent event) {
059                Id<Person> personId = event.getPersonId();
060                if (reconstructor.scheduleBuilders.containsKey(personId)) {
061                        Id<Vehicle> vehicleId = event.getVehicleId();
062                        drives.put(vehicleId, new Drive(personId, event.getTime()));
063                }
064        }
065
066        @Override
067        public void handleEvent(LinkLeaveEvent event) {
068                Drive drive = drives.get(event.getVehicleId());
069                if (drive != null) {
070                        updateDrive(drive, event.getLinkId(), event.getTime());
071                }
072        }
073
074        @Override
075        public void handleEvent(VehicleLeavesTrafficEvent event) {
076                Drive drive = drives.remove(event.getVehicleId());
077                if (drive != null) {
078                        updateDrive(drive, event.getLinkId(), event.getTime());
079                        double duration = event.getTime() - drive.departureTime;
080                        Link[] links = drive.links.toArray(new Link[drive.links.size()]);
081                        double[] linkTTs = createLinkTTs(drive);
082                        VrpPathWithTravelData vrpPath = new VrpPathWithTravelDataImpl(drive.departureTime, duration, links,
083                                        linkTTs);
084                        reconstructor.scheduleBuilders.get(drive.personId).addDrive(vrpPath);
085                }
086        }
087
088        private void updateDrive(Drive drive, Id<Link> linkId, double leaveTime) {
089                drive.links.add(reconstructor.links.get(linkId));
090                drive.linkLeaveTimes.add(leaveTime);
091        }
092
093        private double[] createLinkTTs(Drive drive) {
094                double[] linkTTs = new double[drive.linkLeaveTimes.size()];
095                double prevLLT = drive.departureTime;
096                for (int i = 0; i < drive.linkLeaveTimes.size(); i++) {
097                        double currentLLT = drive.linkLeaveTimes.get(i);
098                        linkTTs[i] = currentLLT - prevLLT;
099                        prevLLT = currentLLT;
100                }
101                return linkTTs;
102        }
103
104        @Override
105        public void reset(int iteration) {
106        }
107}
