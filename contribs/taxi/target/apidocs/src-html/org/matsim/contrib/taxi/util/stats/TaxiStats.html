020package org.matsim.contrib.taxi.util.stats;
021
022import org.apache.commons.math3.stat.descriptive.DescriptiveStatistics;
023import org.matsim.contrib.taxi.schedule.TaxiTask.TaxiTaskType;
024import org.matsim.contrib.util.*;
025
026public class TaxiStats {
027        public final String id;
028
029        // all requests submissions made within the analyzed time period
030        public final DescriptiveStatistics passengerWaitTime = new DescriptiveStatistics();
031
032        // duration of all task types within a given time period
033        // each vehicle's contribution is between 0 and N s (where N is the length of time period)
034        // (vehicle may not operate all the time)
035        // similar, yet slightly less accurate, results can be obtained by averaging time profile
036        // values in a given time period
037        public final EnumAdder<TaxiTaskType, Long> taskTimeSumsByType = new LongEnumAdder<>(TaxiTaskType.class);
038
039        // all drives that started within the analyzed time period
040        // in the case of hourly stats, expect high variations:
041        // can be 1.0 if a single empty drive started just before the end of this hour;
042        // can also be 0.0 if a single occupied drive started just after the beginning of this hour
043        public final DescriptiveStatistics vehicleEmptyDriveRatio = new DescriptiveStatistics();
044
045        // vehicles' operation may be of different lengths, which may bias these stats
046        // if this effect is not desired, consider using taskTypeSums instead
047        public final DescriptiveStatistics vehicleStayRatio = new DescriptiveStatistics();
048
049        public TaxiStats(String id) {
050                this.id = id;
051        }
052
053        public double getFleetEmptyDriveRatio() {
054                double empty = taskTimeSumsByType.get(TaxiTaskType.EMPTY_DRIVE).doubleValue();
055                double occupied = taskTimeSumsByType.get(TaxiTaskType.OCCUPIED_DRIVE).doubleValue();
056                return empty / (empty + occupied);
057        }
058
059        public double getFleetStayRatio() {
060                double stay = taskTimeSumsByType.get(TaxiTaskType.STAY).doubleValue();
061                double total = taskTimeSumsByType.getTotal().doubleValue();
062                return stay / total;
063        }
064
065        public double getOccupiedDriveRatio() {
066                double occupied = taskTimeSumsByType.get(TaxiTaskType.OCCUPIED_DRIVE).doubleValue();
067                double total = taskTimeSumsByType.getTotal().doubleValue();
068                return occupied / total;
069        }
070}
