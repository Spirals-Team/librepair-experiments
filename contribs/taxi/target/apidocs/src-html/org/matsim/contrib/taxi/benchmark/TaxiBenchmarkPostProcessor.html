020package org.matsim.contrib.taxi.benchmark;
021
022import java.io.*;
023import java.text.DecimalFormat;
024import java.util.*;
025
026import org.apache.commons.lang3.StringUtils;
027import org.matsim.core.utils.io.IOUtils;
028
029public class TaxiBenchmarkPostProcessor {
030        private static class Experiment {
031                private final String id;
032                private final List<Stats> stats = new ArrayList<>();
033
034                private Experiment(String id) {
035                        this.id = id;
036                }
037        }
038
039        /**
040         * represents a single row in a file created by MultiRunStats all stats in such a file have the same values for 'n'
041         * and 'm'
042         */
043        private static class Stats {
044                // private final String cfg;
045                private final int n;
046                private final int m;
047
048                private final double[] values;
049
050                private Stats(Scanner sc, int count) {
051                        // cfg = sc.next();
052                        n = sc.nextInt();
053                        m = sc.nextInt();
054
055                        values = new double[count];
056                        for (int i = 0; i < count; i++) {
057                                values[i] = sc.nextDouble();
058                        }
059                }
060        }
061
062        private static final Experiment EMPTY_COLUMN = new Experiment("empty column");
063
064        private final String[] header;
065        private final Experiment[] experiments;
066        private final String[] statsColumns;
067
068        public TaxiBenchmarkPostProcessor(String[] header, String... ids) {
069                this.header = header;
070
071                experiments = new Experiment[ids.length];
072                for (int i = 0; i < experiments.length; i++) {
073                        String id = ids[i];
074                        experiments[i] = id == null ? EMPTY_COLUMN : new Experiment(ids[i]);
075                }
076
077                if (!header[0].equals("n") || !header[1].equals("m")) {
078                        throw new IllegalArgumentException("Incompatibile header");
079                }
080                statsColumns = Arrays.copyOfRange(header, 2, header.length);
081        }
082
083        public void process(String dir, String subDirPrefix, String file) {
084                for (Experiment e : experiments) {
085                        if (e != EMPTY_COLUMN) {
086                                readFile(dir + subDirPrefix + e.id + "/" + file + ".txt", e);
087                        }
088                }
089
090                for (int i = 0; i < statsColumns.length; i++) {
091                        writeValues(dir + file, i);
092                }
093        }
094
095        private void readFile(String file, Experiment experiment) {
096                try (Scanner sc = new Scanner(new File(file))) {
097                        String header = sc.nextLine();
098                        if (!header.split("\t").equals(this.header)) {
099                                throw new RuntimeException("Incompatibile header");
100                        }
101
102                        if (!sc.hasNext()) {
103                                throw new RuntimeException("No stats");
104                        }
105
106                        Stats s0 = new Stats(sc, statsColumns.length);
107                        experiment.stats.add(s0);
108
109                        while (sc.hasNext()) {
110                                Stats stats = new Stats(sc, statsColumns.length);
111
112                                if (stats.m != s0.m || stats.n != s0.n) {
113                                        throw new RuntimeException("The file must contain result for the same 'm' and 'n'");
114                                }
115
116                                experiment.stats.add(stats);
117                        }
118                } catch (FileNotFoundException e) {
119                        throw new RuntimeException(e);
120                }
121        }
122
123        private void writeValues(String file, int column) {
124                String field = statsColumns[column];
125                PrintWriter pw = new PrintWriter(IOUtils.getBufferedWriter(file + "_" + field + ".txt"));
126                StringBuffer lineId = new StringBuffer(StringUtils.leftPad(field, 20));
127                StringBuffer lineN = new StringBuffer(StringUtils.leftPad("n", 20));
128                StringBuffer lineM = new StringBuffer(StringUtils.leftPad("m", 20));
129                StringBuffer lineRatio = new StringBuffer(StringUtils.leftPad("ratio", 20));
130
131                for (Experiment e : experiments) {
132                        if (e == EMPTY_COLUMN) {
133                                lineId.append('\t');
134                                lineN.append('\t');
135                                lineM.append('\t');
136                                lineRatio.append('\t');
137                        } else {
138                                Stats s = e.stats.get(0);
139                                double ratio = (double)s.n / s.m;
140                                lineId.append('\t').append(e.id);
141                                lineN.append('\t').append(s.n);
142                                lineM.append('\t').append(s.m);
143                                lineRatio.append('\t').append(String.format("%.2f", ratio));
144                        }
145                }
146
147                pw.println(lineId.toString());
148                pw.println(lineN.toString());
149                pw.println(lineM.toString());
150                pw.println(lineRatio.toString());
151
152                int statsCount = experiments[0].stats.size();
153                DecimalFormat format = new DecimalFormat("#.###");
154
155                for (int i = 0; i < statsCount; i++) {
156                        // String cfg0 = experiments[0].stats.get(i).cfg;
157                        // pw.printf("%20s", cfg0);
158
159                        for (Experiment e : experiments) {
160                                if (e == EMPTY_COLUMN) {
161                                        pw.print('\t');// insert one empty column
162                                } else {
163                                        Stats s = e.stats.get(i);
164
165                                        // if (!cfg0.equals(s.cfg)) {
166                                        // throw new RuntimeException();
167                                        // }
168
169                                        pw.print("\t" + format.format(s.values[column]));
170                                }
171                        }
172
173                        pw.println();
174                }
175
176                pw.close();
177        }
178}
