020package org.matsim.contrib.taxi.util.stats;
021
022import java.util.List;
023
024import org.apache.commons.math3.stat.descriptive.DescriptiveStatistics;
025import org.matsim.contrib.taxi.schedule.TaxiTask;
026import org.matsim.contrib.util.*;
027import org.matsim.core.utils.io.IOUtils;
028
029public class TaxiStatsWriter {
030        private final List<TaxiStats> taxiStats;
031
032        public TaxiStatsWriter(List<TaxiStats> taxiStats) {
033                this.taxiStats = taxiStats;
034        }
035
036        public void write(String file) {
037                try (CompactCSVWriter writer = new CompactCSVWriter(IOUtils.getBufferedWriter(file))) {
038                        writePassengerWaitTimeStats(writer);
039                        writeVehicleEmptyDriveRatioStats(writer);
040                        writeVehicleWaitRatioStats(writer);
041                        writeTaskTypeSums(writer);
042                }
043        }
044
045        private void writePassengerWaitTimeStats(CompactCSVWriter writer) {
046                writer.writeNext("Passenger Wait Time [s]");
047                writer.writeNext(getStatsSubheader("n"));
048
049                for (TaxiStats s : taxiStats) {
050                        CSVLineBuilder lineBuilder = new CSVLineBuilder().add(s.id).add(s.passengerWaitTime.getN() + "");
051                        addStats(lineBuilder, "%.1f", "%.0f", s.passengerWaitTime);
052                        writer.writeNext(lineBuilder);
053                }
054                writer.writeNextEmpty();
055        }
056
057        private void writeVehicleEmptyDriveRatioStats(CompactCSVWriter writer) {
058                writer.writeNext("Vehicle Empty Drive Ratio");
059                writer.writeNext(getStatsSubheader("fleetAvg"));
060
061                for (TaxiStats s : taxiStats) {
062                        CSVLineBuilder lineBuilder = new CSVLineBuilder().add(s.id).//
063                                        addf("%.4f", s.getFleetEmptyDriveRatio());
064                        addStats(lineBuilder, "%.4f", "%.3f", s.vehicleEmptyDriveRatio);
065                        writer.writeNext(lineBuilder);
066                }
067                writer.writeNextEmpty();
068        }
069
070        private void writeVehicleWaitRatioStats(CompactCSVWriter writer) {
071                writer.writeNext("Vehicle Wait Ratio");
072                writer.writeNext(getStatsSubheader("fleetAvg"));
073
074                for (TaxiStats s : taxiStats) {
075                        CSVLineBuilder lineBuilder = new CSVLineBuilder().add(s.id).//
076                                        addf("%.4f", s.getFleetStayRatio());
077                        addStats(lineBuilder, "%.4f", "%.3f", s.vehicleStayRatio);
078                        writer.writeNext(lineBuilder);
079                }
080                writer.writeNextEmpty();
081        }
082
083        private String[] getStatsSubheader(String header2) {
084                return new String[] { "hour", header2, "avg", "sd", null, //
085                                "min", "2%ile", "5%ile", "25%ile", "50%ile", "75%ile", "95%ile", "98%ile", "max" };
086        }
087
088        private void addStats(CSVLineBuilder lineBuilder, String format1, String format2, DescriptiveStatistics stats) {
089                lineBuilder.addf(format1, stats.getMean()).//
090                                addf(format1, stats.getStandardDeviation()).//
091                                addEmpty().//
092                                addf(format2, stats.getMin()). //
093                                addf(format2, stats.getPercentile(2)). //
094                                addf(format2, stats.getPercentile(5)). //
095                                addf(format2, stats.getPercentile(25)). //
096                                addf(format2, stats.getPercentile(50)). //
097                                addf(format2, stats.getPercentile(75)). //
098                                addf(format2, stats.getPercentile(95)). //
099                                addf(format2, stats.getPercentile(98)). //
100                                addf(format2, stats.getMax());
101        }
102
103        private void writeTaskTypeSums(CompactCSVWriter writer) {
104                writer.writeNext("Total duration of tasks by type [h]");
105                CSVLineBuilder headerBuilder = new CSVLineBuilder().add("hour");
106                for (TaxiTask.TaxiTaskType t : TaxiTask.TaxiTaskType.values()) {
107                        headerBuilder.add(t.name());
108                }
109                writer.writeNext(headerBuilder.add("all"));
110
111                for (TaxiStats s : taxiStats) {
112                        CSVLineBuilder lineBuilder = new CSVLineBuilder().add(s.id);
113                        for (TaxiTask.TaxiTaskType t : TaxiTask.TaxiTaskType.values()) {
114                                lineBuilder.addf("%.2f", s.taskTimeSumsByType.get(t).doubleValue() / 3600);
115                        }
116                        lineBuilder.addf("%.2f", s.taskTimeSumsByType.getTotal().doubleValue() / 3600);
117                        writer.writeNext(lineBuilder);
118                }
119                writer.writeNextEmpty();
120        }
121}
