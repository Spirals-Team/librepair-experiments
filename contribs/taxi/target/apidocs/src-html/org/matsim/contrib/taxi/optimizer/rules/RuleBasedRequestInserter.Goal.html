019package org.matsim.contrib.taxi.optimizer.rules;
020
021import java.util.Collection;
022import java.util.Iterator;
023import java.util.stream.Stream;
024
025import org.matsim.api.core.v01.network.Link;
026import org.matsim.api.core.v01.network.Network;
027import org.matsim.contrib.dvrp.data.Vehicle;
028import org.matsim.contrib.dvrp.passenger.PassengerRequests;
029import org.matsim.contrib.taxi.data.TaxiRequest;
030import org.matsim.contrib.taxi.optimizer.BestDispatchFinder;
031import org.matsim.contrib.taxi.optimizer.UnplannedRequestInserter;
032import org.matsim.contrib.taxi.schedule.TaxiStayTask;
033import org.matsim.contrib.taxi.scheduler.TaxiScheduler;
034import org.matsim.core.mobsim.framework.MobsimTimer;
035import org.matsim.core.router.util.TravelDisutility;
036import org.matsim.core.router.util.TravelTime;
037
038/**
039 * @author michalm
040 */
041public class RuleBasedRequestInserter implements UnplannedRequestInserter {
042        private final TaxiScheduler scheduler;
043        private final BestDispatchFinder dispatchFinder;
044        private final MobsimTimer timer;
045
046        private final IdleTaxiZonalRegistry idleTaxiRegistry;
047        private final UnplannedRequestZonalRegistry unplannedRequestRegistry;
048
049        private final RuleBasedTaxiOptimizerParams params;
050
051        public RuleBasedRequestInserter(TaxiScheduler scheduler, MobsimTimer timer, Network network, TravelTime travelTime,
052                        TravelDisutility travelDisutility, RuleBasedTaxiOptimizerParams params,
053                        IdleTaxiZonalRegistry idleTaxiRegistry, UnplannedRequestZonalRegistry unplannedRequestRegistry) {
054                this(scheduler, timer, new BestDispatchFinder(scheduler, network, timer, travelTime, travelDisutility), params,
055                                idleTaxiRegistry, unplannedRequestRegistry);
056        }
057
058        public RuleBasedRequestInserter(TaxiScheduler scheduler, MobsimTimer timer, BestDispatchFinder dispatchFinder,
059                        RuleBasedTaxiOptimizerParams params, IdleTaxiZonalRegistry idleTaxiRegistry,
060                        UnplannedRequestZonalRegistry unplannedRequestRegistry) {
061                this.scheduler = scheduler;
062                this.timer = timer;
063                this.params = params;
064                this.dispatchFinder = dispatchFinder;
065                this.idleTaxiRegistry = idleTaxiRegistry;
066                this.unplannedRequestRegistry = unplannedRequestRegistry;
067        }
068
069        @Override
070        public void scheduleUnplannedRequests(Collection<TaxiRequest> unplannedRequests) {
071                if (isReduceTP(unplannedRequests)) {
072                        scheduleIdleVehiclesImpl(unplannedRequests);// reduce T_P to increase throughput (demand > supply)
073                } else {
074                        scheduleUnplannedRequestsImpl(unplannedRequests);// reduce T_W (regular NOS)
075                }
076        }
077
078        public enum Goal {
079                MIN_WAIT_TIME, MIN_PICKUP_TIME, DEMAND_SUPPLY_EQUIL;
080        };
081
082        private boolean isReduceTP(Collection<TaxiRequest> unplannedRequests) {
083                switch (params.goal) {
084                        case MIN_PICKUP_TIME:
085                                return true;
086
087                        case MIN_WAIT_TIME:
088                                return false;
089
090                        case DEMAND_SUPPLY_EQUIL:
091                                double now = timer.getTimeOfDay();
092                                long awaitingReqCount = unplannedRequests.stream().filter(r -> PassengerRequests.isUrgent(r, now))
093                                                .count();
094                                return awaitingReqCount > idleTaxiRegistry.getVehicleCount();
095
096                        default:
097                                throw new IllegalStateException();
098                }
099        }
100
101        // request-initiated scheduling
102        private void scheduleUnplannedRequestsImpl(Collection<TaxiRequest> unplannedRequests) {
103                // vehicles are not immediately removed so calculate 'idleCount' locally
104                int idleCount = idleTaxiRegistry.getVehicleCount();
105
106                Iterator<TaxiRequest> reqIter = unplannedRequests.iterator();
107                while (reqIter.hasNext() && idleCount > 0) {
108                        TaxiRequest req = reqIter.next();
109
110                        Stream<Vehicle> selectedVehs = idleCount > params.nearestVehiclesLimit//
111                                        ? idleTaxiRegistry.findNearestVehicles(req.getFromLink().getFromNode(), params.nearestVehiclesLimit)
112                                        : idleTaxiRegistry.vehicles();
113
114                        BestDispatchFinder.Dispatch<TaxiRequest> best = dispatchFinder.findBestVehicleForRequest(req, selectedVehs);
115                        if (best == null) {
116                                // XXX NOTE:
117                                // There may be no idle vehicle in the registry despite idleCount > 0
118                                // Some vehicles may not be idle because they have been assigned another customer,
119                                // while for others the time window ends (t1).
120                                // Their statuses will be updated in this time step (we are just before the sim step,
121                                // triggered by a MobsimBeforeSimStepEvent). Consequently, they will be
122                                // removed from this registry, but till then they are there.
123                                return;
124                        }
125
126                        scheduler.scheduleRequest(best.vehicle, best.destination, best.path);
127
128                        reqIter.remove();
129                        unplannedRequestRegistry.removeRequest(req);
130                        idleCount--;
131                }
132        }
133
134        // vehicle-initiated scheduling
135        private void scheduleIdleVehiclesImpl(Collection<TaxiRequest> unplannedRequests) {
136                Iterator<Vehicle> vehIter = idleTaxiRegistry.vehicles().iterator();
137                while (vehIter.hasNext() && !unplannedRequests.isEmpty()) {
138                        Vehicle veh = vehIter.next();
139                        Link link = ((TaxiStayTask)veh.getSchedule().getCurrentTask()).getLink();
140
141                        Stream<TaxiRequest> selectedReqs = unplannedRequests.size() > params.nearestRequestsLimit
142                                        ? unplannedRequestRegistry.findNearestRequests(link.getToNode(), params.nearestRequestsLimit)
143                                        : unplannedRequests.stream();
144
145                        BestDispatchFinder.Dispatch<TaxiRequest> best = dispatchFinder.findBestRequestForVehicle(veh, selectedReqs);
146
147                        scheduler.scheduleRequest(best.vehicle, best.destination, best.path);
148
149                        unplannedRequests.remove(best.destination);
150                        unplannedRequestRegistry.removeRequest(best.destination);
151                }
152        }
153}
