020package org.matsim.contrib.taxi.util.stats;
021
022import java.util.List;
023
024import org.apache.commons.math3.stat.descriptive.DescriptiveStatistics;
025import org.matsim.contrib.dvrp.data.*;
026import org.matsim.contrib.dvrp.schedule.*;
027import org.matsim.contrib.dvrp.schedule.Schedule.ScheduleStatus;
028import org.matsim.contrib.taxi.data.TaxiRequest;
029import org.matsim.contrib.taxi.schedule.*;
030import org.matsim.contrib.taxi.schedule.TaxiTask.TaxiTaskType;
031import org.matsim.contrib.util.LongEnumAdder;
032
033public class TaxiStatsCalculator {
034        private final int hours;
035        private final TaxiStats[] hourlyStats;
036        private final TaxiStats dailyStats = new TaxiStats(TaxiStatsCalculators.DAILY_STATS_ID);
037        private final List<TaxiStats> taxiStats;
038
039        public TaxiStatsCalculator(Iterable<? extends Vehicle> vehicles) {
040                hours = TaxiStatsCalculators.calcHourCount(vehicles);
041                hourlyStats = new TaxiStats[hours];
042                for (int h = 0; h < hours; h++) {
043                        hourlyStats[h] = new TaxiStats(h + "");
044                }
045
046                taxiStats = TaxiStatsCalculators.createStatsList(hourlyStats, dailyStats);
047
048                for (Vehicle v : vehicles) {
049                        updateStatsForVehicle(v);
050                }
051        }
052
053        public List<TaxiStats> getTaxiStats() {
054                return taxiStats;
055        }
056
057        public TaxiStats getDailyStats() {
058                return dailyStats;
059        }
060
061        private void updateStatsForVehicle(Vehicle vehicle) {
062                Schedule schedule = vehicle.getSchedule();
063                if (schedule.getStatus() == ScheduleStatus.UNPLANNED) {
064                        return;// do not evaluate - the vehicle is unused
065                }
066
067                @SuppressWarnings("unchecked")
068                LongEnumAdder<TaxiTaskType>[] vehicleHourlySums = new LongEnumAdder[hours];
069
070                for (Task t : schedule.getTasks()) {
071                        TaxiTask tt = (TaxiTask)t;
072                        int[] hourlyDurations = TaxiStatsCalculators.calcHourlyDurations((int)t.getBeginTime(),
073                                        (int)t.getEndTime());
074                        int fromHour = TaxiStatsCalculators.getHour(t.getBeginTime());
075                        for (int i = 0; i < hourlyDurations.length; i++) {
076                                includeTaskIntoHourlySums(vehicleHourlySums, fromHour + i, tt, hourlyDurations[i]);
077                        }
078
079                        if (tt.getTaxiTaskType() == TaxiTaskType.PICKUP) {
080                                TaxiRequest req = ((TaxiPickupTask)t).getRequest();
081                                double waitTime = Math.max(t.getBeginTime() - req.getEarliestStartTime(), 0);
082                                int hour = TaxiStatsCalculators.getHour(req.getEarliestStartTime());
083                                hourlyStats[hour].passengerWaitTime.addValue(waitTime);
084                                dailyStats.passengerWaitTime.addValue(waitTime);
085                        }
086                }
087
088                includeVehicleHourlySumsIntoStats(vehicleHourlySums);
089        }
090
091        private void includeTaskIntoHourlySums(LongEnumAdder<TaxiTaskType>[] hourlySums, int hour, TaxiTask task,
092                        int duration) {
093                if (duration > 0) {
094                        if (hourlySums[hour] == null) {
095                                hourlySums[hour] = new LongEnumAdder<>(TaxiTaskType.class);
096                        }
097                        hourlySums[hour].add(task.getTaxiTaskType(), duration);
098                }
099        }
100
101        private void includeVehicleHourlySumsIntoStats(LongEnumAdder<TaxiTaskType>[] vehicleHourlySums) {
102                LongEnumAdder<TaxiTaskType> vehicleDailySums = new LongEnumAdder<>(TaxiTaskType.class);
103
104                for (int h = 0; h < hours; h++) {
105                        LongEnumAdder<TaxiTaskType> vhs = vehicleHourlySums[h];
106                        if (vhs != null && vhs.getLongTotal() > 0) {
107                                updateTaxiStats(hourlyStats[h], vhs);
108                                vehicleDailySums.addAll(vhs);
109                        }
110                }
111
112                updateTaxiStats(dailyStats, vehicleDailySums);
113        }
114
115        private void updateTaxiStats(TaxiStats taxiStats, LongEnumAdder<TaxiTaskType> vehicleSums) {
116                updateEmptyDriveRatio(taxiStats.vehicleEmptyDriveRatio, vehicleSums);
117                updateStayRatio(taxiStats.vehicleStayRatio, vehicleSums);
118                taxiStats.taskTimeSumsByType.addAll(vehicleSums);
119        }
120
121        private void updateEmptyDriveRatio(DescriptiveStatistics emptyDriveRatioStats,
122                        LongEnumAdder<TaxiTaskType> durations) {
123                double empty = durations.getLong(TaxiTaskType.EMPTY_DRIVE);
124                double occupied = durations.getLong(TaxiTaskType.OCCUPIED_DRIVE);
125
126                if (empty != 0 || occupied != 0) {
127                        double emptyRatio = empty / (empty + occupied);
128                        emptyDriveRatioStats.addValue(emptyRatio);
129                }
130        }
131
132        private void updateStayRatio(DescriptiveStatistics stayRatioStats, LongEnumAdder<TaxiTaskType> durations) {
133                double total = durations.getLongTotal();
134                if (total != 0) {
135                        double stayRatio = durations.getLong(TaxiTaskType.STAY) / total;
136                        stayRatioStats.addValue(stayRatio);
137                }
138        }
139}
