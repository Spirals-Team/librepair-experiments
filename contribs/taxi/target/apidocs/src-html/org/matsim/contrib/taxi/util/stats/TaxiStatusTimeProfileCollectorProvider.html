020package org.matsim.contrib.taxi.util.stats;
021
022import java.awt.Color;
023import java.awt.Paint;
024
025import org.jfree.data.xy.DefaultTableXYDataset;
026import org.matsim.contrib.dvrp.data.Fleet;
027import org.matsim.contrib.taxi.data.TaxiRequest.TaxiRequestStatus;
028import org.matsim.contrib.taxi.passenger.SubmittedTaxiRequestsCollector;
029import org.matsim.contrib.util.timeprofile.TimeProfileCharts;
030import org.matsim.contrib.util.timeprofile.TimeProfileCharts.ChartType;
031import org.matsim.contrib.util.timeprofile.TimeProfileCollector;
032import org.matsim.contrib.util.timeprofile.TimeProfileCollector.ProfileCalculator;
033import org.matsim.contrib.util.timeprofile.TimeProfiles;
034import org.matsim.core.controler.MatsimServices;
035import org.matsim.core.mobsim.framework.listeners.MobsimListener;
036
037import com.google.inject.Inject;
038import com.google.inject.Provider;
039
040public class TaxiStatusTimeProfileCollectorProvider implements Provider<MobsimListener> {
041        private final Fleet fleet;
042        private final SubmittedTaxiRequestsCollector requestCollector;
043        private final MatsimServices matsimServices;
044
045        @Inject
046        public TaxiStatusTimeProfileCollectorProvider(Fleet fleet, MatsimServices matsimServices,
047                        SubmittedTaxiRequestsCollector requestCollector) {
048                this.fleet = fleet;
049                this.requestCollector = requestCollector;
050                this.matsimServices = matsimServices;
051        }
052
053        @Override
054        public MobsimListener get() {
055                ProfileCalculator calc = TimeProfiles.combineProfileCalculators(
056                                TaxiTimeProfiles.createCurrentTaxiTaskTypeCounter(fleet), //
057                                TaxiTimeProfiles.createRequestsWithStatusCounter(requestCollector.getRequests().values(),
058                                                TaxiRequestStatus.UNPLANNED));
059
060                TimeProfileCollector collector = new TimeProfileCollector(calc, 300, "taxi_status_time_profiles",
061                                matsimServices);
062
063                collector.setChartCustomizer((chart, chartType) -> {
064                        TimeProfileCharts.changeSeriesColors(chart, new Paint[] { //
065                                        new Color(91, 155, 213), // EMPTY_DRIVE
066                                        new Color(237, 125, 49), // PICKUP
067                                        new Color(165, 165, 165), // OCCUPIED_DRIVE
068                                        new Color(255, 192, 0), // DROPOFF
069                                        new Color(112, 173, 71), // STAY
070                                        new Color(37, 94, 145) }); // UNPLANNED (requests)
071                        if (chartType == ChartType.StackedArea) {
072                                ((DefaultTableXYDataset)chart.getXYPlot().getDataset()).removeSeries(5);
073                        }
074                });
075
076                collector.setChartTypes(ChartType.Line, ChartType.StackedArea);
077                return collector;
078        }
079}
