042/**
043 * @author michalm
044 */
045public class VehicleAssignmentProblem<D> {
046        public static interface AssignmentCost<D> {
047                double calc(VehicleData.Entry departure, DestEntry<D> dest, PathData pathData);
048        }
049
050        private final TravelTime travelTime;
051        private final LeastCostPathCalculator router;
052
053        private final OneToManyPathSearch forwardPathSearch;
054        private final OneToManyPathSearch backwardPathSearch;
055
056        private final StraightLineKnnFinder<VehicleData.Entry, DestEntry<D>> destinationFinder;
057        private final StraightLineKnnFinder<DestEntry<D>, VehicleData.Entry> vehicleFinder;
058
059        private AssignmentCost<D> assignmentCost;
060        private VehicleData vData;
061        private AssignmentDestinationData<D> dData;
062
063        public VehicleAssignmentProblem(TravelTime travelTime, MultiNodePathCalculator router,
064                        BackwardMultiNodePathCalculator backwardRouter) {
065                // we do not need Euclidean router when there is not kNN filtering
066                this(travelTime, router, backwardRouter, null, -1, -1);
067        }
068
069        public VehicleAssignmentProblem(TravelTime travelTime, MultiNodePathCalculator multiNodeRouter,
070                        BackwardMultiNodePathCalculator backwardMultiNodeRouter, LeastCostPathCalculator router,
071                        int nearestDestinationLimit, int nearestVehicleLimit) {
072                this.travelTime = travelTime;
073                this.router = router;
074
075                forwardPathSearch = OneToManyPathSearch.create(multiNodeRouter);
076                backwardPathSearch = OneToManyPathSearch.create(backwardMultiNodeRouter);
077
078                // TODO this kNN is slow
079                destinationFinder = nearestDestinationLimit < 0 ? null
080                                : new StraightLineKnnFinder<>(nearestDestinationLimit, veh -> veh.link, dest -> dest.link);
081                vehicleFinder = nearestVehicleLimit < 0 ? null
082                                : new StraightLineKnnFinder<>(nearestVehicleLimit, dest -> dest.link, veh -> veh.link);
083        }
084
085        public List<Dispatch<D>> findAssignments(VehicleData vData, AssignmentDestinationData<D> dData,
086                        AssignmentCost<D> assignmentCost) {
087                this.vData = vData;
088                this.dData = dData;
089                this.assignmentCost = assignmentCost;
090
091                PathData[][] pathDataMatrix = createPathDataMatrix();
092                double[][] costMatrix = createCostMatrix(pathDataMatrix);
093                int[] assignments = new HungarianAlgorithm(costMatrix).execute();
094                return createDispatches(assignments, pathDataMatrix, travelTime);
095        }
096
097        // private static int calcPathsForVehiclesCount = 0;
098        // private static int calcPathsForDestinationsCount = 0;
099
100        private PathData[][] createPathDataMatrix() {
101                PathData[][] pathDataMatrix = (PathData[][])Array.newInstance(PathData.class, vData.getSize(), dData.getSize());
102
103                if (dData.getSize() > vData.getSize()) {
104                        calcPathsForVehicles(pathDataMatrix);
105                        // calcPathsForVehiclesCount++;
106                } else {
107                        calcPathsForDestinations(pathDataMatrix);
108                        // calcPathsForDestinationsCount++;
109                }
110
111                // if ( (calcPathsForDestinationsCount + calcPathsForVehiclesCount) % 100 == 0) {
112                // System.err.println("PathsForDestinations = " + calcPathsForDestinationsCount
113                // + " PathsForVehicles = " + calcPathsForVehiclesCount);
114                // System.err.println("dests = " + dData.getSize() + " vehs = " + vData.getSize()
115                // + " idleVehs = " + vData.getIdleCount());
116                // }
117
118                return pathDataMatrix;
119        }
120
121        private void calcPathsForVehicles(PathData[][] pathDataMatrix) {
122                for (int v = 0; v < vData.getSize(); v++) {
123                        VehicleData.Entry departure = vData.getEntry(v);
124
125                        List<DestEntry<D>> filteredDests = destinationFinder == null ? dData.getEntries()
126                                        : destinationFinder.findNearest(departure, dData.getEntries().stream());
127                        List<Link> toLinks = Lists.transform(filteredDests, dest -> dest.link);
128                        PathData[] paths = forwardPathSearch.calcPathDataArray(departure.link, toLinks, departure.time);
129
130                        for (int i = 0; i < filteredDests.size(); i++) {
131                                int d = filteredDests.get(i).idx;
132                                pathDataMatrix[v][d] = paths[i];
133                        }
134                }
135        }
136
137        // TODO does not support adv reqs
138        private void calcPathsForDestinations(PathData[][] pathDataMatrix) {
139                for (int d = 0; d < dData.getSize(); d++) {
140                        DestEntry<D> dest = dData.getEntry(d);
141
142                        List<VehicleData.Entry> filteredVehs = vehicleFinder == null ? vData.getEntries()
143                                        : vehicleFinder.findNearest(dest, vData.getEntries().stream());
144                        List<Link> toLinks = Lists.transform(filteredVehs, veh -> veh.link);
145                        PathData[] paths = backwardPathSearch.calcPathDataArray(dest.link, toLinks, dest.time);
146
147                        for (int i = 0; i < filteredVehs.size(); i++) {
148                                int v = filteredVehs.get(i).idx;
149                                pathDataMatrix[v][d] = paths[i];
150                        }
151                }
152        }
153
154        private double[][] createCostMatrix(PathData[][] pathDataMatrix) {
155
156                double[][] costMatrix = new double[vData.getSize()][dData.getSize()];
157
158                for (int v = 0; v < vData.getSize(); v++) {
159                        VehicleData.Entry departure = vData.getEntry(v);
160                        for (int r = 0; r < dData.getSize(); r++) {
161                                costMatrix[v][r] = assignmentCost.calc(departure, dData.getEntry(r), pathDataMatrix[v][r]);
162                        }
163                }
164
165                return costMatrix;
166        }
167
168        private List<Dispatch<D>> createDispatches(int[] assignments, PathData[][] pathDataMatrix, TravelTime travelTime) {
169                List<Dispatch<D>> dispatches = new ArrayList<>(Math.min(vData.getSize(), dData.getSize()));
170                for (int v = 0; v < assignments.length; v++) {
171                        int d = assignments[v];
172                        if (d == -1 || // no request assigned
173                                        d >= dData.getSize()) {// non-existing (dummy) request assigned
174                                continue;
175                        }
176
177                        VehicleData.Entry departure = vData.getEntry(v);
178                        DestEntry<D> dest = dData.getEntry(d);
179                        PathData pathData = pathDataMatrix[v][d];
180
181                        // TODO if null is frequent we may be more efficient by increasing the neighbourhood
182                        VrpPathWithTravelData vrpPath = pathData == null ? //
183                                        VrpPaths.calcAndCreatePath(departure.link, dest.link, departure.time, router, travelTime)
184                                        : VrpPaths.createPath(departure.link, dest.link, departure.time, pathData, travelTime);
185
186                        dispatches.add(new Dispatch<>(departure.vehicle, dest.destination, vrpPath));
187                }
188
189                return dispatches;
190        }
