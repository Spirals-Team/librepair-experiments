020package org.matsim.contrib.taxi.util.stats;
021
022import java.util.List;
023
024import org.matsim.contrib.util.*;
025import org.matsim.contrib.util.histogram.*;
026import org.matsim.core.utils.io.IOUtils;
027
028public class TaxiHistogramsWriter {
029        private final List<TaxiStats> taxiStats;
030
031        public TaxiHistogramsWriter(List<TaxiStats> taxiStats) {
032                this.taxiStats = taxiStats;
033        }
034
035        public void write(String file) {
036                try (CompactCSVWriter writer = new CompactCSVWriter(IOUtils.getBufferedWriter(file))) {
037                        writePassengerWaitTime(writer);
038                        writeVehicleEmptyDriveRatio(writer);
039                        writeVehicleStayRatio(writer);
040                        writeHourlyWaitRatioCounters(writer);
041                }
042        }
043
044        private void writePassengerWaitTime(CompactCSVWriter writer) {
045                writeHistogramHeader(writer, "Passenger Wait Time [min]", new UniformHistogram(2.5, 25));
046                for (TaxiStats s : taxiStats) {
047                        writeHistogramValues(writer, s.id, UniformHistogram.create(2.5 * 60, 25, s.passengerWaitTime.getValues()));
048                }
049                writer.writeNextEmpty();
050        }
051
052        private void writeVehicleEmptyDriveRatio(CompactCSVWriter writer) {
053                writeHistogramHeader(writer, "Vehicle Empty Drive Ratio", new UniformHistogram(0.05, 20));
054                for (TaxiStats s : taxiStats) {
055                        writeHistogramValues(writer, s.id, UniformHistogram.create(0.05, 20, s.vehicleEmptyDriveRatio.getValues()));
056                }
057                writer.writeNextEmpty();
058        }
059
060        private void writeVehicleStayRatio(CompactCSVWriter writer) {
061                writeHistogramHeader(writer, "Vehicle Wait Ratio", new UniformHistogram(0.05, 20));
062                for (TaxiStats s : taxiStats) {
063                        writeHistogramValues(writer, s.id, UniformHistogram.create(0.05, 20, s.vehicleStayRatio.getValues()));
064                }
065                writer.writeNextEmpty();
066        }
067
068        private void writeHourlyWaitRatioCounters(CompactCSVWriter writer) {
069                // [0, 0.01) approx. "always busy"; 0pct is impossible due to 1-sec stay tasks
070                // [0.01, 0.25) and [0.25, 0.5) both busy for most of the time
071                // [0.5, 0.75) and [0.75, 1) both idle for most of the time
072                // [1, 1] //== always idle
073                double[] bounds = { 0, 0.01, 0.25, 0.5, 0.75, 1, 1.00000001 };
074
075                writeHistogramHeader(writer, "Vehicle Wait Ratio Counts", new BoundedHistogram(bounds));
076                for (TaxiStats s : taxiStats) {
077                        writeHistogramValues(writer, s.id, BoundedHistogram.create(bounds, s.vehicleStayRatio.getValues()));
078                }
079                writer.writeNextEmpty();
080        }
081
082        private void writeHistogramHeader(CompactCSVWriter writer, String header, Histogram<?> histogram) {
083                writer.writeNext(header);
084                CSVLineBuilder lineBuilder = new CSVLineBuilder().add("hour");
085                for (int i = 0; i < histogram.getBinCount(); i++) {
086                        lineBuilder.addf("%.2f+", histogram.getBin(i));
087                }
088                writer.writeNext(lineBuilder);
089        }
090
091        private void writeHistogramValues(CompactCSVWriter writer, String hour, Histogram<?> histogram) {
092                CSVLineBuilder lineBuilder = new CSVLineBuilder().add(hour);
093                for (int i = 0; i < histogram.getBinCount(); i++) {
094                        lineBuilder.add(histogram.getCount(i) + "");
095                }
096                writer.writeNext(lineBuilder);
097        }
098}
