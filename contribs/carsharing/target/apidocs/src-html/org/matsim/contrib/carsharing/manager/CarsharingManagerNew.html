058        @Override
059        public List<PlanElement> reserveAndrouteCarsharingTrip(final Plan plan, String carsharingType, Leg legToBeRouted, Double time) {
060                Network network = scenario.getNetwork();
061                Person person = plan.getPerson();
062                Link startLink = network.getLinks().get(legToBeRouted.getRoute().getStartLinkId());
063                Link destinationLink = network.getLinks().get(legToBeRouted.getRoute().getEndLinkId());
064                
065                
066                //=== get the vehicle for the trip if the agent has it already ===
067                CSVehicle vehicle = getVehicleAtLocation(startLink,     plan.getPerson(), carsharingType);              
068
069                boolean willHaveATripFromLocation = willUseTheVehicleLaterFromLocation(destinationLink.getId(), plan, legToBeRouted);
070                double durationOfNextActivity = getDurationOfNextActivity(plan, legToBeRouted, time);           
071                boolean keepTheCar;
072                if (((CarsharingRoute)legToBeRouted.getRoute()).isOldRoute()) 
073                        keepTheCar = ((CarsharingRoute)legToBeRouted.getRoute()).isKeepthecar();
074                else            
075                        keepTheCar = keepTheCarModel.keepTheCarDuringNextActivity(durationOfNextActivity, plan.getPerson(), carsharingType);    
076                //TODO: create a method for getting the search distance
077                double searchDistance = 1000.0;
078                if (vehicle != null) {
079                        
080                        if ((willHaveATripFromLocation && keepTheCar) || (willHaveATripFromLocation && carsharingType.equals("twoway"))) {
081                                ((CarsharingRoute)legToBeRouted.getRoute()).setKeepthecar(true);
082                                return this.routerProvider.routeCarsharingTrip(plan, time, legToBeRouted, carsharingType, vehicle,
083                                        startLink, destinationLink, true, true);
084                        }
085                        else {
086                                
087                                if (carsharingType.equals("oneway")) {
088                                
089                                        CompanyContainer companyContainer = this.carsharingSupplyContainer.getCompany(vehicle.getCompanyId());
090                                        VehiclesContainer vehiclesContainer = companyContainer.getVehicleContainer(carsharingType);
091                                        Link parkingLocation = vehiclesContainer.findClosestAvailableParkingLocation(destinationLink, searchDistance);                                  
092                                        
093                                        if (parkingLocation == null)
094                                                return null;
095                                        destinationLink = parkingLocation;
096                                        vehiclesContainer.reserveParking(destinationLink);
097                                        
098                                        
099                                }
100                                else if (carsharingType.equals("twoway")) {
101                                        CarsharingStation parkingStation = ((TwoWayContainer)this.carsharingSupplyContainer.getCompany(vehicle.getCompanyId()).
102                                                        getVehicleContainer(carsharingType)).getTwowaycarsharingstationsMap().
103                                                        get(((StationBasedVehicle)vehicle).getStationId());
104                                        destinationLink = parkingStation.getLink();
105                                }                       
106                                
107                                return this.routerProvider.routeCarsharingTrip(plan, time, legToBeRouted, carsharingType, vehicle,
108                                                startLink, destinationLink, false, true);                               
109                        }                       
110                }
111                else {
112                        
113                        //=== agent does not hold the vehicle, therefore must find a one from the supply side===
114                        //=== here he chooses the company, type of vehicle and in the end vehicle ===
115                        //=== possibly this could be moved to one method which decides based on the supply which vehicle to take
116                        String typeOfVehicle = chooseVehicleType.getPreferredVehicleType(plan, legToBeRouted);
117                        
118                        Set<CSVehicle> offeredVehicles = new HashSet<>();
119                        for (CompanyAgent companyAgent : this.carsharingSupplyContainer.getCompanyAgents().values()) {
120                                
121                                CSVehicle offeredVehicle = companyAgent.vehicleRequest(person.getId(), startLink, destinationLink, carsharingType, typeOfVehicle);
122                                
123                                if (offeredVehicle != null)
124                                        offeredVehicles.add(offeredVehicle);                            
125                        }
126                        
127                        
128                        //TODO: offer the possible cars to the agent, from which the agent can choose
129                        //CSVehicle chosenVehicle = vehicleChoiceAgent.chooseVehicle(offeredVehicles, startLink, legToBeRouted, time, plan.getPerson());
130                        
131                        CSVehicle chosenVehicle = vehicleChoiceAgent.chooseVehicleActivityTimeIncluded(offeredVehicles, 
132                                        startLink, legToBeRouted, time, plan.getPerson(), durationOfNextActivity, keepTheCar);
133                        
134                        
135                        if (chosenVehicle == null) {
136                                eventsManager.processEvent(new NoVehicleCarSharingEvent(time, carsharingType, "", startLink, destinationLink));
137
138                                return null;
139                        }
140                        else {
141                                
142                                CompanyContainer companyContainer = this.carsharingSupplyContainer.getCompany(chosenVehicle.getCompanyId());
143                                VehiclesContainer vehiclesContainer = companyContainer.getVehicleContainer(carsharingType);
144                                Link stationLink = vehiclesContainer.getVehicleLocation(chosenVehicle);
145                                
146                                if (false == companyContainer.reserveVehicle(chosenVehicle)) {
147                                        eventsManager.processEvent(new NoVehicleCarSharingEvent(time, carsharingType, chosenVehicle.getCompanyId(), startLink, destinationLink));
148
149                                        return null;
150                                }
151                        
152                                eventsManager.processEvent(new StartRentalEvent(time, carsharingType, 
153                                                chosenVehicle.getCompanyId(), startLink, stationLink, destinationLink, person.getId(), chosenVehicle.getVehicleId()));
154                        
155                                if ((willHaveATripFromLocation && keepTheCar) || (willHaveATripFromLocation && carsharingType.equals("twoway"))) {
156                                        ((CarsharingRoute)legToBeRouted.getRoute()).setKeepthecar(true);
157
158                                        return this.routerProvider.routeCarsharingTrip(plan, time, legToBeRouted, carsharingType, chosenVehicle,
159                                                        stationLink, destinationLink, true, false);
160                                }
161                                else {
162                                        if ((carsharingType.equals("oneway")) || (carsharingType.equals("freefloating"))) {
163                                                Link parkingStationLink = this.carsharingSupplyContainer.findClosestAvailableParkingSpace(
164                                                                destinationLink, carsharingType, chosenVehicle.getCompanyId(), searchDistance);
165                                                if (parkingStationLink == null)
166                                                        return null;
167                                        
168                                                vehiclesContainer.reserveParking(parkingStationLink);
169                                        
170                                                destinationLink = parkingStationLink;
171                                        }
172        
173                                        return this.routerProvider.routeCarsharingTrip(plan, time, legToBeRouted, carsharingType, chosenVehicle,
174                                                        stationLink, destinationLink, false, false);                            
175                                }               
176                                
177                        }                               
178                        
179                }                                       
180        }

182        private double getDurationOfNextActivity(Plan plan, Leg legToBeRouted, double time) {
183                
184                int index = plan.getPlanElements().indexOf(legToBeRouted);
185                
186                Activity a = (Activity) plan.getPlanElements().get(index + 1);
187                
188                return a.getEndTime() - time > 0.0 ? a.getEndTime() - time : 0.0;               
189        }
190
191        private boolean willUseTheVehicleLaterFromLocation(Id<Link> linkId, Plan plan, Leg currentLeg) {
192                boolean willUseVehicle = false;
193
194                String mode = currentLeg.getMode();
195                List<PlanElement> planElements = plan.getPlanElements();
196
197                int index = planElements.indexOf(currentLeg) + 1;
198
199                for (int i = index; i < planElements.size(); i++) {
200
201                        if (planElements.get(i) instanceof Leg) {
202
203                                if (((Leg)planElements.get(i)).getMode().equals(mode)) {
204
205                                        if (((Leg)planElements.get(i)).getRoute().getStartLinkId().toString().equals(linkId.toString())) {
206
207                                                willUseVehicle = true;
208                                        }
209                                }
210                        }
211                }
212                return willUseVehicle;
213        }       
214        private CSVehicle getVehicleAtLocation(Link currentLink, Person person, String carsharingType) {
215
216                return this.currentDemand.getVehicleOnLink(person.getId(), currentLink, carsharingType);        
217                
218        }       
219        @Override
220        public void freeParkingSpot(String vehicleId, Id<Link> linkId) {
221                
222                CSVehicle vehicle = this.carsharingSupplyContainer.getVehicleWithId(vehicleId);
223                CompanyContainer companyContainer = this.carsharingSupplyContainer.getCompany(vehicle.getCompanyId());
224                VehiclesContainer vehiclesContainer = companyContainer.getVehicleContainer(vehicle.getCsType());
225                
226                ((OneWayContainer)vehiclesContainer).freeParkingSpot(vehicle);                  
227        }
228
229        @Override
230        public boolean parkVehicle(String vehicleId, Id<Link> linkId) {
231                CSVehicle vehicle = this.carsharingSupplyContainer.getVehicleWithId(vehicleId);
232                Network network = scenario.getNetwork();
233                Link link = network.getLinks().get(linkId);
234                this.carsharingSupplyContainer.getCompany(vehicle.getCompanyId()).parkVehicle(vehicle, link);
235                return false;
236        }
237
238        @Override
239        public void notifyIterationStarts(IterationStartsEvent event) {
240                this.carsharingSupplyContainer.populateSupply();
241                this.currentDemand.reset();
242        }
