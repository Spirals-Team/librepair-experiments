001package org.matsim.contrib.carsharing.replanning;
002
003import java.util.ArrayList;
004import java.util.Arrays;
005import java.util.Collection;
006import java.util.Collections;
007import java.util.List;
008
009import org.matsim.api.core.v01.Id;
010import org.matsim.api.core.v01.Scenario;
011import org.matsim.api.core.v01.TransportMode;
012import org.matsim.api.core.v01.population.Person;
013import org.matsim.api.core.v01.population.Plan;
014import org.matsim.contrib.carsharing.manager.demand.membership.MembershipContainer;
015import org.matsim.core.population.PersonUtils;
016import org.matsim.core.population.algorithms.PermissibleModesCalculator;
017
018public class CarsharingSubTourPermissableModesCalculator implements PermissibleModesCalculator{
019
020        private final Scenario scenario;
021        private final List<String> availableModes;
022        private final List<String> availableModesWithoutCar;
023        private MembershipContainer memberships;
024        public CarsharingSubTourPermissableModesCalculator(final Scenario scenario, final String[] availableModes, 
025                        MembershipContainer memberships) {
026                this.scenario = scenario;
027                this.availableModes = Arrays.asList(availableModes);
028                this.memberships = memberships;
029                if ( this.availableModes.contains( TransportMode.car ) ) {
030                        final List<String> l = new ArrayList<String>( this.availableModes );
031                        while ( l.remove( TransportMode.car ) ) {}
032                        this.availableModesWithoutCar = Collections.unmodifiableList( l );
033                }
034                else {
035                        this.availableModesWithoutCar = this.availableModes;
036                }
037        }
038        
039        @Override
040        public Collection<String> getPermissibleModes(Plan plan) {
041                final Person person;
042                Id<Person> personId;
043                List<String> l = new ArrayList<String>();
044                try {
045                        person = plan.getPerson();
046                        personId = person.getId();
047                }
048                catch (ClassCastException e) {
049                        throw new IllegalArgumentException( "I need a PersonImpl to get car availability" );
050                }
051                
052                 if (Boolean.parseBoolean(scenario.getConfig().getModule("TwoWayCarsharing").getParams().get("useTwoWayCarsharing"))
053                
054                                 && this.memberships.getPerPersonMemberships().get(personId).getMembershipsPerCSType().containsKey("twoway")) {
055                         
056                         
057                         l.add("twoway");
058                         
059                 }
060                
061                return l;
062        }
063
064}
