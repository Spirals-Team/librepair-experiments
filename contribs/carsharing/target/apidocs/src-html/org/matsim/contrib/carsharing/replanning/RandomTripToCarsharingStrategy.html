001package org.matsim.contrib.carsharing.replanning;
002
003import org.matsim.api.core.v01.Scenario;
004import org.matsim.api.core.v01.population.HasPlansAndId;
005import org.matsim.api.core.v01.population.Person;
006import org.matsim.api.core.v01.population.Plan;
007import org.matsim.api.core.v01.replanning.PlanStrategyModule;
008import org.matsim.contrib.carsharing.manager.demand.membership.MembershipContainer;
009import org.matsim.core.replanning.PlanStrategy;
010import org.matsim.core.replanning.PlanStrategyImpl;
011import org.matsim.core.replanning.ReplanningContext;
012import org.matsim.core.replanning.modules.ReRoute;
013import org.matsim.core.replanning.selectors.RandomPlanSelector;
014
015import com.google.inject.Inject;
016import org.matsim.core.router.TripRouter;
017
018import javax.inject.Provider;
019
020
021public class RandomTripToCarsharingStrategy implements PlanStrategy{
022
023        private final PlanStrategyImpl strategy;
024        @Inject
025        public RandomTripToCarsharingStrategy(final Scenario scenario, Provider<TripRouter> tripRouterProvider, MembershipContainer memberships) {
026                this.strategy = new PlanStrategyImpl( new RandomPlanSelector<Plan, Person>() );
027                        
028                //addStrategyModule( new TripsToLegsModule(controler.getConfig() ) );   //lets try without this, not sure if it is needed
029                CarsharingTripModeChoice smc = new CarsharingTripModeChoice(tripRouterProvider, scenario, memberships);
030                addStrategyModule(smc );
031                addStrategyModule( new ReRoute(scenario, tripRouterProvider) );
032        }
033        public void addStrategyModule(final PlanStrategyModule module) {
034                strategy.addStrategyModule(module);
035        }
036        @Override
037        public void run(HasPlansAndId<Plan, Person> person) {
038                strategy.run(person);
039                
040        }
041
042        @Override
043        public void init(ReplanningContext replanningContext) {
044                strategy.init(replanningContext);
045                
046        }
047
048        @Override
049        public void finish() {
050                // TODO Auto-generated method stub
051                strategy.finish();
052        }
053        @Override
054        public String toString() {
055                return strategy.toString();
056        }
057}
