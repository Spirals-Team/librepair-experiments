001package org.matsim.contrib.carsharing.manager.supply;
002
003import java.util.HashMap;
004import java.util.Map;
005import java.util.Set;
006
007import org.matsim.api.core.v01.Scenario;
008import org.matsim.api.core.v01.network.Link;
009import org.matsim.api.core.v01.network.Network;
010import org.matsim.contrib.carsharing.config.CarsharingConfigGroup;
011import org.matsim.contrib.carsharing.config.FreeFloatingConfigGroup;
012import org.matsim.contrib.carsharing.config.FreefloatingAreasReader;
013import org.matsim.contrib.carsharing.readers.CarsharingXmlReaderNew;
014import org.matsim.contrib.carsharing.vehicles.CSVehicle;
015import org.matsim.core.config.ConfigGroup;
016
017import com.google.inject.Inject;
018
019/**
020 * @author balac
021 */
022public class CarsharingSupplyContainer implements CarsharingSupplyInterface {
023        private Map<String, CompanyContainer> companies = new HashMap<String, CompanyContainer>();
024        private Map<String, CompanyAgent> companyAgents = new HashMap<>();
025        private Map<String, CSVehicle> allVehicles = new HashMap<String, CSVehicle>();
026        private Map<CSVehicle, Link> allVehicleLocations = new HashMap<CSVehicle, Link>();
027        private Set<String> companyNames;
028        private Scenario scenario;
029        
030        public CarsharingSupplyContainer(Scenario scenario) {
031                this.scenario = scenario;
032                //populateSupply();
033        }       
034        
035        
036        @Override
037        public Map<CSVehicle, Link> getAllVehicleLocations() {
038                return allVehicleLocations;
039        }
040
041        
042        @Override
043        public Map<String, CSVehicle> getAllVehicles() {
044                return allVehicles;
045        }
046        
047        
048        @Override
049        public CompanyContainer getCompany(String companyId) {
050                
051                return this.companies.get(companyId);
052        }
053        
054        
055        
056        @Override
057        public CSVehicle getVehicleWithId (String vehicleId) {
058                
059                return this.allVehicles.get(vehicleId);
060        }
061
062        
063        @Override
064        public CSVehicle findClosestAvailableVehicle(Link startLink, String carsharingType, String typeOfVehicle,
065                        String companyId, double searchDistance) {
066                
067                CompanyContainer companyContainer = this.companies.get(companyId);
068                VehiclesContainer vehiclesContainer = companyContainer.getVehicleContainer(carsharingType);
069                return vehiclesContainer.findClosestAvailableVehicle(startLink, typeOfVehicle, searchDistance);         
070        }               
071
072        
073        @Override
074        public Link findClosestAvailableParkingSpace(Link destinationLink, String carsharingType,
075                        String companyId, double searchDistance) {
076                CompanyContainer companyContainer = this.companies.get(companyId);
077                VehiclesContainer vehiclesContainer = companyContainer.getVehicleContainer(carsharingType);
078                return vehiclesContainer.findClosestAvailableParkingLocation(destinationLink, searchDistance);  
079        }
080
081        
082        @Override
083        public void populateSupply() {
084                Network network = this.scenario.getNetwork();
085
086                final CarsharingConfigGroup configGroup = (CarsharingConfigGroup)
087                                scenario.getConfig().getModule( CarsharingConfigGroup.GROUP_NAME );
088
089                final FreeFloatingConfigGroup ffConfigGroup = (FreeFloatingConfigGroup)
090                                this.scenario.getConfig().getModule(FreeFloatingConfigGroup.GROUP_NAME);
091
092                CarsharingXmlReaderNew reader = new CarsharingXmlReaderNew(network);
093
094                String areasFile = ffConfigGroup.getAreas();
095                if (areasFile != null) {
096                        FreefloatingAreasReader ffAreasReader = new FreefloatingAreasReader();
097                        ffAreasReader.parse(ConfigGroup.getInputFileURL(scenario.getConfig().getContext(), areasFile));
098                        reader.setFreefloatingAreas(ffAreasReader.getFreefloatingAreas());
099                }
100
101                reader.readFile(configGroup.getvehiclelocations());
102                this.companies = reader.getCompanies();
103                this.allVehicleLocations = reader.getAllVehicleLocations();
104                this.allVehicles = reader.getAllVehicles();
105                this.companyNames = reader.getCompanyNames();
106                
107                for (String companyName : this.companyNames) {
108                        
109                        CompanyAgent companyAGent = new CompanyAgentImpl(this.companies.get(companyName), "");
110                        this.companyAgents.put(companyName, companyAGent);
111                }
112                
113        }
114
115        
116        @Override
117        public Set<String> getCompanyNames() {
118                return companyNames;
119        }
120
121
122        @Override
123        public Map<String, CompanyAgent> getCompanyAgents() {
124                return companyAgents;
125        }       
126}
