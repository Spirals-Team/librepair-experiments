001package org.matsim.contrib.carsharing.manager.demand;
002
003import java.util.HashMap;
004import java.util.Map;
005
006import org.matsim.api.core.v01.Id;
007import org.matsim.api.core.v01.Scenario;
008import org.matsim.api.core.v01.events.LinkLeaveEvent;
009import org.matsim.api.core.v01.events.PersonEntersVehicleEvent;
010import org.matsim.api.core.v01.events.PersonLeavesVehicleEvent;
011import org.matsim.api.core.v01.events.handler.LinkLeaveEventHandler;
012import org.matsim.api.core.v01.events.handler.PersonEntersVehicleEventHandler;
013import org.matsim.api.core.v01.events.handler.PersonLeavesVehicleEventHandler;
014import org.matsim.api.core.v01.network.Network;
015import org.matsim.api.core.v01.population.Person;
016import org.matsim.contrib.carsharing.events.EndRentalEvent;
017import org.matsim.contrib.carsharing.events.StartRentalEvent;
018import org.matsim.contrib.carsharing.events.handlers.EndRentalEventHandler;
019import org.matsim.contrib.carsharing.events.handlers.StartRentalEventHandler;
020import org.matsim.contrib.carsharing.manager.supply.CarsharingSupplyInterface;
021import org.matsim.vehicles.Vehicle;
022
023import com.google.inject.Inject;
024/** 
025 * @author balac
026 */
027public class DemandHandler implements PersonLeavesVehicleEventHandler, 
028PersonEntersVehicleEventHandler, LinkLeaveEventHandler, StartRentalEventHandler, EndRentalEventHandler {
029        
030        @Inject Scenario scenario;
031        @Inject CarsharingSupplyInterface carsharingSupplyContainer;
032
033        private Map<Id<Person>, AgentRentals> agentRentalsMap = new HashMap<Id<Person>, AgentRentals>();        
034
035        private Map<Id<Vehicle>, VehicleRentals> vehicleRentalsMap = new HashMap<Id<Vehicle>, VehicleRentals>();
036
037        private Map<Id<Vehicle>, Id<Person>> vehiclePersonMap = new HashMap<Id<Vehicle>, Id<Person>>();
038
039        private Map<Id<Person>, Double> enterVehicleTimes = new HashMap<Id<Person>, Double>();  
040
041
042        @Override
043        public void reset(int iteration) {
044                agentRentalsMap = new HashMap<Id<Person>, AgentRentals>();      
045
046                vehicleRentalsMap = new HashMap<Id<Vehicle>, VehicleRentals>();
047
048                vehiclePersonMap = new HashMap<Id<Vehicle>, Id<Person>>();
049
050                enterVehicleTimes = new HashMap<Id<Person>, Double>();
051        }
052
053        @Override
054        public void handleEvent(EndRentalEvent event) {
055                
056                AgentRentals agentRentals = this.agentRentalsMap.get(event.getPersonId());
057                
058                RentalInfo info = agentRentals.getStatsPerVehicle().get(event.getvehicleId());
059                agentRentals.getStatsPerVehicle().remove(event.getvehicleId());
060                info.setEndTime(event.getTime());
061                info.setEndLinkId(event.getLinkId());
062
063                agentRentals.getArr().add(info);
064
065                Id<Vehicle> vehicleId = Id.create(event.getvehicleId(), Vehicle.class);
066                if (!this.vehicleRentalsMap.containsKey(vehicleId)) {
067                        this.vehicleRentalsMap.put(vehicleId, new VehicleRentals(vehicleId));
068                }
069
070                this.vehicleRentalsMap.get(vehicleId).getRentals().add(info);
071}
072
073        @Override
074        public void handleEvent(StartRentalEvent event) {
075                RentalInfo info = new RentalInfo();
076                info.setCarsharingType(event.getCarsharingType());
077                info.setAccessStartTime(event.getTime());
078                info.setStartTime(event.getTime());
079                info.setOriginLinkId(event.getOriginLinkId());
080                info.setPickupLinkId(event.getPickuplinkId());
081
082                if (agentRentalsMap.containsKey(event.getPersonId())) {
083                        AgentRentals agentRentals = this.agentRentalsMap.get(event.getPersonId());
084                        agentRentals.getStatsPerVehicle().put(event.getvehicleId(), info);
085                        
086                }
087                else {
088                        AgentRentals agentRentals = new AgentRentals(event.getPersonId());
089                        agentRentalsMap.put(event.getPersonId(), agentRentals);                 
090                        agentRentals.getStatsPerVehicle().put(event.getvehicleId(), info);                      
091                }               
092        }
093
094        @Override
095        public void handleEvent(LinkLeaveEvent event) {
096                if (carsharingTrip(event.getVehicleId())) {
097                        Id<Person> personId = this.vehiclePersonMap.get(event.getVehicleId());
098                        Network network = this.scenario.getNetwork();
099
100                        if (agentRentalsMap.containsKey(personId)) {
101                                AgentRentals agentRentals = this.agentRentalsMap.get(personId);
102                                RentalInfo info = agentRentals.getStatsPerVehicle().get(event.getVehicleId().toString());
103                                info.setVehId(event.getVehicleId());
104                                info.setDistance(info.getDistance() + network.getLinks().get(event.getLinkId()).getLength());
105                        }
106                }
107                
108        }
109
110        @Override
111        public void handleEvent(PersonEntersVehicleEvent event) {
112                if (carsharingTrip(event.getVehicleId())) {
113                        this.vehiclePersonMap.put(event.getVehicleId(), event.getPersonId());
114                        this.enterVehicleTimes.put(event.getPersonId(), event.getTime());
115                        
116                        Id<Person> personId = this.vehiclePersonMap.get(event.getVehicleId());
117
118                        if (agentRentalsMap.containsKey(event.getPersonId())) {
119                                AgentRentals agentRentals = this.agentRentalsMap.get(personId);
120                                RentalInfo info = agentRentals.getStatsPerVehicle().get(event.getVehicleId().toString());
121
122                                if (info.getAccessEndTime() == 0.0)
123                                        info.setAccessEndTime(event.getTime());
124                        }
125                }
126                
127        }
128
129        @Override
130        public void handleEvent(PersonLeavesVehicleEvent event) {
131                
132                if (carsharingTrip(event.getVehicleId())) {
133                        double enterTime = this.enterVehicleTimes.get(event.getPersonId());     
134                        double totalTime = event.getTime() - enterTime;
135
136                        Id<Person> personId = this.vehiclePersonMap.get(event.getVehicleId());
137
138                        if (agentRentalsMap.containsKey(event.getPersonId())) {
139                                AgentRentals agentRentals = this.agentRentalsMap.get(personId);
140                                RentalInfo info = agentRentals.getStatsPerVehicle().get(event.getVehicleId().toString());
141                                info.setInVehicleTime(info.getInVehicleTime() + totalTime);
142                        }
143                }
144        }
145
146        private boolean carsharingTrip(Id<Vehicle> vehicleId) {
147
148                return this.carsharingSupplyContainer.getAllVehicles().containsKey(vehicleId.toString());               
149        }
150
151        public Map<Id<Person>, AgentRentals> getAgentRentalsMap() {
152                return agentRentalsMap;
153        }
154
155        public Map<Id<Vehicle>, VehicleRentals> getVehicleRentalsMap() {
156                return this.vehicleRentalsMap;
157        }
158}
