001package org.matsim.contrib.carsharing.manager.demand;
002
003import org.matsim.api.core.v01.Id;
004import org.matsim.api.core.v01.network.Link;
005import org.matsim.vehicles.Vehicle;
006/** 
007 * @author balac
008 */
009public class RentalInfo {
010        private String carsharingType;
011        public String getCarsharingType() {
012                return carsharingType;
013        }
014        public void setCarsharingType(String carsharingType) {
015                this.carsharingType = carsharingType;
016        }
017        public double getStartTime() {
018                return startTime;
019        }
020        public void setStartTime(double startTime) {
021                this.startTime = startTime;
022        }
023        public double getEndTime() {
024                return endTime;
025        }
026        public void setEndTime(double endTime) {
027                this.endTime = endTime;
028        }
029        public Id<Link> getOriginLinkId() {
030                return originLinkId;
031        }
032        public void setOriginLinkId(Id<Link> originLinkId) {
033                this.originLinkId = originLinkId;
034        }
035        public Id<Link> getPickupLinkId() {
036                return pickupLinkId;
037        }
038        public void setPickupLinkId(Id<Link> pickupLinkId) {
039                this.pickupLinkId = pickupLinkId;
040        }
041        public Id<Link> getDropoffLinkId() {
042                return dropoffLinkId;
043        }
044        public void setDropoffLinkId(Id<Link> dropoffLinkId) {
045                this.dropoffLinkId = dropoffLinkId;
046        }
047        public Id<Link> getEndLinkId() {
048                return endLinkId;
049        }
050        public void setEndLinkId(Id<Link> endLinkId) {
051                this.endLinkId = endLinkId;
052        }
053        public double getDistance() {
054                return distance;
055        }
056        public void setDistance(double distance) {
057                this.distance = distance;
058        }
059        public double getInVehicleTime() {
060                return inVehicleTime;
061        }
062        public void setInVehicleTime(double inVehicleTime) {
063                this.inVehicleTime = inVehicleTime;
064        }
065        public double getAccessStartTime() {
066                return accessStartTime;
067        }
068        public void setAccessStartTime(double accessStartTime) {
069                this.accessStartTime = accessStartTime;
070        }
071        public double getAccessEndTime() {
072                return accessEndTime;
073        }
074        public void setAccessEndTime(double accessEndTime) {
075                this.accessEndTime = accessEndTime;
076        }
077        public double getEgressStartTime() {
078                return egressStartTime;
079        }
080        public void setEgressStartTime(double egressStartTime) {
081                this.egressStartTime = egressStartTime;
082        }
083        public double getEgressEndTime() {
084                return egressEndTime;
085        }
086        public void setEgressEndTime(double egressEndTime) {
087                this.egressEndTime = egressEndTime;
088        }
089        public Id<Vehicle> getVehId() {
090                return vehId;
091        }
092        public void setVehId(Id<Vehicle> vehId) {
093                this.vehId = vehId;
094        }
095        private double startTime = 0.0;
096        private double endTime = 0.0;
097        private Id<Link> originLinkId = null;
098        private Id<Link> pickupLinkId = null;
099        private Id<Link> dropoffLinkId = null;
100        private Id<Link> endLinkId = null;
101        private double distance = 0.0;
102        private double inVehicleTime = 0.0;
103        private double accessStartTime = 0.0;
104        private double accessEndTime = 0.0;
105        private double egressStartTime = 0.0;
106        private double egressEndTime = 0.0;
107        private Id<Vehicle> vehId = null;
108        public String toString() {
109                
110                return carsharingType + "," + Double.toString(startTime) + "," + Double.toString(endTime) + "," +
111                                originLinkId + "," + pickupLinkId + "," +       dropoffLinkId + "," + endLinkId + "," + 
112                                Double.toString(distance) + "," + Double.toString(inVehicleTime) + "," +
113                                Double.toString(accessEndTime - accessStartTime) + "," + Double.toString(egressEndTime - egressStartTime) +
114                "," + vehId;
115        }
116}
