001package org.matsim.contrib.carsharing.manager.routers;
002
003import org.matsim.api.core.v01.Id;
004import org.matsim.api.core.v01.network.Link;
005import org.matsim.api.core.v01.population.Leg;
006import org.matsim.api.core.v01.population.Person;
007import org.matsim.api.core.v01.population.PopulationFactory;
008import org.matsim.api.core.v01.population.Route;
009import org.matsim.core.network.NetworkUtils;
010import org.matsim.core.population.routes.NetworkRoute;
011import org.matsim.core.population.routes.RouteFactories;
012import org.matsim.core.router.util.LeastCostPathCalculator;
013import org.matsim.core.router.util.LeastCostPathCalculator.Path;
014import org.matsim.core.utils.geometry.CoordUtils;
015import org.matsim.vehicles.Vehicle;
016
017public class RouterUtils {
018        
019        public static Leg createCarLeg(PopulationFactory pf, LeastCostPathCalculator pathCalculator,
020                        Person person, Link startLink, Link destinationLink, String mode, 
021                        String vehicleId, double now) {
022                
023                
024                RouteFactories routeFactory = ((PopulationFactory)pf).getRouteFactories() ;
025                
026                Vehicle vehicle = null ;
027                Path path = pathCalculator.calcLeastCostPath(startLink.getToNode(), destinationLink.getFromNode(), 
028                                now, person, vehicle ) ;
029                
030                NetworkRoute carRoute = routeFactory.createRoute(NetworkRoute.class, startLink.getId(), destinationLink.getId() );
031                carRoute.setLinkIds(startLink.getId(), NetworkUtils.getLinkIds( path.links), destinationLink.getId());
032                carRoute.setTravelTime( path.travelTime );              
033                
034                carRoute.setVehicleId( Id.create( (vehicleId), Vehicle.class) ) ;
035
036                Leg carLeg = pf.createLeg(mode);
037                carLeg.setTravelTime( path.travelTime );
038                carLeg.setRoute(carRoute);
039                
040                return carLeg;
041        }
042
043        public static Leg createWalkLeg(PopulationFactory pf, Link startLink, 
044                        Link destinationLink, String mode, double now) {
045                
046                RouteFactories routeFactory = ((PopulationFactory)pf).getRouteFactories() ;
047                
048                Route routeWalk = routeFactory.createRoute( Route.class, startLink.getId(), destinationLink.getId() ) ; 
049                
050                double egressDist = CoordUtils.calcEuclideanDistance(startLink.getCoord(), destinationLink.getCoord()) * 1.3;
051                egressDist = egressDist > 0 ? egressDist : 1; 
052                routeWalk.setTravelTime( (egressDist / 1.0));
053                routeWalk.setDistance(egressDist);      
054
055                final Leg walkLeg = pf.createLeg( mode );
056                walkLeg.setRoute(routeWalk);
057
058                return walkLeg;         
059        }
060        
061}
