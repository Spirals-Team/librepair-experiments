001package org.matsim.contrib.carsharing.manager.demand;
002
003import java.util.Collection;
004import java.util.HashMap;
005import java.util.Map;
006
007import org.matsim.api.core.v01.Coord;
008import org.matsim.api.core.v01.network.Link;
009import org.matsim.api.core.v01.network.Network;
010import org.matsim.contrib.carsharing.vehicles.CSVehicle;
011import org.matsim.core.utils.collections.QuadTree;
012/** 
013 * @author balac
014 */
015public class CarsharingCurrentRentalsInfo {
016        
017        private Map<String, QuadTree<CSVehicle>> currentRentals = new HashMap<String, QuadTree<CSVehicle>>(); 
018        
019        
020        private Network network;
021        
022        public CarsharingCurrentRentalsInfo(Network network) {
023                
024                this.network = network;
025        }
026        
027        public void addType(String carsharingType) {
028                
029                double minx = (1.0D / 0.0D);
030            double miny = (1.0D / 0.0D);
031                double maxx = (-1.0D / 0.0D);
032                double maxy = (-1.0D / 0.0D);
033
034            for (Link l : this.network.getLinks().values()) {
035                
036                    if (l.getCoord().getX() < minx) minx = l.getCoord().getX();
037                    if (l.getCoord().getY() < miny) miny = l.getCoord().getY();
038                    if (l.getCoord().getX() > maxx) maxx = l.getCoord().getX();
039                    if (l.getCoord().getY() <= maxy) continue; maxy = l.getCoord().getY();
040                }
041                minx -= 1.0D; miny -= 1.0D; maxx += 1.0D; maxy += 1.0D;
042                
043                currentRentals.put(carsharingType, new QuadTree<CSVehicle>(minx, miny, maxx, maxy));
044        }
045        
046        public boolean hasVehicleOnLink(Link link, String type) {
047                QuadTree<CSVehicle> vehicleLocations = currentRentals.get(type);
048                Coord coord = link.getCoord();
049                Collection<CSVehicle> vehicles = vehicleLocations.getDisk(coord.getX(), coord.getY(), 0.0);
050                if (vehicles.isEmpty())
051                        return false;
052                else
053                        return true;
054        }
055        
056        public CSVehicle getVehicleOnLink(Link link, String type) {
057                QuadTree<CSVehicle> vehicleLocations = currentRentals.get(type);
058                if (vehicleLocations != null) {
059                        Coord coord = link.getCoord();
060                        Collection<CSVehicle> vehicles = vehicleLocations.getDisk(coord.getX(), coord.getY(), 0.0);
061                        
062                        if (!vehicles.isEmpty())
063                                return (CSVehicle) vehicles.iterator().next();
064                }
065
066                return null;
067        }
068
069        
070        public boolean addVehicle(Link link, CSVehicle vehicle, String type) {
071                QuadTree<CSVehicle> vehicleLocations = currentRentals.get(type);
072                Coord coord = link.getCoord();
073                
074                return vehicleLocations.put(coord.getX(), coord.getY(), vehicle);
075        }
076        
077        public boolean removeVehicle(Link link, CSVehicle vehicle, String type) {
078                QuadTree<CSVehicle> vehicleLocations = currentRentals.get(type);                
079                Coord coord = link.getCoord();
080                
081                return vehicleLocations.remove(coord.getX(), coord.getY(), vehicle);
082        }
083        
084        public Map<String, QuadTree<CSVehicle>> getCurrentRentals() {
085                return currentRentals;
086        }
087        
088}
