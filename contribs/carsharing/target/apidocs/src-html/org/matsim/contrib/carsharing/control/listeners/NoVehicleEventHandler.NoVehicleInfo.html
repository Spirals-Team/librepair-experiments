001package org.matsim.contrib.carsharing.control.listeners;
002
003import java.util.ArrayList;
004
005import org.matsim.api.core.v01.Id;
006import org.matsim.api.core.v01.network.Link;
007import org.matsim.contrib.carsharing.events.NoVehicleCarSharingEvent;
008import org.matsim.contrib.carsharing.events.handlers.NoVehicleCarSharingEventHandler;
009
010public class NoVehicleEventHandler implements NoVehicleCarSharingEventHandler {
011        ArrayList<NoVehicleInfo> noVehicle = new ArrayList<NoVehicleInfo>();
012
013        @Override
014        public void reset(int iteration) {
015                // TODO Auto-generated method stub
016                noVehicle = new ArrayList<NoVehicleInfo>();
017        }
018
019        @Override
020        public void handleEvent(NoVehicleCarSharingEvent event) {
021                // TODO Auto-generated method stub
022                NoVehicleInfo info = new NoVehicleInfo();
023                info.originLinkId = event.getOriginLinkId();
024                info.destinationLinkId = event.getDestinationLinkId();
025                info.type = event.getCarsharingType();
026                noVehicle.add(info);
027        }
028
029        public ArrayList<NoVehicleInfo> info() {
030                return this.noVehicle;
031        }
032
033        public class NoVehicleInfo {
034                Id<Link> originLinkId = null;
035
036                Id<Link> destinationLinkId = null;
037
038                String type = null;
039
040                public String toString() {
041                        return originLinkId.toString() + " - " + destinationLinkId.toString() + " " + type;
042                }
043        }
044}
