001package org.matsim.contrib.carsharing.replanning;
002
003import java.util.Collections;
004import java.util.List;
005import java.util.Random;
006
007import org.matsim.api.core.v01.Id;
008import org.matsim.api.core.v01.Scenario;
009import org.matsim.api.core.v01.population.Leg;
010import org.matsim.api.core.v01.population.Person;
011import org.matsim.api.core.v01.population.Plan;
012import org.matsim.contrib.carsharing.manager.demand.membership.MembershipContainer;
013import org.matsim.contrib.carsharing.manager.demand.membership.PersonMembership;
014import org.matsim.core.population.PopulationUtils;
015import org.matsim.core.population.algorithms.PlanAlgorithm;
016import org.matsim.core.router.StageActivityTypes;
017import org.matsim.core.router.TripRouter;
018import org.matsim.core.router.TripStructureUtils;
019import org.matsim.core.router.TripStructureUtils.Trip;
020/**
021 * @author balacm
022 */
023public final class ChooseRandomTripMode implements PlanAlgorithm {
024        
025        private final String[] possibleModes;
026        //private boolean ignoreCarAvailability = true;
027
028        private final Random rng;
029        private final Scenario scenario;
030        private MembershipContainer memberships; 
031
032        private final StageActivityTypes stageActivityTypes;
033        public ChooseRandomTripMode(final Scenario scenario, final String[] possibleModes, 
034                        final Random rng, final StageActivityTypes stageActivityTypes, MembershipContainer memberships) {
035                this.possibleModes = possibleModes.clone();
036                this.rng = rng;
037                this.stageActivityTypes = stageActivityTypes;
038                this.scenario = scenario;
039                this.memberships = memberships;
040        }
041        @Override
042        public void run(Plan plan) {
043                Id<Person> personId = plan.getPerson().getId();
044                List<Trip> t = TripStructureUtils.getTrips(plan, stageActivityTypes);
045                boolean ffcard = false;
046                boolean owcard = false;
047                int cnt = t.size();
048                if (cnt == 0) {
049                        return;
050                }
051
052                int rndIdx = this.rng.nextInt(cnt);
053                
054                for (Leg l:t.get(rndIdx).getLegsOnly()) {
055                        if (l.getMode().equals( "car" ) || l.getMode().equals( "bike" ) || l.getMode().equals("twoway")) {
056                                return;
057                        }
058                }
059                
060                PersonMembership personMemmbership = this.memberships.getPerPersonMemberships().get(personId);
061                
062                if (personMemmbership != null) {
063                        if (personMemmbership.getMembershipsPerCSType().containsKey("freefloating"))
064                                ffcard = true;
065                        if (personMemmbership.getMembershipsPerCSType().containsKey("oneway"))
066                                owcard = true;
067                }
068                
069                        //don't change the trips between the same links
070                        if (!t.get(rndIdx).getOriginActivity().getLinkId().toString().equals(t.get(rndIdx).getDestinationActivity().getLinkId().toString()))
071                                setRandomTripMode(t.get(rndIdx), plan, ffcard, owcard);
072                        else return;
073        }
074
075        private void setRandomTripMode(final Trip trip, final Plan plan, boolean ffcard, boolean owcard) {              
076                
077                //carsharing is the new trip
078                if (possibleModes.length == 2) {
079                if(rng.nextBoolean()) {
080                        if (owcard)
081                        TripRouter.insertTrip(
082                                        plan,
083                                        trip.getOriginActivity(),
084                                        Collections.singletonList( PopulationUtils.createLeg("oneway") ),
085                                        trip.getDestinationActivity());
086                }
087                else {
088                        if (ffcard)
089                        TripRouter.insertTrip(
090                                        plan,
091                                        trip.getOriginActivity(),
092                                        Collections.singletonList( PopulationUtils.createLeg("freefloating") ),
093                                        trip.getDestinationActivity());
094                }
095                }
096                else if (possibleModes.length == 1 && possibleModes[0] != null){
097                        if ((possibleModes[0].equals("freefloating") && ffcard) || (possibleModes[0].equals("oneway") && owcard))
098                        TripRouter.insertTrip(
099                                        plan,
100                                        trip.getOriginActivity(),
101                                        Collections.singletonList( PopulationUtils.createLeg(possibleModes[0]) ),
102                                        trip.getDestinationActivity());
103                }
104                else
105                        TripRouter.insertTrip(
106                                        plan,
107                                        trip.getOriginActivity(),
108                                        trip.getTripElements(),
109                                        trip.getDestinationActivity());
110        
111        }
112
113
114}
