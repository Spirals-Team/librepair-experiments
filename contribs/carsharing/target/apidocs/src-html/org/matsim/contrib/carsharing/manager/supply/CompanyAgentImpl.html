001package org.matsim.contrib.carsharing.manager.supply;
002
003import org.matsim.api.core.v01.Id;
004import org.matsim.api.core.v01.network.Link;
005import org.matsim.api.core.v01.population.Person;
006import org.matsim.contrib.carsharing.vehicles.CSVehicle;
007
008public class CompanyAgentImpl implements CompanyAgent {
009        
010        private CompanyContainer companyContainer;
011        
012        public CompanyAgentImpl(CompanyContainer companyContainer, String strategyType) {
013                this.companyContainer = companyContainer;
014        }
015        
016        @Override
017        public CSVehicle vehicleRequest(Id<Person> personId, Link locationLink, Link destinationLink,
018                        String carsharingType, String vehicleType) {
019
020                VehiclesContainer vehiclesContainer = companyContainer.getVehicleContainer(carsharingType);
021                
022                if (vehiclesContainer != null) {
023                        
024                        //Depending on the company strategy
025                        //here the company just provides the closest vehicle in the search radius
026                        CSVehicle vehicle = vehiclesContainer.findClosestAvailableVehicle(locationLink, vehicleType, 1000.0);
027                        
028                        return vehicle;
029                }
030                
031                else
032                        return null;
033                
034        }       
035        
036}
