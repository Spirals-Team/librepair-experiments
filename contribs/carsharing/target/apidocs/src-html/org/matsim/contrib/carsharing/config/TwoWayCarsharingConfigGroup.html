001package org.matsim.contrib.carsharing.config;
002
003import org.matsim.core.config.ReflectiveConfigGroup;
004/** 
005 * @author balac
006 */
007
008public class TwoWayCarsharingConfigGroup extends ReflectiveConfigGroup {
009        public static final String GROUP_NAME = "TwoWayCarsharing";
010        
011        private String travelingTwoWayCarsharing = null;
012        
013        private String constantTwoWayCarsharing = null;
014        
015        private String vehiclelocationsInputFile = null;
016        
017        private Double searchDistance = null;
018        
019        private String rentalPriceTimeTwoWayCarsharing = null;
020        
021        private String timeFeeTwoWayCarsharing = null;
022        
023        private String distanceFeeTwoWayCarsharing = null;
024        
025        private boolean useTwoWayCarsharing = false;
026        
027        public TwoWayCarsharingConfigGroup() {
028                super(GROUP_NAME);
029        }
030
031        @StringGetter( "travelingTwoWayCarsharing" )
032        public String getUtilityOfTravelling() {
033                return this.travelingTwoWayCarsharing;
034        }
035
036        @StringSetter( "travelingTwoWayCarsharing" )
037        public void setUtilityOfTravelling(final String travelingTwoWayCarsharing) {
038                this.travelingTwoWayCarsharing = travelingTwoWayCarsharing;
039        }
040
041        @StringGetter( "constantTwoWayCarsharing" )
042        public String constantTwoWayCarsharing() {
043                return this.constantTwoWayCarsharing;
044        }
045
046        @StringSetter( "constantTwoWayCarsharing" )
047        public void setConstantTwoWayCarsharing(final String constantTwoWayCarsharing) {
048                this.constantTwoWayCarsharing = constantTwoWayCarsharing;
049        }
050        
051        @StringGetter( "rentalPriceTimeTwoWayCarsharing" )
052        public String getRentalPriceTimeTwoWayCarsharing() {
053                return this.rentalPriceTimeTwoWayCarsharing;
054        }
055
056        @StringSetter( "rentalPriceTimeTwoWayCarsharing" )
057        public void setRentalPriceTimeTwoWayCarsharing(final String rentalPriceTimeTwoWayCarsharing) {
058                this.rentalPriceTimeTwoWayCarsharing = rentalPriceTimeTwoWayCarsharing;
059        }
060        
061        @StringGetter( "vehiclelocationsTwoWayCarsharing" )
062        public String getvehiclelocations() {
063                return this.vehiclelocationsInputFile;
064        }
065
066        @StringSetter( "vehiclelocationsTwoWayCarsharing" )
067        public void setvehiclelocations(final String vehiclelocationsInputFile) {
068                this.vehiclelocationsInputFile = vehiclelocationsInputFile;
069        }
070        
071        @StringGetter( "searchDistanceTwoWayCarsharing" )
072        public Double getsearchDistance() {
073                return this.searchDistance;
074        }
075
076        @StringSetter( "searchDistanceTwoWayCarsharing" )
077        public void setsearchDistance(final String searchDistance) {
078                this.searchDistance = Double.parseDouble(searchDistance);
079        }
080        
081        @StringGetter( "timeFeeTwoWayCarsharing" )
082        public String timeFeeTwoWayCarsharing() {
083                return this.timeFeeTwoWayCarsharing;
084        }
085
086        @StringSetter( "timeFeeTwoWayCarsharing" )
087        public void setTimeFeeTwoWayCarsharing(final String timeFeeTwoWayCarsharing) {
088                this.timeFeeTwoWayCarsharing = timeFeeTwoWayCarsharing;
089        }
090        
091        @StringGetter( "distanceFeeTwoWayCarsharing" )
092        public String distanceFeeTwoWayCarsharing() {
093                return this.distanceFeeTwoWayCarsharing;
094        }
095
096        @StringSetter( "distanceFeeTwoWayCarsharing" )
097        public void setDistanceFeeTwoWayCarsharing(final String distanceFeeTwoWayCarsharing) {
098                this.distanceFeeTwoWayCarsharing = distanceFeeTwoWayCarsharing;
099        }
100        
101        @StringGetter( "useTwoWayCarsharing" )
102        public boolean useTwoWayCarsharing() {
103                return this.useTwoWayCarsharing;
104        }
105
106        @StringSetter( "useTwoWayCarsharing" )
107        public void setUseTwoWayCarsharing(final boolean useTwoWayCarsharing) {
108                this.useTwoWayCarsharing = useTwoWayCarsharing;
109        }
110        
111        
112}
