001package org.matsim.contrib.carsharing.manager.supply.costs;
002
003import java.util.HashMap;
004import java.util.Map;
005
006import org.matsim.contrib.carsharing.manager.demand.RentalInfo;
007/** 
008 * @author balac
009 */
010public class CompanyCosts {     
011        
012        private Map<String, CostCalculation> perTypeCostCalculator = new HashMap<String, CostCalculation>();
013        
014        public CompanyCosts(Map<String, CostCalculation> perTypeCostCalculator) {
015                
016                this.perTypeCostCalculator = perTypeCostCalculator;
017        }
018        
019        public double calcCost(String carsharingType, RentalInfo rentalInfo) {
020                
021                return this.perTypeCostCalculator.get(carsharingType).getCost(rentalInfo);
022        }
023
024}
