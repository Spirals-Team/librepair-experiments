045        public static CostsCalculatorContainer createCompanyCostsStructure(Set<String> companies) {
046                
047                CostsCalculatorContainer companyCostsContainer = new CostsCalculatorContainer();
048                
049                for (String s : companies) {
050                        
051                        Map<String, CostCalculation> costCalculations = new HashMap<String, CostCalculation>();
052                        
053                        //=== here customizable cost structures come in ===
054                        //===what follows is just an example!! and should be modified according to the study at hand===
055                        costCalculations.put("freefloating", new CostCalculationExample());
056                        costCalculations.put("twoway", new CostCalculationExample());
057                        costCalculations.put("oneway", new CostCalculationExample());
058                        CompanyCosts companyCosts = new CompanyCosts(costCalculations);
059                        
060                        companyCostsContainer.getCompanyCostsMap().put(s, companyCosts);
061                }
062                
063                return companyCostsContainer;
064                
065        }
066        public static AbstractModule createRoutingModule() {
067
068                //=== routing moduels for carsharing trips ===
069        return new AbstractModule() {
070
071            @Override
072            public void install() {
073                addRoutingModuleBinding("twoway").toInstance(new TwoWayCarsharingRoutingModule());
074                addRoutingModuleBinding("freefloating").toInstance(new FreeFloatingRoutingModule());
075                addRoutingModuleBinding("oneway").toInstance(new OneWayCarsharingRoutingModule());
076                bind(MainModeIdentifier.class).toInstance(new MainModeIdentifier() {
077                    final MainModeIdentifier defaultModeIdentifier = new MainModeIdentifierImpl();
078
079                    @Override
080                    public String identifyMainMode(
081                            final List<? extends PlanElement> tripElements) {
082                        // we still need to provide a way to identify our trips
083                        // as being twowaycarsharing trips.
084                        // This is for instance used at re-routing.
085                        for ( PlanElement pe : tripElements ) {
086                            if ( pe instanceof Leg && ((Leg) pe).getMode().equals( "twoway" ) ) {
087                                return "twoway";
088                            }
089                            else if ( pe instanceof Leg && ((Leg) pe).getMode().equals( "oneway" ) ) {
090                                return "oneway";
091                            }
092                            else if ( pe instanceof Leg && ((Leg) pe).getMode().equals( "freefloating" ) ) {
093                                return "freefloating";
094                            }
095                        }
096                        // if the trip doesn't contain a carsharing leg,
097                        // fall back to the default identification method.
098                        return defaultModeIdentifier.identifyMainMode( tripElements );
099                    }
100                });
101            }
102        };
103        }
