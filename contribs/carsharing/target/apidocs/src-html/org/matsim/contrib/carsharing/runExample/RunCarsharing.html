054public class RunCarsharing {
055
056        public static void main(String[] args) {
057                Logger.getLogger( "org.matsim.core.controler.Injector" ).setLevel(Level.OFF);
058
059                final Config config = ConfigUtils.loadConfig(args[0]);
060                
061                if(Integer.parseInt(config.getModule("qsim").getValue("numberOfThreads")) > 1)
062                        Logger.getLogger( "org.matsim.core.controler" ).warn("Carsharing contrib is not stable for parallel qsim!! If the error occures please use 1 as the number of threads.");
063                
064                CarsharingUtils.addConfigModules(config);
065
066                final Scenario sc = ScenarioUtils.loadScenario(config);
067
068                final Controler controler = new Controler( sc );
069                
070                installCarSharing(controler);
071                
072                controler.run();
073
074        }
075
076        public static void installCarSharing(final Controler controler) {               
077                
078                final Scenario scenario = controler.getScenario();
079                CarsharingXmlReaderNew reader = new CarsharingXmlReaderNew(scenario.getNetwork());
080                
081                final CarsharingConfigGroup configGroup = (CarsharingConfigGroup)
082                                scenario.getConfig().getModule( CarsharingConfigGroup.GROUP_NAME );
083
084                reader.readFile(configGroup.getvehiclelocations());
085                
086                Set<String> carsharingCompanies = reader.getCompanies().keySet();
087                
088                MembershipReader membershipReader = new MembershipReader();
089                
090                membershipReader.readFile(configGroup.getmembership());
091
092                final MembershipContainer memberships = membershipReader.getMembershipContainer();
093                
094                final CostsCalculatorContainer costsCalculatorContainer = CarsharingUtils.createCompanyCostsStructure(carsharingCompanies);
095                
096                final CarsharingListener carsharingListener = new CarsharingListener();
097                //final CarsharingSupplyInterface carsharingSupplyContainer = new CarsharingSupplyContainer(controler.getScenario());
098                
099                final KeepingTheCarModel keepingCarModel = new KeepingTheCarModelExample();
100                final ChooseTheCompany chooseCompany = new ChooseTheCompanyExample();
101                final ChooseVehicleType chooseCehicleType = new ChooseVehicleTypeExample();
102                final RouterProvider routerProvider = new RouterProviderImpl();
103                final CurrentTotalDemand currentTotalDemand = new CurrentTotalDemandImpl(controler.getScenario().getNetwork());
104                //final CarsharingManagerInterface carsharingManager = new CarsharingManagerNew();
105                final RouteCarsharingTrip routeCarsharingTrip = new RouteCarsharingTripImpl();
106                final VehicleChoiceAgent vehicleChoiceAgent = new VehicleChoiceAgentImpl();
107                //===adding carsharing objects on supply and demand infrastructure ===
108                
109                controler.addOverridingModule(new AbstractModule() {
110
111                        @Override
112                        public void install() {
113                                bind(KeepingTheCarModel.class).toInstance(keepingCarModel);
114                                bind(ChooseTheCompany.class).toInstance(chooseCompany);
115                                bind(ChooseVehicleType.class).toInstance(chooseCehicleType);
116                                bind(RouterProvider.class).toInstance(routerProvider);
117                                bind(CurrentTotalDemand.class).toInstance(currentTotalDemand);
118                                bind(RouteCarsharingTrip.class).toInstance(routeCarsharingTrip);
119                                bind(CostsCalculatorContainer.class).toInstance(costsCalculatorContainer);
120                                bind(MembershipContainer.class).toInstance(memberships);
121                            bind(CarsharingSupplyInterface.class).to(CarsharingSupplyContainer.class);
122                            bind(CarsharingManagerInterface.class).to(CarsharingManagerNew.class);
123                            bind(VehicleChoiceAgent.class).toInstance(vehicleChoiceAgent);
124                            bind(DemandHandler.class).asEagerSingleton();
125                        }                       
126                        @Provides @Singleton
127                        CarsharingSupplyContainer provideCarsharingSupplyContainer(Scenario scenario) {
128                            return new CarsharingSupplyContainer(scenario);
129                        }
130                        
131                        
132                
133                });             
134                
135                //=== carsharing specific replanning strategies ===
136                
137                controler.addOverridingModule( new AbstractModule() {
138                        @Override
139                        public void install() {
140                                this.addPlanStrategyBinding("RandomTripToCarsharingStrategy").to( RandomTripToCarsharingStrategy.class ) ;
141                                this.addPlanStrategyBinding("CarsharingSubtourModeChoiceStrategy").to( CarsharingSubtourModeChoiceStrategy.class ) ;
142                        }
143                });
144                
145                //=== adding qsimfactory, controller listeners and event handlers
146                controler.addOverridingModule(new AbstractModule() {
147                        @Override
148                        public void install() {
149                                bindMobsim().toProvider(CarsharingQsimFactoryNew.class);
150                        addControlerListenerBinding().toInstance(carsharingListener);
151                        addControlerListenerBinding().to(CarsharingManagerNew.class);                   
152                                //bindScoringFunctionFactory().to(CarsharingScoringFunctionFactory.class);                    
153                        addEventHandlerBinding().to(PersonArrivalDepartureHandler.class);
154                        addEventHandlerBinding().to(DemandHandler.class);
155                        }
156                });
157                //=== adding carsharing specific scoring factory ===
158                controler.addOverridingModule(new AbstractModule() {
159                        
160                        @Override
161                        public void install() {
162                                        
163                                bindScoringFunctionFactory().to(CarsharingScoringFunctionFactory.class);        
164                        }
165                });
166
167                //=== routing moduels for carsharing trips ===
168
169                controler.addOverridingModule(CarsharingUtils.createRoutingModule());                   
170        }
