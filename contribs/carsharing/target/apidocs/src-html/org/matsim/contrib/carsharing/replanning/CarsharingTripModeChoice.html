001package org.matsim.contrib.carsharing.replanning;
002
003import org.matsim.api.core.v01.Scenario;
004import org.matsim.contrib.carsharing.manager.demand.membership.MembershipContainer;
005import org.matsim.core.gbl.MatsimRandom;
006import org.matsim.core.population.algorithms.PlanAlgorithm;
007import org.matsim.core.replanning.modules.AbstractMultithreadedModule;
008import org.matsim.core.router.TripRouter;
009
010import javax.inject.Provider;
011
012/**
013 * @author balacm
014 */
015public class CarsharingTripModeChoice extends AbstractMultithreadedModule{
016
017        /*package*/ final static String CONFIG_MODULE = "ft";
018        /*package*/ final static String CONFIG_PARAM_MODES = "modes";
019        /*package*/ final static String CONFIG_PARAM_IGNORECARAVAILABILITY = "ignoreCarAvailability";
020
021        private final Provider<TripRouter> tripRouterProvider;
022
023        private String[] availableModes = null;
024        private final Scenario scenario;
025        private MembershipContainer memberships; 
026        public CarsharingTripModeChoice(Provider<TripRouter> tripRouterProvider, final Scenario scenario, MembershipContainer memberships ) {
027                super(scenario.getConfig().global().getNumberOfThreads());
028                this.tripRouterProvider = tripRouterProvider;
029                this.memberships = memberships;
030                // try to get the modes from the "changeLegMode" module of the config file
031
032                this.scenario = scenario;
033                // try to get the modes from the "changeLegMode" module of the config file
034
035                
036                if (Boolean.parseBoolean(this.scenario.getConfig().getModule("OneWayCarsharing").getValue("useOneWayCarsharing") )) {
037                        
038                        this.availableModes = new String[1];
039                        this.availableModes[0] = "oneway";
040                }
041                if (Boolean.parseBoolean(this.scenario.getConfig().getModule("FreeFloating").getValue("useFreeFloating") )) {
042                        if (this.availableModes == null) {
043                                this.availableModes = new String[1];
044                                this.availableModes[0] = "freefloating";
045                        }
046                        else {
047                                this.availableModes = new String[2];
048                                this.availableModes[0] = "oneway";
049                        this.availableModes[1] = "freefloating";
050                        }
051                }
052                
053                if (!Boolean.parseBoolean(this.scenario.getConfig().getModule("FreeFloating").getValue("useFreeFloating") ) && !Boolean.parseBoolean(this.scenario.getConfig().getModule("OneWayCarsharing").getValue("useOneWayCarsharing") )) {
054                        this.availableModes = new String[1];
055                }
056                                
057
058                
059
060        }
061        
062        
063        @Override
064        public PlanAlgorithm getPlanAlgoInstance() {
065                final TripRouter tripRouter = tripRouterProvider.get();
066                ChooseRandomTripMode algo = new ChooseRandomTripMode(this.scenario, this.availableModes,
067                                MatsimRandom.getLocalInstance(), tripRouter.getStageActivityTypes(), this.memberships);
068                return algo;
069        }
070
071}
