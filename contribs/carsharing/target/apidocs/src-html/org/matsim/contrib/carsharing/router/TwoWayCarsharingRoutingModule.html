001package org.matsim.contrib.carsharing.router;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import org.matsim.api.core.v01.population.Leg;
007import org.matsim.api.core.v01.population.Person;
008import org.matsim.api.core.v01.population.PlanElement;
009import org.matsim.core.population.PopulationUtils;
010import org.matsim.core.router.EmptyStageActivityTypes;
011import org.matsim.core.router.RoutingModule;
012import org.matsim.core.router.StageActivityTypes;
013import org.matsim.facilities.Facility;
014
015public class TwoWayCarsharingRoutingModule implements RoutingModule {
016        
017        
018        @Override
019        public List<? extends PlanElement> calcRoute(Facility fromFacility,
020                        Facility toFacility, double departureTime, Person person) {
021                final List<PlanElement> trip = new ArrayList<PlanElement>();
022                
023                final Leg leg1 = PopulationUtils.createLeg("twoway");
024                CarsharingRoute route1 = new CarsharingRoute(fromFacility.getLinkId(), toFacility.getLinkId());
025                leg1.setRoute(route1);          
026                trip.add(leg1);         
027                return trip;
028        }
029                
030        @Override
031        public StageActivityTypes getStageActivityTypes() {
032                // TODO Auto-generated method stub
033                
034                return EmptyStageActivityTypes.INSTANCE;
035        }
036}
