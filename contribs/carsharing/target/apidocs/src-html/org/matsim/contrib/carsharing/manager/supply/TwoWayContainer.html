001package org.matsim.contrib.carsharing.manager.supply;
002
003import java.util.Collection;
004import java.util.Map;
005
006import org.matsim.api.core.v01.Coord;
007import org.matsim.api.core.v01.network.Link;
008import org.matsim.contrib.carsharing.stations.CarsharingStation;
009import org.matsim.contrib.carsharing.stations.TwoWayCarsharingStation;
010import org.matsim.contrib.carsharing.vehicles.CSVehicle;
011import org.matsim.contrib.carsharing.vehicles.StationBasedVehicle;
012import org.matsim.core.utils.collections.QuadTree;
013import org.matsim.core.utils.geometry.CoordUtils;
014/** 
015 * @author balac
016 */
017public class TwoWayContainer implements VehiclesContainer{
018        
019        
020        private QuadTree<CarsharingStation> twvehicleLocationQuadTree;
021        private Map<String, CarsharingStation> twowaycarsharingstationsMap;
022        private Map<CSVehicle, Link> twvehiclesMap;
023
024        public TwoWayContainer(QuadTree<CarsharingStation> twvehicleLocationQuadTree,
025                        Map<String, CarsharingStation> twowaycarsharingstationsMap, Map<CSVehicle, Link> twvehiclesMap) {
026                
027                this.twvehicleLocationQuadTree = twvehicleLocationQuadTree;
028                this.twowaycarsharingstationsMap = twowaycarsharingstationsMap;
029                this.twvehiclesMap = twvehiclesMap;
030                
031        }
032        
033        
034        public boolean reserveVehicle(CSVehicle vehicle) {
035                String stationId = ((StationBasedVehicle)vehicle).getStationId();
036
037                if (stationId == null) {
038                        return false;
039                }
040
041                this.twvehiclesMap.remove(vehicle);
042
043                CarsharingStation station = twowaycarsharingstationsMap.get(stationId);                 
044                ((TwoWayCarsharingStation)station).removeCar(vehicle);                          
045
046                return true;
047        }
048
049        public void parkVehicle(CSVehicle vehicle, Link link) {
050                twvehiclesMap.put(vehicle, link);
051                String stationId = ((StationBasedVehicle)vehicle).getStationId();
052                CarsharingStation station = twowaycarsharingstationsMap.get(stationId);                 
053
054                ((TwoWayCarsharingStation)station).addCar(vehicle.getType(),  vehicle);         
055        }
056
057        public Map<String, CarsharingStation> getTwowaycarsharingstationsMap() {
058                return twowaycarsharingstationsMap;
059        }
060
061        @Override
062        public Link getVehicleLocation(CSVehicle vehicle) {
063                return twvehiclesMap.get(vehicle);
064        }
065
066        @Override
067        public CSVehicle findClosestAvailableVehicle(Link startLink, String typeOfVehicle, double searchDistance) {
068                        Collection<CarsharingStation> location = 
069                                this.twvehicleLocationQuadTree.getDisk(startLink.getCoord().getX(), 
070                                                startLink.getCoord().getY(), searchDistance);
071                if (location.isEmpty()) return null;
072
073                CarsharingStation closest = null;
074                double closestFound = searchDistance;
075
076                for(CarsharingStation station: location) {
077                        
078                        Coord coord = station.getLink().getCoord();
079                        
080                        if (CoordUtils.calcEuclideanDistance(startLink.getCoord(), coord) < closestFound 
081                                        && ((TwoWayCarsharingStation)station).getNumberOfVehicles(typeOfVehicle) > 0) {
082                                closest = station;
083                                closestFound = CoordUtils.calcEuclideanDistance(startLink.getCoord(), coord);
084                        }
085                }
086                
087                if (closest != null) {
088                        CSVehicle vehicleToBeUsed = ((TwoWayCarsharingStation)closest).getVehicles(typeOfVehicle).get(0);
089                        return vehicleToBeUsed;
090                }
091                
092                return null;
093        }
094
095        @Override
096        public Link findClosestAvailableParkingLocation(Link destinationLink, double searchDistance) {
097                return null;
098        }
099
100
101        @Override
102        public void reserveParking(Link destinationLink) {
103                // TODO Auto-generated method stub
104                
105        }               
106}
