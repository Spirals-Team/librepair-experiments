001package org.matsim.contrib.carsharing.control.listeners;
002
003import java.io.BufferedWriter;
004import java.io.IOException;
005import java.util.ArrayList;
006import java.util.Map;
007
008import javax.inject.Inject;
009
010import org.matsim.api.core.v01.Id;
011import org.matsim.api.core.v01.population.Person;
012import org.matsim.contrib.carsharing.manager.demand.AgentRentals;
013import org.matsim.contrib.carsharing.manager.demand.DemandHandler;
014import org.matsim.contrib.carsharing.manager.demand.RentalInfo;
015import org.matsim.contrib.carsharing.manager.supply.CarsharingSupplyInterface;
016import org.matsim.contrib.carsharing.vehicles.CSVehicle;
017import org.matsim.core.controler.MatsimServices;
018import org.matsim.core.controler.events.IterationEndsEvent;
019import org.matsim.core.controler.listener.IterationEndsListener;
020import org.matsim.core.utils.io.IOUtils;
021
022/** 
023 * 
024 * @author balac
025 */
026public class CarsharingListener implements IterationEndsListener{
027
028        @Inject private MatsimServices controler;
029        @Inject private DemandHandler demandHandler;
030        @Inject private CarsharingSupplyInterface carsharingSupply;
031
032        ArrayList<Integer> rentalsPerIteration = new ArrayList<>();
033
034        @Override
035        public void notifyIterationEnds(IterationEndsEvent event) {             
036                int numberOfRentals = 0;
037
038                Map<Id<Person>, AgentRentals> agentRentalsMap = demandHandler.getAgentRentalsMap();
039                final BufferedWriter outLink = IOUtils.getBufferedWriter(this.controler.getControlerIO().getIterationFilename(event.getIteration(), "CS.txt"));
040                try {
041                        outLink.write("personID,carsharingType,startTime,endTIme,startLink,pickupLink,dropoffLink,endLink,distance,inVehicleTime,accessTime,egressTime,vehicleID,"
042                                        + "companyID,vehicleType");
043                        outLink.newLine();              
044                
045                for (Id<Person> personId: agentRentalsMap.keySet()) {
046                        
047                        for (RentalInfo i : agentRentalsMap.get(personId).getArr()) {
048                                CSVehicle vehicle = this.carsharingSupply.getAllVehicles().get(i.getVehId().toString());                
049                                numberOfRentals++;
050                                outLink.write(personId + "," + i.toString() + "," + vehicle.getCompanyId() + "," + vehicle.getType());
051                                outLink.newLine();
052                        }
053                        
054                }
055                rentalsPerIteration.add(numberOfRentals);
056
057                outLink.flush();
058                outLink.close();
059                
060                } catch (IOException e) {
061                        // TODO Auto-generated catch block
062                        e.printStackTrace();
063                }       
064                
065                if (event.getIteration() == controler.getConfig().controler().getLastIteration()) {
066                        final BufferedWriter outLinkStats = IOUtils.getBufferedWriter(this.controler.getControlerIO().getOutputFilename("CS.txt"));
067                        try {
068                                outLinkStats.write("iteration,numberOfRentals");
069                        
070                                outLinkStats.newLine();
071                                int k = 0;
072                                for (Integer i : rentalsPerIteration) {
073                                        outLinkStats.write(k + "," + i);
074                                        k++;
075                                        outLinkStats.newLine();
076                                }
077                                
078                                outLinkStats.flush();
079                                outLinkStats.close();
080                        } catch (IOException e) {
081                                // TODO Auto-generated catch block
082                                e.printStackTrace();
083                        }
084                        
085                }
086                
087                
088        }
089}
