001package org.matsim.contrib.carsharing.manager.routers;
002
003import java.util.ArrayList;
004import java.util.Arrays;
005import java.util.List;
006import java.util.Map;
007
008import org.matsim.api.core.v01.Scenario;
009import org.matsim.api.core.v01.TransportMode;
010import org.matsim.api.core.v01.network.Link;
011import org.matsim.api.core.v01.population.Activity;
012import org.matsim.api.core.v01.population.Leg;
013import org.matsim.api.core.v01.population.Person;
014import org.matsim.api.core.v01.population.Plan;
015import org.matsim.api.core.v01.population.PlanElement;
016import org.matsim.api.core.v01.population.PopulationFactory;
017import org.matsim.contrib.carsharing.router.CarsharingRoute;
018import org.matsim.contrib.carsharing.vehicles.CSVehicle;
019import org.matsim.core.population.routes.NetworkRoute;
020import org.matsim.core.router.costcalculators.TravelDisutilityFactory;
021import org.matsim.core.router.util.LeastCostPathCalculator;
022import org.matsim.core.router.util.LeastCostPathCalculatorFactory;
023import org.matsim.core.router.util.TravelDisutility;
024import org.matsim.core.router.util.TravelTime;
025
026import com.google.inject.Inject;
027/** 
028 * @author balac
029 */
030public class RouteCarsharingTripImpl implements RouteCarsharingTrip {
031
032        @Inject private Scenario scenario;
033        @Inject private LeastCostPathCalculatorFactory pathCalculatorFactory ;
034        
035        @Inject private Map<String, TravelTime> travelTimes ;
036        @Inject private Map<String, TravelDisutilityFactory> travelDisutilityFactories ;
037        
038        private final ArrayList<String> carsharingLegs = new ArrayList<>(Arrays.asList("oneway", "twoway",
039                        "freefloating"));
040        
041        private final String[] carsharingVehicleLegs = {"oneway_vehicle", "twoway_vehicle",
042                        "freefloating_vehicle"};
043        private final String[] accessCSLegs = {"access_walk_ow", "access_walk_tw",
044        "access_walk_ff"};
045        
046        private final String[] egressCSLegs = {"egress_walk_ow", "egress_walk_tw",
047        "egress_walk_ff"};
048        
049        private final String[] csInteraction = {"ow_interaction", "tw_interaction",
050        "ff_interaction"};
051        
052        @Override
053        public List<PlanElement> routeCarsharingTrip(Plan plan, Leg legToBeRouted, double time, 
054                        CSVehicle vehicle, Link vehicleLinkLocation, Link parkingLocation, boolean keepTheCarForLaterUse, boolean hasVehicle) {
055                PopulationFactory pf = scenario.getPopulation().getFactory();
056                TravelTime travelTime = travelTimes.get( TransportMode.car ) ;
057                
058                TravelDisutility travelDisutility = travelDisutilityFactories.get( TransportMode.car ).createTravelDisutility(travelTime) ;
059                LeastCostPathCalculator pathCalculator = pathCalculatorFactory.createPathCalculator(scenario.getNetwork(),
060                                travelDisutility, travelTime ) ;
061                
062                String mainMode = legToBeRouted.getMode();
063                int index = carsharingLegs.indexOf(mainMode);
064                final List<PlanElement> trip = new ArrayList<PlanElement>();            
065
066                Person person = plan.getPerson();
067                CarsharingRoute route = (CarsharingRoute) legToBeRouted.getRoute();
068                final Link currentLink = scenario.getNetwork().getLinks().get(route.getStartLinkId());
069                final Link destinationLink = scenario.getNetwork().getLinks().get(route.getEndLinkId());
070                
071                if (hasVehicle) {
072                        //=== car leg                   
073
074                        trip.add(RouterUtils.createCarLeg(pf, pathCalculator,
075                                        person, currentLink, parkingLocation, carsharingVehicleLegs[index], 
076                                        vehicle.getVehicleId(), time));         
077                        
078                        if (!keepTheCarForLaterUse) {                   
079
080                                Activity activityE = scenario.getPopulation().getFactory().createActivityFromLinkId(csInteraction[index],
081                                                parkingLocation.getId());
082                                activityE.setMaximumDuration(0);
083                                
084                                trip.add(activityE);
085                                
086                                trip.add( RouterUtils.createWalkLeg(pf, 
087                                                parkingLocation, destinationLink, egressCSLegs[index], time) );         
088                        }
089                
090                }
091                else {          
092                        
093                        String ffVehId = vehicle.getVehicleId();                        
094                        trip.add( RouterUtils.createWalkLeg(scenario.getPopulation().getFactory(),
095                                        currentLink, vehicleLinkLocation, accessCSLegs[index], time) );
096                        
097                        Activity activityS = scenario.getPopulation().getFactory().createActivityFromLinkId(csInteraction[index],
098                                        vehicleLinkLocation.getId());
099                        activityS.setMaximumDuration(0);
100                        
101                        trip.add(activityS);
102                        // === car leg: ===                                                     
103                        
104                                                
105                        
106                        if (!keepTheCarForLaterUse)  {  
107                                trip.add(RouterUtils.createCarLeg(pf, pathCalculator,
108                                                person, vehicleLinkLocation, parkingLocation, carsharingVehicleLegs[index],
109                                                ffVehId, time));
110                                Activity activityE = scenario.getPopulation().getFactory().createActivityFromLinkId(csInteraction[index],
111                                                parkingLocation.getId());
112                                activityE.setMaximumDuration(0);
113                                
114                                trip.add(activityE);
115                                
116                                trip.add( RouterUtils.createWalkLeg(pf, 
117                                                parkingLocation, destinationLink, egressCSLegs[index], time) );
118                        }
119                        else {
120                                trip.add(RouterUtils.createCarLeg(pf, pathCalculator,
121                                                person, vehicleLinkLocation, destinationLink, carsharingVehicleLegs[index],
122                                                ffVehId, time));
123                                
124                        }
125                        
126                }                       
127                return trip;
128        }
129
130}
