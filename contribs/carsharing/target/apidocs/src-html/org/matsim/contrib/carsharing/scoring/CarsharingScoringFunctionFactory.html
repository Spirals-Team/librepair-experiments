001package org.matsim.contrib.carsharing.scoring;
002
003import org.matsim.api.core.v01.Scenario;
004import org.matsim.api.core.v01.population.Person;
005import org.matsim.contrib.carsharing.manager.demand.DemandHandler;
006import org.matsim.contrib.carsharing.manager.supply.CarsharingSupplyInterface;
007import org.matsim.contrib.carsharing.manager.supply.costs.CostsCalculatorContainer;
008import org.matsim.core.scoring.ScoringFunction;
009import org.matsim.core.scoring.ScoringFunctionFactory;
010import org.matsim.core.scoring.SumScoringFunction;
011import org.matsim.core.scoring.functions.CharyparNagelActivityScoring;
012import org.matsim.core.scoring.functions.CharyparNagelAgentStuckScoring;
013import org.matsim.core.scoring.functions.CharyparNagelLegScoring;
014import org.matsim.core.scoring.functions.ScoringParametersForPerson;
015import org.matsim.core.scoring.functions.SubpopulationScoringParameters;
016
017import javax.inject.Inject;
018
019public class CarsharingScoringFunctionFactory implements ScoringFunctionFactory {
020        
021        private final Scenario scenario;
022        private final ScoringParametersForPerson params;
023        private final DemandHandler demandHandler;
024        private final CostsCalculatorContainer costsCalculatorContainer;
025        private final CarsharingSupplyInterface carsharingSupplyContainer;
026        @Inject
027        CarsharingScoringFunctionFactory( final Scenario sc, final DemandHandler demandHandler,
028                        final CostsCalculatorContainer costsCalculatorContainer, final CarsharingSupplyInterface carsharingSupplyContainer) {
029                this.scenario = sc;
030                this.params = new SubpopulationScoringParameters( sc );
031                this.demandHandler = demandHandler;
032                this.costsCalculatorContainer = costsCalculatorContainer;
033                this.carsharingSupplyContainer = carsharingSupplyContainer;
034        }
035
036
037        @Override
038        public ScoringFunction createNewScoringFunction(Person person) {
039                SumScoringFunction scoringFunctionSum = new SumScoringFunction();
040            //this is the main difference, since we need a special scoring for carsharing legs
041
042                scoringFunctionSum.addScoringFunction(
043            new CarsharingLegScoringFunction( params.getScoringParameters( person ),
044                                                                         this.scenario.getConfig(),
045                                                                         this.scenario.getNetwork(), this.demandHandler, this.costsCalculatorContainer, 
046                                                                         this.carsharingSupplyContainer, person));
047                scoringFunctionSum.addScoringFunction(
048                                new CharyparNagelLegScoring(
049                                                params.getScoringParameters( person ),
050                                                this.scenario.getNetwork())
051                            );
052                //the remaining scoring functions can be changed and adapted to the needs of the user
053                scoringFunctionSum.addScoringFunction(
054                                new CharyparNagelActivityScoring(
055                                                params.getScoringParameters(
056                                                                person ) ) );
057                scoringFunctionSum.addScoringFunction(
058                                new CharyparNagelAgentStuckScoring(
059                                                params.getScoringParameters(
060                                                                person ) ) );
061            return scoringFunctionSum;
062          }
063}
