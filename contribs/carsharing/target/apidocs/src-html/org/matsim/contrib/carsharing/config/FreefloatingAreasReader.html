001package org.matsim.contrib.carsharing.config;
002
003import java.util.ArrayList;
004import java.util.HashMap;
005import java.util.Map;
006import java.util.Stack;
007
008import org.geotools.referencing.crs.DefaultGeographicCRS;
009import org.matsim.api.core.v01.Coord;
010import org.matsim.contrib.carsharing.qsim.FreefloatingAreas;
011import org.matsim.core.utils.gis.PolygonFeatureFactory;
012import org.matsim.core.utils.io.MatsimXmlParser;
013import org.matsim.core.utils.misc.Counter;
014import org.opengis.feature.simple.SimpleFeature;
015import org.xml.sax.Attributes;
016
017public class FreefloatingAreasReader extends MatsimXmlParser {
018        private PolygonFeatureFactory polygonFeatureFactory;
019
020        private Map<String, FreefloatingAreas> freefloatingAreas = new HashMap<String, FreefloatingAreas>();
021
022        private Counter counter;
023
024        private String companyName;
025
026        private String idString = null;
027
028        private ArrayList<Coord> coords;
029
030        public FreefloatingAreasReader() {
031                this.polygonFeatureFactory = new PolygonFeatureFactory.Builder()
032                                .setName("freefloating_area")
033                                .setCrs(DefaultGeographicCRS.WGS84)
034                                .addAttribute("id", String.class)
035                                .create();
036        }
037
038        @Override
039        public void startTag(String name, Attributes atts, Stack<String> context) {
040                if (name.equals("areas")) {
041                        counter = new Counter("reading freefloating area # ");
042                }
043
044                if (name.equals("company")) {
045                        this.companyName = atts.getValue("name");
046                }
047
048                if (name.equals("area")) {
049                        counter.incCounter();
050                        this.idString = atts.getValue("id");
051                        this.coords = new ArrayList<Coord>();
052                }
053
054                if (name.equals("node")) {
055                        final String x = atts.getValue("x");
056                        final String y = atts.getValue("y");
057
058                        Coord coord = new Coord(
059                                        Double.parseDouble(x),
060                                        Double.parseDouble(y)
061                        );
062
063                        this.coords.add(coord);
064                }
065        }
066
067        @Override
068        public void endTag(final String name, final String content, final Stack<String> context) {
069                if (name.equals("area")) {
070                        Coord[] coordsArray = this.coords.toArray(new Coord[this.coords.size()]);
071                        SimpleFeature area = this.polygonFeatureFactory.createPolygon(coordsArray);
072                        area.setAttribute("id", this.idString);
073
074                        this.addArea(area);
075                }
076
077                if (name.equals("areas")) {
078                        counter.printCounter();
079                }
080        }
081
082        public Map<String, FreefloatingAreas> getFreefloatingAreas() {
083                return this.freefloatingAreas;
084        }
085
086        public void addArea(SimpleFeature area) {
087                // TODO: what was the purpose of this check?
088                if (area.getName().getLocalPart() == "freefloating_area") {
089                        if (this.getFreefloatingAreas().containsKey(this.companyName) == false) {
090                                this.getFreefloatingAreas().put(this.companyName, new FreefloatingAreas());
091                        }
092
093                        this.getFreefloatingAreas().get(this.companyName).add(area);
094                }
095        }
096}
