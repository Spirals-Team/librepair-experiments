001package org.matsim.contrib.carsharing.qsim;
002
003import java.util.ArrayList;
004
005import org.geotools.referencing.crs.DefaultGeographicCRS;
006import org.matsim.api.core.v01.Coord;
007import org.matsim.core.utils.gis.PointFeatureFactory;
008import org.opengis.feature.simple.SimpleFeature;
009
010import com.vividsolutions.jts.geom.Coordinate;
011import com.vividsolutions.jts.geom.MultiPolygon;
012import com.vividsolutions.jts.geom.Point;
013import com.vividsolutions.jts.operation.distance.DistanceOp;
014
015public class FreefloatingAreas {
016        private PointFeatureFactory pointFeatureFactory;
017
018        private SimpleFeature areas = null;
019
020        private ArrayList<String> ids = new ArrayList<String>();
021
022        public FreefloatingAreas() {
023                this.pointFeatureFactory = new PointFeatureFactory.Builder()
024                                .setName("point")
025                                .setCrs(DefaultGeographicCRS.WGS84)
026                                .create();
027        }
028
029        public void add(SimpleFeature area) {
030                String id = (String) area.getAttribute("id");
031
032                if (this.ids.contains(id) == false) {
033                        this.ids.add(id);
034
035                        if (this.areas == null) {
036                                this.areas = area;
037                        } else {
038                                // merge geometries!
039                                MultiPolygon oldPolygons = (MultiPolygon) this.areas.getAttribute("the_geom");
040                                MultiPolygon newPolygon = (MultiPolygon) area.getAttribute("the_geom");
041                                MultiPolygon unitedPolygons = (MultiPolygon) oldPolygons.union(newPolygon);
042                                this.areas.setAttribute("the_geom", unitedPolygons);
043                        }
044                }
045        }
046
047        public boolean contains(Coord coord) {
048                if (this.areas == null) {
049                        return true;
050                }
051
052                MultiPolygon polygons = (MultiPolygon) this.areas.getAttribute("the_geom");
053                SimpleFeature pointFeature = this.pointFeatureFactory.createPoint(coord, new Object[0], null);
054                Point point = (Point) pointFeature.getAttribute("the_geom");
055
056                return polygons.contains(point);
057        }
058
059        public Coord[] nearestPoints(Coord coord) {
060                if (this.areas == null) {
061                        return new Coord[]{coord};
062                }
063
064                MultiPolygon polygons = (MultiPolygon) this.areas.getAttribute("the_geom");
065                SimpleFeature pointFeature = this.pointFeatureFactory.createPoint(coord, new Object[0], null);
066                Point point = (Point) pointFeature.getAttribute("the_geom");
067
068                Coordinate[] coordinates = DistanceOp.nearestPoints(polygons, point);
069                Coord[] coords = new Coord[coordinates.length];
070
071                int index = 0;
072                for (Coordinate coordinate : coordinates) {
073                        Coord nearestCoord = new Coord(coordinate.x, coordinate.y);
074                        coords[index] = nearestCoord;
075                        index++;
076                }
077
078                return coords;
079        }
080}
