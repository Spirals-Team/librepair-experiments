001package org.matsim.contrib.carsharing.qsim;
002
003import java.util.Collection;
004import java.util.HashMap;
005import java.util.Map;
006
007import org.matsim.api.core.v01.Id;
008import org.matsim.api.core.v01.network.Link;
009import org.matsim.contrib.carsharing.manager.supply.CarsharingSupplyInterface;
010import org.matsim.contrib.carsharing.vehicles.CSVehicle;
011import org.matsim.core.mobsim.framework.AgentSource;
012import org.matsim.core.mobsim.qsim.QSim;
013import org.matsim.vehicles.Vehicle;
014import org.matsim.vehicles.VehicleType;
015import org.matsim.vehicles.VehicleUtils;
016
017/** 
018 * @author balac
019 */
020public class ParkCSVehicles implements AgentSource {
021        private QSim qsim;
022        private Map<String, VehicleType> modeVehicleTypes;
023        private Collection<String> mainModes;
024        private CarsharingSupplyInterface carsharingSupply;
025        //private final static Logger log = Logger.getLogger(ParkCSVehicles.class);
026        
027        public ParkCSVehicles(QSim qSim,
028                        CarsharingSupplyInterface carsharingSupply) {
029                
030                this.qsim = qSim;  
031                this.modeVehicleTypes = new HashMap<String, VehicleType>();
032                this.mainModes = qsim.getScenario().getConfig().qsim().getMainModes();
033        
034                for (String mode : mainModes) {
035                        modeVehicleTypes.put(mode, VehicleUtils.getDefaultVehicleType());
036                }
037                this.carsharingSupply =  carsharingSupply;
038                modeVehicleTypes.put("twoway", VehicleUtils.getDefaultVehicleType());
039                modeVehicleTypes.put("freefloating", VehicleUtils.getDefaultVehicleType());
040
041                modeVehicleTypes.put("oneway", VehicleUtils.getDefaultVehicleType());
042                
043                
044        }
045
046        @Override
047        public void insertAgentsIntoMobsim() {
048                Map<CSVehicle, Link> allVehicleLocations = this.carsharingSupply.getAllVehicleLocations();
049
050                for (CSVehicle vehicle : allVehicleLocations.keySet()) {
051                        
052                        qsim.createAndParkVehicleOnLink(VehicleUtils.getFactory().createVehicle(Id.create(vehicle.getVehicleId(), Vehicle.class),
053                                        modeVehicleTypes.get(vehicle.getCsType())), allVehicleLocations.get(vehicle).getId());
054                }
055                
056                
057        }
058
059}
