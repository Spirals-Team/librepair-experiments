018package org.matsim.contrib.networkEditor.utils;
019
020import org.matsim.api.core.v01.Coord;

022import org.opengis.feature.simple.SimpleFeature;
023
024import com.vividsolutions.jts.algorithm.CGAlgorithms;
025import com.vividsolutions.jts.algorithm.RobustLineIntersector;
026import com.vividsolutions.jts.geom.Coordinate;
027import com.vividsolutions.jts.geom.GeometryFactory;
028import com.vividsolutions.jts.geom.LineSegment;
029import com.vividsolutions.jts.geom.LinearRing;
030import com.vividsolutions.jts.geom.Point;
031import com.vividsolutions.jts.geom.Polygon;
032
033/**
034 * @author danielmaxx
035 */
036public class GeometryTools {
037
038        static public Point getCentroid(Polygon poly){
039                return poly.getCentroid();
040        }
041
042        static public double getArea(Polygon poly) {
043                return poly.getArea();
044        }
045
046        static public boolean intersects(LinearRing ring, LineSegment segment) {
047                Coordinate coordinates[] = ring.getCoordinates();
048                final int points = coordinates.length;
049                RobustLineIntersector intersector = new RobustLineIntersector();
050                for(int i=0; i<coordinates.length; ++i) {
051                        LineSegment aux = new LineSegment(coordinates[i], coordinates[(i+1)%points]);
052                        intersector.computeIntersection(segment.p0, segment.p1, aux.p0, aux.p1);
053                        if(intersector.hasIntersection()) {
054                                return true;
055                        }
056                }
057                return false;
058        }
059
060        static public boolean intersectRectangle(Coordinate p1, Coordinate p2, LineSegment segment) {
061                return intersects(getRectangle(p1, p2), segment);
062        }
063
064        static public boolean isInside(LinearRing ring, LineSegment segment) {
065                final boolean p1 = CGAlgorithms.isPointInRing(segment.p0, ring.getCoordinates());
066                final boolean p2 = CGAlgorithms.isPointInRing(segment.p1, ring.getCoordinates());
067                return p1&&p2;
068        }
069
070        static public LinearRing getRectangle(Coordinate p1, Coordinate p2) {
071                if(p1.x > p2.x) {
072                        Coordinate aux = p1;
073                        p1 = p2;
074                        p2 = aux;
075                }
076                if(p1.y < p2.y) {
077                        double aux = p2.y;
078                        p2.y = p1.y;
079                        p1.y = aux;
080                }
081                Coordinate rectangle[] = new Coordinate[5];
082                rectangle[0] = p1;
083                rectangle[1] = new Coordinate(p2.x, p1.y);
084                rectangle[2] = p2;
085                rectangle[3] = new Coordinate(p1.x, p2.y);
086                rectangle[4] = p1;
087                return new GeometryFactory().createLinearRing(rectangle);
088        }
089
090        static public LinearRing featureToLinearRing(SimpleFeature ft) {
091                return (LinearRing)ft.getDefaultGeometry();
092        }
093
094        static public java.awt.Polygon toJavaPolygon(LinearRing ring){
095                java.awt.Polygon poly = new java.awt.Polygon();
096                Coordinate coordinates[] = ring.getCoordinates();
097                for(int i=0; i<coordinates.length; ++i)
098                        poly.addPoint((int)coordinates[i].x, (int)coordinates[i].y);
099                return poly;
100        }
101
102        static public Coordinate MATSimCoordToCoordinate(Coord c) {
103                return new Coordinate(c.getX(), c.getY());
104        }
105        static public Coordinate MATSimCoordToCoordinate(MutableCoord c) {
106                return new Coordinate(c.getX(), c.getY());
107        }
108
109}
