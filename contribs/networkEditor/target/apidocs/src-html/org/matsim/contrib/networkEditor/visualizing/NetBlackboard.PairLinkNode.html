474        private void clickEvent(java.awt.event.MouseEvent evt) {//GEN-FIRST:event_clickEvent
475                if (SwingUtilities.isRightMouseButton(evt)) {
476                        if(this.line.active) {
477                                line.active = false;
478                        } else {
479                                centerAt(inverseTransform(new MutableCoord((double) evt.getPoint().x, (double) evt.getPoint().y)));
480                                this.zoom(0.05);
481                        }
482                        this.repaint();
483                } else if (SwingUtilities.isLeftMouseButton(evt)) {
484                        if (this.net == null) {
485                                return;
486                        }
487                        if(actualMode == Mode.NONE)
488                                return;
489                        else if(actualMode == Mode.SELECTION) {
490                                MutableCoord mousePos = new MutableCoord((double) this.getMousePosition().x, (double) this.getMousePosition().y);
491                                this.setActiveSomething(mousePos);
492                                this.controls.updateTable();
493                                /*if(activeLink != null)
494                    addLinkInSelectedLinkList(activeLink, true);*/
495                        } else if(actualMode == Mode.PAINTING) {
496                                if(line.active == false) {
497                                        MutableCoord c = new MutableCoord((double) this.getMousePosition().x, (double) this.getMousePosition().y);
498                                        LinkNodeCoord lnc = setLinePoint(c);
499                                        line.start = inverseTransform(lnc.coord);
500                                        line.linkClosestToStart = null;//lnc.link;
501                                        line.nodeClosestToStart = lnc.node;
502                                        //System.out.println(line.start);
503                                        line.active = true;
504                                } else {
505                                        MutableCoord c = new MutableCoord((double) this.getMousePosition().x, (double) this.getMousePosition().y);
506                                        LinkNodeCoord lnc = setLinePoint(c);
507                                        line.end = lnc.coord;
508                                        line.start = transform(line.start);
509                                        line.linkClosestToEnd = null;//lnc.link;
510                                        line.nodeClosestToEnd = lnc.node;
511                                        createLink(line);
512                                        line = new LineOnBoard();
513                                }
514                        } else if(actualMode == Mode.CUTTING) {
515                                if(activeLink == null) {
516                                        javax.swing.JOptionPane.showMessageDialog(this, "You must select a link.", null, javax.swing.JOptionPane.INFORMATION_MESSAGE);
517                                        return;
518                                }
519                                MutableCoord c = new MutableCoord((double) this.getMousePosition().x, (double) this.getMousePosition().y);
520                                if(this.splitActiveLink(inverseTransform(c))==false)
521                                        javax.swing.JOptionPane.showMessageDialog(this, "Invalid selection, link not cut.", null, javax.swing.JOptionPane.WARNING_MESSAGE);
522                        }
523                        this.repaint();
524                }
525        }//GEN-LAST:event_clickEvent

570        private void formKeyPressed(java.awt.event.KeyEvent evt) {//GEN-FIRST:event_formKeyPressed
571                int keyCode = evt.getKeyCode();
572                if(keyCode == java.awt.event.KeyEvent.VK_DOWN) {
573                        moveViewBox(0, getHeight()*0.02);
574                        repaint();
575                } else if(keyCode == java.awt.event.KeyEvent.VK_UP) {
576                        moveViewBox(0, -getHeight()*0.02);
577                        repaint();
578                } else if(keyCode == java.awt.event.KeyEvent.VK_LEFT) {
579                        moveViewBox(-getWidth()*0.02, 0);
580                        repaint();
581                } else if(keyCode == java.awt.event.KeyEvent.VK_RIGHT) {
582                        moveViewBox(getWidth()*0.02, 0);
583                        repaint();
584                } else if(keyCode == java.awt.event.KeyEvent.VK_CONTROL) {
585                        isControlPressed = true;
586                }
587        }//GEN-LAST:event_formKeyPressed

919                final double midX = (selectionSquare.start.getX()+selectionSquare.end.getX())/2.0;
920                final double midY = (selectionSquare.start.getY()+selectionSquare.end.getY())/2.0;
921                MutableCoord mid = inverseTransform(new MutableCoord(midX, midY));
922                final double dist = calcEuclideanDistance(mid, inverseTransform(selectionSquare.end));
923                final MutableCoord coord = mid;
924                //System.out.println("mid = " + mid + ", dist = " + dist);
925                Collection<Node> nodes = getNearestNodes2(net,coord, dist*8);
926                MutableCoord start = new MutableCoord(selectionSquare.start.getX(), selectionSquare.start.getY()), end = new MutableCoord(selectionSquare.end.getX(), selectionSquare.end.getY());
927                for(Node node : nodes) {
928                        for(Link l : node.getInLinks().values()) {
929                                if(insideSquare(l, start, end))
930                                        this.addLinkInSelectedLinkList(l, false);
931                        }
932                        for(Link l : node.getOutLinks().values()) {
933                                if(insideSquare(l, start, end))
934                                        this.addLinkInSelectedLinkList(l, false);
935                        }
936                }
937        }

1104                                lnc.link = p.link;
1105                        }
1106                } else {
1107                        lnc.coord = mouse;
1108                }
1109                return lnc;

1151         * Transform a coordinate from the screen coordinate system to the network
1152         * coordinate system
1153         * @param coord
1154         * @return The transformed coordinate
1155         */
1156        protected MutableCoord transform(MutableCoord coord) {
1157                double x = coord.getX() - curX, y = coord.getY() - curY;
1158                final double boxSizeX = offX + offX*0.05;
1159                final double boxSizeY = offY + offY*0.05;
1160                x = (x*this.getWidth())/boxSizeX + 10;
1161                y = (y*this.getHeight())/boxSizeY + 10;
1162                y = getHeight() - y;
1163                MutableCoord result = new MutableCoord(x, y);
1164                return result;
1165        }
