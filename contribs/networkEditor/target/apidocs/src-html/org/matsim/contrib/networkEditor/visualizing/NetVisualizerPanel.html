148        private void initBoard() {
149                board = new NetBlackboard();
150                board.setSize(this.editorPanel.getWidth()-50, this.editorPanel.getHeight()-50);
151                this.editorPanel.add(board);
152
153                controls = new NetControls(board);
154                controls.setSize(this.controlPanel.getWidth(), this.controlPanel.getHeight()-30);
155                controlPanel.add(controls);
156
157                board.setNetControls(controls);
158                controls.setNetBlackboard(board);
159
160                this.revalidate();
161        }

170        public boolean loadNetFromFile(String path) {
171                Config config = ConfigUtils.createConfig();
172                Scenario sc = ScenarioUtils.createScenario(config);
173                try {
174                        new MatsimNetworkReader(sc.getNetwork()).readFile(path);
175                        board.setNetwork((Network)sc.getNetwork());
176                        board.repaint();
177                } catch (Exception e) {
178                        e.printStackTrace();
179                        return false;
180                }        
181                return true;
182        }

184        public boolean loadCountsFromFile(String path) {
185                if(board.net == null)

189                try {
190                        countsReader.readFile(path);            
191                } catch (Exception e) {
192                        e.printStackTrace();
193                        return false;
194                }
195                return true;
196        }

211        public boolean saveNetworkAsESRI(String path) {
212                if(board.net == null)
213                        return false;
214                String crs = this.crs;
215                if (crs == null) {
216                        crs = "WGS84"; // better than nothing...
217                }
218//              String UTM19N_PROJCS = "PROJCS[\"WGS_1984_UTM_Zone_19N\",GEOGCS[\"GCS_WGS_1984\",DATUM[\"D_WGS_1984\",SPHEROID[\"WGS_1984\",6378137,298.257223563]],PRIMEM[\"Greenwich\",0],UNIT[\"Degree\",0.017453292519943295]],PROJECTION[\"Transverse_Mercator\"],PARAMETER[\"latitude_of_origin\",0],PARAMETER[\"central_meridian\",-69],PARAMETER[\"scale_factor\",0.9996],PARAMETER[\"false_easting\",500000],PARAMETER[\"false_northing\",0],UNIT[\"Meter\",1]]";
219//              String crs = UTM19N_PROJCS; // the coordinate reference system to be used.
220                FeatureGeneratorBuilderImpl builder = new FeatureGeneratorBuilderImpl(board.net, crs);
221                builder.setWidthCoefficient(0.01);
222                builder.setFeatureGeneratorPrototype(PolygonFeatureGenerator.class);
223                builder.setWidthCalculatorPrototype(CapacityBasedWidthCalculator.class);
224                try {
225                        new Links2ESRIShape(board.net, path, builder).write();
226                }catch(Exception e) {
227                        e.printStackTrace();
228                        return false;
229                }
230                return true;
231        }

233        public boolean saveCounts(String path) {
234                if(board.net == null)
235                        return false;
236                if(board.counts == null)
237                        return false;
238                if(board.counts.getName() == null)
239                        board.counts.setName(board.net.getName()+" Counts\n");
240                CountsWriter countWriter = new CountsWriter(board.counts);
241                try {
242                        countWriter.write(path);
243                } catch(Exception e) {
244                        e.printStackTrace();
245                        return false;
246                }
247                return true;
248        }

250        public boolean loadNetFromOSM(final String path, final String crs) {
251                //REVISAR UTM33N a UTM19N
252                //        String UTM19N_PROJCS = "PROJCS[\"WGS_1984_UTM_Zone_19N\",GEOGCS[\"GCS_WGS_1984\",DATUM[\"D_WGS_1984\",SPHEROID[\"WGS_1984\",6378137,298.257223563]],PRIMEM[\"Greenwich\",0],UNIT[\"Degree\",0.017453292519943295]],PROJECTION[\"Transverse_Mercator\"],PARAMETER[\"latitude_of_origin\",0],PARAMETER[\"central_meridian\",-69],PARAMETER[\"scale_factor\",0.9996],PARAMETER[\"false_easting\",500000],PARAMETER[\"false_northing\",0],UNIT[\"Meter\",1]]";
253                //      String crs = UTM19N_PROJCS; // the coordinate reference system to be used.
254
255                this.crs = crs;
256                String osm = path;
257
258                Scenario sc = ScenarioUtils.createScenario(ConfigUtils.createConfig()) ;
259                Network net = sc.getNetwork();
260
261                CoordinateTransformation ct = TransformationFactory.getCoordinateTransformation(TransformationFactory.WGS84, crs);
262
263                OsmNetworkReader onr = new OsmNetworkReader(net,ct); //constructs a new openstreetmap reader
264                try {
265                        onr.parse(osm); //starts the conversion from osm to matsim
266                } catch (UncheckedIOException e) {
267                        e.printStackTrace();
268                        return false;
269                }
270                //at this point we already have a matsim network...
271                new NetworkCleaner().run(net); //but may be there are isolated (not connected) links. The network cleaner removes those links
272                board.setNetwork((Network)net);
273                board.repaint();
274                return true;
275        }
