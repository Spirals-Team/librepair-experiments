232        
233        public String getIdentifiedCrsString() {
234                return this.identifiedCrsString;
235        }
236
237        private void lookupCRS(final String desc) {
238                try {
239                        CoordinateReferenceSystem crs = MGC.getCRS(desc);
240                        lblInfo.setText(crs.getName().toString());
241                        identifiedCrsString = desc;
242                        btnOk.setEnabled(true);
243                } catch (IllegalArgumentException e) {
244                        lblInfo.setText("not recognized");
245                        btnOk.setEnabled(false);
246                }
247        }
