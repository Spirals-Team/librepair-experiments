018package org.matsim.contrib.networkEditor.visualizing;
019
020import java.util.Stack;
021
022import org.matsim.api.core.v01.network.Link;
023import org.matsim.api.core.v01.network.Network;
024import org.matsim.api.core.v01.network.NetworkFactory;
025import org.matsim.api.core.v01.network.Node;
026import org.matsim.core.network.NetworkUtils;
027
028/**
029 *
030 * @author danielmaxx
031 */
032public class DifferenceManager {
033    
034    private Stack<Difference> back, front;
035    Network net;
036    public enum type{CREATE, EDIT, DELETE, SPLIT};
037    
038    DifferenceManager() {
039        back = new Stack<Difference>();
040        front = new Stack<Difference>();
041    }
042    DifferenceManager(Network net) {
043        back = new Stack<Difference>();
044        front = new Stack<Difference>();
045        this.net = net;
046    }
047
048
049    public void saveState(Link link, type currentType, Node nodeAtStart, Node nodeAtEnd, Link linkAtStartAdded1, Link linkAtStartAdded2, Link linkAtEndAdded1, Link linkAtEndAdded2, Link linkAtStartDeleted, Link linkAtEndDeleted) {
050        Difference diff = new Difference(link, currentType, net, nodeAtStart,  nodeAtEnd, linkAtStartAdded1, linkAtStartAdded2, linkAtEndAdded1, linkAtEndAdded2, linkAtStartDeleted, linkAtEndDeleted);
051        back.push(diff);
052        front.clear();
053    }
054
055    public void undo() {
056        if(back.empty())
057            return;
058        back.peek().revert();
059        front.push(back.pop());
060    }
061
062    public void redo() {
063        if(front.empty())
064            return;
065        front.peek().apply();
066        back.push(front.pop());
067    }
068
069    public int getBackSize() {
070        return back.size();
071    }
072
073    public int getFrontSize() {
074        return front.size();
075    }
076
077    public Link cloneLink(Link link) {
078        final Network network = net;
079        NetworkFactory r = net.getFactory();
080        return NetworkUtils.createLink(link.getId(), link.getFromNode(), link.getToNode(), network, link.getLength(), link.getFreespeed(), link.getCapacity(), link.getNumberOfLanes());
081    }
082
083    class Difference {        
084        public Link link;
085        public Network net;
086        public Node nodeAtStart, nodeAtEnd;
087        public Link linkAtStartAdded1, linkAtStartAdded2;
088        public Link linkAtEndAdded1, linkAtEndAdded2;
089        public Link linkAtStartDeleted, linkAtEndDeleted;
090        public type currentType;
091
092        Difference() {}
093        Difference(Link link, type currentType, Network net, Node nAS, Node nAE, Link lASA1, Link lASA2, Link lAEA1, Link lAEA2, Link lASD, Link lAED) {
094            this.link = link;
095            this.net = net;
096            this.currentType = currentType;
097            nodeAtStart = nAS;
098            nodeAtEnd = nAE;
099            linkAtStartAdded1 = lASA1;
100            linkAtStartAdded2 = lASA2;
101            linkAtEndAdded1 = lAEA1;
102            linkAtEndAdded2 = lAEA2;
103            linkAtStartDeleted = lASD;
104            linkAtEndDeleted = lAED;
105        }
106        
107        Difference(Network net) {
108            this.net = net;
109        }
110        public void revert() {
111            //System.out.println("revert: " + currentType);
112            if(currentType == type.CREATE)
113                deleteLink();
114            else if(currentType == type.EDIT)
115                editLink();
116            else if(currentType == type.DELETE)
117                createLink();
118        }
119        public void apply() {
120            //System.out.println("apply: " + currentType);
121            if(currentType == type.CREATE)
122                createLink();
123            else if(currentType == type.EDIT)
124                editLink();
125            else if(currentType == type.DELETE)
126                deleteLink();
127        }
128
129        private void deleteLink() {
130            if(link != null)
131                net.removeLink(link.getId());
132            if(this.linkAtStartDeleted!=null)
133                net.addLink(linkAtStartDeleted);
134            if(this.linkAtEndDeleted!=null)
135                net.addLink(linkAtEndDeleted);
136            if(this.linkAtStartAdded1!=null)
137                net.removeLink(linkAtStartAdded1.getId());
138            if(this.linkAtStartAdded2!=null)
139                net.removeLink(linkAtStartAdded2.getId());
140            if(this.linkAtEndAdded1!=null)
141                net.removeLink(linkAtEndAdded1.getId());
142            if(this.linkAtEndAdded2!=null)
143                net.removeLink(linkAtEndAdded2.getId());
144            if(this.nodeAtStart!=null)
145                net.removeNode(nodeAtStart.getId());
146            if(this.nodeAtEnd!=null)
147                net.removeNode(nodeAtEnd.getId());
148        }
149
150        private void editLink() {
151            if(link == null) return;
152            Link toEdit = net.getLinks().get(link.getId());
153            Link oldLink = cloneLink(toEdit);
154            toEdit.setCapacity(link.getCapacity());
155            toEdit.setFreespeed(link.getFreespeed());
156            toEdit.setLength(link.getLength());
157            toEdit.setNumberOfLanes(link.getNumberOfLanes());
158            link = oldLink;
159        }
160
161        private void createLink() {            
162            if(this.linkAtStartDeleted!=null)
163                net.removeLink(linkAtStartDeleted.getId());
164            if(this.linkAtEndDeleted!=null)
165                net.removeLink(linkAtEndDeleted.getId());
166            if(this.nodeAtStart!=null)
167                net.addNode(nodeAtStart);
168            if(this.nodeAtEnd!=null)
169                net.addNode(nodeAtEnd);
170            if(this.linkAtStartAdded1!=null)
171                net.addLink(linkAtStartAdded1);
172            if(this.linkAtStartAdded2!=null)
173                net.addLink(linkAtStartAdded2);
174            if(this.linkAtEndAdded1!=null)
175                net.addLink(linkAtEndAdded1);
176            if(this.linkAtEndAdded2!=null)
177                net.addLink(linkAtEndAdded2);
178            if(link != null)
179                net.addLink(link);
180        }
181
182    }
183
184}
