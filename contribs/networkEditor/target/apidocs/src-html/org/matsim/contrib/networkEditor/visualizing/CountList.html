165    private void jButton1ActionPerformed(java.awt.event.ActionEvent evt) {//GEN-FIRST:event_jButton1ActionPerformed
166        if(board.selectedLinkSize()!=1) return;
167        currentCount = board.getCountingStation(board.activeLink);
168        if(currentCount == null) {
169            String name = javax.swing.JOptionPane.showInputDialog("Name of counting station");
170            currentCount = board.addCountingStation(board.activeLink, name);
171        }
172        observer = new VolumeInputObserver(this);
173        VolumeInput volume_input = new VolumeInput(observer);
174        volume_input.setLocation(this.controls.board.getX()+100, this.controls.board.getY()+100);
175        volume_input.run();
176        updateTable();
177    }//GEN-LAST:event_jButton1ActionPerformed

215    public void updateTable() {
216        if(board.activeLink == null) {
217            tableModel = new CountTableModel();
218            jLabel1.setText("");
219        } else {
220            currentCount = board.counts.getCount(board.activeLink.getId());
221            if(currentCount == null) {
222                tableModel = new CountTableModel();
223                jLabel1.setText("(no existing counts)");
224            } else {
225                tableModel = new CountTableModel(currentCount);
226                jLabel1.setText(currentCount.getCsLabel());
227            }
228        }
229        jTable1.setModel(tableModel);
230        jTable1.getModel().addTableModelListener(new CountTableModelListener(this));
231        jTable1.repaint();
232
233    }

259class CountTableModel extends AbstractTableModel {
260
261    public int rows, cols;
262    Object table[][];
263    boolean editable[][];
264    String columnName[];
265    Count<Link> counts;
266    public CountTableModel(){
267        init(0);
268    }
269    public CountTableModel(Count<Link> counts) {
270        this.counts = counts;
271        init(counts.getVolumes().size());
272        int i=0;
273        for(Map.Entry<Integer, Volume> entry : counts.getVolumes().entrySet()) {
274            table[i][0] = entry.getKey().toString();
275            table[i][1] = Double.toString(entry.getValue().getValue());
276            ++i;
277        }
278    }
279    private void init(int r) {
280        table = new Object[r][2];
281        editable = new boolean[r][2];
282        for(int i=0; i<editable.length; ++i) {
283            editable[i][0] = false;
284            editable[i][1] = true;
285        }
286        columnName = new String[2];
287        columnName[0] = "Hour";
288        columnName[1] = "Volume";
289        rows = r;
290        cols = 2;
291    }
292
293    @Override
294                public int getRowCount(){
295        return rows;
296    }
297
298    @Override
299                public int getColumnCount() {
300        return cols;
301    }
302    @Override
303    public String getColumnName(int columnIndex) {
304        return columnName[columnIndex];
305    }
306    @Override
307    public Class<?> getColumnClass(int columnIndex) {
308        return this.getValueAt(0, columnIndex).getClass();
309    }
310    @Override
311    public boolean isCellEditable(int rowIndex, int columnIndex) {
312        return editable[rowIndex][columnIndex];
313    }
314    @Override
315    public Object getValueAt(int rowIndex, int columnIndex) {
316        return table[rowIndex][columnIndex];
317    }
318    @Override
319    public void setValueAt(Object aValue, int rowIndex, int columnIndex) {
320        if(counts == null)
321            return;
322        if(columnIndex == 0) {
323            int oldVal = Integer.valueOf(table[rowIndex][columnIndex].toString());
324            //System.out.println(oldVal);
325            double vol = counts.getVolumes().get(oldVal).getValue();
326            counts.getVolumes().remove(oldVal);
327            int val = Integer.valueOf(aValue.toString());
328            table[rowIndex][columnIndex] = val;
329            counts.createVolume(val, vol);
330        } else {
331            System.out.println(aValue.toString());
332            double val = Double.parseDouble(aValue.toString());
333            table[rowIndex][columnIndex] = val;
334            int h = counts.getVolumes().get(Integer.valueOf(table[rowIndex][0].toString())).getHourOfDayStartingWithOne();
335            counts.getVolumes().remove(h);
336            counts.createVolume(h, val);
337        }
338        fireTableCellUpdated(rowIndex, columnIndex);
339    }
340
341}

343class CountTableModelListener implements TableModelListener {
344    public CountList cList;
345
346    CountTableModelListener(CountList c) {
347        cList = c;
348    }
349
350    @Override
351                public void tableChanged(TableModelEvent evt) {
352        CountTableModel tableModel = (CountTableModel)evt.getSource();
353        if (evt.getType() == TableModelEvent.UPDATE) {
354            int column = evt.getColumn();
355            int row = evt.getFirstRow();
356            String rawdata = tableModel.getValueAt(row, column).toString();
357            double data = 0;
358            try {
359                data = Double.parseDouble(rawdata);
360            } catch (Exception e) {
361                javax.swing.JOptionPane.showMessageDialog(cList, "Values must be numeric.", "Format error", javax.swing.JOptionPane.WARNING_MESSAGE);
362                cList.updateTable();
363                //e.printStackTrace();
364                return;
365            }
366            int hour = Integer.parseInt(tableModel.getValueAt(row, 0).toString());
367            tableModel.setValueAt(data, row, column);
368            cList.board.editCount(cList.board.activeLink, hour, data);
369            cList.updateTable();
370        }
371    }
372 }

374class VolumeInputObserver implements Observer {
375
376    private CountList dad;
377
378    public VolumeInputObserver(CountList dad) {
379        this.dad = dad;
380    }
381
382    @Override
383                public void update(Observable obs, Object arg) {
384        if(arg!=null) {
385            MyVolume myvol = (MyVolume)arg;
386            if(myvol.h_ < 1 || myvol.h_ > 24)
387                javax.swing.JOptionPane.showMessageDialog(dad, "Invalid Hour", "Error", javax.swing.JOptionPane.ERROR_MESSAGE);
388            if(myvol.v_ < 0.0)
389                javax.swing.JOptionPane.showMessageDialog(dad, "Invalid Volume", "Error", javax.swing.JOptionPane.ERROR_MESSAGE);
390            if(dad.existCount(myvol.h_)) {
391                if(javax.swing.JOptionPane.showConfirmDialog(dad, "Hour already exists, replace it?", "Warning", javax.swing.JOptionPane.YES_NO_OPTION) == javax.swing.JOptionPane.OK_OPTION)
392                    dad.addVolume(myvol.h_, myvol.v_);
393            } else
394                dad.addVolume(myvol.h_, myvol.v_);
395        }
396    }
