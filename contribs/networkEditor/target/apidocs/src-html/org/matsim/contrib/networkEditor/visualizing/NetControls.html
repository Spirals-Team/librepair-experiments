556        private void checkToggle(int cT) {
557                if(selected[cT]==false) {
558                        setToogle(cT);
559                } else {
560                        selected[cT]=false;
561                        toggles[cT].setSelected(false);
562                }
563                updateBoardState();
564        }
565
566        private void setToogle(int t) {
567                for(int i=0; i<this.numToggle; ++i){
568                        selected[i] = i==t;
569                        if(i>0)
570                                toggles[i].setSelected(i==t);
571                }
572        }
573
574        private void updateBoardState() {
575                NetBlackboard.Mode modes[] = NetBlackboard.Mode.values();
576                for(int i=0; i<selected.length; ++i) {
577                        if(selected[i]==true) {
578                                board.setMode(modes[i]);
579                                break;
580                        }
581                }
582        }

621        public void updateTable() {
622                tableModel = new MyTableModel();
623                if(board.activeLink != null) tableModel = new MyTableModel(board.activeLink);
624                if(board.activeNode != null) tableModel = new MyTableModel(board.activeNode);
625                jTable1.setModel(tableModel);
626                jTable1.setVisible(true);
627                jTable1.getModel().addTableModelListener(new MyTableModelListener(this));
628                //jTable1.pre
629                jTable1.repaint();
630                this.countList.updateTable();
631        }

633        public void updateButtons() {
634                if(board.diffManager.getBackSize() < 1)

638                if(board.diffManager.getFrontSize() < 1)
639                        this.redoButton.setEnabled(false);
640                else
641                        this.redoButton.setEnabled(true);
642        }

646class MyTableModel extends AbstractTableModel {
647        private String table[][];

651        public enum mode{LINK, NODE, NONE};

675                //editable[2][1] = true;
676
677                table[3][0] = "Free speed";

679                editable[3][1] = true;
680
681                table[4][0] = "Free speed travel time";

684
685                table[5][0] = "Length";
686                table[5][1] = Double.toString(activeLink.getLength());
687                editable[5][1] = true;
688
689                table[6][0] = "Number of lanes";

704        public MyTableModel(Node activeNode) {
705                init(3,2);
706                this.activeNode = activeNode;
707                actualMode = mode.NODE;
708
709                table[0][0] = "Id";
710                table[0][1] = activeNode.getId().toString();
711                table[1][0] = "X";
712                table[1][1] = Double.toString(activeNode.getCoord().getX());
713                table[2][0] = "Y";
714                table[2][1] = Double.toString(activeNode.getCoord().getY());
715        }

717        private void init(int r, int c) {
718                table = new String[r][c];
719                editable = new boolean[r][c];
720                for(int i=0; i<editable.length; ++i) for(int j=0; j<editable[i].length; ++j) editable[i][j] = false;
721                columnName = new String[c];
722                columnName[0] = "Property";
723                columnName[1] = "Value";
724                rows = r;
725                cols = c;
726        }

728        /**

731         * should display.  This method should be quick, as it

755        /**
756         * Returns the name of the column at <code>columnIndex</code>.  This is used
757         * to initialize the table's column header name.  Note: this name does
758         * not need to be unique; two columns in a table can have the same name.
759         *
760         * @param       columnIndex     the index of the column
761         * @return  the name of the column
762         */
763        @Override
764        public String getColumnName(int columnIndex) {
765                return columnName[columnIndex];
766        }

768        /**
769         * Returns the most specific superclass for all the cell values 
770         * in the column.  This is used by the <code>JTable</code> to set up a 
771         * default renderer and editor for the column.
772         *
773         * @param columnIndex  the index of the column
774         * @return the common ancestor class of the object values in the model.
775         */
776        @Override
777        public Class<?> getColumnClass(int columnIndex) {
778                return this.getValueAt(0, columnIndex).getClass();
779        }

781        /**
782         * Returns true if the cell at <code>rowIndex</code> and
783         * <code>columnIndex</code>
784         * is editable.  Otherwise, <code>setValueAt</code> on the cell will not
785         * change the value of that cell.
786         *
787         * @param       rowIndex        the row whose value to be queried
788         * @param       columnIndex     the column whose value to be queried
789         * @return      true if the cell is editable
790         * @see #setValueAt
791         */
792        @Override
793        public boolean isCellEditable(int rowIndex, int columnIndex) {
794                return editable[rowIndex][columnIndex];
795        }
796
797        /**
798         * Returns the value for the cell at <code>columnIndex</code> and
799         * <code>rowIndex</code>.
800         *
801         * @param       rowIndex        the row whose value is to be queried
802         * @param       columnIndex     the column whose value is to be queried
803         * @return      the value Object at the specified cell
804         */
805        @Override
806        public Object getValueAt(int rowIndex, int columnIndex) {
807                return table[rowIndex][columnIndex];
808        }
809
810        /**
811         * Sets the value in the cell at <code>columnIndex</code> and
812         * <code>rowIndex</code> to <code>aValue</code>.
813         *
814         * @param       aValue           the new value
815         * @param       rowIndex         the row whose value is to be changed
816         * @param       columnIndex      the column whose value is to be changed
817         * @see #getValueAt
818         * @see #isCellEditable
819         */
820        @Override
821        public void setValueAt(Object aValue, int rowIndex, int columnIndex) {
822                table[rowIndex][columnIndex] = aValue.toString();
823                fireTableCellUpdated(rowIndex, columnIndex);
824        }

828class MyTableModelListener implements TableModelListener {
829        public NetControls controls;
830
831        MyTableModelListener(NetControls c) {
832                controls = c;
833        }
834
835        @Override
836        public void tableChanged(TableModelEvent evt) {
837                MyTableModel tableModel = (MyTableModel)evt.getSource();
838                //System.out.println("FUCK THIS!");
839                if (evt.getType() == TableModelEvent.UPDATE) {
840                        int column = evt.getColumn();
841                        int row = evt.getFirstRow();
842                        //System.out.println("row: " + row + " column: " + column);
843                        if(tableModel.actualMode == MyTableModel.mode.NONE) return;
844                        if(tableModel.actualMode == MyTableModel.mode.NODE) return;
845                        if(tableModel.actualMode == MyTableModel.mode.LINK){
846                                if(tableModel.isCellEditable(row, column)== false) return;
847                                String rawdata = (String)tableModel.getValueAt(row, column);
848                                double data = 0.0;
849                                try {
850                                        data = Double.parseDouble(rawdata);
851                                } catch (Exception e) {
852                                        javax.swing.JOptionPane.showMessageDialog(controls, "Los valores deben ser numéricos", "Error de formato", javax.swing.JOptionPane.WARNING_MESSAGE);
853                                        controls.updateTable();
854                                        //e.printStackTrace();
855                                        return;
856                                }
857                                controls.diffManager.saveState(controls.diffManager.cloneLink(tableModel.activeLink), DifferenceManager.type.EDIT, null, null, null, null, null, null, null, null);
858                                if(row == 0) {
859                                        tableModel.activeLink.setCapacity(data);
860                                        controls.board.setMinMaxCaps();
861                                } else if(row == 3)
862                                        tableModel.activeLink.setFreespeed(data);
863                                else if(row == 5)
864                                        tableModel.activeLink.setLength(data);
865                                else if(row == 6)
866                                        tableModel.activeLink.setNumberOfLanes(data);
867                                controls.updateTable();
868                                controls.updateButtons();
869                        }
870                }
871        }
