023package org.matsim.contrib.wagonSim.schedule;
024
025import java.io.BufferedReader;
026import java.io.IOException;
027import java.text.ParseException;
028import java.util.Date;
029import java.util.LinkedHashMap;
030import java.util.Map;
031
032import org.matsim.api.core.v01.Id;
033import org.matsim.api.core.v01.network.Node;
034import org.matsim.contrib.wagonSim.Utils;
035import org.matsim.contrib.wagonSim.WagonSimConstants;
036import org.matsim.contrib.wagonSim.schedule.OTTDataContainer.Locomotive;
037import org.matsim.contrib.wagonSim.schedule.OTTDataContainer.StationData;
038import org.matsim.core.utils.io.IOUtils;
039import org.matsim.pt.transitSchedule.api.TransitStopFacility;
040
041/**
042 * @author balmermi @ Senozon AG
043 * @since 2013-07-08
044 */
045public class OTTScheduleParser {
046
047        //////////////////////////////////////////////////////////////////////
048        // variables
049        //////////////////////////////////////////////////////////////////////
050        
051        private final OTTDataContainer dataContainer;
052
053        private static final String VERKEHRSTAG = "Verkehrstag";
054        private static final String ZUGNUMMER = "Zugnummer";
055        private static final String IST_AN = "IST AN";
056        private static final String IST_AB = "IST AB";
057        private static final String VERSPAETUNG_AN = "Verspaetung AN";
058        private static final String VERSPAETUNG_AB = "Verspaetung AB";
059        private static final String BETRIEBSSTELLE = "Betriebstelle";
060        private static final String ZUGART = "Zugart";
061        
062        private static final double DEFAULTFLAG = 999999.0;
063
064        //////////////////////////////////////////////////////////////////////
065        // constructors
066        //////////////////////////////////////////////////////////////////////
067        
068        public OTTScheduleParser(OTTDataContainer dataContainer) {
069                this.dataContainer = dataContainer;
070        }
071
072        //////////////////////////////////////////////////////////////////////
073        // methods
074        //////////////////////////////////////////////////////////////////////
075        
076        public final void parse(String ottFile, Map<Id<Node>, Id<Node>> nodeMap) throws IOException {
077
078                BufferedReader br = IOUtils.getBufferedReader(ottFile);
079                int currRow = 0;
080                String curr_line;
081
082                // read header an build lookup
083                curr_line = br.readLine(); currRow++;
084                String [] header = curr_line.split(";");
085                Map<String, Integer> lookup = new LinkedHashMap<String,Integer>(header.length);
086                for (int i=0; i<header.length; i++) { lookup.put(Utils.removeSurroundingQuotes(header[i].trim()),i); }
087
088                // to check for undefined time 
089                Date dateZero = new Date(0);
090
091                // parse rows and store nodes
092                StationData prevStationData = null;
093                while ((curr_line = br.readLine()) != null) {
094                        currRow++;
095                        String [] row = curr_line.split(";");
096
097                        Date day = null;
098                        try { day = WagonSimConstants.DATE_FORMAT_DDMMYYYYHHMMSS.parse(Utils.removeSurroundingQuotes(row[lookup.get(VERKEHRSTAG)].trim())); }
099                        catch (ParseException e) { throw new RuntimeException("row "+currRow+": Column '"+VERKEHRSTAG+"' not well fomatted. Bailing out."); }
100                        Integer locNr = Integer.parseInt(Utils.removeSurroundingQuotes(row[lookup.get(ZUGNUMMER)].trim()));
101                        Date arrival = null;
102                        try { arrival = WagonSimConstants.DATE_FORMAT_YYYYMMDDHHMMSS.parse(Utils.removeSurroundingQuotes(row[lookup.get(IST_AN)].trim())); }
103                        catch (ParseException e) { throw new RuntimeException("row "+currRow+": Column '"+IST_AN+"' not well fomatted. Bailing out."); }
104                        Date departure = null;
105                        try { departure = WagonSimConstants.DATE_FORMAT_YYYYMMDDHHMMSS.parse(Utils.removeSurroundingQuotes(row[lookup.get(IST_AB)].trim())); }
106                        catch (ParseException e) { throw new RuntimeException("row "+currRow+": Column '"+IST_AB+"' not well fomatted. Bailing out."); }
107                        double delayArrival = Double.parseDouble(Utils.removeSurroundingQuotes(row[lookup.get(VERSPAETUNG_AN)].trim()));
108                        double delayDeparture = Double.parseDouble(Utils.removeSurroundingQuotes(row[lookup.get(VERSPAETUNG_AB)].trim()));
109                        Id<TransitStopFacility> station = Id.create(Utils.removeSurroundingQuotes(row[lookup.get(BETRIEBSSTELLE)].trim()), TransitStopFacility.class);
110                        int locType = Integer.parseInt(Utils.removeSurroundingQuotes(row[lookup.get(ZUGART)].trim()));
111                        
112                        Locomotive locomotive = dataContainer.locomotives.get(Id.create(locNr.longValue(), Locomotive.class));
113                        if (locomotive == null) { locomotive = new Locomotive(locNr,locType); dataContainer.locomotives.put(locomotive.id,locomotive); prevStationData = null; }
114                        if (locomotive.type.intValue() != locType) { throw new RuntimeException("row "+currRow+": alredy given locomotive type="+locomotive.type+" does not fit with locType="+locType+". Bailing out."); }
115
116                        // adapt default arrival to the given departure
117                        if (delayArrival == DEFAULTFLAG) {
118                                if (!arrival.before(dateZero)) { throw new RuntimeException("row "+currRow+": default arrival ("+delayArrival+") does not fit to arrival time ("+arrival.toString()+"). Bailing out."); }
119                                if (delayDeparture == DEFAULTFLAG) { throw new RuntimeException("row "+currRow+": default given for arrival ("+delayArrival+") and departure ("+delayDeparture+"). Bailing out."); }
120                                arrival = new Date(departure.getTime());
121                        }
122                        
123                        // adapt default departure to the given arrival
124                        if (delayDeparture == DEFAULTFLAG) {
125                                if (!departure.before(dateZero)) { throw new RuntimeException("row "+currRow+": default departure ("+delayDeparture+") does not fit to departure time ("+departure.toString()+"). Bailing out."); }
126                                if (delayArrival == DEFAULTFLAG) { throw new RuntimeException("row "+currRow+": default given for arrival ("+delayArrival+") and departure ("+delayDeparture+"). Bailing out."); }
127                                departure = new Date(arrival.getTime());
128                        }
129
130                        // check if there is the same time twice
131                        if (locomotive.trips.containsKey(arrival)) { throw new RuntimeException("row "+currRow+": locomotive id="+locomotive.id+" already arrived at a station at "+IST_AN+"="+arrival.toString()+". Bailing out."); }
132                        
133                        // reset departure and delayDeparture if it's earlier than arrival
134                        if (departure.before(arrival)) {
135                                long delta = (arrival.getTime() - departure.getTime())/1000l; // delta in seconds
136                                delayDeparture += delta;
137                                departure = new Date(arrival.getTime());
138                        }
139                        
140                        // replace station ids according to nodeMap
141                        Id<Node> mappedStationId = nodeMap.get(Id.create(station, Node.class));
142                        if (mappedStationId != null) {
143                                station = Id.create(mappedStationId.toString(), TransitStopFacility.class);
144                        }
145                        
146                        // take care that there is no DEFAULTFLAG ANYMORE
147                        if (delayArrival == DEFAULTFLAG) { delayArrival = delayDeparture; }
148                        if (delayDeparture == DEFAULTFLAG) { delayDeparture = delayArrival; }
149
150                        if ((prevStationData != null) && (prevStationData.stationId.equals(station))) {
151                                prevStationData.delayDeparture = delayDeparture;
152                                prevStationData.departure = departure;
153                                System.out.println("row "+currRow+": station "+station.toString()+" is the same as before. Merging station data into one (to remove loop trips).");
154                        }
155                        else {
156                                StationData stationData = new StationData();
157                                stationData.stationId = station;
158                                stationData.delayArrival = delayArrival;
159                                stationData.arrival = arrival;
160                                stationData.delayDeparture = delayDeparture;
161                                stationData.departure = departure;
162                                locomotive.trips.put(stationData.arrival,stationData);
163                                prevStationData = stationData;
164                        }
165                }
166        }
167}
