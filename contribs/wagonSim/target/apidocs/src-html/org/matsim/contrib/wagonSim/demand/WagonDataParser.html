023package org.matsim.contrib.wagonSim.demand;
024
025import java.io.BufferedReader;
026import java.io.IOException;
027import java.text.ParseException;
028import java.util.Date;
029import java.util.HashMap;
030import java.util.LinkedHashMap;
031import java.util.Map;
032
033import org.matsim.contrib.wagonSim.WagonSimConstants;
034import org.matsim.contrib.wagonSim.demand.WagonDataContainer.Wagon;
035import org.matsim.core.utils.io.IOUtils;
036
037/**
038 * @author balmermi
039 * @since 2013-08-19
040 */
041public class WagonDataParser {
042
043        //////////////////////////////////////////////////////////////////////
044        // variables
045        //////////////////////////////////////////////////////////////////////
046
047        private final WagonDataContainer dataContainer;
048        private final Date referenceDate;
049
050//      private static final String GV_ART_NEU = "GV_Art_neu";
051        private static final String DATUM_VS = "\"Datum_VS\"";
052//      private static final String UICBAHN_VS = "UICBahn_VS";
053//      private static final String BHFNR_VS = "BhfNr_VS";
054//      private static final String BHFNR_EM = "BhfNr_EM";
055//      private static final String TRANSPORT_ART = "Transport_Art";
056//      private static final String BEF_ART = "Bef_Art";
057        private static final String NETTO_GEW = "\"Netto_Gew\"";
058//      private static final String BRUTTO_GEW = "Brutto_Gew";
059        private static final String VONZELLE = "\"vonzelle\"";
060        private static final String NACHZELLE = "\"nachzelle\"";
061        private static final String WAGENLAENGE = "\"Wagenlaenge\"";
062        private static final String TARA_WAGEN = "\"Tara_Wagen\"";
063//      private static final String TYP = "Typ";
064//      private static final String NETTO_NEU = "netto_neu";
065        
066        //////////////////////////////////////////////////////////////////////
067        // constructors
068        //////////////////////////////////////////////////////////////////////
069        
070        public WagonDataParser(WagonDataContainer dataContainer, Date referenceDate) {
071                this.dataContainer = dataContainer;
072                this.referenceDate = referenceDate;
073        }
074
075        //////////////////////////////////////////////////////////////////////
076        // methods
077        //////////////////////////////////////////////////////////////////////
078        
079        @SuppressWarnings("deprecation")
080        public final void parse(String wagonDataFile) throws IOException {
081                
082                BufferedReader br = IOUtils.getBufferedReader(wagonDataFile);
083                int currRow = 0;
084                String curr_line;
085
086                // read header and build lookup
087                curr_line = br.readLine(); currRow++;
088                String [] header = curr_line.split(";");
089                Map<String, Integer> lookup = new LinkedHashMap<String,Integer>(header.length);
090                for (int i=0; i<header.length; i++) { lookup.put(header[i].trim(),i); }
091                
092                Map<String,Integer> currIdIndices = new HashMap<String, Integer>();
093
094                // parse rows and store nodes
095                while ((curr_line = br.readLine()) != null) {
096                        currRow++;
097                        String [] row = curr_line.split(";");
098
099                        Date day = null;
100                        try { day = WagonSimConstants.DATE_FORMAT_DDMMYYYYHHMMSS.parse(row[lookup.get(DATUM_VS)].trim()); }
101                        catch (ParseException e) { throw new RuntimeException("row "+currRow+": Column '"+DATUM_VS+"' not well fomatted. Bailing out."); }
102
103                        // only the same day
104                        if (referenceDate.getYear() != day.getYear()) { continue; }
105                        if (referenceDate.getMonth() != day.getMonth()) { continue; }
106                        if (referenceDate.getDate() != day.getDate()) { continue; }
107                        
108                        Double weightLoad = Double.parseDouble(row[lookup.get(NETTO_GEW)].trim());
109                        String fromZoneId = row[lookup.get(VONZELLE)].trim();
110                        String toZoneId = row[lookup.get(NACHZELLE)].trim();
111                        Double length = Double.parseDouble(row[lookup.get(WAGENLAENGE)].trim())/100.0;
112                        Double weight = Double.parseDouble(row[lookup.get(TARA_WAGEN)].trim());
113                        
114                        String idString = fromZoneId.toString()+"-"+toZoneId.toString();
115                        Integer index = currIdIndices.get(idString);
116                        if (index == null) { index = 0; currIdIndices.put(idString,index); }
117                        currIdIndices.put(idString,index+1);
118                        idString += "_"+index;
119                        
120                        Wagon wagon = new Wagon(idString);
121                        wagon.weight = weight;
122                        wagon.length = length;
123                        wagon.weightLoad = weightLoad;
124                        wagon.fromZoneId = fromZoneId;
125                        wagon.toZoneId = toZoneId;
126
127                        // setting depTime to the next day
128                        wagon.depTime = (referenceDate.getHours()+24)*3600.0 + referenceDate.getMinutes()*60.0 + referenceDate.getSeconds();
129
130                        dataContainer.wagons.put(wagon.id,wagon);
131                }
132        }
133}
