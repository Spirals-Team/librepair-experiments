055/**
056 * Plugs a handler to the {@link Controler} which observes the locomotives (pt-vehicles), more precisely their
057 * load. Furthermore this class provides access to the current-vehicleload (and those of the last iteration).
058 * 
059 * @author droeder
060 *
061 */
062public final class WagonSimVehicleLoadListener implements StartupListener, IterationStartsListener, IterationEndsListener {
063
064        private static final Logger log = Logger
065                        .getLogger(WagonSimVehicleLoadListener.class);
066        private VehicleLoadHandler handler;
067        private ObjectAttributes wagonAttribs;
068        private VehicleLoad lastLoad;
069
070        public WagonSimVehicleLoadListener(ObjectAttributes wagonAttribs) {
071                this.wagonAttribs = wagonAttribs;
072        }
073
074        @Override
075        public void notifyStartup(StartupEvent event) {
076                this.handler = new VehicleLoadHandler(wagonAttribs);
077                handler.vehicleLoad = new VehicleLoad(event.getServices().getScenario().getTransitSchedule());
078                handler.reset(-1);
079                lastLoad = handler.vehicleLoad;
080                event.getServices().getEvents().addHandler(handler);
081        }
082
083
084        @Override
085        public void notifyIterationStarts(IterationStartsEvent event) {
086                handler.vehicleLoad = new VehicleLoad(event.getServices().getScenario().getTransitSchedule());
087                handler.reset(event.getIteration());
088        }
089        
090        @Override
091        public void notifyIterationEnds(IterationEndsEvent event) {
092                this.lastLoad = handler.vehicleLoad;
093                BufferedWriter w = IOUtils.getBufferedWriter(
094                                event.getServices().getControlerIO().getIterationFilename(event.getIteration(), "locomotiveLoadPerDepartureAndStop.txt"));
095                try {
096                        w.write(this.lastLoad.toString());
097                        w.flush();
098                        w.close();
099                } catch (IOException e) {
100                        e.printStackTrace();
101                }
102        }
103        
104        /**
105         * @return
106         */
107        public final VehicleLoad getLoadOfLastIter() {
108                return lastLoad;
109        }
110        
111        /**
112         * @return
113         */
114        public VehicleLoad getCurrentLoad() {
115                return this.handler.vehicleLoad;
116        }

119        public static final String getIdentifier(String vehId, String stopId, int stopIndex){
120                String delimeter = "---";
121                return new String(vehId + delimeter + 
122                                stopId + delimeter +
123                                String.valueOf(stopIndex));
124        }
125        
126        public static final boolean freeCapacityInVehicle(
127                        String vehId,
128                        String stopId,
129                        int stopIndex,
130                        String wagonId,
131                        VehicleLoad vehLoad, 
132                        ObjectAttributes locomotiveAttribs,
133                        ObjectAttributes wagonAttribs) throws NullPointerException{
134                String vehDepId = WagonSimVehicleLoadListener.getIdentifier(vehId, stopId, stopIndex);
135                double expectedLengthOfTheTrain = vehLoad.getAdditionalLengthAtStopForVehicle(vehDepId);
136                double expectedWeightOfTheTrain = vehLoad.getAdditionalWeightAtStopForVehicle(vehDepId);
137                Double maxLoadOfTrain = (Double) locomotiveAttribs.getAttribute(vehId, WagonSimConstants.TRAIN_MAX_WEIGHT);
138                Double maxLengthOfTrain = (Double) locomotiveAttribs.getAttribute(vehId, WagonSimConstants.TRAIN_MAX_LENGTH);
139                Double lengthOfWagon = (Double) wagonAttribs.getAttribute(wagonId, WagonSimConstants.WAGON_LENGTH);
140                Double weightOfWagon = (Double) wagonAttribs.getAttribute(wagonId, WagonSimConstants.WAGON_GROSS_WEIGHT);
141// check if the data is available (it should). otherwise, throw error 
142                if((lengthOfWagon == null) || (weightOfWagon == null) || (maxLengthOfTrain == null) || (maxLoadOfTrain == null) ){
143                        throw new NullPointerException("Either the locomotive or the wagon or the necessary attributes are unknown.");
144                }
145                if((maxLengthOfTrain < (expectedLengthOfTheTrain + lengthOfWagon)) || 
146                                (maxLoadOfTrain < (expectedWeightOfTheTrain + weightOfWagon))){
147                        return false;
148                }
149                return true;
150        }

153        private class VehicleLoadHandler implements TransitDriverStartsEventHandler,
154                                                                                                                PersonEntersVehicleEventHandler,
155                                                                                                                PersonLeavesVehicleEventHandler,
156                                                                                                                VehicleArrivesAtFacilityEventHandler{
157                
158                private VehicleLoad vehicleLoad;
159                private List<Id> ptVehId;
160                private Map<Id, Integer> veh2StopIndex;
161                private Map<Id, Id> vehAtFacility;
162                private ObjectAttributes oa;
163                private List<Id> ptDrivers;
164
165                VehicleLoadHandler(ObjectAttributes oa){
166                        this.oa = oa;
167                }
168                
169                @Override
170                public void reset(int iteration) {
171                        vehicleLoad.reset();
172                        this.ptVehId = new ArrayList<Id>();
173                        this.veh2StopIndex = new HashMap<Id, Integer>();
174                        this.vehAtFacility = new HashMap<Id, Id>();
175                        this.ptDrivers = new ArrayList<Id>();
176                }
177
178                @Override
179                public void handleEvent(VehicleArrivesAtFacilityEvent event) {
180                        // handle only ``pt''-vehicles
181                        if(!ptVehId.contains(event.getVehicleId())) return;
182                        Id veh = event.getVehicleId();
183                        LoadPerDeparturePerStop oldLoad = null;
184                        // get the previous location/load, but not for the first stop
185                        if(vehAtFacility.containsKey(veh)){
186                                String id = getIdentifier(veh.toString(), vehAtFacility.get(veh).toString(), veh2StopIndex.get(veh));
187                                oldLoad = vehicleLoad.load.get(id);
188                        }
189                        // personsEnter/leave-events do not know about position. Thus, store this information.
190                        incStopIndex(veh);
191                        this.vehAtFacility.put(veh, event.getFacilityId());
192                        // copy the load to the next stop-departure-combination, for all except the first stop
193                        if(!(oldLoad == null)){
194                                String id = getIdentifier(veh.toString(), vehAtFacility.get(veh).toString(), veh2StopIndex.get(veh));
195                                vehicleLoad.entering(id, oldLoad.weight, oldLoad.length);
196                        }
197                }
198
199                @Override
200                public void handleEvent(PersonEntersVehicleEvent event) {
201                        // handle only ``pt''-vehicles, but not their drivers
202                        if(!ptVehId.contains(event.getVehicleId())) return;
203                        if(ptDrivers.contains(event.getPersonId())) return;
204                        Id veh = event.getVehicleId();
205                        String id = getIdentifier(veh.toString(), vehAtFacility.get(veh).toString(), veh2StopIndex.get(veh));
206                        Double weight = (Double) this.oa.getAttribute(event.getPersonId().toString(), WagonSimConstants.WAGON_GROSS_WEIGHT);
207                        Double length = (Double) this.oa.getAttribute(event.getPersonId().toString(), WagonSimConstants.WAGON_LENGTH);
208                        this.vehicleLoad.entering(id, weight, length);
209                }
210                
211                @Override
212                public void handleEvent(PersonLeavesVehicleEvent event) {
213                        // handle only ``pt''-vehicles, but not their drivers
214                        if(!ptVehId.contains(event.getVehicleId())) return;
215                        if(ptDrivers.contains(event.getPersonId())) return;
216                        Id veh = event.getVehicleId();
217                        String id = getIdentifier(veh.toString(), vehAtFacility.get(veh).toString(), veh2StopIndex.get(veh));
218                        Double weight = (Double) this.oa.getAttribute(event.getPersonId().toString(), WagonSimConstants.WAGON_GROSS_WEIGHT);
219                        Double length = (Double) this.oa.getAttribute(event.getPersonId().toString(), WagonSimConstants.WAGON_LENGTH);
220                        this.vehicleLoad.leaving(id, weight, length);
221                }
222
223                @Override
224                public void handleEvent(TransitDriverStartsEvent event) {
225                        // intialize information about the vehicles
226                        this.ptVehId.add(event.getVehicleId());
227                        this.veh2StopIndex.put(event.getVehicleId(), -1);
228                        this.ptDrivers.add(event.getDriverId());
229                }
230                
231                private void incStopIndex(Id vehId){
232                        Integer index = this.veh2StopIndex.get(vehId);
233                        if(index == null) throw new RuntimeException("you tried to increase the stopIndex for a vehicle which does not exist...");
234                        this.veh2StopIndex.put(vehId, index + 1);
235                }
236
237
238                
239        }

242        public static final class VehicleLoad {
243                
244                private Map<String, LoadPerDeparturePerStop> load;
245
246                public VehicleLoad(TransitSchedule schedule) {
247                        init(schedule);
248                }
249
250                /**
251                 * @param schedule
252                 */
253                private void init(TransitSchedule schedule) {
254                        this.load = new HashMap<String, LoadPerDeparturePerStop>();
255                        for(TransitLine l: schedule.getTransitLines().values()){
256                                for(TransitRoute r: l.getRoutes().values()){
257                                        for(Departure d: r.getDepartures().values()){
258                                                for(int i = 0; i < r.getStops().size(); i++){
259                                                        TransitRouteStop s = r.getStops().get(i);
260                                                        String id = getIdentifier(d.getVehicleId().toString(), s.getStopFacility().getId().toString(), i);
261                                                        LoadPerDeparturePerStop lo = this.load.put(id, new LoadPerDeparturePerStop(id));
262                                                        if(!(lo == null)){
263                                                                throw new RuntimeException("the identifier of this departure-stop-combination is not unique! " + id);
264                                                        }
265                                                }
266                                        }
267                                }
268                        }
269                        reset();
270                }
271                
272                /**
273                 * 
274                 */
275                void reset() {
276                        for(String s: load.keySet()){
277                                load.get(s).reset();
278                        }
279                }
280                
281                public void entering(String id, double weight, double length){
282                        LoadPerDeparturePerStop l = this.load.get(id);
283                        l.sumUp(weight, length);
284                }
285                
286                public void leaving(String id, double weight, double length){
287                        LoadPerDeparturePerStop l = this.load.get(id);
288                        l.subtract(weight, length);
289                }
290                
291                public final double getAdditionalWeightAtStopForVehicle(String id){
292                        return load.get(id).weight;
293                }
294                
295                public final double getAdditionalLengthAtStopForVehicle(String id){
296                        return load.get(id).length;
297                }
298                
299                @Override
300                public String toString(){
301                        StringBuffer b = new StringBuffer();
302                        b.append("id\tlength\tweight\n");
303                        for(LoadPerDeparturePerStop l: this.load.values()){
304                                b.append(l.toString() + "\n");
305                        }
306                        return b.toString();
307                }
308                
309        }

311        /**
312         * small inner helper-class to store weight/length for a current departure at a current stop (of a TransitVehicle)
313         * @author droeder
314         *
315         */
316        private static class LoadPerDeparturePerStop{
317                private Double weight = null;
318                private Double length = null;
319                private String id;
320                
321                public LoadPerDeparturePerStop(String id){
322                        this.id = id;
323                }
324                
325        
326
327                void reset() {
328                        weight = 0.;
329                        length = 0.;
330                }
331                
332                void sumUp(Double weight, Double length) throws NullPointerException{
333                        if(weight == null || length == null){
334                                throw new NullPointerException("weight: " + weight + "\t length" + length );
335                        }
336                        this.weight += weight;
337                        this.length += length;
338                }
339                 
340                void subtract(Double weight, Double length) throws NullPointerException{
341                        if(weight == null || length == null){
342                                throw new NullPointerException("weight: " + weight + "\t length" + length );
343                        }
344                        this.weight -= weight;
345                        this.length -= length;
346//                      if(this.weight < 0 || this.length < 0){
347//                              // sometimes this number is below zero (e.g. 10E-14), due roinding error
348//                              log.warn("weight and length must not below zero. id: " + id + "\t weight:" + this.weight + "\t length:" + this.length);
349//                      }
350                }
351                
352                @Override
353                public String toString(){
354                        return new String(id + "\t" + length + "\t" + weight);
355                }
356                
357        }
