020package org.matsim.contrib.wagonSim.pt.router;
021
022import org.apache.log4j.Logger;
023import org.matsim.contrib.wagonSim.mobsim.qsim.framework.listeners.WagonSimVehicleLoadListener;
024import org.matsim.core.utils.collections.Tuple;
025import org.matsim.core.utils.misc.Counter;
026import org.matsim.pt.router.*;
027import org.matsim.pt.router.TransitRouterNetwork.TransitRouterNetworkLink;
028import org.matsim.pt.router.TransitRouterNetwork.TransitRouterNetworkNode;
029import org.matsim.pt.transitSchedule.api.*;
030import org.matsim.utils.objectattributes.ObjectAttributes;
031
032import javax.inject.Provider;
033import java.util.LinkedList;
034import java.util.List;
035
036/**
037 * @author droeder
038 *
039 */
040public final class WagonSimRouterFactoryImpl implements Provider<TransitRouter> {
041
042        private static final Logger log = Logger
043                        .getLogger(WagonSimRouterFactoryImpl.class);
044        private WagonSimVehicleLoadListener vehLoad;
045        private TransitRouterConfig config;
046        private TransitSchedule schedule;
047        private TransitRouterNetwork routerNetwork;
048        private PreparedTransitSchedule preparedTransitSchedule;
049        private ObjectAttributes locomotiveAttribs;
050        private ObjectAttributes wagonAttribs;
051
052        public WagonSimRouterFactoryImpl(WagonSimVehicleLoadListener vehLoad, 
053                                TransitSchedule schedule, 
054                                TransitRouterConfig config,
055                                ObjectAttributes wagonAttribs,
056                                ObjectAttributes locomotiveAttribs) {
057                this.vehLoad = vehLoad;
058                this.schedule = schedule;
059                this.config = config;
060                this.routerNetwork = createRouterNetwork(this.schedule, this.config.getBeelineWalkConnectionDistance());
061                this.preparedTransitSchedule = new PreparedTransitSchedule(schedule);
062                this.locomotiveAttribs = locomotiveAttribs;
063                this.wagonAttribs = wagonAttribs;
064        }
065
066        @Override
067        public TransitRouter get() {
068                WagonSimRouterNetworkTravelDistutilityAndTravelTime tt = 
069                                new WagonSimRouterNetworkTravelDistutilityAndTravelTime(
070                                                config, 
071                                                preparedTransitSchedule, 
072                                                vehLoad.getLoadOfLastIter(),
073                                                this.locomotiveAttribs,
074                                                this.wagonAttribs);
075                return new TransitRouterImpl(config, preparedTransitSchedule, routerNetwork, tt, tt);
076        }
077        
078        /**
079         * This is c&p from {@link TransitRouterNetwork#createFromSchedule(TransitSchedule, double)}. However,
080         * this method creates a {@link TransitRouterNetworkLink} per Departure (not only per route).
081         * 
082         * @param schedule
083         * @param maxBeelineWalkConnectionDistance
084         * @return
085         */
086        public static TransitRouterNetwork createRouterNetwork(final TransitSchedule schedule, final double maxBeelineWalkConnectionDistance) {
087                log.info("start creating transit network");
088                final TransitRouterNetwork network = new TransitRouterNetwork();
089                final Counter linkCounter = new Counter(" link #");
090                final Counter nodeCounter = new Counter(" node #");
091                // build nodes and links connecting the nodes according to the transit routes
092                for (TransitLine line : schedule.getTransitLines().values()) {
093                        for (TransitRoute route : line.getRoutes().values()) {
094                                for(Departure d: route.getDepartures().values()){
095                                        TransitRouterNetworkNode prevNode = null;
096                                        TransitRoute r = schedule.getFactory().createTransitRoute(route.getId(), route.getRoute(), route.getStops(), route.getTransportMode());
097                                        r.addDeparture(d);
098                                        for (TransitRouteStop stop : route.getStops()) {
099                                                TransitRouterNetworkNode node = network.createNode(stop, r, line);
100                                                nodeCounter.incCounter();
101                                                if (prevNode != null) {
102                                                        network.createLink(prevNode, node, r, line);
103                                                        linkCounter.incCounter();
104                                                }
105                                                prevNode = node;
106                                        }
107                                }
108                        }
109                }
110                network.finishInit(); // not nice to call "finishInit" here before we added all links...
111                // in my view, it would be possible to completely do without finishInit: do the
112                // additions to the central data structures as items come in, not near the end.  I would
113                // prefer that because nobody could forget the "finishInit".  kai, apr'10
114                // well, not really. finishInit creates the quadtree, for this, the extent must be known,
115                // which is not at the very start, so the quadtree data structure cannot be updated as
116                // links come in. mrieser, dec'10
117                log.info("add transfer links");
118
119                List<Tuple<TransitRouterNetworkNode, TransitRouterNetworkNode>> toBeAdded = new LinkedList<Tuple<TransitRouterNetworkNode, TransitRouterNetworkNode>>();
120                // connect all stops with walking links if they're located less than beelineWalkConnectionDistance from each other
121                for (TransitRouterNetworkNode node : network.getNodes().values()) {
122                        if (node.getInLinks().size() > 0) { // only add links from this node to other nodes if agents actually can arrive here
123                                for (TransitRouterNetworkNode node2 : network.getNearestNodes(node.stop.getStopFacility().getCoord(), maxBeelineWalkConnectionDistance)) {
124                                        if ((node != node2) && (node2.getOutLinks().size() > 0)) { // only add links to other nodes when agents can depart there
125                                                if ((node.line != node2.line) || (node.stop.getStopFacility() != node2.stop.getStopFacility())) {
126                                                        // do not yet add them to the network, as this would change in/out-links
127                                                        toBeAdded.add(new Tuple<TransitRouterNetworkNode, TransitRouterNetworkNode>(node, node2));
128                                                }
129                                        }
130                                }
131                        }
132                }
133                log.info(toBeAdded.size() + " transfer links to be added.");
134                for (Tuple<TransitRouterNetworkNode, TransitRouterNetworkNode> tuple : toBeAdded) {
135                        network.createLink(tuple.getFirst(), tuple.getSecond(), null, null);
136                        linkCounter.incCounter();
137                }
138
139                log.info("transit router network statistics:");
140                log.info(" # nodes: " + network.getNodes().size());
141                log.info(" # links total:     " + network.getLinks().size());
142                log.info(" # transfer links:  " + toBeAdded.size());
143
144                return network;
145        }
146}
