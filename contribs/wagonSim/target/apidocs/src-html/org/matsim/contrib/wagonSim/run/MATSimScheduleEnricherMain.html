049/**
050 * @author balmermi
051 *
052 */
053public class MATSimScheduleEnricherMain {
054
055        //////////////////////////////////////////////////////////////////////
056        // variables
057        //////////////////////////////////////////////////////////////////////
058        
059        private static final Logger log = Logger.getLogger(MATSimScheduleEnricherMain.class);
060        
061        private final Scenario scenario;
062        private final ObjectAttributes vehicleAttributes;
063        
064        //////////////////////////////////////////////////////////////////////
065        // constructors
066        //////////////////////////////////////////////////////////////////////
067        
068        public MATSimScheduleEnricherMain(String networkFile, String transitScheduleFile, String transitVehiclesFile, String transitVehicleAttributesFile) {
069                Config config = Utils.getDefaultWagonSimConfig();
070                scenario = ScenarioUtils.createScenario(config);
071                new NetworkReaderMatsimV1(scenario.getNetwork()).readFile(networkFile);
072                new TransitScheduleReader(scenario).readFile(transitScheduleFile);
073                new VehicleReaderV1(((MutableScenario)scenario).getTransitVehicles()).readFile(transitVehiclesFile);
074                vehicleAttributes = new ObjectAttributes();
075                new ObjectAttributesXmlReader(vehicleAttributes).readFile(transitVehicleAttributesFile);
076        }
077
078        //////////////////////////////////////////////////////////////////////
079        
080        public final void enrich(String shuntingTableFile, double minDwellTime) throws IOException {
081                ShuntingTableToMATSimScheduleEnricher enricher = new ShuntingTableToMATSimScheduleEnricher(scenario,vehicleAttributes);
082
083                Map<Id<TransitLine>, Map<Id<Node>, Boolean>> shuntingTable = Utils.parseShuntingTable(shuntingTableFile);
084                enricher.enrich(shuntingTable,minDwellTime);
085        }
086
087        //////////////////////////////////////////////////////////////////////
088
089        public final Scenario getScenario() {
090                return this.scenario;
091        }
092        
093        //////////////////////////////////////////////////////////////////////
094        
095        public final ObjectAttributes getVehicleAttributes() {
096                return this.vehicleAttributes;
097        }
098        
099        //////////////////////////////////////////////////////////////////////
100        // main
101        //////////////////////////////////////////////////////////////////////
102        
103        /**
104         * @param args
105         * @throws IOException 
106         */
107        public static void main(String[] args) throws IOException {
108
109//              args = new String[] {
110//                              "D:/Users/balmermi/Documents/eclipse/output/sbb/schedulePerformance/network.ott.performance.xml.gz",
111//                              "D:/Users/balmermi/Documents/eclipse/output/sbb/schedulePerformance/transitSchedule.ott.performance.xml.gz",
112//                              "D:/Users/balmermi/Documents/eclipse/output/sbb/schedulePerformance/transitVehicles.ott.performance.xml.gz",
113//                              "D:/Users/balmermi/Documents/eclipse/output/sbb/schedulePerformance/transitVehicleAttributes.ott.performance.xml.gz",
114//                              "D:/Users/balmermi/Documents/eclipse/output/sbb/schedulePerformance/shuntingTable.ott.performance.txt",
115//                              "D:/Users/balmermi/Documents/eclipse/output/sbb/schedulePerformance/shuntingTimes.ott.performance.txt",
116//                              "999999",
117//                              "D:/Users/balmermi/Documents/eclipse/output/sbb/schedulePerformanceEnriched",
118//              };
119                
120                if (args.length != 8) {
121                        log.error(MATSimScheduleEnricherMain.class.getCanonicalName()+" networkFile transitScheduleFile transitVehiclesFile transitVehicleAttributesFile shuntingTableFile shuntingTimesFile minDwellTime outputBase");
122                        System.exit(-1);
123                }
124                
125                String networkFile = args[0];
126                String transitScheduleFile = args[1];
127                String transitVehiclesFile = args[2];
128                String transitVehicleAttributesFile = args[3];
129                String shuntingTableFile = args[4];
130                String shuntingTimesFile = args[5];
131                Double minDwellTime = Double.parseDouble(args[6]);
132                String outputBase = args[7];
133                
134                log.info("Main: "+MATSimScheduleEnricherMain.class.getCanonicalName());
135                log.info("networkFile: "+networkFile);
136                log.info("transitScheduleFile: "+transitScheduleFile);
137                log.info("transitVehiclesFile: "+transitVehiclesFile);
138                log.info("transitVehicleAttributesFile: "+transitVehicleAttributesFile);
139                log.info("shuntingTableFile: "+shuntingTableFile);
140                log.info("shuntingTimesFile: "+shuntingTimesFile);
141                log.info("outputBase: "+outputBase);
142                
143                MATSimScheduleEnricherMain enricher = new MATSimScheduleEnricherMain(networkFile, transitScheduleFile, transitVehiclesFile, transitVehicleAttributesFile);
144                enricher.enrich(shuntingTableFile, minDwellTime);
145                
146                Map<Id<TransitStopFacility>, Double> shuntingTimes = Utils.parseShuntingTimes(shuntingTimesFile);
147                
148                if (!Utils.prepareFolder(outputBase)) {
149                        throw new RuntimeException("Could not prepare output folder for one of the three reasons: (i) folder exists and is not empty, (ii) it's a path to an existing file or (iii) the folder could not be created. Bailing out.");
150                }
151                
152                new NetworkWriter(enricher.getScenario().getNetwork()).write(outputBase+"/network.enriched.xml.gz");
153                new TransitScheduleWriter(enricher.getScenario().getTransitSchedule()).writeFile(outputBase+"/transitSchedule.enriched.xml.gz");
154                Utils.writeShuntingTimes(enricher.getScenario(), shuntingTimes,outputBase+"/shuntingTimes.enriched.txt");
155                new VehicleWriterV1(((MutableScenario)enricher.getScenario()).getTransitVehicles()).writeFile(outputBase+"/transitVehicles.enriched.xml.gz");
156                Utils.writeShuntingTable(enricher.getScenario().getTransitSchedule(),outputBase+"/shuntingTable.enriched.txt");
157                new ObjectAttributesXmlWriter(enricher.getVehicleAttributes()).writeFile(outputBase+"/transitVehicleAttributes.enriched.xml.gz");
158        }
