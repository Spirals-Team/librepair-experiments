077        public final void parseRbFile(String rbFile) throws IOException {
078                log.info("--- START parsing rbFile ---");
079                BufferedReader br = IOUtils.getBufferedReader(rbFile);
080                int currRow = 0;
081                String curr_line;
082
083                // read header an build lookup
084                curr_line = br.readLine(); currRow++;
085                String [] header = curr_line.split(";");
086                Map<String, Integer> lookup = new LinkedHashMap<String,Integer>(header.length);
087                for (int i=0; i<header.length; i++) { lookup.put(header[i].trim(),i); }
088
089                while ((curr_line = br.readLine()) != null) {
090                        currRow++;
091                        String [] row = curr_line.split(";");
092
093                        String name = Utils.removeSurroundingQuotes(row[lookup.get(RB_VERSCHUBKNOTEN)].trim())+RB_NODE_POSTFIX;
094                        int maxTrainCreation = Integer.parseInt(Utils.removeSurroundingQuotes(row[lookup.get(RB_MAXZUGBILDUNGEN)].trim()));
095                        int maxWagonShuntings = Integer.parseInt(Utils.removeSurroundingQuotes(row[lookup.get(RB_MAXWAGENUMSTELLUNGEN)].trim()));
096                        int shuntingTime = Integer.parseInt(Utils.removeSurroundingQuotes(row[lookup.get(RB_WAGENUMSTELLZEIT)].trim()));
097                        
098                        if (dataContainer.productionNodes.containsKey(Id.create(name, Node.class))) {
099                                throw new RuntimeException("row "+currRow+": Production node '"+name+"' already exists. Bailing out.");
100                        }
101                        RbNode rbNode = new RbNode(name);
102                        dataContainer.productionNodes.put(rbNode.id,rbNode);
103                        rbNode.maxTrainCreation = maxTrainCreation;
104                        rbNode.maxWagonShuntings = maxWagonShuntings;
105                        rbNode.shuntingTime = shuntingTime;
106                }
107//              dataContainer.printProductionNodes();
108                log.info("--- END   parsing rbFile ---");
109        }

114        private static final String BEZEICHNUNG = "Bezeichnung";

119        public final void parseDeliveryAndTimeVariationFiles(String deliveryTypeFile, String timeVariationFile) throws IOException {
120                log.info("--- START parsing deliveryTypeFile ---");
121                BufferedReader br = IOUtils.getBufferedReader(deliveryTypeFile);
122                int currRow = 0;
123                String curr_line;
124
125                // read header an build lookup
126                curr_line = br.readLine(); currRow++;
127                String [] header = curr_line.split(";");
128                Map<String, Integer> lookup = new LinkedHashMap<String,Integer>(header.length);
129                for (int i=0; i<header.length; i++) { lookup.put(header[i].trim(),i); }
130
131                while ((curr_line = br.readLine()) != null) {
132                        currRow++;
133                        String [] row = curr_line.split(";");
134
135                        int deliveryTypeId = Integer.parseInt(Utils.removeSurroundingQuotes(row[lookup.get(ID_VERSANDTYP)].trim()));
136                        String desc = Utils.removeSurroundingQuotes(row[lookup.get(BEZEICHNUNG)].trim());
137                        
138                        if (dataContainer.rcpDeliveryTypes.containsKey(deliveryTypeId)) { throw new RuntimeException("row "+currRow+": delivery type '"+deliveryTypeId+"' already exists. Bailing out."); }
139                        RcpDeliveryType rcpDeliveryType = new RcpDeliveryType(deliveryTypeId,desc);
140                        dataContainer.rcpDeliveryTypes.put(rcpDeliveryType.id,rcpDeliveryType);
141                }
142//              dataContainer.printRcpDeliveryTypes();
143                log.info("--- END   parsing deliveryTypeFile ---");
144                
145                log.info("--- START parsing timeVariationFile ---");
146                br = IOUtils.getBufferedReader(timeVariationFile);
147                currRow = 0;
148                curr_line =null;
149
150                // read header an build lookup
151                curr_line = br.readLine(); currRow++;
152                header = curr_line.split(";");
153                lookup = new LinkedHashMap<String,Integer>(header.length);
154                for (int i=0; i<header.length; i++) { lookup.put(header[i].trim(),i); }
155
156                // parse rows and store nodes
157                while ((curr_line = br.readLine()) != null) {
158                        currRow++;
159                        String [] row = curr_line.split(";");
160
161                        String deliveryTypeId = Utils.removeSurroundingQuotes(row[lookup.get(VERSANDTYP)].trim());
162                        int hour = Integer.parseInt(Utils.removeSurroundingQuotes(row[lookup.get(STUNDE)].trim()));
163                        double fraction = Double.parseDouble(Utils.removeSurroundingQuotes(row[lookup.get(ANTEIL)].trim()));
164
165                        RcpDeliveryType rcpDeliveryType = dataContainer.rcpDeliveryTypes.get(deliveryTypeId);
166                        if (!dataContainer.rcpDeliveryTypes.containsKey(deliveryTypeId)) { throw new RuntimeException("row "+currRow+": delivery type '"+deliveryTypeId+"' does not exist. Bailing out."); }
167                        rcpDeliveryType.hourlyDistribution[hour] = fraction;
168                }
169//              dataContainer.printRcpDeliveryTypes();
170                log.info("--- END   parsing timeVariationFile ---");
171        }

176        private static final String RCP_VERSCHUBKNOTEN = "verschubknoten";
177        private static final String RCP_VERSCHUBKNOTEN_EMPFANG = "verschubknoten_empfang";
178        private static final String RCP_GRENZE = "grenze";
179        private static final String RCP_VERSANDTYP = "versandtyp";
180        
181        public final void parseRcpFile(String rcpFile) throws IOException {
182                log.info("--- START parsing rcpFile ---");
183                BufferedReader br = IOUtils.getBufferedReader(rcpFile);
184                int currRow = 0;
185                String curr_line;
186
187                // read header an build lookup
188                curr_line = br.readLine(); currRow++;
189                String [] header = curr_line.split(";");
190                Map<String, Integer> lookup = new LinkedHashMap<String,Integer>(header.length);
191                for (int i=0; i<header.length; i++) { lookup.put(header[i].trim(),i); }
192
193                while ((curr_line = br.readLine()) != null) {
194                        currRow++;
195                        String [] row = curr_line.split(";");
196
197                        String name = Utils.removeSurroundingQuotes(row[lookup.get(RCP_BEDIENKNOTEN)].trim())+RCP_NODE_POSTFIX;
198                        
199                        Id<Node> rbNodeId = Id.create(Utils.removeSurroundingQuotes(row[lookup.get(RCP_VERSCHUBKNOTEN)].trim())+RB_NODE_POSTFIX, Node.class);
200                        ProductionNode rbNode = dataContainer.productionNodes.get(rbNodeId);
201                        if (rbNode == null) { throw new RuntimeException("row "+currRow+": "+RCP_VERSCHUBKNOTEN+"="+rbNodeId+" does not exist. Bailing out."); }
202                        if (!(rbNode instanceof RbNode)) { throw new RuntimeException("row "+currRow+": "+RCP_VERSCHUBKNOTEN+"="+rbNodeId+" is not of type RB. Bailing out."); }
203                        
204                        Id<Node> receptionNodeId = Id.create(Utils.removeSurroundingQuotes(row[lookup.get(RCP_VERSCHUBKNOTEN_EMPFANG)].trim())+RB_NODE_POSTFIX, Node.class);
205                        ProductionNode receptionNode = dataContainer.productionNodes.get(receptionNodeId);
206                        if (receptionNode == null) { throw new RuntimeException("row "+currRow+": "+RCP_VERSCHUBKNOTEN_EMPFANG+"="+receptionNodeId+" does not exist. Bailing out."); }
207                        
208                        int border = Integer.parseInt(Utils.removeSurroundingQuotes(row[lookup.get(RCP_GRENZE)].trim()));
209                        boolean isBorder = false;
210                        if (border == 1) { isBorder = true; }
211                        
212                        String deliveryTypeId = Utils.removeSurroundingQuotes(row[lookup.get(RCP_VERSANDTYP)].trim());
213                        RcpDeliveryType rcpDeliveryType = dataContainer.rcpDeliveryTypes.get(deliveryTypeId);
214                        if (!dataContainer.rcpDeliveryTypes.containsKey(deliveryTypeId)) { throw new RuntimeException("row "+currRow+": delivery type '"+deliveryTypeId+"' does not exist. Bailing out."); }
215                        
216                        if (dataContainer.productionNodes.containsKey(Id.create(name, Node.class))) { throw new RuntimeException("row "+currRow+": Production node '"+name+"' already exists. Bailing out."); }
217                        RcpNode rcpNode = new RcpNode(name);
218                        dataContainer.productionNodes.put(rcpNode.id,rcpNode);
219                        rcpNode.parentNode = rbNode;
220                        rbNode.siblingNodes.add(rcpNode);
221                        rcpNode.parentReceptionNode = receptionNode;
222                        rcpNode.isBorder = isBorder;
223                        rcpNode.deliveryType = rcpDeliveryType;
224                }
225//              dataContainer.printProductionNodes();
226                log.info("--- END   parsing rcpFile ---");
227        }

279        private static final String ROUTE_VONKNOTEN = "VonKnoten";
280        private static final String ROUTE_NACHKNOTEN = "NachKnoten";
281        private static final String ROUTE_ZWISCHENKNOTENNR = "ZwischenKnotenNr";
282        private static final String ROUTE_ZWISCHENKNOTEN = "Zwischenknoten";
283        
284        public final void parseRouteFile(String routeFile) throws IOException {
285                log.info("--- START parsing routeFile ---");
286                BufferedReader br = IOUtils.getBufferedReader(routeFile);
287                int currRow = 0;
288                String curr_line;
289
290                // read header an build lookup
291                curr_line = br.readLine(); currRow++;
292                String [] header = curr_line.split(";");
293                Map<String, Integer> lookup = new LinkedHashMap<String,Integer>(header.length);
294                for (int i=0; i<header.length; i++) { lookup.put(header[i].trim(),i); }
295
296                while ((curr_line = br.readLine()) != null) {
297                        currRow++;
298                        String [] row = curr_line.split(";");
299
300                        String from = Utils.removeSurroundingQuotes(row[lookup.get(ROUTE_VONKNOTEN)].trim());
301                        String to = Utils.removeSurroundingQuotes(row[lookup.get(ROUTE_NACHKNOTEN)].trim());
302                        int index = Integer.parseInt(Utils.removeSurroundingQuotes(row[lookup.get(ROUTE_ZWISCHENKNOTENNR)].trim()));
303                        String via = Utils.removeSurroundingQuotes(row[lookup.get(ROUTE_ZWISCHENKNOTEN)].trim());
304
305                        Id<Node> fromId = Id.create(from+RB_NODE_POSTFIX, Node.class);
306                        ProductionNode fromNode = dataContainer.productionNodes.get(fromId);
307                        if (fromNode == null) { throw new RuntimeException("row "+currRow+": from node id="+from+" is not an RB node. Bailing out."); }
308
309                        Id<Node> toId = Id.create(to+RB_NODE_POSTFIX, Node.class);
310                        ProductionNode toNode = dataContainer.productionNodes.get(toId);
311                        if (toNode == null) { throw new RuntimeException("row "+currRow+": to node id="+to+" is not an RB node. Bailing out."); }
312
313                        Id<Node> viaId = Id.create(via+RB_NODE_POSTFIX, Node.class);
314                        ProductionNode viaNode = dataContainer.productionNodes.get(viaId);
315                        if (viaNode == null) { throw new RuntimeException("row "+currRow+": via node id="+via+" is not an RB node. Bailing out."); }
316
317                        Id<Connection> routeId = Id.create(fromNode.id.toString()+"---"+toNode.id.toString(), Connection.class);
318                        Connection route = dataContainer.connections.get(routeId);
319                        if (route == null) {
320                                if (index != 1) { throw new RuntimeException("row "+currRow+": file is not sorted according to the via node index. Bailing out."); }
321                                route = new Connection(routeId,fromNode,toNode);
322                                dataContainer.connections.put(route.id,route);
323                                route.viaNodes.add(viaNode);
324                        }
325                        else {
326                                if (index != route.viaNodes.size()+1) { throw new RuntimeException("row "+currRow+": file is not sorted according to the via node index. Bailing out."); }
327                                route.viaNodes.add(viaNode);
328                        }
329                }
330                for (Connection c : dataContainer.connections.values()) {
331                        if (!c.viaNodes.get(c.viaNodes.size()-1).id.equals(c.toNode.id)) { throw new RuntimeException("in route file it is expected that the last via node is identical to the toNode. This is not the case for route="+c.toString()); }
332                }
333                
334                dataContainer.printConnections();
335                log.info("--- END   parsing routeFile ---");
336        }
