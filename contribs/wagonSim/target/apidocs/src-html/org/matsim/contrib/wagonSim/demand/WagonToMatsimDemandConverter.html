050/**
051 * @author balmermi
052 * @since 2013-08-19
053 */
054public class WagonToMatsimDemandConverter {
055
056        //////////////////////////////////////////////////////////////////////
057        // variables
058        //////////////////////////////////////////////////////////////////////
059        
060        private final Scenario scenario;
061        private final ObjectAttributes wagonAttributes;
062        private final Map<String, Id<Node>> zoneToNodeMap;
063        private final ObjectAttributes transitVehicleAttributes;
064
065        private static final Logger log = Logger.getLogger(WagonToMatsimDemandConverter.class);
066
067        //////////////////////////////////////////////////////////////////////
068        // constructors
069        //////////////////////////////////////////////////////////////////////
070
071        public WagonToMatsimDemandConverter(Scenario scenario, ObjectAttributes wagonAttributes, ObjectAttributes transitVehicleAttributes, Map<String, Id<Node>> zoneToNodeMap) {
072                this.scenario = scenario;
073                this.wagonAttributes = wagonAttributes;
074                this.transitVehicleAttributes = transitVehicleAttributes;
075                this.zoneToNodeMap = zoneToNodeMap;
076        }
077
078        //////////////////////////////////////////////////////////////////////
079        // methods
080        //////////////////////////////////////////////////////////////////////
081        
082        private final <T> void printIdSet(Set<T> ids, String desc) {
083                System.out.println("--- START printing "+ids.size()+" "+desc+" ids ---");
084                for (T id : ids) { System.out.println(id.toString()); }
085                System.out.println("--- END   printing "+ids.size()+" "+desc+" ids ---");
086        }
087
088        //////////////////////////////////////////////////////////////////////
089        
090        public final void convert(WagonDataContainer dataContainer) {
091
092                Config config = scenario.getConfig();
093                Population population = scenario.getPopulation();
094                PopulationFactory factory = population.getFactory();
095
096//              WagonSimVehicleLoadListener listener = new WagonSimVehicleLoadListener(wagonAttributes);
097//              TransitRouterFactory routerFactory = new WagonSimRouterFactoryImpl(
098//                              listener,
099//                              scenario.getTransitSchedule(),
100//                              new TransitRouterConfig(config.planCalcScore(), config.plansCalcRoute(), config.transitRouter(), config.vspExperimental()),
101//                              wagonAttributes,
102//                              transitVehicleAttributes);
103//
104//              TransitRouter router = routerFactory.createTransitRouter();
105                
106                TransitRouterImpl router = (TransitRouterImpl) new TransitRouterImplFactory(
107                                scenario.getTransitSchedule(),
108                                new TransitRouterConfig(scenario.getConfig())).get();
109
110                
111                Set<String> fromZonesNotConnected = new HashSet<>();
112                Set<String> toZonesNotConnected = new HashSet<>();
113                Set<Id<Node>> fromNodesNotInPseudoNetwork = new HashSet<>();
114                Set<Id<Node>> toNodesNotInPseudoNetwork = new HashSet<>();
115                Set<Id<Node>> fromNodesInPseudoNetwork = new HashSet<>();
116                Set<Id<Node>> toNodesInPseudoNetwork = new HashSet<>();
117                Set<Id<Person>> wagonWithoutRoute = new HashSet<>();
118                int nofWagonsMissed = 0;
119                
120                for (Wagon wagon : dataContainer.wagons.values()) {
121                        Id<Node> fromNodeId = zoneToNodeMap.get(wagon.fromZoneId);
122                        Id<Node> toNodeId = zoneToNodeMap.get(wagon.toZoneId);
123
124                        if (fromNodeId == null) { fromZonesNotConnected.add(wagon.fromZoneId); nofWagonsMissed++; continue; }
125                        if (toNodeId == null) { toZonesNotConnected.add(wagon.toZoneId); nofWagonsMissed++; continue; }
126                        
127                        if (!scenario.getNetwork().getNodes().keySet().contains(fromNodeId)) { fromNodesNotInPseudoNetwork.add(fromNodeId); nofWagonsMissed++; continue; }
128                        if (!scenario.getNetwork().getNodes().keySet().contains(toNodeId)) { toNodesNotInPseudoNetwork.add(toNodeId); nofWagonsMissed++; continue; }
129
130                        fromNodesInPseudoNetwork.add(fromNodeId);
131                        toNodesInPseudoNetwork.add(toNodeId);
132                        
133                        Person person = factory.createPerson(Id.create(wagon.id, Person.class));
134                        Plan plan = factory.createPlan();
135                        person.addPlan(plan);
136                        Activity origin = factory.createActivityFromCoord(WagonSimConstants.ORIGIN,scenario.getNetwork().getNodes().get(fromNodeId).getCoord());
137                        origin.setEndTime(wagon.depTime);
138                        plan.addActivity(origin);
139                        Leg leg = factory.createLeg(TransportMode.pt);
140                        plan.addLeg(leg);
141                        Activity destination = factory.createActivityFromCoord(WagonSimConstants.DESTINATION,scenario.getNetwork().getNodes().get(toNodeId).getCoord());
142                        plan.addActivity(destination);
143                        final Person person1 = person;
144
145                        if (router.calcRoute( new FakeFacility(origin.getCoord()), new FakeFacility(destination.getCoord()), wagon.depTime, person1 ) != null) {
146                                population.addPerson(person);
147                                wagonAttributes.putAttribute(person.getId().toString(),WagonSimConstants.WAGON_GROSS_WEIGHT,wagon.weight+wagon.weightLoad);
148                                wagonAttributes.putAttribute(person.getId().toString(),WagonSimConstants.WAGON_LENGTH,wagon.length);
149                        }
150                        else { wagonWithoutRoute.add(person.getId()); }
151                }
152                
153                printIdSet(fromZonesNotConnected,"not connected from-zone");
154                printIdSet(toZonesNotConnected,"not connected to-zone");
155                Set<String> zonesNotConnected = new HashSet<>(fromZonesNotConnected); zonesNotConnected.addAll(toZonesNotConnected); printIdSet(zonesNotConnected,"not connected zone");
156                printIdSet(fromNodesNotInPseudoNetwork,"not defined from-node");
157                printIdSet(toNodesNotInPseudoNetwork,"not defined to-node");
158                Set<Id<Node>> nodesNotInPseudoNetwork = new HashSet<>(fromNodesNotInPseudoNetwork); nodesNotInPseudoNetwork.addAll(toNodesNotInPseudoNetwork); printIdSet(nodesNotInPseudoNetwork,"not defined node");
159                printIdSet(fromNodesInPseudoNetwork,"demand origin node");
160                printIdSet(toNodesInPseudoNetwork,"demand destination node");
161                Set<Id<Node>> nodesInPseudoNetwork = new HashSet<>(fromNodesInPseudoNetwork); nodesInPseudoNetwork.addAll(toNodesInPseudoNetwork); printIdSet(nodesInPseudoNetwork,"demand origin or destination node");
162                printIdSet(wagonWithoutRoute,"no route wagon");
163                
164                log.info("number of wagons in the container: "+dataContainer.wagons.size());
165                log.info("number of agents (wagons) created: "+population.getPersons().size());
166                log.info("number of agents (wagons) missed: "+nofWagonsMissed+" ("+wagonWithoutRoute.size()+" of them do not find a route)");
167                log.info("number of from-zones not mapped to a node: "+fromZonesNotConnected.size());
168                log.info("number of to-zones not mapped to a node: "+toZonesNotConnected.size());
169                log.info("number of zones not mapped to a node: "+zonesNotConnected.size());
170                log.info("number of from-Nodes not defined in the schedule (OTT Data): "+fromNodesNotInPseudoNetwork.size());
171                log.info("number of to-Nodes not defined in the schedule (OTT Data): "+toNodesNotInPseudoNetwork.size());
172                log.info("number of nodes not defined in the schedule (OTT Data): "+nodesNotInPseudoNetwork.size());
173                log.info("number of Nodes as demand origin: "+fromNodesInPseudoNetwork.size());
174                log.info("number of Nodes as demand destination: "+fromNodesInPseudoNetwork.size());
175                log.info("number of Nodes as demand origin or destination: "+nodesInPseudoNetwork.size());
176        }
