023package org.matsim.contrib.wagonSim.run;
024
025import org.apache.log4j.Logger;
026import org.matsim.api.core.v01.Scenario;
027import org.matsim.contrib.wagonSim.Utils;
028import org.matsim.contrib.wagonSim.network.NEMOInfraDataContainer;
029import org.matsim.contrib.wagonSim.network.NEMOInfraParser;
030import org.matsim.contrib.wagonSim.network.NEMOInfraToMATSimNetworkConverter;
031import org.matsim.core.config.ConfigUtils;
032import org.matsim.core.network.io.NetworkWriter;
033import org.matsim.core.scenario.ScenarioUtils;
034import org.matsim.utils.objectattributes.ObjectAttributes;
035import org.matsim.utils.objectattributes.ObjectAttributesXmlWriter;
036
037/**
038 * @author balmermi
039 *
040 */
041public class NEMOInfraToMATSimNetworkConverterMain {
042
043        //////////////////////////////////////////////////////////////////////
044        // variables
045        //////////////////////////////////////////////////////////////////////
046        
047        private static final Logger log = Logger.getLogger(NEMOInfraToMATSimNetworkConverterMain.class);
048        
049        private final Scenario scenario = ScenarioUtils.createScenario(ConfigUtils.createConfig());
050        private final ObjectAttributes nodeAttributes = new ObjectAttributes();
051        private final ObjectAttributes linkAttributes = new ObjectAttributes();
052        
053        //////////////////////////////////////////////////////////////////////
054        // constructors
055        //////////////////////////////////////////////////////////////////////
056        
057        public NEMOInfraToMATSimNetworkConverterMain() {
058        }
059
060        //////////////////////////////////////////////////////////////////////
061        // methods
062        //////////////////////////////////////////////////////////////////////
063        
064        public final void convertFromFile(String nemoInfraXmlFile) {
065                NEMOInfraDataContainer dataContainer = new NEMOInfraDataContainer();
066                new NEMOInfraParser(dataContainer).readFile(nemoInfraXmlFile);
067                NEMOInfraToMATSimNetworkConverter converter = new NEMOInfraToMATSimNetworkConverter(scenario.getNetwork(),nodeAttributes,linkAttributes);
068                converter.convert(dataContainer);
069                converter.makeNetworkBiDirectional();
070                log.info("network contains "+scenario.getNetwork().getNodes().size()+" nodes and "+scenario.getNetwork().getLinks().size()+" links.");
071                if (converter.validateNetwork()) { log.info("network is valid."); }
072                else { log.warn("network is not yet valid."); }
073        }
074        
075        //////////////////////////////////////////////////////////////////////
076
077        public final Scenario getScenario() {
078                return this.scenario;
079        }
080        
081        //////////////////////////////////////////////////////////////////////
082        
083        public final ObjectAttributes getNodeAttributes() {
084                return this.nodeAttributes;
085        }
086        
087        //////////////////////////////////////////////////////////////////////
088        
089        public final ObjectAttributes getLinkAttributes() {
090                return this.linkAttributes;
091        }
092        
093        //////////////////////////////////////////////////////////////////////
094        // main
095        //////////////////////////////////////////////////////////////////////
096        
097        /**
098         * @param args
099         */
100        public static void main(String[] args) {
101
102//              args = new String[] {
103//                              "S:/raw/europe/ch/ch/sbb/0002/20130524_Daten_Infrastruktur/._infra.xml",
104//                              "D:/Users/balmermi/Documents/eclipse/output/sbb/networkNemoInfra",
105//              };
106                
107                if (args.length != 2) {
108                        log.error(NEMOInfraToMATSimNetworkConverterMain.class.getCanonicalName()+" nemoInfraXmlFile outputBase");
109                        System.exit(-1);
110                }
111                
112                String nemoInfraXmlFile = args[0];
113                String outputBase = args[1];
114                
115                log.info("Main: "+NEMOInfraToMATSimNetworkConverterMain.class.getCanonicalName());
116                log.info("NemoInfraXmlFile: "+nemoInfraXmlFile);
117                log.info("outputBase: "+outputBase);
118                
119                NEMOInfraToMATSimNetworkConverterMain converter = new NEMOInfraToMATSimNetworkConverterMain();
120                converter.convertFromFile(nemoInfraXmlFile);
121                
122                if (!Utils.prepareFolder(outputBase)) {
123                        throw new RuntimeException("Could not prepare output folder for one of the three reasons: (i) folder exists and is not empty, (ii) it's a path to an existing file or (iii) the folder could not be created. Bailing out.");
124                }
125                
126                new NetworkWriter(converter.getScenario().getNetwork()).write(outputBase+"/network.infra.xml.gz");
127                new ObjectAttributesXmlWriter(converter.getNodeAttributes()).writeFile(outputBase+"/nodeAttributes.infra.xml.gz");
128                new ObjectAttributesXmlWriter(converter.getLinkAttributes()).writeFile(outputBase+"/linkAttributes.infra.xml.gz");
129        }
130}
