001package org.matsim.contrib.wagonSim.schedule.mapping;
002
003import org.matsim.api.core.v01.Id;
004import org.matsim.api.core.v01.network.Link;
005import org.matsim.api.core.v01.network.Node;
006
007/**
008 * @author mrieser / senozon
009 */
010public class AddLink implements NetworkEdit {
011
012        private final Id<Link> linkId;
013        private final Id<Node> fromNodeId;
014        private final Id<Node> toNodeId;
015        
016        public AddLink(final Id<Link> linkId, final Id<Node> fromNodeId, final Id<Node> toNodeId) {
017                this.linkId = linkId;
018                this.fromNodeId = fromNodeId;
019                this.toNodeId = toNodeId;
020        }
021        
022        public Id<Link> getLinkId() {
023                return linkId;
024        }
025        
026        public Id<Node> getFromNodeId() {
027                return fromNodeId;
028        }
029        
030        public Id<Node> getToNodeId() {
031                return toNodeId;
032        }
033
034}
