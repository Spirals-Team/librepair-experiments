058public final class WagonSimTripRouterFactoryImpl implements Provider<TripRouter> {
059
060        @SuppressWarnings("unused")
061        private static final Logger log = Logger
062                        .getLogger(WagonSimTripRouterFactoryImpl.class);
063//      private TripRouterFactoryImpl delegate;
064        private RoutingModule walkRouter;
065        private TransitSchedule transitSchedule;
066        private Provider<TransitRouter> routerFactory;
067        private Map<Id<TransitStopFacility>, Double> minShuntingTimes;
068        private Network network;
069
070        public WagonSimTripRouterFactoryImpl(
071                        final Scenario scenario,
072//                      final TravelDisutilityFactory disutilityFactory,
073//                      final TravelTime travelTime,
074//                      final LeastCostPathCalculatorFactory leastCostAlgoFactory,
075                        final Provider<TransitRouter> transitRouterFactory,
076                        Map<Id<TransitStopFacility>, Double> minShuntingTimes) {
077//              this.delegate = new TripRouterFactoryImpl(scenario, disutilityFactory, travelTime, leastCostAlgoFactory, transitRouterFactory);
078                this.routerFactory = transitRouterFactory;
079                this.transitSchedule = scenario.getTransitSchedule();
080                this.network = scenario.getNetwork();
081                this.walkRouter = createWalkRouter(scenario.getPopulation().getFactory(), scenario.getConfig().plansCalcRoute());
082                this.minShuntingTimes = minShuntingTimes;
083        }
084        
085        private RoutingModule createWalkRouter(PopulationFactory populationFactory, PlansCalcRouteConfigGroup routeConfigGroup){
086                return DefaultRoutingModules.createTeleportationRouter( TransportMode.transit_walk, populationFactory, 
087                                routeConfigGroup.getModeRoutingParams().get( TransportMode.walk ) );
088        }
089
090        @Override
091        public TripRouter get() {
092//              TripRouter tripRouter = this.delegate.get();
093                TripRouter tripRouter = new TripRouter();
094                tripRouter.setRoutingModule(TransportMode.pt, 
095                                new WagonSimRouterWrapper(routerFactory.get(), transitSchedule, network, walkRouter, minShuntingTimes));
096                return tripRouter;
097        }
098        
099        private static class WagonSimRouterWrapper implements RoutingModule{
100
101                private TransitRouterWrapper delegate;
102                private Map<Id<TransitStopFacility>, Double> minShuntingTimes;
103
104                public WagonSimRouterWrapper(
105                                final TransitRouter router,
106                                final TransitSchedule transitSchedule,
107                                final Network network,
108                                final RoutingModule walkRouter,
109                                Map<Id<TransitStopFacility>, Double> minShuntingTimes) {
110                        this.delegate = new TransitRouterWrapper(router, transitSchedule, network, walkRouter);
111                        this.minShuntingTimes = minShuntingTimes;
112                }
113                
114                @Override
115                public List<? extends PlanElement> calcRoute(Facility fromFacility,
116                                Facility toFacility, double departureTime, Person person) {
117                        List<? extends PlanElement> pe = this.delegate.calcRoute(fromFacility, toFacility, departureTime, person);
118                        return duplicateStageActivities(pe);
119                }
120
121                /**
122                 * @param pe
123                 * @return
124                 */
125                private List<? extends PlanElement> duplicateStageActivities(List<? extends PlanElement> oldElements) {
126                        List<PlanElement> pe = new ArrayList<PlanElement>();
127                        for(int i = 1; i < oldElements.size() ; i += 2){
128                                Leg l = (Leg) oldElements.get(i-1);
129                                Activity a = (Activity) oldElements.get(i);
130                                if(l.getMode().equals(TransportMode.transit_walk) && a.getType().equals(PtConstants.TRANSIT_ACTIVITY_TYPE)){
131                                        // the initial walk, do not change
132                                        pe.add(l);
133                                        pe.add(a);
134                                }else{
135                                        pe.add(l); // that's a pt leg
136                                        
137                                        // get the minimum shunting time of the stop facility
138                                        Id<TransitStopFacility> stopFacId = ((ExperimentalTransitRoute)l.getRoute()).getEgressStopId();
139                                        Double minShuntingTime = minShuntingTimes.get(stopFacId);
140                                        if (minShuntingTime == null) {  throw new RuntimeException("stationId="+stopFacId+" does not have a minimum shunting time defined. Bailing out."); } 
141                                        
142                                        if(i<oldElements.size()-2){
143                                                // handle all stages except the last one
144                                                pe.add(a); // insert an additional stage act
145                                                Leg ll = PopulationUtils.createLeg(TransportMode.transit_walk);
146                                                Route r = RouteUtils.createGenericRouteImpl(a.getLinkId(), a.getLinkId());
147                                                r.setTravelTime(minShuntingTime);
148                                                ll.setRoute(r);
149                                                ll.setTravelTime(minShuntingTime);
150                                                pe.add(ll); // insert an additional walk leg
151                                        }
152                                        pe.add(a); // that's the original stage act
153                                }
154                        }
155                        return pe;
156                }
157                @Override
158                public StageActivityTypes getStageActivityTypes() {
159                        return this.delegate.getStageActivityTypes();
160                }
161                
162        }
