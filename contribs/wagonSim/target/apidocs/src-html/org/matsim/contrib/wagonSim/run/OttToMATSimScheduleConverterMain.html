046/**
047 * @author balmermi
048 *
049 */
050public class OttToMATSimScheduleConverterMain {
051
052        //////////////////////////////////////////////////////////////////////
053        // variables
054        //////////////////////////////////////////////////////////////////////
055        
056        private static final Logger log = Logger.getLogger(OttToMATSimScheduleConverterMain.class);
057        
058        private final Scenario scenario;
059        private final ObjectAttributes vehicleAttributes = new ObjectAttributes();
060        
061        //////////////////////////////////////////////////////////////////////
062        // constructors
063        //////////////////////////////////////////////////////////////////////
064        
065        public OttToMATSimScheduleConverterMain() {
066                Config config = Utils.getDefaultWagonSimConfig();
067                scenario = ScenarioUtils.createScenario(config);
068        }
069
070        //////////////////////////////////////////////////////////////////////
071        // methods
072        //////////////////////////////////////////////////////////////////////
073        
074        public final void convertFromFiles(String ottFile, Network infraNetwork, String nodeMapFile, String trainTypesFile, boolean isPerformance) throws IOException {
075                Map<Id<Node>, Id<Node>> nodeMap = Utils.parseNodeMapFile(nodeMapFile);
076                log.info("node map file contains "+nodeMap.size()+" mappings.");
077                ObjectAttributes trainTypes = Utils.parseTrainTypesFile(trainTypesFile);
078                OTTDataContainer dataContainer = new OTTDataContainer();
079                new OTTScheduleParser(dataContainer).parse(ottFile, nodeMap);
080                
081                OTTDataToMATSimScheduleConverter converter = new OTTDataToMATSimScheduleConverter(scenario,vehicleAttributes);
082                converter.convert(dataContainer,infraNetwork,trainTypes,isPerformance);
083        }
084        
085        //////////////////////////////////////////////////////////////////////
086
087        public final Scenario getScenario() {
088                return this.scenario;
089        }
090        
091        //////////////////////////////////////////////////////////////////////
092        
093        public final ObjectAttributes getVehicleAttributes() {
094                return this.vehicleAttributes;
095        }
096        
097        //////////////////////////////////////////////////////////////////////
098        // main
099        //////////////////////////////////////////////////////////////////////
100        
101        /**
102         * @param args
103         * @throws IOException 
104         */
105        public static void main(String[] args) throws IOException {
106
107//              args = new String[] {
108//                              "S:/raw/europe/ch/ch/sbb/0002/20130715_Fahrplandaten/OTT_20120308.csv",
109//                              "S:/raw/europe/ch/ch/sbb/0002/20130715_Fahrplandaten/Traintypes.csv",
110//                              "S:/raw/europe/ch/ch/sbb/0002/20130524_Daten_Infrastruktur/._infra.xml",
111//                              "S:/raw/europe/ch/ch/sbb/0002/20130828_nodeMergeList/NodeMap.csv",
112//                              "true", // true := performance schedule; false := target schedule
113//                              "D:/Users/balmermi/Documents/eclipse/output/sbb/schedulePerformance",
114//              };
115                
116                if (args.length != 6) {
117                        log.error(OttToMATSimScheduleConverterMain.class.getCanonicalName()+" ottFile trainTypesFile nemoInfraXmlFile nodeMapFile isPerformance outputBase");
118                        System.exit(-1);
119                }
120                
121                String ottFile = args[0];
122                String trainTypesFile = args[1];
123                String nemoInfraXmlFile = args[2];
124                String nodeMapFile = args[3];
125                boolean isPerformance = Boolean.parseBoolean(args[4]);
126                String outputBase = args[5];
127                
128                log.info("Main: "+OttToMATSimScheduleConverterMain.class.getCanonicalName());
129                log.info("ottFile: "+ottFile);
130                log.info("trainTypesFile: "+trainTypesFile);
131                log.info("nemoInfraXmlFile: "+nemoInfraXmlFile);
132                log.info("nodeMapFile: "+nodeMapFile);
133                log.info("isPerformance: "+isPerformance);
134                log.info("outputBase: "+outputBase);
135
136                NEMOInfraToMATSimNetworkConverterMain networkConverter = new NEMOInfraToMATSimNetworkConverterMain();
137                networkConverter.convertFromFile(nemoInfraXmlFile);
138                
139                OttToMATSimScheduleConverterMain converter = new OttToMATSimScheduleConverterMain();
140                converter.convertFromFiles(ottFile,networkConverter.getScenario().getNetwork(),nodeMapFile,trainTypesFile,isPerformance);
141                
142                MATSimInfraOttNetworkMergerMain merger = new MATSimInfraOttNetworkMergerMain();
143                merger.mergeNetworks(networkConverter.getScenario().getNetwork(),converter.getScenario().getNetwork());
144                
145                if (!Utils.prepareFolder(outputBase)) {
146                        throw new RuntimeException("Could not prepare output folder for one of the three reasons: (i) folder exists and is not empty, (ii) it's a path to an existing file or (iii) the folder could not be created. Bailing out.");
147                }
148                
149                if (!Utils.prepareFolder(outputBase+"/merged")) {
150                        throw new RuntimeException("Could not prepare output folder for one of the three reasons: (i) folder exists and is not empty, (ii) it's a path to an existing file or (iii) the folder could not be created. Bailing out.");
151                }
152                
153                new NetworkWriter(merger.getMergedNetwork()).write(outputBase+"/merged/network.merged.xml.gz");
154                
155                if (isPerformance) {
156                        new NetworkWriter(converter.getScenario().getNetwork()).write(outputBase+"/network.ott.performance.xml.gz");
157                        new TransitScheduleWriter(converter.getScenario().getTransitSchedule()).writeFile(outputBase+"/transitSchedule.ott.performance.xml.gz");
158                        Utils.writeShuntingTimes(converter.getScenario(),null,outputBase+"/shuntingTimes.ott.performance.txt");
159                        new VehicleWriterV1(((MutableScenario)converter.getScenario()).getTransitVehicles()).writeFile(outputBase+"/transitVehicles.ott.performance.xml.gz");
160                        Utils.writeShuntingTable(converter.getScenario().getTransitSchedule(),outputBase+"/shuntingTable.ott.performance.txt");
161                        new ObjectAttributesXmlWriter(converter.getVehicleAttributes()).writeFile(outputBase+"/transitVehicleAttributes.ott.performance.xml.gz");
162                }
163                else {
164                        new NetworkWriter(converter.getScenario().getNetwork()).write(outputBase+"/network.ott.target.xml.gz");
165                        new TransitScheduleWriter(converter.getScenario().getTransitSchedule()).writeFile(outputBase+"/transitSchedule.ott.target.xml.gz");
166                        Utils.writeShuntingTimes(converter.getScenario(),null,outputBase+"/shuntingTimes.ott.target.txt");
167                        new VehicleWriterV1(((MutableScenario)converter.getScenario()).getTransitVehicles()).writeFile(outputBase+"/transitVehicles.ott.target.xml.gz");
168                        Utils.writeShuntingTable(converter.getScenario().getTransitSchedule(),outputBase+"/shuntingTable.ott.target.txt");
169                        new ObjectAttributesXmlWriter(converter.getVehicleAttributes()).writeFile(outputBase+"/transitVehicleAttributes.ott.target.xml.gz");
170                }
171        }
