048public class NEMOInfraToMATSimNetworkConverter {

050        //////////////////////////////////////////////////////////////////////
051        // variables
052        //////////////////////////////////////////////////////////////////////
053
054
055
056        private static final Set<String> defaultModes = new HashSet<String>();
057        private final Network network;
058        private final ObjectAttributes nodeAttributes;
059        private final ObjectAttributes linkAttributes;
060        
061        //////////////////////////////////////////////////////////////////////
062        // constructors
063        //////////////////////////////////////////////////////////////////////
064
065        public NEMOInfraToMATSimNetworkConverter(Network network, ObjectAttributes nodeAttributes, ObjectAttributes linkAttributes) {
066                this.network = network;
067                this.nodeAttributes = nodeAttributes;
068                this.linkAttributes = linkAttributes;
069                defaultModes.add("train");
070        }
071
072        //////////////////////////////////////////////////////////////////////
073        // methods
074        //////////////////////////////////////////////////////////////////////
075        
076        public static final Set<String> getDefaultModes() {
077                defaultModes.add(WagonSimConstants.DEFAULT_LINK_MODE);
078                return defaultModes;
079        }
080        
081        //////////////////////////////////////////////////////////////////////
082
083        private final void convertNodes(NEMOInfraDataContainer dataContainer) {
084                NetworkFactory factory = network.getFactory();
085                
086                for (NEMOInfraNode nemoNode : dataContainer.nodes.values()) {
087                        Node node = factory.createNode(nemoNode.id,nemoNode.coord);
088                        network.addNode(node);
089                        
090                        nodeAttributes.putAttribute(node.getId().toString(), WagonSimConstants.NODE_STATION,nemoNode.isStation);
091                        nodeAttributes.putAttribute(node.getId().toString(), WagonSimConstants.NODE_VALID,nemoNode.isValid);
092                        
093                        NEMOInfraCountry country = dataContainer.countries.get(nemoNode.countryId);
094                        nodeAttributes.putAttribute(node.getId().toString(), WagonSimConstants.NODE_COUNTRY_ID, country.id);
095                        nodeAttributes.putAttribute(node.getId().toString(), WagonSimConstants.NODE_NAME, country.name);
096                        nodeAttributes.putAttribute(node.getId().toString(), WagonSimConstants.NODE_HOME_COUNTRY, country.isHomeCountry);
097                        
098                        NEMOInfraNodeCluster nodeCluster = dataContainer.nodeClusters.get(nemoNode.clusterId);
099                        nodeAttributes.putAttribute(node.getId().toString(), WagonSimConstants.NODE_CLUSTER_ID, nodeCluster.id.toString());
100                }
101        }
102
103        //////////////////////////////////////////////////////////////////////
104        
105        private final void convertLinks(NEMOInfraDataContainer dataContainer) {
106                NetworkFactory factory = network.getFactory();
107                
108                Set<Id<Link>> nemoLinkIdsUsed = new HashSet<>();
109                Set<String> nemoTrackIdsUsed = new HashSet<>();
110                
111                for (NEMOInfraDirection nemoDir : dataContainer.directions.values()) {
112                        NEMOInfraTrack nemoTrack = dataContainer.tracks.get(nemoDir.trackId);
113                        nemoTrackIdsUsed.add(nemoTrack.id);
114                        NEMOInfraLink nemoLink = dataContainer.links.get(nemoTrack.linkId);
115                        nemoLinkIdsUsed.add(nemoLink.id);
116                        
117                        Node fromNode = network.getNodes().get(nemoLink.fromNodeId);
118                        Node toNode = network.getNodes().get(nemoLink.toNodeId);
119                        boolean dir = nemoDir.direction;
120                        
121                        Link link = null;
122                        if (dir) { link = factory.createLink(nemoDir.id,fromNode,toNode); }
123                        else { link = factory.createLink(nemoDir.id,toNode,fromNode); }
124
125                        network.addLink(link);
126                        
127                        link.setLength(nemoLink.length);
128                        link.setAllowedModes(defaultModes);
129                        link.setNumberOfLanes(1);
130                        link.setCapacity(WagonSimConstants.DEFAULT_CAPACITY);
131                        
132                        linkAttributes.putAttribute(link.getId().toString(), WagonSimConstants.LINK_SIMULATE, nemoLink.isSimuLink);
133                        linkAttributes.putAttribute(link.getId().toString(), WagonSimConstants.LINK_GLOBAL, nemoLink.isGlobal);
134                        linkAttributes.putAttribute(link.getId().toString(), WagonSimConstants.LINK_CLOSED, nemoLink.isClosed);
135                        linkAttributes.putAttribute(link.getId().toString(), WagonSimConstants.LINK_VALID, nemoLink.isValid);
136                        linkAttributes.putAttribute(link.getId().toString(), WagonSimConstants.LINK_MAXTRAINLENGTH, nemoLink.maxTrainLength);
137                        
138                        NEMOInfraLinkType linkType = dataContainer.linkTypes.get(nemoLink.typeId);
139                        linkAttributes.putAttribute(link.getId().toString(), WagonSimConstants.LINK_TYPE, linkType.id);
140                        linkAttributes.putAttribute(link.getId().toString(), WagonSimConstants.LINK_VFACTOR, linkType.vFactor);
141                        link.setFreespeed(linkType.velocity); // m/s
142
143                        NEMOInfraLinkOwner linkOwner = dataContainer.linkOwners.get(nemoLink.ownerId);
144                        linkAttributes.putAttribute(link.getId().toString(), WagonSimConstants.LINK_OWNERID, linkOwner.id);
145                        linkAttributes.putAttribute(link.getId().toString(), WagonSimConstants.LINK_OWNERNAME, linkOwner.owner);
146                }
147                
148                if (nemoTrackIdsUsed.size() != dataContainer.tracks.size()) {
149                        throw new RuntimeException("Inconsistent data: the amount of track refered by the directions (="+nemoTrackIdsUsed.size()+") is not equal to the given number of tracks (="+dataContainer.tracks.size()+"). Bailing out.");
150                }
151                if (nemoLinkIdsUsed.size() != dataContainer.links.size()) {
152                        throw new RuntimeException("Inconsistent data: the amount of links refered by the tracks (="+nemoLinkIdsUsed.size()+") is not equal to the given number of links (="+dataContainer.links.size()+"). Bailing out.");
153                }
154        }
155        
156        //////////////////////////////////////////////////////////////////////
157        
158        public final void makeNetworkBiDirectional() {
159                NetworkFactory factory = network.getFactory();
160
161                Set<Id<Link>> linkIds = new HashSet<>(network.getLinks().keySet());
162                for (Id<Link> linkId : linkIds) {
163                        Link link = network.getLinks().get(linkId);
164                        boolean hasOtherDirection = false;
165                        for (Link otherLink : link.getToNode().getOutLinks().values()) {
166                                if (otherLink.getToNode().getId().equals(link.getFromNode().getId())) {
167                                        hasOtherDirection = true;
168                                        break;
169                                }
170                        }
171                        if (!hasOtherDirection) {
172                                Link l = factory.createLink(Id.create(link.getId().toString()+"-r", Link.class), link.getToNode(), link.getFromNode());
173                                l.setLength(link.getLength());
174                                l.setFreespeed(link.getFreespeed());
175                                l.setCapacity(link.getCapacity());
176                                l.setNumberOfLanes(link.getNumberOfLanes());
177                                l.setAllowedModes(link.getAllowedModes());
178                                network.addLink(l);
179                        }
180                }
181        }

183        //////////////////////////////////////////////////////////////////////
184        
185        public final boolean validateNetwork() {
186                boolean isValid = true;
187                for (Link link : network.getLinks().values()) {
188                        for (Link otherLink : link.getFromNode().getOutLinks().values()) {
189                                // check for parallel links
190                                if (!otherLink.getId().equals(link.getId()) && otherLink.getToNode().getId().equals(link.getToNode().getId())) {
191                                        System.out.println(link.toString()+" <==> "+otherLink.toString());
192                                        isValid = false;
193                                }
194                        }
195                }
196                return isValid;
197        }
198
199        //////////////////////////////////////////////////////////////////////
200
201        public final void convert(NEMOInfraDataContainer dataContainer) {
202                convertNodes(dataContainer);
203                convertLinks(dataContainer);
204        }
