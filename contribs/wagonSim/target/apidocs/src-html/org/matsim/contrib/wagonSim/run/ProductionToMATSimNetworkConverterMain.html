053        private final ObjectAttributes nodeAttributes = new ObjectAttributes();
054        private final ObjectAttributes linkAttributes = new ObjectAttributes();

066        
067        public final void convertFromFiles(String rbFile, String rcpFile, String satFile, String deliveryTypeFile, String timeVariationFile, String routeFile, Network infraNetwork) throws IOException {
068                ProductionDataContainer dataContainer = new ProductionDataContainer();
069                ProductionParser productionParser = new ProductionParser(dataContainer);
070                productionParser.parseDeliveryAndTimeVariationFiles(deliveryTypeFile,timeVariationFile);
071                productionParser.parseRbFile(rbFile);
072                productionParser.parseRcpFile(rcpFile);
073                productionParser.parseSatFile(satFile);
074                productionParser.parseRouteFile(routeFile);
075                
076                ProductionToMATSimNetworkConverter converter = new ProductionToMATSimNetworkConverter(scenario.getNetwork(),nodeAttributes,linkAttributes);
077                converter.convert(dataContainer,infraNetwork);
078        }
079        
080        //////////////////////////////////////////////////////////////////////
081
082        public final Scenario getScenario() {
083                return this.scenario;
084        }
085        
086        //////////////////////////////////////////////////////////////////////
087        
088        public final ObjectAttributes getNodeAttributes() {
089                return this.nodeAttributes;
090        }
091        
092        //////////////////////////////////////////////////////////////////////
093        
094        public final ObjectAttributes getLinkAttributes() {
095                return this.linkAttributes;
096        }
