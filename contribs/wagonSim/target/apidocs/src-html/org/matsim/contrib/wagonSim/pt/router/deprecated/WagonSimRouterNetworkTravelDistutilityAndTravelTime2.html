088        @Override
089        public double getLinkTravelTime(Link link, double time, Person person, Vehicle vehicle) {
090                double tt = 0;
091                TransitRouterNetworkLink l = (TransitRouterNetworkLink) link;
092                if(l.getRoute() == null){
093                        // it's a transfer-link, time is fixed by definition
094                        tt = transitRouterConfig.getAdditionalTransferTime();
095                }else{
096                        double nextDeparture = getNextDeparture(l, time, person, vehicle);
097                        // calc the traveltime
098                        TransitRouteStop toStop = l.toNode.stop;
099                        double arrivalOffset = (toStop.getArrivalOffset() != Time.UNDEFINED_TIME) ? toStop.getArrivalOffset() : toStop.getDepartureOffset();
100                        tt = (nextDeparture - time) + (arrivalOffset - l.getFromNode().stop.getDepartureOffset());
101                }
102                return tt;
103        }

105        @Override
106        public double getLinkTravelDisutility(Link link, double time,
107                        Person person, Vehicle vehicle, CustomDataManager dataManager) {
108                double disutility = 0;
109                TransitRouterNetworkLink l = (TransitRouterNetworkLink) link;
110                if(l.getRoute() == null){
111                        // it is a transfer-link
112                        disutility = - (getLinkTravelTime(link, time, person, vehicle) * transitRouterConfig.getMarginalUtilityOfTravelTimePt_utl_s()) 
113                                        - transitRouterConfig.getUtilityOfLineSwitch_utl();
114                }else{
115                        // calcNextDeparture
116                        double nextDepartureTime = getNextDeparture(l, time, person, vehicle);
117                        TransitRouteStop fromStop = l.fromNode.stop;
118                        // calc the different components of the disutility
119                        double fromStopArrivalOffset = (fromStop.getArrivalOffset() != Time.UNDEFINED_TIME) ? fromStop.getArrivalOffset() : fromStop.getDepartureOffset();
120                        double vehArrivalTime = nextDepartureTime - (fromStop.getDepartureOffset() - fromStopArrivalOffset);
121                        double offVehWaitTime=0;
122                        if (time < vehArrivalTime) offVehWaitTime = vehArrivalTime-time;
123                        disutility = - (getLinkTravelTime(link,time, person, vehicle) - offVehWaitTime)      * transitRouterConfig.getMarginalUtilityOfTravelTimePt_utl_s() // time in the vehicle 
124                                        -offVehWaitTime   * transitRouterConfig.getMarginalUtilityOfWaitingPt_utl_s() // time outside the vehicle
125                                        -link.getLength() * transitRouterConfig.getMarginalUtilityOfTravelDistancePt_utl_m(); // travelled distance
126                        // check if we have to punish this departure
127                        Departure d = getEarliestPossibleDeparture(l, time, person, vehicle);
128                        if(punishDeparture(link, person, d)){
129                                if(d.getDepartureTime() + fromStop.getDepartureOffset() > time){
130                                        // punish only departures that happen ``today''
131                                        disutility -= ADDITIONAL_DISUTILITY;
132                                }
133                        }
134                }
135                return disutility;
136        }

152        private double getNextDeparture(TransitRouterNetworkLink l, double time,
153                        Person person, Vehicle vehicle){
154                // init the departure-time-cache
155                initDepartures(l);
156                // get the earliest departure
157                Departure d = getEarliestPossibleDeparture(l, time, person, vehicle);
158                double nextDeparture = d.getDepartureTime() + l.getFromNode().stop.getDepartureOffset();
159                // maybe we got a departure somewhere back in time, thus check the next mornings
160                while(nextDeparture < time){
161                        nextDeparture += 24.0*3600;
162                }
163                return nextDeparture;
164        }
165        /**
166         * 
167         * 
168         * @param link
169         * @param time
170         * @param person
171         * @param vehicle
172         * @return, a) a possible/unpunished departure, b) the first possible departure, 
173         * c) the very first departure (even if (dep + offset < time)) 
174         */
175        private Departure getEarliestPossibleDeparture(TransitRouterNetworkLink l, double time,
176                        Person person, Vehicle vehicle) {
177                String id = l.getLine().getId().toString() + "---" + l.getRoute().getId().toString();
178                double offset = l.getFromNode().getStop().getDepartureOffset();
179                double check = Double.MIN_VALUE;
180                Departure temp = null;
181                
182                for(Departure d : sortedDepartures.get(id)){
183                        // check consistency
184                        if(d.getDepartureTime() < check){
185                                log.warn("seems departures are not sorted...");
186                        }
187                        // we will not miss this departure
188                        if(d.getDepartureTime() + offset >= time){
189                                // and there is no additional Disutility. that is, it is the earliest departure we can and
190                                // we are allowed to catch
191                                if(!punishDeparture(l, person, d)){
192                                        return d;
193                                }
194                                if(temp == null){
195                                        // store the earliest possible departure
196                                        temp = d;
197                                }
198                        }
199                        check = d.getDepartureTime();
200                }
201                // check if we got a possible departure. If not, return the first of the day.
202                temp = (temp == null) ? sortedDepartures.get(id).get(0) : temp;
203                // we didn't find a reasonable, not punished departure. Thus, we use the earliest possible time for further search.
204                return temp;
205        }

207        /**
208         * in some way this provides a functionality like {@link PreparedTransitSchedule}, 
209         * but we have access to the departure-ids
210         * 
211         * @param l
212         */
213        private void initDepartures(TransitRouterNetworkLink l) {
214                String id = l.getLine().getId().toString() + "---" + l.getRoute().getId().toString();
215                if(!sortedDepartures.containsKey(id)){
216                        List<Departure> temp = new ArrayList<Departure>(l.getRoute().getDepartures().values());
217                        Collections.sort(temp, new MyComparator());
218                        this.sortedDepartures.put(id, temp);
219                }
220        }

280        private static class MyComparator implements Comparator{
281                @Override
282                public int compare(Object o1, Object o2) {
283                        if((!(o1 instanceof DepartureImpl)) || (!(o2 instanceof DepartureImpl))){
284                                return 0;
285                        }
286                        Departure d1 = (Departure) o1;
287                        Departure d2 = (Departure) o2;
288                        
289                        if(d1.getDepartureTime() < d2.getDepartureTime()) return -1;
290                        if(d1.getDepartureTime() == d2.getDepartureTime()) return 0;
291                        return 1;
292                }
293                
294        }
