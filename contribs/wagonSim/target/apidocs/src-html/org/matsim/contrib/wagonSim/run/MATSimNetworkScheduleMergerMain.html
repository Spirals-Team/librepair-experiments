054 */
055public class MATSimNetworkScheduleMergerMain {
056
057        //////////////////////////////////////////////////////////////////////
058        // variables
059        //////////////////////////////////////////////////////////////////////
060        
061        private static final Logger log = Logger.getLogger(MATSimNetworkScheduleMergerMain.class);
062        
063        private final Scenario scenario = ScenarioUtils.createScenario(ConfigUtils.createConfig());
064        
065        //////////////////////////////////////////////////////////////////////
066        // constructors
067        //////////////////////////////////////////////////////////////////////
068        
069        public MATSimNetworkScheduleMergerMain() {
070                scenario.getConfig().transit().setUseTransit(true);
071        }
072
073        //////////////////////////////////////////////////////////////////////
074        // methods
075        //////////////////////////////////////////////////////////////////////
076        
077        public final void mergeSchedule(String networkFilename, String transitScheduleFilename, String netEditsFilename) {
078                log.info("Load network " + networkFilename);
079                new MatsimNetworkReader(scenario.getNetwork()).readFile(networkFilename);
080                log.info("Load transit schedule " + transitScheduleFilename);
081                new TransitScheduleReader(scenario).readFile(transitScheduleFilename);
082
083                log.info("Load network edits " + netEditsFilename);
084                List<NetworkEdit> edits = new ArrayList<NetworkEdit>();
085                new NetworkEditsReader(edits).readFile(netEditsFilename);
086
087                log.info("apply edits to network");
088                new ApplyToNetwork(scenario.getNetwork()).applyEdits(edits);
089                log.info("apply edits to transit schedule");
090                new ApplyToTransitSchedule(scenario.getTransitSchedule()).applyEdits(edits);
091
092                log.info("change all links to mode pt");
093                Set<String> ptMode = new HashSet<String>();
094                ptMode.add(TransportMode.pt);
095                for (Link link : scenario.getNetwork().getLinks().values()) {
096                        link.setAllowedModes(ptMode);
097                }
098
099                log.info("fix link lengths…");
100                for (Link link : scenario.getNetwork().getLinks().values()) {
101                        double dist = CoordUtils.calcEuclideanDistance(link.getFromNode().getCoord(), link.getToNode().getCoord());
102                        if (link.getLength() < dist) {
103                                link.setLength(dist);
104                        }
105                }
106
107                log.info("set all transit lines to wait at stop for departure time…");
108                for (TransitLine line : scenario.getTransitSchedule().getTransitLines().values()) {
109                        for (TransitRoute route : line.getRoutes().values()) {
110                                for (TransitRouteStop stop : route.getStops()) {
111                                        stop.setAwaitDepartureTime(true);
112                                }
113                        }
114                }
115
116                log.info("set default link speeds and capacities…");
117                for (Link link : scenario.getNetwork().getLinks().values()) {
118                        link.setFreespeed(10.0);
119                        link.setCapacity(3000.0);
120                }
121
122                log.info("adapt network to schedule");
123                NetworkAdaption na = new NetworkAdaption(scenario);
124                na.adaptLinkFreespeeds();
125                na.adaptLinkCapacities();
126        }
127        
128        //////////////////////////////////////////////////////////////////////
129
130        public final Scenario getMergedScenario() {
131                return this.scenario;
132        }
