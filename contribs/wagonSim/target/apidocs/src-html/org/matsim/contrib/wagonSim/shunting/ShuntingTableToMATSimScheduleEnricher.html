023package org.matsim.contrib.wagonSim.shunting;
024
025import java.util.ArrayList;
026import java.util.Collection;
027import java.util.Collections;
028import java.util.List;
029import java.util.Map;
030import java.util.Set;
031
032import org.matsim.api.core.v01.Id;
033import org.matsim.api.core.v01.Scenario;
034import org.matsim.api.core.v01.TransportMode;
035import org.matsim.api.core.v01.network.Link;
036import org.matsim.api.core.v01.network.NetworkFactory;
037import org.matsim.api.core.v01.network.Node;
038import org.matsim.contrib.wagonSim.WagonSimConstants;
039import org.matsim.core.population.routes.NetworkRoute;
040import org.matsim.core.population.routes.RouteUtils;
041import org.matsim.pt.transitSchedule.api.Departure;
042import org.matsim.pt.transitSchedule.api.TransitLine;
043import org.matsim.pt.transitSchedule.api.TransitRoute;
044import org.matsim.pt.transitSchedule.api.TransitRouteStop;
045import org.matsim.utils.objectattributes.ObjectAttributes;
046
047/**
048 * @author balmermi
049 * @since 2013-07-08
050 */
051public class ShuntingTableToMATSimScheduleEnricher {
052
053        //////////////////////////////////////////////////////////////////////
054        // variables
055        //////////////////////////////////////////////////////////////////////
056
057        private final Scenario scenario;
058        private final ObjectAttributes vehicleAttributes;
059        
060        //////////////////////////////////////////////////////////////////////
061        // constructors
062        //////////////////////////////////////////////////////////////////////
063
064        public ShuntingTableToMATSimScheduleEnricher(Scenario scenario, ObjectAttributes vehicleAttributes) {
065                this.scenario = scenario;
066                this.vehicleAttributes = vehicleAttributes;
067        }
068
069        //////////////////////////////////////////////////////////////////////
070        // methods
071        //////////////////////////////////////////////////////////////////////
072        
073        public final void enrich(Map<Id<TransitLine>,Map<Id<Node>, Boolean>> shuntingTable, double minDwellTime) {
074                
075                NetworkFactory factory = scenario.getNetwork().getFactory();
076                Set<String> transitModes = Collections.singleton(TransportMode.pt);
077                
078                for (TransitLine transitLine : scenario.getTransitSchedule().getTransitLines().values()) {
079
080                        Map<Id<Node>, Boolean> shuntingInfoMap = shuntingTable.get(transitLine.getId());
081                        List<TransitRoute> newTransitRoutes = new ArrayList<TransitRoute>();
082                        
083                        for (TransitRoute transitRoute : transitLine.getRoutes().values()) {
084                                
085                                List<Id<Link>> routeLinkIds = new ArrayList<Id<Link>>();
086                                routeLinkIds.add(transitRoute.getRoute().getStartLinkId());
087                                routeLinkIds.addAll(transitRoute.getRoute().getLinkIds());
088                                routeLinkIds.add(transitRoute.getRoute().getEndLinkId());
089                                
090                                List<TransitRouteStop> transitRouteStops = new ArrayList<TransitRouteStop>();
091                                transitRouteStops.add(transitRoute.getStops().get(0));
092
093                                for (int i=1; i<transitRoute.getStops().size()-1; i++) {
094                                        TransitRouteStop stop = transitRoute.getStops().get(i);
095                                        
096                                        boolean replaceShuntingWithDwelling = false;
097                                        
098                                        double dwellTime = stop.getDepartureOffset()-stop.getArrivalOffset();
099                                        if (dwellTime < 1.0) { dwellTime = 1.0; }
100
101                                        if (shuntingInfoMap != null) {
102                                                Boolean shuntingAllowed = shuntingInfoMap.get(Id.create(stop.getStopFacility().getId(), Node.class));
103                                                if (shuntingAllowed != null) {
104                                                        if (!shuntingAllowed) {
105                                                                replaceShuntingWithDwelling = true;
106                                                        }
107                                                }
108                                                else if (dwellTime < minDwellTime) {
109                                                        replaceShuntingWithDwelling = true;
110                                                }
111                                        }
112                                        else if (dwellTime < minDwellTime) {
113                                                replaceShuntingWithDwelling = true;
114                                        }
115                                        
116                                        if (replaceShuntingWithDwelling) {
117                                                Link dwellLink = scenario.getNetwork().getLinks().get(Id.create(stop.getStopFacility().getId(), Link.class));
118                                                if (dwellLink == null) {
119                                                        Node node = scenario.getNetwork().getLinks().get(stop.getStopFacility().getLinkId()).getToNode();
120                                                        dwellLink = factory.createLink(Id.create(stop.getStopFacility().getId(), Link.class), node, node);
121                                                        dwellLink.setLength(WagonSimConstants.DEFAULT_LENGTH_LOOPLINK);
122                                                        dwellLink.setFreespeed(WagonSimConstants.DEFAULT_FREESPEED);
123                                                        dwellLink.setCapacity(WagonSimConstants.DEFAULT_CAPACITY);
124                                                        dwellLink.setNumberOfLanes(WagonSimConstants.DEFAULT_NUMLANES);
125                                                        dwellLink.setAllowedModes(transitModes);
126                                                        scenario.getNetwork().addLink(dwellLink);
127                                                }
128
129                                                boolean found = false;
130                                                for (int j=0; j<routeLinkIds.size(); j++) {
131                                                        Id<Link> linkId = routeLinkIds.get(j);
132                                                        if (linkId.equals(stop.getStopFacility().getLinkId())) {
133                                                                routeLinkIds.add(j+1,dwellLink.getId());
134                                                                found = true;
135                                                                break;
136                                                        }
137                                                }
138                                                if (!found) { throw new RuntimeException("Transit route stops and network route does not fit together. bailing out."); }
139                                                
140                                                for (Departure departure : transitRoute.getDepartures().values()) {
141                                                        double speed = dwellLink.getLength() / dwellTime;
142                                                        this.vehicleAttributes.putAttribute(departure.getVehicleId().toString(),dwellLink.getId().toString(),speed);
143                                                }
144                                        }
145                                        else {
146                                                transitRouteStops.add(stop);
147                                        }
148                                }
149                                transitRouteStops.add(transitRoute.getStops().get(transitRoute.getStops().size()-1));
150                                
151                                NetworkRoute networkRoute = RouteUtils.createNetworkRoute(routeLinkIds,scenario.getNetwork());
152                                TransitRoute newTransitRoute = scenario.getTransitSchedule().getFactory().createTransitRoute(Id.create(transitLine.getId(), TransitRoute.class), networkRoute,transitRouteStops,TransportMode.pt);
153                                for (Departure departure : transitRoute.getDepartures().values()) { newTransitRoute.addDeparture(departure); }
154                                newTransitRoutes.add(newTransitRoute);
155                        }
156
157                        Collection<TransitRoute> transitRoutes = transitLine.getRoutes().values();
158                        for (TransitRoute transitRoute : transitRoutes) { transitLine.removeRoute(transitRoute); }
159                        for (TransitRoute transitRoute : newTransitRoutes) { transitLine.addRoute(transitRoute); }
160                }
161        }
162}
