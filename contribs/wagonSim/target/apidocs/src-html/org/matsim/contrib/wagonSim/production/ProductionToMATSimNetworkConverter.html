050public class ProductionToMATSimNetworkConverter {

058        public static final double DEFAULT_CAPACITY = 99999.0;
059        private static final Set<String> defaultModes = new HashSet<String>();
060        private static final double RB_COORD_SHIFT = 500.0;
061        private static final double RCP_COORD_SHIFT = 300.0;
062        private static final double SAT_COORD_SHIFT = 100.0;
063        
064        private final Network network;
065        private final ObjectAttributes nodeAttributes;
066        private final ObjectAttributes linkAttributes;
067        
068        private static final Logger log = Logger.getLogger(ProductionToMATSimNetworkConverter.class);
069
070        //////////////////////////////////////////////////////////////////////
071        // constructors
072        //////////////////////////////////////////////////////////////////////
073
074        public ProductionToMATSimNetworkConverter(Network network, ObjectAttributes nodeAttributes, ObjectAttributes linkAttributes) {
075                this.network = network;
076                this.nodeAttributes = nodeAttributes;
077                this.linkAttributes = linkAttributes;
078                defaultModes.add(TransportMode.pt);
079        }
080
081        //////////////////////////////////////////////////////////////////////
082        // methods
083        //////////////////////////////////////////////////////////////////////
084        
085        public static final Set<String> getDefaultModes() {
086                defaultModes.add(TransportMode.pt);
087                return defaultModes;
088        }

092        private final void convertNodes(ProductionDataContainer dataContainer, Network infraNetwork) {
093                log.info("--- START converting nodes ---");
094                NetworkFactory factory = network.getFactory();
095                
096                for (ProductionNode pNode : dataContainer.productionNodes.values()) {
097                        String nodeId = null;
098                        int index = pNode.id.toString().indexOf(ProductionParser.RB_NODE_POSTFIX);
099                        if (index == -1) { index = pNode.id.toString().indexOf(ProductionParser.RCP_NODE_POSTFIX); }
100                        if (index == -1) { index = pNode.id.toString().indexOf(ProductionParser.SAT_NODE_POSTFIX); }
101                        if (index == -1) { throw new RuntimeException("Production node id="+pNode.id+" is neither a RB nor a RCP nor a SAT node. Bailing out."); }
102                        nodeId = pNode.id.toString().substring(0,index);
103                        Node infraNode = infraNetwork.getNodes().get(Id.create(nodeId, Node.class));
104                        if (infraNode == null) { throw new RuntimeException("Production node id="+pNode.id+": A node id="+nodeId+" is not found in the infrastructure network. Bailing out."); }
105                        
106                        if (pNode instanceof RbNode) {
107                                RbNode n = (RbNode)pNode;
108                                Coord c = new Coord(infraNode.getCoord().getX() + RB_COORD_SHIFT, infraNode.getCoord().getY() + RB_COORD_SHIFT);
109                                Node node = factory.createNode(n.id,c);
110                                network.addNode(node);
111                                
112                                if (n.parentNode != null) { throw new RuntimeException("RB node id="+n.id+" must not contain a partent node. Bailing out."); }
113                                if (n.parentReceptionNode != null) { throw new RuntimeException("RB node id="+n.id+" must not contain a parent reception node. Bailing out."); }
114                                if (n.siblingNodes.isEmpty()) { log.info("RB node id="+n.id+" does not contain any RCP nodes."); }
115                                
116                                nodeAttributes.putAttribute(n.id.toString(),WagonSimConstants.NODE_PRODUCTIONNODE_TYPE,ProductionParser.RB_NODE_POSTFIX);
117                                nodeAttributes.putAttribute(n.id.toString(),WagonSimConstants.NODE_INFRANODE_ID,infraNode.getId().toString());
118                                nodeAttributes.putAttribute(n.id.toString(),WagonSimConstants.NODE_MAXTRAINCREATION,n.maxTrainCreation);
119                                nodeAttributes.putAttribute(n.id.toString(),WagonSimConstants.NODE_MAXWAGONSHUNTINGS,n.maxWagonShuntings);
120                                nodeAttributes.putAttribute(n.id.toString(),WagonSimConstants.NODE_SHUNTINGTIME,n.shuntingTime);
121                        }
122                        else if (pNode instanceof RcpNode) {
123                                RcpNode n = (RcpNode)pNode;
124                                Coord c = new Coord(infraNode.getCoord().getX() + RCP_COORD_SHIFT, infraNode.getCoord().getY() + RCP_COORD_SHIFT);
125                                Node node = factory.createNode(n.id,c);
126                                network.addNode(node);
127                                
128                                if (n.parentNode == null) { throw new RuntimeException("RCP node id="+n.id+" must contain a partent node. Bailing out."); }
129                                if (n.parentReceptionNode == null) { throw new RuntimeException("RCP node id="+n.id+" must contain a parent reception node. Bailing out."); }
130                                if (n.siblingNodes.isEmpty()) { log.info("RCP node id="+n.id+" does not contain any SAT nodes."); }
131                                
132                                nodeAttributes.putAttribute(n.id.toString(),WagonSimConstants.NODE_PRODUCTIONNODE_TYPE,ProductionParser.RCP_NODE_POSTFIX);
133                                nodeAttributes.putAttribute(n.id.toString(),WagonSimConstants.NODE_INFRANODE_ID,infraNode.getId().toString());
134                                nodeAttributes.putAttribute(n.id.toString(),WagonSimConstants.NODE_PARENTRECEPTIONNODE_ID,n.parentReceptionNode.id.toString());
135                                nodeAttributes.putAttribute(n.id.toString(),WagonSimConstants.NODE_ISBORDER,n.isBorder);
136                                nodeAttributes.putAttribute(n.id.toString(),WagonSimConstants.NODE_DELIVERYTYPE_ID,n.deliveryType.id.toString());
137                                nodeAttributes.putAttribute(n.id.toString(),WagonSimConstants.NODE_DELIVERYTYPE_DESC,n.deliveryType.desc);
138                                nodeAttributes.putAttribute(n.id.toString(),WagonSimConstants.NODE_DELIVERYTYPE_DISTR,Arrays.toString(n.deliveryType.hourlyDistribution));
139                        }
140                        else if (pNode instanceof SatNode) {
141                                SatNode n = (SatNode)pNode;
142                                Coord c = new Coord(infraNode.getCoord().getX() + SAT_COORD_SHIFT, infraNode.getCoord().getY() + SAT_COORD_SHIFT);
143                                Node node = factory.createNode(n.id,c);
144                                network.addNode(node);
145                                
146                                if (n.parentNode == null) { throw new RuntimeException("SAT node id="+n.id+" must contain a partent node. Bailing out."); }
147                                if (n.parentReceptionNode == null) { throw new RuntimeException("SAT node id="+n.id+" must contain a parent reception node. Bailing out."); }
148                                if (!n.siblingNodes.isEmpty()) { throw new RuntimeException("SAT node id="+n.id+" must not contain any sibling nodes. Bailing out."); }
149                                
150                                nodeAttributes.putAttribute(n.id.toString(),WagonSimConstants.NODE_PRODUCTIONNODE_TYPE,ProductionParser.SAT_NODE_POSTFIX);
151                                nodeAttributes.putAttribute(n.id.toString(),WagonSimConstants.NODE_INFRANODE_ID,infraNode.getId().toString());
152                                nodeAttributes.putAttribute(n.id.toString(),WagonSimConstants.NODE_PARENTRECEPTIONNODE_ID,n.parentReceptionNode.id.toString());
153                                nodeAttributes.putAttribute(n.id.toString(),WagonSimConstants.NODE_MINSERVICE,n.minService);
154                        }
155                        else { throw new RuntimeException("Production node id="+pNode.id+" is of unknown type. Bailing out."); }
156                }
157                log.info("--- END   converting nodes ---");
158        }

162        private final void convertHierarchyLinks(ProductionDataContainer dataContainer) {
163                log.info("--- START converting Hierarchy links ---");
164                NetworkFactory factory = network.getFactory();
165                
166                for (ProductionNode pNode : dataContainer.productionNodes.values()) {
167                        Node fromNode = network.getNodes().get(pNode.id);
168                        
169                        if (pNode.parentNode != null) {
170                                Node toNode = network.getNodes().get(pNode.parentNode.id);
171                                Link link = factory.createLink(Id.create(fromNode.getId().toString()+"-"+toNode.getId().toString(), Link.class),fromNode,toNode);
172                                network.addLink(link);
173                                link.setLength(CoordUtils.calcEuclideanDistance(fromNode.getCoord(),toNode.getCoord()));
174                                link.setAllowedModes(defaultModes);
175                                link.setNumberOfLanes(1);
176                                link.setCapacity(DEFAULT_CAPACITY);
177                        }
178                        for (ProductionNode sNode : pNode.siblingNodes) {
179                                Node toNode = network.getNodes().get(sNode.id);
180                                Link link = factory.createLink(Id.create(fromNode.getId().toString()+"-"+toNode.getId().toString(), Link.class),fromNode,toNode);
181                                network.addLink(link);
182                                link.setLength(CoordUtils.calcEuclideanDistance(fromNode.getCoord(),toNode.getCoord()));
183                                link.setAllowedModes(defaultModes);
184                                link.setNumberOfLanes(1);
185                                link.setCapacity(DEFAULT_CAPACITY);
186                        }
187                }
188                log.info("--- END   converting Hierarchy links ---");
189        }
190
191        //////////////////////////////////////////////////////////////////////
192        
193        private final void convertRoutesToLinks(ProductionDataContainer dataContainer) {
194                log.info("--- START converting routes to links ---");
195                NetworkFactory factory = network.getFactory();
196                
197                for (Connection c : dataContainer.connections.values()) {
198                        Node fromNode = network.getNodes().get(c.fromNode.id);
199                        Node toNode = network.getNodes().get(c.viaNodes.get(0).id);
200                        Id<Link> linkId = Id.create(fromNode.getId().toString()+"-"+toNode.getId().toString(), Link.class);
201                        if (!network.getLinks().containsKey(linkId)) {
202                                Link link = factory.createLink(linkId, fromNode, toNode);
203                                network.addLink(link);
204                                link.setLength(CoordUtils.calcEuclideanDistance(fromNode.getCoord(),toNode.getCoord()));
205                                link.setAllowedModes(defaultModes);
206                                link.setNumberOfLanes(1);
207                                link.setCapacity(DEFAULT_CAPACITY);
208                        }
209                        for (int i=0; i<c.viaNodes.size()-1; i++) {
210                                fromNode = network.getNodes().get(c.viaNodes.get(i).id);
211                                toNode = network.getNodes().get(c.viaNodes.get(i+1).id);
212                                linkId = Id.create(fromNode.getId().toString()+"-"+toNode.getId().toString(), Link.class);
213                                if (!network.getLinks().containsKey(linkId)) {
214                                        Link link = factory.createLink(linkId,fromNode,toNode);
215                                        network.addLink(link);
216                                        link.setLength(CoordUtils.calcEuclideanDistance(fromNode.getCoord(),toNode.getCoord()));
217                                        link.setAllowedModes(defaultModes);
218                                        link.setNumberOfLanes(1);
219                                        link.setCapacity(DEFAULT_CAPACITY);
220                                }
221                        }
222                }
223                log.info("--- END   converting routes to links ---");
224        }
225        
226        //////////////////////////////////////////////////////////////////////
227        
228        public final void convert(ProductionDataContainer dataContainer, Network infraNetwork) {
229                convertNodes(dataContainer,infraNetwork);
230                convertHierarchyLinks(dataContainer);
231                convertRoutesToLinks(dataContainer);
232        }
