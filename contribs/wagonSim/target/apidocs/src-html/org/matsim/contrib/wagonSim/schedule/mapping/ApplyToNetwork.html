001package org.matsim.contrib.wagonSim.schedule.mapping;
002
003import java.util.List;
004
005import org.matsim.api.core.v01.network.Link;
006import org.matsim.api.core.v01.network.Network;
007import org.matsim.api.core.v01.network.Node;
008
009/**
010 * @author mrieser / senozon
011 */
012public class ApplyToNetwork {
013
014        private final Network network;
015        
016        public ApplyToNetwork(final Network network) {
017                this.network = network;
018        }
019        
020        public void applyEdits(final List<NetworkEdit> edits) {
021                for (NetworkEdit edit : edits) {
022                        if (edit instanceof MergeNodes) {
023                                doMergeNodes((MergeNodes) edit);
024                        } else if (edit instanceof AddLink) {
025                                doAddLink((AddLink) edit);
026                        } else if (edit instanceof ReplaceLink) {
027                                doReplaceLink((ReplaceLink) edit);
028                        }
029                }
030        }
031        
032        private void doMergeNodes(final MergeNodes edit) {
033                Node node = this.network.getNodes().get(edit.getNodeId());
034                Node targetNode = this.network.getNodes().get(edit.getTargetNodeId());
035                for (Link inLink : node.getInLinks().values()) {
036                        inLink.setToNode(targetNode);
037                }
038                node.getInLinks().clear();
039                for (Link outLink : node.getOutLinks().values()) {
040                        outLink.setFromNode(targetNode);
041                }
042                node.getOutLinks().clear();
043                this.network.removeNode(node.getId());
044        }
045        
046        private void doAddLink(final AddLink edit) {
047                Node fromNode = this.network.getNodes().get(edit.getFromNodeId());
048                Node toNode = this.network.getNodes().get(edit.getToNodeId());
049                Link link = this.network.getFactory().createLink(edit.getLinkId(), fromNode, toNode);
050                this.network.addLink(link);
051        }
052        
053        private void doReplaceLink(final ReplaceLink edit) {
054                if (edit.getReplacementLinkIds().size() != 1 || !edit.getLinkId().equals(edit.getReplacementLinkIds().get(0))) {
055                        this.network.removeLink(edit.getLinkId());
056                }
057        }
058}
