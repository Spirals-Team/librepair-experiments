159        private final void updateIgnoredAttributes(final String entity, final Attributes atts, final String [] usedAtts) {
160                Set<String> tmpAtts = new HashSet<String>();
161                for (int i=0; i<usedAtts.length; i++) { tmpAtts.add(usedAtts[i]); }
162                for (int i=0; i<atts.getLength(); i++) {
163                        if (!tmpAtts.contains(atts.getQName(i))) {
164                                if (!ignoredAtts.containsKey(entity)) { ignoredAtts.put(entity,new HashSet<String>()); }
165                                if (!ignoredAtts.get(entity).contains(atts.getQName(i))) {
166                                        ignoredAtts.get(entity).add(atts.getQName(i));
167                                }
168                        }
169                }
170        }

174        final Boolean stringToBoolean(String string) {
175                if (string.equalsIgnoreCase(Boolean.TRUE.toString()) || string.equalsIgnoreCase(Boolean.FALSE.toString())) {
176                        return Boolean.valueOf(string);
177                }
178                return null;
179        }

183        private static final String LAND_ID_LAND = "id_Land";
184        private static final String LAND_BEZEICHNUNG = "bezeichnung";
185        private static final String LAND_INLAND = "inland";
186        
187        private final void startLand(final Attributes atts) {
188                updateIgnoredAttributes(LAND,atts,new String []{ LAND_ID_LAND, LAND_BEZEICHNUNG, LAND_INLAND });
189
190                int id_Land = Integer.parseInt(atts.getValue(LAND_ID_LAND));
191                String bezeichnung = atts.getValue(LAND_BEZEICHNUNG);
192                boolean inland = stringToBoolean(atts.getValue(LAND_INLAND));
193                
194                NEMOInfraCountry country = new NEMOInfraCountry(id_Land);
195                country.name = bezeichnung;
196                country.isHomeCountry = inland;
197                
198                if (dataContainer.countries.put(country.id,country) != null) {
199                        throw new RuntimeException("At Entity '"+LAND+"': A country with id="+id_Land+" exists at least twice. Bailing out.");
200                }
201        }

205        private static final String KNOTEN_BSCODE = "bscode";
206        private static final String KNOTEN_NAME = "name";
207        private static final String KNOTEN_XKOORD = "xkoord";
208        private static final String KNOTEN_YKOORD = "ykoord";
209        private static final String KNOTEN_LAND = "land";
210        private static final String KNOTEN_BAHNHOF = "bahnhof";
211        private static final String KNOTEN_GUELTIG = "gueltig";
212        private static final String KNOTEN_CLUSTER = "cluster";
213        
214        private final void startKnoten(final Attributes atts) {
215                updateIgnoredAttributes(KNOTEN,atts,new String []{ KNOTEN_BSCODE, KNOTEN_NAME, KNOTEN_XKOORD, KNOTEN_YKOORD, KNOTEN_LAND, KNOTEN_BAHNHOF, KNOTEN_GUELTIG, KNOTEN_CLUSTER });
216
217                String bscode = atts.getValue(KNOTEN_BSCODE);
218                String name = atts.getValue(KNOTEN_NAME);
219                int xkoord = Integer.parseInt(atts.getValue(KNOTEN_XKOORD));
220                int ykoord = Integer.parseInt(atts.getValue(KNOTEN_YKOORD));
221                int land = Integer.parseInt(atts.getValue(KNOTEN_LAND));
222                boolean bahnhof = stringToBoolean(atts.getValue(KNOTEN_BAHNHOF));
223                boolean gueltig = stringToBoolean(atts.getValue(KNOTEN_GUELTIG));
224                String cluster = atts.getValue(KNOTEN_CLUSTER);
225                if (cluster.equals("")) { cluster = "restCluster"; } // assign all node without a cluster to a restCluster (much better that call it "")
226                
227                NEMOInfraNode node = new NEMOInfraNode(bscode);
228                node.name = name;
229                node.coord = new Coord((double) xkoord, (double) ykoord);
230                node.countryId = Integer.toString(land);
231                node.isStation = bahnhof;
232                node.isValid = gueltig;
233                node.clusterId = cluster;
234
235                if (dataContainer.nodes.put(node.id,node) != null) {
236                        throw new RuntimeException("At Entity '"+KNOTEN+"': A node with id="+bscode+" exists at least twice. Bailing out.");
237                }
238
239                NEMOInfraNodeCluster nodeCluster = dataContainer.nodeClusters.get(node.clusterId);
240                if (nodeCluster == null) { 
241                        nodeCluster = new NEMOInfraNodeCluster(cluster);
242                        dataContainer.nodeClusters.put(nodeCluster.id,nodeCluster);
243                }
244                nodeCluster.nodeIds.add(node.id);
245        }

250        private static final String STRECKENKATEGORIE_BEZEICHNUNG = "bezeichnung";
251        private static final String STRECKENKATEGORIE_VFAKTOR = "vFaktor";
252        
253        private final void startStreckenkategorie(final Attributes atts) {
254                updateIgnoredAttributes(STRECKENKATEGORIE,atts,new String []{ STRECKENKATEGORIE_ID_STRECKENKATEGORIE, STRECKENKATEGORIE_BEZEICHNUNG, STRECKENKATEGORIE_VFAKTOR });
255
256                int id_Streckenkategorie = Integer.parseInt(atts.getValue(STRECKENKATEGORIE_ID_STRECKENKATEGORIE));
257                double velocity = Double.parseDouble(atts.getValue(STRECKENKATEGORIE_BEZEICHNUNG).split(" ")[0])/3.6; // format: "[0-9]+ km / h"; extract number and convert to m/s
258                double vFaktor = Double.parseDouble(atts.getValue(STRECKENKATEGORIE_VFAKTOR));
259                
260                NEMOInfraLinkType linkType = new NEMOInfraLinkType(id_Streckenkategorie);
261                linkType.velocity = velocity;
262                linkType.vFactor = vFaktor;
263                
264                if (dataContainer.linkTypes.put(linkType.id,linkType) != null) {
265                        throw new RuntimeException("At Entity '"+STRECKENKATEGORIE+"': A link type with id="+id_Streckenkategorie+" exists at least twice. Bailing out.");
266                }
267        }

291        private static final String KANTE_VONKNOTEN = "vonKnoten";
292        private static final String KANTE_NACHKNOTEN = "nachKnoten";
293        private static final String KANTE_STRECKESIMU = "streckeSIMU";
294        private static final String KANTE_LAENGE = "laenge";
295        private static final String KANTE_ZWEIGLEISIG = "zweigleisig";
296        private static final String KANTE_GLOBALBEREICH = "globalbereich";
297        private static final String KANTE_STRECKENKATEGORIE = "streckenkategorie";
298        private static final String KANTE_STRECKENKST = "streckenKST";
299        private static final String KANTE_GESPERRT = "gesperrt";
300        private static final String KANTE_GUELTIG = "gueltig";
301        private static final String KANTE_ZUGLAENGE_MAX = "zuglaenge_max";
302
303        private final void startKante(final Attributes atts) {
304                updateIgnoredAttributes(KANTE,atts,new String []{ KANTE_ID_KANTE, KANTE_VONKNOTEN, KANTE_NACHKNOTEN, KANTE_STRECKESIMU, KANTE_LAENGE, KANTE_ZWEIGLEISIG, KANTE_GLOBALBEREICH, KANTE_STRECKENKATEGORIE, KANTE_STRECKENKST, KANTE_GESPERRT, KANTE_GUELTIG, KANTE_ZUGLAENGE_MAX });
305
306                int id_Kante = Integer.parseInt(atts.getValue(KANTE_ID_KANTE));
307                String vonKnoten = atts.getValue(KANTE_VONKNOTEN);
308                String nachKnoten = atts.getValue(KANTE_NACHKNOTEN);
309                int streckeSIMU = Integer.parseInt(atts.getValue(KANTE_STRECKESIMU));
310                if ((streckeSIMU != 0) && (streckeSIMU != 1)) {
311                        throw new RuntimeException("At Entity '"+KANTE+"': Attribute "+KANTE_STRECKESIMU+"="+streckeSIMU+" not allowed. Bailing out.");
312                }
313                int laenge = Integer.parseInt(atts.getValue(KANTE_LAENGE));
314                if (laenge <= 0) {
315                        throw new RuntimeException("At Entity '"+KANTE+"': Attribute "+KANTE_LAENGE+"="+streckeSIMU+" not allowed (must be greater than zero). Bailing out.");
316                }
317                boolean zweigleisig = stringToBoolean(atts.getValue(KANTE_ZWEIGLEISIG));
318                boolean globalbereich = stringToBoolean(atts.getValue(KANTE_GLOBALBEREICH));
319                int streckenkategorie = Integer.parseInt(atts.getValue(KANTE_STRECKENKATEGORIE));
320                int streckenKST = Integer.parseInt(atts.getValue(KANTE_STRECKENKST));
321                boolean gesperrt = stringToBoolean(atts.getValue(KANTE_GESPERRT));
322                boolean gueltig = stringToBoolean(atts.getValue(KANTE_GUELTIG));
323                int zuglaenge_max = Integer.parseInt(atts.getValue(KANTE_ZUGLAENGE_MAX));
324                if (zuglaenge_max <= 0) {
325                        throw new RuntimeException("At Entity '"+KANTE+"': Attribute "+KANTE_ZUGLAENGE_MAX+"="+zuglaenge_max+" not allowed (must be greater than zero). Bailing out.");
326                }
327                
328                NEMOInfraLink link = new NEMOInfraLink(id_Kante);
329                link.fromNodeId = Id.create(vonKnoten, Node.class);
330                link.toNodeId = Id.create(nachKnoten, Node.class);
331                if (streckeSIMU == 0) { link.isSimuLink = false; } else { link.isSimuLink = true; }
332                link.length = (double)laenge;
333                link.hasTwoTracks = zweigleisig;
334                link.isGlobal = globalbereich;
335                link.typeId = Integer.toString(streckenkategorie);
336                link.ownerId = Integer.toString(streckenKST);
337                link.isClosed = gesperrt;
338                link.isValid = gueltig;
339                link.maxTrainLength = (double)zuglaenge_max;
340                
341                if (dataContainer.links.put(link.id,link) != null) {
342                        throw new RuntimeException("At Entity '"+KANTE+"': A link id="+id_Kante+" exists at least twice. Bailing out.");
343                }
344        }

388        /* (non-Javadoc)
389         * @see org.matsim.core.utils.io.MatsimXmlParser#startTag(java.lang.String, org.xml.sax.Attributes, java.util.Stack)
390         */
391        @Override
392        public void startTag(String name, Attributes atts, Stack<String> context) {
393                if (LAND.equals(name)) { startLand(atts); }
394                else if (KNOTEN.equals(name)) { startKnoten(atts); }
395                else if (STRECKENKATEGORIE.equals(name)) { startStreckenkategorie(atts); }
396                else if (STRECKENKST.equals(name)) { startStreckenKST(atts); }
397                else if (KANTE.equals(name)) { startKante(atts); }
398                else if (GLEIS.equals(name)) { startGleis(atts); }
399                else if (FAHRTRICHTUNG.equals(name)) { startFahrtrichtung(atts); }
400                else {
401                        if (!ignoredEntities.contains(name)) { ignoredEntities.add(name); }
402                }
403        }

407        /* (non-Javadoc)
408         * @see org.matsim.core.utils.io.MatsimXmlParser#endTag(java.lang.String, java.lang.String, java.util.Stack)
409         */
410        @Override
411        public void endTag(String name, String content, Stack<String> context) {
412                if (NEMOINFRA.equals(name)) {
413                        log.info("--- START: list of ignored entities ---");
414                        for (String entity : ignoredEntities) { log.info(entity); }
415                        log.info("--- END:   list of ignored entities ---");
416                        ignoredEntities.clear();
417                        log.info("--- START: list of ignored attributes ---");
418                        for (Entry<String,Set<String>> e : ignoredAtts.entrySet()) { log.info(e.getKey()+": "+e.getValue().toString()); }
419                        log.info("--- END:   list of ignored attributes ---");
420                        ignoredAtts.clear();
421                        
422                        boolean isValid = true;
423                        boolean isValidTmp = true;
424                        log.info("--- START: validation ---");
425                        isValidTmp = dataContainer.validateCountries(); log.info("countries: isValid="+isValidTmp); if (!isValidTmp) { isValid = false; }
426                        isValidTmp = dataContainer.validateLinkTypes(); log.info("link types: isValid="+isValidTmp); if (!isValidTmp) { isValid = false; }
427                        isValidTmp = dataContainer.validateLinkOwners(); log.info("link owners: isValid="+isValidTmp); if (!isValidTmp) { isValid = false; }
428                        isValidTmp = dataContainer.validateNodeClusters(); log.info("node clusters: isValid="+isValidTmp); if (!isValidTmp) { isValid = false; }
429                        isValidTmp = dataContainer.validateNodes(); log.info("nodes: isValid="+isValidTmp); if (!isValidTmp) { isValid = false; }
430                        isValidTmp = dataContainer.validateLinks(); log.info("links: isValid="+isValidTmp); if (!isValidTmp) { isValid = false; }
431                        isValidTmp = dataContainer.validateTracks(); log.info("tracks: isValid="+isValidTmp); if (!isValidTmp) { isValid = false; }
432                        isValidTmp = dataContainer.validateDirections(); log.info("directions: isValid="+isValidTmp); if (!isValidTmp) { isValid = false; }
433                        log.info("--- End:   validation ---");
434                        
435                        if (!isValid) { throw new RuntimeException(NEMOINFRA+" data validation failed. Bailing out."); }
436                }
437        }
