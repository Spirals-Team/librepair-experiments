081        private final void convertStations(OTTDataContainer dataContainer, Network infraNetwork) {
082                TransitScheduleFactory factory = scenario.getTransitSchedule().getFactory();
083
084                // get station ids
085                Set<Id<TransitStopFacility>> stationIds = new HashSet<>();
086                for (Locomotive locomotive : dataContainer.locomotives.values()) {
087                        for (StationData stationData : locomotive.trips.values()) {
088                                stationIds.add(stationData.stationId);
089                        }
090                }
091
092                for (Id<TransitStopFacility> stationId : stationIds) {
093                        Node node = infraNetwork.getNodes().get(Id.create(stationId, Node.class));
094                        if (node != null) {
095                                TransitStopFacility stopFacility = factory.createTransitStopFacility(stationId,node.getCoord(),false);
096                                stopFacility.setName(node.getId().toString());
097                                scenario.getTransitSchedule().addStopFacility(stopFacility);
098                        }
099                        else { throw new RuntimeException("node id="+stationId.toString()+" not found in the network. Bailing out."); }
100                }
101        }

105        @SuppressWarnings("deprecation")
106        private final Date extractStartDate(OTTDataContainer dataContainer, boolean isPerformance) {
107                Date startDate = null;
108                for (Locomotive locomotive : dataContainer.locomotives.values()) {
109                        for (StationData stationData : locomotive.trips.values()) {
110                                if (isPerformance) {
111                                        if (startDate == null) { startDate = new Date(stationData.arrival.getTime()); }
112                                        else if (stationData.arrival.getTime() < startDate.getTime()) { startDate.setTime(stationData.arrival.getTime()); }
113                                }
114                                else {
115                                        long time = stationData.arrival.getTime() - (stationData.delayArrival.longValue()*1000L);
116                                        if (startDate == null) { startDate = new Date(time); }
117                                        else if (time < startDate.getTime()) { startDate.setTime(time); }
118                                }
119                        }
120                }
121                
122                startDate.setHours(0);
123                startDate.setMinutes(0);
124                startDate.setSeconds(0);
125                return startDate;
126        }

130        private final void convertSchedules(OTTDataContainer dataContainer, ObjectAttributes trainTypes, boolean isPerformance) {
131                TransitScheduleFactory scheduleFactory = scenario.getTransitSchedule().getFactory();
132                VehiclesFactory vehiclesFactory = ((MutableScenario)scenario).getTransitVehicles().getFactory();
133                
134                VehicleType vehicleType = vehiclesFactory.createVehicleType(Id.create(WagonSimConstants.DEFAULT_VEHICLE_TYPE, VehicleType.class));
135                VehicleCapacity vehicleCapacity = vehiclesFactory.createVehicleCapacity();
136                // we do not use this capacity. Therefore it should infinite, otherwise this capacity may exceed
137                // before ``our'' capacities are exceeded // dr, oct'13
138                vehicleCapacity.setSeats(999999);
139                vehicleCapacity.setStandingRoom(999999);
140                // we defined the vehicle-enter/leave-time is implicit included in transfer-times which
141                // are defined in the transitrouterconfig (for handling see WagonSimTripRouterFactoryImpl#WagonSimRouterWrapper)
142                // dr, oct'13
143                vehicleType.setAccessTime(0);
144                vehicleType.setEgressTime(0);
145                vehicleType.setCapacity(vehicleCapacity);
146                ((MutableScenario)scenario).getTransitVehicles().addVehicleType(vehicleType);
147                
148                Date startDate = extractStartDate(dataContainer,isPerformance);
149                System.out.println("startDate="+startDate.toString());
150
151                for (Locomotive locomotive : dataContainer.locomotives.values()) {
152
153                        Departure departure = null;
154                        List<TransitRouteStop> transitRouteStops = new ArrayList<TransitRouteStop>();
155                        for (StationData stationData : locomotive.trips.values()) {
156
157                                TransitStopFacility stopFacility = scenario.getTransitSchedule().getFacilities().get(stationData.stationId);
158                                if (stopFacility == null) { throw new RuntimeException("locomotive id="+locomotive.id+": station id="+stationData.stationId+" not found. Bailing out."); }
159                                
160                                double arrivalDelay = Double.NaN;
161                                double departureDelay = Double.NaN;
162                                if (departure == null) {
163                                        double lineDepartureOffset = (stationData.departure.getTime()-startDate.getTime())/1000.0;
164                                        if (!isPerformance) { lineDepartureOffset -= stationData.delayDeparture; }
165                                        departure = scheduleFactory.createDeparture(Id.create(locomotive.id, Departure.class), lineDepartureOffset);
166                                        arrivalDelay = 0.0;
167                                }
168                                else {
169                                        arrivalDelay = (stationData.arrival.getTime()-startDate.getTime())/1000.0 - departure.getDepartureTime();
170                                        if (!isPerformance) { arrivalDelay -= stationData.delayArrival; }
171                                }
172                                departureDelay = (stationData.departure.getTime()-startDate.getTime())/1000.0 - departure.getDepartureTime();
173                                if (!isPerformance) { departureDelay -= stationData.delayDeparture; }
174                                
175
176                                if (departureDelay < arrivalDelay) { throw new RuntimeException("locomotive id="+locomotive.id+": arrival="+stationData.arrival.toString()+" does not fit with departure="+stationData.departure.toString()+". ("+departureDelay+"<"+arrivalDelay+") Bailing out."); }
177
178                                TransitRouteStop stop = scheduleFactory.createTransitRouteStop(stopFacility,arrivalDelay,departureDelay);
179                                stop.setAwaitDepartureTime(true);
180                                transitRouteStops.add(stop);
181                        }
182                        
183                        if (transitRouteStops.size() > 1) {
184                                
185                                // check if train type is given
186                                if (trainTypes.getAttribute(locomotive.type.toString(),WagonSimConstants.TRAIN_MAX_SPEED) == null) {
187                                        throw new RuntimeException("locomotive id="+locomotive.id+": type="+locomotive.type+" is not defined by the train type table. Bailing out.");
188                                }
189                                
190                                TransitLine line = scheduleFactory.createTransitLine(Id.create(locomotive.id, TransitLine.class));
191                                scenario.getTransitSchedule().addTransitLine(line);
192                                TransitRoute route = scheduleFactory.createTransitRoute(Id.create(line.getId(), TransitRoute.class), null, transitRouteStops, TransportMode.pt);
193                                line.addRoute(route);
194                                
195                                Vehicle vehicle = vehiclesFactory.createVehicle(Id.create(route.getId(), Vehicle.class), vehicleType);
196                                ((MutableScenario)scenario).getTransitVehicles().addVehicle(vehicle);
197                                departure.setVehicleId(vehicle.getId());
198                                route.addDeparture(departure);
199                                this.vehicleAttributes.putAttribute(vehicle.getId().toString(),WagonSimConstants.TRAIN_TYPE, locomotive.type);
200                                this.vehicleAttributes.putAttribute(vehicle.getId().toString(),WagonSimConstants.TRAIN_MAX_SPEED, trainTypes.getAttribute(locomotive.type.toString(),WagonSimConstants.TRAIN_MAX_SPEED));
201                                this.vehicleAttributes.putAttribute(vehicle.getId().toString(),WagonSimConstants.TRAIN_MAX_WEIGHT, trainTypes.getAttribute(locomotive.type.toString(),WagonSimConstants.TRAIN_MAX_WEIGHT));
202                                this.vehicleAttributes.putAttribute(vehicle.getId().toString(),WagonSimConstants.TRAIN_MAX_LENGTH, trainTypes.getAttribute(locomotive.type.toString(),WagonSimConstants.TRAIN_MAX_LENGTH));
203
204                                // the next day
205                                vehicle = vehiclesFactory.createVehicle(Id.create(route.getId()+".1", Vehicle.class),vehicleType);
206                                ((MutableScenario)scenario).getTransitVehicles().addVehicle(vehicle);
207                                departure = scheduleFactory.createDeparture(Id.create(vehicle.getId(), Departure.class), departure.getDepartureTime()+24*3600);
208                                departure.setVehicleId(vehicle.getId());
209                                route.addDeparture(departure);
210                                this.vehicleAttributes.putAttribute(vehicle.getId().toString(),WagonSimConstants.TRAIN_TYPE, locomotive.type);
211                                this.vehicleAttributes.putAttribute(vehicle.getId().toString(),WagonSimConstants.TRAIN_MAX_SPEED, trainTypes.getAttribute(locomotive.type.toString(),WagonSimConstants.TRAIN_MAX_SPEED));
212                                this.vehicleAttributes.putAttribute(vehicle.getId().toString(),WagonSimConstants.TRAIN_MAX_WEIGHT, trainTypes.getAttribute(locomotive.type.toString(),WagonSimConstants.TRAIN_MAX_WEIGHT));
213                                this.vehicleAttributes.putAttribute(vehicle.getId().toString(),WagonSimConstants.TRAIN_MAX_LENGTH, trainTypes.getAttribute(locomotive.type.toString(),WagonSimConstants.TRAIN_MAX_LENGTH));
214
215                                // the day after the next day
216                                vehicle = vehiclesFactory.createVehicle(Id.create(route.getId()+".2", Vehicle.class),vehicleType);
217                                ((MutableScenario)scenario).getTransitVehicles().addVehicle(vehicle);
218                                departure = scheduleFactory.createDeparture(Id.create(vehicle.getId(), Departure.class), departure.getDepartureTime()+24*3600);
219                                departure.setVehicleId(vehicle.getId());
220                                route.addDeparture(departure);
221                                this.vehicleAttributes.putAttribute(vehicle.getId().toString(),WagonSimConstants.TRAIN_TYPE, locomotive.type);
222                                this.vehicleAttributes.putAttribute(vehicle.getId().toString(),WagonSimConstants.TRAIN_MAX_SPEED, trainTypes.getAttribute(locomotive.type.toString(),WagonSimConstants.TRAIN_MAX_SPEED));
223                                this.vehicleAttributes.putAttribute(vehicle.getId().toString(),WagonSimConstants.TRAIN_MAX_WEIGHT, trainTypes.getAttribute(locomotive.type.toString(),WagonSimConstants.TRAIN_MAX_WEIGHT));
224                                this.vehicleAttributes.putAttribute(vehicle.getId().toString(),WagonSimConstants.TRAIN_MAX_LENGTH, trainTypes.getAttribute(locomotive.type.toString(),WagonSimConstants.TRAIN_MAX_LENGTH));
225                        }
226                        else if (transitRouteStops.size() == 1) {
227                                System.out.println("locomotive id="+locomotive.id+": only one station given. Therefore, no transitLine created.");
228                        }
229                        else {
230                                System.out.println("locomotive id="+locomotive.id+": no station is given. Therefore, no transitLine created.");
231                        }
232                }
233        }

237        private final void convertPseudoNetwork() {
238                List<Id<Node>> nodeIds = new ArrayList<Id<Node>>(scenario.getNetwork().getNodes().keySet());
239                for (Id<Node> id : nodeIds) { scenario.getNetwork().removeNode(id); }
240                new CreatePseudoNetwork(scenario.getTransitSchedule(),scenario.getNetwork(),"").createNetwork();
241
242                for (Link l : scenario.getNetwork().getLinks().values()) {
243                        l.setCapacity(WagonSimConstants.DEFAULT_CAPACITY);
244                        l.setFreespeed(WagonSimConstants.DEFAULT_FREESPEED);
245                }
246        }

250        private final void createVehicleLinkSpeedAttributes() {
251                for (TransitLine transitLine : scenario.getTransitSchedule().getTransitLines().values()) {
252                        for (TransitRoute transitRoute : transitLine.getRoutes().values()) {
253                                Set<Id<Vehicle>> vehIds = new HashSet<>();
254                                for (Departure departure : transitRoute.getDepartures().values()) {
255                                        vehIds.add(departure.getVehicleId());
256                                }
257                                
258                                Iterator<TransitRouteStop> iterator = transitRoute.getStops().iterator();
259                                double departure = iterator.next().getDepartureOffset();
260                                while (iterator.hasNext()) {
261                                        TransitRouteStop routeStop = iterator.next();
262                                        double arrival = routeStop.getArrivalOffset();
263                                        Link link = scenario.getNetwork().getLinks().get(routeStop.getStopFacility().getLinkId());
264                                        double speed = link.getLength() / (arrival - departure);
265                                        if (speed >= 200.0) {
266                                                System.out.println("line="+transitLine.getId()+";route="+transitRoute.getId()+"stop="+routeStop.getStopFacility().getId()+": lLenth="+link.getLength()+";arr="+arrival+";dep="+departure+"");
267                                        }
268
269                                        for (Id<Vehicle> vehId : vehIds) {
270                                                this.vehicleAttributes.putAttribute(vehId.toString(),link.getId().toString(),speed);
271                                        }
272                                        
273                                        departure = routeStop.getDepartureOffset();
274                                }
275                        }
276                }
277        }
278
279        //////////////////////////////////////////////////////////////////////
280        
281        public final void convert(OTTDataContainer dataContainer, Network infraNetwork, ObjectAttributes trainTypes, boolean isPerformance) {
282                convertStations(dataContainer,infraNetwork);
283                convertSchedules(dataContainer,trainTypes,isPerformance);
284                convertPseudoNetwork();
285                createVehicleLinkSpeedAttributes();
286        }
