057        public static final Controler init(Scenario scenario, final ObjectAttributes vehicleLinkSpeedAttributes, Map<Id<TransitStopFacility>, Double> minShuntingTimes) {
058//              super(scenario);
059                final Controler controler = new Controler( scenario ) ;
060                Config config = controler.getConfig() ;
061                
062                final WagonSimVehicleLoadListener listener = new WagonSimVehicleLoadListener(scenario.getPopulation().getPersonAttributes());
063                controler.addOverridingModule(new AbstractModule() {
064                        @Override
065                        public void install() {
066                                bindMobsim().toProvider(new Provider<Mobsim>() {
067                                        @Override
068                                        public Mobsim get() {
069                                                return new WagonSimQSimFactory(vehicleLinkSpeedAttributes, listener).createMobsim(controler.getScenario(), controler.getEvents());
070                                        }
071                                });
072                        }
073                });
074                controler.addControlerListener(listener);
075                Provider<TransitRouter> routerFactory =
076                                new WagonSimRouterFactoryImpl(
077                                                listener, 
078                                                scenario.getTransitSchedule(), new 
079                                                TransitRouterConfig(config.planCalcScore(), config.plansCalcRoute(), config.transitRouter(), config.vspExperimental()), 
080                                                scenario.getPopulation().getPersonAttributes(), 
081                                                vehicleLinkSpeedAttributes);
082                controler.setTripRouterFactory(new WagonSimTripRouterFactoryImpl(scenario, routerFactory, minShuntingTimes));
083                
084                controler.addControlerListener(new WagonSimAnalysisListener());
085                
086                return controler ;
087        }
