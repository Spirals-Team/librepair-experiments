054        public static final <T> void writeObjectIds(final Set<Id<T>> objectIds, final String outFile, final String headerName) throws IOException {
055                BufferedWriter writer = org.matsim.core.utils.io.IOUtils.getBufferedWriter(outFile);
056                if (headerName != null) { writer.write(headerName+"\n"); }
057                TreeSet<Id<T>> sortedIds = new TreeSet<>(objectIds);
058                for (Id<T> oId : sortedIds) { writer.write(oId.toString()+"\n"); }
059                writer.flush();
060                writer.close();
061        }

065        public static final boolean prepareFolder(String folder) {
066                File dir = new File(folder);
067                if (dir.exists()) {
068                        if (dir.isFile()) { return false; }
069                        else if (dir.list().length > 0) { return false; }
070                        else { return true; }
071                }
072                else if (dir.mkdirs()) { return true; }
073                else { return false; }
074        }

078        public static final String removeSurroundingQuotes(String string) {
079                if (string.startsWith("\"") && string.endsWith("\"")) { return string.substring(1,string.length()-1); }
080                return string;
081        }

085        public static final Map<Id<Node>, Id<Node>> parseNodeMapFile(String nodeMapFile) throws IOException {
086                Map<Id<Node>, Id<Node>> nodeMap = new HashMap<Id<Node>, Id<Node>>();
087                BufferedReader br = IOUtils.getBufferedReader(nodeMapFile);
088                int currRow = 0;
089                String curr_line;
090                while ((curr_line = br.readLine()) != null) {
091                        currRow++;
092                        String [] row = curr_line.split(";");
093                        Id<Node> origNodeId = Id.create(Utils.removeSurroundingQuotes(row[0].trim()), Node.class);
094                        Id<Node> mapNodeId = Id.create(Utils.removeSurroundingQuotes(row[1].trim()), Node.class);
095                        if (nodeMap.put(origNodeId,mapNodeId) != null) { throw new RuntimeException("row "+currRow+": node id="+origNodeId+" already mapped to a node. Bailing out."); }
096                }
097                return nodeMap;
098        }

102        public static final ObjectAttributes parseTrainTypesFile(String trainTypesFile) throws IOException {
103                ObjectAttributes trainTypes = new ObjectAttributes();
104
105                BufferedReader br = IOUtils.getBufferedReader(trainTypesFile);
106                String curr_line;
107
108                // read header an build lookup
109                curr_line = br.readLine();
110                String [] header = curr_line.split(";");
111                Map<String, Integer> lookup = new LinkedHashMap<String,Integer>(header.length);
112                for (int i=0; i<header.length; i++) { lookup.put(Utils.removeSurroundingQuotes(header[i].trim()),i); }
113
114                // parse rows and store nodes
115                while ((curr_line = br.readLine()) != null) {
116                        String [] row = curr_line.split(";");
117                        
118                        String type = Utils.removeSurroundingQuotes(row[lookup.get(WagonSimConstants.TRAIN_TYPE)].trim());
119                        Double maxSpeed = Double.parseDouble(Utils.removeSurroundingQuotes(row[lookup.get(WagonSimConstants.TRAIN_MAX_SPEED)].trim()))/3.6;
120                        Double maxWeight = Double.parseDouble(Utils.removeSurroundingQuotes(row[lookup.get(WagonSimConstants.TRAIN_MAX_WEIGHT)].trim()))*1000.0;
121                        Double maxLength = Double.parseDouble(Utils.removeSurroundingQuotes(row[lookup.get(WagonSimConstants.TRAIN_MAX_LENGTH)].trim()));
122
123                        trainTypes.putAttribute(type, WagonSimConstants.TRAIN_MAX_SPEED, maxSpeed);
124                        trainTypes.putAttribute(type, WagonSimConstants.TRAIN_MAX_WEIGHT, maxWeight);
125                        trainTypes.putAttribute(type, WagonSimConstants.TRAIN_MAX_LENGTH, maxLength);
126                }
127                return trainTypes;
128        }

132        public static final Map<String, Id<Node>> parseZoneToNodeMapFile(String zoneToNodeMapFile) throws IOException {
133                Map<String, Id<Node>> zoneToNodeMap = new HashMap<>();
134
135                BufferedReader br = IOUtils.getBufferedReader(zoneToNodeMapFile);
136                int currRow = 0;
137                String curr_line;
138
139                // read header an build lookup
140                curr_line = br.readLine(); currRow++;
141                String [] header = curr_line.split(";");
142                Map<String, Integer> lookup = new LinkedHashMap<String,Integer>(header.length);
143                for (int i=0; i<header.length; i++) { lookup.put(Utils.removeSurroundingQuotes(header[i].trim()),i); }
144
145                // parse rows and store nodes
146                while ((curr_line = br.readLine()) != null) {
147                        currRow++;
148                        String [] row = curr_line.split(";");
149                        
150                        String zoneId = Utils.removeSurroundingQuotes(row[lookup.get("GVVerkehrszelle")].trim());
151                        Id<Node> nodeId = Id.create(Utils.removeSurroundingQuotes(row[lookup.get("Abfertigungsstelle")].trim()), Node.class);

153                        if (zoneToNodeMap.put(zoneId,nodeId) != null) { throw new RuntimeException("row "+currRow+": zone id="+zoneId+" already mapped to a node. Bailing out."); }
154                }
155                return zoneToNodeMap;

160        public static final void writeShuntingTable(TransitSchedule schedule, String outputFile) throws IOException {
161                BufferedWriter bw = IOUtils.getBufferedWriter(outputFile);
162                bw.write(""+WagonSimConstants.SHUNTING_TABLE_LOCID+"\t"+WagonSimConstants.SHUNTING_TABLE_NODEID+"\t"+WagonSimConstants.SHUNTING_TABLE_SHUNTINGFLAG+"\n");
163                for (TransitLine transitLine : schedule.getTransitLines().values()) {
164                        Id locId = transitLine.getId();
165                        if (transitLine.getRoutes().size() != 1) { throw new RuntimeException("lineId="+locId+" must define exactly one transit route. Bailing out."); }
166                        TransitRoute transitRoute = transitLine.getRoutes().values().iterator().next();
167                        for (TransitRouteStop stop : transitRoute.getStops()) {
168                                bw.write(locId.toString()+"\t"+stop.getStopFacility().getId().toString()+"\ttrue\n");
169                        }
170                }
171                bw.flush();
172                bw.close();
173        }

177        public static final Map<Id<TransitLine>, Map<Id<Node>, Boolean>> parseShuntingTable(String shuntingTableFile) throws IOException {
178                Map<Id<TransitLine>, Map<Id<Node>, Boolean>> shuntingTable = new HashMap<>();
179                BufferedReader br = IOUtils.getBufferedReader(shuntingTableFile);
180                String curr_line;
181
182                // read header an build lookup
183                curr_line = br.readLine();
184                String [] header = curr_line.split("\t");
185                Map<String, Integer> lookup = new LinkedHashMap<String,Integer>(header.length);
186                for (int i=0; i<header.length; i++) { lookup.put(Utils.removeSurroundingQuotes(header[i].trim()),i); }
187
188                // parse rows
189                while ((curr_line = br.readLine()) != null) {
190                        String [] row = curr_line.split("\t");
191                        Id<TransitLine> locId = Id.create(Utils.removeSurroundingQuotes(row[lookup.get(WagonSimConstants.SHUNTING_TABLE_LOCID)].trim()), TransitLine.class);
192                        Id<Node> nodeId = Id.create(Utils.removeSurroundingQuotes(row[lookup.get(WagonSimConstants.SHUNTING_TABLE_NODEID)].trim()), Node.class);
193                        Boolean shuntingAllowed = new Boolean(Utils.removeSurroundingQuotes(row[lookup.get(WagonSimConstants.SHUNTING_TABLE_SHUNTINGFLAG)].trim()));
194                        Map<Id<Node>,Boolean> map = shuntingTable.get(locId);
195                        if (map == null) {
196                                map = new HashMap<>();
197                                shuntingTable.put(locId, map);
198                        }
199                        map.put(nodeId, shuntingAllowed);
200                }
201                return shuntingTable;
202        }

206        public static final Map<Id<TransitStopFacility>, Double> parseShuntingTimes(String shuntingTimesFile) throws IOException {
207                Map<Id<TransitStopFacility>, Double> minShuntingTimes = new HashMap<>();
208                BufferedReader br = IOUtils.getBufferedReader(shuntingTimesFile);
209                String curr_line;
210
211                // read header an build lookup
212                curr_line = br.readLine();
213                String [] header = curr_line.split("\t");
214                Map<String, Integer> lookup = new LinkedHashMap<String,Integer>(header.length);
215                for (int i=0; i<header.length; i++) { lookup.put(Utils.removeSurroundingQuotes(header[i].trim()),i); }
216
217                // parse rows
218                while ((curr_line = br.readLine()) != null) {
219                        String [] row = curr_line.split("\t");
220                        Id<TransitStopFacility> stationId = Id.create(Utils.removeSurroundingQuotes(row[lookup.get(WagonSimConstants.STOP_STATION_ID)].trim()), TransitStopFacility.class);
221                        Double minShuntingTime = new Double(Utils.removeSurroundingQuotes(row[lookup.get(WagonSimConstants.STOP_MIN_SHUNTING_TIME)].trim()));
222                        minShuntingTimes.put(stationId, minShuntingTime);
223                }
224                return minShuntingTimes;
225        }

229        public static final void writeShuntingTimes(Scenario scenario, Map<Id<TransitStopFacility>,Double> shuntingTimes, String outputFile) throws IOException {
230                BufferedWriter bw = IOUtils.getBufferedWriter(outputFile);
231                bw.write(WagonSimConstants.STOP_STATION_ID+"\t"+WagonSimConstants.STOP_MIN_SHUNTING_TIME+"\n");
232                for (TransitStopFacility stopFacility : scenario.getTransitSchedule().getFacilities().values()) {
233                        if ((shuntingTimes != null) && shuntingTimes.containsKey(stopFacility.getId())) {
234                                bw.write(stopFacility.getId().toString()+"\t"+shuntingTimes.get(stopFacility.getId())+"\n");
235                        }
236                        else {
237                                bw.write(stopFacility.getId().toString()+"\t"+scenario.getConfig().transitRouter().getAdditionalTransferTime()+"\n");
238                        }
239                }
240                bw.flush();
241                bw.close();
242        }

246        public static final Config getDefaultWagonSimConfig() {
247                Config config = ConfigUtils.createConfig();
248                config.global().setNumberOfThreads(1);
249                config.transit().setUseTransit(true);
250                config.scenario().setUseVehicles(true);
251                config.controler().setFirstIteration(0);
252                // don't know which number is good here. From my point of view
253                // this should be configurable from outside // dr, oct'13
254                config.controler().setLastIteration(50);
255                config.controler().setWriteEventsInterval(1);
256                config.controler().setWritePlansInterval(1);
257                config.linkStats().setAverageLinkStatsOverIterations(0);
258                config.linkStats().setWriteLinkStatsInterval(1);
259                config.transitRouter().setMaxBeelineWalkConnectionDistance(0.0);
260                config.transitRouter().setAdditionalTransferTime(15*60); // default minimum shunting time
261                config.transitRouter().setSearchRadius(0);
262                config.transitRouter().setExtensionRadius(0);
263                // should be very slow, otherwise we will get a walk-connection
264                config.plansCalcRoute().setTeleportedModeSpeed(TransportMode.walk, 0.000001);
265
266                StrategySettings settings = new StrategySettings(Id.create("1", StrategySettings.class));
267                // I think we should use SelectRandom here, as I'm not really sure what will
268                // happen when we use ChangeExpBeta (own scoring). //dr, oct'13
269                settings.setStrategyName(DefaultPlanStrategiesModule.DefaultSelector.SelectRandom.toString());
270                settings.setWeight(0.8);
271                config.strategy().addStrategySettings(settings);
272                settings = new StrategySettings(Id.create("2", StrategySettings.class));
273                settings.setStrategyName(DefaultPlanStrategiesModule.DefaultStrategy.ReRoute.toString());
274                settings.setWeight(0.2);
275                config.strategy().addStrategySettings(settings);
276                config.strategy().setFractionOfIterationsToDisableInnovation(0.8);
277                config.strategy().setMaxAgentPlanMemorySize(5);
278                
279                ActivityParams params = new ActivityParams(WagonSimConstants.ORIGIN);
280                params.setTypicalDuration(12*3600);
281                config.planCalcScore().addActivityParams(params);
282                params = new ActivityParams(WagonSimConstants.DESTINATION);
283                params.setTypicalDuration(12*3600);
284                config.planCalcScore().addActivityParams(params);
285
286                config.qsim().setStartTime(0.0);
287                config.qsim().setEndTime(4*24*3600.0);
288                config.qsim().setStuckTime(3600.0);
289                return config;
290        }
