001package org.matsim.contrib.eventsBasedPTRouter.vehicleOccupancy;
002
003public class VehicleOccupancyDataArray implements VehicleOccupancyData {
004
005        //Attributes
006        private double[] vehicleOccupancies;
007        private int[] numTimes;
008        private int lastPosition;
009
010        //Constructors
011        public VehicleOccupancyDataArray(int numSlots) {
012                vehicleOccupancies = new double[numSlots];
013                numTimes = new int[numSlots];
014                resetVehicleOccupancies();
015        }
016
017        //Methods
018        @Override
019        public double getVehicleOccupancy(int timeSlot) {
020                if(timeSlot>lastPosition)
021                        return 1;
022                else
023                        while(vehicleOccupancies[timeSlot]==-1)
024                                timeSlot++;
025                return vehicleOccupancies[timeSlot];
026        }
027        @Override
028        public synchronized void addVehicleOccupancy(int timeSlot, double vehicleOccupancy) {
029                if(lastPosition<timeSlot)
030                        lastPosition = timeSlot;
031                vehicleOccupancies[timeSlot] = ((vehicleOccupancies[timeSlot]!=-1?vehicleOccupancies[timeSlot]*numTimes[timeSlot]:0)+vehicleOccupancy)/++numTimes[timeSlot];            
032        }
033        @Override
034        public void resetVehicleOccupancies() {
035                for(int i=0; i<vehicleOccupancies.length; i++) {
036                        vehicleOccupancies[i] = -1;
037                        numTimes[i] = 0;
038                }
039                lastPosition = -1;
040        }
041
042}
