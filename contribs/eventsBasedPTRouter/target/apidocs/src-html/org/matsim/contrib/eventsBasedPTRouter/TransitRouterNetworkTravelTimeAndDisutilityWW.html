041/**
042 * TravelTime and TravelDisutility calculator to be used with the transit network used for transit routing.
043 * This version only considers waiting times at stops, and reads travel time between stops from the links
044 * that make up a route.
045 *
046 * @author sergioo
047 */
048public class TransitRouterNetworkTravelTimeAndDisutilityWW extends TransitRouterNetworkTravelTimeAndDisutility implements TravelDisutility {
049
050        private final Map<Id<Link>, double[]> linkTravelTimes = new HashMap<Id<Link>, double[]>();
051        private final Map<Id<Link>, double[]> linkWaitingTimes = new HashMap<Id<Link>, double[]>();
052        private final double timeSlot;
053        private final int numSlots;
054        private Link previousLink;
055        private double previousTime;
056        private double cachedTravelTime;
057
058        public TransitRouterNetworkTravelTimeAndDisutilityWW(final TransitRouterConfig config, Network network, TransitRouterNetworkWW routerNetwork, TravelTime travelTime, WaitTime waitTime, TravelTimeCalculatorConfigGroup tTConfigGroup, QSimConfigGroup qSimConfigGroup, PreparedTransitSchedule preparedTransitSchedule) {
059                this(config, network, routerNetwork, travelTime, waitTime, tTConfigGroup, qSimConfigGroup.getStartTime(), qSimConfigGroup.getEndTime(), preparedTransitSchedule);
060        }
061        public TransitRouterNetworkTravelTimeAndDisutilityWW(final TransitRouterConfig config, Network network, TransitRouterNetworkWW routerNetwork, TravelTime travelTime, WaitTime waitTime, TravelTimeCalculatorConfigGroup tTConfigGroup, double startTime, double endTime, PreparedTransitSchedule preparedTransitSchedule) {
062                super(config, preparedTransitSchedule);
063                timeSlot = tTConfigGroup.getTraveltimeBinSize();
064                numSlots = (int) ((endTime-startTime)/timeSlot);
065                for(TransitRouterNetworkWW.TransitRouterNetworkLink link:routerNetwork.getLinks().values())
066                        if(link.route!=null) {
067                                double[] times = new double[numSlots];
068                                for(int slot = 0; slot<numSlots; slot++) {
069                                        double linksTime = travelTime.getLinkTravelTime(network.getLinks().get(link.fromNode.stop.getStopFacility().getLinkId()), startTime+slot*timeSlot, null, null);
070                                        for(Id<Link> linkId:link.route.getRoute().getSubRoute(link.fromNode.stop.getStopFacility().getLinkId(), link.toNode.stop.getStopFacility().getLinkId()).getLinkIds())
071                                                linksTime += travelTime.getLinkTravelTime(network.getLinks().get(linkId), startTime+slot*timeSlot, null, null);
072                                        times[slot] = linksTime;
073                                }
074                                linkTravelTimes.put(link.getId(), times);
075                        }
076                        else if(link.toNode.route!=null) {
077                                double[] times = new double[numSlots];
078                                for(int slot = 0; slot<numSlots; slot++)
079                                        times[slot] = waitTime.getRouteStopWaitTime(link.toNode.line.getId(), link.toNode.route.getId(), link.fromNode.stop.getStopFacility().getId(), startTime+slot*timeSlot);
080                                linkWaitingTimes.put(link.getId(), times);
081                        }
082        }
083        @Override
084        public double getLinkTravelTime(final Link link, final double time, Person person, Vehicle vehicle) {
085                previousLink = link;
086                previousTime = time;
087                TransitRouterNetworkWW.TransitRouterNetworkLink wrapped = (TransitRouterNetworkWW.TransitRouterNetworkLink) link;
088                if (wrapped.route != null)
089                        //in line link
090                        cachedTravelTime = linkTravelTimes.get(wrapped.getId())[time/timeSlot<numSlots?(int)(time/timeSlot):(numSlots-1)];
091                else if(wrapped.toNode.route!=null)
092                        //wait link
093                        cachedTravelTime = linkWaitingTimes.get(wrapped.getId())[time/timeSlot<numSlots?(int)(time/timeSlot):(numSlots-1)];
094                else if(wrapped.fromNode.route==null)
095                        //walking link
096                        cachedTravelTime = wrapped.getLength()/this.config.getBeelineWalkSpeed();
097                else
098                        //inside link
099                        cachedTravelTime = 0;
100                return cachedTravelTime;
101        }
102        @Override
103        public double getLinkTravelDisutility(final Link link, final double time, final Person person, final Vehicle vehicle, final CustomDataManager dataManager) {
104                boolean cachedTravelDisutility = false; 
105                if(previousLink==link && previousTime==time)
106                        cachedTravelDisutility = true;
107                TransitRouterNetworkWW.TransitRouterNetworkLink wrapped = (TransitRouterNetworkWW.TransitRouterNetworkLink) link;
108                if (wrapped.route != null)
109                        return -(cachedTravelDisutility?cachedTravelTime:linkTravelTimes.get(wrapped.getId())[time/timeSlot<numSlots?(int)(time/timeSlot):(numSlots-1)]) * this.config.getMarginalUtilityOfTravelTimePt_utl_s() 
110                                       - link.getLength() * this.config.getMarginalUtilityOfTravelDistancePt_utl_m();
111                else if (wrapped.toNode.route!=null)
112                        // it's a wait link
113                        return -(cachedTravelDisutility?cachedTravelTime:linkWaitingTimes.get(wrapped.getId())[time/timeSlot<numSlots?(int)(time/timeSlot):(numSlots-1)]) * this.config.getMarginalUtilityOfWaitingPt_utl_s()
114                                        - this.config.getUtilityOfLineSwitch_utl();
115                else if(wrapped.fromNode.route==null)
116                        // it's a transfer link (walk)
117                        return -(cachedTravelDisutility?cachedTravelTime:wrapped.getLength()/this.config.getBeelineWalkSpeed())*this.config.getMarginalUtilityOfTravelTimeWalk_utl_s();
118                else
119                        //inside link
120                        return 0;
121        }
122        @Override
123        public double getLinkTravelDisutility(Link link, double time, Person person, Vehicle vehicle) {
124                TransitRouterNetworkWW.TransitRouterNetworkLink wrapped = (TransitRouterNetworkWW.TransitRouterNetworkLink) link;
125                if (wrapped.route != null)
126                        return - linkTravelTimes.get(wrapped.getId())[time/timeSlot<numSlots?(int)(time/timeSlot):(numSlots-1)] * this.config.getMarginalUtilityOfTravelTimePt_utl_s() 
127                                        - link.getLength() * this.config.getMarginalUtilityOfTravelDistancePt_utl_m();
128                else if (wrapped.toNode.route!=null)
129                        // it's a wait link
130                        return - linkWaitingTimes.get(wrapped.getId())[time/timeSlot<numSlots?(int)(time/timeSlot):(numSlots-1)] * this.config.getMarginalUtilityOfWaitingPt_utl_s()
131                                        - this.config.getUtilityOfLineSwitch_utl();
132                else if(wrapped.fromNode.route==null)
133                        // it's a transfer link (walk)
134                        return -wrapped.getLength()/this.config.getBeelineWalkSpeed() * this.config.getMarginalUtilityOfTravelTimeWalk_utl_s();
135                else
136                        //inside link
137                        return 0;
138        }
139        @Override
140        public double getLinkMinimumTravelDisutility(Link link) {
141                return 0;
142        }
