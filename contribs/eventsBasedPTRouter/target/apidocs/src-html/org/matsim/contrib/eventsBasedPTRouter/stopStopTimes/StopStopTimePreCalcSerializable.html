018/**
019 * <B>When StopStopTimes are generated externally, e.g. by a regression.</B>
020 * <p></p>
021 * See constructor for input format.
022 * </P>
023 */
024public class StopStopTimePreCalcSerializable implements StopStopTime, Serializable {
025
026    private final Map<String, Map<String, Map<Integer, Tuple<Double, Double>>>> stopStopTimes = new HashMap<>(5000);
027    private final Scenario scenario;
028
029    public boolean isLogarithmic() {
030        return logarithmic;
031    }
032
033    private final boolean logarithmic;
034    private int nfeCounter = 0;
035    private int aiobCounter = 0;
036    private int errorCounter = 0;
037
038    //Constructors
039

041    /**
042     * Populates a StopStopTime object with stop to stop travel times and their variances
043     * recorded at particular times.
044     * Needs at minimum one record per stop-to-stop combination.
045     * Times at which records are recorded needn't be regular intervals, because a TreeMap is used.
046     *
047     * @param inputFile   path to the tab-separated file.Format is fromStopId (String), toStopId (String), time of record (seconds, double),
048     *                    travelTime (seconds, double), travelTimeVariance (seconds**2, double). No headings or row numbers.
049     * @param logarithmic if times are recorded as logarithms (normally distributed residuals)
050     */
051    public StopStopTimePreCalcSerializable(String inputFile, Scenario scenario, boolean logarithmic) {
052        this.logarithmic = logarithmic;
053        this.scenario = scenario;
054        BufferedReader reader = IOUtils.getBufferedReader(inputFile);
055        String txt = "";
056        while (true) {
057            try {
058                txt = reader.readLine();
059                if (txt == null)
060                    break;
061                String[] split = txt.split("\t");
062//                get the map from this stop id
063                Map<String, Map<Integer, Tuple<Double, Double>>> toMap = stopStopTimes.get(split[0]);
064                if (toMap == null) {
065                    toMap = new HashMap<>();
066                    stopStopTimes.put(split[0], toMap);
067                }
068
069                Map<Integer, Tuple<Double, Double>> timeData = toMap.get(split[1]);
070                if (timeData == null) {
071                    timeData = new TreeMap<>();
072                    toMap.put(split[1], timeData);
073                }
074
075                timeData.put(Integer.parseInt(split[2]), new Tuple<>(Double.parseDouble(split[3]), Double.parseDouble(split[4])));
076
077            } catch (IOException e) {
078                e.printStackTrace();
079            } catch (ArrayIndexOutOfBoundsException e) {
080                if (aiobCounter < 10) {
081                    System.err.println("Seems one of the lines in the StopStopTime input file is missing a value. Skipping it.");
082                } else {
083                    System.err.println("Seems one of the lines in the StopStopTime input file is missing a value. Skipping further warnings...");
084
085                }
086                aiobCounter++;
087            } catch (NumberFormatException e) {
088                if (nfeCounter < 10) {
089                    System.err.println("Some values in the StopStopTime input file are of the wrong type. Skipping it.");
090                } else {
091                    System.err.println("Some values in the StopStopTime input file are of the wrong type. Skipping further warnings...");
092                }
093                nfeCounter++;
094            }
095        }
096        System.out.println("\n\n*************************************************************************\n\n");
097        System.out.println("STOPSTOPTIMES LOADED, filling in the blanks from the schedule and network");
098        System.out.println("\n\n*************************************************************************\n\n");
099
100        for (TransitLine transitLine : scenario.getTransitSchedule().getTransitLines().values()) {
101            for (TransitRoute route : transitLine.getRoutes().values()) {
102                TRANSITSTOPS:
103                for (int s = 0; s < route.getStops().size() - 1; s++) {
104                    String origin = route.getStops().get(s).getStopFacility().getId().toString();
105                    String destination = route.getStops().get(s + 1).getStopFacility().getId().toString();
106                    Map<String, Map<Integer, Tuple<Double, Double>>> toMap = stopStopTimes.get(origin);
107                    if (toMap == null) {
108                        toMap = new HashMap<>();
109                        stopStopTimes.put(origin, toMap);
110                    }
111
112                    Map<Integer, Tuple<Double, Double>> timeData = toMap.get(destination);
113                    if (timeData == null) {
114                        timeData = new TreeMap<>();
115                        toMap.put(destination, timeData);
116
117                        try {
118                            List<TransitRouteStop> stops;
119                            stops = scenario.getTransitSchedule().getTransitLines().get(transitLine.getId()).getRoutes().get(route.getId()).getStops();
120                            if (stops == null)
121                                throw new NullPointerException();
122                            Link fromLink = null;
123                            Link toLink = null;
124                            for (TransitRouteStop tss : stops) {
125                                if (tss.getStopFacility().getId().toString().equals(origin)) {
126                                    fromLink = scenario.getNetwork().getLinks().get(tss.getStopFacility().getLinkId());
127                                }
128                                if (tss.getStopFacility().getId().toString().equals(destination)) {
129                                    toLink = scenario.getNetwork().getLinks().get(tss.getStopFacility().getLinkId());
130                                }
131                            }
132                            if (fromLink == null || toLink == null)
133                                throw new NullPointerException();
134                            NetworkRoute networkRoute = scenario.getTransitSchedule().getTransitLines().get(transitLine.getId())
135                                    .getRoutes().get(route.getId()).getRoute();
136                            NetworkRoute subRoute = networkRoute.getSubRoute(fromLink.getId(), toLink.getId());
137                            List<Id<Link>> linkIds = new ArrayList<>();
138                            linkIds.addAll(subRoute.getLinkIds());
139                            linkIds.add(toLink.getId());
140                            double freeSpeedTravelTime = 0;
141                            for (Id<Link> id : linkIds) {
142                                Link link = scenario.getNetwork().getLinks().get(id);
143                                freeSpeedTravelTime += link.getLength() / link.getFreespeed();
144                            }
145                            if (logarithmic)
146                                timeData.put(0, new Tuple<Double, Double>(Math.log(freeSpeedTravelTime), 0.005));
147                            else
148                                timeData.put(0, new Tuple<Double, Double>(freeSpeedTravelTime, 0.0));
149
150                        } catch (NullPointerException ne) {
151                            System.err.printf("Couldnt create STOP-STOP entry for for from: %s, to: %s, route: %s, line: %s\n",
152                                    origin, destination, route.getId().toString(), transitLine.getId().toString());
153                            continue TRANSITSTOPS;
154                        }
155
156                    } else {
157                        //value already exists (assuming singular stop-stop network routes)
158                        continue TRANSITSTOPS;
159                    }
160
161                }
162            }
163        }
164    }

167    /**
168     * Retrieves the interpolated travel time for a combination of stops at a particular time, or the first/last variance observation if one cannot be  interpolated.
169     * <p></p>
170     * Returns the error value of infinity for a compbination that doesn't appear in the map
171     *
172     * @param stopOId
173     * @param stopDId
174     * @param time
175     * @return the travel time, or Double.POSITIVE_INFINITY
176     */
177    public double getStopStopTime(Id stopOId, Id stopDId, double time) {
178        Map<String, Map<Integer, Tuple<Double, Double>>> toMap = stopStopTimes.get(stopOId.toString());
179
180        if (toMap == null) {
181            return errorValue(stopOId, stopDId);
182        }
183
184        TreeMap<Integer, Tuple<Double, Double>> timedObservations = (TreeMap<Integer, Tuple<Double, Double>>) toMap.get(stopDId.toString());
185
186        if (timedObservations == null) {
187            return errorValue(stopOId, stopDId);
188        }
189
190        Map.Entry<Integer, Tuple<Double, Double>> ceilingEntry = timedObservations.ceilingEntry((int) time);
191        Map.Entry<Integer, Tuple<Double, Double>> floorEntry = timedObservations.floorEntry((int) time);
192        if (ceilingEntry == null) {
193            if (floorEntry == null) {
194                return errorValue(stopOId, stopDId);
195            } else {
196                return floorEntry.getValue().getFirst();
197            }
198        } else {
199            if (floorEntry == null) {
200                return ceilingEntry.getValue().getFirst();
201            } else {
202                // I have both, so can interpolate a travel time
203                double x1 = floorEntry.getKey();
204                double x2 = ceilingEntry.getKey();
205                double y1 = floorEntry.getValue().getFirst();
206                double y2 = ceilingEntry.getValue().getFirst();
207                double m = (y2 - y1) / (x2 - x1);
208                double x = time - x1;
209                return y1 + m * x;
210            }
211        }
212    }
213
214    private double errorValue(Id stopOId, Id stopDId) {
215        if (errorCounter < 10) {
216            System.err.println("No StopStop data for origin stop " + stopOId.toString() + ", destination stop " + stopDId.toString() + ". Returning estimation...");
217        } else if (errorCounter == 10) {
218            System.err.println("No StopStop data for origin stop " + stopOId.toString() + ". Skipping further warnings...");
219        }
220        errorCounter++;
221
222        return Double.POSITIVE_INFINITY;
223    }

225    /**
226     * Retrieves the interpolated travel time variance for a combination of stops at a particular time, or the first/last variance observation if one cannot be  interpolated.
227     * <p></p>
228     * Returns the error value of infinity for a compbination that doesn't appear in the map
229     *
230     * @param stopOId
231     * @param stopDId
232     * @param time    seconds till the start of the sim
233     * @return the travel time, or Double.POSITIVE_INFINITY
234     */
235    public double getStopStopTimeVariance(Id stopOId, Id stopDId, double time) {
236        Map<String, Map<Integer, Tuple<Double, Double>>> toMap = stopStopTimes.get(stopOId.toString());
237
238        if (toMap == null) {
239            return errorValue(stopOId, stopDId);
240        }
241
242        TreeMap<Integer, Tuple<Double, Double>> timedObservations = (TreeMap<Integer, Tuple<Double, Double>>) toMap.get(stopDId.toString());
243
244        if (timedObservations == null) {
245            return errorValue(stopOId, stopDId);
246        }
247
248        Map.Entry<Integer, Tuple<Double, Double>> ceilingEntry = timedObservations.ceilingEntry((int) time);
249        Map.Entry<Integer, Tuple<Double, Double>> floorEntry = timedObservations.floorEntry((int) time);
250        if (ceilingEntry == null) {
251            if (floorEntry == null) {
252                return 0;
253            } else {
254                return floorEntry.getValue().getSecond();
255            }
256        } else {
257            if (floorEntry == null) {
258                return ceilingEntry.getValue().getSecond();
259            } else {
260                // I have both, so can interpolate a travel time
261                double x1 = floorEntry.getKey();
262                double x2 = ceilingEntry.getKey();
263                double y1 = floorEntry.getValue().getSecond();
264                double y2 = ceilingEntry.getValue().getSecond();
265                double m = (y2 - y1) / (x2 - x1);
266                double x = time - x1;
267                return y1 + m * x;
268            }
269        }
270    }
