001package org.matsim.contrib.eventsBasedPTRouter.stopStopTimes;
002
003public class StopStopTimeDataArray implements StopStopTimeData {
004
005        /**
006         * 
007         */
008        private static final long serialVersionUID = 1L;
009        //Attributes
010        private double[] stopStopTimeMeans;
011        private double[] stopStopTimeSquares;
012        private int[] numTimes;
013
014        //Constructors
015        public StopStopTimeDataArray(int numSlots) {
016                stopStopTimeSquares = new double[numSlots];
017                stopStopTimeMeans = new double[numSlots];
018                numTimes = new int[numSlots];
019                resetStopStopTimes();
020        }
021
022        //Methods
023        @Override
024        public int getNumData(int timeSlot) {
025                return numTimes[timeSlot<stopStopTimeMeans.length?timeSlot:(stopStopTimeMeans.length-1)];
026        }
027        @Override
028        public double getStopStopTime(int timeSlot) {
029                return stopStopTimeMeans[timeSlot<stopStopTimeMeans.length?timeSlot:(stopStopTimeMeans.length-1)];
030        }
031        @Override
032        public double getStopStopTimeVariance(int timeSlot) {
033                int index = timeSlot<stopStopTimeMeans.length?timeSlot:(stopStopTimeMeans.length-1);
034                return stopStopTimeSquares[index]-Math.pow(stopStopTimeMeans[index], 2);
035        }
036        @Override
037        public synchronized void addStopStopTime(int timeSlot, double stopStopTime) {
038                stopStopTimeSquares[timeSlot] = (stopStopTimeSquares[timeSlot]*numTimes[timeSlot]+Math.pow(stopStopTime, 2))/(numTimes[timeSlot]+1);
039                stopStopTimeMeans[timeSlot] = (stopStopTimeMeans[timeSlot]*numTimes[timeSlot]+stopStopTime)/++numTimes[timeSlot];               
040        }
041        @Override
042        public void resetStopStopTimes() {
043                for(int i=0; i<stopStopTimeMeans.length; i++) {
044                        stopStopTimeMeans[i] = 0;
045                        numTimes[i] = 0;
046                }
047        }
048
049}
