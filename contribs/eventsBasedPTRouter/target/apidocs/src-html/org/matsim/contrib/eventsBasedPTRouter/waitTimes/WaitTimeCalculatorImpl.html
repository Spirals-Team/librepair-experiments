047/**
048 * Save waiting times of agents while mobsim is running
049 * 
050 * @author sergioo
051 */
052
053@Singleton
054public class WaitTimeCalculatorImpl implements WaitTimeCalculator, PersonDepartureEventHandler, PersonEntersVehicleEventHandler, TransitDriverStartsEventHandler, VehicleArrivesAtFacilityEventHandler, Provider<WaitTime> {
055
056        //Attributes
057        private final double timeSlot;
058        private final Map<Tuple<Id<TransitLine>, Id<TransitRoute>>, Map<Id<TransitStopFacility>, WaitTimeData>> waitTimes = new HashMap<Tuple<Id<TransitLine>, Id<TransitRoute>>, Map<Id<TransitStopFacility>, WaitTimeData>>(1000);
059        private final Map<Tuple<Id<TransitLine>, Id<TransitRoute>>, Map<Id<TransitStopFacility>, double[]>> scheduledWaitTimes = new HashMap<Tuple<Id<TransitLine>, Id<TransitRoute>>, Map<Id<TransitStopFacility>, double[]>>(1000);
060        private final Map<Id<Person>, Double> agentsWaitingData = new HashMap<Id<Person>, Double>();
061        private Map<Id<Vehicle>, Tuple<Id<TransitLine>, Id<TransitRoute>>> linesRoutesOfVehicle = new HashMap<Id<Vehicle>, Tuple<Id<TransitLine>, Id<TransitRoute>>>();
062        private Map<Id<Vehicle>, Id<TransitStopFacility>> stopOfVehicle = new HashMap<Id<Vehicle>, Id<TransitStopFacility>>();
063        
064        //Constructors
065        @Inject
066        public WaitTimeCalculatorImpl(final TransitSchedule transitSchedule, final Config config, EventsManager eventsManager) {
067                this(transitSchedule, config.travelTimeCalculator().getTraveltimeBinSize(), (int) (config.qsim().getEndTime()-config.qsim().getStartTime()));
068                eventsManager.addHandler(this);
069        }
070        public WaitTimeCalculatorImpl(final TransitSchedule transitSchedule, final int timeSlot, final int totalTime) {
071                this.timeSlot = timeSlot;
072                for(TransitLine line:transitSchedule.getTransitLines().values())
073                        for(TransitRoute route:line.getRoutes().values()) {
074                                double[] sortedDepartures = new double[route.getDepartures().size()];
075                                int d=0;
076                                for(Departure departure:route.getDepartures().values())
077                                        sortedDepartures[d++] = departure.getDepartureTime();
078                                Arrays.sort(sortedDepartures);
079                                Map<Id<TransitStopFacility>, WaitTimeData> stopsMap = new HashMap<Id<TransitStopFacility>, WaitTimeData>(100);
080                                Map<Id<TransitStopFacility>, double[]> stopsScheduledMap = new HashMap<Id<TransitStopFacility>, double[]>(100);
081                                for(TransitRouteStop stop:route.getStops()) {
082                                        stopsMap.put(stop.getStopFacility().getId(), new WaitTimeDataArray((int) (totalTime/timeSlot)+1));
083                                        double[] cacheWaitTimes = new double[(int) (totalTime/timeSlot)+1];
084                                        for(int i=0; i<cacheWaitTimes.length; i++) {
085                                                double endTime = timeSlot*(i+1);
086                                                if(endTime>24*3600)
087                                                        endTime-=24*3600;
088                                                cacheWaitTimes[i] = Time.UNDEFINED_TIME;
089                                                SORTED_DEPARTURES:
090                                                for(double departure:sortedDepartures) {
091                                                        double arrivalTime = departure+(stop.getArrivalOffset()!=Time.UNDEFINED_TIME?stop.getArrivalOffset():stop.getDepartureOffset()); 
092                                                        if(arrivalTime>=endTime) {
093                                                                cacheWaitTimes[i] = arrivalTime-endTime;
094                                                                break SORTED_DEPARTURES;
095                                                        }
096                                                }
097                                                if(cacheWaitTimes[i]==Time.UNDEFINED_TIME)
098                                                        cacheWaitTimes[i] = sortedDepartures[0]+24*3600+(stop.getArrivalOffset()!=Time.UNDEFINED_TIME?stop.getArrivalOffset():stop.getDepartureOffset())-endTime;
099                                        }
100                                        stopsScheduledMap.put(stop.getStopFacility().getId(), cacheWaitTimes);
101                                }
102                                Tuple<Id<TransitLine>, Id<TransitRoute>> key = new Tuple<Id<TransitLine>, Id<TransitRoute>>(line.getId(), route.getId());
103                                waitTimes.put(key, stopsMap);
104                                scheduledWaitTimes.put(key, stopsScheduledMap);
105                        }
106        }
107
108        //Methods
109        @Override
110        public WaitTime get() {
111                return new WaitTime() {
112                        /**
113                         *
114                         */
115                        private static final long serialVersionUID = 1L;
116
117                        @Override
118                        public double getRouteStopWaitTime(Id<TransitLine> lineId, Id<TransitRoute> routeId, Id<TransitStopFacility> stopId, double time) {
119                                return WaitTimeCalculatorImpl.this.getRouteStopWaitTime(lineId, routeId, stopId, time);
120                        }
121                };
122        }
123        @Override
124        public double getRouteStopWaitTime(Id<TransitLine> lineId, Id<TransitRoute> routeId, Id<TransitStopFacility> stopId, double time) {
125                Tuple<Id<TransitLine>, Id<TransitRoute>> key = new Tuple<Id<TransitLine>, Id<TransitRoute>>(lineId, routeId);
126                WaitTimeData waitTimeData = waitTimes.get(key).get(stopId);
127                if(waitTimeData.getNumData((int) (time/timeSlot))==0) {
128                        double[] waitTimes = scheduledWaitTimes.get(key).get(stopId);
129                        return waitTimes[(int) (time/timeSlot)<waitTimes.length?(int) (time/timeSlot):(waitTimes.length-1)];
130                }
131                else
132                        return waitTimeData.getWaitTime((int) (time/timeSlot));
133        }
134        @Override
135        public void reset(int iteration) {
136                for(Map<Id<TransitStopFacility>, WaitTimeData> routeData:waitTimes.values())
137                        for(WaitTimeData waitTimeData:routeData.values())
138                                waitTimeData.resetWaitTimes();
139                agentsWaitingData.clear();
140                linesRoutesOfVehicle.clear();
141                stopOfVehicle.clear();
142        }
143        @Override
144        public void handleEvent(PersonDepartureEvent event) {
145                if(event.getLegMode().equals("pt") && agentsWaitingData.get(event.getPersonId())==null)
146                        agentsWaitingData.put(event.getPersonId(), event.getTime());
147                else if(agentsWaitingData.get(event.getPersonId())!=null)
148                        new RuntimeException("Departing with old data");
149        }
150        @Override
151        public void handleEvent(PersonEntersVehicleEvent event) {
152                Double startWaitingTime = agentsWaitingData.get(event.getPersonId());
153                if(startWaitingTime!=null) {
154                        Tuple<Id<TransitLine>, Id<TransitRoute>> lineRoute = linesRoutesOfVehicle.get(event.getVehicleId());
155                        WaitTimeData data = waitTimes.get(lineRoute).get(stopOfVehicle.get(event.getVehicleId()));
156                        data.addWaitTime((int) (startWaitingTime/timeSlot), event.getTime()-startWaitingTime);
157                        agentsWaitingData.remove(event.getPersonId());
158                }
159        }
160
161        @Override
162        public void handleEvent(VehicleArrivesAtFacilityEvent event) {
163                if(linesRoutesOfVehicle.get(event.getVehicleId())!=null)
164                        stopOfVehicle.put(event.getVehicleId(), event.getFacilityId());
165        }
166
167        @Override
168        public void handleEvent(TransitDriverStartsEvent event) {
169                linesRoutesOfVehicle.put(event.getVehicleId(), new Tuple<Id<TransitLine>, Id<TransitRoute>>(event.getTransitLineId(), event.getTransitRouteId()));
170        }
