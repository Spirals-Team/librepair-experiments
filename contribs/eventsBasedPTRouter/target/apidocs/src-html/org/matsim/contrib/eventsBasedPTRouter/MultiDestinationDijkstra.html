044/**
045 * Implementation of <a
046 * href="http://en.wikipedia.org/wiki/Dijkstra%27s_algorithm">Dijkstra's
047 * shortest-path algorithm</a> on a time-dependent network with arbitrary
048 * non-negative cost functions (e.g. negative link cost are not allowed). So
049 * 'shortest' in our context actually means 'least-cost'.
050 *
051 * <p>
052 * For every router, there exists a class which computes some preprocessing data
053 * and is passed to the router class constructor in order to accelerate the
054 * routing procedure. The one used for Dijkstra is
055 * {@link org.matsim.core.router.util.PreProcessDijkstra}.
056 * </p>
057 * <br>
058 *
059 * <h2>Code example:</h2>
060 * <p>
061 * <code>PreProcessDijkstra preProcessData = new PreProcessDijkstra();<br>
062 * preProcessData.run(network);<br>
063 * TravelCost costFunction = ...<br>
064 * LeastCostPathCalculator routingAlgo = new Dijkstra(network, costFunction, preProcessData);<br>
065 * routingAlgo.calcLeastCostPath(fromNode, toNode, startTime);</code>
066 * </p>
067 * <p>
068 * If you don't want to preprocess the network, you can invoke Dijkstra as
069 * follows:
070 * </p>
071 * <p>
072 * <code> LeastCostPathCalculator routingAlgo = new Dijkstra(network, costFunction);</code>
073 * </p>
074 * 
075 * <h2>Important note</h2>
076 * This class is NOT threadsafe!
077 *
078 * @see org.matsim.core.router.util.PreProcessDijkstra
079 * @see org.matsim.core.router.AStarEuclidean
080 * @see org.matsim.core.router.AStarLandmarks
081 * @author lnicolas
082 * @author mrieser
083 */
084public class MultiDestinationDijkstra {

150        /**
151         * Constructor.
152         *
153         * @param network
154         *            The network on which to route.
155         * @param costFunction
156         *            Determines the link cost defining the cheapest route.
157         * @param timeFunction
158         *            Determines the travel time on each link.
159         * @param preProcessData
160         *            The pre processing data used during the routing phase.
161         */
162        public MultiDestinationDijkstra(final Network network, final TransitTravelDisutility costFunction, final TravelTime timeFunction,
163                        final PreProcessDijkstra preProcessData) {
164
165                this.network = network;
166                this.costFunction = costFunction;
167                this.timeFunction = timeFunction;
168                this.preProcessData = preProcessData;
169
170                this.nodeData = new HashMap<Id<Node>, DijkstraNodeData>((int)(network.getNodes().size() * 1.1), 0.95f);
171
172                if (preProcessData != null) {
173                        if (preProcessData.containsData() == false) {
174                                this.pruneDeadEnds = false;
175                                log.warn("The preprocessing data provided to router class Dijkstra contains no data! Please execute its run(...) method first!");
176                                log.warn("Running without dead-end pruning.");
177                        } else {
178                                this.pruneDeadEnds = true;
179                        }
180                } else {
181                        this.pruneDeadEnds = false;
182                }
183        }

189                        this.modeRestriction = modeRestriction.toArray(new String[modeRestriction.size()]);
190                }

193        /**
194         * Calculates the cheapest route from Node 'fromNode' to Node 'toNode' at
195         * starting time 'startTime'.
196         *
197         * @param fromNode
198         *            The Node at which the route should start.
199         * @param toNode
200         *            The Node at which the route should end.
201         * @param startTime
202         *            The time at which the route should start. <i>Note:</i> Using
203         *            {@link Time#UNDEFINED_TIME} does not imply "time is not relevant",
204         *            rather, {@link Path#travelTime} will return {@link Double#NaN}.
205         * @see org.matsim.core.router.util.LeastCostPathCalculator#calcLeastCostPath(org.matsim.core.network.Node,
206         *      org.matsim.core.network.Node, double)
207         */
208        public Map<Id<Node>, Path> calcLeastCostPath(final Node fromNode, final Set<Node> toNodes, final double startTime, final Person person) {
209
210                Map<Id<Node>, Double> arrivalTimes = new HashMap<Id<Node>, Double>();
211                boolean stillSearching = true;
212                int finishedNodes = 0;
213                augmentIterationId(); // this call makes the class not threadsafe
214                this.person = person;
215                
216                if (this.pruneDeadEnds == true) {
217                        for(Node toNode:toNodes)
218                                deadEndEntryNodes.add(getPreProcessData(toNode).getDeadEndEntryNode());
219                }
220
221                PseudoRemovePriorityQueue<Node> pendingNodes = new PseudoRemovePriorityQueue<Node>(500);
222                initFromNode(fromNode, startTime, pendingNodes);
223
224                while (stillSearching) {
225                        Node outNode = pendingNodes.poll();
226
227                        if (outNode == null) {
228                                log.warn("No route was found from node " + fromNode.getId() + " to all nodes: " + finishedNodes + " of " + toNodes.size());
229                                stillSearching = false;
230                        }
231                        else {
232                                if (toNodes.contains(outNode)) {
233                                        DijkstraNodeData outData = getData(outNode);
234                                        arrivalTimes.put(outNode.getId(), outData.getTime());
235                                        finishedNodes++;
236                                }
237                                relaxNode(outNode, pendingNodes);
238                                if(finishedNodes==toNodes.size())
239                                        stillSearching = false;
240                        }
241                }
242
243                // now construct and return the path
244                return constructPaths(fromNode, toNodes, startTime, arrivalTimes);
245
246        }
247
248        /**
249         * Constructs the path after the algorithm has been run.
250         *
251         * @param fromNode
252         *            The node where the path starts.
253         * @param toNode
254         *            The node where the path ends.
255         * @param startTime
256         *            The time when the trip starts.
257         * @param preProcessData
258         *            The time when the trip ends.
259         */
260        protected Map<Id<Node>, Path> constructPaths(Node fromNode, Set<Node> toNodes, double startTime, Map<Id<Node>, Double> arrivalTimes) {
261                Map<Id<Node>, Path> paths = new HashMap<Id<Node>, Path>();
262                for(Node toNode:toNodes) {
263                        Double arrivalTime = arrivalTimes.get(toNode.getId());
264                        Path path = null;
265                        if(arrivalTime != null) {
266                                ArrayList<Node> nodes = new ArrayList<Node>();
267                                ArrayList<Link> links = new ArrayList<Link>();
268                
269                                nodes.add(0, toNode);
270                                Link tmpLink = getData(toNode).getPrevLink();
271                                if (tmpLink != null) {
272                                        while (tmpLink.getFromNode() != fromNode) {
273                                                links.add(0, tmpLink);
274                                                nodes.add(0, tmpLink.getFromNode());
275                                                tmpLink = getData(tmpLink.getFromNode()).getPrevLink();
276                                        }
277                                        links.add(0, tmpLink);
278                                        nodes.add(0, tmpLink.getFromNode());
279                                }
280                
281                                DijkstraNodeData toNodeData = getData(toNode);
282                                path = new Path(nodes, links, arrivalTime - startTime, toNodeData.getCost());
283                        }
284                        paths.put(toNode.getId(), path);
285                }
286                return paths;
287        }

289        /**
290         * Initializes the first node of a route.
291         *
292         * @param fromNode
293         *            The Node to be initialized.
294         * @param toNode
295         *            The Node at which the route should end.
296         * @param startTime
297         *            The time we start routing.
298         * @param pendingNodes
299         *            The pending nodes so far.
300         */
301        /*package*/ void initFromNode(final Node fromNode, final double startTime,
302                        final PseudoRemovePriorityQueue<Node> pendingNodes) {
303                DijkstraNodeData data = getData(fromNode);
304                visitNode(fromNode, data, pendingNodes, startTime, 0, null);
305        }

307        /**
308         * Expands the given Node in the routing algorithm; may be overridden in
309         * sub-classes.
310         *
311         * @param outNode
312         *            The Node to be expanded.
313         * @param toNode
314         *            The target Node of the route.
315         * @param pendingNodes
316         *            The set of pending nodes so far.
317         */
318        protected void relaxNode(final Node outNode, final PseudoRemovePriorityQueue<Node> pendingNodes) {
319
320                DijkstraNodeData outData = getData(outNode);
321                double currTime = outData.getTime();
322                double currCost = outData.getCost();
323                if (this.pruneDeadEnds) {
324                        PreProcessDijkstra.DeadEndData ddOutData = getPreProcessData(outNode);
325
326                        for (Link l : outNode.getOutLinks().values()) {
327                                relaxNodeLogic(l, pendingNodes, currTime, currCost, ddOutData);
328                        }
329                } else { // this.pruneDeadEnds == false
330                        for (Link l : outNode.getOutLinks().values()) {
331                                relaxNodeLogic(l, pendingNodes, currTime, currCost, null);
332                        }                               
333                }
334        }
335        
336        /**
337         * Logic that was previously located in the relaxNode(...) method. 
338         * By doing so, the FastDijkstra can overwrite relaxNode without copying the logic. 
339         */
340        /*package*/ void relaxNodeLogic(final Link l, final PseudoRemovePriorityQueue<Node> pendingNodes,
341                        final double currTime, final double currCost,
342                        final PreProcessDijkstra.DeadEndData ddOutData) {
343                if (this.pruneDeadEnds) {
344                        if (canPassLink(l)) {
345                                Node n = l.getToNode();
346                                PreProcessDijkstra.DeadEndData ddData = getPreProcessData(n);
347                                /* IF the current node n is not in a dead end
348                                 * OR it is in the same dead end as the fromNode
349                                 * OR it is in the same dead end as the toNode
350                                 * THEN we add the current node to the pending nodes */
351                                if(ddData.getDeadEndEntryNode()==null || ddOutData.getDeadEndEntryNode()!=null)
352                                        addToPendingNodes(l, n, pendingNodes, currTime, currCost);
353                                else {
354                                        TO_NODES:
355                                        for(Node deadEndEntryNode:deadEndEntryNodes)
356                                                if(deadEndEntryNode != null && deadEndEntryNode.getId() == ddData.getDeadEndEntryNode().getId()) {
357                                                        addToPendingNodes(l, n, pendingNodes, currTime, currCost);
358                                                        break TO_NODES;
359                                                }
360                                }
361                        }
362                } else {
363                        if (canPassLink(l)) {
364                                addToPendingNodes(l, l.getToNode(), pendingNodes, currTime, currCost);
365                        }
366                }
367        }

369        /**
370         * Adds some parameters to the given Node then adds it to the set of pending
371         * nodes.
372         *
373         * @param l
374         *            The link from which we came to this Node.
375         * @param n
376         *            The Node to add to the pending nodes.
377         * @param pendingNodes
378         *            The set of pending nodes.
379         * @param currTime
380         *            The time at which we started to traverse l.
381         * @param currCost
382         *            The cost at the time we started to traverse l.
383         * @param toNode
384         *            The target Node of the route.
385         * @return true if the node was added to the pending nodes, false otherwise
386         *              (e.g. when the same node already has an earlier visiting time).
387         */
388        protected boolean addToPendingNodes(final Link l, final Node n,
389                        final PseudoRemovePriorityQueue<Node> pendingNodes, final double currTime,
390                        final double currCost) {
391
392                double travelTime = this.timeFunction.getLinkTravelTime(l, currTime, person, null);
393                double travelCost = this.costFunction.getLinkTravelDisutility(l, currTime, this.person, null, null);
394                DijkstraNodeData data = getData(n);
395                double nCost = data.getCost();
396                if (!data.isVisited(getIterationId())) {
397                        visitNode(n, data, pendingNodes, currTime + travelTime, currCost
398                                        + travelCost, l);
399                        return true;
400                }
401                double totalCost = currCost + travelCost;
402                if (totalCost < nCost) {
403                        revisitNode(n, data, pendingNodes, currTime + travelTime, totalCost, l);
404                        return true;
405                }
406
407                return false;
408        }

410        /**
411         * @param link
412         * @return <code>true</code> if the link can be passed with respect to a possible mode restriction set
413         *
414         * @see #setModeRestriction(Set)
415         */
416        protected boolean canPassLink(final Link link) {
417                if (this.modeRestriction == null) {
418                        return true;
419                }
420                for (String mode : this.modeRestriction) {
421                        if (link.getAllowedModes().contains(mode)) {
422                                return true;
423                        }
424                }
425                return false;
426        }

428        /**
429         * Changes the position of the given Node n in the pendingNodes queue and
430         * updates its time and cost information.
431         *
432         * @param n
433         *            The Node that is revisited.
434         * @param data
435         *            The data for n.
436         * @param pendingNodes
437         *            The nodes visited and not processed yet.
438         * @param time
439         *            The time of the visit of n.
440         * @param cost
441         *            The accumulated cost at the time of the visit of n.
442         * @param outLink
443         *            The link from which we came visiting n.
444         */
445        void revisitNode(final Node n, final DijkstraNodeData data,
446                        final PseudoRemovePriorityQueue<Node> pendingNodes, final double time, final double cost,
447                        final Link outLink) {
448                pendingNodes.remove(n);
449
450                data.visit(outLink, cost, time, getIterationId());
451                pendingNodes.add(n, getPriority(data));
452        }
453
454        /**
455         * Inserts the given Node n into the pendingNodes queue and updates its time
456         * and cost information.
457         *
458         * @param n
459         *            The Node that is revisited.
460         * @param data
461         *            The data for n.
462         * @param pendingNodes
463         *            The nodes visited and not processed yet.
464         * @param time
465         *            The time of the visit of n.
466         * @param cost
467         *            The accumulated cost at the time of the visit of n.
468         * @param outLink
469         *            The node from which we came visiting n.
470         */
471        protected void visitNode(final Node n, final DijkstraNodeData data,
472                        final PseudoRemovePriorityQueue<Node> pendingNodes, final double time, final double cost,
473                        final Link outLink) {
474                data.visit(outLink, cost, time, getIterationId());
475                pendingNodes.add(n, getPriority(data));
476        }

478        /**
479         * Augments the iterationID and checks whether the visited information in
480         * the nodes in the nodes have to be reset.
481         */
482        protected void augmentIterationId() {
483                if (getIterationId() == Integer.MAX_VALUE) {
484                        this.iterationID = Integer.MIN_VALUE + 1;
485                        resetNetworkVisited();
486                } else {
487                        this.iterationID++;
488                }
489        }
490
491        /**
492         * @return iterationID
493         */
494        /*package*/ int getIterationId() {
495                return this.iterationID;
496        }
497
498        /**
499         * Resets all nodes in the network as if they have not been visited yet.
500         */
501        private void resetNetworkVisited() {
502                for (Node node : this.network.getNodes().values()) {
503                        DijkstraNodeData data = getData(node);
504                        data.resetVisited();
505                }
506        }
507
508        /**
509         * The value used to sort the pending nodes during routing.
510         * This implementation compares the total effective travel cost
511         * to sort the nodes in the pending nodes queue during routing.
512         */
513        protected double getPriority(final DijkstraNodeData data) {
514                return data.getCost();
515        }
516
517        /**
518         * Returns the data for the given node. Creates a new NodeData if none exists
519         * yet.
520         *
521         * @param n
522         *            The Node for which to return the data.
523         * @return The data for the given Node
524         */
525        protected DijkstraNodeData getData(final Node n) {
526                DijkstraNodeData r = this.nodeData.get(n.getId());
527                if (null == r) {
528                        r = new DijkstraNodeData();
529                        this.nodeData.put(n.getId(), r);
530                }
531                return r;
532        }
533
534        protected PreProcessDijkstra.DeadEndData getPreProcessData(final Node n) {
535                return this.preProcessData.getNodeData(n);
536        }
