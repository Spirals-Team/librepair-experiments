138        /**
139         * Looks to me like an implementation of the Link interface, with get(Transit)Route and get(Transit)Line on top.
140         * To recall: TransitLine is something like M44.  But it can have more than one route, e.g. going north, going south,
141         * long route, short route. That is, presumably we have one such TransitRouterNetworkLink per TransitRoute. kai/manuel, feb'12
142         */
143        public static final class TransitRouterNetworkLink implements Link {
144
145                final TransitRouterNetworkNode fromNode;
146                final TransitRouterNetworkNode toNode;
147                final TransitRoute route;
148                final TransitLine line;
149                final Id<Link> id;
150                private double length;
151
152                public TransitRouterNetworkLink(final Id<Link> id, final TransitRouterNetworkNode fromNode, final TransitRouterNetworkNode toNode, final TransitRoute route, final TransitLine line, Network network) {
153                        this.id = id;
154                        this.fromNode = fromNode;
155                        this.toNode = toNode;
156                        this.route = route;
157                        this.line = line;
158                        if(route==null)
159                                this.length = CoordUtils.calcEuclideanDistance(this.toNode.stop.getStopFacility().getCoord(), this.fromNode.stop.getStopFacility().getCoord());
160                        else {
161                                this.length = 0;
162                                for(Id<Link> linkId:route.getRoute().getSubRoute(fromNode.stop.getStopFacility().getLinkId(), toNode.stop.getStopFacility().getLinkId()).getLinkIds())
163                                        this.length += network.getLinks().get(linkId).getLength();
164                                this.length += network.getLinks().get(toNode.stop.getStopFacility().getLinkId()).getLength();
165                        }
166                }

288        public TransitRouterNetworkNode createNode(final TransitRouteStop stop, final TransitRoute route, final TransitLine line) {
289                Id<Node> id = null;
290                if(line==null && route==null)
291                        id = Id.createNodeId(stop.getStopFacility().getId().toString());
292                else
293                        id = Id.createNodeId("number:"+nextNodeId++);
294                final TransitRouterNetworkNode node = new TransitRouterNetworkNode(id, stop, route, line);
295                if(this.nodes.get(node.getId())!=null)
296                        throw new RuntimeException();
297                this.nodes.put(node.getId(), node);
298                return node;
299        }

301        public TransitRouterNetworkLink createLink(final Network network, final TransitRouterNetworkNode fromNode, final TransitRouterNetworkNode toNode) {
302                final TransitRouterNetworkLink link = new TransitRouterNetworkLink(Id.createLinkId(this.nextLinkId++), fromNode, toNode, null, null, network);
303                this.links.put(link.getId(), link);
304                fromNode.outgoingLinks.put(link.getId(), link);
305                toNode.ingoingLinks.put(link.getId(), link);
306                return link;
307        }

310                this.getLinks().put(link.getId(), link);

323        public void finishInit() {
324                double minX = Double.POSITIVE_INFINITY;
325                double minY = Double.POSITIVE_INFINITY;
326                double maxX = Double.NEGATIVE_INFINITY;
327                double maxY = Double.NEGATIVE_INFINITY;
328                for (TransitRouterNetworkNode node : getNodes().values())
329                        if(node.line == null) {
330                                Coord c = node.stop.getStopFacility().getCoord();
331                                if (c.getX() < minX)
332                                        minX = c.getX();
333                                if (c.getY() < minY)
334                                        minY = c.getY();
335                                if (c.getX() > maxX)
336                                        maxX = c.getX();
337                                if (c.getY() > maxY)
338                                        maxY = c.getY();
339                        }
340                QuadTree<TransitRouterNetworkNode> quadTree = new QuadTree<TransitRouterNetworkNode>(minX, minY, maxX, maxY);
341                for (TransitRouterNetworkNode node : getNodes().values()) {
342                        if(node.line == null) { 
343                                Coord c = node.stop.getStopFacility().getCoord();
344                                quadTree.put(c.getX(), c.getY(), node);
345                        }
346                }
347                this.qtNodes = quadTree;
348        }
349        public static TransitRouterNetworkWW createFromSchedule(final Network network, final TransitSchedule schedule, final double maxBeelineWalkConnectionDistance) {
350                log.info("start creating transit network");
351                final TransitRouterNetworkWW transitNetwork = new TransitRouterNetworkWW();
352                final Counter linkCounter = new Counter(" link #");
353                final Counter nodeCounter = new Counter(" node #");
354                int numTravelLinks = 0, numWaitingLinks = 0, numInsideLinks = 0, numTransferLinks = 0;
355                Map<Id<TransitStopFacility>, TransitRouterNetworkNode> stops = new HashMap<Id<TransitStopFacility>, TransitRouterNetworkNode>();
356                TransitRouterNetworkNode nodeSR, nodeS;
357                // build stop nodes
358                for (TransitLine line : schedule.getTransitLines().values())
359                        for (TransitRoute route : line.getRoutes().values())
360                                for (TransitRouteStop stop : route.getStops()) {
361                                        nodeS = stops.get(stop.getStopFacility().getId());
362                                        if(nodeS == null) {
363                                                nodeS = transitNetwork.createNode(stop, null, null);
364                                                nodeCounter.incCounter();
365                                                stops.put(stop.getStopFacility().getId(), nodeS);
366                                        }
367                                }
368                transitNetwork.finishInit();
369                // build transfer links
370                log.info("add transfer links");
371                // connect all stops with walking links if they're located less than beelineWalkConnectionDistance from each other
372                for (TransitRouterNetworkNode node : transitNetwork.getNodes().values())
373                        for (TransitRouterNetworkNode node2 : transitNetwork.getNearestNodes(node.stop.getStopFacility().getCoord(), maxBeelineWalkConnectionDistance))
374                                if (node!=node2) {
375                                        transitNetwork.createLink(network, node, node2);
376                                        linkCounter.incCounter();
377                                        numTransferLinks++;
378                                }
379                // build nodes and links connecting the nodes according to the transit routes
380                log.info("add travel, waiting and inside links");
381                for (TransitLine line : schedule.getTransitLines().values())
382                        for (TransitRoute route : line.getRoutes().values()) {
383                                TransitRouterNetworkNode prevNode = null;
384                                for (TransitRouteStop stop : route.getStops()) {
385                                        nodeS = stops.get(stop.getStopFacility().getId());
386                                        nodeSR = transitNetwork.createNode(stop, route, line);
387                                        nodeCounter.incCounter();
388                                        if (prevNode != null) {
389                                                transitNetwork.createLink(network, prevNode, nodeSR, route, line);
390                                                linkCounter.incCounter();
391                                                numTravelLinks++;
392                                        }
393                                        prevNode = nodeSR;
394                                        transitNetwork.createLink(network, nodeS, nodeSR);
395                                        linkCounter.incCounter();
396                                        numWaitingLinks++;
397                                        transitNetwork.createLink(network, nodeSR, nodeS);
398                                        linkCounter.incCounter();
399                                        numInsideLinks++;
400                                }
401                        }
402                log.info("transit router network statistics:");
403                log.info(" # nodes: " + transitNetwork.getNodes().size());
404                log.info(" # links total:     " + transitNetwork.getLinks().size());
405                log.info(" # travel links:  " + numTravelLinks);
406                log.info(" # waiting links:  " + numWaitingLinks);
407                log.info(" # inside links:  " + numInsideLinks);
408                log.info(" # transfer links:  " + numTransferLinks);
409                return transitNetwork;
410        }
