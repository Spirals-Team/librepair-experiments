001package org.matsim.contrib.eventsBasedPTRouter.vehicleOccupancy;
002
003import org.matsim.api.core.v01.Id;
004import org.matsim.api.core.v01.events.PersonEntersVehicleEvent;
005import org.matsim.api.core.v01.events.PersonLeavesVehicleEvent;
006import org.matsim.api.core.v01.events.TransitDriverStartsEvent;
007import org.matsim.api.core.v01.events.handler.PersonEntersVehicleEventHandler;
008import org.matsim.api.core.v01.events.handler.PersonLeavesVehicleEventHandler;
009import org.matsim.api.core.v01.events.handler.TransitDriverStartsEventHandler;
010import org.matsim.core.api.experimental.events.VehicleDepartsAtFacilityEvent;
011import org.matsim.core.api.experimental.events.handler.VehicleDepartsAtFacilityEventHandler;
012import org.matsim.core.config.Config;
013import org.matsim.core.utils.collections.Tuple;
014import org.matsim.pt.transitSchedule.api.TransitLine;
015import org.matsim.pt.transitSchedule.api.TransitRoute;
016import org.matsim.pt.transitSchedule.api.TransitSchedule;
017import org.matsim.pt.transitSchedule.api.TransitStopFacility;
018import org.matsim.vehicles.Vehicle;
019import org.matsim.vehicles.VehicleCapacity;
020import org.matsim.vehicles.Vehicles;
021
022import java.util.HashMap;
023import java.util.Map;
024
025public class VehicleOccupancyCalculator implements VehicleDepartsAtFacilityEventHandler, PersonEntersVehicleEventHandler, PersonLeavesVehicleEventHandler, TransitDriverStartsEventHandler {
026
027        private final Map<Tuple<Id<TransitLine>, Id<TransitRoute>>, Map<Id<TransitStopFacility>, VehicleOccupancyData>> vehicleOccupancy = new HashMap<Tuple<Id<TransitLine>, Id<TransitRoute>>, Map<Id<TransitStopFacility>, VehicleOccupancyData>>(1000);
028        private double timeSlot;
029        private Map<Id<Vehicle>, Integer> ptVehicles = new HashMap<Id<Vehicle>, Integer>();
030        private Map<Id<Vehicle>, Integer> capacities = new HashMap<Id<Vehicle>, Integer>();
031        private Map<Id<Vehicle>, Tuple<Id<TransitLine>, Id<TransitRoute>>> linesRoutesOfVehicle = new HashMap<Id<Vehicle>, Tuple<Id<TransitLine>, Id<TransitRoute>>>();
032        private final Vehicles vehicles;
033        
034        //Constructors
035        public VehicleOccupancyCalculator(final TransitSchedule transitSchedule, final Vehicles vehicles, final Config config) {
036                this(transitSchedule, vehicles, config.travelTimeCalculator().getTraveltimeBinSize(), (int) (config.qsim().getEndTime()-config.qsim().getStartTime()));
037        }
038        public VehicleOccupancyCalculator(final TransitSchedule transitSchedule, final Vehicles vehicles, final int timeSlot, final int totalTime) {
039                this.timeSlot = timeSlot;
040                for(TransitLine line:transitSchedule.getTransitLines().values())
041                        for(TransitRoute route:line.getRoutes().values()) {
042                                Map<Id<TransitStopFacility>, VehicleOccupancyData> routeMap = new HashMap<Id<TransitStopFacility>, VehicleOccupancyData>(100);
043                                vehicleOccupancy.put(new Tuple<Id<TransitLine>, Id<TransitRoute>>(line.getId(), route.getId()), routeMap);
044                                for(int s=0; s<route.getStops().size()-1; s++) {
045                                        routeMap.put(route.getStops().get(s).getStopFacility().getId(), new VehicleOccupancyDataArray((int) (totalTime/timeSlot)+1));
046                                }
047                        }
048                this.vehicles = vehicles;
049        }
050
051        //Methods
052        public VehicleOccupancy getVehicleOccupancy() {
053                return new VehicleOccupancy() {
054                        @Override
055                        public double getVehicleOccupancy(Id<TransitStopFacility> stopOId, Id<TransitLine> lineId, Id<TransitRoute> routeId, double time) {
056                                return VehicleOccupancyCalculator.this.getVehicleOccupancy(stopOId, lineId, routeId, time);
057                        }
058                };
059        }
060        private double getVehicleOccupancy(Id<TransitStopFacility> stopOId, Id<TransitLine> lineId, Id<TransitRoute> routeId, double time) {
061                return vehicleOccupancy.get(new Tuple<Id<TransitLine>, Id<TransitRoute>>(lineId, routeId)).get(stopOId).getVehicleOccupancy((int) (time/timeSlot));
062        }
063        @Override
064        public void reset(int iteration) {
065                for(Map<Id<TransitStopFacility>, VehicleOccupancyData> map:vehicleOccupancy.values())
066                        for(VehicleOccupancyData vehicleOcupancyData:map.values())
067                                vehicleOcupancyData.resetVehicleOccupancies();
068                this.ptVehicles.clear();
069        }
070        @Override
071        public void handleEvent(VehicleDepartsAtFacilityEvent event) {
072                Integer num = ptVehicles.get(event.getVehicleId());
073                if(num!=null) {
074                        VehicleOccupancyData vehicleOccupancyData = vehicleOccupancy.get(linesRoutesOfVehicle.get(event.getVehicleId())).get(event.getFacilityId());
075                        if(vehicleOccupancyData!=null)
076                                vehicleOccupancyData.addVehicleOccupancy((int) (event.getTime()/timeSlot), num/(double)capacities.get(event.getVehicleId()));
077                }
078        }
079        @Override
080        public void handleEvent(PersonLeavesVehicleEvent event) {
081                Integer num = this.ptVehicles.get(event.getVehicleId());
082                if(num!=null)
083                        this.ptVehicles.put(event.getVehicleId(), num-1);
084        }
085        @Override
086        public void handleEvent(PersonEntersVehicleEvent event) {
087                Integer num = this.ptVehicles.get(event.getVehicleId());
088                if(num!=null)
089                        this.ptVehicles.put(event.getVehicleId(), num+1);
090        }
091        @Override
092        public void handleEvent(TransitDriverStartsEvent event) {
093                this.ptVehicles.put(event.getVehicleId(), 0);
094                VehicleCapacity vehicleCapacity = vehicles.getVehicles().get(event.getVehicleId()).getType().getCapacity();
095                this.capacities.put(event.getVehicleId(), vehicleCapacity.getSeats()+vehicleCapacity.getStandingRoom());
096                linesRoutesOfVehicle.put(event.getVehicleId(), new Tuple<Id<TransitLine>, Id<TransitRoute>>(event.getTransitLineId(), event.getTransitRouteId()));
097        }
098        
099
100}
