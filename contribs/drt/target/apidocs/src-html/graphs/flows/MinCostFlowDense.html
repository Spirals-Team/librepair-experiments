036/**
037 * Maximum flow of minimum cost with potentials 
038 */
039public class MinCostFlowDense {
040
041        static class Edge {
042                int to, f, cap, cost, rev;
043
044                Edge(int to, int cap, int cost, int rev) {
045                        this.to = to;
046                        this.cap = cap;
047                        this.cost = cost;
048                        this.rev = rev;
049                }
050        }
051
052        public static void addEdge(List<Edge>[] graph, int s, int t, int cap, int cost) {
053                graph[s].add(new Edge(t, cap, cost, graph[t].size()));
054                graph[t].add(new Edge(s, 0, -cost, graph[s].size() - 1));
055        }
056
057        static void bellmanFord(List<Edge>[] graph, int s, int[] dist) {
058                int n = graph.length;
059                Arrays.fill(dist, Integer.MAX_VALUE);
060                dist[s] = 0;
061                boolean[] inqueue = new boolean[n];
062                int[] q = new int[n];
063                int qt = 0;
064                q[qt++] = s;
065                for (int qh = 0; (qh - qt) % n != 0; qh++) {
066                        int u = q[qh % n];
067                        inqueue[u] = false;
068                        for (int i = 0; i < graph[u].size(); i++) {
069                                Edge e = graph[u].get(i);
070                                if (e.cap <= e.f)
071                                        continue;
072                                int v = e.to;
073                                int ndist = dist[u] + e.cost;
074                                if (dist[v] > ndist) {
075                                        dist[v] = ndist;
076                                        if (!inqueue[v]) {
077                                                inqueue[v] = true;
078                                                q[qt++ % n] = v;
079                                        }
080                                }
081                        }
082                }
083        }
084
085        public static int[] minCostFlow(List<Edge>[] graph, int s, int t, int maxf, boolean runBellmanFord) {
086                int n = graph.length;
087                int[] prio = new int[n];
088                int[] curflow = new int[n];
089                int[] prevedge = new int[n];
090                int[] prevnode = new int[n];
091                int[] pot = new int[n];
092
093                if (runBellmanFord) {
094                        bellmanFord(graph, s, pot); // bellmanFord invocation can be skipped if edges costs are non-negative
095                }
096
097                int flow = 0;
098                int flowCost = 0;
099                while (flow < maxf) {
100                        Arrays.fill(prio, Integer.MAX_VALUE);
101                        prio[s] = 0;
102                        boolean[] finished = new boolean[n];
103                        curflow[s] = Integer.MAX_VALUE;
104                        for (int i = 0; i < n && !finished[t]; i++) {
105                                int u = -1;
106                                for (int j = 0; j < n; j++)
107                                        if (!finished[j] && (u == -1 || prio[u] > prio[j]))
108                                                u = j;
109                                if (prio[u] == Integer.MAX_VALUE)
110                                        break;
111                                finished[u] = true;
112                                for (int k = 0; k < graph[u].size(); k++) {
113                                        Edge e = graph[u].get(k);
114                                        if (e.f >= e.cap)
115                                                continue;
116                                        int v = e.to;
117                                        int nprio = prio[u] + e.cost + pot[u] - pot[v];
118                                        if (prio[v] > nprio) {
119                                                prio[v] = nprio;
120                                                prevnode[v] = u;
121                                                prevedge[v] = k;
122                                                curflow[v] = Math.min(curflow[u], e.cap - e.f);
123                                        }
124                                }
125                        }
126                        if (prio[t] == Integer.MAX_VALUE)
127                                break;
128                        for (int i = 0; i < n; i++)
129                                if (finished[i])
130                                        pot[i] += prio[i] - prio[t];
131                        int df = Math.min(curflow[t], maxf - flow);
132                        flow += df;
133                        for (int v = t; v != s; v = prevnode[v]) {
134                                Edge e = graph[prevnode[v]].get(prevedge[v]);
135                                e.f += df;
136                                graph[v].get(e.rev).f -= df;
137                                flowCost += df * e.cost;
138                        }
139                }
140                return new int[] { flow, flowCost };
141        }
142
143        // Usage example
144        public static void main(String[] args) {
145                @SuppressWarnings("unchecked")
146                List<Edge>[] graph = Stream.generate(ArrayList::new).limit(3).toArray(List[]::new);
147                addEdge(graph, 0, 1, 3, 1);
148                addEdge(graph, 0, 2, 2, 1);
149                addEdge(graph, 1, 2, 2, 1);
150                int[] res = minCostFlow(graph, 0, 2, Integer.MAX_VALUE, true);
151                int flow = res[0];
152                int flowCost = res[1];
153                System.out.println(4 == flow);
154                System.out.println(6 == flowCost);
155        }
156}
