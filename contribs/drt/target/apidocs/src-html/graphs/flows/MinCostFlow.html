037/**
038 * Maximum flow of minimum cost with potentials in O(min(E^2*V*logV, E*logV*FLOW))
039 */
040public class MinCostFlow {
041
042        public static class Edge {
043                int to, f, cap, cost, rev;
044
045                Edge(int to, int cap, int cost, int rev) {
046                        this.to = to;
047                        this.cap = cap;
048                        this.cost = cost;
049                        this.rev = rev;
050                }
051
052                public int getFlow() {
053                        return f;
054                }
055
056                public int getTo() {
057                        return to;
058                }
059
060                @Override
061                public String toString() {
062                        return cap + "/" + cost + "->" + to;
063                }
064        }
065
066        public static void addEdge(List<Edge>[] graph, int s, int t, int cap, int cost) {
067                graph[s].add(new Edge(t, cap, cost, graph[t].size()));
068                graph[t].add(new Edge(s, 0, -cost, graph[s].size() - 1));
069        }
070
071        static void bellmanFord(List<Edge>[] graph, int s, int[] dist) {
072                int n = graph.length;
073                Arrays.fill(dist, Integer.MAX_VALUE);
074                dist[s] = 0;
075                boolean[] inqueue = new boolean[n];
076                int[] q = new int[n];
077                int qt = 0;
078                q[qt++] = s;
079                for (int qh = 0; (qh - qt) % n != 0; qh++) {
080                        int u = q[qh % n];
081                        inqueue[u] = false;
082                        for (int i = 0; i < graph[u].size(); i++) {
083                                Edge e = graph[u].get(i);
084                                if (e.cap <= e.f)
085                                        continue;
086                                int v = e.to;
087                                int ndist = dist[u] + e.cost;
088                                if (dist[v] > ndist) {
089                                        dist[v] = ndist;
090                                        if (!inqueue[v]) {
091                                                inqueue[v] = true;
092                                                q[qt++ % n] = v;
093                                        }
094                                }
095                        }
096                }
097        }
098
099        public static int[] minCostFlow(List<Edge>[] graph, int s, int t, int maxf, boolean runBellmanFord) {
100                int n = graph.length;
101                int[] prio = new int[n];
102                int[] curflow = new int[n];
103                int[] prevedge = new int[n];
104                int[] prevnode = new int[n];
105                int[] pot = new int[n];
106
107                if (runBellmanFord) {
108                        bellmanFord(graph, s, pot); // bellmanFord invocation can be skipped if edges costs are non-negative
109                }
110
111                int flow = 0;
112                int flowCost = 0;
113                while (flow < maxf) {
114                        PriorityQueue<Long> q = new PriorityQueue<>();
115                        q.add((long)s);
116                        Arrays.fill(prio, Integer.MAX_VALUE);
117                        prio[s] = 0;
118                        boolean[] finished = new boolean[n];
119                        curflow[s] = Integer.MAX_VALUE;
120                        while (!finished[t] && !q.isEmpty()) {
121                                long cur = q.remove();
122                                int u = (int)(cur & 0xFFFF_FFFFL);
123                                int priou = (int)(cur >>> 32);
124                                if (priou != prio[u])
125                                        continue;
126                                finished[u] = true;
127                                for (int i = 0; i < graph[u].size(); i++) {
128                                        Edge e = graph[u].get(i);
129                                        if (e.f >= e.cap)
130                                                continue;
131                                        int v = e.to;
132                                        int nprio = prio[u] + e.cost + pot[u] - pot[v];
133                                        if (prio[v] > nprio) {
134                                                prio[v] = nprio;
135                                                q.add(((long)nprio << 32) + v);
136                                                prevnode[v] = u;
137                                                prevedge[v] = i;
138                                                curflow[v] = Math.min(curflow[u], e.cap - e.f);
139                                        }
140                                }
141                        }
142                        if (prio[t] == Integer.MAX_VALUE)
143                                break;
144                        for (int i = 0; i < n; i++)
145                                if (finished[i])
146                                        pot[i] += prio[i] - prio[t];
147                        int df = Math.min(curflow[t], maxf - flow);
148                        flow += df;
149                        for (int v = t; v != s; v = prevnode[v]) {
150                                Edge e = graph[prevnode[v]].get(prevedge[v]);
151                                e.f += df;
152                                graph[v].get(e.rev).f -= df;
153                                flowCost += df * e.cost;
154                        }
155                }
156                return new int[] { flow, flowCost };
157        }
158
159        // Usage example
160        public static void main(String[] args) {
161                @SuppressWarnings("unchecked")
162                List<Edge>[] graph = Stream.generate(ArrayList::new).limit(3).toArray(List[]::new);
163                addEdge(graph, 0, 1, 3, 1);
164                addEdge(graph, 0, 2, 2, 1);
165                addEdge(graph, 1, 2, 2, 1);
166                int[] res = minCostFlow(graph, 0, 2, Integer.MAX_VALUE, true);
167                int flow = res[0];
168                int flowCost = res[1];
169                System.out.println(4 == flow);
170                System.out.println(6 == flowCost);
171        }
172}
