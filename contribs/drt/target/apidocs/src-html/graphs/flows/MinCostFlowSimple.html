033/**
034 * Maximum flow of minimum cost in O(V^3*FLOW)
035 */
036public class MinCostFlowSimple {
037
038        public static int[] minCostFlow(int[][] cap, int[][] cost, int s, int t) {
039                int n = cap.length;
040                int[] d = new int[n];
041                int[] p = new int[n];
042                for (int flow = 0, flowCost = 0;; ++flow) {
043                        Arrays.fill(d, Integer.MAX_VALUE);
044                        d[s] = 0;
045                        for (int i = 0; i < n - 1; i++)
046                                for (int j = 0; j < n; j++)
047                                        for (int k = 0; k < n; k++)
048                                                if (cap[j][k] > 0 && d[j] < Integer.MAX_VALUE && d[k] > d[j] + cost[j][k]) {
049                                                        d[k] = d[j] + cost[j][k];
050                                                        p[k] = j;
051                                                }
052                        if (d[t] == Integer.MAX_VALUE)
053                                return new int[] { flowCost, flow };
054                        flowCost += d[t];
055                        for (int v = t; v != s; v = p[v]) {
056                                --cap[p[v]][v];
057                                ++cap[v][p[v]];
058                        }
059                }
060        }
061
062        public static void addEdge(int[][] cap, int[][] cost, int u, int v, int edgeCapacity, int edgeCost) {
063                cap[u][v] = edgeCapacity;
064                cost[u][v] = edgeCost;
065                cost[v][u] = -edgeCost;
066        }
067
068        // Usage example
069        public static void main(String[] args) {
070                int n = 3;
071                int[][] capacity = new int[n][n];
072                int[][] cost = new int[n][n];
073                addEdge(capacity, cost, 0, 1, 3, 1);
074                addEdge(capacity, cost, 0, 2, 2, 1);
075                addEdge(capacity, cost, 1, 2, 2, 1);
076                int[] costFlow = minCostFlow(capacity, cost, 0, 2);
077                System.out.println(6 == costFlow[0]);
078                System.out.println(4 == costFlow[1]);
079        }
080}
