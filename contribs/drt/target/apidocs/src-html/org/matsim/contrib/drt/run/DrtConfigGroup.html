107        @PositiveOrZero
108        private double stopDuration = Double.NaN;// seconds
109
110        @PositiveOrZero
111        private double maxWaitTime = Double.NaN;// seconds
112
113        // max arrival time defined as:
114        // maxTravelTimeAlpha * unshared_ride_travel_time(fromLink, toLink) + maxTravelTimeBeta,
115        // where unshared_ride_travel_time(fromLink, toLink) is calculated with FastAStarEuclidean
116        // (hence AStarEuclideanOverdoFactor needs to be specified)
117        @Min(1)
118        private double maxTravelTimeAlpha = Double.NaN;// [-]
119
120        @PositiveOrZero
121        private double maxTravelTimeBeta = Double.NaN;// [s]
122
123        private boolean changeStartLinkToLastLinkInSchedule = false;
124
125        @PositiveOrZero
126        private int rebalancingInterval = 0;// [s], if 0 then no rebalancing
127
128        private boolean idleVehiclesReturnToDepots = false;
129
130        @NotNull
131        private OperationalScheme operationalScheme = OperationalScheme.door2door;
132
133        @PositiveOrZero // used only for stopbased DRT scheme
134        private double maxWalkDistance = 0;// [m];
135
136        @PositiveOrZero
137        private double estimatedDrtSpeed = 25. / 3.6;// [m/s]
138
139        @Min(1)
140        private double estimatedBeelineDistanceFactor = 1.3;// [-]
141
142        @NotNull
143        private String vehiclesFile = null;
144
145        // used only for stopbased DRT scheme
146        private String transitStopFile = null; // only for stopbased DRT scheme
147
148        private boolean plotDetailedCustomerStats = true;
149        private boolean plotDetailedVehicleStats = false;
150        private boolean printDetailedWarnings = true;
151
152        @Positive
153        private int numberOfThreads = Math.min(Runtime.getRuntime().availableProcessors(),
154                        ParallelPathDataProvider.MAX_THREADS);
155
156        public enum OperationalScheme {
157                stopbased, door2door
158        }

169                map.put(MAX_TRAVEL_TIME_ALPHA,

171                map.put(MAX_TRAVEL_TIME_BETA,
172                                MAX_TRAVEL_TIME_BETA_EXP);
173                map.put(CHANGE_START_LINK_TO_LAST_LINK_IN_SCHEDULE,
174                                CHANGE_START_EXP);
175                map.put(VEHICLES_FILE,
176                                VEH_FILE_EXP);
177                map.put(PLOT_CUST_STATS, CUST_STATS_EXP);
178                map.put(PLOT_VEH_STATS,
179                                VEH_STATS_EXP);
180                map.put(REBALANCING_INTERVAL,
181                                REB_INT_EXP);
182                map.put(IDLE_VEHICLES_RETURN_TO_DEPOTS,
183                                IDLE_VEHICLES_RETURN_TO_DEPOTS_EXP);
184                map.put(OPERATIONAL_SCHEME, OP_SCHEME_EXP);
185                map.put(MAX_WALK_DISTANCE, MAX_WALK_EXP);
186                map.put(TRANSIT_STOP_FILE, TRANSIT_STOP_FILE_EXP);
187                map.put(ESTIMATED_DRT_SPEED, ESTIMATED_DRT_SPEED_EXP);
188                map.put(ESTIMATED_BEELINE_DISTANCE_FACTOR,
189                                ESTIMATED_BEELINE_DISTANCE_FACTOR_EXP);
190                map.put(NUMBER_OF_THREADS,
191                                NUMBER_OF_THREADS_EXP);
192                map.put(PRINT_WARNINGS,
193                                PRINT_WARNINGS_EXP);
194                return map;
195        }
