049/**
050 * @author michalm
051 */
052public class StopBasedPathDataProvider implements PrecalculatablePathDataProvider {
053        private final double stopDuration;
054
055        private final ManyToManyPathData manyToManyPathData;
056
057        // ==== recalculated by calcPathData()
058        private Map<Id<Link>, PathData> pathsToPickupMap;
059        private Map<Id<Link>, PathData> pathsFromPickupMap;
060        private Map<Id<Link>, PathData> pathsToDropoffMap;
061        private Map<Id<Link>, PathData> pathsFromDropoffMap;
062
063        @Inject
064        public StopBasedPathDataProvider(@Named(DvrpRoutingNetworkProvider.DVRP_ROUTING) Network network,
065                        @Named(DvrpTravelTimeModule.DVRP_ESTIMATED) TravelTime travelTime,
066                        @Named(DefaultDrtOptimizer.DRT_OPTIMIZER) TravelDisutility travelDisutility,
067                        @Named(TransportMode.drt) TransitSchedule schedule, TravelTimeCalculatorConfigGroup ttcConfig,
068                        DrtConfigGroup drtCfg) {
069                stopDuration = drtCfg.getStopDuration();
070
071                List<Link> stopLinks = schedule.getFacilities().values().stream()
072                                .map(tsf -> network.getLinks().get(tsf.getLinkId()))//
073                                .distinct()// more than one stop can be located on a link
074                                .collect(ImmutableList.toImmutableList());
075                manyToManyPathData = new ManyToManyPathData(network, travelTime, travelDisutility, stopLinks,
076                                new TimeDiscretizer(ttcConfig), drtCfg.getNumberOfThreads());
077        }
078
079        @Override
080        public void precalculatePathData(DrtRequest drtRequest, Collection<VehicleData.Entry> vEntries) {
081                Link pickup = drtRequest.getFromLink();
082                Link dropoff = drtRequest.getToLink();
083
084                final double earliestPickupTime = drtRequest.getEarliestStartTime(); // optimistic
085                final double minTravelTime = 15 * 60; // FIXME inaccurate temp solution: fixed 15 min
086                final double earliestDropoffTime = earliestPickupTime + minTravelTime + stopDuration;
087
088                // NOTE: all paths are calculated forward from startTime (no backward Dijkstra used)
089                pathsToPickupMap = manyToManyPathData.getIncomingPathData(pickup.getId(), earliestPickupTime);
090                pathsFromPickupMap = manyToManyPathData.getOutgoingPathData(pickup.getId(), earliestPickupTime);
091                pathsToDropoffMap = manyToManyPathData.getIncomingPathData(dropoff.getId(), earliestDropoffTime);
092                pathsFromDropoffMap = manyToManyPathData.getOutgoingPathData(dropoff.getId(), earliestDropoffTime);
093        }
094
095        @Override
096        public PathDataSet getPathDataSet(DrtRequest drtRequest, Entry vEntry) {
097                return PrecalculatablePathDataProvider.getPathDataSet(drtRequest, vEntry, pathsToPickupMap, pathsFromPickupMap,
098                                pathsToDropoffMap, pathsFromDropoffMap);
099        }
100}
