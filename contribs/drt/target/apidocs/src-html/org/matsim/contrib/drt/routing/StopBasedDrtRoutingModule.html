051/**
052 * @author jbischoff
053 */
054public class StopBasedDrtRoutingModule implements RoutingModule {
055        public interface AccessEgressStopFinder {
056                Pair<TransitStopFacility, TransitStopFacility> findStops(Facility<?> fromFacility, Facility<?> toFacility);
057        }
058
059        private final StageActivityTypes drtStageActivityType = new DrtStageActivityType();
060        private final PopulationFactory populationFactory;
061        private final RoutingModule walkRouter;
062        private final AccessEgressStopFinder stopFinder;
063        private final DrtConfigGroup drtconfig;
064
065        @Inject
066        public StopBasedDrtRoutingModule(PopulationFactory populationFactory,
067                        @Named(TransportMode.walk) RoutingModule walkRouter, AccessEgressStopFinder stopFinder,
068                        DrtConfigGroup drtCfg) {
069                this.populationFactory = populationFactory;
070                this.walkRouter = walkRouter;
071                this.stopFinder = stopFinder;
072                this.drtconfig = drtCfg;
073        }
074
075        @Override
076        public List<? extends PlanElement> calcRoute(Facility<?> fromFacility, Facility<?> toFacility, double departureTime,
077                        Person person) {
078                List<PlanElement> legList = new ArrayList<>();
079                Pair<TransitStopFacility, TransitStopFacility> stops = stopFinder.findStops(fromFacility, toFacility);
080                TransitStopFacility accessFacility = stops.getLeft();
081                if (accessFacility == null) {
082                        if (drtconfig.isPrintDetailedWarnings()) {
083                                Logger.getLogger(getClass())
084                                                .error("No access stop found, agent will walk. Agent Id:\t" + person.getId());
085                        }
086                        return (walkRouter.calcRoute(fromFacility, toFacility, departureTime, person));
087                }
088                TransitStopFacility egressFacility = stops.getRight();
089                if (egressFacility == null) {
090                        if (drtconfig.isPrintDetailedWarnings()) {
091                                Logger.getLogger(getClass())
092                                                .error("No egress stop found, agent will walk. Agent Id:\t" + person.getId());
093                        }
094                        return (walkRouter.calcRoute(fromFacility, toFacility, departureTime, person));
095                }
096                legList.addAll(walkRouter.calcRoute(fromFacility, accessFacility, departureTime, person));
097                Leg walkLeg = (Leg)legList.get(0);
098                Activity drtInt1 = populationFactory.createActivityFromCoord(DrtStageActivityType.DRT_STAGE_ACTIVITY,
099                                accessFacility.getCoord());
100                drtInt1.setMaximumDuration(1);
101                drtInt1.setLinkId(accessFacility.getLinkId());
102                legList.add(drtInt1);
103
104                Route drtRoute = RouteUtils.createGenericRouteImpl(accessFacility.getLinkId(), egressFacility.getLinkId());
105                drtRoute.setDistance(drtconfig.getEstimatedBeelineDistanceFactor()
106                                * CoordUtils.calcEuclideanDistance(accessFacility.getCoord(), egressFacility.getCoord()));
107                drtRoute.setTravelTime(drtRoute.getDistance() / drtconfig.getEstimatedDrtSpeed());
108
109                if (drtRoute.getStartLinkId() == drtRoute.getEndLinkId()) {
110                        if (drtconfig.isPrintDetailedWarnings()) {
111                                Logger.getLogger(getClass())
112                                                .error("Start and end stop are the same, agent will walk. Agent Id:\t" + person.getId());
113                        }
114                        return (walkRouter.calcRoute(fromFacility, toFacility, departureTime, person));
115
116                }
117
118                Leg drtLeg = PopulationUtils.createLeg(TransportMode.drt);
119                drtLeg.setDepartureTime(departureTime + walkLeg.getTravelTime() + 1);
120                drtLeg.setTravelTime(drtRoute.getTravelTime());
121                drtLeg.setRoute(drtRoute);
122
123                legList.add(drtLeg);
124
125                Activity drtInt2 = populationFactory.createActivityFromCoord(DrtStageActivityType.DRT_STAGE_ACTIVITY,
126                                egressFacility.getCoord());
127                drtInt2.setMaximumDuration(1);
128                drtInt2.setLinkId(egressFacility.getLinkId());
129                legList.add(drtInt2);
130                legList.addAll(walkRouter.calcRoute(egressFacility, toFacility,
131                                drtLeg.getDepartureTime() + drtLeg.getTravelTime() + 1, person));
132                for (PlanElement pE : legList) {
133                        if (pE instanceof Leg) {
134                                if (((Leg)pE).getMode().equals(TransportMode.walk)) {
135                                        ((Leg)pE).setMode(DrtStageActivityType.DRT_WALK);
136                                }
137                        }
138                }
139
140                return legList;
141        }
142
143        @Override
144        public StageActivityTypes getStageActivityTypes() {
145                return drtStageActivityType;
146        }
147
148        static Coord getFacilityCoord(Facility<?> facility, Network network) {
149                Coord coord = facility.getCoord();
150                if (coord == null) {
151                        coord = network.getLinks().get(facility.getLinkId()).getCoord();
152                        if (coord == null)
153                                throw new RuntimeException("From facility has neither coordinates nor link Id. Should not happen.");
154                }
155                return coord;
156        }
157}
