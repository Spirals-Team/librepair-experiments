052/**
053 * @author michalm
054 */
055public class DefaultDrtOptimizer implements DrtOptimizer {
056        private static final Logger log = Logger.getLogger(DefaultDrtOptimizer.class);
057
058        public static final String DRT_OPTIMIZER = "drt_optimizer";
059
060        private final DrtConfigGroup drtCfg;
061        private final Fleet fleet;
062        private final DrtScheduleInquiry scheduleInquiry;
063        private final DrtScheduleTimingUpdater scheduleTimingUpdater;
064        private final RebalancingStrategy rebalancingStrategy;
065        private final MobsimTimer mobsimTimer;
066        private final EventsManager eventsManager;
067        private final DrtRequestValidator requestValidator;
068        private final DepotFinder depotFinder;
069        private final EmptyVehicleRelocator relocator;
070        private final UnplannedRequestInserter requestInserter;
071
072        private final Collection<DrtRequest> unplannedRequests = new TreeSet<DrtRequest>(
073                        PassengerRequests.ABSOLUTE_COMPARATOR);
074        private boolean requiresReoptimization = false;
075
076        @Inject
077        public DefaultDrtOptimizer(DrtConfigGroup drtCfg, Fleet fleet, MobsimTimer mobsimTimer, EventsManager eventsManager,
078                        DrtRequestValidator requestValidator, DepotFinder depotFinder, RebalancingStrategy rebalancingStrategy,
079                        DrtScheduleInquiry scheduleInquiry, DrtScheduleTimingUpdater scheduleTimingUpdater,
080                        EmptyVehicleRelocator relocator, UnplannedRequestInserter requestInserter) {
081                this.drtCfg = drtCfg;
082                this.fleet = fleet;
083                this.mobsimTimer = mobsimTimer;
084                this.eventsManager = eventsManager;
085                this.requestValidator = requestValidator;
086                this.depotFinder = drtCfg.getIdleVehiclesReturnToDepots() ? depotFinder : null;
087                this.rebalancingStrategy = drtCfg.getRebalancingInterval() != 0 ? rebalancingStrategy : null;
088                this.scheduleInquiry = scheduleInquiry;
089                this.scheduleTimingUpdater = scheduleTimingUpdater;
090                this.relocator = relocator;
091                this.requestInserter = requestInserter;
092        }
093
094        @Override
095        public void notifyMobsimBeforeSimStep(@SuppressWarnings("rawtypes") MobsimBeforeSimStepEvent e) {
096                if (requiresReoptimization) {
097                        for (Vehicle v : fleet.getVehicles().values()) {
098                                scheduleTimingUpdater.updateTimings(v);
099                        }
100
101                        requestInserter.scheduleUnplannedRequests(unplannedRequests);
102                        requiresReoptimization = false;
103                }
104
105                if (rebalancingStrategy != null && e.getSimulationTime() % drtCfg.getRebalancingInterval() == 0) {
106                        rebalanceFleet();
107                }
108        }
109
110        private void rebalanceFleet() {
111                // right now we relocate only idle vehicles (vehicles that are being relocated cannot be relocated)
112                Stream<? extends Vehicle> rebalancableVehicles = fleet.getVehicles().values().stream()
113                                .filter(scheduleInquiry::isIdle);
114                List<Relocation> relocations = rebalancingStrategy.calcRelocations(rebalancableVehicles,
115                                mobsimTimer.getTimeOfDay());
116
117                if (!relocations.isEmpty()) {
118                        log.debug("Fleet rebalancing: #relocations=" + relocations.size());
119                        for (Relocation r : relocations) {
120                                Link currentLink = ((DrtStayTask)r.vehicle.getSchedule().getCurrentTask()).getLink();
121                                if (currentLink != r.link) {
122                                        relocator.relocateVehicle(r.vehicle, r.link);
123                                }
124                        }
125                }
126        }
127
128        @Override
129        public void requestSubmitted(Request request) {
130                DrtRequest drtRequest = (DrtRequest)request;
131                if (!requestValidator.validateDrtRequest(drtRequest)) {
132                        drtRequest.setRejected(true);
133                        eventsManager.processEvent(new DrtRequestRejectedEvent(mobsimTimer.getTimeOfDay(), drtRequest.getId()));
134                        return;
135                }
136
137                unplannedRequests.add(drtRequest);
138                requiresReoptimization = true;
139        }
140
141        @Override
142        public void nextTask(Vehicle vehicle) {
143                scheduleTimingUpdater.updateBeforeNextTask(vehicle);
144
145                vehicle.getSchedule().nextTask();
146
147                // if STOP->STAY then choose the best depot
148                if (depotFinder != null && Depots.isSwitchingFromStopToStay(vehicle)) {
149                        Link depotLink = depotFinder.findDepot(vehicle);
150                        if (depotLink != null) {
151                                relocator.relocateVehicle(vehicle, depotLink);
152                        }
153                }
154        }
155
156        @Override
157        public void vehicleEnteredNextLink(Vehicle vehicle, Link nextLink) {
158                // scheduler.updateTimeline(vehicle);
159
160                // TODO we may here possibly decide whether or not to reoptimize
161                // if (delays/speedups encountered) {requiresReoptimization = true;}
162        }
163}
