001package org.matsim.contrib.drt.run;
002
003import org.matsim.api.core.v01.Scenario;
004import org.matsim.api.core.v01.TransportMode;
005import org.matsim.contrib.drt.data.validator.DefaultDrtRequestValidator;
006import org.matsim.contrib.drt.data.validator.DrtRequestValidator;
007import org.matsim.contrib.drt.optimizer.DefaultDrtOptimizer;
008import org.matsim.contrib.drt.optimizer.depot.DepotFinder;
009import org.matsim.contrib.drt.optimizer.depot.NearestStartLinkAsDepot;
010import org.matsim.contrib.drt.optimizer.rebalancing.mincostflow.MinCostFlowRebalancingModule;
011import org.matsim.contrib.drt.routing.DefaultAccessEgressStopFinder;
012import org.matsim.contrib.drt.routing.DrtMainModeIdentifier;
013import org.matsim.contrib.drt.routing.DrtRoutingModule;
014import org.matsim.contrib.drt.routing.StopBasedDrtRoutingModule;
015import org.matsim.contrib.drt.routing.StopBasedDrtRoutingModule.AccessEgressStopFinder;
016import org.matsim.contrib.dvrp.data.Fleet;
017import org.matsim.contrib.dvrp.data.file.FleetProvider;
018import org.matsim.contrib.dvrp.router.TimeAsTravelDisutility;
019import org.matsim.core.config.ConfigUtils;
020import org.matsim.core.controler.AbstractModule;
021import org.matsim.core.router.MainModeIdentifier;
022import org.matsim.core.router.costcalculators.TravelDisutilityFactory;
023import org.matsim.core.scenario.ScenarioUtils;
024import org.matsim.pt.transitSchedule.api.TransitSchedule;
025import org.matsim.pt.transitSchedule.api.TransitScheduleReader;
026
027import com.google.inject.name.Names;
028
029public final class DrtModule extends AbstractModule {
030
031        @Override
032        public void install() {
033                DrtConfigGroup drtCfg = DrtConfigGroup.get(getConfig());
034                bind(Fleet.class).toProvider(new FleetProvider(drtCfg.getVehiclesFileUrl(getConfig().getContext())))
035                                .asEagerSingleton();
036                bind(DrtRequestValidator.class).to(DefaultDrtRequestValidator.class);
037                bind(DepotFinder.class).to(NearestStartLinkAsDepot.class);
038                bind(TravelDisutilityFactory.class).annotatedWith(Names.named(DefaultDrtOptimizer.DRT_OPTIMIZER))
039                                .toInstance(timeCalculator -> new TimeAsTravelDisutility(timeCalculator));
040
041                install(new MinCostFlowRebalancingModule(1000));
042
043                switch (drtCfg.getOperationalScheme()) {
044                        case door2door:
045                                addRoutingModuleBinding(TransportMode.drt).to(DrtRoutingModule.class).asEagerSingleton();
046                                break;
047
048                        case stopbased:
049                                final Scenario scenario2 = ScenarioUtils.createScenario(ConfigUtils.createConfig());
050                                new TransitScheduleReader(scenario2)
051                                                .readFile(drtCfg.getTransitStopsFileUrl(getConfig().getContext()).getFile());
052                                bind(TransitSchedule.class).annotatedWith(Names.named(TransportMode.drt))
053                                                .toInstance(scenario2.getTransitSchedule());
054                                bind(MainModeIdentifier.class).to(DrtMainModeIdentifier.class).asEagerSingleton();
055                                addRoutingModuleBinding(TransportMode.drt).to(StopBasedDrtRoutingModule.class).asEagerSingleton();
056                                bind(AccessEgressStopFinder.class).to(DefaultAccessEgressStopFinder.class).asEagerSingleton();
057                                break;
058
059                        default:
060                                throw new IllegalStateException();
061                }
062        }
063}
