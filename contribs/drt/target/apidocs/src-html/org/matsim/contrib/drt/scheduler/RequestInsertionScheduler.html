056public class RequestInsertionScheduler {
057        private final Fleet fleet;
058        private final double stopDuration;
059        private final MobsimTimer timer;
060        private final TravelTime travelTime;
061        private final DrtScheduleTimingUpdater scheduleTimingUpdater;
062        private final DrtTaskFactory taskFactory;
063
064        @Inject
065        public RequestInsertionScheduler(DrtConfigGroup drtCfg, Fleet fleet, MobsimTimer timer,
066                        @Named(DvrpTravelTimeModule.DVRP_ESTIMATED) TravelTime travelTime,
067                        DrtScheduleTimingUpdater scheduleTimingUpdater, DrtTaskFactory taskFactory) {
068                this.fleet = fleet;
069                this.stopDuration = drtCfg.getStopDuration();
070                this.timer = timer;
071                this.travelTime = travelTime;
072                this.scheduleTimingUpdater = scheduleTimingUpdater;
073                this.taskFactory = taskFactory;
074        }
075
076        public void initSchedules(boolean changeStartLinkToLastLinkInSchedule) {
077                ((FleetImpl)fleet).resetSchedules();
078                for (Vehicle veh : fleet.getVehicles().values()) {
079                        if (changeStartLinkToLastLinkInSchedule) {
080                                Vehicles.changeStartLinkToLastLinkInSchedule(veh);
081                        }
082                        veh.getSchedule().addTask(taskFactory.createStayTask(veh, veh.getServiceBeginTime(),
083                                        veh.getServiceEndTime(), veh.getStartLink()));
084                }
085        }
086
087        public void scheduleRequest(VehicleData.Entry vehicleEntry, DrtRequest request, InsertionWithPathData insertion) {
088                insertPickup(vehicleEntry, request, insertion);
089                insertDropoff(vehicleEntry, request, insertion);
090        }
091
092        private void insertPickup(VehicleData.Entry vehicleEntry, DrtRequest request, InsertionWithPathData insertion) {
093                Schedule schedule = vehicleEntry.vehicle.getSchedule();
094                List<Stop> stops = vehicleEntry.stops;
095
096                DrtTask currentTask = (DrtTask)schedule.getCurrentTask();
097                Task beforePickupTask;
098
099                if (insertion.pickupIdx == 0 && currentTask.getDrtTaskType() == DrtTaskType.DRIVE) {
100                        LinkTimePair diversion = ((OnlineDriveTaskTracker)currentTask.getTaskTracker()).getDiversionPoint();
101                        if (diversion != null) { // divert vehicle
102                                beforePickupTask = currentTask;
103                                VrpPathWithTravelData vrpPath = VrpPaths.createPath(vehicleEntry.start.link, request.getFromLink(),
104                                                vehicleEntry.start.time, insertion.pathToPickup, travelTime);
105                                ((OnlineDriveTaskTracker)beforePickupTask.getTaskTracker()).divertPath(vrpPath);
106                        } else { // too late for diversion
107                                if (request.getFromLink() != vehicleEntry.start.link) { // add a new drive task
108                                        VrpPathWithTravelData vrpPath = VrpPaths.createPath(vehicleEntry.start.link, request.getFromLink(),
109                                                        vehicleEntry.start.time, insertion.pathToPickup, travelTime);
110                                        beforePickupTask = taskFactory.createDriveTask(vehicleEntry.vehicle, vrpPath);
111                                        schedule.addTask(currentTask.getTaskIdx() + 1, beforePickupTask);
112                                } else { // no need for a new drive task
113                                        beforePickupTask = currentTask;
114                                }
115                        }
116                } else { // insert pickup after an existing stop/stay task
117                        DrtStayTask stayTask = null;
118                        DrtStopTask stopTask = null;
119                        if (insertion.pickupIdx == 0) {
120                                if (currentTask.getDrtTaskType() == DrtTaskType.STAY) {
121                                        stayTask = (DrtStayTask)currentTask; // ongoing stay task
122                                        double now = timer.getTimeOfDay();
123                                        if (stayTask.getEndTime() > now) { // stop stay task; a new stop/drive task can be inserted now
124                                                stayTask.setEndTime(now);
125                                        }
126                                } else {
127                                        stopTask = (DrtStopTask)currentTask; // ongoing stop task
128                                }
129                        } else {
130                                stopTask = stops.get(insertion.pickupIdx - 1).task; // future stop task
131                        }
132
133                        if (stopTask != null && request.getFromLink() == stopTask.getLink()) { // no detour; no new stop task
134                                // add pickup request to stop task
135                                stopTask.addPickupRequest(request);
136                                request.setPickupTask(stopTask);
137
138                                /// ADDED
139                                //// TODO this is copied, but has not been updated !!!!!!!!!!!!!!!
140                                // add drive from pickup
141                                if (insertion.pickupIdx == insertion.dropoffIdx) {
142                                        // remove drive i->i+1 (if there is one)
143                                        if (insertion.pickupIdx < stops.size()) {// there is at least one following stop
144                                                DrtStopTask nextStopTask = stops.get(insertion.pickupIdx).task;
145                                                if (stopTask.getTaskIdx() + 2 != nextStopTask.getTaskIdx()) {// there must a drive task in
146                                                                                                                                                                                // between
147                                                        throw new RuntimeException();
148                                                }
149                                                if (stopTask.getTaskIdx() + 2 == nextStopTask.getTaskIdx()) {// there must a drive task in
150                                                                                                                                                                                // between
151                                                        int driveTaskIdx = stopTask.getTaskIdx() + 1;
152                                                        schedule.removeTask(schedule.getTasks().get(driveTaskIdx));
153                                                }
154                                        }
155
156                                        Link toLink = request.getToLink(); // pickup->dropoff
157
158                                        VrpPathWithTravelData vrpPath = VrpPaths.createPath(request.getFromLink(), toLink,
159                                                        stopTask.getEndTime(), insertion.pathFromPickup, travelTime);
160                                        Task driveFromPickupTask = taskFactory.createDriveTask(vehicleEntry.vehicle, vrpPath);
161                                        schedule.addTask(stopTask.getTaskIdx() + 1, driveFromPickupTask);
162
163                                        // update timings
164                                        // TODO should be enough to update the timeline only till dropoffIdx...
165                                        scheduleTimingUpdater.updateTimingsStartingFromTaskIdx(vehicleEntry.vehicle,
166                                                        stopTask.getTaskIdx() + 2, driveFromPickupTask.getEndTime());
167                                        ///////
168                                }
169
170                                return;
171                        } else {
172                                StayTask stayOrStopTask = stayTask != null ? stayTask : stopTask;
173
174                                // remove drive i->i+1 (if there is one)
175                                if (insertion.pickupIdx < stops.size()) {// there is at least one following stop
176                                        DrtStopTask nextStopTask = stops.get(insertion.pickupIdx).task;
177
178                                        // check: if there is at most one drive task in between
179                                        if (stayOrStopTask.getTaskIdx() + 2 != nextStopTask.getTaskIdx() //
180                                                        && stayTask != null && stayTask.getTaskIdx() + 1 != nextStopTask.getTaskIdx()) {
181                                                throw new RuntimeException();
182                                        }
183                                        if (stayOrStopTask.getTaskIdx() + 2 == nextStopTask.getTaskIdx()) {
184                                                // removing the drive task that is in between
185                                                int driveTaskIdx = stayOrStopTask.getTaskIdx() + 1;
186                                                schedule.removeTask(schedule.getTasks().get(driveTaskIdx));
187                                        }
188                                }
189
190                                if (stayTask != null && request.getFromLink() == stayTask.getLink()) {
191                                        // the bus stays where it is
192                                        beforePickupTask = stayTask;
193                                } else {// add drive task to pickup location
194                                        // insert drive i->pickup
195                                        VrpPathWithTravelData vrpPath = VrpPaths.createPath(stayOrStopTask.getLink(), request.getFromLink(),
196                                                        stayOrStopTask.getEndTime(), insertion.pathToPickup, travelTime);
197                                        beforePickupTask = taskFactory.createDriveTask(vehicleEntry.vehicle, vrpPath);
198                                        schedule.addTask(stayOrStopTask.getTaskIdx() + 1, beforePickupTask);
199                                }
200                        }
201                }
202
203                // insert pickup stop task
204                double startTime = beforePickupTask.getEndTime();
205                int taskIdx = beforePickupTask.getTaskIdx() + 1;
206                DrtStopTask pickupStopTask = taskFactory.createStopTask(vehicleEntry.vehicle, startTime,
207                                startTime + stopDuration, request.getFromLink());
208                schedule.addTask(taskIdx, pickupStopTask);
209                pickupStopTask.addPickupRequest(request);
210                request.setPickupTask(pickupStopTask);
211
212                // add drive from pickup
213                Link toLink = insertion.pickupIdx == insertion.dropoffIdx ? request.getToLink() // pickup->dropoff
214                                : stops.get(insertion.pickupIdx).task.getLink(); // pickup->i+1
215
216                VrpPathWithTravelData vrpPath = VrpPaths.createPath(request.getFromLink(), toLink, startTime + stopDuration,
217                                insertion.pathFromPickup, travelTime);
218                Task driveFromPickupTask = taskFactory.createDriveTask(vehicleEntry.vehicle, vrpPath);
219                schedule.addTask(taskIdx + 1, driveFromPickupTask);
220
221                // update timings
222                // TODO should be enough to update the timeline only till dropoffIdx...
223                scheduleTimingUpdater.updateTimingsStartingFromTaskIdx(vehicleEntry.vehicle, taskIdx + 2,
224                                driveFromPickupTask.getEndTime());
225        }

227        private void insertDropoff(VehicleData.Entry vehicleEntry, DrtRequest request, InsertionWithPathData insertion) {
228                Schedule schedule = vehicleEntry.vehicle.getSchedule();
229                List<Stop> stops = vehicleEntry.stops;
230
231                Task driveToDropoffTask;
232                if (insertion.pickupIdx == insertion.dropoffIdx) { // no drive to dropoff
233                        int pickupTaskIdx = request.getPickupTask().getTaskIdx();
234                        driveToDropoffTask = schedule.getTasks().get(pickupTaskIdx + 1);
235                } else {
236                        DrtStopTask stopTask = stops.get(insertion.dropoffIdx - 1).task;
237                        if (request.getToLink() == stopTask.getLink()) { // no detour; no new stop task
238                                // add dropoff request to stop task
239                                stopTask.addDropoffRequest(request);
240                                request.setDropoffTask(stopTask);
241                                return;
242                        } else { // add drive task to dropoff location
243
244                                // remove drive j->j+1 (if j is not the last stop)
245                                if (insertion.dropoffIdx < stops.size()) {
246                                        DrtStopTask nextStopTask = stops.get(insertion.dropoffIdx).task;
247                                        if (stopTask.getTaskIdx() + 2 != nextStopTask.getTaskIdx()) {
248                                                throw new IllegalStateException();
249                                        }
250                                        int driveTaskIdx = stopTask.getTaskIdx() + 1;
251                                        schedule.removeTask(schedule.getTasks().get(driveTaskIdx));
252                                }
253
254                                // insert drive i->dropoff
255                                VrpPathWithTravelData vrpPath = VrpPaths.createPath(stopTask.getLink(), request.getToLink(),
256                                                stopTask.getEndTime(), insertion.pathToDropoff, travelTime);
257                                driveToDropoffTask = taskFactory.createDriveTask(vehicleEntry.vehicle, vrpPath);
258                                schedule.addTask(stopTask.getTaskIdx() + 1, driveToDropoffTask);
259                        }
260                }
261
262                // insert dropoff stop task
263                double startTime = driveToDropoffTask.getEndTime();
264                int taskIdx = driveToDropoffTask.getTaskIdx() + 1;
265                DrtStopTask dropoffStopTask = taskFactory.createStopTask(vehicleEntry.vehicle, startTime,
266                                startTime + stopDuration, request.getToLink());
267                schedule.addTask(taskIdx, dropoffStopTask);
268                dropoffStopTask.addDropoffRequest(request);
269                request.setDropoffTask(dropoffStopTask);
270
271                // add drive from dropoff
272                if (insertion.dropoffIdx == stops.size()) {// bus stays at dropoff
273                        if (taskIdx + 2 == schedule.getTaskCount()) {// remove stay task from the end of schedule,
274                                DrtStayTask oldStayTask = (DrtStayTask)schedule.getTasks().get(taskIdx + 1);
275                                schedule.removeTask(oldStayTask);
276                        }
277                        if (taskIdx + 1 == schedule.getTaskCount()) {
278
279                                schedule.addTask(taskFactory.createStayTask(vehicleEntry.vehicle, dropoffStopTask.getEndTime(),
280                                                vehicleEntry.vehicle.getServiceEndTime(), dropoffStopTask.getLink()));
281                        } else {
282                                throw new RuntimeException();
283                        }
284                } else {
285                        Link toLink = stops.get(insertion.dropoffIdx).task.getLink(); // dropoff->j+1
286
287                        VrpPathWithTravelData vrpPath = VrpPaths.createPath(request.getToLink(), toLink, startTime + stopDuration,
288                                        insertion.pathFromDropoff, travelTime);
289                        Task driveFromDropoffTask = taskFactory.createDriveTask(vehicleEntry.vehicle, vrpPath);
290                        schedule.addTask(taskIdx + 1, driveFromDropoffTask);
291
292                        // update timings
293                        scheduleTimingUpdater.updateTimingsStartingFromTaskIdx(vehicleEntry.vehicle, taskIdx + 2,
294                                        driveFromDropoffTask.getEndTime());
295                }
296        }
