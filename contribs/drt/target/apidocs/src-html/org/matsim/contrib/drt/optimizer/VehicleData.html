039/**
040 * @author michalm
041 */
042public class VehicleData {
043        public static class Entry {
044                public final Vehicle vehicle;
045                public final LinkTimePair start;
046                public final int startOccupancy;
047                public final ImmutableList<Stop> stops;
048
049                public Entry(Vehicle vehicle, LinkTimePair start, int startOccupancy, ImmutableList<Stop> stops) {
050                        this.vehicle = vehicle;
051                        this.start = start;
052                        this.startOccupancy = startOccupancy;
053                        this.stops = stops;
054                }
055        }
056
057        public static class Stop {
058                public final DrtStopTask task;
059                public final double maxArrivalTime;// relating to max pass drive time (for dropoff requests)
060                public final double maxDepartureTime;// relating to pass max wait time (for pickup requests)
061                public final int occupancyChange;// diff in pickups and dropoffs
062                public final int outgoingOccupancy;
063
064                public Stop(DrtStopTask task, int outputOccupancy) {
065                        this.task = task;
066                        this.outgoingOccupancy = outputOccupancy;
067
068                        maxArrivalTime = calcMaxArrivalTime();
069                        maxDepartureTime = calcMaxDepartureTime();
070                        occupancyChange = task.getPickupRequests().size() - task.getDropoffRequests().size();
071                }
072
073                private double calcMaxArrivalTime() {
074                        double maxTime = Double.MAX_VALUE;
075                        for (DrtRequest r : task.getDropoffRequests()) {
076                                double reqMaxArrivalTime = r.getLatestArrivalTime();
077                                if (reqMaxArrivalTime < maxTime) {
078                                        maxTime = reqMaxArrivalTime;
079                                }
080                        }
081                        return maxTime;
082                }
083
084                private double calcMaxDepartureTime() {
085                        double maxTime = Double.MAX_VALUE;
086                        for (DrtRequest r : task.getPickupRequests()) {
087                                double reqMaxDepartureTime = r.getLatestStartTime();
088                                if (reqMaxDepartureTime < maxTime) {
089                                        maxTime = reqMaxDepartureTime;
090                                }
091                        }
092                        return maxTime;
093                }
094
095                @Override
096                public String toString() {
097                        return "VehicleData.Stop for: " + task.toString();
098                }
099        }
100
101        public interface EntryFactory {
102                Entry create(Vehicle vehicle, double currentTime);
103        }
104
105        private final double currentTime;
106        private final EntryFactory entryFactory;
107        private final Map<Id<Vehicle>, Entry> entries;
108
109        public VehicleData(double currentTime, Stream<? extends Vehicle> vehicles, EntryFactory entryFactory,
110                        ForkJoinPool forkJoinPool) {
111                this.currentTime = currentTime;
112                this.entryFactory = entryFactory;
113                try {
114                        entries = forkJoinPool.submit(() -> vehicles.parallel()//
115                                        .map(v -> entryFactory.create(v, currentTime))//
116                                        .filter(Objects::nonNull)//
117                                        .collect(Collectors.toMap(e -> e.vehicle.getId(), e -> e)))//
118                                        .get();
119                } catch (InterruptedException | ExecutionException e) {
120                        throw new RuntimeException(e);
121                }
122        }
123
124        public void updateEntry(Vehicle vehicle) {
125                Entry e = entryFactory.create(vehicle, currentTime);
126                if (e != null) {
127                        entries.put(vehicle.getId(), e);
128                } else {
129                        entries.remove(vehicle.getId());
130                }
131        }
132
133        public int getSize() {
134                return entries.size();
135        }
136
137        public Collection<Entry> getEntries() {
138                return Collections.unmodifiableCollection(entries.values());
139        }
140}
