020package org.matsim.contrib.drt.routing;
021
022import java.util.Collections;
023import java.util.List;
024
025import org.matsim.api.core.v01.Coord;
026import org.matsim.api.core.v01.TransportMode;
027import org.matsim.api.core.v01.network.Network;
028import org.matsim.api.core.v01.population.Leg;
029import org.matsim.api.core.v01.population.Person;
030import org.matsim.api.core.v01.population.PlanElement;
031import org.matsim.api.core.v01.population.Route;
032import org.matsim.contrib.drt.run.DrtConfigGroup;
033import org.matsim.core.config.Config;
034import org.matsim.core.population.PopulationUtils;
035import org.matsim.core.population.routes.RouteUtils;
036import org.matsim.core.router.EmptyStageActivityTypes;
037import org.matsim.core.router.RoutingModule;
038import org.matsim.core.router.StageActivityTypes;
039import org.matsim.core.utils.geometry.CoordUtils;
040import org.matsim.facilities.Facility;
041
042import com.google.inject.Inject;
043
044public class DrtRoutingModule implements RoutingModule {
045        private final DrtConfigGroup drtconfig;
046        private StageActivityTypes stageActivityTypes;
047        private final Network network;
048
049        @Inject
050        public DrtRoutingModule(Config config, Network network) {
051                this.drtconfig = DrtConfigGroup.get(config);
052                this.network = network;
053        }
054
055        @Override
056        public List<? extends PlanElement> calcRoute(Facility<?> fromFacility, Facility<?> toFacility, double departureTime,
057                        Person person) {
058                Route route = RouteUtils.createGenericRouteImpl(fromFacility.getLinkId(), toFacility.getLinkId());
059                Coord fromCoord = fromFacility.getCoord();
060                Coord toCoord = toFacility.getCoord();
061
062                if (fromCoord == null) {
063                        fromCoord = network.getLinks().get(fromFacility.getLinkId()).getCoord();
064                }
065                if (toCoord == null) {
066                        toCoord = network.getLinks().get(toFacility.getLinkId()).getCoord();
067                }
068                double distanceEstimate = CoordUtils.calcEuclideanDistance(fromCoord, toCoord)
069                                * drtconfig.getEstimatedBeelineDistanceFactor();
070                route.setDistance(distanceEstimate);
071                route.setTravelTime(distanceEstimate / drtconfig.getEstimatedDrtSpeed());
072
073                Leg leg = PopulationUtils.createLeg(TransportMode.drt);
074                leg.setDepartureTime(departureTime);
075                leg.setTravelTime(route.getTravelTime());
076                leg.setRoute(route);
077                if (fromFacility.getLinkId().equals(toFacility.getLinkId())) {
078                        leg.setMode(TransportMode.walk);
079                }
080
081                return Collections.singletonList(leg);
082        }
083
084        /**
085         * @param stageActivityTypes
086         *            the stageActivityTypes to set
087         */
088        public void setStageActivityTypes(StageActivityTypes stageActivityTypes) {
089                this.stageActivityTypes = stageActivityTypes;
090        }
091
092        @Override
093        public StageActivityTypes getStageActivityTypes() {
094                return this.stageActivityTypes != null ? this.stageActivityTypes : EmptyStageActivityTypes.INSTANCE;
095        }
096}
