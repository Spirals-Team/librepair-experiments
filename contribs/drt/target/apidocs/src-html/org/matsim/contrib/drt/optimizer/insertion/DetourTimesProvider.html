019package org.matsim.contrib.drt.optimizer.insertion;
020
021import java.util.ArrayList;
022import java.util.List;
023
024import org.matsim.api.core.v01.network.Link;
025import org.matsim.contrib.drt.data.DrtRequest;
026import org.matsim.contrib.drt.optimizer.VehicleData.Entry;
027import org.matsim.contrib.drt.optimizer.VehicleData.Stop;
028
029/**
030 * @author michalm
031 */
032public class DetourTimesProvider {
033        class DetourTimesSet {
034                // times[0] is a special entry; times[i] corresponds to stop i-1, for 1 <= i <= stopCount
035                public final Double[] timesToPickup;// times[0] start->pickup
036                public final Double[] timesFromPickup;// times[0] pickup->dropoff
037                public final Double[] timesToDropoff;// times[0] NaN
038                public final Double[] timesFromDropoff;// times[0] NaN
039
040                public DetourTimesSet(Double[] timesToPickup, Double[] timesFromPickup, Double[] timesToDropoff,
041                                Double[] timesFromDropoff) {
042                        this.timesToPickup = timesToPickup;
043                        this.timesFromPickup = timesFromPickup;
044                        this.timesToDropoff = timesToDropoff;
045                        this.timesFromDropoff = timesFromDropoff;
046                }
047        }
048
049        private final double stopDuration;
050        private final DetourTimeEstimator detourTimeEstimator;
051
052        public DetourTimesProvider(DetourTimeEstimator detourTimeEstimator, double stopDuration) {
053                this.detourTimeEstimator = detourTimeEstimator;
054                this.stopDuration = stopDuration;
055        }
056
057        public DetourTimesSet getDetourTimesSet(DrtRequest drtRequest, Entry vEntry) {
058                ArrayList<Link> links = new ArrayList<>(vEntry.stops.size() + 1);
059                links.add(null);// special link
060                for (Stop s : vEntry.stops) {
061                        links.add(s.task.getLink());
062                }
063
064                double earliestPickupTime = drtRequest.getEarliestStartTime();// over-optimistic
065
066                // calc backward TTs from pickup to ends of all stop + start
067                links.set(0, vEntry.start.link);
068                Double[] timesToPickup = estimateTimesBackwards(drtRequest.getFromLink(), links, earliestPickupTime);
069
070                // calc forward TTs from pickup to beginnings of all stops + dropoff
071                links.set(0, drtRequest.getToLink());
072                Double[] timesFromPickup = estimateTimesForwards(drtRequest.getFromLink(), links, earliestPickupTime);
073
074                double pickupToDropoffTime = timesFromPickup[0];// only if no other passengers on board (optimistic)
075                double minTravelTime = pickupToDropoffTime;
076                double earliestDropoffTime = earliestPickupTime + minTravelTime + stopDuration; // over-optimistic
077
078                // calc backward TTs from dropoff to ends of all stops
079                links.set(0, drtRequest.getToLink());// TODO change to null (after nulls are supported by OneToManyPathSearch)
080                Double[] timesToDropoff = estimateTimesBackwards(drtRequest.getToLink(), links, earliestDropoffTime);
081                timesToDropoff[0] = null;
082
083                // calc forward TTs from dropoff to beginnings of all stops
084                Double[] timesFromDropoff = estimateTimesForwards(drtRequest.getToLink(), links, earliestDropoffTime);
085                timesFromDropoff[0] = null;
086
087                return new DetourTimesSet(timesToPickup, timesFromPickup, timesToDropoff, timesFromDropoff);
088        }
089
090        private Double[] estimateTimesBackwards(Link fromLink, List<Link> toLinks, double startTime) {
091                Double[] times = new Double[toLinks.size()];
092                for (int i = 0; i < times.length; i++) {
093                        times[i] = detourTimeEstimator.estimateTime(toLinks.get(i), fromLink);
094                }
095                return times;
096        }
097
098        private Double[] estimateTimesForwards(Link fromLink, List<Link> toLinks, double startTime) {
099                Double[] times = new Double[toLinks.size()];
100                for (int i = 0; i < times.length; i++) {
101                        times[i] = detourTimeEstimator.estimateTime(fromLink, toLinks.get(i));
102                }
103                return times;
104        }
105}
