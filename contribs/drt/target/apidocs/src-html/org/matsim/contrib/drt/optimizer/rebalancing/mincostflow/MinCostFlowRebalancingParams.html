019package org.matsim.contrib.drt.optimizer.rebalancing.mincostflow;
020
021import java.util.Map;
022
023import javax.validation.constraints.Positive;
024import javax.validation.constraints.PositiveOrZero;
025
026import org.matsim.core.config.ReflectiveConfigGroup;
027
028/**
029 * @author michalm
030 */
031public class MinCostFlowRebalancingParams extends ReflectiveConfigGroup {
032        public static boolean isRebalancingEnabled(MinCostFlowRebalancingParams params) {
033                return params != null && params.getInterval() > 0;
034        }
035
036        public static final String SET_NAME = "minCostFlowRebalancing";
037
038        public static final String INTERVAL = "interval";
039        static final String INTERVAL_EXP = "Specifies how often empty vehicle rebalancing is executed."
040                        + " 0 s means no rebalancing. Default is 1800 s.";
041
042        public static final String MIN_SERVICE_TIME = "minServiceTime";
043        static final String MIN_SERVICE_TIME_EXP = //
044                        "Minimum remaining service time of an idle/busy vehicle to be considered as rebalancable/soon-idle (respectively)."
045                                        + " Default is 3600 s. In general, should be higher than interval (e.g. 2 x interval).";
046
047        public static final String MAX_TIME_BEFORE_IDLE = "maxTimeBeforeIdle";
048        static final String MAX_TIME_BEFORE_IDLE_EXP = //
049                        "Maximum remaining time before busy vehicle becomes idle to be considered as soon-idle vehicle."
050                                        + " Default is 900 s. In general should be lower than interval (e.g. 0.5 x interval)";
051
052        public static final String TARGET_ALPHA = "targetAlpha";
053        static final String TARGET_ALPHA_EXP = "alpha coefficient in linear target calculation."
054                        + " In general, should be lower then 1.0 to prevent over-reacting and high empty milleage.";
055
056        public static final String TARGET_BETA = "targetBeta";
057        static final String TARGET_BETA_EXP = "beta constant in linear target calculation."
058                        + " In general, should be lower then 1.0 to prevent over-reacting and high empty milleage.";
059
060        public static final String CELL_SIZE = "cellSize";
061        static final String CELL_SIZE_EXP = "size of square cells used for demand aggregation."
062                        + " Depends on demand, supply and network. Often used with values in the range of 500 - 2000 m";
063
064        @PositiveOrZero
065        private int interval = 1800;// [s], if 0 then no rebalancing
066
067        @Positive
068        public double minServiceTime = 2 * interval;// [s]
069
070        @PositiveOrZero
071        public double maxTimeBeforeIdle = 0.5 * interval;// [s], if 0 then soon-idle vehicle will not be considered
072
073        @PositiveOrZero
074        public double targetAlpha = Double.NaN;
075
076        @PositiveOrZero
077        public double targetBeta = Double.NaN;
078
079        @Positive
080        public double cellSize = Double.NaN;// [m]
081
082        public MinCostFlowRebalancingParams() {
083                super(SET_NAME);
084        }
085
086        @Override
087        public Map<String, String> getComments() {
088                Map<String, String> map = super.getComments();
089                map.put(INTERVAL, INTERVAL_EXP);
090                map.put(MIN_SERVICE_TIME, MIN_SERVICE_TIME_EXP);
091                map.put(MAX_TIME_BEFORE_IDLE, MAX_TIME_BEFORE_IDLE_EXP);
092                map.put(TARGET_ALPHA, TARGET_ALPHA_EXP);
093                map.put(TARGET_BETA, TARGET_BETA_EXP);
094                map.put(CELL_SIZE, CELL_SIZE_EXP);
095                return map;
096        }
097
098        /**
099         * @return -- {@value #INTERVAL_EXP}
100         */
101        @StringGetter(INTERVAL)
102        public int getInterval() {
103                return interval;
104        }
105
106        /**
107         * @param interval
108         *            -- {@value #INTERVAL_EXP}
109         */
110        @StringSetter(INTERVAL)
111        public void setInterval(int interval) {
112                this.interval = interval;
113        }
114
115        /**
116         * @return -- {@value #MIN_SERVICE_TIME_EXP}
117         */
118        @StringGetter(MIN_SERVICE_TIME)
119        public double getMinServiceTime() {
120                return minServiceTime;
121        }
122
123        /**
124         * @param minServiceTime
125         *            -- {@value #MIN_SERVICE_TIME_EXP}
126         */
127        @StringSetter(MIN_SERVICE_TIME)
128        public void setMinServiceTime(double minServiceTime) {
129                this.minServiceTime = minServiceTime;
130        }
131
132        /**
133         * @return -- {@value #MAX_TIME_BEFORE_IDLE_EXP}
134         */
135        @StringGetter(MAX_TIME_BEFORE_IDLE)
136        public double getMaxTimeBeforeIdle() {
137                return maxTimeBeforeIdle;
138        }
139
140        /**
141         * @param maxTimeBeforeIdle--
142         *            {@value #MAX_TIME_BEFORE_IDLE_EXP}
143         */
144        @StringSetter(MAX_TIME_BEFORE_IDLE)
145        public void setMaxTimeBeforeIdle(double maxTimeBeforeIdle) {
146                this.maxTimeBeforeIdle = maxTimeBeforeIdle;
147        }
148
149        /**
150         * @return -- {@value #TARGET_ALPHA_EXP}
151         */
152        @StringGetter(TARGET_ALPHA)
153        public double getTargetAlpha() {
154                return targetAlpha;
155        }
156
157        /**
158         * @param targetAlpha
159         *            -- {@value #TARGET_ALPHA_EXP}
160         */
161        @StringSetter(TARGET_ALPHA)
162        public void setTargetAlpha(double targetAlpha) {
163                this.targetAlpha = targetAlpha;
164        }
165
166        /**
167         * @return -- {@value #TARGET_BETA_EXP}
168         */
169        @StringGetter(TARGET_BETA)
170        public double getTargetBeta() {
171                return targetBeta;
172        }
173
174        /**
175         * @param targetBeta
176         *            -- {@value #TARGET_BETA_EXP}
177         */
178        @StringSetter(TARGET_BETA)
179        public void setTargetBeta(double targetBeta) {
180                this.targetBeta = targetBeta;
181        }
182
183        /**
184         * @return -- {@value #CELL_SIZE_EXP}
185         */
186        @StringGetter(CELL_SIZE)
187        public double getCellSize() {
188                return cellSize;
189        }
190
191        /**
192         * @param cellSize
193         *            -- {@value #CELL_SIZE_EXP}
194         */
195        @StringSetter(CELL_SIZE)
196        public void setCellSize(double cellSize) {
197                this.cellSize = cellSize;
198        }
199}
