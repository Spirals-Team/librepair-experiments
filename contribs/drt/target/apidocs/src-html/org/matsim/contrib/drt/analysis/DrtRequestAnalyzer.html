062/**
063 *
064 */
065public class DrtRequestAnalyzer implements DrtRequestRejectedEventHandler, DrtRequestScheduledEventHandler,
066                DrtRequestSubmittedEventHandler, PersonEntersVehicleEventHandler {
067
068        
069        private final Map<Id<Request>,DrtRequestSubmittedEvent> submittedRequests = new HashMap<>();
070        private final Map<Id<Request>,Tuple<Double,Double>> waitTimeCompare = new HashMap<>();
071        private final Map<Id<Person>,DrtRequestScheduledEvent> scheduledRequests = new HashMap<>();
072        private final List<String> rejections = new ArrayList<>();
073        private final Network network;
074        private final DrtConfigGroup drtCfg;
075        
076        @Inject
077        public DrtRequestAnalyzer(EventsManager events, Network network, DrtConfigGroup drtCfg) {
078                events.addHandler(this);
079                this.network = network;
080                this.drtCfg = drtCfg;
081        }
082        
083        
084        /* (non-Javadoc)
085         * @see org.matsim.core.events.handler.EventHandler#reset(int)
086         */
087        @Override
088        public void reset(int iteration) {
089                submittedRequests.clear();
090                scheduledRequests.clear();
091                waitTimeCompare.clear();
092                rejections.clear();
093        }
094
095        /* (non-Javadoc)
096         * @see org.matsim.api.core.v01.events.handler.PersonEntersVehicleEventHandler#handleEvent(org.matsim.api.core.v01.events.PersonEntersVehicleEvent)
097         */
098        @Override
099        public void handleEvent(PersonEntersVehicleEvent event) {
100                if (this.scheduledRequests.containsKey(event.getPersonId())){
101                        DrtRequestScheduledEvent scheduled = scheduledRequests.remove(event.getPersonId());
102                        DrtRequestSubmittedEvent submission  = this.submittedRequests.remove(scheduled.getRequestId());
103                        double actualWaitTime = event.getTime() - submission.getTime();
104                        double estimatedWaitTime =  scheduled.getPickupTime() - submission.getTime();
105                        waitTimeCompare.put(submission.getRequestId(), new Tuple<>(actualWaitTime,estimatedWaitTime));
106                        
107                }
108        }
109
110        /* (non-Javadoc)
111         * @see org.matsim.contrib.drt.passenger.events.DrtRequestSubmittedEventHandler#handleEvent(org.matsim.contrib.drt.passenger.events.DrtRequestScheduledEvent)
112         */
113        @Override
114        public void handleEvent(DrtRequestScheduledEvent event) {
115                DrtRequestSubmittedEvent submission  = this.submittedRequests.get(event.getRequestId());
116                if (submission!=null){
117                        this.scheduledRequests.put(submission.getPersonId(),event);
118                }
119                else throw new RuntimeException("Vehicle allocation without submission?");
120        }
121
122        /* (non-Javadoc)
123         * @see org.matsim.contrib.drt.passenger.events.DrtRequestScheduledEventHandler#handleEvent(org.matsim.contrib.drt.passenger.events.DrtRequestSubmittedEvent)
124         */
125        @Override
126        public void handleEvent(DrtRequestSubmittedEvent event) {
127                this.submittedRequests.put(event.getRequestId(), event);
128        }
129
130        /* (non-Javadoc)
131         * @see org.matsim.contrib.drt.passenger.events.DrtRequestRejectedEventHandler#handleEvent(org.matsim.contrib.drt.passenger.events.DrtRequestRejectedEvent)
132         */
133        @Override
134        public void handleEvent(DrtRequestRejectedEvent event) {
135                DrtRequestSubmittedEvent submission = this.submittedRequests.remove(event.getRequestId());
136                Coord fromCoord = network.getLinks().get(submission.getFromLinkId()).getCoord();
137                Coord toCoord = network.getLinks().get(submission.getToLinkId()).getCoord();
138                this.rejections.add(submission.getTime()+";"+submission.getPersonId()+";"+submission.getFromLinkId()+";"+submission.getToLinkId()+";"+fromCoord.getX()+";"+fromCoord.getY()+";"+toCoord.getX()+";"+toCoord.getY());
139        }
140        
141        /**
142         * @return the waitTimeCompare
143         */
144        public Map<Id<Request>, Tuple<Double, Double>> getWaitTimeCompare() {
145                return waitTimeCompare;
146        }
147        
148        /**
149         * @return the rejections
150         */
151        public List<String> getRejections() {
152                return rejections;
153        }
154        
155        public void writeAndPlotWaitTimeEstimateComparison(String plotFileName, String textFileName) {
156                BufferedWriter bw = IOUtils.getBufferedWriter(textFileName);
157
158                XYSeries times = new XYSeries("waittimes", true, true);
159
160                try {
161                        bw.append("RequestId;actualWaitTime;estimatedWaitTime;deviate");
162                        for (Entry<Id<Request>, Tuple<Double, Double>> e : this.waitTimeCompare.entrySet()){
163                                bw.newLine();
164                                double first = e.getValue().getFirst();
165                                double second = e.getValue().getSecond();
166                                bw.append(e.getKey().toString()+";"+first+";"+second+";"+(first-second));
167                                times.add(first, second);
168                        }
169                        bw.flush();
170                        bw.close();
171                        
172                        
173                        final JFreeChart chart2 = DensityScatterPlots.createPlot("Wait times", "Actual wait time [s]",
174                                        "Initially planned wait time [s]", times, Pair.of(0., drtCfg.getMaxWaitTime()));
175//                      xAxis.setLowerBound(0);
176//                      yAxis.setLowerBound(0);
177                        ChartUtilities.writeChartAsPNG(new FileOutputStream(plotFileName), chart2, 1500, 1500);
178                } catch (IOException e) {
179                        // TODO Auto-generated catch block
180                        e.printStackTrace();
181                }
182        }
183}
