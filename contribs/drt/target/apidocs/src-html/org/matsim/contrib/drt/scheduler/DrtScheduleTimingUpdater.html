019package org.matsim.contrib.drt.scheduler;
020
021import java.util.List;
022
023import org.matsim.contrib.drt.run.DrtConfigGroup;
024import org.matsim.contrib.drt.schedule.DrtTask;
025import org.matsim.contrib.dvrp.data.Vehicle;
026import org.matsim.contrib.dvrp.path.VrpPathWithTravelData;
027import org.matsim.contrib.dvrp.schedule.DriveTask;
028import org.matsim.contrib.dvrp.schedule.Schedule;
029import org.matsim.contrib.dvrp.schedule.Schedule.ScheduleStatus;
030import org.matsim.contrib.dvrp.schedule.Schedules;
031import org.matsim.contrib.dvrp.schedule.Task;
032import org.matsim.contrib.dvrp.tracker.TaskTrackers;
033import org.matsim.core.mobsim.framework.MobsimTimer;
034import org.matsim.core.utils.misc.Time;
035
036import com.google.inject.Inject;
037
038/**
039 * @author michalm
040 */
041public class DrtScheduleTimingUpdater {
042        private final double stopDuration;
043        private final MobsimTimer timer;
044
045        @Inject
046        public DrtScheduleTimingUpdater(DrtConfigGroup drtCfg, MobsimTimer timer) {
047                this.stopDuration = drtCfg.getStopDuration();
048                this.timer = timer;
049        }
050
051        /**
052         * Check and decide if the schedule should be updated due to if vehicle is Update timings (i.e. beginTime and
053         * endTime) of all tasks in the schedule.
054         */
055        public void updateBeforeNextTask(Vehicle vehicle) {
056                Schedule schedule = vehicle.getSchedule();
057                // Assumption: there is no delay as long as the schedule has not been started (PLANNED)
058                if (schedule.getStatus() != ScheduleStatus.STARTED) {
059                        return;
060                }
061
062                updateTimingsStartingFromCurrentTask(vehicle, timer.getTimeOfDay());
063        }
064
065        public void updateTimings(Vehicle vehicle) {
066                Schedule schedule = vehicle.getSchedule();
067                if (schedule.getStatus() != ScheduleStatus.STARTED) {
068                        return;
069                }
070
071                double predictedEndTime = TaskTrackers.predictEndTime(schedule.getCurrentTask(), timer.getTimeOfDay());
072                updateTimingsStartingFromCurrentTask(vehicle, predictedEndTime);
073        }
074
075        private void updateTimingsStartingFromCurrentTask(Vehicle vehicle, double newEndTime) {
076                Schedule schedule = vehicle.getSchedule();
077                Task currentTask = schedule.getCurrentTask();
078                if (currentTask.getEndTime() != newEndTime) {
079                        currentTask.setEndTime(newEndTime);
080                        updateTimingsStartingFromTaskIdx(vehicle, currentTask.getTaskIdx() + 1, newEndTime);
081                }
082        }
083
084        void updateTimingsStartingFromTaskIdx(Vehicle vehicle, int startIdx, double newBeginTime) {
085                Schedule schedule = vehicle.getSchedule();
086                List<? extends Task> tasks = schedule.getTasks();
087
088                for (int i = startIdx; i < tasks.size(); i++) {
089                        DrtTask task = (DrtTask)tasks.get(i);
090                        double calcEndTime = calcNewEndTime(vehicle, task, newBeginTime);
091
092                        if (Time.isUndefinedTime(calcEndTime)) {
093                                schedule.removeTask(task);
094                                i--;
095                        } else if (calcEndTime < newBeginTime) {// 0 s is fine (e.g. last 'wait')
096                                throw new IllegalStateException();
097                        } else {
098                                task.setBeginTime(newBeginTime);
099                                task.setEndTime(calcEndTime);
100                                newBeginTime = calcEndTime;
101                        }
102                }
103        }
104
105        private double calcNewEndTime(Vehicle vehicle, DrtTask task, double newBeginTime) {
106                switch (task.getDrtTaskType()) {
107                        case STAY: {
108                                if (Schedules.getLastTask(vehicle.getSchedule()).equals(task)) {// last task
109                                        // even if endTime=beginTime, do not remove this task!!! A DRT schedule should end with WAIT
110                                        return Math.max(newBeginTime, vehicle.getServiceEndTime());
111                                } else {
112                                        // if this is not the last task then some other task (e.g. DRIVE or PICKUP)
113                                        // must have been added at time submissionTime <= t
114                                        double oldEndTime = task.getEndTime();
115                                        if (oldEndTime <= newBeginTime) {// may happen if the previous task is delayed
116                                                return Time.UNDEFINED_TIME;// remove the task
117                                        } else {
118                                                return oldEndTime;
119                                        }
120                                }
121                        }
122
123                        case DRIVE: {
124                                // cannot be shortened/lengthen, therefore must be moved forward/backward
125                                VrpPathWithTravelData path = (VrpPathWithTravelData)((DriveTask)task).getPath();
126                                // TODO one may consider recalculation of SP!!!!
127                                return newBeginTime + path.getTravelTime();
128                        }
129
130                        case STOP: {
131                                // TODO does not consider prebooking!!!
132                                double duration = stopDuration;
133                                return newBeginTime + duration;
134                        }
135
136                        default:
137                                throw new IllegalStateException();
138                }
139        }
140
141}
