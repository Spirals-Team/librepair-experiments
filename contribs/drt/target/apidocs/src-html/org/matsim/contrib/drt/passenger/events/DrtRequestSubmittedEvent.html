020package org.matsim.contrib.drt.passenger.events;
021
022import java.util.Map;
023
024import org.matsim.api.core.v01.Id;
025import org.matsim.api.core.v01.events.Event;
026import org.matsim.api.core.v01.network.Link;
027import org.matsim.api.core.v01.population.Person;
028import org.matsim.contrib.dvrp.data.Request;
029
030/**
031 * @author michalm
032 */
033public class DrtRequestSubmittedEvent extends Event {
034        public static final String EVENT_TYPE = "DrtRequest submitted";
035
036        public static final String ATTRIBUTE_REQUEST = "request";
037        public static final String ATTRIBUTE_PERSON = "person";
038        public static final String ATTRIBUTE_FROM_LINK = "fromLink";
039        public static final String ATTRIBUTE_TO_LINK = "toLink";
040        public static final String ATTRIBUTE_UNSHARED_RIDE_TIME = "unsharedRideTime";
041        public static final String ATTRIBUTE_UNSHARED_RIDE_DISTANCE = "unsharedRideDistance";
042
043        private final Id<Request> requestId;
044        private final Id<Person> personId;
045        private final Id<Link> fromLinkId;
046        private final Id<Link> toLinkId;
047        private final double unsharedRideTime;
048        private final double unsharedRideDistance;
049
050        public DrtRequestSubmittedEvent(double time, Id<Request> requestId, Id<Person> personId, Id<Link> fromLinkId,
051                        Id<Link> toLinkId, double unsharedRideTime, double unsharedRideDistance) {
052                super(time);
053                this.requestId = requestId;
054                this.personId = personId;
055                this.fromLinkId = fromLinkId;
056                this.toLinkId = toLinkId;
057                this.unsharedRideTime = unsharedRideTime;
058                this.unsharedRideDistance = unsharedRideDistance;
059        }
060
061        @Override
062        public String getEventType() {
063                return EVENT_TYPE;
064        }
065
066        /**
067         * the ID of the initial request submitted
068         */
069        public Id<Request> getRequestId() {
070                return requestId;
071        }
072
073        /**
074         * the Person Id that submitted the request
075         */
076        public Id<Person> getPersonId() {
077                return personId;
078        }
079
080        /**
081         * the request's origin
082         */
083        public Id<Link> getFromLinkId() {
084                return fromLinkId;
085        }
086
087        /**
088         * the request's destination
089         */
090        public Id<Link> getToLinkId() {
091                return toLinkId;
092        }
093
094        /**
095         * the estimated traveltime it would take to ride without any detours
096         */
097        public double getUnsharedRideTime() {
098                return unsharedRideTime;
099        }
100
101        /**
102         * the estimated distance it would take to ride without any detours
103         */
104        public double getUnsharedRideDistance() {
105                return unsharedRideDistance;
106        }
107
108        @Override
109        public Map<String, String> getAttributes() {
110                Map<String, String> attr = super.getAttributes();
111                attr.put(ATTRIBUTE_REQUEST, requestId + "");
112                attr.put(ATTRIBUTE_PERSON, personId + "");
113                attr.put(ATTRIBUTE_FROM_LINK, fromLinkId + "");
114                attr.put(ATTRIBUTE_TO_LINK, toLinkId + "");
115                attr.put(ATTRIBUTE_UNSHARED_RIDE_TIME, unsharedRideTime + "");
116                attr.put(ATTRIBUTE_UNSHARED_RIDE_DISTANCE, unsharedRideDistance + "");
117                return attr;
118        }
119}
