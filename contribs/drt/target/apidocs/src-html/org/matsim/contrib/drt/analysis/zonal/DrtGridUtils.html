023package org.matsim.contrib.drt.analysis.zonal;

025import java.util.HashMap;
026import java.util.Map;
027
028import org.matsim.api.core.v01.network.Network;
029import org.matsim.core.network.NetworkUtils;

031import com.vividsolutions.jts.geom.Coordinate;
032import com.vividsolutions.jts.geom.Geometry;
033import com.vividsolutions.jts.geom.GeometryFactory;
034import com.vividsolutions.jts.geom.Polygon;
035
036/**
037 * @author  jbischoff
038 *
039 */
040/**
041 *
042 */
043public class DrtGridUtils {
044
045        public static Map<String,Geometry> createGridFromNetwork(Network network, double cellsize){
046                
047                double[] boundingbox = NetworkUtils.getBoundingBox(network.getNodes().values());
048                double minX = (Math.floor(boundingbox[0] / cellsize)*cellsize);
049                double maxX = (Math.floor(boundingbox[2] / cellsize)*cellsize);
050                double minY = (Math.floor(boundingbox[1] / cellsize)*cellsize);
051                double maxY = (Math.floor(boundingbox[3] / cellsize)*cellsize);
052                GeometryFactory gf = new GeometryFactory();
053                Map<String,Geometry> grid = new HashMap<>();
054                int cell = 0;
055                for (double lx = minX;lx<maxX;lx +=cellsize ){
056                                
057                        for (double by = minY;by<maxY;by+=cellsize){
058                                cell++;
059                                Coordinate p1 = new Coordinate(lx, by);
060                                Coordinate p2 = new Coordinate(lx+cellsize, by);
061                                Coordinate p3 = new Coordinate(lx+cellsize, by+cellsize);
062                                Coordinate p4 = new Coordinate(lx, by+cellsize);
063                                Coordinate [] ca = {p1, p2, p3, p4, p1};
064                                Polygon p = new Polygon(gf.createLinearRing(ca), null, gf); 
065                                grid.put(cell+"", p);
066                        }
067                }
068                
069                return grid;
070        } 
071}
