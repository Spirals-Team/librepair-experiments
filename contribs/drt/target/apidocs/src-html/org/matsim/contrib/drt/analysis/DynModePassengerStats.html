057/**
058 * @author jbischoff
059 *
060 */
061public class DynModePassengerStats implements PersonEntersVehicleEventHandler, PersonDepartureEventHandler,
062                PersonArrivalEventHandler, LinkEnterEventHandler, ActivityEndEventHandler, DrtRequestSubmittedEventHandler {
063
064        final private Map<Id<Person>, Double> departureTimes = new HashMap<>();
065        final private Map<Id<Person>, Id<Link>> departureLinks = new HashMap<>();
066        final private List<DynModeTrip> drtTrips = new ArrayList<>();
067        final private Map<Id<Vehicle>, Map<Id<Person>, MutableDouble>> inVehicleDistance = new HashMap<>();
068        final private Map<Id<Vehicle>, double[]> vehicleDistances = new HashMap<>();
069        final private Map<Id<Person>, DynModeTrip> currentTrips = new HashMap<>();
070        final private Map<Id<Person>, Double> unsharedDistances = new HashMap<>();
071        final private Map<Id<Person>, Double> unsharedTimes = new HashMap<>();
072        final String mode;
073        private final int maxcap ;
074        
075
076        final private Network network;
077        /*
078         * (non-Javadoc)
079         * 
080         * @see org.matsim.core.events.handler.EventHandler#reset(int)
081         */
082
083        /**
084         * 
085         */
086        @Inject
087        public DynModePassengerStats(Network network, EventsManager events, Config config, Fleet fleet) {
088                this.mode = ((DvrpConfigGroup)config.getModules().get(DvrpConfigGroup.GROUP_NAME)).getMode();
089                this.network = network;
090                events.addHandler(this);
091                maxcap = DynModeTripsAnalyser.findMaxCap(fleet);
092        }
093
094        public DynModePassengerStats(Network network, String mode, int maxcap) {
095                this.mode = mode;
096                this.network = network;
097                this.maxcap = maxcap;
098        }
099
100        @Override
101        public void reset(int iteration) {
102                drtTrips.clear();
103                departureTimes.clear();
104                departureLinks.clear();
105                inVehicleDistance.clear();
106                currentTrips.clear();
107                vehicleDistances.clear();
108                unsharedDistances.clear();
109                unsharedTimes.clear();
110        }
111
112        /*
113         * (non-Javadoc)
114         * 
115         * @see org.matsim.api.core.v01.events.handler.ActivityEndEventHandler#handleEvent(org.matsim.api.core.v01.events.
116         * ActivityEndEvent)
117         */
118        @Override
119        public void handleEvent(ActivityEndEvent event) {
120                if (event.getActType().equals(VrpAgentLogic.BEFORE_SCHEDULE_ACTIVITY_TYPE)) {
121                        Id<Vehicle> vid = Id.createVehicleId(event.getPersonId().toString());
122                        this.inVehicleDistance.put(vid, new HashMap<Id<Person>, MutableDouble>());
123                        this.vehicleDistances.put(vid, new double[3+maxcap]);
124                }
125        }
126
127        /*
128         * (non-Javadoc)
129         * 
130         * @see org.matsim.api.core.v01.events.handler.LinkEnterEventHandler#handleEvent(org.matsim.api.core.v01.events.
131         * LinkEnterEvent)
132         */
133        @Override
134        public void handleEvent(LinkEnterEvent event) {
135                if (inVehicleDistance.containsKey(event.getVehicleId())) {
136                        double distance = network.getLinks().get(event.getLinkId()).getLength();
137                        for (MutableDouble d : inVehicleDistance.get(event.getVehicleId()).values()) {
138                                d.add(distance);
139                        }
140                        this.vehicleDistances.get(event.getVehicleId())[0] += distance; // overall distance drive
141                        int occupancy = inVehicleDistance.get(event.getVehicleId()).size();
142                        this.vehicleDistances.get(event.getVehicleId())[1] += distance
143                                        * occupancy; // overall revenue distance
144                        if (occupancy > 0) {
145                                this.vehicleDistances.get(event.getVehicleId())[2] += distance; // overall occupied distance
146                                this.vehicleDistances.get(event.getVehicleId())[2+occupancy] += distance; // overall occupied distance with n passengers
147                                
148                        }
149                }
150
151        }
152
153        /*
154         * (non-Javadoc)
155         * 
156         * @see org.matsim.api.core.v01.events.handler.PersonArrivalEventHandler#handleEvent(org.matsim.api.core.v01.events.
157         * PersonArrivalEvent)
158         */
159        @Override
160        public void handleEvent(PersonArrivalEvent event) {
161                if (event.getLegMode().equals(mode)) {
162                        DynModeTrip trip = currentTrips.remove(event.getPersonId());
163                        if (trip != null) {
164                                double distance = inVehicleDistance.get(trip.getVehicle()).remove(event.getPersonId()).doubleValue();
165                                trip.setTravelDistance(distance);
166                                trip.setArrivalTime(event.getTime());
167                                trip.setToLink(event.getLinkId());
168                                Coord toCoord = this.network.getLinks().get(event.getLinkId()).getCoord();
169                                trip.setToCoord(toCoord);
170                                trip.setInVehicleTravelTime(event.getTime() - trip.getDepartureTime() - trip.getWaitTime());
171                        } else {
172                                throw new NullPointerException("Arrival without departure?");
173                        }
174                }
175
176        }
177
178        /*
179         * (non-Javadoc)
180         * 
181         * @see
182         * org.matsim.api.core.v01.events.handler.PersonDepartureEventHandler#handleEvent(org.matsim.api.core.v01.events.
183         * PersonDepartureEvent)
184         */
185        @Override
186        public void handleEvent(PersonDepartureEvent event) {
187                if (event.getLegMode().equals(mode)) {
188                        this.departureTimes.put(event.getPersonId(), event.getTime());
189                        this.departureLinks.put(event.getPersonId(), event.getLinkId());
190                }
191        }
192
193        /*
194         * (non-Javadoc)
195         * 
196         * @see
197         * org.matsim.api.core.v01.events.handler.PersonEntersVehicleEventHandler#handleEvent(org.matsim.api.core.v01.events
198         * .PersonEntersVehicleEvent)
199         */
200        @Override
201        public void handleEvent(PersonEntersVehicleEvent event) {
202                if (this.departureTimes.containsKey(event.getPersonId())) {
203                        double departureTime = this.departureTimes.remove(event.getPersonId());
204                        double waitTime = event.getTime() - departureTime;
205                        Id<Link> departureLink = this.departureLinks.remove(event.getPersonId());
206                        double unsharedDistance = this.unsharedDistances.remove(event.getPersonId());
207                        double unsharedTime = this.unsharedTimes.remove(event.getPersonId());
208                        Coord departureCoord = this.network.getLinks().get(departureLink).getCoord();
209                        DynModeTrip trip = new DynModeTrip(departureTime, event.getPersonId(), event.getVehicleId(), departureLink,
210                                        departureCoord, waitTime);
211                        trip.setUnsharedDistanceEstimate_m(unsharedDistance);
212                        trip.setUnsharedTimeEstimate_m(unsharedTime);
213                        this.drtTrips.add(trip);
214                        this.currentTrips.put(event.getPersonId(), trip);
215                        this.inVehicleDistance.get(event.getVehicleId()).put(event.getPersonId(), new MutableDouble());
216                }
217        }
218
219        /**
220         * @return the drtTrips
221         */
222        public List<DynModeTrip> getDrtTrips() {
223                return drtTrips;
224        }
225
226        /**
227         * @return the vehicleDistances
228         */
229        public Map<Id<Vehicle>, double[]> getVehicleDistances() {
230                return vehicleDistances;
231        }
232
233        /*
234         * (non-Javadoc)
235         * 
236         * @see org.matsim.contrib.drt.passenger.events.DrtRequestSubmittedEventHandler#handleEvent(org.matsim.contrib.drt.
237         * passenger.events.DrtRequestScheduledEvent)
238         */
239        @Override
240        public void handleEvent(DrtRequestSubmittedEvent event) {
241                this.unsharedDistances.put(event.getPersonId(), event.getUnsharedRideDistance());
242                this.unsharedTimes.put(event.getPersonId(), event.getUnsharedRideTime());
243        }
244}
