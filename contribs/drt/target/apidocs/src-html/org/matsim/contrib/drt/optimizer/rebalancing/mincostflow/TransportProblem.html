019package org.matsim.contrib.drt.optimizer.rebalancing.mincostflow;
020
021import java.util.ArrayList;
022import java.util.Collections;
023import java.util.List;
024import java.util.function.ToIntBiFunction;
025import java.util.stream.Stream;
026
027import org.apache.commons.lang3.tuple.Pair;
028import org.apache.commons.lang3.tuple.Triple;
029
030import graphs.flows.MinCostFlow;
031import graphs.flows.MinCostFlow.Edge;
032
033/**
034 * @author michalm
035 */
036public class TransportProblem<P, C> {
037        private final ToIntBiFunction<P, C> costFunction;
038
039        public TransportProblem(ToIntBiFunction<P, C> costFunction) {
040                this.costFunction = costFunction;
041        }
042
043        public List<Triple<P, C, Integer>> solve(List<Pair<P, Integer>> supply, List<Pair<C, Integer>> demand) {
044                final int P = supply.size();
045                final int C = demand.size();
046                final int N = P + C + 2;
047
048                // N nodes, which indices are:
049                // 0 - source
050                // 1..P - producers 1..P
051                // P+1..P+C - consumers 1..C
052                // P+C+1 - sink
053
054                @SuppressWarnings("unchecked")
055                List<Edge>[] graph = Stream.generate(ArrayList::new).limit(N).toArray(List[]::new);
056
057                // source -> producers
058                int totalSupply = 0;
059                for (int i = 0; i < P; i++) {
060                        int supplyValue = supply.get(i).getValue();
061                        MinCostFlow.addEdge(graph, 0, 1 + i, supplyValue, 0);
062                        totalSupply += supplyValue;
063                }
064
065                // producers --> consumers
066                for (int i = 0; i < P; i++) {
067                        Pair<P, Integer> producer = supply.get(i);
068                        for (int j = 0; j < C; j++) {
069                                Pair<C, Integer> consumer = demand.get(j);
070                                int capacity = Math.min(producer.getValue(), consumer.getValue());
071                                int cost = costFunction.applyAsInt(producer.getKey(), consumer.getKey());
072                                MinCostFlow.addEdge(graph, 1 + i, 1 + P + j, capacity, cost);
073                        }
074                }
075
076                // consumers -> sink
077                int totalDemand = 0;
078                for (int j = 0; j < C; j++) {
079                        int demandValue = demand.get(j).getValue();
080                        MinCostFlow.addEdge(graph, 1 + P + j, N - 1, demandValue, 0);
081                        totalDemand += demandValue;
082                }
083
084                // solve min cost flow problem
085                int[] result = MinCostFlow.minCostFlow(graph, 0, N - 1, Math.min(totalSupply, totalDemand), false);
086                if (result[0] == 0) {
087                        return Collections.emptyList();
088                }
089
090                // extract flows
091                List<Triple<P, C, Integer>> flows = new ArrayList<>();
092                for (int i = 0; i < P; i++) {
093                        P from = supply.get(i).getKey();
094                        for (Edge e : graph[1 + i]) {
095                                int flow = e.getFlow();
096                                if (flow > 0) {
097                                        int j = e.getTo() - (1 + P);
098                                        C to = demand.get(j).getKey();
099                                        flows.add(Triple.of(from, to, flow));
100                                }
101                        }
102                }
103                return flows;
104        }
105}
