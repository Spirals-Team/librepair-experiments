019package org.matsim.contrib.drt.optimizer;
020
021import java.util.ArrayList;
022import java.util.List;
023
024import javax.inject.Inject;
025
026import org.matsim.contrib.drt.optimizer.VehicleData.Entry;
027import org.matsim.contrib.drt.optimizer.VehicleData.EntryFactory;
028import org.matsim.contrib.drt.optimizer.VehicleData.Stop;
029import org.matsim.contrib.drt.run.DrtConfigGroup;
030import org.matsim.contrib.drt.schedule.DrtDriveTask;
031import org.matsim.contrib.drt.schedule.DrtStayTask;
032import org.matsim.contrib.drt.schedule.DrtStopTask;
033import org.matsim.contrib.drt.schedule.DrtTask;
034import org.matsim.contrib.drt.schedule.DrtTask.DrtTaskType;
035import org.matsim.contrib.dvrp.data.Vehicle;
036import org.matsim.contrib.dvrp.schedule.Schedule;
037import org.matsim.contrib.dvrp.schedule.Schedule.ScheduleStatus;
038import org.matsim.contrib.dvrp.tracker.OnlineDriveTaskTracker;
039import org.matsim.contrib.dvrp.util.LinkTimePair;
040
041import com.google.common.collect.ImmutableList;
042
043/**
044 * @author michalm
045 */
046public class VehicleDataEntryFactoryImpl implements EntryFactory {
047        private final double lookAhead;
048
049        @Inject
050        public VehicleDataEntryFactoryImpl(DrtConfigGroup drtCfg) {
051                lookAhead = drtCfg.getMaxWaitTime() - drtCfg.getStopDuration();
052                if (lookAhead < 0) {
053                        throw new IllegalArgumentException(
054                                        DrtConfigGroup.MAX_WAIT_TIME + " must not be smaller than " + DrtConfigGroup.STOP_DURATION);
055                }
056        }
057
058        public Entry create(Vehicle vehicle, double currentTime) {
059                if (!isEligibleForRequestInsertion(vehicle, currentTime)) {
060                        return null;
061                }
062
063                Schedule schedule = vehicle.getSchedule();
064                @SuppressWarnings("unchecked")
065                List<DrtTask> tasks = (List<DrtTask>)schedule.getTasks();
066                DrtTask currentTask = (DrtTask)schedule.getCurrentTask();
067
068                LinkTimePair start;
069                int nextTaskIdx;
070                if (schedule.getStatus() == ScheduleStatus.STARTED) {
071                        switch (currentTask.getDrtTaskType()) {
072                                case DRIVE:
073                                        DrtDriveTask driveTask = (DrtDriveTask)currentTask;
074                                        start = ((OnlineDriveTaskTracker)driveTask.getTaskTracker()).getDiversionPoint();
075                                        if (start == null) { // too late to divert a vehicle
076                                                start = new LinkTimePair(driveTask.getPath().getToLink(), driveTask.getEndTime());
077                                        }
078                                        break;
079
080                                case STOP:
081                                        DrtStopTask stopTask = (DrtStopTask)currentTask;
082                                        start = new LinkTimePair(stopTask.getLink(), stopTask.getEndTime());
083                                        break;
084
085                                case STAY:
086                                        DrtStayTask stayTask = (DrtStayTask)currentTask;
087                                        start = new LinkTimePair(stayTask.getLink(), currentTime);
088                                        break;
089
090                                default:
091                                        throw new RuntimeException();
092                        }
093
094                        nextTaskIdx = currentTask.getTaskIdx() + 1;
095                } else { // PLANNED
096                        start = new LinkTimePair(vehicle.getStartLink(), vehicle.getServiceBeginTime());
097                        nextTaskIdx = 0;
098                }
099
100                List<DrtStopTask> stopTasks = new ArrayList<>();
101                for (DrtTask task : tasks.subList(nextTaskIdx, tasks.size())) {
102                        if (task.getDrtTaskType() == DrtTaskType.STOP) {
103                                stopTasks.add((DrtStopTask)task);
104                        }
105                }
106
107                Stop[] stops = new Stop[stopTasks.size()];
108                int outputOccupancy = 0;
109                for (int i = stops.length - 1; i >= 0; i--) {
110                        Stop s = stops[i] = new Stop(stopTasks.get(i), outputOccupancy);
111                        outputOccupancy -= s.occupancyChange;
112                }
113
114                return new Entry(vehicle, start, outputOccupancy, ImmutableList.copyOf(stops));
115        }
116
117        public boolean isEligibleForRequestInsertion(Vehicle vehicle, double currentTime) {
118                return !(currentTime + lookAhead < vehicle.getServiceBeginTime()//
119                                || currentTime >= vehicle.getServiceEndTime());
120        }
121}
