019package org.matsim.contrib.drt.optimizer.insertion;
020
021import org.matsim.contrib.dvrp.path.OneToManyPathSearch.PathData;
022
023/**
024 * @author michalm
025 */
026public class InsertionWithPathData implements InsertionWithDetourTimes {
027        private final int pickupIdx;
028        private final int dropoffIdx;
029        private final PathData pathToPickup;
030        private final PathData pathFromPickup;
031        private final PathData pathToDropoff;// null if dropoff inserted directly after pickup
032        private final PathData pathFromDropoff;// null if dropoff inserted at the end
033
034        InsertionWithPathData(int pickupIdx, int dropoffIdx, PathData pathToPickup, PathData pathFromPickup,
035                        PathData pathToDropoff, PathData pathFromDropoff) {
036                this.pickupIdx = pickupIdx;
037                this.dropoffIdx = dropoffIdx;
038                this.pathToPickup = pathToPickup;
039                this.pathFromPickup = pathFromPickup;
040                this.pathToDropoff = pathToDropoff;
041                this.pathFromDropoff = pathFromDropoff;
042        }
043
044        @Override
045        public int getPickupIdx() {
046                return pickupIdx;
047        }
048
049        @Override
050        public int getDropoffIdx() {
051                return dropoffIdx;
052        }
053
054        @Override
055        public double getTimeToPickup() {
056                return pathToPickup.getTravelTime();
057        }
058
059        @Override
060        public double getTimeFromPickup() {
061                return pathFromPickup.getTravelTime();
062        }
063
064        @Override
065        public double getTimeToDropoff() {
066                return pathToDropoff.getTravelTime();// NullPointerException if dropoff inserted directly after pickup
067        }
068
069        @Override
070        public double getTimeFromDropoff() {
071                return pathFromDropoff.getTravelTime();// NullPointerException if dropoff inserted at the end
072        }
073
074        public PathData getPathToPickup() {
075                return pathToPickup;
076        }
077
078        public PathData getPathFromPickup() {
079                return pathFromPickup;
080        }
081
082        public PathData getPathToDropoff() {
083                return pathToDropoff;
084        }
085
086        public PathData getPathFromDropoff() {
087                return pathFromDropoff;
088        }
089
090        @Override
091        public String toString() {
092                return "Insertion: pickupIdx=" + pickupIdx + ", dropoffIdx=" + dropoffIdx;
093        }
094}
