072        public static Map<Double, List<DynModeTrip>> splitTripsIntoBins(Collection<DynModeTrip> trips, int startTime,
073                        int endTime, int binSize_s) {
074                LinkedList<DynModeTrip> alltrips = new LinkedList<>();
075                alltrips.addAll(trips);
076                Collections.sort(alltrips);
077                DynModeTrip currentTrip = alltrips.pollFirst();
078                if (currentTrip.getDepartureTime() > endTime) {
079                        Logger.getLogger(DynModeTripsAnalyser.class).error("wrong end / start Times for analysis");
080                }
081                Map<Double, List<DynModeTrip>> splitTrips = new TreeMap<>();
082                for (int time = startTime; time < endTime; time = time + binSize_s) {
083                        List<DynModeTrip> currentList = new ArrayList<>();
084                        splitTrips.put(Double.valueOf(time), currentList);
085                        while (currentTrip.getDepartureTime() < time + binSize_s) {
086                                currentList.add(currentTrip);
087                                currentTrip = alltrips.pollFirst();
088                                if (currentTrip == null) {
089                                        return splitTrips;
090                                }
091                        }
092
093                }
094
095                return splitTrips;
096
097        }

099        public static String summarizeTrips(List<DynModeTrip> trips, String delimiter) {
100                DescriptiveStatistics waitStats = new DescriptiveStatistics();
101                DescriptiveStatistics rideStats = new DescriptiveStatistics();
102                DescriptiveStatistics distanceStats = new DescriptiveStatistics();
103                DescriptiveStatistics directDistanceStats = new DescriptiveStatistics();
104
105                DescriptiveStatistics traveltimes = new DescriptiveStatistics();
106
107                DecimalFormat format = new DecimalFormat();
108                format.setDecimalFormatSymbols(new DecimalFormatSymbols(Locale.US));
109                format.setMinimumIntegerDigits(1);
110                format.setMaximumFractionDigits(2);
111                format.setGroupingUsed(false);
112
113                for (DynModeTrip trip : trips) {
114                        if (trip.getToLinkId() == null) {
115                                continue;
116                        }
117                        waitStats.addValue(trip.getWaitTime());
118                        rideStats.addValue(trip.getInVehicleTravelTime());
119                        distanceStats.addValue(trip.getTravelDistance());
120                        directDistanceStats.addValue(trip.getUnsharedDistanceEstimate_m());
121                        traveltimes.addValue(trip.getInVehicleTravelTime() + trip.getWaitTime());
122                }
123                String value = format.format(waitStats.getValues().length) + delimiter + format.format(waitStats.getMean())
124                                + delimiter + format.format(waitStats.getMax()) + delimiter + format.format(waitStats.getPercentile(95))
125                                + delimiter + format.format(waitStats.getPercentile(75)) + delimiter
126                                + format.format(waitStats.getPercentile(50)) + delimiter + format.format(rideStats.getMean())
127                                + delimiter + format.format(distanceStats.getMean()) + delimiter
128                                + format.format(directDistanceStats.getMean()) + delimiter + format.format(traveltimes.getMean());
129                return value;
130        }

132        public static double getDirectDistanceMean(List<DynModeTrip> trips) {
133
134                DescriptiveStatistics directDistanceStats = new DescriptiveStatistics();
135
136                for (DynModeTrip trip : trips) {
137                        if (trip.getToLinkId() == null) {
138                                continue;
139                        }
140
141                        directDistanceStats.addValue(trip.getUnsharedDistanceEstimate_m());
142                }
143                return directDistanceStats.getMean();

146        public static void analyseDetours(Network network, List<DynModeTrip> trips, DrtConfigGroup drtCfg,
147                        String fileName) {
148                if (trips == null)
149                        return;
150
151                List<String> detours = new ArrayList<String>();
152                XYSeries distances = new XYSeries("distances");
153                XYSeries travelTimes = new XYSeries("travel times");
154                XYSeries rideTimes = new XYSeries("ride times");
155
156                for (DynModeTrip trip : trips) {
157                        if (trip.getToLinkId() == null) {
158                                continue; // unfinished trip (simulation stopped before arrival)
159                        }
160
161                        double travelTime = trip.getInVehicleTravelTime() + trip.getWaitTime();
162                        distances.add(trip.getTravelDistance(), trip.getUnsharedDistanceEstimate_m());
163                        travelTimes.add(travelTime, trip.getUnsharedTimeEstimate_m());
164                        rideTimes.add(trip.getInVehicleTravelTime(), trip.getUnsharedTimeEstimate_m());
165
166                        double distanceDetour = trip.getTravelDistance() / trip.getUnsharedDistanceEstimate_m();
167                        double timeDetour = travelTime / trip.getUnsharedTimeEstimate_m();
168                        detours.add(trip.getPerson() + ";" + trip.getTravelDistance() + ";" + trip.getUnsharedDistanceEstimate_m()
169                                        + ";" + distanceDetour + ";" + travelTime + ";" + trip.getUnsharedTimeEstimate_m() + ";"
170                                        + timeDetour);
171                }
172
173                collection2Text(detours, fileName + ".csv",
174                                "person;distance;unsharedDistance;distanceDetour;time;unsharedTime;timeDetour");
175
176                final JFreeChart chart = DensityScatterPlots.createPlot("Travelled Distances", "travelled distance [m]",
177                                "unshared ride distance [m]", distances);
178                ChartSaveUtils.saveAsPNG(chart, fileName + "_distancePlot", 1500, 1500);
179
180                final JFreeChart chart2 = DensityScatterPlots.createPlot("Travel Times", "travel time [s]",
181                                "unshared ride time [s]", travelTimes,
182                                Pair.of(drtCfg.getMaxTravelTimeAlpha(), drtCfg.getMaxTravelTimeBeta()));
183                ChartSaveUtils.saveAsPNG(chart2, fileName + "_travelTimePlot", 1500, 1500);
184
185                final JFreeChart chart3 = DensityScatterPlots.createPlot("Ride Times", "ride time [s]",
186                                "unshared ride time [s]", rideTimes,
187                                Pair.of(drtCfg.getMaxTravelTimeAlpha(), drtCfg.getMaxTravelTimeBeta()));
188                ChartSaveUtils.saveAsPNG(chart3, fileName + "_rideTimePlot", 1500, 1500);
189        }

191        public static void analyseWaitTimes(String fileName, List<DynModeTrip> trips, int binsize_s) {
192                Collections.sort(trips);
193                if (trips.size() == 0)
194                        return;
195                int startTime = ((int)(trips.get(0).getDepartureTime() / binsize_s)) * binsize_s;
196                int endTime = ((int)(trips.get(trips.size() - 1).getDepartureTime() / binsize_s) + binsize_s) * binsize_s;
197                Map<Double, List<DynModeTrip>> splitTrips = splitTripsIntoBins(trips, startTime, endTime, binsize_s);
198
199                DecimalFormat format = new DecimalFormat();
200                format.setDecimalFormatSymbols(new DecimalFormatSymbols(Locale.US));
201                format.setMinimumIntegerDigits(1);
202                format.setMaximumFractionDigits(2);
203                format.setGroupingUsed(false);
204
205                SimpleDateFormat sdf2 = new SimpleDateFormat("HH:mm:ss");
206
207                BufferedWriter bw = IOUtils.getBufferedWriter(fileName + ".csv");
208                TimeSeriesCollection dataset = new TimeSeriesCollection();
209                TimeSeriesCollection datasetrequ = new TimeSeriesCollection();
210                TimeSeries averageWaitC = new TimeSeries("average");
211                TimeSeries medianWait = new TimeSeries("median");
212                TimeSeries p_5Wait = new TimeSeries("5th percentile");
213                TimeSeries p_95Wait = new TimeSeries("95th percentile");
214                TimeSeries requests = new TimeSeries("Ride requests");
215
216                try {
217                        bw.write("timebin;trips;average_wait;min;p_5;p_25;median;p_75;p_95;max");
218                        for (Entry<Double, List<DynModeTrip>> e : splitTrips.entrySet()) {
219                                long rides = 0;
220                                double averageWait = 0;
221                                double min = 0;
222                                double p_5 = 0;
223                                double p_25 = 0;
224                                double median = 0;
225                                double p_75 = 0;
226                                double p_95 = 0;
227                                double max = 0;
228                                if (!e.getValue().isEmpty()) {
229                                        DescriptiveStatistics stats = new DescriptiveStatistics();
230                                        for (DynModeTrip t : e.getValue()) {
231                                                stats.addValue(t.getWaitTime());
232                                        }
233                                        rides = stats.getN();
234                                        averageWait = stats.getMean();
235                                        min = stats.getMin();
236                                        p_5 = stats.getPercentile(5);
237                                        p_25 = stats.getPercentile(25);
238                                        median = stats.getPercentile(50);
239                                        p_75 = stats.getPercentile(75);
240                                        p_95 = stats.getPercentile(95);
241                                        max = stats.getMax();
242
243                                }
244                                Minute h = new Minute(sdf2.parse(Time.writeTime(e.getKey())));
245
246                                medianWait.addOrUpdate(h, Double.valueOf(median));
247                                averageWaitC.addOrUpdate(h, Double.valueOf(averageWait));
248                                p_5Wait.addOrUpdate(h, Double.valueOf(p_5));
249                                p_95Wait.addOrUpdate(h, Double.valueOf(p_95));
250                                requests.addOrUpdate(h, rides * 3600. / binsize_s);// normalised [req/h]
251                                bw.newLine();
252                                bw.write(Time.writeTime(e.getKey()) + ";" + rides + ";" + format.format(averageWait) + ";"
253                                                + format.format(min) + ";" + format.format(p_5) + ";" + format.format(p_25) + ";"
254                                                + format.format(median) + ";" + format.format(p_75) + ";" + format.format(p_95) + ";"
255                                                + format.format(max));
256
257                        }
258                        bw.flush();
259                        bw.close();
260                        dataset.addSeries(averageWaitC);
261                        dataset.addSeries(medianWait);
262                        dataset.addSeries(p_5Wait);
263                        dataset.addSeries(p_95Wait);
264                        datasetrequ.addSeries(requests);
265                        JFreeChart chart = chartProfile(splitTrips.size(), dataset, "Waiting times", "Wait time (s)");
266                        JFreeChart chart2 = chartProfile(splitTrips.size(), datasetrequ, "Ride requests per hour",
267                                        "Requests per hour (req/h)");
268                        ChartSaveUtils.saveAsPNG(chart, fileName, 1500, 1000);
269                        ChartSaveUtils.saveAsPNG(chart2, fileName + "_requests", 1500, 1000);
270
271                } catch (IOException | ParseException e) {
272
273                        e.printStackTrace();
274                }
275
276        }

278        private static JFreeChart chartProfile(int length, TimeSeriesCollection dataset, String descriptor, String yax) {
279                JFreeChart chart = ChartFactory.createTimeSeriesChart(descriptor, "Time", yax, dataset);
280
281                // ChartFactory.createXYLineChart("TimeProfile", "Time", "Wait Time
282                // [s]", dataset,
283                // PlotOrientation.VERTICAL, true, false, false);
284
285                XYPlot plot = chart.getXYPlot();
286                plot.setRangeGridlinesVisible(false);
287                plot.setDomainGridlinesVisible(false);
288                plot.setBackgroundPaint(Color.white);
289
290                NumberAxis yAxis = (NumberAxis)plot.getRangeAxis();
291                yAxis.setAutoRange(true);
292
293                XYItemRenderer renderer = plot.getRenderer();
294                for (int s = 0; s < length; s++) {
295                        renderer.setSeriesStroke(s, new BasicStroke(2));
296                }
297
298                return chart;
299        }

301        public static <T> void collection2Text(Collection<T> c, String filename, String header) {
302                BufferedWriter bw = IOUtils.getBufferedWriter(filename);
303                try {
304                        if (header != null) {
305                                bw.write(header);
306                                bw.newLine();
307                        }
308                        for (Iterator<T> iterator = c.iterator(); iterator.hasNext();) {
309
310                                bw.write(iterator.next().toString());
311                                bw.newLine();
312                        }
313                        bw.flush();
314                        bw.close();
315                } catch (IOException e) {
316                        // TODO Auto-generated catch block
317                        e.printStackTrace();
318                }
319        }

325        public static void writeVehicleDistances(Map<Id<Vehicle>, double[]> vehicleDistances, String iterationFilename) {
326                String header = "vehicleId;drivenDistance_m;occupiedDistance_m;emptyDistance_m;revenueDistance_pm";
327                BufferedWriter bw = IOUtils.getBufferedWriter(iterationFilename);
328                DecimalFormat format = new DecimalFormat();
329                format.setDecimalFormatSymbols(new DecimalFormatSymbols(Locale.US));
330                format.setMinimumIntegerDigits(1);
331                format.setMaximumFractionDigits(2);
332                format.setGroupingUsed(false);
333                try {
334                        bw.write(header);
335                        for (Entry<Id<Vehicle>, double[]> e : vehicleDistances.entrySet()) {
336                                double drivenDistance = e.getValue()[0];
337                                double revenueDistance = e.getValue()[1];
338                                double occDistance = e.getValue()[2];
339                                double emptyDistance = drivenDistance - occDistance;
340                                bw.newLine();
341                                bw.write(e.getKey().toString() + ";" + format.format(drivenDistance) + ";" + format.format(occDistance)
342                                                + ";" + format.format(emptyDistance) + ";" + format.format(revenueDistance));
343                        }
344                        bw.flush();
345                        bw.close();
346                } catch (IOException e) {
347                        // TODO Auto-generated catch block
348                        e.printStackTrace();
349                }
350
351        }

358        public static String summarizeVehicles(Map<Id<Vehicle>, double[]> vehicleDistances, String del) {
359                DecimalFormat format = new DecimalFormat();
360                format.setDecimalFormatSymbols(new DecimalFormatSymbols(Locale.US));
361                format.setMinimumIntegerDigits(1);
362                format.setMaximumFractionDigits(2);
363                format.setGroupingUsed(false);
364
365                DescriptiveStatistics driven = new DescriptiveStatistics();
366                DescriptiveStatistics revenue = new DescriptiveStatistics();
367                DescriptiveStatistics occupied = new DescriptiveStatistics();
368                DescriptiveStatistics empty = new DescriptiveStatistics();
369        
370
371                for (double[] dist : vehicleDistances.values()) {
372                        driven.addValue(dist[0]);
373                        revenue.addValue(dist[1]);
374                        occupied.addValue(dist[2]);
375                        double emptyD = dist[0] - dist[2];
376                        empty.addValue(emptyD);
377                }
378                double d_r_d_t = revenue.getSum() / driven.getSum();
379                // bw.write("iteration;vehicles;totalDistance;totalEmptyDistance;emptyRatio;totalRevenueDistance;averageDrivenDistance;averageEmptyDistance;averageRevenueDistance");
380                String result = vehicleDistances.size() + del + format.format(driven.getSum()) + del
381                                + format.format(empty.getSum()) + del + format.format(empty.getSum() / driven.getSum()) + del
382                                + format.format(revenue.getSum()) + del + format.format(driven.getMean()) + del
383                                + format.format(empty.getMean()) + del + format.format(revenue.getMean()) + del
384                                + format.format(d_r_d_t); 
385                return result;
386        }

411        public static String summarizeDetailedOccupancyStats(Map<Id<Vehicle>, double[]> vehicleDistances, String del, int maxcap) {
412                DecimalFormat format = new DecimalFormat();
413                format.setDecimalFormatSymbols(new DecimalFormatSymbols(Locale.US));
414                format.setMinimumIntegerDigits(1);
415                format.setMaximumFractionDigits(2);
416                format.setGroupingUsed(false);
417                
418                double[] sum = new double[maxcap+1];
419                
420                for (double[] dist : vehicleDistances.values()) {
421                        double emptyD = dist[0] - dist[2];
422                        sum[0] += emptyD;
423                        for (int i = 3;i<maxcap+3;i++) {
424                                sum[i-2]+=dist[i];
425                        }
426                }
427                String result = "";
428                for (int i = 0;i<=maxcap;i++) {
429                        result = result+";"+format.format(sum[i]);
430                }
431                
432                return result;
433        }
