023package org.matsim.contrib.drt.analysis.zonal;
024
025import java.util.Collections;
026import java.util.HashMap;
027import java.util.Map;
028
029import javax.inject.Inject;
030
031import org.apache.commons.lang3.mutable.MutableInt;
032import org.apache.log4j.Logger;
033import org.matsim.api.core.v01.events.PersonDepartureEvent;
034import org.matsim.api.core.v01.events.handler.PersonDepartureEventHandler;
035import org.matsim.contrib.dvrp.run.DvrpConfigGroup;
036import org.matsim.core.api.experimental.events.EventsManager;
037import org.matsim.core.config.Config;
038import org.matsim.core.utils.misc.Time;
039
040/**
041 * @author  jbischoff
042 *
043 */
044/**
045 *
046 */
047public class ZonalDemandAggregator implements PersonDepartureEventHandler {
048
049        private final DrtZonalSystem zonalSystem;
050        private final String mode;
051        private final int binsize = 1800; 
052        private Map<Double,Map<String,MutableInt>> departures = new HashMap<>();
053        private Map<Double,Map<String,MutableInt>> previousIterationDepartures = new HashMap<>();
054        /**
055         * 
056         */
057        @Inject
058        public ZonalDemandAggregator(EventsManager events, DrtZonalSystem zonalSystem, Config config) {
059                this.zonalSystem= zonalSystem;
060                events.addHandler(this);
061                DvrpConfigGroup dvrpconfig = DvrpConfigGroup.get(config);
062                this.mode = dvrpconfig.getMode();
063                
064        }
065        
066        @Override
067        public void reset(int iteration){
068                previousIterationDepartures.clear();
069                previousIterationDepartures.putAll(departures);
070                departures.clear();
071                prepareZones();
072
073        }
074
075        /* (non-Javadoc)
076         * @see org.matsim.api.core.v01.events.handler.PersonDepartureEventHandler#handleEvent(org.matsim.api.core.v01.events.PersonDepartureEvent)
077         */
078        @Override
079        public void handleEvent(PersonDepartureEvent event) {
080                if (event.getLegMode().equals(mode)){
081                        Double bin = getBinForTime(event.getTime());
082                        
083                        String zoneId = zonalSystem.getZoneForLinkId(event.getLinkId());
084                        if (zoneId == null) {
085                                Logger.getLogger(getClass()).error("No zone found for linkId "+ event.getLinkId().toString());
086                                return;
087                        }
088                        if (departures.containsKey(bin)){
089                        this.departures.get(bin).get(zoneId).increment();
090                        }
091                        else Logger.getLogger(getClass()).error("Time "+Time.writeTime(event.getTime())+" / bin "+bin+" is out of boundary"); 
092                        }
093                }
094        
095        
096        private void prepareZones(){
097                for (int i = 0;i<(3600/binsize)*36;i++){
098                        
099                        Map<String,MutableInt> zonesPerSlot = new HashMap<>();
100                        for (String zone : zonalSystem.getZones().keySet()){
101                                zonesPerSlot.put(zone, new MutableInt());
102                        }
103                        departures.put(Double.valueOf(i),zonesPerSlot);
104
105                }
106        }
107        
108        private Double getBinForTime(double time){
109                
110                return Math.floor(time/binsize); 
111        }
112        
113        public Map<String,MutableInt> getExpectedDemandForTimeBin(double time){
114                Double bin = getBinForTime(time);
115                return previousIterationDepartures.getOrDefault(bin, Collections.emptyMap());
116        }
117        
118}
