019package org.matsim.contrib.drt.analysis;
020
021import java.awt.Color;
022import java.awt.Shape;
023import java.awt.geom.Ellipse2D;
024
025import org.apache.commons.lang3.tuple.Pair;
026import org.jfree.chart.ChartFactory;
027import org.jfree.chart.JFreeChart;
028import org.jfree.chart.plot.XYPlot;
029import org.jfree.chart.renderer.xy.XYLineAndShapeRenderer;
030import org.jfree.data.xy.XYSeries;
031import org.jfree.data.xy.XYSeriesCollection;
032
033/**
034 * @author michalm
035 */
036public class DensityScatterPlots {
037        private static final Shape CIRCLE = new Ellipse2D.Double(-3, -3, 6, 6);
038
039        public static JFreeChart createPlot(String title, String xAxisLabel, String yAxisLabel, XYSeries series) {
040                return createPlot(title, xAxisLabel, yAxisLabel, series, null);
041        }
042
043        public static JFreeChart createPlot(String title, String xAxisLabel, String yAxisLabel, XYSeries series,
044                        Pair<Double, Double> lineCoeffs) {
045                XYSeriesCollection dataset = new XYSeriesCollection();
046                dataset.addSeries(series);
047                double maxValue = Math.max(series.getMaxX(), series.getMaxY());
048
049                // y=x
050                XYSeries lineXY = new XYSeries("y = x");
051                lineXY.add(0, 0);
052                lineXY.add(maxValue, maxValue);
053                dataset.addSeries(lineXY);
054
055                if (lineCoeffs != null) {
056                        // a*y+b=x
057                        double a = lineCoeffs.getLeft();
058                        double b = lineCoeffs.getRight();
059                        String namePrefix = a == 0 ? "" : (a + " * y + ");
060                        XYSeries lineABXY = new XYSeries(namePrefix + b + " = x");
061                        lineABXY.add(b, 0);
062                        if (a == 0) {
063                                lineABXY.add(b, maxValue);
064                        } else {
065                                lineABXY.add(maxValue, (maxValue - b) / a);
066                        }
067                        dataset.addSeries(lineABXY);
068                }
069
070                final JFreeChart chart = ChartFactory.createScatterPlot(title, xAxisLabel, yAxisLabel, dataset);
071                XYPlot xyPlot = (XYPlot)chart.getPlot();
072
073                XYLineAndShapeRenderer renderer = (XYLineAndShapeRenderer)xyPlot.getRenderer(0);
074                renderer.setSeriesPaint(0, new Color(255, 0, 0, 50));
075                renderer.setSeriesShape(0, CIRCLE);
076                renderer.setSeriesLinesVisible(0, false);
077                renderer.setSeriesShapesVisible(0, true);
078                renderer.setSeriesVisibleInLegend(0, false);
079
080                for (int i = 1; i < dataset.getSeriesCount(); i++) {
081                        renderer.setSeriesPaint(i, new Color(0, 0, 0));
082                        renderer.setSeriesLinesVisible(i, true);
083                        renderer.setSeriesShapesVisible(i, false);
084                        renderer.setSeriesVisibleInLegend(i, false);
085                }
086
087                xyPlot.getDomainAxis().setUpperBound(maxValue);
088                xyPlot.getRangeAxis().setUpperBound(maxValue);
089                xyPlot.getDomainAxis().setLowerBound(0);
090                xyPlot.getRangeAxis().setLowerBound(0);
091
092                return chart;
093        }
094}
