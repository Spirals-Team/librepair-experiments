023package org.matsim.contrib.drt.analysis.zonal;
024
025import java.util.HashMap;
026import java.util.Map;
027import java.util.Map.Entry;
028
029import javax.inject.Inject;
030
031import org.apache.log4j.Logger;
032import org.matsim.api.core.v01.Coord;
033import org.matsim.api.core.v01.Id;
034import org.matsim.api.core.v01.network.Link;
035import org.matsim.api.core.v01.network.Network;
036import org.matsim.contrib.dvrp.router.DvrpRoutingNetworkProvider;
037import org.matsim.core.utils.geometry.geotools.MGC;
038
039import com.google.inject.Provider;
040import com.google.inject.name.Named;
041import com.vividsolutions.jts.geom.Geometry;
042import com.vividsolutions.jts.geom.Point;
043
044/**
045 * @author jbischoff
046 *
047 */
048public class DrtZonalSystem {
049
050        private final Map<Id<Link>, String> link2zone = new HashMap<>();
051        private final Network network;
052        private final Map<String, Geometry> zones;
053
054        /**
055         * 
056         */
057        public DrtZonalSystem(Network network, double cellSize) {
058                this.network = network;
059                zones = DrtGridUtils.createGridFromNetwork(network, cellSize);
060
061        }
062
063        public DrtZonalSystem(Network network, Map<String, Geometry> zones) {
064                this.network = network;
065                this.zones = zones;
066
067        }
068
069        public Geometry getZone(String zone) {
070                return zones.get(zone);
071        }
072
073        public String getZoneForLinkId(Id<Link> linkId) {
074                if (this.link2zone.containsKey(linkId)) {
075                        return link2zone.get(linkId);
076                }
077
078                Point linkCoord = MGC.coord2Point(network.getLinks().get(linkId).getCoord());
079
080                for (Entry<String, Geometry> e : zones.entrySet()) {
081                        if (e.getValue().contains(linkCoord)) {
082                                link2zone.put(linkId, e.getKey());
083                                return e.getKey();
084                        }
085                }
086                link2zone.put(linkId, null);
087                return null;
088
089        }
090
091        /**
092         * @return the zones
093         */
094        public Map<String, Geometry> getZones() {
095                return zones;
096        }
097
098        public Coord getZoneCentroid(String zoneId) {
099
100                Geometry zone = zones.get(zoneId);
101                if (zone == null) {
102                        Logger.getLogger(getClass()).error("Zone " + zoneId + " not found.");
103                        return null;
104                }
105                Coord c = MGC.point2Coord(zone.getCentroid());
106                return c;
107        }
108
109        public static class DrtZonalSystemProvider implements Provider<DrtZonalSystem> {
110                @Inject
111                @Named(DvrpRoutingNetworkProvider.DVRP_ROUTING)
112                private Network network;
113
114                private final double cellSize;
115
116                public DrtZonalSystemProvider(double cellSize) {
117                        this.cellSize = cellSize;
118                }
119
120                @Override
121                public DrtZonalSystem get() {
122                        return new DrtZonalSystem(network, cellSize);
123                }
124        }
125}
