020package org.matsim.contrib.drt.optimizer.insertion;
021
022import java.util.Collection;
023import java.util.Comparator;
024import java.util.List;
025import java.util.Map;
026import java.util.Optional;
027import java.util.concurrent.ForkJoinPool;
028
029import org.apache.commons.math3.stat.descriptive.SummaryStatistics;
030import org.apache.log4j.Logger;
031import org.matsim.contrib.drt.data.DrtRequest;
032import org.matsim.contrib.drt.optimizer.VehicleData.Entry;
033import org.matsim.contrib.drt.optimizer.insertion.DetourLinksProvider.DetourLinksSet;
034import org.matsim.contrib.drt.optimizer.insertion.InsertionGenerator.Insertion;
035import org.matsim.contrib.drt.optimizer.insertion.SingleVehicleInsertionProblem.BestInsertion;
036import org.matsim.contrib.drt.run.DrtConfigGroup;
037import org.matsim.core.mobsim.framework.MobsimTimer;
038
039/**
040 * @author michalm
041 */
042public class ParallelMultiVehicleInsertionProblem implements MultiVehicleInsertionProblem {
043        private final PrecalculablePathDataProvider pathDataProvider;
044        private final DrtConfigGroup drtCfg;
045        private final MobsimTimer timer;
046        private final InsertionCostCalculator insertionCostCalculator;
047        private final ForkJoinPool forkJoinPool;
048        private final DetourLinksStats detourLinksStats = new DetourLinksStats();
049
050        public ParallelMultiVehicleInsertionProblem(PrecalculablePathDataProvider pathDataProvider, DrtConfigGroup drtCfg,
051                        MobsimTimer timer, ForkJoinPool forkJoinPool) {
052                this.pathDataProvider = pathDataProvider;
053                this.drtCfg = drtCfg;
054                this.timer = timer;
055                this.forkJoinPool = forkJoinPool;
056                insertionCostCalculator = new InsertionCostCalculator(drtCfg, timer);
057        }
058
059        @Override
060        public Optional<BestInsertion> findBestInsertion(DrtRequest drtRequest, Collection<Entry> vEntries) {
061                DetourLinksProvider detourLinksProvider = new DetourLinksProvider(drtCfg, timer, drtRequest);
062                detourLinksProvider.findInsertionsAndLinks(forkJoinPool, vEntries);
063
064                detourLinksStats.updateStats(vEntries, detourLinksProvider);
065                Map<Entry, List<Insertion>> filteredInsertions = detourLinksProvider.getFilteredInsertions();
066                if (filteredInsertions.isEmpty()) {
067                        return Optional.empty();
068                }
069
070                pathDataProvider.precalculatePathData(drtRequest, detourLinksProvider.getDetourLinksSet());
071
072                return forkJoinPool.submit(() -> filteredInsertions.entrySet().parallelStream()//
073                                .map(e -> new SingleVehicleInsertionProblem(pathDataProvider, insertionCostCalculator)
074                                                .findBestInsertion(drtRequest, e.getKey(), e.getValue()))//
075                                .filter(Optional::isPresent)//
076                                .map(Optional::get)//
077                                .min(Comparator.comparing(i -> i.cost)))//
078                                .join();
079        }
080
081        public void shutdown() {
082                forkJoinPool.shutdown();
083                detourLinksStats.printStats();
084        }
085
086        private static class DetourLinksStats {
087                private static final Logger log = Logger.getLogger(DetourLinksStats.class);
088
089                private final SummaryStatistics toPickupStats = new SummaryStatistics();
090                private final SummaryStatistics fromPickupStats = new SummaryStatistics();
091                private final SummaryStatistics toDropoffStats = new SummaryStatistics();
092                private final SummaryStatistics fromDropoffStats = new SummaryStatistics();
093                private final SummaryStatistics vEntriesStats = new SummaryStatistics();
094                private final SummaryStatistics insertionStats = new SummaryStatistics();
095                private final SummaryStatistics insertionAtEndStats = new SummaryStatistics();
096                private final SummaryStatistics insertionAtEndWhenNoStopsStats = new SummaryStatistics();
097
098                private void updateStats(Collection<Entry> vEntries, DetourLinksProvider detourLinksProvider) {
099                        addSet(detourLinksProvider.getDetourLinksSet(), vEntries.size());
100                        updateInsertionStats(vEntries, detourLinksProvider.getFilteredInsertions());
101                }
102
103                private void addSet(DetourLinksSet set, int vEntriesCount) {
104                        toPickupStats.addValue(set.pickupDetourStartLinks.size());
105                        fromPickupStats.addValue(set.pickupDetourEndLinks.size());
106                        toDropoffStats.addValue(set.dropoffDetourStartLinks.size());
107                        fromDropoffStats.addValue(set.dropoffDetourEndLinks.size());
108                        vEntriesStats.addValue(vEntriesCount);
109                }
110
111                private void updateInsertionStats(Collection<Entry> vEntries,
112                                Map<Entry, List<Insertion>> filteredInsertionsPerVehicle) {
113                        int insertionCount = 0;
114                        int insertionAtEndCount = 0;
115                        int insertionAtEndToEmptyCount = 0;
116
117                        for (java.util.Map.Entry<Entry, List<Insertion>> e : filteredInsertionsPerVehicle.entrySet()) {
118                                List<Insertion> insertions = e.getValue();
119                                insertionCount += insertions.size();
120
121                                Insertion lastInsertion = insertions.get(insertions.size() - 1);
122                                if (lastInsertion.pickupIdx == lastInsertion.dropoffIdx
123                                                && lastInsertion.pickupIdx == e.getKey().stops.size()) {
124                                        insertionAtEndCount++;
125                                        if (lastInsertion.pickupIdx == 0) {
126                                                insertionAtEndToEmptyCount++;
127                                        }
128                                }
129                        }
130
131                        insertionStats.addValue(insertionCount);
132                        insertionAtEndStats.addValue(insertionAtEndCount);
133                        insertionAtEndWhenNoStopsStats.addValue(insertionAtEndToEmptyCount);
134                }
135
136                private void printStats() {
137                        log.debug("toPickupStats:\n" + toPickupStats);
138                        log.debug("fromPickupStats:\n" + fromPickupStats);
139                        log.debug("toDropoffStats:\n" + toDropoffStats);
140                        log.debug("fromDropoffStats:\n" + fromDropoffStats);
141                        log.debug("vEntriesStats:\n" + vEntriesStats);
142                        log.debug("insertionStats:\n" + insertionStats);
143                        log.debug("insertionAtEndStats:\n" + insertionAtEndStats);
144                        log.debug("insertionAtEndWhenNoStopsStats:\n" + insertionAtEndWhenNoStopsStats);
145                }
146        }
