019package org.matsim.contrib.drt.optimizer.rebalancing.mincostflow;
020
021import java.util.ArrayList;
022import java.util.Collections;
023import java.util.HashMap;
024import java.util.List;
025import java.util.Map;
026import java.util.stream.Stream;
027
028import javax.inject.Inject;
029
030import org.apache.commons.lang3.tuple.Pair;
031import org.matsim.api.core.v01.network.Link;
032import org.matsim.contrib.drt.analysis.zonal.DrtZonalSystem;
033import org.matsim.contrib.drt.optimizer.rebalancing.RebalancingStrategy;
034import org.matsim.contrib.dvrp.data.Fleet;
035import org.matsim.contrib.dvrp.data.Vehicle;
036import org.matsim.contrib.dvrp.schedule.Schedule;
037import org.matsim.contrib.dvrp.schedule.Schedules;
038import org.matsim.contrib.dvrp.schedule.StayTask;
039import org.matsim.contrib.dvrp.schedule.Task.TaskStatus;
040
041/**
042 * @author michalm
043 */
044public class MinCostFlowRebalancingStrategy implements RebalancingStrategy {
045        public interface RebalancingTargetCalculator {
046                int estimate(String zone, double time);
047        }
048
049        private static final double MAX_REMAINING_TIME_TO_IDLENESS = 1800;// for soon-idle vehicles
050        private static final double MIN_REMAINING_SERVICE_TIME = 3600;// for idle vehicles
051
052        private final RebalancingTargetCalculator rebalancingTargetCalculator;
053        private final DrtZonalSystem zonalSystem;
054        private final Fleet fleet;
055        private final MinCostRelocationCalculator minCostRelocationCalculator;
056
057        @Inject
058        public MinCostFlowRebalancingStrategy(RebalancingTargetCalculator rebalancingTargetCalculator,
059                        DrtZonalSystem zonalSystem, Fleet fleet, MinCostRelocationCalculator minCostRelocationCalculator) {
060                this.rebalancingTargetCalculator = rebalancingTargetCalculator;
061                this.zonalSystem = zonalSystem;
062                this.fleet = fleet;
063                this.minCostRelocationCalculator = minCostRelocationCalculator;
064        }
065
066        @Override
067        public List<Relocation> calcRelocations(Stream<? extends Vehicle> rebalancableVehicles, double time) {
068                Map<String, List<Vehicle>> rebalancableVehiclesPerZone = groupRebalancableVehicles(rebalancableVehicles, time);
069                if (rebalancableVehiclesPerZone.isEmpty()) {
070                        return Collections.emptyList();
071                }
072                Map<String, List<Vehicle>> soonIdleVehiclesPerZone = groupSoonIdleVehicles(time);
073                return calculateMinCostRelocations(time, rebalancableVehiclesPerZone, soonIdleVehiclesPerZone);
074        }
075
076        private Map<String, List<Vehicle>> groupRebalancableVehicles(Stream<? extends Vehicle> rebalancableVehicles,
077                        double time) {
078                Map<String, List<Vehicle>> rebalancableVehiclesPerZone = new HashMap<>();
079                rebalancableVehicles.filter(v -> v.getServiceEndTime() > time + MIN_REMAINING_SERVICE_TIME).forEach(v -> {
080                        Link link = ((StayTask)v.getSchedule().getCurrentTask()).getLink();
081                        String zone = zonalSystem.getZoneForLinkId(link.getId());
082                        if (zone != null) {
083                                // zonePerVehicle.put(v.getId(), zone);
084                                rebalancableVehiclesPerZone.computeIfAbsent(zone, z -> new ArrayList<>()).add(v);
085                        }
086                });
087                return rebalancableVehiclesPerZone;
088        }
089
090        // also include vehicles being right now relocated or recharged
091        private Map<String, List<Vehicle>> groupSoonIdleVehicles(double time) {
092                Map<String, List<Vehicle>> soonIdleVehiclesPerZone = new HashMap<>();
093                for (Vehicle v : fleet.getVehicles().values()) {
094                        Schedule s = v.getSchedule();
095                        StayTask stayTask = (StayTask)Schedules.getLastTask(s);
096                        if (stayTask.getStatus() == TaskStatus.PLANNED
097                                        && stayTask.getBeginTime() < time + MAX_REMAINING_TIME_TO_IDLENESS
098                                        && stayTask.getBeginTime() < time + MIN_REMAINING_SERVICE_TIME) {// XXX a separate constant???
099                                String zone = zonalSystem.getZoneForLinkId(stayTask.getLink().getId());
100                                if (zone != null) {
101                                        soonIdleVehiclesPerZone.computeIfAbsent(zone, z -> new ArrayList<>()).add(v);
102                                }
103                        }
104                }
105                return soonIdleVehiclesPerZone;
106        }
107
108        private List<Relocation> calculateMinCostRelocations(double time,
109                        Map<String, List<Vehicle>> rebalancableVehiclesPerZone,
110                        Map<String, List<Vehicle>> soonIdleVehiclesPerZone) {
111                List<Pair<String, Integer>> supply = new ArrayList<>();
112                List<Pair<String, Integer>> demand = new ArrayList<>();
113
114                for (String z : zonalSystem.getZones().keySet()) {
115                        int rebalancable = rebalancableVehiclesPerZone.getOrDefault(z, Collections.emptyList()).size();
116                        int soonIdle = soonIdleVehiclesPerZone.getOrDefault(z, Collections.emptyList()).size();
117                        int target = rebalancingTargetCalculator.estimate(z, time);
118
119                        int delta = Math.min(rebalancable + soonIdle - target, rebalancable);
120                        if (delta < 0) {
121                                demand.add(Pair.of(z, -delta));
122                        } else if (delta > 0) {
123                                supply.add(Pair.of(z, delta));
124                        }
125                }
126
127                return minCostRelocationCalculator.calcRelocations(supply, demand, rebalancableVehiclesPerZone);
128        }
129}
