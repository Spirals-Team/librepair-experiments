019package org.matsim.contrib.drt.optimizer.rebalancing.mincostflow;
020
021import javax.inject.Inject;
022
023import org.apache.commons.lang3.mutable.MutableInt;
024import org.matsim.contrib.drt.analysis.zonal.ZonalDemandAggregator;
025import org.matsim.contrib.drt.optimizer.rebalancing.mincostflow.MinCostFlowRebalancingStrategy.RebalancingTargetCalculator;
026import org.matsim.contrib.drt.run.DrtConfigGroup;
027
028/**
029 * @author michalm
030 */
031public class LinearRebalancingTargetCalculator implements RebalancingTargetCalculator {
032        private final ZonalDemandAggregator demandAggregator;
033        private final MinCostFlowRebalancingParams params;
034
035        @Inject
036        public LinearRebalancingTargetCalculator(ZonalDemandAggregator demandAggregator, DrtConfigGroup drtCfg) {
037                this.demandAggregator = demandAggregator;
038                params = drtCfg.getMinCostFlowRebalancing();
039        }
040
041        // FIXME targets should be calculated more intelligently
042        @Override
043        public int estimate(String zone, double time) {
044                // XXX this "time+60" (taken from old code) means probably "in the next time bin"
045                MutableInt expectedDemand = demandAggregator.getExpectedDemandForTimeBin(time + 60).get(zone);
046                if (expectedDemand == null || expectedDemand.intValue() == 0) {
047                        return 0;// for larger zones we may assume that target is at least 1 (or in some cases) ??????
048                }
049                return (int)Math.round(params.getTargetAlpha() * expectedDemand.intValue() + params.getTargetBeta());
050        }
051}
