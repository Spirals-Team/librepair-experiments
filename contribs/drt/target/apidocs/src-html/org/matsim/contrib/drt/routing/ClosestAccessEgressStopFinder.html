042/**
043 * @author michalm
044 */
045public class ClosestAccessEgressStopFinder implements AccessEgressStopFinder {
046
047        private final Network network;
048        private final Map<Id<TransitStopFacility>, TransitStopFacility> stops;
049        private final double maxWalkDistance;
050        private final double walkBeelineFactor;
051
052        @Inject
053        public ClosestAccessEgressStopFinder(@Named(TransportMode.drt) TransitSchedule transitSchedule,
054                        DrtConfigGroup drtconfig, PlansCalcRouteConfigGroup planscCalcRouteCfg, Network network) {
055                this.network = network;
056                this.stops = transitSchedule.getFacilities();
057                this.maxWalkDistance = drtconfig.getMaxWalkDistance();
058                this.walkBeelineFactor = planscCalcRouteCfg.getModeRoutingParams().get(TransportMode.walk)
059                                .getBeelineDistanceFactor();;
060        }
061
062        @Override
063        public Pair<TransitStopFacility, TransitStopFacility> findStops(Facility<?> fromFacility, Facility<?> toFacility) {
064                TransitStopFacility accessFacility = findClosestStop(fromFacility);
065                if (accessFacility == null) {
066                        return new ImmutablePair<>(null, null);
067                }
068
069                TransitStopFacility egressFacility = findClosestStop(toFacility);
070                return new ImmutablePair<TransitStopFacility, TransitStopFacility>(accessFacility, egressFacility);
071        }
072
073        private TransitStopFacility findClosestStop(Facility<?> facility) {
074                Coord coord = StopBasedDrtRoutingModule.getFacilityCoord(facility, network);
075
076                TransitStopFacility closest = stops.values().stream()//
077                                .max(Comparator.comparing(s -> DistanceUtils.calculateSquaredDistance(coord, s.getCoord())))//
078                                .orElse(null);
079
080                if (closest != null
081                                && walkBeelineFactor * DistanceUtils.calculateDistance(coord, closest.getCoord()) <= maxWalkDistance) {
082                        return closest;
083                } else {
084                        return null;
085                }
086        }
087}
