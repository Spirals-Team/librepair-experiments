023package org.matsim.contrib.drt.analysis.zonal;
024
025import java.util.HashMap;
026import java.util.LinkedList;
027import java.util.Map;
028
029import javax.inject.Inject;
030
031import org.matsim.api.core.v01.Id;
032import org.matsim.api.core.v01.events.ActivityEndEvent;
033import org.matsim.api.core.v01.events.ActivityStartEvent;
034import org.matsim.api.core.v01.events.handler.ActivityEndEventHandler;
035import org.matsim.api.core.v01.events.handler.ActivityStartEventHandler;
036import org.matsim.contrib.drt.vrpagent.DrtActionCreator;
037import org.matsim.contrib.dvrp.data.Vehicle;
038import org.matsim.contrib.dvrp.vrpagent.VrpAgentLogic;
039import org.matsim.core.api.experimental.events.EventsManager;
040
041/**
042 * @author  jbischoff
043 *
044 */
045/**
046 *
047 */
048public class ZonalIdleVehicleCollector implements ActivityStartEventHandler, ActivityEndEventHandler {
049
050        private Map<String,LinkedList<Id<Vehicle>>> vehiclesPerZone = new HashMap<>();
051        private Map<Id<Vehicle>,String> zonePerVehicle = new HashMap<>();
052        private final DrtZonalSystem zonalSystem; 
053        
054        /**
055         * 
056         */
057        @Inject
058        public ZonalIdleVehicleCollector(EventsManager events, DrtZonalSystem zonalSystem) {
059                events.addHandler(this);
060                this.zonalSystem = zonalSystem;
061                for (String z : zonalSystem.getZones().keySet()){
062                        vehiclesPerZone.put(z,new LinkedList<Id<Vehicle>>());
063                }
064        }
065        
066        
067        @Override
068        public void handleEvent(ActivityStartEvent event) {
069                if (event.getActType().equals(DrtActionCreator.DRT_STAY_NAME)){
070                        String zone = zonalSystem.getZoneForLinkId(event.getLinkId());
071                        if (zone!=null){
072                                Id<Vehicle> vid = Id.create(event.getPersonId(), Vehicle.class);
073                                vehiclesPerZone.get(zone).add(vid);
074                                zonePerVehicle.put(vid, zone);
075                        }
076                }
077                
078                if (event.getActType().equals(VrpAgentLogic.AFTER_SCHEDULE_ACTIVITY_TYPE)){
079                        String zone = zonalSystem.getZoneForLinkId(event.getLinkId());
080                        if (zone!=null){
081                                Id<Vehicle> vid = Id.create(event.getPersonId(), Vehicle.class);
082                                zonePerVehicle.remove(vid);
083                                vehiclesPerZone.get(zone).remove(vid);
084                        }
085                }
086                
087        }
088
089
090
091        /* (non-Javadoc)
092         * @see org.matsim.api.core.v01.events.handler.ActivityEndEventHandler#handleEvent(org.matsim.api.core.v01.events.ActivityEndEvent)
093         */
094        @Override
095        public void handleEvent(ActivityEndEvent event) {
096                if (event.getActType().equals(DrtActionCreator.DRT_STAY_NAME)){
097                        String zone = zonalSystem.getZoneForLinkId(event.getLinkId());
098                        if (zone!=null){
099                                
100                                Id<Vehicle> vid = Id.create(event.getPersonId(), Vehicle.class);
101                                zonePerVehicle.remove(vid);
102                                vehiclesPerZone.get(zone).remove(vid);
103
104                        }
105                }       
106        }
107        
108        public LinkedList<Id<Vehicle>> getIdleVehiclesPerZone(String zone){
109                return this.vehiclesPerZone.get(zone);
110        }
111        
112        
113
114}
