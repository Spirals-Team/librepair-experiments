020package org.matsim.contrib.drt.optimizer.insertion;
021
022import java.util.ArrayList;
023import java.util.List;
024
025import org.matsim.contrib.drt.data.DrtRequest;
026import org.matsim.contrib.drt.optimizer.VehicleData;
027
028/**
029 * Generates all possible pickup and dropoff insertion point pairs that do not violate the vehicle capacity. In order to
030 * generate insertion points for a given vehicle, a {@code VehicleData.Entry} must be prepared. It contains the
031 * information about the vehicle current state (location, time and occupancy; additionally, the vehicle can be queried
032 * about the current task etc. ) and the sequence of planned stops (of lenght {@code N}.
033 * <p>
034 * Pickup insertion points are indexed in the following way:
035 * <ul>
036 * <li>{@code pickupIdx = 0} - pickup inserted at/after the current position (e.g. ongoing stop, stay or drive, the
037 * latter resulting in immediate diversion)</li>
038 * <li>{@code 0 < pickupIdx <= N} - pickup inserted at/after stop {@code pickupIdx - 1}</li>
039 * </ul>
040 * <p>
041 * Dropoff insertion points are indexed in the following way:
042 * <ul>
043 * <li>{@code dropoffIdx = pickupIdx} - dropoff inserted directly after pickup</li>
044 * <li>{@code pickupIdx < dropoffIdx <= N} - dropoff inserted at/after stop {@code dropoffIdx - 1}</li>
045 * </ul>
046 * <p>
047 * A pickup/dropoff is inserted at (i.e. included into) a given stop if they are on the same link. Otherwise, it is
048 * inserted after that stop. In the latter case, the sequence of stops is changed from: <br/>
049 * {@code ... --> stop i --> stop i + 1 --> ...} <br/>
050 * to: <br/>
051 * {@code ... --> stop i --> new stop --> stop i + 1 --> ...}.
052 * <p>
053 * If a pickup/dropoff is inserted at stop {@code i} (pickup/dropoff is on the same link as the stop), insertion after
054 * stop {@code i-1} will not be generated (as that would be equivalent to/duplicate of the former one).
055 * 
056 * @author michalm
057 */
058public class InsertionGenerator {
059        public static class Insertion {
060                public final int pickupIdx;
061                public final int dropoffIdx;
062
063                public Insertion(int pickupIdx, int dropoffIdx) {
064                        this.pickupIdx = pickupIdx;
065                        this.dropoffIdx = dropoffIdx;
066                }
067        }
068
069        public List<Insertion> generateInsertions(DrtRequest drtRequest, VehicleData.Entry vEntry) {
070                int stopCount = vEntry.stops.size();
071                List<Insertion> insertions = new ArrayList<>();
072                int occupancy = vEntry.startOccupancy;
073                for (int i = 0; i < stopCount; i++) {// insertions up to before last stop
074                        if (occupancy < vEntry.vehicle.getCapacity() // only not fully loaded arcs
075                                        && drtRequest.getFromLink() != vEntry.stops.get(i).task.getLink()) {// next stop at different link
076                                generateDropoffInsertions(drtRequest, vEntry, i, insertions);
077                        }
078                        // else: skip this stop and evaluate only insertion _after_ stop i
079
080                        occupancy = vEntry.stops.get(i).outgoingOccupancy;
081                }
082
083                generateDropoffInsertions(drtRequest, vEntry, stopCount, insertions);// last stop
084                return insertions;
085        }
086
087        private void generateDropoffInsertions(DrtRequest drtRequest, VehicleData.Entry vEntry, int i,
088                        List<Insertion> insertions) {
089                int stopCount = vEntry.stops.size();
090                for (int j = i; j < stopCount; j++) {// insertions up to before last stop
091                        // no need to check the capacity constraints if i == j
092                        if (j > i && // i -> pickup -> i+1 && j -> dropoff -> j+1
093                                        vEntry.stops.get(j - 1).outgoingOccupancy == vEntry.vehicle.getCapacity()) {
094                                return;// stop iterating -- cannot insert dropoff after node j
095                        }
096
097                        if (drtRequest.getToLink() != vEntry.stops.get(j).task.getLink()) {// next stop at different link
098                                insertions.add(new Insertion(i, j));
099                        }
100                        // else: skip this stop and evaluate only insertion _after_ stop i
101                }
102
103                insertions.add(new Insertion(i, stopCount));// insertion after last stop
104        }
105}
