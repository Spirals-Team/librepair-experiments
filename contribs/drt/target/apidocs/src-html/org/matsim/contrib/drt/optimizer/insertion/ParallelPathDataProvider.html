024import java.util.concurrent.ExecutionException;
025import java.util.concurrent.ExecutorService;
026import java.util.concurrent.Executors;
027import java.util.concurrent.Future;
028
029import javax.inject.Inject;
030import javax.inject.Named;
031
032import org.matsim.api.core.v01.Id;
033import org.matsim.api.core.v01.network.Link;
034import org.matsim.api.core.v01.network.Network;

036import org.matsim.contrib.drt.optimizer.DefaultDrtOptimizer;

044import org.matsim.contrib.dvrp.trafficmonitoring.DvrpTravelTimeModule;

048import org.matsim.core.router.util.TravelTime;

050import com.google.common.collect.ImmutableList;

052/**
053 * @author michalm
054 */
055public class ParallelPathDataProvider implements PrecalculatablePathDataProvider, MobsimBeforeCleanupListener {
056        public static final int MAX_THREADS = 4;
057
058        private final OneToManyPathSearch toPickupPathSearch;
059        private final OneToManyPathSearch fromPickupPathSearch;
060        private final OneToManyPathSearch toDropoffPathSearch;
061        private final OneToManyPathSearch fromDropoffPathSearch;
062
063        private final double stopDuration;
064
065        private final ExecutorService executorService;
066
067        // ==== recalculated by precalculatePathData()
068        private Map<Id<Link>, PathData> pathsToPickupMap;
069        private Map<Id<Link>, PathData> pathsFromPickupMap;
070        private Map<Id<Link>, PathData> pathsToDropoffMap;
071        private Map<Id<Link>, PathData> pathsFromDropoffMap;
072
073        @Inject
074        public ParallelPathDataProvider(@Named(DvrpRoutingNetworkProvider.DVRP_ROUTING) Network network,
075                        @Named(DvrpTravelTimeModule.DVRP_ESTIMATED) TravelTime travelTime,
076                        @Named(DefaultDrtOptimizer.DRT_OPTIMIZER) TravelDisutility travelDisutility, DrtConfigGroup drtCfg) {
077                toPickupPathSearch = OneToManyPathSearch.createBackwardSearch(network, travelTime, travelDisutility);
078                fromPickupPathSearch = OneToManyPathSearch.createForwardSearch(network, travelTime, travelDisutility);
079                toDropoffPathSearch = OneToManyPathSearch.createBackwardSearch(network, travelTime, travelDisutility);
080                fromDropoffPathSearch = OneToManyPathSearch.createForwardSearch(network, travelTime, travelDisutility);
081                stopDuration = drtCfg.getStopDuration();
082                executorService = Executors.newFixedThreadPool(Math.min(drtCfg.getNumberOfThreads(), MAX_THREADS));
083        }
084
085        @Override
086        public void precalculatePathData(DrtRequest drtRequest, Collection<VehicleData.Entry> vEntries) {
087                Map<Id<Link>, Link> startLinks = new HashMap<>();
088                Map<Id<Link>, Link> stopLinks = new HashMap<>();
089
090                for (VehicleData.Entry vEntry : vEntries) {
091                        startLinks.put(vEntry.start.link.getId(), vEntry.start.link);
092                        for (Stop s : vEntry.stops) {
093                                // TODO consider a stop filter (replacement for/addition to DrtVehicleFilter)
094                                // the filtering could distinguish between pickup/dropoff...
095                                Link l = s.task.getLink();
096                                stopLinks.put(l.getId(), l);
097                        }
098                }
099
100                Link pickup = drtRequest.getFromLink();
101                Link dropoff = drtRequest.getToLink();
102
103                double earliestPickupTime = drtRequest.getEarliestStartTime(); // optimistic
104                double minTravelTime = 15 * 60; // FIXME inaccurate temp solution: fixed 15 min
105                double earliestDropoffTime = earliestPickupTime + minTravelTime + stopDuration;
106
107                ImmutableList<Link> stopLinkList = ImmutableList.copyOf(stopLinks.values());
108
109                Future<Map<Id<Link>, PathData>> pathsToPickupFuture = executorService.submit(() -> {
110                        ImmutableList<Link> startAndStopLinkList = ImmutableList
111                                        .<Link> builderWithExpectedSize(startLinks.values().size() + stopLinkList.size())
112                                        .addAll(startLinks.values()).addAll(stopLinkList).build();
113                        // calc backward dijkstra from pickup to ends of all stop + start
114                        // TODO exclude inserting pickup after fully occupied stops
115                        return toPickupPathSearch.calcPathDataMap(pickup, startAndStopLinkList, earliestPickupTime);
116                });
117
118                Future<Map<Id<Link>, PathData>> pathsFromPickupFuture = executorService.submit(() -> {
119                        ImmutableList<Link> dropoffAndStopLinkList = ImmutableList
120                                        .<Link> builderWithExpectedSize(stopLinkList.size() + 1).add(drtRequest.getToLink())
121                                        .addAll(stopLinkList).build();
122                        // calc forward dijkstra from pickup to beginnings of all stops + dropoff
123                        // TODO exclude inserting before fully occupied stops (unless the new request's dropoff is located there)
124                        return fromPickupPathSearch.calcPathDataMap(pickup, dropoffAndStopLinkList, earliestPickupTime);
125                });
126
127                Future<Map<Id<Link>, PathData>> pathsToDropoffFuture = executorService.submit(() -> {
128                        // calc backward dijkstra from dropoff to ends of all stops
129                        // TODO exclude inserting dropoff after fully occupied stops (unless the new request's dropoff is located
130                        // there)
131                        return toDropoffPathSearch.calcPathDataMap(dropoff, stopLinkList, earliestDropoffTime);
132                });
133
134                Future<Map<Id<Link>, PathData>> pathsFromDropoffFuture = executorService.submit(() -> {
135                        // calc forward dijkstra from dropoff to beginnings of all stops
136                        // TODO exclude inserting dropoff before fully occupied stops
137                        return fromDropoffPathSearch.calcPathDataMap(dropoff, stopLinkList, earliestDropoffTime);
138                });
139
140                try {
141                        pathsToPickupMap = pathsToPickupFuture.get();
142                        pathsFromPickupMap = pathsFromPickupFuture.get();
143                        pathsToDropoffMap = pathsToDropoffFuture.get();
144                        pathsFromDropoffMap = pathsFromDropoffFuture.get();
145                } catch (InterruptedException | ExecutionException e) {
146                        throw new RuntimeException(e);
147                }
148        }
149
150        @Override
151        public PathDataSet getPathDataSet(DrtRequest drtRequest, Entry vEntry) {
152                return PrecalculatablePathDataProvider.getPathDataSet(drtRequest, vEntry, pathsToPickupMap, pathsFromPickupMap,
153                                pathsToDropoffMap, pathsFromDropoffMap);
154        }
155
156        @Override
157        public void notifyMobsimBeforeCleanup(@SuppressWarnings("rawtypes") MobsimBeforeCleanupEvent e) {
158                executorService.shutdown();
159        }
160}
