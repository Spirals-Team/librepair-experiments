066public final class DrtControlerCreator {
067
068        public static Controler createControler(Config config, boolean otfvis) {
069                adjustConfig(config);
070                Scenario scenario = ScenarioUtils.loadScenario(config);
071                Controler controler = new Controler(scenario);
072                return addDrtToControler(controler, otfvis);
073        }
074
075        public static Controler createControler(Scenario scenario, boolean otfvis) {
076                // yy I know that this one breaks the sequential loading of the building blocks, but I would like to be able
077                // to modify the scenario before I pass it to the controler. kai, oct'17
078                adjustConfig(scenario.getConfig());
079                Controler controler = new Controler(scenario);
080                return addDrtToControler(controler, otfvis);
081        }
082
083        public static Controler addDrtToControler(Controler controler, boolean otfvis) {
084                controler.addOverridingModule(new DvrpModule(DrtControlerCreator::createModuleForQSimPlugin, Arrays
085                                .asList(DrtOptimizer.class, DefaultUnplannedRequestInserter.class, ParallelPathDataProvider.class)));
086                controler.addOverridingModule(new DrtModule());
087                controler.addOverridingModule(new DrtAnalysisModule());
088                if (otfvis) {
089                        controler.addOverridingModule(new OTFVisLiveModule());
090                }
091                return controler;
092        }
093
094        public static void adjustConfig(Config config) {
095                DrtConfigGroup drtCfg = DrtConfigGroup.get(config);
096                if (drtCfg.getOperationalScheme().equals(DrtConfigGroup.OperationalScheme.stopbased)) {
097                        if (config.planCalcScore().getActivityParams(DrtStageActivityType.DRT_STAGE_ACTIVITY) == null) {
098                                ActivityParams params = new ActivityParams(DrtStageActivityType.DRT_STAGE_ACTIVITY);
099                                params.setTypicalDuration(1);
100                                params.setScoringThisActivityAtAll(false);
101                                config.planCalcScore().getScoringParametersPerSubpopulation().values()
102                                                .forEach(k -> k.addActivityParams(params));
103                                config.planCalcScore().addActivityParams(params);
104                                Logger.getLogger(DrtControlerCreator.class).info(
105                                                "drt interaction scoring parameters not set. Adding default values (activity will not be scored).");
106                        }
107                        if (!config.planCalcScore().getModes().containsKey(DrtStageActivityType.DRT_WALK)) {
108                                ModeParams drtWalk = new ModeParams(DrtStageActivityType.DRT_WALK);
109                                ModeParams walk = config.planCalcScore().getModes().get(TransportMode.walk);
110                                drtWalk.setConstant(walk.getConstant());
111                                drtWalk.setMarginalUtilityOfDistance(walk.getMarginalUtilityOfDistance());
112                                drtWalk.setMarginalUtilityOfTraveling(walk.getMarginalUtilityOfTraveling());
113                                drtWalk.setMonetaryDistanceRate(walk.getMonetaryDistanceRate());
114                                config.planCalcScore().getScoringParametersPerSubpopulation().values()
115                                                .forEach(k -> k.addModeParams(drtWalk));
116                                Logger.getLogger(DrtControlerCreator.class)
117                                                .info("drt_walk scoring parameters not set. Adding default values (same as for walk mode).");
118                        }
119                }
120
121                config.addConfigConsistencyChecker(new DrtConfigConsistencyChecker());
122                config.checkConsistency();
123        }
124
125        public static com.google.inject.Module createModuleForQSimPlugin(Config config) {
126                return new com.google.inject.AbstractModule() {
127                        @Override
128                        protected void configure() {
129                                bind(MobsimTimer.class).toProvider(MobsimTimerProvider.class).asEagerSingleton();
130                                DvrpTravelDisutilityProvider.bindTravelDisutilityForOptimizer(binder(),
131                                                DefaultDrtOptimizer.DRT_OPTIMIZER);
132
133                                bind(DrtOptimizer.class).to(DefaultDrtOptimizer.class).asEagerSingleton();
134                                bind(VrpOptimizer.class).to(DrtOptimizer.class);
135
136                                bind(DefaultUnplannedRequestInserter.class).asEagerSingleton();
137                                bind(UnplannedRequestInserter.class).to(DefaultUnplannedRequestInserter.class);
138                                bind(VehicleData.EntryFactory.class).to(VehicleDataEntryFactoryImpl.class).asEagerSingleton();
139
140                                bind(DrtTaskFactory.class).to(DrtTaskFactoryImpl.class).asEagerSingleton();
141
142                                bind(EmptyVehicleRelocator.class).asEagerSingleton();
143                                bind(DrtScheduleInquiry.class).asEagerSingleton();
144                                bind(RequestInsertionScheduler.class).asEagerSingleton();
145                                bind(DrtScheduleTimingUpdater.class).asEagerSingleton();
146
147                                bind(DynActionCreator.class).to(DrtActionCreator.class).asEagerSingleton();
148
149                                bind(PassengerRequestCreator.class).to(DrtRequestCreator.class).asEagerSingleton();
150
151                                bind(ParallelPathDataProvider.class).asEagerSingleton();
152                                bind(PrecalculatablePathDataProvider.class).to(ParallelPathDataProvider.class);
153                        }
154                };
155        }
