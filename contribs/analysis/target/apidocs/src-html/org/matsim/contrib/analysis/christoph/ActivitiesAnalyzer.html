050/**
051 * Counts the number of activities performed at a time based on events.
052 * Agents who only perform a single activity create no events. Information about
053 * them is collected using a IterationStartListener.
054 *
055 * @author cdobler
056 */
057public class ActivitiesAnalyzer implements ActivityStartEventHandler, ActivityEndEventHandler, 
058                StartupListener, BeforeMobsimListener, IterationEndsListener {
059
060        public static String defaultActivitiesFileName = "activityCounts";
061        
062        private final boolean autoConfig;
063        
064        private double endTime = 30*3600;
065        
066        private String activitiesFileName = defaultActivitiesFileName;
067        private final Set<Id> observedAgents;
068        private boolean createGraphs;
069        private final Map<String, LinkedList<ActivityData>> activityCountData = new TreeMap<String, LinkedList<ActivityData>>();
070        private final LinkedList<ActivityData> overallCount = new LinkedList<ActivityData>();
071        
072        /**
073         * This is how most people will probably will use this class.
074         * It has to be created an registered as ControlerListener.
075         * Then, it auto-configures itself (register as events handler,
076         * get paths to output files, ...).
077         */
078        public ActivitiesAnalyzer() {
079                
080                this.autoConfig = true;
081                this.createGraphs = true;
082                this.observedAgents = null;
083                
084                reset(0);
085        }
086        
087        public ActivitiesAnalyzer(String activitiesFileName, Set<String> activityTypes, boolean createGraphs) {
088                this(activitiesFileName, activityTypes, null, createGraphs);
089        }
090        
091        public ActivitiesAnalyzer(String activitiesFileName, Set<String> activityTypes, Set<Id> observedAgents, boolean createGraphs) {
092                
093                this.autoConfig = false;
094                
095                this.activitiesFileName = activitiesFileName;
096                this.createGraphs = createGraphs;
097                
098                // use all activity defined in the set
099                for (String activityType : activityTypes) {
100                        this.activityCountData.put(activityType, new LinkedList<ActivityData>());
101                }
102                
103                if (observedAgents != null) {
104                        // make a copy to prevent people changing the set over the iterations
105                        this.observedAgents = new HashSet<Id>(observedAgents);                  
106                } else this.observedAgents = null;
107                
108                reset(0);
109        }
110        
111        public void setCreateGraphs(boolean createGraphs) {
112                this.createGraphs = createGraphs;
113        }
114        
115        public void setEndTime(double endTime) {
116                this.endTime = endTime;
117        }
118        
119        @Override
120        public void handleEvent(ActivityEndEvent event) {
121                
122                if (observedAgents != null && !observedAgents.contains(event.getPersonId())) return;
123                
124                LinkedList<ActivityData> list = this.activityCountData.get(event.getActType());
125                
126                // ignore not observed activity types
127                if (list == null) return;
128                
129                changeCount(event.getTime(), list, -1);
130                changeCount(event.getTime(), this.overallCount, -1);
131        }
132
133
134        @Override
135        public void handleEvent(ActivityStartEvent event) {
136                
137                if (observedAgents != null && !observedAgents.contains(event.getPersonId())) return;
138                
139                LinkedList<ActivityData> list = this.activityCountData.get(event.getActType());
140
141                // ignore not observed activity types
142                if (list == null) return;
143                
144                changeCount(event.getTime(), list, 1);
145                changeCount(event.getTime(), this.overallCount, 1);
146        }
147        
148        private void changeCount(double time, LinkedList<ActivityData> list, int delta) {
149                
150                ActivityData activityData = list.getLast();
151                
152                /*
153                 * If there is already another entry for the same time step, re-use it.
154                 * Otherwise create a new one.
155                 */
156                if (time == activityData.time) {
157                        activityData.activityCount += delta;
158                } else {
159                        list.add(new ActivityData(time, activityData.activityCount + delta));
160                }
161        }
162        
163        @Override
164        public void reset(final int iter) {
165                for (List<ActivityData> list : this.activityCountData.values()) {
166                        list.clear();
167                        list.add(new ActivityData(0.0, 0));
168                }
169                this.overallCount.clear();
170                this.overallCount.add(new ActivityData(0.0, 0));
171        }
172        
173
174        @Override
175        public void notifyStartup(StartupEvent event) {
176                
177                MatsimServices controler = event.getServices();
178                
179                if (autoConfig) {
180                        // use all activity types defined in the config
181                        Set<String> activityTypes = new TreeSet<String>(event.getServices().getConfig().planCalcScore().getActivityTypes());
182                        for (String activityType : activityTypes) {
183                                this.activityCountData.put(activityType, new LinkedList<ActivityData>());
184                        }
185                        
186                        controler.getEvents().addHandler(this);
187                }
188        }
189        
190        @Override
191        public void notifyBeforeMobsim(BeforeMobsimEvent event) {
192                
193                ActivityData overallActivityData = this.overallCount.getLast();
194
195        for (Person person : event.getServices().getScenario().getPopulation().getPersons().values()) {
196                        
197                        if (this.observedAgents != null && !this.observedAgents.contains(person.getId())) continue;
198                        
199                        Plan plan = person.getSelectedPlan();
200                        Activity firstActivity = (Activity) plan.getPlanElements().get(0);
201                        LinkedList<ActivityData> list = activityCountData.get(firstActivity.getType());
202                        
203                        // ignore not observed activity types
204                        if (list == null) continue;
205                        
206                        ActivityData activityData = list.getLast();
207                        activityData.activityCount += 1;
208                        overallActivityData.activityCount += 1;
209                }
210        }
211
212        @Override
213        public void notifyIterationEnds(IterationEndsEvent event) {
214                
215                OutputDirectoryHierarchy outputDirectoryHierarchy = event.getServices().getControlerIO();
216                
217                try {
218                        for (String activityType : this.activityCountData.keySet()) {
219                                String fileName = outputDirectoryHierarchy.getIterationFilename(event.getIteration(), this.activitiesFileName + "_" + activityType + ".txt");
220                                BufferedWriter activitiesWriter = IOUtils.getBufferedWriter(fileName);
221                                
222                                activitiesWriter.write("TIME");
223                                activitiesWriter.write("\t");
224                                activitiesWriter.write(activityType.toUpperCase());
225                                activitiesWriter.write("\n");
226                                
227                                List<ActivityData> list = this.activityCountData.get(activityType);
228                                for (ActivityData activityData : list) {
229                                        activitiesWriter.write(String.valueOf(activityData.time));
230                                        activitiesWriter.write("\t");
231                                        activitiesWriter.write(String.valueOf(activityData.activityCount));
232                                        activitiesWriter.write("\n");
233                                }
234                                
235                                activitiesWriter.flush();
236                                activitiesWriter.close();
237                        }
238                } catch (IOException e) {
239                        throw new UncheckedIOException(e);
240                }
241                
242                if (this.createGraphs) {
243                        // create chart when data of more than one iteration is available.
244                        XYLineChart chart;
245                                                
246                        /*
247                         * number of performed activities
248                         */
249                        chart = new XYLineChart("Number of performed Activities", "time", "# activities");
250                        for (String activityType : this.activityCountData.keySet()) {
251                                List<ActivityData> list = this.activityCountData.get(activityType);
252                                int length = list.size();
253                                
254                                double[] times = new double[length * 2 - 1];
255                                double[] counts = new double[length * 2 - 1];
256                                Iterator<ActivityData> iter = list.iterator();
257                                int i = 0;
258                                double lastValue = 0.0;
259                                while (iter.hasNext()) {
260                                        ActivityData activityData = iter.next();
261                                        times[i] = Math.min(activityData.time, this.endTime) / 3600.0;
262                                        counts[i] = activityData.activityCount;
263                                        if (i > 0) {
264                                                times[i - 1] = Math.min(activityData.time, this.endTime) / 3600.0;
265                                                counts[i - 1] = lastValue;
266                                        }
267                                        lastValue = activityData.activityCount;
268                                        i += 2;
269                                }
270                                chart.addSeries(activityType, times, counts);
271                        }
272                        int length = this.overallCount.size();
273                        double[] times = new double[length * 2 - 1];
274                        double[] counts = new double[length * 2 - 1];
275                        Iterator<ActivityData> iter = this.overallCount.iterator();
276                        int i = 0;
277                        double lastValue = 0.0;
278                        while (iter.hasNext()) {
279                                ActivityData activityData = iter.next();
280                                times[i] = Math.min(activityData.time, this.endTime) / 3600.0;
281                                counts[i] = activityData.activityCount;
282                                if (i > 0) {
283                                        times[i - 1] = Math.min(activityData.time, this.endTime) / 3600.0;
284                                        counts[i - 1] = lastValue;
285                                }
286                                lastValue = activityData.activityCount;
287                                i += 2;
288                        }
289                        chart.addSeries("overall", times, counts);
290                        
291                    NumberAxis domainAxis = (NumberAxis) chart.getChart().getXYPlot().getDomainAxis();
292                    domainAxis.setTickLabelFont(new Font("SansSerif", Font.PLAIN, 11));
293                    domainAxis.setStandardTickUnits(NumberAxis.createIntegerTickUnits());
294                    domainAxis.setAutoRange(false);
295                    domainAxis.setRange(0, endTime / 3600.0);
296                        
297                        chart.addMatsimLogo();
298                        String fileName = outputDirectoryHierarchy.getIterationFilename(event.getIteration(), this.activitiesFileName + ".png");
299                        chart.saveAsPng(fileName, 800, 600);
300                }
301        }
302
303        private static class ActivityData {
304                
305                public final double time;
306                public int activityCount;
307
308                public ActivityData(double time, int activityCount) {
309                        this.time = time;
310                        this.activityCount = activityCount;
311                }
312        }
