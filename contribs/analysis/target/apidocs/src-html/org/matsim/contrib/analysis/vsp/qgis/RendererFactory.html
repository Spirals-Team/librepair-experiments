001package org.matsim.contrib.analysis.vsp.qgis;
002
003import org.matsim.contrib.analysis.vsp.qgis.utils.ColorRangeUtils;
004
005/**
006 * @author gthunig on 22.05.2017.
007 */
008public class RendererFactory {
009
010    public static GraduatedSymbolRenderer createNoiseRenderer(VectorLayer layer, double receiverPointGap) {
011        return new GraduatedSymbolRenderer(layer, 45d, 75d, 8, (int) receiverPointGap);
012    }
013
014    public static GraduatedSymbolRenderer createDensitiesRenderer(VectorLayer layer, int populationThreshold, int symbolSize) {
015        return new GraduatedSymbolRenderer(layer, (double)populationThreshold, (double)populationThreshold, 1,
016                symbolSize, ColorRangeUtils.ColorRange.DENSITY);
017    }
018}
