001package org.matsim.contrib.analysis.vsp.qgis.utils;
002
003import org.apache.log4j.Logger;
004
005import java.awt.*;
006
007/**
008 * @author gthunig on 22.05.2017.
009 */
010public class ColorRangeUtils {
011    public static final Logger log = Logger.getLogger(ColorRangeUtils.class);
012
013    public enum ColorRange {
014        DEFAULT_RED_TO_BLUE, RED_TO_GREEN, GREEN_TO_RED, GREEN_TO_BLUE, BLUE_TO_GREEN, BLUE_TO_RED, RED_TO_BLUE,
015        BLACK_TO_WHITE, WHITE_TO_BLACK, DENSITY
016    }
017
018    public static Color getColor(ColorRange colorRange, double power) {
019        switch (colorRange) {
020            case DEFAULT_RED_TO_BLUE:
021                return getDefaultColorFromRedToBlue(power);
022
023            case RED_TO_GREEN:
024                return getColorFromRedToGreen(power);
025
026            case GREEN_TO_BLUE:
027                return getColorFromGreenToBlue(power);
028
029            case BLUE_TO_RED:
030                return getColorFromBlueToRed(power);
031
032            case GREEN_TO_RED:
033                return getColorFromRedToGreen(invert(power));
034
035            case BLUE_TO_GREEN:
036                return getColorFromGreenToBlue(invert(power));
037
038            case RED_TO_BLUE:
039                return getColorFromBlueToRed(invert(power));
040
041            case BLACK_TO_WHITE:
042                return getColorFromBlackToWhite(power);
043
044            case WHITE_TO_BLACK:
045                return getColorFromBlackToWhite(invert(power));
046
047            case DENSITY:
048                return getColorInDensityRange();
049
050            default:
051                log.error("No such color range");
052                return new Color(0,0,0);
053        }
054    }
055
056
057
058    private static Color getDefaultColorFromRedToBlue(double power) {
059        DefaultColorRange colorRange = new DefaultColorRange();
060        return colorRange.getColor(power);
061    }
062
063    private static Color getColorFromRedToGreen(double power) {
064        double H = power * 0.35; // Hue
065        double S = 0.9; // Saturation
066        double B = 0.9; // Brightness
067
068        return Color.getHSBColor((float)H, (float)S, (float)B);
069    }
070
071    private static Color getColorFromGreenToBlue(double power) {
072        double H = 0.35 + power * 0.33; // Hue
073        double S = 0.9; // Saturation
074        double B = 0.9; // Brightness
075
076        return Color.getHSBColor((float)H, (float)S, (float)B);
077    }
078
079    private static Color getColorFromBlueToRed(double power) {
080        double H = 0.68 + power * 0.32; // Hue
081        double S = 0.9; // Saturation
082        double B = 0.9; // Brightness
083
084        return Color.getHSBColor((float)H, (float)S, (float)B);
085    }
086
087    private static Color getColorFromBlackToWhite(double power) {
088        int rgbValue = (int)(255 * power);
089        return new Color(rgbValue);
090    }
091
092    private static Color getColorInDensityRange() {
093        return new Color(255, 255, 255, 255);
094    }
095
096    private static double invert(double power) {
097        return 1-power;
098    }
099}
