001package org.matsim.contrib.analysis.vsp.qgis;
002
003
004/**
005 * Renderer for QGis raster layers.
006 * So far, no project or use case specific settings exist, so this is the only class that renders raster data.
007 * 
008 * @author dhosse
009 *
010 */
011public class QGisRasterRenderer extends QGisRenderer {
012        
013        private double opacity;
014        
015        private int alphaBand;
016        private int blueBand;
017        private int greenBand;
018        private int redBand;
019        
020        private QGisConstants.rasterRendererType type;
021        
022        public QGisRasterRenderer(double opacity, int alpha, int blue, int green, QGisConstants.rasterRendererType type, int red, QGisLayer layer){
023                
024                super(QGisConstants.renderingType.singleSymbol, layer);
025                
026                this.opacity = opacity;
027                this.alphaBand = alpha;
028                this.blueBand = blue;
029                this.greenBand = green;
030                this.type = type;
031                this.redBand = red;
032                
033        }
034        
035        @Override
036        public void init() {
037                
038        }
039        
040        public double getOpacity() {
041                return opacity;
042        }
043        public void setOpacity(double opacity) {
044                this.opacity = opacity;
045        }
046        public int getAlphaBand() {
047                return alphaBand;
048        }
049        public void setAlphaBand(int alphaBand) {
050                this.alphaBand = alphaBand;
051        }
052        public int getBlueBand() {
053                return blueBand;
054        }
055        public void setBlueBand(int blueBand) {
056                this.blueBand = blueBand;
057        }
058        public int getGreenBand() {
059                return greenBand;
060        }
061        public void setGreenBand(int greenBand) {
062                this.greenBand = greenBand;
063        }
064        public int getRedBand() {
065                return redBand;
066        }
067        public void setRedBand(int redBand) {
068                this.redBand = redBand;
069        }
070        public QGisConstants.rasterRendererType getType() {
071                return type;
072        }
073        public void setType(QGisConstants.rasterRendererType type) {
074                this.type = type;
075        }
076
077}
