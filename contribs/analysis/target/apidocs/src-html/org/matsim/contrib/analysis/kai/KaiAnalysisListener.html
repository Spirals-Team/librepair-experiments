021package org.matsim.contrib.analysis.kai;
022
023import org.matsim.core.controler.events.IterationEndsEvent;
024import org.matsim.core.controler.events.StartupEvent;
025import org.matsim.core.controler.listener.IterationEndsListener;
026import org.matsim.core.controler.listener.StartupListener;
027
028import javax.inject.Inject;
029
030/**
031 * 
032 * 
033 * @author nagel
034 *
035 */
036public class KaiAnalysisListener implements StartupListener, IterationEndsListener {
037        // NOTE: My excel opens tab-separated txt files directly (from the command line).  It does not do this with comma-separated or semicolon-separated.
038        // So tab-separated is the way to go. kai, sep'13
039        
040        KNAnalysisEventsHandler calcLegTimes = null ;
041        
042        // the default constructor can be injected even without annotation!  kai, may'18
043        
044        @Override
045        public void notifyStartup(StartupEvent event) {
046                
047                this.calcLegTimes = new KNAnalysisEventsHandler( event.getServices().getScenario() ) ;
048                event.getServices().getEvents().addHandler( this.calcLegTimes ) ;
049
050        }
051
052        @Override
053        public void notifyIterationEnds(IterationEndsEvent event) {
054                // moved this to iteration end since I also want to analyze population scores. kai, mar'14
055
056                int iteration = event.getIteration() ;
057
058                this.calcLegTimes.writeStats(event.getServices().getControlerIO().getIterationFilename(iteration, "stats_"));
059
060                // trips are from "true" activity to "true" activity.  legs may also go
061                // from/to ptInteraction activity.  This, in my opinion "legs" is the correct (matsim) term
062                // kai, jul'11
063
064        }
065
066}
