001package org.matsim.contrib.analysis.vsp.qgis;
002
003import java.io.IOException;
004import java.util.ArrayList;
005import java.util.List;
006
007import org.apache.log4j.Logger;
008import org.matsim.core.api.internal.MatsimWriter;
009import org.matsim.core.controler.OutputDirectoryLogging;
010import org.matsim.core.utils.io.AbstractMatsimWriter;
011
012import com.vividsolutions.jts.geom.Envelope;
013/**
014 * Writer that creates a QuantumGIs project file (*.qgs). 
015 *
016 * @author dhosse
017 *
018 */
019
020public class QGisWriter extends AbstractMatsimWriter implements MatsimWriter {
021
022        private static final Logger log = Logger.getLogger( QGisWriter.class );
023        
024        private QGisFileWriter handler;
025        
026        private List<QGisLayer> layers = new ArrayList<>();
027        
028        protected SRS srs = null;
029        
030        private QGisConstants.units unit;
031
032        private String workingDirectory;
033        private String title = "";
034        private String projectname = "";
035        
036        private Envelope envelope;
037        
038        /**
039         * Creates a new instance of a QGis project file (*.qgs) writer.
040         * Coordinate reference system and working directory are mandatory for writing the file.
041         * Layers have to be added separately with the method {@code addLayer}. If no layers were
042         * added, the writer creates an empty project file.
043         * </p>
044         * After calling the constructor you have to set the envelope (starting view) manually.
045         * 
046         * @param crs coordinate reference system of the network / spatial data
047         * @param workingDir the directory in which all generated files (shapefiles, qgs file) are put.
048         * 
049         */
050        public QGisWriter(String crs, String workingDir){
051
052                setCrs(crs);
053                this.workingDirectory = workingDir;
054                this.handler = new QGisFileWriter(this);
055                this.setUnit(QGisConstants.units.meters);
056                
057        }
058        
059        /**
060         * 
061         * @param srs String representation of the spatial reference system that you want to assign.
062         */
063        private void setCrs(String srs){
064                this.srs = SRS.createSpatialRefSys(srs);
065        }
066        
067        /**
068         * Adds a new layer to the layers list. 
069         * 
070         * @param layer the layer you want to add
071         */
072        public void addLayer(QGisLayer layer){
073                
074                this.layers.add(layer);
075                
076        }
077        
078        /**
079         * Adds a new layer to the layers list at the specified position.
080         * 
081         * @param position the index you want to add the layer at
082         * @param layer the layer you want to add
083         */
084        public void addLayer(int position, QGisLayer layer){
085                
086                this.layers.add(position, layer);
087                
088        }
089        
090        @Override
091        public void write(String filename) {
092                
093                OutputDirectoryLogging.catchLogEntries();
094                try {
095                        OutputDirectoryLogging.initLoggingWithOutputDirectory(this.workingDirectory);
096                } catch (IOException e1) {
097                        e1.printStackTrace();
098                }
099                
100                log.info("Writing QuantumGIS project file (*.qgs) to " + this.workingDirectory + filename + "...");
101                log.info("Make sure all your input files exist. This writer will do absolutely no file writing except for the project file.");
102                
103                this.printQGisProjectSettings();
104                
105                this.openFile(this.workingDirectory + filename);
106                
107                try {
108                
109                        // write file header (contains document type, qgis version etc)
110                        this.handler.writeHeaderAndStartElement(this.writer);
111                        
112                        // writes the project title (if specified, else empty field)
113                        this.handler.writeTitle(this.writer);
114                        
115                        // writes the layer tree (all layers of the project in order of painting)
116                        this.handler.writeLayerTreeGroup(this.writer);
117                        
118                        // writes basic map information, such as envelope of view and spatial reference system
119                        this.handler.writeMapCanvas(this.writer);
120                        
121                        // TODO: what does this do?
122                        this.handler.writeLayerTreeCanvas(this.writer);
123                        
124                        if(this.layers.size() > 0){
125                                
126                                // writes the map layers inside the project file
127                                // order has nothing to do with drawing order
128                                // contains specifications of layer (id, datasource, srs, renderer etc)
129                                this.handler.writeProjectLayers(this.writer);
130                                
131                        }
132                        
133                        // writes project properties like gui settings and the type of paths that's used
134                        this.handler.writeProperties(this.writer);
135                        
136                        this.handler.endFile(this.writer);
137                        this.writer.flush();
138                        this.writer.close();
139                        
140                } catch (IOException e) {
141                        
142                        e.printStackTrace();
143                        
144                }
145                
146                log.info("...Done.");
147                
148                OutputDirectoryLogging.closeOutputDirLogging();
149                
150        }
151
152        private void printQGisProjectSettings() {
153                
154                log.info("QGis project uses the following configuration:");
155                
156                log.info("PROJECT VERSION:\t\t\t" + QGisConstants.currentVersion);
157                log.info("EXTENT (MINX, MINY, MAXX, MAXY):\t" + this.envelope.getMinX() + ", " +
158                                                                                                                this.envelope.getMinY() + ", " +
159                                                                                                                this.envelope.getMaxX() + ", " +
160                                                                                                                this.envelope.getMaxY());
161                log.info("SPATIAL REFERENCE SYSTEM:\t\t" + this.srs.getDescription());
162                log.info("NUMBER OF LAYERS:\t\t\t" + this.layers.size());
163                log.info("LAYERS (IN DRAWING ORDER):");
164                
165                log.info("############################################################");
166                
167                String srs = null;
168                
169                for(QGisLayer layer : this.layers){
170                        
171                        srs = layer.getSrs() != null ? layer.getSrs().getDescription() : "[project]";
172                        
173                        log.info("name:\t\t\t\t" + layer.getName());
174                        log.info("input type:\t\t\t" + layer.getInputType().toString());
175                        log.info("layer type:\t\t\t" + layer.getType().toString());
176                        
177                        if(layer.getLayerClass() != null){
178                                
179                                log.info("layer class:\t\t\t" + layer.getLayerClass().toString());
180                                
181                        }
182                        
183                        if(layer.getRenderer() != null){
184                                
185                                log.info("rendering type:\t\t\t" + layer.getRenderer().getRenderingType().toString());
186                                log.info("renderer:\t\t\t\t" + layer.getRenderer().getClass().getSimpleName());
187                                
188                        }
189                        
190                        log.info("spatial reference system:\t\t" + srs);
191                        
192                        log.info("############################################################");
193                        
194                }
195                
196        }
197

199                this.workingDirectory = workingDir;
200        }

203                return this.layers;
204        }
205

207                return this.envelope;
208        }
209        
210        public String getWorkingDir(){
211                return this.workingDirectory;
212        }
213

215         * Sets the starting view on the map when opening the project file.
216         * 
217         * @param envelope with minx, miny, maxx and maxy coordinates of the starting view
218         */
219        public void setEnvelope(Envelope envelope){
220                this.envelope = envelope;
221        }
222        
223        /**
224         * 
225         * @return The spatial reference system that was created for the project file
226         */
227        public SRS getSRS(){
228                return this.srs;
229        }
230

232                return unit;
233        }
234

236                this.unit = unit;
237        }
238

240                return title;
241        }
242
243        public void setTitle(String title) {
244                this.title = title;
245        }
246
247        public String getProjectname() {
248                return projectname;
249        }
250
251        public void setProjectname(String projectname) {
252                this.projectname = projectname;
253        }
