003import java.io.BufferedWriter;

007/**

014public class QGisFileWriter {
015        
016        private final QGisWriter writer;
017        
018        public QGisFileWriter(QGisWriter writer){
019                this.writer = writer;
020        }
021        
022        public void writeHeaderAndStartElement(BufferedWriter out) throws IOException{
023                
024                out.write("<!DOCTYPE qgis PUBLIC 'http://mrcc.com/qgis.dtd' 'SYSTEM'>\n");
025                out.write("<qgis projectname=\"" + this.writer.getProjectname() + "\" version=\"" + QGisConstants.currentVersion + "\">\n");
026                
027        }
028        
029        public void writeTitle(BufferedWriter out) throws IOException{
030                
031                out.write("\t<title>" + this.writer.getTitle() + "</title>\n");
032                
033        }
034        
035        public void writeLayerTreeGroup(BufferedWriter out) throws IOException{
036                
037                out.write("\t<layer-tree-group expanded=\"1\" checked=\"Qt::Checked\" name=\"\">\n");
038                
039                out.write("\t\t<customproperties/>\n");
040                
041                for(QGisLayer layer : this.writer.getLayers()){
042                        
043                        writeLayerTreeLayer(out, layer);
044                        
045                }
046                
047                out.write("\t</layer-tree-group>\n");
048                
049        }
050        
051        private void writeLayerTreeLayer(BufferedWriter out, QGisLayer layer) throws IOException{
052                
053                out.write("\t\t<layer-tree-layer expanded=\"1\" checked=\"Qt::Checked\" id=\"" + layer.getId().toString() + "\" name=\"" + layer.getName() + "\">\n");
054                out.write("\t\t\t<customproperties/>\n");
055                out.write("\t\t</layer-tree-layer>\n");
056                
057        }
058        
059        public void writeMapCanvas(BufferedWriter out) throws IOException{
060                
061                out.write("\t<mapcanvas>\n");
062                
063                out.write("\t\t<units>" + this.writer.getUnit().toString() + "</units>\n");
064                out.write("\t\t<extent>\n");
065                
066                out.write("\t\t\t<xmin>" + this.writer.getEnvelope().getMinX() + "</xmin>\n");
067                out.write("\t\t\t<ymin>" + this.writer.getEnvelope().getMinY() + "</ymin>\n");
068                out.write("\t\t\t<xmax>" + this.writer.getEnvelope().getMaxX() + "</xmax>\n");
069                out.write("\t\t\t<ymax>" + this.writer.getEnvelope().getMaxY() + "</ymax>\n");
070                
071                out.write("\t\t</extent>\n");
072                out.write("\t\t<projections>1</projections>\n");
073                this.writeDestinationSrs(out);
074                out.write("\t\t<layer_coordinate_transform_info/>\n");
075                
076                out.write("\t</mapcanvas>\n");
077                
078        }

139        private void writeMapLayer(BufferedWriter out, QGisLayer layer) throws IOException{
140                
141                if(layer.getType().equals(QGisConstants.layerType.vector)){
142                        
143                        writeVectorLayer(out, layer);
144                        
145                } else if(layer.getType().equals(QGisConstants.layerType.raster)){
146                        
147                        writeRasterLayer(out, layer);
148                        
149                } else {
150                        throw new RuntimeException("Cannot write map layer. Unknown format.");
151                }
152                
153                out.write("\t\t</maplayer>\n");
154                
155        }

157        private void writeVectorLayer(BufferedWriter out, QGisLayer layer) throws IOException {
158                
159                VectorLayer vlayer = (VectorLayer) layer;
160                
161                if(vlayer.getGeometryType().equals(QGisConstants.geometryType.No_geometry)){
162                        
163                        out.write("\t\t<maplayer minimumScale=\"0\" maximumScale=\"1e+08\" geometry=\"" + vlayer.getGeometryType().toString().replace("_", " ") + "\" type=\"" + vlayer.getType() + "\" hasScaleBasedVisibilityFlag=\"0\">\n");
164                        
165                } else{
166                        
167                        out.write("\t\t<maplayer minimumScale=\"0\" maximumScale=\"1e+08\" simplifyDrawingHints=\"1\" minLabelScale=\"0\" maxLabelScale=\"1e+08\" simplifyDrawingTol=\"1\" geometry=\"" + vlayer.getGeometryType().toString() + "\" simplifyMaxScale=\"1\" type=\"vector\" hasScaleBasedVisibilityFlag=\"0\" simplifyLocal=\"1\" scaleBasedLabelVisibilityFlag=\"0\">\n");
168                
169                }
170
171                out.write("\t\t\t<id>" + vlayer.getId().toString() + "</id>\n");
172                
173                String relP = Paths.get(this.writer.getWorkingDir()).relativize(Paths.get(layer.getPath())).toString();
174                
175                if(isCsvDataLayer(vlayer)){
176                        
177                        String datasource = relP + "?type=csv&amp;useHeader=" + vlayer.headerUsed() + "&amp;";
178                        
179                        if(vlayer.getDelimiter() != null){
180                                
181                                datasource += "delimiter=" + vlayer.getDelimiter() + "&amp;";
182                        }
183                        
184                        
185                        datasource += "quote='&amp;escape='&amp;skipEmptyField=Yes&amp;xField=" + vlayer.getXField() +
186                                        "&amp;yField=" + vlayer.getYField() + "&amp;spatialIndex=no&amp;subsetIndex=no&amp;watchFile=no\n";
187                        
188                        out.write("\t\t\t<datasource>" + datasource + "</datasource>\n");
189                        
190                } else{
191                        
192                        relP.replace("file:/", "");
193                        
194                        out.write("\t\t\t<datasource>" + relP + "</datasource>\n");
195                        
196                }
197                
198                out.write("\t\t\t<title></title>\n");
199                out.write("\t\t\t<abstract></abstract>\n");
200                out.write("\t\t\t<keywordList>\n");
201                
202                out.write("\t\t\t\t<value></value>\n");
203                
204                out.write("\t\t\t</keywordList>\n");
205                out.write("\t\t\t<layername>" + vlayer.getName() + "</layername>\n");
206                out.write("\t\t\t<srs>\n");
207                
208                out.write("\t\t\t\t<spatialrefsys>\n");
209
210                SRS srs = vlayer.getSrs();
211                
212                if(srs == null){
213                        
214                        srs = this.writer.getSRS();
215                        
216                }
217                
218                out.write("\t\t\t\t\t<proj4>" + srs.getProj4() + "</proj4>\n");
219                out.write("\t\t\t\t\t<srsid>" + srs.getSrsid() + "</srsid>\n");
220                out.write("\t\t\t\t\t<srid>" + srs.getSrid() + "</srid>\n");
221                out.write("\t\t\t\t\t<authid>" + srs.getAuthid() + "</authid>\n");
222                out.write("\t\t\t\t\t<description>" + srs.getDescription() + "</description>\n");
223                out.write("\t\t\t\t\t<projectionacronym>" + srs.getProjectionacronym() + "</projectionacronym>\n");
224                out.write("\t\t\t\t\t<ellipsoidacronym>" + srs.getEllipsoidacronym() + "</ellipsoidacronym>\n");
225                out.write("\t\t\t\t\t<geographicflag>true</geographicflag>\n");
226                
227                out.write("\t\t\t\t</spatialrefsys>\n");
228                
229                out.write("\t\t\t</srs>\n");
230                
231                if(vlayer.getInputType().equals(QGisConstants.inputType.csv)&&vlayer.getRenderer() != null){
232                        
233                        out.write("\t\t\t<provider encoding=\"System\">delimitedtext</provider>\n");
234                        
235                } else{
236                        
237                        out.write("\t\t\t<provider encoding=\"System\">ogr</provider>\n");
238                        
239                }
240                
241                if(vlayer.getVectorJoins().isEmpty()){
242                        
243                        out.write("\t\t\t<vectorjoins/>\n");
244                        
245                } else{
246                        
247                        out.write("\t\t\t<vectorjoins>\n");
248                        
249                        for(VectorJoin vj : vlayer.getVectorJoins()){
250                                
251                                out.write("\t\t\t\t<join joinFieldName=\"" + vj.getJoinFieldName() +"\" targetFieldName=\"" + vj.getTargetFieldName() + "\""
252                                                + " memoryCache=\"1\" joinLayerId=\"" + vj.getJoinLayerId().toString() + "\"/>\n");
253                                
254                        }
255                        
256                        out.write("\t\t\t</vectorjoins>\n");
257                        
258                }
259                
260                if(vlayer.getRenderer() != null){
261                        
262                        writeGeometryLayer(out,vlayer);
263                        
264                        out.write("\t\t\t<featureBlendMode>0</featureBlendMode>\n");
265                        out.write("\t\t\t<layerTransparency>" + Integer.toString(vlayer.getLayerTransparency()) + "</layerTransparency>\n");
266
267                }
268                
269                out.write("\t\t\t<editform></editform>\n");
270                out.write("\t\t\t<editforminit></editforminit>\n");
271                out.write("\t\t\t<featformsuppress>0</featformsuppress>\n");
272                out.write("\t\t\t<annotationform></annotationform>\n");
273                out.write("\t\t\t<editorlayout>generatedlayout</editorlayout>\n");
274                out.write("\t\t\t<excludeAttributesWMS/>\n");
275                out.write("\t\t\t<excludeAttributesWFS/>\n");
276                out.write("\t\t\t<attributeactions/>\n");
277                
278        }

280        private boolean isCsvDataLayer(VectorLayer vlayer) {
281                
282                return(vlayer.getInputType().equals(QGisConstants.inputType.csv) && !vlayer.getGeometryType().equals(QGisConstants.geometryType.No_geometry));
283        
284        }

286        private void writeRasterLayer(BufferedWriter out, QGisLayer layer)throws IOException {
287                
288                RasterLayer rlayer = (RasterLayer) layer;
289                
290                out.write("\t\t<maplayer minimumScale=\"0\" maximumScale=\"1e+08\" type=\"" + rlayer.getType() + "\" hasScaleBasedVisibilityFlag=\"0\">\n");
291                out.write("\t\t\t<id>" + rlayer.getId().toString() + "</id>\n");
292                
293                String relP = Paths.get(this.writer.getWorkingDir()).relativize(Paths.get(layer.getPath())).toString();
294                
295                out.write("\t\t\t<datasource>" + relP + "</datasource>\n");
296                out.write("\t\t\t<layername>" + layer.getName() + "</layername>\n");
297                
298                out.write("\t\t\t<srs>\n");
299                
300                out.write("\t\t\t\t<spatialrefsys>\n");
301                
302                SRS srs = layer.getSrs();
303                
304                if(srs == null){
305                        
306                        srs = this.writer.getSRS();
307                        
308                }
309                
310                out.write("\t\t\t\t\t<proj4>" + srs.getProj4() + "</proj4>\n");
311                out.write("\t\t\t\t\t<srsid>" + srs.getSrsid() + "</srsid>\n");
312                out.write("\t\t\t\t\t<srid>" + srs.getSrid() + "</srid>\n");
313                out.write("\t\t\t\t\t<authid>" + srs.getAuthid() + "</authid>\n");
314                out.write("\t\t\t\t\t<description>" + srs.getDescription() + "</description>\n");
315                out.write("\t\t\t\t\t<projectionacronym>" + srs.getProjectionacronym() + "</projectionacronym>\n");
316                out.write("\t\t\t\t\t<ellipsoidacronym>" + srs.getEllipsoidacronym() + "</ellipsoidacronym>\n");
317                out.write("\t\t\t\t\t<geographicflag>true</geographicflag>\n");
318                
319                out.write("\t\t\t\t</spatialrefsys>\n");
320                
321                out.write("\t\t\t</srs>\n");
322                out.write("\t\t\t<customproperties>\n");
323                
324                out.write("\t\t\t\t<property key=\"identify/format\" value=\"Value\"/>\n");
325                
326                out.write("\t\t\t</customproperties>\n");
327                out.write("\t\t\t<provider>gdal</provider>\n");
328                out.write("\t\t\t<noData>\n");
329                
330                out.write("\t\t\t\t<noDataList bandNo=\"1\" useSrcNoData=\"0\"/>\n");
331                out.write("\t\t\t\t<noDataList bandNo=\"2\" useSrcNoData=\"0\"/>\n");
332                out.write("\t\t\t\t<noDataList bandNo=\"3\" useSrcNoData=\"0\"/>\n");
333                
334                out.write("\t\t\t</noData>\n");
335                out.write("\t\t\t<pipe>\n");
336                
337                QGisRasterRenderer renderer = (QGisRasterRenderer)layer.getRenderer();
338                
339                out.write("\t\t\t\t<rasterrenderer opacity=\"" + renderer.getOpacity() + "\" alphaBand=\"" + renderer.getAlphaBand() +
340                                "\" blueBand=\"" + renderer.getBlueBand() + "\" greenBand=\"" + renderer.getGreenBand() + "\" type=\"" + renderer.getType() +
341                                "\" redBand=\"" + renderer.getRedBand() + "\">\n");
342                
343                out.write("\t\t\t\t\t<rasterTransparency/>\n");
344                
345                out.write("\t\t\t\t</rasterrenderer>\n");
346                out.write("\t\t\t\t<brightnesscontrast brightness=\"0\" contrast=\"0\"/>\n");
347                out.write("\t\t\t\t<huesaturation colorizeGreen=\"128\" colorizeOn=\"0\" colorizeRed=\"255\" colorizeBlue=\"128\" grayscaleMode=\"0\" saturation=\"0\" colorizeStrength=\"100\"/>\n");
348                out.write("\t\t\t\t<rasterresampler maxOversampling=\"2\"/>\n");
349                
350                out.write("\t\t\t</pipe>\n");
351                out.write("\t\t\t<blendMode>0</blendMode>\n");
352                
353        }

355        private void writeGeometryLayer(BufferedWriter out, VectorLayer layer) throws IOException {
356        
357                QGisRenderer qRenderer = layer.getRenderer();
358                
359                if(qRenderer.getRenderingType().equals(QGisConstants.renderingType.categorizedSymbol)){
360
361                        //TODO no classes for categorized renderer created so far...
362                        
363                } else if(qRenderer.getRenderingType().equals(QGisConstants.renderingType.graduatedSymbol)){
364                        
365                        GraduatedSymbolRenderer renderer = (GraduatedSymbolRenderer)qRenderer;
366                        
367                        out.write("\t\t\t<renderer-v2 attr=\"" + renderer.getRenderingAttribute() + "\" symbollevels=\"0\" type=\"" + renderer.getRenderingType().toString() + "\">\n");
368                        
369                        out.write("\t\t\t\t<ranges>\n");
370                        
371                        for(int i = 0; i < renderer.getRanges().length; i++){
372
373                            Range currentRange = renderer.getRanges()[i];
374                                out.write("\t\t\t\t\t<range symbol=\"" + i);
375                                if (currentRange.isLowerBoundSet())
376                    out.write("\" lower=\"" + currentRange.getLowerBound());
377                if (currentRange.isUpperBoundSet())
378                    out.write("\" upper=\"" + currentRange.getUpperBound());
379                out.write("\" label=\"" + renderer.getRanges()[i].getLabel() + "\"/>\n");
380                                
381                        }
382                        
383                        out.write("\t\t\t\t</ranges>\n");
384                        
385                } else if(qRenderer.getRenderingType().equals(QGisConstants.renderingType.RuleRenderer)){
386                        
387            RuleBasedRenderer renderer = (RuleBasedRenderer)qRenderer;
388
389            out.write("\t\t\t<renderer-v2 attr=\"" + renderer.getRenderingAttribute() + "\" symbollevels=\"0\" type=\"" + renderer.getRenderingType().toString() + "\">\n");
390
391            out.write("\t\t\t\t<rules>\n");
392
393            for(Rule rule : renderer.getRules()) {
394
395                out.write("\t\t\t\t\t<rule filter=\"" + rule.getFilter()
396                        + "\" symbol=\"" + rule.getSymbol() + "\" label=\"" + rule.getLabel() + "\"/>\n");
397
398            }
399
400            out.write("\t\t\t\t</rules>\n");
401                        
402                } else if(qRenderer.getRenderingType().equals(QGisConstants.renderingType.singleSymbol)){
403                        
404                        out.write("\t\t\t<renderer-v2 symbollevels=\"0\" type=\"" + qRenderer.getRenderingType().toString() + "\">\n");
405                        
406                }
407                
408                out.write("\t\t\t\t<symbols>\n");
409                
410                for(int i = 0; i < qRenderer.getSymbolLayers().size(); i++){
411                        
412                        if(layer.getGeometryType().equals(QGisConstants.geometryType.Line)){
413                                
414                                writeLineLayer(out, layer, i);
415                                
416                        } else if(layer.getGeometryType().equals(QGisConstants.geometryType.Point)){
417                                
418                                writePointLayer(out, layer, i);
419                                
420                        } else{
421                                throw new RuntimeException("Unknown geometry type! Cannot write symbol layer!");
422                        }
423                        
424                }
425                
426                out.write("\t\t\t\t</symbols>\n");
427                out.write("\t\t\t\t<rotation/>\n");
428                out.write("\t\t\t\t<sizescale scalemethod=\"area\"/>\n");
429                
430                out.write("\t\t\t</renderer-v2>\n");
431                
432        }

434        private void writePointLayer(BufferedWriter out, QGisLayer layer, int idx) throws IOException {
435
436                QGisPointSymbolLayer psl = (QGisPointSymbolLayer)layer.getRenderer().getSymbolLayers().get(idx);
437                
438                String color = Integer.toString(psl.getColor().getRed()) + "," 
439                                + Integer.toString(psl.getColor().getGreen()) + ","
440                                + Integer.toString(psl.getColor().getBlue()) + ","
441                                + Integer.toString(psl.getColor().getAlpha());
442                
443                String colorBorder = Integer.toString(psl.getColorBorder().getRed()) + ","
444                                + Integer.toString(psl.getColorBorder().getGreen()) + ","
445                                + Integer.toString(psl.getColorBorder().getBlue()) + ","
446                                + Integer.toString(psl.getColorBorder().getAlpha());
447                
448                String offset = Double.toString(psl.getOffset()[0]) + "," + Double.toString(psl.getOffset()[1]);
449                String offsetMapUnitScale = Double.toString(psl.getOffsetMapUnitScale()[0]) + "," +
450                                Double.toString(psl.getOffsetMapUnitScale()[1]); 
451                
452                out.write("\t\t\t\t\t<symbol alpha=\"" + psl.getLayerTransparency() + "\" type=\"" + psl.getSymbolType().toString().toLowerCase() + "\" name=\"" + idx + "\">\n");
453                
454                out.write("\t\t\t\t\t\t<layer pass=\"0\" class=\"" + layer.getLayerClass().toString() + "\" locked=\"0\">\n");
455                
456                out.write("\t\t\t\t\t\t\t<prop k=\"angle\" v=\"0\"/>\n");
457                out.write("\t\t\t\t\t\t\t<prop k=\"color\" v=\"" + color + "\"/>\n");
458                out.write("\t\t\t\t\t\t\t<prop k=\"color_border\" v=\"" + colorBorder + "\"/>\n");
459                out.write("\t\t\t\t\t\t\t<prop k=\"horizontal_anchor_point\" v=\"1\"/>\n");
460                out.write("\t\t\t\t\t\t\t<prop k=\"name\" v=\"" + psl.getPointLayerSymbol().toString() + "\"/>\n");
461                out.write("\t\t\t\t\t\t\t<prop k=\"offset\" v=\"" + offset + "\"/>\n");
462                out.write("\t\t\t\t\t\t\t<prop k=\"offset_map_unit_scale\" v=\"" + offsetMapUnitScale + "\"/>\n");
463                out.write("\t\t\t\t\t\t\t<prop k=\"offset_unit\" v=\"" + psl.getSizeUnits().toString() + "\"/>\n");
464                out.write("\t\t\t\t\t\t\t<prop k=\"outline_style\" v=\"" + psl.getPenStyle().toString() + "\"/>\n");
465                out.write("\t\t\t\t\t\t\t<prop k=\"outline_width\" v=\"0\"/>\n");
466                out.write("\t\t\t\t\t\t\t<prop k=\"outline_width_map_unit_scale\" v=\"0,0\"/>\n");
467                out.write("\t\t\t\t\t\t\t<prop k=\"outline_width_unit\" v=\"" + psl.getSizeUnits().toString() + "\"/>\n");
468                out.write("\t\t\t\t\t\t\t<prop k=\"scale_method\" v=\"area\"/>\n");
469                out.write("\t\t\t\t\t\t\t<prop k=\"size\" v=\"" + Double.toString(psl.getSize()) + "\"/>\n");
470                out.write("\t\t\t\t\t\t\t<prop k=\"size_map_unit_scale\" v=\"0,0\"/>\n");
471                out.write("\t\t\t\t\t\t\t<prop k=\"size_unit\" v=\"" + psl.getSizeUnits().toString() + "\"/>\n");
472                out.write("\t\t\t\t\t\t\t<prop k=\"vertical_anchor_point\" v=\"1\"/>\n");
473                
474                out.write("\t\t\t\t\t\t</layer>\n");
475                
476                out.write("\t\t\t\t\t</symbol>\n");
477                        
478        }

480        private void writeLineLayer(BufferedWriter out, QGisLayer layer, int idx) throws IOException {
481                
482                QGisLineSymbolLayer lsl = (QGisLineSymbolLayer)layer.getRenderer().getSymbolLayers().get(0);
483                
484                String color = Integer.toString(lsl.getColor().getRed()) + "," 
485                                + Integer.toString(lsl.getColor().getBlue()) + "," +
486                                Integer.toString(lsl.getColor().getGreen()) + ";" +
487                                Integer.toString(lsl.getColor().getAlpha());
488                
489                out.write("\t\t\t\t\t<symbol alpha=\"1\" type=\"" + lsl.getSymbolType().toString().toLowerCase() + "\" name=\"" + idx + "\">\n");
490                
491                out.write("\t\t\t\t\t\t<layer pass=\"0\" class=\"" + layer.getLayerClass().toString() + "\" locked=\"0\">\n");
492                
493                out.write("\t\t\t\t\t\t\t<prop k=\"capstyle\" v=\"square\"/>\n");
