001package org.matsim.contrib.analysis.vsp.qgis;
002
003import org.matsim.core.utils.geometry.transformations.TransformationFactory;
004
005public class SRS{
006
007        private String proj4;
008        private String srsid;
009        private String srid;
010        private String authid;
011        private String description;
012        private String projectionacronym;
013        private String ellipsoidacronym;
014
015        private SRS(String proj4, String srsid, String srid, String authid, String description, String projectionacronym, String ellipsoidacronym){
016                this.proj4 = proj4;
017                this.srsid = srsid;
018                this.srid = srid;
019                this.authid = authid;
020                this.description = description;
021                this.projectionacronym = projectionacronym;
022                this.ellipsoidacronym = ellipsoidacronym;
023        }
024
025        /**
026         * Creates a new SRS from an input string (e.g. {@code TransformationFactory.WGS84}).
027         *
028         * @param srs String representation of the spatial reference system you want to use here
029         * @return
030         */
031        public static SRS createSpatialRefSys(String srs){
032
033                switch (srs) {
034                        case TransformationFactory.DHDN_GK4:
035
036                                return new SRS("+proj=tmerc +lat_0=0 +lon_0=12 +k=1 +x_0=4500000 +y_0=0 +ellps=bessel +towgs84=598.1,73.7,418.2,0.202,0.045,-2.455,6.7 +units=m +no_defs",
037                                                "2648", "31468", "EPSG:31468", "DHDN / Gauss-Kruger zone 4", "tmerc", "bessel");
038
039                        case TransformationFactory.WGS84:
040
041                                return new SRS("+proj=longlat +datum=WGS84 +no_defs",
042                                                "3452", "4326", "EPSG:4326", "WGS 84", "longlat", "WGS84");
043
044                        case TransformationFactory.WGS84_SA_Albers:
045
046                                return new SRS("+proj=aea +lat_1=-18 +lat_2=-32 +lat_0=0 +lon_0=24 +x_0=0 +y_0=0 +datum=WGS84 +units=m +no_defs",
047                                                "100000", "0", "USER:100000", "WGS84_SA_Albers", "aea", "");
048
049                        case "WGS84_Pseudo_Mercator":
050
051                                return new SRS("+proj=merc +a=6378137 +b=6378137 +lat_ts=0.0 +lon_0=0.0 +x_0=0.0 +y_0=0 +k=1.0 +units=m +nadgrids=@null +wktext  +no_defs",
052                                                "3857", "3857", "EPSG:3857", "WGS 84 / Pseudo Mercator", "merc", "WGS84");
053
054                        // TODO the following is a quick fixing to make ESPG:21037 useable; right now I don't
055                        // see, however, why we need this class instead of just using the coordinate systems from MGC
056                        case "EPSG:21037":
057
058                                return new SRS("+proj=utm +zone=37 +south +ellps=clrk80 +units=m +no_defs",
059                                                "21037", "21037", "EPSG:21037", "Arc 1960 / UTM zone 3Ss", "", "");
060
061                        // TODO the following is a quick fixing to make TransformationFactory.WGS84_UTM31N useable; right now I don't
062                        // see, however, why we need this class instead of just using the coordinate systems from MGC
063                        case TransformationFactory.WGS84_UTM31N:  // EPSG:32641, e.g. for Hasselt
064
065                                return new SRS("+proj=utm +zone=31 +ellps=WGS84 +datum=WGS84 +units=m +no_defs",
066                                                "32631", "32631", "EPSG:32631", "WGS 84 / UTM zone 31N", "", "");
067
068                        // TODO the following is a quick fixing to make TransformationFactory.WGS84_UTM33N useable... see above
069                        case TransformationFactory.WGS84_UTM33N:  // EPSG:32641, e.g. for Hasselt
070
071                                return new SRS("+proj=utm +zone=33 +ellps=WGS84 +datum=WGS84 +units=m +no_defs",
072                                                "32633", "32633", "EPSG:32633", "WGS 84 / UTM zone 33N", "", "");
073
074                        default:
075                                try {
076                                        return new SRS(
077                                                        getProj4FromSrs(srs),            // Proj4
078                                                        getSrsidFromSrs(srs),            // Srsid
079                                                        getSridFromSrs(srs),                // Srid
080                                                        getAuthidFromSrs(srs),            // Authid
081                                                        getDescriptionFromSrs(srs),        // Description
082                                                        getProjectionacronymFromSrs(srs),    // Projectionacronym
083                                                        getEllipsoidacronymFromSrs(srs)    // Ellipsoidacronym
084                                        );
085                                } catch (Exception e) {
086                                        e.printStackTrace();
087                                        throw new RuntimeException("Unsupported coordinate system.");
088                                }
089                }
090        }
091
092        private static String getProj4FromSrs(String srs) {
093                return (new org.osgeo.proj4j.CRSFactory()).createFromName(srs).getParameterString();
094        }
095
096        private static String getSrsidFromSrs(String srs) {
097                return splitSrsString(srs);
098        }
099
100        private static String splitSrsString(String srs) {
101                String[] splitString = srs.split(":");
102                return splitString[1];
103        }
104
105        private static String getSridFromSrs(String srs) {
106                return splitSrsString(srs);
107        }
108
109        private static String getAuthidFromSrs(String srs) {
110                return srs;
111        }
112
113        private static String getDescriptionFromSrs(String srs) {
114                return srs;
115        }
116
117        private static String getProjectionacronymFromSrs(String srs) {
118                return (new org.osgeo.proj4j.CRSFactory()).createFromName(srs).getProjection().getName();
119        }
120
121        private static String getEllipsoidacronymFromSrs(String srs) {
122                return (new org.osgeo.proj4j.CRSFactory()).createFromName(srs).getProjection().getEllipsoid().getName();
123        }
124
125        public String getProj4() {
126                return proj4;
127        }
128
129        public String getSrsid() {
130                return srsid;
131        }
132
133        public String getSrid() {
134                return srid;
135        }
136
137        public String getAuthid() {
138                return authid;
139        }
140
141        public String getDescription() {
142                return description;
143        }
144
145        public String getProjectionacronym() {
146                return projectionacronym;
147        }
148
149        public String getEllipsoidacronym() {
150                return ellipsoidacronym;
151        }
152
153        @Override
154        public String toString() {
155                return
156                                "Proj4: '" + proj4 + "'\n" +
157                                                "Srsid: '" + srsid + "'\n" +
158                                                "Srid: '" + srid + "'\n" +
159                                                "Authid: '" + authid + "'\n" +
160                                                "Description: '" + description + "'\n" +
161                                                "Projectionacronym: '" + projectionacronym + "'\n" +
162                                                "Ellipsoidacronym: '" + ellipsoidacronym + "'";
163        }
164}
