001package org.matsim.contrib.analysis.vsp.qgis;
002
003
004import org.matsim.contrib.analysis.vsp.qgis.utils.ColorRangeUtils;
005
006import java.awt.*;
007
008/**
009 * Renderer that draws graduated symbols.
010 * </p>
011 * This type of renderer needs a rendering attribute (to classify the symbols),
012 * ranges (from which to which value a symbol is drawn in a specific way) and at least
013 * two symbol layers.
014 * 
015 * @author dhosse, gthunig
016 *
017 */
018public class GraduatedSymbolRenderer extends QGisRenderer {
019
020        private String renderingAttribute;
021        private boolean useHeader;
022        private String fileHeader;
023
024    private Range[] ranges;
025    private double lowerBound;
026    private double upperBound;
027    private int range;
028    private int symbolSize;
029    private ColorRangeUtils.ColorRange colorRange;
030
031    public GraduatedSymbolRenderer(VectorLayer layer, Double lowerBound, Double upperBound,
032                                 Integer range, int symbolSize) {
033        this(layer, lowerBound, upperBound, range, symbolSize, ColorRangeUtils.ColorRange.RED_TO_GREEN);
034    }
035
036    public GraduatedSymbolRenderer(VectorLayer layer, Double lowerBound, Double upperBound,
037                                   Integer range, int symbolSize, ColorRangeUtils.ColorRange colorRange) {
038        super(QGisConstants.renderingType.graduatedSymbol, layer);
039        this.fileHeader = layer.getHeader();
040        this.useHeader = fileHeader != null;
041        this.lowerBound = lowerBound;
042        this.upperBound = upperBound;
043        this.range = range;
044        this.symbolSize = symbolSize;
045        this.colorRange = colorRange;
046        init();
047    }
048        
049        public Range[] getRanges() {
050            return ranges;
051    }
052
053        String getRenderingAttribute(){
054                return this.renderingAttribute;
055        }
056
057        public void setRenderingAttribute(String attr){
058                if(this.useHeader){
059                        if(this.fileHeader.contains(attr)){
060                                this.renderingAttribute = attr;
061                        } else{
062                                throw new RuntimeException("Rendering attribute " + attr + " does not exist in header!");
063                        }
064                } else{
065                        throw new RuntimeException("The input file for this renderer has no header. Use method \"setRenderingAttribute(int columnIndex)\" instead!");
066                }
067        }
068        
069        void setRenderingAttribute(int columnIndex){
070                if(!this.useHeader){
071                        this.renderingAttribute = "field_" + Integer.toString(columnIndex);
072                } else{
073                        throw new RuntimeException("The input file for this renderer has a header. Use method \"setRenderingAttribute(String attr)\" instead!");
074                }
075        }
076
077    @Override
078    public void init() {
079        double spread = this.upperBound - this.lowerBound;
080        double stepSize = spread / (this.range - 2);
081
082        this.ranges = new Range[this.range];
083        this.ranges[0] = createFirstRange();
084
085        for (int i = 0; i < range - 2; i++) {
086            Double lowerBoundary = this.lowerBound + i * stepSize;
087            Double upperBoundary = this.lowerBound + (i+1) * stepSize;
088            this.ranges[i+1] = new Range(lowerBoundary, upperBoundary, lowerBoundary.toString() + " - " + upperBoundary.toString());
089        }
090
091        if (range > 1)
092            this.ranges[this.range - 1] = createLastRange();
093
094        createQGisPointSymbolLayers();
095    }
096
097    private Range createFirstRange() {
098            Range firstRange = new Range("Smaller than " + String.valueOf(lowerBound));
099            firstRange.setLowerBound(-100);
100            firstRange.setUpperBound(lowerBound);
101            return firstRange;
102    }
103
104    private Range createLastRange() {
105        Range lastRange = new Range("Greater than " + String.valueOf(lowerBound));
106        lastRange.setLowerBound(upperBound);
107        lastRange.setUpperBound(100);
108        return lastRange;
109    }
110
111    private void createQGisPointSymbolLayers() {
112
113        double colorRangeStep = 1.0 / (double)(range-1);
114        double sizeMapUnitScale[] = {0,0};
115
116        for (int i = 0; i < range; i++) {
117            QGisPointSymbolLayer currentLevel = new QGisPointSymbolLayer();
118            currentLevel.setId(i);
119            currentLevel.setColor(ColorRangeUtils.getColor(colorRange, colorRangeStep*i));
120            currentLevel.setColorBorder(new Color(0,0,0,255));
121            currentLevel.setPointLayerSymbol(QGisConstants.pointLayerSymbol.rectangle);
122            currentLevel.setSize(symbolSize);
123            currentLevel.setSizeUnits(QGisConstants.sizeUnits.MapUnit);
124            currentLevel.setSizeMapUnitScale(sizeMapUnitScale);
125            currentLevel.setPenStyle(QGisConstants.penstyle.no);
126            currentLevel.setLayerTransparency(1);
127            this.addSymbolLayer(currentLevel);
128        }
129    }
130
131}
