001package org.matsim.contrib.analysis.vsp.qgis;
002
003import org.apache.log4j.Logger;
004import org.matsim.contrib.analysis.vsp.qgis.utils.AccessibilityRuleWithFilterCreator;
005import org.matsim.contrib.analysis.vsp.qgis.utils.ColorRangeUtils;
006
007import java.awt.*;
008
009/**
010 *
011 * @author gthunig on 17.05.2017
012 *
013 */
014public class RuleBasedRenderer extends QGisRenderer {
015    public static final Logger log = Logger.getLogger(RuleBasedRenderer.class);
016
017    private String nameOfAccessibilityLayer;
018    private double lowerBound;
019    private double upperBound;
020
021    private String nameOfDensityLayer;
022    private double minimumDensity;
023
024    private int range;
025    private int symbolSize;
026    private ColorRangeUtils.ColorRange colorRange;
027
028    private Rule[] rules;
029
030    private String renderingAttribute;
031    private boolean useHeader;
032    private String fileHeader;
033
034    public RuleBasedRenderer(VectorLayer layer, double lowerBound, double upperBound, int range, int symbolSize,
035                             String nameOfAccessibilityLayer, String nameOfDensityLayer, double minimumDensity) {
036        this(layer, lowerBound, upperBound, range, symbolSize, ColorRangeUtils.ColorRange.RED_TO_GREEN,
037                nameOfAccessibilityLayer, nameOfDensityLayer, minimumDensity);
038    }
039
040    public RuleBasedRenderer(VectorLayer layer, double lowerBound, double upperBound, int range, int symbolSize,
041                             ColorRangeUtils.ColorRange colorRange,
042                             String nameOfAccessibilityLayer, String nameOfDensityLayer, double minimumDensity) {
043        super(QGisConstants.renderingType.RuleRenderer, layer);
044        this.fileHeader = layer.getHeader();
045        this.useHeader = fileHeader != null;
046        this.lowerBound = lowerBound;
047        this.upperBound = upperBound;
048        this.range = range;
049        this.symbolSize = symbolSize;
050        this.colorRange = colorRange;
051        this.nameOfAccessibilityLayer = nameOfAccessibilityLayer;
052        this.nameOfDensityLayer = nameOfDensityLayer;
053        this.minimumDensity = minimumDensity;
054        init();
055    }
056
057    public Rule[] getRules() {
058        return rules;
059    }
060
061    public String getRenderingAttribute(){
062
063        return this.renderingAttribute;
064
065    }
066
067    public void setRenderingAttribute(String attr){
068
069        if(this.useHeader){
070
071            if(this.fileHeader.contains(attr)){
072
073                this.renderingAttribute = attr;
074
075            } else{
076
077                throw new RuntimeException("Rendering attribute " + attr + " does not exist in header!");
078
079            }
080
081        } else{
082
083            throw new RuntimeException("The input file for this renderer has no header. Use method \"setRenderingAttribute(int columnIndex)\" instead!");
084
085        }
086
087    }
088
089    public void setRenderingAttribute(int columnIndex){
090
091        if(!this.useHeader){
092
093            this.renderingAttribute = "field_" + Integer.toString(columnIndex);
094
095        } else{
096
097            throw new RuntimeException("The input file for this renderer has a header. Use method \"setRenderingAttribute(String attr)\" instead!");
098
099        }
100
101    }
102
103    @Override
104    public void init() {
105        this.rules = new Rule[range];
106        createRules();
107
108        createQGisPointSymbolLayers();
109    }
110
111    private void createQGisPointSymbolLayers() {
112
113        double colorRangeStep = 1.0 / (double)(range-1);
114        double sizeMapUnitScale[] = {0,0};
115
116        for (int i = 0; i < range; i++) {
117            QGisPointSymbolLayer currentLevel = new QGisPointSymbolLayer();
118            currentLevel.setId(i);
119            currentLevel.setColor(ColorRangeUtils.getColor(colorRange, colorRangeStep*i));
120            currentLevel.setColorBorder(new Color(0,0,0,255));
121            currentLevel.setPointLayerSymbol(QGisConstants.pointLayerSymbol.rectangle);
122            currentLevel.setSize(symbolSize);
123            currentLevel.setSizeUnits(QGisConstants.sizeUnits.MapUnit);
124            currentLevel.setSizeMapUnitScale(sizeMapUnitScale);
125            currentLevel.setPenStyle(QGisConstants.penstyle.no);
126            currentLevel.setLayerTransparency(1);
127            this.addSymbolLayer(currentLevel);
128        }
129    }
130
131    private void createRules() {
132        createFirstRule();
133        createMidRules();
134        createLastRule();
135    }
136
137    private void createFirstRule() {
138        rules[0] = AccessibilityRuleWithFilterCreator.createRuleWithoutLowerBound(nameOfAccessibilityLayer,
139                lowerBound, nameOfDensityLayer, minimumDensity, 0, String.valueOf(lowerBound));
140    }
141
142    private void createMidRules() {
143        for (int i = 1; i < range -1; i++) {
144            createMidRule(i);
145        }
146    }
147
148    private void createMidRule(int i) {
149        double lowerBound = calcLowerBound(i);
150        double upperBound = calcUpperBound(i);
151        rules[i] = AccessibilityRuleWithFilterCreator.createRule(nameOfAccessibilityLayer, lowerBound, upperBound,
152                nameOfDensityLayer, minimumDensity, i,
153                String.valueOf(lowerBound) + " - " + String.valueOf(upperBound));
154    }
155
156    private void createLastRule() {
157        double lowerBound = calcLowerBound(range - 1);
158        rules[range -1] = AccessibilityRuleWithFilterCreator.createRuleWithoutUpperBound(nameOfAccessibilityLayer,
159                lowerBound, nameOfDensityLayer, minimumDensity,
160                range -1, "> " + String.valueOf(lowerBound));
161    }
162
163    private double calcLowerBound(int i) {
164        return lowerBound + ((upperBound-lowerBound)/(range-2)) * (i - 1);
165    }
166
167    private double calcUpperBound(int i) {
168        return calcLowerBound(i + 1);
169    }

171
172
173}
