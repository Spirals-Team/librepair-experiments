001package org.matsim.contrib.analysis.vsp.qgis;
002
003import java.io.BufferedReader;
004import java.io.File;
005import java.io.FileNotFoundException;
006import java.io.FileReader;
007import java.io.IOException;
008import java.text.SimpleDateFormat;
009import java.util.Date;
010
011import org.matsim.api.core.v01.Id;
012import org.matsim.core.utils.io.IOUtils;
013
014public abstract class QGisLayer {
015        
016        private Id<QGisLayer> id;
017        private String name;
018        private String path;
019        
020        private QGisConstants.inputType inputType;
021        
022        private QGisConstants.layerClass layerClass;
023        
024        // must be specified for all layers (except for data layers)
025        private QGisRenderer renderer;
026        
027        private QGisConstants.layerType type;
028        
029        private SRS srs;
030        
031        /**
032         * Creates a new instance of a qgis layer.
033         * For each geometry layer, a {@code QGisRenderer} must be created (for use cases look at package layerTemplates).
034         * </p>
035         * If the layer input is a csv file, you also need to specify a delimiter (e.g. , or ;) and the header names
036         * of the columns that contain the x and y coordinates (by setXField and setYField).
037         * 
038         * @param name a unique name for the layer (as it is displayed in qgis later)
039         * @param path the path to the input file
040         * @param geometryType the type of geometry (none, point, line, polygon)
041         */
042        public QGisLayer(String name, String path){
043                
044                this.name = name;
045                this.path = path;
046                this.id = Id.create(name + new SimpleDateFormat("yyyyMMdd").format(new Date()), QGisLayer.class);
047                
048                if(this.path.contains(QGisConstants.inputType.csv.toString())){
049                        
050                        this.inputType = QGisConstants.inputType.csv;
051                        
052                } else if(this.path.contains(QGisConstants.inputType.shp.toString())){
053                        
054                        this.inputType = QGisConstants.inputType.shp;
055                        
056                } else if(this.path.contains(QGisConstants.inputType.xml.toString())){
057                        
058                        this.inputType = QGisConstants.inputType.xml;
059                        
060                } else{
061                        
062                        throw new RuntimeException("Unsupported input type! Cannot create layer.");
063                        
064                }
065                
066        }
067        
068        public Id<QGisLayer> getId(){
069                return this.id;
070        }
071        
072        protected void setId(Id<QGisLayer> id){
073                this.id = id;
074        }
075        
076        public String getName(){
077                return this.name;
078        }
079        
080        protected void setName(String name){
081                this.name = name;
082        }
083        
084        public QGisRenderer getRenderer() {
085                return renderer;
086        }
087
088        protected void setRenderer(QGisRenderer renderer) {
089                this.renderer = renderer;
090        }
091
092        public QGisConstants.layerClass getLayerClass() {
093                return layerClass;
094        }
095
096        public void setLayerClass(QGisConstants.layerClass layerClass) {
097                this.layerClass = layerClass;
098        }
099        
100        public String getPath(){
101                return this.path;
102        }
103        
104        public QGisConstants.inputType getInputType(){
105                return this.inputType;
106        }
107
108        public QGisConstants.layerType getType() {
109                return type;
110        }
111
112        public void setType(QGisConstants.layerType type) {
113                this.type = type;
114        }
115
116        public SRS getSrs() {
117                return srs;
118        }
119
120        public void setSrs(String srs) {
121                this.srs = SRS.createSpatialRefSys(srs);
122        }
123        
124}
