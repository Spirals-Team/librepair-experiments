023package org.matsim.contrib.analysis.vsp.traveltimedistance;
024
025import org.matsim.api.core.v01.Scenario;
026import org.matsim.core.config.ConfigUtils;
027import org.matsim.core.network.io.MatsimNetworkReader;
028import org.matsim.core.population.io.PopulationReader;
029import org.matsim.core.scenario.ScenarioUtils;
030import org.matsim.core.utils.geometry.CoordinateTransformation;
031import org.matsim.core.utils.geometry.transformations.TransformationFactory;
032
033/**
034 * @author  jbischoff
035 * An example how to validate car travel times in MATSim against an external source. In this example, the Here Maps API is used.
036 * 
037 */
038
039public class RunTraveltimeValidationExample {
040
041        /**
042         * 
043         * @param args
044         *  Arguments to pass:
045  <ol type="1">
046        <li>A MATSim Plans file</li>
047        <li>A MATSim Events file</li>
048        <li>A MATSim Network file</li>
049        <li>EPSG-Code of your coordinate system</li>
050        <li>HERE Maps APP ID, see here.com</li>
051        <li>HERE Maps APP code, see here.com</li>
052        <li>Output folder location</li>
053        <li>The date to validate travel times for, format: YYYY-MM-DD</li>
054
055        <li>(Optional: The number of trips to validate)</li>
056
057        </ol>
058         *  
059         */
060        public static void main(String[] args) {
061                String plans = args[0];
062                String events = args[1];
063                String network = args [2];
064                String epsg = args[3];
065                String appId = args[4];
066                String appCode = args[5];
067                Integer tripsToValidate = null;
068                String outputfolder = args[6];
069                String date = args[7];
070                if (args.length>8){
071                        tripsToValidate = Integer.parseInt(args[8]);
072                }
073                
074                Scenario scenario = ScenarioUtils.createScenario(ConfigUtils.createConfig());
075                new MatsimNetworkReader(scenario.getNetwork()).readFile(network);
076                new PopulationReader(scenario).readFile(plans);
077                CoordinateTransformation transformation = TransformationFactory.getCoordinateTransformation(epsg, TransformationFactory.WGS84);
078                HereMapsRouteValidator validator = new HereMapsRouteValidator(outputfolder, appId, appCode, date, transformation);
079                validator.setWriteDetailedFiles(true);
080                TravelTimeValidationRunner runner;
081                if (tripsToValidate != null){
082                runner = new TravelTimeValidationRunner(scenario.getNetwork(), scenario.getPopulation(), events, outputfolder, validator, tripsToValidate);
083                }
084                else  {
085                        runner = new TravelTimeValidationRunner(scenario.getNetwork(), scenario.getPopulation(), events, outputfolder,  validator);
086                }
087                runner.run();
088        }
089        
090}
