021import java.io.BufferedReader;
022import java.io.BufferedWriter;
023import java.io.File;
024import java.io.IOException;
025import java.io.InputStreamReader;
026import java.net.MalformedURLException;
027import java.net.URL;
028import java.text.DecimalFormat;
029import java.text.NumberFormat;
030import java.util.Locale;
031
032import org.json.simple.JSONArray;
033import org.json.simple.JSONObject;
034import org.json.simple.parser.JSONParser;
035import org.json.simple.parser.ParseException;
036import org.matsim.api.core.v01.Coord;
037import org.matsim.core.utils.collections.Tuple;
038import org.matsim.core.utils.geometry.CoordinateTransformation;
039import org.matsim.core.utils.io.IOUtils;

042/**
043 * @author jbischoff this class requests travel times and distances between two
044 *         coordinates from HERE Maps. Please replace the API code with your own
045 *         and obey here's usage policy
046 *
047 */
048public class HereMapsRouteValidator implements TravelTimeDistanceValidator {
049
050        final String appId;
051        final String appCode;
052        final String outputPath;
053        final String date;
054        final CoordinateTransformation transformation;
055        boolean writeDetailedFiles = true;
056
057        /**
058         * 
059         * @param outputFolder
060         *            folder to write gzipped json files
061         * @param appId
062         *            your here App ID
063         * @param appCode
064         *            your here App Code
065         * @param date
066         *            a date to run the validation for, format: 2017-06-08
067         * @param transformation
068         *            A coordinate transformation to WGS 84
069         */
070        public HereMapsRouteValidator(String outputFolder, String appId, String appCode, String date,
071                        CoordinateTransformation transformation) {
072                this.outputPath = outputFolder;
073                this.appId = appId;
074                this.appCode = appCode;
075                this.date = date;
076                this.transformation = transformation;
077                File outDir = new File(outputFolder);
078                if (!outDir.exists()) {
079                        outDir.mkdirs();
080                }
081
082        }
083
084        /*
085         * (non-Javadoc)
086         * 
087         * @see org.matsim.contrib.analysis.vsp.traveltimes.TravelTimeValidator#
088         * getTravelTime(org.matsim.contrib.analysis.vsp.traveltimes.CarTrip)
089         */
090        @Override
091        public Tuple<Double, Double> getTravelTime(CarTrip trip) {
092
093                long travelTime = 0;
094                long distance = 0;
095                Coord from = transformation.transform(trip.getDepartureLocation());
096                Coord to = transformation.transform(trip.getArrivalLocation());
097                String filename = outputPath + "/" + trip.getPersonId() + "_" + trip.getDepartureTime() + ".json.gz";
098                Locale locale = new Locale("en", "UK");
099                String pattern = "###.#####";
100
101                DecimalFormat df = (DecimalFormat) NumberFormat.getNumberInstance(locale);
102                df.applyPattern(pattern);
103
104                String urlString = "http://route.cit.api.here.com/routing/7.2/calculateroute.json?app_id=" + appId
105                                + "&app_code=" + appCode + "&waypoint0=geo!" + df.format(from.getY()) + "," + df.format(from.getX())
106                                + "&waypoint1=geo!" + df.format(to.getY()) + "," + df.format(to.getX()) + "&departure=" + date + "T"
107                                + Time.writeTime(trip.getDepartureTime()) + "&mode=fastest;car;traffic:enabled";
108
109                try {
110                        System.out.println(urlString);
111                        URL url = new URL(urlString);
112                        BufferedReader in = new BufferedReader(new InputStreamReader(url.openStream()));
113                        JSONParser jp = new JSONParser();
114
115                        JSONObject jsonObject = (JSONObject) jp.parse(in);
116                        JSONObject route = (JSONObject) ((JSONArray) ((JSONObject) jsonObject.get("response")).get("route")).get(0);
117                        JSONObject summary = (JSONObject) route.get("summary");
118                        travelTime = (long) summary.get("travelTime");
119                        distance = (long) summary.get("distance");
120
121                        // System.out.println(travelTime + " "+ baseTime + " "+distance);
122                        if (writeDetailedFiles){
123                        BufferedWriter bw = IOUtils.getBufferedWriter(filename);
124                        bw.write(jsonObject.toString());
125                        bw.flush();
126                        bw.close();
127                        }
128                } catch (MalformedURLException e) {
129                } catch (IOException e) {
130                } catch (ParseException e) {
131                }
132
133                return new Tuple<Double, Double>((double) travelTime, (double) distance);
134        }
135
136        /**
137         * @return the writeDetailedFiles
138         */
139        public boolean isWriteDetailedFiles() {
140                return writeDetailedFiles;
141        }
142
143        /**
144         * @param writeDetailedFiles
145         *            the writeDetailedFiles to set
146         */
147        public void setWriteDetailedFiles(boolean writeDetailedFiles) {
148                this.writeDetailedFiles = writeDetailedFiles;
149        }
150
151}
