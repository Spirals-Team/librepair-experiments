001package org.matsim.contrib.analysis.vsp.qgis;
002
003/**
004 * @author gthunig on 17.05.2017.
005 */
006public class Rule {
007
008    private String filter;
009    private int symbol;
010    private String label;
011
012    public Rule(String filter, int symbol, String label){
013
014        this.filter = filter;
015        this.symbol = symbol;
016        this.label = label;
017
018    }
019
020    public String getFilter() {
021        return filter;
022    }
023
024    public int getSymbol() {
025        return symbol;
026    }
027
028    public String getLabel() {
029        return label;
030    }
031}
