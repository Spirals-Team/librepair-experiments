001package org.matsim.contrib.analysis.vsp.qgis;
002
003import java.io.BufferedWriter;
004import java.io.IOException;
005
006import org.matsim.core.utils.io.IOUtils;
007
008public class QGisMapnikFileCreator {
009        
010        public static void writeMapnikFile(String osmMapnikFile) {
011                try {
012                        BufferedWriter writer = IOUtils.getBufferedWriter( osmMapnikFile ) ;
013
014                        writer.write("<GDAL_WMS>\n");
015                        writer.write("\t<Service name=\"TMS\">\n");
016                        writer.write("\t\t<ServerUrl>http://tile.openstreetmap.org/${z}/${x}/${y}.png</ServerUrl>\n");
017                        writer.write("\t</Service>\n");
018                        writer.write("\t<DataWindow>\n");
019                        //TODO the following four lines seem to have to be made adjustable to transfer the file created here to another region (dz, dez'14)
020                        writer.write("\t\t<UpperLeftX>-20037508.34</UpperLeftX>\n");
021                        writer.write("\t\t<UpperLeftY>20037508.34</UpperLeftY>\n");
022                        writer.write("\t\t<LowerRightX>20037508.34</LowerRightX>\n");
023                        writer.write("\t\t<LowerRightY>-20037508.34</LowerRightY>\n");
024                        writer.write("\t\t<TileLevel>18</TileLevel>\n");
025                        writer.write("\t\t<TileCountX>1</TileCountX>\n");
026                        writer.write("\t\t<TileCountY>1</TileCountY>\n");
027                        writer.write("\t\t<YOrigin>top</YOrigin>\n");
028                        writer.write("\t</DataWindow>\n");
029                        writer.write("\t<Projection>EPSG:3857</Projection>\n");
030                        writer.write("\t<BlockSizeX>256</BlockSizeX>\n");
031                        writer.write("\t<BlockSizeY>256</BlockSizeY>\n");
032                        writer.write("\t<BandsCount>3</BandsCount>\n");
033                        writer.write("\t<Cache>\n");
034                        writer.write("\t\t<Path>/tmp/cache_osm_mapnik</Path>\n");
035                        writer.write("\t</Cache>\t\n");
036                        writer.write("</GDAL_WMS>\n");
037
038                        writer.flush();
039                        writer.close();
040                } catch (IOException e) {
041                        e.printStackTrace();
042                        throw new RuntimeException( "writing Mapnik file did not work") ;
043                }
044        }
045
046}
