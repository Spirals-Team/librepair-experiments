021package org.matsim.contrib.analysis.filters.population;
022
023import org.matsim.api.core.v01.population.Person;
024import org.matsim.core.population.algorithms.PersonAlgorithm;
025import org.matsim.facilities.filters.Filter;
026
027/**
028 * Interface for filtering persons.
029 *
030 * @author ychen
031 */
032public interface PersonFilter extends Filter, PersonAlgorithm {
033        /**
034         * judges whether the Person will be selected or not
035         *
036         * @param person person being judged
037         * @return true if the Person meets the criterion of the filter
038         */
039        boolean judge(Person person);
040
041        /**
042         * sends the person to the next PersonAlgorithm, which could be another filter.
043         *
044         * @param person person to be handled
045         */
046        @Override
047        void run(Person person);
048
049}
