021package org.matsim.contrib.analysis.filters.population;
022
023import java.util.List;
024import java.util.Map;
025
026import org.matsim.api.core.v01.Coord;
027import org.matsim.api.core.v01.Id;
028import org.matsim.api.core.v01.network.Link;
029import org.matsim.api.core.v01.network.Network;
030import org.matsim.api.core.v01.population.Activity;
031import org.matsim.api.core.v01.population.Leg;
032import org.matsim.api.core.v01.population.Person;
033import org.matsim.api.core.v01.population.Plan;
034import org.matsim.core.population.algorithms.PersonAlgorithm;
035import org.matsim.core.population.routes.NetworkRoute;
036import org.matsim.core.utils.geometry.CoordUtils;
037
038/**
039 * Filters all persons out whose routes do not lead at least once through
040 * one of the links within the <code>areaOfInterest</code>, passing only
041 * persons to the next algorithm that travel at least once on a link of
042 * the <code>areaOfInterest</code>. For the case a leg is missing route
043 * information, and thus missing the information which links are used,
044 * an {@linkplain #setAlternativeAOI(Coord, double) alternative area of
045 * interest can be specified}, given by its center and a radius. In that
046 * case it is tested whether the bee-line connecting the start and end
047 * point of the leg leads through the alternative aoi-circle.
048 *
049 * @author laemmel
050 * @author mrieser
051 */
052public class PersonIntersectAreaFilter extends AbstractPersonFilter {
053
054        private final Map<Id<Link>, Link> areaOfInterest;
055        private Coord aoiCenter = null;
056        private double aoiRadius = 0.0;
057        private final Network network;
058
059        public PersonIntersectAreaFilter(final PersonAlgorithm nextAlgorithm, final Map<Id<Link>, Link> areaOfInterest, final Network network) {
060                this.nextAlgorithm = nextAlgorithm;
061                this.areaOfInterest = areaOfInterest;
062                this.network = network;
063        }
064
065        /**
066         * Sets an alternative area of interest, given by its center and a radius.
067         * This alternative aoi is used when a route is missing in a leg and thus no
068         * links are available to decide. In that case it is checked if the bee-line
069         * between from and to activity cuts the circle with radius <code>aoiRadius</code>
070         * around <code>aoiCenter</code>.
071         *
072         * @param aoiCenter
073         * @param aoiRadius
074         */
075        public void setAlternativeAOI(final Coord aoiCenter, final double aoiRadius) {
076                this.aoiCenter = aoiCenter;
077                this.aoiRadius = aoiRadius;
078        }
079
080
081        @Override
082        public boolean judge(final Person person) {
083                for (Plan plan : person.getPlans()) {
084                        for (int i = 1, n = plan.getPlanElements().size(); i < n; i+=2) {
085                                Leg leg = (Leg) plan.getPlanElements().get(i);
086                                if (leg.getRoute() == null) {
087                                        if (judgeByBeeline((Activity) plan.getPlanElements().get(i-1), (Activity) plan.getPlanElements().get(i+1))) {
088                                                return true;
089                                        }
090                                }
091                                else if (leg.getRoute() instanceof NetworkRoute) {
092                                        List<Id<Link>> linkIds = ((NetworkRoute) leg.getRoute()).getLinkIds();
093                                        if (linkIds.size() == 0) {
094                                                if (judgeByBeeline((Activity) plan.getPlanElements().get(i-1), (Activity) plan.getPlanElements().get(i+1))) {
095                                                        return true;
096                                                }
097                                        }
098                                        else {
099                                                for (Id<Link> link : linkIds) {
100                                                        if (this.areaOfInterest.containsKey(link)) {
101                                                                return true;
102                                                        }
103                                                }
104                                                // test departure link
105                                                Id<Link> linkId = ((Activity) plan.getPlanElements().get(i-1)).getLinkId();
106                                                if ((linkId != null) && (this.areaOfInterest.containsKey(linkId))) {
107                                                        return true;
108                                                }
109                                                // test arrival link
110                                                linkId = ((Activity) plan.getPlanElements().get(i+1)).getLinkId();
111                                                if ((linkId != null) && (this.areaOfInterest.containsKey(linkId))) {
112                                                        return true;
113                                                }
114                                        }
115                                }
116                                else { // leg.getRoute() instanceof GenericRoute
117                                        if (judgeByBeeline((Activity) plan.getPlanElements().get(i-1), (Activity) plan.getPlanElements().get(i+1))) {
118                                                return true;
119                                        }
120                                }
121                        }
122                }
123                return false;
124        }
125
126        private boolean judgeByBeeline(final Activity fromAct, final Activity toAct) {
127                if (this.aoiCenter == null) {
128                        // we cannot use the bee-line decision if we don't know the alternative aoi-center
129                        return false;
130                }
131                Coord fromCoord = fromAct.getCoord();
132                Coord toCoord = toAct.getCoord();
133
134                if (fromCoord == null) {
135                        fromCoord = this.network.getLinks().get(fromAct.getLinkId()).getCoord();
136                }
137                if (toCoord == null) {
138                        toCoord = this.network.getLinks().get(toAct.getLinkId()).getCoord();
139                }
140
141                return (CoordUtils.distancePointLinesegment(fromCoord, toCoord, this.aoiCenter) <= this.aoiRadius);
142        }
143
144}
