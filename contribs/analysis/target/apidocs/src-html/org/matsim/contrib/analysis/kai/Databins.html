001package org.matsim.contrib.analysis.kai;
002
003import java.util.Map;
004import java.util.Map.Entry;
005import java.util.Set;
006import java.util.TreeMap;
007
008import org.apache.log4j.Logger;
009
010/** 
011 * Data structure that allows to register data boundaries, and then to count data in the thereby defined categories. 
012 * The data structure allows multiple keys, so it is, for example, possible to count different trip purposes into the same 
013 * categories.
014 * 
015 * @author nagel
016 *
017 * @param <K>
018 */
019public final class Databins<K> {
020        private static final Logger log = Logger.getLogger(Databins.class) ;
021        private Map<K,double[]> delegate = new TreeMap<>() ;
022        private double[] dataBoundaries ;
023        private final String typeName;
024        public Databins( String type, double[] dataBoundaries ) {
025                this.typeName = type ;
026                this.dataBoundaries = dataBoundaries ;
027        }
028        public double getValue(K key, int idx) {
029                if ( delegate.get(key)==null ) {
030                        instantiate(key) ;
031                }
032                return this.delegate.get(key)[idx] ;
033        }
034        public Set<Entry<K, double[]>> entrySet() {
035                return this.delegate.entrySet() ;
036        }
037        public double[] getDataBoundaries() {
038                return dataBoundaries ;
039        }
040        public double[] getValues(K key) {
041                if ( delegate.get(key)==null ) {
042                        instantiate(key) ;
043                }
044                return delegate.get(key) ;
045        }
046        public void instantiate(K key) {
047                double[] array = new double[ dataBoundaries.length] ;
048                delegate.put( key, array ) ;
049        }
050        public void addValue(K key, int idx, Double val) {
051                if ( delegate.get(key)==null ) {
052                        instantiate(key) ;
053                }
054                delegate.get(key)[idx] += val ;
055        }
056        public void inc(K key, int idx) {
057                if ( delegate.get(key)==null ) {
058                        instantiate(key) ;
059                }
060                delegate.get(key)[idx] ++ ;
061        }
062        public int getIndex(double dblVal) {
063                int ii = dataBoundaries.length-1 ;
064                for ( ; ii>=0 ; ii-- ) {
065                        if ( dataBoundaries[ii] <= dblVal ) 
066                                return ii ;
067                }
068                log.warn("statistics contains value that smaller than the smallest category; adding it to smallest category" ) ;
069                log.warn("statType: " + typeName + "; val: " + dblVal ) ;
070                return 0 ;
071        }
072        public void clear() {
073                this.delegate.clear();
074        }
