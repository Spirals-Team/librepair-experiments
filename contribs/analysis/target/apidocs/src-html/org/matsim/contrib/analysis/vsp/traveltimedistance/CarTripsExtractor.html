023package org.matsim.contrib.analysis.vsp.traveltimedistance;
024
025import java.util.ArrayList;
026import java.util.HashMap;
027import java.util.List;
028import java.util.Map;
029import java.util.Set;
030
031import org.matsim.api.core.v01.Coord;
032import org.matsim.api.core.v01.Id;
033import org.matsim.api.core.v01.TransportMode;
034import org.matsim.api.core.v01.events.LinkEnterEvent;
035import org.matsim.api.core.v01.events.PersonArrivalEvent;
036import org.matsim.api.core.v01.events.PersonDepartureEvent;
037import org.matsim.api.core.v01.events.PersonEntersVehicleEvent;
038import org.matsim.api.core.v01.events.handler.LinkEnterEventHandler;
039import org.matsim.api.core.v01.events.handler.PersonArrivalEventHandler;
040import org.matsim.api.core.v01.events.handler.PersonDepartureEventHandler;
041import org.matsim.api.core.v01.events.handler.PersonEntersVehicleEventHandler;
042import org.matsim.api.core.v01.network.Network;
043import org.matsim.api.core.v01.population.Person;
044import org.matsim.vehicles.Vehicle;
045
046/**
047 * @author  jbischoff
048 *
049 */
050/**
051 *
052 */
053public class CarTripsExtractor implements PersonDepartureEventHandler, PersonArrivalEventHandler, PersonEntersVehicleEventHandler, LinkEnterEventHandler {
054
055        
056        private final Set<Id<Person>> personsWithPlan;
057        private final Network network;
058        private final Map<Id<Person>,Double> lastDepartureTime = new HashMap<>();
059        private final Map<Id<Person>,Coord> lastDepartureLocation = new HashMap<>();
060        private final Map<Id<Person>,Id<Vehicle>> vehicle2pax  = new HashMap<>();;
061        private final Map<Id<Vehicle>,Double> distanceTraveled = new HashMap<>();
062        private final List<CarTrip> trips = new ArrayList<>();
063        
064        
065        public CarTripsExtractor(Set<Id<Person>> personsWithPlan, Network network) {
066                this.personsWithPlan = personsWithPlan;
067                this.network = network;
068        }
069
070        /* (non-Javadoc)
071         * @see org.matsim.core.events.handler.EventHandler#reset(int)
072         */
073        @Override
074        public void reset(int iteration) {
075                lastDepartureLocation.clear();
076                lastDepartureTime.clear();
077                vehicle2pax.clear();
078                distanceTraveled.clear();
079                trips.clear();
080        }
081
082        /* (non-Javadoc)
083         * @see org.matsim.api.core.v01.events.handler.PersonArrivalEventHandler#handleEvent(org.matsim.api.core.v01.events.PersonArrivalEvent)
084         */
085        @Override
086        public void handleEvent(PersonArrivalEvent event) {
087                if (event.getLegMode().equals(TransportMode.car)){
088                        if (this.personsWithPlan.contains(event.getPersonId()))
089                        {
090                                Coord arrivalCoord = network.getLinks().get(event.getLinkId()).getCoord();
091                                Coord departureCoord = this.lastDepartureLocation.remove(event.getPersonId());
092                                double departureTime = this.lastDepartureTime.remove(event.getPersonId());
093                                Id<Vehicle> vehicleId = vehicle2pax.remove(event.getPersonId());
094                                double distance = this.distanceTraveled.remove(vehicleId);
095                                CarTrip trip = new CarTrip(event.getPersonId(), departureTime, event.getTime(), distance, departureCoord, arrivalCoord);
096                                trip.setActualTravelTime(event.getTime()-departureTime);
097                                this.trips.add(trip);
098                        }
099                }
100        }
101
102        /* (non-Javadoc)
103         * @see org.matsim.api.core.v01.events.handler.PersonDepartureEventHandler#handleEvent(org.matsim.api.core.v01.events.PersonDepartureEvent)
104         */
105        @Override
106        public void handleEvent(PersonDepartureEvent event) {
107                if (event.getLegMode().equals(TransportMode.car)){
108                        if (this.personsWithPlan.contains(event.getPersonId()))
109                        {
110                                Coord departureCoord = network.getLinks().get(event.getLinkId()).getCoord();
111                                this.lastDepartureLocation.put(event.getPersonId(), departureCoord );
112                                this.lastDepartureTime.put(event.getPersonId(), event.getTime());
113
114                        }
115                }
116        }
117        /**
118         * @return the trips
119         */
120        public List<CarTrip> getTrips() {
121                return trips;
122        }
123
124        /* (non-Javadoc)
125         * @see org.matsim.api.core.v01.events.handler.LinkEnterEventHandler#handleEvent(org.matsim.api.core.v01.events.LinkEnterEvent)
126         */
127        @Override
128        public void handleEvent(LinkEnterEvent event) {
129                if (distanceTraveled.containsKey(event.getVehicleId())){
130                        double length =  network.getLinks().get(event.getLinkId()).getLength();
131                        double distance = distanceTraveled.get(event.getVehicleId()) + length;
132                        distanceTraveled.put(event.getVehicleId(), distance);
133                }
134        }
135
136        /* (non-Javadoc)
137         * @see org.matsim.api.core.v01.events.handler.PersonEntersVehicleEventHandler#handleEvent(org.matsim.api.core.v01.events.PersonEntersVehicleEvent)
138         */
139        @Override
140        public void handleEvent(PersonEntersVehicleEvent event) {
141                if (this.lastDepartureTime.containsKey(event.getPersonId())){
142                        this.vehicle2pax.put(event.getPersonId(), event.getVehicleId());
143                        this.distanceTraveled.put(event.getVehicleId(), 0.0);
144                }
145        }
146}
