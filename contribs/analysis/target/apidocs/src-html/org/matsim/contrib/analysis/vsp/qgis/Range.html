001package org.matsim.contrib.analysis.vsp.qgis;
002
003public class Range {
004
005        private double lowerBound;
006    boolean lowerBoundSet;
007    private double upperBound;
008    boolean upperBoundSet;
009    private String label;
010
011        public Range(double lowerBound, double upperBound, String label){
012                this.setLowerBound(lowerBound);
013                this.setUpperBound(upperBound);
014                this.label = label;
015        }
016
017    public Range(String label){
018        this.label = label;
019    }
020
021    public void setLowerBound(double lowerBound) {
022        this.lowerBound = lowerBound;
023        this.lowerBoundSet = true;
024    }
025
026    public double getLowerBound() {
027                return lowerBound;
028        }
029
030    public boolean isLowerBoundSet() {
031        return lowerBoundSet;
032    }
033
034    public void setUpperBound(double upperBound) {
035        this.upperBound = upperBound;
036        this.upperBoundSet = true;
037    }
038
039    public double getUpperBound() {
040                return upperBound;
041        }
042
043    public boolean isUpperBoundSet() {
044        return upperBoundSet;
045    }
046
047    public String getLabel() {
048                return label;
049        }
050
051}
