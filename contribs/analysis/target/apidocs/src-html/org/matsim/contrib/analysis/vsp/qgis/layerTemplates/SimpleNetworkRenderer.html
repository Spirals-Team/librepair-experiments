001package org.matsim.contrib.analysis.vsp.qgis.layerTemplates;
002
003import java.awt.Color;
004
005import org.matsim.contrib.analysis.vsp.qgis.QGisConstants;
006import org.matsim.contrib.analysis.vsp.qgis.QGisLineSymbolLayer;
007import org.matsim.contrib.analysis.vsp.qgis.QGisPointSymbolLayer;
008import org.matsim.contrib.analysis.vsp.qgis.QGisRenderer;
009import org.matsim.contrib.analysis.vsp.qgis.VectorLayer;
010
011public class SimpleNetworkRenderer extends QGisRenderer {
012
013        private QGisConstants.geometryType geometryType;
014        
015        /**
016         * Instantiates a renderer for drawing a simple network (e.g. links).
017         * The specifications for the symbol layer are made in the private method {@code init()}; 
018         * 
019         * @param gType The type of geometry that is to be drawn.
020         */
021        public SimpleNetworkRenderer(VectorLayer layer) {
022                
023                super(QGisConstants.renderingType.singleSymbol,layer);
024                this.geometryType = layer.getGeometryType();
025                
026                init();
027                
028        }
029        
030        @Override
031        public void init(){
032                
033                if(this.geometryType.equals(QGisConstants.geometryType.Point)){
034                        
035                        QGisPointSymbolLayer psl = new QGisPointSymbolLayer();
036                        psl.setId(0);
037                        psl.setColor(new Color(0,0,0,255));
038                        psl.setColorBorder(new Color(0,0,0,255));
039                        psl.setPenStyle(QGisConstants.penstyle.solid);
040                        psl.setLayerTransparency(1);
041                        psl.setSize(0.25);
042                        psl.setPointLayerSymbol(QGisConstants.pointLayerSymbol.circle);
043                        psl.setSizeUnits(QGisConstants.sizeUnits.MM);
044                        this.addSymbolLayer(psl);
045                        
046                } else if(this.geometryType.equals(QGisConstants.geometryType.Line)){
047                
048                        QGisLineSymbolLayer lsl = new QGisLineSymbolLayer();
049                        lsl.setId(0);
050                        lsl.setPenStyle(QGisConstants.penstyle.solid);
051                        lsl.setSizeUnits(QGisConstants.sizeUnits.MM);
052                        lsl.setColor(new Color(0,0,0,255));
053                        lsl.setLayerTransparency(0);
054                        lsl.setWidth(0.25);
055                        
056                        this.addSymbolLayer(lsl);
057                
058                } else{
059                        
060                        throw new RuntimeException("Unsupported geometry type!");
061                        
062                }
063                        
064        }
065
066}
