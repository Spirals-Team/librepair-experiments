023package org.matsim.contrib.noise.data;
024
025import java.util.ArrayList;
026import java.util.Collections;
027import java.util.HashMap;
028import java.util.List;
029import java.util.Map;
030
031import org.matsim.api.core.v01.Coord;
032import org.matsim.api.core.v01.Id;
033import org.matsim.api.core.v01.network.Link;
034import org.matsim.api.core.v01.population.Person;
035
036/**
037 * 
038 * Extends the basic information of a receiver point towards data required for the computation of noise.
039 * 
040 * @author ikaddoura
041 *
042 */
043public class NoiseReceiverPoint extends ReceiverPoint {
044        
045        public NoiseReceiverPoint(Id<ReceiverPoint> id, Coord coord) {
046                super(id, coord);
047        }
048
049        private Map<Id<Person>, List<PersonActivityInfo>> personId2actInfos = null;//new HashMap<>(0);
050        
051        // initialization
052        private Map<Id<Link>, Double> linkId2distanceCorrection = null;//new HashMap<>(0);
053        private Map<Id<Link>, Double> linkId2angleCorrection = null;//new HashMap<>(0);
054                        
055        // time-specific information
056        private double finalImmission = 0.;
057        private double affectedAgentUnits = 0.;
058        private double damageCosts;
059        private double damageCostsPerAffectedAgentUnit;
060
061        public Map<Id<Person>, List<PersonActivityInfo>> getPersonId2actInfos() {
062                if(personId2actInfos == null) {
063                        personId2actInfos = new HashMap<>(4);
064                }
065                return Collections.unmodifiableMap(personId2actInfos);
066        }
067
068        public void addPersonActInfo(Id<Person> person, PersonActivityInfo info) {
069                if(personId2actInfos == null) {
070                        personId2actInfos = new HashMap<>(4);
071                }
072                List<PersonActivityInfo> infos = this.personId2actInfos.get(person);
073                if(infos == null) {
074                        infos = new ArrayList<>();
075                        this.personId2actInfos.put(person, infos);
076                }
077                infos.add(info);
078        }
079        
080        public Map<Id<Link>, Double> getLinkId2distanceCorrection() {
081                if(linkId2distanceCorrection == null) {
082                        linkId2distanceCorrection = new HashMap<>();
083                }
084                return Collections.unmodifiableMap(linkId2distanceCorrection);
085        }
086
087        public void setLinkId2distanceCorrection(Id<Link> linkId,  Double distanceCorrection) {
088                if(linkId2distanceCorrection == null) {
089                        linkId2distanceCorrection = new HashMap<>();
090                }
091                this.linkId2distanceCorrection.put(linkId, distanceCorrection);
092        }
093
094        public Map<Id<Link>, Double> getLinkId2angleCorrection() {
095                if(linkId2angleCorrection== null) {
096                        linkId2angleCorrection = new HashMap<>();
097                }
098                return Collections.unmodifiableMap(linkId2angleCorrection);
099        }
100
101        public void setLinkId2angleCorrection(Id<Link> linkId, Double angleCorrection) {
102                if(linkId2angleCorrection== null) {
103                        linkId2angleCorrection = new HashMap<>();
104                }
105                this.linkId2angleCorrection.put(linkId, angleCorrection);
106        }
107
108        public double getFinalImmission() {
109                return finalImmission;
110        }
111
112        public void setFinalImmission(double finalImmission) {
113                this.finalImmission = finalImmission;
114        }
115
116        public double getDamageCosts() {
117                return damageCosts;
118        }
119
120        public void setDamageCosts(double damageCosts) {
121                this.damageCosts = damageCosts;
122        }
123
124        public double getDamageCostsPerAffectedAgentUnit() {
125                return damageCostsPerAffectedAgentUnit;
126        }
127
128        public void setDamageCostsPerAffectedAgentUnit(
129                        double damageCostsPerAffectedAgentUnit) {
130                this.damageCostsPerAffectedAgentUnit = damageCostsPerAffectedAgentUnit;
131        }
132
133        public double getAffectedAgentUnits() {
134                return affectedAgentUnits;
135        }
136
137        public void setAffectedAgentUnits(double affectedAgentsUnits) {
138                this.affectedAgentUnits = affectedAgentsUnits;
139        }
140
141        @Override
142        public String toString() {
143                return "NoiseReceiverPoint [personId2actInfos=" + personId2actInfos
144                                + ", linkId2distanceCorrection=" + linkId2distanceCorrection
145                                + ", linkId2angleCorrection=" + linkId2angleCorrection
146//                              + ", linkId2IsolatedImmission=" + linkId2IsolatedImmission
147//                              + ", linkId2IsolatedImmissionPlusOneCar=" + linkId2IsolatedImmissionPlusOneCar
148//                              + ", linkId2IsolatedImmissionPlusOneHGV=" + linkId2IsolatedImmissionPlusOneHGV 
149                                + ", finalImmission=" + finalImmission 
150                                + ", affectedAgentUnits=" + affectedAgentUnits
151                                + ", damageCosts=" + damageCosts 
152                                + ", damageCostsPerAffectedAgentUnit=" + damageCostsPerAffectedAgentUnit + "]";
153        }
154
155        public void reset() {
156                resetTimeInterval();
157                this.personId2actInfos = null;
158        }
159        
160        public void resetTimeInterval() {
161//              linkId2IsolatedImmission.clear();
162                this.setFinalImmission(0.);
163                this.setAffectedAgentUnits(0.);
164                this.setDamageCosts(0.);
165                this.setDamageCostsPerAffectedAgentUnit(0.);
166        }
167}
