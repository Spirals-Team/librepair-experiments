023package org.matsim.contrib.noise.handler;
024
025import java.util.HashMap;
026import java.util.Map;
027
028import org.matsim.api.core.v01.Id;
029import org.matsim.api.core.v01.events.LinkEnterEvent;
030import org.matsim.api.core.v01.events.LinkLeaveEvent;
031import org.matsim.api.core.v01.events.PersonArrivalEvent;
032import org.matsim.api.core.v01.events.handler.LinkEnterEventHandler;
033import org.matsim.api.core.v01.events.handler.LinkLeaveEventHandler;
034import org.matsim.api.core.v01.events.handler.PersonArrivalEventHandler;
035import org.matsim.contrib.noise.data.NoiseContext;
036import org.matsim.contrib.noise.data.NoiseLink;
037import org.matsim.vehicles.Vehicle;
038
039import com.google.inject.Inject;
040
041/**
042 * @author ikaddoura
043 *
044 */
045public class LinkSpeedCalculation implements LinkEnterEventHandler, LinkLeaveEventHandler, PersonArrivalEventHandler{
046
047        @Inject
048        private NoiseContext noiseContext;
049        
050        private Map<Id<Vehicle>, Double> vehicleId2enterTime = new HashMap<>();
051
052        @Override
053        public void reset(int iteration) {
054                this.vehicleId2enterTime.clear();
055        }
056
057        @Override
058        public void handleEvent(LinkLeaveEvent event) {
059                
060                if (this.vehicleId2enterTime.get(event.getVehicleId()) != null) {
061                        double traveltime = event.getTime() - this.vehicleId2enterTime.get(event.getVehicleId());
062                        
063                        boolean isHGV = false;
064                        for (String hgvPrefix : this.noiseContext.getNoiseParams().getHgvIdPrefixesArray()) {
065                                if (event.getVehicleId().toString().startsWith(hgvPrefix)) {
066                                        isHGV = true;
067                                        break;
068                                }
069                        }
070                        
071                        NoiseLink noiseLink = this.noiseContext.getNoiseLinks().get(event.getLinkId());
072                        if (isHGV || this.noiseContext.getBusVehicleIDs().contains(event.getVehicleId())) {
073                                // HGV or Bus
074                                if (noiseLink != null) {
075                                        double travelTimeSum = noiseLink.getTravelTimeHGV_sec() + traveltime;
076                                        noiseLink.setTravelTimeHGV_Sec(travelTimeSum);
077                                        int hgvAgents = noiseLink.getHgvAgentsLeaving() + 1;
078                                        noiseLink.setHgvAgentsLeaving(hgvAgents);
079                                        
080                                } else {
081                                        noiseLink = new NoiseLink(event.getLinkId());
082                                        noiseLink.setTravelTimeHGV_Sec(traveltime);
083                                        noiseLink.setHgvAgentsLeaving(1);
084                                        this.noiseContext.getNoiseLinks().put(event.getLinkId(), noiseLink);
085                                }
086                                                        
087                        } else {
088                                // Car
089                                if (noiseLink != null) {
090                                        double travelTimeSum = noiseLink.getTravelTimeCar_sec() + traveltime;
091                                        noiseLink.setTravelTimeCar_Sec(travelTimeSum);
092                                        int carAgents = noiseLink.getCarAgentsLeaving() + 1;
093                                        noiseLink.setCarAgentsLeaving(carAgents);
094                                        
095                                } else {
096                                        noiseLink = new NoiseLink(event.getLinkId());
097                                        noiseLink.setTravelTimeCar_Sec(traveltime);
098                                        noiseLink.setCarAgentsLeaving(1);
099                                        this.noiseContext.getNoiseLinks().put(event.getLinkId(), noiseLink);
100                                }
101                                        
102                        }
103                        
104                } else {
105                        // the person has just departed, don't count this vehicle
106                }
107        }
108        
109        @Override
110        public void handleEvent(PersonArrivalEvent event) {
111                // the person has arrived and is no longer traveling
112                this.vehicleId2enterTime.remove(event.getPersonId());
113        }
114
115        @Override
116        public void handleEvent(LinkEnterEvent event) {
117                // the person has entered the link, store the time
118                this.vehicleId2enterTime.put(event.getVehicleId(), event.getTime());
119        }
120
121        public void setNoiseContext(NoiseContext noiseContext) {
122                this.noiseContext = noiseContext;
123        }
124
125}
