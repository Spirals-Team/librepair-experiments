128        public final void run() {
129                // lv final. kai
130
131                readValues();
132                
133                if (receiverPointsFile != null) readReceiverPoints();
134                if (networkFile != null) loadNetwork();
135                
136                if (receiverPointsFile != null) {
137                        writeFileReceiverPoint();
138                } else if (networkFile != null) {
139                        writeFileLink();
140                } else {
141                        throw new RuntimeException("Either provide the receiver points file or the network file! Aborting...");
142                }
143        }

151        private void writeFileLink() {
152                int lineCounter = 0 ;
153
154                String outputFile = this.outputDirectory;
155
156                for (int i = 0; i < this.labels.length; i++) {
157                        outputFile = outputFile + this.labels[i] + "_";                         
158                }
159                outputFile = outputFile + "merged_" + this.outputFormat.toString() + ".csv.gz";
160
161                try ( BufferedWriter bw = IOUtils.getBufferedWriter(outputFile) ) {
162                        // so-called "try-with-resources". Kai
163
164                        log.info(" Writing merged file to " + outputFile + "...") ;
165
166                        // write headers
167                        switch( this.outputFormat ) {
168                        // yy should probably become different classes. kai
169                        case xyt1t2t3etc:
170                                bw.write("Id");
171                                for (String label : this.label2time2rp2value.keySet()) {
172                                        for (double time = startTime; time <= endTime; time = time + timeBinSize) {
173                                                bw.write(";" + label + "_" + Time.writeTime(time, Time.TIMEFORMAT_HHMMSS));
174                                        }
175                                }
176                                break;
177                        case xyt:
178                                bw.write("Id;time");
179                                for (String label : this.label2time2rp2value.keySet()) {
180                                        bw.write(";" + label);
181                                }
182                                break;
183                        default:
184                                throw new RuntimeException("not implemented") ;
185                        }
186
187                        bw.newLine();
188
189                        // fill table
190                        switch( this.outputFormat ) {
191                        case xyt1t2t3etc:       
192
193                                for (Id<Link> rp : this.network.getLinks().keySet()) {
194                                        bw.write(rp.toString());
195
196                                        for (String label : this.label2time2rp2value.keySet()) {
197                                                for (double time = startTime; time <= endTime; time = time + timeBinSize) {
198                                                        bw.write(";" + this.label2time2rp2value.get(label).get(time).get(rp.toString()));
199                                                }
200                                        }
201                                        bw.newLine();
202                                }                               
203                                break;
204                        case xyt:
205
206                                for (Id<Link> rp : this.network.getLinks().keySet()) {
207
208                                        for (double time = startTime; time <= endTime; time = time + timeBinSize) {
209
210                                                boolean writeThisLine = false;
211                                                String lineToWrite = rp.toString();
212
213                                                for (String label : this.label2time2rp2value.keySet()) {
214                                                        double value = this.label2time2rp2value.get(label).get(time).get(rp.toString());
215                                                        if (value > this.threshold) {
216                                                                writeThisLine = true;
217                                                        }
218                                                        lineToWrite = lineToWrite + ";" + value;
219                                                }
220
221                                                // only write the line if at least one value is larger than threshold
222                                                if (writeThisLine) {
223                                                        bw.write(lineToWrite);
224                                                        bw.newLine();
225                                                        lineCounter ++ ;
226                                                        if (lineCounter % 10000 == 0.) {
227                                                                log.info("# " + lineCounter);
228                                                        }
229                                                }
230                                        }
231                                }       
232                                break ;
233                        default:
234                                throw new RuntimeException("not implemented") ;
235                        }
236
237                        bw.close();
238                        log.info("Output written to " + outputFile);
239
240
241                } catch (IOException e1) {
242                        e1.printStackTrace();
243                }
244        }

246        private void writeFileReceiverPoint() {
247                int lineCounter = 0 ;
248
249                String outputFile = this.outputDirectory;
250
251                for (int i = 0; i < this.labels.length; i++) {
252                        outputFile = outputFile + this.labels[i] + "_";                         
253                }
254                outputFile = outputFile + "merged_" + this.outputFormat.toString() + ".csv.gz";
255
256                try ( BufferedWriter bw = IOUtils.getBufferedWriter(outputFile) ) {
257                        // so-called "try-with-resources". Kai
258
259                        log.info(" Writing merged file to " + outputFile + "...") ;
260
261                        // write headers
262                        switch( this.outputFormat ) {
263                        // yy should probably become different classes. kai
264                        case xyt1t2t3etc:
265                                bw.write("Receiver Point Id;x;y");
266                                for (String label : this.label2time2rp2value.keySet()) {
267                                        for (double time = startTime; time <= endTime; time = time + timeBinSize) {
268                                                bw.write(";" + label + "_" + Time.writeTime(time, Time.TIMEFORMAT_HHMMSS));
269                                        }
270                                }
271                                break;
272                        case xyt:
273                                bw.write("Receiver Point Id;x;y;time");
274                                for (String label : this.label2time2rp2value.keySet()) {
275                                        bw.write(";" + label);
276                                }
277                                break;
278                        default:
279                                throw new RuntimeException("not implemented") ;
280                        }
281
282                        bw.newLine();
283
284                        // fill table
285                        switch( this.outputFormat ) {
286                        case xyt1t2t3etc:       
287
288                                for (Id<ReceiverPoint> rp : this.rp2Coord.keySet()) {
289                                        bw.write(rp.toString() + ";" + rp2Coord.get(rp).getX() + ";" + rp2Coord.get(rp).getY());
290
291                                        for (String label : this.label2time2rp2value.keySet()) {
292                                                for (double time = startTime; time <= endTime; time = time + timeBinSize) {
293                                                        if (this.label2time2rp2value.get(label).get(time).get(rp.toString()) == null) {
294                                                                throw new RuntimeException("null!");
295                                                        }
296                                                        bw.write(";" + this.label2time2rp2value.get(label).get(time).get(rp.toString()));
297                                                }
298                                        }
299                                        bw.newLine();
300                                }                               
301                                break;
302                        case xyt:
303
304                                for (Id<ReceiverPoint> rp : this.rp2Coord.keySet()) {
305
306                                        for (double time = startTime; time <= endTime; time = time + timeBinSize) {
307
308                                                boolean writeThisLine = false;
309                                                String lineToWrite = rp.toString() + ";" + rp2Coord.get(rp).getX() + ";" + rp2Coord.get(rp).getY() + ";" + String.valueOf(time);
310
311                                                for (String label : this.label2time2rp2value.keySet()) {
312                                                        double value = this.label2time2rp2value.get(label).get(time).get(rp);
313                                                        if (value > this.threshold) {
314                                                                writeThisLine = true;
315                                                        }
316                                                        lineToWrite = lineToWrite + ";" + value;
317                                                }
318
319                                                // only write the line if at least one value is larger than threshold
320                                                if (writeThisLine) {
321                                                        bw.write(lineToWrite);
322                                                        bw.newLine();
323                                                        lineCounter ++ ;
324                                                        if (lineCounter % 10000 == 0.) {
325                                                                log.info("# " + lineCounter);
326                                                        }
327                                                }
328                                        }
329                                }       
330                                break ;
331                        default:
332                                throw new RuntimeException("not implemented") ;
333                        }
334
335                        bw.close();
336                        log.info("Output written to " + outputFile);
337
338
339                } catch (IOException e1) {
340                        e1.printStackTrace();
341                }
342        }

344        private void readReceiverPoints() {
345
346                BufferedReader br = IOUtils.getBufferedReader(this.receiverPointsFile);
347                String line;
348                try {
349                        line = br.readLine();
350                } catch (IOException e) {
351                        e.printStackTrace();
352                }
353
354                int lineCounter = 0;
355
356                log.info("Reading receiver points file...");
357
358                try {
359                        while( (line = br.readLine()) != null){
360
361                                if (lineCounter % 10000 == 0.) {
362                                        log.info("# " + lineCounter);
363                                }
364
365                                String[] columns = line.split(this.separator);
366                                Id<ReceiverPoint> rpId = null;
367                                double x = 0;
368                                double y = 0;
369
370                                for(int i = 0; i < columns.length; i++){
371
372                                        switch(i){
373                                        case 0: rpId = Id.create(columns[i], ReceiverPoint.class);
374                                        break;
375                                        case 1: x = Double.valueOf(columns[i]);
376                                        break;
377                                        case 2: y = Double.valueOf(columns[i]);
378                                        break;
379                                        default: throw new RuntimeException("More than three columns. Aborting...");
380                                        }
381                                }
382                                lineCounter++;
383                                this.rp2Coord.put(rpId, new Coord(x, y));
384                        }
385                } catch (NumberFormatException | IOException e) {
386                        e.printStackTrace();
387                }
388        }

390        private void readValues() {
391                for (int ll = 0; ll < this.labels.length; ll++) {
392
393                        Map<Double, Map<String, Double>> time2rp2value = new HashMap<>();
394
395                        String workingDirectory = this.workingDirectories[ll];
396                        String label = this.labels[ll];
397
398                        log.info("Reading " + label + "...");
399
400                        double totalValueSum = 0.;
401
402                        for (double time = startTime; time <= endTime; time = time + timeBinSize) {
403                                log.info("Reading time bin: " + time);
404
405                                double valueSumTimeBin = 0.;
406                                String fileName = workingDirectory + label + "_" + Double.toString(time) + ".csv";
407
408                                try ( BufferedReader br = IOUtils.getBufferedReader(fileName) ) {
409                                        // this will automagically use the *.gz version if a non-gzipped version does not exist. kai, jan'15
410
411                                        String line = br.readLine();
412
413                                        Map<String, Double> rp2value = new HashMap<>();
414                                        int lineCounter = 0;
415
416                                        log.info("Reading lines ");
417                                        while ((line = br.readLine()) != null) {
418
419                                                if (lineCounter % 10000 == 0.) {
420                                                        log.info("# " + lineCounter);
421                                                }
422
423                                                String[] columns = line.split(separator);
424                                                String rp = null;
425                                                Double value = null;
426                                                for (int column = 0; column < columns.length; column++) {
427                                                        if (column == 0) {
428                                                                rp = columns[column];
429                                                        } else if (column == 1) {
430                                                                value = Double.valueOf(columns[column]);
431                                                                valueSumTimeBin+=value;
432                                                        } else {
433                                                                // throw new RuntimeException("More than two columns. Aborting...");
434                                                        }
435                                                        rp2value.put(rp, value);
436
437                                                }
438                                                lineCounter++;
439                                                time2rp2value.put(time, rp2value);
440                                        }
441                                        log.info("sum of all values in this time bin: " + valueSumTimeBin);
442                                } catch (NumberFormatException | IOException e) {
443                                        e.printStackTrace();
444                                }
445                                
446                                totalValueSum+=valueSumTimeBin;
447                                log.info("total sum of all values: " + totalValueSum);
448                        }
449
450                        this.label2time2rp2value.put(label, time2rp2value);
451                }
452        }
