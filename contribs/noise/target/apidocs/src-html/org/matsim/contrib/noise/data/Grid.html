053/**
054 * Computes a grid of receiver points and provides some basic spatial functionality,
055 * e.g. the nearest receiver point for the coordinates of each 'considered' activity type. 
056 * 
057 * @author lkroeger, ikaddoura
058 *
059 */
060public class Grid {
061        
062        private static final Logger log = Logger.getLogger(Grid.class);
063                        
064        private final Scenario scenario;
065        private final NoiseConfigGroup noiseParams;
066                
067        private final Map<Id<Person>, List<Coord>> personId2consideredActivityCoords = new HashMap<Id<Person>, List<Coord>>();
068        
069        private final Set<Coord> consideredActivityCoordsForSpatialFunctionality = new HashSet <Coord>();
070        private final Set<Coord> consideredActivityCoordsForReceiverPointGrid = new HashSet <Coord>();
071        
072        private final Set<String> consideredActivitiesForSpatialFunctionality = new HashSet<String>();
073        private final Set<String> consideredActivitiesForReceiverPointGrid = new HashSet<String>();
074        
075        private double xCoordMin = Double.MAX_VALUE;
076        private double xCoordMax = Double.MIN_VALUE;
077        private double yCoordMin = Double.MAX_VALUE;
078        private double yCoordMax = Double.MIN_VALUE;
079        
080        private final Map<Coord,Id<ReceiverPoint>> activityCoord2receiverPointId = new HashMap<Coord, Id<ReceiverPoint>>();
081        private Map<Id<ReceiverPoint>, ReceiverPoint> receiverPoints;
082        
083        public Grid(Scenario scenario) {
084                this.scenario = scenario;       
085
086                if ((NoiseConfigGroup) this.scenario.getConfig().getModule("noise") == null) {
087                        throw new RuntimeException("Could not find a noise config group. "
088                                        + "Check if the custom module is loaded, e.g. 'ConfigUtils.loadConfig(configFile, new NoiseConfigGroup())'"
089                                        + " Aborting...");
090                }
091                
092                this.noiseParams = (NoiseConfigGroup) this.scenario.getConfig().getModule("noise");
093                
094                this.receiverPoints = new HashMap<Id<ReceiverPoint>, ReceiverPoint>();
095                
096                String[] consideredActTypesForDamagesArray = noiseParams.getConsideredActivitiesForDamageCalculationArray();
097                for (int i = 0; i < consideredActTypesForDamagesArray.length; i++) {
098                        this.consideredActivitiesForSpatialFunctionality.add(consideredActTypesForDamagesArray[i]);
099                }
100                
101                String[] consideredActTypesForReceiverPointGridArray = noiseParams.getConsideredActivitiesForReceiverPointGridArray();
102                for (int i = 0; i < consideredActTypesForReceiverPointGridArray.length; i++) {
103                        this.consideredActivitiesForReceiverPointGrid.add(consideredActTypesForReceiverPointGridArray[i]);
104                }
105
106//              this.noiseParams.checkGridParametersForConsistency();
107                initialize();
108        }
109
110        private void initialize() {
111                setActivityCoords();
112                
113                if (this.noiseParams.getReceiverPointsCSVFile() == null) {
114                        log.info("Creating receiver point square grid...");
115                        createGrid();
116                } else {
117                        log.info("Loading receiver points based on provided point coordinates in " + this.noiseParams.getReceiverPointsCSVFile());
118                        loadGrid();
119                }
120                
121                setActivityCoord2NearestReceiverPointId();
122                
123                // delete unnecessary information
124                this.consideredActivityCoordsForReceiverPointGrid.clear();
125                this.consideredActivityCoordsForSpatialFunctionality.clear();
126        }
127
128        private void setActivityCoords () {
129                StageActivityTypes stages = new StageActivityTypesImpl(PtConstants.TRANSIT_ACTIVITY_TYPE);
130                for (Person person: scenario.getPopulation().getPersons().values()) {
131
132                        for(Activity activity: TripStructureUtils.getActivities(person.getSelectedPlan(), stages)){
133                                if (this.consideredActivitiesForSpatialFunctionality.contains(activity.getType()) || consideredActivityPrefix(activity.getType(), this.consideredActivitiesForSpatialFunctionality)) {
134                                        List<Coord> activityCoordinates = personId2consideredActivityCoords.get(person.getId());//new ArrayList<Coord>();
135                                        
136                                        if (activityCoordinates == null) {
137                                                activityCoordinates = new ArrayList<>();
138                                                personId2consideredActivityCoords.put(person.getId(), activityCoordinates);
139                                        }
140                                        
141                                        activityCoordinates.add(activity.getCoord());
142                                        
143                                        consideredActivityCoordsForSpatialFunctionality.add(activity.getCoord());
144                                }
145                                
146                                if (this.consideredActivitiesForReceiverPointGrid.contains(activity.getType()) || consideredActivityPrefix(activity.getType(), consideredActivitiesForReceiverPointGrid)) {
147                                        consideredActivityCoordsForReceiverPointGrid.add(activity.getCoord());
148                                }
149                        }
150                }
151        }
152        
153        private boolean consideredActivityPrefix(String type, Set<String> list) {
154                for (String consideredActivity : list) {
155                        if (consideredActivity.endsWith("*")) {
156                                if (type.startsWith(consideredActivity.substring(0, consideredActivity.length() - 1))) {
157                                        return true;
158                                }
159                        }
160                }
161                return false;
162        }
163
164        private void loadGrid() {
165                                
166                String gridCSVFile = this.noiseParams.getReceiverPointsCSVFile();
167                
168                Map<Id<ReceiverPoint>, Coord> gridPoints = null;
169                try {
170                        gridPoints = readCSVFile(gridCSVFile, ",", 0, 1, 2);
171                } catch (IOException e) {
172                        e.printStackTrace();
173                }
174                
175                CoordinateTransformation ct = TransformationFactory.getCoordinateTransformation(this.noiseParams.getReceiverPointsCSVFileCoordinateSystem(), this.scenario.getConfig().global().getCoordinateSystem());
176                
177                for (Id<ReceiverPoint> id : gridPoints.keySet()) {
178                        
179                        Coord coord = gridPoints.get(id);
180                        Coord transformedCoord = ct.transform(coord);
181                        
182                        ReceiverPoint rp = new ReceiverPoint(id, transformedCoord);                     
183                        receiverPoints.put(id, rp);
184                }
185                
186                log.info("Total number of receiver points: " + receiverPoints.size());
187        }
188
189        
190        private void createGrid() {
191                
192                if (this.noiseParams.getReceiverPointsGridMinX() == 0. && this.noiseParams.getReceiverPointsGridMinY() == 0. && this.noiseParams.getReceiverPointsGridMaxX() == 0. && this.noiseParams.getReceiverPointsGridMaxY() == 0.) {
193                        
194                        log.info("Creating receiver points for the entire area between the minimum and maximium x and y activity coordinates of all activity locations.");
195                                                
196                        for (Coord coord : consideredActivityCoordsForReceiverPointGrid) {
197                                if (coord.getX() < xCoordMin) {
198                                        xCoordMin = coord.getX();
199                                }
200                                if (coord.getX() > xCoordMax) {
201                                        xCoordMax = coord.getX();
202                                }
203                                if (coord.getY() < yCoordMin) {
204                                        yCoordMin = coord.getY();
205                                }
206                                if (coord.getY() > yCoordMax) {
207                                        yCoordMax = coord.getY();
208                                }
209                        }
210                        
211                } else {
212                        
213                        xCoordMin = this.noiseParams.getReceiverPointsGridMinX();
214                        xCoordMax = this.noiseParams.getReceiverPointsGridMaxX();
215                        yCoordMin = this.noiseParams.getReceiverPointsGridMinY();
216                        yCoordMax = this.noiseParams.getReceiverPointsGridMaxY();
217                        
218                        log.info("Creating receiver points for the area between the coordinates (" + xCoordMin + "/" + yCoordMin + ") and (" + xCoordMax + "/" + yCoordMax + ").");                     
219                }
220                
221                createReceiverPoints();         
222        }
223        
224        private void createReceiverPoints() {
225
226                Counter counter = new Counter("create receiver point #");
227                
228                for (double y = yCoordMax + 100. ; y > yCoordMin - 100. - noiseParams.getReceiverPointGap() ; y = y - noiseParams.getReceiverPointGap()) {
229                
230                        for (double x = xCoordMin - 100. ; x < xCoordMax + 100. + noiseParams.getReceiverPointGap() ; x = x + noiseParams.getReceiverPointGap()) {
231                                
232                                Id<ReceiverPoint> id = Id.create(counter.getCounter(), ReceiverPoint.class);
233                                Coord coord = new Coord(x, y);
234                                ReceiverPoint rp = new ReceiverPoint(id, coord);
235                                receiverPoints.put(id, rp);
236                                counter.incCounter();
237                        }
238                }
239                counter.printCounter();
240                log.info("Total number of receiver points: " + receiverPoints.size());
241        }
242        
243        private void setActivityCoord2NearestReceiverPointId () {
244                double gap = noiseParams.getReceiverPointGap();
245                Counter counter = new Counter("fill quadtree #") ;
246                QuadTree<ReceiverPoint> qTree = new QuadTree<>(xCoordMin - 4*gap, yCoordMin - 4* gap, xCoordMax + 4*gap, yCoordMax + 4*gap);
247                for(ReceiverPoint p: receiverPoints.values()) {
248                        qTree.put(p.getCoord().getX(), p.getCoord().getY(), p);
249                        counter.incCounter();
250                }
251                counter.printCounter();
252                
253                counter = new Counter("compute nearest receiver-points #");
254                for (Coord coord : consideredActivityCoordsForSpatialFunctionality) {
255                        
256                        // TODO maybe add a check here so we consider only the rp in the 9 surrounding cells?
257                        ReceiverPoint rp = qTree.getClosest(coord.getX(), coord.getY());
258                        if(rp != null) {
259                                if(activityCoord2receiverPointId.put(coord, rp.getId()) != null){
260                                        log.warn("this must not happen");
261                                }
262                        }
263                        
264                        counter.incCounter();
265                }
266                counter.printCounter();
267        }

269        private Map<Id<ReceiverPoint>, Coord> readCSVFile(String file, String separator, int idColumn, int xCoordColumn, int yCoordColumn) throws IOException {
270                
271                Map<Id<ReceiverPoint>, Coord> id2Coord = new HashMap<>();
272                
273                BufferedReader br = IOUtils.getBufferedReader(file);
274                String line = null;
275                try {
276                        line = br.readLine();
277                } catch (IOException e) {
278                        e.printStackTrace();
279                }
280                
281                String[] headers = line.split(separator);
282                if (idColumn >= 0) log.info("id: " + headers[idColumn]);
283                log.info("xCoord: " + headers[xCoordColumn]);
284                log.info("yCoord: " + headers[yCoordColumn]);
285                
286                int lineCounter = 0;
287
288                while( (line = br.readLine()) != null) {
289
290                        if (lineCounter % 1000000 == 0.) {
291                                log.info("# " + lineCounter);
292                        }
293
294                        String[] columns = line.split(separator);
295                        if (line.isEmpty() || line.equals("") || columns.length != headers.length) {
296                                log.warn("Skipping line " + lineCounter + ". Line is empty or the columns are inconsistent with the headers: [" + line.toString() + "]");
297                        
298                        } else {
299                                String id = null;
300                                double x = 0;
301                                double y = 0;
302
303                                for (int column = 0; column < columns.length; column++){                                        
304                                        if (column == idColumn) {
305                                                id = columns[column];
306                                        } else if (column == xCoordColumn) {
307                                                x = Double.valueOf(columns[column]);
308                                        } else if (column == yCoordColumn) {
309                                                y = Double.valueOf(columns[column]);
310                                        }
311                                }
312                                if (idColumn >= 0) {
313                                        id2Coord.put(Id.create(id, ReceiverPoint.class), new Coord(x,y));
314                                } else {
315                                        id2Coord.put(Id.create(String.valueOf(lineCounter), ReceiverPoint.class), new Coord(x,y));
316                                }
317                                
318                                lineCounter++;
319                        }                       
320                }
321                
322                log.info("Done. Number of read lines: " + lineCounter);
323                
324                return id2Coord;
325        }
