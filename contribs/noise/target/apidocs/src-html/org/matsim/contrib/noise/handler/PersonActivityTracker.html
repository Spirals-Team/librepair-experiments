047/**
048 * 
049 * A handler which keeps track of each agent's activities throughout the day if the activity is of a certain predefined activity type (considered activity type).
050 * This handler is required for the calculation of noise damages.
051 * 
052 * @author ikaddoura
053 *
054 */
055
056public class PersonActivityTracker implements ActivityEndEventHandler , ActivityStartEventHandler {
057
058        private static final Logger log = Logger.getLogger(PersonActivityTracker.class);
059        
060        private final NoiseContext noiseContext;
061        
062        private final Set<String> consideredActivityTypes = new HashSet<String>();
063        private Map<Id<Person>, Integer> personId2currentActNr = new HashMap<Id<Person>, Integer>();
064                
065        public PersonActivityTracker(NoiseContext noiseContext) {
066                this.noiseContext = noiseContext;
067                
068                String[] consideredActTypesArray = noiseContext.getGrid().getGridParams().getConsideredActivitiesForDamageCalculationArray();
069                for (int i = 0; i < consideredActTypesArray.length; i++) {
070                        this.consideredActivityTypes.add(consideredActTypesArray[i]);
071                }
072                
073                if (this.consideredActivityTypes.size() == 0) {
074                        log.warn("Not considering any activity type for the noise damage computation. This event handler can be disabled.");
075                }       
076                
077                setFirstActivities();
078                
079        }
080
081        @Override
082        public void reset(int iteration) {
083                
084                this.personId2currentActNr.clear();
085                setFirstActivities();
086        }
087        
088        private int countWarn = 0;
089        private void setFirstActivities() {
090                
091                log.info("Receiving first activities from the selected plans...");
092                for (Person person : this.noiseContext.getScenario().getPopulation().getPersons().values()) {
093                        Plan plan = person.getSelectedPlan();
094
095                        if (!plan.getPlanElements().isEmpty() && plan.getPlanElements().get(0) instanceof Activity) {
096                                Activity firstActivity = (Activity) plan.getPlanElements().get(0);
097
098                                if (this.consideredActivityTypes.contains(firstActivity.getType()) || consideredActivityPrefix(firstActivity.getType(), this.consideredActivityTypes)) {
099                                        
100                                        Id<ReceiverPoint> rpId = noiseContext.getGrid().getActivityCoord2receiverPointId().get(noiseContext.getGrid().getPersonId2listOfConsideredActivityCoords().get(person.getId()).get(0));
101                                        this.personId2currentActNr.put(person.getId(), 0);
102                                        
103                                        PersonActivityInfo actInfo = new PersonActivityInfo();
104                                        actInfo.setStartTime(0.);
105                                        actInfo.setEndTime(30 * 3600.);
106                                        actInfo.setActivityType(firstActivity.getType());
107                                        
108                                        if (rpId == null) {
109                                                if (countWarn == 0) {
110                                                        log.warn("Please note that population units are only calculated for a predefined area. Thus, not all agents' activities are mapped to a receiver point. "
111                                                                        + "The border receiver points should not be used for analysis. "
112                                                                        + "This message is only given once.");
113                                                        countWarn++;
114                                                }
115                                        } else {
116                                                this.noiseContext.getReceiverPoints().get(rpId).addPersonActInfo(person.getId(), actInfo);
117                                        }
118                                }
119                        }
120                }
121                log.info("Receiving first activities from the selected plans... Done.");
122        }
123        
124        private boolean consideredActivityPrefix(String type, Set<String> consideredActivityTypes2) {
125                for (String consideredActivity : consideredActivityTypes2) {
126                        if (consideredActivity.endsWith("*")) {
127                                if (type.startsWith(consideredActivity.substring(0, consideredActivity.length() - 1))) {
128                                        return true;
129                                }
130                        }
131                }
132                return false;
133        }
134        
135        public void handleEvent(ActivityStartEvent event) {
136                                                
137                if (!(this.noiseContext.getScenario().getPopulation().getPersons().get(event.getPersonId()) != null)) {
138                } else {
139                
140                        if (!event.getActType().toString().equals(PtConstants.TRANSIT_ACTIVITY_TYPE)) {
141                                
142                                if (this.consideredActivityTypes.contains(event.getActType()) || consideredActivityPrefix(event.getActType(), this.consideredActivityTypes)) {
143//                                      Logger.getLogger(this.getClass()).warn( "event:" + event ) ;
144//                                      Logger.getLogger(this.getClass()).warn( "personId:" + event.getDriverId() ) ;
145//                                      Logger.getLogger(this.getClass()).warn( "map:" + this.personId2currentActNr ) ;
146//                                      Logger.getLogger(this.getClass()).warn( "nr:" + this.personId2currentActNr.get( event.getDriverId() ) ) ;
147                                        
148//                                      int newActNr = this.personId2currentActNr.get(event.getDriverId()) + 1;
149                                        // I had null pointer exceptions with the previous line.  Presumably, some agents were not initialized.  Thus
150                                        // replacing it with the following lines. kai, jul'15
151                                        int newActNr = 0 ;
152                                        if ( this.personId2currentActNr.get(event.getPersonId())!= null ) {
153                                                newActNr = this.personId2currentActNr.get(event.getPersonId()) + 1;
154                                        }
155                                        
156                                        this.personId2currentActNr.put(event.getPersonId(), newActNr);
157                                                                                
158                                        Coord coord = noiseContext.getGrid().getPersonId2listOfConsideredActivityCoords().get(event.getPersonId()).get(this.personId2currentActNr.get(event.getPersonId()));
159                                        Id<ReceiverPoint> rpId = noiseContext.getGrid().getActivityCoord2receiverPointId().get(coord);
160                                        
161                                        PersonActivityInfo actInfo = new PersonActivityInfo();
162                                        actInfo.setStartTime(event.getTime());
163                                        actInfo.setEndTime(30 * 3600.); // assuming this activity to be the last one in the agents' plan, will be overwritten if it is not the last activity
164                                        actInfo.setActivityType(event.getActType());
165                                        NoiseReceiverPoint nrp = null;
166                                        if ((nrp = this.noiseContext.getReceiverPoints().get(rpId) )!= null) {
167                                                nrp.addPersonActInfo(event.getPersonId(), actInfo);
168                                        }
169                                }
170                        }
171                }
172        }
173
174        @Override
175        public void handleEvent(ActivityEndEvent event) {
176                                
177                if (!(this.noiseContext.getScenario().getPopulation().getPersons().get(event.getPersonId()) != null)) {
178                } else {
179                        
180                        if (!event.getActType().toString().equals(PtConstants.TRANSIT_ACTIVITY_TYPE)) {
181                                
182                                if (this.consideredActivityTypes.contains(event.getActType()) || consideredActivityPrefix(event.getActType(), this.consideredActivityTypes)) {
183                                                                                
184                                        Coord coord = noiseContext.getGrid().getPersonId2listOfConsideredActivityCoords().get(event.getPersonId()).get(this.personId2currentActNr.get(event.getPersonId()));
185                                        Id<ReceiverPoint> rpId = noiseContext.getGrid().getActivityCoord2receiverPointId().get(coord);
186
187                                        if (this.noiseContext.getReceiverPoints().get(rpId) != null && this.noiseContext.getReceiverPoints().get(rpId).getPersonId2actInfos().get(event.getPersonId()) != null) {
188                                                for (PersonActivityInfo actInfo : this.noiseContext.getReceiverPoints().get(rpId).getPersonId2actInfos().get(event.getPersonId())) {
189                                                        if (actInfo.getEndTime() == 30 * 3600.) {
190                                                                actInfo.setEndTime(event.getTime());
191                                                        }
192                                                }       
193                                        }
194                                }
195                        } 
196                }               
197        }
