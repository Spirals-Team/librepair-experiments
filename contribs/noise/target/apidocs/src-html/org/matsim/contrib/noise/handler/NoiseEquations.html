038        public static double calculateMittelungspegelLm(int n, double p) {
039
040                //      Der Beurteilungspegel L_r ist bei Straßenverkehrsgeraeuschen gleich dem Mittelungspegel L_m.
041                //  L_r = L_m = 10 * lg(( 1 / T_r ) * (Integral)_T_r(10^(0,1*1(t))dt))
042                //      L_m,e ist der Mittelungspegel im Abstand von 25m von der Achse der Schallausbreitung
043                                
044                //      M ... traffic volume
045                //      p ... share of hdv in %
046                
047                if (p > 1) {
048                        throw new RuntimeException("p has to be <= 1. For an HGV share of 1%, p should be 0.01. Aborting...");
049                }
050                                
051                double pInPercentagePoints = p * 100.;  
052                double mittelungspegel = 37.3 + 10* Math.log10(n * (1 + (0.082 * pInPercentagePoints)));
053                
054                return mittelungspegel;
055        }
056        
057        public static double calculateGeschwindigkeitskorrekturDv (double vCar , double vHdv , double p) {
058                
059                //  v ... speed in kilometers per hour
060                //      p ... share of hdv, in percentage points
061                
062                if (p > 1) {
063                        throw new RuntimeException("p has to be <= 1. For an HGV share of 1%, p should be 0.01. Aborting...");
064                }
065                
066                double pInPercentagePoints = p * 100.;  
067
068                double lCar = calculateLCar(vCar);
069                double lHdv = calculateLHdv(vHdv);
070
071                double d = lHdv - lCar; 
072                double geschwindigkeitskorrekturDv = lCar - 37.3 + 10* Math.log10( (100.0 + (Math.pow(10.0, (0.1 * d)) - 1) * pInPercentagePoints ) / (100 + 8.23 * pInPercentagePoints));
073                
074                return geschwindigkeitskorrekturDv;
075        }

077        public static double calculateResultingNoiseImmission (Collection<Double> collection){
078                
079                double resultingNoiseImmission = 0.;
080                
081                if (collection.size() > 0) {
082                        double sumTmp = 0.;
083                        for (double noiseImmission : collection) {
084                                if (noiseImmission > 0.) {
085                                        sumTmp = sumTmp + (Math.pow(10, (0.1 * noiseImmission)));
086                                }
087                        }
088                        resultingNoiseImmission = 10 * Math.log10(sumTmp);
089                        if (resultingNoiseImmission < 0) {
090                                resultingNoiseImmission = 0.;
091                        }
092                }
093                return resultingNoiseImmission;
094        }
095        
096        public static double calculateLCar(double vCar) {
097                
098                double lCar = 27.7 + (10 * Math.log10(1.0 + Math.pow(0.02 * vCar, 3.0)));
099                return lCar;
100        }
101
102        public static double calculateLHdv(double vHdv) {
103                
104                double lHdv = 23.1 + (12.5 * Math.log10(vHdv));
105                return lHdv;
106        }

108        public static double calculateDistanceCorrection(double distance) {
109                double correctionTermDs = 15.8 - (10 * Math.log10(distance)) - (0.0142 * (Math.pow(distance, 0.9)));
110                return correctionTermDs;
111        }

113        public static double calculateAngleCorrection(double angle) {
114                double angleCorrection = 10 * Math.log10((angle) / (180));
115                return angleCorrection;
116        }

118        public static double calculateDamageCosts(double noiseImmission, double affectedAgentUnits, double timeInterval, double annualCostRate, double timeBinSize) {
119
120                String daytimeType = "NIGHT";
121                
122                if (timeInterval > 6 * 3600 && timeInterval <= 18 * 3600) {
123                        daytimeType = "DAY";
124                } else if (timeInterval > 18 * 3600 && timeInterval <= 22 * 3600) {
125                        daytimeType = "EVENING";
126                }
127                
128                double lautheitsgewicht = 0;
129                
130                if (daytimeType == "DAY"){
131                        if (noiseImmission < 50){
132                        } else {
133                                lautheitsgewicht = Math.pow(2.0 , 0.1 * (noiseImmission - 50));
134                        }
135                } else if (daytimeType == "EVENING"){
136                        if (noiseImmission < 45){
137                        } else {
138                                lautheitsgewicht = Math.pow(2.0 , 0.1 * (noiseImmission - 45));
139                        }
140                } else if (daytimeType == "NIGHT"){
141                        if (noiseImmission < 40){
142                        } else {
143                                lautheitsgewicht = Math.pow(2.0 , 0.1 * (noiseImmission - 40));
144                        }
145                        
146                } else {
147                        throw new RuntimeException("Neither day, evening nor night. Aborting...");
148                }
149                
150                double laermEinwohnerGleichwert = lautheitsgewicht * affectedAgentUnits;
151                double damageCosts = ( annualCostRate * laermEinwohnerGleichwert / 365. ) * ( timeBinSize / (24.0 * 3600) );
152                
153                return damageCosts;
154        }

156        public static double calculateShareOfResultingNoiseImmission (double noiseImmission , double resultingNoiseImmission){
157                double shareOfResultingNoiseImmission = Math.pow(((Math.pow(10, (0.05 * noiseImmission))) / (Math.pow(10, (0.05 * resultingNoiseImmission)))), 2);
158                return shareOfResultingNoiseImmission;  
159        }
160        
161        public static double calculateShare(int nVehicleType1, double lVehicleType1, int nVehicleType2, double lVehicleType2) { 
162                double share = ((nVehicleType1 * Math.pow(10, 0.1 * lVehicleType1)) / ((nVehicleType1 * Math.pow(10, 0.1 * lVehicleType1)) + (nVehicleType2 * Math.pow(10, 0.1 * lVehicleType2))));
163                return share;
164        }

166        public static double calculateResultingNoiseImmissionPlusOneVehicle(double finalImmission, double immissionIsolatedLink, double immissionIsolatedLinkPlusOneVehicle) {
167                double noiseImmissionPlusOneVehicle = Double.NEGATIVE_INFINITY;
168                if (finalImmission != 0.) {
169                        if (immissionIsolatedLink == 0.) {
170                                noiseImmissionPlusOneVehicle = 10 * Math.log10( Math.pow(10, (0.1 * immissionIsolatedLinkPlusOneVehicle)) + Math.pow(10, (0.1 * finalImmission)) );
171                        } else {
172                                noiseImmissionPlusOneVehicle = 10 * Math.log10( Math.pow(10, (0.1 * immissionIsolatedLinkPlusOneVehicle)) - Math.pow(10, (0.1 * immissionIsolatedLink)) + Math.pow(10, (0.1 * finalImmission)) );
173                        }
174                } else {
175                        noiseImmissionPlusOneVehicle = immissionIsolatedLinkPlusOneVehicle;
176                }
177                return noiseImmissionPlusOneVehicle;
178        }
