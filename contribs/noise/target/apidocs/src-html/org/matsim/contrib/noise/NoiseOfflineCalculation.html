020package org.matsim.contrib.noise;
021
022import org.apache.log4j.Logger;
023import org.matsim.api.core.v01.Scenario;
024import org.matsim.contrib.noise.data.NoiseContext;
025import org.matsim.contrib.noise.handler.LinkSpeedCalculation;
026import org.matsim.contrib.noise.handler.NoiseTimeTracker;
027import org.matsim.contrib.noise.handler.PersonActivityTracker;
028import org.matsim.core.api.experimental.events.EventsManager;
029import org.matsim.core.config.ConfigUtils;
030import org.matsim.core.controler.OutputDirectoryLogging;
031import org.matsim.core.events.EventsUtils;
032import org.matsim.core.events.MatsimEventsReader;
033import org.matsim.core.events.algorithms.EventWriterXML;
034
035import java.io.File;
036import java.io.IOException;
037
038/**
039 * (1) Computes noise emissions, immissions, person activities and damages based on a standard events file.
040 * (2) Optionally throws noise immission damage events for the causing agent and the affected agent.
041 *
042 * @author ikaddoura
043 *
044 */
045public class NoiseOfflineCalculation {
046        private static final Logger log = Logger.getLogger(NoiseOfflineCalculation.class);
047
048        private String outputDirectory;
049        private Scenario scenario;
050        
051        private NoiseContext noiseContext = null;
052        private NoiseTimeTracker timeTracker = null;
053
054        public NoiseOfflineCalculation(Scenario scenario, String analysisOutputDirectory) {
055                this.scenario = scenario;
056                this.outputDirectory = analysisOutputDirectory;
057                
058                if (!outputDirectory.endsWith("/")) {
059                        outputDirectory = outputDirectory + "/";
060                }
061                
062//              NoiseConfigGroup noiseParameters = (NoiseConfigGroup) scenario.getConfig().getModules().get(NoiseConfigGroup.GROUP_NAME);
063                NoiseConfigGroup noiseParameters = ConfigUtils.addOrGetModule(scenario.getConfig(), NoiseConfigGroup.class) ;
064                if (noiseParameters.isInternalizeNoiseDamages()) {
065                        log.warn("If you intend to internalize noise damages, please run the online noise computation."
066                                        + " This is an offline noise calculation which can only be used for analysis purposes.");
067                        noiseParameters.setInternalizeNoiseDamages(false);
068                }
069        }
070
071        public void run() {
072                        
073                String outputFilePath = outputDirectory + "noise-analysis_it." + this.scenario.getConfig().controler().getLastIteration() + "/";
074                File file = new File(outputFilePath);
075                file.mkdirs();
076                
077                OutputDirectoryLogging.catchLogEntries();
078                try {
079                        OutputDirectoryLogging.initLoggingWithOutputDirectory(outputFilePath);
080                } catch (IOException e1) {
081                        e1.printStackTrace();
082                }
083                                        
084                noiseContext = new NoiseContext(scenario);
085                NoiseWriter.writeReceiverPoints(noiseContext, outputFilePath + "/receiverPoints/", false);
086                                
087                EventsManager events = EventsUtils.createEventsManager();
088
089                timeTracker = new NoiseTimeTracker();
090                timeTracker.setNoiseContext(noiseContext);
091                timeTracker.setEvents(events);
092                timeTracker.setOutputFilePath(outputFilePath);
093                
094                events.addHandler(timeTracker);
095                
096                if (noiseContext.getNoiseParams().isUseActualSpeedLevel()) {
097                        LinkSpeedCalculation linkSpeedCalculator = new LinkSpeedCalculation();
098                        linkSpeedCalculator.setNoiseContext(noiseContext);
099                        events.addHandler(linkSpeedCalculator); 
100                }
101                
102                EventWriterXML eventWriter = null;
103                if (noiseContext.getNoiseParams().isThrowNoiseEventsAffected() || noiseContext.getNoiseParams().isThrowNoiseEventsCaused()) {
104                        eventWriter = new EventWriterXML(outputFilePath + this.scenario.getConfig().controler().getLastIteration() + ".events_NoiseImmission_Offline.xml.gz");
105                        events.addHandler(eventWriter); 
106                }
107                
108                if (noiseContext.getNoiseParams().isComputePopulationUnits()) {
109                        PersonActivityTracker actTracker = new PersonActivityTracker(noiseContext);
110                        events.addHandler(actTracker);
111                }
112                
113                log.info("Reading events file...");
114                MatsimEventsReader reader = new MatsimEventsReader(events);
115                reader.readFile(this.scenario.getConfig().controler().getOutputDirectory() + "/ITERS/it." + this.scenario.getConfig().controler().getLastIteration() + "/" + this.scenario.getConfig().controler().getLastIteration() + ".events.xml.gz");
116                log.info("Reading events file... Done.");
117                
118                timeTracker.computeFinalTimeIntervals();
119
120                if (noiseContext.getNoiseParams().isThrowNoiseEventsAffected() || noiseContext.getNoiseParams().isThrowNoiseEventsCaused()) {
121                        eventWriter.closeFile();
122                }
123                log.info("Noise calculation completed.");
124                
125        }
126
127        public NoiseTimeTracker getTimeTracker() {
128                return timeTracker;
129        }
130
131        public NoiseContext getNoiseContext() {
132                return noiseContext;
133        }
134        
135}
