023package org.matsim.contrib.noise.data;
024
025import java.util.ArrayList;
026import java.util.List;
027
028import org.matsim.api.core.v01.Id;
029import org.matsim.api.core.v01.network.Link;
030import org.matsim.vehicles.Vehicle;
031
032/**
033 * 
034 * Contains the relevant information for a single link, i.e. time-specific data such as number of vehicles, noise emission or damages.
035 * 
036 * @author ikaddoura
037 *
038 */
039public class NoiseLink {
040
041        private final Id<Link> id;      
042        private List<Id<Vehicle>> enteringVehicleIds = new ArrayList<Id<Vehicle>>();
043        private int carAgentsEntering = 0; // carAgents x scaleFactor = cars
044        private int hgvAgentsEntering = 0; // hgvAgents x scaleFactor = hgv
045        private int carAgentsLeaving = 0; 
046        private int hgvAgentsLeaving = 0;
047        private double travelTimeCar_Sec = 0.;
048        private double travelTimeHGV_Sec = 0.;
049        private double emission = 0.;
050        
051        private double emissionPlusOneCar = 0.;
052        private double emissionPlusOneHGV = 0.;
053        private double immissionPlusOneCar = 0.;
054        private double immissionPlusOneHGV = 0.;
055        
056        private double damageCost = 0.; 
057        private double averageDamageCostPerCar = 0.; 
058        private double averageDamageCostPerHgv = 0.; 
059        private double marginalDamageCostPerCar = 0.; 
060        private double marginalDamageCostPerHgv = 0.; 
061        
062        public NoiseLink(Id<Link> linkId) {
063                this.id = linkId;
064        }
065        public int getCarAgentsLeaving() {
066                return carAgentsLeaving;
067        }
068        public void setCarAgentsLeaving(int carAgentsLeaving) {
069                this.carAgentsLeaving = carAgentsLeaving;
070        }
071        public int getHgvAgentsLeaving() {
072                return hgvAgentsLeaving;
073        }
074        public void setHgvAgentsLeaving(int hgvAgentsLeaving) {
075                this.hgvAgentsLeaving = hgvAgentsLeaving;
076        }
077        public Id<Link> getId() {
078                return id;
079        }
080        public List<Id<Vehicle>> getEnteringVehicleIds() {
081                return enteringVehicleIds;
082        }
083        public void setEnteringVehicleIds(List<Id<Vehicle>> enteringVehicleIds) {
084                this.enteringVehicleIds = enteringVehicleIds;
085        }
086        public int getCarAgentsEntering() {
087                return carAgentsEntering;
088        }
089        public void setCarAgentsEntering(int carsEntering) {
090                this.carAgentsEntering = carsEntering;
091        }
092        public int getHgvAgentsEntering() {
093                return hgvAgentsEntering;
094        }
095        public void setHgvAgentsEntering(int hgvEntering) {
096                this.hgvAgentsEntering = hgvEntering;
097        }
098        public double getDamageCost() {
099                return damageCost;
100        }
101        public void setDamageCost(double damageCost) {
102                this.damageCost = damageCost;
103        }
104        public double getAverageDamageCostPerCar() {
105                return averageDamageCostPerCar;
106        }
107        public void setAverageDamageCostPerCar(double damageCostPerCar) {
108                this.averageDamageCostPerCar = damageCostPerCar;
109        }
110        public double getAverageDamageCostPerHgv() {
111                return averageDamageCostPerHgv;
112        }
113        public void setAverageDamageCostPerHgv(double damageCostPerHgv) {
114                this.averageDamageCostPerHgv = damageCostPerHgv;
115        }
116        public double getEmission() {
117                return emission;
118        }
119        public void setEmission(double emission) {
120                this.emission = emission;
121        }
122        public double getEmissionPlusOneCar() {
123                return emissionPlusOneCar;
124        }
125        public void setEmissionPlusOneCar(double emissionPlusOneCar) {
126                this.emissionPlusOneCar = emissionPlusOneCar;
127        }
128        public double getEmissionPlusOneHGV() {
129                return emissionPlusOneHGV;
130        }
131        public void setEmissionPlusOneHGV(double emissionPlusOneHGV) {
132                this.emissionPlusOneHGV = emissionPlusOneHGV;
133        }
134        public double getImmissionPlusOneCar() {
135                return immissionPlusOneCar;
136        }
137        public void setImmissionPlusOneCar(double immissionPlusOneCar) {
138                this.immissionPlusOneCar = immissionPlusOneCar;
139        }
140        public double getImmissionPlusOneHGV() {
141                return immissionPlusOneHGV;
142        }
143        public void setImmissionPlusOneHGV(double immissionPlusOneHGV) {
144                this.immissionPlusOneHGV = immissionPlusOneHGV;
145        }
146        public double getMarginalDamageCostPerCar() {
147                return marginalDamageCostPerCar;
148        }
149        public void setMarginalDamageCostPerCar(double marginalDamageCostPerCar) {
150                this.marginalDamageCostPerCar = marginalDamageCostPerCar;
151        }
152        public double getMarginalDamageCostPerHgv() {
153                return marginalDamageCostPerHgv;
154        }
155        public void setMarginalDamageCostPerHgv(double marginalDamageCostPerHgv) {
156                this.marginalDamageCostPerHgv = marginalDamageCostPerHgv;
157        }
158        public double getTravelTimeCar_sec() {
159                return travelTimeCar_Sec;
160        }
161        public void setTravelTimeCar_Sec(double travelTimeCar_Sec) {
162                this.travelTimeCar_Sec = travelTimeCar_Sec;
163        }
164        public double getTravelTimeHGV_sec() {
165                return travelTimeHGV_Sec;
166        }
167        public void setTravelTimeHGV_Sec(double travelTimeHGV_Sec) {
168                this.travelTimeHGV_Sec = travelTimeHGV_Sec;
169        }
170        @Override
171        public String toString() {
172                return "NoiseLink [id=" + id + ", enteringVehicleIds="
173                                + enteringVehicleIds + ", carAgents=" + carAgentsEntering
174                                + ", hgvAgents=" + hgvAgentsEntering + ", emission=" + emission
175                                + ", emissionPlusOneCar=" + emissionPlusOneCar
176                                + ", emissionPlusOneHGV=" + emissionPlusOneHGV
177                                + ", immissionPlusOneCar=" + immissionPlusOneCar
178                                + ", immissionPlusOneHGV=" + immissionPlusOneHGV
179                                + ", damageCost=" + damageCost + ", averageDamageCostPerCar="
180                                + averageDamageCostPerCar + ", averageDamageCostPerHgv="
181                                + averageDamageCostPerHgv + ", marginalDamageCostPerCar="
182                                + marginalDamageCostPerCar + ", marginalDamageCostPerHgv="
183                                + marginalDamageCostPerHgv + ", travelTimeCar="
184                                + travelTimeCar_Sec + ", travelTimeHGV="
185                                + travelTimeHGV_Sec + "]";
186        }
187        
188        
189}
