023package org.matsim.contrib.noise.examples;
024
025import org.matsim.api.core.v01.Scenario;
026import org.matsim.contrib.noise.NoiseConfigGroup;
027import org.matsim.contrib.noise.NoiseModule;
028import org.matsim.contrib.noise.data.NoiseAllocationApproach;
029import org.matsim.contrib.noise.utils.ProcessNoiseImmissions;
030import org.matsim.core.config.Config;
031import org.matsim.core.config.ConfigUtils;
032import org.matsim.core.controler.Controler;
033import org.matsim.core.controler.OutputDirectoryHierarchy;
034import org.matsim.core.scenario.ScenarioUtils;
035
036import java.io.IOException;
037
038/**
039 * An example how to use the noise module during a MATSim run (= online noise computation).
040 * 
041 * The {@link NoiseConfigGroup} specifies parameters that are relevant for the noise computation and if noise damages are internalized.
042 * For the internalization of noise damages, there is an average and a marginal cost pricing approach, see {@link NoiseAllocationApproach}.
043 * 
044 * For an example of how to compute noise levels, damages etc. for a final iteration (= offline noise computation), see {@link NoiseOfflineCalculationExample}. 
045 * 
046 * @author ikaddoura
047 *
048 */
049public class NoiseOnlineControlerExample {
050        
051        private static final String configFile = "./test/input/org/matsim/contrib/noise/config.xml";
052
053        public static void main(String[] args) throws IOException {
054                                
055                Config config = ConfigUtils.loadConfig(configFile, new NoiseConfigGroup());
056                Scenario scenario = ScenarioUtils.loadScenario(config);
057                
058                Controler controler = new Controler(scenario);
059                controler.addOverridingModule(new NoiseModule(scenario));
060                
061                controler.getConfig().controler().setOverwriteFileSetting(OutputDirectoryHierarchy.OverwriteFileSetting.deleteDirectoryIfExists);
062                controler.run();
063                                
064                // optionally process the output data
065                String workingDirectory = controler.getConfig().controler().getOutputDirectory() + "/ITERS/it." + controler.getConfig().controler().getLastIteration() + "/immissions/";
066                String receiverPointsFile = controler.getConfig().controler().getOutputDirectory() + "/receiverPoints/receiverPoints.csv";
067                NoiseConfigGroup noiseParameters = ConfigUtils.addOrGetModule(config, NoiseConfigGroup.class ) ;
068                ProcessNoiseImmissions processNoiseImmissions = new ProcessNoiseImmissions(workingDirectory, receiverPointsFile, noiseParameters.getReceiverPointGap());
069                processNoiseImmissions.run();   
070        }
071        
072}
