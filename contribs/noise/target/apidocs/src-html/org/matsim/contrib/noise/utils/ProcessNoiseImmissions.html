039import java.io.BufferedReader;
040import java.io.BufferedWriter;
041import java.io.FileWriter;
042import java.io.IOException;
043import java.util.HashMap;
044import java.util.Map;

046/**
047 * @author ikaddoura
048 *
049 */
050public class ProcessNoiseImmissions {
051        
052        private static final Logger log = Logger.getLogger(ProcessNoiseImmissions.class);
053
054        private final double startTime = 3600.;
055        private final double timeBinSize = 3600.;
056        private final double endTime = 24. * 3600.;
057        
058        private final double receiverPointGap;
059        private final String workingDirectory;
060        private String receiverPointsFile;
061                
062        private final String separator = ";";
063        private final String label = "immission";
064        
065        private final String outputPath;
066        
067        private BufferedWriter bw;
068        private Map<Double, Map<Id<ReceiverPoint>, Double>> time2rp2value = new HashMap<Double, Map<Id<ReceiverPoint>, Double>>();
069        
070        public ProcessNoiseImmissions(String workingDirectory, String receiverPointsFile, double receiverPointGap) {
071                this.workingDirectory = workingDirectory;
072                this.receiverPointsFile = receiverPointsFile;
073                this.receiverPointGap = receiverPointGap;
074                this.outputPath = workingDirectory;
075        }
076
077        public static void main(String[] args) {
078                String workingDirectory = "/Users/ihab/Documents/workspace/runs-svn/cn/output/cn/ITERS/it.100/immissions/";
079                String receiverPointsFile = "/Users/ihab/Documents/workspace/runs-svn/cn/output/cn/receiverPoints/receiverPoints.csv";
080
081                ProcessNoiseImmissions readNoiseFile = new ProcessNoiseImmissions(workingDirectory, receiverPointsFile, 100);
082                readNoiseFile.run();
083        }
084        
085        public void run() {
086                
087                String outputFile = outputPath + label + "_processed.csv";
088                
089                try {
090                        
091                        for (double time = startTime; time <= endTime; time = time + timeBinSize) {
092                                
093                                log.info("Reading time bin: " + time);
094
095                                String fileName = workingDirectory + label + "_" + Double.toString(time) + ".csv";
096                                BufferedReader br = IOUtils.getBufferedReader(fileName);
097                                
098                                String line = null;
099                                line = br.readLine();
100
101                                Map<Id<ReceiverPoint>, Double> rp2value = new HashMap<Id<ReceiverPoint>, Double>();
102                                int lineCounter = 0;
103                                log.info("Reading lines ");
104                                while ((line = br.readLine()) != null) {
105                                        
106                                        if (lineCounter % 10000 == 0.) {
107                                                log.info("# " + lineCounter);
108                                        }
109                                        
110                                        String[] columns = line.split(separator);
111                                        Id<ReceiverPoint> rp = null;
112                                        Double value = null;
113                                        for (int column = 0; column < columns.length; column++) {
114                                                if (column == 0) {
115                                                        rp = Id.create(columns[column], ReceiverPoint.class);
116                                                } else if (column == 1) {
117                                                        value = Double.valueOf(columns[column]); 
118                                                } else {
119//                                                      throw new RuntimeException("More than two columns. Aborting...");
120                                                }
121                                                rp2value.put(rp, value);
122                                                
123                                        }
124                                        lineCounter++;
125                                        time2rp2value.put(time, rp2value);
126                                }
127                        }
128                        
129                        BufferedReader br = IOUtils.getBufferedReader(this.receiverPointsFile);
130                        String line = br.readLine();
131                        
132                        Map<Id<ReceiverPoint>, Coord> rp2Coord = new HashMap<Id<ReceiverPoint>, Coord>();
133                        int lineCounter = 0;
134                        
135                        log.info("Reading receiver points file");
136                        
137                        while( (line = br.readLine()) != null){
138                                
139                                if (lineCounter % 10000 == 0.) {
140                                        log.info("# " + lineCounter);
141                                }
142                                
143                                String[] columns = line.split(this.separator);
144                                Id<ReceiverPoint> rpId = null;
145                                double x = 0;
146                                double y = 0;
147                                
148                                for(int i = 0; i < columns.length; i++){
149                                        
150                                        switch(i){
151                                        
152                                        case 0: rpId = Id.create(columns[i], ReceiverPoint.class);
153                                                        break;
154                                        case 1: x = Double.valueOf(columns[i]);
155                                                        break;
156                                        case 2: y = Double.valueOf(columns[i]);
157                                                        break;
158                                        default: throw new RuntimeException("More than three columns. Aborting...");
159                                        
160                                        }
161                                        
162                                }
163                                
164                                lineCounter++;
165                                rp2Coord.put(rpId, new Coord(x, y));
166                                
167                        }
168                        
169                        bw = new BufferedWriter(new FileWriter(outputFile));
170                        
171                        // write headers
172                        bw.write("Receiver Point Id;x;y");
173                        
174                        for (double time = startTime; time <= endTime; time = time + timeBinSize) {
175                                bw.write(";" + label + "_" + Time.writeTime(time, Time.TIMEFORMAT_HHMMSS));
176                        }
177                        
178                        bw.write(";Lden;L_6-9;L_16-19");
179
180                        bw.newLine();
181
182                        // fill table
183                        for (Id<ReceiverPoint> rp : time2rp2value.get(endTime).keySet()) {
184                                bw.write(rp.toString() + ";" + rp2Coord.get(rp).getX() + ";" + rp2Coord.get(rp).getY());
185                                
186                                for (double time = startTime; time <= endTime; time = time + timeBinSize) {
187                                        bw.write(";" + time2rp2value.get(time).get(rp));
188                                }
189                                
190                                // aggregate time intervals
191        
192                                double termDay = 0.;
193                                // day: 7-19
194                                for (double time = 8 * 3600.; time <= 19 * 3600.; time = time + timeBinSize) {
195                                        termDay = termDay + Math.pow(10, time2rp2value.get(time).get(rp) / 10);
196                                }
197                                
198                                double termEvening = 0.;
199                                // evening: 19-23
200                                for (double time = 20 * 3600.; time <= 23 * 3600.; time = time + timeBinSize) {
201                                        termEvening = termEvening + Math.pow(10, (time2rp2value.get(time).get(rp) + 5) / 10);
202                                }
203
204                                double termNight = 0.;
205                                // night: 23-7
206                                
207                                // nightA: 23-24
208                                for (double time = 24 * 3600.; time <= 24 * 3600.; time = time + timeBinSize) {
209                                        termNight = termNight + Math.pow(10, (time2rp2value.get(time).get(rp) + 10) / 10);
210                                }
211                                // nightB: 0-7
212                                for (double time = 1 * 3600.; time <= 7 * 3600.; time = time + timeBinSize) {
213                                        termNight = termNight + Math.pow(10, (time2rp2value.get(time).get(rp) + 10) / 10);
214                                }
215                        
216                                double Lden = 10 * Math.log10(1./24. * (termDay + termEvening + termNight));
217                                bw.write(";" + Lden);
218                                
219                                double term69 = 0.;
220                                for (double time = 7 * 3600.; time <= 9 * 3600.; time = time + timeBinSize) {
221                                        term69 = term69 + Math.pow(10, (time2rp2value.get(time).get(rp)) / 10);
222                                }
223                                double L_69 = 10 * Math.log10(1./3. * term69);
224                                bw.write(";" + L_69);
225                                
226                                double term1619 = 0.;
227                                for (double time = 17 * 3600.; time <= 19 * 3600.; time = time + timeBinSize) {
228                                        term1619 = term1619 + Math.pow(10, (time2rp2value.get(time).get(rp)) / 10);
229                                }
230                                double L_1619 = 10 * Math.log10(1./3. * term1619);
231                                bw.write(";" + L_1619);
232                                
233                                bw.newLine();
234                        }                               
235                        
236                        bw.close();
237                        log.info("Output written to " + outputFile);
238                }
239                
240                catch (IOException e1) {
241                        e1.printStackTrace();
242                }
243
244                String qGisProjectFile = "immission.qgs";
245                
246                QGisWriter writer = new QGisWriter(TransformationFactory.DHDN_GK4, workingDirectory);
247                        
248// ################################################################################################################################################
249                Envelope envelope = new Envelope(4568808,5803042,4622772,5844280);
250                writer.setEnvelope(envelope);
251                                
252                VectorLayer noiseLayer = new VectorLayer("noise", outputFile, QGisConstants.geometryType.Point, true);
253                noiseLayer.setDelimiter(";");
254                noiseLayer.setXField("x");
255                noiseLayer.setYField("y");
256
257        GraduatedSymbolRenderer renderer = RendererFactory.createNoiseRenderer(noiseLayer, this.receiverPointGap);
258                renderer.setRenderingAttribute("Lden");
259                
260                writer.addLayer(noiseLayer);
261                
262// ################################################################################################################################################
263                
264                writer.write(qGisProjectFile);
265                
266        }
267
