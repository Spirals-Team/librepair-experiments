020package org.matsim.contrib.noise.examples;
021
022import org.matsim.api.core.v01.Scenario;
023import org.matsim.contrib.noise.NoiseConfigGroup;
024import org.matsim.contrib.noise.NoiseOfflineCalculation;
025import org.matsim.contrib.noise.utils.MergeNoiseCSVFile;
026import org.matsim.contrib.noise.utils.ProcessNoiseImmissions;
027import org.matsim.core.config.Config;
028import org.matsim.core.config.ConfigUtils;
029import org.matsim.core.scenario.ScenarioUtils;
030
031/**
032 * 
033 * An example how to compute noise levels, damages etc. for a single iteration (= offline noise computation). 
034 * 
035 * @author ikaddoura
036 *
037 */
038public class NoiseOfflineCalculationExample {
039        
040        private static String runDirectory = "pathTo/RunDirectory/";
041        private static String outputDirectory = "pathTo/analysis-output-directory/";
042        private static int lastIteration = 100;
043                                
044        public static void main(String[] args) {
045        
046                Config config = ConfigUtils.createConfig(new NoiseConfigGroup());
047                config.network().setInputFile(runDirectory + "output_network.xml.gz");
048                config.plans().setInputFile(runDirectory + "output_plans.xml.gz");
049                config.controler().setOutputDirectory(runDirectory);
050                config.controler().setLastIteration(lastIteration);             
051                                                
052                // adjust the default noise parameters
053//              NoiseConfigGroup noiseParameters = (NoiseConfigGroup) config.getModules().get(NoiseConfigGroup.GROUP_NAME);
054                NoiseConfigGroup noiseParameters = ConfigUtils.addOrGetModule(config,NoiseConfigGroup.class) ;
055                noiseParameters.setReceiverPointGap(12345789.);
056                // ...
057                
058                Scenario scenario = ScenarioUtils.loadScenario(config);
059                
060                NoiseOfflineCalculation noiseCalculation = new NoiseOfflineCalculation(scenario, outputDirectory);
061                noiseCalculation.run();
062                
063                // some processing of the output data
064                if (!outputDirectory.endsWith("/")) outputDirectory = outputDirectory + "/";
065                
066                String outputFilePath = outputDirectory + "noise-analysis_it." + scenario.getConfig().controler().getLastIteration() + "/";
067                ProcessNoiseImmissions process = new ProcessNoiseImmissions(outputFilePath + "immissions/", outputFilePath + "receiverPoints/receiverPoints.csv", noiseParameters.getReceiverPointGap());
068                process.run();
069                                
070                final String[] labels = { "immission", "consideredAgentUnits" , "damages_receiverPoint" };
071                final String[] workingDirectories = { outputFilePath + "/immissions/" , outputFilePath + "/consideredAgentUnits/" , outputFilePath + "/damages_receiverPoint/" };
072
073                MergeNoiseCSVFile merger = new MergeNoiseCSVFile() ;
074                merger.setReceiverPointsFile(outputFilePath + "receiverPoints/receiverPoints.csv");
075                merger.setOutputDirectory(outputFilePath);
076                merger.setTimeBinSize(noiseParameters.getTimeBinSizeNoiseComputation());
077                merger.setWorkingDirectory(workingDirectories);
078                merger.setLabel(labels);
079                merger.run();
080        }
081}
