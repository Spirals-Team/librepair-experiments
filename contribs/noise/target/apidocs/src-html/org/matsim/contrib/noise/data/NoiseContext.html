126        private void checkConsistency() {
127                
128                if (this.scenario.getPopulation() == null || this.scenario.getPopulation().getPersons().isEmpty()) {
129                        throw new RuntimeException("The noise computation requires a loaded population to identify passenger cars."
130                                        + " Please provide a scenario which contains a loaded population. Aborting...");
131                }
132                List<String> consideredActivitiesForDamagesList = new ArrayList<String>();
133                List<String> consideredActivitiesForReceiverPointGridList = new ArrayList<String>();
134
135                for (int i = 0; i < this.grid.getGridParams().getConsideredActivitiesForDamageCalculationArray().length; i++) {
136                        consideredActivitiesForDamagesList.add(this.grid.getGridParams().getConsideredActivitiesForDamageCalculationArray()[i]);
137                }
138                
139                for (int i = 0; i < this.grid.getGridParams().getConsideredActivitiesForReceiverPointGridArray().length; i++) {
140                        consideredActivitiesForReceiverPointGridList.add(this.grid.getGridParams().getConsideredActivitiesForReceiverPointGridArray()[i]);
141                }
142                
143                if (this.noiseParams.isComputeNoiseDamages()) {
144                        
145                        if (consideredActivitiesForDamagesList.size() == 0) {
146                                log.warn("Not considering any activity type for the noise damage computation."
147                                                + "The computation of noise damages should be disabled.");
148                        }
149                        
150                        if (this.grid.getGridParams().getReceiverPointsGridMaxX() != 0.
151                                        || this.grid.getGridParams().getReceiverPointsGridMinX() != 0.
152                                        || this.grid.getGridParams().getReceiverPointsGridMaxY() != 0.
153                                        || this.grid.getGridParams().getReceiverPointsGridMinY() != 0.) {
154                                log.warn("In order to keep track of the agent activities, the grid of receiver points should not be limited to a set of predefined coordinates."
155                                                + "For a grid covering all activity locations, set the minimum and maximum x/y parameters to 0.0. "
156                                                + "There will be more agents mapped to the receiver points at the edges. Only the inner receiver points should be used for analysis.");
157                        }
158                                                
159                        if (this.grid.getGridParams().getReceiverPointsGridMinX() == 0. && this.grid.getGridParams().getReceiverPointsGridMinY() == 0. && this.grid.getGridParams().getReceiverPointsGridMaxX() == 0. && this.grid.getGridParams().getReceiverPointsGridMaxY() == 0.) {
160                                for (String type : consideredActivitiesForDamagesList) {
161                                        if (!consideredActivitiesForReceiverPointGridList.contains(type)) {
162                                                throw new RuntimeException("An activity type which is considered for the damage calculation (" + type
163                                                                + ") should also be considered for the minimum and maximum coordinates of the receiver point grid area. Aborting...");
164                                        }
165                                }
166                        }
167                }
168        }

170        private void setRelevantLinkInfo() {
171
172                Counter cnt = new Counter("set relevant link-info # ");
173                // go through all rp's and throw them away. We need noise-rps from here on.
174                ConcurrentLinkedQueue<ReceiverPoint> rps = new ConcurrentLinkedQueue<>(this.grid.getAndClearReceiverPoints().values());
175                ReceiverPoint rp = null;
176                while((rp = rps.poll()) != null){
177                        
178                        NoiseReceiverPoint nrp = new NoiseReceiverPoint(rp.getId(), rp.getCoord());
179
180                        // get the zone grid cell around the receiver point
181                        Set<Id<Link>> potentialLinks = new HashSet<>();
182                        Tuple<Integer,Integer>[] zoneTuples = getZoneTuplesForLinks(nrp.getCoord());
183                        for(Tuple<Integer, Integer> key: zoneTuples) {
184                                List<Id<Link>> links = zoneTuple2listOfLinkIds.get(key);
185                                if(links != null) {
186                                        potentialLinks.addAll(links);
187                                }
188                        }
189
190                        // go through these potential relevant link Ids
191                        Set<Id<Link>> relevantLinkIds = new HashSet<>();
192                        for (Id<Link> linkId : potentialLinks){
193                                if (!(relevantLinkIds.contains(linkId))) {
194                                        Link candidateLink = scenario.getNetwork().getLinks().get(linkId);
195                                        //maybe replace the disctance-calculation to remove dupolicated code. Not absolute sure, since tests are failing, when method is replaced.
196                                        //double distance = CoordUtils.distancePointLinesegment(candidateLink.getFromNode().getCoord(), candidateLink.getToNode().getCoord(), nrp.getCoord());
197                                        double distance = calcDistance(nrp, candidateLink);
198                                        
199                                        if (distance < noiseParams.getRelevantRadius()){
200                                                
201                                                relevantLinkIds.add(linkId);
202                                                // wouldn't it be good to check distance < minDistance here? DR20180215
203                                                if (distance == 0) {
204                                                        double minimumDistance = 5.;
205                                                        distance = minimumDistance;
206                                                        log.warn("Distance between " + linkId + " and " + rp.getId() + " is 0. The calculation of the correction term Ds requires a distance > 0. Therefore, setting the distance to a minimum value of " + minimumDistance + ".");
207                                                }
208                                                double correctionTermDs = NoiseEquations.calculateDistanceCorrection(distance);
209                                                double correctionTermAngle = calculateAngleImmissionCorrection(nrp.getCoord(), scenario.getNetwork().getLinks().get(linkId));
210                                                
211                                                nrp.setLinkId2distanceCorrection(linkId, correctionTermDs);
212                                                nrp.setLinkId2angleCorrection(linkId, correctionTermAngle);
213                                        }
214                                }
215                        }
216                        
217                        this.noiseReceiverPoints.put(nrp.getId(), nrp);
218                        cnt.incCounter();
219                }
220                cnt.printCounter();
221        }

223        /**
224         * @param nrp
225         * @param candidateLink
226         * @return
227         */
228        private double calcDistance(NoiseReceiverPoint nrp, Link candidateLink) {
229                double pointCoordX = nrp.getCoord().getX();
230                double pointCoordY = nrp.getCoord().getY();
231                double fromCoordX = candidateLink.getFromNode().getCoord().getX();
232                double fromCoordY = candidateLink.getFromNode().getCoord().getY();
233                double toCoordX = candidateLink.getToNode().getCoord().getX();
234                double toCoordY = candidateLink.getToNode().getCoord().getY();
235        
236                double lotPointX = 0.;
237                double lotPointY = 0.;
238        
239                double vectorX = toCoordX - fromCoordX;
240                if (vectorX == 0.) {
241                        vectorX = 0.00000001;
242                        // dividing by zero is not possible
243                }
244                
245                double vectorY = toCoordY - fromCoordY;
246                double vector = vectorY/vectorX;
247                if (vector == 0.) {
248                        vector = 0.00000001;
249                        // dividing by zero is not possible
250                }
251        
252                double vector2 = (-1) * (1/vector);
253                double yAbschnitt = fromCoordY - (fromCoordX * vector);
254                double yAbschnittOriginal = fromCoordY - (fromCoordX * vector);
255        
256                double yAbschnitt2 = pointCoordY - (pointCoordX * vector2);
257        
258                double xValue = 0.;
259                double yValue = 0.;
260        
261                if (yAbschnitt<yAbschnitt2) {
262                        yAbschnitt2 = yAbschnitt2 - yAbschnitt;
263                        yAbschnitt = 0;
264                        xValue = yAbschnitt2 / (vector - vector2);
265                        yValue = yAbschnittOriginal + (xValue*vector);
266                } else if(yAbschnitt2<yAbschnitt) {
267                        yAbschnitt = yAbschnitt - yAbschnitt2;
268                        yAbschnitt2 = 0;
269                        xValue = yAbschnitt / (vector2 - vector);
270                        yValue = yAbschnittOriginal + (xValue*vector);
271                }
272        
273                lotPointX = xValue;
274                lotPointY = yValue;
275                double distance = 0.;
276                
277                if(((xValue>fromCoordX)&&(xValue<toCoordX))||((xValue>toCoordX)&&(xValue<fromCoordX))||((yValue>fromCoordY)&&(yValue<toCoordY))||((yValue>toCoordY)&&(yValue<fromCoordY))) {
278                        // no edge solution
279                        distance = Math.sqrt((Math.pow(lotPointX-pointCoordX, 2))+(Math.pow(lotPointY-pointCoordY, 2)));
280                } else {
281                        // edge solution (Randloesung)
282                        double distanceToFromNode = Math.sqrt((Math.pow(fromCoordX-pointCoordX, 2))+(Math.pow(fromCoordY-pointCoordY, 2)));
283                        double distanceToToNode = Math.sqrt((Math.pow(toCoordX-pointCoordX, 2))+(Math.pow(toCoordY-pointCoordY, 2)));
284                        if (distanceToFromNode > distanceToToNode) {
285                                distance = distanceToToNode;
286                        } else {
287                                distance = distanceToFromNode;
288                        }
289                }
290                return distance;
291        }

297        private Tuple<Integer, Integer>[] getZoneTuplesForLinks(Coord coord) {
298                Tuple<Integer, Integer> zoneTuple = getZoneTupleForLinks(coord);
299                int x = zoneTuple.getFirst();
300                int y = zoneTuple.getSecond();
301                return new Tuple[] {
302                                zoneTuple,
303                                new Tuple<Integer, Integer>(x-1, y-1),
304                                new Tuple<Integer, Integer>(x, y-1),
305                                new Tuple<Integer, Integer>(x+1, y-1),
306                                new Tuple<Integer, Integer>(x-1, y),
307                                new Tuple<Integer, Integer>(x+1, y),
308                                new Tuple<Integer, Integer>(x-1, y+1),
309                                new Tuple<Integer, Integer>(x, y+1),
310                                new Tuple<Integer, Integer>(x+1, y+1)
311                };
312        }

323        private void setLinksToZones() {
324                Counter cnt = new Counter("set links to zones #");
325                for (Link link : scenario.getNetwork().getLinks().values()){
326                        
327                        // split up the link into link segments with the following length
328                        double partLength = 0.25 * noiseParams.getRelevantRadius();
329                        int parts = (int) (link.getLength()/partLength);
330
331                        double fromX = link.getFromNode().getCoord().getX();
332                        double fromY = link.getFromNode().getCoord().getY();
333                        double toX = link.getToNode().getCoord().getX();
334                        double toY = link.getToNode().getCoord().getY();
335                        double vectorX = toX - fromX;
336                        double vectorY = toY - fromY;
337                        
338                        // collect the coordinates of this link
339                        Set<Coord> coords = new HashSet<Coord>();
340                        coords.add(link.getFromNode().getCoord());
341                        coords.add(link.getToNode().getCoord());
342                        for (int i = 1 ; i<parts ; i++) {
343                                double x = fromX + (i*((1./(parts))*vectorX));
344                                double y = fromY + (i*((1./(parts))*vectorY));
345                                Coord  coordTmp = new Coord(x, y);
346                                coords.add(coordTmp);
347                        }
348                        
349                        // get zone grid cells for these coordinates
350                        Set<Tuple<Integer,Integer>> relevantTuples = new HashSet<Tuple<Integer,Integer>>();
351                        for (Coord coord : coords) {
352                                relevantTuples.add(getZoneTupleForLinks(coord));
353                        }
354                        
355                        // go through these zone grid cells and save the link Id                        
356                        for(Tuple<Integer,Integer> tuple : relevantTuples) {
357                                List<Id<Link>> linkIds = zoneTuple2listOfLinkIds.get(tuple);
358                                if(linkIds == null) {
359                                        linkIds = new ArrayList<>();
360                                        zoneTuple2listOfLinkIds.put(tuple, linkIds);
361                                }
362                                linkIds.add(link.getId());
363                        }
364                        cnt.incCounter();
365                }
366                cnt.printCounter();
367        }

369        private Tuple<Integer,Integer> getZoneTupleForLinks(Coord coord) {
370                 
371                double xCoord = coord.getX();
372                double yCoord = coord.getY();
373                
374                int xDirection = (int) ((xCoord - xCoordMinLinkNode) / (noiseParams.getRelevantRadius() / 1.)); 
375                int yDirection = (int) ((yCoordMaxLinkNode - yCoord) / noiseParams.getRelevantRadius() / 1.);
376                
377                Tuple<Integer,Integer> zoneDefinition = new Tuple<Integer, Integer>(xDirection, yDirection);
378                return zoneDefinition;
379        }

381        private double calculateAngleImmissionCorrection(Coord receiverPointCoord, Link link) {
382
383                double angle = 0;
384                
385                double pointCoordX = receiverPointCoord.getX();
386                double pointCoordY = receiverPointCoord.getY();
387                
388                double fromCoordX = link.getFromNode().getCoord().getX();
389                double fromCoordY = link.getFromNode().getCoord().getY();
390                double toCoordX = link.getToNode().getCoord().getX();
391                double toCoordY = link.getToNode().getCoord().getY();
392
393                if (pointCoordX == fromCoordX && pointCoordY == fromCoordY) {
394                        // receiver point is situated on the link (fromNode)
395                        // assume a maximum angle immission correction for this case
396                        angle = 0;
397                        
398                } else if (pointCoordX == toCoordX && pointCoordY == toCoordY) {
399                        // receiver point is situated on the link (toNode)
400                        // assume a zero angle immission correction for this case
401                        angle = 180;            
402                        
403                } else {
404                        // all other cases
405                        double sc = (fromCoordX - pointCoordX) * (toCoordX - pointCoordX) + (fromCoordY - pointCoordY) * (toCoordY - pointCoordY);
406                        double cosAngle = sc / (
407                                        Math.sqrt(
408                                                        Math.pow(fromCoordX - pointCoordX, 2) + Math.pow(fromCoordY - pointCoordY, 2)
409                                                        )
410                                                        *
411                                        Math.sqrt(
412                                                        Math.pow(toCoordX - pointCoordX, 2) + Math.pow(toCoordY - pointCoordY, 2)
413                                                        )
414                                        );
415                        angle = Math.toDegrees(Math.acos(cosAngle));
416
417                        if (sc > 0) {
418                                // spitzer winkel
419                                                
420                                if (angle > 90) {
421                                        angle = 180 - angle;
422                                }
423                                
424                        } else if (sc < 0) {
425                                // stumpfer winkel
426                                
427                                if (angle < 90) {
428                                        angle = 180 - angle;
429                                }
430                                
431                        } else {
432                                angle = 0.;
433                        }
434                }
435                        
436                // since zero is not defined
437                if (angle == 0.) {
438                        // zero degrees is not defined
439                        angle = 0.0000000001;
440                }
441                                        
442//              System.out.println(receiverPointCoord + " // " + link.getId() + "(" + link.getFromNode().getCoord() + "-->" + link.getToNode().getCoord() + " // " + angle);
443                double immissionCorrection = NoiseEquations.calculateAngleCorrection(angle);
444                return immissionCorrection;
445        }
