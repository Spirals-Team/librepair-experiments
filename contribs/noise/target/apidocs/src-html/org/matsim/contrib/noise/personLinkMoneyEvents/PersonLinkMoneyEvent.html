020package org.matsim.contrib.noise.personLinkMoneyEvents;
021
022import java.util.Map;
023
024import org.matsim.api.core.v01.Id;
025import org.matsim.api.core.v01.events.Event;
026import org.matsim.api.core.v01.events.PersonMoneyEvent;
027import org.matsim.api.core.v01.network.Link;
028import org.matsim.api.core.v01.population.Person;
029import org.matsim.core.api.internal.HasPersonId;
030
031/**
032 * The idea is to have a standard events format to be used by various external cost pricing approaches.
033 * 
034 * The normal {@link PersonMoneyEvent} is not sufficient since we need refer the toll to a specific link
035 * and the link information may not be available when the money event is thrown. Also, the routing relevant time
036 * may be different from the time when the event is thrown.
037 * 
038 * @author ikaddoura
039*/
040
041public class PersonLinkMoneyEvent extends Event implements HasPersonId{
042
043        public static final String ATTRIBUTE_AMOUNT = "amount";
044
045        public static final String EVENT_TYPE = "personLinkMoney";
046        public static final String ATTRIBUTE_PERSON = "person";
047        public static final String ATTRIBUTE_LINK = "link";
048        public static final String ATTRIBUTE_RELEVANT_TIME = "relevantTime";
049        public static final String ATTRIBUTE_DESCRIPTION = "description";
050        
051        private final Id<Person> personId;
052        private final Id<Link> linkId;
053        private final double amount;
054        private final double relevantTime;
055        private final String description;
056        
057        public PersonLinkMoneyEvent(final double time, final Id<Person> agentId, final Id<Link> linkId, final double amount, final double relevantTime, final String description) {
058                super(time);
059                this.personId = agentId;
060                this.linkId = linkId;
061                this.amount = amount;
062                this.relevantTime = relevantTime;
063                this.description = description;
064        }
065
066        public Id<Person> getPersonId() {
067                return this.personId;
068        }
069        
070        public double getAmount() {
071                return this.amount;
072        }
073
074        public Id<Link> getLinkId() {
075                return linkId;
076        }
077
078        public double getRelevantTime() {
079                return relevantTime;
080        }
081
082        public String getDescription() {
083                return description;
084        }
085
086        @Override
087        public String getEventType() {
088                return EVENT_TYPE;
089        }
090        
091        @Override
092        public Map<String, String> getAttributes() {
093                Map<String, String> attr = super.getAttributes();
094                attr.put(ATTRIBUTE_AMOUNT, Double.toString(this.amount));
095                attr.put(ATTRIBUTE_PERSON, this.personId.toString());
096                attr.put(ATTRIBUTE_LINK, this.linkId.toString());
097                attr.put(ATTRIBUTE_RELEVANT_TIME, Double.toString(this.relevantTime));
098                attr.put(ATTRIBUTE_DESCRIPTION, this.description);
099                return attr;
100        }
101
102}
