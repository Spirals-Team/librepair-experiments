041/**
042 * This calculates transport-times and transport-costs to cover the distance from one location to another. 
043 * It calculates these values based on a {@link Network} to serve as {@link VehicleRoutingTransportCosts} in the {@link VehicleRoutingProblem}.
044 * 
045 * <p>It can be used with multiple threads. Note that each thread gets its own leastCostPathCalculator. It is created only once and cached afterwards. Thus
046 * it requires a threadSafe leastCostPathCalculatorFactory (the calculator itself does not need to be thread-safe). 
047 * 
048 * <p>If the client of this class requests cost-information to get from from-location to to-location at a certain time and with a certain vehicle, it looks up whether there
049 * is already an entry in the cache. If so, it returns the cached value, if not it calculates new values with a leastCostPathCalculator defined in here. It looks up the cached values
050 * according to the {@link TransportDataKey}.
051 * 
052 * <p>Two TransportDataKeys are equal if the following data are equal</br>
053 * - from-locations</br>
054 * - to-locations</br>
055 * - time and</br>
056 * - vehicleTypes
057 * 
058 * <p>Keep in mind that if you have many locations, small time-bins and many vehicleTypes, calculations get very time- and memory-consuming.
059 * 
060 * @author stefan schröder
061 *
062 */
063public class NetworkBasedTransportCosts implements VehicleRoutingTransportCosts{

112        /**
113         * The key to cache {@link TransportData}. 
114         * 
115         * <p>Two TransportDataKeys are equal if the following data are equal</br>
116         * - from-locations</br>
117         * - to-locations</br>
118         * - time and</br>
119         * - vehicleTypes
120         *  
121         * <p>The time-value is usually a representative value for a certain time-bin. If the time-bin's width is 1, it calculates and caches each and every transport-data which might be 
122         * a) very time-consuming and b) very memory-consuming.
123         * 
124         * @author stefan schröder
125         *
126         */
127        static class TransportDataKey {
128                private final String from;
129                private final String to;
130                private final double time;
131                private final String vehicleType;
132
133                public TransportDataKey(String from, String to, double time, String vehicleType) {
134                        super();
135                        this.from = from;
136                        this.to = to;
137                        this.time = time;
138                        this.vehicleType = vehicleType;
139                }
140                public String getFrom() {
141                        return from;
142                }
143                public String getTo() {
144                        return to;
145                }
146                public double getTime() {
147                        return time;
148                }
149                
150                public String getVehicleType() {
151                        return vehicleType;
152                }
153                @Override
154                public int hashCode() {
155                        final int prime = 31;
156                        int result = 1;
157                        result = prime * result + ((from == null) ? 0 : from.hashCode());
158                        long temp;
159                        temp = Double.doubleToLongBits(time);
160                        result = prime * result + (int) (temp ^ (temp >>> 32));
161                        result = prime * result + ((to == null) ? 0 : to.hashCode());
162                        result = prime * result
163                                        + ((vehicleType == null) ? 0 : vehicleType.hashCode());
164                        return result;
165                }
166                @Override
167                public boolean equals(Object obj) {
168                        if (this == obj)
169                                return true;
170                        if (obj == null)
171                                return false;
172                        if (getClass() != obj.getClass())
173                                return false;
174                        TransportDataKey other = (TransportDataKey) obj;
175                        if (from == null) {
176                                if (other.from != null)
177                                        return false;
178                        } else if (!from.equals(other.from))
179                                return false;
180                        if (Double.doubleToLongBits(time) != Double
181                                        .doubleToLongBits(other.time))
182                                return false;
183                        if (to == null) {
184                                if (other.to != null)
185                                        return false;
186                        } else if (!to.equals(other.to))
187                                return false;
188                        if (vehicleType == null) {
189                                if (other.vehicleType != null)
190                                        return false;
191                        } else if (!vehicleType.equals(other.vehicleType))
192                                return false;
193                        return true;
194                }
195

225        static class BaseVehicleTransportCosts implements TravelDisutility{
226                
227                private TravelTime travelTime;
228                
229                private Map<String,VehicleTypeVarCosts> typeSpecificCosts = new HashMap<String, NetworkBasedTransportCosts.VehicleTypeVarCosts>();
230                
231                /**
232                 * Constructs travelDisutility according to the builder.
233                 *
234                 */
235                private BaseVehicleTransportCosts(Map<String, VehicleTypeVarCosts> typeSpecificCosts, TravelTime travelTime){
236                        this.travelTime = travelTime;
237                        this.typeSpecificCosts = typeSpecificCosts;
238                }
239
240                @Override
241                public double getLinkTravelDisutility(Link link, double time, Person person, org.matsim.vehicles.Vehicle vehicle) {
242                        VehicleTypeVarCosts typeCosts = typeSpecificCosts.get(vehicle.getType().getId().toString());
243                        if(typeCosts == null) throw new IllegalStateException("type specific costs for " + vehicle.getType().getId().toString() + " are missing.");
244                        double tt = travelTime.getLinkTravelTime(link, time, person, vehicle);
245                        return typeCosts.perMeter*link.getLength() + typeCosts.perSecond*tt;
246                }
247
248                @Override
249                public double getLinkMinimumTravelDisutility(Link link) {
250                        double minDisutility = Double.MAX_VALUE;
251                        double free_tt = link.getLength()/link.getFreespeed();
252                        for(VehicleTypeVarCosts c : typeSpecificCosts.values()){
253                                double disu = c.perMeter*link.getLength() + c.perSecond*free_tt;
254                                if(disu < minDisutility) minDisutility=disu;
255                        }
256                        return minDisutility;
257                }

532        /**
533         * Gets the transport-time.
534         * 
535         * <p>If <code>fromId.equals(toId)</code> it returns 0.0. Otherwise, it looks up in the cache whether the transport-time has already been computed
536         * (see {@link TransportDataKey}, {@link TransportData}). If so, it returns
537         * the cached travel-time. If not, it computes and caches new values with the leastCostPathCalc defined in here.  
538         * 
539         * @Throws {@link IllegalStateException} if vehicle is null
540         */
541        @Override
542        public double getTransportTime(Location fromId, Location toId, double departureTime, Driver driver, Vehicle vehicle) {
543                if(fromId.equals(toId)){
544                        return 0.0;
545                }
546                if(vehicle == null) {
547                        vehicle = getDefaultVehicle(fromId);
548                }
549                String typeId = vehicle.getType().getTypeId();
550                int timeSlice = getTimeSlice(departureTime);
551                TransportDataKey transportDataKey = makeKey(fromId.getId(),toId.getId(),timeSlice,typeId);
552                TransportData data = costCache.get(transportDataKey);
553                double transportTime;
554                if(data != null){
555                        transportTime = data.transportTime;
556                }
557                else{
558                        informStartCalc();
559                        Id<Link> fromLinkId = Id.create(fromId.getId(), Link.class);
560                        Id<Link> toLinkId = Id.create(toId.getId(), Link.class);
561                        Link fromLink = network.getLinks().get(fromLinkId);
562                        Link toLink = network.getLinks().get(toLinkId);
563                        org.matsim.vehicles.Vehicle matsimVehicle = getMatsimVehicle(vehicle);
564                        LeastCostPathCalculator router = createLeastCostPathCalculator();
565                        Path path = router.calcLeastCostPath(fromLink.getToNode(), toLink.getFromNode(), departureTime, null, matsimVehicle);
566//                      if(path == null) return Double.MAX_VALUE;
567                        double additionalCostTo = travelDisutility.getLinkTravelDisutility(toLink, departureTime+path.travelTime, null, matsimVehicle);
568                        double additionalTimeTo = travelTime.getLinkTravelTime(toLink, departureTime+path.travelTime, null, matsimVehicle);
569                        transportTime = path.travelTime;
570                        TransportData newData = new TransportData(path.travelCost+additionalCostTo,path.travelTime+additionalTimeTo);
571                        TransportData existingData = costCache.putIfAbsent(transportDataKey, newData);
572                        ttMemorizedCounter.incCounter();
573                        if(existingData == null){
574                                existingData = newData;
575                        }
576                        transportTime = existingData.transportTime;
577                        informEndCalc();
578                }
579                return transportTime;
580        }

595        /**
596         * Gets the transport-costs.
597         * 
598         * <p>If <code>fromId.equals(toId)</code> it returns 0.0. Otherwise, it looks up in the cache whether the transport-costs have already been computed 
599         * (see {@link TransportDataKey}, {@link TransportData}). If so, it returns
600         * the cached travel-cost value. If not, it computes and caches new values with the leastCostPathCalc defined in here.  
601         * 
602         * @Throws {@link IllegalStateException} if vehicle is null
603         */
604        @Override
605        public double getTransportCost(Location fromId, Location toId, double departureTime, Driver driver, Vehicle vehicle) {
606                if(fromId ==  null || toId == null) throw new IllegalStateException(
607                "either fromId ("+fromId+") or toId ("+toId+") is null [departureTime="+departureTime+"][vehicle="+vehicle+"]");
608        if(fromId.equals(toId)){
609                        return 0.0;
610                }
611                if(vehicle == null) {
612                        vehicle = getDefaultVehicle(fromId);
613                }
614                Id<Link> fromLinkId = Id.create(fromId.getId(), Link.class);
615                Id<Link> toLinkId = Id.create(toId.getId(), Link.class);
616                Link fromLink = network.getLinks().get(fromLinkId);
617                Link toLink = network.getLinks().get(toLinkId);
618                LeastCostPathCalculator router = createLeastCostPathCalculator();
619        
620                int timeSlice = getTimeSlice(departureTime);
621                String typeId = vehicle.getType().getTypeId();
622                TransportDataKey transportDataKey = makeKey(fromId.getId(),toId.getId(),timeSlice, typeId);
623                TransportData data = costCache.get(transportDataKey);
624                double transportCost;
625                if(data != null){
626                        transportCost = data.transportCosts;
627                }
628                else{
629                        informStartCalc();
630                        org.matsim.vehicles.Vehicle matsimVehicle = getMatsimVehicle(vehicle);
631                        Path path = router.calcLeastCostPath(fromLink.getToNode(), toLink.getFromNode(), departureTime, null, matsimVehicle);
632//                      if(path == null) return Double.MAX_VALUE;
633                        double additionalCostTo = travelDisutility.getLinkTravelDisutility(toLink, departureTime+path.travelTime, null, matsimVehicle);
634                        double additionalTimeTo = travelTime.getLinkTravelTime(toLink, departureTime+path.travelTime,null,matsimVehicle);
635                        TransportData newData = new TransportData(path.travelCost+additionalCostTo,path.travelTime+additionalTimeTo);
636                        TransportData existingData = costCache.putIfAbsent(transportDataKey, newData);
637                        ttMemorizedCounter.incCounter();
638                        if(existingData == null){
639                                //succeeded
640                                existingData = newData;
641                        }
642                        transportCost = existingData.transportCosts;
643                        informEndCalc();
644                }
645                return transportCost;
646        }

680        private org.matsim.vehicles.Vehicle getMatsimVehicle(Vehicle vehicle) {
681                String typeId = vehicle.getType().getTypeId();
682                if(matsimVehicles.containsKey(typeId)){
683                        return matsimVehicles.get(typeId);
684                }
685                org.matsim.vehicles.Vehicle matsimVehicle = new MatsimVehicleWrapper(vehicle);
686                matsimVehicles.put(typeId,matsimVehicle);
687                return matsimVehicle;
688        }

698        private LeastCostPathCalculator createLeastCostPathCalculator(){
699                LeastCostPathCalculator router = routerCache.get(Thread.currentThread().getId());
700                if(router == null){
701                        LeastCostPathCalculator newRouter = leastCostPathCalculatorFactory.createPathCalculator(network, travelDisutility, travelTime);
702                        router = routerCache.putIfAbsent(Thread.currentThread().getId(), newRouter);
703                        if(router == null){
704                                router = newRouter;
705                        }
706                }
707                return router;
708        }
