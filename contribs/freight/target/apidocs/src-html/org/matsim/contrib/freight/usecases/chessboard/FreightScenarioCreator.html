047        //carriers
048        Carriers carriers = new Carriers();

050        for(int i=1;i<10;i++){
051            Id<Link> homeId = Id.createLinkId("i("+i+",9)R");

063            carriers.addCarrier(carrier_);

088    private static void createCustomers(Carrier carrier, Network network) {
089        List<Id<Link>> innerCityLinks = createInnerCityLinks(network);
090        List<Id<Link>> outerCityLinks = createOuterCityLinks(network);
091
092        for(int i=0;i<20;i++){
093            CarrierService.Builder serviceBuilder = CarrierService.Builder.newInstance(Id.create((i + 1),CarrierService.class), drawLocationLinkId(innerCityLinks, outerCityLinks));
094            serviceBuilder.setCapacityDemand(1);
095            serviceBuilder.setServiceDuration(5*60);
096            serviceBuilder.setServiceStartTimeWindow(TimeWindow.newInstance(6*60*60, 15*60*60));
097            carrier.getServices().add(serviceBuilder.build());
098        }
099    }
100
101    private static Id<Link> drawLocationLinkId(List<Id<Link>> innerCityLinks, List<Id<Link>> outerCityLinks) {
102        double probInner = 0.5;
103        double randomFigure = random.nextDouble();
104        if(randomFigure <= probInner){
105            int randomIndex = random.nextInt(innerCityLinks.size());
106            return innerCityLinks.get(randomIndex);
107        }
108        else{
109            int randomIndex = random.nextInt(outerCityLinks.size());
110            return outerCityLinks.get(randomIndex);
111        }
112    }
113
114    private static List<Id<Link>> createOuterCityLinks(Network network) {
115        List<Id<Link>> inner = new InnerOuterCityScenarioCreator().getInnerCityLinks();
116        List<Id<Link>> outer = new ArrayList<Id<Link>>();
117        for(Id<Link> id : network.getLinks().keySet()){
118            if(!inner.contains(id)){
119                outer.add(id);
120            }
121        }
122        return outer;
123    }
124
125    private static List<Id<Link>> createInnerCityLinks(Network network) {
126        List<Id<Link>> inner = new InnerOuterCityScenarioCreator().getInnerCityLinks();
127        List<Id<Link>> innerCityLinkIds = new ArrayList<Id<Link>>();
128        for(Id<Link> id : inner){
129            if(network.getLinks().containsKey(id)){
130                innerCityLinkIds.add(id);
131            }
132        }
133        return innerCityLinkIds;
134    }

136    private static void createFleet(Id<Link> homeId, Carrier carrier) {
137        Id<Link> oppositeId = getOpposite(homeId);
138
139        //light
140        carrier.getCarrierCapabilities().getCarrierVehicles().add(getLightVehicle(carrier.getId(), homeId, "a"));
141        carrier.getCarrierCapabilities().getCarrierVehicles().add(getLightVehicle(carrier.getId(), oppositeId, "b"));
142
143        //heavy
144        carrier.getCarrierCapabilities().getCarrierVehicles().add(getHeavyVehicle(carrier.getId(), homeId, "a"));
145        carrier.getCarrierCapabilities().getCarrierVehicles().add(getHeavyVehicle(carrier.getId(), oppositeId, "b"));
146
147        carrier.getCarrierCapabilities().setFleetSize(FleetSize.INFINITE);
148    }
149
150    public static Id<Link> getOpposite(Id<Link> homeId) {
151        if(homeId.toString().startsWith("i")){
152            String opposite = "i(" + homeId.toString().substring(2,4);
153            if(homeId.toString().substring(4,5).equals("0")) opposite += "9)R";
154            else opposite += "0)";
155            return Id.createLinkId(opposite);
156        }
157        else{
158            String opposite = "j(";
159            if(homeId.toString().substring(2,3).equals("0")) {
160                opposite += "9," + homeId.toString().substring(4,6);
161            }
162            else opposite += "0," + homeId.toString().substring(4,6) + "R";
163            return Id.createLinkId(opposite);
164        }
165    }

167    private static CarrierVehicle getLightVehicle(Id<?> id, Id<Link> homeId, String depot) {
168        CarrierVehicle.Builder vBuilder = CarrierVehicle.Builder.newInstance(Id.create(("carrier_"+id.toString()+"_lightVehicle_" + depot) ,Vehicle.class), homeId);
169        vBuilder.setEarliestStart(6*60*60);
170        vBuilder.setLatestEnd(16*60*60);
171        vBuilder.setType(createLightType());
172        return vBuilder.build();
173    }

180        typeBuilder.setCostPerTimeUnit(0.008);
