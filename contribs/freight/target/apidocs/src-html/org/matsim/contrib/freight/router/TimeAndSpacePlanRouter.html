001package org.matsim.contrib.freight.router;
002
003import org.matsim.api.core.v01.network.Network;
004import org.matsim.contrib.freight.carrier.Carrier;
005import org.matsim.contrib.freight.carrier.CarrierPlan;
006import org.matsim.contrib.freight.carrier.Carriers;
007import org.matsim.contrib.freight.carrier.ScheduledTour;
008import org.matsim.core.router.util.LeastCostPathCalculator;
009import org.matsim.core.router.util.TravelTime;
010
011/**
012 * Router routing carrierPlans in time and space.
013 * 
014 * @author sschroeder
015 *
016 */
017public class TimeAndSpacePlanRouter {
018
019        private LeastCostPathCalculator router;
020        
021        private Network network;
022        
023        private TravelTime travelTime;
024        
025        public TimeAndSpacePlanRouter(LeastCostPathCalculator router, Network network, TravelTime travelTime) {
026                super();
027                this.router = router;
028                this.network = network;
029                this.travelTime = travelTime;
030        }
031
032        /**
033         * Routes all scheduled tours within the plan in time and space.
034         * 
035         * @param plan
036         * @see TimeAndSpaceTourRouter
037         */
038        public void run(CarrierPlan plan) {
039                if(plan == null) throw new IllegalStateException("plan is missing.");
040                for(ScheduledTour tour : plan.getScheduledTours()){
041                        new TimeAndSpaceTourRouter(router, network, travelTime).route(tour);
042                }       
043        }
044        
045        public void run(Carriers carriers){
046                for(Carrier carrier : carriers.getCarriers().values()){
047                        CarrierPlan p = carrier.getSelectedPlan();
048                        if(p != null){
049                                run(p);
050                        }
051                }
052        }
053
054}
