001package org.matsim.contrib.freight.carrier;
002
003
004/**
005 * A concrete assignment of a tour, a vehicle and a departure time.
006 * 
007 * @author sschroeder, mzilske
008 *
009 */
010public class ScheduledTour {
011
012        
013        /**
014         * Returns a new instance of ScheduledTour.
015         * 
016         * <p>Look at the builder. It might be easier to build a scheduled tour. 
017         * You get the builder this way: ScheduledTour.Builder.newInstance(carrierVehicle).
018         * 
019         * @param tour
020         * @param vehicle
021         * @param departureTime
022         * @return a scheduledTour
023         * @see ScheduledTour
024         */
025        public static ScheduledTour newInstance(Tour tour, CarrierVehicle vehicle, double departureTime){
026                return new ScheduledTour(tour,vehicle,departureTime);
027        }
028        
029        private final Tour tour;
030
031        private final CarrierVehicle vehicle;
032
033        private final double departureTime;
034
035        private ScheduledTour(final Tour tour, final CarrierVehicle vehicle, final double departureTime) {
036                this.tour = tour;
037                this.vehicle = vehicle;
038                this.departureTime = departureTime;
039        }
040
041        public Tour getTour() {
042                return tour;
043        }
044
045        public CarrierVehicle getVehicle() {
046                return vehicle;
047        }
048
049        public double getDeparture() {
050                return departureTime;
051        }
052
053        @Override
054        public String toString() {
055                return "[tour="+tour+"][vehicle="+vehicle+"][departureTime="+departureTime+"]";
056        }
057
058}
