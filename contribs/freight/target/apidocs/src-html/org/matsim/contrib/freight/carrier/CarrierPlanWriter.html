001package org.matsim.contrib.freight.carrier;
002
003import java.io.BufferedWriter;
004import java.io.IOException;
005import java.util.Collection;
006import java.util.HashMap;
007import java.util.Map;
008
009import com.graphhopper.jsprit.core.problem.job.Shipment;
010
011import org.apache.log4j.Logger;
012import org.matsim.api.core.v01.Id;
013import org.matsim.api.core.v01.network.Link;
014import org.matsim.contrib.freight.carrier.Tour.Leg;
015import org.matsim.contrib.freight.carrier.Tour.ShipmentBasedActivity;
016import org.matsim.contrib.freight.carrier.Tour.TourElement;
017import org.matsim.core.population.routes.NetworkRoute;
018import org.matsim.core.utils.io.MatsimXmlWriter;
019import org.matsim.core.utils.misc.Time;
020
021/**
022 * A writer that writes carriers and their plans in an xml-file.
023 * 
024 * @author sschroeder
025 *
026 */
027public class CarrierPlanWriter extends MatsimXmlWriter {
028
029        private static Logger logger = Logger.getLogger(CarrierPlanWriter.class);
030
031        private Collection<Carrier> carriers;
032
033        private int idCounter = 0;
034
035        private Map<CarrierShipment, Id<Shipment>> registeredShipments = new HashMap<CarrierShipment, Id<Shipment>>();
036
037        /**
038         * Constructs the writer with the carriers to be written.
039         * 
040         * @param carriers to be written
041         */
042        public CarrierPlanWriter(Collection<Carrier> carriers) {
043                super();
044                this.carriers = carriers;
045        }
046
047        /**
048         * Writes carriers and their plans into a xml-file.
049         * 
050         * @param filename should be the target xml-file
051         */
052        public void write(String filename) {
053                logger.info("write carrier plans");
054                try {
055                        openFile(filename);
056                        writeXmlHead();
057                        startCarriers(this.writer);
058                        for (Carrier carrier : carriers) {
059                                startCarrier(carrier, this.writer);
060                                writeVehicles(carrier, this.writer);
061                                writeShipments(carrier, this.writer);
062                                writePlans(carrier, this.writer);
063                                endCarrier(this.writer);
064                        }
065                        endCarriers(this.writer);
066                        close();
067                        logger.info("done");
068                } catch (IOException e) {
069                        e.printStackTrace();
070                        logger.error(e);
071                        System.exit(1);
072                }
073        }
074
075        private void startCarriers(BufferedWriter writer) throws IOException {
076                writer.write("\t<carriers>\n");
077        }
078
079        private void startCarrier(Carrier carrier, BufferedWriter writer)
080                        throws IOException {
081                writer.write("\t\t<carrier id=\"" + carrier.getId() + "\">\n");
082        }
083
084        private void writeVehicles(Carrier carrier, BufferedWriter writer)
085                        throws IOException {
086                writer.write("\t\t\t<vehicles>\n");
087                for (CarrierVehicle v : carrier.getCarrierCapabilities().getCarrierVehicles()) {
088                        writer.write("\t\t\t\t<vehicle id=\"" + v.getVehicleId()
089                                        + "\" linkId=\"" + v.getLocation() + "\"" + "\" typeId=\""
090                                        + v.getVehicleTypeId().toString()
091                                        + "\" earliestStart=\"" + getTime(v.getEarliestStartTime())
092                                        + "\" latestEnd=\"" + getTime(v.getLatestEndTime())
093                                        + "\"/>\n");
094                }
095                writer.write("\t\t\t</vehicles>\n\n");
096        }
097
098        private void writeShipments(Carrier carrier, BufferedWriter writer)
099                        throws IOException {
100                writer.write("\t\t\t<shipments>\n");
101                for (CarrierShipment s : carrier.getShipments()) {
102                        // CarrierShipment s = contract.getShipment();
103                        Id<Shipment> shipmentId = Id.create(++idCounter, Shipment.class);
104                        registeredShipments.put(s, shipmentId);
105                        writer.write("\t\t\t\t<shipment ");
106                        writer.write("id=\"" + shipmentId + "\" ");
107                        writer.write("from=\"" + s.getFrom() + "\" ");
108                        writer.write("to=\"" + s.getTo() + "\" ");
109                        writer.write("size=\"" + s.getSize() + "\" ");
110                        writer.write("startPickup=\""
111                                        + getTime(s.getPickupTimeWindow().getStart()) + "\" ");
112                        writer.write("endPickup=\""
113                                        + getTime(s.getPickupTimeWindow().getEnd()) + "\" ");
114                        writer.write("startDelivery=\""
115                                        + getTime(s.getDeliveryTimeWindow().getStart()) + "\" ");
116                        writer.write("endDelivery=\""
117                                        + getTime(s.getDeliveryTimeWindow().getEnd()) + "\" ");
118                        writer.write("pickupServiceTime=\""
119                                        + getTime(s.getPickupServiceTime()) + "\" ");
120                        writer.write("deliveryServiceTime=\""
121                                        + getTime(s.getDeliveryServiceTime()) + "\"/>\n");
122                }
123                writer.write("\t\t\t</shipments>\n\n");
124        }
125
126        private String getTime(double time) {
127                return Time.writeTime(time);
128        }
129
130        private void writePlans(Carrier carrier, BufferedWriter writer)
131                        throws IOException {
132                if (carrier.getSelectedPlan() == null) {
133                        return;
134                }
135                
136                for(CarrierPlan plan : carrier.getPlans()){
137                        writer.write("\t\t\t<plan");
138                        if(plan.getScore() != null){
139                                writer.write(" score=\"" + plan.getScore().toString() + "\"");
140                        }
141                        if(carrier.getSelectedPlan() != null){
142                                if(plan == carrier.getSelectedPlan()){
143                                        writer.write(" selected=\"true\"");
144                                }
145                                else{
146                                        writer.write(" selected=\"false\"");
147                                }
148                        }
149                        else{
150                                writer.write(" selected=\"false\"");
151                        }
152                        writer.write(">\n");
153                                        
154                        for (ScheduledTour tour : plan.getScheduledTours()) {
155                                writer.write("\t\t\t\t<tour ");
156                                writer.write("vehicleId=\"" + tour.getVehicle().getVehicleId()
157                                                + "\">\n");
158                                writer.write("\t\t\t\t\t<act type=\"" + FreightConstants.START
159                                                + "\" end_time=\"" + Time.writeTime(tour.getDeparture())
160                                                + "\"/>\n");
161                                for (TourElement tourElement : tour.getTour().getTourElements()) {
162                                        if (tourElement instanceof Leg) {
163                                                Leg leg = (Leg) tourElement;
164                                                writer.write("\t\t\t\t\t<leg dep_time=\""
165                                                                + Time.writeTime(leg.getExpectedDepartureTime())
166                                                                + "\" transp_time=\""
167                                                                + Time.writeTime(leg.getExpectedTransportTime())
168                                                                + "\">");
169                                                if (leg.getRoute() != null) {
170                                                        writer.write("\n");
171                                                        writer.write("\t\t\t\t\t\t<route>");
172                                                        boolean firstLink = true;
173                                                        for (Id<Link> id : ((NetworkRoute) leg.getRoute()).getLinkIds()) {
174                                                                if (firstLink) {
175                                                                        writer.write(id.toString());
176                                                                        firstLink = false;
177                                                                } else {
178                                                                        writer.write(" " + id.toString());
179                                                                }
180                                                        }
181                                                        writer.write("</route>\n");
182                                                        writer.write("\t\t\t\t\t</leg>\n");
183                                                } else {
184                                                        writer.write("</leg>\n");
185                                                }
186                                        }
187                                        if (tourElement instanceof ShipmentBasedActivity) {
188                                                ShipmentBasedActivity act = (ShipmentBasedActivity) tourElement;
189                                                writer.write("\t\t\t\t\t<act ");
190                                                writer.write("type=\"" + act.getActivityType() + "\" ");
191                                                writer.write("shipmentId=\""
192                                                                + registeredShipments.get(act.getShipment())
193                                                                + "\" ");
194                                                writer.write("/>\n");
195                                        }
196
197                                }
198                                writer.write("\t\t\t\t\t<act type=\"" + FreightConstants.END
199                                                + "\"/>\n");
200                                writer.write("\t\t\t\t</tour>\n");
201                        }
202                        writer.write("\t\t\t</plan>\n\n");
203                }
204                
205        }
206
207        private void endCarrier(BufferedWriter writer) throws IOException {
208                writer.write("\t\t</carrier>\n\n");
209                registeredShipments.clear();
210        }
211
212        private void endCarriers(BufferedWriter writer) throws IOException {
213                writer.write("\t</carriers>\n");
214
215        }
216}
