036public class InitialCarrierPlanCreator {
037
038    private Network network;
039
040    public InitialCarrierPlanCreator(Network network) {
041        this.network = network;
042    }
043
044    public CarrierPlan createPlan(Carrier carrier){
045//              VehicleRoutingProblem.Builder vrpBuilder = MatsimJspritFactory.createRoutingProblemBuilder(carrier, network);
046//              NetworkBasedTransportCosts.Builder costsBuilder = NetworkBasedTransportCosts.Builder.newInstance(network, carrier.getCarrierCapabilities().getVehicleTypes());
047//              NetworkBasedTransportCosts costs = costsBuilder.build();
048//              vrpBuilder.setRoutingCost(costs);
049//              VehicleRoutingProblem vrp = vrpBuilder.build();
050//              
051//              VehicleRoutingAlgorithm vra = VehicleRoutingAlgorithms.readAndCreateAlgorithm(vrp, "input/usecases/chessboard/vrpalgo/ini_algorithm_v2.xml");
052////            vra.getAlgorithmListeners().addListener(new AlgorithmSearchProgressChartListener("output/"+carrier.getId()+".png"));
053//              Collection<VehicleRoutingProblemSolution> solutions = vra.searchSolutions();
054//              
055//              CarrierPlan plan = MatsimJspritFactory.createPlan(carrier, Solutions.bestOf(solutions));
056//              
057
058
059        //construct the routing problem - here the interface to jsprit comes into play
060        VehicleRoutingProblem.Builder vrpBuilder = MatsimJspritFactory.createRoutingProblemBuilder(carrier, network);
061
062        //******
063        //Define transport-costs
064        //******
065        //construct network-based routing costs
066        //by default travelTimes are calculated with freeSpeed and vehicleType.maxVelocity on the network
067        NetworkBasedTransportCosts.Builder tpcostsBuilder = NetworkBasedTransportCosts.Builder.newInstance(network, carrier.getCarrierCapabilities().getVehicleTypes());
068        //sets time-dependent travelTimes
069        //                              tpcostsBuilder.setTravelTime(travelTimes);
070        //sets time-slice to build time-dependent tpcosts and traveltime matrices
071        //                              tpcostsBuilder.setTimeSliceWidth(900);
072        //                              tpcostsBuilder.setFIFO(true);
073        //assign netBasedCosts to RoutingProblem
074        NetworkBasedTransportCosts netbasedTransportcosts = tpcostsBuilder.build();
075
076        //set transport-costs
077        vrpBuilder.setRoutingCost(netbasedTransportcosts);
078
079        //******
080        //Define activity-costs
081        //******
082        //should be inline with activity-scoring
083        VehicleRoutingActivityCosts activitycosts = new VehicleRoutingActivityCosts(){
084
085            private double penalty4missedTws = 0.01;
086
087            @Override
088            public double getActivityCost(TourActivity act, double arrivalTime, Driver arg2, Vehicle vehicle) {
089                double tooLate = Math.max(0, arrivalTime - act.getTheoreticalLatestOperationStartTime());
090                double waiting = Math.max(0, act.getTheoreticalEarliestOperationStartTime() - arrivalTime);
091                //                                              double waiting = 0.;
092                double service = act.getOperationTime()*vehicle.getType().getVehicleCostParams().perTimeUnit;
093                return penalty4missedTws*tooLate + vehicle.getType().getVehicleCostParams().perTimeUnit*waiting + service;
094                //                                              //                              return penalty4missedTws*tooLate;
095                //                                              return 0.0;
096            }
097
098                        @Override
099                        public double getActivityDuration(TourActivity tourAct, double arrivalTime, Driver driver,
100                                        Vehicle vehicle) {
101                                double activityDuration = Math.max(0, tourAct.getEndTime() - tourAct.getArrTime()); //including waiting times
102                                return activityDuration;
103                        }
104
105        };
106        vrpBuilder.setActivityCosts(activitycosts);
107
108        //build the problem
109        VehicleRoutingProblem vrp = vrpBuilder.build();
110
111        //configure the algorithm
112        AlgorithmConfig algorithmConfig = new AlgorithmConfig();
113        AlgorithmConfigXmlReader xmlReader = new AlgorithmConfigXmlReader(algorithmConfig);
114        xmlReader.read("input/usecases/chessboard/vrpalgo/ini_algorithm_v2.xml");
115
116        StateManager stateManager = new StateManager(vrp);
117        stateManager.updateLoadStates();
118
119        ConstraintManager constraintManager = new ConstraintManager(vrp,stateManager);
120        constraintManager.addLoadConstraint();
121        
122        Boolean addDefaultCostCalculators = true;
123        
124        VehicleRoutingAlgorithm vra = VehicleRoutingAlgorithms.readAndCreateAlgorithm(vrp, algorithmConfig, 0, null, stateManager, constraintManager, addDefaultCostCalculators);
125
126        //get configures algorithm
127        //                              VehicleRoutingAlgorithm vra = VehicleRoutingAlgorithms.readAndCreateAlgorithm(vrp, vrpAlgorithmConfig);
128        //                              vra.getAlgorithmListeners().addListener(new AlgorithmSearchProgressChartListener("output/"+carrierPlan.getCarrier().getId() + "_" + carrierPlan.hashCode() + ".png"));
129        //add initial-solution - which is the initialSolution for the vehicle-routing-algo
130        //                              vra.addInitialSolution(MatsimJspritFactory.createSolution(carrierPlan, network));
131
132        //solve problem
133        Collection<VehicleRoutingProblemSolution> solutions = vra.searchSolutions();
134
135        //get best
136        VehicleRoutingProblemSolution solution = Solutions.bestOf(solutions);
137
138        //              SolutionPlotter.plotSolutionAsPNG(vrp, solution, "output/sol_"+System.currentTimeMillis()+".png", "sol");
139
140        //create carrierPlan from solution
141        CarrierPlan plan = MatsimJspritFactory.createPlan(carrier, solution);
142        NetworkRouter.routePlan(plan, netbasedTransportcosts);
143        return plan;
144    }
145
146    public static void main(String[] args) {
147
148        Config config = new Config();
149        config.addCoreModules();
150        Scenario scenario = ScenarioUtils.createScenario(config);
151        new MatsimNetworkReader(scenario.getNetwork()).readFile("input/usecases/chessboard/network/grid9x9_cap20.xml");
152
153        Carriers carriers = new Carriers();
154        new CarrierPlanXmlReaderV2(carriers).readFile("input/usecases/chessboard/freight/carrierPlansWithoutRoutes_10minTW.xml");
155
156        CarrierVehicleTypes types = new CarrierVehicleTypes();
157        new CarrierVehicleTypeReader(types).readFile("input/usecases/chessboard/freight/vehicleTypes.xml");
158        new CarrierVehicleTypeLoader(carriers).loadVehicleTypes(types);
159
160        for(Carrier carrier : carriers.getCarriers().values()){
161            CarrierPlan plan = new InitialCarrierPlanCreator(scenario.getNetwork()).createPlan(carrier);
162            carrier.setSelectedPlan(plan);
163        }
164
165        new CarrierPlanXmlWriterV2(carriers).write("input/usecases/chessboard/freight/carrierPlans_10minTW.xml");
166    }
