013package org.matsim.contrib.freight.utils;
014
015import org.matsim.api.core.v01.Scenario;
016import org.matsim.api.core.v01.network.Network;
017import org.matsim.core.config.Config;
018import org.matsim.core.network.io.MatsimNetworkReader;
019import org.matsim.core.scenario.ScenarioUtils;
020
021/**
022 * Class that consolidates matsim.config and scenario.
023 * 
024 * @author sschroeder
025 *
026 */
027public class MatsimStuffLoader {
028        
029        /**
030         * Class to store Matsim.Config and Matsim.Scenario
031         * 
032         * @author stefan
033         *
034         */
035        public static class MatsimStuff {
036                public final Config config;
037                public final Scenario scenario;
038                
039                public MatsimStuff(Config config, Scenario scenario) {
040                        super();
041                        this.config = config;
042                        this.scenario = scenario;
043                }
044                
045                public Network getNetwork(){
046                        return scenario.getNetwork();
047                }
048        }
049        
050        /**
051         * Reads and creates a network and sets up the required matsim.config and matsim.scenario.
052         * 
053         * <p>The network can be retrieved by calling <code>matsimStuff.scenario.getNetwork();</code>
054         * 
055         * @param networkFilename
056         * @return
057         */
058        public static MatsimStuff loadNetworkAndGetStuff(String networkFilename){
059                Config config = new Config();
060                config.addCoreModules();
061                Scenario scenario = ScenarioUtils.createScenario(config);
062                new MatsimNetworkReader(scenario.getNetwork()).readFile(networkFilename);
063                return new MatsimStuff(config, scenario);
064        }
065
066}
