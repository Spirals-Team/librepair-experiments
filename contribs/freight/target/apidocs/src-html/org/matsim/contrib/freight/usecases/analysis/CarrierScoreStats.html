041/**
042 * As you can see, it is basically a copy of {@link org.matsim.analysis.ScoreStatsControlerListener}. However, it is modified to score {@link Carrier}s
043 * rather than Persons. (Oct'13, schroeder)
044 * 
045 * 
046 * <p>Calculates at the end of each iteration the following statistics:
047 * <ul>
048 * <li>average score of the selected plan</li>
049 * <li>average of the score of the worst plan of each agent</li>
050 * <li>average of the score of the best plan of each agent</li>
051 * <li>average of the average score of all plans of each agent</li>
052 * </ul>
053 * Plans with undefined scores
054 * are not included in the statistics. The calculated values are written to a file, each iteration on
055 * a separate line.
056 *
057 * @author mrieser
058 */
059public class CarrierScoreStats implements StartupListener, IterationEndsListener, ShutdownListener {
060
061        final private static int INDEX_WORST = 0;
062        final private static int INDEX_BEST = 1;
063        final private static int INDEX_AVERAGE = 2;
064        final private static int INDEX_EXECUTED = 3;
065
066        private BufferedWriter out;
067        final private String fileName;
068        
069        private final boolean createPNG;
070        private double[][] history = null;
071        private int minIteration = 0;
072        
073        private Carriers carriers;
074
075        private final static Logger log = Logger.getLogger(CarrierScoreStats.class);
076
077        /**
078         * Creates a new ScoreStats instance.
079         *
080         * @param filename including the path, excluding the file type extension
081         * @param createPNG true if in every iteration, the scorestats should be visualized in a graph and written to disk.
082         * @throws UncheckedIOException
083         */
084        public CarrierScoreStats(Carriers carriers, final String filename, final boolean createPNG) throws UncheckedIOException {
085                this.carriers = carriers;
086                this.fileName = filename;
087                this.createPNG = createPNG;
088        }
089
090        @Override
091        public void notifyStartup(final StartupEvent event) {
092                if (fileName.toLowerCase(Locale.ROOT).endsWith(".txt")) {
093                        this.out = IOUtils.getBufferedWriter(fileName);
094                } else {
095                        this.out = IOUtils.getBufferedWriter(fileName + ".txt");
096                }
097                try {
098                        this.out.write("ITERATION\tavg. EXECUTED\tavg. WORST\tavg. AVG\tavg. BEST\n");
099                } catch (IOException e) {
100                        throw new UncheckedIOException(e);
101                }
102                this.minIteration = event.getServices().getConfig().controler().getFirstIteration();
103                int maxIter = event.getServices().getConfig().controler().getLastIteration();
104                int iterations = maxIter - this.minIteration;
105                if (iterations > 5000) iterations = 5000; // limit the history size
106                this.history = new double[4][iterations+1];
107        }
108
109        @Override
110        public void notifyIterationEnds(final IterationEndsEvent event) {
111                double sumScoreWorst = 0.0;
112                double sumScoreBest = 0.0;
113                double sumAvgScores = 0.0;
114                double sumExecutedScores = 0.0;
115                int nofScoreWorst = 0;
116                int nofScoreBest = 0;
117                int nofAvgScores = 0;
118                int nofExecutedScores = 0;
119//              int nofExecutedIvPlans = 0;
120//              int nofExecutedOevPlans = 0;
121
122//              for (Person person : this.population.getPersons().values()) {
123                for (Carrier carrier : carriers.getCarriers().values()){
124                        CarrierPlan worstPlan = null;
125                        CarrierPlan bestPlan = null;
126                        double worstScore = Double.POSITIVE_INFINITY;
127                        double bestScore = Double.NEGATIVE_INFINITY;
128                        double sumScores = 0.0;
129                        double cntScores = 0;
130                        for (CarrierPlan plan : carrier.getPlans()) {
131
132                                if (plan.getScore() == null) {
133                                        continue;
134                                }
135                                double score = plan.getScore();
136
137                                // worst plan
138                                if (worstPlan == null) {
139                                        worstPlan = plan;
140                                        worstScore = score;
141                                } else if (score < worstScore) {
142                                        worstPlan = plan;
143                                        worstScore = score;
144                                }
145
146                                // best plan
147                                if (bestPlan == null) {
148                                        bestPlan = plan;
149                                        bestScore = score;
150                                } else if (score > bestScore) {
151                                        bestPlan = plan;
152                                        bestScore = score;
153                                }
154
155                                // avg. score
156                                sumScores += score;
157                                cntScores++;
158
159                                // executed plan?
160                                if (carrier.getSelectedPlan().equals(plan)) {
161                                        sumExecutedScores += score;
162                                        nofExecutedScores++;
163//                                      if (plan.getType() == Plan.Type.CAR) {
164//                                              nofExecutedIvPlans ++;
165//                                      }
166//                                      else if (plan.getType() == Plan.Type.PT) {
167//                                              nofExecutedOevPlans++;
168//                                      }
169                                }
170                        }
171
172                        if (worstPlan != null) {
173                                nofScoreWorst++;
174                                sumScoreWorst += worstScore;
175                        }
176                        if (bestPlan != null) {
177                                nofScoreBest++;
178                                sumScoreBest += bestScore;
179                        }
180                        if (cntScores > 0) {
181                                sumAvgScores += (sumScores / cntScores);
182                                nofAvgScores++;
183                        }
184                }
185                log.info("-- avg. score of the executed plan of each agent: " + (sumExecutedScores / nofExecutedScores));
186//              log.info("-- number of executed plans: "  + nofExecutedScores);
187//              log.info("-- number of executed iv plans: "  + nofExecutedIvPlans);
188//              log.info("-- number of executed oev plans: "  + nofExecutedOevPlans);
189//              log.info("-- modal split iv: "  + ((nofExecutedScores == 0) ? 0 : ((double)nofExecutedIvPlans / (double)nofExecutedScores * 100d)) +
190//                              " % oev: " + ((nofExecutedScores == 0) ? 0 : ((double)nofExecutedOevPlans / (double)nofExecutedScores * 100d)) + " %");
191                log.info("-- avg. score of the worst plan of each agent: " + (sumScoreWorst / nofScoreWorst));
192                log.info("-- avg. of the avg. plan score per agent: " + (sumAvgScores / nofAvgScores));
193                log.info("-- avg. score of the best plan of each agent: " + (sumScoreBest / nofScoreBest));
194
195                try {
196                        this.out.write(event.getIteration() + "\t" + (sumExecutedScores / nofExecutedScores) + "\t" +
197                                        (sumScoreWorst / nofScoreWorst) + "\t" + (sumAvgScores / nofAvgScores) + "\t" + (sumScoreBest / nofScoreBest) + "\n");
198                        this.out.flush();
199                } catch (IOException e) {
200                        e.printStackTrace();
201                }
202
203                if (this.history != null) {
204                        int index = event.getIteration() - this.minIteration;
205                        this.history[INDEX_WORST][index] = (sumScoreWorst / nofScoreWorst);
206                        this.history[INDEX_BEST][index] = (sumScoreBest / nofScoreBest);
207                        this.history[INDEX_AVERAGE][index] = (sumAvgScores / nofAvgScores);
208                        this.history[INDEX_EXECUTED][index] = (sumExecutedScores / nofExecutedScores);
209
210                        if (this.createPNG && event.getIteration() != this.minIteration) {
211                                // create chart when data of more than one iteration is available.
212                                XYLineChart chart = new XYLineChart("Score Statistics", "iteration", "score");
213                                double[] iterations = new double[index + 1];
214                                for (int i = 0; i <= index; i++) {
215                                        iterations[i] = i + this.minIteration;
216                                }
217                                double[] values = new double[index + 1];
218                                System.arraycopy(this.history[INDEX_WORST], 0, values, 0, index + 1);
219                                chart.addSeries("avg. worst score", iterations, values);
220                                System.arraycopy(this.history[INDEX_BEST], 0, values, 0, index + 1);
221                                chart.addSeries("avg. best score", iterations, values);
222                                System.arraycopy(this.history[INDEX_AVERAGE], 0, values, 0, index + 1);
223                                chart.addSeries("avg. of plans' average score", iterations, values);
224                                System.arraycopy(this.history[INDEX_EXECUTED], 0, values, 0, index + 1);
225                                chart.addSeries("avg. executed score", iterations, values);
226                                chart.addMatsimLogo();
227                                chart.saveAsPng(this.fileName + ".png", 800, 600);
228                        }
229                        if (index == (this.history[0].length - 1)) {
230                                // we cannot store more information, so disable the graph feature.
231                                this.history = null;
232                        }
233                }
234        }
235
236        @Override
237        public void notifyShutdown(final ShutdownEvent controlerShudownEvent) {
238                try {
239                        this.out.close();
240                } catch (IOException e) {
241                        e.printStackTrace();
242                }
243
244        }
