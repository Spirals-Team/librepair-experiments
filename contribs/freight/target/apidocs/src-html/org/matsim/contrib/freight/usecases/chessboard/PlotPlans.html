001package org.matsim.contrib.freight.usecases.chessboard;
002
003import com.graphhopper.jsprit.analysis.toolbox.Plotter;
004import com.graphhopper.jsprit.core.problem.VehicleRoutingProblem;
005import com.graphhopper.jsprit.core.problem.solution.VehicleRoutingProblemSolution;
006
007import org.matsim.api.core.v01.Id;
008import org.matsim.api.core.v01.Scenario;
009import org.matsim.contrib.freight.carrier.Carrier;
010import org.matsim.contrib.freight.carrier.CarrierPlanXmlReaderV2;
011import org.matsim.contrib.freight.carrier.CarrierVehicleTypeLoader;
012import org.matsim.contrib.freight.carrier.CarrierVehicleTypeReader;
013import org.matsim.contrib.freight.carrier.CarrierVehicleTypes;
014import org.matsim.contrib.freight.carrier.Carriers;
015import org.matsim.contrib.freight.jsprit.MatsimJspritFactory;
016import org.matsim.core.config.Config;
017import org.matsim.core.network.io.MatsimNetworkReader;
018import org.matsim.core.scenario.ScenarioUtils;
019
020public class PlotPlans {
021
022    public static void main(String[] args) {
023        Config config = new Config();
024        config.addCoreModules();
025        Scenario scenario = ScenarioUtils.createScenario(config);
026        new MatsimNetworkReader(scenario.getNetwork()).readFile("input/usecases/chessboard/network/grid9x9.xml");
027
028
029        final Carriers carriers = new Carriers();
030//              new CarrierPlanXmlReaderV2(carriers).read("input/usecases/chessboard/freight/singleCarrierTwentyActivities.xml");
031        new CarrierPlanXmlReaderV2(carriers).readFile("output/ITERS/it.140/140.carrierPlans.xml");
032
033        CarrierVehicleTypes types = new CarrierVehicleTypes();
034        new CarrierVehicleTypeReader(types).readFile("input/usecases/chessboard/freight/vehicleTypes.xml");
035        new CarrierVehicleTypeLoader(carriers).loadVehicleTypes(types);
036
037        final Carrier carrier = carriers.getCarriers().get(Id.create("carrier1",Carrier.class));
038        VehicleRoutingProblem.Builder vrpBuilder = MatsimJspritFactory.createRoutingProblemBuilder(carrier, scenario.getNetwork());
039        VehicleRoutingProblem vrp = vrpBuilder.build();
040
041        VehicleRoutingProblemSolution solution = MatsimJspritFactory.createSolution(carrier.getSelectedPlan(), vrp);
042
043        new Plotter(vrp,solution.getRoutes()).plot("output/plot_140", "carrier1");
044
045    }
046
047}
