013package org.matsim.contrib.freight.jsprit;
014
015import org.matsim.contrib.freight.carrier.CarrierPlan;
016import org.matsim.contrib.freight.router.TimeAndSpacePlanRouter;
017
018/**
019 * Router that routes {@link CarrierPlan}.
020 * 
021 * @author stefan schr√∂der
022 *
023 */
024public class NetworkRouter {
025        
026        /**
027         * Routes the {@link CarrierPlan} with the router defined in {@link NetworkBasedTransportCosts}.
028         * 
029         * <p>Note that this changes the plan, i.e. it adds routes to the input-plan.
030         * 
031         * @param {@link CarrierPlan}
032         * @param {@link NetworkBasedTransportCosts}
033         */
034        public static void routePlan(CarrierPlan plan, NetworkBasedTransportCosts netbasedTransportCosts){
035                new TimeAndSpacePlanRouter(netbasedTransportCosts.getRouter(), netbasedTransportCosts.getNetwork(), netbasedTransportCosts.getTravelTime()).run(plan);
036        }
037
038}
