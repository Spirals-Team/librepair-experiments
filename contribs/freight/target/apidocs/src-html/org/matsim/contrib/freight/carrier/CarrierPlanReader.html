122        @Override
123        public void startTag(String name, Attributes atts, Stack<String> context) {
124                if (name.equals(CARRIER)) {
125                        String id = atts.getValue(ID);
126                        currentCarrier = CarrierImpl.newInstance(Id.create(id, Carrier.class)); 
127                }
128                if (name.equals(SHIPMENTS)) {
129                        currentShipments = new HashMap<String, CarrierShipment>();
130                }
131                if (name.equals(SHIPMENT)) {
132                        String from = atts.getValue(FROM);
133                        String to = atts.getValue(TO);
134                        int size = getInt(atts.getValue(SIZE));
135                        String startPickup = atts.getValue("startPickup");
136                        String endPickup = atts.getValue("endPickup");
137                        String startDelivery = atts.getValue("startDelivery");
138                        String endDelivery = atts.getValue("endDelivery");
139                        String pickupServiceTime = atts.getValue("pickupServiceTime");
140                        String deliveryServiceTime = atts.getValue("deliveryServiceTime");
141                        CarrierShipment.Builder shipmentBuilder = CarrierShipment.Builder.newInstance(Id.create(from, Link.class), Id.create(to, Link.class), size);
142                        if (startPickup == null) {
143                                shipmentBuilder.setPickupTimeWindow(TimeWindow.newInstance(0.0, Integer.MAX_VALUE)).setDeliveryTimeWindow(TimeWindow.newInstance(0.0, Integer.MAX_VALUE));
144                        } else {
145                                shipmentBuilder.setPickupTimeWindow(TimeWindow.newInstance(getDouble(startPickup), getDouble(endPickup))).
146                                                setDeliveryTimeWindow(TimeWindow.newInstance(getDouble(startDelivery), getDouble(endDelivery)));
147                        }
148                        if (pickupServiceTime != null) shipmentBuilder.setPickupServiceTime(getDouble(pickupServiceTime)); 
149                        if (deliveryServiceTime != null) shipmentBuilder.setDeliveryServiceTime(getDouble(deliveryServiceTime));
150                        CarrierShipment shipment = shipmentBuilder.build();
151                        currentShipments.put(atts.getValue(ID), shipment);
152                        currentCarrier.getShipments().add(shipment);
153                }
154
155                if (name.equals(VEHICLES)) {
156                        vehicles = new HashMap<String, CarrierVehicle>();
157                }
158                if (name.equals(VEHICLE)) {
159                        String vId = atts.getValue(ID);
160                        String linkId = atts.getValue(LINKID);
161                        String startTime = atts.getValue(VEHICLESTART);
162                        String endTime = atts.getValue(VEHICLEEND);
163                        String typeId = atts.getValue("typeId");
164                        if (typeId == null) {
165                                logger.warn("no vehicle type. set type='default' -> defaultVehicleType (see CarrierVehicleTypeImpl)");
166                                typeId = "default";
167                        }
168                        CarrierVehicle.Builder vehicleBuilder = CarrierVehicle.Builder.newInstance(Id.create(vId, Vehicle.class), Id.create(linkId, Link.class));
169                        vehicleBuilder.setTypeId(Id.create(typeId, VehicleType.class));
170                        vehicleBuilder.setType(CarrierVehicleType.Builder.newInstance(Id.create(typeId, VehicleType.class)).build());
171                        if (startTime != null) vehicleBuilder.setEarliestStart(getDouble(startTime));
172                        if (endTime != null) vehicleBuilder.setLatestEnd(getDouble(endTime));
173                        CarrierVehicle vehicle = vehicleBuilder.build();
174                        currentCarrier.getCarrierCapabilities().getCarrierVehicles().add(vehicle);
175                        vehicles.put(vId, vehicle);
176                }
177                if(name.equals("plan")){
178                        String score = atts.getValue("score");
179                        if(score != null){
180                                currentScore = getDouble(score);
181                        }
182                        else{
183                                currentScore = null;
184                        }
185                        String selected = atts.getValue("selected");
186                        if(selected == null ) {
187                                this.selected = false;
188                        }
189                        else if(selected.equals("true")){
190                                this.selected = true;
191                        }
192                        else{
193                                this.selected = false;
194                        }
195                        scheduledTours = new ArrayList<ScheduledTour>();
196                }
197                if (name.equals("tour")) {
198                        String vehicleId = atts.getValue("vehicleId");
199                        currentVehicle = vehicles.get(vehicleId);
200                        currentTourBuilder = Tour.Builder.newInstance();
201                }
202                if (name.equals("leg")) {
203                        currentLegDepTime = getDouble(atts.getValue("dep_time"));
204                        currentLegTransTime = getDouble(atts.getValue("transp_time"));
205                }
206                if (name.equals(ACTIVITY)) {
207                        if (atts.getValue(TYPE).equals("start")) {
208                                currentStartTime = getDouble(atts.getValue("end_time"));
209                                previousActLoc = currentVehicle.getLocation();
210                                currentTourBuilder.scheduleStart(currentVehicle.getLocation(),TimeWindow.newInstance(currentVehicle.getEarliestStartTime(), currentVehicle.getLatestEndTime()));
211                        } else if (atts.getValue(TYPE).equals("pickup")) {
212                                String id = atts.getValue(SHIPMENTID);
213                                CarrierShipment s = currentShipments.get(id);
214                                finishLeg(s.getFrom());
215                                currentTourBuilder.schedulePickup(s);
216                                previousActLoc = s.getFrom();
217                        } else if (atts.getValue(TYPE).equals("delivery")) {
218                                String id = atts.getValue(SHIPMENTID);
219                                CarrierShipment s = currentShipments.get(id);
220                                finishLeg(s.getTo());
221                                currentTourBuilder.scheduleDelivery(s);
222                                previousActLoc = s.getTo();
223                        } else if (atts.getValue(TYPE).equals("end")) {
224                                finishLeg(currentVehicle.getLocation());
225                                currentTourBuilder.scheduleEnd(currentVehicle.getLocation(), TimeWindow.newInstance(currentVehicle.getEarliestStartTime(),currentVehicle.getLatestEndTime()));
226                        }
227                }
228        }
229
230        private void finishLeg(Id<Link> toLocation) {
231                NetworkRoute route = null;
232                if (previousRouteContent != null) {
233                        List<Id<Link>> linkIds = NetworkUtils.getLinkIds(previousRouteContent);
234                        route = RouteUtils.createLinkNetworkRouteImpl(previousActLoc, toLocation);
235                        if (!linkIds.isEmpty()) {
236                                route.setLinkIds(previousActLoc, linkIds, toLocation);
237                        }
238                }
239                currentTourBuilder.addLeg(currentTourBuilder.createLeg(route, currentLegDepTime, currentLegTransTime));
240                previousRouteContent = null;
241        }
242
243        private double getDouble(String timeString) {
244                if (timeString.contains(":")) {
245                        return Time.parseTime(timeString);
246                } else {
247                        return Double.parseDouble(timeString);
248                }
249
250        }
251
252        @Override
253        public void endTag(String name, String content, Stack<String> context) {
254                if (name.equals("route")) {
255                        this.previousRouteContent = content;
256                }
257                if (name.equals("carrier")) {
258                        carriers.getCarriers().put(currentCarrier.getId(), currentCarrier);
259                }
260                if (name.equals("plan")) {
261                        currentPlan = new CarrierPlan(currentCarrier, scheduledTours);
262                        currentPlan.setScore(currentScore);
263                        currentCarrier.getPlans().add(currentPlan);
264                        if(this.selected){
265                                currentCarrier.setSelectedPlan(currentPlan);
266                        }
267                }
268                if (name.equals("tour")) {
269                        ScheduledTour sTour = ScheduledTour.newInstance(currentTourBuilder.build(), currentVehicle, currentStartTime);
270                        scheduledTours.add(sTour);
271                }
272        }
273
274        private int getInt(String value) {
275                return Integer.parseInt(value);
276        }
