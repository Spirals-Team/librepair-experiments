029package org.matsim.contrib.freight.mobsim;
030
031import com.google.inject.Provider;
032import org.matsim.api.core.v01.Scenario;
033import org.matsim.contrib.freight.CarrierConfig;
034import org.matsim.core.api.experimental.events.EventsManager;
035import org.matsim.core.config.groups.QSimConfigGroup;
036import org.matsim.core.mobsim.framework.Mobsim;
037import org.matsim.core.mobsim.qsim.QSim;
038import org.matsim.core.mobsim.qsim.QSimUtils;
039import org.matsim.core.mobsim.qsim.agents.DefaultAgentFactory;
040
041import javax.inject.Inject;
042import java.util.Collection;
043
044
045public class FreightQSimFactory implements Provider<Mobsim> {
046
047        private final Scenario scenario;
048        private EventsManager eventsManager;
049        private CarrierAgentTracker carrierAgentTracker;
050        private CarrierConfig carrierConfig;
051
052        @Inject
053        public FreightQSimFactory(Scenario scenario, EventsManager eventsManager, CarrierAgentTracker carrierAgentTracker, CarrierConfig carrierConfig) {
054                this.scenario = scenario;
055                this.eventsManager = eventsManager;
056                this.carrierAgentTracker = carrierAgentTracker;
057                this.carrierConfig = carrierConfig;
058        }
059
060        @Override
061        public Mobsim get() {
062                QSimConfigGroup conf = scenario.getConfig().qsim();
063                if (conf == null) {
064                        throw new NullPointerException(
065                                        "There is no configuration set for the QSim. Please add the module 'qsim' to your config file.");
066                }
067                final QSim sim = QSimUtils.createDefaultQSim(scenario, eventsManager);
068                Collection<MobSimVehicleRoute> vRoutes = carrierAgentTracker.createPlans();
069                FreightAgentSource agentSource = new FreightAgentSource(vRoutes, new DefaultAgentFactory(sim), sim);
070                sim.addAgentSource(agentSource);
071                if (carrierConfig.getPhysicallyEnforceTimeWindowBeginnings()) {
072                        WithinDayActivityReScheduling withinDayActivityRescheduling = new WithinDayActivityReScheduling(agentSource, carrierAgentTracker);
073                        sim.addQueueSimulationListeners(withinDayActivityRescheduling);
074                }
075                return sim;
076        }
077
078}
