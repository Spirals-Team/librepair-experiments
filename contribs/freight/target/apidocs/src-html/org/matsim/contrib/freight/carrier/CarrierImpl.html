114        @Override
115        public CarrierPlan createCopyOfSelectedPlanAndMakeSelected() {
116                CarrierPlan newPlan = CarrierImpl.copyPlan(this.selectedPlan) ;
117                this.setSelectedPlan( newPlan ) ;
118                return newPlan ;
119        }
120
121        public static CarrierPlan copyPlan(CarrierPlan plan2copy) {
122                List<ScheduledTour> tours = new ArrayList<ScheduledTour>();
123                for (ScheduledTour sTour : plan2copy.getScheduledTours()) {
124                        double depTime = sTour.getDeparture();
125                        CarrierVehicle vehicle = sTour.getVehicle();
126                        Tour tour = sTour.getTour().duplicate();
127                        tours.add(ScheduledTour.newInstance(tour, vehicle, depTime));
128                }
129                CarrierPlan copiedPlan = new CarrierPlan(plan2copy.getCarrier(), tours);
130                double initialScoreOfCopiedPlan = plan2copy.getScore();
131                copiedPlan.setScore(initialScoreOfCopiedPlan);
132                return copiedPlan;
133        
134        }
135
136        @Override
137        public boolean removePlan(CarrierPlan p) {
138                return this.plans.remove(p);
139        }
