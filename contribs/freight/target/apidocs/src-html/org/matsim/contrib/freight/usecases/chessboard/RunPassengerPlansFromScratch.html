001package org.matsim.contrib.freight.usecases.chessboard;
002
003import org.matsim.core.config.Config;
004import org.matsim.core.config.groups.PlanCalcScoreConfigGroup.ActivityParams;
005import org.matsim.core.config.groups.StrategyConfigGroup.StrategySettings;
006import org.matsim.core.controler.Controler;
007import org.matsim.core.controler.OutputDirectoryHierarchy;
008
009public class RunPassengerPlansFromScratch {
010        
011        public static void main(String[] args) {
012                String NETWORK_FILENAME = "input/usecases/chessboard/network/grid9x9.xml";
013                String PLANS_FILENAME = "input/usecases/chessboard/passenger/passengerPlans.xml";
014                Config config = new Config();
015                config.addCoreModules();
016                
017                ActivityParams workParams = new ActivityParams("work");
018                workParams.setTypicalDuration(60 * 60 * 8);
019                config.planCalcScore().addActivityParams(workParams);
020                ActivityParams homeParams = new ActivityParams("home");
021                homeParams.setTypicalDuration(16 * 60 * 60);
022                config.planCalcScore().addActivityParams(homeParams);
023                config.global().setCoordinateSystem("EPSG:32632");
024                config.controler().setFirstIteration(0);
025                config.controler().setLastIteration(2);
026                config.network().setInputFile(NETWORK_FILENAME);
027                config.plans().setInputFile(PLANS_FILENAME);
028                
029                StrategySettings bestScore = new StrategySettings();
030                bestScore.setStrategyName("BestScore");
031                bestScore.setWeight(0.5);
032                
033                StrategySettings reRoute = new StrategySettings();
034                reRoute.setStrategyName("ReRoute");
035                reRoute.setWeight(0.5);
036//              reRoute.setDisableAfter(300);
037                
038                config.strategy().setMaxAgentPlanMemorySize(5);
039                config.strategy().addStrategySettings(bestScore);
040                config.strategy().addStrategySettings(reRoute);
041//              
042                Controler controler = new Controler(config);
043                controler.getConfig().controler().setWriteEventsInterval(1);
044        controler.getConfig().controler().setCreateGraphs(false);
045                //Select how to react of not empty output directory
046        controler.getConfig().controler().setOverwriteFileSetting(OutputDirectoryHierarchy.OverwriteFileSetting.overwriteExistingFiles);
047//              controler.getConfig().controler().setOverwriteFileSetting(OutputDirectoryHierarchy.OverwriteFileSetting.failIfDirectoryExists);
048
049                controler.run();
050
051        }
052
053}
