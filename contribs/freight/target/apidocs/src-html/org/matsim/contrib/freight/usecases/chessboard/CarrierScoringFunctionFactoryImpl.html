179    static class DriversLegScoring implements SumScoringFunction.BasicScoring, SumScoringFunction.LegScoring {
180
181        private double score = 0.0;
182
183        private final Network network;
184
185        private final Carrier carrier;
186
187        private Set<CarrierVehicle> employedVehicles;
188
189        public DriversLegScoring(Carrier carrier, Network network) {
190            super();
191            this.network = network;
192            this.carrier = carrier;
193            employedVehicles = new HashSet<CarrierVehicle>();
194        }
195
196
197        @Override
198        public void finish() {
199
200        }
201
202
203        @Override
204        public double getScore() {
205            return score;
206        }
207
208        private double getTimeParameter(CarrierVehicle vehicle) {
209            return vehicle.getVehicleType().getVehicleCostInformation().perTimeUnit;
210        }
211
212
213        private double getDistanceParameter(CarrierVehicle vehicle) {
214            return vehicle.getVehicleType().getVehicleCostInformation().perDistanceUnit;
215        }
216
217
218        private CarrierVehicle getVehicle(Id vehicleId) {
219            for(CarrierVehicle cv : carrier.getCarrierCapabilities().getCarrierVehicles()){
220                if(cv.getVehicleId().equals(vehicleId)){
221                    return cv;
222                }
223            }
224            return null;
225        }
226
227        @Override
228        public void handleLeg(Leg leg) {
229            if(leg.getRoute() instanceof NetworkRoute){
230                NetworkRoute nRoute = (NetworkRoute) leg.getRoute();
231                Id vehicleId = nRoute.getVehicleId();
232                CarrierVehicle vehicle = getVehicle(vehicleId);
233                if(vehicle == null) throw new IllegalStateException("vehicle with id " + vehicleId + " is missing");
234                if(!employedVehicles.contains(vehicle)){
235                    employedVehicles.add(vehicle);
236                }
237                double distance = 0.0;
238                double toll = 0.;
239                if(leg.getRoute() instanceof NetworkRoute){
240                    Link startLink = network.getLinks().get(leg.getRoute().getStartLinkId());
241                    distance += startLink.getLength();
242                    for(Id linkId : ((NetworkRoute) leg.getRoute()).getLinkIds()){
243                        distance += network.getLinks().get(linkId).getLength();
244
245                    }
246                    distance += network.getLinks().get(leg.getRoute().getEndLinkId()).getLength();
247
248                }
249
250                double distanceCosts = distance*getDistanceParameter(vehicle);
251                assert distanceCosts >= 0.0 : "distanceCosts must be positive";
252                score += (-1) * distanceCosts;
253                double timeCosts = leg.getTravelTime()*getTimeParameter(vehicle);
254                assert timeCosts >= 0.0 : "timeCosts must be positive";
255                score += (-1) * timeCosts;
256
257            }
258        }
259
260    }

263    static class TollScoring implements SumScoringFunction.BasicScoring, SumScoringFunction.ArbitraryEventScoring {
264
265        private double score = 0.;
266
267        private Carrier carrier;
268
269        private Network network;
270
271        private VehicleTypeDependentRoadPricingCalculator roadPricing;
272
273        public TollScoring(Carrier carrier, Network network, VehicleTypeDependentRoadPricingCalculator roadPricing) {
274            this.carrier = carrier;
275            this.roadPricing = roadPricing;
276            this.network = network;
277        }
278
279        @Override
280        public void handleEvent(Event event) {
281            if(event instanceof LinkEnterEvent){
282                CarrierVehicle carrierVehicle = getVehicle(((LinkEnterEvent) event).getVehicleId());
283                if(carrierVehicle == null) throw new IllegalStateException("carrier vehicle missing");
284                double toll = roadPricing.getTollAmount(carrierVehicle.getVehicleType().getId(),network.getLinks().get(((LinkEnterEvent) event).getLinkId()),event.getTime());
285                if(toll > 0.) System.out.println("bing: vehicle " + carrierVehicle.getVehicleId() + " paid toll " + toll + "");
286                score += (-1) * toll;
287            }
288        }
289
290        private CarrierVehicle getVehicle(Id<Vehicle> vehicleId) {
291            for(CarrierVehicle v : carrier.getCarrierCapabilities().getCarrierVehicles()){
292                if(v.getVehicleId().equals(vehicleId)){
293                    return v;
294                }
295            }
296            return null;
297        }
298
299        @Override
300        public void finish() {
301
302        }
303
304        @Override
305        public double getScore() {
306            return score;
307        }
308    }
