013package org.matsim.contrib.freight.jsprit;
014
015import java.util.ArrayList;
016import java.util.Collection;
017import java.util.Collections;
018import java.util.HashMap;
019import java.util.Map;
020
021import com.graphhopper.jsprit.core.problem.vehicle.VehicleType;
022
023import org.matsim.api.core.v01.Id;
024import org.matsim.api.core.v01.network.Link;
025import org.matsim.roadpricing.RoadPricingScheme;
026import org.matsim.roadpricing.RoadPricingSchemeImpl.Cost;
027
028
029/**
030 * Calculator that manages and calculates vehicle type dependent road pricing schemas.
031 * 
032 * @author stefan schr√∂der
033 *
034 */
035public class VehicleTypeDependentRoadPricingCalculator {
036        
037        static interface TollCalculator {
038                double getTollAmount(Cost cost, Link link);
039        }
040        
041        static class CordonCalc implements TollCalculator {
042
043                
044                @Override
045                public double getTollAmount(Cost cost, Link link) {
046                        if(cost == null) return 0.0;
047                        return cost.amount;
048                }
049                
050        }
051        
052        static class DistanceCalc implements TollCalculator{
053
054                @Override
055                public double getTollAmount(Cost cost, Link link) {
056                        if(cost == null) return 0.0;
057                        return cost.amount*link.getLength();
058                }
059                
060        }
061        
062        
063        private Map<Id<org.matsim.vehicles.VehicleType>, Collection<RoadPricingScheme>> schemes = new HashMap<>();
064
065        private Map<String,TollCalculator> calculators = new HashMap<String, TollCalculator>();
066        
067        /**
068         * Gets an unmodifiable list of {@link RoadPricingScheme} for a {@link VehicleType} with input-id.
069         */
070        public Collection<RoadPricingScheme> getPricingSchemes(Id<org.matsim.vehicles.VehicleType> vehicleType){
071                Collection<RoadPricingScheme> collection = schemes.get(vehicleType);
072                if(collection == null) return Collections.unmodifiableCollection(Collections.<RoadPricingScheme>emptyList());
073                return Collections.unmodifiableCollection(collection);
074        }
075        
076        /**
077         * Gets the toll-amount for vehicleType on {@link Link} link at time 'time'.
078         * 
079         * <p>In case of <code>RoadPricingScheme.TOLL_TYPE_CORDON</code> the toll amount on a link is equivalent to <code>rps.getLinkCostInfo(...).amount</code>.
080         * 
081         * <p>In case of <code>RoadPricingScheme.TOLL_TYPE_DISTANCE</code> the toll amount on a link is equivalent to <code>rps.getLinkCostInfo(...).amount*link.getLength()</code>.
082         * 
083         * @param vehicleType
084         * @param link
085         * @param time
086         * @return
087         */
088        public double getTollAmount(Id<org.matsim.vehicles.VehicleType> vehicleType, Link link, double time){ 
089                Collection<RoadPricingScheme> pricingSchemes = getPricingSchemes(vehicleType);
090                if(pricingSchemes == null) return 0.0; 
091                double toll = 0.0;
092                for(RoadPricingScheme rps : pricingSchemes){
093                        Cost linkCostInfo = rps.getLinkCostInfo(link.getId(), time, null, null);
094                        toll += calculators.get(rps.getType()).getTollAmount(linkCostInfo,link);
095                }
096                return toll; 
097        }
098        
099        
100        public void addPricingScheme(String vehicleTypeId, RoadPricingScheme pricingScheme){
101                Id<org.matsim.vehicles.VehicleType> typeId = Id.create(vehicleTypeId, org.matsim.vehicles.VehicleType.class);
102                addPricingScheme(typeId, pricingScheme);
103        }
104        
105        /**
106         * Adds a {@link RoadPricingScheme} for vehicleTypeId.
107         * 
108         * <p>Currently, only Cordon and Distance tolls can be calculated.
109         * 
110         * <p>In case of <code>RoadPricingScheme.TOLL_TYPE_CORDON</code> the toll amount on a link is equivalent to <code>rps.getLinkCostInfo(...).amount</code>.
111         * 
112         * <p>In case of <code>RoadPricingScheme.TOLL_TYPE_DISTANCE</code> the toll amount on a link is equivalent to <code>rps.getLinkCostInfo(...).amount*link.getLength()</code>.
113         * 
114         * 
115         * @param vehicleTypeId
116         * @param {@link RoadPricingScheme}
117         */
118        public void addPricingScheme(Id<org.matsim.vehicles.VehicleType> vehicleTypeId, RoadPricingScheme pricingScheme){
119                if(!schemes.containsKey(vehicleTypeId)){
120                        schemes.put(vehicleTypeId, new ArrayList<RoadPricingScheme>());
121                }
122                schemes.get(vehicleTypeId).add(pricingScheme);
123                if(pricingScheme.getType().equals(RoadPricingScheme.TOLL_TYPE_CORDON)){
124                        calculators.put(pricingScheme.getType(), new CordonCalc());
125                }
126                else if(pricingScheme.getType().equals(RoadPricingScheme.TOLL_TYPE_DISTANCE)){
127                        calculators.put(pricingScheme.getType(), new DistanceCalc());
128                }
129        else throw new IllegalStateException("toll type specification missing. if you use xml add specification to root node like this\n"
130            + "<roadpricing type=\"cordon\" name=\"cordon\"> if it is a cordon toll and \n<roadpricing type=\"distance\" name=\"distance\"> if it is a distance toll");
131        }
132        
133        /**
134         * Removes {@link RoadPricingScheme} for vehicleTypeId.
135         * 
136         * @param vehicleTypeId
137         */
138        public void removePricingSchemes(Id<org.matsim.vehicles.VehicleType> vehicleTypeId){
139                schemes.remove(vehicleTypeId);
140        }
141
142        /**
143         * Checks whether there is a {@link RoadPricingScheme} for vehicleType 'typeId'.
144         * 
145         * @param typeId
146         * @return true if there is a pricing-scheme, false otherwise
147         */
148        public boolean containsKey(Id<org.matsim.vehicles.VehicleType> typeId) {
149                return schemes.containsKey(typeId);
150        }
151        
152        /**
153         * Gets an unmodifiable map of pricing schemes.
154         */
155        public Map<Id<org.matsim.vehicles.VehicleType>,Collection<RoadPricingScheme>> getSchemes(){
156                return Collections.unmodifiableMap(schemes);
157        }
158}
