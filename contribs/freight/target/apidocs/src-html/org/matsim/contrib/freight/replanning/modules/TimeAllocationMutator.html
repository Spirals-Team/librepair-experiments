019package org.matsim.contrib.freight.replanning.modules;
020
021import java.util.ArrayList;
022import java.util.Collection;
023
024import org.matsim.contrib.freight.carrier.CarrierPlan;
025import org.matsim.contrib.freight.carrier.ScheduledTour;
026import org.matsim.core.gbl.MatsimRandom;
027import org.matsim.core.replanning.ReplanningContext;
028import org.matsim.core.replanning.modules.GenericPlanStrategyModule;
029
030/**
031 * @author nagel
032 *
033 */
034public class TimeAllocationMutator implements GenericPlanStrategyModule<CarrierPlan> {
035
036    private double probability = 1.;
037
038    /**
039     * max. departure time mutation: +/- 0.5 * mutationRange
040     */
041    private double mutationRange = 3600. * 3.;
042
043    public TimeAllocationMutator() {
044    }
045
046    public TimeAllocationMutator(double probability, double mutationRange) {
047        this.probability = probability;
048        this.mutationRange = mutationRange;
049    }
050
051    @Override
052        public void handlePlan(CarrierPlan carrierPlan) {
053                Collection<ScheduledTour> newTours = new ArrayList<ScheduledTour>() ;
054                for ( ScheduledTour tour : carrierPlan.getScheduledTours() ) {
055            if(MatsimRandom.getRandom().nextDouble() < probability) {
056                double departureTime = tour.getDeparture() + (MatsimRandom.getRandom().nextDouble() - 0.5) * mutationRange;
057                if ( departureTime < tour.getVehicle().getEarliestStartTime() ) {
058                    departureTime = tour.getVehicle().getEarliestStartTime();
059                }
060                newTours.add(ScheduledTour.newInstance(tour.getTour(), tour.getVehicle(), departureTime));
061            }
062            else newTours.add(tour);
063                }
064                carrierPlan.getScheduledTours().clear(); 
065                carrierPlan.getScheduledTours().addAll( newTours ) ;
066        }
067
068        @Override
069        public void prepareReplanning(ReplanningContext replanningContext) {
070        }
071
072        @Override
073        public void finishReplanning() {
074        }
075
076}
