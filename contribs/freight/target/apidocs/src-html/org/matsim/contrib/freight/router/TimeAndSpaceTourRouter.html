024/**
025 * Router routing scheduledTours.
026 * 
027 * @author sschroeder
028 *
029 */
030public class TimeAndSpaceTourRouter {
031        
032        static class MatsimVehicleAdapter implements Vehicle {
033
034                private CarrierVehicle carrierVehicle;
035
036                public MatsimVehicleAdapter(CarrierVehicle vehicle) {
037                        this.carrierVehicle = vehicle;
038                }
039
040                @Override
041                public Id<Vehicle> getId() {
042                        return carrierVehicle.getVehicleId();
043                }
044
045                @Override
046                public VehicleType getType() {
047                        return carrierVehicle.getVehicleType();
048                }
049
050                public CarrierVehicle getCarrierVehicle() {
051                        return carrierVehicle;
052                }
053
054        }
055        
056        
057        @SuppressWarnings("unused")
058        private static Logger logger = Logger.getLogger(TimeAndSpaceTourRouter.class);
059        
060        private LeastCostPathCalculator router;
061        
062        private Network network;
063        
064        private TravelTime travelTime;
065
066        /**
067         * Constructs the timeAndSpaceRouter with a leastCostPathCalculator, network and travelTime.
068         * @param router
069         * @param network
070         * @param travelTime
071         * @see LeastCostPathCalculator, Network, TravelTime
072         */
073        public TimeAndSpaceTourRouter(LeastCostPathCalculator router, Network network, TravelTime travelTime) {
074                super();
075                this.router = router;
076                this.network = network;
077                this.travelTime = travelTime;
078        }
079        
080        /**
081         * Routes a scheduledTour in time and space.
082         * 
083         * <p>Uses a leastCostPathCalculator to calculate a route/path from one activity to another. It starts at the departureTime of 
084         * the scheduledTour and determines activity arrival and departure times considering activities time-windows.
085         * @param tour
086         */
087        public void route(ScheduledTour tour) {
088                MatsimVehicleAdapter matsimVehicle = new MatsimVehicleAdapter(tour.getVehicle());
089                double currTime = tour.getDeparture();
090                Id<Link> prevLink = tour.getTour().getStartLinkId();
091                Leg prevLeg = null;
092                for(TourElement e : tour.getTour().getTourElements()){
093                        if(e instanceof Leg){
094                                prevLeg = (Leg) e;
095                                prevLeg.setDepartureTime(currTime);
096                        }
097                        if(e instanceof TourActivity){
098                                TourActivity act = (TourActivity) e;
099                                route(prevLeg, prevLink, act.getLocation(), null, matsimVehicle);
100                                double expectedArrival = currTime + prevLeg.getExpectedTransportTime();
101                                act.setExpectedArrival(expectedArrival);
102                                double startAct = Math.max(expectedArrival, act.getTimeWindow().getStart()); 
103                                currTime = startAct + act.getDuration();
104                                prevLink = act.getLocation();
105                        }
106                }
107                Id<Link> endLink = tour.getTour().getEndLinkId();
108                route(prevLeg,prevLink,endLink, null, matsimVehicle);
109        }
110        
111        private void route(Leg prevLeg, Id<Link> fromLinkId, Id<Link> toLinkId, Person person, Vehicle vehicle) {
112                if(fromLinkId.equals(toLinkId)){
113                        prevLeg.setExpectedTransportTime(0);
114                        NetworkRoute route = RouteUtils.createLinkNetworkRouteImpl(fromLinkId, toLinkId);
115                        route.setDistance(0.0);
116                        route.setTravelTime(0.0);
117//                      route.setVehicleId(vehicle.getId());
118                        prevLeg.setRoute(route);
119                        return;
120                }
121                Path path = router.calcLeastCostPath(network.getLinks().get(fromLinkId).getToNode(), network.getLinks().get(toLinkId).getFromNode(), prevLeg.getExpectedDepartureTime(), person, vehicle);
122                double travelTime = path.travelTime;
123                
124                /*
125                 *ACHTUNG. Konsistenz zu VRP 
126                 */
127                double toLinkTravelTime = this.travelTime.getLinkTravelTime(network.getLinks().get(toLinkId),prevLeg.getExpectedDepartureTime()+travelTime, person, vehicle);
128                travelTime += toLinkTravelTime;
129                prevLeg.setExpectedTransportTime(travelTime);
130                NetworkRoute route = createRoute(fromLinkId,path,toLinkId);
131//              route.setVehicleId(vehicle.getId());
132                prevLeg.setRoute(route);
133        }
134        
135        private NetworkRoute createRoute(Id<Link> fromLink, Path path, Id<Link> toLink) {
136                NetworkRoute route = RouteUtils.createLinkNetworkRouteImpl(fromLink, toLink);
137                route.setLinkIds(fromLink, getLinkIds(path.links), toLink);
138                return route;
139        }
140        
141        private List<Id<Link>> getLinkIds(List<Link> links) {
142                List<Id<Link>> linkIds = new ArrayList<Id<Link>>();
143                for(Link l : links){
144                        linkIds.add(l.getId());
145                }
146                return linkIds;
147        }
