001package org.matsim.contrib.freight.jsprit;
002
003import org.matsim.api.core.v01.network.Link;
004import org.matsim.api.core.v01.population.Person;
005import org.matsim.core.router.util.TravelTime;
006import org.matsim.vehicles.Vehicle;
007
008public class FiFoTravelTime implements TravelTime {
009
010        private TravelTime travelTime;
011        
012        private int binSize;
013        
014        public FiFoTravelTime(TravelTime travelTime, int binSize) {
015                super();
016                this.travelTime = travelTime;
017                this.binSize = binSize;
018        }
019
020        public double getLinkTravelTime(Link link, double time, Vehicle vehicle) {
021                double tt = getTravelTime(link, time, vehicle);
022                if(getTimeBin(time) == getTimeBin(time+tt)){
023                        return tt;
024                }
025                else{
026                        double totalTravelTime = 0.0;
027                        double distanceToTravel = link.getLength();
028                        double currentTime = time;
029                        boolean distanceTraveled = false;
030                        while(!distanceTraveled){
031                                double nextTimeThreshold = getNextTimeBin(currentTime);
032                                if(currentTime < nextTimeThreshold){
033                                        double speed = calculateCurrentSpeed(link,time, vehicle);
034                                        double maxReachableDistanceInThisTimeBin = (nextTimeThreshold-currentTime)*speed;
035                                        if(distanceToTravel > maxReachableDistanceInThisTimeBin){
036                                                distanceToTravel = distanceToTravel - maxReachableDistanceInThisTimeBin;
037                                                totalTravelTime += (nextTimeThreshold-currentTime);
038                                                currentTime = nextTimeThreshold;
039                                        }
040                                        else{ //<= maxReachableDistance
041                                                totalTravelTime += distanceToTravel/speed;
042                                                distanceTraveled = true;
043                                        }
044                                }
045                        }
046                        return totalTravelTime;
047                }
048        }
049
050        private double getTravelTime(Link link, double time, Vehicle vehicle) {
051                double linkTravelTime = travelTime.getLinkTravelTime(link, time, null, vehicle);
052                return linkTravelTime;
053        }
054                
055        private double calculateCurrentSpeed(Link link, double time, Vehicle vehicle) {
056                double speed = link.getLength()/getTravelTime(link, time, vehicle);
057                if(speed > vehicle.getType().getMaximumVelocity()){
058                        speed = vehicle.getType().getMaximumVelocity();
059                }
060                return speed;
061        }
062        
063        private double getTimeBin(double currentTime){
064                return (int)currentTime/binSize;
065        }
066
067        private double getNextTimeBin(double currentTime) {
068                double lastThreshold = Math.floor(currentTime/binSize) * binSize;
069                return lastThreshold + binSize;
070        }
071
072        @Override
073        public double getLinkTravelTime(Link link, double time, Person person, Vehicle vehicle) {
074                return getLinkTravelTime(link, time, vehicle);
075        }
076}
