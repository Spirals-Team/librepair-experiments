001package org.matsim.contrib.freight.usecases.chessboard;
002
003import org.matsim.api.core.v01.Scenario;
004import org.matsim.api.core.v01.network.Network;
005import org.matsim.contrib.freight.carrier.*;
006import org.matsim.contrib.freight.controler.CarrierModule;
007import org.matsim.contrib.freight.replanning.CarrierPlanStrategyManagerFactory;
008import org.matsim.contrib.freight.scoring.CarrierScoringFunctionFactory;
009import org.matsim.contrib.freight.usecases.analysis.CarrierScoreStats;
010import org.matsim.contrib.freight.usecases.analysis.LegHistogram;
011import org.matsim.contrib.freight.usecases.chessboard.CarrierScoringFunctionFactoryImpl.DriversActivityScoring;
012import org.matsim.contrib.freight.usecases.chessboard.CarrierScoringFunctionFactoryImpl.DriversLegScoring;
013import org.matsim.core.api.experimental.events.EventsManager;
014import org.matsim.core.config.Config;
015import org.matsim.core.config.ConfigUtils;
016import org.matsim.core.controler.Controler;
017import org.matsim.core.controler.MatsimServices;
018import org.matsim.core.controler.listener.IterationEndsListener;
019import org.matsim.core.replanning.GenericPlanStrategyImpl;
020import org.matsim.core.replanning.GenericStrategyManager;
021import org.matsim.core.replanning.selectors.BestPlanSelector;
022import org.matsim.core.scenario.ScenarioUtils;
023import org.matsim.core.scoring.SumScoringFunction;
024
025public class RunPassengerAlongWithCarriers {
026
027    public static void main(String[] args) {
028
029
030        String configFile = "input/usecases/chessboard/passenger/config.xml";
031        Config config = ConfigUtils.loadConfig(configFile);
032        Scenario scenario = ScenarioUtils.loadScenario(config);
033
034        Controler controler = new Controler(config);
035        final Carriers carriers = new Carriers();
036        new CarrierPlanXmlReaderV2(carriers).readFile("input/usecases/chessboard/freight/carrierPlans.xml");
037
038        CarrierVehicleTypes types = new CarrierVehicleTypes();
039        new CarrierVehicleTypeReader(types).readFile("input/usecases/chessboard/freight/vehicleTypes.xml");
040        new CarrierVehicleTypeLoader(carriers).loadVehicleTypes(types);
041
042        CarrierPlanStrategyManagerFactory strategyManagerFactory = createStrategyManagerFactory(types, controler);
043        CarrierScoringFunctionFactory scoringFunctionFactory = createScoringFunctionFactory(scenario.getNetwork());
044
045        CarrierModule carrierController = new CarrierModule(carriers, strategyManagerFactory, scoringFunctionFactory);
046
047        controler.addOverridingModule(carrierController);
048        prepareFreightOutputDataAndStats(scenario, controler.getEvents(), controler, carriers);
049
050        controler.run();
051
052    }
053
054
055    private static void prepareFreightOutputDataAndStats(Scenario scenario, EventsManager eventsManager, MatsimServices controler, final Carriers carriers) {
056        final LegHistogram freightOnly = new LegHistogram(900);
057        freightOnly.setPopulation(scenario.getPopulation());
058        freightOnly.setInclPop(false);
059        final LegHistogram withoutFreight = new LegHistogram(900);
060        withoutFreight.setPopulation(scenario.getPopulation());
061
062        CarrierScoreStats scores = new CarrierScoreStats(carriers, "output/carrier_scores", true);
063
064        eventsManager.addHandler(withoutFreight);
065        eventsManager.addHandler(freightOnly);
066        controler.addControlerListener(scores);
067        controler.addControlerListener((IterationEndsListener) event -> {
068            //write plans
069            String dir = event.getServices().getControlerIO().getIterationPath(event.getIteration());
070            new CarrierPlanXmlWriterV2(carriers).write(dir + "/" + event.getIteration() + ".carrierPlans.xml");
071
072            //write stats
073            freightOnly.writeGraphic(dir + "/" + event.getIteration() + ".legHistogram_freight.png");
074            freightOnly.reset(event.getIteration());
075
076            withoutFreight.writeGraphic(dir + "/" + event.getIteration() + ".legHistogram_withoutFreight.png");
077            withoutFreight.reset(event.getIteration());
078        });
079    }
080
081
082    private static CarrierScoringFunctionFactory createScoringFunctionFactory(final Network network) {
083        return carrier -> {
084            SumScoringFunction sf = new SumScoringFunction();
085            DriversLegScoring driverLegScoring = new DriversLegScoring(carrier, network);
086            DriversActivityScoring actScoring = new DriversActivityScoring();
087            sf.addScoringFunction(driverLegScoring);
088            sf.addScoringFunction(actScoring);
089            return sf;
090        };
091    }
092
093
094    private static CarrierPlanStrategyManagerFactory createStrategyManagerFactory(final CarrierVehicleTypes types, final MatsimServices controler) {
095        return () -> {
096            final GenericStrategyManager<CarrierPlan, Carrier> strategyManager = new GenericStrategyManager<>();
097            strategyManager.addStrategy(new GenericPlanStrategyImpl<>(new BestPlanSelector<>()), null, 0.95);
098            strategyManager.addStrategy(new SelectBestPlanAndOptimizeItsVehicleRouteFactory(controler.getScenario().getNetwork(), types, controler.getLinkTravelTimes()).createStrategy(), null, 0.05);
099            return strategyManager;
100        };
101    }
102
103}
