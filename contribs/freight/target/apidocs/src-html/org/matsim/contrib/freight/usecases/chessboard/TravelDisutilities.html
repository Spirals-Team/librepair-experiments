001package org.matsim.contrib.freight.usecases.chessboard;
002
003import org.matsim.api.core.v01.Id;
004import org.matsim.api.core.v01.network.Link;
005import org.matsim.api.core.v01.population.Person;
006import org.matsim.contrib.freight.carrier.CarrierVehicleType;
007import org.matsim.contrib.freight.carrier.CarrierVehicleTypes;
008import org.matsim.contrib.freight.jsprit.VehicleTypeDependentRoadPricingCalculator;
009import org.matsim.core.router.util.TravelDisutility;
010import org.matsim.core.router.util.TravelTime;
011import org.matsim.vehicles.Vehicle;
012import org.matsim.vehicles.VehicleType;
013
014public class TravelDisutilities {
015        
016        public static TravelDisutility createBaseDisutility(final CarrierVehicleTypes vehicleTypes, final TravelTime travelTime){
017                
018                return new TravelDisutility() {
019
020                        @Override
021                        public double getLinkTravelDisutility(Link link, double time, Person person, org.matsim.vehicles.Vehicle vehicle) {
022                                CarrierVehicleType type = vehicleTypes.getVehicleTypes().get(vehicle.getType().getId());
023                                if(type == null) throw new IllegalStateException("vehicle "+vehicle.getId()+" has no type");
024                                double tt = travelTime.getLinkTravelTime(link, time, person, vehicle);
025                                return type.getVehicleCostInformation().perDistanceUnit*link.getLength() + type.getVehicleCostInformation().perTimeUnit*tt;
026                        }
027
028                        @Override
029                        public double getLinkMinimumTravelDisutility(Link link) {
030                                double minDisutility = Double.MAX_VALUE;
031                                double free_tt = link.getLength()/link.getFreespeed();
032                                for(CarrierVehicleType type : vehicleTypes.getVehicleTypes().values()){
033                                        double disu = type.getVehicleCostInformation().perDistanceUnit*link.getLength() + type.getVehicleCostInformation().perTimeUnit*free_tt;
034                                        if(disu < minDisutility) minDisutility=disu;
035                                }
036                                return minDisutility;
037                        }
038                };
039        }
040
041
042    public static TravelDisutility withToll(final TravelDisutility base, final VehicleTypeDependentRoadPricingCalculator roadPricing){
043
044        return new TravelDisutility() {
045
046            @Override
047            public double getLinkTravelDisutility(Link link, double time, Person person, Vehicle vehicle) {
048                double costs = base.getLinkTravelDisutility(link, time, person, vehicle);
049                Id<VehicleType> typeId = vehicle.getType().getId();
050                double toll = roadPricing.getTollAmount(typeId, link, time);
051                return costs + toll;
052            }
053
054            @Override
055            public double getLinkMinimumTravelDisutility(Link link) {
056                return base.getLinkMinimumTravelDisutility(link);
057            }
058
059        };
060
061    }
062        
063
