026
027public class PassengerScenarioCreator {
028        
029        static int agentCounter = 1;
030        
031        static int nuOfAgentsPerHomeLink = 1;
032        
033        public static void main(String[] args) {
034                
035                Config config = new Config();
036                config.addCoreModules();
037                Scenario scenario = ScenarioUtils.createScenario(config);
038                new MatsimNetworkReader(scenario.getNetwork()).readFile("input/usecases/chessboard/network/grid9x9.xml");
039                
040                Population population = scenario.getPopulation(); 
041                
042                for(int i=1;i<10;i++){
043                        Id<Link> homeId = Id.create("i("+i+",9)R", Link.class);
044                        Id<Link> workId = Id.create("i("+i+",0)", Link.class);
045                        List<Person> persons = createPersons(homeId,workId,scenario);
046                        for(Person p : persons) population.addPerson(p);
047
048                        Id<Link> homeIdR = Id.create("i("+i+",0)", Link.class);
049                        Id<Link> workIdR = Id.create("i("+i+",9)R", Link.class);
050                        List<Person> personsR = createPersons(homeIdR,workIdR,scenario);
051                        for(Person p : personsR) population.addPerson(p);
052                }
053                
054                for(int i=1;i<10;i++){
055                        Id<Link> homeId = Id.create("j(0,"+i+")R", Link.class);
056                        Id<Link> workId = Id.create("j(9,"+i+")", Link.class);
057                        List<Person> persons = createPersons(homeId,workId,scenario);
058                        for(Person p : persons) population.addPerson(p);
059                        
060                        Id<Link> homeIdR = Id.create("j(9,"+i+")", Link.class);
061                        Id<Link> workIdR = Id.create("j(0,"+i+")R", Link.class);
062                        List<Person> personsR = createPersons(homeIdR,workIdR,scenario);
063                        for(Person p : personsR) population.addPerson(p);
064                }
065                
066                new PopulationWriter(population, scenario.getNetwork()).write("input/usecases/chessboard/passenger/passengerPlansV2.xml");
067        }
068
069        private static List<Person> createPersons(Id<Link> homeId, Id<Link> workId,Scenario scenario) {
070                LeastCostPathCalculator lcpa = new DijkstraFactory().createPathCalculator(scenario.getNetwork(), 
071                                new FreespeedTravelTimeAndDisutility(-1.0, -1.0, -1.0), new FreespeedTravelTimeAndDisutility(-1.0, -1.0, -1.0));
072        PopulationFactory popFactory = (PopulationFactory) scenario.getPopulation().getFactory();
073                List<Person> persons = new ArrayList<Person>();
074                for(int agent=0;agent<nuOfAgentsPerHomeLink;agent++){
075                        
076                        Person person = popFactory.createPerson(Id.create(agentCounter, Person.class));
077                        agentCounter++;
078                        Plan plan = popFactory.createPlan();
079                        plan.setPerson(person);
080
081                        Activity act1 = popFactory.createActivityFromLinkId("home", homeId);
082                        act1.setEndTime(8*60*60);
083                        plan.addActivity(act1);
084                        Leg leg1 = popFactory.createLeg(TransportMode.car);
085                        Path path1 = lcpa.calcLeastCostPath(scenario.getNetwork().getLinks().get(homeId).getToNode(), scenario.getNetwork().getLinks().get(workId).getFromNode(), act1.getEndTime(), person, null);
086                        NetworkRoute linkNetworkRoute = RouteUtils.createLinkNetworkRouteImpl(homeId, getLinkIds(path1), workId);
087                        leg1.setRoute(linkNetworkRoute);
088                        plan.addLeg(leg1);
089
090                        Activity act2 = popFactory.createActivityFromLinkId("work", workId);
091                        act2.setMaximumDuration(8*60*60);
092                        plan.addActivity(act2);
093
094                        Leg leg2 = popFactory.createLeg(TransportMode.car);
095                        Path path2 = lcpa.calcLeastCostPath(scenario.getNetwork().getLinks().get(workId).getToNode(), scenario.getNetwork().getLinks().get(homeId).getFromNode(), act1.getEndTime(), person, null);
096                        NetworkRoute linkNetworkRoute2 = RouteUtils.createLinkNetworkRouteImpl(workId, getLinkIds(path2), homeId);
097                        leg2.setRoute(linkNetworkRoute2);
098                        plan.addLeg(leg2);
099
100                        plan.addActivity(popFactory.createActivityFromLinkId("home", homeId));
101
102                        person.addPlan(plan);
103                        person.setSelectedPlan(plan);
104                        persons.add(person);
105                }
106                return persons;
107        }
108
109        private static List<Id<Link>> getLinkIds(Path path1) {
110                List<Id<Link>> links = new ArrayList<Id<Link>>();
111                for(Link l : path1.links){ links.add(l.getId()); }
112                return links;
113        }
