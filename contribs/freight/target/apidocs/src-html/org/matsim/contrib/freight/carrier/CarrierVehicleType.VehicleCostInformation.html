001package org.matsim.contrib.freight.carrier;
002
003import org.matsim.api.core.v01.Id;
004import org.matsim.vehicles.EngineInformation;
005import org.matsim.vehicles.VehicleType;
006import org.matsim.vehicles.VehicleTypeImpl;
007
008/**
009 * The carrier vehicle type.
010 * 
011 * I decided to put vehicle cost information into the type (which is indeed not a physical attribute of the type). Thus physical and
012 * non physical attributes are used. This is likely to be changed in future.
013 * 
014 * @author sschroeder
015 *
016 */
017public class CarrierVehicleType extends ForwardingVehicleType {
018
019        /**
020         * A builder building the type.
021         * 
022         * @author sschroeder
023         *
024         */
025        public static class Builder {
026                
027                /**
028                 * Returns a new instance of builder initialized with the typeId.
029                 * 
030                 * The defaults are [fix=0.0][perDistanceUnit=1.0][perTimeUnit=0.0].
031                 * 
032                 * @param typeId
033                 * @return a type builder
034                 */
035                public static Builder newInstance(Id<VehicleType> typeId){
036                        return new Builder(typeId);
037                }
038                
039                private Id<VehicleType> typeId;
040                private double fix = 0.0;
041                private double perDistanceUnit = 1.0;
042                private double perTimeUnit = 0.0;
043                private String description;
044                private EngineInformation engineInfo;
045                private int capacity = 0;
046                private double maxVeloInMeterPerSeconds = Double.MAX_VALUE;
047                
048                
049                private Builder(Id<VehicleType> typeId){
050                        this.typeId = typeId;
051                }
052                
053                /**
054                 * Sets fixed costs of vehicle.
055                 * 
056                 * <p>By default it is 0.
057                 * @param fix
058                 * @return
059                 */
060                public Builder setFixCost(double fix){
061                        this.fix = fix;
062                        return this;
063                }
064                
065                /**
066                 * Sets costs per distance-unit.
067                 * 
068                 * <p>By default it is 1.
069                 * 
070                 * @param perDistanceUnit
071                 * @return
072                 */
073                public Builder setCostPerDistanceUnit(double perDistanceUnit){
074                        this.perDistanceUnit = perDistanceUnit;
075                        return this;
076                }
077                
078                /**
079                 * Sets costs per time-unit.
080                 * 
081                 * <p>By default it is 0.
082                 * 
083                 * @param perTimeUnit
084                 * @return
085                 */
086                public Builder setCostPerTimeUnit(double perTimeUnit){
087                        this.perTimeUnit = perTimeUnit;
088                        return this;
089                }
090                
091                /**
092                 * Sets description.
093                 * 
094                 * @param description
095                 * @return this builder
096                 */
097                public Builder setDescription(String description){
098                        this.description = description;
099                        return this;
100                }
101                
102                /**
103                 * Sets the capacity of vehicle-type.
104                 * 
105                 * <p>By defaul the capacity is 0.
106                 * 
107                 * @param capacity
108                 * @return this builder
109                 */
110                public Builder setCapacity(int capacity){
111                        this.capacity = capacity;
112                        return this;
113                }
114                
115                /**
116                 * Builds the type.
117                 * 
118                 * @return {@link CarrierVehicleType}
119                 */
120                public CarrierVehicleType build(){
121                        return new CarrierVehicleType(this);
122                }
123
124                /**
125                 * Sets {@link VehicleCostInformation}
126                 * 
127                 * <p>The defaults are [fix=0.0][perDistanceUnit=1.0][perTimeUnit=0.0].
128                 * 
129                 * @param info
130                 * @return this builder
131                 */
132                public Builder setVehicleCostInformation(VehicleCostInformation info) {
133                        fix = info.fix;
134                        perDistanceUnit = info.perDistanceUnit;
135                        perTimeUnit = info.perTimeUnit;
136                        return this;
137                }
138
139                /**
140                 * Sets {@link EngineInformation}
141                 * 
142                 * @param engineInfo
143                 * @return this builder
144                 */
145                public Builder setEngineInformation(EngineInformation engineInfo) {
146                        this.engineInfo = engineInfo;
147                        return this;
148                }
149
150                public Builder setMaxVelocity(double veloInMeterPerSeconds) {
151                        this.maxVeloInMeterPerSeconds  = veloInMeterPerSeconds;
152                        return this;
153                }
154        }
155        
156        public static class VehicleCostInformation {
157
158                public final double fix;
159                public final double perDistanceUnit;
160                public final double perTimeUnit;
161
162                public VehicleCostInformation(double fix, double perDistanceUnit, double perTimeUnit) {
163                        super();
164                        this.fix = fix;
165                        this.perDistanceUnit = perDistanceUnit;
166                        this.perTimeUnit = perTimeUnit;
167                }
168
169        }
170
171        private VehicleCostInformation vehicleCostInformation;
172
173        private int capacity;
174        
175        private CarrierVehicleType(Builder builder){
176                super(new VehicleTypeImpl(builder.typeId));
177                this.vehicleCostInformation = new VehicleCostInformation(builder.fix, builder.perDistanceUnit, builder.perTimeUnit);
178                if(builder.engineInfo != null) super.setEngineInformation(builder.engineInfo);
179                if(builder.description != null) super.setDescription(builder.description);
180                capacity = builder.capacity;
181                super.setMaximumVelocity(builder.maxVeloInMeterPerSeconds);
182        }
183
184        /**
185         * Returns the cost values for this vehicleType.
186         * 
187         * If cost values are not explicitly set, the defaults are [fix=0.0][perDistanceUnit=1.0][perTimeUnit=0.0].
188         * 
189         * @return vehicleCostInformation
190         */
191        public VehicleCostInformation getVehicleCostInformation() {
192                return vehicleCostInformation;
193        }
194        
195        /**
196         * Returns the capacity of carrierVehicleType.
197         * 
198         * <p>This might be replaced in future by a more complex concept of capacity (considering volume and different units).
199         * 
200         * @return integer
201         */
202        public int getCarrierVehicleCapacity(){
203                return capacity;
204        }
205        
206}
