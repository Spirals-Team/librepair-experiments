001package org.matsim.contrib.freight.usecases.chessboard;
002
003import java.util.ArrayList;
004import java.util.Arrays;
005import java.util.List;
006
007import org.matsim.api.core.v01.Id;
008import org.matsim.api.core.v01.network.Link;
009
010public class InnerOuterCityScenarioCreator {
011        
012
013        public List<Id<Link>> getInnerCityLinks(){
014                List<Id<Link>> innerCityLinks = new ArrayList<>();
015                /*
016                 * j(6,7)R, j(5,7), ..., j(3,7)[R|]
017                 * .
018                 * .
019                 * .
020                 * j(6,3), j(5,3), ..., j(3,3)
021                 * 
022                 * i(3,6),i(4,6), ..., i(7,6)
023                 * .
024                 * .
025                 * .
026                 * i(3,3),i(4,3),...., i(7,3)
027                 */
028                
029                for(int i=3;i<8;i++){
030                        for(int j=3;j<8;j++){
031                                innerCityLinks.add(Id.create("j("+i+","+j+")", Link.class));
032                                innerCityLinks.add(Id.create("j("+i+","+j+")R", Link.class));
033                                innerCityLinks.add(Id.create("i("+i+","+j+")R", Link.class));
034                                innerCityLinks.add(Id.create("i("+i+","+j+")", Link.class));
035                        }
036                }
037                return innerCityLinks;
038        }
039        
040        public List<Id<Link>> getAccessLinksToInnerCity(){
041                List<Id<Link>> accessLinks = new ArrayList<Id<Link>>();
042                List<String> linkStrings = Arrays.asList("j(4,7)R","j(6,7)R","j(5,3)","j(3,3)",
043                                "i(3,4)","i(3,6)","i(7,3)R","i(7,5)R");
044                for(String idString : linkStrings) accessLinks.add(Id.create(idString, Link.class));
045                return accessLinks;
046        }
047
048}
