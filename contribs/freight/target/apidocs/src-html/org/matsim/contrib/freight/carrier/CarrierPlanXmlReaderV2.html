001package org.matsim.contrib.freight.carrier;
002
003import org.apache.log4j.Logger;
004import org.matsim.core.api.internal.MatsimReader;
005
006/**
007 * A reader that reads carriers and their plans.
008 * 
009 * @author sschroeder
010 *
011 */
012public class CarrierPlanXmlReaderV2 implements MatsimReader {
013
014        private static Logger logger = Logger.getLogger(CarrierPlanXmlReaderV2.class);
015        
016        private CarrierPlanXmlParserV2 delegate;
017
018        /**
019         * Constructs a reader with an empty carriers-container for the carriers to be constructed. 
020         * 
021         * @param carriers which is a map that stores carriers
022         */
023        public CarrierPlanXmlReaderV2(Carriers carriers) {
024                this.delegate = new CarrierPlanXmlParserV2(carriers);
025        }
026        
027        /**
028         * Reads a xml-file that contains carriers and their plans.
029         * 
030         * <p> Builds carriers and plans, and stores them in the carriers-object coming with this constructor.
031         * 
032         * @param filename
033         */
034        @Override
035        public void readFile(String filename) {
036                logger.info("read carrier plans");
037                this.delegate.setValidating(false);
038                this.delegate.readFile(filename);
039                logger.info("done");
040
041        }
042
043}
