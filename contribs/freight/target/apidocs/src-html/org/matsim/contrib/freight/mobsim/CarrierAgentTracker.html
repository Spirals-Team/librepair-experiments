039/**
040 * This keeps track of all carrierAgents during simulation.
041 * 
042 * @author mzilske, sschroeder
043 *
044 */
045public class CarrierAgentTracker implements ActivityStartEventHandler, ActivityEndEventHandler, PersonDepartureEventHandler, PersonArrivalEventHandler,  LinkEnterEventHandler, VehicleEntersTrafficEventHandler, VehicleLeavesTrafficEventHandler {
046
047        private final Carriers carriers;
048
049        private final EventsManager eventsManager;
050        
051        private Vehicle2DriverEventHandler delegate = new Vehicle2DriverEventHandler();
052
053        private final Collection<CarrierAgent> carrierAgents = new ArrayList<CarrierAgent>();
054        
055        private Map<Id<Person>, CarrierAgent> driverAgentMap = new HashMap<Id<Person>, CarrierAgent>();
056
057        public CarrierAgentTracker(Carriers carriers, Network network, CarrierScoringFunctionFactory carrierScoringFunctionFactory) {
058                this.carriers = carriers;
059                createCarrierAgents(carrierScoringFunctionFactory);
060                eventsManager = EventsUtils.createEventsManager();
061        }
062
063        private void createCarrierAgents(CarrierScoringFunctionFactory carrierScoringFunctionFactory) {
064                for (Carrier carrier : carriers.getCarriers().values()) {
065                        ScoringFunction carrierScoringFunction = carrierScoringFunctionFactory.createScoringFunction(carrier);
066                        CarrierAgent carrierAgent = new CarrierAgent(this, carrier, carrierScoringFunction, delegate);
067                        carrierAgents.add(carrierAgent);
068                }
069        }
070
071        public EventsManager getEventsManager() {
072                return eventsManager;
073        }
074
075        /**
076         * Returns the entire set of selected carrier plans.
077         * 
078         * @return collection of plans
079         * @see Plan, CarrierPlan
080         */
081        Collection<MobSimVehicleRoute> createPlans() {
082                List<MobSimVehicleRoute> vehicleRoutes = new ArrayList<MobSimVehicleRoute>();
083                for (CarrierAgent carrierAgent : carrierAgents) {
084                        List<MobSimVehicleRoute> plansForCarrier = carrierAgent.createFreightDriverPlans();
085                        vehicleRoutes.addAll(plansForCarrier);
086                }
087                return vehicleRoutes;
088        }
089
090        /**
091         * Request all carrier agents to score their plans.
092         * 
093         */
094        public void scoreSelectedPlans() {
095                for (Carrier carrier : carriers.getCarriers().values()) {
096                        CarrierAgent agent = findCarrierAgent(carrier.getId());
097                        agent.scoreSelectedPlan();
098                }
099        }
100
101        @Override
102        public void reset(int iteration) {
103                delegate.reset(iteration);
104        }
105
106        private CarrierAgent findCarrierAgent(Id<Carrier> id) {
107                for (CarrierAgent agent : carrierAgents) {
108                        if (agent.getId().equals(id)) {
109                                return agent;
110                        }
111                }
112                return null;
113        }
114
115        private void processEvent(Event event) {
116                eventsManager.processEvent(event);
117        }
118
119        /**
120         * Informs the world that a shipment has been picked up.
121         * 
122         * <p>Is called by carrierAgent in charge of picking up shipments. It throws an ShipmentPickedupEvent which can be listened to
123         * with an ShipmentPickedUpListener.
124         * 
125         * @param carrierId
126         * @param driverId
127         * @param shipment
128         * @param time
129         * @see ShipmentPickedUpEvent, ShipmentPickedUpEventHandler
130         */
131        public void notifyPickedUp(Id<Carrier> carrierId, Id<Person> driverId, CarrierShipment shipment, double time) {
132                processEvent(new ShipmentPickedUpEvent(carrierId, driverId, shipment, time));
133        }
134
135        public void notifyDelivered(Id<Carrier> carrierId, Id<Person> driverId, CarrierShipment shipment, double time) {
136                processEvent(new ShipmentDeliveredEvent(carrierId, driverId, shipment,time));
137        }
138
139        @Override
140        public void handleEvent(ActivityEndEvent event) {
141                CarrierAgent carrierAgent = getCarrierAgent(event.getPersonId());
142                if(carrierAgent == null) return;
143                carrierAgent.handleEvent(event);
144        }
145
146        @Override
147        public void handleEvent(LinkEnterEvent event) {
148                CarrierAgent carrierAgent = getCarrierAgent(delegate.getDriverOfVehicle(event.getVehicleId()));
149                if(carrierAgent == null) return;
150                carrierAgent.handleEvent(event);
151        }
152
153        @Override
154        public void handleEvent(ActivityStartEvent event) {
155                CarrierAgent carrierAgent = getCarrierAgent(event.getPersonId());
156                if(carrierAgent == null) return;
157                carrierAgent.handleEvent(event);
158        }
159
160
161        @Override
162        public void handleEvent(PersonArrivalEvent event) {
163                CarrierAgent carrierAgent = getCarrierAgent(event.getPersonId());
164                if(carrierAgent == null) return;
165                carrierAgent.handleEvent(event);
166        }
167
168        @Override
169        public void handleEvent(PersonDepartureEvent event) {
170                CarrierAgent carrierAgent = getCarrierAgent(event.getPersonId());
171                if(carrierAgent == null) return;
172                carrierAgent.handleEvent(event);
173        }
174
175        private CarrierAgent getCarrierAgent(Id<Person> driverId) {
176                if(driverAgentMap.containsKey(driverId)){
177                        return driverAgentMap.get(driverId);
178                }
179                for(CarrierAgent ca : carrierAgents){
180                        if(ca.getDriverIds().contains(driverId)){
181                                driverAgentMap.put(driverId, ca);
182                                return ca;
183                        }
184                }
185                return null;    
186        }
187        
188        CarrierDriverAgent getDriver(Id<Person> driverId){
189                CarrierAgent carrierAgent = getCarrierAgent(driverId);
190                if(carrierAgent == null) throw new IllegalStateException("missing carrier agent. cannot find carrierAgent to driver " + driverId);
191                return carrierAgent.getDriver(driverId);
192        }
193
194        @Override
195        public void handleEvent(VehicleLeavesTrafficEvent event) {
196                delegate.handleEvent(event);
197        }
198
199        @Override
200        public void handleEvent(VehicleEntersTrafficEvent event) {
201                delegate.handleEvent(event);
202        }
203}
