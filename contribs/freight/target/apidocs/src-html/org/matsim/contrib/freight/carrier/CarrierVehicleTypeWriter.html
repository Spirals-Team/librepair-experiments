001package org.matsim.contrib.freight.carrier;
002
003import java.io.BufferedWriter;
004import java.io.IOException;
005
006import org.apache.log4j.Logger;
007import org.matsim.contrib.freight.carrier.CarrierVehicleType.VehicleCostInformation;
008import org.matsim.core.utils.io.MatsimXmlWriter;
009import org.matsim.vehicles.EngineInformation;
010
011/**
012 * A writer that writes carriers and their plans in an xml-file.
013 * 
014 * @author sschroeder
015 *
016 */
017public class CarrierVehicleTypeWriter extends MatsimXmlWriter {
018
019        private static Logger logger = Logger.getLogger(CarrierVehicleTypeWriter.class);
020
021        private CarrierVehicleTypes vehicleTypes;
022
023        
024        public CarrierVehicleTypeWriter(CarrierVehicleTypes carrierVehicleTypes) {
025                super();
026                this.vehicleTypes = carrierVehicleTypes;
027        }
028
029        
030        public void write(String filename) {
031                logger.info("write vehicle-types");
032                try {
033                        openFile(filename);
034                        writeXmlHead();
035                        writeTypes(this.writer);
036                        close();
037                        logger.info("done");
038                } catch (IOException e) {
039                        e.printStackTrace();
040                        logger.error(e);
041                        System.exit(1);
042                }
043        }
044
045        private void writeTypes(BufferedWriter writer)throws IOException {
046                writer.write("\t<vehicleTypes>\n");
047                for(CarrierVehicleType type : vehicleTypes.getVehicleTypes().values()){
048                        writer.write("\t\t<vehicleType id=\"" + type.getId() + "\">\n");
049                        writer.write("\t\t\t<description>" + type.getDescription() + "</description>\n");
050                        EngineInformation engineInformation = type.getEngineInformation();
051                        if(engineInformation != null) writer.write("\t\t\t<engineInformation fuelType=\"" + engineInformation.getFuelType().toString() + "\" gasConsumption=\"" + engineInformation.getGasConsumption() + "\"/>\n");
052                        writer.write("\t\t\t<capacity>" + type.getCarrierVehicleCapacity() + "</capacity>\n");
053                        VehicleCostInformation vehicleCostInformation = type.getVehicleCostInformation();
054                        if(vehicleCostInformation == null) throw new IllegalStateException("vehicleCostInformation is missing.");
055                        writer.write("\t\t\t<costInformation fix=\"" + vehicleCostInformation.fix + "\" perMeter=\"" + vehicleCostInformation.perDistanceUnit + 
056                                        "\" perSecond=\"" + vehicleCostInformation.perTimeUnit + "\"/>\n");
057                        writer.write("\t\t</vehicleType>\n");
058                }
059                writer.write("\t</vehicleTypes>\n\n");
060        }
061
062        
063}
