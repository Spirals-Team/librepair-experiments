001package roadTypeMapping;
002
003import java.util.HashMap;
004import java.util.Map;
005
006/**
007 * Created by molloyj on 01.12.2017.
008 * class to mimic the old roadTypeMapping that berlin uses with VISUM
009 */
010public class VisumHbefaRoadTypeMapping implements HbefaRoadTypeMapping {
011
012    Map<String, String> mapping = new HashMap<>();
013
014    @Override
015    public String get(String roadType, double freeVelocity) {
016        return mapping.get(roadType);
017    }
018
019    public void put(String visumRtNr, String hbefaRtName) {
020        mapping.put(visumRtNr, hbefaRtName);
021    }
022
023    public static VisumHbefaRoadTypeMapping emptyMapping() {
024        return new VisumHbefaRoadTypeMapping();
025    }
026}
