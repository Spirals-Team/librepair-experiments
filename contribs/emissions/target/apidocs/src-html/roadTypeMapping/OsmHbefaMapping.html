001package roadTypeMapping;
002
003import java.util.HashMap;
004import java.util.Map;
005
006/**
007 * Created by molloyj on 01.12.2017.
008 *
009 *
010 * handled OSM road types:
011 *    motorway,trunk,primary,secondary, tertiary, unclassified,residential,service
012 *    motorway_link, trunk_link,primary_link, secondary_link
013 *    tertiary_link, living_street, pedestrian,track,road
014 *
015 * Hbefa categories and respective speeds
016 *    URB/MW-Nat./80 - 130
017 *    URB/MW-City/60 - 110
018 *    URB/Trunk-Nat./70 - 110
019 *    URB/Trunk-City/50 - 90
020 *    URB/Distr/50 - 80
021 *    URB/Local/50 - 60
022 *    URB/Access/30 - 50
023 *
024 * Conversions from OSM to hbefa types
025 *    motorway;MW
026 *    primary;Trunk
027 *    secondary;Distr
028 *    tertiary;Local
029 *    residential;Access
030 *    living;Access
031
032 */
033public class OsmHbefaMapping implements HbefaRoadTypeMapping {
034    private static final int MAX_SPEED = 130;
035    Map<String, Hbefa> hbfeaMap = new HashMap<>();
036
037    public static class Hbefa {
038        String name;
039        int min;
040        int max;
041
042        public Hbefa(String name, int min, int max) {
043            this.name = name;
044            this.min = min;
045            this.max = max;
046        }
047    }
048
049
050    public static OsmHbefaMapping build() {
051        OsmHbefaMapping mapping = new OsmHbefaMapping();
052        mapping.hbfeaMap.put("motorway-Nat.", new Hbefa("MW-Nat.",80,130));
053        mapping.hbfeaMap.put("motorway", new Hbefa("MW-City",60,90));
054        mapping.hbfeaMap.put("primary-Nat.", new Hbefa("Trunk-Nat.",80,110));
055        mapping.hbfeaMap.put("primary", new Hbefa("Trunk-City",50,80));
056        mapping.hbfeaMap.put("secondary", new Hbefa("Distr",50,80));
057        mapping.hbfeaMap.put("tertiary", new Hbefa("Local",50,60));
058        mapping.hbfeaMap.put("residential", new Hbefa("Access",30,50));
059        mapping.hbfeaMap.put("service", new Hbefa("Access",30,50));
060        mapping.hbfeaMap.put("living", new Hbefa("Access",30,50));
061
062        return mapping;
063    }
064
065    @Override
066    public String get(String roadType, double freeVelocity) {
067        
068        return getHEBFAtype(roadType,freeVelocity);
069
070
071    }
072
073    private int getSpeedCat(double speed) {
074
075            int speedCat = Double.isInfinite(speed) ? MAX_SPEED : (int) Math.round(speed * 3.6);
076            speedCat = ((speedCat + 5) / 10) * 10;
077            speedCat = Math.min(MAX_SPEED, speedCat);
078            return speedCat;
079
080    }
081
082    private String getHEBFAtype(String roadType, double freeVelocity) {
083
084
085        String[] ss = roadType.split("_"); //want to remove
086        String type = ss[0];
087
088        //TODO: could make distinction between national and city, based on shapefile, or regions.
089
090        if (type.equals("unclassified") || type.equals("road")) {
091            if (freeVelocity <= 50) type = "living";
092            else if (freeVelocity == 60) type = "tertiary";
093            else if (freeVelocity == 70) type = "secondary";
094            else if (freeVelocity <= 90) type = "primary";
095            else type = "motorway";
096        }
097
098        //specify that if speed > 90 and primary or motorway, then Nat.
099        if (type.equals("motorway") || type.equals("primary") && freeVelocity >= 90) {
100            type += "-Nat.";
101        }
102        if (hbfeaMap.get(type) == null) {
103            throw new RuntimeException("'"+ type +"' not in hbefa map");
104        }
105        int min_speed = hbfeaMap.get(type).min;
106        int max_speed = hbfeaMap.get(type).max;
107        int capped_speed = (int) Math.min(Math.max(min_speed, freeVelocity), max_speed);
108
109        return "URB/" + hbfeaMap.get(type).name + "/" + capped_speed;
110
111
112    }
