022package org.matsim.contrib.emissions;
023
024import java.util.HashMap;
025import java.util.Map;
026import org.apache.log4j.Logger;

028import org.matsim.api.core.v01.events.LinkLeaveEvent;
029import org.matsim.api.core.v01.events.VehicleEntersTrafficEvent;

031import org.matsim.api.core.v01.events.handler.LinkLeaveEventHandler;
032import org.matsim.api.core.v01.events.handler.VehicleEntersTrafficEventHandler;

034import org.matsim.api.core.v01.network.Link;

037import org.matsim.core.api.experimental.events.EventsManager;

043
044/**
045 * @author benjamin
046 */
047public class ColdEmissionHandler implements LinkLeaveEventHandler, VehicleLeavesTrafficEventHandler, VehicleEntersTrafficEventHandler {
048
049    private final Logger logger = Logger.getLogger(ColdEmissionHandler.class);
050
051    private final Vehicles vehicles;
052    private final Network network;
053    private final ColdEmissionAnalysisModule coldEmissionAnalysisModule;
054
055    private int zeroLinkLengthWarnCnt = 0;
056    private int nonCarWarn = 0;
057
058    private final Map<Id<Vehicle>, Double> vehicleId2stopEngineTime = new HashMap<>();
059    private final Map<Id<Vehicle>, Double> vehicleId2accumulatedDistance = new HashMap<>();
060    private final Map<Id<Vehicle>, Double> vehicleId2parkingDuration = new HashMap<>();
061    private final Map<Id<Vehicle>, Id<Link>> vehicleId2coldEmissionEventLinkId = new HashMap<>();
062    
063    public ColdEmissionHandler(
064            Vehicles vehicles,
065            Network network,
066            ColdEmissionAnalysisModuleParameter parameterObject2,
067            EventsManager emissionEventsManager, Double emissionEfficiencyFactor) {
068
069        this.vehicles = vehicles;
070        this.network = network;
071        this.coldEmissionAnalysisModule = new ColdEmissionAnalysisModule(parameterObject2, emissionEventsManager, emissionEfficiencyFactor);
072    }
073
074    @Override
075    public void reset(int iteration) {
076        vehicleId2stopEngineTime.clear();
077        vehicleId2accumulatedDistance.clear();
078        vehicleId2parkingDuration.clear();
079        vehicleId2coldEmissionEventLinkId.clear();
080        coldEmissionAnalysisModule.reset();
081    }
082
083    @Override
084    public void handleEvent(LinkLeaveEvent event) {
085                Id<Vehicle> vehicleId = event.getVehicleId();
086        Id<Link> linkId = event.getLinkId();
087        Link link = this.network.getLinks().get(linkId);
088        double linkLength = link.getLength();
089
090        if (linkLength == 0.) {
091            if (zeroLinkLengthWarnCnt == 0 ){
092                logger.warn("Length of the link "+ linkId + " is zero. No emissions will be estimated for this link. Make sure, this is intentional.");
093                logger.warn(Gbl.ONLYONCE);
094                zeroLinkLengthWarnCnt++;
095            }
096            return;
097        }
098
099        Double previousDistance = this.vehicleId2accumulatedDistance.get(vehicleId);
100        if (previousDistance != null) {
101            double distance = previousDistance + linkLength;
102            double parkingDuration = this.vehicleId2parkingDuration.get(vehicleId);
103            Id<Link> coldEmissionEventLinkId = this.vehicleId2coldEmissionEventLinkId.get(vehicleId);
104
105            Vehicle vehicle = vehicles.getVehicles().get(vehicleId);
106
107            if ((distance / 1000) > 1.0) {
108                this.coldEmissionAnalysisModule.calculateColdEmissionsAndThrowEvent(
109                        coldEmissionEventLinkId,
110                        vehicle,
111                        event.getTime(),
112                        parkingDuration,
113                        2
114                );
115                this.vehicleId2accumulatedDistance.remove(vehicleId);
116            } else {
117                this.vehicleId2accumulatedDistance.put(vehicleId, distance);
118            }
119        }
120    }
121
122    @Override
123    public void handleEvent(VehicleLeavesTrafficEvent event) {
124        if (!event.getNetworkMode().equals("car")) {
125            if( nonCarWarn <=1) {
126                logger.warn("non-car modes are supported, however, not properly tested yet.");
127                logger.warn(Gbl.ONLYONCE);
128                nonCarWarn++;
129            }
130        }
131        Id<Vehicle> vehicleId = event.getVehicleId();
132        Double stopEngineTime = event.getTime();
133        this.vehicleId2stopEngineTime.put(vehicleId, stopEngineTime);
134    }
135
136    // TODO actually, the engine starts before with the PersonEntersVehicleEvent
137    @Override
138    public void handleEvent(VehicleEntersTrafficEvent event) {
139        if (!event.getNetworkMode().equals("car")) {
140            if( nonCarWarn <=1) {
141                logger.warn("non-car modes are supported, however, not properly tested yet.");
142                logger.warn(Gbl.ONLYONCE);
143                nonCarWarn++;
144            }
145        }
146        Id<Link> linkId = event.getLinkId();
147        Id<Vehicle> vehicleId = event.getVehicleId();
148        double startEngineTime = event.getTime();
149        this.vehicleId2coldEmissionEventLinkId.put(vehicleId, linkId);
150
151        double parkingDuration;
152        if (this.vehicleId2stopEngineTime.containsKey(vehicleId)) {
153            double stopEngineTime = this.vehicleId2stopEngineTime.get(vehicleId);
154            parkingDuration = startEngineTime - stopEngineTime;
155
156        } else { //parking duration is assumed to be at least 12 hours when parking overnight
157            parkingDuration = 43200.0;
158        }
159        this.vehicleId2parkingDuration.put(vehicleId, parkingDuration);
160        this.vehicleId2accumulatedDistance.put(vehicleId, 0.0);
161
162        Vehicle vehicle = vehicles.getVehicles().get(vehicleId);
163        VehicleType vt = vehicle.getType();
164
165        this.coldEmissionAnalysisModule.calculateColdEmissionsAndThrowEvent(
166                linkId,
167                vehicle,
168                startEngineTime,
169                parkingDuration,
170                1
171        );
172    }
173}
