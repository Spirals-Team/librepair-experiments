020package org.matsim.contrib.emissions.example;
021
022import org.matsim.contrib.emissions.utils.EmissionsConfigGroup;
023import org.matsim.core.config.Config;
024import org.matsim.core.config.ConfigWriter;
025import org.matsim.core.config.groups.*;
026import org.matsim.core.config.groups.PlanCalcScoreConfigGroup.ActivityParams;
027import org.matsim.core.config.groups.StrategyConfigGroup.StrategySettings;
028import org.matsim.core.controler.Controler;
029import org.matsim.core.controler.MatsimServices;
030
031/**
032 * 
033 * Creates a config file 
034 * with necessary emission input files for the {@link org.matsim.contrib.emissions.utils.EmissionsConfigGroup EmissionsConfigGroup}.
035 * 
036 * This config file is used by the {@link org.matsim.contrib.emissions.example.RunEmissionToolOfflineExampleV2 OfflineExample} and
037 * the {@link org.matsim.contrib.emissions.example.RunEmissionToolOnlineExampleV2 OnlineExample}
038 * 
039 * @author benjamin, julia
040 *
041 */
042public class CreateEmissionConfig {
043
044                private static final String inputPath = "./test/input/org/matsim/contrib/emissions/";
045                private static final String networkFile = //inputPath + 
046                                "sample_network.xml";
047                private static final String plansFile = //inputPath + 
048                                "sample_population.xml";
049                private static final String emissionVehicleFile = //inputPath + 
050                                "sample_emissionVehicles.xml";
051                
052                private static final String roadTypeMappingFile = //inputPath +
053                                "sample_roadTypeMapping.txt";
054                
055                private static final String averageFleetWarmEmissionFactorsFile = //inputPath +
056                                "sample_EFA_HOT_vehcat_2005average.txt";
057                private static final String averageFleetColdEmissionFactorsFile = //inputPath +
058                                "sample_EFA_ColdStart_vehcat_2005average.txt";
059                
060                private static final boolean isUsingDetailedEmissionCalculation = true;
061                private static final String detailedWarmEmissionFactorsFile = //inputPath +
062                                "sample_EFA_HOT_SubSegm_2005detailed.txt";
063                private static final String detailedColdEmissionFactorsFile = //inputPath +
064                                "sample_EFA_ColdStart_SubSegm_2005detailed.txt";
065                
066                private static final String outputPath = "./test/output/";
067                private static final String configFilePath = inputPath + "config_v2.xml";
068                
069                private static final int numberOfIterations = 6;
070                
071                
072                public static void main(String[] args) {
073                        
074                        Config config = new Config();
075                        config.addCoreModules();
076                        MatsimServices controler = new Controler(config);
077                        
078                // controlerConfigGroup
079                        ControlerConfigGroup ccg = controler.getConfig().controler();
080                        ccg.setOutputDirectory(outputPath);
081                        ccg.setFirstIteration(0);
082                        ccg.setLastIteration(numberOfIterations-1);
083                        
084                // planCalcScoreConfigGroup
085                        PlanCalcScoreConfigGroup pcs = controler.getConfig().planCalcScore();
086                        ActivityParams homeP = new ActivityParams("home");
087                        homeP.setTypicalDuration(12 * 3600);
088                        pcs.addActivityParams(homeP);
089                        ActivityParams workP = new ActivityParams("work");
090                        workP.setTypicalDuration(8 * 3600);
091                        pcs.addActivityParams(workP);
092
093                // strategy
094                        StrategyConfigGroup scg = controler.getConfig().strategy();
095                        StrategySettings strategySettings = new StrategySettings();
096                        strategySettings.setStrategyName("ChangeExpBeta");
097                        strategySettings.setWeight(1.0);
098                        scg.addStrategySettings(strategySettings);
099                        
100                // network
101                        NetworkConfigGroup ncg = controler.getConfig().network();
102                        ncg.setInputFile(networkFile);
103                        
104                // plans
105                        PlansConfigGroup pcg = controler.getConfig().plans();
106                        pcg.setInputFile(plansFile);
107                        
108                // define emission tool input files     
109                EmissionsConfigGroup ecg = new EmissionsConfigGroup() ;
110                controler.getConfig().addModule(ecg);
111                ecg.setEmissionRoadTypeMappingFile(roadTypeMappingFile);
112                // emission vehicles are now set in the default vehicle container
113                config.vehicles().setVehiclesFile(emissionVehicleFile);
114
115                ecg.setUsingVehicleTypeIdAsVehicleDescription(false);
116
117                ecg.setAverageWarmEmissionFactorsFile(averageFleetWarmEmissionFactorsFile);
118                ecg.setAverageColdEmissionFactorsFile(averageFleetColdEmissionFactorsFile);
119                ecg.setUsingDetailedEmissionCalculation(isUsingDetailedEmissionCalculation);
120                ecg.setDetailedWarmEmissionFactorsFile(detailedWarmEmissionFactorsFile);
121                ecg.setDetailedColdEmissionFactorsFile(detailedColdEmissionFactorsFile);
122
123                ecg.setWritingEmissionsEvents(false);
124                ecg.setEmissionCostMultiplicationFactor(1.0);
125                ecg.setConsideringCO2Costs(true);
126                ecg.setEmissionEfficiencyFactor(1.0);
127                
128           // write config     
129                ConfigWriter cw = new ConfigWriter(config);
130                        cw.write(configFilePath);
131                        
132
133
134        }
135
136}
