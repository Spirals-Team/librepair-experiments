020package org.matsim.contrib.emissions.types;
021
022/**
023 * @author benjamin
024 *
025 */
026public enum ColdPollutant {
027        
028        /*TODO: CO2 not directly available for cold emissions; thus it could be calculated through FC, CO, and HC as follows:
029        get("FC")*0.865 - get("CO")*0.429 - get("HC")*0.866) / 0.273;*/
030        CO("CO"), FC("FC"), HC("HC"), NMHC("NMHC"), NOX("NOx"), NO2("NO2"), PM("PM");
031        
032        private final String key;
033
034        ColdPollutant(String key) {
035                this.key = key;
036        }
037
038        public String getText() {
039                return key;
040        }
041        
042        //TODO: Is this really neccessary? WarmPollutant does not have this method...jk, bk mai'13
043        public static ColdPollutant getValue(String key){
044                for(ColdPollutant cp : ColdPollutant.values()){
045                        String cpString = cp.getText();
046                        if(cpString.equals(key)){
047                                return cp;
048                        }
049                }
050                return null;
051        }
052}
