039public class EmissionUtils {
040        private static final Logger logger = Logger.getLogger(EmissionUtils.class);
041
042        private final SortedSet<String> listOfPollutants;
043
044        public EmissionUtils(){
045                
046                listOfPollutants = new TreeSet<>();
047                for(WarmPollutant wp : WarmPollutant.values()){
048                        listOfPollutants.add(wp.toString());
049                }
050                for(ColdPollutant cp : ColdPollutant.values()){
051                        listOfPollutants.add(cp.toString());
052                }
053        }
054
055        public SortedMap<String, Double> sumUpEmissions(Map<WarmPollutant, Double> warmEmissions, Map<ColdPollutant, Double> coldEmissions) {
056                SortedMap<String, Double> pollutant2sumOfEmissions = new TreeMap<>();
057                SortedMap<String, Double> warmPollutant2emissions = convertWarmPollutantMap2String(warmEmissions);
058                SortedMap<String, Double> coldPollutant2emissions = convertColdPollutantMap2String(coldEmissions);
059                double sumOfPollutant;
060        
061                for(String pollutant : listOfPollutants){
062                        if(warmPollutant2emissions.containsKey(pollutant)){
063                                if(coldPollutant2emissions.containsKey(pollutant)){
064                                        sumOfPollutant = warmPollutant2emissions.get(pollutant) + coldPollutant2emissions.get(pollutant);
065                                } else{
066                                        sumOfPollutant = warmPollutant2emissions.get(pollutant);
067                                }
068                        } else if(coldPollutant2emissions.containsKey(pollutant)){
069                                sumOfPollutant = coldPollutant2emissions.get(pollutant);
070                        } else {
071                                sumOfPollutant = 0.0;
072                                logger.warn("Pollutant " + pollutant + " is not found in the emission events file [probably an old file]." +
073                                                "Therefore setting it to " + sumOfPollutant);
074                        }
075                        pollutant2sumOfEmissions.put(pollutant, sumOfPollutant);
076                }
077                return pollutant2sumOfEmissions;
078        }
079        
080        public <T> Map<Id<T>, SortedMap<String, Double>> sumUpEmissionsPerId(
081                        Map<Id<T>, Map<WarmPollutant, Double>> warmEmissions,
082                        Map<Id<T>, Map<ColdPollutant, Double>> coldEmissions) {
083
084                Map<Id<T>, SortedMap<String, Double>> totalEmissions = new HashMap<>();
085                Set<Id<T>> warmColdIds = new HashSet<>();
086                warmColdIds.addAll(warmEmissions.keySet());
087                warmColdIds.addAll(coldEmissions.keySet());
088
089                for (Id<T> id : warmColdIds) {
090                        if (warmEmissions.containsKey(id)) {
091                                if (coldEmissions.containsKey(id)) {
092                                        SortedMap<String, Double> idSumOfEmissions = sumUpEmissions(warmEmissions.get(id), coldEmissions.get(id));
093                                        totalEmissions.put(id, idSumOfEmissions);
094                                } else {
095                                        SortedMap<String, Double> warmPollutantString2Values = convertWarmPollutantMap2String(warmEmissions.get(id));
096                                        totalEmissions.put(id, warmPollutantString2Values);
097                                }
098                        } else {
099                                SortedMap<String, Double> coldPollutantString2Values = convertColdPollutantMap2String(coldEmissions.get(id));
100                                totalEmissions.put(id, coldPollutantString2Values);
101                        }
102                }
103                return totalEmissions;
104        }
105
106        public Map<Id<Person>, SortedMap<String, Double>> setNonCalculatedEmissionsForPopulation(Population population, Map<Id<Person>, SortedMap<String, Double>> totalEmissions) {
107                Map<Id<Person>, SortedMap<String, Double>> personId2Emissions = new HashMap<>();
108
109                for(Person person : population.getPersons().values()){
110                        Id<Person> personId = person.getId();
111                        SortedMap<String, Double> emissionType2Value;
112                        if(totalEmissions.get(personId) == null){ // person not in map (e.g. pt user)
113                                emissionType2Value = new TreeMap<>();
114                                for(String pollutant : listOfPollutants){
115                                        emissionType2Value.put(pollutant, 0.0);
116                                }
117                        } else { // person in map, but some emissions are not set; setting these to 0.0 
118                                emissionType2Value = totalEmissions.get(personId);
119                                for(String pollutant : listOfPollutants){ 
120                                        if(emissionType2Value.get(pollutant) == null){
121                                                emissionType2Value.put(pollutant, 0.0);
122                                        } // else do nothing
123                                }
124                        }
125                        personId2Emissions.put(personId, emissionType2Value);
126                }
127                return personId2Emissions;
128        }
129        
130        public Map<Id<Link>, SortedMap<String, Double>> setNonCalculatedEmissionsForNetwork(Network network, Map<Id<Link>, SortedMap<String, Double>> totalEmissions) {
131                Map<Id<Link>, SortedMap<String, Double>> linkId2Emissions = new HashMap<>();
132
133                for(Link link: network.getLinks().values()){
134                        Id<Link> linkId = link.getId();
135                        SortedMap<String, Double> emissionType2Value;
136                        
137                        if(totalEmissions.get(linkId) == null){
138                                emissionType2Value = new TreeMap<>();
139                                for(String pollutant : listOfPollutants){
140                                        emissionType2Value.put(pollutant, 0.0);
141                                }
142                        } else {
143                                emissionType2Value = totalEmissions.get(linkId);
144                                for(String pollutant : listOfPollutants){ 
145                                        if(emissionType2Value.get(pollutant) == null){
146                                                emissionType2Value.put(pollutant, 0.0);
147                                        } else {
148                                                emissionType2Value.put(pollutant, emissionType2Value.get(pollutant));
149                                        }
150                                }
151                        }
152                        linkId2Emissions.put(linkId, emissionType2Value);
153                }
154                return linkId2Emissions;
155        }
156
157        public <T> SortedMap<String, Double> getTotalEmissions(Map<Id<T>, SortedMap<String, Double>> person2TotalEmissions) {
158                SortedMap<String, Double> totalEmissions = new TreeMap<>();
159
160                for(Id<T> personId : person2TotalEmissions.keySet()){
161                        SortedMap<String, Double> individualEmissions = person2TotalEmissions.get(personId);
162                        double sumOfPollutant;
163                        for(String pollutant : individualEmissions.keySet()){
164                                if(totalEmissions.containsKey(pollutant)){
165                                        sumOfPollutant = totalEmissions.get(pollutant) + individualEmissions.get(pollutant);
166                                } else {
167                                        sumOfPollutant = individualEmissions.get(pollutant);
168                                }
169                                totalEmissions.put(pollutant, sumOfPollutant);
170                        }
171                }
172                return totalEmissions;
173        }
174
175        public SortedSet<String> getListOfPollutants() {
176                return listOfPollutants;
177        }
178
179        public SortedMap<String, Double> convertWarmPollutantMap2String(Map<WarmPollutant, Double> warmEmissions) {
180                SortedMap<String, Double> warmPollutantString2Values = new TreeMap<>();
181                for (Entry<WarmPollutant, Double> entry: warmEmissions.entrySet()){
182                        String pollutant = entry.getKey().toString();
183                        Double value = entry.getValue();
184                        warmPollutantString2Values.put(pollutant, value);
185                }
186                return warmPollutantString2Values;
187        }
188
189        public SortedMap<String, Double> convertColdPollutantMap2String(Map<ColdPollutant, Double> coldEmissions) {
190                SortedMap<String, Double> coldPollutantString2Values = new TreeMap<>();
191                for (Entry<ColdPollutant, Double> entry: coldEmissions.entrySet()){
192                        String pollutant = entry.getKey().toString();
193                        Double value = entry.getValue();
194                        coldPollutantString2Values.put(pollutant, value);
195                }
196                return coldPollutantString2Values;
197        }
