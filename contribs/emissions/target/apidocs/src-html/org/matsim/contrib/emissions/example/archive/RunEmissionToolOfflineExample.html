020package org.matsim.contrib.emissions.example.archive;
021
022import org.matsim.api.core.v01.Scenario;
023import org.matsim.contrib.emissions.EmissionModule;
024import org.matsim.contrib.emissions.utils.EmissionsConfigGroup;
025import org.matsim.core.api.experimental.events.EventsManager;
026import org.matsim.core.config.Config;
027import org.matsim.core.config.ConfigUtils;
028import org.matsim.core.config.ConfigReader;
029import org.matsim.core.events.EventsUtils;
030import org.matsim.core.events.MatsimEventsReader;
031import org.matsim.core.events.algorithms.EventWriterXML;
032import org.matsim.core.scenario.ScenarioUtils;
033
034
035/**
036 * 
037 * Use the config file as created by the 
038 * {@link org.matsim.contrib.emissions.example.CreateEmissionConfig CreateEmissionConfig} to calculate 
039 * emissions based on the link leave events of an events file. Results are written into an emission event file.
040 * Archived: Nov'16
041 *
042 * @author benjamin, julia
043 */
044public class RunEmissionToolOfflineExample {
045        
046        private final static String runDirectory = "./test/output/";
047        private static final String configFile = "./test/input/org/matsim/contrib/emissions/config.xml";
048        private final static Integer lastIteration = getLastIteration();
049        
050        private static final String eventsPath = runDirectory + "ITERS/it." + lastIteration + "/" +  lastIteration;
051        private static final String eventsFile = eventsPath + ".events.xml.gz";
052        private static final String emissionEventOutputFile = eventsPath + ".emission.events.offline.xml.gz";
053        
054        // =======================================================================================================              
055        
056        public static void main (String[] args) throws Exception{
057                Config config = ConfigUtils.loadConfig(configFile, new EmissionsConfigGroup());
058
059                Scenario scenario = ScenarioUtils.loadScenario(config);
060
061                // following is only for backward compatibility in which vehicle description is null;
062                // for the new scenarios, setting vehicle description should be preferred.; Amit, sep 2016.
063                EmissionsConfigGroup ecg = (EmissionsConfigGroup)scenario.getConfig().getModules().get(EmissionsConfigGroup.GROUP_NAME);
064                ecg.setUsingVehicleTypeIdAsVehicleDescription(true);
065
066                EventsManager eventsManager = EventsUtils.createEventsManager();
067                EmissionModule emissionModule = new EmissionModule(scenario, eventsManager);
068
069                EventWriterXML emissionEventWriter = new EventWriterXML(emissionEventOutputFile);
070                emissionModule.getEmissionEventsManager().addHandler(emissionEventWriter);
071
072                MatsimEventsReader matsimEventsReader = new MatsimEventsReader(eventsManager);
073                matsimEventsReader.readFile(eventsFile);
074                
075                emissionEventWriter.closeFile();
076
077                emissionModule.writeEmissionInformation();
078        }
079
080        private static int getLastIteration() {
081                Config config = new Config();
082                config.addCoreModules();
083                ConfigReader configReader = new ConfigReader(config);
084                configReader.readFile(RunEmissionToolOfflineExample.configFile);
085        return config.controler().getLastIteration();
086        }
087}
