107                        WarmEmissionAnalysisModuleParameter parameterObject,
108                        EventsManager emissionEventsManager, Double emissionEfficiencyFactor) {
109                
110                if(parameterObject == null){
111                        logger.error("No warm emission analysis module parameter set. Aborting...");
112                        System.exit(0);
113                }
114                if(emissionEventsManager == null){
115                        logger.error("Event manager not set. Please check the configuration of your scenario. Aborting..." );

122                this.emissionEfficiencyFactor = emissionEfficiencyFactor;

145        public Map<WarmPollutant, Double> checkVehicleInfoAndCalculateWarmEmissions(
146                        Vehicle vehicle,
147                        String roadType,
148                        double freeVelocity,
149                        double linkLength,
150                        double travelTime) {
151
152                if(this.ecg.isUsingVehicleTypeIdAsVehicleDescription() ) {
153                        if(vehicle.getType().getDescription()==null) { // emission specification is in vehicle type id
154                                vehicle.getType().setDescription(EmissionSpecificationMarker.BEGIN_EMISSIONS
155                                                +vehicle.getType().getId().toString()+ EmissionSpecificationMarker.END_EMISSIONS);
156                        } else if( vehicle.getType().getDescription().contains(EmissionSpecificationMarker.BEGIN_EMISSIONS.toString()) ) {
157                                // emission specification is in vehicle type id and in vehicle description too.
158                        } else {
159                                String vehicleDescription = vehicle.getType().getDescription() + EmissionSpecificationMarker.BEGIN_EMISSIONS
160                                                + vehicle.getType().getId().toString()+ EmissionSpecificationMarker.END_EMISSIONS;
161                                vehicle.getType().setDescription(vehicleDescription);
162                        }
163                }
164
165                Map<WarmPollutant, Double> warmEmissions = new HashMap<>();
166                if(vehicle == null ||
167                                (vehicle.getType() == null && vehicle.getType().getDescription() == null) // if both are null together; no vehicle type information.
168                                ) {
169                        throw new RuntimeException("Vehicle type description for vehicle " + vehicle + " is missing. " +
170                                        "Please make sure that requirements for emission vehicles in "
171                                        + EmissionsConfigGroup.GROUP_NAME + " config group are met. Aborting...");
172                }
173
174                String vehicleDescription = vehicle.getType().getDescription();
175
176                Tuple<HbefaVehicleCategory, HbefaVehicleAttributes> vehicleInformationTuple = convertVehicleTypeId2VehicleInformationTuple(vehicleDescription);
177                if (vehicleInformationTuple.getFirst() == null){
178                        throw new RuntimeException("Vehicle category for vehicle " + vehicle + " is not valid. " +
179                                        "Please make sure that requirements for emission vehicles in " + 
180                                        EmissionsConfigGroup.GROUP_NAME + " config group are met. Aborting...");
181                }
182
183                warmEmissions = calculateWarmEmissions(vehicle.getId(), travelTime, roadType, freeVelocity, linkLength, vehicleInformationTuple);
184
185                // a basic apporach to introduce emission reduced cars:
186                if(emissionEfficiencyFactor != null){
187                        warmEmissions = rescaleWarmEmissions(warmEmissions);
188                }
189                return warmEmissions;
190        }
191        
192        private Map<WarmPollutant, Double> rescaleWarmEmissions(Map<WarmPollutant, Double> warmEmissions) {
193                Map<WarmPollutant, Double> rescaledWarmEmissions = new HashMap<>();
194                
195                for(WarmPollutant wp : warmEmissions.keySet()){
196                        Double orgValue = warmEmissions.get(wp);
197                        Double rescaledValue = emissionEfficiencyFactor * orgValue;
198                        rescaledWarmEmissions.put(wp, rescaledValue);
199                }
200                return rescaledWarmEmissions;
201        }

203        private Map<WarmPollutant, Double> calculateWarmEmissions(
204                        Id<Vehicle> vehicleId,
205                        double travelTime,
206                        String roadType,
207                        double freeVelocity,
208                        double linkLength,
209                        Tuple<HbefaVehicleCategory, HbefaVehicleAttributes> vehicleInformationTuple) {
210
211                Map<WarmPollutant, Double> warmEmissionsOfEvent = new HashMap<>();
212
213                String hbefaRoadTypeName = this.roadTypeMapping.get(roadType, freeVelocity);
214
215                HbefaWarmEmissionFactorKey keyFreeFlow = new HbefaWarmEmissionFactorKey();
216                HbefaWarmEmissionFactorKey keyStopAndGo = new HbefaWarmEmissionFactorKey();
217
218                if(vehicleInformationTuple.getFirst().equals(HbefaVehicleCategory.HEAVY_GOODS_VEHICLE)){
219                        keyFreeFlow.setHbefaVehicleCategory(HbefaVehicleCategory.HEAVY_GOODS_VEHICLE);
220                        keyStopAndGo.setHbefaVehicleCategory(HbefaVehicleCategory.HEAVY_GOODS_VEHICLE);
221                } else if (vehicleInformationTuple.getFirst().equals(HbefaVehicleCategory.MOTORCYCLE)) {
222                        keyFreeFlow.setHbefaVehicleCategory(HbefaVehicleCategory.MOTORCYCLE);
223                        keyStopAndGo.setHbefaVehicleCategory(HbefaVehicleCategory.MOTORCYCLE);
224                } else if(vehicleInformationTuple.getFirst().equals(HbefaVehicleCategory.ZERO_EMISSION_VEHICLE)) {
225                        for (WarmPollutant warmPollutant : WarmPollutant.values()) {
226                                warmEmissionsOfEvent.put( warmPollutant, 0.0 );
227                        }
228                        return warmEmissionsOfEvent;
229                } else {
230                        keyFreeFlow.setHbefaVehicleCategory(HbefaVehicleCategory.PASSENGER_CAR);
231                        keyStopAndGo.setHbefaVehicleCategory(HbefaVehicleCategory.PASSENGER_CAR);
232                }
233
234                keyFreeFlow.setHbefaRoadCategory(hbefaRoadTypeName);
235                keyStopAndGo.setHbefaRoadCategory(hbefaRoadTypeName);
236                keyFreeFlow.setHbefaTrafficSituation(HbefaTrafficSituation.FREEFLOW);
237                keyStopAndGo.setHbefaTrafficSituation(HbefaTrafficSituation.STOPANDGO);
238
239                if(this.detailedHbefaWarmTable != null){ // check if detailed emission factors file is set in config
240                        HbefaVehicleAttributes hbefaVehicleAttributes = new HbefaVehicleAttributes();
241                        hbefaVehicleAttributes.setHbefaTechnology(vehicleInformationTuple.getSecond().getHbefaTechnology());
242                        hbefaVehicleAttributes.setHbefaSizeClass(vehicleInformationTuple.getSecond().getHbefaSizeClass());
243                        hbefaVehicleAttributes.setHbefaEmConcept(vehicleInformationTuple.getSecond().getHbefaEmConcept());
244                        keyFreeFlow.setHbefaVehicleAttributes(hbefaVehicleAttributes);
245                        keyStopAndGo.setHbefaVehicleAttributes(hbefaVehicleAttributes);
246                }
247                
248                double linkLength_km = linkLength / 1000;
249                double travelTime_h = travelTime / 3600;
250                double freeFlowSpeed_kmh = freeVelocity * 3.6;
251                double averageSpeed_kmh = linkLength_km / travelTime_h;
252                
253                double freeFlowSpeedFromTable_kmh;
254                double stopGoSpeedFromTable_kmh;
255                double efFreeFlow_gpkm;
256                double efStopGo_gpkm;
257
258                for (WarmPollutant warmPollutant : WarmPollutant.values()) {
259                        double generatedEmissions;
260
261                        keyFreeFlow.setHbefaComponent(warmPollutant);
262                        keyStopAndGo.setHbefaComponent(warmPollutant);
263                        
264                        if(this.detailedHbefaWarmTable != null){
265                                if(this.detailedHbefaWarmTable.get(keyFreeFlow) != null && this.detailedHbefaWarmTable.get(keyStopAndGo) != null){
266                                        stopGoSpeedFromTable_kmh = this.detailedHbefaWarmTable.get(keyStopAndGo).getSpeed();
267                                        efFreeFlow_gpkm = this.detailedHbefaWarmTable.get(keyFreeFlow).getWarmEmissionFactor();
268                                        efStopGo_gpkm = this.detailedHbefaWarmTable.get(keyStopAndGo).getWarmEmissionFactor();
269                                        freeFlowSpeedFromTable_kmh = this.detailedHbefaWarmTable.get(keyFreeFlow).getSpeed();
270
271                                } else {
272                                        vehAttributesNotSpecifiedCnt++;
273                                        stopGoSpeedFromTable_kmh = this.avgHbefaWarmTable.get(keyStopAndGo).getSpeed();
274                                        efFreeFlow_gpkm = this.avgHbefaWarmTable.get(keyFreeFlow).getWarmEmissionFactor();
275                                        efStopGo_gpkm = this.avgHbefaWarmTable.get(keyStopAndGo).getWarmEmissionFactor();
276                                        freeFlowSpeedFromTable_kmh = this.avgHbefaWarmTable.get(keyFreeFlow).getSpeed();
277
278                    int maxWarnCnt = 3;
279                    if(vehAttributesNotSpecifiedCnt <= maxWarnCnt) {
280                                                logger.warn("Detailed vehicle attributes are not specified correctly for vehicle " + vehicleId + ": " + 
281                                                                "`" + vehicleInformationTuple.getSecond() + "'. Using fleet average values instead.");
282                                                if(vehAttributesNotSpecifiedCnt == maxWarnCnt) logger.warn(Gbl.FUTURE_SUPPRESSED);
283                                        }
284//                                      vehAttributesNotSpecified.add(personId);
285                                }
286                        } else {
287                                stopGoSpeedFromTable_kmh = this.avgHbefaWarmTable.get(keyStopAndGo).getSpeed();
288                                efFreeFlow_gpkm = this.avgHbefaWarmTable.get(keyFreeFlow).getWarmEmissionFactor();
289                                efStopGo_gpkm = this.avgHbefaWarmTable.get(keyStopAndGo).getWarmEmissionFactor();
290                                freeFlowSpeedFromTable_kmh = this.avgHbefaWarmTable.get(keyFreeFlow).getSpeed();
291//                              vehAttributesNotSpecified.add(personId);
292                        }
293                        
294                        if(averageSpeed_kmh <= 0.0){
295                                throw new RuntimeException("Average speed has been calculated to 0.0 or a negative value. Aborting...");
296                        }
297                        if ((averageSpeed_kmh - freeFlowSpeed_kmh) > 1.0){
298                                throw new RuntimeException("Average speed has been calculated to be greater than free flow speed; this might produce negative warm emissions. Aborting...");
299                        }
300                        /* NOTE: the following comparision does not make sense since HBEFA assumes free flow speeds to be different from speed limits.
301                         * For instance, for RUR/MW/80/Freeflow HBEFA assumes a free flow speed of 82.80 kmh.
302                         * benjamin, amit 01'2014
303                         * */           
304//                      if(freeFlowSpeedFromTable_kmh - freeFlowSpeed_kmh > 1.0 || freeFlowSpeedFromTable_kmh - freeFlowSpeed_kmh <-1.0){
305//                              logger.warn("The given free flow speed does not match the table's value. Please check consistency of your scenario!");
306//                              logger.info("Using given speed value to avoid negative emission values...");    
307//                      }
308                        if((averageSpeed_kmh - freeFlowSpeed_kmh) >= -1.0) { // both speeds are assumed to be not very different > only freeFlow on link
309                                generatedEmissions = linkLength_km * efFreeFlow_gpkm;
310                                freeFlowCounter++;
311                                freeFlowKmCounter = freeFlowKmCounter + linkLength_km;
312                        } else if ((averageSpeed_kmh - stopGoSpeedFromTable_kmh) <= 0.0) { // averageSpeed is less than stopGoSpeed > only stop&go on link
313                                generatedEmissions = linkLength_km * efStopGo_gpkm;
314                                stopGoCounter++;
315                                stopGoKmCounter = stopGoKmCounter + linkLength_km;
316                        } else {
317                                double distanceStopGo_km = (linkLength_km * stopGoSpeedFromTable_kmh * (freeFlowSpeed_kmh - averageSpeed_kmh)) / (averageSpeed_kmh * (freeFlowSpeed_kmh - stopGoSpeedFromTable_kmh));
318                                double distanceFreeFlow_km = linkLength_km - distanceStopGo_km;
319
320                                generatedEmissions = (distanceFreeFlow_km * efFreeFlow_gpkm) + (distanceStopGo_km * efStopGo_gpkm);
321                                fractionCounter++;
322                                stopGoKmCounter = stopGoKmCounter + distanceStopGo_km;
323                                freeFlowKmCounter = freeFlowKmCounter + distanceFreeFlow_km;
324                        }
325                        kmCounter = kmCounter + linkLength_km;
326                        warmEmissionsOfEvent.put(warmPollutant, generatedEmissions);
327                }
328                emissionEventCounter++;
329//              vehicleIdSet.add(personId);
330                return warmEmissionsOfEvent;
331        }

333        private Tuple<HbefaVehicleCategory, HbefaVehicleAttributes> convertVehicleTypeId2VehicleInformationTuple(String vehicleDescription) {
334                Tuple<HbefaVehicleCategory, HbefaVehicleAttributes> vehicleInformationTuple;
335                HbefaVehicleCategory hbefaVehicleCategory = null;
336                HbefaVehicleAttributes hbefaVehicleAttributes = new HbefaVehicleAttributes();
337
338                int startIndex = vehicleDescription.indexOf(EmissionSpecificationMarker.BEGIN_EMISSIONS.toString()) + EmissionSpecificationMarker.BEGIN_EMISSIONS.toString().length();
339                int endIndex = vehicleDescription.lastIndexOf(EmissionSpecificationMarker.END_EMISSIONS.toString());
340
341                String[] vehicleInformationArray = vehicleDescription.substring(startIndex, endIndex).split(";");
342
343                for(HbefaVehicleCategory vehCat : HbefaVehicleCategory.values()){
344                        if(vehCat.toString().equals(vehicleInformationArray[0])){
345                                hbefaVehicleCategory = vehCat;
346                        }
347                }
348
349                if(vehicleInformationArray.length == 4){
350                        hbefaVehicleAttributes.setHbefaTechnology(vehicleInformationArray[1]);
351                        hbefaVehicleAttributes.setHbefaSizeClass(vehicleInformationArray[2]);
352                        hbefaVehicleAttributes.setHbefaEmConcept(vehicleInformationArray[3]);
353                } // else interpretation as "average vehicle"
354
355                vehicleInformationTuple = new Tuple<>(hbefaVehicleCategory, hbefaVehicleAttributes);
356                return vehicleInformationTuple;
357        }
