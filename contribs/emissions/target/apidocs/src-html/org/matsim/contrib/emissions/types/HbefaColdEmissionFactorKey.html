021package org.matsim.contrib.emissions.types;
022
023/**
024 * @author benjamin
025 * @author julia
026 * 
027 **/
028public class HbefaColdEmissionFactorKey {
029        
030        private HbefaVehicleCategory hbefaVehicleCategory;
031        private ColdPollutant hbefaComponent;
032        private Integer hbefaParkingTime;
033        private Integer hbefaDistance;
034        private HbefaVehicleAttributes hbefaVehicleAttributes = new HbefaVehicleAttributes();
035        
036        public HbefaColdEmissionFactorKey(){
037        }
038
039        HbefaVehicleCategory getHbefaVehicleCategory() {
040                return hbefaVehicleCategory;
041        }
042
043        public void setHbefaVehicleCategory(HbefaVehicleCategory hbefaVehicleCategory) {
044                this.hbefaVehicleCategory = hbefaVehicleCategory;
045        }
046
047        ColdPollutant getHbefaComponent() {
048                return hbefaComponent;
049        }
050
051        public void setHbefaComponent(ColdPollutant hbefaComponent) {
052                this.hbefaComponent = hbefaComponent;
053        }
054
055        Integer getHbefaParkingTime() {
056                return hbefaParkingTime;
057        }
058
059        public void setHbefaParkingTime(Integer hbefaParkingTime) {
060                this.hbefaParkingTime = hbefaParkingTime;
061        }
062
063        Integer getHbefaDistance() {
064                return hbefaDistance;
065        }
066
067        public void setHbefaDistance(Integer hbefaDistance) {
068                this.hbefaDistance = hbefaDistance;
069        }
070                
071        HbefaVehicleAttributes getHbefaVehicleAttributes() {
072                return hbefaVehicleAttributes;
073        }
074
075        public void setHbefaVehicleAttributes(HbefaVehicleAttributes hbefaVehicleAttributes) {
076                this.hbefaVehicleAttributes = hbefaVehicleAttributes;
077        }
078
079        /* need to implement the "equals" method in order to be able to construct an "equal" key
080         later on (e.g. from data available in the simulation)*/
081        @Override
082        public boolean equals(Object obj) {
083                if(this == obj) {
084                      return true;
085                 }
086                 if (!(obj instanceof HbefaColdEmissionFactorKey)) {
087                        return false; 
088                 }
089                 HbefaColdEmissionFactorKey key = (HbefaColdEmissionFactorKey) obj;
090                 return hbefaVehicleCategory.equals(key.getHbefaVehicleCategory())
091                 && hbefaComponent.equals(key.getHbefaComponent())
092                 && hbefaParkingTime.equals(key.getHbefaParkingTime())
093                 && hbefaDistance.equals(key.getHbefaDistance())
094                 && hbefaVehicleAttributes.equals(key.getHbefaVehicleAttributes());
095        }
096
097        // if "equals" is implemented, "hashCode" also needs to be implemented
098        @Override
099        public int hashCode(){
100                return toString().hashCode();
101        } 
102        
103
104        // needed for "hashCode" method
105        @Override
106        public String toString(){
107                return hbefaVehicleCategory + "; " 
108                + hbefaComponent + "; "
109                + hbefaParkingTime + "; "
110                + hbefaDistance+ "; "
111                + hbefaVehicleAttributes;
112        }
113}
