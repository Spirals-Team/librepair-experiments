040/**
041 * 2 categories for distance driven AFTER coldstart:
042 * <ul>
043 * <li> 0 - 1 km </li>
044 * <li> 1 - 2 km </li>
045 * </ul>
046 * 
047 * 13 categories for parking time BEFORE coldstart:
048 * <ul>
049 * <li> 0 - 1 h [1]</li>
050 * <li> 1 - 2 h [2]</li>
051 * <li> ... </li>
052 * <li> 11 - 12 h [12]</li>
053 * <li> > 12 h [13]</li>
054 * </ul>
055 * 
056 * Remarks:
057 * <ul>
058 * <li>HBEFA 3.1 does not provide further distance categories for cold start emission factors when average amient temperature is assumed <br>
059 * <li>HBEFA 3.1 does not provide cold start emission factors for Heavy Goods Vehicles; thus, HGV are assumed to produce the same cold start emission factors as passenger cars <br>
060 * <li>In the current implementation, vehicles emit one part of their cold start emissions when the engine is started (distance class 0 - 1 km);
061 * after reaching 1 km, the rest of their cold start emissions is emitted (difference between distance class 1 - 2 km and distance class 0 - 1 km)
062 * </ul>
063 * 
064 * 
065 * @author benjamin
066 */
067public class ColdEmissionAnalysisModule {
068        private static final Logger logger = Logger.getLogger(ColdEmissionAnalysisModule.class);
069        
070        private final Map<HbefaColdEmissionFactorKey, HbefaColdEmissionFactor> avgHbefaColdTable;
071        private final Map<HbefaColdEmissionFactorKey, HbefaColdEmissionFactor> detailedHbefaColdTable;
072        
073        private final EventsManager eventsManager;
074        private final Double emissionEfficiencyFactor;
075        private final EmissionsConfigGroup ecg;
076        
077        private int vehInfoWarnHDVCnt = 0;
078        private int vehAttributesNotSpecifiedCnt = 0;
079        private static final int maxWarnCnt = 3;
080        private int vehInfoWarnMotorCylceCnt = 0;
081
082        public static class ColdEmissionAnalysisModuleParameter {
083                public final Map<HbefaColdEmissionFactorKey, HbefaColdEmissionFactor> avgHbefaColdTable;
084                public final Map<HbefaColdEmissionFactorKey, HbefaColdEmissionFactor> detailedHbefaColdTable;
085                private final EmissionsConfigGroup ecg;

090                        this.avgHbefaColdTable = avgHbefaColdTable;
091                        this.detailedHbefaColdTable = detailedHbefaColdTable;

097                        ColdEmissionAnalysisModuleParameter parameterObject,
098                        EventsManager emissionEventsManager, Double emissionEfficiencyFactor) {
099
100                this.avgHbefaColdTable = parameterObject.avgHbefaColdTable;
101                this.detailedHbefaColdTable = parameterObject.detailedHbefaColdTable;
102                this.ecg = parameterObject.ecg;
103                this.eventsManager = emissionEventsManager;
104                this.emissionEfficiencyFactor = emissionEfficiencyFactor;
105        }

113        public void calculateColdEmissionsAndThrowEvent(
114                        Id<Link> coldEmissionEventLinkId,
115                        Vehicle vehicle,
116                        double eventTime,
117                        double parkingDuration,
118                        int distance_km) {
119
120                if(this.ecg.isUsingVehicleTypeIdAsVehicleDescription() ) {
121                        if(vehicle.getType().getDescription()==null) { // emission specification is in vehicle type id
122                                vehicle.getType().setDescription(EmissionSpecificationMarker.BEGIN_EMISSIONS
123                                                +vehicle.getType().getId().toString()+ EmissionSpecificationMarker.END_EMISSIONS);
124                        } else if( vehicle.getType().getDescription().contains(EmissionSpecificationMarker.BEGIN_EMISSIONS.toString()) ) {
125                                // emission specification is in vehicle type id and in vehicle description too.
126                        } else {
127                                String vehicleDescription = vehicle.getType().getDescription() + EmissionSpecificationMarker.BEGIN_EMISSIONS
128                                                + vehicle.getType().getId().toString()+ EmissionSpecificationMarker.END_EMISSIONS;
129                                vehicle.getType().setDescription(vehicleDescription);
130                        }
131                }
132
133                String vehicleDescription = vehicle.getType().getDescription();
134
135                Map<ColdPollutant, Double> coldEmissions = new HashMap<>();
136                if(vehicle.getType().getDescription() == null){
137                        throw new RuntimeException("Vehicle type description for vehicle " + vehicle + "is missing. " +
138                                        "Please make sure that requirements for emission vehicles in "
139                                        + EmissionsConfigGroup.GROUP_NAME + " config group are met. Aborting...");
140                }
141                Tuple<HbefaVehicleCategory, HbefaVehicleAttributes> vehicleInformationTuple = convertVehicleDescription2VehicleInformationTuple(vehicleDescription);
142                if (vehicleInformationTuple.getFirst() == null){
143                        throw new RuntimeException("Vehicle category for vehicle " + vehicle + " is not valid. " +
144                                        "Please make sure that requirements for emission vehicles in " + 
145                                        EmissionsConfigGroup.GROUP_NAME + " config group are met. Aborting...");
146                }
147
148                coldEmissions = getColdPollutantDoubleMap(vehicle.getId(), parkingDuration, vehicleInformationTuple, distance_km);
149
150                // a basic apporach to introduce emission reduced cars:
151                if(emissionEfficiencyFactor != null){
152                        coldEmissions = rescaleColdEmissions(coldEmissions);
153                }
154                Event coldEmissionEvent = new ColdEmissionEvent(eventTime, coldEmissionEventLinkId, vehicle.getId(), coldEmissions);
155                this.eventsManager.processEvent(coldEmissionEvent);
156        }
157
158        private Map<ColdPollutant, Double> rescaleColdEmissions(Map<ColdPollutant, Double> coldEmissions) {
159                Map<ColdPollutant, Double> rescaledColdEmissions = new HashMap<>();
160                
161                for(ColdPollutant wp : coldEmissions.keySet()){
162                        Double orgValue = coldEmissions.get(wp);
163                        Double rescaledValue = emissionEfficiencyFactor * orgValue;
164                        rescaledColdEmissions.put(wp, rescaledValue);
165                }
166                return rescaledColdEmissions;
167        }

169    private Map<ColdPollutant, Double> getColdPollutantDoubleMap(Id<Vehicle> vehicleId, double parkingDuration, Tuple<HbefaVehicleCategory, HbefaVehicleAttributes> vehicleInformationTuple, int distance_km) {
170        final Map<ColdPollutant, Double> coldEmissionsOfEvent = new HashMap<>();
171
172        HbefaColdEmissionFactorKey key = new HbefaColdEmissionFactorKey();
173
174        if(vehicleInformationTuple.getFirst().equals(HbefaVehicleCategory.HEAVY_GOODS_VEHICLE)){
175            key.setHbefaVehicleCategory(HbefaVehicleCategory.HEAVY_GOODS_VEHICLE);
176            key.setHbefaVehicleCategory(HbefaVehicleCategory.PASSENGER_CAR);
177            if(vehInfoWarnHDVCnt < maxWarnCnt) {
178                vehInfoWarnHDVCnt++;
179                logger.warn("HBEFA 3.1 does not provide cold start emission factors for " +
180                        HbefaVehicleCategory.HEAVY_GOODS_VEHICLE +
181                        ". Setting vehicle category to " + HbefaVehicleCategory.PASSENGER_CAR + "...");
182                if(vehInfoWarnHDVCnt == maxWarnCnt) logger.warn(Gbl.FUTURE_SUPPRESSED);
183            }
184        } else if(vehicleInformationTuple.getFirst().equals(HbefaVehicleCategory.ZERO_EMISSION_VEHICLE)) {
185                        for (ColdPollutant cp : ColdPollutant.values()){
186                                coldEmissionsOfEvent.put( cp, 0.0 );
187                        }
188                        return coldEmissionsOfEvent;
189                } else if (vehicleInformationTuple.getFirst().equals(HbefaVehicleCategory.MOTORCYCLE)) {
190                        if(vehInfoWarnMotorCylceCnt == 0) {
191                                vehInfoWarnMotorCylceCnt++;
192                                logger.warn("HBEFA 3.1 does not provide cold start emission factors for " +
193                                                HbefaVehicleCategory.MOTORCYCLE +
194                                                ". Setting cold emissions to zero.");
195                                logger.warn(Gbl.ONLYONCE + "\t" + Gbl.FUTURE_SUPPRESSED);
196                        }
197                        for (ColdPollutant cp : ColdPollutant.values()){
198                                coldEmissionsOfEvent.put( cp, 0.0 );
199                        }
200                        return coldEmissionsOfEvent;
201                } else {
202            key.setHbefaVehicleCategory(HbefaVehicleCategory.PASSENGER_CAR);
203        }
204
205        int parkingDuration_h = Math.max(1, (int) (parkingDuration / 3600));
206        if (parkingDuration_h >= 12) parkingDuration_h = 13;
207
208        key.setHbefaParkingTime(parkingDuration_h);
209
210        for (ColdPollutant coldPollutant : ColdPollutant.values()) {
211            double generatedEmissions;
212            if (distance_km == 1) {
213               generatedEmissions = getTableEmissions(vehicleId, vehicleInformationTuple, 1, key, coldPollutant);
214            } else {
215               generatedEmissions = getTableEmissions(vehicleId, vehicleInformationTuple, 2, key, coldPollutant) - getTableEmissions(vehicleId, vehicleInformationTuple, 1, key, coldPollutant);
216            }
217            coldEmissionsOfEvent.put(coldPollutant, generatedEmissions);
218        }
219        return coldEmissionsOfEvent;
220    }

222    private double getTableEmissions(Id<Vehicle> vehicleId, Tuple<HbefaVehicleCategory, HbefaVehicleAttributes> vehicleInformationTuple, int distance_km, HbefaColdEmissionFactorKey key, ColdPollutant coldPollutant) {
223        key.setHbefaDistance(distance_km);
224        double generatedEmissions;
225
226        key.setHbefaComponent(coldPollutant);
227
228        if(this.detailedHbefaColdTable != null){ // check if detailed emission factors file is set in config
229            HbefaVehicleAttributes hbefaVehicleAttributes = new HbefaVehicleAttributes();
230            hbefaVehicleAttributes.setHbefaTechnology(vehicleInformationTuple.getSecond().getHbefaTechnology());
231            hbefaVehicleAttributes.setHbefaSizeClass(vehicleInformationTuple.getSecond().getHbefaSizeClass());
232            hbefaVehicleAttributes.setHbefaEmConcept(vehicleInformationTuple.getSecond().getHbefaEmConcept());
233            key.setHbefaVehicleAttributes(hbefaVehicleAttributes);
234
235            if(this.detailedHbefaColdTable.containsKey(key)){
236                generatedEmissions = this.detailedHbefaColdTable.get(key).getColdEmissionFactor();
237            } else {
238                generatedEmissions = this.avgHbefaColdTable.get(key).getColdEmissionFactor();
239
240                if(vehAttributesNotSpecifiedCnt < maxWarnCnt) {
241                    vehAttributesNotSpecifiedCnt++;
242                    logger.warn("Detailed vehicle attributes are not specified correctly for vehicle " + vehicleId + ": " +
243                            "`" + vehicleInformationTuple.getSecond() + "'. Using fleet average values instead.");
244                    if(vehAttributesNotSpecifiedCnt == maxWarnCnt) logger.warn(Gbl.FUTURE_SUPPRESSED);
245                }
246            }
247        } else {
248            generatedEmissions = this.avgHbefaColdTable.get(key).getColdEmissionFactor();
249        }
250        return generatedEmissions;
251    }

253    private Tuple<HbefaVehicleCategory, HbefaVehicleAttributes> convertVehicleDescription2VehicleInformationTuple(String vehicleDescription) {
254                Tuple<HbefaVehicleCategory, HbefaVehicleAttributes> vehicleInformationTuple;
255                HbefaVehicleCategory hbefaVehicleCategory = null;
256                HbefaVehicleAttributes hbefaVehicleAttributes = new HbefaVehicleAttributes();
257
258                int startIndex = vehicleDescription.indexOf(EmissionSpecificationMarker.BEGIN_EMISSIONS.toString()) + EmissionSpecificationMarker.BEGIN_EMISSIONS.toString().length();
259                int endIndex = vehicleDescription.lastIndexOf(EmissionSpecificationMarker.END_EMISSIONS.toString());
260
261                String[] vehicleInformationArray = vehicleDescription.substring(startIndex, endIndex).split(";");
262
263                for(HbefaVehicleCategory vehCat : HbefaVehicleCategory.values()){
264                        if(vehCat.toString().equals(vehicleInformationArray[0])){
265                                hbefaVehicleCategory = vehCat;
266                        }
267                }
268
269                if(vehicleInformationArray.length == 4){
270                        hbefaVehicleAttributes.setHbefaTechnology(vehicleInformationArray[1]);
271                        hbefaVehicleAttributes.setHbefaSizeClass(vehicleInformationArray[2]);
272                        hbefaVehicleAttributes.setHbefaEmConcept(vehicleInformationArray[3]);
273                } // else interpretation as "average vehicle"
274
275                vehicleInformationTuple = new Tuple<>(hbefaVehicleCategory, hbefaVehicleAttributes);
276                return vehicleInformationTuple;
277        }
