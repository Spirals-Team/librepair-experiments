020package org.matsim.contrib.dvrp.run;
021
022import org.apache.log4j.Logger;
023import org.matsim.contrib.dynagent.run.DynQSimConfigConsistencyChecker;
024import org.matsim.core.config.Config;
025import org.matsim.core.config.consistency.ConfigConsistencyChecker;
026
027public class DvrpConfigConsistencyChecker implements ConfigConsistencyChecker {
028        private static final Logger log = Logger.getLogger(DvrpConfigConsistencyChecker.class);
029
030        @Override
031        public void checkConsistency(Config config) {
032                new DynQSimConfigConsistencyChecker().checkConsistency(config);
033
034                if (!config.qsim().isInsertingWaitingVehiclesBeforeDrivingVehicles()) {
035                        // Typically, vrp paths are calculated from startLink to endLink
036                        // (not from startNode to endNode). That requires making some assumptions
037                        // on how much time travelling on the first and last links takes.
038                        // The current implementation assumes:
039                        // (a) free-flow travelling on the last link, which is actually the case in QSim, and
040                        // (b) a 1-second stay on the first link (spent on moving over the first node).
041                        // The latter expectation is assumes that departing vehicles must be inserted before driving ones
042                        // (though that still does not guarantee 1-second stay since the vehicle may need to wait if the next
043                        // link is fully congested)
044                        log.warn(" 'QSim.insertingWaitingVehiclesBeforeDrivingVehicles' should be true in order to get"
045                                        + " more precise travel time estimates. See comments in DvrpConfigConsistencyChecker");
046                }
047                if (config.qsim().isRemoveStuckVehicles()) {
048                        throw new RuntimeException("Stuck DynAgents cannot be removed from simulation");
049                }
050        }
051}
