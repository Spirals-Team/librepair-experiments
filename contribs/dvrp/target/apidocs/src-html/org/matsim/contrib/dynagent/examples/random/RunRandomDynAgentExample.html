020package org.matsim.contrib.dynagent.examples.random;
021
022import org.matsim.api.core.v01.Scenario;
023import org.matsim.contrib.dynagent.run.*;
024import org.matsim.contrib.otfvis.OTFVisLiveModule;
025import org.matsim.core.config.*;
026import org.matsim.core.config.groups.QSimConfigGroup.*;
027import org.matsim.core.controler.Controler;
028import org.matsim.core.controler.OutputDirectoryHierarchy.OverwriteFileSetting;
029import org.matsim.core.scenario.ScenarioUtils;
030import org.matsim.vis.otfvis.OTFVisConfigGroup;
031import org.matsim.vis.otfvis.OTFVisConfigGroup.ColoringScheme;
032
033public class RunRandomDynAgentExample {
034        public static void run(boolean otfvis) {
035                String netFile = "./src/main/resources/grid_network.xml";
036                run(netFile, otfvis);
037        }
038
039        public static void run(String netFile, boolean otfvis) {
040                OTFVisConfigGroup otfvisConfig = new OTFVisConfigGroup();
041                otfvisConfig.setColoringScheme(ColoringScheme.byId);
042                otfvisConfig.setDrawNonMovingItems(true);
043
044                Config config = ConfigUtils.createConfig(otfvisConfig);
045                config.qsim().setSimStarttimeInterpretation(StarttimeInterpretation.onlyUseStarttime);
046                config.qsim().setSnapshotStyle(SnapshotStyle.queue);
047                config.network().setInputFile(netFile);
048                config.controler().setOutputDirectory("./test/output/");
049                config.controler().setOverwriteFileSetting(OverwriteFileSetting.deleteDirectoryIfExists);
050                config.controler().setLastIteration(0);
051                config.addConfigConsistencyChecker(new DynQSimConfigConsistencyChecker());
052                config.checkConsistency();
053
054                Scenario scenario = ScenarioUtils.loadScenario(config);
055
056                Controler controler = new Controler(scenario);
057                controler.addOverridingModule(new DynQSimModule<>(RandomDynQSimProvider.class));
058
059                if (otfvis) {
060                        controler.addOverridingModule(new OTFVisLiveModule());
061                }
062
063                controler.run();
064        }
065
066        public static void main(String[] args) {
067                run(true);
068        }
069}
