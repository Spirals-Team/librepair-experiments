041/**
042 * Used for offline estimation of travel times for VrpOptimizer by means of the exponential moving average. The
043 * weighting decrease, alpha, must be in (0,1]. We suggest small values of alpha, e.g. 0.05.
044 * 
045 * The averaging starts from the initial travel time estimates. If not provided, the free-speed TTs is used as the
046 * initial estimates
047 * 
048 * @author michalm
049 */
050public class DvrpOfflineTravelTimeEstimator implements DvrpTravelTimeEstimator, MobsimBeforeCleanupListener {
051        private final TravelTime observedTT;
052        private final Network network;
053
054        private final int interval;
055        private final int intervalCount;
056        private final Map<Id<Link>, double[]> linkTTs;
057        private final double alpha;
058
059        @Inject
060        public DvrpOfflineTravelTimeEstimator(@Named(DvrpTravelTimeModule.DVRP_INITIAL) TravelTime initialTT,
061                        @Named(DvrpTravelTimeModule.DVRP_OBSERVED) TravelTime observedTT,
062                        @Named(DvrpRoutingNetworkProvider.DVRP_ROUTING) Network network, TravelTimeCalculatorConfigGroup ttCalcConfig,
063                        DvrpConfigGroup dvrpConfig) {
064                this(initialTT, observedTT, network, ttCalcConfig, dvrpConfig.getTravelTimeEstimationAlpha());
065        }
066
067        public DvrpOfflineTravelTimeEstimator(TravelTime initialTT, TravelTime observedTT, Network network,
068                        TravelTimeCalculatorConfigGroup ttCalcConfig, double travelTimeEstimationAlpha) {
069                this.observedTT = observedTT;
070                this.network = network;
071
072                alpha = travelTimeEstimationAlpha;
073                if (alpha > 1 || alpha <= 0) {
074                        throw new IllegalArgumentException("travelTimeEstimationAlpha must be in (0,1]");
075                }
076
077                interval = ttCalcConfig.getTraveltimeBinSize();
078                intervalCount = TimeBinUtils.getTimeBinCount(ttCalcConfig.getMaxTime(), interval);
079
080                linkTTs = Maps.newHashMapWithExpectedSize(network.getLinks().size());
081                init(initialTT);
082        }
083
084        private void init(TravelTime initialTT) {
085                for (Link link : network.getLinks().values()) {
086                        double[] tt = new double[intervalCount];
087                        updateTTs(link, tt, initialTT, 1.);
088                        linkTTs.put(link.getId(), tt);
089                }
090        }
091
092        @Override
093        public double getLinkTravelTime(Link link, double time, Person person, Vehicle vehicle) {
094                // TODO TTC is more flexible (simple averaging vs linear interpolation, etc.)
095                int idx = TimeBinUtils.getTimeBinIndex(time, interval, intervalCount);
096                return linkTTs.get(link.getId())[idx];
097        }
098
099        @Override
100        public void notifyMobsimBeforeCleanup(@SuppressWarnings("rawtypes") MobsimBeforeCleanupEvent e) {
101                for (Link link : network.getLinks().values()) {
102                        updateTTs(link, linkTTs.get(link.getId()), observedTT, alpha);
103                }
104        }
105
106        private void updateTTs(Link link, double[] tt, TravelTime travelTime, double alpha) {
107                for (int i = 0; i < intervalCount; i++) {
108                        double oldEstimatedTT = tt[i];
109                        double experiencedTT = travelTime.getLinkTravelTime(link, i * interval, null, null);
110                        tt[i] = alpha * experiencedTT + (1 - alpha) * oldEstimatedTT;
111                }
112        }
113}
