020package org.matsim.contrib.util.random;
021
022import com.google.common.collect.*;
023
024public class WeightedRandomSelectionTable<R, C, V> {
025        public static <R, C, V> WeightedRandomSelectionTable<R, C, V> createWithArrayTable(Iterable<? extends R> rowKeys,
026                        Iterable<? extends C> colKeys) {
027                Table<R, C, WeightedRandomSelection<V>> selectionTable = ArrayTable.create(rowKeys, colKeys);
028                return new WeightedRandomSelectionTable<R, C, V>(selectionTable);
029        }
030
031        public static <R, C, V> WeightedRandomSelectionTable<R, C, V> createWithHashBasedTable() {
032                Table<R, C, WeightedRandomSelection<V>> selectionTable = HashBasedTable.create();
033                return new WeightedRandomSelectionTable<R, C, V>(selectionTable);
034        }
035
036        // (K1, K2) -> random selection of V
037        private final Table<R, C, WeightedRandomSelection<V>> selectionTable;
038
039        private WeightedRandomSelectionTable(Table<R, C, WeightedRandomSelection<V>> selectionTable) {
040                this.selectionTable = selectionTable;
041        }
042
043        public void add(R rowKey, C colKey, V value, double weight) {
044                WeightedRandomSelection<V> selection = selectionTable.get(rowKey, colKey);
045
046                if (selection == null) {
047                        selection = new WeightedRandomSelection<>();
048                        selectionTable.put(rowKey, colKey, selection);
049                }
050
051                selection.add(value, weight);
052        }
053
054        public boolean contains(R rowKey, C colKey) {
055                return selectionTable.get(rowKey, colKey) != null;
056        }
057
058        public V select(R rowKey, C colKey) {
059                WeightedRandomSelection<V> selection = selectionTable.get(rowKey, colKey);
060                return selection == null ? null : selection.select();
061        }
062}
