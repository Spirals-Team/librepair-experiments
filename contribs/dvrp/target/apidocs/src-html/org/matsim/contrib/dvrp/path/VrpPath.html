025/**
026 * Contains relevant information about a dvrp path. But not information about where we are. This information is
027 * accessible via {@link OnlineDriveTaskTracker#getCurrentLinkIdx()}
028 * 
029 * @author (of documentation) nagel
030 */
031public interface VrpPath extends Iterable<Link> {
032        int getLinkCount();
033
034        Link getLink(int idx);
035
036        double getLinkTravelTime(int idx);
037
038        void setLinkTravelTime(int idx, double linkTT);
039
040        Link getFromLink();
041
042        Link getToLink();
043}
