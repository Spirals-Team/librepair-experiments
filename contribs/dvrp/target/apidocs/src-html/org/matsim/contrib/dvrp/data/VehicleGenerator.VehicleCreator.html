020package org.matsim.contrib.dvrp.data;
021
022import java.util.*;
023
024import org.apache.commons.math3.stat.descriptive.rank.Max;
025import org.matsim.contrib.util.random.*;
026
027/**
028 * @author michalm
029 */
030public class VehicleGenerator {
031        public interface VehicleCreator {//move to generator
032                Vehicle createVehicle(double t0, double t1);
033        }
034        
035        private final UniformRandom uniform = RandomUtils.getGlobalUniform();
036        private final List<VehicleImpl> vehicles = new ArrayList<>();
037
038        private final double minWorkTime;
039        private final double maxWorkTime;
040        private final VehicleCreator vehicleCreator;
041
042        private Queue<VehicleImpl> activeVehicles;
043        private double previousTime;
044        private double currentTime;
045
046        public VehicleGenerator(double minWorkTime, double maxWorkTime, VehicleCreator vehicleCreator) {
047                this.minWorkTime = minWorkTime;
048                this.maxWorkTime = maxWorkTime;
049                this.vehicleCreator = vehicleCreator;
050        }
051
052        public void generateVehicles(double[] vehicleCounts, double startTime, double periodDuration) {
053                if (periodDuration > minWorkTime) {
054                        throw new IllegalArgumentException();
055                }
056
057                initQueue(vehicleCounts);
058
059                // only iteration 0; in order to have zero vehicles before startTime
060                previousTime = startTime;
061                currentTime = startTime;
062
063                for (int i = 0; i < vehicleCounts.length; i++) {
064                        removeVehiclesOnT1();
065                        reachExpectedVehicleCount(vehicleCounts[i]);
066                        previousTime = currentTime;
067                        currentTime += periodDuration;
068                }
069
070                // get down to 0 after the last vehicle count
071                currentTime = previousTime;
072                reachExpectedVehicleCount(0);
073        }
074
075        // reach the expected vehicle count at currentTime
076        private void reachExpectedVehicleCount(double expectedVehicleCount) {
077                int vehsToAdd = calculateNumberOfVehiclesToAdd(expectedVehicleCount);
078                if (vehsToAdd > 0) {
079                        addVehicles(vehsToAdd);
080                } else {
081                        removeVehiclesBeforeT1(-vehsToAdd);
082                }
083        }
084
085        private void initQueue(double[] vehicleCounts) {
086                int queueCapacity = (int)new Max().evaluate(vehicleCounts) + 1;
087                activeVehicles = new PriorityQueue<>(queueCapacity, Vehicles.T1_COMPARATOR);
088        }
089
090        private void removeVehiclesOnT1() {
091                while (!activeVehicles.isEmpty()) {
092                        if (activeVehicles.peek().getServiceEndTime() >= currentTime) {
093                                return;
094                        }
095
096                        activeVehicles.poll();
097                }
098        }
099
100        private int calculateNumberOfVehiclesToAdd(double expectedVehicleCount) {
101                if (expectedVehicleCount < 0) {
102                        throw new IllegalArgumentException();
103                }
104
105                return (int)uniform.floorOrCeil(expectedVehicleCount) - activeVehicles.size();
106        }
107
108        private void addVehicles(int count) {
109                for (int i = 0; i < count; i++) {
110                        double t0 = Math.floor(uniform.nextDouble(previousTime, currentTime));
111                        double workTime = Math.round(uniform.nextDouble(minWorkTime, maxWorkTime));
112                        VehicleImpl veh = (VehicleImpl)vehicleCreator.createVehicle(t0, t0 + workTime);
113                        activeVehicles.add(veh);
114                        vehicles.add(veh);
115                }
116        }
117
118        private void removeVehiclesBeforeT1(int count) {
119                for (int i = 0; i < count; i++) {
120                        VehicleImpl veh = activeVehicles.poll();
121                        double t1 = Math.floor(uniform.nextDouble(previousTime, currentTime));
122                        veh.setServiceEndTime(t1);
123                }
124        }
125
126        public List<? extends Vehicle> getVehicles() {
127                return vehicles;
128        }
129}
