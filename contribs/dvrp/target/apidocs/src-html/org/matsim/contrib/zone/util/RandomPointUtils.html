020package org.matsim.contrib.zone.util;
021
022import org.matsim.contrib.util.random.*;
023import org.matsim.core.utils.geometry.geotools.MGC;
024
025import com.google.common.base.Predicate;
026import com.vividsolutions.jts.geom.*;
027import com.vividsolutions.jts.geom.prep.PreparedGeometry;
028
029public class RandomPointUtils {
030        public static Point getRandomPointInGeometry(final Geometry geometry) {
031                return getRandomPointInEnvelope(p -> geometry.contains(p), geometry.getEnvelopeInternal());
032        }
033
034        public static Point getRandomPointInGeometry(final PreparedGeometry geometry) {
035                return getRandomPointInEnvelope(p -> geometry.contains(p), geometry.getGeometry().getEnvelopeInternal());
036        }
037
038        private static Point getRandomPointInEnvelope(Predicate<Point> contains, Envelope envelope) {
039                UniformRandom uniform = RandomUtils.getGlobalUniform();
040                double minX = envelope.getMinX();
041                double maxX = envelope.getMaxX();
042                double minY = envelope.getMinY();
043                double maxY = envelope.getMaxY();
044
045                Point p = null;
046                do {
047                        double x = uniform.nextDouble(minX, maxX);
048                        double y = uniform.nextDouble(minY, maxY);
049                        p = MGC.xy2Point(x, y);
050                } while (!contains.apply(p));
051
052                return p;
053        }
054}
