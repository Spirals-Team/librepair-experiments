019package org.matsim.contrib.dvrp.path;
020
021import java.util.Collection;
022import java.util.Collections;
023import java.util.List;
024import java.util.Map;
025import java.util.function.Supplier;
026import java.util.stream.Collectors;
027import java.util.stream.IntStream;
028
029import org.apache.log4j.Logger;
030import org.matsim.api.core.v01.Id;
031import org.matsim.api.core.v01.network.Link;
032import org.matsim.api.core.v01.network.Network;
033import org.matsim.contrib.dvrp.path.OneToManyPathSearch.PathData;
034import org.matsim.contrib.dvrp.util.TimeDiscretizer;
035import org.matsim.contrib.util.ExecutorServiceWithResource;
036import org.matsim.core.router.util.TravelDisutility;
037import org.matsim.core.router.util.TravelTime;
038
039import com.google.common.collect.ArrayTable;
040import com.google.common.collect.ImmutableList;
041import com.google.common.collect.Table;
042
043/**
044 * @author michalm
045 */
046public class ManyToManyPathData {
047        private static final Logger log = Logger.getLogger(ManyToManyPathData.class);
048
049        private final Table<Id<Link>, Id<Link>, PathData>[] tables;
050        private final TimeDiscretizer discretizer;
051
052        public ManyToManyPathData(Network network, TravelTime travelTime, TravelDisutility travelDisutility,
053                        List<Link> links, TimeDiscretizer discretizer, int threads) {
054                this.discretizer = discretizer;
055                this.tables = createTables(links);
056                updateTable(links, threads,
057                                () -> OneToManyPathSearch.createForwardSearch(network, travelTime, travelDisutility));
058        }
059
060        private Table<Id<Link>, Id<Link>, PathData>[] createTables(List<Link> links) {
061                ImmutableList<Id<Link>> linkIds = links.stream().map(l -> l.getId()).collect(ImmutableList.toImmutableList());
062                @SuppressWarnings("unchecked")
063                Table<Id<Link>, Id<Link>, PathData>[] tables = new Table[discretizer.getIntervalCount()];
064                for (int i = 0; i < tables.length; i++) {
065                        tables[i] = ArrayTable.create(linkIds, linkIds);
066                }
067                return tables;
068        }
069
070        private void updateTable(List<Link> links, int threads, Supplier<OneToManyPathSearch> oneToManyPathSearchProvider) {
071                log.info("Matrix calculation started");
072                ExecutorServiceWithResource<OneToManyPathSearch> executorService = new ExecutorServiceWithResource<>(//
073                                IntStream.range(0, threads)//
074                                                .mapToObj(i -> oneToManyPathSearchProvider.get())//
075                                                .collect(Collectors.toList()));
076
077                executorService.submitRunnablesAndWait(//
078                                IntStream.range(0, tables.length).boxed() // in each table (i)
079                                                .flatMap(i -> links.stream() // for each link (fromLink)
080                                                                .map(fromLink -> (search -> updateRow(search, fromLink, links, i)))));
081
082                executorService.shutdown();
083                log.info("Matrix calculation finished");
084        }
085
086        private void updateRow(OneToManyPathSearch search, Link fromLink, Collection<Link> toLinks, int timeIdx) {
087                int startTime = timeIdx * discretizer.getTimeInterval();
088                Map<Id<Link>, PathData> pathData = search.calcPathDataMap(fromLink, toLinks, startTime);
089                for (Map.Entry<Id<Link>, PathData> e : pathData.entrySet()) {
090                        tables[timeIdx].put(fromLink.getId(), e.getKey(), e.getValue());
091                }
092        }
093
094        public PathData getPathData(Id<Link> fromLink, Id<Link> toLink, double startTime) {
095                return tables[discretizer.getIdx(startTime)].get(fromLink, toLink);
096        }
097
098        public Map<Id<Link>, PathData> getOutgoingPathData(Id<Link> fromLink, double startTime) {
099                // to avoid delegation via Collections.unmodifiableMap(), one could change ArrayTable to ImmutableTable
100                // (guava would likely choose DenseImmutableTable instead of SparseImmutableTable -- though no explicit choice)
101                // same for getIncomingPathData
102                return Collections.unmodifiableMap(tables[discretizer.getIdx(startTime)].row(fromLink));
103        }
104
105        public Map<Id<Link>, PathData> getIncomingPathData(Id<Link> toLink, double startTime) {
106                return Collections.unmodifiableMap(tables[discretizer.getIdx(startTime)].column(toLink));
107        }
108}
