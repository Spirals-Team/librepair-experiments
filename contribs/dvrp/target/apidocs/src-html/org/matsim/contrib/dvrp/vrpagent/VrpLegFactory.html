019package org.matsim.contrib.dvrp.vrpagent;
020
021import org.matsim.contrib.dvrp.data.Vehicle;
022import org.matsim.contrib.dvrp.optimizer.VrpOptimizerWithOnlineTracking;
023import org.matsim.contrib.dvrp.schedule.DriveTask;
024import org.matsim.contrib.dvrp.tracker.TaskTrackers;
025import org.matsim.core.mobsim.framework.MobsimTimer;
026
027/**
028 * @author michalm
029 */
030public interface VrpLegFactory {
031        /**
032         * @param vehicle
033         *            for which the leg is created
034         * @return fully initialised VrpLeg (e.g. with online tracking, etc.)
035         */
036        VrpLeg create(Vehicle vehicle);
037
038        static VrpLeg createWithOfflineTracker(Vehicle vehicle, MobsimTimer timer) {
039                DriveTask driveTask = (DriveTask)vehicle.getSchedule().getCurrentTask();
040                VrpLeg leg = new VrpLeg(driveTask.getPath());
041                TaskTrackers.initOfflineTaskTracking(driveTask, timer);
042                return leg;
043        }
044
045        static VrpLeg createWithOnlineTracker(Vehicle vehicle, VrpOptimizerWithOnlineTracking optimizer,
046                        MobsimTimer timer) {
047                DriveTask driveTask = (DriveTask)vehicle.getSchedule().getCurrentTask();
048                VrpLeg leg = new VrpLeg(driveTask.getPath());
049                TaskTrackers.initOnlineDriveTaskTracking(vehicle, leg, optimizer, timer);
050                return leg;
051        }
052}
