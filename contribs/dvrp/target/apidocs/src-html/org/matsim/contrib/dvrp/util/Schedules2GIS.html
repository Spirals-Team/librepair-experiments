020package org.matsim.contrib.dvrp.util;
021
022import java.io.File;
023import java.util.ArrayList;
024import java.util.Collection;
025import java.util.List;
026import java.util.stream.Stream;
027
028import org.matsim.api.core.v01.Coord;
029import org.matsim.api.core.v01.network.Link;
030import org.matsim.contrib.dvrp.data.Vehicle;
031import org.matsim.contrib.dvrp.path.VrpPath;
032import org.matsim.contrib.dvrp.schedule.DriveTask;
033import org.matsim.contrib.dvrp.schedule.Schedules;
034import org.matsim.core.utils.geometry.geotools.MGC;
035import org.matsim.core.utils.gis.PolylineFeatureFactory;
036import org.matsim.core.utils.gis.ShapeFileWriter;
037import org.opengis.feature.simple.SimpleFeature;
038
039import com.vividsolutions.jts.geom.Coordinate;
040
041public class Schedules2GIS {
042        private final Iterable<? extends Vehicle> vehicles;
043        private final PolylineFeatureFactory factory;
044
045        public Schedules2GIS(Iterable<? extends Vehicle> vehicles, String coordSystem) {
046                this.vehicles = vehicles;
047
048                factory = new PolylineFeatureFactory.Builder().//
049                                setCrs(MGC.getCRS(coordSystem)).//
050                                setName("vrp_route").//
051                                addAttribute("VEH_ID", Integer.class).//
052                                addAttribute("VEH_NAME", String.class).//
053                                addAttribute("ROUTE_ID", Integer.class).//
054                                addAttribute("ARC_IDX", Integer.class).//
055                                create();
056        }
057
058        public void write(String vrpOutDir) {
059                new File(vrpOutDir).mkdir();
060                String file = vrpOutDir + "\\route_";
061
062                for (Vehicle v : vehicles) {
063                        Stream<DriveTask> drives = Schedules.driveTasks(v.getSchedule());
064                        Collection<SimpleFeature> features = new ArrayList<>();
065
066                        drives.forEach(drive -> {
067                                Coordinate[] coords = createLineString(drive);
068                                if (coords != null) {
069                                        features.add(this.factory.createPolyline(coords,
070                                                        new Object[] { v.getId(), v.getId(), drive.getTaskIdx() }, null));
071                                }
072                        });
073
074                        if (!features.isEmpty()) {
075                                ShapeFileWriter.writeGeometries(features, file + v.getId() + ".shp");
076                        }
077                }
078        }
079
080        private Coordinate[] createLineString(DriveTask driveTask) {
081                VrpPath path = driveTask.getPath();
082
083                if (path.getLinkCount() == 1) {
084                        return null;
085                }
086
087                List<Coordinate> coordList = new ArrayList<>();
088
089                for (Link l : path) {
090                        Coord c = l.getToNode().getCoord();
091                        coordList.add(new Coordinate(c.getX(), c.getY()));
092                }
093
094                return coordList.toArray(new Coordinate[coordList.size()]);
095        }
096}
