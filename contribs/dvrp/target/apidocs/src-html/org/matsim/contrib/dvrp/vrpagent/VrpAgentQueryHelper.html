020package org.matsim.contrib.dvrp.vrpagent;

022import java.util.*;
023
024import org.matsim.api.core.v01.TransportMode;
025import org.matsim.api.core.v01.population.*;
026import org.matsim.contrib.dvrp.data.Vehicle;
027import org.matsim.contrib.dvrp.path.VrpPaths;
028import org.matsim.contrib.dvrp.schedule.*;
029import org.matsim.contrib.dvrp.schedule.Schedule.ScheduleStatus;
030import org.matsim.contrib.dynagent.DynAgent;
031import org.matsim.core.mobsim.framework.MobsimAgent;
032import org.matsim.utils.objectattributes.attributable.Attributes;
033import org.matsim.vis.otfvis.OnTheFlyServer.NonPlanAgentQueryHelper;
034
035import com.google.inject.Inject;
036
037/**
038 * The class is designed for inheritance. Overrride createLeg() and createActivity() to obtain different visualisation
039 * of VRP schedules/vehicles
040 * 
041 * @author michalm
042 */
043public class VrpAgentQueryHelper implements NonPlanAgentQueryHelper {
044        protected final PopulationFactory populFactory;
045
046        @Inject
047        public VrpAgentQueryHelper(PopulationFactory populFactory) {
048                this.populFactory = populFactory;
049        }
050
051        @Override
052        public Activity getCurrentActivity(MobsimAgent mobsimAgent) {
053                Vehicle vehicle = getVehicle(mobsimAgent);
054                Schedule schedule = vehicle.getSchedule();
055
056                if (schedule.getStatus() == ScheduleStatus.STARTED) {
057                        Task currentTask = schedule.getCurrentTask();
058                        if (currentTask instanceof StayTask) {
059                                return createActivity((StayTask)currentTask);
060                        }
061                }
062
063                return null;
064        }
065
066        @Override
067        public Plan getPlan(MobsimAgent mobsimAgent) {
068                return new VrpSchedulePlan(getVehicle(mobsimAgent));
069        }
070
071        private List<PlanElement> initPlanElements(Vehicle vehicle) {
072                List<PlanElement> planElements = new ArrayList<>();
073                Schedule schedule = vehicle.getSchedule();
074
075                if (schedule.getStatus() == ScheduleStatus.STARTED) {
076                        for (Task t : schedule.getTasks()) {
077                                if (t instanceof DriveTask) {
078                                        planElements.add(createLeg((DriveTask)t));
079                                } else {
080                                        planElements.add(createActivity((StayTask)t));
081                                }
082                        }
083                }
084
085                return planElements;
086        }
087
088        private Vehicle getVehicle(MobsimAgent mobsimAgent) {
089                return ((VrpAgentLogic)((DynAgent)mobsimAgent).getAgentLogic()).getVehicle();
090        }
091
092        protected Leg createLeg(DriveTask task) {
093                Leg leg = populFactory.createLeg(TransportMode.car);
094                leg.setRoute(VrpPaths.createNetworkRoute(task.getPath(), populFactory.getRouteFactories()));
095                return leg;
096        }
097
098        protected Activity createActivity(StayTask task) {
099                Activity act = populFactory.createActivityFromLinkId("s", task.getLink().getId());
100                act.setStartTime(task.getBeginTime());
101                act.setEndTime(task.getEndTime());
102                return act;
103        }
104
105        private final class VrpSchedulePlan implements Plan {
106                private List<PlanElement> unmodifiablePlanElements;
107
108                private VrpSchedulePlan(Vehicle vehicle) {
109                        unmodifiablePlanElements = Collections.unmodifiableList(initPlanElements(vehicle));
110                }
111
112                @Override
113                public List<PlanElement> getPlanElements() {
114                        return unmodifiablePlanElements;
115                }
116
117                @Override
118                public Person getPerson() {
119                        throw new UnsupportedOperationException();
120                }
121
122                public Double getScore() {
123                        throw new UnsupportedOperationException();
124                }
125
126                @Override
127                public Map<String, Object> getCustomAttributes() {
128                        throw new UnsupportedOperationException();
129                }
130
131                @Override
132                public void addLeg(Leg leg) {
133                        throw new UnsupportedOperationException();
134                }
135
136                @Override
137                public void addActivity(Activity act) {
138                        throw new UnsupportedOperationException();
139                }
140
141                @Override
142                public String getType() {
143                        throw new UnsupportedOperationException();
144                }
145
146                @Override
147                public void setType(String type) {
148                        throw new UnsupportedOperationException();
149                }
150
151                @Override
152                public void setScore(Double score) {
153                        throw new UnsupportedOperationException();
154                }
155
156                @Override
157                public void setPerson(Person person) {
158                        throw new UnsupportedOperationException();
159                }
160
161                @Override
162                public Attributes getAttributes() {
163                        throw new UnsupportedOperationException();
164                }
165        }
166}
