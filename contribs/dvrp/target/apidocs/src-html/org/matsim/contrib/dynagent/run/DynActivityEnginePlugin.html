001package org.matsim.contrib.dynagent.run;
002
003import java.util.Collection;
004import java.util.Collections;
005
006import org.matsim.core.config.Config;
007import org.matsim.core.mobsim.qsim.AbstractQSimPlugin;
008import org.matsim.core.mobsim.qsim.interfaces.ActivityHandler;
009import org.matsim.core.mobsim.qsim.interfaces.MobsimEngine;
010
011import com.google.inject.AbstractModule;
012import com.google.inject.Module;
013
014public class DynActivityEnginePlugin extends AbstractQSimPlugin {
015        public DynActivityEnginePlugin(Config config) {
016                super(config);
017        }
018
019        @Override
020        public Collection<? extends Module> modules() {
021                return Collections.singletonList(new AbstractModule() {
022                        @Override
023                        public void configure() {
024                                bind(DynActivityEngine.class).asEagerSingleton();
025                        }
026                });
027        }
028
029        @Override
030        public Collection<Class<? extends ActivityHandler>> activityHandlers() {
031                return Collections.singletonList(DynActivityEngine.class);
032        }
033
034        @Override
035        public Collection<Class<? extends MobsimEngine>> engines() {
036                return Collections.singletonList(DynActivityEngine.class);
037        }
038}
