001package org.matsim.contrib.zone.io;
002
003import java.util.*;
004
005import org.matsim.api.core.v01.Id;
006import org.matsim.contrib.zone.Zone;
007import org.matsim.core.utils.collections.Tuple;
008import org.matsim.core.utils.io.MatsimXmlWriter;
009
010public class ZoneXmlWriter extends MatsimXmlWriter {
011        private Map<Id<Zone>, Zone> zones;
012
013        public ZoneXmlWriter(Map<Id<Zone>, Zone> zones) {
014                this.zones = zones;
015        }
016
017        public void write(String file) {
018                openFile(file);
019                writeDoctype("zones", "http://matsim.org/files/dtd/zones_v1.dtd");
020                writeStartTag("zones", Collections.<Tuple<String, String>> emptyList());
021                writeZones();
022                writeEndTag("zones");
023                close();
024        }
025
026        private void writeZones() {
027                for (Zone z : zones.values()) {
028                        List<Tuple<String, String>> atts = new ArrayList<>();
029                        atts.add(new Tuple<String, String>("id", z.getId().toString()));
030
031                        String type = z.getType();
032                        if (type != null && !type.isEmpty()) {
033                                atts.add(new Tuple<String, String>("type", type));
034                        }
035
036                        writeStartTag("zone", atts, true);
037                }
038        }
039}
