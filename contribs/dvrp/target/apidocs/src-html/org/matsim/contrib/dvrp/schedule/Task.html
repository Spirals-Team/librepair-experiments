020package org.matsim.contrib.dvrp.schedule;
021
022import org.matsim.contrib.dvrp.tracker.TaskTracker;
023
024/**
025 * Root class of Task hierarchy. <br/>
026 * <br/>
027 * Design comments:
028 * <ul>
029 * <li>This interface makes fairly strong assumptions on its implementation. I am wondering a bit if the design purpose
030 * might not be better expressed by removing the interfaces and simply have an implementation class hierarchy. A bit
031 * like the MATSim events, which also originally were interfaces with implementations behind them, and at some point we
032 * united this again. kai, feb'17
033 * </ul>
034 * 
035 * @author michalm
036 * @author (of documentation) nagel
037 */
038public interface Task {
039        public enum TaskStatus {
040                PLANNED, STARTED, PERFORMED;
041        }
042
043        TaskStatus getStatus();
044
045        /**
046         * Returns the begin time of the task (inclusive). Should be equal to the end time of the previous task.
047         */
048        double getBeginTime();
049
050        /**
051         * Returns the end time of the task (exclusive). Should be equal to the begin time of the next task.<br/>
052         * <br/>
053         * This is a scheduled (planned) end time. To get a more precise estimate on the expected end time of the current
054         * task, use {@link TaskTracker#predictEndTime()}. The actual end time is a result of simulation.<br/>
055         * <br/>
056         * When the task ends, its end time (and the begin/end times of the following tasks) should be updated accordingly
057         * (see {@link Task#setBeginTime(double)} and {@link Task#setEndTime(double)}).
058         */
059        double getEndTime();
060
061        /**
062         * Index of the task in the schedule. Managed by ScheduleImpl through a package-protected variable.
063         */
064        int getTaskIdx();
065
066        void setBeginTime(double beginTime);
067
068        void setEndTime(double endTime);
069
070        /**
071         * A TaskTracker predicts the task end time. The prediction may be different than the planned and actual end times.
072         */
073        TaskTracker getTaskTracker();
074
075        /**
076         * adds the TaskTracker to the ongoing Task (preferably at the beginning of its execution). Only ongoing tasks
077         * (TaskStatus == STARTED) can be tracked.
078         */
079        void initTaskTracker(TaskTracker taskTracker);
080}
