020package org.matsim.contrib.dvrp.path;
021
022import java.util.Collection;
023import java.util.Collections;
024import java.util.List;
025import java.util.Map;
026
027import org.matsim.api.core.v01.Id;
028import org.matsim.api.core.v01.network.Link;
029import org.matsim.api.core.v01.network.Network;
030import org.matsim.api.core.v01.network.Node;
031import org.matsim.contrib.locationchoice.router.BackwardFastMultiNodeDijkstra;
032import org.matsim.contrib.locationchoice.router.BackwardFastMultiNodeDijkstraFactory;
033import org.matsim.contrib.locationchoice.router.BackwardMultiNodePathCalculator;
034import org.matsim.core.router.FastMultiNodeDijkstraFactory;
035import org.matsim.core.router.InitialNode;
036import org.matsim.core.router.MultiNodePathCalculator;
037import org.matsim.core.router.RoutingNetworkImaginaryNode;
038import org.matsim.core.router.util.LeastCostPathCalculator.Path;
039import org.matsim.core.router.util.TravelDisutility;
040import org.matsim.core.router.util.TravelTime;
041
042import com.google.common.collect.ImmutableList;
043import com.google.common.collect.Maps;
044
045public class OneToManyPathSearch {
046        public static OneToManyPathSearch createForwardSearch(Network network, TravelTime travelTime,
047                        TravelDisutility travelDisutility) {
048                return create((MultiNodePathCalculator)new FastMultiNodeDijkstraFactory(true).createPathCalculator(network,
049                                travelDisutility, travelTime));
050        }
051
052        public static OneToManyPathSearch createBackwardSearch(Network network, TravelTime travelTime,
053                        TravelDisutility travelDisutility) {
054                return create((BackwardMultiNodePathCalculator)new BackwardFastMultiNodeDijkstraFactory(true)
055                                .createPathCalculator(network, travelDisutility, travelTime));
056        }
057
058        public static OneToManyPathSearch create(MultiNodePathCalculator multiNodeDijkstra) {
059                return new OneToManyPathSearch(multiNodeDijkstra);
060        }
061
062        public static class PathData {
063                final Path path;// shortest path
064                private final double firstAndLastLinkTT;// at both the first and last links
065
066                public PathData(Path path, double firstAndLastLinkTT) {
067                        this.path = new Path(null, ImmutableList.copyOf(path.links), path.travelTime, path.travelCost);
068                        this.firstAndLastLinkTT = firstAndLastLinkTT;
069                }
070
071                public double getTravelTime() {
072                        return path.travelTime + firstAndLastLinkTT;
073                }
074        }
075
076        private static class ToNode extends InitialNode {
077                private Path path;
078
079                private ToNode(Node node, double initialCost, double initialTime) {
080                        super(node, initialCost, initialTime);
081                }
082        }
083
084        private final MultiNodePathCalculator multiNodeDijkstra;// forward or backward
085        private final boolean forward;
086
087        private OneToManyPathSearch(MultiNodePathCalculator multiNodeDijkstra) {
088                this.multiNodeDijkstra = multiNodeDijkstra;
089                this.forward = !(multiNodeDijkstra instanceof BackwardFastMultiNodeDijkstra);
090        }
091
092        public PathData[] calcPathDataArray(Link fromLink, List<Link> toLinks, double startTime) {
093                Node fromNode = getFromNode(fromLink);
094                Map<Id<Node>, ToNode> toNodes = createToNodes(fromLink, toLinks);
095                calculatePaths(fromNode, toNodes, startTime);
096                return createPathDataArray(fromLink, toLinks, startTime, toNodes);
097        }
098
099        public Map<Id<Link>, PathData> calcPathDataMap(Link fromLink, Collection<Link> toLinks, double startTime) {
100                Node fromNode = getFromNode(fromLink);
101                Map<Id<Node>, ToNode> toNodes = createToNodes(fromLink, toLinks);
102                calculatePaths(fromNode, toNodes, startTime);
103                return createPathDataMap(fromLink, toLinks, startTime, toNodes);
104        }
105
106        private Map<Id<Node>, ToNode> createToNodes(Link fromLink, Collection<Link> toLinks) {
107                Map<Id<Node>, ToNode> toNodes = Maps.newHashMapWithExpectedSize(toLinks.size());
108                for (Link toLink : toLinks) {
109                        if (toLink != fromLink) {
110                                Node toNode = getToNode(toLink);
111                                toNodes.putIfAbsent(toNode.getId(), new ToNode(toNode, 0, 0));
112                        }
113                }
114                return toNodes;
115        }
116
117        private void calculatePaths(Node fromNode, Map<Id<Node>, ToNode> toNodes, double startTime) {
118                RoutingNetworkImaginaryNode imaginaryNode = new RoutingNetworkImaginaryNode(toNodes.values());
119                multiNodeDijkstra.setSearchAllEndNodes(true);
120                multiNodeDijkstra.calcLeastCostPath(fromNode, imaginaryNode, startTime, null, null);
121
122                // get path for each ToNode
123
124                // XXX in most cases we need costs/times, while paths could be constructed lazily only when needed
125                // TODO add getCost/Time() to MultiNodeDijkstra
126                for (ToNode toNode : toNodes.values()) {
127                        toNode.path = multiNodeDijkstra.constructPath(fromNode, toNode.node, startTime);
128                }
129        }
130
131        private PathData[] createPathDataArray(Link fromLink, List<Link> toLinks, double startTime,
132                        Map<Id<Node>, ToNode> toNodes) {
133                PathData[] pathDataArray = new PathData[toLinks.size()];
134                for (int i = 0; i < pathDataArray.length; i++) {
135                        pathDataArray[i] = createPathData(fromLink, toLinks.get(i), startTime, toNodes);
136                }
137                return pathDataArray;
138        }
139
140        private Map<Id<Link>, PathData> createPathDataMap(Link fromLink, Collection<Link> toLinks, double startTime,
141                        Map<Id<Node>, ToNode> toNodes) {
142                Map<Id<Link>, PathData> pathDataMap = Maps.newHashMapWithExpectedSize(toLinks.size());
143                for (Link toLink : toLinks) {
144                        pathDataMap.put(toLink.getId(), createPathData(fromLink, toLink, startTime, toNodes));
145                }
146                return pathDataMap;
147        }
148
149        private PathData createPathData(Link fromLink, Link toLink, double startTime, Map<Id<Node>, ToNode> toNodes) {
150                if (toLink == fromLink) {
151                        return createZeroPathData(fromLink);
152                } else {
153                        ToNode toNode = toNodes.get(getToNode(toLink).getId());
154                        return new PathData(toNode.path, getFirstAndLastLinkTT(fromLink, toLink, toNode.path, startTime));
155                }
156        }
157
158        private PathData createZeroPathData(Link fromLink) {
159                List<Node> singleNodeList = Collections.singletonList(getFromNode(fromLink));
160                List<Link> emptyLinkList = Collections.emptyList();
161                return new PathData(new Path(singleNodeList, emptyLinkList, 0, 0), 0);
162        }
163
164        private Node getToNode(Link toLink) {
165                return forward ? toLink.getFromNode() : toLink.getToNode();
166        }
167
168        private Node getFromNode(Link fromLink) {
169                return forward ? fromLink.getToNode() : fromLink.getFromNode();
170        }
171
172        private double getFirstAndLastLinkTT(Link fromLink, Link toLink, Path path, double time) {
173                double lastLinkTT = forward ? //
174                                VrpPaths.getLastLinkTT(toLink, time + path.travelTime) : VrpPaths.getLastLinkTT(fromLink, time);
175                return VrpPaths.FIRST_LINK_TT + lastLinkTT;
176        }
177}
