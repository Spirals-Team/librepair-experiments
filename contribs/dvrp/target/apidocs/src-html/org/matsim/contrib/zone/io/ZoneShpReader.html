020package org.matsim.contrib.zone.io;
021
022import java.util.*;
023
024import org.matsim.api.core.v01.Id;
025import org.matsim.contrib.zone.Zone;
026import org.matsim.core.utils.gis.ShapeFileReader;
027import org.opengis.feature.simple.SimpleFeature;
028
029import com.vividsolutions.jts.geom.MultiPolygon;
030
031public class ZoneShpReader {
032        private final Map<Id<Zone>, Zone> zones;
033
034        public ZoneShpReader(Map<Id<Zone>, Zone> zones) {
035                this.zones = zones;
036        }
037
038        public void readZones(String file) {
039                readZones(file, ZoneShpWriter.ID_HEADER);
040        }
041
042        public void readZones(String file, String idHeader) {
043                Collection<SimpleFeature> features = ShapeFileReader.getAllFeatures(file);
044                if (features.size() != zones.size()) {
045                        throw new RuntimeException("Features#: " + features.size() + "; zones#: " + zones.size());
046                }
047
048                for (SimpleFeature ft : features) {
049                        String id = ft.getAttribute(idHeader).toString();
050                        Zone z = zones.get(Id.create(id, Zone.class));
051                        z.setMultiPolygon((MultiPolygon)ft.getDefaultGeometry());
052                }
053        }
054}
