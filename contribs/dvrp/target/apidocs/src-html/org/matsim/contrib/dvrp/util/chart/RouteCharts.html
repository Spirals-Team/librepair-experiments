053public class RouteCharts {
054        public static JFreeChart chartRoutes(Collection<? extends Vehicle> vehicles) {
055                CoordDataset lData = new CoordDataset();
056                int i = 0;
057                for (Vehicle v : vehicles) {
058                        Schedule schedule = v.getSchedule();
059                        lData.addSeries(Integer.toString(i++), ScheduleCoordSources.createCoordSource(schedule));
060                }
061
062                JFreeChart chart = ChartFactory.createXYLineChart("Routes", "X", "Y", lData, PlotOrientation.VERTICAL, true,
063                                true, false);
064
065                XYPlot plot = (XYPlot)chart.getPlot();
066                plot.setRangeGridlinesVisible(false);
067                plot.setDomainGridlinesVisible(false);
068                plot.setBackgroundPaint(Color.white);
069
070                NumberAxis yAxis = (NumberAxis)plot.getRangeAxis();
071                yAxis.setAutoRangeIncludesZero(false);
072
073                XYLineAndShapeRenderer renderer = (XYLineAndShapeRenderer)plot.getRenderer();
074                renderer.setSeriesShapesVisible(0, true);
075                renderer.setSeriesLinesVisible(0, false);
076                renderer.setSeriesItemLabelsVisible(0, true);
077
078                renderer.setBaseItemLabelGenerator(new XYItemLabelGenerator() {
079                        public String generateLabel(XYDataset dataset, int series, int item) {
080                                return ((CoordDataset)dataset).getText(series, item);
081                        }
082                });
083
084                for (int j = 1; j <= vehicles.size(); j++) {
085                        renderer.setSeriesShapesVisible(j, true);
086                        renderer.setSeriesLinesVisible(j, true);
087                        renderer.setSeriesItemLabelsVisible(j, true);
088                }
089
090                return chart;
091        }
092
093        public static JFreeChart chartRoutesByStatus(List<? extends Vehicle> vehicles) {
094                CoordDataset nData = new CoordDataset();
095
096                for (int i = 0; i < vehicles.size(); i++) {
097                        Schedule schedule = vehicles.get(i).getSchedule();
098                        Map<TaskStatus, CoordSource> vsByStatus = createLinkSourceByStatus(schedule);
099                        nData.addSeries(i + "-PR", vsByStatus.get(TaskStatus.PERFORMED));
100                        nData.addSeries(i + "-ST", vsByStatus.get(TaskStatus.STARTED));
101                        nData.addSeries(i + "-PL", vsByStatus.get(TaskStatus.PLANNED));
102                }
103
104                JFreeChart chart = ChartFactory.createXYLineChart("Routes", "X", "Y", nData, PlotOrientation.VERTICAL, false,
105                                true, false);
106
107                XYPlot plot = (XYPlot)chart.getPlot();
108                plot.setRangeGridlinesVisible(false);
109                plot.setDomainGridlinesVisible(false);
110                plot.setBackgroundPaint(Color.white);
111
112                NumberAxis yAxis = (NumberAxis)plot.getRangeAxis();
113                yAxis.setAutoRangeIncludesZero(false);
114
115                XYLineAndShapeRenderer renderer = (XYLineAndShapeRenderer)plot.getRenderer();
116                renderer.setSeriesShapesVisible(0, true);
117                renderer.setSeriesLinesVisible(0, false);
118                renderer.setSeriesItemLabelsVisible(0, true);
119
120                renderer.setBaseItemLabelGenerator(new LabelGenerator());
121
122                Paint[] paints = DefaultDrawingSupplier.DEFAULT_PAINT_SEQUENCE;
123                Shape[] shapes = DefaultDrawingSupplier.DEFAULT_SHAPE_SEQUENCE;
124
125                for (int i = 0; i < vehicles.size(); i++) {
126                        int s = 3 * i;
127
128                        renderer.setSeriesItemLabelsVisible(s + 1, true);
129                        renderer.setSeriesItemLabelsVisible(s + 2, true);
130                        renderer.setSeriesItemLabelsVisible(s + 3, true);
131
132                        renderer.setSeriesShapesVisible(s + 1, true);
133                        renderer.setSeriesShapesVisible(s + 2, true);
134                        renderer.setSeriesShapesVisible(s + 3, true);
135
136                        renderer.setSeriesLinesVisible(s + 1, true);
137                        renderer.setSeriesLinesVisible(s + 2, true);
138                        renderer.setSeriesLinesVisible(s + 3, true);
139
140                        renderer.setSeriesPaint(s + 1, paints[(i + 1) % paints.length]);
141                        renderer.setSeriesPaint(s + 2, paints[(i + 1) % paints.length]);
142                        renderer.setSeriesPaint(s + 3, paints[(i + 1) % paints.length]);
143
144                        renderer.setSeriesShape(s + 1, shapes[(i + 1) % shapes.length]);
145                        renderer.setSeriesShape(s + 2, shapes[(i + 1) % shapes.length]);
146                        renderer.setSeriesShape(s + 3, shapes[(i + 1) % shapes.length]);
147
148                        renderer.setSeriesStroke(s + 2, new BasicStroke(3));
149                        renderer.setSeriesStroke(s + 3,
150                                        new BasicStroke(1, BasicStroke.CAP_ROUND, BasicStroke.JOIN_ROUND, 1, new float[] { 5f, 5f }, 0));
151                }
152
153                return chart;
154        }
155
156        private static Map<TaskStatus, CoordSource> createLinkSourceByStatus(Schedule schedule) {
157                Stream<DriveTask> tasks = Schedules.driveTasks(schedule);
158
159                // creating lists of DriveTasks
160                Map<TaskStatus, List<DriveTask>> taskListByStatus = tasks.collect(Collectors.groupingBy(t -> t.getStatus()));
161
162                // creating LinkSources
163                Map<TaskStatus, CoordSource> linkSourceByStatus = new EnumMap<>(TaskStatus.class);
164                for (TaskStatus ts : TaskStatus.values()) {
165                        linkSourceByStatus.put(ts, ScheduleCoordSources.createCoordSource(taskListByStatus.get(ts)));
166                }
167
168                return linkSourceByStatus;
169        }
170
171        private static class LabelGenerator implements XYItemLabelGenerator {
172                public String generateLabel(XYDataset dataset, int series, int item) {
173                        if (series == 0) {
174                                return "D";
175                        }
176
177                        if (item == 0) {
178                                return null;
179                        }
180
181                        return ((CoordDataset)dataset).getText(series, item);
182                }
183        }
