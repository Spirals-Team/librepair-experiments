001package org.matsim.contrib.util.distance;
002
003import org.matsim.api.core.v01.BasicLocation;
004import org.matsim.api.core.v01.Coord;
005
006public class DistanceUtils {
007        public static double calculateDistance(BasicLocation<?> fromLocation, BasicLocation<?> toLocation) {
008                return calculateDistance(fromLocation.getCoord(), toLocation.getCoord());
009        }
010
011        public static double calculateSquaredDistance(BasicLocation<?> fromLocation, BasicLocation<?> toLocation) {
012                return calculateSquaredDistance(fromLocation.getCoord(), toLocation.getCoord());
013        }
014
015        /**
016         * @return distance (for distance-based comparison/sorting, consider using the squared distance)
017         */
018        public static double calculateDistance(Coord fromCoord, Coord toCoord) {
019                return Math.sqrt(calculateSquaredDistance(fromCoord, toCoord));
020        }
021
022        /**
023         * @return SQUARED distance (to avoid unnecessary Math.sqrt() calls when comparing distances)
024         */
025        public static double calculateSquaredDistance(Coord fromCoord, Coord toCoord) {
026                double deltaX = toCoord.getX() - fromCoord.getX();
027                double deltaY = toCoord.getY() - fromCoord.getY();
028                return deltaX * deltaX + deltaY * deltaY;
029        }
030}
