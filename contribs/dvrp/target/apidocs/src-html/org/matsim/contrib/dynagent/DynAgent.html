061        private DynActivity dynActivity;
062
063        // =====
064
065        public DynAgent(Id<Person> id, Id<Link> startLinkId, EventsManager events, DynAgentLogic agentLogic) {
066                this.id = id;
067                this.currentLinkId = startLinkId;
068                this.agentLogic = agentLogic;
069                this.events = events;
070
071                // initial activity
072                dynActivity = this.agentLogic.computeInitialActivity(this);
073                state = Time.isUndefinedTime(dynActivity.getEndTime()) ? MobsimAgent.State.ABORT : MobsimAgent.State.ACTIVITY;
074        }
075
076        private void computeNextAction(DynAction oldDynAction, double now) {
077                oldDynAction.finalizeAction(now);
078
079                state = null;// !!! this is important
080                dynActivity = null;
081                dynLeg = null;
082
083                DynAction nextDynAction = agentLogic.computeNextAction(oldDynAction, now);
084
085                if (nextDynAction instanceof DynActivity) {
086                        dynActivity = (DynActivity)nextDynAction;
087                        state = MobsimAgent.State.ACTIVITY;
088                        events.processEvent(new ActivityStartEvent(now, id, currentLinkId, null, dynActivity.getActivityType()));
089                } else {
090                        dynLeg = (DynLeg)nextDynAction;
091                        state = MobsimAgent.State.LEG;
092                }
093        }
094
095        @Override
096        public void endActivityAndComputeNextState(double now) {
097                events.processEvent(new ActivityEndEvent(now, id, currentLinkId, null, dynActivity.getActivityType()));
098                computeNextAction(dynActivity, now);
099        }
100
101        @Override
102        public void endLegAndComputeNextState(double now) {
103                events.processEvent(new PersonArrivalEvent(now, id, currentLinkId, dynLeg.getMode()));
104                computeNextAction(dynLeg, now);
105        }
106
107        @Override
108        public void setStateToAbort(double now) {
109                this.state = MobsimAgent.State.ABORT;
110        }
111
112        public DynAgentLogic getAgentLogic() {
113                return agentLogic;
114        }
115
116        public DynAction getCurrentAction() {
117                switch (state) {
118                        case ACTIVITY:
119                                return dynActivity;
120
121                        case LEG:
122                                return dynLeg;
123
124                        default:
125                                throw new IllegalStateException();
126                }
127        }
128

137        }
138

140        public String getMode() {

142        }
