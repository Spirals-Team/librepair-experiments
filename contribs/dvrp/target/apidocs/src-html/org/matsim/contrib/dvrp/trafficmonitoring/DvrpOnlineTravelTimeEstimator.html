019package org.matsim.contrib.dvrp.trafficmonitoring;
020
021import javax.inject.Inject;
022
023import org.matsim.api.core.v01.network.Link;
024import org.matsim.api.core.v01.population.Person;
025import org.matsim.contrib.dvrp.run.DvrpConfigGroup;
026import org.matsim.core.mobsim.framework.MobsimTimer;
027import org.matsim.core.mobsim.framework.events.MobsimInitializedEvent;
028import org.matsim.core.mobsim.framework.listeners.MobsimInitializedListener;
029import org.matsim.core.mobsim.qsim.QSim;
030import org.matsim.vehicles.Vehicle;
031import org.matsim.withinday.trafficmonitoring.WithinDayTravelTime;
032
033/**
034 * Online estimation of travel times for VrpOptimizer by combining WithinDayTravelTime and
035 * DvrpOfflineTravelTimeEstimator. The beta coefficient is provided in seconds and should be either 0 (no online
036 * estimation) or positive (mixed online-offline estimation). If beta is sufficiently large, 'beta >> 0', only the
037 * currently observed TT is used
038 * 
039 * @author michalm
040 */
041public class DvrpOnlineTravelTimeEstimator implements DvrpTravelTimeEstimator, MobsimInitializedListener {
042        private final WithinDayTravelTime withinDayTT;
043        private final DvrpOfflineTravelTimeEstimator offlineTTEstimator;
044        private MobsimTimer mobsimTimer;
045        private final double beta;
046
047        @Inject
048        public DvrpOnlineTravelTimeEstimator(WithinDayTravelTime withinDayTT,
049                        DvrpOfflineTravelTimeEstimator offlineTTEstimator, DvrpConfigGroup dvrpConfig) {
050                this.withinDayTT = withinDayTT;
051                this.offlineTTEstimator = offlineTTEstimator;
052
053                beta = dvrpConfig.getTravelTimeEstimationBeta();
054                if (beta < 0) {
055                        throw new IllegalArgumentException("travelTimeEstimationBeta must be zero or positive");
056                }
057        }
058
059        @Override
060        public double getLinkTravelTime(Link link, double time, Person person, Vehicle vehicle) {
061                double currentTime = mobsimTimer.getTimeOfDay();
062                double correction = Math.min(1, Math.max(0, 1 - (time - currentTime) / beta));
063                double currentTT = withinDayTT.getLinkTravelTime(link, currentTime, person, vehicle);
064                double offlineTT = offlineTTEstimator.getLinkTravelTime(link, time, person, vehicle);
065                return correction * currentTT + (1 - correction) * offlineTT;
066
067                // XXX Alternatively:
068                // double currentOfflineTT = offlineTTEstimator.getLinkTravelTime(link, currentTime, person, vehicle);
069                // return correction * currentTT * offlineTT / currentOfflineTT + (1-correction) * offlineTT
070        }
071
072        @Override
073        public void notifyMobsimInitialized(@SuppressWarnings("rawtypes") MobsimInitializedEvent e) {
074                mobsimTimer = ((QSim)e.getQueueSimulation()).getSimTimer();
075        }
076}
