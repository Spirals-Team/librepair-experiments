020package org.matsim.contrib.dvrp.tracker;
021
022import org.matsim.contrib.dvrp.data.Vehicle;
023import org.matsim.contrib.dvrp.optimizer.VrpOptimizerWithOnlineTracking;
024import org.matsim.contrib.dvrp.schedule.DriveTask;
025import org.matsim.contrib.dvrp.schedule.StayTask;
026import org.matsim.contrib.dvrp.schedule.Task;
027import org.matsim.contrib.dvrp.schedule.Task.TaskStatus;
028import org.matsim.contrib.dvrp.vrpagent.VrpLeg;
029import org.matsim.contrib.dynagent.DynAction;
030import org.matsim.contrib.dynagent.DynActivity;
031import org.matsim.core.mobsim.framework.MobsimTimer;
032
033/**
034 * General assumptions:
035 * <ul>
036 * <li>An offline tracker knows/uses only the corresponding task and the schedule (i.e. plan)</li>
037 * <li>An online tracker knows/uses also the corresponding {@link DynAction} (i.e. execution)</li>
038 * </ul>
039 * 
040 * @author michalm
041 */
042public class TaskTrackers {
043        public static void initOnlineDriveTaskTracking(Vehicle vehicle, VrpLeg vrpDynLeg,
044                        VrpOptimizerWithOnlineTracking optimizer, MobsimTimer timer) {
045                initOnlineDriveTaskTracking(vehicle, vrpDynLeg,
046                                new OnlineDriveTaskTrackerImpl(vehicle, vrpDynLeg, optimizer, timer));
047        }
048
049        public static void initOnlineDriveTaskTracking(Vehicle vehicle, VrpLeg vrpDynLeg,
050                        OnlineDriveTaskTracker onlineTracker) {
051                DriveTask driveTask = (DriveTask)vehicle.getSchedule().getCurrentTask();
052                driveTask.initTaskTracker(onlineTracker);
053                vrpDynLeg.initOnlineTracking(onlineTracker);
054        }
055
056        public static void initOnlineStayTaskTracking(StayTask stayTask, final DynActivity dynActivity) {
057                stayTask.initTaskTracker(() -> dynActivity.getEndTime());
058        }
059
060        public static void initOfflineTaskTracking(final Task task, final MobsimTimer timer) {
061                task.initTaskTracker(() -> TaskTrackers.predictEndTimeOffline(task, timer.getTimeOfDay()));
062        }
063
064        public static double predictEndTime(Task task, double currentTime) {
065                if (task.getStatus() != TaskStatus.STARTED) {
066                        throw new IllegalStateException();
067                }
068
069                TaskTracker tracker = task.getTaskTracker();
070                return tracker != null ? //
071                                tracker.predictEndTime() : predictEndTimeOffline(task, currentTime);
072        }
073
074        private static double predictEndTimeOffline(Task task, double currentTime) {
075                return Math.max(task.getEndTime(), currentTime);
076        }
077}
