020package org.matsim.contrib.util.timeprofile;
021
022import java.util.ArrayList;
023import java.util.List;
024import java.util.function.BiConsumer;
025import java.util.function.Function;
026
027import org.jfree.chart.JFreeChart;
028import org.matsim.contrib.util.CompactCSVWriter;
029import org.matsim.contrib.util.chart.ChartSaveUtils;
030import org.matsim.contrib.util.timeprofile.TimeProfileCharts.ChartType;
031import org.matsim.core.controler.MatsimServices;
032import org.matsim.core.mobsim.framework.events.MobsimBeforeCleanupEvent;
033import org.matsim.core.mobsim.framework.events.MobsimBeforeSimStepEvent;
034import org.matsim.core.mobsim.framework.listeners.MobsimBeforeCleanupListener;
035import org.matsim.core.mobsim.framework.listeners.MobsimBeforeSimStepListener;
036import org.matsim.core.utils.io.IOUtils;
037import org.matsim.core.utils.misc.Time;
038
039public class TimeProfileCollector implements MobsimBeforeSimStepListener, MobsimBeforeCleanupListener {
040        public interface ProfileCalculator {
041                String[] getHeader();
042
043                Object[] calcValues();
044        }
045
046        private final ProfileCalculator calculator;
047        private final List<Double> times = new ArrayList<>();
048        private final List<Object[]> timeProfile = new ArrayList<>();
049        private final int interval;
050        private final String outputFile;
051        private final MatsimServices matsimServices;
052
053        private Function<Object[], String[]> valuesToStringsConverter = TimeProfiles::combineValuesIntoStrings;
054        private BiConsumer<JFreeChart, ChartType> chartCustomizer;
055        private ChartType[] chartTypes = { ChartType.Line };
056
057        public TimeProfileCollector(ProfileCalculator calculator, int interval, String outputFile,
058                        MatsimServices matsimServices) {
059                this.calculator = calculator;
060                this.interval = interval;
061                this.outputFile = outputFile;
062                this.matsimServices = matsimServices;
063        }
064
065        @Override
066        public void notifyMobsimBeforeSimStep(@SuppressWarnings("rawtypes") MobsimBeforeSimStepEvent e) {
067                if (e.getSimulationTime() % interval == 0) {
068                        times.add(e.getSimulationTime());
069                        timeProfile.add(calculator.calcValues());
070                }
071        }
072
073        public void setValuesToStringsConverter(Function<Object[], String[]> valuesToStringsConverter) {
074                this.valuesToStringsConverter = valuesToStringsConverter;
075        }
076
077        public void setChartCustomizer(BiConsumer<JFreeChart, ChartType> chartCustomizer) {
078                this.chartCustomizer = chartCustomizer;
079        }
080
081        public void setChartTypes(ChartType... chartTypes) {
082                this.chartTypes = chartTypes;
083        }
084
085        @Override
086        public void notifyMobsimBeforeCleanup(@SuppressWarnings("rawtypes") MobsimBeforeCleanupEvent e) {
087                String file = matsimServices.getControlerIO().getIterationFilename(matsimServices.getIterationNumber(),
088                                outputFile);
089                String timeFormat = interval % 60 == 0 ? Time.TIMEFORMAT_HHMM : Time.TIMEFORMAT_HHMMSS;
090
091                try (CompactCSVWriter writer = new CompactCSVWriter(IOUtils.getBufferedWriter(file + ".txt"))) {
092                        writer.writeNext("time", calculator.getHeader());
093                        for (int i = 0; i < timeProfile.size(); i++) {
094                                writer.writeNext(Time.writeTime(times.get(i), timeFormat), //
095                                                valuesToStringsConverter.apply(timeProfile.get(i)));
096                        }
097                }
098
099                for (ChartType t : chartTypes) {
100                        generateImage(t);
101                }
102        }
103
104        private void generateImage(ChartType chartType) {
105                JFreeChart chart = TimeProfileCharts.chartProfile(calculator.getHeader(), times, timeProfile, chartType);
106                if (chartCustomizer != null) {
107                        chartCustomizer.accept(chart, chartType);
108                }
109
110                String imageFile = matsimServices.getControlerIO().getIterationFilename(matsimServices.getIterationNumber(),
111                                outputFile + "_" + chartType.name());
112                ChartSaveUtils.saveAsPNG(chart, imageFile, 1500, 1000);
113        }
114}
