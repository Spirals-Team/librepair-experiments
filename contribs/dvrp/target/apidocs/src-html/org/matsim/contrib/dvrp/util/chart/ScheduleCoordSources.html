020package org.matsim.contrib.dvrp.util.chart;
021
022import java.util.List;
023import java.util.stream.Collectors;
024
025import org.matsim.api.core.v01.Coord;
026import org.matsim.contrib.dvrp.schedule.DriveTask;
027import org.matsim.contrib.dvrp.schedule.Schedule;
028import org.matsim.contrib.dvrp.schedule.Schedules;
029import org.matsim.contrib.util.chart.CoordDataset.CoordSource;
030
031public class ScheduleCoordSources {
032        // n DriveTasks -> n+1 Links
033        public static CoordSource createCoordSource(final List<DriveTask> tasks) {
034                return new CoordSource() {
035
036                        @Override
037                        public Coord getCoord(int item) {
038                                if (item == 0) {
039                                        return tasks.get(0).getPath().getFromLink().getCoord();
040                                }
041
042                                return tasks.get(item - 1).getPath().getToLink().getCoord();
043                        }
044
045                        @Override
046                        public int getCount() {
047                                int size = tasks.size();
048                                return size == 0 ? 0 : size + 1;
049                        }
050                };
051        }
052
053        // Schedule -> n DriveTasks -> n+1 Links
054        public static CoordSource createCoordSource(Schedule schedule) {
055                List<DriveTask> driveTasks = Schedules.driveTasks(schedule).collect(Collectors.toList());
056                return ScheduleCoordSources.createCoordSource(driveTasks);
057        }
058}
