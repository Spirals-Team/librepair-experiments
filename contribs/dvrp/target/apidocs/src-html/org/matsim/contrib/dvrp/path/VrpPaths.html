024import org.matsim.api.core.v01.Id;
025import org.matsim.api.core.v01.network.Link;
026import org.matsim.contrib.dvrp.path.OneToManyPathSearch.PathData;

033public class VrpPaths {
034        /**
035         * ASSUMPTION: A vehicle enters and exits links at their ends (link.getToNode())
036         */
037        public static VrpPathWithTravelData calcAndCreatePath(Link fromLink, Link toLink, double departureTime,
038                        LeastCostPathCalculator router, TravelTime travelTime) {
039                Path path = null;
040                if (fromLink != toLink) {
041                        // calc path for departureTime+1 (we need 1 second to move over the node)
042                        path = router.calcLeastCostPath(fromLink.getToNode(), toLink.getFromNode(), departureTime + 1, null, null);
043                }
044
045                return VrpPaths.createPath(fromLink, toLink, departureTime, path, travelTime);
046        }
047
048        public static VrpPathWithTravelData createZeroLengthPath(Link fromTolink, double departureTime) {
049                return new VrpPathWithTravelDataImpl(departureTime, 0, new Link[] { fromTolink }, new double[] { 0 });
050        }
051
052        public static VrpPathWithTravelData createPath(Link fromLink, Link toLink, double departureTime, PathData pathData,
053                        TravelTime travelTime) {
054                return createPath(fromLink, toLink, departureTime, pathData.path, travelTime);
055        }
056
057        public static VrpPathWithTravelData createPath(Link fromLink, Link toLink, double departureTime, Path path,
058                        TravelTime travelTime) {
059                if (fromLink == toLink) {
060                        return createZeroLengthPath(fromLink, departureTime);
061                }
062
063                int count = path.links.size();
064                if (count > 0) {
065                        if (fromLink.getToNode() != path.links.get(0).getFromNode()) {
066                                throw new IllegalArgumentException("fromLink and path are not connected; fromLink: " + fromLink
067                                                + "\n path beg" + path.links.get(0));
068                        }
069                        if (path.links.get(count - 1).getToNode() != toLink.getFromNode()) {
070                                throw new IllegalArgumentException("path and toLink are not connected; path end:"
071                                                + path.links.get(count - 1).toString() + "\n toLink: " + toLink.toString());
072                        }
073                }
074
075                Link[] links = new Link[count + 2];
076                double[] linkTTs = new double[count + 2];
077
078                // we start at the end of fromLink
079                // actually, in QSim, it usually takes 1 second to move over the first node
080                // (when INSERTING_WAITING_VEHICLES_BEFORE_DRIVING_VEHICLES is ON;
081                // otherwise it can take much longer)
082                double currentTime = departureTime;
083                links[0] = fromLink;
084                double linkTT = FIRST_LINK_TT;
085                linkTTs[0] = linkTT;
086                currentTime += linkTT;
087
088                for (int i = 1; i <= count; i++) {
089                        Link link = path.links.get(i - 1);
090                        links[i] = link;
091                        linkTT = travelTime.getLinkTravelTime(link, currentTime, null, null);
092                        linkTTs[i] = linkTT;
093                        currentTime += linkTT;
094                }
095
096                // there is no extra time spent on queuing at the end of the last link
097                links[count + 1] = toLink;
098                linkTT = getLastLinkTT(toLink, currentTime);// as long as we cannot divert from the last link this is okay
099                linkTTs[count + 1] = linkTT;
100                double totalTT = FIRST_LINK_TT + path.travelTime + linkTT;
101
102                return new VrpPathWithTravelDataImpl(departureTime, totalTT, links, linkTTs);
103        }
104
105        static final double FIRST_LINK_TT = 1;
106
107        static double getLastLinkTT(Link lastLink, double time) {
108                // XXX imprecise if qsimCfg.timeStepSize != 1
109                return Math.floor(lastLink.getLength() / lastLink.getFreespeed(time));
110        }
111
112        /**
113         * Used for OTFVis. Does not contain info on timing, distance and cost. Can be extended...
114         * 
115         * @param path
116         * @param routeFactories
117         * @return
118         */
119        public static NetworkRoute createNetworkRoute(VrpPath path, RouteFactories routeFactories) {
120                Id<Link> fromLinkId = path.getFromLink().getId();
121                Id<Link> toLinkId = path.getToLink().getId();
122                NetworkRoute route = routeFactories.createRoute(NetworkRoute.class, fromLinkId, toLinkId);
123
124                int length = path.getLinkCount();
125                if (length >= 2) {// means: fromLink != toLink
126                        // all except the first and last ones (== fromLink and toLink)
127                        ArrayList<Id<Link>> linkIdList = new ArrayList<>(length - 2);
128                        for (int i = 1; i < length - 1; i++) {
129                                linkIdList.add(path.getLink(i).getId());
130                        }
131                        route.setLinkIds(fromLinkId, linkIdList, toLinkId);
132                }
133
134                return route;
135        }
136
137        /**
138         * @return The distance of a VRP path Includes the to link, but not the from link
139         */
140        public static double calcDistance(VrpPath path) {
141                double distance = 0.0;
142                for (int i = 1; i < path.getLinkCount(); i++) {
143                        distance += path.getLink(i).getLength();
144                }
145                return distance;
146        }
