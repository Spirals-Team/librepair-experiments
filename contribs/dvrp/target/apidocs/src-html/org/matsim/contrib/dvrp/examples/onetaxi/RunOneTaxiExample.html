020package org.matsim.contrib.dvrp.examples.onetaxi;
021
022import java.util.Collections;
023
024import org.matsim.api.core.v01.Scenario;
025import org.matsim.contrib.dvrp.data.file.FleetProvider;
026import org.matsim.contrib.dvrp.optimizer.VrpOptimizer;
027import org.matsim.contrib.dvrp.passenger.PassengerRequestCreator;
028import org.matsim.contrib.dvrp.run.DvrpConfigConsistencyChecker;
029import org.matsim.contrib.dvrp.run.DvrpConfigGroup;
030import org.matsim.contrib.dvrp.run.DvrpModule;
031import org.matsim.contrib.dvrp.vrpagent.VrpAgentLogic.DynActionCreator;
032import org.matsim.contrib.otfvis.OTFVisLiveModule;
033import org.matsim.core.config.Config;
034import org.matsim.core.config.ConfigGroup;
035import org.matsim.core.config.ConfigUtils;
036import org.matsim.core.controler.Controler;
037import org.matsim.core.scenario.ScenarioUtils;
038import org.matsim.vis.otfvis.OTFVisConfigGroup;
039
040/**
041 * @author michalm
042 */
043public final class RunOneTaxiExample {
044        private static final String CONFIG_FILE = "./src/main/resources/one_taxi/one_taxi_config.xml";
045        private static final String TAXIS_FILE = "one_taxi_vehicles.xml";
046
047        public static void run(boolean otfvis, int lastIteration) {
048                // load config
049                Config config = ConfigUtils.loadConfig(CONFIG_FILE, new DvrpConfigGroup(), new OTFVisConfigGroup());
050                config.controler().setLastIteration(lastIteration);
051                config.addConfigConsistencyChecker(new DvrpConfigConsistencyChecker());
052                config.checkConsistency();
053
054                // load scenario
055                Scenario scenario = ScenarioUtils.loadScenario(config);
056
057                // setup controler
058                Controler controler = new Controler(scenario);
059                controler.addOverridingModule(new DvrpModule(//
060                                cfg -> new com.google.inject.AbstractModule() {
061                                        @Override
062                                        protected void configure() {
063                                                // optimizer that dispatches taxis
064                                                bind(VrpOptimizer.class).to(OneTaxiOptimizer.class).asEagerSingleton();
065                                                // converts departures of the "taxi" mode into taxi requests
066                                                bind(PassengerRequestCreator.class).to(OneTaxiRequestCreator.class).asEagerSingleton();
067                                                // converts scheduled tasks into simulated actions (legs and activities)
068                                                bind(DynActionCreator.class).to(OneTaxiActionCreator.class).asEagerSingleton();
069                                        }
070                                }, //
071                                Collections.emptySet()));// no mobsim listeners necessary in OneTaxiExample
072
073                controler.addOverridingModule(
074                                FleetProvider.createModule(ConfigGroup.getInputFileURL(config.getContext(), TAXIS_FILE)));
075
076                if (otfvis) {
077                        controler.addOverridingModule(new OTFVisLiveModule()); // OTFVis visualisation
078                }
079
080                // run simulation
081                controler.run();
082        }
083
084        public static void main(String... args) {
085                run(true, 0); // switch to 'false' to turn off visualisation
086        }
087}
