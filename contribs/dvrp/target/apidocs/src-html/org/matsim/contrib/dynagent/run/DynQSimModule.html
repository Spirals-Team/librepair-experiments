020package org.matsim.contrib.dynagent.run;
021
022import java.util.*;
023
024import org.matsim.core.config.Config;
025import org.matsim.core.controler.AbstractModule;
026import org.matsim.core.mobsim.framework.Mobsim;
027import org.matsim.core.mobsim.qsim.*;
028import org.matsim.core.mobsim.qsim.changeeventsengine.NetworkChangeEventsPlugin;
029import org.matsim.core.mobsim.qsim.messagequeueengine.MessageQueuePlugin;
030import org.matsim.core.mobsim.qsim.pt.TransitEnginePlugin;
031import org.matsim.core.mobsim.qsim.qnetsimengine.QNetsimEnginePlugin;
032
033import com.google.inject.Provides;
034
035public class DynQSimModule<T extends Mobsim> extends AbstractModule {
036        private final Class<? extends javax.inject.Provider<? extends T>> providerClass;
037
038        public DynQSimModule(Class<? extends javax.inject.Provider<? extends T>> providerClass) {
039                this.providerClass = providerClass;
040        }
041
042        @Override
043        public void install() {
044                bindMobsim().toProvider(providerClass);
045        }
046
047        @Provides
048        private Collection<AbstractQSimPlugin> provideQSimPlugins(Config config) {
049                return createQSimPlugins(config);
050        }
051
052        public static Collection<AbstractQSimPlugin> createQSimPlugins(Config config) {
053                final Collection<AbstractQSimPlugin> plugins = new ArrayList<>();
054                plugins.add(new MessageQueuePlugin(config));
055                plugins.add(new DynActivityEnginePlugin(config));
056                plugins.add(new QNetsimEnginePlugin(config));
057                if (config.network().isTimeVariantNetwork()) {
058                        plugins.add(new NetworkChangeEventsPlugin(config));
059                }
060                if (config.transit().isUseTransit()) {
061                        plugins.add(new TransitEnginePlugin(config));
062                }
063                plugins.add(new TeleportationPlugin(config));
064                plugins.add(new PopulationPlugin(config));
065                return plugins;
066        }
067}
