020package org.matsim.contrib.dvrp.passenger;
021
022import org.matsim.contrib.dvrp.schedule.StayTask;
023import org.matsim.contrib.dynagent.*;
024import org.matsim.core.mobsim.framework.MobsimPassengerAgent;
025
026public class SinglePassengerPickupActivity extends AbstractDynActivity implements PassengerPickupActivity {
027        private final PassengerEngine passengerEngine;
028        private final DynAgent driver;
029        private final PassengerRequest request;
030        private final double pickupDuration;
031
032        private boolean passengerAboard = false;
033        private double endTime;
034
035        public SinglePassengerPickupActivity(PassengerEngine passengerEngine, DynAgent driver, StayTask pickupTask,
036                        PassengerRequest request, double pickupDuration, String activityType) {
037                super(activityType);
038
039                this.passengerEngine = passengerEngine;
040                this.driver = driver;
041                this.request = request;
042                this.pickupDuration = pickupDuration;
043
044                double now = pickupTask.getBeginTime();
045
046                passengerAboard = passengerEngine.pickUpPassenger(this, driver, request, now);
047
048                if (passengerAboard) {
049                        endTime = now + pickupDuration;
050                } else {
051                        setEndTimeIfWaitingForPassenger(now);
052                }
053        }
054
055        @Override
056        public double getEndTime() {
057                return endTime;
058        }
059
060        @Override
061        public void doSimStep(double now) {
062                if (!passengerAboard) {
063                        setEndTimeIfWaitingForPassenger(now);// TODO use DynActivityEngine.END_ACTIVITY_LATER instead?
064                }
065        }
066
067        private void setEndTimeIfWaitingForPassenger(double now) {
068                // try to predict the passenger's arrival time
069                endTime = Math.max(now, request.getEarliestStartTime()) + pickupDuration;
070
071                if (endTime == now) {// happens only if pickupDuration == 0
072                        endTime += 1; // to prevent the driver departing now (before picking up the passenger)
073                }
074        }
075
076        @Override
077        public void notifyPassengerIsReadyForDeparture(MobsimPassengerAgent passenger, double now) {
078                if (passenger != request.getPassenger()) {
079                        throw new IllegalArgumentException("I am waiting for a different passenger!");
080                }
081
082                passengerAboard = passengerEngine.pickUpPassenger(this, driver, request, now);
083
084                if (!passengerAboard) {
085                        throw new IllegalStateException("The passenger is not on the link or not available for departure!");
086                }
087
088                endTime = now + pickupDuration;
089        }
090}
