020package org.matsim.contrib.zone;
021
022import org.matsim.api.core.v01.*;
023import org.matsim.api.core.v01.network.*;
024
025public class SquareGrid {
026        private final Network network;
027        private final double cellSize;
028
029        private double minX;
030        private double minY;
031        private double maxX;
032        private double maxY;
033
034        private int cols;
035        private int rows;
036
037        private Zone[] zones;
038
039        public SquareGrid(Network network, double cellSize) {
040                this.network = network;
041                this.cellSize = cellSize;
042
043                initBounds();
044
045                cols = (int)Math.ceil((maxX - minX) / cellSize);
046                rows = (int)Math.ceil((maxY - minY) / cellSize);
047
048                initZones();
049        }
050
051        // This method's content has been copied from NetworkImpl
052        private void initBounds() {
053                minX = Double.POSITIVE_INFINITY;
054                minY = Double.POSITIVE_INFINITY;
055                maxX = Double.NEGATIVE_INFINITY;
056                maxY = Double.NEGATIVE_INFINITY;
057                for (Node n : network.getNodes().values()) {
058                        if (n.getCoord().getX() < minX) {
059                                minX = n.getCoord().getX();
060                        }
061                        if (n.getCoord().getY() < minY) {
062                                minY = n.getCoord().getY();
063                        }
064                        if (n.getCoord().getX() > maxX) {
065                                maxX = n.getCoord().getX();
066                        }
067                        if (n.getCoord().getY() > maxY) {
068                                maxY = n.getCoord().getY();
069                        }
070                }
071                minX -= 1.0;// TODO use epsilon instead
072                minY -= 1.0;
073                maxX += 1.0;
074                maxY += 1.0;
075                // yy the above four lines are problematic if the coordinate values are much smaller than one. kai, oct'15
076        }
077
078        private void initZones() {
079                zones = new Zone[rows * cols];
080                double x0 = minX + cellSize / 2;
081                double y0 = minY + cellSize / 2;
082
083                for (int r = 0; r < rows; r++) {
084                        for (int c = 0; c < cols; c++) {
085                                int idx = r * cols + c;
086                                Coord centroid = new Coord(c * cellSize + x0, r * cellSize + y0);
087                                zones[idx] = new Zone(Id.create(idx, Zone.class), "square", centroid);
088                        }
089                }
090        }
091
092        public Zone[] getZones() {
093                return zones;
094        }
095
096        public Zone getZone(Coord coord) {
097                int r = (int)((coord.getY() - minY) / cellSize);// == Math.floor
098                int c = (int)((coord.getX() - minX) / cellSize);// == Math.floor
099                return zones[r * cols + c];
100        }
101}
