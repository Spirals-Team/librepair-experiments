020package org.matsim.contrib.dvrp.examples.onetruck;
021
022import java.util.Collection;
023
024import org.matsim.api.core.v01.Scenario;
025import org.matsim.api.core.v01.network.Network;
026import org.matsim.contrib.dvrp.data.file.FleetProvider;
027import org.matsim.contrib.dvrp.optimizer.VrpOptimizer;
028import org.matsim.contrib.dvrp.router.DvrpRoutingNetworkProvider;
029import org.matsim.contrib.dvrp.run.DvrpConfigConsistencyChecker;
030import org.matsim.contrib.dvrp.run.DvrpQSimPluginsProvider;
031import org.matsim.contrib.dvrp.trafficmonitoring.DvrpTravelTimeModule;
032import org.matsim.contrib.dvrp.vrpagent.VrpAgentLogic.DynActionCreator;
033import org.matsim.contrib.otfvis.OTFVisLiveModule;
034import org.matsim.core.config.Config;
035import org.matsim.core.config.ConfigGroup;
036import org.matsim.core.config.ConfigUtils;
037import org.matsim.core.controler.AbstractModule;
038import org.matsim.core.controler.Controler;
039import org.matsim.core.mobsim.qsim.AbstractQSimPlugin;
040import org.matsim.core.scenario.ScenarioUtils;
041import org.matsim.vis.otfvis.OTFVisConfigGroup;
042
043import com.google.inject.TypeLiteral;
044import com.google.inject.name.Names;
045
046/**
047 * @author michalm
048 */
049public final class RunOneTruckExample {
050        private static final String CONFIG_FILE = "./src/main/resources/one_truck/one_truck_config.xml";
051        private static final String TRUCK_FILE = "one_truck_vehicles.xml";
052
053        public static void run(boolean otfvis, int lastIteration) {
054                // load config
055                Config config = ConfigUtils.loadConfig(CONFIG_FILE, new OTFVisConfigGroup());
056                config.controler().setLastIteration(lastIteration);
057                config.addConfigConsistencyChecker(new DvrpConfigConsistencyChecker());
058                config.checkConsistency();
059
060                // load scenario
061                Scenario scenario = ScenarioUtils.loadScenario(config);
062
063                // setup controler
064                Controler controler = new Controler(scenario);
065                controler.addOverridingModule(new AbstractModule() {
066                        public void install() {
067                                addTravelTimeBinding(DvrpTravelTimeModule.DVRP_ESTIMATED).to(networkTravelTime());
068                                bind(Network.class).annotatedWith(Names.named(DvrpRoutingNetworkProvider.DVRP_ROUTING))
069                                                .to(Network.class);
070                                bind(new TypeLiteral<Collection<AbstractQSimPlugin>>() {})
071                                                .toProvider(createQSimPluginsProvider(getConfig()));
072                        }
073                });
074                controler.addOverridingModule(
075                                FleetProvider.createModule(ConfigGroup.getInputFileURL(config.getContext(), TRUCK_FILE)));
076
077                if (otfvis) {
078                        controler.addOverridingModule(new OTFVisLiveModule()); // OTFVis visualisation
079                }
080
081                // run simulation
082                controler.run();
083        }
084
085        private static DvrpQSimPluginsProvider createQSimPluginsProvider(Config config) {
086                return new DvrpQSimPluginsProvider(config, cfg -> {
087                        return new com.google.inject.AbstractModule() {
088                                @Override
089                                protected void configure() {
090                                        bind(OneTruckRequestCreator.class).asEagerSingleton();
091                                        bind(VrpOptimizer.class).to(OneTruckOptimizer.class).asEagerSingleton();
092                                        bind(DynActionCreator.class).to(OneTruckActionCreator.class).asEagerSingleton();
093                                }
094                        };
095                }).addListener(OneTruckRequestCreator.class)//
096                                .setAddPassengerEnginePlugin(false);
097        }
098
099        public static void main(String... args) {
100                run(false, 0); // switch to 'false' to turn off visualisation
101        }
102}
