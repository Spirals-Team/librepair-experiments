020package org.matsim.contrib.zone.util;
021
022import java.util.*;
023
024import org.matsim.api.core.v01.*;
025import org.matsim.contrib.zone.Zone;
026import org.matsim.core.utils.geometry.geotools.MGC;
027
028import com.vividsolutions.jts.geom.*;
029import com.vividsolutions.jts.index.SpatialIndex;
030import com.vividsolutions.jts.index.quadtree.Quadtree;
031
032public class ZoneFinderImpl implements ZoneFinder {
033        private final SpatialIndex quadTree = new Quadtree();
034        private final double expansionDistance;
035
036        public ZoneFinderImpl(Map<Id<Zone>, Zone> zones, double expansionDistance) {
037                this.expansionDistance = expansionDistance;
038
039                for (Zone z : zones.values()) {
040                        quadTree.insert(z.getMultiPolygon().getEnvelopeInternal(), z);
041                }
042        }
043
044        @Override
045        @SuppressWarnings("unchecked")
046        public Zone findZone(Coord coord) {
047                Point point = MGC.coord2Point(coord);
048                Envelope env = point.getEnvelopeInternal();
049
050                Zone zone = getSmallestZoneContainingPoint(quadTree.query(env), point);
051                if (zone != null) {
052                        return zone;
053                }
054
055                if (expansionDistance > 0) {
056                        env.expandBy(expansionDistance);
057                        zone = getNearestZone(quadTree.query(env), point);
058                }
059
060                return zone;
061        }
062
063        private Zone getSmallestZoneContainingPoint(List<Zone> zones, Point point) {
064                if (zones.size() == 1) {// almost 100% cases
065                        return zones.get(0);
066                }
067
068                double minArea = Double.MAX_VALUE;
069                Zone smallestZone = null;
070
071                for (Zone z : zones) {
072                        if (z.getMultiPolygon().contains(point)) {
073                                double area = z.getMultiPolygon().getArea();
074                                if (area < minArea) {
075                                        minArea = area;
076                                        smallestZone = z;
077                                }
078                        }
079                }
080
081                return smallestZone;
082        }
083
084        private Zone getNearestZone(List<Zone> zones, Point point) {
085                if (zones.size() == 1) {
086                        return zones.get(0);
087                }
088
089                double minDistance = Double.MAX_VALUE;
090                Zone nearestZone = null;
091
092                for (Zone z : zones) {
093                        double distance = z.getMultiPolygon().distance(point);
094                        if (distance <= expansionDistance) {
095                                if (distance < minDistance) {
096                                        minDistance = distance;
097                                        nearestZone = z;
098                                }
099                        }
100                }
101
102                return nearestZone;
103        }
104
105}
