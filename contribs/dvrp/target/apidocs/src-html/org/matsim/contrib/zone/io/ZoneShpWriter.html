020package org.matsim.contrib.zone.io;
021
022import java.util.*;
023
024import org.matsim.api.core.v01.Id;
025import org.matsim.contrib.zone.Zone;
026import org.matsim.core.utils.geometry.geotools.MGC;
027import org.matsim.core.utils.gis.*;
028import org.opengis.feature.simple.SimpleFeature;
029import org.opengis.referencing.crs.CoordinateReferenceSystem;
030
031public class ZoneShpWriter {
032        public static final String ID_HEADER = "ID";
033
034        private final Map<Id<Zone>, Zone> zones;
035        private final String coordinateSystem;
036
037        public ZoneShpWriter(Map<Id<Zone>, Zone> zones, String coordinateSystem) {
038                this.zones = zones;
039                this.coordinateSystem = coordinateSystem;
040        }
041
042        public void write(String shpFile) {
043                CoordinateReferenceSystem crs = MGC.getCRS(coordinateSystem);
044
045                PolygonFeatureFactory factory = new PolygonFeatureFactory.Builder().addAttribute(ID_HEADER, String.class)
046                                .setCrs(crs).setName("zone").create();
047
048                List<SimpleFeature> features = new ArrayList<>();
049                for (Zone z : zones.values()) {
050                        String id = z.getId() + "";
051                        features.add(factory.createPolygon(z.getMultiPolygon(), new Object[] { id }, id));
052                }
053
054                ShapeFileWriter.writeGeometries(features, shpFile);
055        }
056}
