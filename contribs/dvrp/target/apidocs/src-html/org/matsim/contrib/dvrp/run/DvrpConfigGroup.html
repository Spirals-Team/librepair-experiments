020package org.matsim.contrib.dvrp.run;
021
022import java.util.Map;
023
024import javax.annotation.Nullable;
025import javax.validation.constraints.Max;
026import javax.validation.constraints.NotNull;
027import javax.validation.constraints.Positive;
028import javax.validation.constraints.PositiveOrZero;
029
030import org.matsim.api.core.v01.TransportMode;
031import org.matsim.core.config.Config;
032import org.matsim.core.config.ReflectiveConfigGroup;
033
034public class DvrpConfigGroup extends ReflectiveConfigGroup {
035
036        public static final String GROUP_NAME = "dvrp";
037        @SuppressWarnings("deprecation")
038        public static DvrpConfigGroup get(Config config) {
039                return (DvrpConfigGroup)config.getModule(GROUP_NAME);// will fail if not in the config
040        }
041
042        public static final String MODE = "mode";
043        static final String MODE_EXP = "Mode which will be handled by PassengerEngine and VrpOptimizer "
044                        + "(passengers'/customers' perspective)";
045
046        public static final String NETWORK_MODE = "networkMode";
047        static final String NETWORK_MODE_EXP = "Mode of which the network will be used for routing vehicles, calculating trave times, "
048                        + "etc. (fleet operator's perspective). " + "Default is car.";
049
050        public static final String TRAVEL_TIME_ESTIMATION_ALPHA = "travelTimeEstimationAlpha";
051        static final String TRAVEL_TIME_ESTIMATION_ALPHA_EXP = "Used for OFFLINE estimation of travel times for VrpOptimizer"
052                        + " by means of the exponential moving average."
053                        + " The weighting decrease, alpha, must be in (0,1]."
054                        + " We suggest small values of alpha, e.g. 0.05."
055                        + " The averaging starts from the initial travel time estimates. If not provided,"
056                        + " the free-speed TTs is used as the initial estimates";
057
058        public static final String TRAVEL_TIME_ESTIMATION_BETA = "travelTimeEstimationBeta";
059        static final String TRAVEL_TIME_ESTIMATION_BETA_EXP = "Used for ONLINE estimation of travel times for VrpOptimizer"
060                        + " by combining WithinDayTravelTime and DvrpOfflineTravelTimeEstimator."
061                        + " The beta coefficient is provided in seconds and should be either 0 (no online estimation)"
062                        + " or positive (mixed online-offline estimation)."
063                        /////
064                        + " For 'beta = 0', only the offline estimate is used:" + " 'onlineTT(t) = offlineTT(t)',"
065                        + " where 'offlineTT(t)' in the offline estimate for TT at time 't',"
066                        /////
067                        + " For 'beta > 0', estimating future TTs at time 't',"
068                        + " uses the currently observed TT to correct the offline estimates is made:"
069                        + " where 'currentTT' is the currently observed TT,"
070                        + " and 'correction = min(1, max(0, 1 - (time - currentTime) / beta))'"
071                        ////
072                        + " The rule is that correction decreases linearly from 1 (when 'time = currentTime')"
073                        + " to 0 (when 'time = currentTime + beta'"
074                        + " For 'time > currentTime + beta' correction is 0,"
075                        + " whereas if 'time < currentTime' it is 1."
076                        ////
077                        + " If beta is sufficiently large, 'beta >> 0', only the currently observed TT is used.";
078        // In DVRP 'time < currentTime' may only happen for backward path search, a adding proper search termination
079        // criterion should prevent this from happening
080
081        @NotNull
082        private String mode = null; // travel mode (passengers'/customers' perspective)
083
084        @Nullable
085        private String networkMode = TransportMode.car; // used for building routes, calculating travel times, etc.
086        // (dispatcher's perspective); null ==> no filtering (routing network equals scenario.network)
087
088        @Positive
089        @Max(1)
090        private double travelTimeEstimationAlpha = 0.05; // [-], 1 ==> TTs from the last iteration only
091
092        @PositiveOrZero
093        private double travelTimeEstimationBeta = 0; // [s], 0 ==> only offline TT estimation
094
095        public DvrpConfigGroup() {
096                super(GROUP_NAME);
097        }
098
099        @Override
100        public Map<String, String> getComments() {
101                Map<String, String> map = super.getComments();
102                map.put(MODE, MODE_EXP);
103                map.put(NETWORK_MODE, NETWORK_MODE_EXP);
104                map.put(TRAVEL_TIME_ESTIMATION_ALPHA, //
105                                TRAVEL_TIME_ESTIMATION_ALPHA_EXP);
106                map.put(TRAVEL_TIME_ESTIMATION_BETA,
107                                TRAVEL_TIME_ESTIMATION_BETA_EXP);
108                return map;
109        }
110
111        /**
112         * 
113         * @return -- {@value #MODE_EXP}}
114         */
115        @StringGetter(MODE)
116        public String getMode() {
117                return mode;
118        }
119
120        /**
121         * 
122         * @param -- {@value #MODE_EXP}
123         */
124        @StringSetter(MODE)
125        public void setMode(String mode) {
126                this.mode = mode;
127        }
128        /**
129         * 
130         * @return -- {@value #NETWORK_MODE_EXP}
131         */
132        @StringGetter(NETWORK_MODE)
133        public String getNetworkMode() {
134                return networkMode;
135        }
136
137        /**
138         * 
139         * @param -- {@value #NETWORK_MODE_EXP}
140         */
141        @StringSetter(NETWORK_MODE)
142        public void setNetworkMode(String routingMode) {
143                this.networkMode = routingMode;
144        }
145
146        /**
147         * 
148         * @return -- {@value #TRAVEL_TIME_ESTIMATION_ALPHA_EXP}
149         */
150        @StringGetter(TRAVEL_TIME_ESTIMATION_ALPHA)
151        public double getTravelTimeEstimationAlpha() {
152                return travelTimeEstimationAlpha;
153        }
154
155        /**
156         * 
157         * @value -- {@value #TRAVEL_TIME_ESTIMATION_ALPHA_EXP}
158         */
159        @StringSetter(TRAVEL_TIME_ESTIMATION_ALPHA)
160        public void setTravelTimeEstimationAlpha(double travelTimeEstimationAlpha) {
161                this.travelTimeEstimationAlpha = travelTimeEstimationAlpha;
162        }
163
164        /**
165         * 
166         * @return -- {@value #TRAVEL_TIME_ESTIMATION_BETA_EXP}
167         */
168        @StringGetter(TRAVEL_TIME_ESTIMATION_BETA)
169        public double getTravelTimeEstimationBeta() {
170                return travelTimeEstimationBeta;
171        }
172
173        /**
174         * 
175         * @param -- {@value #TRAVEL_TIME_ESTIMATION_BETA_EXP}
176         */
177        @StringSetter(TRAVEL_TIME_ESTIMATION_BETA)
178        public void setTravelTimeEstimationBeta(double travelTimeEstimationBeta) {
179                this.travelTimeEstimationBeta = travelTimeEstimationBeta;
180        }
181}
