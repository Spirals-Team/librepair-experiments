020package org.matsim.contrib.socnetgen.sna.snowball.social;
021
022import com.vividsolutions.jts.geom.Point;
023import org.matsim.api.core.v01.Coord;
024import org.matsim.contrib.socnetgen.sna.graph.social.SocialPerson;
025import org.matsim.contrib.socnetgen.sna.graph.social.SocialVertex;
026import org.matsim.contrib.socnetgen.sna.snowball.SampledVertexDecorator;
027
028import java.util.List;
029
030/**
031 * @author illenberger
032 *
033 */
034public class SocialSampledVertexDecorator<V extends SocialVertex> extends SampledVertexDecorator<V> implements SocialVertex {
035
036        protected SocialSampledVertexDecorator(V delegate) {
037                super(delegate);
038        }
039
040        @SuppressWarnings("unchecked")
041        @Override
042        public List<? extends SocialSampledEdgeDecorator<?>> getEdges() {
043                return (List<? extends SocialSampledEdgeDecorator<?>>) super.getEdges();
044        }
045
046        @SuppressWarnings("unchecked")
047        @Override
048        public List<? extends SocialSampledVertexDecorator<V>> getNeighbours() {
049                return (List<? extends SocialSampledVertexDecorator<V>>) super.getNeighbours();
050        }
051
052        @Override
053        public SocialPerson getPerson() {
054                return getDelegate().getPerson();
055        }
056
057        /**
058         * @deprecated
059         */
060        @Override
061        public Coord getCoordinate() {
062                return getDelegate().getCoordinate();
063        }
064
065        @Override
066        public Point getPoint() {
067                return getDelegate().getPoint();
068        }
069        
070        public String toString() {
071                StringBuilder builder = new StringBuilder();
072                builder.append("id=");
073                builder.append(getDelegate().getPerson().getId().toString());
074                builder.append(", sampled=");
075                builder.append(String.valueOf(getIterationSampled()));
076                builder.append(", detected=");
077                builder.append(String.valueOf(getIterationDetected()));
078                return builder.toString();
079        }
080
081}
