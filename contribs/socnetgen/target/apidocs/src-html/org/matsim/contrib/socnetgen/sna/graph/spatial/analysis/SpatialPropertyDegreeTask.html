026import java.io.IOException;
027import java.util.HashSet;

029import java.util.Set;

031import org.apache.commons.math.stat.descriptive.DescriptiveStatistics;

040import org.matsim.contrib.socnetgen.sna.graph.spatial.SpatialGraph;
041import org.matsim.contrib.socnetgen.sna.graph.spatial.SpatialVertex;
042
043import com.vividsolutions.jts.geom.Point;
044
045/**
046 * @author illenberger
047 *
048 */
049public class SpatialPropertyDegreeTask extends ModuleAnalyzerTask<Degree> {
050
051        private Discretizer discretizer = new DummyDiscretizer();
052        
053        private SpatialCostFunction costFunction;
054        
055        private Set<Point> opportunities;
056        
057        public SpatialPropertyDegreeTask(SpatialCostFunction costFunction, Set<Point> points) {
058                this.costFunction = costFunction;
059                this.opportunities = points;
060                setModule(Degree.getInstance());
061        }
062        
063        public void setDiscretizer(Discretizer discretizer) {
064                this.discretizer = discretizer;
065        }
066        
067        @Override
068        public void analyze(Graph g, Map<String, DescriptiveStatistics> statsMap) {
069                if(outputDirectoryNotNull()) {
070                        try {
071                                SpatialGraph graph = (SpatialGraph) g;
072                                
073                                if(opportunities == null) {
074                                        opportunities = new HashSet<Point>();
075                                        for(SpatialVertex vertex : graph.getVertices())
076                                                opportunities.add(vertex.getPoint());
077                                }
078                                
079                                /*
080                                 * distance-degree correlation
081                                 */
082                                TObjectDoubleHashMap<SpatialVertex> yVals = Distance.getInstance().vertexMean(graph.getVertices());
083                                TObjectDoubleHashMap kVals =  module.values(graph.getVertices());
084                                TDoubleDoubleHashMap correl = VertexPropertyCorrelation.mean(yVals, kVals, discretizer);
085                                StatsWriter.writeHistogram(correl, "k", "d_mean", getOutputDirectory() + "d_mean_k.mean.txt");
086                                
087                                TDoubleObjectHashMap<DescriptiveStatistics> stat = VertexPropertyCorrelation.statistics(yVals, kVals, discretizer);
088                                StatsWriter.writeBoxplotStats(stat, getOutputDirectory() + "d_mean_k.boxplot.txt");
089                                StatsWriter.writeStatistics(stat, "k", getOutputDirectory() + "d_k.stats.txt");
090                                
091                                yVals = Distance.getInstance().vertexMedian(graph.getVertices());
092                                correl = VertexPropertyCorrelation.mean(yVals, kVals, discretizer);
093                                StatsWriter.writeHistogram(correl, "k", "d_median", getOutputDirectory() + "d_median_k.mean.txt");
094                                
095                                stat = VertexPropertyCorrelation.statistics(yVals, kVals, discretizer);
096                                StatsWriter.writeBoxplotStats(stat, getOutputDirectory() + "d_median_k.boxplot.txt");
097
098                        } catch (IOException e) {
099                                e.printStackTrace();
100                        }
101                }
102        }
103
104}
