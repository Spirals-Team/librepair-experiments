020package org.matsim.contrib.socnetgen.sna.snowball.analysis;
021
022import gnu.trove.iterator.TObjectDoubleIterator;
023import gnu.trove.map.hash.TObjectDoubleHashMap;
024
025import java.util.Collection;
026import java.util.Set;
027
028import org.matsim.contrib.common.stats.DescriptivePiStatistics;
029import org.matsim.contrib.socnetgen.sna.graph.Graph;
030import org.matsim.contrib.socnetgen.sna.graph.Vertex;
031import org.matsim.contrib.socnetgen.sna.graph.analysis.Transitivity;
032import org.matsim.contrib.socnetgen.sna.math.DescriptivePiStatisticsFactory;
033import org.matsim.contrib.socnetgen.sna.snowball.SampledGraph;
034import org.matsim.contrib.socnetgen.sna.snowball.SampledVertex;
035
036
037/**
038 * A class that provides functionality to estimate triangular configurations in a
039 * graph.
040 * 
041 * @author illenberger
042 *
043 */
044public class EstimatedTransitivity extends Transitivity {
045
046        private final PiEstimator piEstimator;
047
048        private DescriptivePiStatisticsFactory facory;
049        
050        private boolean estimEdges;
051        
052        private boolean enableCache;
053        
054        private static int cachedSize;
055        
056        private static TObjectDoubleHashMap<SampledVertex> cachedValues;
057        
058        /**
059         * Creates a new estimated transitivity object with <tt>piEstimator</tt> as
060         * the pi-estimator and <tt>factor</tt> as the factor for creating new
061         * {@link DescriptivePiStatistics} object that do the actual estimation of
062         * properties. If the switch <tt>estimEdges</tt> is set to <tt>true</tt>
063         * missing edges between neighbours are also estimated.
064         * 
065         * @param piEstimator
066         *            the pi-estimator
067         * @param factory
068         *            a factory for creating new instance of
069         *            {@link DescriptivePiStatistics}
070         * @param estimEdges
071         *            <tt>true</tt> if missing edges between neighbours are to be
072         *            estimated.
073         */
074        public EstimatedTransitivity(PiEstimator piEstimator, DescriptivePiStatisticsFactory factory, boolean estimEdges) {
075                this.piEstimator = piEstimator;
076                this.facory = factory;
077                this.estimEdges = estimEdges;
078        }
079
080        /**
081         * Enables caching functionality. If
082         * {@link #localClusteringCoefficients(Collection)} is called multiple times
083         * with a set of vertices of same size the calculation is done only once.
084         * 
085         * @param flag <tt>true</tt> to enable caching.
086         */
087        @Deprecated
088        public void enableCaching(boolean flag) {
089                enableCache = flag;
090        }
091
092        /**
093         * Estimates the local clustering coefficient for all sampled vertices.
094         * 
095         * @param vertices
096         *            a set of sampled vertices
097         * @return a object-double map with all sampled vertices and their estimated
098         *         local clustering coefficients.
099         */
100        @SuppressWarnings("unchecked")
101        @Override
102        public <V extends Vertex> TObjectDoubleHashMap<V> localClusteringCoefficients(Collection<V> vertices) {
103                if(enableCache && vertices.size() == cachedSize) {
104                        return (TObjectDoubleHashMap<V>) cachedValues;
105                }
106                
107                Set<SampledVertex> sampledVertices = SnowballPartitions.<SampledVertex>createSampledPartition((Collection<SampledVertex>)vertices);
108                
109                TObjectDoubleHashMap<SampledVertex> coefficients = new TObjectDoubleHashMap<SampledVertex>();
110                
111                for(SampledVertex vertex : sampledVertices) {
112                        int k = vertex.getEdges().size();
113                        if (k == 0 || k == 1) {
114                                coefficients.put(vertex, 0.0);
115                        } else {
116                                double c = 0;
117                                int n_edge = countAdjacentEdges(vertex);
118                                
119                                if (estimEdges) {
120                                        c = 2 * estimateAdjacentEdges(vertex, n_edge) / (double) (k * (k - 1));
121                                } else {
122                                        c = 2 * n_edge / (double) (k * (k - 1));
123                                }
124                                
125                                coefficients.put(vertex, c);
126                        }
127                }
128                
129                cachedSize = vertices.size();
130                cachedValues = coefficients;
131                
132                return (TObjectDoubleHashMap<V>) coefficients;
133        }
134
135        protected double estimateAdjacentEdges(SampledVertex vertex, int n_edge) {
136                int n = vertex.getNeighbours().size();
137                int n_sampled = 0;
138                for (Vertex neighbor : vertex.getNeighbours()) {
139                        if (((SampledVertex) neighbor).isSampled())
140                                n_sampled++;
141                }
142
143                int n_notsampled = n - n_sampled;
144
145                double N_edge = n_sampled * n_notsampled + 0.5 * n_sampled * (n_sampled - 1);
146                double p_edge = 0;
147        
148                if (N_edge > 0)
149                        p_edge = n_edge / N_edge;
150
151                double n_edge_estim = n_edge + n_notsampled * (n_notsampled - 1) * 0.5 * p_edge;
152                
153                return n_edge_estim;
154        }
155        
156        /**
157         * Returns a descriptive statistics object containing the estimated local
158         * clustering coefficients of vertices sampled in or up to the next to last
159         * iteration each associated a pi-value.
160         * 
161         * @param vertices a set of sampled vertices
162         * @return a descriptive statistics object.
163         */
164        @SuppressWarnings("unchecked")
165        @Override
166        public DescriptivePiStatistics statistics(Set<? extends Vertex> vertices) {
167                DescriptivePiStatistics stats = facory.newInstance();
168                int iteration = SnowballStatistics.getInstance().lastIteration((Set<? extends SampledVertex>) vertices);
169                
170                TObjectDoubleHashMap<? extends Vertex> values = localClusteringCoefficients(vertices);
171                TObjectDoubleIterator<? extends Vertex> it = values.iterator();
172                
173                for (int i = 0; i < values.size(); i++) {
174                        it.advance();
175                        if (((SampledVertex) it.key()).getIterationSampled() <= iteration - 1) {
176                                double p = piEstimator.probability((SampledVertex) it.key(), iteration - 1);
177                                if (p > 0)
178                                        stats.addValue(it.value(), p);
179                        }
180                }
181                return stats;
182        }
183
184        @Override
185        public double globalClusteringCoefficient(Graph graph) {
186                double n_tripples = 0;
187                double n_triangles = 0;
188
189                SampledGraph sampledGraph = (SampledGraph) graph;
190                int iteration = SnowballStatistics.getInstance().lastIteration(sampledGraph.getVertices());
191                
192                for(SampledVertex v : sampledGraph.getVertices()) {
193                        if(v.isSampled() && v.getIterationSampled() < iteration) {
194                                int k = v.getNeighbours().size();
195                                if(k > 1) {
196                                        int n_2 = k*(k-1)/2;
197                                        double n_3 = countAdjacentEdges(v);
198                                        if(estimEdges)
199                                                n_3 = estimateAdjacentEdges(v, (int)n_3);
200                                        
201                                        double p = piEstimator.probability(v, iteration - 1);
202                                        
203                                        n_tripples += n_2 * 1/p;
204                                        n_triangles += n_3 * 1/p; 
205                                }
206                                 
207                        }
208                }
209                 
210                return n_triangles/n_tripples;
211        }
212        
213}
