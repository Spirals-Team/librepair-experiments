020package org.matsim.contrib.socnetgen.sna.snowball.sim;
021
022import org.matsim.contrib.socnetgen.sna.graph.analysis.AnalyzerTask;
023import org.matsim.contrib.socnetgen.sna.snowball.SampledVertexDecorator;
024import org.matsim.contrib.socnetgen.sna.snowball.analysis.PiEstimator;
025
026import java.io.File;
027import java.util.Collection;
028import java.util.Map;
029
030
031/**
032 * An analyzer that analysis the sampled graph in the following intervals:
033 * <ul>
034 * <li>after 100 additionally sampled vertices until 1000 sampled vertices</li>
035 * <li>after 1000 additionally sampled vertices until 10000 sampled vertices</li>
036 * <li>after 10000 additionally sampled vertices.</li>
037 * </ul>
038 * 
039 * @author illenberger
040 * 
041 */
042public class IntervalSampleAnalyzer extends SampleAnalyzer {
043
044        private int prevN = 0;
045        
046        /**
047         * @see {@link SampleAnalyzer#SampleAnalyzer(Map, Collection, String)}
048         */
049        public IntervalSampleAnalyzer(Map<String, AnalyzerTask> tasks, Collection<PiEstimator> estimators,
050                        String output) {
051                super(tasks, estimators, output);
052        }
053
054        /**
055         * Analysis the sampled graph after a given number of vertices has been
056         * sampled.
057         * 
058         * @param sampler
059         *            a snowball sampler
060         * @param vertex
061         *            the last sampled vertex
062         * @return <tt>true</tt>
063         */
064        @Override
065        public boolean afterSampling(Sampler<?, ?, ?> sampler, SampledVertexDecorator<?> vertex) {
066                int n = sampler.getNumSampledVertices();
067                if (n > prevN) {
068                        if (n < 200) {
069                                if(n % 10 == 0)
070                                        dump(sampler);
071                        } else if (n < 1000) {
072                                if (n % 100 == 0)
073                                        dump(sampler);
074                        } else if (n < 10000) {
075                                if (n % 1000 == 0)
076                                        dump(sampler);
077                        } else {
078                                if (n % 5000 == 0)
079                                        dump(sampler);
080                        }
081                }
082                prevN = n;
083                return true;
084        }
085
086        protected void dump(Sampler<?, ?, ?> sampler) {
087                File file = makeDirectories(String.format("%1$s/vertex.%2$s", getRootDirectory(), sampler
088                                .getNumSampledVertices()));
089                analyze(sampler.getSampledGraph(), file.getAbsolutePath());
090        }
091
092        /**
093         * Returns always <tt>true</tt>
094         * 
095         * @return <tt>true</tt>
096         */
097        @Override
098        public boolean beforeSampling(Sampler<?, ?, ?> sampler, SampledVertexDecorator<?> vertex) {
099                return true;
100        }
101
102        /**
103         * Makes a dump with the maximum network sampled.
104         */
105        @Override
106        public void endSampling(Sampler<?, ?, ?> sampler) {
107                dump(sampler);
108        }
109
110}
