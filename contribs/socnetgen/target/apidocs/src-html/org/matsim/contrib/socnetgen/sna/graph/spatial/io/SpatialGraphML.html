020package org.matsim.contrib.socnetgen.sna.graph.spatial.io;
021
022import com.vividsolutions.jts.geom.Coordinate;
023import com.vividsolutions.jts.geom.GeometryFactory;
024import com.vividsolutions.jts.geom.Point;
025import org.apache.log4j.Logger;
026import org.matsim.contrib.common.gis.CRSUtils;
027import org.matsim.contrib.socnetgen.sna.graph.spatial.SpatialVertex;
028import org.matsim.core.utils.collections.Tuple;
029import org.opengis.referencing.crs.CoordinateReferenceSystem;
030import org.xml.sax.Attributes;
031
032import java.util.List;
033
034/**
035 * Utility-class for de- and encoding GraphML data.
036 * 
037 * @author illenberger
038 *
039 */
040public class SpatialGraphML {
041        
042        private static final Logger logger = Logger.getLogger(SpatialGraphML.class);
043
044        public static final String COORD_X_ATTR = "x";
045        
046        public static final String COORD_Y_ATTR = "y";
047        
048        public static final String SRID_ATTR = "srid";
049        
050        private static final GeometryFactory geometryFactory = new GeometryFactory();
051        
052        /**
053         * Creates a point out of the data in <tt>attrs</tt>.
054         * 
055         * @param attrs
056         *            attribute data which is expected to store xy-coordinate
057         *            information.
058         * 
059         * @return a new point.
060         */
061        public static Point newPoint(Attributes attrs) {
062                String xstr = attrs.getValue(COORD_X_ATTR);
063                String ystr = attrs.getValue(COORD_Y_ATTR);
064                
065                if(xstr != null && ystr != null) {
066                        double x = Double.parseDouble(xstr);
067                        double y = Double.parseDouble(ystr);
068                
069                        return geometryFactory.createPoint(new Coordinate(x, y));
070                } else {
071                        return null;
072                }
073        }
074        
075        /**
076         * Retrieves the coordinate reference system from the information stored in
077         * <tt>attrs</tt>.
078         * 
079         * @param attrs
080         *            attribute data which is expected to store the SRID.
081         * 
082         * @return a coordinate reference system.
083         */
084        public static CoordinateReferenceSystem newCRS(Attributes attrs) {
085                String value = attrs.getValue(SRID_ATTR);
086                if(value == null) {
087                        logger.warn("No SRID attribute specified. Assuming CH1903LV03.");
088                        value = "21781";
089                }
090                int code = Integer.parseInt(value);
091                return CRSUtils.getCRS(code);
092        }
093        
094        public static void addPointData(SpatialVertex v, List<Tuple<String, String>> attrs) {
095                if(v.getPoint() != null) {
096                        attrs.add(new Tuple<String, String>(SpatialGraphML.COORD_X_ATTR, String.valueOf(v.getPoint().getX())));
097                        attrs.add(new Tuple<String, String>(SpatialGraphML.COORD_Y_ATTR, String.valueOf(v.getPoint().getY())));
098                }
099        }
100}
