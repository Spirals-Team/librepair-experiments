024package org.matsim.contrib.socnetgen.sna.graph;
025
026import java.util.ArrayList;
027import java.util.List;
028import java.util.RandomAccess;
029
030
031/**
032 * Representation of a vertex without any further attributes.
033 * 
034 * @author illenberger
035 * 
036 */
037public class SparseVertex implements Vertex {
038
039        private ArrayList<SparseEdge> edges;
040
041        private ArrayList<SparseVertex> neighbours;
042
043        /**
044         * Creates a new isolated vertex.
045         */
046        protected SparseVertex() {
047                edges = new ArrayList<SparseEdge>();
048                neighbours = new ArrayList<SparseVertex>();
049        }
050
051        /**
052         * Returns the list of edges connected to this vertex. The list implements
053         * the {@link RandomAccess} interface.
054         * 
055         * @see {@link Vertex#getEdges()}
056         * 
057         * @return the list of edges connected to this vertex.
058         */
059        public List<? extends SparseEdge> getEdges() {
060                return edges;
061        }
062
063        /**
064         * Returns the list of adjacent vertices. The list implements the
065         * {@link RandomAccess} interface.
066         * 
067         * @see {@link Vertex#getNeighbours()}.
068         * 
069         * @return the list of adjacent vertices.
070         */
071        public List<? extends SparseVertex> getNeighbours() {
072                return neighbours;
073        }
074
075        /**
076         * Connects this vertex to an edge. This method does not check if <tt>e</tt>
077         * is already connected to this vertex!
078         * 
079         * @param e
080         *            the edge this vertex is to be connected to.
081         */
082        void addEdge(SparseEdge e) {
083                edges.add(e);
084                neighbours.add(e.getOpposite(this));
085        }
086
087        /**
088         * Disconnects this vertex from an edge.
089         * 
090         * @param e
091         *            the edge this vertex is to be disconnected from.
092         * @return <tt>true</tt> if this vertex has been disconnected from
093         *         <tt>e</tt>, <tt>false</tt> if this vertex is not connected
094         *         to edge <tt>e</tt>.
095         */
096        boolean removeEdge(SparseEdge e) {
097                boolean removedEdge = edges.remove(e);
098                boolean removedNeighbour = neighbours.remove(e.getOpposite(this));
099                
100                if(removedEdge && removedNeighbour)
101                        return true;
102                else if(!removedEdge && !removedNeighbour)
103                        return false;
104                else
105                        /*
106                         * Actually, this should never happen. Think about, if this is really necessary.
107                         */
108                        throw new RuntimeException("Graph connectivity appears to be inconsitent!");
109        }
110        
111        /**
112         * Calls {@link ArrayList#trimToSize()} on all internal lists.
113         */
114        void optimize() {
115                edges.trimToSize();
116                neighbours.trimToSize();
117        }
118}
