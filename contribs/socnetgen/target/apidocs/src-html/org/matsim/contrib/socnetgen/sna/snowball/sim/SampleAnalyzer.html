020package org.matsim.contrib.socnetgen.sna.snowball.sim;
021
022import org.apache.log4j.Level;
023import org.apache.log4j.Logger;
024import org.matsim.contrib.socnetgen.sna.graph.analysis.AnalyzerTask;
025import org.matsim.contrib.socnetgen.sna.graph.analysis.GraphAnalyzer;
026import org.matsim.contrib.socnetgen.sna.snowball.SampledGraphProjection;
027import org.matsim.contrib.socnetgen.sna.snowball.analysis.PiEstimator;
028
029import java.io.File;
030import java.io.IOException;
031import java.util.Collection;
032import java.util.Map;
033import java.util.Map.Entry;
034
035/**
036 * Abstract base class for sample analyzers. Organizes the analyzers output into
037 * directories and updates each estimator before analysis.
038 * 
039 * @author illenberger
040 * 
041 */
042public abstract class SampleAnalyzer implements SamplerListener {
043
044        private Map<String, AnalyzerTask> tasks;
045
046        private Collection<PiEstimator> estimators;
047
048        private String rootDirectory;
049
050        /**
051         * Creates a new sample analyzer.
052         * 
053         * @param tasks
054         *            A map of analyzer tasks. The output of an analyzer task will
055         *            be written into a directory named with the entry key.
056         * @param estimators
057         *            A collection of estimators that need to be updated before
058         *            analysis.
059         * @param rootDirectory
060         *            The root directory for the analysis output.
061         */
062        public SampleAnalyzer(Map<String, AnalyzerTask> tasks, Collection<PiEstimator> estimators,
063                        String rootDirectory) {
064                this.tasks = tasks;
065                this.rootDirectory = rootDirectory;
066                this.estimators = estimators;
067        }
068
069        /**
070         * Returns the root directory for analysis output.
071         * 
072         * @return the root directory for analysis output.
073         */
074        protected String getRootDirectory() {
075                return rootDirectory;
076        }
077
078        /**
079         * Analysis the sampled graph will all analyzer tasks.
080         * 
081         * @param graph
082         *            A sampled graph.
083         * @param output
084         *            The output directory for the analysis output.
085         */
086        protected void analyze(SampledGraphProjection<?, ?, ?> graph, String output) {
087                try {
088                        /*
089                         * Prevent the analyzer tasks from logging.
090                         */
091                        Level level = Logger.getRootLogger().getLevel(); 
092                        Logger.getRootLogger().setLevel(Level.WARN);
093                        /*
094                         * Update all estimators.
095                         */
096                        for (PiEstimator estimator : estimators) {
097                                estimator.update(graph);
098                        }
099                        /*
100                         * Analyze the graph with each analyzer taks.
101                         */
102                        for (Entry<String, AnalyzerTask> task : tasks.entrySet()) {
103                                /*
104                                 * Create output directories.
105                                 */
106                                String taskOutput = String.format("%1$s/%2$s/", output, task.getKey());
107                                File file = makeDirectories(taskOutput);
108                                task.getValue().setOutputDirectoy(file.getAbsolutePath());
109                                /*
110                                 * Analyze.
111                                 */
112                                GraphAnalyzer.analyze(graph, task.getValue(), file.getAbsolutePath());
113//                              GraphAnalyzer.writeStats(stats, file.getAbsolutePath() + "/stats.txt");
114                        }
115                        
116                        Logger.getRootLogger().setLevel(level);
117                        
118                } catch (IOException e) {
119                        e.printStackTrace();
120                }
121        }
122
123        protected File makeDirectories(String path) {
124                File file = new File(path);
125                file.mkdirs();
126                return file;
127        }
128}
