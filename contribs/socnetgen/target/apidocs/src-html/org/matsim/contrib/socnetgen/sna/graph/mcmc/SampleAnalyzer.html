020package org.matsim.contrib.socnetgen.sna.graph.mcmc;
021
022import org.apache.log4j.Logger;
023import org.matsim.contrib.socnetgen.sna.graph.Edge;
024import org.matsim.contrib.socnetgen.sna.graph.Graph;
025import org.matsim.contrib.socnetgen.sna.graph.GraphBuilder;
026import org.matsim.contrib.socnetgen.sna.graph.Vertex;
027import org.matsim.contrib.socnetgen.sna.graph.analysis.AnalyzerTask;
028import org.matsim.contrib.socnetgen.sna.graph.analysis.GraphAnalyzer;
029import org.matsim.contrib.socnetgen.sna.graph.io.GraphMLWriter;
030import org.matsim.contrib.socnetgen.sna.graph.matrix.AdjacencyMatrix;
031
032import java.io.File;
033import java.io.IOException;
034import java.util.Locale;
035
036/**
037 * @author illenberger
038 *
039 */
040public class SampleAnalyzer<G extends Graph, E extends Edge, V extends Vertex> implements SamplerListener<V> {
041
042        private static final Logger logger = Logger.getLogger(SampleAnalyzer.class);
043        
044        private final G graph;
045        
046        private final GraphBuilder<G, V, E> builder;
047        
048        private GraphMLWriter writer;
049        
050        private AnalyzerTask analyzerTask;
051        
052        private final String rootDirectory;
053        
054        private long infoInteraval;
055        
056        private long analysisInterval;
057        
058        private long maxIteration;
059        
060        public SampleAnalyzer(G graph, GraphBuilder<G, V, E> builder, String rootDirectory) {
061                this.graph = graph;
062                this.builder = builder;
063                this.rootDirectory = rootDirectory;
064        }
065        
066        public void setWriter(GraphMLWriter writer) {
067                this.writer = writer;
068        }
069
070        public void setAnalyzerTask(AnalyzerTask analyzerTask) {
071                this.analyzerTask = analyzerTask;
072        }
073
074        public void setInfoInteraval(long infoInteraval) {
075                this.infoInteraval = infoInteraval;
076        }
077
078        public void setAnalysisInterval(long analysisInterval) {
079                this.analysisInterval = analysisInterval;
080        }
081
082        public void setMaxIteration(long maxIteration) {
083                this.maxIteration = maxIteration;
084        }
085
086        @Override
087        public boolean beforeSampling(AdjacencyMatrix<V> y, long iteration) {
088                if(iteration % analysisInterval == 0) {
089                        analyze(y, iteration);
090                }
091                
092                if(iteration % infoInteraval == 0) {
093                        double m = y.countEdges();
094                        double k_mean = AdjacencyMatrixStatistics.getMeanDegree(y);
095                        double c_local = AdjacencyMatrixStatistics.getLocalClusteringCoefficient(y);
096                        double c_global = AdjacencyMatrixStatistics.getGlobalClusteringCoefficient(y);
097                
098                        logger.info(String.format(Locale.US, "m=%1$s, <k>=%2$.4f, <c_local>=%3$.4f, <c_global>=%4$.4f.", m, k_mean, c_local, c_global));
099                }
100                
101                // yyyy the following feels a bit weird: I think it gives the responsibility to end the sampling to a class that
102                // otherwise is purely observing.  kai, nov'16
103                if(iteration > maxIteration)
104                        return false;
105                else
106                        return true;
107        }
108
109        private void analyze(AdjacencyMatrix<V> y, long iteration) {
110                y.synchronizeEdges(graph, builder);
111        
112                File file = new File(String.format("%1$s/%2$d/", rootDirectory, iteration));
113                file.mkdirs();
114                
115                try {
116                        if(writer != null)
117                                writer.write(graph, String.format("%1$s/graph.graphml", file.getAbsolutePath()));
118                        
119                        if(analyzerTask != null)
120                                GraphAnalyzer.analyze(graph, analyzerTask, file.getAbsolutePath());
121                } catch (IOException e) {
122                        e.printStackTrace();
123                }
124        }
125}
