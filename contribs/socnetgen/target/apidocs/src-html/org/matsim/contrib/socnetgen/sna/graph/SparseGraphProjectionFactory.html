020package org.matsim.contrib.socnetgen.sna.graph;
021
022/**
023 * Implementation of GraphProjectionFactory to create instances of GraphProjection,
024 * VertexDecorator and EdgeDecorator.
025 * 
026 * @author illenberger
027 *
028 */
029public class SparseGraphProjectionFactory<G extends Graph, V extends Vertex, E extends Edge> implements GraphProjectionFactory<G, V, E, GraphProjection<G,V,E>, VertexDecorator<V>, EdgeDecorator<E>>{
030
031        /**
032         * Creates and returns an edge decorator that decorates <tt>delegate</tt>.
033         * 
034         * @param delegate the original edge.
035         * 
036         * @return an edge decorator.
037         */
038        @Override
039        public EdgeDecorator<E> createEdge(E delegate) {
040                return new EdgeDecorator<E>(delegate);
041        }
042
043        /**
044         * Creates and returns an empty graph projection on <tt>delegate</tt>.
045         * 
046         * @param delegate the original graph.
047         * 
048         * @return an empty graph projection.
049         */
050        @Override
051        public GraphProjection<G, V, E> createGraph(G delegate) {
052                return new GraphProjection<G, V, E>(delegate);
053        }
054
055        /**
056         * Creates and returns a vertex decorator that decorates <tt>delegate</tt>.
057         * 
058         * @param delegate the original vertex.
059         * 
060         * @return a new vertex decorator.
061         */
062        @Override
063        public VertexDecorator<V> createVertex(V delegate) {
064                return new VertexDecorator<V>(delegate);
065        }
066
067        @Override
068        public GraphProjection<G, V, E> copyGraph(GraphProjection<G, V, E> graph) {
069                throw new UnsupportedOperationException("Seems like someone is using this method...");
070        }
071
072        @Override
073        public VertexDecorator<V> copyVertex(VertexDecorator<V> vertex) {
074                throw new UnsupportedOperationException("Seems like someone is using this method...");
075        }
076
077        @Override
078        public EdgeDecorator<E> copyEdge(EdgeDecorator<E> edge) {
079                throw new UnsupportedOperationException("Seems like someone is using this method...");
080        }
081
082}
