024package org.matsim.contrib.socnetgen.sna.graph.mcmc;
025
026import org.matsim.contrib.socnetgen.sna.graph.Vertex;
027import org.matsim.contrib.socnetgen.sna.graph.matrix.AdjacencyMatrix;
028
029
030/**
031 * @author illenberger
032 *
033 */
034public class GibbsEdgeFlip extends AbstractGibbsSampler {
035
036        /**
037         * 
038         */
039        public GibbsEdgeFlip() {
040                super();
041        }
042
043        /**
044         * @param seed
045         */
046        public GibbsEdgeFlip(long seed) {
047                super(seed);
048        }
049
050        protected int[][] edges;
051        
052        protected EdgeSwitchCondition switchCondition;
053        
054        public <V extends Vertex> void sample(AdjacencyMatrix<V> y, EnsembleProbability d, SamplerListener<V> handler, EdgeSwitchCondition condition) {
055                switchCondition = condition;
056                sample(y, d, handler);
057        }
058        
059        @Override
060        public <V extends Vertex> void sample(AdjacencyMatrix<V> y, EnsembleProbability d, SamplerListener<V> handler) {
061                if(switchCondition == null)
062                        switchCondition = new GibbsEdgeSwitch.DefaultSwitchCondition();
063                
064                int N = y.getVertexCount();
065                int M = y.countEdges();
066                edges = new int[M][2];
067                int idx_ij = 0;
068                for(int i = 0; i < N; i++) {
069                        for(int j = i+1; j < N; j++) {
070                                if(y.getEdge(i, j)) {
071                                        edges[idx_ij][0] = i;
072                                        edges[idx_ij][1] = j;
073                                        idx_ij++;
074                                }
075                        }
076                }
077                super.sample(y, d, handler);
078        }
079        
080        @Override
081        public <V extends Vertex> boolean step(AdjacencyMatrix<V> y, EnsembleProbability d) {
082                boolean accept = false;
083                int idx_ij = random.nextInt(edges.length);
084                int i = edges[idx_ij][0];
085                int j = edges[idx_ij][1];
086                
087                int idx_uv = random.nextInt(edges.length);
088                while(idx_uv == idx_ij) {
089                        idx_uv = random.nextInt(edges.length);
090                }
091                
092                int u = edges[idx_uv][0];
093                int v = edges[idx_uv][1];
094                
095                if(i != u && j != v && j != u && i != v && !y.getEdge(i, u) && !y.getEdge(j, v)) {
096                        
097                        if(switchCondition.allowSwitch(y, i, j, i, u) && switchCondition.allowSwitch(y, u, v, j, v)) {
098                        /*
099                         * ************************************************
100                         * TODO: CHECK THIS!
101                         */
102                        double p_change = d.ratio(y, i, u, false)
103                                        * d.ratio(y, j, v, false)
104                                        * 1/d.ratio(y, i, j, true)
105                                        * 1/d.ratio(y, u, v, true); 
106                        double p = 1 / (1 + p_change);
107                        /*
108                         * ************************************************
109                         */
110                        if(random.nextDouble() <= p) {
111                                y.removeEdge(i, j);
112                                y.removeEdge(u, v);
113                                y.addEdge(i, u);
114                                y.addEdge(j, v);
115                                
116                                edges[idx_ij][0] = i;
117                                edges[idx_ij][1] = u;
118                                edges[idx_uv][0] = j;
119                                edges[idx_uv][1] = v;
120                                
121                                accept = true;
122                        }
123                        }
124                }
125                
126                return accept;
127        }
128        
129        public static class DefaultSwitchCondition implements EdgeSwitchCondition {
130
131                @Override
132                public boolean allowSwitch(AdjacencyMatrix<?> y, int i, int j, int u, int v) {
133                        return true;
134                }
135                
136        }
137}
