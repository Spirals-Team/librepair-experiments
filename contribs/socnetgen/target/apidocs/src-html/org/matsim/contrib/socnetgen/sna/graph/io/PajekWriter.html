024package org.matsim.contrib.socnetgen.sna.graph.io;
025
026import gnu.trove.map.hash.TObjectIntHashMap;
027
028import java.io.BufferedWriter;
029import java.io.FileWriter;
030import java.io.IOException;
031import java.util.ArrayList;
032import java.util.List;
033
034import org.matsim.contrib.socnetgen.sna.graph.Edge;
035import org.matsim.contrib.socnetgen.sna.graph.Graph;
036import org.matsim.contrib.socnetgen.sna.graph.Vertex;
037import org.matsim.core.utils.collections.Tuple;
038
039
040/**
041 * @author illenberger
042 *
043 */
044public class PajekWriter<G extends Graph, V extends Vertex, E extends Edge> {
045
046        private static final String NEW_LINE = "\r\n";
047        
048        private static final String WHITESPACE = " ";
049        
050        private static final String QUOTE = "\"";
051        
052        private static final String ZERO = "0";
053        
054        private TObjectIntHashMap<V> pajekVertexIds;
055        
056        public void write(G g, String file) throws IOException {
057                write(g, new EmptyAttributes(), file);
058        }
059        
060        @SuppressWarnings("unchecked")
061        public void write(G g, PajekAttributes<V, E> attrs, String file) throws IOException {
062                BufferedWriter writer = new BufferedWriter(new FileWriter(file));
063                
064                writer.write("*Vertices ");
065                writer.write(String.valueOf(g.getVertices().size()));
066                writer.write(NEW_LINE);
067                
068                pajekVertexIds = new TObjectIntHashMap<V>();
069                int counter = 0;
070                for(Vertex v : g.getVertices()) {
071                        counter++;
072                        pajekVertexIds.put((V) v, counter);
073                        /*
074                         * Pajek id
075                         */
076                        writer.write(String.valueOf(counter));
077                        writer.write(WHITESPACE);
078                        /*
079                         * Label
080                         */
081                        writer.write(QUOTE);
082                        writer.write(getVertexLabel((V) v));
083                        writer.write(QUOTE);
084                        writer.write(WHITESPACE);
085                        /*
086                         * Coordinates
087                         */
088                        writer.write(getVertexX((V)v));
089                        writer.write(WHITESPACE);
090                        writer.write(getVertexY((V)v));
091                        writer.write(WHITESPACE);
092                        /*
093                         * Vertex shape
094                         */
095                        writer.write(getVertexShape((V)v));
096                        /*
097                         * Vertex attributes
098                         */
099                        for(String att : attrs.getVertexAttributes()) {
100                                writer.write(WHITESPACE);
101                                writer.write(att);
102                                writer.write(WHITESPACE);
103                                writer.write(attrs.getVertexValue((V)v, att));
104                        }
105                        
106                        writer.write(NEW_LINE);
107                }
108                
109                writer.write("*Edges");
110                writer.write(NEW_LINE);
111                
112                for(Edge e : g.getEdges()) {
113                        Tuple<V, V> t = (Tuple<V, V>) e.getVertices();                          
114                        
115                        writer.write(String.valueOf(pajekVertexIds.get(t.getFirst())));
116                        writer.write(WHITESPACE);
117                        writer.write(String.valueOf(pajekVertexIds.get(t.getSecond())));
118                        /*
119                         * Edge attributes
120                         */
121                        if (attrs.getEdgeAttributes() != null) {
122                                for (String att : attrs.getEdgeAttributes()) {
123                                        writer.write(WHITESPACE);
124                                        writer.write(att);
125                                        writer.write(WHITESPACE);
126                                        writer.write(attrs.getEdgeValue((E) e, att));
127                                }
128                        }
129                        writer.write(NEW_LINE);
130                }
131                
132                writer.close();
133        }
134        
135        protected String getVertexLabel(V v) {
136                return String.valueOf(pajekVertexIds.get(v));
137        }
138        
139        protected String getVertexX(V v) {
140                return ZERO;
141        }
142        
143        protected String getVertexY(V v) {
144                return ZERO;
145        }
146        
147        protected String getVertexShape(V v) {
148                return ZERO;
149        }
150        
151        private class EmptyAttributes implements PajekAttributes<V, E> {
152
153                public List<String> getEdgeAttributes() {
154                        return new ArrayList<String>();
155                }
156
157                public String getEdgeValue(E e, String attribute) {
158                        return null;
159                }
160
161                public List<String> getVertexAttributes() {
162                        return new ArrayList<String>();
163                }
164
165                public String getVertexValue(V v, String attribute) {
166                        return null;
167                }
168                
169        }
170}
