099        public void analyze(Graph graph, Map<String, DescriptiveStatistics> statsMap) {
100                String subKey = key + "_local";
101                
102                DescriptiveStatistics stats = module.statistics(graph.getVertices());
103                printStats(stats, subKey);
104                statsMap.put(subKey, stats);
105                
106                if(outputDirectoryNotNull()) {
107                        try {
108                                writeHistograms(stats, new LinearDiscretizer(0.05), subKey, false);
109                                writeHistograms(stats, subKey, 100, 20);
110                        } catch (IOException e) {
111                                e.printStackTrace();
112                        }
113                }
114                
115                subKey = key + "_global";
116                stats = new DescriptiveStatistics();
117                stats.addValue(module.globalClusteringCoefficient(graph));
118                printStats(stats, subKey);
119                statsMap.put(subKey, stats);
120        }
