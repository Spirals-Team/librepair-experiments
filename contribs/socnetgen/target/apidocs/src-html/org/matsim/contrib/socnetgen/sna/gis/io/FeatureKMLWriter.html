059public class FeatureKMLWriter {
060
061        private Colorizable colorizable = new DefaultColorizer();
062
063        private boolean drawContours = true;
064
065        private final ObjectFactory objectFactory = new ObjectFactory();
066        
067        private int alpha = 200;
068        
069        private MathTransform transform;
070
071        public Colorizable getColorizable() {
072                return colorizable;
073        }
074
075        public void setColorizable(Colorizable colorizable) {
076                this.colorizable = colorizable;
077        }
078
079        public void write(Set<Geometry> features, String filename) {
080                CoordinateReferenceSystem sourceCRS = CRSUtils.getCRS(features.iterator().next().getSRID());
081                try {
082                        transform = CRS.findMathTransform(sourceCRS, DefaultGeographicCRS.WGS84);
083                } catch (FactoryException e) {
084                        // TODO Auto-generated catch block
085                        e.printStackTrace();
086                }
087                
088                KMZWriter kmzWriter = new KMZWriter(filename);
089
090                DocumentType documentType = objectFactory.createDocumentType();
091                FolderType polyFolderType = objectFactory.createFolderType();
092                polyFolderType.setName("Zones");
093
094                FolderType labelFolderType = objectFactory.createFolderType();
095                labelFolderType.setName("Labels");
096                /*
097                 * Create a polygon style type for each color.
098                 */
099                Set<Color> colors = getColors(features);
100                Map<Color, StyleType> styleTypes = new HashMap<Color, StyleType>();
101                for(Color color : colors) {
102                        PolyStyleType polyStyleType = objectFactory.createPolyStyleType();
103                        polyStyleType.setColor(new byte[]{(byte) alpha, (byte)color.getBlue(), (byte)color.getGreen(), (byte)color.getRed()});
104                        polyStyleType.setOutline(drawContours);
105                        
106                        StyleType styleType = objectFactory.createStyleType();
107                        styleType.setPolyStyle(polyStyleType);
108                        styleType.setId(String.valueOf(color.hashCode()));
109                        documentType.getAbstractStyleSelectorGroup().add(objectFactory.createStyle(styleType));
110                        
111                        styleTypes.put(color, styleType);
112                }
113                /*
114                 * Create default style type.
115                 */
116                PolyStyleType polyStyleType = objectFactory.createPolyStyleType();
117                polyStyleType.setFill(false);
118                polyStyleType.setOutline(true);
119                
120                StyleType contourStyle = objectFactory.createStyleType();
121                contourStyle.setPolyStyle(polyStyleType);
122                contourStyle.setId("contourStyle");
123                documentType.getAbstractStyleSelectorGroup().add(objectFactory.createStyle(contourStyle));
124//              /*
125//               * Create style type for the lable. 
126//               */
127//              IconStyleType iconStyleType = objectFactory.createIconStyleType();
128//              iconStyleType.setScale(0.0);
129                StyleType lableStyleType = objectFactory.createStyleType();
130//              lableStyleType.setIconStyle(iconStyleType);
131                lableStyleType.setId("label");
132                documentType.getAbstractStyleSelectorGroup().add(objectFactory.createStyle(lableStyleType));
133                /*
134                 * Draw polygons.
135                 */
136                for(Geometry feature : features) {
137                        LinearRingType linearRingType = objectFactory.createLinearRingType();
138                        
139                        Coordinate[] coordinates = feature.getCoordinates();
140                        for(Coordinate coord : coordinates) {
141                                linearRingType.getCoordinates().add(makeCoordinateString(coord));
142                        }
143                        linearRingType.getCoordinates().add(makeCoordinateString(coordinates[0]));
144                        
145                        BoundaryType boundaryType = objectFactory.createBoundaryType();
146                        boundaryType.setLinearRing(linearRingType);
147                        
148                        PolygonType polygonType = objectFactory.createPolygonType();
149                        polygonType.setOuterBoundaryIs(boundaryType);
150                        
151                        PlacemarkType polyPlacemarkType = objectFactory.createPlacemarkType();
152                        polyPlacemarkType.setAbstractGeometryGroup(objectFactory.createPolygon(polygonType));
153                        
154                        Color color = colorizable.getColor(feature);
155                        if(color != null) {             
156                                polyPlacemarkType.setStyleUrl(styleTypes.get(color).getId());
157                        } else {
158                                polyPlacemarkType.setStyleUrl(contourStyle.getId());
159                        }
160                        polyFolderType.getAbstractFeatureGroup().add(objectFactory.createPlacemark(polyPlacemarkType));
161                }
162                
163                documentType.getAbstractFeatureGroup().add(objectFactory.createFolder(polyFolderType));
164                documentType.getAbstractFeatureGroup().add(objectFactory.createFolder(labelFolderType));
165                
166                KmlType kmlType = objectFactory.createKmlType();
167                kmlType.setAbstractFeatureGroup(objectFactory.createDocument(documentType));
168                
169                kmzWriter.writeMainKml(kmlType);
170                kmzWriter.close();
171        }
172
173        private Set<Color> getColors(Set<Geometry> features) {
174                Set<Color> colors = new HashSet<Color>();
175                for (Geometry feature : features) {
176                        Color color = colorizable.getColor(feature);
177                        if (color != null)
178                                colors.add(color);
179                }
180                return colors;
181
182        }
183        
184        private String makeCoordinateString(Coordinate coordinate) {
185                if(transform != null) {
186                        double[] points = new double[]{coordinate.x, coordinate.y};
187                        try {
188                                transform.transform(points, 0, points, 0, 1);
189                        } catch (TransformException e) {
190                                e.printStackTrace();
191                                return null;
192                        }
193                        
194                        StringBuffer buffer = new StringBuffer(50);
195                        buffer.append(Double.toString(points[0]));
196                        buffer.append(",");
197                        buffer.append(Double.toString(points[1]));
198                        return buffer.toString();
199                } else {
200                        StringBuffer buffer = new StringBuffer(50);
201                        buffer.append(Double.toString(coordinate.x));
202                        buffer.append(",");
203                        buffer.append(Double.toString(coordinate.y));
204                        return coordinate.toString();
205                }
206        }

209
210                @Override
211                public Color getColor(Object object) {
212                        return null;
213                }
214                
215        }
