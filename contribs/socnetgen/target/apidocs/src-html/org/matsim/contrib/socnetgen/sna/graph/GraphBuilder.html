020package org.matsim.contrib.socnetgen.sna.graph;
021
022
023/**
024 * A GraphBuilder allows to create and modify graphs. It ensures that the
025 * connectivity of vertices and edges conforms to the definition of the
026 * underlying graph implementation.
027 * 
028 * @author illenberger
029 * 
030 */
031public interface GraphBuilder <G extends Graph, V extends Vertex, E extends Edge> {
032
033        /**
034         * Creates and returns an empty graph.
035         * 
036         * @return an empty graph.
037         */
038        public G createGraph();
039
040        /**
041         * Creates and adds a new vertex to a graph.
042         * 
043         * @param graph
044         *            the graph a new vertex should be inserted in.
045         * @return the newly inserted vertex, or <tt>null</tt> is the insertion of a
046         *         vertex would violate the definition of the underlying graph.
047         */
048        public V addVertex(G graph);
049
050        /**
051         * Creates and inserts a new edge into the graph and assures the correct
052         * connectivity between vertices and edges.
053         * 
054         * @param graph
055         *            the graph a new edge should be inserted in.
056         * @param vertex1
057         *            ,vertex2 the vertices between which the edge should be
058         *            inserted.
059         * @return the newly inserted edge, or <tt>null</tt> if the insertion of an
060         *         edge at the specified position would violate the definition of
061         *         the underlying graph.
062         */
063        public E addEdge(G graph, V vertex1, V vertex2);
064        
065        /**
066         * Removes a vertex from a graph.
067         * 
068         * @param graph
069         *            the graph the vertex should be removed from.
070         * @param vertex
071         *            the vertex to be removed.
072         * @return <tt>true</tt> if the vertex has been removed, or <tt>false</tt>
073         *         if the vertex is not part of graph <tt>graph</tt> or the vertex
074         *         is not isolated.
075         */
076        public boolean removeVertex(G graph, V vertex);
077
078        /**
079         * Removes an edge from a graph and assures the correct release of
080         * connections between vertices and edges.
081         * 
082         * @param graph
083         *            the graph the edge should be removed from.
084         * @param edge
085         *            the edge to be removed.
086         * @return <tt>true</tt> if the edge has been removed, or <tt>false</tt> if
087         *         the edge is not part of graph <tt>graph</tt> or if the removal
088         *         would violate the definition of the underlying graph.
089         */
090        public boolean removeEdge(G graph, E edge);
091        
092        public G copyGraph(G graph);
093        
094}
