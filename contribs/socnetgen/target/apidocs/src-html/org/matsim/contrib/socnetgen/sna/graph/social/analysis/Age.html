020package org.matsim.contrib.socnetgen.sna.graph.social.analysis;
021
022import gnu.trove.list.array.TDoubleArrayList;
023import gnu.trove.map.hash.TDoubleDoubleHashMap;
024import gnu.trove.map.hash.TDoubleObjectHashMap;
025import gnu.trove.map.hash.TObjectDoubleHashMap;
026
027import java.util.Set;
028
029import org.apache.commons.math.stat.correlation.PearsonsCorrelation;
030import org.apache.commons.math.stat.descriptive.DescriptiveStatistics;
031import org.matsim.contrib.common.stats.Correlations;
032import org.matsim.contrib.common.stats.LinearDiscretizer;
033import org.matsim.contrib.socnetgen.sna.graph.Vertex;
034import org.matsim.contrib.socnetgen.sna.graph.analysis.AbstractVertexProperty;
035import org.matsim.contrib.socnetgen.sna.graph.social.SocialEdge;
036import org.matsim.contrib.socnetgen.sna.graph.social.SocialVertex;
037
038/**
039 * @author illenberger
040 *
041 */
042public class Age extends AbstractVertexProperty {
043
044        private static Age instance;
045        
046        public static Age getInstance() {
047                if(instance == null)
048                        instance = new Age();
049                return instance;
050        }
051        
052        public TObjectDoubleHashMap<Vertex> values(Set<? extends Vertex> vertices) {
053                TObjectDoubleHashMap<Vertex> values = new TObjectDoubleHashMap<Vertex>();
054                for(Vertex vertex : vertices) {
055                        int age = ((SocialVertex) vertex).getPerson().getAge();
056                        if(age > -1)
057                                values.put(vertex, age);
058                }
059                
060                return values;
061        }
062        
063        public TDoubleDoubleHashMap correlation(Set<? extends SocialVertex> vertices) {
064                TDoubleArrayList values1 = new TDoubleArrayList(vertices.size() * 15);
065                TDoubleArrayList values2 = new TDoubleArrayList(vertices.size() * 15);
066                
067                for(SocialVertex vertex : vertices) {
068                        int age1 = vertex.getPerson().getAge();
069                        if(age1 > -1) {
070                                for(SocialVertex neighbor : vertex.getNeighbours()) {
071                                        int age2 = neighbor.getPerson().getAge();
072                                        if(age2 > -1) {
073                                                values1.add(age1);
074                                                values2.add(age2);
075                                        }
076                                }
077                        }
078                }
079                
080                return Correlations.mean(values1.toArray(), values2.toArray());
081        }
082        
083        public double correlationCoefficient(Set<? extends SocialEdge> edges) {
084                TDoubleArrayList values1 = new TDoubleArrayList(edges.size());
085                TDoubleArrayList values2 = new TDoubleArrayList(edges.size());
086                
087                for(SocialEdge edge : edges) {
088                        double a1 = edge.getVertices().getFirst().getPerson().getAge();
089                        double a2 = edge.getVertices().getSecond().getPerson().getAge();
090                        if(a1 > 0 && a2 > 0) {
091                                values1.add(a1);
092                                values2.add(a2);
093                                
094                                values1.add(a2);
095                                values2.add(a1);
096                        }
097                }
098
099                if(values1.isEmpty())
100                        return Double.NaN;
101                else
102                        return new PearsonsCorrelation().correlation(values1.toArray(), values2.toArray());
103        }
104        
105        public TDoubleObjectHashMap<DescriptiveStatistics> boxplot(Set<? extends SocialVertex> vertices) {
106                TDoubleArrayList values1 = new TDoubleArrayList(vertices.size() * 15);
107                TDoubleArrayList values2 = new TDoubleArrayList(vertices.size() * 15);
108                
109                for(SocialVertex vertex : vertices) {
110                        int age1 = vertex.getPerson().getAge();
111                        if(age1 > -1) {
112                                for(SocialVertex neighbor : vertex.getNeighbours()) {
113                                        int age2 = neighbor.getPerson().getAge();
114                                        if(age2 > -1) {
115                                                values1.add(age1);
116                                                values2.add(age2);
117                                        }
118                                }
119                        }
120                }
121                
122                return Correlations.statistics(values1.toArray(), values2.toArray(), new LinearDiscretizer(5.0));
123        }
124}
