090                 * load population
091                 */
092                Scenario scenario = ScenarioUtils.createScenario(ConfigUtils.createConfig());

097                popReader.readFile(config.plans().getInputFile()) ;
098                Population population = scenario.getPopulation();
099                /*
100                 * read parameters
101                 */
102

104                long randomSeed = config.global().getRandomSeed() ;
105                
106//              long iterations = (long)Double.parseDouble(config.getParam(MODULE_NAME, "iterations"));

108

110                long logInterval = sConfig.getLogInterval() ;
111                
112//              long sampleInterval = (long)Double.parseDouble(config.getParam(MODULE_NAME, "sampleinterval"));
113                long sampleInterval = sConfig.getSampleInterval() ; 
114                
115//              String output = config.getParam(MODULE_NAME, "output");
116                String output = sConfig.getOutput() ;
117                
118//              double theta_distance = Double.parseDouble(config.getParam(MODULE_NAME, "theta_distance"));
119                double theta_distance = sConfig.getThetaDistance() ;
120
121//              double theta_age = Double.parseDouble(config.getParam(MODULE_NAME, "theta_age"));
122                double theta_age = sConfig.getThetaAge() ;
123                
124//              double theta_gender = Double.parseDouble(config.getParam(MODULE_NAME, "theta_gender"));
125                double theta_gender = sConfig.getThetaGender() ;
126
127//              double theta_triangles = Double.parseDouble(config.getParam(MODULE_NAME, "theta_triangles"));
128//              double theta_triangles = sConfig.getThetaTriangles() ;
129                // not needed here, but exists in config. kai, nov'16
130                
131//              boolean conservePk = Boolean.parseBoolean(config.getParam(MODULE_NAME, "conservePk"));
132                boolean conservePk = sConfig.isConservingPk() ;
133                
134                /*
135                 * initialize graph
136                 */
137                logger.info("Initializing graph...");
138                Set<Person> persons = new HashSet<>(population.getPersons().values());
139                SocialSparseGraphFactory factory = new SocialSparseVertexPool(persons, CRSUtils.getCRS(21781));
140                SocialSparseGraphBuilder builder = new SocialSparseGraphBuilder(factory);
141                SocialSparseGraph graph = createGraph(config.getParam(MODULE_NAME, "graphtype"), persons, randomSeed, builder, config);
142                /*
143                 * initialize ergm
144                 */
145                logger.info("Initializing ergm...");
146                AdjacencyMatrix<SocialSparseVertex> y = new AdjacencyMatrix<>(graph, false);
147                
148                Ergm ergm = new Ergm();
149//              ergm.addComponent(new EdgePowerLawDistance(y, theta_distance, 10 * persons.size()));
150//              ErgmDensity eden = new ErgmDensity();
151//              eden.setTheta(6.9);
152//              ergm.addComponent(eden);
153                ergm.addComponent(new ErgmLnDistance(y, theta_distance));
154                ergm.addComponent(new ErgmAge(theta_age));
155                ergm.addComponent(new ErgmGender(theta_gender));
156//              ergm.addComponent(new ErgmTriangles(theta_triangles));
157                /*
158                 * initialize sampler
159                 */
160//              GibbsEdgeSwitch sampler = new GibbsEdgeSwitch(randomSeed);
161                GibbsEdgeFlip sampler = new GibbsEdgeFlip(randomSeed);
162//              GibbsEdgeInsert sampler = new GibbsEdgeInsert(randomSeed);
163                
164                SampleAnalyzer<SocialSparseGraph, SocialSparseEdge, SocialSparseVertex> analyzer = new SampleAnalyzer<>(graph, builder, output);
165                analyzer.setAnalysisInterval(sampleInterval);
166                analyzer.setInfoInteraval(logInterval);
167                analyzer.setMaxIteration(iterations);
168                analyzer.setWriter(new SocialGraphMLWriter());
169                /*
170                 * initialize analyzer task
171                 */
172                AnalyzerTaskArray array = new AnalyzerTaskArray();
173                array.addAnalyzerTask(new TopologyAnalyzerTask(), "topo");
174                AnalyzerTaskComposite spatialTask = new AnalyzerTaskComposite();
175                spatialTask.addTask(new SpatialAnalyzerTask());
176//              spatialTask.addTask(new ExtendedSpatialAnalyzerTask());
177                array.addAnalyzerTask(spatialTask, "spatial");
178                array.addAnalyzerTask(new SocialAnalyzerTask(), "social");
179                
180                analyzer.setAnalyzerTask(array);
181                /*
182                 * sample...
183                 */
184                if(conservePk)
185                        sampler.sample(y, ergm, analyzer);//, new ConserveDegreeDistribution());
186                else
187                        sampler.sample(y, ergm, analyzer);
188                logger.info("Done.");
189        }

191        public static SocialSparseGraph createGraph(String type, Set<Person> persons, long randomSeed, SocialSparseGraphBuilder builder, Config config) {
192                SocialSparseGraph graph = null;
193                if("ba".equalsIgnoreCase(type)) {
194                        BarabasiAlbertGenerator<SocialSparseGraph, SocialSparseVertex, SocialSparseEdge> generator = new BarabasiAlbertGenerator<>(builder);
195                        graph = generator.generate(5, 5, persons.size() - 5 , randomSeed);
196                        
197                } else if("random".equalsIgnoreCase(type)) {
198                        ErdosRenyiGenerator<SocialSparseGraph, SocialSparseVertex, SocialSparseEdge> generator = new ErdosRenyiGenerator<>(builder);
199                        double p = 14.8/(persons.size() - 1);
200                        generator.setRandomDrawMode(true);
201                        graph = generator.generate(persons.size(), p, randomSeed);
202                        
203                } else if("lognormal".equalsIgnoreCase(type)) {
204                        double sigma = Double.parseDouble(config.getParam(MODULE_NAME, "k_sigma"));
205                        double mu = Double.parseDouble(config.getParam(MODULE_NAME, "k_mu"));
206                        int k_max = Integer.parseInt(config.getParam(MODULE_NAME, "k_max"));
207                        UnivariateRealFunction function = new LogNormalDistribution(sigma, mu, 1);
208                        RandomGraphGenerator<SocialSparseGraph, SocialSparseVertex, SocialSparseEdge> generator = new RandomGraphGenerator<>(function, builder, randomSeed);
209                        graph = generator.generate(persons.size(), k_max);
210                        
211                } else if("empty".equalsIgnoreCase(type)) {
212                        graph = builder.createGraph();
213                        for(int i = 0; i < persons.size(); i++) {
214                                builder.addVertex(graph);
215                        }
216                        
217                } else {
218                        throw new IllegalArgumentException("No graph type specified!");
219                }
220                
221                return graph;
222        }
