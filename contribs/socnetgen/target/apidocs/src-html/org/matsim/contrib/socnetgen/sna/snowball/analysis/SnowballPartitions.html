020package org.matsim.contrib.socnetgen.sna.snowball.analysis;
021
022import gnu.trove.map.hash.TIntObjectHashMap;
023
024import java.util.ArrayList;
025import java.util.Collection;
026import java.util.HashSet;
027import java.util.List;
028import java.util.Set;
029
030import org.matsim.contrib.socnetgen.sna.snowball.SampledVertex;
031
032/**
033 * Utility class for creating subsets of snowball sampled vertices.
034 * 
035 * @author illenberger
036 * 
037 */
038public class SnowballPartitions {
039
040        /**
041         * Creates a subset of sampled vertices.
042         * 
043         * @param <V>
044         *            the vertex type.
045         * @param vertices
046         *            a set of vertices.
047         * @return a subset of sampled vertices.
048         */
049        public static <V extends SampledVertex> Set<V> createSampledPartition(Collection<V> vertices) {
050                Set<V> partition = new HashSet<V>();
051                for (V vertex : vertices) {
052                        if (vertex.isSampled())
053                                partition.add(vertex);
054                }
055                return partition;
056        }
057
058        /**
059         * Creates a subset of vertices sampled in a specific iteration.
060         * 
061         * @param <V>
062         *            the vertex type.
063         * @param vertices
064         *            a set of vertices.
065         * @param iteration
066         *            the snowball iteration.
067         * @return a subset of vertices sampled in iteration <tt>iteration</tt>.
068         */
069        public static <V extends SampledVertex> Set<V> createSampledPartition(Collection<V> vertices, int iteration) {
070                Set<V> partition = new HashSet<V>();
071                for (V vertex : vertices) {
072                        if (vertex.isSampled()) {
073                                if (vertex.getIterationSampled() == iteration)
074                                        partition.add(vertex);
075                        }
076                }
077                return partition;
078        }
079
080        /**
081         * Creates a subset of vertices detected in a specific iteration.
082         * 
083         * @param <V>
084         *            the vertex type.
085         * @param vertices
086         *            a set of vertices.
087         * @param iteration
088         *            the snowball iteration.
089         * @return a subset of vertices detected in a specific iteration.
090         */
091        public static <V extends SampledVertex> Set<V> createDetectedPartition(Collection<V> vertices, int iteration) {
092                Set<V> partition = new HashSet<V>();
093                for (V vertex : vertices) {
094                        if (vertex.isDetected()) {
095                                if (vertex.getIterationDetected() == iteration)
096                                        vertices.add(vertex);
097                        }
098                }
099                return partition;
100        }
101
102        /**
103         * Creates a list of sets of sampled vertices where the list index
104         * corresponds to the iteration the vertices have been sampled.
105         * 
106         * @param <V>
107         *            the vertex type.
108         * @param vertices
109         *            a set of vertices.
110         * @return a list of sets of sampled vertices where the list index
111         *         corresponds to the iteration the vertices have been sampled.
112         */
113        public static <V extends SampledVertex> List<Set<V>> createSampledPartitions(Collection<V> vertices) {
114                TIntObjectHashMap<Set<V>> partitions = new TIntObjectHashMap<Set<V>>();
115                for (V vertex : vertices) {
116                        if (vertex.isSampled()) {
117                                int it = vertex.getIterationSampled();
118                                Set<V> partition = partitions.get(it);
119                                if (partition == null) {
120                                        partition = new HashSet<V>();
121                                        partitions.put(it, partition);
122                                }
123
124                                partition.add(vertex);
125                        }
126                }
127
128                List<Set<V>> list = new ArrayList<Set<V>>(partitions.size());
129                for (int i = 0; i < partitions.size() - 1; i++) {
130                        list.add(partitions.get(i));
131                }
132
133                return list;
134        }
135}
