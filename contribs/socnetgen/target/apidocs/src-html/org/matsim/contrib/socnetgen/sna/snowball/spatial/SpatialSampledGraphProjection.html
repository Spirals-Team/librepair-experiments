020package org.matsim.contrib.socnetgen.sna.snowball.spatial;
021
022import org.matsim.contrib.socnetgen.sna.graph.SparseVertex;
023import org.matsim.contrib.socnetgen.sna.graph.spatial.SpatialEdge;
024import org.matsim.contrib.socnetgen.sna.graph.spatial.SpatialGraph;
025import org.matsim.contrib.socnetgen.sna.graph.spatial.SpatialVertex;
026import org.matsim.contrib.socnetgen.sna.snowball.SampledGraphProjection;
027import org.opengis.referencing.crs.CoordinateReferenceSystem;
028
029import java.util.Set;
030
031
032/**
033 * Extension of {@link SampledGraphProjection} that implements
034 * {@link SpatialGraph}.
035 * 
036 * @see {@link SampledGraphProjection}
037 * @see {@link SpatialGraph}
038 * @author illenberger
039 * 
040 */
041public class SpatialSampledGraphProjection <G extends SpatialGraph, V extends SpatialVertex, E extends SpatialEdge> extends
042                SampledGraphProjection<G, V, E> implements SpatialGraph {
043
044        /**
045         * Creates an empty spatial sampled graph projection.
046         * 
047         * @param delegate the original graph.
048         */
049        public SpatialSampledGraphProjection(G delegate) {
050                super(delegate);
051        }
052        
053        /**
054         * @see {@link SampledGraphProjection#getEdges()}
055         */
056        @SuppressWarnings("unchecked")
057        @Override
058        public Set<? extends SpatialSampledEdgeDecorator<E>> getEdges() {
059                return (Set<? extends SpatialSampledEdgeDecorator<E>>) super.getEdges();
060        }
061
062        /**
063         * @see {@link SampledGraphProjection#getVertices()}
064         */
065        @SuppressWarnings("unchecked")
066        @Override
067        public Set<? extends SpatialSampledVertexDecorator<V>> getVertices() {
068                return (Set<? extends SpatialSampledVertexDecorator<V>>) super.getVertices();
069        }
070
071        /**
072         * @see {@link SampledGraphProjection#getEdge(SparseVertex, SparseVertex)}
073         */
074        @Override
075        public SpatialSampledEdgeDecorator<E> getEdge(SparseVertex v_i, SparseVertex v_j) {
076                return (SpatialSampledEdgeDecorator<E>) super.getEdge(v_i, v_j);
077        }
078
079        /**
080         * @see {@link SampledGraphProjection#getVertex(SpatialVertex)}
081         */
082        @Override
083        public SpatialSampledVertexDecorator<V> getVertex(V v) {
084                return (SpatialSampledVertexDecorator<V>) super.getVertex(v);
085        }
086
087        /**
088         * @see {@link SpatialGraph#getCoordinateReferenceSysten()}
089         */
090        @Override
091        public CoordinateReferenceSystem getCoordinateReferenceSysten() {
092                return getDelegate().getCoordinateReferenceSysten();
093        }
094
095}
