130        /**
131         * Starts the snowball sampling. The sampler will draw the seed vertices
132         * from the seed generator and expand to all vertices indicated as
133         * responding by the response generator until all reachable vertices have
134         * been sampled or the sampler listener terminates the sampling process.
135         * 
136         * @param graph The graph to sample.
137         */
138        public void run(G graph) {
139                List<TaggedVertex> recruits = init(graph);
140                /*
141                 * loop until no recruits are available
142                 */
143                while (!recruits.isEmpty()) {
144                        logger.info(String.format("Sampling iteration %1$s.", iteration));
145
146                        List<TaggedVertex> newRecruits = new LinkedList<TaggedVertex>();
147                        Collections.shuffle(recruits, random);
148                        for (TaggedVertex vertex : recruits) {
149                                if (!listener.beforeSampling(this, vertex.getProjection()))
150                                        return;
151                                /*
152                                 * check if the vertex is responsive
153                                 */
154                                if (vertex.isResponsive()) {
155                                        /*
156                                         * expand and collect new recruits
157                                         */
158                                        expand(vertex, newRecruits);
159                                }
160                                /*
161                                 * notify listener and terminate if applicable
162                                 */
163                                if (!listener.afterSampling(this, vertex.getProjection()))
164                                        return;
165                        }
166                        recruits = newRecruits;
167                        iteration++;
168                }
169
170                listener.endSampling(this);
171
172                logger.info("Done.");
173        }
174
175        @SuppressWarnings("unchecked")
176        private List<TaggedVertex> init(G graph) {
177                iteration = 0;
178                numSampledVertices = 0;
179                /*
180                 * 
181                 */
182                if (listener == null)
183                        listener = new DefaultListener();
184                /*
185                 * create an internally tagged graph for the sampler
186                 */
187                TaggedGraphBuilder tBuilder = new TaggedGraphBuilder();
188                taggedGraph = tBuilder.decorateGraph(graph);
189                /*
190                 * create a sampled graph
191                 */
192                if(builder == null)
193                        builder = new SampledGraphProjectionBuilder<G, V, E>();
194                sampledGraph = builder.createGraph(graph);
195                /*
196                 * tag the responsive vertices
197                 */
198                if (responseGenerator == null)
199                        responseGenerator = new AllResponding();
200
201                Set<V> responsive = responseGenerator.apply((Set<V>) graph.getVertices());
202                for (V vertex : responsive) {
203                        taggedGraph.getVertex(vertex).setResponsive(true);
204                }
205                /*
206                 * draw the seed vertices
207                 */
208                Set<V> seeds = seedGenerator.apply(responsive);
209                List<TaggedVertex> taggedSeeds = new LinkedList<TaggedVertex>();
210                for (V vertex : seeds) {
211                        TaggedVertex taggedVertex = taggedGraph.getVertex(vertex);
212                        SampledVertexDecorator<V> sampledVertex = builder.addVertex(sampledGraph, vertex);
213                        sampledVertex.detect(-1);
214                        sampledVertex.setSeed(sampledVertex);
215                        taggedVertex.setProjection(sampledVertex);
216                        taggedSeeds.add(taggedVertex);
217
218                }
219
220                return taggedSeeds;
221        }
222
223        private void expand(TaggedVertex vertex, List<TaggedVertex> recruits) {
224                /*
225                 * sample vertex
226                 */
227                vertex.getProjection().sample(iteration);
228                numSampledVertices++;
229                /*
230                 * expand to neighbors
231                 */
232                for (TaggedEdge edge : vertex.getEdges()) {
233                        if (edge.getProjection() == null) {
234                                /*
235                                 * edge has not been sampled
236                                 */
237                                TaggedVertex neighbour = edge.getOpposite(vertex);
238                                SampledVertexDecorator<V> sampledNeighbour;
239                                if (neighbour.getProjection() == null) {
240                                        /*
241                                         * neighbor has not been sampled
242                                         */
243                                        sampledNeighbour = builder.addVertex(sampledGraph, neighbour.getDelegate());
244                                        sampledNeighbour.detect(iteration);
245                                        sampledNeighbour.setSeed(vertex.getProjection().getSeed());
246                                        neighbour.setProjection(sampledNeighbour);
247                                        /*
248                                         * recruit
249                                         */
250                                        recruits.add(neighbour);
251                                } else {
252                                        sampledNeighbour = neighbour.getProjection();
253                                }
254                                /*
255                                 * sample edge
256                                 */
257                                SampledEdgeDecorator<E> sampledEdge = builder.addEdge(sampledGraph, vertex.getProjection(),
258                                                sampledNeighbour, edge.getDelegate());
259                                edge.setProjection(sampledEdge);
260                        }
261                }
262        }
