020package org.matsim.contrib.socnetgen.sna.graph.analysis;
021
022import gnu.trove.iterator.TObjectDoubleIterator;
023import gnu.trove.map.hash.TObjectDoubleHashMap;
024
025import java.util.Collection;
026import java.util.HashSet;
027import java.util.List;
028import java.util.Set;
029
030import org.apache.commons.math.stat.descriptive.DescriptiveStatistics;
031import org.matsim.contrib.socnetgen.sna.graph.Graph;
032import org.matsim.contrib.socnetgen.sna.graph.Vertex;
033
034/**
035 * A class that provides functionality to analyze triangular configurations in a
036 * graph.
037 * 
038 * @author illenberger
039 * 
040 */
041public class Transitivity extends AbstractVertexProperty {
042
043        private static Transitivity instance;
044        
045        public static Transitivity getInstance() {
046                if(instance == null)
047                        instance = new Transitivity();
048                return instance;
049        }
050        
051        /**
052         * Calculates the distribution of local clustering coefficients.
053         * 
054         * @param vertices
055         *            a set of vertices.
056         * 
057         * @return the distribution of local clustering coefficients.
058         */
059        public DescriptiveStatistics localClusteringDistribution(Set<? extends Vertex> vertices) {
060                DescriptiveStatistics stats = new DescriptiveStatistics();
061                TObjectDoubleHashMap<?> values = values(vertices);
062                TObjectDoubleIterator<?> it = values.iterator();
063                for(int i = 0; i < values.size(); i++) {
064                        it.advance();
065                        stats.addValue(it.value());
066                }
067                return stats;
068        }
069
070        /**
071         * Calculates the local clustering coefficient of each vertex in
072         * <tt>vertices</tt>. The local clustering coefficient is defined as:
073         * <ul>
074         * <li>C = 0 if k = 0 or k = 1,
075         * <li>C = 2y/k*(k-1) if k > 1, where y is the number of edges connecting
076         * neighbors of the vertex.
077         * </ul>
078         * 
079         * @param vertices
080         *            a collection of vertices the local clustering coefficients are
081         *            to be calculated.
082         * @return an object-double map containing the vertex as key and the
083         *         clustering coefficient as value.
084         */
085        public <V extends Vertex> TObjectDoubleHashMap<V> localClusteringCoefficients(Collection<V> vertices) {
086                TObjectDoubleHashMap<V> cc = new TObjectDoubleHashMap<V>();
087                for (V v : vertices) {
088                        int k = v.getEdges().size();
089                        if (k == 0 || k == 1) {
090                                cc.put(v, 0.0);
091                        } else {
092                                cc.put(v, 2 * countAdjacentEdges(v) / (double) (k * (k - 1)));
093                        }
094                }
095
096                return cc;
097        }
098
099        /**
100         * Counts the number of edges that connect neighbors of <tt>v</tt>.
101         * 
102         * @param v
103         *            a vertex.
104         * @return the number of edges that connect neighbors of <tt>v</tt>.
105         */
106        protected int countAdjacentEdges(Vertex v) {
107                int edgecount = 0;
108                Set<Vertex> n1s = new HashSet<Vertex>(v.getNeighbours());
109                int n_Neighbours1 = v.getNeighbours().size();
110                for (int i1 = 0; i1 < n_Neighbours1; i1++) {
111                        Vertex n1 = v.getNeighbours().get(i1);
112                        int n_Neighbours2 = n1.getNeighbours().size();
113                        for (int i2 = 0; i2 < n_Neighbours2; i2++) {
114                                Vertex n2 = n1.getNeighbours().get(i2);
115                                if (n2 != v) {
116                                        if (n1s.contains(n2))
117                                                edgecount++;
118                                }
119                        }
120                        n1s.remove(n1);
121                }
122                return edgecount;
123        }
124
125        /**
126         * Calculates the global clustering coefficient, which is defined as the
127         * three times the number of triangles over the number of connected triples.
128         * 
129         * @param graph
130         *            the graph the global clustering coefficient is calculated for.
131         * @return the global clustering coefficient.
132         */
133        public double globalClusteringCoefficient(Graph graph) {
134                int n_tripples = 0;
135                int n_triangles = 0;
136                for (Vertex v : graph.getVertices()) {
137                        List<? extends Vertex> n1s = v.getNeighbours();
138                        for (int i = 0; i < n1s.size(); i++) {
139                                List<? extends Vertex> n2s = n1s.get(i).getNeighbours();
140                                for (int k = 0; k < n2s.size(); k++) {
141                                        if (!n2s.get(k).equals(v)) {
142                                                n_tripples++;
143                                                if (n2s.get(k).getNeighbours().contains(v))
144                                                        n_triangles++;
145                                        }
146                                }
147                        }
148                }
149
150                return n_triangles / (double) n_tripples;
151        }
152
153        /**
154         * Calls {@link Transitivity#localClusteringCoefficients(Collection)}.
155         */
156        @SuppressWarnings("unchecked")
157        @Override
158        public TObjectDoubleHashMap<Vertex> values(Set<? extends Vertex> vertices) {
159                return (TObjectDoubleHashMap<Vertex>) localClusteringCoefficients(vertices);
160        }
161}
