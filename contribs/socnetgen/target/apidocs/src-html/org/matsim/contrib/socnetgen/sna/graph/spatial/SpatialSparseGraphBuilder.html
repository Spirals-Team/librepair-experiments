020package org.matsim.contrib.socnetgen.sna.graph.spatial;
021
022import com.vividsolutions.jts.geom.Point;
023import org.matsim.contrib.socnetgen.sna.graph.AbstractSparseGraphBuilder;
024import org.matsim.contrib.socnetgen.sna.graph.GraphFactory;
025import org.opengis.referencing.crs.CoordinateReferenceSystem;
026
027
028/**
029 * An extension to AbstractSparseGraphBuilder to build SpatialSparseGraphs.
030 * 
031 * @author illenberger
032 *
033 */
034public class SpatialSparseGraphBuilder extends AbstractSparseGraphBuilder<SpatialSparseGraph, SpatialSparseVertex, SpatialSparseEdge> {
035        
036        /**
037         * @param factory
038         */
039        public SpatialSparseGraphBuilder(GraphFactory<SpatialSparseGraph, SpatialSparseVertex, SpatialSparseEdge> factory) {
040                super(factory);
041                // TODO Auto-generated constructor stub
042        }
043
044        /**
045         * Creates a new builder where all spatial information refers to the
046         * coordinated reference system <tt>crs</tt>.
047         * 
048         * @param crs
049         *            a coordinate reference system.
050         */
051        public SpatialSparseGraphBuilder(CoordinateReferenceSystem crs) {
052                super(new SpatialSparseGraphFactory(crs));
053        }
054
055        /**
056         * Creates and adds a new vertex to a graph. The spatial reference id of
057         * <tt>point</tt> will be set to the one of the builders spatial reference
058         * system.
059         * 
060         * @param graph
061         *            the graph a new vertex should be inserted in.
062         * @param point
063         *            the point in space where the vertex is located.
064         * 
065         * @return the newly inserted vertex, or <tt>null</tt> is the insertion of a
066         *         vertex would violate the definition of the underlying graph.
067         */
068        public SpatialSparseVertex addVertex(SpatialSparseGraph graph, Point point) {
069                SpatialSparseVertex vertex = ((SpatialSparseGraphFactory)getFactory()).createVertex(point);
070                if(insertVertex(graph, vertex))
071                        return vertex;
072                else
073                        return null;
074        }
075}
