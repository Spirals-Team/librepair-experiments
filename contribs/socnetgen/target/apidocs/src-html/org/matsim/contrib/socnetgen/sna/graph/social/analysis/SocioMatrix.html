020package org.matsim.contrib.socnetgen.sna.graph.social.analysis;
021
022import java.io.BufferedWriter;
023import java.io.FileWriter;
024import java.io.IOException;
025import java.util.List;
026import java.util.Locale;
027
028/**
029 * @author illenberger
030 *
031 */
032public class SocioMatrix<K> {
033
034        private double[][] m;
035        
036        private List<K> keys;
037        
038        public SocioMatrix(List<K> keys) {
039                m = new double[keys.size()][keys.size()];
040                this.keys = keys;
041        }
042        
043        public List<K> getKeys() {
044                return keys;
045        }
046        
047        public void setValue(double value, K key1, K key2) {
048                int i = keys.indexOf(key1);
049                int j = keys.indexOf(key2);
050                
051                m[i][j] = value;
052        }
053        
054        public double getValue(K key1, K key2) {
055                int i = keys.indexOf(key1);
056                int j = keys.indexOf(key2);
057                
058                return m[i][j];
059        }
060        
061        public void adjustValue(K key1, K key2, double adjustValue) {
062                int i = keys.indexOf(key1);
063                int j = keys.indexOf(key2);
064                
065                m[i][j] += adjustValue;
066        }
067        
068        double[][] getMatrix() {
069                return m;
070        }
071        
072        public void toFile(String file) throws IOException {
073                BufferedWriter writer = new BufferedWriter(new FileWriter(file));
074                
075                for(K value : keys) {
076                        writer.write("\t");
077                        writer.write(value.toString());
078                }
079                writer.newLine();
080                
081                for(int i = 0; i < m.length; i++) {
082                        writer.write(keys.get(i).toString());
083                        
084                        for(int j = 0; j < m.length; j++) {
085                                writer.write("\t");
086                                writer.write(String.format(Locale.US, "%1$.2f", m[i][j]));
087                                
088                        }
089                        writer.newLine();
090                }
091                
092                writer.close();
093        }
094}
