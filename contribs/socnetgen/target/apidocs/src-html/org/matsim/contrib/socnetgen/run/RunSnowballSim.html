045    /**
046     * @param args
047     */
048    public static void main(String[] args) {
049        /*
050                 * Load config.
051                 */
052        Config config = loadConfig(args[0]);
053                /*
054                 * Load graph.
055                 */
056        Graph graph = loadGraph(config.getParam(MODULENAME, "graphFile"));
057                /*
058                 * Get random seed.
059                 */
060        long randomSeed = config.global().getRandomSeed();
061                /*
062                 * Init random seed generator.
063                 */
064        int numSeeds = Integer.parseInt(config.getParam(MODULENAME, "seeds"));
065        VertexFilter<Vertex> seedGenerator = new PropDegreeFixedSizePartition<Vertex>(numSeeds, randomSeed);
066                /*
067                 * Init response rate generator.
068                 */
069        double responseRate;
070        String str = config.getParam(MODULENAME, "responseRate");
071        if (str.endsWith("%")) {
072            str = str.substring(0, str.length() - 1);
073            responseRate = Double.parseDouble(str) / 100.0;
074        } else
075            responseRate = Double.parseDouble(str);
076
077        VertexFilter<Vertex> reponseGenerator = new RandomPartition<Vertex>(responseRate, randomSeed);
078                /*
079                 * Init estimators.
080                 */
081        PiEstimator estimator = new SimplePiEstimator(graph.getVertices().size());
082        RDSEstimator rdsEstimator = new RDSEstimator(graph.getVertices().size());
083        Map<String, AnalyzerTask> analyzers = loadAnalyzers(graph, estimator, rdsEstimator);
084        List<PiEstimator> estimators = new ArrayList<PiEstimator>(2);
085        estimators.add(estimator);
086        estimators.add(rdsEstimator);
087                /*
088                 * Get output directory.
089                 */
090        String output = config.getParam(MODULENAME, "output");
091                /*
092                 * Init sample analyzers.
093                 */
094//              final int interval = Integer.parseInt(config.getParam(MODULENAME, "interval"));
095//              IntervalSampleAnalyzer intervalAnalyzer = new IntervalSampleAnalyzer(analyzers, estimators, output);
096        IntervalSampleAnalyzer intervalAnalyzer = new LogIntervalSampleAnalyzer(analyzers, estimators, output, 1.5, 100, numSeeds);
097        IterationSampleAnalyzer iterationAnalyzer = new IterationSampleAnalyzer(analyzers, estimators, output);
098        FinalSampleAnalyzer completeAnalyzer = new FinalSampleAnalyzer(analyzers, estimators, output);
099//              ConnectionSampleAnalyzer connectionAnalyzer = new ConnectionSampleAnalyzer(numSeeds, analyzers, output);
100                /*
101                 * Init sampler listener.
102                 */
103        SamplerListenerComposite listeners = new SamplerListenerComposite();
104//              /*
105//               * Add estimators to listener.
106//               */
107//              for(EstimatorSet estimator : estimators.values()) {
108//                      if(estimator.distribution instanceof SamplerListener) {
109//                              listeners.addComponent((SamplerListener)estimator.distribution);
110//                      }
111//              }
112                /*
113                 * Add analyzers to listener.
114                 */
115        listeners.addComponent(intervalAnalyzer);
116        listeners.addComponent(iterationAnalyzer);
117        listeners.addComponent(completeAnalyzer);
118//              listeners.addComponent(connectionAnalyzer);
119//              listeners.addComponent(new DegreeGrowth(output));
120//              listeners.addComponent(new DegreeShare(output));
121                /*
122                 * Init and run sampler.
123                 */
124        SnowballSampler<Graph, Vertex, Edge> sampler = new SnowballSampler<Graph, Vertex, Edge>(randomSeed);
125        sampler.setSeedGenerator(seedGenerator);
126        sampler.setResponseGenerator(reponseGenerator);
127        sampler.setListener(listeners);
128
129        sampler.run(graph);
130    }

132    public static Config loadConfig(String file) {
133        Config config = new Config();
134        config.addCoreModules();
135        ConfigReader reader = new ConfigReader(config);
136        reader.readFile(file);
137        return config;
138    }

140    private static Graph loadGraph(String file) {
141        SparseGraphMLReader reader = new SparseGraphMLReader();
142        return reader.readGraph(file);

206    private static AnalyzerTask createDefaultEstimAnalyzer(PiEstimator estimator, DescriptivePiStatisticsFactory factory) {
207        AnalyzerTaskComposite tasks = new AnalyzerTaskComposite();
208
209        DegreeTask estimDegree = new DegreeTask();
210        estimDegree.setModule(new EstimatedDegree(estimator, factory));
211        tasks.addTask(estimDegree);
212
213        TransitivityTask estimTransitivity = new TransitivityTask();
214        EstimatedTransitivity trans = new EstimatedTransitivity(estimator, factory, true);
215        trans.enableCaching(false);
216        estimTransitivity.setModule(trans);
217        tasks.addTask(estimTransitivity);
218
219        tasks.addTask(new EstimatorTask(estimator));
220
221        return tasks;
222    }
