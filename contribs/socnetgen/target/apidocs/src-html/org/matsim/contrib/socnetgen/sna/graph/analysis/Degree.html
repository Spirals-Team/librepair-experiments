020package org.matsim.contrib.socnetgen.sna.graph.analysis;
021
022import gnu.trove.map.hash.TObjectDoubleHashMap;
023
024import java.util.Set;
025
026import org.apache.commons.math.stat.descriptive.DescriptiveStatistics;
027import org.matsim.contrib.socnetgen.sna.graph.Edge;
028import org.matsim.contrib.socnetgen.sna.graph.Graph;
029import org.matsim.contrib.socnetgen.sna.graph.Vertex;
030
031/**
032 * A class that provides functionality to analyze degree related
033 * graph-properties.
034 * 
035 * @author illenberger
036 * 
037 */
038public class Degree extends AbstractVertexProperty {
039
040        private static Degree instance;
041        
042        public static Degree getInstance() {
043                if(instance == null)
044                        instance = new Degree();
045                return instance;
046        }
047        
048        /**
049         * Returns a descriptive statistics object containing the elements of <tt>vertices</tt>. The
050         * graph is treated as undirected.
051         * 
052         * @param vertices
053         *            a collection of vertices.
054         *            
055         * @return a descriptive statistics object.
056         */
057        public DescriptiveStatistics statistics(Set<? extends Vertex> vertices) {
058                DescriptiveStatistics distribution = new DescriptiveStatistics();
059                for (Vertex v : vertices)
060                        distribution.addValue(v.getEdges().size());
061
062                return distribution;
063        }
064
065        /**
066         * Returns a Vertex-double-map containing the degree of each vertex.
067         * 
068         * @param vertices
069         *            a set of vertices
070         * @return a Vertex-double-map containing the degree of each vertex.
071         */
072        public TObjectDoubleHashMap<Vertex> values(Set<? extends Vertex> vertices) {
073                TObjectDoubleHashMap<Vertex> values = new TObjectDoubleHashMap<Vertex>();
074                for (Vertex v : vertices)
075                        values.put(v, v.getEdges().size());
076
077                return values;
078        }
079
080        /**
081         * Calculates the degree correlation of graph <tt>graph</tt>.<br>
082         * See: M. E. J. Newman. Assortative mixing in networks. Physical Review
083         * Letters, 89(20), 2002.
084         * 
085         * @param g
086         *            the graph the degree correlation is to be calculated.
087         * @return the degree correlation.
088         */
089        public double assortativity(Graph graph) {
090                double product = 0;
091                double sum = 0;
092                double squareSum = 0;
093
094                for (Edge e : graph.getEdges()) {
095                        Vertex v_i = e.getVertices().getFirst();
096                        Vertex v_j = e.getVertices().getSecond();
097                        
098                        int k_i = v_i.getEdges().size();
099                        int k_j = v_j.getEdges().size();
100
101                        sum += 0.5 * (k_i + k_j);
102                        squareSum += 0.5 * (Math.pow(k_i, 2) + Math.pow(k_j, 2));
103                        product += k_i * k_j;
104                }
105
106                double norm = 1 / (double) graph.getEdges().size();
107                return ((norm * product) - Math.pow(norm * sum, 2)) / ((norm * squareSum) - Math.pow(norm * sum, 2));
108        }
109}
