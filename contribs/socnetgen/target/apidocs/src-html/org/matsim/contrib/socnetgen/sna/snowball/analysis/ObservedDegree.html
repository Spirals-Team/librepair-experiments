077        /**
078         * Calculated the Pearson correlation coefficient of the degrees on either
079         * end of all sampled edges, i.e., edges where both vertices are sampled.
080         * 
081         * @param graph
082         *            a sampled graph
083         * @return the Pearson correlation coefficient of the degrees on either end
084         *         of all sampled edges.
085         */
086        @Override
087        public double assortativity(Graph g) {
088                SampledGraph graph = (SampledGraph) g;
089
090                double product = 0;
091                double sum = 0;
092                double squareSum = 0;
093                double M = 0;
094                
095                for (SampledEdge e : graph.getEdges()) {
096                        SampledVertex v_i = e.getVertices().getFirst();
097                        SampledVertex v_j = e.getVertices().getSecond();
098                        if (v_i.isSampled() && v_j.isSampled()) {
099                                int k_i = v_i.getEdges().size();
100                                int k_j = v_j.getEdges().size();
101
102                                sum += 0.5 * (k_i + k_j);
103                                squareSum += 0.5 * (Math.pow(k_i, 2) + Math.pow(k_j, 2));
104                                product += k_i * k_j;
105                                
106                                M++;
107                        }
108                }
109
110                double norm = 1 / M;
111                return ((norm * product) - Math.pow(norm * sum, 2)) / ((norm * squareSum) - Math.pow(norm * sum, 2));
