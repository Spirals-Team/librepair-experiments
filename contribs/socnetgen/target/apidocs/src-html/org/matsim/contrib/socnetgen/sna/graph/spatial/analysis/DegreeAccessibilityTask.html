041/**
042 * @author illenberger
043 *
044 */
045public class DegreeAccessibilityTask extends ModuleAnalyzerTask<Accessibility> {
046
047        public DegreeAccessibilityTask(Accessibility module) {
048                setModule(module);
049        }
050        
051        public DegreeAccessibilityTask(SpatialCostFunction function) {
052                setModule(new Accessibility(function));
053        }
054        
055        @Override
056        public void analyze(Graph g, Map<String, DescriptiveStatistics> statsMap) {
057                SpatialGraph graph = (SpatialGraph) g;
058                
059                TObjectDoubleHashMap<Vertex> xVals = module.values(graph.getVertices());
060                TObjectDoubleHashMap<Vertex> yVals = Degree.getInstance().values(graph.getVertices());
061                
062                double[] xArray = new double[xVals.size()];
063                double[] yArray = new double[xVals.size()];
064                TObjectDoubleIterator<Vertex> it = xVals.iterator();
065                for(int i = 0; i < xVals.size(); i++) {
066                        it.advance();
067                        xArray[i] = it.value();
068                        yArray[i] = yVals.get(it.key());
069                }
070                double r = new PearsonsCorrelation().correlation(xArray, yArray);
071                
072                printStats(singleValueStats("r_ka", r, statsMap), "r_kA");
073                
074                if(outputDirectoryNotNull()) {
075                        
076                        
077                        TDoubleDoubleHashMap correl = VertexPropertyCorrelation.mean(yVals, xVals, FixedSampleSizeDiscretizer.create(xVals.values(), 20, 100));
078//                      TDoubleDoubleHashMap correl = VertexPropertyCorrelation.mean(yVals, xVals, new LinearDiscretizer(5.0));
079                        try {
080                                StatsWriter.writeHistogram(correl, "A", "k", getOutputDirectory() + "k_mean_A.txt");
081                        } catch (IOException e) {
082                                e.printStackTrace();
083                        }
084                        
085                        it = xVals.iterator();
086                        for(int i = 0; i < xVals.size(); i++) {
087                                it.advance();
088                                double A = it.value();
089                                double k = yVals.get(it.key());
090                                
091                                yVals.put(it.key(), k/A);
092                        }
093                        
094                        correl = VertexPropertyCorrelation.mean(yVals, xVals, FixedSampleSizeDiscretizer.create(xVals.values(), 20, 100));
095
096                        try {
097                                StatsWriter.writeHistogram(correl, "A", "c_i", getOutputDirectory() + "c_i_A.txt");
098                        } catch (IOException e) {
099                                e.printStackTrace();
100                        }
101                }
102                
103        }
104
105}
