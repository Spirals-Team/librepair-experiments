051
052        /**
053         * Creates new estimated degree object configured with <tt>estimator</tt> as
054         * pi-estimator and <tt>factory</tt> as the factory for creating new
055         * instances of {@link DescriptivePiStatistics} which does the actual
056         * estimation of degree properties.
057         * 
058         * @param estimator a pi-estimator
059         * @param factory a factory for creating new instance of {@link DescriptivePiStatistics}.
060         */
061        public EstimatedDegree(PiEstimator estimator, DescriptivePiStatisticsFactory factory) {
062                this.piEstimator = estimator;
063                this.factory = factory;
064        }
065        
066        /**
067         * Returns a descriptive statistics object containing all sampled vertices
068         * each associated a pi-value.
069         * 
070         * @param vertices
071         *            a set of sampled vertices
072         * @return a descriptive statistics object.
073         */
074        @SuppressWarnings("unchecked")
075        @Override
076        public DescriptivePiStatistics statistics(Set<? extends Vertex> vertices) {
077                DescriptivePiStatistics stats = factory.newInstance();
078        
079                int p0 = 0;
080                
081                Set<SampledVertex> samples = SnowballPartitions.<SampledVertex>createSampledPartition((Set<SampledVertex>)vertices);
082        
083                for(SampledVertex vertex : samples) {
084                        double p = piEstimator.probability(vertex);
085                        if(p > 0) {
086                                stats.addValue(vertex.getNeighbours().size(), p);
087                        } else
088                                p0++;
089                }
090                
091                if(p0 > 0)
092                        logger.warn(String.format("There are %1$s vertices with probability 0!", p0));
093                
094                return stats;
095        }
096
097        /**
098         * @param vertices
099         *            a set of sampled vertices
100         * @return an object-double map containing all sampled vertices and their
101         *         degree.
102         */
103        @SuppressWarnings("unchecked")
104        @Override
105        public TObjectDoubleHashMap<Vertex> values(Set<? extends Vertex> vertices) {
106                return super.values(SnowballPartitions.createSampledPartition((Set<SampledVertex>)vertices));
107        }
