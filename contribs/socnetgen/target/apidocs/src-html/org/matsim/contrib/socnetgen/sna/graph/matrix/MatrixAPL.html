020package org.matsim.contrib.socnetgen.sna.graph.matrix;
021
022import gnu.trove.list.array.TDoubleArrayList;
023import gnu.trove.list.array.TIntArrayList;
024
025import java.util.ArrayList;
026import java.util.List;
027
028import org.apache.commons.math.stat.descriptive.DescriptiveStatistics;
029import org.apache.log4j.Logger;
030import org.matsim.contrib.common.util.ProgressLogger;
031import org.matsim.contrib.socnetgen.sna.util.MultiThreading;
032
033/**
034 * @author illenberger
035 * 
036 */
037public class MatrixAPL {
038
039        private static final Logger logger = Logger.getLogger(MatrixAPL.class);
040        
041        private int numThreads;
042        
043        private boolean calcDistr = true;
044        
045        public MatrixAPL() {
046                this.numThreads = MultiThreading.getNumAllowedThreads();
047        }
048        
049        public MatrixAPL(int numThreads) {
050                this.numThreads = numThreads;
051        }
052        
053        public void setCalcAPLDistribution(boolean flag) {
054                this.calcDistr = flag;
055        }
056        
057        public DescriptiveStatistics apl(AdjacencyMatrix<?> y) {
058                List<APLThread> threads = new ArrayList<APLThread>();
059                int size = (int) Math.floor(y.getVertexCount() / (double) numThreads);
060                int i_start = 0;
061                int i_stop = size;
062                for (int i = 0; i < numThreads - 1; i++) {
063                        threads.add(new APLThread(y, new SingelPathDijkstra(y, new CostFunction()), i_start, i_stop, calcDistr));
064                        i_start = i_stop;
065                        i_stop += size;
066                }
067                threads.add(new APLThread(y, new SingelPathDijkstra(y, new CostFunction()), i_start, y.getVertexCount(), calcDistr));
068                /*
069                 * start threads
070                 */
071                logger.info(String.format("Calculating average path lenth on %1$s threads.", numThreads));
072                ProgressLogger.init(y.getVertexCount(), 1, 5);
073                for (APLThread thread : threads) {
074                        thread.start();
075                }
076                /*
077                 * wait for threads
078                 */
079                for (APLThread thread : threads) {
080                        try {
081                                thread.join();
082                        } catch (InterruptedException e) {
083                                e.printStackTrace();
084                        }
085                }
086                /*
087                 * merge results
088                 */
089                DescriptiveStatistics stats = new DescriptiveStatistics();
090                if(calcDistr) {
091                        for (APLThread thread : threads) {
092                                TDoubleArrayList vals = thread.getValues();
093                                for(int i = 0; i < vals.size(); i++) {
094                                        stats.addValue(vals.get(i));
095                                }
096                        }
097                } else {
098                        long lengthSum = 0;
099                        long pathCount = 0;
100                        for (APLThread thread : threads) {
101                                lengthSum += thread.lengthSum;
102                                pathCount += thread.pathCount;
103                        }
104                        stats.addValue(lengthSum/(double)pathCount);
105                }
106                
107                return stats;
108        }
109
110
111        private class APLThread extends Thread {
112
113                private final AdjacencyMatrix<?> y;
114
115                private final SingelPathDijkstra dijkstra;
116
117                private final int startIdx;
118
119                private final int endIdx;
120
121                private final TDoubleArrayList stats;
122
123                private final boolean calcDistr;
124                
125                private long lengthSum;
126                
127                private long pathCount;
128
129                APLThread(AdjacencyMatrix<?> y, SingelPathDijkstra dijkstra, int startIdx, int endIdx, boolean calcDistr) {
130                        this.y = y;
131                        this.dijkstra = dijkstra;
132                        this.startIdx = startIdx;
133                        this.endIdx = endIdx;
134                        this.calcDistr = calcDistr;
135                        if(calcDistr)
136                                this.stats = new TDoubleArrayList((int) ((endIdx - startIdx) * y.getVertexCount() / 2.0));
137                        else
138                                this.stats = new TDoubleArrayList();
139                }
140                
141                TDoubleArrayList getValues() {
142                        return stats;
143                }
144                
145                @Override
146                public void run() {
147                        for (int i = startIdx; i < endIdx; i++) {
148                                dijkstra.run(i, -1);
149                                for (int j = i + 1; j < y.getVertexCount(); j++) {
150                                        TIntArrayList path = dijkstra.getPath(i, j);
151                                        if (path != null) {
152                                                if (calcDistr)
153                                                        stats.add(path.size());
154                                                else {
155                                                        lengthSum += path.size();
156                                                        pathCount++;
157                                                }
158                                        }
159                                }
160                                ProgressLogger.step();
161                        }
162                }
163
164        }
165        
166        private class CostFunction implements EdgeCostFunction {
167
168                @Override
169                public double edgeCost(int i, int j) {
170                        return 1;
171                }
172                
173        }
174}
