024package org.matsim.contrib.socnetgen.sna.graph;
025
026import java.util.HashMap;
027import java.util.Map;
028import java.util.Set;
029
030/**
031 * This class allows to extract subgraphs from existing graphs by projecting
032 * specific vertices and edges onto a new graph. A GraphProjection works similar
033 * to a decorator, however, the connectivity of vertices and edges can differ
034 * from the original graph.
035 * 
036 * @author illenberger
037 * 
038 */
039public class GraphProjection<G extends Graph, V extends Vertex, E extends Edge>
040                extends SparseGraph {
041
042        private G delegate;
043
044        private Map<V, VertexDecorator<V>> vMapping = new HashMap<V, VertexDecorator<V>>();
045
046        private Map<E, EdgeDecorator<E>> eMapping = new HashMap<E, EdgeDecorator<E>>();
047
048        /**
049         * Creates a new empty projection from graph <tt>delegate</tt>.
050         * 
051         * @param delegate
052         *            the original graph.
053         */
054        public GraphProjection(G delegate) {
055                this.delegate = delegate;
056        }
057
058        /**
059         * Returns the original graph.
060         * 
061         * @return the original graph.
062         */
063        public G getDelegate() {
064                return delegate;
065        }
066
067        /**
068         * Returns the decorator representing the projection of vertex <tt>v</tt>.
069         * 
070         * @param v
071         *            the original vertex.
072         * @return the decorater of vertex <tt>v</tt>, or <tt>null</tt> if there is
073         *         no projection for <tt>v</tt>.
074         */
075        public VertexDecorator<V> getVertex(V v) {
076                return vMapping.get(v);
077        }
078
079
080        /**
081         * Returns the decorator representing the projection of edge <tt>e</tt>.
082         * 
083         * @param v
084         *            the original edge.
085         * @return the decorater of edge <tt>e</tt>, or <tt>null</tt> if there is
086         *         no projection for <tt>e</tt>.
087         */
088        public EdgeDecorator<E> getEdge(E e) {
089                return eMapping.get(e);
090        }
091
092        /**
093         * Associates vertex <tt>delegate</tt> with its decorator <tt>decorator</tt>
094         * .
095         * 
096         * @param delegate
097         *            the original vertex.
098         * @param decorator
099         *            the decorator of the original vertex.
100         */
101        void setMapping(V delegate, VertexDecorator<V> decorator) {
102                vMapping.put(delegate, decorator);
103        }
104
105        /**
106         * Removes the mapping for <tt>delegate</tt>.
107         * 
108         * @param delegate
109         *            the original vertex.
110         */
111        void removeMapping(V delegate) {
112                vMapping.remove(delegate);
113        }
114
115        /**
116         * Associates edge <tt>delegate</tt> with its decorator <tt>decorator</tt>.
117         * 
118         * @param delegate
119         *            the original edge.
120         * @param decorator
121         *            the decorator of the original edge.
122         */
123        void setMapping(E delegate, EdgeDecorator<E> decorator) {
124                eMapping.put(delegate, decorator);
125        }
126
127        /**
128         * Removes the mapping for <tt>delegate</tt>.
129         * 
130         * @param delegate
131         *            the original edge.
132         */
133        void removeMapping(E delegate) {
134                eMapping.remove(delegate);
135        }
136
137        /**
138         * @see {@link Graph#getEdges()}.
139         */
140        @SuppressWarnings("unchecked")
141        @Override
142        public Set<? extends EdgeDecorator<E>> getEdges() {
143                return (Set<? extends EdgeDecorator<E>>) super.getEdges();
144        }
145
146        /**
147         * @see {@link Graph#getVertices()}.
148         */
149        @SuppressWarnings("unchecked")
150        @Override
151        public Set<? extends VertexDecorator<V>> getVertices() {
152                return (Set<? extends VertexDecorator<V>>) super.getVertices();
153        }
154
155        /**
156         * @see SparseGraph#getEdge(SparseVertex,
157         *      SparseVertex)
158         */
159        @SuppressWarnings("unchecked")
160        @Override
161        public EdgeDecorator<E> getEdge(SparseVertex v_i, SparseVertex v_j) {
162                return (EdgeDecorator<E>) super.getEdge(v_i, v_j);
163        }
164}
