038/**
039 * Implementation of {@link KMLObjectStyle} that displays an icon at the
040 * position of the placemark representing a vertex. Per default vertices will be
041 * drawn as a dot ("icon18.png" in the main matsim package) colored by

115        private Map<Vertex, StyleType> initVertexStyles(Graph graph) {
116                /*
117                 * Get the color for all vertices.
118                 */
119                Map<Vertex, Color> vertexColorMapping = new HashMap<Vertex, Color>();
120                Set<Color> colors = new HashSet<Color>();
121                for (Vertex v : graph.getVertices()) {
122                        Color c = vertexColorizer.getColor(v);
123                        colors.add(c);
124                        vertexColorMapping.put(v, c);
125                }
126                /*
127                 * Create s style type for all unique colors.
128                 */
129                LinkType kmlIconLink = kmlFactory.createLinkType();
130                kmlIconLink.setHref(ICON_HREF);
131                Map<Color, StyleType> colorStyles = new HashMap<Color, StyleType>();
132                for (Color c : colors) {
133                        IconStyleType kmlIconStyle = kmlFactory.createIconStyleType();
134                        kmlIconStyle.setIcon(kmlIconLink);
135                        kmlIconStyle.setScale(0.75);
136                        kmlIconStyle.setColor(new byte[] { (byte) c.getAlpha(), (byte) c.getBlue(), (byte) c.getGreen(),
137                                        (byte) c.getRed() });
138
139                        StyleType kmlStyle = kmlFactory.createStyleType();
140                        kmlStyle.setId(String.format("vertex%1$s", Integer.toHexString(c.getRGB())));
141                        kmlStyle.setIconStyle(kmlIconStyle);
142
143                        colorStyles.put(c, kmlStyle);
144                }
145                /*
146                 * Assign each vertex its style type.
147                 */
148                Map<Vertex, StyleType> vertexStyleMapping = new HashMap<Vertex, StyleType>();
149                for (Entry<Vertex, Color> entry : vertexColorMapping.entrySet()) {
150                        vertexStyleMapping.put(entry.getKey(), colorStyles.get(entry.getValue()));
151                }
152
153                return vertexStyleMapping;
154        }
155
156        /**
157         * @see {@link KMLObjectStyle#getStyle(Object)}
158         */
159        @Override
160        public StyleType getStyle(Object object) {
161                if (vertexStyles == null)
162                        vertexStyles = initVertexStyles(graph);
163
164                return vertexStyles.get(object);
165        }
