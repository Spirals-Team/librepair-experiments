119        /**
120         * Reads a zone layer from a shape file. The shape file should be a
121         * collection features of type "Zone".
122         * 
123         * @param filename the path to the shape file
124         * @return a zone layer initialized with the features read from the shape file.
125         * @throws IOException
126         */
127        public static ZoneLayer<Map<String, Object>> read(String filename) throws IOException {
128                Set<Zone<Map<String, Object>>> zones = new HashSet<Zone<Map<String, Object>>>();
129                for(SimpleFeature feature : EsriShapeIO.readFeatures(filename)) {
130                        Zone<Map<String, Object>> zone = new Zone<Map<String, Object>>((Geometry) feature.getDefaultGeometry());
131                        Map<String, Object> map = new HashMap<String, Object>(feature.getAttributeCount());
132                        for(Property prop : feature.getProperties()) {
133                                map.put(prop.getName().getLocalPart(), prop.getValue());
134                        }
135                        zone.setAttribute(map);
136                        zones.add(zone);
137                }
138                
139                return new ZoneLayer<Map<String, Object>>(zones);
140        }

142        public static ZoneLayer<Double> read(String filename, String key) throws IOException {
143                Set<Zone<Double>> zones = new HashSet<Zone<Double>>();
144                for(SimpleFeature feature : EsriShapeIO.readFeatures(filename)) {
145                        Zone<Double> zone = new Zone<Double>((Geometry) feature.getDefaultGeometry());
146//                      feature.getFeatureType().
147                        Object obj = feature.getAttribute(key);
148                        if (obj != null) {
149                                double val = Double.parseDouble(obj.toString()); // FIXME
150                                zone.setAttribute(val);
151                        }
152                        zones.add(zone);
153                }
154                
155                return new ZoneLayer<Double>(zones);
156        }

158        public static <T> void write(ZoneLayer<T> layer, String filename) throws IOException {
159                CoordinateReferenceSystem crs = layer.getCRS();
160                SimpleFeatureTypeBuilder typeBuilder = new SimpleFeatureTypeBuilder();
161                typeBuilder.setCRS(crs);
162                typeBuilder.setName("zone");
163                typeBuilder.add("the_geom", MultiPolygon.class);
164                typeBuilder.add("value", Double.class);
165                SimpleFeatureType featureType = typeBuilder.buildFeatureType();
166        
167                DefaultFeatureCollection collection = new DefaultFeatureCollection();
168        SimpleFeatureBuilder featureBuilder = new SimpleFeatureBuilder(featureType);
169
170        for(Zone<T> zone : layer.getZones()) {
171                featureBuilder.add(zone.getGeometry());
172                featureBuilder.add(zone.getAttribute());
173                SimpleFeature feature = featureBuilder.buildFeature(null);
174                collection.add(feature);
175        }
176
177        File newFile = new File(filename);
178
179        ShapefileDataStoreFactory dataStoreFactory = new ShapefileDataStoreFactory();
180
181        Map<String, Serializable> params = new HashMap<String, Serializable>();
182        params.put("url", newFile.toURI().toURL());
183        params.put("create spatial index", Boolean.TRUE);
184
185        ShapefileDataStore newDataStore = (ShapefileDataStore) dataStoreFactory.createNewDataStore(params);
186        newDataStore.createSchema(featureType);
187
188//        newDataStore.forceSchemaCRS(layer.getCRS());
189                
190        Transaction transaction = new DefaultTransaction("create");
191
192                String typeName = newDataStore.getTypeNames()[0];
193                SimpleFeatureSource featureSource = newDataStore.getFeatureSource(typeName);
194
195                SimpleFeatureStore featureStore = (SimpleFeatureStore) featureSource;
196
197                featureStore.setTransaction(transaction);
198                try {
199                        featureStore.addFeatures(collection);
200                        transaction.commit();
201
202                } catch (Exception problem) {
203                        problem.printStackTrace();
204                        transaction.rollback();
205
206                } finally {
207                        transaction.close();
208                }
209        }

211        public static void writeWithAttributes(ZoneLayer<Map<String, Object>> layer, String filename) throws IOException {
212                CoordinateReferenceSystem crs = layer.getCRS();
213                SimpleFeatureTypeBuilder typeBuilder = new SimpleFeatureTypeBuilder();
214                typeBuilder.setCRS(crs);
215                typeBuilder.setName("zone");
216                typeBuilder.add("the_geom", MultiPolygon.class);
217                /*
218                 * TODO: Only those attributes of this zone are written.
219                 */
220                Zone<Map<String, Object>> azone = layer.getZones().iterator().next();
221                List<String> keys = new ArrayList<>(azone.getAttribute().keySet());
222                for(String key : keys) {
223                        typeBuilder.add(key, azone.getAttribute().get(key).getClass());
224                }
225                
226                SimpleFeatureType featureType = typeBuilder.buildFeatureType();
227
228                DefaultFeatureCollection collection = new DefaultFeatureCollection();
229        SimpleFeatureBuilder featureBuilder = new SimpleFeatureBuilder(featureType);
230
231        for(Zone<Map<String, Object>> zone : layer.getZones()) {
232                featureBuilder.add(zone.getGeometry());
233                for(String key : keys) {
234                        featureBuilder.add(zone.getAttribute().get(key));
235                }
236                SimpleFeature feature = featureBuilder.buildFeature(null);
237                collection.add(feature);
238        }
239
240        File newFile = new File(filename);
241
242        ShapefileDataStoreFactory dataStoreFactory = new ShapefileDataStoreFactory();
243
244        Map<String, Serializable> params = new HashMap<String, Serializable>();

246        params.put("create spatial index", Boolean.TRUE);
247
248        ShapefileDataStore newDataStore = (ShapefileDataStore) dataStoreFactory.createNewDataStore(params);

250
251//        newDataStore.forceSchemaCRS(layer.getCRS());
252                
253        Transaction transaction = new DefaultTransaction("create");

255                String typeName = newDataStore.getTypeNames()[0];

266                        problem.printStackTrace();
267                        transaction.rollback();
268
269                } finally {
270                        transaction.close();
271                }
