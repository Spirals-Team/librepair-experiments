020package org.matsim.contrib.socnetgen.sna.graph.analysis;
021
022import org.apache.commons.math.stat.descriptive.DescriptiveStatistics;
023import org.matsim.contrib.common.stats.LinearDiscretizer;
024import org.matsim.contrib.socnetgen.sna.graph.Graph;
025
026import java.io.IOException;
027import java.util.Map;
028
029/**
030 * An AnalyzerTaks that calculated degree related measurements.
031 * 
032 * @author illenberger
033 * 
034 */
035public class DegreeTask extends ModuleAnalyzerTask<Degree> {
036
037        public static final String KEY = "k";
038
039        public static final String MEAN_DEGREE = "k_mean";
040
041        public static final String MIN_DEGREE = "k_min";
042
043        public static final String MAX_DEGREE = "k_max";
044
045        public static final String DEGREE_CORRELATION = "r_k";
046
047        /**
048         * Creates a new DegreeTask with an instance of {@link Degree} used for
049         * analysis.
050         */
051        public DegreeTask() {
052                setKey(KEY);
053                setModule(Degree.getInstance());
054        }
055
056        /**
057         * Determines the degree distribution and the degree correlation of a graph.
058         * Writes the histogram of the degree distribution into the output directory
059         * (if specified).
060         * 
061         * @param graph
062         *            a graph.
063         * @param stats
064         *            a map where the results of the analysis are stored.
065         */
066        @Override
067        public void analyze(Graph graph, Map<String, DescriptiveStatistics> statsMap) {
068                DescriptiveStatistics stats = module.statistics(graph.getVertices());
069                printStats(stats, key);
070                statsMap.put(key, stats);
071                if (outputDirectoryNotNull()) {
072                        try {
073                                writeHistograms(stats, new LinearDiscretizer(1.0), key, false);
074                                writeHistograms(stats, new LinearDiscretizer(5.0), key + "_5", false);
075                                writeHistograms(stats, key, 13, 1);
076                        } catch (IOException e) {
077                                e.printStackTrace();
078                        }
079                }
080
081                stats = new DescriptiveStatistics();
082                stats.addValue(module.assortativity(graph));
083                statsMap.put("r_" + key, stats);
084                printStats(stats, "r_" + key);
085        }
086
087}
