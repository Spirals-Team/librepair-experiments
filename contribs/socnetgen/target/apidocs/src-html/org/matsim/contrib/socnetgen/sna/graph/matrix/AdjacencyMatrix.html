040/**
041 * Matrix-representation of an undirected and unweighted graph.
042 * 
043 * @author illenberger
044 * 
045 */
046public class AdjacencyMatrix<V extends Vertex> {
047
048        private ArrayList<TIntArrayList> rows;
049
050        private ArrayList<TIntIntHashMap> commonNeighbors;
051
052        private List<V> vertices;
053
054        private TObjectIntHashMap<Vertex> vertexIndicies;
055
056        /**
057         * Creates a new empty adjacency matrix.
058         */
059        public AdjacencyMatrix() {
060                this(false);
061        }
062
063        /**
064         * Creates a new empty adjacency matrix.
065         * 
066         * @param trackCommonNeighbors
067         *            Indicates if this matrix caches common neighbors of vertices
068         *            during insertion and removal of edges. The method
069         *            {@link #getCommonNeighbors(int, int)} requires this switch to
070         *            be set to <tt>true</tt>.
071         */
072        public AdjacencyMatrix(boolean trackCommonNeighbors) {
073                rows = new ArrayList<TIntArrayList>(1000);
074                if (trackCommonNeighbors)
075                        commonNeighbors = new ArrayList<TIntIntHashMap>(1000);
076        }
077
078        /**
079         * Creates a new adjacency matrix out of graph <tt>g</tt>.
080         * 
081         * @param g
082         *            a graph
083         */
084        public AdjacencyMatrix(Graph g) {
085                this(g, false);
086        }
087
088        /**
089         * Creates a new adjacency matrix out of graph <tt>g</tt>.
090         * 
091         * @param g
092         *            a graph
093         * @param trackCommonNeighbors
094         *            Indicates if this matrix caches common neighbors of vertices
095         *            during insertion and removal of edges. The method
096         *            {@link #getCommonNeighbors(int, int)} requires this switch to
097         *            be set to <tt>true</tt>.
098         */
099        @SuppressWarnings("unchecked")
100        public AdjacencyMatrix(Graph g, boolean trackCommonNeighbors) {
101                rows = new ArrayList<TIntArrayList>(g.getVertices().size());
102                if (trackCommonNeighbors)
103                        commonNeighbors = new ArrayList<TIntIntHashMap>(g.getVertices().size());
104
105                vertices = new ArrayList<V>(g.getVertices().size());
106                /*
107                 * create vertices
108                 */
109                vertexIndicies = new TObjectIntHashMap<Vertex>();
110                int idx = 0;
111                for (Vertex v : g.getVertices()) {
112                        vertexIndicies.put(v, idx);
113                        vertices.add((V) v);
114                        addVertex();
115                        idx++;
116                }
117                /*
118                 * create edges
119                 */
120                for (Edge e : g.getEdges()) {
121                        Tuple<? extends Vertex, ? extends Vertex> p = e.getVertices();
122                        int i = -1;
123                        if (vertexIndicies.contains(p.getFirst()))
124                                i = vertexIndicies.get(p.getFirst());
125                        int j = -1;
126                        if (vertexIndicies.contains(p.getSecond()))
127                                j = vertexIndicies.get(p.getSecond());
128                        if (i > -1 && j > -1) {
129                                addEdge(i, j);
130                        } else {
131                                throw new IllegalArgumentException(String.format("Indices i=%1$s, j=%2$s not allowed!", i, j));
132                        }
133                }
134        }
135
136        /**
137         * Returns the vertex object associated with index <tt>i</tt>.
138         * 
139         * @param i
140         *            a vertex index.
141         * @return the vertex object associated with index <tt>i</tt> or
142         *         <tt>null</tt> if there is no vertex with index <tt>i</tt>.
143         */
144        public V getVertex(int i) {
145                if (i >= vertices.size())
146                        return null;
147                else
148                        return vertices.get(i);
149        }
150
151        /**
152         * If this matrix has been created from a graph, returns the index
153         * associated with <tt>vertex</tt>.
154         * 
155         * @param vertex
156         *            a vertex
157         * @return the index associated with <tt>vertex</tt>.
158         */
159        public int getIndex(V vertex) {
160                if (vertexIndicies.contains(vertex))
161                        return vertexIndicies.get(vertex);
162                else
163                        return -1;
164        }
165
166        /**
167         * Creates a new graph out of this adjacency matrix.
168         * 
169         * @param <G>
170         *            the type parameter of the graph.
171         * @param <E>
172         *            the type parameter of the edges.
173         * @param builder
174         *            a graph builder used to create the graph.
175         * @return a new graph out of this adjacency matrix.
176         */
177        public <G extends Graph, E extends Edge> G getGraph(GraphBuilder<G, V, E> builder) {
178                G g = builder.createGraph();
179                /*
180                 * create vertices
181                 */
182                List<V> vertexList = new ArrayList<V>(getVertexCount());
183                for (int i = 0; i < getVertexCount(); i++) {
184                        V v = builder.addVertex(g);
185                        if (v != null)
186                                vertexList.add(v);
187                        else
188                                throw new RuntimeException("Cannot add vertices to the graph.");
189
190                }
191                /*
192                 * add edges
193                 */
194                addEdges(g, builder, vertexList);
195
196                return g;
197        }
198
199        /**
200         * Synchronizes the edges of an existing graph with the same set of vertices
201         * with the edges of this matrix. This method will fail if either the number
202         * of vertices of this matrix or the set of vertices of the graph has
203         * changed since the construction of this matrix.
204         * 
205         * @param <G>
206         *            the type parameter of the graph.
207         * @param <E>
208         *            the type parameter of the edges.
209         * @param g
210         *            a graph with the same set of vertices.
211         * @param builder
212         *            a graph builder to add and remove edges to the graph.
213         */
214        @SuppressWarnings("unchecked")
215        public <G extends Graph, E extends Edge> void synchronizeEdges(G g, GraphBuilder<G, V, E> builder) {
216                if (g.getVertices().size() == getVertexCount()) {
217                        /*
218                         * remove all edges
219                         */
220                        Set<Edge> edges = new HashSet<Edge>(g.getEdges());
221                        for (Edge edge : edges)
222                                builder.removeEdge((G) g, (E) edge);
223                        /*
224                         * add edges
225                         */
226                        addEdges(g, builder, vertices);
227                } else {
228                        throw new UnsupportedOperationException("The size of the graph or matrix has changed.");
229                }
230
231        }
232
233        private <G extends Graph, E extends Edge> void addEdges(G g, GraphBuilder<G, V, E> builder, List<V> vertexList) {
234                for (int i = 0; i < rows.size(); i++) {
235                        TIntArrayList row = rows.get(i);
236                        for (int idx = 0; idx < row.size(); idx++) {
237                                int j = row.get(idx);
238                                if (j > i) {
239                                        if (builder.addEdge(g, vertexList.get(i), vertexList.get(j)) == null)
240                                                throw new UnsupportedOperationException(
241                                                                "There seems to be some inconsistency in the vertex set.");
242                                }
243                        }
244                }
245        }
246
247        /**
248         * Returns the size of the matrix.
249         * 
250         * @return the size of the matrix.
251         */
252        public int getVertexCount() {
253                return rows.size();
254        }
255
256        /**
257         * Returns the number of edges.
258         * 
259         * @return the number of edges.
260         */
261        public int countEdges() {
262                int edges = 0;
263                for (TIntArrayList row : rows) {
264                        edges += row.size();
265                }
266                return edges / 2;
267        }
268
269        /**
270         * Returns if vertex <tt>i</tt> and <tt>j</tt> are connected.
271         * 
272         * @param i
273         *            a vertex index.
274         * @param j
275         *            a vertex index.
276         * @return <tt>true</tt> if vertex <tt>i</tt> and <tt>j</tt> are connected,
277         *         <tt>false</tt> otherwise.
278         */
279        public boolean getEdge(int i, int j) {
280                return rows.get(i).contains(j);
281        }
282
283        /**
284         * Adds a new vertex to the matrix and returns its index.
285         * 
286         * @return the index of the new vertex.
287         */
288        public int addVertex() {
289                rows.add(new TIntArrayList(10));
290                if (commonNeighbors != null)
291                        commonNeighbors.add(new TIntIntHashMap());
292
293                return rows.size() - 1;
294        }
295
296        /**
297         * Connects vertex <tt>i</tt> and <tt>j</tt> with an edge.
298         * 
299         * @param i
300         *            a vertex index.
301         * @param j
302         *            a vertex index.
303         */
304        public void addEdge(int i, int j) {
305                if (i != j) {
306                        addEdgeInternal(i, j);
307                        addEdgeInternal(j, i);
308                } else {
309                        throw new UnsupportedOperationException("Self-loops are not allowed.");
310                }
311        }
312
313        private void addEdgeInternal(int i, int j) {
314                TIntArrayList neighbors_i = rows.get(i);
315
316                if (commonNeighbors != null) {
317                        TIntArrayList neighbors_j = rows.get(j);
318
319                        for (int k = 0; k < neighbors_j.size(); k++) {
320                                int neighbor = neighbors_j.get(k);
321                                if (neighbor != i) {
322                                        adjustCommonNeighbors(i, neighbor, 1);
323                                        adjustCommonNeighbors(neighbor, i, 1);
324                                }
325                        }
326                }
327
328                neighbors_i.add(j);
329        }
330
331        /**
332         * Removes the edge between <tt>i</tt> and <tt>j</tt>.
333         * 
334         * @param i
335         *            a vertex index.
336         * @param j
337         *            a vertex index.
338         */
339        public void removeEdge(int i, int j) {
340                removeEdgeInternal(i, j);
341                removeEdgeInternal(j, i);
342        }
343
344        private void removeEdgeInternal(int i, int j) {
345                TIntArrayList neighbors_i = rows.get(i);
346
347                if (commonNeighbors != null) {
348                        TIntArrayList neighbors_j = rows.get(j);
349
350                        for (int k = 0; k < neighbors_j.size(); k++) {
351                                int neighbor = neighbors_j.get(k);
352                                if (neighbor != i) {
353                                        adjustCommonNeighbors(i, neighbor, -1);
354                                        adjustCommonNeighbors(neighbor, i, -1);
355                                }
356                        }
357                }
358
359                neighbors_i.removeAt(neighbors_i.indexOf(j));
360        }
361
362        /**
363         * Returns the number of vertices that are adjacent to <tt>i</tt> and
364         * <tt>j</tt>. The method requires the <tt>trackCommonNeighbors</tt>-switch
365         * to be set to <tt>true</tt> during instantiation of this matrix.
366         * 
367         * @see {@link AdjacencyMatrix#AdjacencyMatrix(boolean)}
368         * @see {@link AdjacencyMatrix#AdjacencyMatrix(Graph, boolean)}
369         * @param i
370         *            a vertex index.
371         * @param j
372         *            a vertex index.
373         * @return the number of vertices that are adjacent to <tt>i</tt> and
374         *         <tt>j</tt>.
375         */
376        public int getCommonNeighbors(int i, int j) {
377                if (commonNeighbors != null)
378                        return commonNeighbors.get(i).get(j);
379                else
380                        throw new java.lang.UnsupportedOperationException("This matrix does not support counting common neighbors.");
381        }
382
383        private void adjustCommonNeighbors(int i, int j, int amount) {
384                TIntIntHashMap neighbours_i = commonNeighbors.get(i);
385                int n = neighbours_i.adjustOrPutValue(j, amount, amount);
386                if (n < 1)
387                        neighbours_i.remove(j);
388        }

390        /**
391         * Returns the number of neighbors of vertex <tt>i</tt>.
392         * 
393         * @param i
394         *            a vertex index.
395         * @return the number of neighbors of vertex <tt>i</tt>.
396         */
397        public int getNeighborCount(int i) {
398                return rows.get(i).size();
399        }
400
401        /**
402         * Returns the indices of the neighbors of vertex <tt>i</tt>.
403         * 
404         * @param i
405         *            a vertex index.
406         * @return the indices of the neighbors of vertex <tt>i</tt>.
407         */
408        public TIntArrayList getNeighbours(int i) {
409                return rows.get(i);
410        }
411
412        /**
413         * Counts the number of triangles connected to vertex <tt>i</tt>.
414         * 
415         * @param i
416         *            a vertex index.
417         * @return the number of triangles connected to vertex <tt>i</tt>.
418         */
419        public int countTriangles(int i) {
420                int triangles = 0;
421                TIntArrayList row = rows.get(i);
422                for (int u = 0; u < row.size(); u++) {
423                        for (int v = u + 1; v < row.size(); v++) {
424                                if (getEdge(row.get(u), row.get(v)))
425                                        triangles++;
426                        }
427                }
428                return triangles;
429        }
430
431        /**
432         * Counts the number of triples centered at vertex <tt>i</tt>.
433         * 
434         * @param i
435         *            a vertex index.
436         * @return the number of triples centered at vertex <tt>i</tt>.
437         */
438        public int countTriples(int i) {
439                int n = rows.get(i).size();
440                return (int) (n * (n - 1) / 2.0);
441        }
