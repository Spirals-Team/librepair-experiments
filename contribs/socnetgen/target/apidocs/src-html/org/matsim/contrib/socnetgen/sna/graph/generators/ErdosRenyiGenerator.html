024package org.matsim.contrib.socnetgen.sna.graph.generators;
025
026import org.apache.log4j.Logger;
027import org.matsim.contrib.common.util.ProgressLogger;
028import org.matsim.contrib.socnetgen.sna.graph.*;
029import org.matsim.contrib.socnetgen.sna.graph.io.GraphMLWriter;
030
031import java.io.FileNotFoundException;
032import java.io.IOException;
033import java.util.*;
034
035/**
036 * Random graph generator that connects vertices with constant probability.
037 *
038 * @author illenberger
039 */
040public class ErdosRenyiGenerator<G extends Graph, V extends Vertex, E extends Edge> {
041
042    private static final Logger logger = Logger.getLogger(ErdosRenyiGenerator.class);
043
044    private GraphBuilder<G, V, E> builder;
045
046    private boolean randomDraw = false;
047
048    /**
049     * Creates a new random graph generator.
050     *
051     * @param builder a graph builder to create new graphs.
052     */
053    public ErdosRenyiGenerator(GraphBuilder<G, V, E> builder) {
054        this.builder = builder;
055    }
056
057    /**
058     * Sets the mode of the generator to "random draw". Per default the generator will iterate over all possible edges
059     * an insert edges with the given probability. If <tt>randomDraw</tt> is set to <tt>true</tt> the generator will
060     * randomly draw <tt>p * n * (n - 1) / 2</tt> distinct pairs of vertices and connect them with edges.
061     *
062     * @param randomDraw <tt>true</tt> if the generator should operate in the "random draw" mode, <tt>false</tt>
063     *                   otherwise.
064     */
065    public void setRandomDrawMode(boolean randomDraw) {
066        this.randomDraw = randomDraw;
067    }
068
069    /**
070     * Creates a new graph with the specified number of vertices that are connected with constant probability.
071     *
072     * @param numVertices the number of vertices.
073     * @param p           the edge probability
074     * @param randomSeed  a random seed
075     * @return a new random graph.
076     */
077    public G generate(int numVertices, double p, long randomSeed) {
078        G g = builder.createGraph();
079
080        for (int i = 0; i < numVertices; i++)
081            builder.addVertex(g);
082
083        return generate(g, p, randomSeed);
084    }
085
086    /**
087     * Connects the vertices of an existing graph with constant probability.
088     *
089     * @param graph      an existing graph
090     * @param p          the edge probability
091     * @param randomSeed a random seed
092     * @return a graph with randomly inserted edges.
093     */
094    public G generate(G graph, double p, long randomSeed) {
095        if (randomDraw)
096            return randomDraw(graph, p, randomSeed);
097        else
098            return enumerate(graph, p, randomSeed);
099    }
100
101    @SuppressWarnings("unchecked")
102    private G enumerate(G graph, double p, long randomSeed) {
103        LinkedList<V> pending = new LinkedList<V>();
104        pending.addAll((Collection<? extends V>) graph.getVertices());
105
106        Random random = new Random(randomSeed);
107        V v1;
108        while ((v1 = pending.poll()) != null) {
109            for (V v2 : pending) {
110                if (random.nextDouble() <= p) {
111                    builder.addEdge(graph, v1, v2);
112                }
113            }
114        }
115
116        return graph;
117    }
118
119    @SuppressWarnings("unchecked")
120    private G randomDraw(G graph, double p, long randomSeed) {
121        long n = graph.getVertices().size();
122        long M = n * (n - 1) / 2;
123        int m = (int) (p * M);
124
125        Random random = new Random(randomSeed);
126        List<V> vertices = new ArrayList<V>((Collection<? extends V>) graph.getVertices());
127        ProgressLogger.init(m, 1, 5);
128        for (int i = 0; i < m; i++) {
129            E edge = null;
130            while (edge == null) {
131                V vi = vertices.get(random.nextInt((int) n));
132                V vj = vertices.get(random.nextInt((int) n));
133                edge = builder.addEdge(graph, vi, vj);
134            }
135            ProgressLogger.step();
136//                      if(i % 10000 == 0)
137//                              logger.info(String.format("Created %1$s of %2$s edges.", i+1, m));
138        }
139        ProgressLogger.terminate();
140        return graph;
141    }
142
143    /**
144     * Main-method to create random graphs.<br> Usage: ErdosRenyiGenerator graphMLFile numVertices probability
145     * [randomSeed]
146     *
147     * @param args command line arguments.
148     * @throws FileNotFoundException
149     * @throws IOException
150     */
151    public static void main(String args[]) throws FileNotFoundException, IOException {
152        int N = Integer.parseInt(args[1]);
153        double p = Double.parseDouble(args[2]);
154        long seed = (long) (Math.random() * 1000);
155        if (args.length > 3)
156            seed = Long.parseLong(args[3]);
157
158        ErdosRenyiGenerator<SparseGraph, SparseVertex, SparseEdge> generator = new ErdosRenyiGenerator<SparseGraph, SparseVertex, SparseEdge>(
159                new SparseGraphBuilder());
160        generator.setRandomDrawMode(true);
161        Graph g = generator.generate(N, p, seed);
162
163        GraphMLWriter writer = new GraphMLWriter();
164        writer.write(g, args[0]);
165    }
166}
