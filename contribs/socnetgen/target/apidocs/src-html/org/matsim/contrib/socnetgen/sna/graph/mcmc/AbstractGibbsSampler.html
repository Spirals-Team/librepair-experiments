020package org.matsim.contrib.socnetgen.sna.graph.mcmc;
021
022import org.apache.log4j.Logger;
023import org.matsim.contrib.socnetgen.sna.graph.Vertex;
024import org.matsim.contrib.socnetgen.sna.graph.matrix.AdjacencyMatrix;
025
026import java.util.Random;
027
028/**
029 * @author illenberger
030 *
031 */
032public abstract class AbstractGibbsSampler {
033
034        private static final Logger logger = Logger.getLogger(AbstractGibbsSampler.class);
035        
036        final protected Random random;
037        
038        private long logInterval = (long) 1E6;
039        
040        public AbstractGibbsSampler() {
041                random = new Random();
042        }
043        
044        public AbstractGibbsSampler(long seed) {
045                random = new Random(seed);
046        }
047        
048        public void setLogInterval(long interval) {
049                this.logInterval = interval;
050        }
051        
052        public <V extends Vertex> void sample(AdjacencyMatrix<V> y, EnsembleProbability p, SamplerListener<V> listener) {
053                long time = System.currentTimeMillis();
054                
055                int accept = 0;
056                long it = 0;
057                while(listener.beforeSampling(y, it)) {
058                        it++;
059                        if(step(y, p))
060                                accept++;
061                        
062                        if(it % logInterval == 0) {
063                                logger.info(String.format("[%1$s] Accepted %2$s of %3$s steps (ratio = %4$.6f).", it, accept, logInterval, accept/(double)logInterval));
064                                accept = 0;
065                        }
066                }
067                                
068                logger.info(String.format("Sampling done in %1$s s.", (System.currentTimeMillis() - time)/1000));
069        }
070        
071        abstract protected <V extends Vertex> boolean step(AdjacencyMatrix<V> m, EnsembleProbability p);
072}
