020package org.matsim.contrib.socnetgen.sna.graph.analysis;
021
022import org.matsim.contrib.socnetgen.sna.graph.*;
023import org.matsim.contrib.socnetgen.sna.graph.io.GraphMLWriter;
024import org.matsim.contrib.socnetgen.sna.graph.io.SparseGraphMLReader;
025
026import java.io.IOException;
027import java.util.HashSet;
028import java.util.Set;
029
030/**
031 * @author illenberger
032 *
033 */
034public class DegreeFilter implements GraphFilter<Graph> {
035
036        private GraphBuilder<Graph, Vertex, Edge> builder;
037        
038        private int threshold;
039        
040        private boolean lessthan;
041
042        public DegreeFilter(GraphBuilder<? extends Graph, ? extends Vertex, ? extends Edge> builder, int threshold) {
043                this.builder = (GraphBuilder<Graph, Vertex, Edge>) builder;
044                this.threshold = threshold;
045        }
046        
047        public void setLessthan(boolean flag) {
048                this.lessthan = flag;
049        }
050        
051        @Override
052        public Graph apply(Graph graph) {
053                Set<Vertex> remove = new HashSet<Vertex>();
054                
055                for(Vertex vertex : graph.getVertices()) {
056                        if(lessthan) {
057                                if(vertex.getNeighbours().size() < threshold)
058                                        remove.add(vertex);
059                        } else {
060                                if (vertex.getNeighbours().size() > threshold)
061                                        remove.add(vertex);
062                        }
063                }
064                
065                for(Vertex vertex : remove) {
066                        Set<Edge> edges = new HashSet<Edge>(vertex.getEdges());
067                        for(Edge edge : edges)
068                                builder.removeEdge(graph, edge);
069                        
070                        builder.removeVertex(graph, vertex);
071                }
072                
073                return graph;
074        }
075
076        public static void main(String args[]) throws IOException {
077                SparseGraphMLReader reader = new SparseGraphMLReader();
078                SparseGraph graph = reader.readGraph(args[0]);
079                
080                DegreeFilter filter = new DegreeFilter(new SparseGraphBuilder(), 3);
081                filter.setLessthan(true);
082                filter.apply(graph);
083                
084                GraphMLWriter writer = new GraphMLWriter();
085                writer.write(graph, args[1]);
086        }
087}
