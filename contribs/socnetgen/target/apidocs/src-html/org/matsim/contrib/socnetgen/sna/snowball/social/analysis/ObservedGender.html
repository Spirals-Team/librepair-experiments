020package org.matsim.contrib.socnetgen.sna.snowball.social.analysis;
021
022import org.matsim.contrib.socnetgen.sna.graph.social.SocialVertex;
023import org.matsim.contrib.socnetgen.sna.graph.social.analysis.Gender;
024import org.matsim.contrib.socnetgen.sna.graph.social.analysis.SocioMatrix;
025import org.matsim.contrib.socnetgen.sna.graph.social.analysis.SocioMatrixBuilder;
026import org.matsim.contrib.socnetgen.sna.snowball.analysis.SnowballPartitions;
027import org.matsim.contrib.socnetgen.sna.snowball.social.SocialSampledVertexDecorator;
028
029import java.util.Map;
030import java.util.Set;
031
032/**
033 * @author illenberger
034 *
035 */
036public class ObservedGender extends Gender {
037
038        @Override
039        public SocioMatrix<String> probaMatrix(Set<? extends SocialVertex> vertices) {
040                Map<SocialVertex, String> egos = super.values(SnowballPartitions.<SocialSampledVertexDecorator<?>>createSampledPartition((Set<SocialSampledVertexDecorator<?>>)vertices));
041                Map<SocialVertex, String> alters = super.values(vertices);
042                return SocioMatrixBuilder.probaMatrix(egos, alters);
043        }
044
045        @Override
046        public Map<SocialVertex, String> values(Set<? extends SocialVertex> vertices) {
047                return super.values(SnowballPartitions.<SocialSampledVertexDecorator<?>>createSampledPartition((Set<SocialSampledVertexDecorator<?>>)vertices, 0));
048        }
049}
