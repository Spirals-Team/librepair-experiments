024package org.matsim.contrib.socnetgen.sna.graph;
025
026import java.util.LinkedHashSet;
027import java.util.Set;
028
029
030/**
031 * Representation of an undirected and unweighted mathematical sparse graph.
032 * 
033 * @author illenberger
034 * 
035 */
036public class SparseGraph implements Graph {
037
038        private LinkedHashSet<SparseVertex> vertices;
039
040        private LinkedHashSet<SparseEdge> edges;
041
042        /**
043         * Creates an empty graph.
044         */
045        public SparseGraph() {
046                vertices = new LinkedHashSet<SparseVertex>();
047                edges = new LinkedHashSet<SparseEdge>();
048        }
049
050        /**
051         * Returns the set of vertices. It is discouraged to do modification to the
052         * returned set. To modify a graph use {@link SparseGraphBuilder}, otherwise
053         * graph consistency can not be guaranteed.
054         * 
055         * @see {@link Graph#getVertices()}
056         */
057        public Set<? extends SparseVertex> getVertices() {
058                return vertices;
059        }
060
061        /**
062         * Returns the set of edges. It is discouraged to do modification to the
063         * returned set. To modify a graph use {@link SparseGraphBuilder}, otherwise
064         * graph consistency can not be guaranteed.
065         * 
066         * @see {@link Graph#getEdges()}
067         */
068        public Set<? extends SparseEdge> getEdges() {
069                return edges;
070        }
071
072        /**
073         * Inserts a vertex into this graph if it is not already part of the graph.
074         * Do not use this call directly. Use
075         * {@link SparseGraphBuilder#addVertex(SparseGraph)}.
076         * 
077         * @param v
078         *            the vertex to be inserted.
079         * @return <tt>true</tt> if the vertex is inserted into this graph,
080         *         <tt>false</tt> otherwise.
081         */
082        boolean insertVertex(SparseVertex v) {
083                return vertices.add(v);
084        }
085
086        /**
087         * Inserts an edge into this graph if it is not already part of the graph.
088         * Do not use this call directly. Use
089         * {@link SparseGraphBuilder#addEdge(SparseGraph, SparseVertex, SparseVertex)}.
090         * 
091         * @param e
092         *            the edge to be inserted.
093         * @return <tt>true</tt> if the edge is inserted into this graph,
094         *         <tt>false</tt> otherwise.
095         */
096        boolean insertEdge(SparseEdge e) {
097                return edges.add(e);
098        }
099
100        /**
101         * Removes a vertex from the graph. Do not use this call directly. Use
102         * {@link SparseGraphBuilder#removeVertex(SparseGraph, SparseVertex)}.
103         * 
104         * @param v
105         *            the vertex to be removed from the graph.
106         * 
107         * @return <tt>true</tt> if the vertex has been removed from the graph,
108         *         <tt>false</tt> if <tt>v</tt> is not part of this graph.
109         */
110        boolean removeVertex(SparseVertex v) {
111                return vertices.remove(v);
112        }
113
114        /**
115         * Removes an edge from the graph. Do not use this call directly. Use
116         * {@link SparseGraphBuilder#removeEdge(SparseGraph, SparseEdge)}.
117         * 
118         * @param e
119         *            the edge to be removed.
120         * @return <tt>true</tt> if the edge has been removed from the graph,
121         *         <tt>false</tt> if <tt>e</tt> is not part of this graph.
122         */
123        boolean removeEdge(SparseEdge e) {              
124                return edges.remove(e);
125        }
126        
127        /**
128         * Optimizes the internal storage structure.
129         */
130        public void optimize() {
131                for (SparseVertex v : vertices)
132                        v.optimize();
133        }
134
135        /**
136         * Returns the edge connecting vertex <tt>v_i</tt> and <tt>v_j</tt>. If
137         * there are multiple edges connecting <tt>v_i</tt> and <tt>v_j</tt> it is
138         * not defined which edge is returned.
139         * 
140         * @param v_i
141         *            a vertex.
142         * @param v_j
143         *            a vertex.
144         * @return the edge connecting vertex <tt>v_i</tt> and <tt>v_j</tt>, or
145         *         <tt>null</tt> if <tt>v_i</tt> and <tt>v_j</tt> are not connected
146         *         with each other.
147         */
148        public SparseEdge getEdge(SparseVertex v_i, SparseVertex v_j) {
149                SparseEdge e = null;
150                int cnt = v_i.getEdges().size();
151                for(int i = 0; i < cnt; i++) {
152                        e = v_i.getEdges().get(i);
153                        if(e.getOpposite(v_i) == v_j) {
154                                return e;
155                        }
156                }
157                
158                return null;
159        }
160
161        /**
162         * Returns a short description for this graph.
163         * 
164         * @return a short description for this grpah.
165         */
166        @Override
167        public String toString() {
168                StringBuilder builder = new StringBuilder(60);
169                builder.append("SparseGraph: ");
170                builder.append(String.valueOf(vertices.size()));
171                builder.append(" vertices, ");
172                builder.append(String.valueOf(edges.size()));
173                builder.append(" edges");
174                return builder.toString();
175        }
176}
