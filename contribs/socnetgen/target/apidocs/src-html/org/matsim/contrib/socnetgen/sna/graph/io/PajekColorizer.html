024package org.matsim.contrib.socnetgen.sna.graph.io;
025
026import org.matsim.contrib.socnetgen.sna.graph.Edge;
027import org.matsim.contrib.socnetgen.sna.graph.Vertex;
028
029import java.io.*;
030import java.util.LinkedList;
031import java.util.List;
032
033
034/**
035 * @author illenberger
036 *
037 */
038public abstract class PajekColorizer<V extends Vertex, E extends Edge> {
039
040        private PajekColorTable colorTable;
041        
042        public PajekColorizer() {
043                colorTable = new PajekColorTable();
044        }
045        
046        public abstract String getVertexFillColor(V v);
047        
048        public abstract String getEdgeColor(E e);
049        
050        public String getColor(double value) {
051                float red = 1;
052                float green = 1;
053                float blue = 1;
054                
055                if(value < 0) {
056                        red = 0;
057                        green = 0;
058                        blue = 0;
059                } else if(value > 0 && value <= 0.25) {
060                        red = (float) (value/0.25);
061                        green = 1;
062                        blue = 0;
063                } else if (value > 0.25 && value <= 0.5) {
064                        red = 1;
065                        green = (float) (1 - (value-0.25)/0.25);
066                        blue = 0;
067                } else if (value > 0.5 && value <= 0.75) {
068                        red = 1;
069                        green = 0;
070                        blue = (float) ((value - 0.5)/0.25);
071                } else if (value > 0.75 && value <= 1) {
072                        red = 0;
073                        green = 0;
074                        blue = 1;
075                }
076                
077                return colorTable.getColor(red, green, blue);
078        }
079
080        private static class PajekColorTable {
081                
082                private static final String WSPACE = " ";
083                
084                private List<Color> colors = new LinkedList<Color>();
085                
086                public PajekColorTable() {
087                        try {
088                                InputStream stream = ClassLoader.getSystemResourceAsStream("org/matsim/contrib/socnetgen/sna/graph/io/PajekColorTable");
089                                BufferedReader reader = new BufferedReader(new InputStreamReader(stream));
090                                String line;
091                                String name = null;
092                                float red = 0;
093                                float green = 0;
094                                float blue = 0;
095                                int argument;
096                                while((line = reader.readLine()) != null) {
097                                        argument = 0;
098                                        for(String token : line.split(WSPACE)) {
099                                                if(token.length() > 0) {
100                                                        argument++;
101                                                        if(argument == 1) {
102                                                                name = token;
103                                                        } else if (argument == 2) {
104                                                                red = Float.parseFloat(token);
105                                                        } else if (argument == 3) {
106                                                                green = Float.parseFloat(token);
107                                                        } else if (argument == 4) {
108                                                                blue = Float.parseFloat(token);
109                                                                break;
110                                                        }
111                                                }
112                                        }
113        
114                                        Color c = new Color();
115                                        c.red = red;
116                                        c.green = green;
117                                        c.blue = blue;
118                                        c.name = name;
119                                        colors.add(c);
120                                }
121                                
122                                
123                        } catch (FileNotFoundException e) {
124                                e.printStackTrace();
125                        } catch (IOException e) {
126                                e.printStackTrace();
127                        }
128                }
129                
130                public String getColor(float r, float g, float b) {
131                        double delta_min = Double.MAX_VALUE;
132                        Color c = null;
133                        for(Color color : colors) {
134                                double delta_red = r - color.red;
135                                double delta_blue = b - color.blue;
136                                double delta_green = g - color.green;
137                                double delta = Math.sqrt(Math.pow(delta_red, 2) + Math.pow(delta_green, 2) + Math.pow(delta_blue, 2));
138                                if(delta < delta_min) {
139                                        delta_min = delta;
140                                        c = color;
141                                }
142                        }
143                        
144                        return c.name;
145                }
146                
147                private class Color {
148                        
149                        private String name;
150                        
151                        private float red;
152                        
153                        private float green;
154                        
155                        private float blue;
156                }
157        }
158
159}
