184        public void run(AdjacencyMatrix<?> y) {
185                int[] sources = new int[y.getVertexCount()];
186                int[] targets = new int[y.getVertexCount()];
187                
188                for(int i = 0; i < sources.length; i++) {
189                        sources[i] = i;
190                        targets[i] = i;
191                }
192                
193                run(y, sources, targets);
194        }

196        /**
197         * Builds the complete shortest path tree for all vertices and
198         * simultaneously calculates closeness, betweenness, radius and diameter.
199         * 
200         * @param y an adjacency matrix.
201         */
202        public void run(AdjacencyMatrix<?> y, int[] sources, int[] targets) {
203                int n = y.getVertexCount();
204                vertexCloseness = new double[n];
205                Arrays.fill(vertexCloseness, Double.POSITIVE_INFINITY);
206                vertexBetweenness = new double[n];
207                edgeBetweenness = new TIntDoubleHashMap[n];
208                diameter = 0;
209                radius = Integer.MAX_VALUE;
210                /*
211                 * create threads
212                 */
213                if(dijkstraFactory == null)
214                        dijkstraFactory = new DijkstraFactory();
215                
216                List<CentralityThread> threads = new ArrayList<CentralityThread>();
217                int size = (int) Math.floor(sources.length / (double) numThreads);
218                int i_start = 0;
219                int i_stop = size;
220                for (int i = 0; i < numThreads - 1; i++) {
221                        int[] subSources = Arrays.copyOfRange(sources, i_start, i_stop);
222                        threads.add(new CentralityThread(y, subSources, targets, dijkstraFactory, calcBetweenness));
223                        i_start = i_stop;
224                        i_stop += size;
225                }
226                int[] subSources = Arrays.copyOfRange(sources, i_start, sources.length);
227                threads.add(new CentralityThread(y, subSources, targets, dijkstraFactory, calcBetweenness));
228                /*
229                 * start threads
230                 */
231                for (CentralityThread thread : threads) {
232                        thread.start();
233                }
234                /*
235                 * wait for threads
236                 */
237                for (CentralityThread thread : threads) {
238                        try {
239                                thread.join();
240                        } catch (InterruptedException e) {
241                                e.printStackTrace();
242                        }
243                }
244                /*
245                 * merge results of threads
246                 */
247                apl = new DescriptiveStatistics();
248                double aplSum = 0;
249                double cnt = 0;
250                for (CentralityThread thread : threads) {
251
252                        if (calcAPLDistribution) {
253                                for (int i = 0; i < thread.pathLengths.size(); i++) {
254                                        apl.addValue(thread.pathLengths.get(i));
255                                }
256                        } else {
257                                for (int i = 0; i < thread.pathLengths.size(); i++) {
258                                        aplSum += thread.pathLengths.get(i);
259                                }
260                        }
261
262                        cnt++;
263                        
264                        for (int i = 0; i < n; i++) {
265                                /*
266                                 * if this thread did not calculate the closeness of i it
267                                 * returns infinity
268                                 */
269                                vertexCloseness[i] = Math.min(vertexCloseness[i], thread.vertexCloseness[i]);
270                                /*
271                                 * merge vertex betweenness values
272                                 */
273                                vertexBetweenness[i] += thread.vertexBetweenness[i];
274                                /*
275                                 * merge edge betweenness values
276                                 */
277                                if (thread.edgeBetweenness[i] != null) {
278                                        TIntDoubleIterator it = thread.edgeBetweenness[i].iterator();
279                                        for (int j = 0; j < thread.edgeBetweenness[i].size(); j++) {
280                                                it.advance();
281                                                /*
282                                                 * since we have an undirected graph, merge betweenness
283                                                 * for both edges
284                                                 */
285                                                if (edgeBetweenness[i] == null)
286                                                        edgeBetweenness[i] = new TIntDoubleHashMap();
287                                                edgeBetweenness[i].adjustOrPutValue(it.key(), it.value(), it.value());
288
289                                                if (edgeBetweenness[it.key()] == null)
290                                                        edgeBetweenness[it.key()] = new TIntDoubleHashMap();
291                                                edgeBetweenness[it.key()].adjustOrPutValue(i, it.value(), it.value());
292                                        }
293                                }
294                                /*
295                                 * get diameter and radius
296                                 */
297                                diameter = Math.max(diameter, thread.diameter);
298                                radius = Math.min(radius, thread.radius);
299                        }
300                }
301                /*
302                 * calculate mean values
303                 */
304                meanVertexCloseness = StatUtils.mean(vertexCloseness);
305                
306                if(!calcAPLDistribution)
307                        apl.addValue(aplSum/cnt);
308                
309                double sum = 0;
310                for (int i = 0; i < y.getVertexCount(); i++)
311                        sum += vertexBetweenness[i];
312                meanVertexBetweenness = sum / (double) y.getVertexCount();
313
314                sum = 0;
315                double count = 0;
316                for (int i = 0; i < n; i++) {
317                        if (edgeBetweenness[i] != null) {
318                                TIntDoubleIterator it = edgeBetweenness[i].iterator();
319                                for (int k = 0; k < edgeBetweenness[i].size(); k++) {
320                                        it.advance();
321                                        sum += it.value();
322                                        count++;
323                                }
324                        }
325                }
326                meanEdgeBetweenness = sum / count;
327        }

329        private static class CentralityThread extends Thread {
330
331                private Dijkstra dijkstra;
332
333                private int[] sources;
334
335                private int[] targets;
336                
337                private boolean calcBetweenness;
338
339                private TIntDoubleHashMap[] edgeBetweenness;
340
341                private double vertexBetweenness[];
342
343                private double vertexCloseness[];
344
345                private int n;
346
347                private static int counter = 0;
348
349                private int diameter;
350
351                private int radius;
352                
353//              private int totalPathLength;
354//              
355//              private int totalPathCount;
356                
357                private TIntArrayList pathLengths;
358                
359                private final Logger logger = Logger.getLogger(CentralityThread.class);
360
361                public CentralityThread(AdjacencyMatrix<?> y, int[] sources, int[] targets, DijkstraFactory dijkstraFactory, boolean calcBetweenness) {
362                        dijkstra = dijkstraFactory.newDijkstra(y); 
363                        this.sources = sources;
364                        this.targets = targets;
365                        this.calcBetweenness = calcBetweenness;
366                        n = y.getVertexCount();
367                        diameter = 0;
368                        radius = Integer.MAX_VALUE;
369//                      pathLengths = new TIntArrayList(sources.length * targets.length);
370                        pathLengths = new TIntArrayList(sources.length + targets.length);
371                        counter = 0;
372                }
373
374                @Override
375                public void run() {
376                        PathAnalyzer pathAnalyzer = new PathAnalyzer();
377                        PathExtractor pathExtractor = new PathExtractor();
378                        /*
379                         * initialize the closeness array with infinity
380                         */
381                        vertexCloseness = new double[n];
382                        Arrays.fill(vertexCloseness, Double.POSITIVE_INFINITY);
383                        /*
384                         * initialize the betweenness arrays with zero
385                         */
386                        vertexBetweenness = new double[n];
387                        edgeBetweenness = new TIntDoubleHashMap[n];
388                        /*
389                         * time measuring
390                         */
391                        long dkTime = 0;
392                        long cTime = 0;
393
394                        for (int idx = 0; idx < sources.length; idx++) {
395                                int i = sources[idx];
396                                /*
397                                 * run the Dijkstra to all nodes
398                                 */
399                                long time = System.currentTimeMillis();
400//                              TIntArrayList reachable = dijkstra.run(i, -1);
401                                dijkstra.run(i, -1);
402
403                                dkTime += System.currentTimeMillis() - time;
404                                /*
405                                 * extract the paths
406                                 */
407                                time = System.currentTimeMillis();
408                                int pathLengthSum = 0;
409                                int eccentricity = 0;
410                                /*
411                                 * iterate over all reachable nodes
412                                 */
413                                int reachedTargets = 0;
414//                              for (int k = 0; k < reachable.size(); k++) {
415                                for (int k = 0; k < targets.length; k++) {
416                                        int j = targets[k];
417                                        /*
418                                         * determine the length and number of paths
419                                         */
420                                        pathAnalyzer.run(dijkstra.getSpanningTree(), j);
421                                        int pathLength = pathAnalyzer.pathLength;
422                                        int pathCount = pathAnalyzer.pathCount;
423                                        if (pathLength > 0) {
424                                                reachedTargets++;
425                                                pathLengthSum += pathLength;
426                                                pathLengths.add(pathLength);
427                                                eccentricity = Math.max(eccentricity, pathLength);
428                                                
429                                                if (calcBetweenness) {
430                                                        /*
431                                                         * extract all paths
432                                                         */
433                                                        int[][] matrix = new int[pathLength][pathCount];
434                                                        pathExtractor.run(dijkstra.getSpanningTree(), matrix, j);
435                                                        /*
436                                                         * increase betweenness values for each passed
437                                                         * vertex and edge
438                                                         */
439                                                        for (int col = 0; col < pathCount; col++) {
440                                                                int prevVertex = i;
441                                                                /*
442                                                                 * reverse the order in case we have some day
443                                                                 * directed graphs...
444                                                                 */
445                                                                for (int row = pathLength - 1; row > -1; row--) {
446                                                                        int vertex = matrix[row][col];
447                                                                        vertexBetweenness[vertex]++;
448
449                                                                        if (edgeBetweenness[prevVertex] == null) {
450                                                                                edgeBetweenness[prevVertex] = new TIntDoubleHashMap();
451                                                                        }
452                                                                        edgeBetweenness[prevVertex].adjustOrPutValue(vertex, 1.0, 1.0);
453
454                                                                        prevVertex = vertex;
455                                                                }
456                                                                /*
457                                                                 * decrease betweenness of target node
458                                                                 */
459                                                                vertexBetweenness[j]--;
460                                                        }
461                                                }
462                                        }
463                                }
464
465                                if (reachedTargets > 0) {
466                                        vertexCloseness[i] = pathLengthSum / (double) reachedTargets;
467//                                      totalPathLength += pathLengthSum;
468//                                      totalPathCount += reachedTargets;
469                                }
470
471                                diameter = Math.max(diameter, eccentricity);
472                                radius = Math.min(radius, eccentricity);
473
474                                cTime += System.currentTimeMillis() - time;
475                                counter++;
476                                if (counter % 1000 == 0) {
477                                        logger.info(String.format("Procesed %1$s vertices, dijkstra took %2$s ms, misc took %3$s ms.",
478                                                        counter, dkTime, cTime));
479                                        dkTime = 0;
480                                        cTime = 0;
481                                }
482                        }
483                }
484        }

486        private static class PathAnalyzer {
487
488                private TIntArrayList[] spanningTree;
489
490                private int pathLength;
491
492                private int pathCount;
493
494                public void run(TIntArrayList[] spanningTree, int i) {
495                        this.spanningTree = spanningTree;
496                        pathLength = 0;
497                        pathCount = 1;
498                        step(i, 1);
499                }
500
501                private void step(int i, int depth) {
502                        if (spanningTree[i] != null && spanningTree[i].size() > 0) {
503                                pathLength = Math.max(depth, pathLength);
504
505                                if (spanningTree[i].size() > 1)
506                                        pathCount += spanningTree[i].size() - 1;
507
508                                for (int k = 0; k < spanningTree[i].size(); k++) {
509                                        step(spanningTree[i].get(k), depth + 1);
510                                }
511                        }
512                }
513        }
514
515        private static class PathExtractor {
516
517                private TIntArrayList[] spanningTree;
518
519                private int[][] matrix;
520
521                private int column;
522
523                public void run(TIntArrayList[] spanningTree, int[][] matrix, int i) {
524                        this.spanningTree = spanningTree;
525                        this.matrix = matrix;
526                        column = 0;
527                        step(i, 0);
528                }
529
530                private void step(int i, int row) {
531                        if (spanningTree[i] != null && spanningTree[i].size() > 0) {
532                                matrix[row][column] = i;
533
534                                step(spanningTree[i].get(0), row + 1);
535
536                                if (spanningTree[i].size() > 1) {
537                                        for (int k = 1; k < spanningTree[i].size(); k++) {
538                                                /*
539                                                 * copy the upper vertex indices from left to right
540                                                 */
541                                                for (int l = 0; l < row + 1; l++) {
542                                                        matrix[l][column + 1] = matrix[l][column];
543                                                }
544                                                column++;
545                                                step(spanningTree[i].get(k), row + 1);
546                                        }
547                                }
548                        }
549                }
