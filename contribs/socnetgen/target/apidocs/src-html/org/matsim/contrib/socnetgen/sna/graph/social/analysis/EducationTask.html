020package org.matsim.contrib.socnetgen.sna.graph.social.analysis;
021
022import gnu.trove.map.hash.TObjectDoubleHashMap;
023
024import java.io.IOException;
025import java.util.Map;
026
027import org.apache.commons.math.stat.descriptive.DescriptiveStatistics;
028import org.matsim.contrib.common.stats.StatsWriter;
029import org.matsim.contrib.socnetgen.sna.graph.Graph;
030import org.matsim.contrib.socnetgen.sna.graph.analysis.ModuleAnalyzerTask;
031import org.matsim.contrib.socnetgen.sna.graph.social.SocialGraph;
032import org.matsim.contrib.socnetgen.sna.graph.social.SocialVertex;
033
034/**
035 * @author illenberger
036 * 
037 */
038public class EducationTask extends ModuleAnalyzerTask<Education> {
039
040        public EducationTask() {
041                setModule(Education.getInstance());
042        }
043        
044        public EducationTask(Education module) {
045                setModule(module);
046        }
047        
048        @Override
049        public void analyze(Graph g, Map<String, DescriptiveStatistics> statsMap) {
050                SocialGraph graph = (SocialGraph) g;
051                
052                if(module instanceof EducationCategorized) {
053                        double r = ((EducationCategorized)module).correlation(graph.getEdges());
054                        DescriptiveStatistics d = singleValueStats("r_edu", r, statsMap);
055                        printStats(d, "r_edu");
056                }
057                
058                if (getOutputDirectory() != null) {
059                        try {
060                                
061                                
062                                Map<SocialVertex, String> values = module.values(graph.getVertices());
063                                TObjectDoubleHashMap<String> hist = LinguisticHistogram.create(values.values());
064                                
065                                StatsWriter.writeLabeledHistogram(hist, "education", "n", String.format("%1$s/education.txt", getOutputDirectory()));
066                                
067                                SocioMatrix<String> m = module.countsMatrix(graph.getVertices());
068                                m.toFile(String.format("%1$s/education.countsMatrix.txt", getOutputDirectory()));
069                                
070                                SocioMatrixBuilder.normalizeTotalSum(m);
071                                m.toFile(String.format("%1$s/education.countsMatrix.normTotal.txt", getOutputDirectory()));
072                                
073                                m = module.countsMatrix(graph.getVertices());
074                                SocioMatrixBuilder.normalizeRowSum(m);
075                                m.toFile(String.format("%1$s/education.countsMatrix.normRow.txt", getOutputDirectory()));
076                                
077                                
078                                m = module.probaMatrix(graph.getVertices());
079                                m.toFile(String.format("%1$s/education.probaMatrix.txt", getOutputDirectory()));
080                        } catch (IOException e) {
081                                e.printStackTrace();
082                        }
083                }
084
085        }
086
087}
