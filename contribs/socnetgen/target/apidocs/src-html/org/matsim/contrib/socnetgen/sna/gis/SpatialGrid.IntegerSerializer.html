026import com.vividsolutions.jts.geom.Point;
027import org.apache.log4j.Logger;
028import org.matsim.core.utils.io.MatsimXmlParser;

036/**
037 * @author illenberger
038 *
039 */
040public class SpatialGrid<T> {
041        
042        private static final Logger logger = Logger.getLogger(SpatialGrid.class);
043
044        private Object[][] matrix;
045        
046        private final double minX;
047        
048        private final double minY;
049        
050        private final double maxX;
051        
052        private final double maxY;
053        
054        private final double resolution;
055        
056        public SpatialGrid(double xmin, double ymin, double xmax, double ymax, double resolution) {
057                minX = xmin;
058                minY = ymin;
059                maxX = xmax;
060                maxY = ymax;
061                this.resolution = resolution;
062                int numXBins = (int)Math.ceil((maxX - minX) / resolution) + 1;
063                int numYBins = (int)Math.ceil((maxY - minY) / resolution) + 1;
064                matrix = new Object[numYBins][numXBins];
065        }
066        
067        public SpatialGrid(SpatialGrid<?> grid) {
068                this(grid.getXmin(), grid.getYmin(), grid.getXmax(), grid.getYmax(), grid.getResolution());
069        }
070        
071        public double getXmin() {
072                return minX;
073        }
074        
075        public double getYmin() {
076                return minY;
077        }
078        
079        public double getXmax() {
080                return maxX;
081        }
082        
083        public double getYmax() {
084                return maxY;
085        }
086        
087        public double getResolution() {
088                return resolution;
089        }
090        
091        public int getNumRows() {
092                return matrix.length;
093        }
094        
095        public int getNumCols(int row) {
096                return matrix[row].length;
097        }
098        
099        @SuppressWarnings("unchecked")
100        public T getValue(Point point) {
101                if(isInBounds(point))
102                        return (T)matrix[getRow(point.getY())][getColumn(point.getX())];
103                else
104                        return null;
105        }
106        
107        public boolean setValue(T value, Point point) {
108                if(isInBounds(point)) {
109                        matrix[getRow(point.getY())][getColumn(point.getX())] = value;
110                        return true;
111                } else
112                        return false;
113        }
114        
115        public boolean isInBounds(Point point) {
116                return point.getX() >= minX && point.getX() <= maxX &&
117                                point.getY() >= minY && point.getY() <= maxY;
118        }
119        
120        @SuppressWarnings("unchecked")
121        public T getValue(int row, int col) {
122                return (T)matrix[row][col];
123        }
124        
125        public boolean setValue(int row, int col, T value) {
126                if(row < matrix.length) {
127                        if(col < matrix[row].length) {
128                                matrix[row][col] = value;
129                                return true;
130                        } else
131                                return false;
132                } else
133                        return false;
134        }
135        
136        public int getRow(double yCoord) {
137                return matrix.length - 1 - (int)Math.floor((yCoord - minY) / resolution);
138        }
139        
140        public int getColumn(double xCoord) {
141                return (int)Math.floor((xCoord - minX) / resolution);
142        }
143        
144        @SuppressWarnings("unchecked")
145        public void toFile(String filename) {
146                toFile(filename, (ObjectSerializer<T>) new DoubleSerializer());
147        }
148        
149        public void toFile(String filename, ObjectSerializer<T> serializer) {
150                GridXMLWriter<T> writer = new GridXMLWriter<T>(serializer);
151                try {
152                        writer.write(this, filename);
153                } catch (IOException e) {
154                        logger.fatal("IOException occured!", e);
155                }
156        }
157        
158        public static SpatialGrid<Double> readFromFile(String filename) {
159                return readFromFile(filename, new DoubleSerializer());
160        }
161        
162        public static <V> SpatialGrid<V> readFromFile(String filename, ObjectSerializer<V> serializer) {
163                GridXMLParser<V> parser = new GridXMLParser<V>(serializer);
164                parser.setValidating(false);
165                try {
166                        parser.readFile(filename);
167                        return parser.grid;
168                } catch (Exception e) {
169                        logger.fatal("Exception during parsing occured!", e);
170                        e.printStackTrace();
171                        return null;
172                }
173        }
174        
175        private static class GridXMLParser<V> extends MatsimXmlParser {
176
177                public static final String GRID_TAG = "grid";
178                
179                public static final String CELL_TAG = "cell";
180                
181                public static final String XMIN_TAG = "xmin";
182                
183                public static final String XMAX_TAG = "xmax";
184                
185                public static final String YMIN_TAG = "ymin";
186                
187                public static final String YMAX_TAG = "ymax";
188                
189                public static final String RES_TAG = "resolution";
190                
191                public static final String ROW_TAG = "row";
192                
193                public static final String COL_TAG = "col";
194                
195                public static final String VALUE_TAG = "value";
196                
197                private SpatialGrid<V> grid;
198                
199                private ObjectSerializer<V> serializer;
200
201                public GridXMLParser(ObjectSerializer<V> serializer) {
202                        super();
203                        this.serializer = serializer;
204                }
205                
206                @Override
207                public void endTag(String name, String content, Stack<String> context) {
208                }
209
210                @Override
211                public void startTag(String name, Attributes atts, Stack<String> context) {
212                        if(CELL_TAG.equalsIgnoreCase(name)) {
213                                int row = (int)getDouble(atts, ROW_TAG);
214                                int col = (int)getDouble(atts, COL_TAG);
215                                String data = atts.getValue(VALUE_TAG);
216                                if(data == null)
217                                        throw new IllegalArgumentException(VALUE_TAG + " must be specified");
218                                else {
219                                        if(!grid.setValue(row, col, serializer.decode(data)))
220                                                logger.warn(String.format("Out of bounds! (row=%1$s, col=%2$s)",row, col));
221                                }
222                        } else if(GRID_TAG.equalsIgnoreCase(name)) {
223                                if(grid != null) {
224                                        logger.fatal("Only one grid per file allowed!");
225                                        throw new UnsupportedOperationException("Only one grid per file allowed!");
226                                } else {
227                                        double xmin = getDouble(atts, XMIN_TAG);
228                                        double xmax = getDouble(atts, XMAX_TAG);
229                                        double ymin = getDouble(atts, YMIN_TAG);
230                                        double ymax = getDouble(atts, YMAX_TAG);
231                                        double resolution = getDouble(atts, RES_TAG);
232                                        grid = new SpatialGrid<V>(xmin, ymin, xmax, ymax, resolution);
233                                }
234                        }
235                }
236                
237                private double getDouble(Attributes atts, String qName) {
238                        String val = atts.getValue(qName);
239                        if(val == null)
240                                throw new IllegalArgumentException(qName + " must be specified!");
241                        else
242                                return Double.parseDouble(val);
243                }
244        }
245        
246        private static class GridXMLWriter<V> extends MatsimXmlWriter {
247                
248                private ObjectSerializer<V> serializer;
249                
250                public GridXMLWriter(ObjectSerializer<V> serializer) {
251                        this.serializer = serializer;
252                }
253                
254                @SuppressWarnings("unchecked")
255                public void write(SpatialGrid<V> grid, String filename) throws IOException {
256                        openFile(filename);
257                        writeXmlHead();
258                        
259                        writer.write("<");
260                        writer.write(GridXMLParser.GRID_TAG);
261                        writer.write(" ");
262                        writer.write(makeAttribute(GridXMLParser.XMIN_TAG, String.valueOf(grid.minX)));
263                        writer.write(makeAttribute(GridXMLParser.YMIN_TAG, String.valueOf(grid.minY)));
264                        writer.write(makeAttribute(GridXMLParser.XMAX_TAG, String.valueOf(grid.maxX)));
265                        writer.write(makeAttribute(GridXMLParser.YMAX_TAG, String.valueOf(grid.maxY)));
266                        writer.write(makeAttribute(GridXMLParser.RES_TAG, String.valueOf(grid.resolution)));
267                        writer.write(">");
268                        writer.write(NL);
269                        
270                        for(int row = 0; row < grid.matrix.length; row++) {
271                                for(int col = 0; col < grid.matrix[row].length; col++) {
272                                        writer.write("\t<");
273                                        writer.write(GridXMLParser.CELL_TAG);
274                                        writer.write(" ");
275                                        writer.write(makeAttribute(GridXMLParser.ROW_TAG, String.valueOf(row)));
276                                        writer.write(makeAttribute(GridXMLParser.COL_TAG, String.valueOf(col)));
277                                        writer.write(makeAttribute(GridXMLParser.VALUE_TAG, serializer.encode((V) grid.matrix[row][col])));
278                                        writer.write("/>");
279                                        writer.write(NL);
280                                }
281                        }
282                        
283                        writer.write("</");
284                        writer.write(GridXMLParser.GRID_TAG);
285                        writer.write(">");
286                        writer.close();
287                }
288                
289                private String makeAttribute(String qName, String value) {
290                        StringBuilder builder = new StringBuilder(25);
291                        builder.append(qName);
292                        builder.append("=\"");
293                        builder.append(value);
294                        builder.append("\" ");
295                        return builder.toString();
296                }
297        }
298        
299        public static interface ObjectSerializer<T> {
300
301                public String encode(T object);
302                
303                public T decode(String data);
304        }
305        
306        public static class DoubleSerializer implements ObjectSerializer<Double> {
307
308                public String encode(Double object) {
309                        if(object == null)
310                                return "0";
311                        else
312                                return String.valueOf(object);
313                }
314
315                public Double decode(String data) {
316                        return new Double(data);
317                }
318
319        }
320        
321        public static class IntegerSerializer implements ObjectSerializer<Integer> {
322
323                public String encode(Integer object) {
324                        if(object == null)
325                                return "0";
326                        else
327                                return String.valueOf(object);
328                }
329
330                public Integer decode(String data) {            
331                        return Integer.parseInt(data);
332                }
333
334        }
