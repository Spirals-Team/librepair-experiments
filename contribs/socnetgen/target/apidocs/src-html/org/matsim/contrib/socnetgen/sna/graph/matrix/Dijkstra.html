020package org.matsim.contrib.socnetgen.sna.graph.matrix;
021
022import gnu.trove.list.array.TIntArrayList;
023
024
025/**
026 * Implementation of the Dijkstra best path algorithm. This algorithm runs on an
027 * adjacency matrix, i.e., the graph is treated as undirected and unweighted.
028 * 
029 * @author illenberger
030 * 
031 */
032public class Dijkstra extends DijkstraEngine {
033
034        private TIntArrayList[] spanningTree;
035
036        /**
037         * Creates a new Dijkstra object with adjacency matrix <tt>y</tt>.
038         * 
039         * @param y
040         *            an adjacency matrix.
041         */
042        public Dijkstra(AdjacencyMatrix<?> y) {
043                super(y);
044        }
045
046        /**
047         * Runs the Dijkstra algorithm from <tt>source</tt> to <tt>target</tt> and
048         * returns all vertices that could be reached from <tt>source</tt>. If
049         * <tt>target</tt> is not reachable or denotes a non-existing vertex index
050         * (e.g. -1) the algorithm runs over the complete connected component. To
051         * obtain a concrete path use {@link #getPath(int, int)}.
052         * 
053         * @param source
054         *            the source vertex index.
055         * @param target
056         *            the target vertex index, or <tt>-1</tt> to build up the
057         *            complete spanning tree.
058         * @return a list of all reachable vertices.
059         */
060        public TIntArrayList run(int source, int target) {
061                spanningTree = new TIntArrayList[y.getVertexCount()];
062                return super.run(source, target);
063        }
064
065        @Override
066        protected void foundCheaper(int i, int j, double cost) {
067                TIntArrayList predecessors = new TIntArrayList();
068                spanningTree[j] = predecessors;
069                predecessors.add(i);
070                
071        }
072
073        
074        @Override
075        protected void foundEqual(int i, int j, double cost) {
076                TIntArrayList predecessors = spanningTree[j];
077                if (predecessors == null) {
078                        predecessors = new TIntArrayList();
079                        spanningTree[j] = predecessors;
080                }
081
082                predecessors.add(i);
083        }
084
085        /**
086         * Returns a shortest path from vertex <tt>i</tt> to <tt>j</tt>, or
087         * <tt>null</tt> if there is no path from <tt>i</tt> to <tt>j</tt>. If there
088         * are multiple shortest paths there is no rule which path is returned.
089         * 
090         * @param i
091         *            the source vertex index.
092         * @param j
093         *            the target vertex index.
094         * @return a shortest path from vertex <tt>i</tt> to <tt>j</tt>, or
095         *         <tt>null</tt> if there is no path from <tt>i</tt> to <tt>j</tt>.
096         *         The path contains the indices of the vertices excluding the
097         *         source vertex.
098         */
099        public TIntArrayList getPath(int i, int j) {
100                TIntArrayList path = new TIntArrayList();
101
102                if (i == j)
103                        return path;
104
105                if (spanningTree[j] == null || spanningTree[j].isEmpty())
106                        return null;
107
108                while (j != i) {
109                        path.add(j);
110                        if (spanningTree[j].size() > 0)
111                                j = spanningTree[j].get(0);
112                        else
113                                return null;
114                }
115                path.reverse();
116
117                return path;
118        }
119
120        /**
121         * Returns the spanning tree. Each element in the array contains a list with
122         * the vertex's predecessors (array indices are vertex indices). A
123         * <tt>null</tt> element or an empty list indicates that this vertex has no
124         * predecessors.
125         * 
126         * @return the spanning tree.
127         */
128        public TIntArrayList[] getSpanningTree() {
129                return spanningTree;
130        }
131}
