020package org.matsim.contrib.socnetgen.sna.graph.spatial.analysis;
021
022import gnu.trove.iterator.TObjectDoubleIterator;
023import gnu.trove.map.hash.TObjectDoubleHashMap;
024
025import java.util.Set;
026
027import org.matsim.contrib.common.util.ProgressLogger;
028import org.matsim.contrib.socnetgen.sna.gis.SpatialCostFunction;
029import org.matsim.contrib.socnetgen.sna.graph.spatial.SpatialVertex;
030import org.matsim.contrib.socnetgen.sna.math.Distribution;
031
032import com.vividsolutions.jts.geom.Point;
033
034/**
035 * @author illenberger
036 *
037 */
038public class LogAccessibility {
039        
040        private static LogAccessibility instance;
041        
042        public static LogAccessibility getInstance() {
043                if(instance == null)
044                        instance = new LogAccessibility();
045                return instance;
046        }
047        
048        public Distribution distribution(Set<? extends SpatialVertex> vertices, SpatialCostFunction costFunction, Set<Point> opportunities) {
049                TObjectDoubleHashMap<SpatialVertex> values = values(vertices, costFunction, opportunities);
050                TObjectDoubleIterator<SpatialVertex> it = values.iterator();
051                Distribution distr = new Distribution();
052                for(int i = 0; i < values.size(); i++) {
053                        it.advance();
054                        distr.add(it.value());
055                }
056                
057                return distr;
058        }
059
060        @SuppressWarnings("unchecked")
061        public TObjectDoubleHashMap<SpatialVertex> values(Set<? extends SpatialVertex> vertices, SpatialCostFunction costFunction, Set<Point> opportunities) {
062                Set<SpatialVertex> spatialVertices = (Set<SpatialVertex>) vertices;
063                TObjectDoubleHashMap<SpatialVertex> values = new TObjectDoubleHashMap<SpatialVertex>(spatialVertices.size());
064
065                ProgressLogger.init(spatialVertices.size(), 1, 5);
066                for (SpatialVertex vertex : spatialVertices) {
067                        if (vertex.getPoint() != null) {
068                                double sum = 0;
069                                for (Point point : opportunities) {
070                                        if (point != null) {
071                                                double c = costFunction.costs(vertex.getPoint(), point);
072                                                sum += Math.exp(-c);
073                                        }
074                                }
075                                values.put(vertex, Math.log(sum));
076                                ProgressLogger.step();
077                        }
078                }
079
080                return values;
081        }
082}
