020package org.matsim.contrib.socnetgen.sna.gis;
021
022import com.vividsolutions.jts.geom.Coordinate;
023import com.vividsolutions.jts.geom.Envelope;
024import com.vividsolutions.jts.geom.GeometryFactory;
025import com.vividsolutions.jts.geom.Point;
026import org.matsim.contrib.common.gis.DistanceCalculator;
027
028import java.util.ArrayList;
029import java.util.Collection;
030import java.util.List;
031import java.util.Set;
032
033/**
034 * @author illenberger
035 * 
036 */
037public class PointUtils {
038
039        private static final GeometryFactory geometryFactory = new GeometryFactory();
040
041        public static Envelope envelope(Collection<Point> points) {
042                double xmin = Double.MAX_VALUE;
043                double xmax = Double.MIN_VALUE;
044                double ymin = Double.MAX_VALUE;
045                double ymax = Double.MIN_VALUE;
046
047                for (Point point : points) {
048                        xmin = Math.min(xmin, point.getX());
049                        xmax = Math.max(xmax, point.getX());
050                        ymin = Math.min(ymin, point.getY());
051                        ymax = Math.max(ymax, point.getY());
052                }
053
054                return new Envelope(xmin, xmax, ymin, ymax);
055        }
056
057        public static Point centerOfMass(Set<Point> points) {
058                double xsum = 0;
059                double ysum = 0;
060
061                for (Point point : points) {
062                        xsum += point.getX();
063                        ysum += point.getY();
064                }
065
066                double n = points.size();
067                return geometryFactory.createPoint(new Coordinate(xsum / n, ysum / n));
068        }
069
070        public static double avrDistance(Set<Point> points, DistanceCalculator calculator) {
071                List<Point> pointList = new ArrayList<Point>(points);
072
073                if (points.size() < 2)
074                        return Double.NaN;
075
076                int n = pointList.size();
077                double sum = 0;
078
079                for (int i = 0; i < n; i++) {
080                        for (int j = i + 1; j < n; j++) {
081                                sum += calculator.distance(pointList.get(i), pointList.get(j));
082                        }
083                }
084
085                return 2 * sum / (double) n * (n - 1);
086        }
087}
