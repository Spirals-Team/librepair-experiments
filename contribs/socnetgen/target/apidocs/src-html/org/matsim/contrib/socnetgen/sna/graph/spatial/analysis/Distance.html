020package org.matsim.contrib.socnetgen.sna.graph.spatial.analysis;
021
022import gnu.trove.list.array.TDoubleArrayList;
023import gnu.trove.map.hash.TObjectDoubleHashMap;
024
025import java.util.HashSet;
026import java.util.Set;
027
028import org.apache.commons.math.stat.StatUtils;
029import org.apache.commons.math.stat.descriptive.DescriptiveStatistics;
030import org.matsim.contrib.socnetgen.sna.graph.spatial.SpatialEdge;
031import org.matsim.contrib.socnetgen.sna.graph.spatial.SpatialVertex;
032import org.matsim.contrib.socnetgen.sna.math.Distribution;
033
034
035/**
036 * @author illenberger
037 *
038 */
039public class Distance {
040
041        private static Distance instance;
042        
043        public static Distance getInstance() {
044                if(instance == null)
045                        instance = new Distance();
046                return instance;
047        }
048        
049        public DescriptiveStatistics statistics(Set<? extends SpatialVertex> vertices) {
050                DescriptiveStatistics stats = new DescriptiveStatistics();
051                for(double val : distribution(vertices).getValues()) {
052                        stats.addValue(val);
053                }
054                return stats;
055        }
056        
057        public Distribution distribution(Set<? extends SpatialVertex> vertices) {
058                Distribution distribution = new Distribution();
059                
060                Set<SpatialEdge> touched = new HashSet<SpatialEdge>();
061                for(SpatialVertex v : vertices) {
062                        for(int i = 0; i < v.getEdges().size(); i++) {
063                                if(touched.add(v.getEdges().get(i))) {
064                                        double d = v.getEdges().get(i).length();
065                                        if(!Double.isNaN(d))
066                                                distribution.add(d);
067                                }
068                        }
069                }
070                
071                return distribution;
072        }
073        
074        public Distribution vertexAccumulatedDistribution(Set<? extends SpatialVertex> vertices) {
075                Distribution distribution = new Distribution();
076                
077                for(SpatialVertex v_i : vertices) {
078                        double sum = 0;
079                        for(SpatialEdge e : v_i.getEdges()) {
080                                double d = e.length();
081                                if(!Double.isNaN(d))
082                                        sum += e.length();
083                        }
084                        distribution.add(sum);
085                }
086                
087                return distribution;
088        }
089        
090        public TObjectDoubleHashMap<SpatialVertex> vertexMean(Set<? extends SpatialVertex> vertices) {
091                TObjectDoubleHashMap<SpatialVertex> values = new TObjectDoubleHashMap<SpatialVertex>();
092                
093                for(SpatialVertex vertex : vertices) {
094                        double sum = 0;
095                        int cnt = 0;
096                        for(SpatialEdge e : vertex.getEdges()) {
097                                double d = e.length();
098                                if(!Double.isNaN(d)) {
099                                        sum += e.length();
100                                        cnt++;
101                                }
102                        }
103                        if(cnt > 0)
104                                values.put(vertex, sum/(double)cnt);
105                }
106                
107                return values;
108        }
109        
110        public TObjectDoubleHashMap<SpatialVertex> vertexMedian(Set<? extends SpatialVertex> vertices) {
111                TObjectDoubleHashMap<SpatialVertex> values = new TObjectDoubleHashMap<SpatialVertex>();
112                
113                for(SpatialVertex vertex : vertices) {
114                        TDoubleArrayList list = new TDoubleArrayList();
115                        for(SpatialEdge e : vertex.getEdges()) {
116                                double d = e.length();
117                                
118                                if(!Double.isNaN(d)) {
119                                        list.add(d);
120                                }
121                        }
122                        if(list.size() > 0)
123                                values.put(vertex, StatUtils.percentile(list.toArray(), 50));
124                }
125                
126                return values;
127        }
128        
129        public Distribution vertexMeanDistribution(Set<? extends SpatialVertex> vertices) {
130                TObjectDoubleHashMap<SpatialVertex> values = vertexMean(vertices);
131                return new Distribution(values.values());
132        }
133        
134        public Distribution vertexMedianDistribution(Set<? extends SpatialVertex> vertices) {
135                TObjectDoubleHashMap<SpatialVertex> values = vertexMedian(vertices);
136                return new Distribution(values.values());
137        }
138}
