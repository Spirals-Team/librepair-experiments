001package org.matsim.contrib.matsim4urbansim.config;
002
003import java.io.File;
004
005import javax.xml.XMLConstants;
006import javax.xml.bind.JAXBContext;
007import javax.xml.bind.JAXBElement;
008import javax.xml.bind.Unmarshaller;
009import javax.xml.validation.SchemaFactory;
010
011import org.apache.log4j.Logger;
012import org.matsim.contrib.matsim4urbansim.matsim4urbansim.jaxbconfigv3.Matsim4UrbansimConfigType;
013import org.matsim.contrib.matsim4urbansim.matsim4urbansim.jaxbconfigv3.ObjectFactory;
014
015public class JAXBUnmarshalV3 {
016
017        // logger
018        private static final Logger log = Logger.getLogger(JAXBUnmarshalV3.class);
019
020        /**
021         * reads matsim config generated from urbansim and inits jaxbv3 object structure
022         * @param matsimConfigFile
023         * @return Matsim4UrbansimConfigType
024         */
025        @SuppressWarnings("unchecked")
026        public Matsim4UrbansimConfigType unmarshal(String matsimConfigFile) {
027
028                Matsim4UrbansimConfigType m4uConfigType = null;
029
030                log.info("Unmaschalling MATSim configuration from: " + matsimConfigFile);
031                log.info("...");
032                try {
033
034                        JAXBContext jaxbContext = JAXBContext.newInstance(ObjectFactory.class);
035                        // create an unmaschaller (write xml file)
036                        Unmarshaller unmarschaller = jaxbContext.createUnmarshaller();
037                        unmarschaller.setSchema(SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI).newSchema(getClass().getResource("/dtd/matsim4urbansim_v3.xsd")));
038
039                        File inputFile = new File( matsimConfigFile );
040                        isFileAvailable(matsimConfigFile, inputFile);
041                        // contains the content of the MATSim config.
042                        Object object = unmarschaller.unmarshal(inputFile);
043                        
044                        // The structure of both objects must match.
045                        if(object.getClass() == Matsim4UrbansimConfigType.class)
046                                m4uConfigType = (Matsim4UrbansimConfigType) object;
047                        else
048                                m4uConfigType = (( JAXBElement<Matsim4UrbansimConfigType>) object).getValue();
049
050                } catch (Exception e) {
051                        System.out.flush() ;
052                        e.printStackTrace();
053                        throw new RuntimeException("unmarschalling failed; aborting ...") ;
054                }
055
056                log.info("... finished unmarschallig");
057                // return initialized object representation of matsim4urbansim config
058                // file
059                return m4uConfigType;
060
061        }
062        
063        
064        private void isFileAvailable(String matsimConfigFile, File file){
065                if(!file.exists()){
066                        log.error(matsimConfigFile + " not found!!!");
067                        System.exit(-1);
068                }
069        }
070        
071        
072}
