001package org.matsim.contrib.matsim4urbansim.utils.network;
002
003import java.util.ArrayList;
004import java.util.List;
005
006import org.apache.log4j.Logger;
007import org.matsim.api.core.v01.network.Network;
008import org.matsim.api.core.v01.network.Node;
009import org.matsim.core.api.internal.NetworkRunnable;
010
011/**
012 * Nodes with no incoming or outgoing links are removed from the network.
013 *
014 * @author aneumann based on mrieser and balmermi
015 */
016public class NetworkRemoveUnusedNodes implements NetworkRunnable {
017
018        private static final Logger log = Logger.getLogger(NetworkRemoveUnusedNodes.class);
019
020        @Override
021        public void run(final Network network) {
022                log.info("running " + this.getClass().getName() + " algorithm...");
023                log.info("  network contains " + network.getNodes().size() + " nodes and " + network.getLinks().size() + " links.");
024
025                /* Reducing the network so it only contains nodes with links attached.
026                 * Loop over all nodes and check if they have in or outLinks, if not, remove them from the network
027                 */
028                List<Node> nodesList = new ArrayList<Node>(network.getNodes().values());
029                for (Node node : nodesList) {
030                        if (node.getInLinks().size() + node.getOutLinks().size() == 0) {
031                                network.removeNode(node.getId());               // removeNode takes care of removing links too in the network
032                        }
033                }
034                log.info("  resulting network contains " + network.getNodes().size() + " nodes and " + network.getLinks().size() + " links.");
035                log.info("done.");
036        }
037
038}
