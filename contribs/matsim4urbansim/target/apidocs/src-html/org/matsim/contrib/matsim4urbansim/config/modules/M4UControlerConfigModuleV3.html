030/**
031 * @author thomas
032 */
033public class M4UControlerConfigModuleV3 extends ConfigGroup{
034        // IMPORTANT: This is just a container for matsim4urbansim config data.  It is _not_ a matsim config group.
035
036        public static final String GROUP_NAME = "matsim4urbansimControler";
037        
038        private boolean usingShapefileLocationDistribution;
039        private String urbansimZoneRandomLocationDistributionShapeFile;
040        private double urbansimZoneRandomLocationDistributionRadius;
041        private boolean usingWarmStart;
042    private boolean usingHotStart;
043    private String warmStartPlansFileLocation;
044    private String hotStartPlansFileLocation;
045    
046        public M4UControlerConfigModuleV3() {
047                super(GROUP_NAME);
048                this.usingHotStart       = false;
049                // This class feels quite dangerous to me; one can have inconsistent entries between the Map and the typed values. kai, apr'13 
050                // The way it (hopefully) works now: as long as the config group is not "materialized", one has to use addParam/getValue.
051                // Once the class is materialized, one can only use the direct getters/setters.  kai, may'13
052        }
053
054        @Override
055        @Deprecated
056        public String getValue(final String key) {
057                throw new RuntimeException(" use direct getter; aborting ... " ) ;
058        }
059
060        @Override
061        @Deprecated
062        public void addParam(String param_name, String value) {
063                throw new RuntimeException(" use direct setter; aborting ... " ) ;
064        }
065        
066        @Override
067        public final Map<String,String> getParams() {
068                Map<String,String> map = new LinkedHashMap<String,String>() ;
069
070                // Maybe we don't want this?  Users should set this via the OPUS GUI, not via the external matsim config file.
071                // ??  kai, may'13
072
073                return map ;
074        }
075
076        public void setUsingShapefileLocationDistribution(boolean useShapefileLocationDistribution){
077                this.usingShapefileLocationDistribution = useShapefileLocationDistribution;
078        }
079        
080        public boolean usingShapefileLocationDistribution(){
081                return this.usingShapefileLocationDistribution;
082        }
083        
084        public void setUrbansimZoneRandomLocationDistributionShapeFile(String shapefile){
085                this.urbansimZoneRandomLocationDistributionShapeFile = shapefile;
086        }
087        
088        public String getUrbansimZoneRandomLocationDistributionShapeFile() {
089                return this.urbansimZoneRandomLocationDistributionShapeFile;
090        }
091        
092        public void setUrbansimZoneRandomLocationDistributionRadius(double radius){
093                this.urbansimZoneRandomLocationDistributionRadius = radius;
094        }
095        
096        public double getUrbanSimZoneRadiusLocationDistribution() {
097                return this.urbansimZoneRandomLocationDistributionRadius;
098        }
099
100    public boolean usingWarmStart(){
101        return this.usingWarmStart;
102    }
103    
104    public void setWarmStart(boolean value){
105        this.usingWarmStart = value;
106    }
107        
108    public void setWarmStartPlansLocation(String path){
109        this.warmStartPlansFileLocation = path;
110    }
111    
112    public String getWarmStartPlansFileLocation(){

114    }
115    
116    public boolean usingHotStart(){
117        return this.usingHotStart;
118    }
119    
120    public void setHotStart(boolean value){
121        this.usingHotStart = value;
122    }
123    
124    public String getHotStartPlansFileLocation(){

126    }
127    
128    public void setHotStartPlansFileLocation(String path){

130    }
