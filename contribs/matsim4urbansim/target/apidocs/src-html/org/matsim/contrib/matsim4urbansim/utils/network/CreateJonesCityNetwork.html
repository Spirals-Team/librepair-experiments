124        /**
125         * @param network
126         */
127        private static void concistencyCheck(Network network) {
128                int nodeSize = network.getNodes().values().size();
129                int linkSize = network.getLinks().values().size();
130                int steps = (int)Math.sqrt(nodeSize);
131                int part1 = nodeSize-(steps*2);
132                int expectedNumberOfLinks = 2*(2*part1 + 2*steps);
133                if(expectedNumberOfLinks != linkSize)
134                        System.out.println("The network might be inconsistent");
135        }

137        /**
138         * @param network
139         * @param reader
140         * @param indexXCoodinate
141         * @param indexYCoodinate
142         * @param indexZoneID
143         * @throws IOException
144         * @throws NumberFormatException
145         */
146        private static Iterator<? extends Node> addNodes2Network(Network network,
147                        BufferedReader reader, final int indexXCoodinate,
148                        final int indexYCoodinate, final int indexZoneID)
149                        throws IOException, NumberFormatException {
150
151                String line;
152                // temporary variables, needed to construct nodes
153                Id<ActivityFacility> zoneID;
154                Coord coord;
155                String[] parts;
156
157                while ((line = reader.readLine()) != null) {
158                        parts = line.split(separator);
159
160                        // get zone ID, UrbanSim sometimes writes IDs as floats!
161                        int zoneIdAsInt = Integer.parseInt(parts[indexZoneID]);
162                        zoneID = Id.create(zoneIdAsInt, ActivityFacility.class);
163                        // get the coordinates of that parcel
164                        // tnicolai remove the 1000.
165                        double x = Double.parseDouble( parts[indexXCoodinate] );
166                        double y = Double.parseDouble( parts[indexYCoodinate] );
167
168                        coord = new Coord(x, y);
169                        final Coord coord1 = coord;
170
171                        // create a new node
172                        Node node = (Node)NetworkUtils.createAndAddNode(network, Id.create(zoneID, Node.class), coord1);
173                        NetworkUtils.setOrigId( node, zoneID.toString() ) ;
174                        NetworkUtils.setType(node,"unknownType");               
175                }
176                return network.getNodes().values().iterator();
177        }
178        
179        /**
180         * @param network
181         */
182        private static void addLinks2Network(Network network) {
183                
184                int zones = network.getNodes().size();
185                int steps = (int)Math.sqrt(zones)-1;
186                double length = stepSize * 1000; // tnicolai: replace the 1000 when the new zones data set is delivered. With this setting the network is created correctley but dosen't match with the home and work locations
187                double maxValue = steps * length;
188                double minValue = 0.;
189                long linkID = 0;
190                
191                for(double x = minValue; x < maxValue; x = x+length ){
192                        System.out.println("X-Coord:" + x);
193                        for(double y = minValue; y < maxValue; y = y+length){
194                                System.out.println("Y-Coord:" + y);
195
196                                Node currentNode = NetworkUtils.getNearestNode(network,new Coord(x, y));
197                                // add link to above neighbor
198                                linkID = addLink(network, maxValue, linkID, currentNode, new Coord(x, y + length));
199                                // add link to left neighbor
200                                linkID = addLink(network, maxValue, linkID, currentNode, new Coord(x + length, y));
201                        }
202                }
203        }
204
205        /**
206         * @param network
207         * @param maxValue
208         * @param linkID
209         * @param currentNode
210         * @param neighbor
211         */
212        private static long addLink(Network network, double maxValue,
213                                                                long linkID, Node currentNode, Coord neighbor) {
214                
215                if (neighbor.getX() < maxValue
216                 && neighbor.getY() < maxValue) {
217                        
218                        final Coord coord = neighbor;
219                        Node neighborNode = NetworkUtils.getNearestNode(network,coord);
220                        if(neighborNode != currentNode){
221                                final Node fromNode = currentNode;
222                                final Node toNode = neighborNode;
223                                final double length1 = length;
224                                final double freespeed = freeSpeed;
225                                final double capacity1 = capacity;
226                                final double numLanes = lanes;
227                                NetworkUtils.createAndAddLink(network,Id.create(linkID++, Link.class), fromNode, toNode, length1, freespeed, capacity1, numLanes );
228                                final Node fromNode1 = neighborNode;
229                                final Node toNode1 = currentNode;
230                                final double length2 = length;
231                                final double freespeed1 = freeSpeed;
232                                final double capacity2 = capacity;
233                                final double numLanes1 = lanes;
234                                NetworkUtils.createAndAddLink(network,Id.create(linkID++, Link.class), fromNode1, toNode1, length2, freespeed1, capacity2, numLanes1 );
235                        }
236                }
237                else
238                        System.out.println("This lies outside the network: x="+ neighbor.getX() + " y="+ neighbor.getY());
239                
240                return linkID;
241        }
