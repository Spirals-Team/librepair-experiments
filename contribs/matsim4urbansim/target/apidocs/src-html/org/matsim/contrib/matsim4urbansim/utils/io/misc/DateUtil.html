001package org.matsim.contrib.matsim4urbansim.utils.io.misc;
002
003import java.text.SimpleDateFormat;
004import java.util.Calendar;
005
006public class DateUtil {
007
008        /**
009         * returns current date time in given dateFormat
010         * @param dateFormat
011         * @return
012         */
013        public static String now(String dateFormat) {
014            Calendar cal = Calendar.getInstance();
015            SimpleDateFormat sdf = new SimpleDateFormat(dateFormat);
016            return sdf.format(cal.getTime());
017         }
018        
019        /**
020         * returns current date time in default dateFormat
021         * @return
022         */
023        public static String now(){
024                Calendar cal = Calendar.getInstance();
025                // Format: YearMonthDay_HourMinute
026            SimpleDateFormat sdf = new SimpleDateFormat("yyyyMMdd_HHmm");
027            return sdf.format(cal.getTime());
028        }
029          
030        /**
031         * Testing
032         * @param arg
033         */
034        public static void  main(String arg[]) {
035                
036           System.out.println( DateUtil.now() );
037           System.out.println(DateUtil.now("dd MMMMM yyyy"));
038           System.out.println(DateUtil.now("yyyyMMdd"));
039           System.out.println(DateUtil.now("dd.MM.yy"));
040           System.out.println(DateUtil.now("MM/dd/yy"));
041           System.out.println(DateUtil.now("yyyy.MM.dd G 'at' hh:mm:ss z"));
042           System.out.println(DateUtil.now("EEE, MMM d, ''yy"));
043           System.out.println(DateUtil.now("h:mm a"));
044           System.out.println(DateUtil.now("H:mm:ss:SSS"));
045           System.out.println(DateUtil.now("K:mm a,z"));
046           System.out.println(DateUtil.now("yyyy.MMMMM.dd GGG hh:mm aaa"));
047        }
048}
