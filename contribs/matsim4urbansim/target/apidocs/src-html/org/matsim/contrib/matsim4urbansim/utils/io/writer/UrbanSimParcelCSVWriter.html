023package org.matsim.contrib.matsim4urbansim.utils.io.writer;
024
025import java.io.BufferedWriter;
026import java.io.File;
027import java.io.IOException;
028import java.nio.file.Files;
029import java.util.Map;
030
031import org.apache.log4j.Logger;
032import org.matsim.api.core.v01.Id;
033import org.matsim.contrib.accessibility.Modes4Accessibility;
034import org.matsim.contrib.matsim4urbansim.config.modules.UrbanSimParameterConfigModuleV3;
035import org.matsim.contrib.matsim4urbansim.constants.InternalConstants;
036import org.matsim.core.config.Config;
037import org.matsim.core.utils.io.IOUtils;
038import org.matsim.core.utils.io.UncheckedIOException;
039
040
041/**
042 * @author thomas
043 *
044 */
045public class UrbanSimParcelCSVWriter {
046
047        private static final Logger log         = Logger.getLogger(UrbanSimParcelCSVWriter.class);
048        private static BufferedWriter parcelWriter = null;
049        public static final String FILE_NAME= "parcels.csv";
050        public static final String ACCESSIBILITY_BY_FREESPEED = "freespeed_accessibility";
051        public static final String ACCESSIBILITY_BY_CAR = "car_accessibility";
052        public static final String ACCESSIBILITY_BY_BIKE = "bike_accessibility";
053        public static final String ACCESSIBILITY_BY_WALK = "walk_accessibility";
054        public static final String ACCESSIBILITY_BY_PT = "pt_accessibility";
055        
056        /**
057         * writes the header for zones csv file
058         * @param config TODO
059         */
060        public static void initUrbanSimZoneWriter(Config config){
061                UrbanSimParameterConfigModuleV3 module = (UrbanSimParameterConfigModuleV3) config.getModule(UrbanSimParameterConfigModuleV3.GROUP_NAME);
062                try{
063                        log.info("Initializing UrbanSimParcelCSVWriter ...");
064                        parcelWriter = IOUtils.getBufferedWriter( module.getMATSim4OpusTemp() + FILE_NAME );
065                        log.info("Writing data into " + module.getMATSim4OpusTemp() + FILE_NAME + " ...");
066                        
067                        // create header
068                        parcelWriter.write( InternalConstants.PARCEL_ID + "," +
069                                                                ACCESSIBILITY_BY_FREESPEED + "," +
070                                                                ACCESSIBILITY_BY_CAR + "," +
071                                                                ACCESSIBILITY_BY_BIKE + "," +
072                                                                ACCESSIBILITY_BY_WALK + "," +
073                                                                ACCESSIBILITY_BY_PT);
074                        parcelWriter.newLine();
075                        
076                        log.info("... done!");
077                }
078                catch(Exception e){
079                        e.printStackTrace();
080                }
081        }
082        
083        /**
084         * writing the parcel data (accessibilities) to csv file
085         * @param node
086         * @param freeSpeedAccessibility
087         * @param carAccessibility
088         * @param walkAccessibility
089         */
090        public static void write(Id parcelID, Map<Modes4Accessibility,Double> accessibilities ) {
091                try{
092                        assert(UrbanSimParcelCSVWriter.parcelWriter != null);
093                        parcelWriter.write( parcelID.toString() ) ;
094                        for ( Modes4Accessibility mode : Modes4Accessibility.values() ) {
095                                parcelWriter.write( "," + accessibilities.get( mode ) ) ;
096                        }
097                        parcelWriter.newLine();
098                }
099                catch(Exception e){
100                        e.printStackTrace();
101                        throw new RuntimeException("could not write") ;
102                }
103        }
104        
105        /**
106         * finalize and close csv file
107         * @param config TODO
108         */
109        public static void close(Config config){
110                UrbanSimParameterConfigModuleV3 module = (UrbanSimParameterConfigModuleV3) config.getModule(UrbanSimParameterConfigModuleV3.GROUP_NAME);
111                try {
112                        log.info("Closing UrbanSimZoneCSVWriterV2 ...");
113                        assert(UrbanSimParcelCSVWriter.parcelWriter != null);
114                        parcelWriter.flush();
115                        parcelWriter.close();
116                        
117                        // copy the zones file to the outputfolder...
118                        log.info("Copying " + module.getMATSim4OpusTemp() + FILE_NAME + " to " + module.getMATSim4OpusOutput() + FILE_NAME);
119            try {
120                Files.copy(new File( module.getMATSim4OpusTemp() + FILE_NAME).toPath(), new File( module.getMATSim4OpusOutput()+ FILE_NAME).toPath());
121            } catch (IOException e) {
122                throw new UncheckedIOException(e);
123            }
124
125            log.info("... done!");
126                } catch (IOException e) {
127                        e.printStackTrace();
128                }       
129        }
130}
