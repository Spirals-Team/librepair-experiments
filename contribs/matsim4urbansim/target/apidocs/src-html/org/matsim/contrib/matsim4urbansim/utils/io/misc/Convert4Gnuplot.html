432         * @param header
433         * @return
434         * @throws NumberFormatException
435         */
436        @SuppressWarnings("all")
437        private static double[] getXAxis(String[] header) throws NumberFormatException{
438                
439                String[] tmp;
440                String tmpName;
441                
442                double xAxis[] = new double[header.length-1];
443                
444                for(int i = 1; i < header.length; i++){
445                        
446                        tmpName = header[i];
447                        tmp = tmpName.split(":");
448                        tmpName = tmp[0];
449                        tmp = tmpName.split("_");
450                        tmpName = tmp[tmp.length-1];
451                        
452                        xAxis[i-1] = Integer.parseInt(tmpName);
453                }
454                return xAxis;
455        }

482        /**
483         * creates an header object. A header object retuns the indices of columns
484         * in ascending order.
485         * 
486         * @param line
487         */
488        private static HeaderObject initHeaderObject(String line){
489                
490                String parts[] = line.split( InternalConstants.TAB );
491
492                int unsortedHeaderArray[] = new int[parts.length];
493                int sortedHeaderArray[] = new int[parts.length];
494                
495                // puts a header like "zone_id:i8       var_2018:f8     var_2019:f8     var_2012:f8"
496                // into two unsorted arrays with the followg structure: [-1, 2018, 2019, 2012]
497                for(int i = 0; i < parts.length; i++){
498                        int index = parts[i].indexOf(":");
499                        parts[i] = parts[i].substring(0, index);
500                        
501                        String key[] = parts[i].split("_");
502                        
503                        try{
504                                unsortedHeaderArray[i] = Integer.parseInt( key[key.length-1]);
505                        }
506                        catch(NumberFormatException nfe){
507                                unsortedHeaderArray[i] = -1; // -> zone id
508                        }
509                        // only copy values to other array
510                        sortedHeaderArray[i] = unsortedHeaderArray[i];
511                }
512                // stores unsorted array as a map of <YEAR, INDEX>
513                Map<Integer,Integer> idxFromKey = HeaderParser.createIdxFromKey( unsortedHeaderArray );
514                // sorted array in chronological order. The zone id is stored as -1 and should be the first value
515                ArrayQuicksort( sortedHeaderArray );
516                
517                return new HeaderObject(idxFromKey, sortedHeaderArray);
518        }

520        /**
521         * sorts a given array
522         * 
523         * @param array
524         * 
525         * @author thomas
526         */
527        public static int[] ArrayQuicksort(int array[]){
528            int i;
529        
530            System.out.println("Values Before the sort:\n");
531            for(i = 0; i < array.length; i++)
532              System.out.print( array[i]+"  ");
533            System.out.println();
534            quick_srt(array,0,array.length-1);
535            System.out.print("Values after the sort:\n");
536            for(i = 0; i <array.length; i++)
537              System.out.print(array[i]+"  ");
538            return array;
539        }

541        private static void quick_srt(int array[],int low, int n){
542            int lo = low;
543            int hi = n;
544            if (lo >= n) {
545              return;
546            }
547            int mid = array[(lo + hi) / 2];
548            while (lo < hi) {
549              while (lo<hi && array[lo] < mid) {
550                lo++;
551              }
552              while (lo<hi && array[hi] > mid) {
553                hi--;
554              }
555              if (lo < hi) {
556                int T = array[lo];
557                array[lo] = array[hi];
558                array[hi] = T;
559              }
560            }
561            if (hi < lo) {
562              int T = hi;
563              hi = lo;
564              lo = T;
565            }
566            quick_srt(array, low, lo);
567            quick_srt(array, lo == low ? lo+1 : lo, n);
568          }

660class HeaderObject{
661        
662        private Map<Integer,Integer> idxFromKey = null;
663        private int sortedHeaderArray[] = null;
664        
665        public HeaderObject(Map<Integer,Integer> idxFromKey, int sortedHeaderArray[]){
666                this.idxFromKey = idxFromKey;
667                this.sortedHeaderArray = sortedHeaderArray;
668        }
669
670        /**
671         * returns the index for zone_id
672         * @return index
673         */
674        public int getZoneId(){
675                int index = idxFromKey.get(-1);
676                return index;
677        }
678        
679        public int getIndexOf(int i){
680                int year = sortedHeaderArray[i];
681                int index = idxFromKey.get(year);
682                return index;
683        }
684        
685        public double[] getSortedHeaderAsDouble(){
686                double tmp[] = new double[sortedHeaderArray.length-1];
687                for (int i = 1; i < sortedHeaderArray.length; i++)
688                        tmp[i-1] = (double) sortedHeaderArray[i];
689                return tmp;
690        }
691        
692}
