050public class RandomLocationDistributor {
051        
052        private static final Logger log = Logger.getLogger(RandomLocationDistributor.class);
053        public static final String ZONE_ID = "ZONE_ID";
054        
055        private Random random           = null;
056        private String shapefile        = null;
057        private double radius           = 0.;
058        private boolean isShapefile = false;
059        private Map<Id<ActivityFacility>, SimpleFeature> featureMap = null;
060        
061        /**
062         * constructor
063         * @param shapefile
064         * @param radius
065         */
066        public RandomLocationDistributor(String shapefile, double radius) {
067                this.random = MatsimRandom.getRandom();
068                this.shapefile = shapefile;
069                this.radius = radius;
070
071                if (this.shapefile != null && Paths.pathExsits(this.shapefile)) {
072                        // try to create a map of zone features from the shape file
073                        this.featureMap = initShapeFeatures(this.shapefile);
074                        // set true if successful
075                        if (this.featureMap != null)
076                                this.isShapefile = true;
077                } else
078                        log.info("No shape-file given. Using radius (" + radius
079                                        + " meter) and the zone centroid to distribute locations.");
080        }
081
082        /**
083         * inits zone features
084         * 
085         * @param shapefile
086         * @return
087         */
088        private Map<Id<ActivityFacility>, SimpleFeature> initShapeFeatures(String shapefile) {
089                try {
090                        SimpleFeatureSource fts = ShapeFileReader.readDataFile(shapefile); // reads in shape file
091                        SimpleFeatureIterator fIt = fts.getFeatures().features();
092                        Map<Id<ActivityFacility>, SimpleFeature> featureMap = new ConcurrentHashMap<>();
093                        
094                        while (fIt.hasNext()) {
095                                SimpleFeature feature = fIt.next();
096                                featureMap.put(Id.create(feature.getAttribute(ZONE_ID).toString(), ActivityFacility.class), feature);
097                        }
098                        fIt.close();
099                        return featureMap;
100                } catch (IOException ioe) {
101                        ioe.printStackTrace();
102                }
103                log.warn("Could not initialize zone freatures from shape-file. Using radius ("
104                                + radius
105                                + " meter) and the zone centroid instead to distribute locations.");
106                return null;
107        }
108        
109        public Coord getRandomLocation(Id<ActivityFacility> zoneID, Coord coord) {
110                if(isShapefile){
111                        SimpleFeature feature = this.featureMap.get(zoneID);
112                        return getRandomPointInFeature(feature);
113                }
114                else
115                        return getRandomPointInRadius(coord);
116        }
117        
118        /**
119         * distributing location within a circle defined by a given zone centroid and radius
120         * 
121         * @param rnd
122         * @param zoneCoordinate
123         * @param radius
124         * @return
125         */
126        private Coord getRandomPointInRadius(Coord zoneCoordinate){
127                Coord p = null;
128                double x, y;
129                double distance;
130                do {
131                        x = (zoneCoordinate.getX() - radius) + (random.nextDouble() * 2 * radius);
132                        y = (zoneCoordinate.getY() - radius) + (random.nextDouble() * 2 * radius);
133                        p = new Coord(x, y);
134                        
135                        distance = CoordUtils.calcEuclideanDistance(zoneCoordinate, p);
136                        
137                } while ( distance > radius );
138                return p;
139        }
140        
141        /**
142         * distributing locations within a zone given by a shapefile
143         * 
144         * @param rnd
145         * @param feature
146         * @return
147         */
148        private Coord getRandomPointInFeature(SimpleFeature feature) {
149                Point p = null;
150                double x, y;
151                do {
152                        x = feature.getBounds().getMinX() + random.nextDouble() * (feature.getBounds().getMaxX() - feature.getBounds().getMinX());
153                        y = feature.getBounds().getMinY() + random.nextDouble() * (feature.getBounds().getMaxY() - feature.getBounds().getMinY());
154                        p = MGC.xy2Point(x, y);
155                } while (!((Geometry) feature.getDefaultGeometry()).contains(p));
156                return new Coord(x, y);
157        }
158        
159        /**
160         * determines if a given coordinate lies within a selected zone.
161         * the shape of the zone is defined by the respective zone geometry in the shape file
162         * 
163         * @param zoneID is an identifyer to select a zone in the shape file
164         * @param coordinate to be tested if it lies within the selected zone
165         * @return true if coordinate lies within the selected zone
166         */
167        public boolean coordinateInZone(Id<ActivityFacility> zoneID, Coord coordinate){
168                Point point = MGC.xy2Point(coordinate.getX(), coordinate.getY());
169                return pointInZone(zoneID, point);
170        }
171        
172        /**
173         * determines if a given point lies within a selected zone.
174         * the shape of the zone is defined by the respective zone geometry in the shape file
175         * 
176         * @param zoneID is an identifyer to select a zone in the shape file
177         * @param point to be tested if it lies within the selected zone
178         * @return true if point lies within the selected zone
179         */
180        public boolean pointInZone(Id<ActivityFacility> zoneID, Point point){
181                SimpleFeature feature = this.featureMap.get(zoneID);
182                boolean withinZoneGeometry = ((Geometry) feature.getDefaultGeometry()).contains(point);
183                return withinZoneGeometry;
184        }
185
186        /**
187         * determines if given coordinates are lying within the radius
188         * 
189         * @param coordinateA a coordinate
190         * @param coordinateB a coordinate
191         * @return true if coordinate lies within the radius
192         */
193        public boolean coordinatesInRadius(Coord coordinateA, Coord coordinateB){
194                double distance = CoordUtils.calcEuclideanDistance(coordinateA, coordinateB);
195                boolean withinRadius = (distance <= radius);
196                return withinRadius;
197        }
