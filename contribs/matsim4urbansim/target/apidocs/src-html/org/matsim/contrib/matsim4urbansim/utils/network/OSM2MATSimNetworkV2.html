042                Scenario sc = ScenarioUtils.createScenario(ConfigUtils.createConfig());
043                // creating an empty matsim network
044                Network network = sc.getNetwork();
045                // using the Belge Lambert 72 projection for the matsim network
046                CoordinateTransformation ct = TransformationFactory
047                                .getCoordinateTransformation(TransformationFactory.WGS84,
048                                                "EPSG:31300");
049                OsmNetworkReader osmReader = new OsmNetworkReader(network, ct);
050
051                osmReader.setKeepPaths(false);
052                osmReader.setScaleMaxSpeed(true);
053
054                // this layer covers the whole area, Belgium and bordering areas
055                // including OSM motorways and trunks
056                // osmReader.setHierarchyLayer(51.671, 2.177, 49.402, 6.764, 2);
057                osmReader.setHierarchyLayer(51.671, 2.177, 49.402, 6.764, 4);

062                
063                // tnicolai: this layer is removed, because of having different layers within the

065                // this layer covers the city of Brussels including OSM 
066                // tertiary roads or greater
067                // osmReader.setHierarchyLayer(50.9515, 4.1748, 50.7312, 4.5909, 5);
068
069                // converting the merged OSM network into matsim format
070                osmReader.parse(INFILE);
071                new NetworkWriter(network).write(OUTFILE);
072                // writing out a cleaned matsim network and loading it 
073                // into the scenario
074                new NetworkCleaner().run(OUTFILE, OUTFILE.split(".xml")[0]
075                                + "_clean.xml.gz");
076                Scenario scenario = (MutableScenario) ScenarioUtils
077                                .createScenario(ConfigUtils.createConfig());
078                new MatsimNetworkReader(scenario.getNetwork()).readFile(OUTFILE.split(".xml")[0]
079                                + "_clean.xml.gz");
080                network = (Network) scenario.getNetwork();
081
082                // simplifying the cleaned network
083                NetworkSimplifier simplifier = new NetworkSimplifier();
084                Set<Integer> nodeTypess2merge = new HashSet<Integer>();
085                nodeTypess2merge.add(new Integer(4));
086                nodeTypess2merge.add(new Integer(5));
087                simplifier.setNodesToMerge(nodeTypess2merge);
088                simplifier.run(network);
089                new NetworkWriter(network).write(OUTFILE.split(".xml")[0]
090                                + "_clean_simple.xml.gz");
091        }
