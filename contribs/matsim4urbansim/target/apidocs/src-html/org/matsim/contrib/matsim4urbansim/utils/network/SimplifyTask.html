001package org.matsim.contrib.matsim4urbansim.utils.network;
002
003import java.util.List;
004import java.util.ListIterator;
005import java.util.Map;
006
007import org.openstreetmap.osmosis.core.container.v0_6.BoundContainer;
008import org.openstreetmap.osmosis.core.container.v0_6.EntityContainer;
009import org.openstreetmap.osmosis.core.container.v0_6.EntityProcessor;
010import org.openstreetmap.osmosis.core.container.v0_6.NodeContainer;
011import org.openstreetmap.osmosis.core.container.v0_6.RelationContainer;
012import org.openstreetmap.osmosis.core.container.v0_6.WayContainer;
013import org.openstreetmap.osmosis.core.domain.v0_6.Way;
014import org.openstreetmap.osmosis.core.domain.v0_6.WayNode;
015import org.openstreetmap.osmosis.core.filter.common.IdTracker;
016import org.openstreetmap.osmosis.core.filter.common.IdTrackerFactory;
017import org.openstreetmap.osmosis.core.filter.common.IdTrackerType;
018import org.openstreetmap.osmosis.core.lifecycle.ReleasableIterator;
019import org.openstreetmap.osmosis.core.store.SimpleObjectStore;
020import org.openstreetmap.osmosis.core.store.SingleClassObjectSerializationFactory;
021import org.openstreetmap.osmosis.core.task.v0_6.Sink;
022import org.openstreetmap.osmosis.core.task.v0_6.SinkSource;
023
024public class SimplifyTask implements SinkSource, EntityProcessor {
025
026        private Sink sink;
027
028        private IdTracker visitedNodes;
029
030        private IdTracker requiredNodes;
031
032        private SimpleObjectStore<NodeContainer> allNodes;
033        
034        private SimpleObjectStore<WayContainer> allWays;
035
036        private int count = 0; // just for debug stats
037
038        private IdTracker availableNodes;
039
040        /**
041         * Creates a new instance.
042         * 
043         * @param idTrackerType
044         *            Defines the id tracker implementation to use.
045         */
046        public SimplifyTask(IdTrackerType idTrackerType) {
047                requiredNodes = IdTrackerFactory.createInstance(idTrackerType);
048                visitedNodes = IdTrackerFactory.createInstance(idTrackerType);
049                availableNodes = IdTrackerFactory.createInstance(idTrackerType);
050                allNodes = new SimpleObjectStore<NodeContainer>(
051                                new SingleClassObjectSerializationFactory(NodeContainer.class),
052                                "afnd", true);
053                allWays = new SimpleObjectStore<WayContainer>(
054                                new SingleClassObjectSerializationFactory(WayContainer.class),
055                                "afwy", true);
056                
057        }
058
059        /**
060         * {@inheritDoc}
061         */
062        public void process(EntityContainer entityContainer) {
063                // Ask the entity container to invoke the appropriate processing method
064                // for the entity type.
065                entityContainer.process(this);
066        }
067
068        /**
069         * {@inheritDoc}
070         */
071        public void process(BoundContainer boundContainer) {
072                // pass on the bounds information unchanged
073                sink.process(boundContainer);
074        }
075
076        /**
077         * {@inheritDoc}
078         */
079        public void process(NodeContainer container) {
080
081                // stuff all nodes into a file
082                availableNodes.set(container.getEntity().getId());
083                allNodes.add(container);
084
085                // debug
086                count++;
087                if (count % 50000 == 0)
088                        System.out.println(count + " nodes processed so far");
089        }
090
091        /**
092         * {@inheritDoc}
093         */
094        public void process(WayContainer container) {
095                Way way = container.getEntity();
096                List<WayNode> wayNodes = way.getWayNodes();
097                if (wayNodes.size() < 2) {
098                
099                } else {
100                        WayNode startNode = wayNodes.get(0);
101                        WayNode endNode = wayNodes.get(wayNodes.size()-1);
102
103                        requiredNodes.set(startNode.getNodeId());
104                        requiredNodes.set(endNode.getNodeId());
105                        
106                        for (WayNode wayNode : wayNodes) {
107                                long nodeId = wayNode.getNodeId();
108                                if (visitedNodes.get(nodeId)) {
109                                        requiredNodes.set(nodeId);
110                                }
111                                visitedNodes.set(nodeId);
112                        }
113
114                        int prevRealNodeIndex = 0;
115                        WayNode prevRealNode = wayNodes.get(prevRealNodeIndex);
116                        for (int i = 1; i < wayNodes.size(); i++) {
117                                WayNode node = wayNodes.get(i);
118                                if (requiredNodes.get(node.getNodeId())) {
119                                        if (prevRealNode.getNodeId() == node.getNodeId()) {
120                                                /* We detected a loop between to "real" nodes.
121                                                 * Set some nodes between the start/end-loop-node to "used" again.
122                                                 * But don't set all of them to "used", as we still want to do some network-thinning.
123                                                 * I decided to use sqrt(.)-many nodes in between...
124                                                 */
125                                                double increment = Math.sqrt(i - prevRealNodeIndex);
126                                                double nextNodeToKeep = prevRealNodeIndex + increment;
127                                                for (double j = nextNodeToKeep; j < i; j += increment) {
128                                                        int index = (int) Math.floor(j);
129                                                        WayNode intermediaryNode = wayNodes.get(index);
130                                                        requiredNodes.set(intermediaryNode.getNodeId());
131                                                }
132                                        }
133                                        prevRealNodeIndex = i;
134                                        prevRealNode = node;
135                                }
136                        }
137
138                        
139                }
140                allWays.add(container);
141        }
142
143        /**
144         * {@inheritDoc}
145         */
146        public void process(RelationContainer container) {
147                // Do nothing (Drop all relations)
148        }
149
150        /**
151         * {@inheritDoc}
152         */
153        public void complete() {
154                ReleasableIterator<NodeContainer> nodeIterator;
155                ReleasableIterator<WayContainer> wayIterator;
156                NodeContainer nodeContainer;
157                long nodeId;
158
159                // Send on only the required nodes
160                nodeIterator = allNodes.iterate();
161                while (nodeIterator.hasNext()) {
162                        nodeContainer = nodeIterator.next();
163                        nodeId = nodeContainer.getEntity().getId();
164
165                        if (requiredNodes.get(nodeId)) {
166                                sink.process(nodeContainer);
167                        }
168                }
169                nodeIterator.release();
170                
171                wayIterator = allWays.iterate();
172                while (wayIterator.hasNext()) {
173                        WayContainer wayContainer = wayIterator.next();
174                        Way way = wayContainer.getEntity().getWriteableInstance();
175                        ListIterator<WayNode> i = way.getWayNodes().listIterator();
176                        while (i.hasNext()) {
177                                WayNode wayNode = i.next();
178                                if (!requiredNodes.get(wayNode.getNodeId()) || !availableNodes.get(wayNode.getNodeId())) {
179                                        i.remove();
180                                }
181                        }
182                        sink.process(wayContainer);
183                }
184                wayIterator.release();
185                
186                sink.complete();
187        }
188
189        /**
190         * {@inheritDoc}
191         */
192        public void release() {
193                allNodes.release();
194                allWays.release();
195                sink.release();
196        }
197
198        /**
199         * {@inheritDoc}
200         */
201        public void setSink(Sink sink) {
202                this.sink = sink;
203        }
204
205        @Override
206        public void initialize(Map<String, Object> metaData) {
207                // TODO Auto-generated method stub
208                
209        }
