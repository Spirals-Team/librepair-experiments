051        private void readLink(Way entry) {
052                List<WayNode> wayNodes = entry.getWayNodes();
053                if (wayNodes.size() < 2) {
054
055                } else {
056                        WayNode fromNode = wayNodes.get(0);
057                        for (int i = 1, n = wayNodes.size(); i < n; i++) {
058                                WayNode toNode = wayNodes.get(i);
059                                org.matsim.api.core.v01.network.Node fromMatsimNode = network.getNodes().get(Id.create(fromNode.getNodeId(), org.matsim.api.core.v01.network.Node.class));
060                                if (fromMatsimNode == null) {
061                                        throw new RuntimeException("Missing node: "+fromNode.getNodeId());
062                                }
063                                org.matsim.api.core.v01.network.Node toMatsimNode = network.getNodes().get(Id.create(toNode.getNodeId(), org.matsim.api.core.v01.network.Node.class));
064                                if (toMatsimNode == null) {
065                                        throw new RuntimeException("Missing node: "+toNode.getNodeId());
066                                }
067                                double length = CoordUtils.calcEuclideanDistance(fromMatsimNode.getCoord(), toMatsimNode.getCoord());
068                                createLink((Network) this.network, entry, fromNode, toNode, length);
069                                fromNode = toNode;
070                        }
071                }
072        }

075        private void readNode(Node osmNode) {
076                Coord osmCoord = new Coord(osmNode.getLongitude(), osmNode.getLatitude());
077                Coord transformedCoord = transform.transform(osmCoord);
078                org.matsim.api.core.v01.network.Node node = network.getFactory().createNode(Id.create(osmNode.getId(), org.matsim.api.core.v01.network.Node.class), transformedCoord);
079                network.addNode(node);
080                osmNode.getTags().add(new Tag("matsim:node-id", node.getId().toString()));
081        }

089        private void createLink(final Network network, final Way way, final WayNode fromNode, final WayNode toNode, final double length) {
090                TagCollectionImpl tagCollection = new TagCollectionImpl(way.getTags());
091                Map<String, String> tags = tagCollection.buildMap();
092                String highway = tags.get("highway");
093
094                OsmHighwayDefaults defaults = this.highwayDefaults.get(highway);
095                if (defaults == null) {
096                        defaults = this.highwayDefaults.get("tertiary");
097                }
098                String origId = Long.toString(way.getId());
099                double nofLanes = defaults.lanes;
100                double laneCapacity = defaults.laneCapacity;
101                double freespeed = defaults.freespeed;
102                double freespeedFactor = defaults.freespeedFactor;
103                boolean oneway = defaults.oneway;
104                boolean onewayReverse = false;
105
106                if (tags.containsKey("oneway")) {
107                        String onewayTag = tags.get("oneway");
108                        if ("yes".equals(onewayTag)) {
109                                oneway = true;
110                        } 
111                }
112
113                if (tags.containsKey("maxspeed")) {
114                        try {
115                                double maxspeed = Double.parseDouble(tags.get("maxspeed"));
116                                if (maxspeed < freespeed) {
117                                        // freespeed doesn't always mean it's the maximum speed allowed.
118                                        // thus only correct freespeed if maxspeed is lower than freespeed.
119                                        freespeed = maxspeed;
120                                }
121                        } catch (NumberFormatException e) {
122                                log.warn("Could not parse freespeed tag:" + e.getMessage() + ". Ignoring it.");
123                        }
124                }
125
126                // check tag "lanes"
127                if (tags.containsKey("lanes")) {
128                        try {
129                                if(Double.parseDouble(tags.get("lanes")) > 0){
130                                        nofLanes = Double.parseDouble(tags.get("lanes"));
131                                }
132                        } catch (Exception e) {
133                                log.warn("Could not parse lanes tag:" + e.getMessage() + ". Ignoring it.");
134                        }
135                }
136
137                // create the link(s)
138                double capacity = nofLanes * laneCapacity;
139
140                if (this.scaleMaxSpeed) {
141                        freespeed = freespeed * freespeedFactor;
142                }
143
144                // only create link, if both nodes were found, node could be null, since nodes outside a layer were dropped
145                long fromNodeNumber = fromNode.getNodeId();
146                long toNodeNumber = toNode.getNodeId();
147                if (!onewayReverse) {
148                        Id<Link> id = Id.create(Long.toString(way.getId())+"_"+Long.toString(fromNodeNumber)+"_"+Long.toString(toNodeNumber), Link.class);
149                        if (!network.getLinks().containsKey(id)) {
150                                final Id<Link> id2 = id;
151                                final double freespeed1 = freespeed;
152                                final double capacity1 = capacity;
153                                final double numLanes = nofLanes;
154                                Link l = NetworkUtils.createAndAddLink(network,id2, network.getNodes().get(Id.create(fromNodeNumber, org.matsim.api.core.v01.network.Node.class)), network.getNodes().get(Id.create(toNodeNumber, org.matsim.api.core.v01.network.Node.class)), length, freespeed1, capacity1, numLanes );
155                                final String id1 = origId;
156                                NetworkUtils.setOrigId( ((Link) l), id1 ) ;
157                                tagWayForward(way, l);
158                        } else {
159                                log.warn("Duplicate link: " + id);
160                        }
161                }
162                if (!oneway) {
163                        Id<Link> id = Id.create(Long.toString(way.getId())+"_"+Long.toString(fromNodeNumber)+"_"+Long.toString(toNodeNumber)+"_R", Link.class);
164                        if (!network.getLinks().containsKey(id)) {
165                                final Id<Link> id2 = id;
166                                final double freespeed1 = freespeed;
167                                final double capacity1 = capacity;
168                                final double numLanes = nofLanes;
169                                Link l = NetworkUtils.createAndAddLink(network,id2, network.getNodes().get(Id.create(toNodeNumber, org.matsim.api.core.v01.network.Node.class)), network.getNodes().get(Id.create(fromNodeNumber, org.matsim.api.core.v01.network.Node.class)), length, freespeed1, capacity1, numLanes );
170                                final String id1 = origId;
171                                NetworkUtils.setOrigId( ((Link) l), id1 ) ;
172                                tagWayBackward(way, l);
173                        } else {
174                                log.warn("Duplicate link: " + id);
175                        }
176                }
177        }

179        private void tagWayBackward(Way way, Link l) {
180                way.getTags().add(new Tag("matsim:backward:link-id:" + l.getId().toString(), l.getId().toString()));
181        }

183        private void tagWayForward(Way way, Link l) {
184                way.getTags().add(new Tag("matsim:forward:link-id:" + l.getId().toString(), l.getId().toString()));
185        }

197                        this.freespeedFactor = freespeedFactor;

216                        }

218                        @Override

222
223                        @Override
224                        public void process(WayContainer way) {
225                                readLink(way.getEntity());
226                                sink.process(way);
227                        }

229                });
230        }
