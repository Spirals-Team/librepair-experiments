001package org.matsim.contrib.matsim4urbansim.utils.helperobjects;
002
003
004public class AccessibilityStorage {
005
006        private double congestedTravelTimeAccessibility = - Double.MAX_VALUE;
007        private double freespeedTravelCostAccessibility = - Double.MAX_VALUE;
008        private double walkTravelTimeAccessibility              = - Double.MAX_VALUE;
009
010        /**
011         * constructor
012         * 
013         * @param conTravelTimeAccessibility
014         * @param freeTravelTimeAccessibility
015         * @param walkTravelTimeAccessibility
016         */
017        public AccessibilityStorage(double conTravelTimeAccessibility, double freeTravelTimeAccessibility, double walkTravelTimeAccessibility){
018                
019                this.congestedTravelTimeAccessibility = conTravelTimeAccessibility;
020                this.freespeedTravelCostAccessibility = freeTravelTimeAccessibility;
021                this.walkTravelTimeAccessibility = walkTravelTimeAccessibility;
022        }
023        
024        public double getCongestedTravelTimeAccessibility(){
025                return this.congestedTravelTimeAccessibility;
026        }
027        public double getFreespeedTravelCostAccessibility(){
028                return this.freespeedTravelCostAccessibility;
029        }
030        public double getWalkTravelTimeAccessibility(){
031                return this.walkTravelTimeAccessibility;
032        }
033
034}
