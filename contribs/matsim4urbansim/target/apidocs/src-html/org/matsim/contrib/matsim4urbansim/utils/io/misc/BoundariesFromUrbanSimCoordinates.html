001package org.matsim.contrib.matsim4urbansim.utils.io.misc;
002
003import java.io.BufferedReader;
004import java.io.FileNotFoundException;
005import java.io.IOException;
006import java.util.Map;
007
008import org.apache.log4j.Logger;
009import org.matsim.contrib.matsim4urbansim.constants.InternalConstants;
010import org.matsim.contrib.matsim4urbansim.utils.io.HeaderParser;
011import org.matsim.core.utils.io.IOUtils;
012
013
014public class BoundariesFromUrbanSimCoordinates {
015        // Logger
016        private static final Logger log = Logger.getLogger(BoundariesFromUrbanSimCoordinates.class);
017        
018        private static final String inputFile = "/Users/thomas/Development/opus_home/matsim4opus/tmp/zone__dataset_table__exported_indicators__2001.tab";
019        
020        private static double xmin = Double.MAX_VALUE;
021        private static double ymin = Double.MAX_VALUE;
022        private static double xmax = -Double.MAX_VALUE;
023        private static double ymax = -Double.MAX_VALUE;
024        
025        /**
026         * @param args
027         */
028        public static void main(String[] args) {
029                
030                long startTime = System.currentTimeMillis();
031                
032                BufferedReader reader = IOUtils.getBufferedReader(inputFile);
033                log.info( "Starting to read urbansim parcels table from " + inputFile );
034                
035                try{
036                        
037                        // read header of facilities table
038                        String line = reader.readLine();
039                
040                        // get and initialize the column number of each header element
041                        Map<String,Integer> idxFromKey = HeaderParser.createIdxFromKey( line, InternalConstants.TAB_SEPERATOR );
042                        final int indexXCoodinate       = idxFromKey.get( InternalConstants.X_COORDINATE );
043                        final int indexYCoodinate       = idxFromKey.get( InternalConstants.Y_COORDINATE );
044                        
045                        String[] parts;
046                        
047                        //
048                        while ( (line = reader.readLine()) != null ) {
049                                parts = line.split(InternalConstants.TAB_SEPERATOR);
050                                
051                                double x = Double.parseDouble( parts[indexXCoodinate] );
052                                double y = Double.parseDouble( parts[indexYCoodinate] );
053                                
054                                xmin = Math.min(xmin, x);
055                                ymin = Math.min(ymin, y);
056                                xmax = Math.max(xmax, x);
057                                ymax = Math.max(ymax, y);
058                        }
059                        
060                        log.info("xmin:" + xmin);
061                        log.info("ymin:" + ymin);
062                        log.info("xmax:" + xmax);
063                        log.info("ymax:" + ymax);
064                        
065                        reader.close();
066                        
067                        log.info("This took " + (System.currentTimeMillis()-startTime)/1000 + " seconds (" + (System.currentTimeMillis()-startTime)/6000 + " minutes).");
068                        
069                } catch (FileNotFoundException e) {
070                        e.printStackTrace();
071                } catch (IOException e) {
072                        e.printStackTrace();
073                }
074        }
075}
