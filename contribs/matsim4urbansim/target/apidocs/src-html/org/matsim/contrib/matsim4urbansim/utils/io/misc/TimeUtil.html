001package org.matsim.contrib.matsim4urbansim.utils.io.misc;
002
003import org.junit.Assert;
004
005/**
006 * see also org.matsim.core.utils.misc.Time for time conversion
007 * @author thomas
008 *
009 */
010public class TimeUtil {
011        
012        public static double convertSeconds2Minutes(double sec){
013                
014                int m = (int)(sec / 60);
015                double s = (sec % 60)/100;
016                double min = m + s;
017                
018                return min;
019        }
020
021        /**
022         * for testing only
023         * 
024         * @param args
025         */
026        public static void main(String[] args) {
027                double s1 = 3600.;
028                double s2 = 93.;
029                double s3 = 7225.;
030                
031                Assert.assertEquals(60., convertSeconds2Minutes(s1), 1e-7);
032                Assert.assertEquals(1.33, convertSeconds2Minutes(s2), 1e-7);
033                Assert.assertEquals(120.25, convertSeconds2Minutes(s3), 1e-7);
034        }
035
036}
