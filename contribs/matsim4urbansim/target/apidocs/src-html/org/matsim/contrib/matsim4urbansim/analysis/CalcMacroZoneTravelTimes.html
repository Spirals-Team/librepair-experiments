043/**
044 * @author droeder
045 *
046 */
047public class CalcMacroZoneTravelTimes implements PersonDepartureEventHandler,
048                                                                                                PersonArrivalEventHandler,
049                                                                                                ActivityStartEventHandler,
050                                                                                                ActivityEndEventHandler{
051
052        private static final Logger log = Logger
053                        .getLogger(CalcMacroZoneTravelTimes.class);
054        private HashMap<Id, LegStore> legStore;
055        private Map<Id<ActivityFacility>, ? extends ActivityFacility> allFacilities;
056        private Map<Id<ActivityFacility>, Id<ActivityFacility>> micro2MacroZone;
057        private Map<String, Matrix> mode2zoneTraveltimes;
058        private Map<String, Matrix> mode2zoneTrips;
059        private int end;
060        private int start; 
061
062        /**
063         * Generates some output for validation reasons for the brussels-scenario. Note it might NOT work for other scenarios...
064         * @param map
065         * @param micro2MacroZone
066         */
067        public CalcMacroZoneTravelTimes(Map<Id<ActivityFacility>, ? extends ActivityFacility> map, Map<Id<ActivityFacility>, Id<ActivityFacility>> micro2MacroZone, int startHour, int endHour) {
068                this.legStore = new HashMap<Id, LegStore>();
069                this.allFacilities = map;
070                this.micro2MacroZone = micro2MacroZone;
071                this.mode2zoneTraveltimes = new HashMap<String, Matrix>();
072                this.mode2zoneTrips = new HashMap<String, Matrix>();
073                this.start = startHour;
074                this.end = endHour;
075        }
076
077        @Override
078        public void reset(int iteration) {
079
080        }
081
082        /**
083         * @param outputPath
084         */
085        public void writeOutput(String outputPath) {
086                //write traveltimes
087                for(java.util.Map.Entry<String, Matrix> e: this.mode2zoneTraveltimes.entrySet()){
088                        BufferedWriter writer = IOUtils.getBufferedWriter(outputPath + "/travelTimesAndTrips_" + String.valueOf(start) + "-" + String.valueOf(end) + "_"+ e.getKey() + ".csv");
089                        
090                        try {
091                                writer.write("fromZone;toZone;totalTravelTime[s];Trips[#]; avgTravelTimes[s];");
092                                writer.newLine();
093                                for(ArrayList<Entry> ee:  e.getValue().getFromLocations().values()){
094                                        for(Entry eee: ee){
095                                                writer.write(eee.getFromLocation() + ";" + eee.getToLocation() + ";");
096                                                writer.write(eee.getValue() + ";");
097                                                writer.write(this.mode2zoneTrips.get(e.getKey()).getEntry(eee.getFromLocation(), eee.getToLocation()).getValue() + ";");
098                                                writer.write(eee.getValue()/this.mode2zoneTrips.get(e.getKey()).getEntry(eee.getFromLocation(), eee.getToLocation()).getValue() + ";");
099                                                writer.newLine();
100                                        }
101                                }
102                                writer.flush();
103                                writer.close();
104                                log.info(outputPath + "/travelTimesAndTrips_" + e.getKey() + ".csv written...");
105                        } catch (IOException ioe) {
106                                ioe.printStackTrace();
107                        }
108                }
109                //write modeshares
110//              BufferedWriter writer = IOUtils.getBufferedWriter(outputPath + "/modeshares.csv");
111        }
112
113        @Override
114        public void handleEvent(ActivityEndEvent event) {
115                // handle only trips that start between 6 and 10 am...
116                if((event.getTime() < (this.start*3600)) || (event.getTime() >= (this.end * 3600))) return;
117                LegStore ls = new LegStore();
118                ActivityFacility fac = allFacilities.get(event.getFacilityId());
119                ls.setFromFacility(((Id) fac.getCustomAttributes().get(InternalConstants.ZONE_ID)));
120                this.legStore.put(event.getPersonId(), ls);
121        }
122
123        @Override
124        public void handleEvent(PersonDepartureEvent event) {
125                // check if agents exists...
126                if(!this.legStore.containsKey(event.getPersonId())) return;
127                this.legStore.get(event.getPersonId()).setDepTime(event.getTime());
128                this.legStore.get(event.getPersonId()).setMode(event.getLegMode());
129        }
130
131        @Override
132        public void handleEvent(PersonArrivalEvent event) {
133                // check if agents exists...
134                if(!this.legStore.containsKey(event.getPersonId())) return;
135                this.legStore.get(event.getPersonId()).setArrTime(event.getTime());
136        }
137        
138        @Override
139        public void handleEvent(ActivityStartEvent event) {
140                // check if agents exists...
141                if(!this.legStore.containsKey(event.getPersonId())) return;
142                LegStore ls = this.legStore.remove(event.getPersonId());
143                ActivityFacility fac = allFacilities.get(event.getFacilityId());
144                ls.setToFacility(((Id) fac.getCustomAttributes().get(InternalConstants.ZONE_ID)));
145                computeLeg(ls);
146        }
147
148        /**
149         * @param ls
150         */
151        private void computeLeg(LegStore ls) {
152                Matrix tt, trip;
153                // get the correct tt-matrix or create a new one if necessary
154                tt = this.mode2zoneTraveltimes.get(ls.getMode());
155                if(tt == null){
156                        tt = new Matrix(ls.getMode()+ "_traveltimes", ls.getMode() + " traveltimes");
157                        init(tt);
158                        this.mode2zoneTraveltimes.put(ls.getMode(), tt);
159                }
160                // same for the trip-matrix
161                trip = this.mode2zoneTrips.get(ls.getMode());
162                if(trip == null){
163                        trip = new Matrix(ls.getMode()+ "_trips", ls.getMode() + " trips");
164                        init(trip);
165                        this.mode2zoneTrips.put(ls.getMode(), trip);
166                }
167                String from, to;
168                // get the macrozone-ids... maybe doublecheck here, if the id exists...
169                from = this.micro2MacroZone.get(ls.getFromFac()).toString();
170                to = this.micro2MacroZone.get(ls.getToFac()).toString();
171                // update the trip-matrix-value
172                Entry ee = trip.getEntry(from, to);
173                if(ee == null) ee = trip.setEntry(from, to, 0);
174                trip.setEntry(from, to, ee.getValue() + 1);
175                // update the the tt-matrix-value
176                ee = tt.getEntry(from, to);
177                if(ee == null) ee = tt.setEntry(from, to, 0);
178                tt.setEntry(from, to, ee.getValue() + ls.getTravelTime());
179        }
180
181        /**
182         * initialize the matrix to get it sorted in a proper way...
183         * @param matrix
184         */
185        private void init(Matrix matrix) {
186                for(int i = 1; i < 8; i++){
187                        for(int j = 1; j < 8; j++){
188                                matrix.createAndAddEntry(Integer.toString(i), Integer.toString(j), 0.);
189                        }
190                }
191        }
192        
193        
194}
195class LegStore{
196        private Id<ActivityFacility> fromFac;
197        private Double depTime;
198        private Double arrTime;
199        private Id<ActivityFacility> toFacility;
200        private String mode;
201
202        public LegStore(){
203                
204        }
205        
206        public void setFromFacility(Id<ActivityFacility> fromFac){
207                this.fromFac = fromFac;
208        }
209        
210        public void setDepTime(Double time){
211                this.depTime = time;
212        }
213        
214        public void setArrTime(Double time){
215                this.arrTime = time;
216        }
217        
218        public void setToFacility(Id<ActivityFacility> toFac){
219                this.toFacility = toFac;
220        }
221        
222        public void setMode(String mode){
223                this.mode = mode;
224        }
225        
226        public Double getTravelTime(){
227                return (this.arrTime - this.depTime);
228        }
229        
230        public Id getToFac(){
231                return this.toFacility;
232        }
233        
234        public Id getFromFac(){
235                return this.fromFac;
236        }
237        
238        public String getMode(){
239                return this.mode;
240        }
241}
