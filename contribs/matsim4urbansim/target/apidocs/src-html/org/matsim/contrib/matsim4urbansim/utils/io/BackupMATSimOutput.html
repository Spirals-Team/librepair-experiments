100        /**
101         * Preparing hot start: Copying recent matsim plans file to a specified location (Matsim config).
102         *                                              Matsim will check this location for plans file on run and activates hot start if the plans file is there
103         */
104        public static void prepareHotStart(Scenario scenario){
105                
106                M4UControlerConfigModuleV3 module = ConfigurationUtils.getMATSim4UrbaSimControlerConfigModule(scenario);
107                UrbanSimParameterConfigModuleV3 uspModule = M4UConfigUtils.getUrbanSimParameterConfigAndPossiblyConvert(scenario.getConfig()) ;
108                
109                String hotStartFile = module.getHotStartPlansFileLocation().trim();
110                if(exists(hotStartFile)){
111                        
112                        String plansFile = uspModule.getMATSim4OpusOutput() + InternalConstants.GENERATED_PLANS_FILE_NAME;
113                        try{
114                                log.info("Preparing hot start for next MATSim run ...");
115                                FileCopy.fileCopy(new File(plansFile), new File(hotStartFile));
116                        } catch (Exception e) {
117                                log.error("Error while copying plans file, i. e. hot start will not work!");
118                                e.printStackTrace();
119                        }
120                        
121                        log.info("Hot start preparation successful!");
122                }else{
123                        log.info("can not prepare hotStart. hotstart-file does not exist: " + hotStartFile);
124                }
125        }
126
127        /**
128         * @param hotStartFile
129         * @return
130         */
131        private static boolean exists(String hotStartFile) {
132                if(hotStartFile.endsWith(".xml") || hotStartFile.endsWith(".xml.gz")){
133                        File f = new File(hotStartFile);
134                        String fileName = f.getName();
135                        String dir = f.getAbsolutePath().replaceAll(fileName, "");
136                        File directory = new File(dir);
137                        if(directory.exists()){
138                                return true;
139                        }
140                }
141                return false;
142        }
