065        static void initAccessibilityConfigGroupParameters(MatsimConfigType matsim4urbansimConfigPart1, Config config){
066
067                // get every single matsim4urbansim/urbansimParameter
068                
069                // add accessibility parameter
070                AccessibilityConfigGroup acm = ConfigUtils.addOrGetModule(config, AccessibilityConfigGroup.GROUP_NAME, AccessibilityConfigGroup.class);
071                
072                acm.setCellSizeCellBasedAccessibility( matsim4urbansimConfigPart1.getCellSize().longValue() );
073                
074                // by shape file
075                if(matsim4urbansimConfigPart1.isAccessibilityComputationAreaFromShapeFile()){
076                        
077                        acm.setAreaOfAccessibilityComputation(AreaOfAccesssibilityComputation.fromShapeFile);
078
079                        if(matsim4urbansimConfigPart1.getStudyAreaBoundaryShapeFile()  != null &&
080                          (new File(matsim4urbansimConfigPart1.getStudyAreaBoundaryShapeFile().getInputFile())).exists())
081                                acm.setShapeFileCellBasedAccessibility( matsim4urbansimConfigPart1.getStudyAreaBoundaryShapeFile().getInputFile() );
082                        else
083                                throw new RuntimeException("Study area boundary shape file not found! Given shape file location:" 
084                        + matsim4urbansimConfigPart1.getStudyAreaBoundaryShapeFile());
085                }
086                // by bounding box
087                if(matsim4urbansimConfigPart1.isAccessibilityComputationAreaFromBoundingBox()){
088                        
089                        acm.setAreaOfAccessibilityComputation(AreaOfAccesssibilityComputation.fromBoundingBox);
090                        
091                        acm.setBoundingBoxBottom(matsim4urbansimConfigPart1.getBoundingBoxBottom()) ;
092                        acm.setBoundingBoxTop(matsim4urbansimConfigPart1.getBoundingBoxTop()) ;
093                        acm.setBoundingBoxLeft(matsim4urbansimConfigPart1.getBoundingBoxLeft()) ;
094                        acm.setBoundingBoxRight(matsim4urbansimConfigPart1.getBoundingBoxRight()) ;
095                }
096                // by network
097                if(matsim4urbansimConfigPart1.isAccessibilityComputationAreaFromNetwork()){
098                        
099                        acm.setAreaOfAccessibilityComputation(AreaOfAccesssibilityComputation.fromNetwork);
100                }
101        }

105         * @param matsim4urbansimConfigPart1

108        static void initM4UControlerConfigModuleV3Parameters(MatsimConfigType matsim4urbansimConfigPart1, Config config){
109                
110                M4UControlerConfigModuleV3 module = getMATSim4UrbaSimControlerConfigAndPossiblyConvert(config);
111                
112                module.setUrbansimZoneRandomLocationDistributionRadius(matsim4urbansimConfigPart1.getUrbansimZoneRandomLocationDistributionByRadius());
113                module.setUrbansimZoneRandomLocationDistributionShapeFile(matsim4urbansimConfigPart1.getUrbansimZoneRandomLocationDistributionByShapeFile());
114                module.setUsingShapefileLocationDistribution( (new File(matsim4urbansimConfigPart1.getUrbansimZoneRandomLocationDistributionByShapeFile())).exists() );
115                log.info("This message affects UrbanSim ZONE applications only:");
116                if(module.usingShapefileLocationDistribution()){
117                        log.info("Found a zone shape file: " + module.getUrbansimZoneRandomLocationDistributionShapeFile());
118                        log.info("This activates the distribution of persons within a zone using the zone boundaries of this shape file."); 
119                }
120                else{
121                        log.info("Persons are distributed within a zone using the zone centroid and a radius of " + module.getUrbanSimZoneRadiusLocationDistribution() + " meter.");
122                        log.info("In order to use exact zone boundaries for your sceanrio provide a zone shape file and enter the file location in the external MATSim config file as follows:");
123                        log.info("<module name=\"matsim4urbansimParameter\" >");
124                        log.info("<param name=\"urbanSimZoneShapefileLocationDistribution\" value=\"/path/to/shapeFile\" />");
125                        log.info("</module>");
126                }
127                if ( matsim4urbansimConfigPart1.getWarmStartPlansFile()==null ) {
128                        module.setWarmStart(false) ;
129                } else if ( matsim4urbansimConfigPart1.getWarmStartPlansFile() == null ) {
130                        module.setWarmStart(false) ;
131                } else {
132                        module.setWarmStart( (new File(matsim4urbansimConfigPart1.getWarmStartPlansFile().getInputFile())).exists() );
133                }
134                module.setWarmStartPlansLocation(matsim4urbansimConfigPart1.getWarmStartPlansFile().getInputFile());
135                module.setHotStart(matsim4urbansimConfigPart1.isUseHotStart());
136                module.setHotStartPlansFileLocation(matsim4urbansimConfigPart1.getHotStartPlansFile().getInputFile());
137                
138                // setting right plans file
139                if(module.usingWarmStart()){
140                        PlansConfigGroup pcg = config.plans();
141                        boolean setHotSart = false;
142                        
143                        // check if hot start is switched on and if the hot start plans file file is available
144                        if(module.usingHotStart()){
145                                File f = new File(module.getHotStartPlansFileLocation());
146                                if(f.exists()){
147                                        log.info("Hot start flag is set to true. Hot start plans file found at : " + module.getHotStartPlansFileLocation());
148                                        pcg.setInputFile(module.getHotStartPlansFileLocation());
149                                        log.info("Setting hot start plans file done!");
150                                        setHotSart = true;
151                                }
152                                else{
153                                        log.warn("Hot start flag is set to true. However, the hot start plans file is not found at given location : " + module.getHotStartPlansFileLocation());
154                                        log.warn("Warm start will be used instead!");
155                                        log.warn("This is be ok for the first time when MATSim is called by UrbanSim. Next time MATSim should start with hot start.");
156                                }
157                        }
158                        
159                        if(!setHotSart){
160                                File f = new File(module.getWarmStartPlansFileLocation());
161                                if(f.exists()){
162                                        log.info("Warm start plans file found at : " + module.getWarmStartPlansFileLocation());
163                                        pcg.setInputFile(module.getWarmStartPlansFileLocation());
164                                        log.info("Setting warm start plans file done!");
165                                }
166                                else{
167                                        throw new RuntimeException("Given warm start plans file not found : " + module.getWarmStartPlansFileLocation() + " Please check the path in your config file!");
168                                }
169                        }
170                }
171        }

278        /**
279         * setting qsim
280         * @param matsim4urbansimConfigPart2 TODO
281         * @param config TODO
282         */
283        static void initQSim(Matsim4UrbansimType matsim4urbansimConfigPart2 , Config config){
284                log.info("Setting QSimConfigGroup to config...");
285
286                QSimConfigGroup qsimCG = config.qsim();
287
288                double popSampling = matsim4urbansimConfigPart2.getPopulationSamplingRate();
289                // Adapting the storageCapFactor has the following reason:
290                // Too low SorageCapacities especially with small sampling 
291                // rates can (eg 1%) lead to strong backlogs on the traffic network. 
292                // This leads to an unstable behavior of the simulation (by breakdowns 
293                // during the learning progress).
294                // The correction fetch factor introduced here raises the 
295                // storage capacity at low sampling rates and becomes flatten 
296                // with increasing sampling rates (at a 100% sample, the 
297                // storage capacity == 1).                      tnicolai nov'11
298                if(popSampling <= 0.){
299                        // yyyyyy how can this happen???? kai, apr'13
300                        double popSamplingBefore = popSampling ;
301                        popSampling = 0.01;
302                        log.warn("Raised popSampling rate from " + popSamplingBefore + 
303                                        " to " + popSampling + " to to avoid errors while calulating the correction factor ...");
304                }
305                
306                log.info("FlowCapFactor and StorageCapFactor are adapted to the population sampling rate (sampling rate = " + popSampling + ").");
307                // setting FlowCapFactor == population sampling rate (no correction factor needed here)
308                qsimCG.setFlowCapFactor( popSampling ); 
309                
310                        
311                qsimCG.setStorageCapFactor( popSampling * Math.pow(popSampling, -0.25) ) ;   // same as: sample / Math.sqrt(Math.sqrt(sample))
312                // yy and same as Math.pow( popSampling, 0.75 ), isn't it? 
313
314                qsimCG.setRemoveStuckVehicles( false );
315                qsimCG.setStuckTime(10.);
316                qsimCG.setEndTime(30.*3600.); // 30h
317
318                log.info("...done!");
319        }

321        static void initStrategy(Config config){
322                // It is no longer possible to configure any of this from the urbansim side.  However, something still needs to be set for
323                // matsim, which is defined here.  
324                // Unfortunately, this makes it a bit tricky to use your own matsim config, since depending on the strategy settings id one
325                // either adds to these strategy settings, or overrides them, or modifies them.  It is, however, also unclear what else to do; this

328                log.info("Setting StrategyConfigGroup to config...");
329
330                config.strategy().setMaxAgentPlanMemorySize( 5 );
331
332                StrategyConfigGroup.StrategySettings changeExpBeta = new StrategyConfigGroup.StrategySettings(Id.create(1, StrategySettings.class));
333                changeExpBeta.setStrategyName(DefaultPlanStrategiesModule.DefaultSelector.ChangeExpBeta.toString());
334                changeExpBeta.setWeight( 0.8 ) ;
335                config.strategy().addStrategySettings(changeExpBeta);
336
337                StrategyConfigGroup.StrategySettings timeAlocationMutator = new StrategyConfigGroup.StrategySettings(Id.create(2, StrategySettings.class));
338                timeAlocationMutator.setStrategyName(DefaultPlanStrategiesModule.DefaultStrategy.TimeAllocationMutator.toString());
339                timeAlocationMutator.setWeight( 0.1 ); 
340//              timeAlocationMutator.setDisableAfter(disableStrategyAfterIteration(config)); // now matsim default
341                config.strategy().addStrategySettings(timeAlocationMutator);
342                config.timeAllocationMutator().setMutationRange(7200.) ;
343
344                StrategyConfigGroup.StrategySettings reroute = new StrategyConfigGroup.StrategySettings(Id.create(3, StrategySettings.class));
345                reroute.setStrategyName(DefaultPlanStrategiesModule.DefaultStrategy.ReRoute.toString());
346                reroute.setWeight( 0.1 );
347//              reroute.setDisableAfter(disableStrategyAfterIteration(config)); // now matsim default
348                config.strategy().addStrategySettings(reroute);
349
350                log.info("...done!");
351        }

404        static Matsim4UrbansimConfigType unmarschal(String matsim4urbansimConfigFilename){
405
406                // JAXBUnmarschal reads the UrbanSim generated MATSim config, validates it against
407                // the current xsd (checks e.g. the presents and data type of parameter) and generates
408                // a Java object representing the config file.
409                JAXBUnmarshalV3 um = new JAXBUnmarshalV3();
410                
411                Matsim4UrbansimConfigType m4uConfigType = null;
412                m4uConfigType = um.unmarshal(matsim4urbansimConfigFilename);
413                
414                if(m4uConfigType == null) {
415                        throw new RuntimeException("Unmarschalling failed. SHUTDOWN MATSim!");
416                }
417
418                return m4uConfigType;
419                
420        }

479        static final void checkConfigConsistencyAndWriteToLog(Config config, final String message) {
480                String newline = System.getProperty("line.separator");// use native line endings for logfile
481                log.info(newline + newline + message + ":");
482                StringWriter writer = new StringWriter();
483                new ConfigWriter(config).writeStream(new PrintWriter(writer), newline);
484                log.info(newline + "Complete config dump:" + newline + writer.getBuffer().toString());
485                log.info("Complete config dump done.");
