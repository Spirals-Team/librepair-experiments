001package org.matsim.contrib.matsim4urbansim.utils.io;
002
003import org.matsim.api.core.v01.Coord;
004import org.matsim.api.core.v01.Id;
005import org.matsim.api.core.v01.TransportMode;
006import org.matsim.api.core.v01.population.Activity;
007import org.matsim.api.core.v01.population.Plan;
008import org.matsim.api.core.v01.population.Activity;
009import org.matsim.contrib.matsim4urbansim.constants.InternalConstants;
010import org.matsim.core.gbl.MatsimRandom;
011import org.matsim.core.population.PopulationUtils;
012import org.matsim.facilities.ActivityFacility;
013
014
015public class CreateHomeWorkHomePlan {
016        
017        /**
018         * Helper method to start a plan by inserting the home location.  This is really only useful together with "completePlanToHwh",
019         * which completes the plan, and benefits from the fact that the Strings for the "home" and the "work" act are now concentrated
020         * here.
021         *
022         * @param plan
023         * @param homeCoord
024         * @param homeLocation
025         *
026         * @author nagel
027         */
028        public static void makeHomePlan( Plan plan, Coord homeCoord, ActivityFacility homeLocation) {
029                final Coord coord = homeCoord;
030                Activity act = PopulationUtils.createAndAddActivityFromCoord(plan, InternalConstants.ACT_HOME, coord) ;
031                act.setFacilityId( homeLocation.getId() );      // tnicolai: added facility id to compute zone2zone trips
032        }
033
034        /**
035         * Helper method to complete a plan with *wh in a consistent way.  Assuming that the first activity is the home activity.
036         *
037         * @param plan
038         * @param workCoord
039         * @param jobLocation
040         *
041         * @author nagel
042         */
043        public static void completePlanToHwh ( Plan plan, Coord workCoord, ActivityFacility jobLocation ) {
044                
045                // complete the first activity (home) by setting end time. 
046                Activity act = (Activity)PopulationUtils.getFirstActivity( plan );
047
048                final double hmDpTime = (6.+2.*MatsimRandom.getRandom().nextDouble())*3600.;
049                act.setEndTime( hmDpTime ) ;
050                // (I just made this random between 6am and 8am to avoid the departure peak at 7.  In particular for pt users, it does not
051                // seem to go away. kai, apr'13)
052                // (tnicolai: make configurable: see actType1.setOpeningTime(7*3600))
053
054                // gather coordinate and facility id needed for last activity
055                Coord homeCoord = act.getCoord();
056                Id homeId = act.getFacilityId();
057                
058                // set Leg
059                PopulationUtils.createAndAddLeg( plan, TransportMode.car );
060                final Coord coord = workCoord;
061                
062                // set second activity (work)
063                act = PopulationUtils.createAndAddActivityFromCoord(plan, InternalConstants.ACT_WORK, coord);
064                act.setFacilityId( jobLocation.getId() );
065//              act.setMaximumDuration( 8.*3600. ) ; // tnicolai: make configurable: actType1.setTypicalDuration(8*60*60);
066                act.setEndTime( hmDpTime + 9.*3600. ) ; // avoid durations except when short (e.g. plugging in hybrid veh).  kai, may'13
067                
068                // set Leg
069                PopulationUtils.createAndAddLeg( plan, TransportMode.car );
070                final Coord coord1 = homeCoord;
071                
072                // set last activity (=first activity) and complete home-work-home plan.
073                PopulationUtils.createAndAddActivityFromCoord(plan, InternalConstants.ACT_HOME, coord1);
074                act = (Activity)PopulationUtils.getLastActivity(plan);
075                act.setFacilityId( homeId );
076        }
077
078}
