024package org.matsim.contrib.matsim4urbansim.gis;
025
026import java.util.Iterator;
027
028import org.matsim.api.core.v01.Coord;
029import org.matsim.api.core.v01.network.Network;
030import org.matsim.api.core.v01.network.Node;
031import org.matsim.contrib.matsim4urbansim.utils.helperobjects.ZoneObject;
032import org.matsim.core.network.NetworkUtils;
033import org.matsim.facilities.ActivityFacilitiesImpl;
034import org.matsim.facilities.ActivityFacility;
035
036
037/**
038 * @author thomas
039 *
040 */
041public class ZoneUtil {
042
043        /**
044         * Mapping zone centroids to their nearest network node and
045         * creating an array containing the follwoing information:
046         * zone id, zone coordinate (centroid) and nearest node 
047         * 
048         * @param network
049         */
050        public static ZoneObject[] mapZoneCentroid2NearestNode(final ActivityFacilitiesImpl zones, final Network network) {
051                
052                assert( network != null );
053                assert( zones != null );
054                int numberOfZones = zones.getFacilities().values().size();
055                ZoneObject zoneArray[] = new ZoneObject[numberOfZones];
056                Iterator<? extends ActivityFacility> zonesIterator = zones.getFacilities().values().iterator();
057
058                int counter = 0;
059                while( zonesIterator.hasNext() ){
060
061                        ActivityFacility zone = zonesIterator.next();
062                        assert (zone != null );
063                        assert( zone.getCoord() != null );
064                        Coord zoneCoordinate = zone.getCoord();
065                        final Coord coord = zoneCoordinate;
066                        Node networkNode = NetworkUtils.getNearestNode(network,coord);
067                        assert( networkNode != null );
068                                
069                        zoneArray[counter] = new ZoneObject(zone.getId(), zoneCoordinate, networkNode);
070                        counter++;
071                }
072                return zoneArray;
073        }
074        
075}
