024
025import org.apache.log4j.Logger;
026import org.matsim.contrib.accessibility.AccessibilityConfigGroup;
027import org.matsim.contrib.matrixbasedptrouter.MatrixBasedPtRouterConfigGroup;
028import org.matsim.contrib.matsim4urbansim.config.modules.M4UControlerConfigModuleV3;
029import org.matsim.contrib.matsim4urbansim.config.modules.UrbanSimParameterConfigModuleV3;
030import org.matsim.contrib.matsim4urbansim.utils.io.Paths;
031import org.matsim.core.config.Config;
032import org.matsim.core.config.ConfigUtils;
033import org.matsim.core.config.consistency.ConfigConsistencyChecker;
034
035/**
036 * @author nagel
037 *
038 */
039public class M4UConfigConsistencyChecker implements ConfigConsistencyChecker {
040        private static final Logger log = Logger.getLogger(M4UConfigConsistencyChecker.class);
041
042        @Override
043        public void checkConsistency(Config config) {
044                boolean problem = false ;
045                
046                @SuppressWarnings("unused")
047                M4UControlerConfigModuleV3 matsim4urbansimModule = (M4UControlerConfigModuleV3) config.getModule(M4UControlerConfigModuleV3.GROUP_NAME) ;
048                @SuppressWarnings("unused")
049                UrbanSimParameterConfigModuleV3 urbansimParameterModule = (UrbanSimParameterConfigModuleV3) config.getModule(UrbanSimParameterConfigModuleV3.GROUP_NAME ) ;
050                @SuppressWarnings("unused")
051                AccessibilityConfigGroup accessibilityConfigModule = ConfigUtils.addOrGetModule(config, AccessibilityConfigGroup.GROUP_NAME, AccessibilityConfigGroup.class) ;
052                MatrixBasedPtRouterConfigGroup ippcm = ConfigUtils.addOrGetModule(config, MatrixBasedPtRouterConfigGroup.GROUP_NAME, MatrixBasedPtRouterConfigGroup.class) ;
053                
054                if ( ippcm.isUsingTravelTimesAndDistances() ) {
055                        if ( !ippcm.isUsingPtStops() ) {
056                        problem = true ;
057                        log.error("As far as I understand, improved pseudo pt will not work when pt stops are switched off.  There is no obvious conceptual " +
058                                        "reason for this; it is just how it is currently implemented.  Either switch off " 
059                                        + MatrixBasedPtRouterConfigGroup.PT_TRAVEL_TIMES_AND_DISTANCES_SWITCH + " or switch on " 
060                                        + MatrixBasedPtRouterConfigGroup.PT_STOPS + " plus provide a pt stops file.  Aborting ...") ;
061                        }
062                }
063                
064                if ( ippcm.isUsingPtStops() ) {
065                        if ( !ippcm.getPtStopsInputFile().isEmpty() && !Paths.pathExsits( ippcm.getPtStopsInputFile() )) {
066                                problem = true ;
067                                log.error( MatrixBasedPtRouterConfigGroup.USING_PT_STOPS + " is set to true but pt stops file not found.  Aborting ... ") ;
068                        }
069                        
070                        if ( ippcm.isUsingTravelTimesAndDistances() ) {
071                                if ( !ippcm.getPtTravelTimesInputFile().isEmpty() && !Paths.pathExsits( ippcm.getPtTravelTimesInputFile() )) {
072                                        problem = true ;
073                                        log.error( MatrixBasedPtRouterConfigGroup.PT_TRAVEL_TIMES_AND_DISTANCES_SWITCH + " is set to true but pt travel times file not found.  Aborting ... ") ;
074                                }
075                                if ( !ippcm.getPtTravelDistancesInputFile().isEmpty() && !Paths.pathExsits( ippcm.getPtTravelDistancesInputFile() )) {
076                                        problem = true ;
077                                        log.error( MatrixBasedPtRouterConfigGroup.PT_TRAVEL_TIMES_AND_DISTANCES_SWITCH + " is set to true but pt distances file not found.  Aborting ... ") ;
078                                }
079                        }
080                }
081
082                if ( config.network().getInputFile().isEmpty() ) {
083                        problem = true ;
084                        log.error("Missing MATSim network! The network must be specified either directly in the " +
085                                        "MATSim4UrbanSim configuration or in an external MATSim configuration.");
086                }
087                
088//              if ( accessibilityConfigModule.getLogitScaleParameter() != 1. ) {
089//                      log.warn("using a logit scale parameter != 1. for accessibility computation.  Not recommended; will make interpretation of results more indirect") ;
090//              }
091//              if ( !accessibilityConfigModule.isUsingCarParametersFromMATSim() ) {
092//                      problem = true ;
093//                      log.error("using car beta parameters not from matsim currently not allowed since interpretation not fully clear") ;
094//              }
095//              if ( !accessibilityConfigModule.isUsingPtParametersFromMATSim() ) {
096//                      problem = true ;
097//                      log.error("using pt beta parameters not from matsim currently not allowed since interpretation not fully clear") ;
098//              }
099//              if ( !accessibilityConfigModule.isUsingWalkParametersFromMATSim() ) {
100//                      problem = true ;
101//                      log.error("using walk beta parameters not from matsim currently not allowed since interpretation not fully clear") ;
102//              }
103//              if ( !accessibilityConfigModule.isUsingBikeParametersFromMATSim() ) {
104//                      problem = true ;
105//                      log.error("using bike beta parameters not from matsim currently not allowed since interpretation not fully clear") ;
106//              }
107                
108                if ( problem ) {
109                        throw new RuntimeException("serious problem in MATSim4UrbanSimConfigConsistencyChecker; aborting ...") ;
110                }
111        }
112
113}
