035/**
036 * @author thomas
037 * 
038 * improvements dec'11:
039 * - adjusting flow- and storage capacities to population sample rate. The
040 * storage capacity includes a fetch factor to avoid backlogs and network breakdown
041 * for small sample rates.
042 * 
043 * improvements jan'12:
044 * - initGlobalSettings sets the number of available processors in the 
045 *      GlobalConfigGroup to speed up MATSim computations. Before that only
046 *      2 processors were used even if there are more.
047 * 
048 * improvements feb'12:
049 * - setting mutationrange = 2h for TimeAllocationMutator (this shifts the departure times)
050 * 
051 * improvements march'12:
052 * - extended the MATSim4UrbanSim configuration, e.g. a standard MATSim config can be loaded
053 * 
054 * improvements aug'12
055 * - extended the MATSim4UrbanSim configuration: 
056 *   - added a switch to select between between radius or shape-file distribution of locations within a zone
057 *   - added a field "project_name" of the UrbanSim scenario as an identifier
058 *   - added a time-of-a-day parameter
059 *   - added beta parameter for mode bike
060 *   
061 * improvements/changes oct'12
062 * - switched to qsim
063 * 
064 * changes dec'12
065 * - switched matsim4urbansim config v2 parameters from v3 are out sourced into external matsim config
066 * - introducing pseudo pt (configurable via external MATSim config)
067 * - introducing new strategy module "changeSingeLegMode" (configurable via external MATSim config)
068 * 
069 * changes jan'13
070 * - added attributes for pt stop, travel times and distances input files 
071 *
072 * changes march'13
073 * - now, the external config overwrites all parameter settings that overlap with the settings 
074 *   made in the MATSim4UrbanSim config (see also test cases in "ConfigLoadingTest")
075 */
076public class M4UConfigurationConverterV4 {
077        // logger
078        static final Logger log = Logger.getLogger(M4UConfigurationConverterV4.class);
079        
080        // MATSim config
081        private Config config = null;
082
083        // JAXB representation of matsim4urbansim config
084        // private final MatsimConfigType matsim4urbansimConfig ;
085        private final Matsim4UrbansimConfigType matsim4urbansimConfig;
086
087        /**
088         * constructor
089         * 
090         * @param config stores MATSim parameters
091         * @param matsim4urbansimConfigFilename path to matsim config file
092         */
093        public M4UConfigurationConverterV4(final String matsim4urbansimConfigFilename){
094                this.matsim4urbansimConfig = M4UConfigUtils.unmarschal(matsim4urbansimConfigFilename); // loading and initializing MATSim config        
095                Assert.assertTrue( this.matsim4urbansimConfig!= null ) ;
096        }
097        
098        /**
099         * Transferring all parameter from matsim4urbansim config to internal MATSim config/scenario
100         * @return boolean true if initialization successful
101         */
102        public boolean init(){
103
104                // creates an empty config to be filled by settings from the MATSim4UrbanSim and external config files
105                this.config = M4UConfigUtils.createEmptyConfigWithSomeDefaults();
106                Assert.assertTrue( this.config != null ) ;
107
108                // get root elements from JAXB matsim4urbansim config object
109                MatsimConfigType matsim4urbansimConfigPart1 = this.matsim4urbansimConfig.getMatsimConfig();
110                Assert.assertTrue( matsim4urbansimConfigPart1 != null ) ;
111                
112                Matsim4UrbansimType matsim4urbansimConfigPart2 = this.matsim4urbansimConfig.getMatsim4Urbansim();
113                Assert.assertTrue( matsim4urbansimConfigPart2 != null ) ;
114
115                // int MATSimConfigType parameters
116                M4UConfigUtils.initAccessibilityConfigGroupParameters(matsim4urbansimConfigPart1, config);
117                M4UConfigUtils.initM4UControlerConfigModuleV3Parameters(matsim4urbansimConfigPart1, config);
118                M4UConfigUtils.initUrbanSimParameterConfigModuleV3Parameters(matsim4urbansimConfigPart2, config);
119                
120                M4UConfigUtils.initNetwork(matsim4urbansimConfigPart1, config);
121                M4UConfigUtils.initControler(matsim4urbansimConfigPart1, config);
122                M4UConfigUtils.initPlanCalcScore(matsim4urbansimConfigPart1, config);
123                M4UConfigUtils.initQSim(matsim4urbansimConfigPart2, config);
124
125                M4UConfigUtils.initStrategy(config) ;
126                // note: ending innovation after 80% of iterations is now switched on in "createEmptyConfigWithSomeDefaults" above.
127
128//              MatrixBasedPtRouterConfigUtils.initMatrixBasedPtRouterParameters(matsim4urbansimConfigPart3, config);
129                // (was, in my view, not doing anything besides consistency checking, which is now in the config group consistency
130                // checker. kai, jul'13)
131
132                // loading the external MATSim config in to the initialized config
133                M4UConfigUtils.loadExternalConfigAndOverwriteMATSim4UrbanSimSettings(matsim4urbansimConfigPart1.getExternalMatsimConfig().getInputFile(), config);
134                // (by the design, the matsim xml config reader over-writes only entries which are explicitly mentioned in the external config)
135
136                // show final settings
137                M4UConfigUtils.printUrbanSimParameterSettings( M4UConfigUtils.getUrbanSimParameterConfigAndPossiblyConvert(config) );
138                M4UConfigUtils.printMATSim4UrbanSimControlerSettings( M4UConfigUtils.getMATSim4UrbaSimControlerConfigAndPossiblyConvert(config) );
139
140                config.addConfigConsistencyChecker( new VspConfigConsistencyCheckerImpl() ) ;
141                config.addConfigConsistencyChecker( new M4UConfigConsistencyChecker() ) ;
142
143                M4UConfigUtils.checkConfigConsistencyAndWriteToLog(config, "at the end of the matsim4urbansim config converter") ;
144
145                return true;
146        }
147
148        public Config getConfig(){
149                        return this.config;
150        }
151        
152}
