118        /**
119         *      calculating and dumping the following outcomes:
120         *  - zone2zone impedances including travel times, travel costs and am walk time
121         *  - logsum computation of workplace accessibility
122         */
123        @Override
124        public void notifyShutdown(ShutdownEvent event) {
125                log.info("Entering notifyShutdown ..." );
126
127                int benchmarkID = this.benchmark.addMeasure("zone-to-zone impedances");
128                
129                // get the controller and scenario
130                MatsimServices controler = event.getServices();
131                Scenario sc = controler.getScenario();
132                
133                double samplingRate = ConfigurationUtils.getUrbanSimParameterConfigModule(sc).getPopulationSampleRate();
134                double inverseOfSamplingRate = 1/samplingRate;
135                
136                double walkSpeedMeterPerMinute = sc.getConfig().plansCalcRoute().getTeleportedModeSpeeds().get(TransportMode.walk) * 60.; // corresponds to 5 km/h
137                double bikeSpeedMeterPerMinute = sc.getConfig().plansCalcRoute().getTeleportedModeSpeeds().get(TransportMode.bike) * 60.; // corresponds to 15 km/h 
138                
139                // get the free-speed car travel times (in seconds)
140                TravelTime ttf = new FreeSpeedTravelTime() ;
141                TravelDisutility tdFree = controler.getTravelDisutilityFactory().createTravelDisutility(ttf ) ;
142                LeastCostPathTreeExtended lcptExtFreeSpeedCarTrvelTime = new LeastCostPathTreeExtended( ttf, tdFree, (RoadPricingSchemeImpl) controler.getScenario().getScenarioElement(RoadPricingScheme.ELEMENT_NAME) ) ;
143                
144                // get the congested car travel time (in seconds)
145                TravelTime ttc = controler.getLinkTravelTimes(); // congested
146                TravelDisutility tdCongested = controler.getTravelDisutilityFactory().createTravelDisutility(ttc ) ;
147                LeastCostPathTreeExtended  lcptExtCongestedCarTravelTime = new LeastCostPathTreeExtended(ttc, tdCongested, (RoadPricingSchemeImpl) controler.getScenario().getScenarioElement(RoadPricingScheme.ELEMENT_NAME) ) ;
148
149                // get travel distance (in meter)
150                LeastCostPathTree lcptTravelDistance             = new LeastCostPathTree( ttf, new TravelDistanceCalculator());
151
152        Network network = (Network) controler.getScenario().getNetwork();
153                double depatureTime = 8.*3600;
154                
155                // od-trip matrix (zonal based)
156                Matrix originDestinationMatrix = new Matrix("tripMatrix", "Zone to Zone origin destination trip matrix");
157                
158                try {
159                        // creating zone-to-zone impedance matrix with header
160                        UrbanSimParameterConfigModuleV3 module = (UrbanSimParameterConfigModuleV3) event.getServices().
161                                        getConfig().getModule(UrbanSimParameterConfigModuleV3.GROUP_NAME);
162                        BufferedWriter travelDataWriter = initZone2ZoneImpedaceWriter(module.getMATSim4OpusTemp() + FILE_NAME); 
163                        
164                        computeZoneToZoneTrips(sc, originDestinationMatrix, depatureTime);
165
166                        log.info("Computing and writing zone2zone impedance matrix ..." );
167
168                        // init array with zone informations
169                        ZoneObject[] zones = ZoneUtil.mapZoneCentroid2NearestNode(this.zones, network);
170                        // init progress bar
171                        ProgressBar bar = new ProgressBar( zones.length );
172                        log.info("Processing " + zones.length + " UrbanSim zones ...");
173
174                        // main for loop, dumping out zone2zone impedances (travel times)
175                        for(int fromZoneIndex = 0; fromZoneIndex < zones.length; fromZoneIndex++){
176                                
177                                // progress bar
178                                bar.update();
179                                
180                                // get nearest network node and zone id for origin zone
181                                Node fromNode = zones[fromZoneIndex].getNearestNode();
182                                Id originZoneID = zones[fromZoneIndex].getZoneID();
183                                
184                                // run dijksrtra for current node as origin
185                                lcptExtCongestedCarTravelTime.calculateExtended(network, fromNode, depatureTime);
186                                lcptExtFreeSpeedCarTrvelTime.calculateExtended(network, fromNode, depatureTime);
187                                lcptTravelDistance.calculate(network, fromNode, depatureTime);
188                                
189                                for(int toZoneIndex = 0; toZoneIndex < zones.length; toZoneIndex++){
190                                        
191                                        // get nearest network node and zone id for destination zone
192                                        Node toNode = zones[toZoneIndex].getNearestNode();
193                                        Id destinationZoneID = zones[toZoneIndex].getZoneID();
194                                        
195                                        // free speed car travel times in minutes
196                                        double arrivalTimeFS = lcptExtFreeSpeedCarTrvelTime.getTree().get( toNode.getId() ).getTime();
197                                        double freeSpeedTravelTime_min = (arrivalTimeFS - depatureTime) / 60.;
198                                        // we guess that any value less than 1.2 leads to errors on the UrbanSim side
199                                        // since ln(0) is not defined or ln(1) = 0 causes trouble as a denominator ...
200                                        if(freeSpeedTravelTime_min < 1.2)
201                                                freeSpeedTravelTime_min = 1.2;
202                                        
203                                        // get generalized travel cost 
204                                        double travelDisutility_util = lcptExtCongestedCarTravelTime.getTree().get( toNode.getId() ).getCost();
205                                        // we guess that any value less than 1.2 leads to errors on the UrbanSim side
206                                        // since ln(0) is not defined or ln(1) = 0 causes trouble as a denominator ...
207                                        // if(travelDisutility_util < 1.2)
208                                        //      travelDisutility_util = 1.2;
209                                        
210                                        // get congested arrival time
211                                        double arrivalTimeCC = lcptExtCongestedCarTravelTime.getTree().get( toNode.getId() ).getTime();
212                                        // congested car travel times in minutes
213                                        double congestedTravelTime_min = (arrivalTimeCC - depatureTime) / 60.;
214                                        // we guess that any value less than 1.2 leads to errors on the UrbanSim side
215                                        // since ln(0) is not defined or ln(1) = 0 causes trouble as a denominator ...
216                                        if(congestedTravelTime_min < 1.2)
217                                                congestedTravelTime_min = 1.2;
218
219                                        // travel distance in meter
220                                        double travelDistance_meter = lcptTravelDistance.getTree().get( toNode.getId() ).getCost();
221                                        double bikeTravelTime_min = travelDistance_meter / bikeSpeedMeterPerMinute;
222                                        // we guess that any value less than 1.2 leads to errors on the UrbanSim side
223                                        // since ln(0) is not defined or ln(1) = 0 causes trouble as a denominator ...
224                                        if(bikeTravelTime_min < 1.2)
225                                                bikeTravelTime_min = 1.2;
226                                        double walkTravelTime_min = travelDistance_meter / walkSpeedMeterPerMinute;
227                                        // we guess that any value less than 1.2 leads to errors on the UrbanSim side
228                                        // since ln(0) is not defined or ln(1) = 0 causes trouble as a denominator ...
229                                        if(walkTravelTime_min < 1.2 )
230                                                walkTravelTime_min = 1.2 ;
231                                        // changed the above values from 12 to 1.2 in jan'15. kai
232                                        
233                                        // pt travel times in minutes
234                                        double ptTravelTime_min = -1.;
235                                        if(this.ptMatrix != null){
236                                                ptTravelTime_min = this.ptMatrix.getTotalTravelTime_seconds(fromNode.getCoord(), toNode.getCoord()) / 60.;
237                                                // we guess that any value less than 1.2 leads to errors on the UrbanSim side
238                                                // since ln(0) is not defined or ln(1) = 0 causes trouble as a denominator ...
239                                                if(ptTravelTime_min < 1.2)
240                                                        ptTravelTime_min = 1.2;
241                                        }
242
243                                        // query trips in OD Matrix
244                                        double trips = 0.0;
245                                        Entry e = originDestinationMatrix.getEntry( originZoneID.toString(), destinationZoneID.toString() );
246                                        if(e != null) // trips are scaled to 100%
247                                                trips = e.getValue() * inverseOfSamplingRate;
248                                        
249                                        // IMPORTANT: Do adapt the travel_data header in "initZone2ZoneImpedaceWriter"
250                                        //                        when changing anything at this call.
251                                        travelDataWriter.write ( originZoneID.toString()                        //origin zone id
252                                                                                + "," + destinationZoneID.toString()    //destination zone id
253                                                                                + "," + freeSpeedTravelTime_min                 //free speed travel times
254                                                                                + "," + travelDisutility_util                   //congested generalized cost
255                                                                                + "," + congestedTravelTime_min                 //congested travel times
256                                                                                + "," + bikeTravelTime_min                              //bike travel times
257                                                                                + "," + walkTravelTime_min                              //walk travel times
258                                                                                + "," + ptTravelTime_min                                //pt travel times
259                                                                                + "," + travelDistance_meter                    //network distance
260                                                                                + "," + trips);                                                 //vehicle trips (scaled to 100%)
261                                        travelDataWriter.newLine();
262                                }
263                        }
264
265                        // finish progress bar
266                        System.out.println();
267                        // flush and close writers
268                        travelDataWriter.flush();
269                        travelDataWriter.close();
270                        
271                        // copy the zones file to the outputfolder...
272                        log.info("Copying " + module.getMATSim4OpusTemp() + FILE_NAME + " to " + module.getMATSim4OpusOutput() + FILE_NAME);
273            try {
274                Files.copy(new File( module.getMATSim4OpusTemp() + FILE_NAME).toPath(), new File( module.getMATSim4OpusOutput() + FILE_NAME).toPath());
275            } catch (IOException e) {
276                throw new UncheckedIOException(e);
277            }
278
279            log.info("... done with writing travel_data.csv" );
280                        
281                        if (this.benchmark != null && benchmarkID > 0) {
282                                this.benchmark.stoppMeasurement(benchmarkID);
283                                log.info("Zone-to-zone impedance measure with " 
284                                                 + zones.length + " zones took "
285                                                 + this.benchmark.getDurationInSeconds(benchmarkID)
286                                                 + " seconds ("
287                                                 + this.benchmark.getDurationInSeconds(benchmarkID) / 60. 
288                                                 + " minutes).");
289                        }
290                } catch (FileNotFoundException e) {
291                        e.printStackTrace();
292                } catch (IOException e) {
293                        e.printStackTrace();
294                }
295                
296                log.info("... done with notifyShutdown.") ;
297        }

299        /**
300         * Returns a BufferedWriter writing a zone-to-zone impedance matrix for UrbanSim
301         * 
302         * @return BufferedWriter
303         * @throws FileNotFoundException
304         * @throws IOException
305         */
306        private BufferedWriter initZone2ZoneImpedaceWriter(String travelDataPath)
307                        throws FileNotFoundException, IOException {
308                BufferedWriter travelDataWriter = IOUtils.getBufferedWriter( travelDataPath );
309                
310                // Travel Data Header
311                travelDataWriter.write ( "from_zone_id:i4,to_zone_id:i4,vehicle_free_speed_travel_time:f4,single_vehicle_to_work_travel_cost:f4," +
312                                "am_single_vehicle_to_work_travel_time:f4,bike_time_in_minutes:f4,walk_time_in_minutes:f4,pt_time_in_minutes:f4," +
313                                "vehicle_travel_distance_in_meter:f4,am_pk_period_drive_alone_vehicle_trips:f4" ) ; 
314                // Logger.getLogger(this.getClass()).warn( "add new fields (this message is shown until all travel data attributes are updated)" );
315                travelDataWriter.newLine();
316                return travelDataWriter;
317        }

319        /**
320         * goes through all person and their plans and stores their trips into a matrix
321         * 
322         * @param sc
323         * @param originDestinationMatrix
324         */
325        private void computeZoneToZoneTrips(Scenario sc, Matrix originDestinationMatrix, double timeOfDay){
326                log.info("Computing zone2zone trip numbers ...") ;
327                // yyyy might make even more sense to do this via events.  kai, feb'11
328                Entry matrixEntry = null;
329                
330                Map<Id<ActivityFacility>, ? extends ActivityFacility> allFacilities; 
331                if(parcels == null)
332                        allFacilities = zones.getFacilities();  // used for UrbanSim Zone Models
333                else
334                        allFacilities = parcels.getFacilities();// used for UrbanSim Parcel Models
335                
336                for ( Person person : sc.getPopulation().getPersons().values() ) {
337
338                        Plan plan = person.getSelectedPlan() ;
339                        
340                        boolean isFirstPlanActivity = true;
341                        String lastZoneId = null;
342                        
343                        if(plan.getPlanElements().size() <= 1) // check if activities available (then size of plan elements > 1)
344                                continue;
345                        
346                        boolean enterTripInODMatrix = true;
347                        
348                        for ( PlanElement pe : plan.getPlanElements() ) {
349                                
350                                if ( pe instanceof Activity ) {
351                                        Activity act = (Activity) pe;
352                                        Id id = act.getFacilityId();
353                                        if( id == null) // that person plan doesn't contain any activity, continue with next person
354                                                continue;
355                                        
356                                        ActivityFacility fac = allFacilities.get(id);
357                                        if(fac == null)
358                                                continue;
359                                        String zone_ID = ((Id) fac.getCustomAttributes().get(InternalConstants.ZONE_ID)).toString();
360
361                                        if (isFirstPlanActivity)
362                                                isFirstPlanActivity = false; 
363                                        else if (enterTripInODMatrix){
364                                                matrixEntry = originDestinationMatrix.getEntry(lastZoneId, zone_ID);
365                                                if(matrixEntry != null){
366                                                        double trips = matrixEntry.getValue() + 1.;
367                                                        originDestinationMatrix.setEntry(lastZoneId, zone_ID, trips);
368                                                }
369                                                else    
370                                                        originDestinationMatrix.createAndAddEntry(lastZoneId, zone_ID, 1.);
371                                                        // see PtPlanToPlanStepBasedOnEvents.addPersonToVehicleContainer (or zone coordinate addition)
372                                        }
373                                        lastZoneId = zone_ID; // stores the first activity (e. g. "home")
374                                }
375//                              else if(pe instanceof Leg){ // tnicolai: enable this part to get time dependent trips
376//                                      
377//                                      Leg leg = (Leg)pe;
378//                                      
379//                                      double departureTime = leg.getDepartureTime();
380//                                      double arrivalTime = departureTime + leg.getTravelTime();
381//                                      
382//                                      enterTripInODMatrix = (timeOfDay >= departureTime && timeOfDay <= arrivalTime);
383//                              }
384                        }
385                }
386                // tnicolai: debugging
387//              double originCounter = 0.;
388//              double destinationCounter = 0.;
389//              for(Id fromId : originDestinationMatrix.getFromLocations().keySet()){
390//                      log.info("From Zone: " + fromId.toString());
391//                      for(Entry e : originDestinationMatrix.getFromLocEntries(fromId)){
392//                              originCounter += e.getValue();
393//                              log.info("To Zone: " + e.getToLocation() + " value = " + e.getValue());
394//                      }
395//              }
396//              
397//              for(Id ToId : originDestinationMatrix.getToLocations().keySet()){
398//                      log.info("To Zone: " + ToId.toString());
399//                      for(Entry e : originDestinationMatrix.getToLocEntries(ToId)){
400//                              destinationCounter += e.getValue();
401//                              log.info("From Zone: " + e.getFromLocation() + " value = " + e.getValue());
402//                      }
403//              }
404//              log.info("Home-to-work trips: " + originCounter);
405//              log.info("Work-to-Home trips: " + destinationCounter);
406//              log.info("Total trips: " + (originCounter + destinationCounter));
407                log.info("DONE with computing zone2zone trip numbers ...") ;
408        }
