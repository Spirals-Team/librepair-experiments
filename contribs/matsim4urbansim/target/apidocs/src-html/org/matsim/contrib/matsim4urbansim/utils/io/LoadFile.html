024package org.matsim.contrib.matsim4urbansim.utils.io;
025
026import java.io.BufferedOutputStream;
027import java.io.File;
028import java.io.FileOutputStream;
029import java.io.IOException;
030import java.io.InputStream;
031import java.net.MalformedURLException;
032import java.net.URL;
033
034import org.apache.log4j.Logger;
035import org.matsim.contrib.matsim4urbansim.constants.InternalConstants;
036
037
038
039/**
040 * @author thomas
041 *
042 */
043@Deprecated // should be replaced by matsim core xml parsing. kai, jul'13
044public class LoadFile {
045        
046        // logger
047        private static final Logger log = Logger.getLogger(LoadFile.class);
048        
049        private String source                   = null;
050        private String destinationPath  = null;
051        private String fileName                 = null;
052                
053        /**
054         * constructor
055         * @param source
056         * @param destination
057         */
058        @Deprecated // should be replaced by matsim core xml parsing. kai, jul'13
059        public LoadFile(String source, String destination, String fileName){
060                this.source = source;
061                this.destinationPath = Paths.checkPathEnding( destination );
062                this.fileName = fileName;
063        }
064        
065        /**
066         * linke loadMATSim4UrbanSimXSD() but returns the canonical path
067         * of the loaded xsd file
068         * @return path of loaded xsd file
069         */
070        @Deprecated // should be replaced by matsim core xml parsing. kai, jul'13
071        public String loadMATSim4UrbanSimXSDString(){
072                
073                try {
074                        File f = loadMATSim4UrbanSimXSD();
075                        return f.getCanonicalPath();
076                } catch (IOException e) {
077                        e.printStackTrace();
078                }
079                return null;
080        }
081        
082        /**
083         * creates a local copy of the MATSim4UrbanSim xsd file from matsim.org
084         * and returns the reference to the created file
085         * @return reference to the created xsd file
086         */
087        @Deprecated // should be replaced by matsim core xml parsing. kai, jul'13
088        public File loadMATSim4UrbanSimXSD(){
089                
090                InputStream is = null;
091                BufferedOutputStream bos = null;
092                boolean isInternetStream = Boolean.TRUE;
093                
094                // make sure output directory exists
095                File dir = new File(this.destinationPath);
096                if(!dir.exists())
097                        dir.mkdirs();
098                File output = new File(destinationPath + fileName);
099                
100                log.info("Trying to load " + this.source + ". In some cases (e.g. network interface up but no connection), this may take a bit.");
101                log.info("The xsd file will be saved in " + this.destinationPath + "."); 
102                
103                // get input stream (from internet)
104                try {
105                        is = new URL(this.source).openStream();
106                        bos = new BufferedOutputStream(new FileOutputStream(output));
107                        
108                        for(int c; (c = is.read()) != -1 ;){
109                                bos.write(c);
110                        }
111                        log.info("Loading successfully.");
112                        
113                        if(bos != null){
114                                bos.flush();
115                                bos.close();
116                        }
117                        if (is != null) is.close();
118                        
119                        return output;
120                        
121                } catch (MalformedURLException e1) {
122                        e1.printStackTrace();
123                } catch (IOException e1) {
124                        // There was a problem getting the (remote) file, just show the error as information for the user
125                        log.error(e1.toString() + ". May not be fatal." ) ;
126                        isInternetStream = Boolean.FALSE;
127                        output = null;  // set to null
128                }
129
130                // if no internet connection, trying to load xsd schema locally
131                if(! isInternetStream){
132                        
133                        // This should work for jar files (dtd folder located in root)
134                        log.info("Trying to access local dtd folder at standard location " + InternalConstants.CURRENT_MATSIM_4_URBANSIM_XSD_LOCALJAR + " ...");
135                        File dtdFile = new File( InternalConstants.CURRENT_MATSIM_4_URBANSIM_XSD_LOCALJAR );
136                        
137                        if (dtdFile.exists() && dtdFile.isFile() && dtdFile.canRead()) {
138                                log.info("Using the local DTD " + dtdFile.getAbsolutePath());
139                                output = new File( dtdFile.getAbsolutePath() );
140                                // URL localUrl = this.getClass().getResource(Constants.MATSIM_4_URBANSIM_XSD_LOCAL_CURRENT);
141                                // output = new File( localUrl.getPath() );
142                                return output; // return path to local xsd file
143                        }
144                        
145                        {
146                                // This works in Eclipse environment for debugging ...
147                                String currentDir = System.getProperty("user.dir");
148                                int index = (currentDir.indexOf("playground") > 0) ? currentDir.indexOf("playground") : currentDir.indexOf("contrib");
149                                String root = currentDir.substring(0, index);
150
151                                dtdFile = new File( root + "/matsim" + InternalConstants.CURRENT_MATSIM_4_URBANSIM_XSD_LOCALJAR );
152                                log.info("Trying to access local dtd folder at standard location " + dtdFile.getAbsolutePath() + " ...");
153
154                                if (dtdFile.exists() && dtdFile.isFile() && dtdFile.canRead()) {
155                                        log.info("Using the local DTD " + dtdFile.getAbsolutePath());
156                                        output = new File( dtdFile.getAbsolutePath() );
157                                        // URL localUrl = this.getClass().getResource(Constants.MATSIM_4_URBANSIM_XSD_LOCAL_CURRENT);
158                                        // output = new File( localUrl.getPath() );
159                                        return output; // return path to local xsd file
160                                }
161                        }
162
163                        {
164                                // This works for me (kai) for command line calling ...
165                                String currentDir = System.getProperty("user.dir");
166                                int index = (currentDir.indexOf("playground") > 0) ? currentDir.indexOf("playground") : currentDir.indexOf("contrib");
167                                String root = currentDir.substring(0, index);
168
169                                dtdFile = new File( root + "/matsim/src/main/resources/" + InternalConstants.CURRENT_MATSIM_4_URBANSIM_XSD_LOCALJAR );
170                                log.info("Trying to access local dtd folder at standard location " + dtdFile.getAbsolutePath() + " ...");
171
172                                if (dtdFile.exists() && dtdFile.isFile() && dtdFile.canRead()) {
173                                        log.info("Using the local DTD " + dtdFile.getAbsolutePath());
174                                        output = new File( dtdFile.getAbsolutePath() );
175                                        // URL localUrl = this.getClass().getResource(Constants.MATSIM_4_URBANSIM_XSD_LOCAL_CURRENT);
176                                        // output = new File( localUrl.getPath() );
177                                        return output; // return path to local xsd file
178                                }                       
179                        }
180                }
181                
182                // could neither get the remote nor the local version of the xsd
183                System.out.flush();
184                log.warn("Could neither get the XSD from the web nor a local one.");
185                System.err.flush() ;
186                return null;
187        }
188}
