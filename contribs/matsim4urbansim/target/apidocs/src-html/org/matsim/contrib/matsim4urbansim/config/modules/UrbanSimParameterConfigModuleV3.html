027public class UrbanSimParameterConfigModuleV3 extends ConfigGroup{
028        // IMPORTANT: This is just a container for matsim4urbansim config data.  It is _not_ a matsim config group.
029        
030        public static final String GROUP_NAME = "urbansimParameter";
031
032        private boolean isParcel;
033        private double populationSampleRate;
034        private int year;
035        private String opusHome;

039        private String matsim4OpusOutput;
040        private String matsim4OpusTemp;

045        private boolean usingZoneBasedAccessibility;
046        private boolean usingGridBasedAccessibility;
047        private boolean isBackup;
048        
049        public UrbanSimParameterConfigModuleV3(){
050                super(GROUP_NAME);
051                // This class feels quite dangerous to me; one can have inconsistent entries between the Map and the typed values. kai, apr'13
052                // The way it (hopefully) works now: as long as the config group is not "materialized", one has to use addParam/getValue.
053                // Once the class is materialized, one can only use the direct getters/setters.  kai, may'13
054        }
055        
056        @Override
057        @Deprecated
058        public String getValue(final String key) {
059                throw new RuntimeException(" use direct getter; aborting ... " ) ;
060        }
061
062        @Override
063        @Deprecated
064        public void addParam(String param_name, String value) {
065                throw new RuntimeException(" use direct setter; aborting ... " ) ;
066        }
067        
068        @Override
069        public final Map<String,String> getParams() {
070                Map<String,String> map = new LinkedHashMap<String,String>() ;
071
072                // The following is correct but ...
073                // ... maybe we don't want this.  Users should set this via the OPUS GUI, not via the external matsim config file.
074                // ??  kai, may'13
075                
076                return map ;
077        }
078
079        public void setSpatialUnitFlag(String spatialUnit){
080                if(spatialUnit.equalsIgnoreCase("parcel"))
081                        this.isParcel = true;
082                else if(spatialUnit.equalsIgnoreCase("zone"))
083                        this.isParcel = false;
084        }
085        
086        public boolean isParcelMode(){

088        }
