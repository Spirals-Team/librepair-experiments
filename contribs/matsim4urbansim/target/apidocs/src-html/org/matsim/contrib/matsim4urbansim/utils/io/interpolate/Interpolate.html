023package org.matsim.contrib.matsim4urbansim.utils.io.interpolate;
024
025import org.apache.log4j.Logger;
026import org.matsim.contrib.accessibility.gis.SpatialGrid;
027import org.matsim.contrib.accessibility.interpolation.Interpolation;
028import org.matsim.contrib.matrixbasedptrouter.utils.BoundingBox;
029
030/**
031 * @author thomas
032 *
033 */
034public class Interpolate {
035        
036        // logger
037        private static final Logger log = Logger.getLogger(Interpolate.class);
038        
039        private static String inputSpatialGrid;
040        private static String outputSpatialGrid;
041        private static double cellSizeInMeter;
042
043        /**
044         * @param args
045         */
046        public static void main(String[] args) {
047                
048                if(args == null || args.length != 2){
049                        log.info("Enter the following parameter:");
050                        log.info("1) input table");
051                        log.info("2) resolution (as side length in merter)");
052                        log.info("Please try again.");
053                        System.exit(0);
054                }
055                
056                inputSpatialGrid = args[0];
057                outputSpatialGrid = args[0].replaceAll(".txt", "_interpolated"+args[1]+"m.txt");
058                cellSizeInMeter  = Double.parseDouble(args[1]);
059
060                log.info("Reading input file (spatial grid).");
061                SpatialGrid inputSG = SpatialGrid.readFromFile(inputSpatialGrid);               
062                
063                log.info("Putting input file (spatial grid) into interpolation routine");
064                Interpolation interpol = new Interpolation(inputSG, Interpolation.BILINEAR);
065                
066                log.info("Creating spatial grid for interpolated values.");
067                BoundingBox boundary = BoundingBox.createBoundingBox(inputSG.getXmin(), inputSG.getYmin(), inputSG.getXmax(), inputSG.getYmax());
068                SpatialGrid outputSG= new SpatialGrid(boundary.getBoundingBox(), cellSizeInMeter);
069                
070                for(double x = outputSG.getXmin(); x < outputSG.getXmax(); x = x+cellSizeInMeter){
071                        for(double y = outputSG.getYmin(); y < outputSG.getYmax(); y = y+cellSizeInMeter){
072
073                                if(inputSG.getValue(x, y) == Double.NaN)
074                                        continue;
075                                
076                                double interpolatedValue = interpol.interpolate(x, y);
077                                outputSG.setValue(interpolatedValue, x, y);
078                        }
079                }
080                outputSG.writeToFile(outputSpatialGrid);
081                log.info("Done...");
082        }
083}
