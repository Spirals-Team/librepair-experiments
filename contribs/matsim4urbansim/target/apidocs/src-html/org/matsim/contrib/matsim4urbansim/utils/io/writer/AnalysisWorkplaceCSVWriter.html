001package org.matsim.contrib.matsim4urbansim.utils.io.writer;
002
003import java.io.BufferedWriter;
004import java.util.Iterator;
005import java.util.List;
006
007import org.apache.log4j.Logger;
008import org.matsim.contrib.accessibility.utils.AggregationObject;
009import org.matsim.contrib.matsim4urbansim.config.modules.UrbanSimParameterConfigModuleV3;
010import org.matsim.contrib.matsim4urbansim.constants.InternalConstants;
011import org.matsim.contrib.matsim4urbansim.utils.helperobjects.SpatialReferenceObject;
012import org.matsim.core.config.Config;
013import org.matsim.core.utils.io.IOUtils;
014
015
016public class AnalysisWorkplaceCSVWriter {
017        
018        private static final Logger log = Logger.getLogger(AnalysisWorkplaceCSVWriter.class);
019        
020        public static final String FILE_NAME= "workplaces.csv";
021        public static final String FILE_NAME_AGGREGATED= "aggregated_workplaces.csv";
022        
023        public static final String WORKPLACES_COUNT = "workplaces";
024        /**
025         * Writing aggregated workplace data to disc
026         * @param jobClusterArray
027         * @param config TODO
028         * @param file
029         */
030        public static void writeAggregatedWorkplaceData2CSV(final AggregationObject[] jobClusterArray, Config config){
031                UrbanSimParameterConfigModuleV3 module = (UrbanSimParameterConfigModuleV3) config.getModule(UrbanSimParameterConfigModuleV3.GROUP_NAME);
032                try{
033                        log.info("Initializing AnalysisWorkplaceCSVWriter ...");
034                        BufferedWriter bwAggregatedWP = IOUtils.getBufferedWriter(module.getMATSim4OpusOutput() + FILE_NAME_AGGREGATED );
035                        log.info("Writing (aggregated workplace) data into " + module.getMATSim4OpusOutput() + FILE_NAME_AGGREGATED + " ...");
036                        
037                        // create header
038                        bwAggregatedWP.write(InternalConstants.ZONE_ID +","+ 
039                                                                 InternalConstants.PARCEL_ID +","+ 
040                                                                 InternalConstants.NEARESTNODE_ID +","+
041                                                                 InternalConstants.NEARESTNODE_X_COORD +","+ 
042                                                                 InternalConstants.NEARESTNODE_Y_COORD +","+
043                                                                 WORKPLACES_COUNT);
044                        bwAggregatedWP.newLine();
045                        
046                        for(int i = 0; i < jobClusterArray.length; i++){
047                                bwAggregatedWP.write(jobClusterArray[i].getZoneID() + "," + 
048                                                                         jobClusterArray[i].getParcelID() + "," +
049                                                                         jobClusterArray[i].getNearestNode().getId()  + "," +
050                                                                         jobClusterArray[i].getNearestNode().getCoord().getX() + "," +
051                                                                         jobClusterArray[i].getNearestNode().getCoord().getY() + "," +
052                                                                         jobClusterArray[i].getNumberOfObjects());
053                                bwAggregatedWP.newLine();
054                        }
055                        
056                        bwAggregatedWP.flush();
057                        bwAggregatedWP.close();
058                        log.info("... done!");
059                }
060                catch(Exception e){ 
061                        e.printStackTrace(); 
062                }
063        }
064        
065        /**
066         * writing raw workplace data to disc
067         * @param jobSampleList
068         * @param config TODO
069         * @param file
070         */
071        public static void writeWorkplaceData2CSV(final List<SpatialReferenceObject> jobSampleList, Config config){
072                UrbanSimParameterConfigModuleV3 module = (UrbanSimParameterConfigModuleV3) config.getModule(UrbanSimParameterConfigModuleV3.GROUP_NAME);
073                try{
074                        log.info("Initializing AnalysisWorkplaceCSVWriter ...");
075                        BufferedWriter bwWorkplaces = IOUtils.getBufferedWriter( module.getMATSim4OpusOutput() + FILE_NAME );
076                        log.info("Writing data into " + module.getMATSim4OpusOutput() + FILE_NAME + " ...");
077                        
078                        
079                        // create header
080                        bwWorkplaces.write(InternalConstants.JOB_ID +","+ 
081                                                                 InternalConstants.PARCEL_ID +","+ 
082                                                                 InternalConstants.ZONE_ID +","+
083                                                                 InternalConstants.X_COORDINATE +","+ 
084                                                                 InternalConstants.Y_COORDINATE);
085                        bwWorkplaces.newLine();
086                        
087                        Iterator<SpatialReferenceObject> jobIterator = jobSampleList.iterator();
088
089                        while(jobIterator.hasNext()){
090                                
091                                SpatialReferenceObject job = jobIterator.next();
092                                
093                                bwWorkplaces.write(job.getObjectID() + "," + 
094                                                                   job.getParcelID() + "," +
095                                                                   job.getZoneID() + "," +
096                                                                   job.getCoord().getX() + "," +
097                                                                   job.getCoord().getY());
098                                bwWorkplaces.newLine();
099                        }
100                        
101                        bwWorkplaces.flush();
102                        bwWorkplaces.close();
103                        log.info("... done!");
104                }
105                catch(Exception e){ 
106                        e.printStackTrace(); 
107                }
108        }
109}
