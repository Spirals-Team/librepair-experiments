024package org.matsim.contrib.matsim4urbansim.utils;
025
026import java.io.File;
027
028import org.apache.log4j.Logger;
029import org.matsim.contrib.matsim4urbansim.config.M4UConfigUtils;
030import org.matsim.contrib.matsim4urbansim.config.modules.UrbanSimParameterConfigModuleV3;
031import org.matsim.contrib.matsim4urbansim.utils.io.Paths;
032import org.matsim.core.config.Config;
033
034
035
036/**
037 * @author thomas
038 *
039 */
040public class OPUSDirectoryUtil {
041        private static final Logger log = Logger.getLogger(OPUSDirectoryUtil.class);
042
043        
044        public static void setTmpDirectories(Config c){
045                UrbanSimParameterConfigModuleV3 module = (UrbanSimParameterConfigModuleV3) c.getModule(UrbanSimParameterConfigModuleV3.GROUP_NAME);
046                String tmp = Paths.checkPathEnding(System.getProperty("java.io.tmpdir"));
047                module.setOpusHome(tmp);
048                module.setOpusDataPath(tmp + "data/");
049                module.setMATSim4Opus(tmp + "opus_matsim/");
050                module.setMATSim4OpusConfig(tmp + "opus_matsim/matsim_config/");
051                module.setMATSim4OpusOutput(tmp + "opus_matsim/output/");
052                module.setMATSim4OpusTemp(tmp + "opus_matsim/tmp/");
053                module.setMATSim4OpusBackup(tmp + "opus_matsim/backup/");
054        }
055        
056        /**
057         * create new temp directories. these will be deleted after each test run.
058         * @param config TODO
059         */
060        public static void createOPUSDirectories(Config config){
061                log.info("Creating temp directories");
062                
063                UrbanSimParameterConfigModuleV3 module = M4UConfigUtils.getUrbanSimParameterConfigAndPossiblyConvert(config);
064                // set temp directory as opus_home
065//              InternalConstants.setOpusHomeDirectory(System.getProperty("java.io.tmpdir"));
066
067                File tempFile = new File(module.getOpusHome());
068                tempFile.mkdirs();
069                
070                tempFile = new File(module.getMATSim4Opus());
071                tempFile.mkdirs();
072                
073                tempFile = new File(module.getMATSim4OpusOutput());
074                tempFile.mkdirs();
075                
076                tempFile = new File(module.getMATSim4OpusTemp());
077                tempFile.mkdirs();
078                
079                tempFile = new File(module.getMATSim4OpusConfig());
080                tempFile.mkdirs();
081                log.info("Finished creating temp directories");
082        }
083        
084        /**
085         * Removes the output directory for the UrbanSim data
086         * if doesn't existed before the test run. 
087         * @param config TODO
088         */
089        public static void cleaningUpOPUSDirectories(Config config){
090                log.info("Removing temp directories");
091                UrbanSimParameterConfigModuleV3 module = M4UConfigUtils.getUrbanSimParameterConfigAndPossiblyConvert(config);
092                File tempFile;// = new File(InternalConstants.getOPUS_HOME());
093//              used only for tests. At least on my System this call leads to a null-pointer-exception
094//              because there are some other directories in the temp-directory. Especially some hidden ones.
095//              \\ DR, jul'13
096//              if(tempFile.exists())
097//                      deleteDirectory(tempFile);
098                tempFile = new File(module.getMATSim4Opus());
099                if(tempFile.exists())
100                        org.matsim.contrib.matrixbasedptrouter.utils.TempDirectoryUtil.deleteDirectory(tempFile);
101                tempFile = new File(module.getMATSim4OpusOutput());
102                if(tempFile.exists())
103                        org.matsim.contrib.matrixbasedptrouter.utils.TempDirectoryUtil.deleteDirectory(tempFile);
104                tempFile = new File(module.getMATSim4OpusTemp());
105                if(tempFile.exists())
106                        org.matsim.contrib.matrixbasedptrouter.utils.TempDirectoryUtil.deleteDirectory(tempFile);
107                tempFile = new File(module.getMATSim4OpusConfig());
108                if(tempFile.exists())
109                        org.matsim.contrib.matrixbasedptrouter.utils.TempDirectoryUtil.deleteDirectory(tempFile);       
110                log.info("Finished removing temp directories");
111        }
112
113}
