032                Scenario scenario = createScenario();

037        private static Scenario createScenario() {
038                Scenario scenario = ScenarioUtils.createScenario(ConfigUtils.createConfig());
039                Map<String, Set<String>> tagKeyValues = new ConcurrentHashMap<String, Set<String>>();
040                tagKeyValues.put("highway", new HashSet<String>(Arrays.asList("motorway","motorway_link","trunk","trunk_link","primary","primary_link","secondary","tertiary","minor","unclassified","residential","living_street")));
041                Set<String> tagKeys = Collections.emptySet();
042                TagFilter tagFilter = new TagFilter("accept-way", tagKeys, tagKeyValues);
043                
044                FastXmlReader reader = new FastXmlReader(new File(inputOSMNetwork ), true, CompressionMethod.None);
045                
046                SimplifyTask simplify = new SimplifyTask(IdTrackerType.BitSet);
047                
048                GeotoolsTransformation gt = new GeotoolsTransformation(TransformationFactory.WGS84, "EPSG:31300");
049                // CoordinateTransformation coordinateTransformation = new WGS84ToMercator.Project(18);
050                // NetworkSink sink = new NetworkSink(scenario.getNetwork(), coordinateTransformation);
051                NetworkSink sink = new NetworkSink(scenario.getNetwork(), gt);
052                sink.setHighwayDefaults(1, "motorway",      2, 120.0/3.6, 1.0, 2000, true);
053                sink.setHighwayDefaults(1, "motorway_link", 1,  80.0/3.6, 1.0, 1500, true);
054                sink.setHighwayDefaults(2, "trunk",         1,  80.0/3.6, 1.0, 2000, true);
055                sink.setHighwayDefaults(2, "trunk_link",    1,  50.0/3.6, 1.0, 1500, true);
056                sink.setHighwayDefaults(3, "primary",       1,  80.0/3.6, 1.0, 1500, true);
057                sink.setHighwayDefaults(3, "primary_link",  1,  60.0/3.6, 1.0, 1500, true);
058                sink.setHighwayDefaults(4, "secondary",     1,  60.0/3.6, 1.0, 1000, true);
059                sink.setHighwayDefaults(5, "tertiary",      1,  45.0/3.6, 1.0,  600, true);
060                sink.setHighwayDefaults(6, "minor",         1,  45.0/3.6, 1.0,  600, true);
061                sink.setHighwayDefaults(6, "unclassified",  1,  45.0/3.6, 1.0,  600, false);
062                sink.setHighwayDefaults(6, "residential",   1,  30.0/3.6, 1.0,  600, false);
063                sink.setHighwayDefaults(6, "living_street", 1,  15.0/3.6, 1.0,  300, false);
064
065                reader.setSink(tagFilter);
066                tagFilter.setSink(simplify);
067                simplify.setSink(sink);
068                sink.setSink(new Sink() {
069
070                        @Override
071                        public void process(EntityContainer entityContainer) {
072                                
073                        }
074
075                        @Override
076                        public void complete() {
077                                
078                        }
079
080                        @Override
081                        public void release() {
082                                
083                        }
084
085                        @Override
086                        public void initialize(Map<String, Object> metaData) {
087                                // TODO Auto-generated method stub
088                                
089                        }
090                        
091                });
092
093                reader.run();
094                return scenario;
095        }
