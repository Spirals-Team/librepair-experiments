001package org.matsim.contrib.matsim4urbansim.zurichscenario;
002
003import java.util.ArrayList;
004import java.util.Iterator;
005import java.util.Map;
006
007import org.apache.log4j.Logger;
008import org.matsim.api.core.v01.Id;
009import org.matsim.api.core.v01.network.Link;
010import org.matsim.api.core.v01.network.Network;
011import org.matsim.api.core.v01.population.Leg;
012import org.matsim.api.core.v01.population.Person;
013import org.matsim.api.core.v01.population.Plan;
014import org.matsim.api.core.v01.population.PlanElement;
015import org.matsim.api.core.v01.population.Population;
016import org.matsim.core.population.PopulationUtils;
017import org.matsim.core.population.routes.NetworkRoute;
018
019public class ZurichUtilities {
020        
021        /** logger */
022        private static final Logger log = Logger.getLogger(ZurichUtilities.class);
023
024        /** scenario/test case identifier */
025        static String CLOSE_UETLIBERGTUNNEL = "uetlibergtunnel";
026        static String CLOSE_BIRMENSDORFERSTRASSE_OUTERRIM = "birmsdorferstrasseouterrim"; // former birmsdorferstrasse
027        static String CLOSE_BIRMENSDORFERSTRASSE_CITYCENTER = "birmsdorferstrassecitycenter";
028        static String CLOSE_SCHWAMENDINGERTUNNEL = "schwamendingertunnel";
029        static String CLOSE_MILCHBUCKTUNNEL = "milchbucktunnel";
030        static String CLOSE_GUBRISTTUNNEL = "gubristtunnel";
031        
032        /** links to remove */
033        static ArrayList<Id<Link>> linksToRemove = null;
034        
035        /**
036         * Takes a link list and removes them from the network to apply a certain, predefined scenario
037         * @param network
038         * @param linkList
039         */
040        static void applyScenario(final Network network){
041                // check whether all links exist in network
042                existsInNetwork(network);
043                // close links
044                removeLinks(network);
045        }
046        
047        /**
048         * Checks whether the links exist in the network
049         * @param network
050         * @param linkList
051         * @return true if all links exist in the network
052         */
053        static void existsInNetwork(final Network network){
054                
055                log.info("Looking whether all links for road closure exist in network ...");
056                
057                if(linksToRemove == null){
058                        log.info("No links defined! Exit.");
059                        return;
060                }
061                
062                Map<Id<Link>, ? extends Link> networkLinks = network.getLinks();
063                Iterator<Id<Link>> linkIterator = linksToRemove.iterator();
064                
065                while(linkIterator.hasNext()){
066                        Id<Link> id = linkIterator.next();
067                        if(networkLinks.containsKey(id))
068                                log.info("Link found in network: " + id.toString());
069                        else
070                                log.warn("Link not found in network: " + id.toString());
071                }
072        }
073
074        /**
075         * this closes a link by setting the free speed and capacity to zero 
076         * and the link length to Double.MAX_VALUE
077         * 
078         * @param network
079         * @param linkList
080         */
081        static void removeLinks(final Network network){
082                
083                if(linksToRemove == null)
084                        return;
085                
086                Iterator<Id<Link>> linkIterator = linksToRemove.iterator();
087                
088                while(linkIterator.hasNext()){
089                        Id<Link> id = linkIterator.next();
090                        
091                        network.removeLink( id );
092                        log.info("Removed link " + id.toString() + " from network ...");
093                }
094        }
095        
096        /**
097         * This method deletes plan elements from existing plans when they contain links
098         * that are removed from the network (see modifyNetwork) to avoid errors during mobsim
099         * 
100         * @param population
101         */
102        public static void deleteRoutesContainingRemovedLinks(final Population population){
103                
104                if(linksToRemove == null)
105                        return;
106                
107                int cnt = 0;
108                for (Person person : PopulationUtils.getSortedPersons(population).values()){
109                        boolean isModified = false;
110                        for(Plan plan : person.getPlans()){
111                                for(PlanElement pe: plan.getPlanElements()){
112                                        if(pe instanceof Leg){
113                                                Leg leg = (Leg)pe;
114                                                if(leg.getRoute() != null){
115                                                        
116                                                        NetworkRoute nr = (NetworkRoute)leg.getRoute();
117                                                        Iterator<Id<Link>> linkIterator = linksToRemove.iterator();
118                                                        boolean setRouteToNull = false;
119                                                        
120                                                        while(linkIterator.hasNext()){
121                                                                Id<Link> linkId = linkIterator.next();
122                                                                
123                                                                for(Id id : nr.getLinkIds()){
124                                                                        if(id.compareTo(linkId) == 0){
125                                                                                leg.setRoute(null);
126                                                                                isModified = true;
127                                                                                setRouteToNull = true;
128                                                                                break;
129                                                                        }                       
130                                                                }
131                                                                if(setRouteToNull)
132                                                                        break;
133                                                        }
134
135                                                }
136                                        }
137                                }
138                        }
139                        if(isModified)
140                                cnt++;
141                }
142                if(cnt > 0){
143                        log.info("Reseted " + cnt + " persons plans from a population of " + population.getPersons().size()  );
144                        log.info("This was necessary because these plans contain removed links!");
145                }
146        }
147
148}
