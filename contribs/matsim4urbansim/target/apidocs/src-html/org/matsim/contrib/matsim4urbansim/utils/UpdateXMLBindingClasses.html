024package org.matsim.contrib.matsim4urbansim.utils;
025
026import com.sun.tools.xjc.XJCFacade;
027import org.apache.log4j.Logger;
028import org.matsim.contrib.matrixbasedptrouter.utils.TempDirectoryUtil;
029import org.matsim.contrib.matsim4urbansim.constants.InternalConstants;
030import org.matsim.contrib.matsim4urbansim.utils.io.LoadFile;
031import org.matsim.contrib.matsim4urbansim.utils.securitymanager.ExitException;
032import org.matsim.contrib.matsim4urbansim.utils.securitymanager.NoExitSecurityManager;
033
034import java.io.File;
035import java.io.IOException;
036
037
038/**
039 * @author thomas
040 * 
041 * This class generates a set of Java classes via JAXB that represents a given schema (xsd file) and copies them into the package "com.matsim.config".
042 * The JAXB executables are located outside the playground in "OPUS_HOME/libs".
043 * 
044 */
045public class UpdateXMLBindingClasses {
046        
047        private static final Logger log = Logger.getLogger(UpdateXMLBindingClasses.class);
048        
049        private static SecurityManager securityManager;
050        
051        private static String helpMessage = null;
052        
053        private static String targetPackage = null;
054        private static String outputDirectory = null;
055        private static String xsdLocation = null;
056
057        /**
058         * @param args
059         */
060        public static void main(String[] args) {
061                
062                // set help message
063                helpMessage = "Please run this program as described below:\n" +
064                                          "java UpdateXMLBindingClasses --jaxbLocation=my path/libs/jaxb-2.1.7/bin/xjc.sh\n" +
065                                          "Optional you can define:\n"+
066                                          "- the desired location of the generated binding classes: --destination=your/destination/path\n" +
067                                          "- the desierd package structure: --package=your.desired.package.structure\n";
068                                          
069                
070                // script to create jaxb bindings
071                log.info("Start generating xml bindings ...");
072                log.info("");
073                
074                if(!checkParameter( args ))
075                        System.exit(0);
076                
077                log.info("");
078                log.info("Parameter overview");
079                log.info("Target package: " + targetPackage);
080                log.info("Destination directory: " + outputDirectory);
081                log.info("XSD location: " + xsdLocation);
082                log.info("");
083                
084                String[] arg = new String[] { "-p", targetPackage, "-d", outputDirectory, xsdLocation };
085                
086                try {
087                        // backup security manager
088                        securityManager = System.getSecurityManager();
089                        System.setSecurityManager( new NoExitSecurityManager() );
090                        
091                        // generate java binding classes
092                        // info: XJCFacade exits with a System.exit() 
093                        XJCFacade.main(arg);
094                        
095                        // restore previous security manager
096                        System.setSecurityManager( securityManager );
097                        
098                } catch (ExitException ee) {
099                        log.warn( ee.getMessage() );
100                        // restore previous security manager
101                        System.setSecurityManager( securityManager );
102                        log.info("Successful finished creating xml bindings ...");
103                }
104                catch(Throwable te){
105                        te.printStackTrace();
106                }
107        }
108        
109        /**
110         * checks correctness of input parameters
111         * @param args
112         * @return
113         */
114        private static boolean checkParameter(String args[]){
115                
116                for ( int i=0 ; i<args.length ; i++ ) {
117                        log.info( "Parameter: " + i + " = " + args[i] ) ;
118                        String[] parts = args[i].split("=");
119                        if ( parts[0].equals("--destination") ) {
120                                outputDirectory = parts[1];
121                        }
122                        else if ( parts[0].equals("--xsd") ) {
123                                xsdLocation = parts[1];
124                        }
125                        else if ( parts[0].equals("--package") ) {
126                                targetPackage = parts[1];
127                        }
128                        else if ( parts[0].equals("--help") )
129                                log.info(helpMessage);
130                }
131                
132                TempDirectoryUtil tempDirectoryUtil = new TempDirectoryUtil() ;
133
134                
135                if(xsdLocation == null){
136                        log.warn("XSD location not given (optional)...");
137                        String tmpDir = tempDirectoryUtil.createCustomTempDirectory("xsd");
138                        // set default location                 
139                        LoadFile loadFile = new LoadFile(InternalConstants.CURRENT_MATSIM_4_URBANSIM_XSD_MATSIMORG, tmpDir, InternalConstants.CURRENT_XSD_FILE_NAME);
140                        xsdLocation = loadFile.loadMATSim4UrbanSimXSDString();
141                        
142                        log.warn("Set xsd default location to: " + xsdLocation);
143                }
144                if(outputDirectory == null){
145                        log.warn("Destination not given (optional)...");
146                        // set default location
147                        outputDirectory = InternalConstants.MATSIM_WORKING_DIRECTORY + "/src/main/java/";
148                        TempDirectoryUtil.createDirectory( outputDirectory );
149                        log.warn("Set default destination to: " + outputDirectory);
150                }
151                if(targetPackage == null){
152                        log.warn("Package name not given (optional)...");
153                        // set default location
154                        targetPackage = getTargetPackage(new UpdateXMLBindingClasses().getClass());//"playground.tnicolai.urbansim.org.matsim.config";
155                        log.warn("Set default package name to: " + targetPackage);
156                }
157
158                if( !(outputDirectory!= null && 
159                          targetPackage != null && 
160                          isValidLocataion(xsdLocation)) )
161                {
162                        log.warn(helpMessage);
163                        return false;
164                }
165                return true;
166        }
167        
168        public static String getTargetPackage(Class<?> c){
169                
170                Package pack = c.getPackage();
171                String name = pack.getName();
172                int dot = name.lastIndexOf(".");
173                name = name.substring(0, dot+1) + InternalConstants.JAXB_PARSER_PACKAGE_NAME;
174                
175                return name;
176        }
177        
178        /**
179         * determines if a path whether exists or not
180         * @return true if path exists otherwise false
181         */
182        private static boolean isValidLocataion(String parameter){
183                
184                try{
185                        if(parameter != null){
186                                File f = new File(parameter);
187                                if( !(f.exists()) ){
188                                        log.error("Location not found! " + f.getCanonicalPath());
189                                        return false;
190                                }
191                        }
192                        else{
193                                log.error("Location not initialized!");
194                                return false;
195                        }
196                }
197                catch(IOException e){}
198                return true;
199        }
