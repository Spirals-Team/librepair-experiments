043public class Benchmark {
044        
045        // logger
046        private static final Logger log = Logger.getLogger(Benchmark.class);
047        // writes the results to the desired location
048        private BufferedWriter benchmarkWriter = null;
049        private String output = null;
050        // counter in order to separate measurement tasks
051        private static int measureID = 0;
052        
053        private ArrayList<MeasurementObject> measurements = null;
054        
055        public Benchmark(){
056                log.info("Initializing ...");
057                measurements = new ArrayList<MeasurementObject>();
058        }
059        
060        public int addMeasure(String name){
061                
062                long startTime = System.currentTimeMillis();
063                
064                log.info("Added new measurement item (id=" + measureID + ").");
065                MeasurementObject mo = new MeasurementObject(name, startTime, measureID);
066                measurements.add(mo);
067                measureID++;
068                
069                return measureID-1;
070        }
071        
072        public int addMeasure(String name, String filePath, boolean readingFile){
073                
074                long startTime = System.currentTimeMillis();
075                
076                log.info("Added new measurement item (id=" + measureID + ").");
077                MeasurementObject mo = new MeasurementObject(name, startTime, new File(filePath), readingFile, measureID);
078                measurements.add(mo);
079                measureID++;
080                
081                return measureID-1;
082        }
083        
084        public void stoppMeasurement(int id){
085                
086                long endTime = System.currentTimeMillis();
087                
088                log.info("Stopping measurement (id=" + measureID + ").");
089                if(id < measurements.size())
090                        measurements.get( id ).stopMeasurement( endTime );
091        }
092        
093        public void dumpResults(String output){
094                
095                try {
096                        this.output = output;
097                        benchmarkWriter = IOUtils.getAppendingBufferedWriter( output );
098                } catch (UncheckedIOException e) {
099                        e.printStackTrace();
100                        benchmarkWriter = null;
101                }
102                
103                log.info("Dumping measurement results to: " + output );
104                if(measurements != null && measurements.size() > 0 && benchmarkWriter != null){
105                        
106                        Iterator<MeasurementObject> it = measurements.iterator();
107                        
108                        // write header
109                        try {
110                                benchmarkWriter.write("Name\t Duration in ms\t Reading File\t Writing File\t File Size in bytes\t File Size in megabytes\t");
111                                benchmarkWriter.newLine();
112                                
113                                while(it.hasNext()){
114                                        
115                                        MeasurementObject mo = it.next();
116                                        
117                                        if(mo.getFile() == null)
118                                                benchmarkWriter.write( mo.getName() + InternalConstants.TAB + String.valueOf(mo.getDuration()) + InternalConstants.TAB + "-" + InternalConstants.TAB + "-" + InternalConstants.TAB + "-" + InternalConstants.TAB + "-");
119                                        else
120                                                benchmarkWriter.write( mo.getName() + InternalConstants.TAB + String.valueOf(mo.getDuration()) + InternalConstants.TAB + (mo.isReading()?mo.getFile().getCanonicalPath():"-") + InternalConstants.TAB + (mo.isReading()?"-":mo.getFile().getCanonicalPath()) + InternalConstants.TAB + String.valueOf(mo.getFileSize()) + InternalConstants.TAB + String.valueOf(mo.getFileSize()/(1024*1024)) );
121                                        benchmarkWriter.newLine();
122                                }
123                                benchmarkWriter.flush();
124                                benchmarkWriter.close();
125                        } catch (IOException e) {
126                                e.printStackTrace();
127                        }
128                }
129                log.info("Done with dumping measurement results.");
130        }
131        
132        public double getDurationInMilliSeconds(int id){
133                if(measurements != null && measurements.size() > id)
134                        return measurements.get(id).getDuration();
135                return -1.;
136        }
137        public double getDurationInSeconds(int id){
138                double duration = getDurationInMilliSeconds(id);
139                if(duration < 0.)
140                        return -1.;
141                return duration/1000.;
142        }
143        
144        private class MeasurementObject {
145                
146                private String name;
147                private long startTime, endtime, duration;      // in milliseconds
148                private File file;
149                private long fileSize;                                  // in bytes
150                private boolean readingFile;                            // false if a file is written ...
151                private int id;
152                
153                public MeasurementObject(String name, long startTime, int id){
154                        this.name = name;
155                        this.startTime = startTime;
156                        this.file = null;
157                        this.fileSize = -1;
158                        this.id = id;
159                }
160                
161                public MeasurementObject(String name, long startTime, File file, boolean readingFile, int id){
162                        this.name = name;
163                        this.startTime = startTime;
164                        this.file = file;
165                        this.readingFile = readingFile;
166                        this.id = id;
167                }
168                
169                public void stopMeasurement(long endTime){
170                        this.endtime = endTime;
171                        this.duration = this.endtime - this.startTime;
172                        
173
174                        if(file != null && file.exists()){
175                                this.fileSize = file.length();
176                        }
177                }
178                
179                // getter methods
180                public long getDuration(){
181                        return this.duration;
182                }
183                public String getName(){
184                        return this.name;
185                }
186                public File getFile(){
187                        return this.file;
188                }
189                public long getFileSize(){
190                        return this.fileSize;
191                }
192                public boolean isReading(){
193                        return this.readingFile;
194                }
195                public int getID(){
196                        return this.id;
197                }
198        }
