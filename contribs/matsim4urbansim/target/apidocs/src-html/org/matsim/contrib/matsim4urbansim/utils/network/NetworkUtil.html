062         * Returns the orthogonal distance between a point and a network link (a straight line).
063         * It assumes that a link has unlimited length.
064         * So it gives just the distance between a point and a line.
065         * 
066         * tnicolai feb'13: not used any more for accessibility calculation
067         * 
068         * @param link
069         * @param pointx
070         * @param pointy
071         * @return
072         */
073        private static double getOrthogonalDistance(Link link, double pointx, double pointy){
074                
075                double ax = link.getFromNode().getCoord().getX();
076                double ay = link.getFromNode().getCoord().getY();
077                double bx = link.getToNode().getCoord().getX();
078                double by = link.getToNode().getCoord().getY();
079
080                double normalzation = Math.sqrt( Math.pow( bx - ax , 2) + Math.pow( by - ay, 2));
081                double distance = Math.abs( ((pointx - ax) * (by - ay)) - ((pointy -ay) * (bx - ax)) );
082                
083                return distance/normalzation;
084        }

198         * This method calculates the distance between a point and a node via a link (which should contains the node) as the sum of
199         * - the orthogonal distance between the point and the link and
200         * - the distance between the intersection point of the orthogonal projection (from the point to the link) to the node.
201         * If the orthogonal projection of the point to the line does not intersects the link, the method returns the euclidean distance between the point and the node.
202         * 
203         * @param link
204         * @param pointx
205         * @param pointy
206         * @param destinationNode
207         * @return
208         */
209        private static Distances getDistance2Node(Link link, double pointx, double pointy, Node destinationNode){
210                
211                // line A B
212                double ax = link.getFromNode().getCoord().getX();
213                double ay = link.getFromNode().getCoord().getY();
214                double bx = link.getToNode().getCoord().getX();
215                double by = link.getToNode().getCoord().getY();
216                
217                // vector ba (r)
218                double bax = bx - ax;
219                double bay = by - ay;
220                // vector pa (v)
221                double pax = pointx - ax;
222                double pay = pointy - ay;
223                
224                // calculation the vector p (projection of point P on line A B)
225                double numerator = pax * bax + pay * bay;
226                double denominator = bax*bax + bay*bay;
227                double fraction = numerator/denominator;
228                double vectorx = bax * fraction;
229                double vectory = bay * fraction;
230                
231                double intersectionx = ax + vectorx;
232                double intersectiony = ay + vectory;
233                
234                // TEST 
235                // is bax or bay == 0?
236                double lambdax = vectorx/bax; // vectorx = intersectionx-ax
237                double lambday = vectory/bay; // vectory = intersectiony-ay
238                double lambdaxInt = Math.rint( lambdax * 1000);
239                double lambdayInt = Math.rint( lambday * 1000);
240                
241                totalCounter++;
242                
243                if(bax == 0 && 0 <= lambday && lambday <= 1){ // for vertical links
244                        // test if lambday is true for x and y
245                        double testx = ax + (lambday * bax);
246                        double testy = ay + (lambday * bay);
247                        if(testx == intersectionx && testy == intersectiony){
248                                othogonalCounter++;
249                                return orthogonalDistancePlusLinkIntersection(pointx, pointy,
250                                                destinationNode, intersectionx, intersectiony);
251                        }
252                }
253                else if(bay == 0 && 0 <= lambdax && lambdax <= 1){ // for horizontal links
254                        // test if lambdax is true for x and y
255                        double testx = ax + (lambdax * bax);
256                        double testy = ay + (lambday * bay);
257                        if(testx == intersectionx && testy == intersectiony){
258                                othogonalCounter++;
259                                return orthogonalDistancePlusLinkIntersection(pointx, pointy,
260                                                destinationNode, intersectionx, intersectiony);
261                        }
262                }
263                // hier liegt der Schnittpunkt "intersection auf der geraden (link)
264                else if(lambdaxInt == lambdayInt && 0 <= lambdax && lambdax <= 1 ){
265                        othogonalCounter++;
266                        return orthogonalDistancePlusLinkIntersection(pointx, pointy,
267                                        destinationNode, intersectionx, intersectiony);
268                }
269                
270                euclidianCounter++;
271                return getEuclidianDistance(pointx, pointy, destinationNode.getCoord().getX(), destinationNode.getCoord().getY());
272                
273                // TEST
274                
275//              double distancePoint2Link = Math.sqrt( (pointx-intersectionx)*(pointx-intersectionx) + (pointy-intersectiony)*(pointy-intersectiony));
276//              double distanceIntersection2Node = Math.sqrt( (intersectionx - destinationNode.getCoord().getX())*(intersectionx - destinationNode.getCoord().getX()) + (intersectiony - destinationNode.getCoord().getY())*(intersectiony - destinationNode.getCoord().getY()) );
277//              return distancePoint2Link + distanceIntersection2Node;
278        }

302        /**
303         * returns the euclidean distance between two coordinates
304         * 
305         * @param origin
306         * @param destination
307         * @return distance
308         */
309        public static double getEuclidianDistance(Coord origin, Coord destination){
310                
311                assert(origin != null);
312                assert(destination != null);
313                
314                double xDiff = origin.getX() - destination.getX();
315                double yDiff = origin.getY() - destination.getY();
316                double distance = Math.sqrt( (xDiff*xDiff) + (yDiff*yDiff) );
317                
318                return distance;
319        }

322         * 
323         * @param x1
324         * @param y1
325         * @param x2
326         * @param y2
327         * @return Distances
328         */
329        public static Distances getEuclidianDistance(double x1, double y1, double x2, double y2){
330                
331                Distances d = new Distances();
332                
333                double xDiff = x1 - x2;
334                double yDiff = y1 - y2;
335                double distance =  Math.sqrt( (xDiff*xDiff) + (yDiff*yDiff) );
336                d.setDisatancePoint2Road(distance);
337                
338                return d;
339        }

341        /**
342         * This method expects the nearest link to a given measure point. 
343         * It calculates the euclidian distance for both nodes of the link, 
344         * "fromNode" and "toNode" and returns the node with shorter distance
345         * 
346         * @param coordFromZone
347         * @param nearestLink
348         */
349        public static Node getNearestNode(Coord coordFromZone, Link nearestLink) {
350                Node toNode = nearestLink.getToNode();
351                Node fromNode= nearestLink.getFromNode();
352                
353                double distanceToNode = getEuclidianDistance(coordFromZone, toNode.getCoord());
354                double distanceFromNode= getEuclidianDistance(coordFromZone, fromNode.getCoord());
355                
356                if(distanceToNode < distanceFromNode)
357                        return toNode;
358                return fromNode;
359        }
