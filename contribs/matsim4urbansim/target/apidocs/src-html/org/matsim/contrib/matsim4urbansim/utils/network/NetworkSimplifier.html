041/**
042 * Simplifies a given network, by merging links.
043 *
044 * @author aneumann
045 *
046 */
047public class NetworkSimplifier {
048
049        private static final Logger log = Logger.getLogger(NetworkSimplifier.class);
050        private boolean mergeLinkStats = false;
051        private Set<Integer> nodeTopoToMerge = new TreeSet<Integer>();
052
053        public void run(final Network network) {
054
055                if(this.nodeTopoToMerge.size() == 0){
056                        throw new RuntimeException("No types of node specified. Please use setNodesToMerge to specify which nodes should be merged");
057                }
058
059                log.info("running " + this.getClass().getName() + " algorithm...");
060
061                log.info("  checking " + network.getNodes().size() + " nodes and " +
062                                network.getLinks().size() + " links for dead-ends...");
063
064                NetworkCalcTopoType nodeTopo = new NetworkCalcTopoType();
065                nodeTopo.run(network);
066
067                for (Node node : network.getNodes().values()) {
068
069                        if(this.nodeTopoToMerge.contains(Integer.valueOf(nodeTopo.getTopoType(node)))){
070
071                                List<Link> iLinks = new ArrayList<Link> (node.getInLinks().values());
072
073                                for (Link iL : iLinks) {
074                                        Link inLink = (Link) iL;
075
076                                        List<Link> oLinks = new ArrayList<Link> (node.getOutLinks().values());
077
078                                        for (Link oL : oLinks) {
079                                                Link outLink = (Link) oL;
080
081                                                if(inLink != null && outLink != null){
082                                                        if(!outLink.getToNode().equals(inLink.getFromNode())){
083
084                                                                if(this.mergeLinkStats){
085
086                                                                        // Try to merge both links by guessing the resulting links attributes
087                                                                        Link link = network.getFactory().createLink(
088                                                                                        Id.create(inLink.getId() + "-" + outLink.getId(), Link.class),
089                                                                                        inLink.getFromNode(),
090                                                                                        outLink.getToNode());
091
092                                                                        // length can be summed up
093                                                                        link.setLength(inLink.getLength() + outLink.getLength());
094
095                                                                        // freespeed depends on total length and time needed for inLink and outLink
096                                                                        link.setFreespeed(
097                                                                                        (inLink.getLength() + outLink.getLength()) /
098                                                                                        (NetworkUtils.getFreespeedTravelTime(inLink) + NetworkUtils.getFreespeedTravelTime(outLink))
099                                                                                        );
100
101                                                                        // the capacity and the new links end is important, thus it will be set to the minimum
102                                                                        link.setCapacity(Math.min(inLink.getCapacity(), outLink.getCapacity()));
103
104                                                                        // number of lanes can be derived from the storage capacity of both links
105                                                                        link.setNumberOfLanes((inLink.getLength() * inLink.getNumberOfLanes()
106                                                                                                        + outLink.getLength() * outLink.getNumberOfLanes())
107                                                                                                        / (inLink.getLength() + outLink.getLength())
108                                                                                                        );
109
110//                                                                      inLink.getOrigId() + "-" + outLink.getOrigId(),
111                                                                        network.addLink(link);
112                                                                        network.removeLink(inLink.getId());
113                                                                        (network).removeLink(outLink.getId());
114
115                                                                } else {
116
117                                                                        // Only merge links with same attributes
118                                                                        if(bothLinksHaveSameLinkStats(inLink, outLink)){
119                                                                                Link newLink = NetworkUtils.createAndAddLink(((Network) network),Id.create(inLink.getId() + "-" + outLink.getId(), Link.class), inLink.getFromNode(), outLink.getToNode(), inLink.getLength() + outLink.getLength(), inLink.getFreespeed(), inLink.getCapacity(), inLink.getNumberOfLanes(), NetworkUtils.getOrigId( inLink ) + "-" + NetworkUtils.getOrigId( outLink ), null);
120
121                                                                                newLink.setAllowedModes(inLink.getAllowedModes());
122
123                                                                                network.removeLink(inLink.getId());
124                                                                                network.removeLink(outLink.getId());
125                                                                        }
126
127                                                                }
128                                                        }
129                                                }
130                                        }
131                                }
132                        }
133
134                }
135
136                org.matsim.core.network.algorithms.NetworkCleaner nc = new org.matsim.core.network.algorithms.NetworkCleaner();
137                nc.run(network);
138
139                nodeTopo = new NetworkCalcTopoType();
140                nodeTopo.run(network);
141
142                log.info("  resulting network contains " + network.getNodes().size() + " nodes and " +
143                                network.getLinks().size() + " links.");
144                log.info("done.");
145        }
146
147        /**
148         * Specify the types of node which should be merged.
149         *
150         * @param nodeTypesToMerge A Set of integer indicating the node types as specified by {@link NetworkCalcTopoType}
151         * @see NetworkCalcTopoType NetworkCalcTopoType for a list of available classifications.
152         */
153        public void setNodesToMerge(Set<Integer> nodeTypesToMerge){
154                this.nodeTopoToMerge.addAll(nodeTypesToMerge);
155        }
156
157        /**
158         *
159         * @param mergeLinkStats If set true, links will be merged despite their different attributes.
160         *  If set false, only links with the same attributes will be merged, thus preserving as much information as possible.
161         *  Default is set false.
162         */
163        public void setMergeLinkStats(boolean mergeLinkStats){
164                this.mergeLinkStats = mergeLinkStats;
165        }
166
167        // helper
168
169        /**
170         * Compare link attributes. Return whether they are the same or not.
171         */
172        private boolean bothLinksHaveSameLinkStats(Link linkA, Link linkB){
173
174                boolean bothLinksHaveSameLinkStats = true;
175
176                if(!linkA.getAllowedModes().equals(linkB.getAllowedModes())){ bothLinksHaveSameLinkStats = false; }
177
178                if(linkA.getFreespeed() != linkB.getFreespeed()){ bothLinksHaveSameLinkStats = false; }
179
180                if(linkA.getCapacity() != linkB.getCapacity()){ bothLinksHaveSameLinkStats = false; }
181
182                if(linkA.getNumberOfLanes() != linkB.getNumberOfLanes()){ bothLinksHaveSameLinkStats = false; }
183
184                return bothLinksHaveSameLinkStats;
185        }
186
187        public static void main(String[] args) {
188
189                Set<Integer> nodeTypesToMerge = new TreeSet<Integer>();
190                nodeTypesToMerge.add(new Integer(4));
191                nodeTypesToMerge.add(new Integer(5));
192
193                Scenario scenario = ScenarioUtils.createScenario(ConfigUtils.createConfig());
194                final Network network = scenario.getNetwork();
195                new MatsimNetworkReader(scenario.getNetwork()).readFile("/Users/thomas/Downloads/belgiumResidential.xml.gz");
196
197                NetworkSimplifier nsimply = new NetworkSimplifier();
198                nsimply.setNodesToMerge(nodeTypesToMerge);
199                nsimply.setMergeLinkStats(false); //default = false
200                nsimply.run(network);
201
202                new NetworkWriter(network).write("/Users/thomas/Downloads/belgiumResidentialSimplifyed2.xml.gz");
203
204        }
