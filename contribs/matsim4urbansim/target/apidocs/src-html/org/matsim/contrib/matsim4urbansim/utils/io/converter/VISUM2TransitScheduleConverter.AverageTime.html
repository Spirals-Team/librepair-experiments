124        private void initTransitStops() {
125                try {
126                        // read zone attribute table and store as hashmap
127                        String line = zoneAttributeReader.readLine();
128                        // get and initialize the column number of each header element
129                        Map<String,Integer> idxFromKey = HeaderParser.createIdxFromKey( line, SEPARATOR);
130                        final int bezirkNummerIDX = idxFromKey.get("$BEZIRK:NR");
131                        final int xCoordIDX = idxFromKey.get("XKOORD");
132                        final int yCoordIDX = idxFromKey.get("YKOORD");
133                        
134                        // temporary variables
135                        int lengths = line.split(SEPARATOR).length;
136                        Coord coord;
137                        String[] parts;
138                        
139                        while ( (line = zoneAttributeReader.readLine()) != null ) {
140                                
141                                parts = line.split(SEPARATOR);
142                                if( parts.length < lengths )
143                                        continue;
144                                
145                                Long bezirkNummer = Long.parseLong( parts[ bezirkNummerIDX ] );
146                                // get the coordinates
147                                coord = new Coord(Double.parseDouble(parts[xCoordIDX]), Double.parseDouble(parts[yCoordIDX]));
148                                
149                                if( ! transitStop.containsKey( bezirkNummer ) )
150                                        transitStop.put(bezirkNummer, coord);
151                                else
152                                        log.warn("A transit stop with a BEZIRK:NR " + bezirkNummer + " already exists!");
153                        }
154                        
155                        log.info("Done creating transit stop HashMap.");
156                        
157                } catch (IOException e) { 
158                        e.printStackTrace(); 
159                } catch (NumberFormatException nfe) {
160                        nfe.printStackTrace();
161                }
162        }

164        private void initVISUMData() {
165                
166                int lineNumber                                  = 0;
167                final int bezikNummerOriginIDX  = 0;
168                final int bezikNummerDestinationIDX = 1;
169                final int valueIDX                              = 2;
170                String[] parts = null;
171                String line  = null;
172                
173                try {
174                        log.info("Reading VISUM station 2 station travel times ...");
175                        boolean skipLine = true;
176                        Counter reiseZeitenCounter = new Counter();
177                        while ( (line = reiseZeitenReader.readLine()) != null ) {
178                                lineNumber ++;
179                                // skip lines
180                                if(skipLine){
181                                        if(line.length() == 0)
182                                                skipLine = false;
183                                        continue;
184                                }
185                                else if(line.startsWith(VISUM_END_TAG))
186                                        break;
187                                
188                                line = line.trim().replaceAll("\\s+", SEPARATOR);
189                                parts = line.split(SEPARATOR);
190                                
191                                Long bezirkNummerOrigin         = Long.parseLong(parts[bezikNummerOriginIDX]);
192                                long bezirkNummerDestination= Long.parseLong(parts[bezikNummerDestinationIDX]);
193                                double reiseZeit                        = Double.parseDouble(parts[valueIDX]);
194                                
195                                if(reiseZeit == EMPTY_ENTRY){
196                                        reiseZeitenCounter.countEmpty();
197                                        continue;
198                                }
199                                else if(visumData.containsKey(bezirkNummerOrigin)){
200                                        VISUMObject vo = new VISUMObject(bezirkNummerDestination, reiseZeit);
201                                        ArrayList<VISUMObject> list = visumData.get(bezirkNummerOrigin);
202                                        list.add(vo);
203                                        reiseZeitenCounter.countTotal();
204                                }
205                                else{
206                                        VISUMObject vo = new VISUMObject(bezirkNummerDestination, reiseZeit);
207                                        ArrayList<VISUMObject> list = new ArrayList<VISUMObject>();
208                                        list.add(vo);
209                                        visumData.put(bezirkNummerOrigin, list);
210                                        reiseZeitenCounter.countOrigins();
211                                }
212                        }
213                        
214                        log.info("Done reading VISUM station 2 station travel times.");
215                        log.info("Reading VISUM waiting times ...");
216                        skipLine = true;
217                        Counter warteZeitenCounter = new Counter();
218                        while ( (line = startWarteZeitenReader.readLine()) != null ) {
219                                // skip lines
220                                if(skipLine){
221                                        if(line.length() == 0)
222                                                skipLine = false;
223                                        continue;
224                                }
225                                else if(line.startsWith(VISUM_END_TAG))
226                                        break;
227
228                                line = line.trim().replaceAll("\\s+", SEPARATOR);
229                                parts = line.split(SEPARATOR);
230                                
231                                Long bezirkNummerOrigin         = Long.parseLong(parts[bezikNummerOriginIDX]);
232                                long bezirkNummerDestination= Long.parseLong(parts[bezikNummerDestinationIDX]);
233                                int warteZeit                           = Integer.parseInt(parts[valueIDX]);
234                                
235                                if(warteZeit == EMPTY_ENTRY){
236                                        warteZeitenCounter.countEmpty();
237                                        continue;
238                                }
239                                else if(visumData.containsKey(bezirkNummerOrigin)){
240                                        ArrayList<VISUMObject> list = visumData.get(bezirkNummerOrigin);
241                                        for(int i = 0; i <list.size(); i++){
242                                                if(list.get(i).getDestinationBezirkNummer() == bezirkNummerDestination){
243                                                        VISUMObject vo = list.get(i);
244                                                        vo.addWarteZeit(warteZeit);
245                                                        break;
246                                                }
247                                        }
248                                        warteZeitenCounter.countTotal();
249                                }
250                                else
251                                        log.warn("BezirkNummer " + bezirkNummerOrigin + " not found!");
252                                
253                                avgWaitingTime.addTime(warteZeit);
254                        }
255
256                        log.info("Reise Zeiten Counter: Total="+reiseZeitenCounter.getTotalCounter() + " Empty="+reiseZeitenCounter.getEmptyCounter() + " Origins="+reiseZeitenCounter.getOriginCounter());
257                        log.info("Warte Zeiten Counter: Total="+warteZeitenCounter.getTotalCounter() + " Empty="+warteZeitenCounter.getEmptyCounter() );
258                        log.info("Average Waiting Time:"+ avgWaitingTime.getAverageTime());
259                        log.info("Done reading VISUM waiting times.");
260
261                } catch (IOException e) { 
262                        e.printStackTrace(); 
263                } catch (NumberFormatException nfe) {
264                        log.error(line);
265                        for(int i = 0; i < parts.length; i++)
266                                System.out.println(parts[i]);
267                        log.error("LineNumber: " + lineNumber);
268                        nfe.printStackTrace();
269                }
270        }

273         * this checks if the transit stops are used in VISUM and vice versa

275        private void consistencyCheck(){
276                
277                log.info("Checking for missing waiting times ...");
278                Iterator<Long> transitStopKeyIterator = transitStop.keySet().iterator();
279                
280                int removedVISUMDataEntries     = 0;
281                
282                ArrayList<Long> removeTransitStops = new ArrayList<Long>();
283                
284                while(transitStopKeyIterator.hasNext()){
285                        
286                        long key = transitStopKeyIterator.next();
287                        if(visumData.containsKey(key)){
288                                // check if each VISUM Object in list if it contains travel and waiting times ...
289                                ArrayList<VISUMObject> list = visumData.get(key);
290                                int localCounter = 0;
291                                for(int i = 0; i < list.size(); i++){
292                                        
293                                        VISUMObject vo = list.get(i);
294                                        
295                                        long destinationKey = vo.getDestinationBezirkNummer();
296                                        
297                                        if(!transitStop.containsKey(destinationKey)){
298                                                if(!removeTransitStops.contains(destinationKey))
299                                                        removeTransitStops.add(destinationKey);
300                                                list.remove(i);
301                                                i--;
302                                                localCounter++;
303                                        }
304                                        else if( (vo.getWarteZeit() < 0) || (vo.getReiseZeit() < 0.) ){
305                                                // missing waiting time remove VISUMObject
306                                                list.remove(i);
307                                                i--;
308                                                // alternatively:
309                                                // add avg waiting time
310                                                // vo.addWarteZeit(avgWaitingTime.getAverageTime());
311                                                localCounter ++;
312                                        }
313                                }
314                                removedVISUMDataEntries += localCounter;
315                                log.warn("Removed " + localCounter + " entries in transit stop " + key + ". These entries had no waiting- or travel times ...");
316                        }
317                        else{
318                                // Apparently the VISUM data does not contain this transit stop. hence it will be removed
319                                if(!removeTransitStops.contains(key))
320                                        removeTransitStops.add(key);
321                        }
322                }
323                
324                for(int i = 0; i < removeTransitStops.size(); i++){
325                        transitStop.remove(removeTransitStops.get(i));
326                        log.warn("Removed transit stop:" + removeTransitStops.get(i) + ". It is not used in VISUM ...");
327                }       
328                log.info(removeTransitStops.size() + " transit stops and " + removedVISUMDataEntries + " VISUM entries are removed.");
329        }

335        private void createTransitSchedule() {
336                
337                log.info("Creating transit schedule ...");
338
339                ArrayList<Long> transitStopKeyList = new ArrayList<Long>(transitStop.keySet());
340                
341                TransitScheduleFactoryImpl tsfi = new TransitScheduleFactoryImpl();
342                TransitSchedule transitSchedule = tsfi.createTransitSchedule();
343                
344                // create transit stops
345                log.info("Creating transit stops ...");
346                for (int i = 0; i < transitStopKeyList.size(); i++) {
347
348                        // get transit stop id (origin)
349                        Long transitStopKey = transitStopKeyList.get(i);
350                        Coord coord = transitStop.get(transitStopKey);
351//                      Link nearestLink = network.getNearestLinkExactly(coord);
352                        
353                        TransitStopFacility stop = tsfi.createTransitStopFacility(Id.create(transitStopKey, TransitStopFacility.class), coord, false);
354//                      ((TransitStopFacilityImpl) stop).setLinkId(nearestLink.getId());
355                        ((TransitStopFacilityImpl) stop).setLinkId(Id.create(transitStopKey +""+ transitStopKey, Link.class));
356                        transitSchedule.addStopFacility(stop);
357                }
358                log.info("Creating transit stops done.");
359                
360                // add transit lines
361                log.info("Creating transit lines ...");
362                long transitLineId = 1;
363                String mode = "pt".intern();
364                for(int i = 0; i < transitStopKeyList.size(); i++){
365                        
366                        
367//                      // for testing 
368//                      if(i == 10)
369//                              break;
370                        
371                        // get transit stop id (origin)
372                        long transitStopA = transitStopKeyList.get(i);
373                        log.info("Processing transit stop " + transitStopA);
374                        
375                        // get related destinations, travel- and waiting times
376                        ArrayList<VISUMObject> list = visumData.get(transitStopA);
377                        Iterator<VISUMObject> listIterator = list.iterator();
378                        
379                        while(listIterator.hasNext()){
380                                
381                                VISUMObject voA2B       = listIterator.next();
382                                
383                                long transitStopB       = voA2B.getDestinationBezirkNummer();
384                                
385                                // transit line A -> B & B -> A
386                                TransitLine transitLine = tsfi.createTransitLine(Id.create(transitLineId, TransitLine.class));
387                                
388                                if(! voA2B.isHandeled() ){                              // direction A -> B
389                                        
390                                        long waitTimeA2B        = voA2B.getWarteZeit() * 60;    // get as seconds
391                                        double travelTimeA2B= voA2B.getReiseZeit() * 60.;       // get as seconds 
392                                        // mark as handled
393                                        voA2B.markAsHandled();
394                                        
395                                        // line A -> B
396                                        createAndAddTransitStopsAndDepartures(tsfi, transitSchedule, mode, 
397                                                                transitStopA, transitStopB, transitLine,
398                                                                travelTimeA2B, waitTimeA2B);
399                                }
400
401                                VISUMObject voB2A = getVISUMObject(transitStopB, transitStopA);
402                                if(voB2A != null && !voB2A.isHandeled()){// direction B -> A
403                                        
404                                        long waitTimeB2A        = voB2A.getWarteZeit() * 60;    // get as seconds
405                                        double travelTimeB2A= voB2A.getReiseZeit() * 60.;       // get as seconds
406                                        // mark as handled
407                                        voB2A.markAsHandled();
408                                        
409                                        // line B -> A (get opposite direction)
410                                        createAndAddTransitStopsAndDepartures(tsfi, transitSchedule, mode, 
411                                                        transitStopB, transitStopA, transitLine,
412                                                        travelTimeB2A, waitTimeB2A);
413                                }
414                                
415                                // add transit line to schedule if not empty
416                                if(transitLine.getRoutes().size() > 0){
417                                        transitLineId++;
418                                        transitSchedule.addTransitLine(transitLine);
419                                }
420                        }
421                        log.info("Created Transit Lines: " + transitLineId);
422                }
423                
424                log.info("Writing TransitSchedule ...");
425                writer = new TransitScheduleWriter(transitSchedule);
426                writer.writeFile(destinationFile);
427                log.info("Writing done!");
428        }

430        /**
431         * @param tsfi
432         * @param transitSchedule
433         * @param mode
434         * @param transitStopOrigin
435         * @param transitStopDestination
436         * @param transitLine
437         * @param travelTime
438         */
439        private void createAndAddTransitStopsAndDepartures(TransitScheduleFactoryImpl tsfi,
440                        TransitSchedule transitSchedule, String mode, long transitStopOrigin,
441                        long transitStopDestination, TransitLine transitLine, double travelTime, long waitTime) {
442                // create transit route stops
443                TransitStopFacility origin          = transitSchedule.getFacilities().get(Id.create(transitStopOrigin, TransitStopFacility.class));
444                TransitStopFacility destination = transitSchedule.getFacilities().get(Id.create(transitStopDestination, TransitStopFacility.class));
445                
446                TransitRouteStop transitRouteStopOrigin         = tsfi.createTransitRouteStop(origin, 0., 0.);
447                TransitRouteStop transitRouteStopDestination= tsfi.createTransitRouteStop(destination, travelTime, 0.);
448                List<TransitRouteStop> stops = new ArrayList<TransitRouteStop>();
449                stops.add(transitRouteStopOrigin);
450                stops.add(transitRouteStopDestination);
451                
452//              Link linkOrigin          = network.getLinks().get(origin.getLinkId());
453//              Link linkDestination = network.getLinks().get(destination.getLinkId());
454//              Node nodeOrigin          = linkOrigin.getToNode();
455//              Node nodeDestination = linkDestination.getToNode();
456//              
457//              Path path = dijkstra.calcLeastCostPath(nodeOrigin, nodeDestination, 0, null, null);
458//              List<Id> linkIds = new ArrayList<Id>();
459//              for(Link l: path.links)
460//                      linkIds.add(l.getId());
461                
462                List<Id<Link>> linkIds = new ArrayList<Id<Link>>();
463                linkIds.add(Id.create(origin.getId() +""+ destination.getId(), Link.class));
464                
465                NetworkRoute route = RouteUtils.createLinkNetworkRouteImpl(origin.getLinkId(), destination.getLinkId());
466                route.setLinkIds(origin.getLinkId(), linkIds, destination.getLinkId());
467
468                TransitRoute transitRoute = tsfi.createTransitRoute(Id.create(transitStopOrigin+"to"+transitStopDestination, TransitRoute.class), route, stops, mode);
469                
470                // set departures
471                int departureId = 1;
472                for(int t = 0; t < secondsOfDay; t += (waitTime*2)){
473                        Departure departure = tsfi.createDeparture(Id.create(getNumberFormat(departureId), Departure.class), t);
474                        departure.setVehicleId(Id.create(departureId, Vehicle.class));
475                        transitRoute.addDeparture(departure);
476                        
477                        departureId++;
478                }
479                
480                transitLine.addRoute(transitRoute);
481        }

483        private VISUMObject getVISUMObject(long origin, long destination){
484                
485                ArrayList<VISUMObject> list = visumData.get( origin );
486                VISUMObject vo = null;
487                
488                for(int i = 0; i < list.size(); i++)
489                        if (list.get(i).getDestinationBezirkNummer() == destination){
490                                vo = list.get(i);
491                                list.remove(i);
492                                break;
493                        }
494                return vo;
495        }

563        public class Counter{
564                
565                private int emptyCounter;
566                private int origins;
567                private int total;
568                
569                public Counter(){
570                        this.emptyCounter       = 0;
571                        this.origins            = 0;
572                        this.total                      = 0;
573                }
574                
575                public void countEmpty(){
576                        this.total++;
577                        this.emptyCounter++;
578                }
579                
580                public void countOrigins(){
581                        this.total++;
582                        this.origins++;
583                }
584                
585                public void countTotal(){
586                        this.total++;
587                }
588                
589                public int getEmptyCounter(){
590                        return this.emptyCounter;
591                }
592                
593                public int getOriginCounter(){
594                        return this.origins;
595                }
596                
597                public int getTotalCounter(){
598                        return this.total;
599                }

602        public class AverageTime{
603                
604                private long sum;
605                private long counter;
606                
607                public AverageTime(){
608                        this.sum        = 0;
609                        this.counter= 0;
610                }
611                
612                public void addTime(long time){
613                        sum += time;
614                        counter++;
615                }
616                
617                public int getAverageTime(){
618                        return (int) (sum/counter);
619                }
