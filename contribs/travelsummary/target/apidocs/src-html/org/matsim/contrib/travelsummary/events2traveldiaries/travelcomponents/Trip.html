001package org.matsim.contrib.travelsummary.events2traveldiaries.travelcomponents;
002
003import org.matsim.api.core.v01.Coord;
004import org.matsim.api.core.v01.Id;
005
006public   class Trip extends TravelComponent {
007        Journey journey;
008        private String mode;
009        private Id line;
010        private Id route;
011        private Coord orig;
012        private Coord dest;
013        private Id boardingStop;
014        private Id alightingStop;
015        private double distance;
016
017        public String toString() {
018                return String
019                                .format("\tTRIP: mode: %s start: %6.0f end: %6.0f distance: %6.0f \n",
020                                                getMode(), getStartTime(), getEndTime(), getDistance());
021        }
022
023        public Id getLine() {
024                return line;
025        }
026
027        public void setLine(Id line) {
028                this.line = line;
029        }
030
031        public Id getRoute() {
032                return route;
033        }
034
035        public void setRoute(Id route) {
036                this.route = route;
037        }
038
039        public Id getBoardingStop() {
040                return boardingStop;
041        }
042
043        public void setBoardingStop(Id boardingStop) {
044                this.boardingStop = boardingStop;
045        }
046
047        public String getMode() {
048                return mode;
049        }
050
051        public void setMode(String mode) {
052                this.mode = mode.trim();
053        }
054
055        public double getDistance() {
056                return distance;
057        }
058
059        public void setDistance(double distance) {
060                this.distance = distance;
061        }
062
063        public Id getAlightingStop() {
064                return alightingStop;
065        }
066
067        public void setAlightingStop(Id alightingStop) {
068                this.alightingStop = alightingStop;
069        }
070
071        public Coord getDest() {
072                return dest;
073        }
074
075        public void setDest(Coord dest) {
076                this.dest = dest;
077        }
078
079        public Coord getOrig() {
080                return orig;
081        }
082
083        public void setOrig(Coord orig) {
084                this.orig = orig;
085        }
086
087        public void incrementDistance(double linkLength) {
088                this.distance += linkLength;
089                
090        }
091
092        public void incrementTime(double linkTime) {
093                this.setEndTime(this.getEndTime()+linkTime);
094                
095        }
096}
