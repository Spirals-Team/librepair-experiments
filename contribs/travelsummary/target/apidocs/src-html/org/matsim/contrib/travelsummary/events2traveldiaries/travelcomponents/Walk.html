001package org.matsim.contrib.travelsummary.events2traveldiaries.travelcomponents;
002
003import org.matsim.api.core.v01.Coord;
004
005public  class Walk extends TravelComponent {
006        public Journey journey;
007        private Coord orig;
008        private Coord dest;
009        private double distance;
010        private boolean accessWalk = false;
011        private boolean egressWalk = false;
012
013        public String toString() {
014                return String.format(
015                                "\tWALK: start: %6.0f end: %6.0f distance: %6.0f \n",
016                                getStartTime(), getEndTime(), getDistance());
017        }
018
019        public boolean isEgressWalk() {
020                return egressWalk;
021        }
022
023        public void setEgressWalk(boolean egressWalk) {
024                this.egressWalk = egressWalk;
025        }
026
027        public boolean isAccessWalk() {
028                return accessWalk;
029        }
030
031        public void setAccessWalk(boolean accessWalk) {
032                this.accessWalk = accessWalk;
033        }
034
035        public Coord getOrig() {
036                return orig;
037        }
038
039        public void setOrig(Coord orig) {
040                this.orig = orig;
041        }
042
043        public Coord getDest() {
044                return dest;
045        }
046
047        public void setDest(Coord dest) {
048                this.dest = dest;
049        }
050
051        public double getDistance() {
052                return distance;
053        }
054
055        public void setDistance(double distance) {
056                this.distance = distance;
057        }
058}
