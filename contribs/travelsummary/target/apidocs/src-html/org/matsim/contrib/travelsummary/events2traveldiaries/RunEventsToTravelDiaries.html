019package org.matsim.contrib.travelsummary.events2traveldiaries;
020
021import org.matsim.api.core.v01.Scenario;
022import org.matsim.core.api.experimental.events.EventsManager;
023import org.matsim.core.config.Config;
024import org.matsim.core.config.ConfigUtils;
025import org.matsim.core.events.EventsManagerImpl;
026import org.matsim.core.events.MatsimEventsReader;
027import org.matsim.core.network.io.MatsimNetworkReader;
028import org.matsim.core.scenario.ScenarioUtils;
029import org.matsim.pt.transitSchedule.api.TransitScheduleReader;
030import org.matsim.vehicles.VehicleReaderV1;
031
032import java.io.File;
033import java.io.IOException;
034
035/**
036 * @author pieterfourie
037 *         <p>
038 *         Running this class with or without arguments prints out instructions for use.
039 *         </p>
040 */
041public class RunEventsToTravelDiaries {
042
043    /**
044     * @param args
045     */
046    public static void main(String[] args) {
047        String eventsFileName = null;
048        Config config = null;
049        String appendage = "";
050        String outputDirectory = null;
051        printHelp();
052        try {
053            config = ConfigUtils.loadConfig(args[0]);
054            outputDirectory = config.controler().getOutputDirectory();
055
056            eventsFileName = args[1];
057
058            if (args.length > 2) {
059                appendage = args[2];
060                if (!(appendage.matches("[_]*[a-zA-Z0-9]*") ||
061                        appendage.matches("[a-zA-Z0-9]*[_]*") ||
062                        appendage.matches("[a-zA-Z0-9]*"))) {
063                    System.err.println("Only alphanumeric and " +
064                            "leading OR trailing underscore characters allowed in " +
065                            "prefix/suffix.");
066                    System.exit(1);
067                }
068            }
069
070            if (args.length > 3) {
071
072                outputDirectory = args[3];
073
074            } else {
075                System.err.println("No output directory specified. " +
076                        "Output directory from config will be used.");
077
078            }
079            //test the output directory first before running the whole analysis
080            File f = new File(outputDirectory);
081            System.out.println("Writing files to " + f.getAbsolutePath());
082            if (!f.exists() && f.canWrite()) {
083                System.err.println("Cannot write to output directory. " +
084                        "Check for existence and permissions");
085                System.exit(1);
086            }
087
088        } catch (ArrayIndexOutOfBoundsException e) {
089            System.exit(1);
090        }
091        Scenario scenario = ScenarioUtils.createScenario(config);
092
093        new MatsimNetworkReader(scenario.getNetwork()).parse(config.network().getInputFileURL(config.getContext()));
094
095        if (config.transit().isUseTransit() ) {
096
097            new TransitScheduleReader(scenario)
098                    .readFile(config.transit().getTransitScheduleFile());
099
100            new VehicleReaderV1(scenario.getTransitVehicles())
101                    .readFile(config.transit().getVehiclesFile());
102
103        }
104
105
106        EventsToTravelDiaries handler =
107                new EventsToTravelDiaries(scenario);
108
109        EventsManager events = new EventsManagerImpl();
110
111        events.addHandler(handler);
112
113        new MatsimEventsReader(events).readFile(eventsFileName);
114
115        try {
116            handler.writeSimulationResultsToTabSeparated(outputDirectory, appendage);
117        } catch (IOException e) {
118            e.printStackTrace();
119        }
120
121    }
122
123    private static void printHelp() {
124        System.out.println(
125                "This class generates tab-separated travel diary tables from MATSim events.\n" +
126                        "It currently works with scenarios that both teleport and fully simulate public\n" +
127                        "transportation.\n\n" +
128                        "It takes the following inputs, in strict order:\n\n" +
129                        "0 (REQUIRED): A config file, specifying the network file, and transit data (if\n" +
130                        "              simulating transit).\n\n" +
131                        "1 (REQUIRED): An events file.\n\n" +
132                        "2 (OPTIONAL): An ALPHANUMERIC prefix or suffix to prepend or append to the beginning or\n" +
133                        "              end of the output file names. Prefixes have a trailing under-\n" +
134                        "              score, suffixes have a leading underscore, e.g. pre_ will\n" +
135                        "              produce pre_matsim_table.txt, and _post will produce\n" +
136                        "              matsim_table_post.txt. Useful when comparing multiple diaries.\n\n" +
137                        "3 (OPTIONAL): An output directory, where the travel diary tables will be\n" +
138                        "              written to, otherwise the output directory from the config\n" +
139                        "              is used.\n\n" +
140                        "4 (OPTIONAL): A maximum number of events to process (for diagnostics).\n\n"
141        );
142    }
143}
