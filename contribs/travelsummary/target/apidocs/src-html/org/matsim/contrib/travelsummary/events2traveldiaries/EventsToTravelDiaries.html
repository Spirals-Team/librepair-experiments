156    @Override
157    public void handleEvent(ActivityEndEvent event) {
158        if (eventCounter.incrementAndGet() < maxEvents) {
159            try {
160                if (isTransitScenario) {
161                    if (transitDriverIds.contains(event.getPersonId()))
162                        return;
163                }
164                TravellerChain chain = chains.get(event.getPersonId());
165                locations.put(event.getPersonId(),
166                        network.getLinks().get(event.getLinkId()).getCoord());
167                if (chain == null) {
168                    chain = new TravellerChain();
169                    chains.put(event.getPersonId(), chain);
170                    Activity act = chain.addActivity();
171                    act.setCoord(network.getLinks().get(event.getLinkId())
172                            .getCoord());
173                    act.setEndTime(event.getTime());
174                    act.setFacility(event.getFacilityId());
175                    act.setStartTime(0.0);
176                    act.setType(event.getActType());
177
178                } else if (!event.getActType().equals(
179                        PtConstants.TRANSIT_ACTIVITY_TYPE)) {
180                    Activity act = chain.getActs().getLast();
181                    act.setEndTime(event.getTime());
182                }
183            } catch (Exception e) {
184                System.err.println(e.getStackTrace());
185                System.err.println(event.toString());
186            }
187        }

190    @Override
191    public void handleEvent(ActivityStartEvent event) {
192        if (eventCounter.incrementAndGet() > maxEvents) return;
193        try {
194            if (isTransitScenario) {
195                if (transitDriverIds.contains(event.getPersonId()))
196                    return;
197            }
198            TravellerChain chain = chains.get(event.getPersonId());
199            boolean beforeInPT = chain.isInPT();
200            if (event.getActType().equals(PtConstants.TRANSIT_ACTIVITY_TYPE)) {
201                chain.setInPT(true);
202
203            } else {
204                chain.setInPT(false);
205                chain.traveling = false;
206                Activity act = chain.addActivity();
207                act.setCoord(network.getLinks().get(event.getLinkId())
208                        .getCoord());
209                act.setFacility(event.getFacilityId());
210                act.setStartTime(event.getTime());
211                act.setType(event.getActType());
212                // end the preceding journey
213                Journey journey = chain.getJourneys().getLast();
214                journey.setDest(act.getCoord());
215                journey.setEndTime(event.getTime());
216                journey.setToAct(act);
217                if (beforeInPT)
218                    journey.getWalks().getLast().setEgressWalk(true);
219            }
220        } catch (Exception e) {
221            System.err.println(e.getStackTrace());
222            System.err.println(event.toString());
223            ;
224        }
225    }

227    @Override
228    public void handleEvent(PersonArrivalEvent event) {
229        if (eventCounter.incrementAndGet() > maxEvents) return;
230        try {
231            if (isTransitScenario) {
232                if (transitDriverIds.contains(event.getPersonId()))
233                    return;
234            }
235            TravellerChain chain = chains.get(event.getPersonId());
236            String legMode = event.getLegMode();
237            String modeType = networkModes.contains(legMode) ? "congested" : legMode;
238            switch (modeType) {
239                case "walk":
240                case "transit_walk": {
241                    Journey journey = chain.getJourneys().getLast();
242                    Walk walk = journey.getWalks().getLast();
243                    walk.setDest(network.getLinks().get(event.getLinkId())
244                            .getCoord());
245                    walk.setEndTime(event.getTime());
246                    walk.setDistance(walk.getDuration() * walkSpeed);
247                    break;
248                }
249                case "congested": {
250                    Journey journey = chain.getJourneys().getLast();
251                    journey.setDest(network.getLinks().get(event.getLinkId())
252                            .getCoord());
253                    journey.setEndTime(event.getTime());
254                    Trip trip = journey.getTrips().getLast();
255                    trip.setDistance(journey.getDistance());
256                    trip.setEndTime(event.getTime());
257                    chain.inCongestedMode = false;
258                    break;
259                }
260                case "pt":
261                    if (isTransitScenario) {
262                        Journey journey = chain.getJourneys().getLast();
263                        Trip trip = journey.getTrips().getLast();
264                        trip.setDest(network.getLinks().get(event.getLinkId())
265                                .getCoord());
266                        trip.setEndTime(event.getTime());
267                        journey.setPossibleTransfer(new Transfer());
268                        journey.getPossibleTransfer().setStartTime(event.getTime());
269                        journey.getPossibleTransfer().setFromTrip(trip);
270                    } else {
271                        Journey journey = chain.getJourneys().getLast();
272                        journey.setEndTime(event.getTime());
273                        journey.setDest(network.getLinks().get(event.getLinkId())
274                                .getCoord());
275                        journey.setEndTime(event.getTime());
276                    }
277                    break;
278                default:
279                    Journey journey = chain.getJourneys().getLast();
280                    journey.setEndTime(event.getTime());
281                    journey.setDest(network.getLinks().get(event.getLinkId())
282                            .getCoord());
283                    journey.setEndTime(event.getTime());
284                    Trip trip = journey.getTrips().getLast();
285                    trip.setDistance(journey.getDistance());
286                    trip.setEndTime(event.getTime());
287                    break;
288            }
289        } catch (Exception e) {
290            System.err.println(e.getStackTrace());
291            System.err.println(event.toString());
292        }
293    }

295    @Override
296    public void handleEvent(PersonDepartureEvent event) {
297        if (eventCounter.incrementAndGet() > maxEvents) return;
298        try {
299            if (transitDriverIds.contains(event.getPersonId()))
300                return;
301            TravellerChain chain = chains.get(event.getPersonId());
302            Journey journey;
303            Trip trip;
304            String legMode = event.getLegMode();
305            String modeType = networkModes.contains(legMode) ? "congested" : legMode;
306            switch (modeType) {
307                case TransportMode.walk:
308                    //fall through to the next
309                case TransportMode.transit_walk:
310                    if (!chain.traveling) {
311                        chain.traveling = true;
312                        journey = chain.addJourney();
313                        journey.setOrig(network.getLinks().get(event.getLinkId())
314                                .getCoord());
315                        journey.setFromAct(chain.getActs().getLast());
316                        journey.setStartTime(event.getTime());
317                        Walk walk = journey.addWalk();
318                        walk.setAccessWalk(true);
319                        walk.setStartTime(event.getTime());
320                        walk.setOrig(journey.getOrig());
321                    } else {
322                        journey = chain.getJourneys().getLast();
323                        Walk walk = journey.addWalk();
324                        walk.setStartTime(event.getTime());
325                        walk.setOrig(network.getLinks().get(event.getLinkId())
326                                .getCoord());
327                        journey.getPossibleTransfer().getWalks().add(walk);
328                    }
329                    break;
330                case "congested":
331                    chain.inCongestedMode = true;
332                    journey = chain.addJourney();
333                    journey.setCarJourney(true);
334                    journey.setOrig(network.getLinks().get(event.getLinkId())
335                            .getCoord());
336                    journey.setFromAct(chain.getActs().getLast());
337                    journey.setStartTime(event.getTime());
338                    trip = journey.addTrip();
339                    trip.setMode(legMode);
340                    trip.setStartTime(event.getTime());
341                    break;
342                case TransportMode.pt:
343                    if (isTransitScenario) {
344                        // person waits till they enter the vehicle
345                        journey = chain.getJourneys().getLast();
346                        Wait wait = journey.addWait();
347                        if (journey.getWaits().size() == 1)
348                            wait.setAccessWait(true);
349                        wait.setStartTime(event.getTime());
350                        wait.setCoord(network.getLinks().get(event.getLinkId())
351                                .getCoord());
352                        if (!wait.isAccessWait()) {
353                            journey.getPossibleTransfer().getWaits().add(wait);
354                        }
355                    } else {
356                        journey = chain.addJourney();
357                        journey.setTeleportJourney(true);
358                        journey.setOrig(network.getLinks().get(event.getLinkId())
359                                .getCoord());
360                        journey.setFromAct(chain.getActs().getLast());
361                        journey.setStartTime(event.getTime());
362                        journey.setMainmode(legMode);
363                        trip = journey.addTrip();
364                        trip.setMode(legMode);
365                        trip.setStartTime(event.getTime());
366                    }
367                    break;
368                default:
369                    journey = chain.addJourney();
370                    journey.setTeleportJourney(true);
371                    journey.setOrig(network.getLinks().get(event.getLinkId())
372                            .getCoord());
373                    journey.setFromAct(chain.getActs().getLast());
374                    journey.setStartTime(event.getTime());
375                    journey.setMainmode(legMode);
376                    trip = journey.addTrip();
377                    trip.setMode(legMode);
378                    trip.setStartTime(event.getTime());
379                    break;
380
381            }
382        } catch (Exception e) {
383            System.err.println(e.getStackTrace());
384            System.err.println(event.toString());
385        }
386    }

404    @Override
405    public void handleEvent(PersonEntersVehicleEvent event) {
406        if (eventCounter.incrementAndGet() > maxEvents) return;
407        try {
408            if (transitDriverIds.contains(event.getPersonId()))
409                return;
410            if (ptVehicles.keySet().contains(event.getVehicleId())) {
411                TravellerChain chain = chains.get(event.getPersonId());
412                Journey journey = chain.getJourneys().getLast();
413                // first, handle the end of the wait
414                journey.getWaits().getLast().setEndTime(event.getTime());
415                // now, create a new trip
416                ptVehicles.get(event.getVehicleId()).addPassenger(
417                        event.getPersonId());
418                Trip trip = journey.addTrip();
419                PTVehicle vehicle = ptVehicles.get(event.getVehicleId());
420                trip.setLine(vehicle.transitLineId);
421                trip.setMode(transitSchedule.getTransitLines()
422                        .get(vehicle.transitLineId).getRoutes()
423                        .get(vehicle.transitRouteId).getTransportMode());
424                trip.setBoardingStop(vehicle.lastStop);
425                trip.setOrig(journey.getWaits().getLast().getCoord());
426                trip.setRoute(ptVehicles.get(event.getVehicleId()).transitRouteId);
427                trip.setStartTime(event.getTime());
428                // check for the end of a transfer
429                if (journey.getPossibleTransfer() != null) {
430                    journey.getPossibleTransfer().setToTrip(trip);
431                    journey.getPossibleTransfer().setEndTime(event.getTime());
432                    journey.addTransfer(journey.getPossibleTransfer());
433                    journey.setPossibleTransfer(null);
434                }
435            } else {
436                // add the person to the map that keeps track of who drives what
437                // vehicle
438                driverIdFromVehicleId.put(event.getVehicleId(),
439                        event.getPersonId());
440            }
441        } catch (Exception e) {
442            System.err.println(e.getStackTrace());
443            System.err.println(event.toString());
444        }
445    }

447    @Override
448    public void handleEvent(PersonLeavesVehicleEvent event) {
449        if (eventCounter.incrementAndGet() > maxEvents) return;
450        if (transitDriverIds.contains(event.getPersonId()))
451            return;
452        try {
453            if (ptVehicles.keySet().contains(event.getVehicleId())) {
454                TravellerChain chain = chains.get(event.getPersonId());
455                chain.traveledVehicle = true;
456                PTVehicle vehicle = ptVehicles.get(event.getVehicleId());
457                double stageDistance = vehicle.removePassenger(event
458                        .getPersonId());
459                Trip trip = chain.getJourneys().getLast().getTrips().getLast();
460                trip.setDistance(stageDistance);
461                trip.setAlightingStop(vehicle.lastStop);
462            } else {
463                driverIdFromVehicleId.remove(event.getVehicleId());
464            }
465
466        } catch (Exception e) {
467            System.err.println(e.getStackTrace());
468            System.err.println(event.toString());
469        }

493    public void handleEvent(LinkLeaveEvent event) {
494        if (eventCounter.incrementAndGet() > maxEvents) return;
495        try {
496            if (ptVehicles.keySet().contains(event.getVehicleId())) {
497                PTVehicle vehicle = ptVehicles.get(event.getVehicleId());
498                if (vehicle.in)
499                    vehicle.in = false;
500                vehicle.incDistance(network.getLinks().get(event.getLinkId())
501                        .getLength());
502
503            } else {
504                TravellerChain chain = chains.get(driverIdFromVehicleId
505                        .get(event.getVehicleId()));
506                if (chain.inCongestedMode) {
507                    Journey journey = chain.getJourneys().getLast();
508                    journey.incrementCarDistance(network.getLinks()
509                            .get(event.getLinkId()).getLength());
510
511                }
512            }
513        } catch (Exception e) {
514            System.err.println(e.getStackTrace());
515            System.err.println(event.toString());

572    public void writeSimulationResultsToTabSeparated(String path, String appendage) throws IOException {
573        String actTableName;
574        String journeyTableName;
575        String transferTableName;
576        String tripTableName;
577        if (appendage.matches("[a-zA-Z0-9]*[_]*")) {
578            actTableName = appendage + "matsim_activities.txt";
579            journeyTableName = appendage + "matsim_journeys.txt";
580            transferTableName = appendage + "matsim_transfers.txt";
581            tripTableName = appendage + "matsim_trips.txt";
582        } else {
583            if (appendage.matches("[a-zA-Z0-9]*"))
584                appendage = "_" + appendage;
585            actTableName = "matsim_activities" + appendage + ".txt";
586            journeyTableName = "matsim_journeys" + appendage + ".txt";
587            transferTableName = "matsim_transfers" + appendage + ".txt";
588            tripTableName = "matsim_trips" + appendage + ".txt";
589        }
590        BufferedWriter activityWriter =
591                IOUtils.getBufferedWriter(path + "/" + actTableName);
592
593        activityWriter.write("activity_id\tperson_id\tfacility_id\ttype\t" +
594                "start_time\tend_time\tx\ty\tsample_selector\n");
595
596        BufferedWriter journeyWriter =
597                IOUtils.getBufferedWriter(path + "/" + journeyTableName);
598        journeyWriter.write("journey_id\tperson_id\tstart_time\t" +
599                "end_time\tdistance\tmain_mode\tfrom_act\tto_act\t" +
600                "in_vehicle_distance\tin_vehicle_time\t" +
601                "access_walk_distance\taccess_walk_time\taccess_wait_time\t" +
602                "first_boarding_stop\tegress_walk_distance\t" +
603                "egress_walk_time\tlast_alighting_stop\t" +
604                "transfer_walk_distance\ttransfer_walk_time\t" +
605                "transfer_wait_time\tsample_selector\n");
606
607        BufferedWriter tripWriter =
608                IOUtils.getBufferedWriter(path + "/" + tripTableName);
609        tripWriter.write("trip_id\tjourney_id\tstart_time\tend_time\t" +
610                "distance\tmode\tline\troute\tboarding_stop\t" +
611                "alighting_stop\tsample_selector\n");
612
613        BufferedWriter transferWriter =
614                IOUtils.getBufferedWriter(path + "/" + transferTableName);
615        transferWriter.write("transfer_id\tjourney_id\tstart_time\t" +
616                "end_time\tfrom_trip\tto_trip\twalk_distance\t" +
617                "walk_time\twait_time\tsample_selector\n");
618
619        //read a static field that increments with every inheriting object constructed
620        Counter counter = new Counter("Output lines written: ");
621        for (Entry<Id, TravellerChain> entry : chains.entrySet()) {
622            String pax_id = entry.getKey().toString();
623            TravellerChain chain = entry.getValue();
624            for (Activity act : chain.getActs()) {
625                try {
626                    activityWriter.write(String.format(
627                            "%d\t%s\t%s\t%s\t%d\t%d\t%f\t%f\t%f\n",
628                            act.getElementId(), pax_id,
629                            act.getFacility(), act.getType(),
630                            (int) act.getStartTime(),
631                            (int) act.getEndTime(),
632                            act.getCoord().getX(),
633                            act.getCoord().getY(),
634                            MatsimRandom.getRandom().nextDouble()
635                    ));
636                } catch (Exception e) {
637                    System.out.println("Couldn't print activity chain!");
638                }
639            }
640            for (Journey journey : chain.getJourneys()) {
641                try {
642                    journeyWriter.write(String.format(
643                            "%d\t%s\t%d\t%d\t%.3f\t%s\t%d\t%d\t%.3f\t%d\t%.3f\t%d\t%d\t%s\t%.3f\t%d\t%s\t%.3f\t%d\t%d\t%f\n",
644                            journey.getElementId(),
645                            pax_id,
646                            (int) journey.getStartTime(),
647                            (int) journey.getEndTime(),
648                            journey.getDistance(),
649                            journey.getMainMode(),
650                            journey.getFromAct().getElementId(),
651                            journey.getToAct().getElementId(),
652                            journey.getInVehDistance(),
653                            (int) journey.getInVehTime(),
654                            journey.getAccessWalkDistance(),
655                            (int) journey.getAccessWalkTime(),
656                            (int) journey.getAccessWaitTime(),
657                            journey.getFirstBoardingStop(),
658                            journey.getEgressWalkDistance(),
659                            (int) journey.getEgressWalkTime(),
660                            journey.getLastAlightingStop(),
661                            journey.getTransferWalkDistance(),
662                            (int) journey.getTransferWalkTime(),
663                            (int) journey.getTransferWaitTime(),
664                            MatsimRandom.getRandom().nextDouble()
665                    ));
666                    counter.incCounter();
667
668                    if (!(journey.isCarJourney() || journey.isTeleportJourney())) {
669                        for (Trip trip : journey.getTrips()) {
670                            tripWriter.write(String.format(
671                                    "%d\t%d\t%d\t%d\t%.3f\t%s\t%s\t%s\t%s\t%s\t%f\n",
672                                    trip.getElementId(),
673                                    journey.getElementId(),
674                                    (int) trip.getStartTime(),
675                                    (int) trip.getEndTime(),
676                                    trip.getDistance(),
677                                    trip.getMode(), trip.getLine(),
678                                    trip.getRoute(), trip.getBoardingStop(),
679                                    trip.getAlightingStop(),
680                                    MatsimRandom.getRandom().nextDouble()
681                            ));
682                            counter.incCounter();
683                        }
684                        for (Transfer transfer : journey.getTransfers()) {
685                            transferWriter.write(String.format(
686                                    "%d\t%d\t%d\t%d\t%d\t%d\t%.3f\t%d\t%d\t%f\n",
687                                    transfer.getElementId(),
688                                    journey.getElementId(),
689                                    (int) transfer.getStartTime(),
690                                    (int) transfer.getEndTime(),
691                                    transfer.getFromTrip()
692                                            .getElementId(),
693                                    transfer.getToTrip()
694                                            .getElementId(),
695
696                                    transfer.getWalkDistance(),
697                                    (int) transfer.getWalkTime(),
698                                    (int) transfer.getWaitTime(),
699                                    MatsimRandom.getRandom().nextDouble()
700
701                            ));
702                            counter.incCounter();
703                        }
704                    } else {
705                        for (Trip trip : journey.getTrips()) {
706
707                            tripWriter.write(String.format(
708                                    "%d\t%d\t%d\t%d\t%.3f\t%s\t%s\t%s\t%s\t%s\t%f\n",
709                                    trip.getElementId(),
710                                    journey.getElementId(),
711                                    (int) trip.getStartTime(),
712                                    (int) trip.getEndTime(),
713                                    journey.isTeleportJourney() ? 0.000 :
714                                            trip.getDistance(),
715                                    trip.getMode(), "", "", "", "",
716                                    MatsimRandom.getRandom().nextDouble()
717
718                            ));
719                            counter.incCounter();
720                        }
721                    }
722                } catch (NullPointerException e) {
723                    setStuck(getStuck() + 1);
724                }
725            }
726
727        }
728
729        activityWriter.close();
730        journeyWriter.close();
731        tripWriter.close();
732        transferWriter.close();
733        counter.printCounter();
734    }
