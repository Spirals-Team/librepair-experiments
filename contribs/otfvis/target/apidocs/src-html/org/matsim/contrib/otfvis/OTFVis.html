096        public static void main(final String[] args) {
097                if (args.length == 0) {
098                        OTFVisGUI.runDialog();
099                } else if ( args[0].equalsIgnoreCase("-convert") || args[0].equalsIgnoreCase("--convert") ) {
100                        convert(args);
101                } else if (args[0].equalsIgnoreCase("-help") || args[0].equalsIgnoreCase("--help") || args[0].equalsIgnoreCase("-?") ) {
102                        convert(args);
103                } else if (args.length == 1) {
104                        String filename = args[0];
105                        play(filename);
106                } else {
107                        printUsage();
108                }
109
110        }
111
112        private static void play(String filename) {
113                String lowerCaseFilename = filename.toLowerCase(Locale.ROOT);
114                if (lowerCaseFilename.endsWith(".mvi")) {
115                        playMVI(filename);
116                } else if ((lowerCaseFilename.endsWith(".xml") || lowerCaseFilename.endsWith(".xml.gz"))) {
117                        FileType type;
118                        type = new MatsimFileTypeGuesser(filename).getGuessedFileType();
119                        if (FileType.Config.equals(type)) {
120                                playConfig(filename);
121                        } else if (FileType.Network.equals(type)) {
122                                playNetwork(filename);
123                        } else {
124                                printUsage();
125                        }
126                }
127        }

141        public static void playConfig(final String[] args) {
142                Config config = ConfigUtils.loadConfig(args[0]);
143                MatsimRandom.reset(config.global().getRandomSeed());
144                log.info("Complete config dump:");
145                StringWriter writer = new StringWriter();
146                new ConfigWriter(config).writeStream(new PrintWriter(writer));
147                log.info("\n\n" + writer.getBuffer().toString());
148                log.info("Complete config dump done.");
149                Scenario scenario = ScenarioUtils.loadScenario(config);
150                playScenario(scenario);
151        }
152        
153        public static void playScenario(Scenario scenario){
154                EventsManager events = EventsUtils.createEventsManager();
155                QSim qSim = QSimUtils.createDefaultQSim(scenario, events);
156
157                OnTheFlyServer server = startServerAndRegisterWithQSim(scenario.getConfig(),scenario, events, qSim);
158                OTFClientLive.run(scenario.getConfig(), server);
159                
160                qSim.run();
161        }

163    public static OnTheFlyServer startServerAndRegisterWithQSim(Config config, Scenario scenario, EventsManager events, QSim qSim) {
164        return startServerAndRegisterWithQSim(config, scenario, events, qSim, null);
165    }

167    public static OnTheFlyServer startServerAndRegisterWithQSim(Config config, Scenario scenario, EventsManager events, QSim qSim,
168            NonPlanAgentQueryHelper nonPlanAgentQueryHelper) {
169                OnTheFlyServer server = OnTheFlyServer.createInstance(scenario, events, qSim, nonPlanAgentQueryHelper);
170
171                if (config.transit().isUseTransit()) {
172
173                        Network network = scenario.getNetwork();
174                        TransitSchedule transitSchedule = scenario.getTransitSchedule();
175                        
176//                      TransitQSimEngine transitEngine = qSim.getTransitEngine();
177//                      AgentTracker agentTracker = transitEngine.getAgentTracker();
178                        
179//                      AgentSnapshotInfoFactory snapshotInfoFactory = qSim.getVisNetwork().getagentsnapshotinfofactory();
180                        SnapshotLinkWidthCalculator linkWidthCalculator = new SnapshotLinkWidthCalculator();
181                        linkWidthCalculator.setLinkWidthForVis( config.qsim().getLinkWidthForVis() );
182                        if (! Double.isNaN(network.getEffectiveLaneWidth())){
183                                linkWidthCalculator.setLaneWidth( network.getEffectiveLaneWidth() );
184                        }
185                        AgentSnapshotInfoFactory snapshotInfoFactory = new AgentSnapshotInfoFactory(linkWidthCalculator);
186
187                        for ( AgentTracker agentTracker : qSim.getAgentTrackers() ) {
188                                FacilityDrawer.Writer facilityWriter = new FacilityDrawer.Writer(network, transitSchedule, agentTracker, snapshotInfoFactory);
189                                server.addAdditionalElement(facilityWriter);
190                        }
191                }
192
193                server.pause();
194                return server;
195        }

197        public static void playNetwork(final String filename) {
198                Config config = ConfigUtils.createConfig();
199                final MutableScenario scenario = (MutableScenario) ScenarioUtils.createScenario(config);
200                new MatsimNetworkReader(scenario.getNetwork()).readFile(filename);
201                EventsManager events = EventsUtils.createEventsManager();
202                final Map<Id<Link>, VisLink> visLinks = new HashMap<>();
203                for (final Id<Link> linkId : scenario.getNetwork().getLinks().keySet()) {
204                        visLinks.put(linkId, new VisLink() {
205                                @Override
206                                public Link getLink() {
207                                        return scenario.getNetwork().getLinks().get(linkId);
208                                }
209
210                                @Override
211                                public Collection<? extends VisVehicle> getAllVehicles() {
212                                        return Collections.emptyList();
213                                }
214
215                                @Override
216                                public VisData getVisData() {
217                                        return new VisData() {
218                                                @Override
219                                                public Collection<AgentSnapshotInfo> addAgentSnapshotInfo(Collection<AgentSnapshotInfo> positions) {
220                                                        return Collections.emptyList();
221                                                }
222                                        };
223                                }
224                        });
225                }
226                OnTheFlyServer server = OnTheFlyServer.createInstance(scenario, events, new VisMobsim() {
227                        @Override
228                        public void run() {
229
230                        }
231
232                        @Override
233                        public void addQueueSimulationListeners(MobsimListener listener) {
234
235                        }
236
237                        @Override
238                        public VisNetwork getVisNetwork() {
239                                return new VisNetwork() {
240                                        @Override
241                                        public Map<Id<Link>, ? extends VisLink> getVisLinks() {
242                                                return visLinks;
243                                        }
244
245                                        @Override
246                                        public Network getNetwork() {
247                                                return scenario.getNetwork();
248                                        }
249                                };
250                        }
251
252                        @Override
253                        public Map<Id<Person>, MobsimAgent> getAgents() {
254                                return Collections.emptyMap();
255                        }
256
257                        @Override
258                        public VisData getNonNetworkAgentSnapshots() {
259                                return new VisData() {
260                                        @Override
261                                        public Collection<AgentSnapshotInfo> addAgentSnapshotInfo(Collection<AgentSnapshotInfo> positions) {
262                                                return Collections.emptyList();
263                                        }
264                                };
265                        }
266                });
267
268                OTFClientLive.run(config, server);
269        }

271        public static void convert(final String[] args) {
272                if ((args.length < 4) || (args.length > 5)) {
273                        printUsage();
274                        return;
275                }
276                String eventFile = args[1];
277                String networkFile = args[2];
278                String mviFile = args[3];
279                int snapshotPeriod = 600;
280                if (args.length == 5) {
281                        snapshotPeriod = Integer.parseInt(args[4]);
282                }
283                Scenario scenario = ScenarioUtils.createScenario(ConfigUtils.createConfig());
284                new MatsimNetworkReader(scenario.getNetwork()).readFile(networkFile);
285                OTFEvent2MVI.convert(scenario, eventFile, mviFile, snapshotPeriod);
286        }
