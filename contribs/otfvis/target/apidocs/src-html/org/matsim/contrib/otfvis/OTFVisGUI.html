210        @Override
211        public void actionPerformed(ActionEvent e) {
212                String command = e.getActionCommand();
213                if (command.equals("Cancel")) {
214                        System.exit(0);
215                } else if (command.equals("Load")) {
216                        // Check configuration
217                        ActionPanel panel = (ActionPanel) tabPane.getSelectedComponent();
218                        if (panel.isFileValid()) {
219                                this.setVisible(false); // We are done here
220                                panel.execute();
221                        } else {
222                                JOptionPane.showMessageDialog(this,
223                                                "Not all Filenames given/valid!");
224                        }
225                }
226        }

228        class ActionPanel extends JPanel {
229                private boolean mustExist;
230                private List<JTextField> fields = new ArrayList<JTextField>();
231
232                public ActionPanel(boolean mustExist) {
233                        this.mustExist = mustExist;
234                }
235
236                public void addTextField(JTextField text) {
237                        fields.add(text);
238                }
239
240                //
241                protected boolean isFieldValid(JTextField field, boolean mustExist) {
242                        boolean valid = true;
243                        String name = field.getText();
244                        // Check if all textfields have been set
245                        valid &= name.length() > 0;
246                        // If they are valid filenames
247                        if (valid) {
248                                File file = new File(name);
249                                boolean exists = file.exists();
250                                if (mustExist) {
251                                        valid &= exists;
252                                } else {
253                                        int answer = JOptionPane.showConfirmDialog(this, name
254                                                        + " exists!\nDo you want to overwrite?");
255                                        valid = (answer == 0);
256                                }
257                        }
258                        return valid;
259                }
260
261                protected boolean isFieldValid2(JTextField field, boolean mustExist) {
262                        boolean valid = true;
263                        // Check if all textfields have been set
264                        valid &= field.getText().length() > 0;
265                        // If they are valid filenames
266                        if (valid && mustExist) {
267                                File file = new File(field.getText());
268                                valid &= file.exists();
269                        }
270                        return valid;
271                }
272
273                public boolean isFileValid() {
274                        boolean valid = true;
275                        for (JTextField field : fields) {
276                                valid &= isFieldValid(field, mustExist);
277                        }
278                        return valid;
279                }
280
281                public void execute() {
282                };
283        }

285        class BrowseAction extends AbstractAction {
286                String type;
287                JTextField text;
288
289                private void setFilter(JFileChooser fc, final String suffix,
290                                final String description) {
291                        fc.setFileFilter(new FileFilter() {
292                                @Override
293                                public boolean accept(File f) {
294                                        return f.isDirectory()
295                                                        || f.getName().toLowerCase(Locale.ROOT)
296                                                                        .endsWith(suffix);
297                                }
298
299                                @Override
300                                public String getDescription() {
301                                        return description;
302                                }
303                        });
304                }
305
306                public BrowseAction(JTextField text, String type) {
307                        super("Browse...", null);
308                        this.text = text;
309                        this.type = type;
310                        putValue(SHORT_DESCRIPTION, "Browse...");
311                }
312
313                public void actionPerformed(ActionEvent e) {
314                        // Which file to browse?
315                        JFileChooser fc = new JFileChooser();
316                        if (type.equals("Net")) {
317                                setFilter(fc, ".xml", "MATSim net file (*.xml)");
318                        } else if (type.equals("Config")) {
319                                setFilter(fc, ".xml", "MATSim config file (*.xml)");
320                        } else if (type.equals("Event")) {
321                                setFilter(fc, ".events.xml.gz",
322                                                "MATSim Event file (*.events.xml.gz)");
323                        } else if (type.equals("MVI")) {
324                                setFilter(fc, ".mvi", "OTFVis movie file (*.mvi)");
325                        }
326
327                        JButton button = (JButton) e.getSource();
328                        Component parent = button.getParent();
329                        int state = fc.showOpenDialog(parent);
330                        if (state == JFileChooser.APPROVE_OPTION) {
331                                String filename = fc.getSelectedFile().getAbsolutePath();
332                                // Set the appropriate text field... we know its a browse button
333                                // parent.get
334                                // filename.;
335                                this.text.setText(filename);
336                        }
337                        System.out.println("No file selected.");
338
339                }
340        }
