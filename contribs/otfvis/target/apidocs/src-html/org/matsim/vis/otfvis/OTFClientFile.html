021package org.matsim.vis.otfvis;
022
023import org.matsim.vis.otfvis.data.OTFClientQuadTree;
024import org.matsim.vis.otfvis.data.OTFConnectionManager;
025import org.matsim.vis.otfvis.data.OTFServerQuadTree;
026import org.matsim.vis.otfvis.data.fileio.OTFFileReader;
027import org.matsim.vis.otfvis.data.fileio.SettingsSaver;
028import org.matsim.vis.otfvis.gui.OTFVisFrame;
029import org.matsim.vis.otfvis.gui.OTFHostControl;
030import org.matsim.vis.otfvis.gui.OTFControlBar;
031import org.matsim.vis.otfvis.gui.OTFTimeLine;
032import org.matsim.vis.otfvis.handler.OTFAgentsListHandler;
033import org.matsim.vis.otfvis.handler.OTFLinkAgentsHandler;
034import org.matsim.vis.otfvis.opengl.drawer.OTFOGLDrawer;
035
036import javax.swing.*;
037import java.awt.*;
038
039/**
040 * This file starts OTFVis using a .mvi file.
041 * 
042 * @author dstrippgen
043 * @author dgrether
044 */
045public class OTFClientFile implements Runnable {
046
047        private final String url;
048        
049        public OTFClientFile(String filename) {
050                super();
051                this.url = filename;
052                
053        }
054
055        @Override
056        public final void run() {
057                SwingUtilities.invokeLater(new Runnable() {
058                        @Override
059                        public void run() {
060                                createDrawer();
061                        }
062                });
063        }
064
065        private void createDrawer() {
066                Component canvas = OTFOGLDrawer.createGLCanvas(new OTFVisConfigGroup());
067                OTFFileReader otfServer = new OTFFileReader(url);
068                final OTFHostControl hostControl = new OTFHostControl(otfServer, canvas);
069                OTFVisConfigGroup otfVisConfig = otfServer.getOTFVisConfig();
070                OTFConnectionManager connect = new OTFConnectionManager();
071                connect.connectWriterToReader(OTFLinkAgentsHandler.Writer.class, OTFLinkAgentsHandler.class);
072                connect.connectWriterToReader(OTFAgentsListHandler.Writer.class, OTFAgentsListHandler.class);
073                OTFServerQuadTree servQ = otfServer.getQuad(connect);
074                OTFClientQuadTree clientQ = servQ.convertToClient(otfServer, connect);
075                OTFOGLDrawer mainDrawer = new OTFOGLDrawer(clientQ, otfVisConfig, canvas, hostControl);
076                OTFControlBar hostControlBar = new OTFControlBar(otfServer, hostControl, mainDrawer);
077                OTFVisFrame otfVisFrame = new OTFVisFrame(canvas, otfServer, hostControlBar, mainDrawer, new SettingsSaver(url));
078                OTFTimeLine timeLine = new OTFTimeLine("time", hostControl);
079                otfVisFrame.getContentPane().add(timeLine, BorderLayout.SOUTH);
080                otfVisFrame.pack();
081        otfVisFrame.setVisible(true);
082        }
083
084}
