055/**
056 * QueryAgentPTBus draws certain public transport related informations.
057 */
058public class QueryAgentPTBus extends AbstractQuery {
059
060        public static class Result implements OTFQueryResult {
061
062                private final List<String> allIds;
063                private float[] vertex = null;
064                private transient FloatBuffer vert;
065                private boolean calcOffset = true;
066
067                public Result(final List<String> allIds) {
068                        this.allIds = allIds;
069                }
070
071                @Override
072                public void draw(OTFOGLDrawer drawer) {
073                        TextRenderer textRenderer = new TextRenderer(new Font("SansSerif", Font.PLAIN, 32), true, false);
074
075                        if(this.vertex == null) return;
076
077                        OGLAgentPointLayer layer = drawer.getCurrentSceneGraph().getAgentPointLayer();
078
079                        if(this.calcOffset) {
080                                float east = (float)drawer.getQuad().offsetEast;
081                                float north = (float)drawer.getQuad().offsetNorth;
082
083                                this.calcOffset = false;
084                                for(int i = 0; i < this.vertex.length; i+=2) {
085                                        this.vertex[i] -=east;
086                                        this.vertex[i+1] -= north;
087                                }
088                                this.vert = Buffers.copyFloatBuffer(FloatBuffer.wrap(this.vertex));
089
090                        }
091
092                        GL2 gl = OTFGLAbstractDrawable.getGl().getGL2();
093                        gl.glEnable(GL.GL_BLEND);
094                        gl.glColor4d(0.6, 0.0,0.2,.2);
095
096                        gl.glEnable(GL.GL_LINE_SMOOTH);
097                        gl.glEnableClientState (GL2.GL_VERTEX_ARRAY);
098                        gl.glLineWidth(1.f*OTFClientControl.getInstance().getOTFVisConfig().getLinkWidth());
099                        gl.glVertexPointer (2, GL.GL_FLOAT, 0, this.vert);
100                        gl.glDrawArrays (GL2.GL_LINE_STRIP, 0, this.vertex.length/2);
101                        gl.glDisableClientState (GL2.GL_VERTEX_ARRAY);
102                        gl.glDisable(GL2.GL_LINE_SMOOTH);
103
104                        for(String id : allIds) {
105                                Point2D.Double pos = layer.getAgentCoords(id.toCharArray());
106                                if (pos == null) continue;
107
108                                //System.out.println("POS: " + pos.x + ", " + pos.y);
109                                gl.glColor4f(0.2f, 0.4f, 0.4f, 0.5f);//Blue
110                                gl.glLineWidth(2);
111                                gl.glBegin(GL.GL_LINE_STRIP);
112                                gl.glVertex3d((float)pos.x + 50, (float)pos.y + 50,0);
113                                gl.glVertex3d((float)pos.x +250, (float)pos.y +250,0);
114                                gl.glEnd();
115                                GLUtils.drawCircle(gl, (float)pos.x, (float)pos.y, 100.f);
116                                InfoText text = new InfoText("Bus " + id, (float)pos.x+ 250, (float)pos.y+ 250);
117                                text.draw(textRenderer, OTFClientControl.getInstance().getMainOTFDrawer().getCanvas(), drawer.getViewBoundsAsQuadTreeRect());
118
119                        }
120                        gl.glDisable(GL.GL_BLEND);
121
122                }
123
124                @Override
125                public void remove() {
126
127                }
128
129                @Override
130                public boolean isAlive() {
131                        return false;
132                }
133
134        }
135
136        private String agentId;
137        private Result result;
138        private final List<String> allIds = new LinkedList<>();
139
140        private Network net = null;
141
142
143        @Override
144        public void setId(String id) {
145                this.agentId = id;
146        }
147
148        private float[] buildRoute(Plan plan) {
149                List<Id<Link>> drivenLinks = new LinkedList<>();
150
151                List<PlanElement> actslegs = plan.getPlanElements();
152                for (PlanElement pe : actslegs) {
153                        if (pe instanceof Activity) {
154                                // handle act
155                                Activity act = (Activity)pe;
156                                drivenLinks.add(act.getLinkId());
157                        } else if (pe instanceof Leg) {
158                                // handle leg
159                                Leg leg = (Leg) pe;
160                                //if (!leg.getMode().equals("car")) continue;
161                                for (Id<Link> linkId : ((NetworkRoute) leg.getRoute()).getLinkIds()) {
162                                        drivenLinks.add(linkId);
163                                }
164                        }
165                }
166
167                if(drivenLinks.size() == 0) return null;
168
169                // convert this to drawable info
170                float[] vertex = new float[drivenLinks.size()*2];
171                int pos = 0;
172                for(Id<Link> linkId : drivenLinks) {
173                        Link link = this.net.getLinks().get(linkId);
174                        Node node = link.getFromNode();
175                        vertex[pos++] = (float)node.getCoord().getX();
176                        vertex[pos++] = (float)node.getCoord().getY();
177                }
178                return vertex;
179        }
180
181        @Override
182        public void installQuery(SimulationViewForQueries simulationView) {
183                this.net = simulationView.getNetwork();
184                this.result = new Result(this.allIds);
185                String prefix = agentId + "-";
186                Collection<Id<Person>> agentIds = simulationView.getMobsimAgents().keySet();
187                for(Id<Person> id : agentIds) {
188                        if(id.toString().startsWith(prefix, 0)) {
189                            allIds.add(id.toString());
190                        }
191                }
192                if (allIds.size()==0) {
193                        return;
194                }
195                MobsimAgent firstAgent = simulationView.getMobsimAgents().get(Id.create(allIds.get(0), Person.class));
196                Plan plan = simulationView.getPlan(firstAgent);
197                this.result.vertex = buildRoute(plan);
198        }
199
200
201        @Override
202        public Type getType() {
203                return OTFQuery.Type.AGENT;
204        }
205
206        @Override
207        public OTFQueryResult query() {
208                return result;
209        }
