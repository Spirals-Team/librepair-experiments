045/**
046 * OGLAgentPointLayer is responsible for drawing the agents/vehicles as point sprites.
047 * It is a very fast way to draw massive (100ks) of agents in realtime.
048 * It does not run too well on ATI cards, though.
049 *
050 * @author dstrippgen
051 *
052 */
053public class OGLAgentPointLayer extends OTFGLAbstractDrawable implements SceneLayer {

055        private final static int BUFFERSIZE = 10000;
056
057        private static FastColorizer redToGreenColorizer = new FastColorizer(
058                                        new double[] { 0.0, 30., 50.}, new Color[] {Color.RED, Color.YELLOW, Color.GREEN});

088        @Override
089        public void onDraw(GL2 gl) {
090                gl.glEnable(GL2.GL_POINT_SPRITE);
091
092                gl.glPointSize(OTFClientControl.getInstance().getOTFVisConfig().getAgentSize() / 10.f);
093
094                gl.glEnableClientState (GL2.GL_COLOR_ARRAY);
095                gl.glEnableClientState (GL2.GL_VERTEX_ARRAY);
096
097                //texture = null;
098                // setting the texture to null means that agents are painted using (software-rendered?) squares.  I have made speed
099                // tests and found on my computer (mac powerbook, with "slow" graphics settings) no difference at all between "null"
100                // and a jpg.  But it looks weird w/o some reasonable icon.  kai, jan'11
101
102                if (texture != null) {
103                        texture.enable(gl);
104                        gl.glEnable(GL2.GL_TEXTURE_2D);
105                        gl.glTexEnvf(GL2.GL_POINT_SPRITE, GL2.GL_COORD_REPLACE, GL2.GL_TRUE);
106                        texture.bind(gl);
107                }
108
109                gl.glDepthMask(false);
110
111                ByteBuffer colors;
112                FloatBuffer vertex;
113                for(int i = 0; i < this.posBuffers.size(); i++) {
114                        colors = this.colBuffers.get(i);
115                        vertex = this.posBuffers.get(i);
116                        int remain = i == this.posBuffers.size()-1 ? this.count %OGLAgentPointLayer.BUFFERSIZE : OGLAgentPointLayer.BUFFERSIZE;
117                        colors.position(0);
118                        vertex.position(0);
119                        gl.glColorPointer (4, GL.GL_UNSIGNED_BYTE, 0, colors);
120                        gl.glVertexPointer (2, GL.GL_FLOAT, 0, vertex);
121                        gl.glDrawArrays (GL.GL_POINTS, 0, remain);
122                }
123
124                gl.glDisableClientState (GL2.GL_COLOR_ARRAY);
125                gl.glDisableClientState (GL2.GL_VERTEX_ARRAY);
126                if (texture != null ) {
127                        texture.disable(gl);
128                }
129
130                gl.glDisable(GL2.GL_POINT_SPRITE);
131        }

135                return 100;

138        public Point2D.Double getAgentCoords(char [] id) {
139                int idNr = Arrays.hashCode(id);
140                Integer i = this.id2coord.get(idNr);
141                if (i != null) {
142                        FloatBuffer vertex = this.posBuffers.get(i / BUFFERSIZE);
143                        int innerIdx = i % BUFFERSIZE;
144                        float x = vertex.get(innerIdx*2);
145                        float y = vertex.get(innerIdx*2+1);
146                        return new Point2D.Double(x,y);
147                }
148                return null;
149        }

151        public void addAgent(AgentSnapshotInfo agInfo) {
152                Color color;
153                switch ( OTFClientControl.getInstance().getOTFVisConfig().getColoringScheme() ) {
154                case bvg:
155                        color = bvgColoringScheme(agInfo);
156                        break;
157                case bvg2:
158                        color = bvg2ColoringScheme(agInfo);
159                        break;
160                case byId:
161                        color = byIdColoringScheme(agInfo);
162                        break;
163                case gtfs:
164                        color = gtfsColoringScheme(agInfo);
165                        break;
166                case standard:
167                        color = standardColoringScheme(agInfo);
168                        break;
169                case taxicab:
170                        color = taxicabColoringScheme(agInfo) ;
171                        break;
172                default:
173                        color = standardColoringScheme(agInfo);
174                        break;
175                }
176
177                if (this.count % OGLAgentPointLayer.BUFFERSIZE == 0) {
178                        this.vertexIN = Buffers.newDirectFloatBuffer(OGLAgentPointLayer.BUFFERSIZE*2);
179                        this.colorsIN = Buffers.newDirectByteBuffer(OGLAgentPointLayer.BUFFERSIZE*4);
180                        this.colBuffers.add(this.colorsIN);
181                        this.posBuffers.add(this.vertexIN);
182                }
183                this.vertexIN.put((float)agInfo.getEasting());
184                this.vertexIN.put((float)agInfo.getNorthing());
185                this.id2coord.put(Arrays.hashCode(agInfo.getId().toString().toCharArray()),this.count);
186
187                this.colorsIN.put( (byte)color.getRed());
188                this.colorsIN.put( (byte)color.getGreen());
189                this.colorsIN.put((byte)color.getBlue());
190                this.colorsIN.put( (byte) ALPHA);
191
192                this.count++;
193        }

195        private static Color taxicabColoringScheme(AgentSnapshotInfo agInfo) {
196        // ===============TAXI COLOURING===============
197        if (agInfo.getId().toString().startsWith("taxi")) {
198            if (agInfo.getAgentState() == AgentState.PERSON_DRIVING_CAR) {
199                return Color.YELLOW ;
200            }
201            else {
202                return Color.BLACK ;
203            }
204        } else {
205            //===============REGULAR COLOURING===============
206            return standardColoringScheme( agInfo ) ;
207        }
208        }
209
210        private static Color standardColoringScheme(AgentSnapshotInfo agInfo) {
211                if ( agInfo.getAgentState()==AgentState.PERSON_DRIVING_CAR ) {
212                        return redToGreenColorizer.getColorZeroOne(agInfo.getColorValueBetweenZeroAndOne());
213                } else if ( agInfo.getAgentState()==AgentState.PERSON_AT_ACTIVITY ) {
214                        return Color.ORANGE;
215                } else if ( agInfo.getAgentState()==AgentState.PERSON_OTHER_MODE ) {
216                        return Color.MAGENTA;
217                } else if ( agInfo.getAgentState()==AgentState.TRANSIT_DRIVER ) {
218                        return Color.BLUE;
219                } else {
220                        return Color.YELLOW;
221                }
222        }
223
224        private static Color byIdColoringScheme(AgentSnapshotInfo agInfo) {
225                String idstr = agInfo.getId().toString() ;
226                int val = 8 ;
227                if ( idstr.hashCode()%val==0 ) {
228                        return Color.red ;
229                } else if (idstr.hashCode()%val==1 ) {
230                        return Color.orange ;
231                } else if (idstr.hashCode()%val==2 ) {
232                        return Color.yellow ;
233                } else if (idstr.hashCode()%val==3 ) {
234                        return Color.green ;
235                } else if (idstr.hashCode()%val==4 ) {
236                        return Color.blue ;
237                } else if (idstr.hashCode()%val==5 ) {
238                        return Color.cyan ;
239                } else if (idstr.hashCode()%val==6 ) {
240                        return Color.magenta ;
241                } else if (idstr.hashCode()%val==7 ) {
242                        return Color.pink ;
243                } else {
244                        return Color.black;
245                }
246
247        }

249        private static int bvg2cnt = 0 ;
250
251        private Color bvg2ColoringScheme(AgentSnapshotInfo agInfo) {
252
253                if ( bvg2cnt < 1 ) {
254                        bvg2cnt++ ;
255                        Logger.getLogger(this.getClass()).info( "using bvg2 coloring scheme ...") ;
256                }
257
258                if ( agInfo.getAgentState()==AgentState.PERSON_DRIVING_CAR ) {
259                        return Color.DARK_GRAY;
260                } else if ( agInfo.getAgentState()==AgentState.PERSON_AT_ACTIVITY ) {
261                        return Color.ORANGE;
262                } else if ( agInfo.getAgentState()==AgentState.TRANSIT_DRIVER ) {
263                        String idstr = agInfo.getId().toString();
264                        if ( idstr.contains("line_") && idstr.contains("-B-") ) {
265                                return Color.MAGENTA;
266                        } else if ( idstr.contains("line_") && idstr.contains("-T-")) {
267                                return Color.RED;
268                        } else if ( idstr.contains("line_SB")) {
269                                return Color.GREEN;
270                        } else if ( idstr.contains("line_U")) {
271                                return Color.BLUE;
272                        } else {
273                                return Color.ORANGE;
274                        }
275                } else {
276                        return Color.YELLOW;
277                }
278        }

280        private static Color bvgColoringScheme(AgentSnapshotInfo agInfo) {
281                if ( agInfo.getAgentState()==AgentState.PERSON_DRIVING_CAR ) {
282                        return Color.DARK_GRAY;
283                } else if ( agInfo.getAgentState()==AgentState.PERSON_AT_ACTIVITY ) {
284                        return Color.ORANGE;
285                } else if ( agInfo.getAgentState()==AgentState.TRANSIT_DRIVER ) {
286                        String idstr = agInfo.getId().toString();
287                        if ( idstr.endsWith("_B") ) {
288                                return Color.MAGENTA;
289                        } else if ( idstr.endsWith("_T") ) {
290                                return Color.RED;
291                        } else if ( idstr.endsWith("_S")) {
292                                return Color.GREEN;
293                        } else if ( idstr.endsWith("_U")) {
294                                return Color.BLUE;
295                        } else {
296                                return Color.ORANGE;
297                        }
298                } else {
299                        return Color.YELLOW;
300                }
301        }

303        private static Color gtfsColoringScheme(AgentSnapshotInfo agInfo) {
304                if ( agInfo.getAgentState()==AgentState.PERSON_DRIVING_CAR ) {
305                        return Color.DARK_GRAY;
306                } else if ( agInfo.getAgentState()==AgentState.PERSON_AT_ACTIVITY ) {
307                        return Color.ORANGE;
308                } else if ( agInfo.getAgentState()==AgentState.TRANSIT_DRIVER ) {
309                        String idstr = agInfo.getId().toString();
310                        if ( idstr.endsWith("bus") ) {
311                                return Color.MAGENTA;
312                        } else if ( idstr.endsWith("tram") ) {
313                                return Color.RED;
314                        } else if ( idstr.endsWith("rail")) {
315                                return Color.GREEN;
316                        } else if ( idstr.endsWith("subway")) {
317                                return Color.BLUE;
318                        } else {
319                                return Color.ORANGE;
320                        }
321                } else {
322                        return Color.YELLOW;
323                }
