043/**
044 * The OTF has a file Reader and a file Writer part.
045 * The Reader is the the mvi playing OTFServer.

049 * @author michaz

051public final class OTFFileReader implements OTFServer {
052
053        private static final Logger log = Logger.getLogger(OTFFileReader.class);
054
055        private final String fileName;
056
057        private File sourceZipFile = null;
058
059        private byte[] currentBuffer = null;
060
061        private double nextTime = -1;
062
063        private final TreeMap<Double, Long> timesteps = new TreeMap<>();
064
065        private OTFVisConfigGroup otfVisConfig;
066
067        public OTFFileReader(final String fname) {
068                this.fileName = fname;
069                this.sourceZipFile = new File(this.fileName);
070                otfVisConfig = readConfigOrUseDefaults();
071                OTFClientControl.getInstance().setOTFVisConfig(otfVisConfig);
072                openAndScanZipFile();
073        }
074
075        private void openAndScanZipFile() {
076                try {
077                        ZipFile zipFile = new ZipFile(this.sourceZipFile, ZipFile.OPEN_READ);
078                        scanZIPFile(zipFile);
079                } catch (IOException e) {
080                        throw new RuntimeException(e);
081                }
082        }
083
084        private void scanZIPFile(ZipFile zipFile) {
085                this.nextTime = -1;
086                Enumeration<? extends ZipEntry> zipFileEntries = zipFile.entries();
087                System.out.println("Scanning timesteps:");
088                Gbl.startMeasurement();
089                while (zipFileEntries.hasMoreElements()) {
090                        ZipEntry entry = zipFileEntries.nextElement();
091                        String currentEntry = entry.getName();
092                        if (currentEntry.contains("step")) {
093                                String[] spliti = StringUtils.explode(currentEntry, '.', 10);
094                                double time_s = Double.parseDouble(spliti[1]);
095                                if (this.nextTime == -1) {
096                                        this.nextTime = time_s;
097                                }
098                                this.timesteps.put(time_s, entry.getSize());
099                                System.out.print(time_s);
100                                System.out.print(", ");
101                        }
102                }
103                System.out.println("");
104                System.out.println("Nr of timesteps: " + this.timesteps.size());
105        }
106
107        @Override
108        public int getLocalTime() {
109                return (int) this.nextTime;
110        }
111
112        @Override
113        public boolean isLive() {
114                return false;
115        }
116
117        @Override
118        public OTFServerQuadTree getQuad(final OTFConnectionManager connect) {
119                log.info("reading quad from file...");
120                return readQuad();
121        }
122
123        private OTFServerQuadTree readQuad() {
124                try {
125                        ZipFile zipFile = new ZipFile(this.sourceZipFile, ZipFile.OPEN_READ);
126                        ZipEntry quadEntry = zipFile.getEntry("quad.bin");
127                        BufferedInputStream is = new BufferedInputStream(zipFile.getInputStream(quadEntry));
128                        OTFServerQuadTree quad = (OTFServerQuadTree) new ObjectInputStream(is).readObject();
129                        log.debug("Read OTFServerQuadI from file, type: " + quad.getClass().getName());
130                        zipFile.close();
131                        return quad;
132                } catch (IOException | ClassNotFoundException e) {
133                        throw new RuntimeException(e);
134                }
135        }
136
137        @Override
138        public byte[] getQuadConstStateBuffer() {
139                try {
140                        ZipFile zipFile = new ZipFile(this.sourceZipFile, ZipFile.OPEN_READ);
141                        ZipEntry entry = zipFile.getEntry("const.bin");
142                        byte[] buffer = new byte[(int) entry.getSize()];
143                        DataInputStream inFile = new DataInputStream(zipFile.getInputStream(entry));
144                        readStateBuffer(inFile, buffer);
145                        inFile.close();
146                        zipFile.close();
147                        return buffer;
148                } catch (IOException e) {
149                        throw new RuntimeException(e);
150                }
151        }
152
153        @Override
154        public byte[] getQuadDynStateBuffer(final Rect bounds) {
155                if (bounds == null) {
156                        log.warn("Bounds are ignored but set! [[I don't understand what this means here.  kai, feb'11]]");
157                }
158                if (this.currentBuffer == null) {
159                        byte[] buffer = null;
160                        try {
161                                buffer = readTimeStep(this.nextTime);
162                        } catch (IOException e) {
163                                e.printStackTrace();
164                        }
165                        this.currentBuffer = buffer;
166                }
167                return this.currentBuffer;
168        }
169
170        private byte[] readTimeStep(final double time_s) throws IOException {
171                int time_string = (int) time_s;
172                ZipFile zipFile = new ZipFile(this.sourceZipFile, ZipFile.OPEN_READ);
173                ZipEntry entry = zipFile.getEntry("step." + time_string + ".bin");
174                byte[] buffer = new byte[(int) this.timesteps.get(time_s) .longValue()];
175                DataInputStream inFile = new DataInputStream(new BufferedInputStream(zipFile.getInputStream(entry), 1000000));
176                readStateBuffer(inFile, buffer);
177                zipFile.close();
178                return buffer;
179        }
180
181        private void readStateBuffer(DataInputStream inputFile, final byte[] result) {
182                try {
183                        double timenextTime = inputFile.readDouble();
184                        int size = inputFile.readInt();
185                        int offset = 0;
186                        int remain = size;
187                        int read = 0;
188                        while ((remain > 0) && (read != -1)) {
189                                read = inputFile.read(result, offset, remain);
190                                remain -= read;
191                                offset += read;
192                        }
193
194                        if (offset != size) {
195                                throw new RuntimeException("READ SIZE did not fit! File corrupted! in second " + timenextTime);
196                        }
197
198                } catch (IOException e) {
199                        throw new RuntimeException(e);
200                }
201        }
202
203        @Override
204        public void requestNewTime(final int time) {
205                Double ceil = timesteps.ceilingKey((double) time);
206                if (ceil != null) {
207                        this.nextTime = ceil;
208                } else if (!timesteps.isEmpty()) {
209                        this.nextTime = timesteps.lastKey();
210                } else {
211                        this.nextTime = -1;
212                }
213                this.currentBuffer = null;
214        }
215
216        @Override
217        public Collection<Double> getTimeSteps() {
218                return this.timesteps.keySet();
219        }
220
221        @Override
222        public void setShowNonMovingItems(boolean showNonMovingItems) {
223                OTFLinkAgentsHandler.showParked = showNonMovingItems;
224        }
225
226        @Override
227        public boolean isFinished() {
228                return getLocalTime() >= timesteps.lastKey();
229        }
230
231        @Override
232        public OTFVisConfigGroup getOTFVisConfig() {
233                return otfVisConfig;
234        }
235
236        private OTFVisConfigGroup readConfigOrUseDefaults() {
237                OTFVisConfigGroup otfVisConfig2 = tryToReadSettingsFromFileNextToMovie();
238                if (otfVisConfig2 == null) {
239                        otfVisConfig2 = new OTFVisConfigGroup();
240                }
241                this.setEffectiveLaneWidthIfNull(otfVisConfig2);
242                return otfVisConfig2;
243        }
244
245        private void setEffectiveLaneWidthIfNull(OTFVisConfigGroup c) {
246                if (c.getEffectiveLaneWidth() == null){
247                        c.setEffectiveLaneWidth(3.75); //default value
248                }
249        }
250        
251        private OTFVisConfigGroup tryToReadSettingsFromFileNextToMovie() {
252                log.debug("Looking for settings in: " + this.fileName);
253                SettingsSaver saver = new SettingsSaver(this.fileName);
254                return saver.tryToReadSettingsFile();
255        }
