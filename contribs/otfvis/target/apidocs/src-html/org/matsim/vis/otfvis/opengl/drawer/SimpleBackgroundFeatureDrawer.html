021package org.matsim.vis.otfvis.opengl.drawer;
022
023import com.jogamp.opengl.GL;
024import com.jogamp.opengl.GL2;
025import com.jogamp.opengl.GLAutoDrawable;
026import com.jogamp.opengl.GLEventListener;
027import com.vividsolutions.jts.geom.*;
028import org.geotools.data.simple.SimpleFeatureIterator;
029import org.geotools.data.simple.SimpleFeatureSource;
030import org.opengis.feature.simple.SimpleFeature;
031
032import java.io.IOException;
033import java.util.ArrayList;
034import java.util.List;
035
036/**
037 * SimpleBackgroundDrawer can draw a geotools feature (e.g. shape file?) on screen.
038 * @author dstrippgen
039 *
040 */
041public class SimpleBackgroundFeatureDrawer implements GLEventListener {
042
043        private static class OTFFeature {
044                final int npoints;
045                float [] xpoints;
046                float [] ypoints;
047                final float[] color;
048                final  int glType;
049
050                OTFFeature(final float[] xpoints, final float[] ypoints, final int npoints, final float[] color, final int glType) {
051                        this.npoints = npoints;
052                        this.color = color;
053                        this.glType = glType;
054                        this.xpoints = xpoints;
055                        this.ypoints = ypoints;
056                }
057        }
058        private final List<OTFFeature> featureList = new ArrayList<>();
059
060        public SimpleBackgroundFeatureDrawer(final SimpleFeatureSource features, final float[] color) throws IOException{
061                SimpleFeatureIterator fIt = features.getFeatures().features();
062                while (fIt.hasNext()) {
063                        final SimpleFeature ft = fIt.next();
064                        featureList.add(toOTFFeature(ft, color));
065                }
066                fIt.close();
067        }
068        
069        private OTFFeature toOTFFeature(final SimpleFeature feature, final float[] color) {
070                final Geometry geo = (Geometry) feature.getDefaultGeometry();
071                final LineString ls;
072                final int glType;
073                if (geo instanceof Polygon) {
074                        ls = ((Polygon) geo).getExteriorRing();
075                        glType = GL2.GL_POLYGON;
076                } else if (geo instanceof MultiPolygon) {
077                        ls = ((Polygon) geo.getGeometryN(0)).getExteriorRing();
078                        glType = GL2.GL_POLYGON;
079                } else if (geo instanceof LineString) {
080                        ls = (LineString) geo;
081                        glType = GL2.GL_LINE_STRIP;
082                } else if (geo instanceof MultiLineString) {
083                        ls = (LineString) geo.getGeometryN(0);
084                        glType = GL2.GL_LINE_STRIP;
085                } else if (geo instanceof Point) {
086                        final GeometryFactory geofac  = new GeometryFactory();
087                        ls = geofac.createLineString(new Coordinate[]{geo.getCoordinate()});
088                        glType = GL.GL_POINTS;
089                } else {
090                        throw new RuntimeException("Could not read Geometry from Feature!!");
091                }
092                final int npoints = ls.getNumPoints();
093                final float [] xpoints = new float[npoints];
094                final float [] ypoints = new float[npoints];
095                for (int i = 0; i < npoints; i++) {
096                        xpoints[i] = (float) (ls.getPointN(i).getCoordinate().x);
097                        ypoints[i] = (float) (ls.getPointN(i).getCoordinate().y);
098                }
099                return new OTFFeature(xpoints, ypoints, npoints, color, glType);
100        }
101
102        @Override
103        public void init(GLAutoDrawable glAutoDrawable) {}
104
105        @Override
106        public void reshape(GLAutoDrawable glAutoDrawable, int i, int i1, int i2, int i3) {}
107
108        @Override
109        public void display(GLAutoDrawable glAutoDrawable) {
110                GL2 gl = (GL2) glAutoDrawable.getGL();
111                for(OTFFeature feature : featureList) {
112                        gl.glColor4f(feature.color[0],feature.color[1],feature.color[2],feature.color[3]);
113                        gl.glBegin(feature.glType);
114                        for (int i =  0; i < feature.npoints; i++) {
115                                gl.glVertex3d(feature.xpoints[i], feature.ypoints[i], 0);       
116                        }
117                        gl.glEnd();
118                }
119        }
120
121        @Override
122        public void dispose(GLAutoDrawable glAutoDrawable) {}
123
124}
