040/**
041 * QueryAgentId is an internal query.
042 * For a given coordinate it tries to find the most likely agent.
043 * It does that by first finding the most likely link to the clicked coordinate and
044 * then the agent nearest to the coordinate on that link.
045 * TODO This might yield unexpected results. It would probably be a better solution to collect
046 * the agents of all surrounding link and then chose the nearest one.
047 *
048 * @author dstrippgen
049 *
050 */
051public class QueryAgentId extends AbstractQuery {
052
053        private final double x;
054        private final double y;
055        private double width = 0;
056        private double height = 0;
057        private Result result;
058
059        public QueryAgentId(double x,double y) {
060                this.x = x;
061                this.y = y;
062        }
063
064        public QueryAgentId(Rectangle2D.Double rect) {
065                this.x = rect.x;
066                this.y = rect.y;
067                this.width = rect.width;
068                this.height = rect.height;
069        }
070
071        @Override
072        public void installQuery(SimulationViewForQueries simulationView) {
073                this.result = new Result();
074                double minDist = Double.POSITIVE_INFINITY;
075                double dist = 0;
076                Collection<AgentSnapshotInfo> positions = new LinkedList<AgentSnapshotInfo>();
077                for(VisLink qlink : simulationView.getVisNetwork().getVisLinks().values()) {
078                        qlink.getVisData().addAgentSnapshotInfo(positions);
079                }
080                positions = simulationView.getNonNetwokAgentSnapshots().addAgentSnapshotInfo(positions);
081                for(AgentSnapshotInfo info : positions) {
082                        if ((info.getAgentState()== AgentState.PERSON_AT_ACTIVITY) && !OTFLinkAgentsHandler.showParked) continue;
083                        java.awt.geom.Point2D.Double xy = OTFServerQuadTree.transform(new Coord(info.getEasting(), info.getNorthing()));
084                        double xDist = xy.getX() - this.x;
085                        double yDist = xy.getY() - this.y;
086                        if (this.width == 0) {
087                                // search for NEAREST agent to given POINT
088                                dist = Math.sqrt(xDist*xDist + yDist*yDist);//"dist_2 = xDist*xDist + yDist*yDist" will work here (no need for "sqrt"), michalm
089                                if(dist < minDist){
090                                        minDist = dist;
091                                        this.result.agentIds.clear();
092                                        this.result.agentIds.add(info.getId().toString());
093                                }
094                        } else {
095                                // search for all agents in given RECT
096                                if( (xDist < this.width) && (yDist < this.height) && (xDist >= 0) && (yDist >= 0) ) {
097                                        this.result.agentIds.add(info.getId().toString());
098                                }
099                        }
100                }
101        }
102
103        @Override
104        public Type getType() {
105                return OTFQuery.Type.OTHER;
106        }
107
108        @Override
109        public void setId(String id) {
110        }
111
112        @Override
113        public OTFQueryResult query() {
114                return this.result;
115        }
116
117        public static class Result implements OTFQueryResult {
118
119                public List<String> agentIds = new ArrayList<String>();
120
121                @Override
122                public void remove() {
123
124                }
125
126                @Override
127                public boolean isAlive() {
128                        return false;
129                }
130
131                @Override
132                public void draw(OTFOGLDrawer drawer) {
133
134                }
135
136        }
137
138}
