323        public static Component createGLCanvas(OTFVisConfigGroup otfVisConfig) {
324                GLCapabilities caps = new GLCapabilities(GLProfile.get(GLProfile.GL2));
325                if (otfVisConfig.isMapOverlayMode()) {
326                        caps.setBackgroundOpaque(false);
327
328                        // A GLJPanel is an OpenGL component which is "more Swing compatible" than a GLCanvas.
329            // The JOGL doc says the tradeoff is that it is slower than a GLCanvas.
330            // We use it if we want to put map tiles behind the agent drawer, because it can be made translucent!
331                        GLJPanel glJPanel = new GLJPanel(caps);
332                        glJPanel.setOpaque(false); // So that the map shines through
333                        return glJPanel;
334                } else {
335                        // This is the default JOGL component. JOGL doc recommends using it if you do not need a GLJPanel.
336                        return new GLCanvas(caps);
337                }
338        }

340        @Override
341        public void display(GLAutoDrawable drawable) {
342                int now = this.hostControlBar.getSimTime();
343                QuadTree.Rect rect;
344                if (nRedrawn > 0) {
345                        rect = this.getViewBoundsAsQuadTreeRect();
346                } else {
347                        // The first time display() is called, it is important that clientQ.getSceneGraph() is called with the whole area rather with what may be visible.
348                        // This is because the display-list based StaticNetLayer is initialized then, and it must contain the whole network.
349                        // Secondly, we can't really know which part is visible the first time, because the window hasn't been opened and doesn't know its size yet.
350                        // So we pass the size of the whole network here and don't rely on anybody else for that.
351                        // michaz May '11
352                        rect = new QuadTree.Rect((float)clientQ.getMinEasting(), (float)clientQ.getMinNorthing(), (float)clientQ.getMaxEasting(), (float)clientQ.getMaxNorthing());
353                }
354                this.currentSceneGraph  = this.clientQ.getSceneGraph(now, rect);
355                if (this.queryHandler != null) {
356                        this.queryHandler.updateQueries();
357                }
358                GL2 gl = drawable.getGL().getGL2();
359                gl.glPolygonMode(GL.GL_FRONT_AND_BACK, GL2.GL_FILL);
360                gl.glEnable(GL.GL_BLEND);
361                gl.glBlendFunc(GL.GL_SRC_ALPHA, GL2.GL_ONE_MINUS_SRC_ALPHA);
362                this.setFrustrum(gl);
363                if (this.currentSceneGraph != null) {
364                        this.currentSceneGraph.draw();
365                }
366
367                if (otfVisConfig.drawTime()) {
368                        drawFrameRate(drawable, now);
369                }
370
371                if((this.currentRect != null) && (this.alpha >= 0.f)){
372                        gl.glEnable(GL2.GL_BLEND);
373                        gl.glBlendFunc(GL2.GL_SRC_ALPHA, GL.GL_ONE_MINUS_SRC_ALPHA);
374                        this.renderFace(gl, this.marker);
375                        gl.glDisable(GL2.GL_BLEND);
376                } else {
377                        this.currentRect = null;
378                        this.alpha = 1.0f;
379                }
380
381                if (this.current == null) {
382                        AWTGLReadBufferUtil glReadBufferUtil = new AWTGLReadBufferUtil(gl.getGLProfile(), true);
383                        this.current = glReadBufferUtil.readPixelsToBufferedImage(gl, false);
384                }
385                gl.glDisable(GL.GL_BLEND);
386                if (viewBounds != null) { // has glInit arrived yet? Otherwise we are not yet fully initialized. :-(
387                        fireChangeListeners();
388                }
389                ++nRedrawn;
390        }

392        private void drawFrameRate(GLAutoDrawable drawable, int now) {
393
394                if (this.statusWidth == 0) {
395                        // Place it at a fixed offset wrt the upper right corner
396                        this.statusWidth = (int) this.textRenderer.getBounds("FPS: 10000.00").getWidth();
397                }
398
399                // Calculate text location and color
400                int x = drawable.getSurfaceWidth() - this.statusWidth - 5;
401                int y = drawable.getSurfaceHeight() - 30;
402
403                // Render the text
404                this.textRenderer.setColor(Color.DARK_GRAY);
405                this.textRenderer.beginRendering(drawable.getSurfaceWidth(), drawable.getSurfaceHeight());
406                this.textRenderer.draw(Time.writeTime(now, ':'), x, y);
407                this.textRenderer.endRendering();
408        }

420        private Point3f getOGLPos(int x, int y) {
421                int[] xy = new int[2]; xy[0] = x; xy[1] = y;
422                ((GLAutoDrawable) canvas).getNativeSurface().convertToPixelUnits(xy);
423                x = xy[0]; y = xy[1];
424                double[] obj_pos = new double[3];
425                float winX, winY;//, winZ = cameraStart.getZ();
426                float posX, posY;//, posZ;
427                double[] w_pos = new double[3];
428        winX = x;
429                winY = viewport[3] - y;
430                GLU glu = new GLU();
431                obj_pos[2]=0; // Check view relative z-koord of layer zero == visnet layer
432                glu.gluProject( obj_pos[0], obj_pos[1],obj_pos[2], modelview,0, projection,0, viewport,0, w_pos,0);
433                glu.gluUnProject( winX, winY, w_pos[2], modelview,0, projection,0, viewport,0, obj_pos,0);
434                posX = (float)obj_pos[0];
435                posY = (float)obj_pos[1];
436                return new Point3f(posX, posY, 0);
437        }

457        @Override
458        public void init(GLAutoDrawable drawable) {
459                GL2 gl = drawable.getGL().getGL2();
460                this.textRenderer = new TextRenderer(new Font("SansSerif", Font.PLAIN, 32), true, false);
461                gl.setSwapInterval(0);
462                float[] components = otfVisConfig.getBackgroundColor().getColorComponents(new float[4]);
463                gl.glClearColor(components[0], components[1], components[2], components[3]);
464                gl.glClear( GL2.GL_COLOR_BUFFER_BIT | GL2.GL_DEPTH_BUFFER_BIT);
465
466                if (!glInited) {
467                        // This method can (and will) be called several times.
468                        // (Gl contexts can change without notice.)
469                        float minEasting = (float)clientQ.getMinEasting();
470                        float minNorthing = (float)clientQ.getMinNorthing();
471                        float maxNorthing = (float)clientQ.getMaxNorthing();
472                        double aspectRatio = (double) drawable.getSurfaceWidth() / (double) drawable.getSurfaceHeight();
473                        double pixelRatio = (double) drawable.getSurfaceHeight() / (double) (maxNorthing-minNorthing);
474                        this.scale = 1.0f / (float) pixelRatio;
475                        this.viewBounds =  new QuadTree.Rect(minEasting, minNorthing, minEasting + (maxNorthing - minNorthing) * aspectRatio, maxNorthing);
476                        if (otfVisConfig.isMapOverlayMode()) {
477                                setScale(nearestPowerOfTwo(this.scale));
478                        }
479                        int time = this.hostControlBar.getSimTime();
480                        QuadTree.Rect rect = new QuadTree.Rect((float)clientQ.getMinEasting(), (float)clientQ.getMinNorthing(), (float)clientQ.getMaxEasting(), (float)clientQ.getMaxNorthing());
481                        this.currentSceneGraph = this.clientQ.getSceneGraph(time, rect);
482                }
483                marker = GLUtils.createTexture(gl, MatsimResource.getAsInputStream("otfvis/marker.png"));
484                currentSceneGraph.glInit();
485                glInited = true;
486        }

500        private void renderFace(GL2 gl, Texture t) {
501                TextureCoords tc = t.getImageTexCoords();
502                float tx1 = tc.left();
503                float ty1 = tc.top();
504                float tx2 = tc.right();
505                float ty2 = tc.bottom();
506                t.enable(gl);
507                t.bind(gl);
508
509                if (button==4) gl.glColor4f(0.8f, 0.2f, 0.2f, alpha);
510                else gl.glColor4f(alpha, alpha, alpha, alpha);
511
512                gl.glBegin(GL2.GL_QUADS);
513                gl.glTexCoord2f(tx1, ty1); gl.glVertex2f(currentRect.x, currentRect.y);
514                gl.glTexCoord2f(tx2, ty1); gl.glVertex2f(currentRect.x, currentRect.y + currentRect.height);
515                gl.glTexCoord2f(tx2, ty2); gl.glVertex2f(currentRect.x + currentRect.width, currentRect.y + currentRect.height);
516                gl.glTexCoord2f(tx1, ty2); gl.glVertex2f(currentRect.x + currentRect.width, currentRect.y);
517                gl.glEnd();
518                t.disable(gl);
519        }

522        public void reshape(GLAutoDrawable drawable, int x, int y, int width, int height) {
523                if (oldWidth != 0.0f) {
524                        double pixelSizeX = (getViewBoundsAsQuadTreeRect().maxX - getViewBoundsAsQuadTreeRect().minX) / oldWidth;
525                        double pixelSizeY = (getViewBoundsAsQuadTreeRect().maxY - getViewBoundsAsQuadTreeRect().minY) / oldHeight;
526                        this.viewBounds = new QuadTree.Rect(getViewBoundsAsQuadTreeRect().minX, getViewBoundsAsQuadTreeRect().maxY - pixelSizeY * height, getViewBoundsAsQuadTreeRect().minX + pixelSizeX * width, getViewBoundsAsQuadTreeRect().maxY);
527                }
528                oldWidth = width;
529                oldHeight = height;
530        }

544                GLU glu = new GLU();
545                gl.glMatrixMode(GL2.GL_PROJECTION);
546                gl.glLoadIdentity();
547                glu.gluOrtho2D(viewBounds.minX, viewBounds.maxX, viewBounds.minY, viewBounds.maxY);
548                gl.glMatrixMode(GL2.GL_MODELVIEW);
549                gl.glLoadIdentity();
550                // update matrices for mouse position calculation
551                gl.glGetDoublev( GL2.GL_MODELVIEW_MATRIX, modelview,0);
552                gl.glGetDoublev( GL2.GL_PROJECTION_MATRIX, projection,0);
553                gl.glGetIntegerv( GL2.GL_VIEWPORT, viewport,0 );
554        }

566        private void showZoomDialog() {
567                this.zoomD = new JDialog();
568                this.zoomD.setUndecorated(true);
569                this.zoomD.setLocationRelativeTo(this.canvas.getParent());
570                Point pD = this.canvas.getLocationOnScreen();
571                this.zoomD.setLocation(pD);
572                this.zoomD.setPreferredSize(this.canvas.getSize());
573                GridLayout gbl = new GridLayout(3,3);
574                this.zoomD.getContentPane().setLayout( gbl );
575                ArrayList<JButton> buttons = new ArrayList<>();
576                final List<ZoomEntry> zooms = otfVisConfig.getZooms();
577                log.debug("Number of zooms: " + otfVisConfig.getZooms().size());
578                for(int i=0; i<zooms.size();i++) {
579                        ZoomEntry z = zooms.get(i);
580                        JButton b = new JButton(z.getName());//icon);
581                        b.setToolTipText(z.getName());
582                        b.setPreferredSize(new Dimension(220, 100));
583                        buttons.add(i, b);
584                        b.setActionCommand(Integer.toString(i));
585                        b.addActionListener( new ActionListener() {
586                                @Override
587                                public void actionPerformed( ActionEvent e ) {
588                                        int num = Integer.parseInt(e.getActionCommand());
589                                        OTFOGLDrawer.this.lastZoom = zooms.get(num);
590                                        Rectangle2D viewBounds1 = OTFOGLDrawer.this.lastZoom.getZoomstart();
591                                        OTFOGLDrawer.this.viewBounds = new Rect(viewBounds1.getMinX(), viewBounds1.getMinY(), viewBounds1.getMaxX(), viewBounds1.getMaxY());
592                                        OTFOGLDrawer.this.zoomD.setVisible(false);
593                                }
594                        } );
595                        this.zoomD.getContentPane().add(b);
596                }
597                JPanel pane = new JPanel();
598                JButton bb = new JButton("Cancel");
599                bb.addActionListener( new ActionListener() {
600                        @Override
601                        public void actionPerformed( ActionEvent e ) {
602                                OTFOGLDrawer.this.lastZoom = null;
603                                OTFOGLDrawer.this.zoomD.setVisible(false);
604                        }
605                } );
606                bb.setPreferredSize(new Dimension(120, 40));
607                pane.add(bb);
608                this.zoomD.getContentPane().add(pane);
609                this.zoomD.doLayout();
610                this.zoomD.pack();
611                for(int i=0; i<zooms.size();i++) {
612                        ZoomEntry z = zooms.get(i);
613                        JButton b = buttons.get(i);
614                        ImageIcon icon = new ImageIcon(ImageUtil.createThumbnail(z.getSnap(),Math.min(z.getSnap().getWidth(),b.getSize().width)-20));
615                        b.setIcon(icon);
616                }
617                this.zoomD.setVisible(true);
618        }
