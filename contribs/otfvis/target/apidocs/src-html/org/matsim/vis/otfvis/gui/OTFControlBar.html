021package org.matsim.vis.otfvis.gui;
022
023import org.matsim.core.gbl.MatsimResource;
024import org.matsim.core.utils.misc.Time;
025import org.matsim.vis.otfvis.OTFClientControl;
026import org.matsim.vis.otfvis.interfaces.OTFServer;
027import org.matsim.vis.otfvis.opengl.drawer.OTFOGLDrawer;
028
029import javax.swing.*;
030import javax.swing.event.ChangeEvent;
031import javax.swing.event.ChangeListener;
032import java.awt.*;
033import java.awt.event.*;
034import java.text.MessageFormat;
035
036public final class OTFControlBar extends JToolBar {
037
038        private static final long serialVersionUID = 1L;
039
040        private JButton playPauseButton;
041
042        private JFormattedTextField timeField;
043
044        private JFormattedTextField scaleField;
045
046        private final OTFHostControl hostControl;
047
048    private boolean synchronizedPlay = true;
049
050        private ImageIcon playIcon = null;
051
052        private ImageIcon pauseIcon = null;
053
054        private OTFAbortGoto progressBar = null;
055
056        private enum PlayOrPause {PLAY, PAUSE}
057
058    private final OTFOGLDrawer drawer;
059
060        /*
061         * The symbol which the play/pause button currently shows.
062         * (NOT the mode we are in!)
063         */
064        private PlayOrPause playOrPause = PlayOrPause.PLAY;
065
066        private OTFServer server;
067
068        public OTFControlBar(OTFServer server, final OTFHostControl hostControl, OTFOGLDrawer mainDrawer)  {
069                this.server = server;
070                this.hostControl = hostControl;
071                this.drawer = mainDrawer;
072                this.setFloatable(false);
073
074                playIcon = new ImageIcon(MatsimResource.getAsImage("otfvis/buttonPlay.png"), "Play");
075                pauseIcon = new ImageIcon(MatsimResource.getAsImage("otfvis/buttonPause.png"), "Pause");
076
077                if (!server.isLive()) {
078                        JButton button2 = new JButton();
079                        button2.putClientProperty("JButton.buttonType","icon");
080                        button2.addActionListener(new ActionListener() {
081                                @Override
082                                public void actionPerformed(ActionEvent e) {
083                                        OTFControlBar.this.hostControl.toStart();
084                                        repaint();
085                                }
086                        });
087                        button2.setToolTipText("restart the server/simulation");
088                        String imgLocation2 = "otfvis/" + "buttonRestart" + ".png";
089                        ImageIcon icon2 = new ImageIcon(MatsimResource.getAsImage(imgLocation2), "Restart");
090                        button2.setIcon(icon2);
091                        add(button2);
092                        JButton button1 = new JButton();
093                        button1.putClientProperty("JButton.buttonType","icon");
094                        button1.addActionListener(new ActionListener() {
095                                @Override
096                                public void actionPerformed(ActionEvent e) {
097                                        int bigStep = OTFClientControl.getInstance().getOTFVisConfig().getBigTimeStep();
098                                        hostControl.requestTimeStep(OTFControlBar.this.hostControl.getSimTime() - bigStep);
099                                }
100                        });
101                        button1.setToolTipText("go several timesteps backwards");
102                        String imgLocation1 = "otfvis/" + "buttonStepBB" + ".png";
103                        ImageIcon icon1 = new ImageIcon(MatsimResource.getAsImage(imgLocation1), "<<");
104                        button1.setIcon(icon1);
105                        add(button1);
106                        JButton button = new JButton();
107                        button.putClientProperty("JButton.buttonType","icon");
108                        button.addActionListener(new ActionListener() {
109                                @Override
110                                public void actionPerformed(ActionEvent e) {
111                                        hostControl.requestTimeStep(OTFControlBar.this.hostControl.getSimTime() - 1);
112                                }
113                        });
114                        button.setToolTipText("go one timestep backwards");
115                        String imgLocation = "otfvis/" + "buttonStepB" + ".png";
116                        ImageIcon icon = new ImageIcon(MatsimResource.getAsImage(imgLocation), "<");
117                        button.setIcon(icon);
118                        add(button);
119                }
120
121                JButton button2 = new JButton();
122                button2.putClientProperty("JButton.buttonType","icon");
123                button2.addActionListener(new ActionListener() {
124                        @Override
125                        public void actionPerformed(ActionEvent e) {
126                                if (playOrPause == PlayOrPause.PLAY) {
127                                        playPauseButton.setIcon(pauseIcon);
128                                        OTFControlBar.this.hostControl.play(synchronizedPlay);
129                                        playOrPause = PlayOrPause.PAUSE;
130                                } else if (playOrPause == PlayOrPause.PAUSE) {
131                                        playPauseButton.setIcon(playIcon);
132                                        OTFControlBar.this.hostControl.pause();
133                                        playOrPause = PlayOrPause.PLAY;
134                                }
135                        }
136                });
137                button2.setToolTipText("press to play simulation continuously");
138                String imgLocation2 = "otfvis/" + "buttonPlay" + ".png";
139                ImageIcon icon2 = new ImageIcon(MatsimResource.getAsImage(imgLocation2), "PLAY");
140                button2.setIcon(icon2);
141                playPauseButton = button2;
142                add(playPauseButton);
143                JButton button1 = new JButton();
144                button1.putClientProperty("JButton.buttonType","icon");
145                button1.addActionListener(new ActionListener() {
146                        @Override
147                        public void actionPerformed(ActionEvent e) {
148                                hostControl.requestTimeStep(OTFControlBar.this.hostControl.getSimTime() + 1);
149                        }
150                });
151                button1.setToolTipText("go one timestep forward");
152                String imgLocation1 = "otfvis/" + "buttonStepF" + ".png";
153                ImageIcon icon1 = new ImageIcon(MatsimResource.getAsImage(imgLocation1), ">");
154                button1.setIcon(icon1);
155                add(button1);
156                JButton button = new JButton();
157                button.putClientProperty("JButton.buttonType","icon");
158                button.addActionListener(new ActionListener() {
159                        @Override
160                        public void actionPerformed(ActionEvent e) {
161                                int bigStep = OTFClientControl.getInstance().getOTFVisConfig().getBigTimeStep();
162                                hostControl.requestTimeStep(OTFControlBar.this.hostControl.getSimTime() + bigStep);
163                        }
164                });
165                button.setToolTipText("go several timesteps forward");
166                String imgLocation = "otfvis/" + "buttonStepFF" + ".png";
167                ImageIcon icon = new ImageIcon(MatsimResource.getAsImage(imgLocation), ">>");
168                button.setIcon(icon);
169                add(button);
170                MessageFormat format = new MessageFormat("{0,number,00}:{1,number,00}:{2,number,00}");
171                timeField = new JFormattedTextField(format);
172                timeField.setMaximumSize(new Dimension(100,30));
173                timeField.setMinimumSize(new Dimension(80,30));
174                timeField.setHorizontalAlignment(JTextField.CENTER);
175                add( timeField );
176                timeField.addActionListener(new ActionListener() {
177                        @Override
178                        public void actionPerformed(ActionEvent e) {
179                                final int newTime_s = (int) Time.parseTime(timeField.getText());
180                                progressBar = new OTFAbortGoto(OTFControlBar.this.server, newTime_s);
181                                progressBar.start();
182                                new Thread() {
183                                        @Override
184                                        public void run() {
185                                                OTFControlBar.this.hostControl.requestTimeStep(newTime_s);
186                                                progressBar.terminate = true;
187                                        }
188                                }.start();
189                        }
190                });
191                hostControl.getSimTimeModel().addChangeListener(new ChangeListener() {
192                        @Override
193                        public void stateChanged(ChangeEvent e) {
194                                SwingUtilities.invokeLater(new Runnable() {
195                                        @Override
196                                        public void run() {
197                                                timeField.setText(Time.writeTime(hostControl.getSimTime()));
198                                        }
199                                });
200                        }
201                });
202                timeField.setText(Time.writeTime(hostControl.getSimTime()));
203
204                if (this.server.isLive()) {
205                        final JCheckBox synchBox = new JCheckBox("Synch");
206                        synchBox.setMnemonic(KeyEvent.VK_V);
207                        synchBox.setSelected(synchronizedPlay);
208                        synchBox.addItemListener(new ItemListener() {
209                                @Override
210                                public void itemStateChanged(ItemEvent e) {
211                                        OTFControlBar.this.hostControl.pause();
212                                        synchronizedPlay = synchBox.isSelected();
213                                        if (playOrPause == PlayOrPause.PAUSE) { // means: playing
214                                                OTFControlBar.this.hostControl.play(synchronizedPlay);
215                                        }
216                                }
217                        });
218                        add(synchBox);
219                }
220
221                JLabel lab = new JLabel("Scale: ");
222                lab.setMaximumSize(new Dimension(100,30));
223                lab.setMinimumSize(new Dimension(80,30));
224                lab.setHorizontalAlignment(JLabel.RIGHT);
225                add(lab);
226                scaleField = new JFormattedTextField("1.0");
227                scaleField.setMaximumSize(new Dimension(50,30));
228                scaleField.setMinimumSize(new Dimension(30,30));
229                scaleField.setHorizontalAlignment(JTextField.CENTER);
230                add( scaleField );
231                scaleField.addActionListener(new ActionListener() {
232                        @Override
233                        public void actionPerformed(ActionEvent e) {
234                                drawer.setScale(Float.parseFloat(scaleField.getText()));
235                        }
236                });
237                drawer.addChangeListener(new ChangeListener() {
238                        @Override
239                        public void stateChanged(ChangeEvent e) {
240                                updateScaleField();
241                        }
242                });
243                updateScaleField();
244        }
245
246        private void updateScaleField() {
247                scaleField.setText(String.valueOf((double) ((float)(Math.round(drawer.getScale()*100))/100)));
248        }
249
250}
