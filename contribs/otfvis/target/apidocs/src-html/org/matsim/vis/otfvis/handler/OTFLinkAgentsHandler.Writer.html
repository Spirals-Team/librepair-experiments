068        public static Point2D.Float calcOrtho(double startx, double starty, double endx, double endy, double len){
069                double dx = endy - starty;
070                double dy = endx -startx;
071                double sqr1 = Math.sqrt(dx*dx +dy*dy);
072
073                dx = dx*len/sqr1;
074                dy = -dy*len/sqr1;
075
076                return new Point2D.Float((float)dx,(float)dy);
077        }

092                private static final transient Collection<AgentSnapshotInfo> positions = new ArrayList<AgentSnapshotInfo>();

094                @Override
095                public void writeConstData(ByteBuffer out) throws IOException {
096                        String id = this.src.getLink().getId().toString();
097                        ByteBufferUtils.putString(out, id);
098                        //subtract minEasting/Northing somehow!
099                        Point2D.Double.Double linkStart = OTFServerQuadTree.transform(this.src.getLink().getFromNode().getCoord());
100                        Point2D.Double.Double linkEnd = OTFServerQuadTree.transform(this.src.getLink().getToNode().getCoord());

102                        out.putFloat((float) linkStart.x); 
103                        out.putFloat((float) linkStart.y);
104                        out.putFloat((float) linkEnd.x);

106                                if ( OTFVisConfigGroup.NUMBER_OF_LANES.equals(OTFClientControl.getInstance().getOTFVisConfig().getLinkWidthIsProportionalTo()) ) {
107                                        out.putInt(NetworkUtils.getNumberOfLanesAsInt(0, this.src.getLink()));
108                                } else if ( OTFVisConfigGroup.CAPACITY.equals(OTFClientControl.getInstance().getOTFVisConfig().getLinkWidthIsProportionalTo()) ) {
109                                        out.putInt( 1 + (int)(2.*this.src.getLink().getCapacity()/3600.) ) ;
110                                        // yyyyyy 3600. is a magic number (the default of the capacity period attribute in Network) but I cannot get to the network (where "capacityPeriod" resides).  
111                                        // Please do better if you know better.  kai, jun'11
112                                } else {
113                                        throw new RuntimeException("I do not understand.  Aborting ..." ) ;
114                                }
115                }

117                @Override
118                public void writeDynData(ByteBuffer out) throws IOException {
119                        out.putFloat((float)0.) ; 
120
121                        positions.clear();
122                        this.src.getVisData().addAgentSnapshotInfo( positions);
123
124                        if (showParked) {
125                                out.putInt(positions.size());
126
127                                for (AgentSnapshotInfo pos : positions) {
128                                        writeAgent(pos, out);
129                                }
130                        } else {
131                                int valid = 0;
132                                for (AgentSnapshotInfo pos : positions) {
133                                        if (pos.getAgentState() != AgentState.PERSON_AT_ACTIVITY)
134                                                valid++;
135                                }
136                                out.putInt(valid);
137
138                                for (AgentSnapshotInfo pos : positions) {
139                                        if (pos.getAgentState() != AgentState.PERSON_AT_ACTIVITY)
140                                                writeAgent(pos, out);
141                                }
142                        }
143                }
144
145                private static void writeAgent(AgentSnapshotInfo pos, ByteBuffer out) {
146                        String id = pos.getId().toString();
147                        ByteBufferUtils.putString(out, id);
148                        Point2D.Double point = OTFServerQuadTree.transform(new Coord(pos.getEasting(), pos.getNorthing()));
149                        out.putFloat((float) point.getX());
150                        out.putFloat((float) point.getY());
151                        out.putInt(pos.getUserDefined());
152                        out.putFloat((float) pos.getColorValueBetweenZeroAndOne());
153                        out.putInt(pos.getAgentState().ordinal());
154                }
155
156                @Override
157                public OTFDataWriter<VisLink> getWriter() {
158                        return new Writer();
159                }
160        }

162        private void readAgent(ByteBuffer in, SceneGraph sceneGraph) {
163                // yyyy there is a very similar method in OTFAgentsListHandler.  with a more robust format, they should be united.  kai, apr'10
164
165                String id = ByteBufferUtils.getString(in);
166                float x = in.getFloat();
167                float y = in.getFloat();
168                int userdefined = in.getInt();
169                float colorValue = in.getFloat();
170                int state = in.getInt();
171
172                AgentSnapshotInfo agInfo = snapshotFactory.createAgentSnapshotInfo(Id.create(id, Person.class), x, y, 0., 0.);
173                agInfo.setColorValueBetweenZeroAndOne(colorValue);
174                agInfo.setUserDefined(userdefined);
175                agInfo.setAgentState(AgentState.values()[state]);
176                sceneGraph.getAgentPointLayer().addAgent(agInfo);
177        }

179        @Override

181                String id = ByteBufferUtils.getString(in);
182                this.quad[0] = new Point2D.Float(in.getFloat(), in.getFloat());
183                this.quad[1] = new Point2D.Float(in.getFloat(), in.getFloat());
184                this.nrLanes = in.getInt();
185                this.id = id.toCharArray();
186        }

188        @Override
189        public void readDynData(ByteBuffer in, SceneGraph graph) throws IOException {
190                this.coloridx = in.getFloat();
191
192                int count = in.getInt();
193                for (int i = 0; i < count; i++) {
194                        readAgent(in, graph);
195                }
196        }

202        public void drawLink() {
203                GL2 gl = GLContext.getCurrentGL().getGL2();
204                if (drawLinkAsLine) {
205                        gl.glLineWidth(1);
206                        gl.glBegin(GL2.GL_LINES);
207                        gl.glVertex2f(quad[0].x, quad[0].y);
208                        gl.glVertex2f(quad[1].x, quad[1].y);
209                        gl.glEnd();
210                } else {
211                        linkWidthCalculator.setLaneWidth(OTFClientControl.getInstance().getOTFVisConfig().getEffectiveLaneWidth());
212                        linkWidthCalculator.setLinkWidthForVis(OTFClientControl.getInstance().getOTFVisConfig().getLinkWidth());
213                        float width = (float) linkWidthCalculator.calculateLinkWidth(this.nrLanes);
214                        final Point2D.Float ortho = calcOrtho(this.quad[0].x, this.quad[0].y, this.quad[1].x, this.quad[1].y, width);
215                        // (yy this is where the width of the links for drawing is set)
216
217                        this.quad[2] = new Point2D.Float(this.quad[0].x + ortho.x, this.quad[0].y + ortho.y);
218                        this.quad[3] = new Point2D.Float(this.quad[1].x + ortho.x, this.quad[1].y + ortho.y);
219                        //Draw quad
220                        TextureCoords co = new TextureCoords(0,0,1,1);
221                        gl.glBegin(GL2.GL_QUADS);
222                        gl.glTexCoord2f(co.right(),co.bottom()); gl.glVertex3f(quad[0].x, quad[0].y, 0);
223                        gl.glTexCoord2f(co.right(),co.top()); gl.glVertex3f(quad[1].x, quad[1].y, 0);
224                        gl.glTexCoord2f(co.left(), co.top()); gl.glVertex3f(quad[3].x, quad[3].y, 0);
225                        gl.glTexCoord2f(co.left(),co.bottom()); gl.glVertex3f(quad[2].x, quad[2].y, 0);
226                        gl.glEnd();
227                }
228        }

230        public void prepareLinkId(Map<Coord, String> linkIds) {
231                double alpha = 0.4;
232                double middleX = alpha*this.quad[0].x + (1.0-alpha)*this.quad[3].x;
233                double middleY = alpha*this.quad[0].y + (1.0-alpha)*this.quad[3].y;
234                String idstr = "" ;
235                if ( id != null ) { // yyyy can't say if this is a meaningful fix but it works for the problem that I have right now.  kai, may'10
236                        idstr = new String(id);
237                }
238                linkIds.put(new Coord(middleX, middleY), idstr);
239        }
