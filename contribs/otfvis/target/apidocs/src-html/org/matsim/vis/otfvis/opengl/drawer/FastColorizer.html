001package org.matsim.vis.otfvis.opengl.drawer;
002
003import org.matsim.vis.otfvis.gui.ValueColorizer;
004
005import java.awt.*;
006
007public class FastColorizer {
008
009        private final int grain;
010        private Color[] fastValues;
011        private double minVal, maxVal, valRange;
012
013        public FastColorizer(double[] ds, Color[] colors) {
014                this(ds, colors, 1000);
015        }
016
017        private FastColorizer(double[] ds, Color[] colors, int grain) {
018                ValueColorizer helper = new ValueColorizer(ds,colors);
019                this.grain = grain;
020                this.fastValues = new Color[grain];
021                this.minVal = ds[0];
022                this.maxVal = ds[ds.length-1];
023                this.valRange = this.maxVal - this.minVal;
024                // calc prerendered Values
025                double step = this.valRange/grain;
026                for(int i = 0; i< grain; i++) {
027                        double value = i*step + this.minVal;
028                        this.fastValues[i] = helper.getColor(value);
029                }
030        }
031
032        public Color getColor(double value) {
033                if (value >= this.maxVal) return this.fastValues[this.grain-1];
034                if (value < this.minVal) return this.fastValues[0];
035                return this.fastValues[(int)((value-this.minVal)*this.grain/this.valRange)];
036        }
037
038        public Color getColorZeroOne( double value ) {
039                if ( value >= 1. ) return this.fastValues[this.grain-1] ;
040                if ( value <= 0. ) return this.fastValues[0] ;
041                return this.fastValues[(int)(value*this.grain)] ;
042        }
043
044}
