021package org.matsim.vis.otfvis.data;
022
023import java.io.Serializable;
024import java.util.ArrayList;
025import java.util.Collection;
026import java.util.HashSet;
027import java.util.Iterator;
028import java.util.LinkedList;
029import java.util.List;
030import java.util.Map;
031
032import org.apache.log4j.Logger;
033import org.matsim.vis.otfvis.caching.SceneLayer;
034import org.matsim.vis.otfvis.interfaces.OTFDataReader;
035import org.matsim.vis.snapshotwriters.VisLink;
036
037/**
038 * The OTFConnectionManager is the most important class when building an OTFVis instance.
039 * It holds pairs of classes. Each of this class-pairs yields as a "From" -> "To" connection between classes.
040 * The whole from-to connections established in a OTFConnectionManager describe the route all data has to
041 * take from the source (normally a QueueLink/Node, etc.) to the actual display on screen.
042 * It is the programmer's responsibility to define a complete chain of responsible objects for all data sent.
043 *
044 *  A chain of responsibility normally consists of
045 *  a DataSource (e.g. QueueLink),
046 *  a DataWriter (e.g. OTFDefaultLinkHandler.Writer)
047 *  a DataReader (e.g. OTFDefaultLinkHandler)
048 *  a Visualizer class (e.g. SimpleStaticNetLayer.QuadDrawer)
049 *  and possibly a layer this Drawer belongs to (e.g. SimpleStaticNetLayer)
050 *  <p></p>
051 *  <ul>
052 *  <li> DataSource -> ByteStream ("Writer")
053 *  <li> ByteStream -> DataSource ("Reader")
054 *  <li> DataSource -> GraphicsObject ("Receiver")
055 *  <li> GraphicsObject -> DrawMechanics ("Layer")
056 *  </ul> 
057 *
058 * @author dstrippgen
059 */
060public class OTFConnectionManager implements Serializable {
061
062        private static final long serialVersionUID = 6481835753628883014L;
063
064        private static final Logger log = Logger.getLogger(OTFConnectionManager.class);
065
066        public OTFConnectionManager() {
067                
068        }
069        
070        private static class Entry implements Serializable {
071
072                private static final long serialVersionUID = -2260651735789627280L;
073
074                Class<?> from, to;
075
076                public Entry(Class<?> from, Class<?> to) {
077                        this.from = from;
078                        this.to = to;
079                }
080
081                @Override
082                public String toString() {
083                        return "(" + from.toString() + "," + to.toString() + ") ";
084                }
085
086                public Class<?> getTo(){
087                        return this.to;
088                }
089
090                public Class<?> getFrom(){
091                        return this.from;
092                }
093
094                @Override
095                public boolean equals(Object obj) {
096                        if (!(obj instanceof Entry)) {
097                                return false;
098                        }
099                        Entry other = (Entry) obj;
100                        return from.equals(other.from) && to.equals(other.to);
101                }
102
103                @Override
104                public int hashCode() {
105                        return from.hashCode() * to.hashCode();
106                }
107
108        }
109
110        private final List<Entry> connections = new LinkedList<Entry>();
111
112        public void connectLinkToWriter(Class<? extends OTFDataWriter<? extends VisLink>> writer) {
113                Entry entry = new Entry(VisLink.class, writer);
114                connections.add(entry);
115        }
116
117        public void connectWriterToReader(Class<? extends OTFDataWriter> writer, Class<? extends OTFDataReader> reader) {
118                Collection<Class<?>> readerClasses = this.getToEntries(writer);
119                if (!readerClasses.isEmpty()) {
120                        throw new RuntimeException("We already have a reader for this writer.");
121                }
122                Entry entry = new Entry(writer, reader);
123                connections.add(entry);
124        }
125
126        public void connectReaderToReceiver(Class<? extends OTFDataReader> reader, Class receiver) {
127                Collection<Class<?>> receiverClasses = this.getToEntries(reader);
128                if (receiverClasses.contains(receiver)) {
129                        log.warn("Trying to connect the same receiver twice. Ignoring: "+receiver);
130                } else {
131                        Entry entry = new Entry(reader, receiver);
132                        connections.add(entry);
133                }
134        }
135
136        public void connectReceiverToLayer(Class receiver, Class<? extends SceneLayer> layer) {
137                Collection<Class<?>> layerClasses = this.getToEntries(receiver);
138                if (layerClasses.contains(layer)) {
139                        log.warn("Trying to connect the same layer twice. Ignoring: "+receiver);
140                } else {
141                        Entry entry = new Entry(receiver, layer);
142                        connections.add(entry);
143                }
144        }
145
146        public void remove(Class<?> from) {
147                Iterator<Entry> iter = connections.iterator();
148                while(iter.hasNext()) {
149                        Entry entry = iter.next();
150                        if (entry.from == from) iter.remove();
151                }
152        }
153
154        public Collection<Class<?>> getToEntries(Class<?> srcClass) {
155                List<Class<?>> classList = new LinkedList<Class<?>>();
156                for(Entry entry : connections) {
157                        if (entry.from.equals(srcClass)) {
158                          classList.add(entry.to);
159                        }
160                }
161                return classList;
162        }
163
164        @SuppressWarnings("unchecked")
165        public Collection<Class<OTFDataReader>> getReadersForWriter(Class<? extends OTFDataWriter> writer) {
166                Collection<Class<?>> classList = getToEntries(writer);
167                Collection<Class<OTFDataReader>> readers = new HashSet<Class<OTFDataReader>>();
168                for(Class<?> entry : classList) {
169                        readers.add((Class<OTFDataReader>) entry);
170                }
171                return readers;
172        }
173
174        public void fillLayerMap(Map<Class<?>, SceneLayer> layers) {
175                Iterator<Entry> iter = connections.iterator();
176                while(iter.hasNext()) {
177                        Entry entry = iter.next();
178                        if (SceneLayer.class.isAssignableFrom(entry.to))
179                                try {
180                                        layers.put(entry.from, (SceneLayer)(entry.to.newInstance()));
181                                } catch (InstantiationException e) {
182                                        e.printStackTrace();
183                                } catch (IllegalAccessException e) {
184                                        e.printStackTrace();
185                                }
186                }
187        }
188
189        public void addEntriesFrom(OTFConnectionManager connect2) {
190                Iterator<Entry> iter = connect2.connections.iterator();
191                while(iter.hasNext()) {
192                        Entry entry = iter.next();
193                        log.info("updating entry: " + entry.from.getCanonicalName() + " to " + entry.to.getName());
194                        this.connections.add(entry);
195                }
196        }
197
198        public void logEntries() {
199                for (Entry e : this.connections){
200                        log.info("writing entry: " + e.getFrom().getCanonicalName() + " to " + e.getTo().getName());
201                }
202        }
203
204        @SuppressWarnings("unchecked")
205        public Collection<Class<OTFWriterFactory<VisLink>>> getLinkWriters() {
206                Collection<Class<OTFWriterFactory<VisLink>>> result = new ArrayList<Class<OTFWriterFactory<VisLink>>>();
207                for (Class<?> clazz : getToEntries(VisLink.class)) {
208                        result.add((Class<OTFWriterFactory<VisLink>>) clazz);
209                }
210                return result;
211        }
212
213}
