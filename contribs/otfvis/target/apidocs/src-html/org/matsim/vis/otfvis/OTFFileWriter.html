020package org.matsim.vis.otfvis;
021
022import java.io.BufferedOutputStream;
023import java.io.DataOutputStream;
024import java.io.FileNotFoundException;
025import java.io.FileOutputStream;
026import java.io.IOException;
027import java.io.ObjectOutputStream;
028import java.nio.ByteBuffer;
029import java.util.ArrayList;
030import java.util.Collection;
031import java.util.zip.ZipEntry;
032import java.util.zip.ZipOutputStream;
033
034import org.matsim.api.core.v01.Scenario;
035import org.matsim.core.config.ConfigUtils;
036import org.matsim.vis.otfvis.data.OTFConnectionManager;
037import org.matsim.vis.otfvis.data.OTFServerQuadTree;
038import org.matsim.vis.otfvis.handler.OTFAgentsListHandler;
039import org.matsim.vis.otfvis.handler.OTFLinkAgentsHandler;
040import org.matsim.vis.snapshotwriters.AgentSnapshotInfo;
041import org.matsim.vis.snapshotwriters.SnapshotWriter;
042import org.matsim.vis.snapshotwriters.VisData;
043/**
044 * The OTF has a file Reader and a file Writer part.
045 * The writer is in charge of writing mvi data into a file.
046 * 
047 * @author dstrippgen 
048 * @author dgrether
049 */
050public final class OTFFileWriter implements SnapshotWriter {
051
052        private static final int BUFFERSIZE = 300000000; // ~300MB
053        private static final int FILE_BUFFERSIZE = 50000000; // ~50MB
054
055        private OTFServerQuadTree quad;
056        private ZipOutputStream zos;
057        private final ByteBuffer buf = ByteBuffer.allocate(BUFFERSIZE);
058        private final OTFConnectionManager connect;
059        private String outFileName;
060        private Collection<AgentSnapshotInfo> positions = new ArrayList<AgentSnapshotInfo>();
061
062        private double lastTime=-1;
063
064        private final OTFAgentsListHandler.Writer writer;
065
066        public OTFFileWriter(Scenario scenario, String outfilename) {
067                this.connect = new OTFConnectionManager();
068                this.connect.connectLinkToWriter(OTFLinkAgentsHandler.Writer.class);
069                this.connect.connectWriterToReader(OTFLinkAgentsHandler.Writer.class, OTFLinkAgentsHandler.class);
070                this.outFileName = outfilename;
071                this.quad = new SnapshotWriterQuadTree(scenario.getNetwork());
072                this.quad.initQuadTree(connect);
073                this.writer = new OTFAgentsListHandler.Writer();
074                this.writer.setSrc(new VisData() {
075
076                        @Override
077                        public Collection<AgentSnapshotInfo> addAgentSnapshotInfo(Collection<AgentSnapshotInfo> inPositions) {
078                                inPositions.addAll(positions);
079                                return inPositions;
080                        }
081                        
082                });
083                if (ConfigUtils.addOrGetModule(scenario.getConfig(), OTFVisConfigGroup.GROUP_NAME, OTFVisConfigGroup.class) != null) {
084                        ConfigUtils.addOrGetModule(scenario.getConfig(), OTFVisConfigGroup.GROUP_NAME, OTFVisConfigGroup.class).setEffectiveLaneWidth(scenario.getNetwork().getEffectiveLaneWidth());
085                }
086                OTFClientControl.getInstance().setOTFVisConfig(ConfigUtils.addOrGetModule(scenario.getConfig(), OTFVisConfigGroup.GROUP_NAME, OTFVisConfigGroup.class));
087                try {
088                        this.zos = new ZipOutputStream(new BufferedOutputStream(new FileOutputStream(this.outFileName), FILE_BUFFERSIZE));
089                        this.writeQuad();
090                        this.writeConstData();
091                } catch (FileNotFoundException e) {
092                        throw new RuntimeException(e);
093                }
094        }
095
096        private void writeQuad() {
097                try {
098                        this.zos.putNextEntry(new ZipEntry("quad.bin"));
099                        this.quad.addAdditionalElement(this.writer);
100                        new ObjectOutputStream(this.zos).writeObject(this.quad);
101                        this.zos.closeEntry();
102                } catch (IOException e) {
103                        throw new RuntimeException(e);
104                }
105        }
106
107        private void writeConstData() {
108                try {
109                        this.zos.putNextEntry(new ZipEntry("const.bin"));
110                        DataOutputStream outFile = new DataOutputStream(this.zos);
111                        this.buf.position(0);
112                        outFile.writeDouble(-1.);
113                        this.quad.writeConstData(this.buf);
114                        outFile.writeInt(this.buf.position());
115                        outFile.write(this.buf.array(), 0, this.buf.position());
116                        this.zos.closeEntry();
117                } catch (IOException e) {
118                        throw new RuntimeException(e);
119                }
120        }
121
122        @Override
123        public void beginSnapshot(final double time) {
124                this.positions.clear();
125                this.lastTime = time;
126        }
127
128        private void writeDynData(final int time_s) {
129                try {
130                        this.zos.putNextEntry(new ZipEntry("step." + time_s + ".bin"));
131                        DataOutputStream outFile = new DataOutputStream(this.zos);
132                        this.buf.position(0);
133                        outFile.writeDouble(time_s);
134                        this.quad.writeDynData(null, this.buf);
135                        outFile.writeInt(this.buf.position());
136                        outFile.write(this.buf.array(), 0, this.buf.position());
137                        this.zos.closeEntry();
138                } catch (IOException e) {
139                        e.printStackTrace();
140                }
141        }
142
143        @Override
144        public void addAgent(final AgentSnapshotInfo position) {
145                if (position.getAgentState() != AgentSnapshotInfo.AgentState.PERSON_AT_ACTIVITY) {
146                        this.positions.add(position);
147                }
148        }
149
150        @Override
151        public void endSnapshot() {
152                writeDynData((int)this.lastTime);
153        }
154
155        @Override
156        public void finish() {
157                try {
158                        this.zos.close();
159                } catch (IOException e) {
160                        throw new RuntimeException(e);
161                }
162        }
163
164}
