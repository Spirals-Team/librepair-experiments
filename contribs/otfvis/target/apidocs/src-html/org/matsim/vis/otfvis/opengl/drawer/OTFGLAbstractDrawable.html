021package org.matsim.vis.otfvis.opengl.drawer;
022
023import com.jogamp.opengl.GL2;
024import com.jogamp.opengl.GLAutoDrawable;
025import com.jogamp.opengl.GLContext;
026
027
028/**
029 * History:<ul>
030 * <li> There used to be an "OTFGLDrawableImpl implements OTFOGLDrawable".
031 * This class here is replacing both the Impl (which was abstract anyway) and the interface (which was only used for this
032 * one abstract implementation.
033 * <li> An advantage of making this abstract instead of an interface is that onDraw can remain protected, only exposing
034 * draw() as the true public interfaces (as it should be as far as I can currently see).
035 * <li> There used to be an "isValid" but as far as I could tell it did not do anything.
036 * </ul>
037 * kai, feb'11
038 * 
039 * @author dstrippgen
040 *
041 */
042public abstract class OTFGLAbstractDrawable {
043
044        public final void draw() {
045                onDraw(GLContext.getCurrentGL().getGL2());
046        }
047        
048        /**
049         * 
050         * This is the OpenGL init command passed down. OpenGL may call init any time, for example when switching
051         * displays, re-creating buffers etc., so layers need to be notified of this.
052         * Particularly, the OGLSimpleStaticNetLayer needs this message to recreate its OpenGL display lists
053         * when the OpenGL context changes.
054         */
055        public final void glInit() {
056                onInit(GLContext.getCurrentGL().getGL2());
057        }
058        
059        protected void onInit(GL2 gl) {
060                // To override.
061        }
062        
063        abstract protected void onDraw( GL2 gl ) ;
064
065        public final static GL2 getGl() {
066                return GLContext.getCurrentGL().getGL2();
067        }
068
069}
