021package org.matsim.vis.otfvis.data;
022
023import java.io.IOException;
024import java.io.Serializable;
025import java.nio.ByteBuffer;
026
027
028/**
029 * OTFDataWriter<SrcData> is the base of all Writer classes. SrcData is most likely a
030 * QueueLink, QueueNode or likewise.
031 * 
032 * @author dstrippgen
033 *
034 * @param <SrcData>
035 */
036public abstract class OTFDataWriter<SrcData> implements Serializable {
037
038        private static final long serialVersionUID = 7593448140900220038L;
039        
040        protected transient SrcData src;
041        
042        abstract public void writeConstData(ByteBuffer out) throws IOException;
043        abstract public void writeDynData(ByteBuffer out) throws IOException;
044        
045        public void setSrc(SrcData src) {
046                this.src = src;
047        }
048        
049        public SrcData getSrc() {
050                return this.src;
051        }
052
053}
