042/**
043 * QueryLinkId find the most likely link (or links) to a given coordinate (or rectangle).
044 * The meaning of "nearest" is that of the smallest distance between the middle point of the link
045 * and the given coordinate. 
046 * TODO This might lead to unexpected behavior with long links. 
047 * 
048 * @author dstrippgen
049 *
050 */
051public class QueryLinkId extends AbstractQuery {
052
053        public static class Result implements OTFQueryResult {
054
055                public Map<Coord, String> linkIds = new HashMap<Coord, String>();
056
057                @Override
058                public void remove() {
059
060                }
061                
062                @Override
063                public boolean isAlive() {
064                        return false;
065                }
066
067                @Override
068                public void draw(OTFOGLDrawer drawer) {
069
070                }
071
072        }
073
074        private final double sx;
075        private final double sy;
076        private double width = 0;
077        private double height = 0;
078        
079        private Result result;
080
081        public QueryLinkId(Rectangle2D.Double rect) {
082                this.sx = rect.x;
083                this.sy = rect.y;
084                this.width = rect.width;
085                this.height = rect.height;
086        }
087
088        class AddIdStringExecutor implements Executor<OTFDataWriter> {
089                private final boolean nearestOnly;
090                private double minDist = Double.POSITIVE_INFINITY;
091                private static final double epsilon = 0.0001;
092                private final double cellWidth;
093                
094                public AddIdStringExecutor(boolean nearestOnly) {
095                        this.nearestOnly = nearestOnly;
096                        cellWidth = OTFClientControl.getInstance().getOTFVisConfig().getLinkWidth();
097                }
098                
099                @Override
100                public void execute(double x, double y, OTFDataWriter writer)  {
101                        Object src = writer.getSrc();
102                        if(src instanceof VisLink) {
103                                Link link = ((VisLink)src).getLink();
104                                double alpha = 0.6;
105                                Coord from = link.getFromNode().getCoord();
106                                java.awt.geom.Point2D.Double transformedFrom = OTFServerQuadTree.transform(from);
107                                double fromX = transformedFrom.getX();
108                                double fromY = transformedFrom.getY();
109                                Coord to = link.getToNode().getCoord();
110                                java.awt.geom.Point2D.Double transformedTo = OTFServerQuadTree.transform(to);
111                                double middleX = alpha*fromX + (1.0-alpha)*transformedTo.getX();
112                                double middleY = alpha*fromY + (1.0-alpha)*transformedTo.getY();
113                                if (nearestOnly) {
114                                        double xDist = middleX - sx;
115                                        double yDist = middleY - sy;
116                                        // search for NEAREST agent to given POINT
117                                        double dist = Math.sqrt(xDist*xDist + yDist*yDist);
118                                        if(dist <= minDist){
119                                                // is this just about the same distance, then put both into account
120                                                if (minDist - dist > epsilon) result.linkIds.clear();
121
122                                                minDist = dist;
123                                                Point2D.Float anchor = OTFLinkAgentsHandler.calcOrtho(fromX, fromY, middleX, middleY, cellWidth/2.);
124                                                result.linkIds.put(new Coord(middleX + anchor.x, middleY + anchor.y), link.getId().toString());
125                                        }
126
127                                } else {
128                                        Point2D.Float anchor = OTFLinkAgentsHandler.calcOrtho(fromX, fromY, middleX, middleY, cellWidth/2.);
129                                        result.linkIds.put(new Coord(middleX + anchor.x, middleY + anchor.y), link.getId().toString());
130                                }
131                        }
132                }
133        }
134        
135        @Override
136        public void installQuery(SimulationViewForQueries simulationView) {
137                this.result = new Result();
138                OTFServerQuadTree quad = simulationView.getNetworkQuadTree();
139                double regionWidth = (quad.getMaxEasting()-quad.getMinEasting())*0.1;
140                double regionHeight = (quad.getMaxNorthing()-quad.getMinNorthing())*0.1;
141
142                QuadTree.Rect rect;
143                if (width == 0) {
144                        rect = new QuadTree.Rect(sx-regionWidth, sy-regionHeight, sx+regionWidth, sy+regionHeight);
145                } else {
146                        rect = new QuadTree.Rect(sx,sy,sx+width,sy+height);
147                }
148                quad.execute(rect, new AddIdStringExecutor(width == 0));
149        }
150
151        @Override
152        public Type getType() {
153                return OTFQuery.Type.OTHER;
154        }
155
156        @Override
157        public void setId(String id) {
158        }
159
160        @Override
161        public OTFQueryResult query() {
162                return result;
163        }
164
165}
