041/**
042 * OTFServerQuad is the quad representation of all elements of the network on the server
043 * side. This QuadTree is mirrored on the client side by OTFClientQuad.
044 *
045 * @author dstrippgen
046 *
047 */
048public abstract class OTFServerQuadTree extends QuadTree<OTFDataWriter> {
049
050        private static final Logger log = Logger.getLogger(OTFServerQuadTree.class);
051
052        private final List<OTFDataWriter> additionalElements = new LinkedList<OTFDataWriter>();
053
054        private static final long serialVersionUID = 1L;
055        private double minEasting;
056        private double maxEasting;
057        private double minNorthing;
058        private double maxNorthing;
059        private double easting;
060        private double northing;
061
062        private  static OTFTransformation otfTransformation = null;
063        private static CoordinateTransformation transformation = new IdentityTransformation();
064
065        public OTFServerQuadTree(Network network) {
066                super(0,0,0,0);
067                this.setBoundingBoxFromNetwork(network);
068        }
069
070        public abstract void initQuadTree(final OTFConnectionManager connect);
071
072        protected void setBoundingBoxFromNetwork(Network n){
073                this.minEasting = Double.POSITIVE_INFINITY;
074                this.maxEasting = Double.NEGATIVE_INFINITY;
075                this.minNorthing = Double.POSITIVE_INFINITY;
076                this.maxNorthing = Double.NEGATIVE_INFINITY;
077
078                for (org.matsim.api.core.v01.network.Node node : n.getNodes().values()) {
079                        Coord coord = transformation.transform(node.getCoord());
080                        this.minEasting = Math.min(this.minEasting, coord.getX());
081                        this.maxEasting = Math.max(this.maxEasting, coord.getX());
082                        this.minNorthing = Math.min(this.minNorthing, coord.getY());
083                        this.maxNorthing = Math.max(this.maxNorthing, coord.getY());
084                }
085                // make sure, the bounding box is bigger than the biggest element, otherwise
086                // requests with null == use biggest bounding box will fail on the leftmost elements
087                this.maxEasting +=1;
088                this.maxNorthing +=1;
089
090                otfTransformation = new OTFTransformation(transformation, this.minEasting, this.minNorthing);
091//              log.debug("offsetEast: " + offsetEast + " offsetNorth: " + offsetNorth + " " + transformation);
092                this.easting = this.maxEasting - this.minEasting;
093                this.northing = this.maxNorthing - this.minNorthing;
094                // set top node
095                setTopNode(0, 0, easting, northing);
096        }
097
098
099        /**
100         * Sets a new top node in case the extremities from the c'tor are not
101         * good anymore, it also clear the QuadTree
102         * @param minX The smallest x coordinate expected
103         * @param minY The smallest y coordinate expected
104         * @param maxX The largest x coordinate expected
105         * @param maxY The largest y coordinate expected
106         */
107        protected void setTopNode(final double minX, final double minY, final double maxX, final double maxY) {
108                this.top = new Node<OTFDataWriter>(minX, minY, maxX, maxY);
109        }
110
111        public void addAdditionalElement(OTFDataWriter element) {
112                this.additionalElements.add(element);
113        }
114
115        public OTFClientQuadTree convertToClient(final OTFServer host, final OTFConnectionManager connect) {
116                final OTFClientQuadTree client = new OTFClientQuadTree(host, 0., 0.,this.easting, this.northing);
117                client.offsetEast = this.minEasting;
118                client.offsetNorth = this.minNorthing;
119
120                this.execute(0.,0.,this.easting, this.northing,
121                                new ConvertToClientExecutor(connect,client));
122
123                for(OTFDataWriter<?> element : this.additionalElements) {
124                        Collection<Class<OTFDataReader>> readerClasses = connect.getReadersForWriter(element.getClass());
125                        for (Class<? extends OTFDataReader> readerClass : readerClasses) {
126                                try {
127                                        Object reader = readerClass.newInstance();
128                                        client.addAdditionalElement((OTFDataReader)reader);
129                                        log.info("Connected additional element writer " + element.getClass().getName() + "(" + element + ")  to " + reader.getClass().getName() + " (" + reader + ")");
130                                } catch (InstantiationException e) {
131                                        throw new RuntimeException(e);
132                                } catch (IllegalAccessException e) {
133                                        throw new RuntimeException(e);
134                                }
135                        }
136
137                }
138                return client;
139        }
140
141        public void writeConstData(ByteBuffer out) {
142
143                for (OTFDataWriter element : this.values()) {
144                        try {
145                                element.writeConstData(out);
146                        } catch (IOException e) {
147                                e.printStackTrace();
148                        }
149                }
150
151                for (OTFDataWriter element : this.additionalElements) {
152                        try {
153                                element.writeConstData(out);
154                        } catch (IOException e) {
155                                e.printStackTrace();
156                        }
157                }
158        }
159
160        public void writeDynData(QuadTree.Rect bounds, ByteBuffer out) {
161
162                this.execute(bounds, new WriteDataExecutor(out,false));
163                // "this" is a quad tree; thus this command somehow marks the relevant leaves
164
165                for(OTFDataWriter element : this.additionalElements) {
166                        try {
167                                element.writeDynData(out);
168                        } catch (IOException e) {
169                                e.printStackTrace();
170                        }
171                }
172        }
173
174        // Internally we hold the coordinates from 0,0 to max -min .. to optimize use of float in visualizer
175        @Override
176        public double getMaxEasting() {
177                return this.maxEasting;
178        }
179
180        @Override
181        public double getMaxNorthing() {
182                return this.maxNorthing;
183        }
184
185        @Override
186        public double getMinEasting() {
187                return this.minEasting;
188        }
189
190        @Override
191        public double getMinNorthing() {
192                return this.minNorthing;
193        }
194
195        public static Point2D.Double transform(Coord coord) {
196                Coord transformedCoord = otfTransformation.transform(coord);
197                Point2D.Double result = new Point2D.Double(transformedCoord.getX(), transformedCoord.getY());
198                return result;
199        }
200        
201        public static CoordinateTransformation getOTFTransformation(){
202                return otfTransformation;
203        }
204
205        public static void setTransformation(CoordinateTransformation transformation) {
206                OTFServerQuadTree.transformation = transformation;
207        }
208
209        private static class ConvertToClientExecutor implements Executor<OTFDataWriter> {
210                final OTFConnectionManager connect;
211                final OTFClientQuadTree client;
212
213                public ConvertToClientExecutor(OTFConnectionManager connect2, OTFClientQuadTree client) {
214                        this.connect = connect2;
215                        this.client = client;
216                }
217                @Override
218                @SuppressWarnings("unchecked")
219                public void execute(double x, double y, OTFDataWriter writer)  {
220                        Collection<Class<?>> readerClasses = this.connect.getToEntries(writer.getClass());
221                        for (Class readerClass : readerClasses) {
222                                try {
223                                        OTFDataReader reader = (OTFDataReader)readerClass.newInstance();
224                                        reader.setSrc(writer);
225                                        this.client.put(x, y, reader);
226                                } catch (Exception e) {
227                                        e.printStackTrace();
228                                        throw new RuntimeException();
229                                }
230                        }
231                }
232        }
233
234
235        private static class WriteDataExecutor implements Executor<OTFDataWriter> {
236                final ByteBuffer out;
237                boolean writeConst;
238
239                public WriteDataExecutor(ByteBuffer out, boolean writeConst) {
240                        this.out = out;
241                        this.writeConst = writeConst;
242                }
243                @Override
244                public void execute(double x, double y, OTFDataWriter writer)  {
245                        try {
246                                if (this.writeConst) writer.writeConstData(this.out);
247                                else writer.writeDynData(this.out);
248                        } catch (IOException e) {
249                                e.printStackTrace();
250                        }
251                }
252        }
