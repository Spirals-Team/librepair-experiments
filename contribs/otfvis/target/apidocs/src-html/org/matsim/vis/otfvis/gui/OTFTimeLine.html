020package org.matsim.vis.otfvis.gui;
021
022import java.awt.Graphics;
023import java.awt.Rectangle;
024import java.awt.event.ActionEvent;
025import java.awt.event.ActionListener;
026import java.util.Collection;
027import java.util.Hashtable;
028
029import javax.swing.BorderFactory;
030import javax.swing.BoundedRangeModel;
031import javax.swing.JButton;
032import javax.swing.JLabel;
033import javax.swing.JSlider;
034import javax.swing.JToolBar;
035import javax.swing.event.ChangeEvent;
036import javax.swing.event.ChangeListener;
037
038import org.matsim.core.utils.misc.Time;
039
040// TODO should not be an OTFDrawer, need to handle invalidate better
041/**
042 * OTFTimeLine is the time line toolbar.
043 * It is only used in case of playing a mvi file.
044 *
045 * @author dstrippgen
046 *
047 */
048public class OTFTimeLine extends JToolBar implements ActionListener, ChangeListener {
049
050        private static final long serialVersionUID = 1L;
051
052        private static class TimestepSlider extends JSlider {
053
054                private static final long serialVersionUID = 1L;
055
056                public TimestepSlider(final BoundedRangeModel model) {
057                        super(model);
058                }
059
060                @Override
061                public void paint(Graphics g) {
062                        super.paint(g);
063                        Rectangle bounds = g.getClipBounds();
064                        bounds.grow(-32, 0);
065                }
066        }
067
068        private final OTFHostControl hostControl;
069
070        private JSlider timestepSlider;
071
072        Hashtable<Integer, JLabel> labelTable = new Hashtable<Integer, JLabel>();
073
074        private int loopStart;
075
076        private int loopEnd;
077
078        public OTFTimeLine(String string, OTFHostControl hostControl) {
079                super(string);
080                this.hostControl = hostControl;
081                addSlider();
082                addLoopButtons();
083                this.setVisible(true);
084        }
085
086        private void addSlider() {
087                //Create the slider.
088                Collection<Double> steps = hostControl.getTimeStepsdrawer();
089                if ((steps == null) || (steps.size() == 0)) {
090                        timestepSlider = new TimestepSlider(hostControl.getSimTimeModel());
091                        return; // nothing to display
092                }
093
094                int min = hostControl.getSimTimeModel().getMinimum();
095                int max = hostControl.getSimTimeModel().getMaximum();
096                timestepSlider = new TimestepSlider(hostControl.getSimTimeModel());
097
098                timestepSlider.getModel().addChangeListener(this);
099                timestepSlider.setMajorTickSpacing((min-max)/10);
100                timestepSlider.setPaintTicks(true);
101
102                //Create the label table.
103                //PENDING: could use images, but we don't have any good ones.
104                Double[] dsteps = steps.toArray(new Double[steps.size()]);
105                labelTable.put(Integer.valueOf( min ),
106                                new JLabel(Time.writeTime(dsteps[0], Time.TIMEFORMAT_HHMM)));
107                //new JLabel(createImageIcon("images/stop.gif")) );
108                labelTable.put(Integer.valueOf( max ),
109                                new JLabel(Time.writeTime(dsteps[dsteps.length-1], Time.TIMEFORMAT_HHMM)) );
110                //new JLabel(createImageIcon("images/fast.gif")) );
111
112                int n = dsteps.length/10 + 1;
113
114                for(int i= n; i< dsteps.length-1; i+=n) {
115                        labelTable.put(Integer.valueOf( dsteps[i].intValue() ),
116                                        new JLabel(Time.writeTime(dsteps[i], Time.TIMEFORMAT_HHMM)) );
117                }
118                timestepSlider.setLabelTable(labelTable);
119
120                timestepSlider.setPaintLabels(true);
121                timestepSlider.setBorder(BorderFactory.createEmptyBorder(0,0,0,10));
122                add(timestepSlider);
123        }
124
125        private void addLoopButtons() {
126                JButton button = new JButton();
127                button.setText("[");
128                button.setActionCommand("setLoopStart");
129                button.addActionListener(this);
130                button.setToolTipText("Sets the loop start time");
131                add(button);
132
133                button = new JButton();
134                button.setText("]");
135                button.setActionCommand("setLoopEnd");
136                button.addActionListener(this);
137                button.setToolTipText("Sets the loop end time");
138                add(button);
139        }
140
141        void replaceLabel(String label, int newEnd) {
142                for (Integer i : labelTable.keySet() ) {
143                        JLabel value = labelTable.get(i);
144                        if(value.getText().equals(label)) {
145                                labelTable.remove(i);
146                                break;
147                        }
148                }
149                labelTable.put(Integer.valueOf(newEnd), new JLabel(label));
150                timestepSlider.setLabelTable(labelTable);
151                timestepSlider.repaint();
152        }
153
154        @Override
155        public void actionPerformed(ActionEvent e) {
156                int time = timestepSlider.getValue();
157                if (e.getActionCommand().equals("setLoopStart")){
158                        loopStart = time;
159                        replaceLabel("[", time);
160                } else if (e.getActionCommand().equals("setLoopEnd")){
161                        loopEnd = time;
162                        replaceLabel("]", time);
163                }
164                hostControl.setLoopBounds(loopStart, loopEnd);
165        }
166
167        /** Listen to the slider. */
168        @Override
169        public void stateChanged(ChangeEvent e) {
170                int gotoTime = hostControl.getSimTime();
171                hostControl.requestTimeStep(gotoTime);
172        }
173
174}
