001package org.matsim.vis.otfvis.opengl.gl;
002
003import com.jogamp.opengl.GL2;
004import com.jogamp.opengl.util.texture.Texture;
005import com.jogamp.opengl.util.texture.TextureIO;
006
007import java.io.IOException;
008import java.io.InputStream;
009
010public class GLUtils {
011
012        static public Texture createTexture(GL2 gl, final InputStream _data) {
013                try (InputStream data = _data) {
014                        Texture t = TextureIO.newTexture(data, true, null);
015                        t.setTexParameteri(gl, GL2.GL_TEXTURE_MIN_FILTER, GL2.GL_LINEAR);
016                        t.setTexParameteri(gl, GL2.GL_TEXTURE_MAG_FILTER, GL2.GL_LINEAR);
017                        return t;
018                } catch (IOException e) {
019                        throw new RuntimeException(e);
020                }
021        }
022
023        private static final int w = 40;
024        private static final double incr = 2*Math.PI/w;
025
026        public static void drawCircle(GL2 gl, float x, float y, float size) {
027                gl.glLineWidth(2);
028                gl.glEnable(GL2.GL_LINE_SMOOTH);
029                gl.glBegin(GL2.GL_LINE_STRIP);
030                float f=0;
031                for (int i=0; i<=w; i++) {
032                        gl.glVertex3d(Math.cos(f)*size + x, Math.sin(f)*size + y,0);
033                        f += incr;
034                }
035                gl.glEnd();
036                gl.glDisable(GL2.GL_LINE_SMOOTH);
037        }
038}
