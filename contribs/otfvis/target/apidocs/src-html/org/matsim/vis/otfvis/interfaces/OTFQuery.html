021package org.matsim.vis.otfvis.interfaces;
022
023import org.matsim.vis.otfvis.SimulationViewForQueries;
024
025/**
026 * An interface for the live version of the OTFVis.
027 * Implementing this interface enables q query to be send to the actual simulation
028 * and be visualized thereafter.<br />
029 * Apart from implementing this interface, the query must be made known to
030 * the queries drop-down box in the class OTFQueryControlBar.
031 * The execution of a query happens in two parts. First the
032 * query() method is called on the server side, with all parameters set.
033 * Data needs to be collected here. It needs to be stored in a way that allows
034 * serialization.
035 * The query object is then transported back to the client, where the draw() method
036 * is called to visualize whatever data has been collected.
037 *
038 * @author dstrippgen
039 */
040public interface OTFQuery {
041
042        /**
043         * Type of Ids that this query will deal with
044         * Right now only LINK and AGENT is used.
045         * The client framework needs this for guessing which ID
046         * to collect for a given query.
047         *
048         * @param Id A String representation of Id
049         */
050        public enum Type {AGENT,LINK,OTHER, CLIENT}
051
052        /**
053         * Sets Id as String.
054         * Together with type this indicates which agent/link to query.
055         *
056         * @param Id A String representation of Id
057         */
058        public void setId(String id);
059
060
061        /**
062         * Is called by the OTFServer framework to issue a query into the simulation.
063         * This method should extract the wanted knowledge from the given
064         * sources of information
065         *
066         * @param net The simulation.
067         * @param plans The Population the simulation fed from.
068         * @param quad The quadtree with writer objects.
069         * @return a query containing results, usually this, but not with live queries, as in
070         * this case the result object should change whenever the result changes, as the RI stream
071         * will actually only transport NEW objects.
072         *
073         */
074        public void installQuery(SimulationViewForQueries queueModel);
075
076        public void uninstall();
077
078
079        /**
080         * Everytime the display needs to be refreshed this
081         * method is called for every active Query.
082         *
083         * @return Type of Ids expected
084         */
085        public Type getType();
086}
