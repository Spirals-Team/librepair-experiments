090                @Override
091                public void draw(OTFOGLDrawer drawer) {
092                        textRenderer = new TextRenderer(new Font("SansSerif", Font.PLAIN, 32), true, false);
093
094                        if(this.vertex == null) return;
095
096                        if(this.calcOffset) {
097                                float east = (float)drawer.getQuad().offsetEast;
098                                float north = (float)drawer.getQuad().offsetNorth;
099
100                                this.calcOffset = false;
101                                for(int i = 0; i < this.vertex.length; i+=2) {
102                                        this.vertex[i] -=east;
103                                        this.vertex[i+1] -= north;
104                                }
105
106                                int maxCount = 0;
107                                for (int aCount : this.count) if (aCount > maxCount) maxCount = aCount;
108
109                                colorizer3 = new FastColorizer(
110                                                new double[] { 0.0, maxCount}, new Color[] {
111                                                                Color.WHITE, Color.BLUE});
112
113                                this.colors = ByteBuffer.allocateDirect(this.count.length*4*2);
114
115                                for (int aCount : this.count) {
116                                        Color mycolor = colorizer3.getColor(aCount);
117                                        this.colors.put((byte) mycolor.getRed());
118                                        this.colors.put((byte) mycolor.getGreen());
119                                        this.colors.put((byte) mycolor.getBlue());
120                                        this.colors.put((byte) 120);
121                                        this.colors.put((byte) mycolor.getRed());
122                                        this.colors.put((byte) mycolor.getGreen());
123                                        this.colors.put((byte) mycolor.getBlue());
124                                        this.colors.put((byte) 120);
125                                }
126
127                                this.vert = Buffers.copyFloatBuffer(FloatBuffer.wrap(this.vertex));
128                                InfoText agentText = new InfoText(this.linkIdString, this.vertex[0], this.vertex[1] );
129                                agentText.draw(textRenderer, OTFClientControl.getInstance().getMainOTFDrawer().getCanvas(), drawer.getViewBoundsAsQuadTreeRect());
130                        }
131
132                        this.vert.position(0);
133                        this.colors.position(0);
134
135                        GL2 gl = OTFGLAbstractDrawable.getGl();
136                        Color color = Color.ORANGE;
137                        gl.glColor4d(color.getRed()/255., color.getGreen()/255.,color.getBlue()/255.,.3);
138                        gl.glColor4d(1., 1.,1.,.3);
139                        gl.glEnable(GL2.GL_BLEND);
140                        gl.glEnable(GL2.GL_LINE_SMOOTH);
141                        gl.glEnableClientState (GL2.GL_COLOR_ARRAY);
142                        gl.glEnableClientState (GL2.GL_VERTEX_ARRAY);
143                        gl.glLineWidth(2.f*OTFClientControl.getInstance().getOTFVisConfig().getLinkWidth());
144                        gl.glVertexPointer (2, GL2.GL_FLOAT, 0, this.vert);
145                        gl.glColorPointer (4, GL2.GL_UNSIGNED_BYTE, 0, this.colors);
146                        gl.glDrawArrays (GL2.GL_LINES, 0, this.vertex.length/2);
147                        gl.glDisableClientState (GL2.GL_VERTEX_ARRAY);
148                        gl.glDisableClientState (GL2.GL_COLOR_ARRAY);
149                        gl.glDisable(GL2.GL_LINE_SMOOTH);
150                        gl.glDisable(GL2.GL_BLEND);
151
152                        drawCaption(drawer);
153                }
154
155                private void drawQuad(GL2 gl, double xs, double xe, double ys, double ye, Color color) {
156                        gl.glColor4d(color.getRed()/255., color.getGreen()/255.,color.getBlue()/255.,color.getAlpha()/255.);
157                        double z = 0;
158                        gl.glBegin(GL2.GL_QUADS);
159                        gl.glVertex3d(xs, ys, z);
160                        gl.glVertex3d(xe, ys, z);
161                        gl.glVertex3d(xe, ye, z);
162                        gl.glVertex3d(xs, ye, z);
163                        gl.glEnd();
164
165                }
166                private void drawCaption(OTFOGLDrawer drawer) {
167                        QuadTree.Rect bounds = drawer.getViewBoundsAsQuadTreeRect();
168
169                        double maxX = bounds.minX + (bounds.maxX -bounds.minX)*0.22;
170                        double minX = bounds.minX + (bounds.maxX -bounds.minX)*0.01;
171                        double maxY = bounds.minY + (bounds.maxY -bounds.minY)*0.15;
172                        double minY = bounds.minY + (bounds.maxY -bounds.minY)*0.01;
173                        GL2 gl = GLContext.getCurrentGL().getGL2();
174                        Color color = new Color(255,255,255,200);
175                        gl.glEnable(GL.GL_BLEND);
176                        drawQuad(gl, minX, maxX, minY, maxY, color);
177                        double horOf = (maxY-minY)/12;
178                        double verOf = (maxX-minX)/12;
179
180                        int maxCount = 0;
181                        for (int aCount : this.count) {
182                                if (aCount > maxCount) maxCount = aCount;
183                        }
184
185                        Color c1 = colorizer3.getColor(0.0);
186                        Color c2 = colorizer3.getColor(maxCount/2.);
187                        Color c3 = colorizer3.getColor(maxCount);
188
189                        double a=1,b=4,c=1,d=3;
190                        drawQuad(gl, minX +a*verOf, minX+b*verOf, minY+c*horOf, minY+d*horOf, c1);
191                        InfoText text1 = new InfoText("Count: 0" , (float)(minX+(b+1)*verOf), (float) (minY+c*horOf));
192                        text1.draw(textRenderer, OTFClientControl.getInstance().getMainOTFDrawer().getCanvas(), drawer.getViewBoundsAsQuadTreeRect());
193                        a=1;b=4;c=5;d=7;
194                        drawQuad(gl, minX +a*verOf, minX+b*verOf, minY+c*horOf, minY+d*horOf, c2);
195                        InfoText text2 = new InfoText("Count: " + (maxCount/2) , (float)(minX+(b+1)*verOf), (float) (minY+c*horOf));
196                        text2.draw(textRenderer, OTFClientControl.getInstance().getMainOTFDrawer().getCanvas(), drawer.getViewBoundsAsQuadTreeRect());
197                        a=1;b=4;c=9;d=11;
198                        drawQuad(gl, minX +a*verOf, minX+b*verOf, minY+c*horOf, minY+d*horOf, c3);
199                        InfoText text3 = new InfoText("Count: " + (maxCount) , (float)(minX+(b+1)*verOf), (float) (minY+c*horOf));
200                        text3.draw(textRenderer, OTFClientControl.getInstance().getMainOTFDrawer().getCanvas(), drawer.getViewBoundsAsQuadTreeRect());
201                }

249        private void addLink(Id<Link> linkId) {
250                Integer count = this.drivenLinks.get(linkId);
251                if (count == null) this.drivenLinks.put(linkId, 1);
252                else  this.drivenLinks.put(linkId, count + 1);
253        }

255        private List<Plan> getPersonsNOW() {
256                List<Plan> actPersons = new ArrayList<>();
257                VisLink link = simulationView.getVisNetwork().getVisLinks().get(this.queryLinkId);
258                Collection<? extends VisVehicle> vehs = link.getAllVehicles();
259                for( VisVehicle veh : vehs) {
260                        if ( veh.getDriver() instanceof PlanAgent ) {
261                                Plan plan = ((PlanAgent)veh.getDriver()).getCurrentPlan() ;
262                                actPersons.add( plan );
263                        }
264                }
265                return actPersons;
266        }
267
268        private List<Plan> getPersonsALL() {
269                List<Plan> actPersons = new ArrayList<>();
270                for (MobsimAgent agent : simulationView.getMobsimAgents().values()) {
271                        Plan plan = simulationView.getPlan(agent);
272                        List<PlanElement> actslegs = plan.getPlanElements();
273                        for (PlanElement pe : actslegs) {
274                                if (pe instanceof Activity) {
275                                        // handle act
276                                        Activity act = (Activity) pe;
277                                        Id<Link> id2 = act.getLinkId();
278                                        if(id2.equals(this.queryLinkId)) {
279                                                actPersons.add(plan);
280                                                break;
281                                        }
282                                } else if (pe instanceof Leg) {
283                                        // handle leg
284                                        Leg leg = (Leg) pe;
285                                        // just look at car routes right now
286                                        if(!leg.getMode().equals(TransportMode.car)) continue;
287                                        for (Id<Link> id2 : ((NetworkRoute) leg.getRoute()).getLinkIds()) {
288                                                if(id2.equals(this.queryLinkId) ) {
289                                                        actPersons.add(plan);
290                                                        break;
291                                                }
292                                        }
293                                }
294                        }
295                }
296                return actPersons;
297        }
298
299        private void collectLinksFromLeg(List<Plan> actPersons) {
300                boolean addthis = false;
301                for (Plan plan : actPersons) {
302                        for (PlanElement pe : plan.getPlanElements()) {
303                                if ( pe instanceof Leg ) {
304                                        Leg leg = (Leg) pe ;
305                                        Route route = leg.getRoute();
306                                        if ( route instanceof NetworkRoute ) { // added in jun09, see below in "collectLinks". kai, jun09
307                                                List<Id<Link>> linkIds = new ArrayList<>();
308                                                for (Id<Link> linkId : ((NetworkRoute) route).getLinkIds() ) {
309                                                        linkIds.add(linkId);
310                                                        if(linkId.equals(this.queryLinkId) ) {
311                                                                // only if this specific route includes link, add the route
312                                                                addthis = true;
313                                                        }
314                                                }
315                                                if(addthis) for (Id<Link> linkId : linkIds) addLink(linkId);
316                                                addthis = false;
317                                        }
318                                }
319                        }
320                }
321        }
322
323        private void collectLinks(List<Plan> actPersons) {
324                for (Plan plan : actPersons) {
325                        for (PlanElement pe : plan.getPlanElements()) {
326                                if (pe instanceof Activity) {
327                                        Activity act = (Activity) pe;
328                                        addLink(act.getLinkId());
329                                } else if (pe instanceof Leg) {
330                                        Leg leg = (Leg) pe;
331                                        Route route = leg.getRoute() ;
332                                        if (route instanceof NetworkRoute) {
333                                                NetworkRoute nr = (NetworkRoute) route ;
334                                                for (Id<Link> linkId : nr.getLinkIds()) {
335                                                        addLink(linkId);
336                                                }
337                                        }
338                                }
339                        }
340                }
341        }

343        @Override
344        public void installQuery(SimulationViewForQueries simulationView) {
345                this.simulationView = simulationView;
346                this.result = new Result();
347                result.linkIdString = this.queryLinkId.toString();
348                this.drivenLinks = new HashMap<>();
349
350                List<Plan> actPersons = nowOnly ? getPersonsNOW() : getPersonsALL();
351
352                if(tripOnly) collectLinksFromLeg(actPersons);
353                else collectLinks(actPersons);
354
355                if(this.drivenLinks.size() == 0) return;
356
357                // convert this to drawable info
358                result.vertex = new float[this.drivenLinks.size()*4];
359                result.count = new int[this.drivenLinks.size()];
360                int pos = 0;
361                for(Id<Link> linkId : this.drivenLinks.keySet()) {
362                        Link link = simulationView.getNetwork().getLinks().get(linkId);
363                        result.count[pos/4] = this.drivenLinks.get(linkId);
364                        Node node = link.getFromNode();
365                        result.vertex[pos++] = (float)node.getCoord().getX();
366                        result.vertex[pos++] = (float)node.getCoord().getY();
367                        node = link.getToNode();
368                        result.vertex[pos++] = (float)node.getCoord().getX();
369                        result.vertex[pos++] = (float)node.getCoord().getY();
370                }
371        }
