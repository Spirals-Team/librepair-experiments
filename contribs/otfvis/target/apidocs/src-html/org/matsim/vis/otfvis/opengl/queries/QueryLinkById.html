020package org.matsim.vis.otfvis.opengl.queries;

022import java.awt.Color;
023import java.nio.ByteBuffer;
024import java.nio.FloatBuffer;
025import java.util.ArrayList;
026import java.util.List;
027
028import com.jogamp.opengl.GL;
029import com.jogamp.opengl.GL2;
030
031import org.apache.log4j.Logger;
032import org.matsim.api.core.v01.Coord;
033import org.matsim.api.core.v01.Id;

035import org.matsim.api.core.v01.network.Network;
036import org.matsim.vis.otfvis.OTFClientControl;
037import org.matsim.vis.otfvis.SimulationViewForQueries;
038import org.matsim.vis.otfvis.data.OTFServerQuadTree;

047/**
048 * @author dgrether
049 *
050 */
051public class QueryLinkById extends AbstractQuery implements OTFQuery {
052
053        private static final Logger log = Logger.getLogger(QueryLinkById.class);
054        
055        private List<Id<Link>> linkIds;
056
057        private transient Result result;
058
059        @Override
060        public void setId(String id) {
061                this.linkIds = QueryUtils.parseIds(id, Link.class);
062        }
063
064        @Override
065        public void installQuery(SimulationViewForQueries simulationView) {
066                Network net = simulationView.getNetwork();
067                this.fillResult(net);
068        }
069
070        private void fillResult(Network net){
071                this.result = new Result();
072                if (this.linkIds.size() == 0) return;
073
074                //get the links from the network
075                List<Link> links = new ArrayList<Link>();
076                for (Id<Link> linkid : this.linkIds){
077                        Link link = net.getLinks().get(linkid);
078                        if (link != null){
079                                links.add(link);
080                                log.info("link id " + linkid + " found in network.");
081                        }
082                        else {
083                                log.info("link id " + linkid + " not found in network.");
084                        }
085                }
086                //fill the result
087                float[] vertex = new float[links.size()*4];
088                byte[] colors = new byte[links.size()*8];
089                
090                int pos = 0;
091                Coord fromCoord, toCoord;
092                Color arielUltra = Color.RED;
093                for (Link l : links){
094                        fromCoord = OTFServerQuadTree.getOTFTransformation().transform(l.getFromNode().getCoord());
095                        toCoord = OTFServerQuadTree.getOTFTransformation().transform(l.getToNode().getCoord());
096                        vertex[pos  * 4 + 0] = (float)fromCoord.getX();
097                        vertex[pos  * 4 + 1] = (float)fromCoord.getY();
098                        vertex[pos  * 4 + 2] = (float)toCoord.getX();
099                        vertex[pos  * 4 + 3] = (float)toCoord.getY();
100                        colors[pos * 8 + 0] = (byte) arielUltra.getRed();
101                        colors[pos * 8 + 1] = (byte) arielUltra.getGreen();
102                        colors[pos * 8 + 2] = (byte) arielUltra.getBlue();
103                        colors[pos * 8 + 3] = (byte) 32;
104                        colors[pos * 8 + 4] = (byte) arielUltra.getRed();
105                        colors[pos * 8 + 5] = (byte) arielUltra.getGreen();
106                        colors[pos * 8 + 6] = (byte) arielUltra.getBlue();
107                        colors[pos * 8 + 7] = (byte) 128;
108                        pos++;
109                }
110                this.result.vertex = vertex;
111                this.result.colors = colors;
112                
113        }
114        
115        
116        @Override
117        public void uninstall() {
118                this.linkIds.clear();
119        }
120
121        @Override
122        public Type getType() {
123                return OTFQuery.Type.LINK;
124        }
125
126        @Override
127        public OTFQueryResult query() {
128                return this.result;
129        }
130
131        private static final class Result implements OTFQueryResult{
132                
133                float[] vertex = null;
134                byte[] colors = null;
135                private transient ByteBuffer cols;
136                private transient FloatBuffer vert;
137
138                @Override
139                public void draw(OTFOGLDrawer drawer) {
140                        if(this.vertex == null) return;
141                        GL2 gl = OTFGLAbstractDrawable.getGl();
142                        this.vert = Buffers.copyFloatBuffer(FloatBuffer.wrap(this.vertex));
143                        this.cols = Buffers.copyByteBuffer(ByteBuffer.wrap(this.colors));
144                        rewindGLBuffers();
145                        prepare(gl);
146                        unPrepare(gl);
147
148                }
149
150                private void prepare(GL2 gl) {
151                        gl.glEnable(GL2.GL_BLEND);
152                        gl.glEnable(GL2.GL_LINE_SMOOTH);
153                        gl.glEnableClientState(GL2.GL_COLOR_ARRAY);
154                        gl.glEnableClientState(GL2.GL_VERTEX_ARRAY);
155                        gl.glLineWidth(1.f * getLineWidth());
156                        gl.glColorPointer(4, GL2.GL_UNSIGNED_BYTE, 0, this.cols);
157                        gl.glVertexPointer(2, GL2.GL_FLOAT, 0, this.vert);
158                        gl.glDrawArrays(GL2.GL_LINES, 0, this.vertex.length);
159                        gl.glDisableClientState(GL2.GL_VERTEX_ARRAY);
160                        gl.glDisableClientState(GL2.GL_COLOR_ARRAY);
161                        gl.glDisable(GL2.GL_LINE_SMOOTH);
162                }
163
164                private void unPrepare(GL gl) {
165                        gl.glDisable(GL.GL_BLEND);
166                }
167
168                private float getLineWidth() {
169                        return OTFClientControl.getInstance().getOTFVisConfig()
170                                        .getLinkWidth();
171                }
172                
173                private void rewindGLBuffers() {
174                        vert.position(0);
175                        cols.position(0);
176                }
177                
178                @Override
179                public void remove() {
180                        
181                }
182
183                @Override
184                public boolean isAlive() {
185                        return true;
186                }
187                
188        }
189        
190}
