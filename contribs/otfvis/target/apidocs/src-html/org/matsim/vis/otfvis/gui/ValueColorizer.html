021package org.matsim.vis.otfvis.gui;
022
023import java.awt.Color;
024import java.io.Serializable;
025
026/**
027 * @author gunnar
028 */
029public class ValueColorizer implements Serializable {
030
031        private static final long serialVersionUID = -2510927168023208677L;
032
033        // MEMBER VARIABLES
034
035        private final double[] values;
036
037        private final Color[] colors;
038
039        // CONSTRUCTION
040
041        public ValueColorizer() {
042                this(new double[] { -1.0, 0.0, 0.1, 0.3, 1.0 }, new Color[] { Color.BLUE, Color.WHITE, Color.GREEN, Color.YELLOW,
043                                Color.RED });
044        }
045
046        public ValueColorizer(double[] values, Color[] colors) {
047                this.values = values.clone();
048                this.colors = colors.clone();
049        }
050
051        // COLOR GENERATION
052
053        private int upperIndex(double val) {
054                int result = 0;
055                while (this.values[result] < val && result < this.values.length - 1)
056                        result++;
057                return result;
058
059        }
060
061        private int bound(double x) {
062                return (int) Math.max(0, Math.min(Math.round(x), 255));
063        }
064
065        public Color getColor(double value) {
066                final int u = upperIndex(value);
067                if (u == 0) {
068                        return this.colors[0];
069                }
070                final double w = (value - this.values[u - 1]) / (this.values[u] - this.values[u - 1]);
071                final int r = bound(w * this.colors[u].getRed() + (1 - w) * this.colors[u - 1].getRed());
072                final int g = bound(w * this.colors[u].getGreen() + (1 - w) * this.colors[u - 1].getGreen());
073                final int b = bound(w * this.colors[u].getBlue() + (1 - w) * this.colors[u - 1].getBlue());
074
075                return new Color(r, g, b);
076        }
077
078}
