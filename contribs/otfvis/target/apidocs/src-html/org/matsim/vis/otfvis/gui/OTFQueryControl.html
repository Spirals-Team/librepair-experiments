058/**
059 * This class is only used with the "live" OTFVis. It represents th Query
060 * issuing GUI element on the lower part of the screen. Alls queries are
061 * initiated and managed here.
062 * 
063 * @author dstrippgen
064 * 
065 */
066public class OTFQueryControl implements GLEventListener {
067
068        private static final Logger log = Logger.getLogger(OTFQueryControl.class);
069
070        private JTextField textField;
071
072        private final OTFServer server;
073
074        private final Map<OTFQueryRemote, OTFQueryResult> queryEntries = new HashMap<>();
075
076        private final Vector<QueryEntry> queries = new Vector<>(Arrays
077                        .asList(new QueryEntry("agentPlan",
078                                                        "show the actual plan of an agent", QueryAgentPlan.class),
079                                        new QueryEntry("agentEvents",
080                                                        "show the actual events of an agent",
081                                                        QueryAgentEvents.class), new QueryEntry(
082                                                        "agentPTBus",
083                                                        "highlight all buses of a given line",
084                                                        QueryAgentPTBus.class), new QueryEntry(
085                                                        "linkSpinneALL", "show Spinne of ALL traffic",
086                                                        QuerySpinne.class),
087                                        new QueryEntry("linkById", "show link(s) by comma separated link id", QueryLinkById.class),
088                                        new QueryEntry("nodeById", "show node(s) by comma separated node id", QueryNodeById.class)));
089
090        private final OTFVisConfigGroup config;
091
092        public OTFQueryControl(OTFServer server, final OTFVisConfigGroup config) {
093                this.config = config;
094                this.server = server;
095        }
096
097        @Override
098        public void init(GLAutoDrawable glAutoDrawable) {
099
100        }
101
102        @Override
103        public void reshape(GLAutoDrawable glAutoDrawable, int i, int i1, int i2, int i3) {
104
105        }
106
107        synchronized void handleIdQuery(String id, String queryName) {
108                AbstractQuery query = createQuery(queryName);
109                query.setId(id);
110                createQuery(query);
111        }
112
113        public void handleClick(Point2D.Double point, int mouseButton) {
114                Rectangle2D.Double origRect = new Rectangle2D.Double(point.x, point.y,
115                                0, 0);
116                // Only handle clicks with the main == zoom button
117                if ((mouseButton == 1) || (mouseButton == 4))
118                        handleClick(origRect, mouseButton);
119        }
120
121        synchronized void removeQueries() {
122                if(server.isLive()) {
123                        ((OTFLiveServer) server).removeQueries();
124                }
125                for (OTFQueryResult query : this.queryEntries.values()) {
126                        query.remove();
127                }
128                this.queryEntries.clear();
129                textField.setText("");
130                ((Component) OTFClientControl.getInstance().getMainOTFDrawer().getCanvas()).repaint();
131        }
132
133        @Override
134        public void display(GLAutoDrawable glAutoDrawable) {
135                for (OTFQueryResult queryResult : this.queryEntries.values()) {
136                        queryResult.draw(OTFClientControl.getInstance().getMainOTFDrawer());
137                }
138        }
139
140        public void handleClick(Rectangle2D.Double origRect, int mouseButton) {
141                if (mouseButton == 3) {
142                        removeQueries();
143                        ((Component) OTFClientControl.getInstance().getMainOTFDrawer().getCanvas()).repaint();
144                } else {
145                        String queryName = this.config.getQueryType();
146                        Type typeOfQuery = getTypeOfQuery(queryName);
147                        if (typeOfQuery == OTFQuery.Type.AGENT) {
148                                List<String> agentIds = resolveId(origRect);
149                                if ((agentIds != null) && (agentIds.size() != 0)) {
150                                        log.debug("AgentId = " + agentIds);
151                                        handleIdQuery(agentIds, queryName);
152                                } else {
153                                        log.debug("No AgentId found!");
154                                }
155                        } else if (typeOfQuery == OTFQuery.Type.LINK) {
156                                QueryLinkId.Result linkIdQuery = (QueryLinkId.Result) createQuery(new QueryLinkId(origRect));
157                                if ((linkIdQuery != null) && (linkIdQuery.linkIds.size() != 0)) {
158                                        log.debug("LinkId = " + linkIdQuery.linkIds);
159                                        handleIdQuery(linkIdQuery.linkIds.values(), queryName);
160                                } else {
161                                        log.debug("No LinkId found!");
162                                }
163                        }
164                }
165        }
166
167        synchronized public void updateQueries() {
168                for (Map.Entry<OTFQueryRemote, OTFQueryResult> queryItem : queryEntries.entrySet()) {
169                        OTFQueryResult queryResult = queryItem.getValue();
170                        if (queryResult.isAlive()) {
171                                OTFQueryRemote queryRemote = queryItem.getKey();
172                                queryResult.remove();
173                                OTFQueryResult newResult = queryRemote.query();
174                                queryItem.setValue(newResult);
175                        }
176                }
177        }
178
179        private OTFQueryResult createQuery(AbstractQuery query) {
180                OTFQueryRemote remoteQuery = doQuery(query);
181                OTFQueryResult queryResult = remoteQuery.query();
182                queryEntries.put(remoteQuery, queryResult);
183                return queryResult;
184        }
185
186        private OTFQueryRemote doQuery(final AbstractQuery query) {
187                return ((OTFLiveServer) server).answerQuery(query);
188        }
189
190        private void handleIdQuery(Collection<String> list, String queryName) {
191                if (!this.config.isMultipleSelect()) {
192                        removeQueries();
193                }
194                StringBuilder infoText = new StringBuilder(textField.getText());
195                for (String id : list) {
196                        if (infoText.length() != 0) {
197                                infoText.append(", ");
198                        }
199                        infoText.append(id);
200                        handleIdQuery(id, queryName);
201                }
202                textField.setText(infoText.toString());
203                ((Component) OTFClientControl.getInstance().getMainOTFDrawer().getCanvas()).repaint();
204        }
205
206        @SuppressWarnings("unchecked")
207        AbstractQuery createQuery(String className) {
208                try {
209                        Class<? extends AbstractQuery> classDefinition = (Class<? extends AbstractQuery>) Class
210                        .forName(className);
211                        return classDefinition.newInstance();
212                } catch (InstantiationException | IllegalAccessException | ClassNotFoundException e) {
213                        throw new RuntimeException(e);
214                }
215        }
216
217        private Type getTypeOfQuery(String queryName) {
218                OTFQuery query = createQuery(queryName);
219                return query.getType();
220        }
221
222        private List<String> resolveId(Double origRect) {
223                QueryAgentId.Result agentIdQuery = (QueryAgentId.Result) createQuery(new QueryAgentId(origRect));
224                if ((agentIdQuery != null) && (agentIdQuery.agentIds.size() != 0)) {
225                        return agentIdQuery.agentIds;
226                }
227                return null;
228        }
229
230        public Vector<QueryEntry> getQueries() {
231                return queries;
232        }
233
234        public void setQueryTextField(JTextField textField2) {
235                this.textField = textField2;
236        }
237
238        @Override
239        public void dispose(GLAutoDrawable glAutoDrawable) {
240
241        }
242
243}
