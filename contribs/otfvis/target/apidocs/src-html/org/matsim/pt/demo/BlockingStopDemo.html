077        private static final int nOfLinks = 13;
078        private static final int nOfStops = nOfLinks-3;
079        private static final int nOfCars = 20;
080        private static final int carsHeading = 30;
081        private static final double startTime = 7.0 * 3600;
082        private static final double busDeparture = 7.0 * 3600 + 3 * 60 + 15;

084        private final MutableScenario scenario = (MutableScenario) ScenarioUtils.createScenario(ConfigUtils.createConfig());

086        private void prepareConfig() {
087                Config config = this.scenario.getConfig();
088                config.transit().setUseTransit(true);
089                config.qsim().setSnapshotStyle(SnapshotStyle.queue);
090                config.qsim().setEndTime(24.0*3600);
091        }

093        private void createNetwork() {
094                Network network = this.scenario.getNetwork();
095                Node[] nodes = new Node[nOfLinks * 2 + 2];
096                for (int i = 0; i <= nOfLinks; i++) {
097                        nodes[i] = network.getFactory().createNode(Id.create(i, Node.class), new Coord(i * 500, 0));
098                        network.addNode(nodes[i]);
099                        nodes[i+nOfLinks+1] = network.getFactory().createNode(Id.create(i+nOfLinks+1, Node.class), new Coord(i * 500, 500));
100                        network.addNode(nodes[i+nOfLinks+1]);
101                }
102                for (int i = 0; i < nOfLinks; i++) {
103                        Link link = network.getFactory().createLink(Id.create(i, Link.class), nodes[i], nodes[i+1]);
104                        link.setLength(500.0);
105                        link.setFreespeed(10.0);
106                        link.setCapacity(1000.0);
107                        link.setNumberOfLanes(1);
108                        network.addLink(link);
109                        link = network.getFactory().createLink(Id.create(i+nOfLinks, Link.class), nodes[i+nOfLinks+1], nodes[i+nOfLinks+2]);
110                        link.setLength(500.0);
111                        link.setFreespeed(10.0);
112                        link.setCapacity(1000.0);
113                        link.setNumberOfLanes(1);
114                        network.addLink(link);
115                }
116        }

118        private void createTransitSchedule() {
119                TransitSchedule schedule = this.scenario.getTransitSchedule();
120                TransitScheduleFactory builder = schedule.getFactory();
121
122                TransitStopFacility[] stops = new TransitStopFacility[nOfStops * 2];
123                ArrayList<TransitRouteStop> stopList = new ArrayList<>(nOfStops);
124
125                // line 1
126                for (int i = 0; i < nOfStops; i++) {
127                        stops[i] = builder.createTransitStopFacility(Id.create(i, TransitStopFacility.class), new Coord(1000 + i * 500, 0), false);
128                        stops[i].setLinkId(Id.create(i+1, Link.class));
129                        schedule.addStopFacility(stops[i]);
130                        stopList.add(builder.createTransitRouteStop(stops[i], 100 + i*70, 120 + i*70));
131                }
132
133                Link startLink = this.scenario.getNetwork().getLinks().get(Id.create(0, Link.class));
134                Link endLink = this.scenario.getNetwork().getLinks().get(Id.create(nOfLinks-1, Link.class));
135                NetworkRoute networkRoute = this.scenario.getPopulation().getFactory().getRouteFactories().createRoute(NetworkRoute.class, startLink.getId(), endLink.getId());
136                ArrayList<Id<Link>> linkIdList = new ArrayList<>(nOfLinks);
137                for (int i = 1; i < nOfLinks-1; i++) {
138                        linkIdList.add(Id.create(i, Link.class));
139                }
140                networkRoute.setLinkIds(startLink.getId(), linkIdList, endLink.getId());
141                TransitRoute tRoute1 = builder.createTransitRoute(Id.create(1, TransitRoute.class), networkRoute, stopList, "bus");
142
143                TransitLine tLine1 = builder.createTransitLine(Id.create(1, TransitLine.class));
144                tLine1.addRoute(tRoute1);
145                schedule.addTransitLine(tLine1);
146
147                Departure dep1 = builder.createDeparture(Id.create(1, Departure.class), busDeparture);
148                dep1.setVehicleId(Id.create("tr_1", Vehicle.class));
149                tRoute1.addDeparture(dep1);
150
151                // line 2
152                stopList = new ArrayList<>(nOfStops);
153                for (int i = 0; i < nOfStops; i++) {
154                        stops[i+nOfStops] = builder.createTransitStopFacility(Id.create(i+nOfStops, TransitStopFacility.class), new Coord(1000 + i * 500, 500), true);
155                        stops[i+nOfStops].setLinkId(Id.create(i+1+nOfLinks, Link.class));
156                        schedule.addStopFacility(stops[i+nOfStops]);
157                        stopList.add(builder.createTransitRouteStop(stops[i+nOfStops], 100 + i*70, 120 + i*70));
158                }
159
160                startLink = this.scenario.getNetwork().getLinks().get(Id.create(nOfLinks, Link.class));
161                endLink = this.scenario.getNetwork().getLinks().get(Id.create(2*nOfLinks-1, Link.class));
162                networkRoute = this.scenario.getPopulation().getFactory().getRouteFactories().createRoute(NetworkRoute.class, startLink.getId(), endLink.getId());
163                linkIdList = new ArrayList<>(nOfLinks);
164                for (int i = nOfLinks+1; i < (2*nOfLinks - 1); i++) {
165                        linkIdList.add(Id.create(i, Link.class));
166                }
167                networkRoute.setLinkIds(startLink.getId(), linkIdList, endLink.getId());
168                TransitRoute tRoute2 = builder.createTransitRoute(Id.create(2, TransitRoute.class), networkRoute, stopList, "bus");
169
170                TransitLine tLine2 = builder.createTransitLine(Id.create(2, TransitLine.class));
171                tLine2.addRoute(tRoute2);
172                schedule.addTransitLine(tLine2);
173
174                Departure dep2 = builder.createDeparture(Id.create(2, Departure.class), busDeparture);
175                dep2.setVehicleId(Id.create("tr_2", Vehicle.class));
176                tRoute2.addDeparture(dep2);
177        }

179        private void createVehicles() {
180                Vehicles vehicles = this.scenario.getTransitVehicles();
181                VehiclesFactory vb = vehicles.getFactory();
182                VehicleType vehicleType = vb.createVehicleType(Id.create("transitVehicleType", VehicleType.class));
183                VehicleCapacity capacity = vb.createVehicleCapacity();
184                capacity.setSeats(101);
185                capacity.setStandingRoom(0);
186                vehicleType.setCapacity(capacity);
187                vehicles.addVehicleType(vehicleType);
188                Id<Vehicle> id = Id.create("tr_1", Vehicle.class);
189                vehicles.addVehicle( vb.createVehicle(id, vehicleType));
190                id = Id.create("tr_2", Vehicle.class);
191                vehicles.addVehicle( vb.createVehicle(id, vehicleType));
192        }

194        private void createPopulation() {
195                TransitSchedule schedule = this.scenario.getTransitSchedule();
196                Population population = this.scenario.getPopulation();
197                PopulationFactory pb = population.getFactory();
198                TransitLine tLine1 = schedule.getTransitLines().get(Id.create(1, TransitLine.class));
199                TransitRoute tRoute1 = tLine1.getRoutes().get(Id.create(1, TransitRoute.class));
200                TransitLine tLine2 = schedule.getTransitLines().get(Id.create(2, TransitLine.class));
201                TransitRoute tRoute2 = tLine2.getRoutes().get(Id.create(2, TransitRoute.class));
202
203                // bus-passengers line 1
204                for (int i = 1; i < nOfStops; i++) {
205                        Person person = pb.createPerson(Id.create(Integer.toString(-i), Person.class));
206                        Plan plan = pb.createPlan();
207                        Activity act1 = pb.createActivityFromLinkId("home", Id.create(i, Link.class));
208                        act1.setEndTime(startTime + i*60);
209                        Leg leg = pb.createLeg(TransportMode.pt);
210                        leg.setRoute(new ExperimentalTransitRoute(schedule.getFacilities().get(Id.create(i-1, TransitStopFacility.class)), tLine1, tRoute1, schedule.getFacilities().get(Id.create(nOfStops-1, TransitStopFacility.class))));
211                        Activity act2 = pb.createActivityFromLinkId("work", Id.create(nOfLinks-1, Link.class));
212
213                        population.addPerson(person);
214                        person.addPlan(plan);
215                        person.setSelectedPlan(plan);
216                        plan.addActivity(act1);
217                        plan.addLeg(leg);
218                        plan.addActivity(act2);
219                }
220
221                // bus-passengers line 2
222                for (int i = 1; i < nOfStops; i++) {
223                        Person person = pb.createPerson(Id.create(Integer.toString(-i-nOfStops), Person.class));
224                        Plan plan = pb.createPlan();
225                        Activity act1 = pb.createActivityFromLinkId("home", Id.create(nOfLinks+i, Link.class));
226                        act1.setEndTime(startTime + i*60);
227                        Leg leg = pb.createLeg(TransportMode.pt);
228                        leg.setRoute(new ExperimentalTransitRoute(schedule.getFacilities().get(Id.create(nOfStops+i-1, TransitStopFacility.class)), tLine2, tRoute2, schedule.getFacilities().get(Id.create(2*nOfStops-1, TransitStopFacility.class))));
229                        Activity act2 = pb.createActivityFromLinkId("work", Id.create(2*nOfLinks-1, Link.class));
230
231                        population.addPerson(person);
232                        person.addPlan(plan);
233                        person.setSelectedPlan(plan);
234                        plan.addActivity(act1);
235                        plan.addLeg(leg);
236                        plan.addActivity(act2);
237                }
238
239                // car-drivers
240                NetworkRoute carRoute1 = population.getFactory().getRouteFactories().createRoute(NetworkRoute.class, Id.create(0, Link.class), Id.create(nOfLinks-1, Link.class));
241                NetworkRoute carRoute2 = population.getFactory().getRouteFactories().createRoute(NetworkRoute.class, Id.create(nOfLinks, Link.class), Id.create(2*nOfLinks-1, Link.class));
242                List<Id<Link>> linkIds1 = new ArrayList<>(nOfLinks - 2);
243                List<Id<Link>> linkIds2 = new ArrayList<>(nOfLinks - 2);
244                for (int i = 1; i<nOfLinks-1; i++) {
245                        linkIds1.add(Id.create(i, Link.class));
246                        linkIds2.add(Id.create(i+nOfLinks, Link.class));
247                }
248                carRoute1.setLinkIds(Id.create(0, Link.class), linkIds1, Id.create(nOfLinks-1, Link.class));
249                carRoute2.setLinkIds(Id.create(nOfLinks, Link.class), linkIds2, Id.create(2*nOfLinks-1, Link.class));
250                for (int i = 0; i < nOfCars; i++) {
251                        Person person = pb.createPerson(Id.create(Integer.toString(i), Person.class));
252                        Plan plan = pb.createPlan();
253                        Activity act1a = pb.createActivityFromLinkId("home", Id.create(0, Link.class));
254                        act1a.setEndTime(startTime + i*carsHeading);
255                        Leg leg1 = pb.createLeg(TransportMode.car);
256                        leg1.setRoute(carRoute1);
257                        Activity act1b = pb.createActivityFromLinkId("work", Id.create(4, Link.class));
258
259                        population.addPerson(person);
260                        person.addPlan(plan);
261                        person.setSelectedPlan(plan);
262                        plan.addActivity(act1a);
263                        plan.addLeg(leg1);
264                        plan.addActivity(act1b);
265
266                        Person person2 = pb.createPerson(Id.create(Integer.toString(i+nOfCars), Person.class));
267                        Plan plan2 = pb.createPlan();
268                        Activity act2a = pb.createActivityFromLinkId("home", Id.create(nOfLinks, Link.class));
269                        act2a.setEndTime(startTime + i*carsHeading);
270                        Leg leg2 = pb.createLeg(TransportMode.car);
271                        leg2.setRoute(carRoute2);
272                        Activity act2b = pb.createActivityFromLinkId("work", Id.create(nOfLinks-1, Link.class));
273
274                        population.addPerson(person2);
275                        person2.addPlan(plan2);
276                        person2.setSelectedPlan(plan2);
277                        plan2.addActivity(act2a);
278                        plan2.addLeg(leg2);
279                        plan2.addActivity(act2b);
280                }
281
282        }

284        private void runSim() {
285                EventsManager events = EventsUtils.createEventsManager();
286                Network network = this.scenario.getNetwork();
287
288                VehicleTracker vehTracker = new VehicleTracker();
289                events.addHandler(vehTracker);
290                TransitRouteAccessEgressAnalysis analysis = new TransitRouteAccessEgressAnalysis(this.scenario.getTransitSchedule().getTransitLines().get(Id.create(1, TransitLine.class)).getRoutes().get(Id.create(1, TransitRoute.class)), vehTracker);
291                events.addHandler(analysis);
292                TravelTimeCalculator ttc = new TravelTimeCalculator(this.scenario.getNetwork(), 120, 7*3600+1800, new TravelTimeCalculatorConfigGroup());
293                events.addHandler(ttc);
294
295                Collection<AbstractModule> overrides = new ArrayList<>() ;
296                overrides.add( new AbstractModule() {
297                        @Override public void install() {
298                                bind(TransitStopHandlerFactory.class).to(SimpleTransitStopHandlerFactory.class).asEagerSingleton();
299                        }
300                } ) ;
301                final QSim sim = QSimUtils.createDefaultQSimWithOverrides(this.scenario, events, overrides);
302//              QSim sim = QSimUtils.createDefaultQSim(this.scenario, events);
303//              sim.getTransitEngine().setTransitStopHandlerFactory(new SimpleTransitStopHandlerFactory());
304
305                // ... but I think it is not working anyways because there is some problem to find the relevant part of the network in otfvis ...
306                // kai, nov'17
307
308                OnTheFlyServer server = OTFVis.startServerAndRegisterWithQSim(this.scenario.getConfig(), this.scenario, events, sim);
309                OTFClientLive.run(this.scenario.getConfig(), server);
310                
311                sim.run();
312
313                System.out.println("TransitRouteAccessEgressAnalysis:");
314                analysis.printStats();
315
316                System.out.println("TravelTimes:");
317                for (int i = 13; i < 26; i++) {
318                        System.out.print("\tlink " + i);
319                }
320                System.out.println();
321                for (int time = 7*3600; time < (7*3600 + 15*60); time += 120) {
322                        System.out.print(Time.writeTime(time));
323                        for (int i = 13; i < 26; i++) {
324                                System.out.print("\t" + ttc.getLinkTravelTimes().getLinkTravelTime(network.getLinks().get(Id.create(i, Link.class)), time, null, null));
325                        }
326                        System.out.println();
327                }
328                System.out.println();
329                for (int i = 0; i < 13; i++) {
330                        System.out.print("\tlink " + i);
331                }
332                System.out.println();
333                for (int time = 7*3600; time < (7*3600 + 15*60); time += 120) {
334                        System.out.print(Time.writeTime(time));
335                        for (int i = 0; i < 13; i++) {
336                                System.out.print("\t" + ttc.getLinkTravelTimes().getLinkTravelTime(network.getLinks().get(Id.create(i, Link.class)), time, null, null));
337                        }
338                        System.out.println();
339                }
340        }

346                createPopulation();
347                createVehicles();
