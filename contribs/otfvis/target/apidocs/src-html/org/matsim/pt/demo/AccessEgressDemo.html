072public class AccessEgressDemo {
073
074        private static final int nOfLinks = 15;
075        private static final int nOfBuses = 20;
076        private static final int nOfAgentsPerStop = 100;
077        private static final int agentInterval = 60;
078        private static final int delayedBus = 9;
079        private static final int heading = 5*60;
080        private static final int delay = 60;
081        private static final double departureTime = 7.0*3600;
082        private static final boolean stopsBlockLane = true;
083
084        private final MutableScenario scenario = (MutableScenario) ScenarioUtils.createScenario(ConfigUtils.createConfig());
085
086        private void prepareConfig() {
087                Config config = this.scenario.getConfig();
088                config.transit().setUseTransit(true);
089                config.qsim().setSnapshotStyle(SnapshotStyle.queue);
090                config.qsim().setEndTime(24.0*3600);
091        }
092
093        private void createNetwork() {
094                Network network = this.scenario.getNetwork();
095                Node[] nodes = new Node[nOfLinks + 1];
096                for (int i = 0; i <= nOfLinks; i++) {
097                        nodes[i] = network.getFactory().createNode(Id.create(i, Node.class), new Coord(i * 500, 0));
098                        network.addNode(nodes[i]);
099                }
100                for (int i = 0; i < nOfLinks; i++) {
101                        Link l = network.getFactory().createLink(Id.create(i, Link.class), nodes[i], nodes[i+1]);
102                        l.setLength(500.0);
103                        l.setFreespeed(10.0);
104                        l.setCapacity(1000.0);
105                        l.setNumberOfLanes(1);
106                        network.addLink(l);
107                }
108        }
109
110        private void createTransitSchedule() {
111                TransitSchedule schedule = this.scenario.getTransitSchedule();
112                TransitScheduleFactory builder = schedule.getFactory();
113                TransitStopFacility[] stops = new TransitStopFacility[nOfLinks];
114                ArrayList<TransitRouteStop> stopList = new ArrayList<>(nOfLinks);
115                for (int i = 0; i < nOfLinks; i++) {
116                        stops[i] = builder.createTransitStopFacility(Id.create(i, TransitStopFacility.class), new Coord((i + 1) * 500, 0), stopsBlockLane);
117                        stops[i].setLinkId(Id.create(i, Link.class));
118                        schedule.addStopFacility(stops[i]);
119                        TransitRouteStop stop = builder.createTransitRouteStop(stops[i], i * 50, i * 50 + 10);
120                        stopList.add(stop);
121                }
122                Link startLink = this.scenario.getNetwork().getLinks().get(Id.create(0, Link.class));
123                Link endLink = this.scenario.getNetwork().getLinks().get(Id.create(nOfLinks - 1, Link.class));
124                NetworkRoute networkRoute = this.scenario.getPopulation().getFactory().getRouteFactories().createRoute(NetworkRoute.class, startLink.getId(), endLink.getId());
125                ArrayList<Id<Link>> linkList = new ArrayList<>(nOfLinks - 2);
126                for (int i = 1; i < nOfLinks -1; i++) {
127                        linkList.add(Id.create(i, Link.class));
128                }
129                networkRoute.setLinkIds(startLink.getId(), linkList, endLink.getId());
130                TransitRoute tRoute = builder.createTransitRoute(Id.create(1, TransitRoute.class), networkRoute, stopList, "bus");
131
132                TransitLine tLine = builder.createTransitLine(Id.create(1, TransitLine.class));
133                tLine.addRoute(tRoute);
134                schedule.addTransitLine(tLine);
135
136                for (int i = 0; i < nOfBuses; i++       ) {
137                        Departure dep = builder.createDeparture(Id.create(i, Departure.class), departureTime + i*heading + (i == delayedBus ? delay : 0));
138                        dep.setVehicleId(Id.create(i, Vehicle.class));
139                        tRoute.addDeparture(dep);
140                }
141        }
142
143        private void createVehicles() {
144                Vehicles vehicles = this.scenario.getTransitVehicles();
145                VehiclesFactory vb = vehicles.getFactory();
146                VehicleType vehicleType = vb.createVehicleType(Id.create("transitVehicleType", VehicleType.class));
147                vehicles.addVehicleType(vehicleType);
148                
149                VehicleCapacity capacity = vb.createVehicleCapacity();
150                capacity.setSeats(101);
151                capacity.setStandingRoom(0);
152                vehicleType.setCapacity(capacity);
153                for (int i = 0; i < nOfBuses; i++) {
154                        vehicles.addVehicle( vb.createVehicle(Id.create(i, Vehicle.class), vehicleType));
155                }
156        }
157
158        private void createPopulation() {
159                TransitSchedule schedule = this.scenario.getTransitSchedule();
160                Population population = this.scenario.getPopulation();
161                PopulationFactory pb = population.getFactory();
162                TransitStopFacility[] stops = schedule.getFacilities().values().toArray(new TransitStopFacility[schedule.getFacilities().size()]);
163                TransitLine tLine = schedule.getTransitLines().get(Id.create(1, TransitLine.class));
164                TransitRoute tRoute = tLine.getRoutes().get(Id.create(1, TransitRoute.class));
165
166                TransitStopFacility lastStop = schedule.getFacilities().get(Id.create(stops.length - 1, TransitStopFacility.class));
167                for (int i = 0; i < stops.length; i++) {
168                        TransitStopFacility stop = stops[i];
169                        if (stop == lastStop) {
170                                continue;
171                        }
172                        for (int j = 0; j < nOfAgentsPerStop; j++) {
173                                Person person = pb.createPerson(Id.create(Integer.toString(i * nOfAgentsPerStop + j), Person.class));
174                                Plan plan = pb.createPlan();
175                                Activity act1 = pb.createActivityFromLinkId("home", stop.getLinkId());
176                                act1.setEndTime(departureTime + j * agentInterval);
177                                Leg leg = pb.createLeg(TransportMode.pt);
178                                leg.setRoute(new ExperimentalTransitRoute(stop, tLine, tRoute, lastStop));
179                                Activity act2 = pb.createActivityFromLinkId("work", Id.create(nOfLinks - 1, Link.class));
180
181                                population.addPerson(person);
182                                person.addPlan(plan);
183                                person.setSelectedPlan(plan);
184                                plan.addActivity(act1);
185                                plan.addLeg(leg);
186                                plan.addActivity(act2);
187                        }
188                }
189        }
190
191        private void runSim() {
192                EventsManager events = EventsUtils.createEventsManager();
193
194                TransitRoute route = this.scenario.getTransitSchedule().getTransitLines().get(Id.create(1, TransitLine.class)).getRoutes().get(Id.create(1, TransitRoute.class));
195                VehicleTracker vehTracker = new VehicleTracker();
196                events.addHandler(vehTracker);
197                TransitRouteAccessEgressAnalysis analysis = new TransitRouteAccessEgressAnalysis(route, vehTracker);
198                events.addHandler(analysis);
199                RouteTimeDiagram diagram = new RouteTimeDiagram();
200                events.addHandler(diagram);
201
202                Collection<AbstractModule> overrides = new ArrayList<>() ;
203                overrides.add( new AbstractModule() {
204                        @Override public void install() {
205                                bind(TransitStopHandlerFactory.class).to(SimpleTransitStopHandlerFactory.class).asEagerSingleton();
206                        }
207                } ) ;
208                final QSim sim = QSimUtils.createDefaultQSimWithOverrides(this.scenario, events, overrides);
209//              QSim sim = QSimUtils.createDefaultQSim(this.scenario, events);
210//              sim.getTransitEngine().setTransitStopHandlerFactory(new SimpleTransitStopHandlerFactory());
211                // ... but I think it is not working anyways because there is some problem to find the relevant part of the network in otfvis ...
212                // kai, nov'17
213                
214                OnTheFlyServer server = OTFVis.startServerAndRegisterWithQSim(this.scenario.getConfig(), this.scenario, events, sim);
215                OTFClientLive.run(this.scenario.getConfig(), server);
216                sim.run();
217
218                System.out.println("TransitRouteAccessEgressAnalysis:");
219                analysis.printStats();
220                System.out.println("Route-Time-Diagram:");
221                diagram.writeData();
222                String filename = "output/routeTimeDiagram.png";
223                System.out.println("writing route-time diagram to: " + filename);
224                diagram.createGraph(filename, route);
225        }
226
227        public void run() {
228                prepareConfig();
229                createNetwork();
230                createTransitSchedule();
231                createVehicles();
232                createPopulation();
233                runSim();
234        }
235
236        public static void main(final String[] args) {
237                new AccessEgressDemo().run();
238        }
