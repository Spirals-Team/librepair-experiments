327        private void runSim() {
328                EventsManager events = EventsUtils.createEventsManager();
329                VehicleTracker vehTracker = new VehicleTracker();
330                events.addHandler(vehTracker);
331                TransitRoute route1 = this.scenario.getTransitSchedule().getTransitLines().get(Id.create(1, TransitLine.class)).getRoutes().get(Id.create(1, TransitRoute.class));
332                TransitRoute route2 = this.scenario.getTransitSchedule().getTransitLines().get(Id.create(2, TransitLine.class)).getRoutes().get(Id.create(1, TransitRoute.class));
333                RouteOccupancy analysis1 = new RouteOccupancy(route1, vehTracker);
334                RouteOccupancy analysis2 = new RouteOccupancy(route2, vehTracker);
335                events.addHandler(analysis1);
336                events.addHandler(analysis2);
337
338                QSim sim = QSimUtils.createDefaultQSim(this.scenario, events);
339                OnTheFlyServer server = OTFVis.startServerAndRegisterWithQSim(scenario.getConfig(), scenario, events, sim);
340                OTFClientLive.run(scenario.getConfig(), server);
341                sim.run();
342
343                System.out.println("stop\t#exitleaving\t#enter\t#inVehicle");
344                int inVehicle = 0;
345                for (TransitRouteStop stop : route1.getStops()) {
346                        Id<TransitStopFacility> stopId = stop.getStopFacility().getId();
347                        int enter = analysis1.getNumberOfEnteringPassengers(stopId);
348                        int leave = analysis1.getNumberOfLeavingPassengers(stopId);
349                        inVehicle = inVehicle + enter - leave;
350                        System.out.println(stopId + "\t" + leave + "\t" + enter + "\t" + inVehicle);
351                }
352
353                System.out.println("stop\t#exitleaving\t#enter\t#inVehicle");
354                inVehicle = 0;
355                for (TransitRouteStop stop : route2.getStops()) {
356                        Id<TransitStopFacility> stopId = stop.getStopFacility().getId();
357                        int enter = analysis2.getNumberOfEnteringPassengers(stopId);
358                        int leave = analysis2.getNumberOfLeavingPassengers(stopId);
359                        inVehicle = inVehicle + enter - leave;
360                        System.out.println(stopId + "\t" + leave + "\t" + enter + "\t" + inVehicle);
361                }
362        }
