029import java.io.*;
030import java.util.zip.Inflater;
031import java.util.zip.DataFormatException;
032
033public class CompressedBlockInputStream extends FilterInputStream {
034    /**
035     * Buffer of compressed data read from the stream
036     */
037    private byte[] inBuf = null;
038
039    /**
040     * Buffer of uncompressed data
041     */
042    private byte[] outBuf = null;
043
044    /**
045     * Offset and length of uncompressed data
046     */
047    private int outOffs = 0;
048    private int outLength = 0;
049
050    /**
051     * Inflater for decompressing
052     */
053    private Inflater inflater = null;
054
055    public CompressedBlockInputStream(InputStream is)
056            throws IOException {
057        super(is);
058        inflater = new Inflater();
059    }
060
061    private void readAndDecompress() throws IOException {
062        // Read the length of the compressed block
063        int ch1 = in.read();
064        int ch2 = in.read();
065        int ch3 = in.read();
066        int ch4 = in.read();
067        if ((ch1 | ch2 | ch3 | ch4) < 0)
068            throw new EOFException();
069        /*
070      Length of data in the input data
071     */
072        int inLength = ((ch1 << 24) + (ch2 << 16) +
073                (ch3 << 8) + (ch4 << 0));
074
075        ch1 = in.read();
076        ch2 = in.read();
077        ch3 = in.read();
078        ch4 = in.read();
079        if ((ch1 | ch2 | ch3 | ch4) < 0)
080            throw new EOFException();
081        outLength = ((ch1 << 24) + (ch2 << 16) +
082                (ch3 << 8) + (ch4 << 0));
083
084        // Make sure we've got enough space to read the block
085        if ((inBuf == null) || (inLength > inBuf.length)) {
086            inBuf = new byte[inLength];
087        }
088
089        if ((outBuf == null) || (outLength > outBuf.length)) {
090            outBuf = new byte[outLength];
091        }
092
093        // Read until we're got the entire compressed buffer.
094        // read(...) will not necessarily block until all
095        // requested data has been read, so we loop until
096        // we're done.
097        int inOffs = 0;
098        while (inOffs < inLength) {
099            int inCount =
100                    in.read(inBuf, inOffs, inLength - inOffs);
101            if (inCount == -1) {
102                throw new EOFException();
103            }
104            inOffs += inCount;
105        }
106
107        inflater.setInput(inBuf, 0, inLength);
108        try {
109            inflater.inflate(outBuf);
110        }
111        catch(DataFormatException dfe) {
112            throw new IOException(
113                    "Data format exception - " +
114                            dfe.getMessage());
115        }
116
117        // Reset the inflator so we can re-use it for the
118        // next block
119        inflater.reset();
120
121        outOffs = 0;
122    }
123
124    public int read() throws IOException {
125        if (outOffs >= outLength) {
126            try {
127                readAndDecompress();
128            }
129            catch(EOFException eof) {
130                return -1;
131            }
132        }
133
134        return outBuf[outOffs++] & 0xff;
135    }
136
137    public int read(byte[] b, int off, int len)
138            throws IOException {
139        int count = 0;
140        while (count < len) {
141            if (outOffs >= outLength) {
142                try {
143                    // If we've read at least one decompressed
144                    // byte and further decompression would
145                    // require blocking, return the count.
146                    if ((count > 0) && (in.available() == 0))
147                        return count;
148                    else
149                        readAndDecompress();
150                }
151                catch(EOFException eof) {
152                    if (count == 0)
153                        count = -1;
154                    return count;
155                }
156            }
157
158            int toCopy =
159                    Math.min(outLength - outOffs, len - count);
160            System.arraycopy(
161                    outBuf, outOffs, b, off + count, toCopy);
162            outOffs += toCopy;
163            count += toCopy;
164        }
165
166        return count;
167    }
168
169    public int available() throws IOException {
170        // This isn't precise, but should be an adequate
171        // lower bound on the actual amount of available data
172        return (outLength - outOffs) + in.available();
173    }
174
175}
