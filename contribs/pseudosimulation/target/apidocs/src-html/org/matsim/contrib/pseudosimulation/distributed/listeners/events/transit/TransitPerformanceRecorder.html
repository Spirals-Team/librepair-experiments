043import java.util.Map;
044
045public class TransitPerformanceRecorder {
046
047    private final Scenario scenario;
048    private final Map<Id<Vehicle>, Vehicle> vehicles;
049    private final Map<Id<VehicleType>, VehicleType> vehicleTypes;
050    private MobSimSwitcher switcher;
051    private TransitPerformance transitPerformance;
052    private VehicleTrackerCollection vehicletrackers;
053    private Map<Id<Departure>, TransitRoute> departureIdToRoute;
054
055
056
057    private boolean shouldReset() {
058        if(switcher == null)
059            return true;
060        return switcher.isQSimIteration();
061    }
062    public TransitPerformanceRecorder(Scenario scenario, EventsManager eventsManager, MobSimSwitcher switcher) {
063        this(scenario, eventsManager);
064        this.switcher = switcher;
065    }
066    public TransitPerformanceRecorder(Scenario scenario, EventsManager eventsManager) {
067//        identifyVehicleRoutes();
068        vehicletrackers = new VehicleTrackerCollection(scenario.getTransitVehicles().getVehicles().size());
069        RidershipHandler handler = new RidershipHandler();
070        eventsManager.addHandler(handler);
071        this.scenario = scenario;
072        this.vehicles = scenario.getTransitVehicles().getVehicles();
073        this.vehicleTypes = scenario.getTransitVehicles().getVehicleTypes();
074        this.transitPerformance = new TransitPerformance();
075    }
076
077    public TransitPerformance getTransitPerformance() {
078        return transitPerformance;
079    }
080
081    private int getVehicleCapacity(Id<Vehicle> vehicleId) {
082        VehicleType type = vehicles.get(vehicleId).getType();
083        VehicleCapacity capacity = vehicleTypes.get(type.getId()).getCapacity();
084        return capacity.getSeats() + capacity.getStandingRoom();
085
086    }
087
088
089    class RidershipHandler implements TransitDriverStartsEventHandler, VehicleArrivesAtFacilityEventHandler, VehicleDepartsAtFacilityEventHandler, PersonEntersVehicleEventHandler, PersonLeavesVehicleEventHandler {
090
091
092        @Override
093        public void reset(int iteration) {
094            if(shouldReset()) {
095                vehicletrackers = new VehicleTrackerCollection(scenario.getTransitVehicles().getVehicles().size());
096                transitPerformance = new TransitPerformance();
097            }
098        }
099
100
101        @Override
102        public void handleEvent(PersonEntersVehicleEvent event) {
103            VehicleTracker tracker = vehicletrackers.get(event.getVehicleId().toString(), event.getTime());
104            //skip car drivers
105            if (tracker == null) return;
106            tracker.ridershipIncrement(event);
107        }
108
109        @Override
110        public void handleEvent(PersonLeavesVehicleEvent event) {
111            VehicleTracker tracker = vehicletrackers.get(event.getVehicleId().toString(), event.getTime());
112//                      skip car drivers
113            if (tracker == null) return;
114            tracker.ridershipDecrement(event);
115        }
116
117        @Override
118        public void handleEvent(TransitDriverStartsEvent event) {
119            FullDeparture fullDeparture = new FullDeparture(event.getTransitLineId(), event.getTransitRouteId(),
120                    event.getVehicleId(), event.getDepartureId());
121            VehicleTracker tracker = new VehicleTracker(fullDeparture, event.getDriverId(), getVehicleCapacity(event.getVehicleId()));
122            vehicletrackers.put(event.getVehicleId().toString(),event.getTime(), tracker);
123        }
124
125        @Override
126        public void handleEvent(VehicleArrivesAtFacilityEvent event) {
127            VehicleTracker tracker = vehicletrackers.get(event.getVehicleId().toString(),event.getTime());
128            DwellEvent dwellEvent = tracker.registerArrival(event);
129            transitPerformance.addVehicleDwellEventAtStop(tracker.getFullDeparture().getLineId(),tracker.getFullDeparture().getRouteId(),event.getFacilityId(),dwellEvent);
130
131        }
132
133        @Override
134        public void handleEvent(VehicleDepartsAtFacilityEvent event) {
135            VehicleTracker tracker = vehicletrackers.get(event.getVehicleId().toString(), event.getTime());
136            tracker.registerDeparture(event);
137
138        }
139    }
140}
