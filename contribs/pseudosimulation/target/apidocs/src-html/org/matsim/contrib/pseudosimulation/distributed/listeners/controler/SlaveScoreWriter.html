001package org.matsim.contrib.pseudosimulation.distributed.listeners.controler;
002
003import org.matsim.contrib.pseudosimulation.distributed.MasterControler;
004import org.matsim.core.controler.events.IterationEndsEvent;
005import org.matsim.core.controler.events.ShutdownEvent;
006import org.matsim.core.controler.events.StartupEvent;
007import org.matsim.core.controler.listener.IterationEndsListener;
008import org.matsim.core.controler.listener.ShutdownListener;
009import org.matsim.core.controler.listener.StartupListener;
010import org.matsim.core.utils.charts.XYLineChart;
011import org.matsim.core.utils.io.IOUtils;
012import org.matsim.core.utils.io.UncheckedIOException;
013
014import java.io.BufferedWriter;
015import java.io.IOException;
016
017public class SlaveScoreWriter implements IterationEndsListener,
018                ShutdownListener, StartupListener {
019        private final MasterControler controler;
020        private BufferedWriter out;
021        final static int INDEX_WORST = 0;
022        final static int INDEX_BEST = 1;
023        final static int INDEX_AVERAGE = 2;
024        final static int INDEX_EXECUTED = 3;
025
026        public SlaveScoreWriter(MasterControler controler) {
027                super();
028                this.controler = controler;
029
030        }
031
032        @Override
033        public void notifyIterationEnds(IterationEndsEvent event) {
034                double[][] history = controler.getSlaveScoreHistory();
035                int idx = event.getIteration() - controler.getConfig().controler().getFirstIteration();
036                try {
037                        out.write(event.getIteration() + "\t"
038                                        + history[INDEX_EXECUTED][idx] + "\t"
039                                        + history[INDEX_WORST][idx] + "\t"
040                                        + history[INDEX_AVERAGE][idx] + "\t"
041                                        + history[INDEX_BEST][idx] + "\n");
042                        out.flush();
043                } catch (IOException e) {
044                        // TODO Auto-generated catch block
045                        e.printStackTrace();
046                }
047                // create chart when data of more than one iteration is available.
048                if(idx<2){
049                        return;
050                }
051                XYLineChart chart = new XYLineChart("SlaveHandler Score Statistics",
052                                "iteration", "score");
053                double[] iterations = new double[idx];
054                for (int i = 0; i < idx; i++) {
055                        iterations[i] = i + controler.getConfig().controler().getFirstIteration()+1;
056                }
057                double[] values = new double[idx];
058                double[] fullhist = new double[idx];
059                int[] series = { INDEX_WORST, INDEX_BEST, INDEX_AVERAGE, INDEX_EXECUTED };
060                String[] seriesNames = { "avg. worst score", "avg. best score",
061                                "avg. of plans' average score", "avg. executed score" };
062                for (int s = 0; s < series.length; s++) {
063                        System.arraycopy(history[series[s]], 1, fullhist, 0,
064                                        fullhist.length);
065                        int valuecounter = 0;
066                        for (int i =0;i<idx;i++) {
067                                values[valuecounter++] = fullhist[i];
068                        }
069                        chart.addSeries(seriesNames[s], iterations, values);
070
071                }
072
073
074
075
076                chart.addMatsimLogo();
077                chart.saveAsPng(controler.getMATSimControler().getControlerIO().getOutputPath()
078                                + "/slaveScoreStats.png", 1200, 800);
079
080        }
081
082        @Override
083        public void notifyShutdown(final ShutdownEvent controlerShudownEvent) {
084                try {
085                        this.out.close();
086                } catch (IOException e) {
087                        e.printStackTrace();
088                }
089
090        }
091
092        @Override
093        public void notifyStartup(StartupEvent event) {
094                String fileName = controler.getMATSimControler().getControlerIO().getOutputPath()
095                                + "/slaveScoreStats.txt";
096                this.out = IOUtils.getBufferedWriter(fileName);
097
098                try {
099                        this.out.write("ITERATION\tavg. EXECUTED\tavg. WORST\tavg. AVG\tavg. BEST\n");
100                        this.out.flush();
101                } catch (IOException e) {
102                        throw new UncheckedIOException(e);
103                }
104
105        }
106}
