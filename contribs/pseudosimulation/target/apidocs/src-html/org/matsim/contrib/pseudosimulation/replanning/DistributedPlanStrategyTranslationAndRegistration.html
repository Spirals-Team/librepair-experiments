001package org.matsim.contrib.pseudosimulation.replanning;
002
003import org.matsim.contrib.pseudosimulation.RunPSim;
004import org.matsim.contrib.pseudosimulation.replanning.factories.DistributedPlanMutatorStrategyFactory;
005import org.matsim.contrib.pseudosimulation.replanning.factories.DistributedPlanSelectorStrategyFactory;
006import org.matsim.core.config.Config;
007import org.matsim.core.config.groups.StrategyConfigGroup;
008import org.matsim.core.controler.AbstractModule;
009import org.matsim.core.controler.Controler;
010import org.matsim.core.replanning.PlanStrategy;
011import org.matsim.core.replanning.strategies.*;
012
013import javax.inject.Provider;
014import java.util.HashMap;
015import java.util.Map;
016
017/**
018 * @author fouriep
019 *         <p></p>
020 *         Essentially a lookup used in config translation. Also registers
021 *         extended strategies with the controler.
022 *         <p></p>
023 *         If a mutating strategy is sent for pseudo-simulation, it needs to be
024 *         marked as such, and registered with the {@link RunPSim}.
025 *         Non-mutating strategies, e.g. selector strategies, should be disabled
026 *         during PSim iterations, and only run during QSim iterations.
027 *         <p></p>
028 *         <p></p>
029 *         This class records strategies that should work with PSim. It creates staretgy delegate instances
030 *         and adds mutated strategies to PSim.
031 *         Each factory is registered during controler
032 *         construction, and the config entries are changed to refer to their
033 *         PSim equivalents in the controler's substituteStrategies() method.
034 *         <p></p>
035 *         <p></p>
036 *         <p></p>
037 *         <p></p>
038 *         <B>NOTE:</B> to save processing overhead, selector strategies are set
039 *         up to always return the person's current selected plan during
040 *         non-QSim iterations.
041 */
042public class DistributedPlanStrategyTranslationAndRegistration {
043    public static final String SUFFIX = "PSIM";
044    public static Map<String, Class<? extends Provider<PlanStrategy>>> SupportedSelectors = new HashMap<>();
045    public static Map<String, Class<? extends Provider<PlanStrategy>>> SupportedMutators = new HashMap<>();
046    public static Map<String, Character> SupportedMutatorGenes = new HashMap<>();
047    public static boolean TrackGenome = false;
048
049    private DistributedPlanStrategyTranslationAndRegistration() {
050    }
051
052    static {
053        initMaps();
054    }
055
056    static void initMaps() {
057        SupportedSelectors.put("KeepLastSelected", KeepLastSelected.class);
058        SupportedSelectors.put("BestScore", SelectBest.class);
059        SupportedSelectors.put("ChangeExpBeta", ChangeExpBeta.class);
060        SupportedSelectors.put("SelectExpBeta", SelectExpBeta.class);
061        SupportedSelectors.put("SelectRandom", SelectRandom.class);
062        SupportedSelectors.put("SelectPathSizeLogit", SelectPathSizeLogit.class);
063
064        SupportedMutators.put("ReRoute", ReRoute.class);
065        SupportedMutators.put("TimeAllocationMutator", TimeAllocationMutator.class);
066        SupportedMutators.put("TimeAllocationMutator_ReRoute", TimeAllocationMutatorReRoute.class);
067        SupportedMutators.put("ChangeSingleTripMode", ChangeSingleTripMode.class);
068        SupportedMutators.put("SubtourModeChoice", SubtourModeChoice.class);
069        SupportedMutators.put("ChangeTripMode", ChangeTripMode.class);
070
071        SupportedMutatorGenes.put("ReRoute", new Character('A'));
072        SupportedMutatorGenes.put("TimeAllocationMutator", new Character('B'));
073        SupportedMutatorGenes.put("TimeAllocationMutator_ReRoute", new Character('C'));
074        SupportedMutatorGenes.put("ChangeLegMode", new Character('D'));
075        SupportedMutatorGenes.put("ChangeSingleLegMode", new Character('E'));
076        SupportedMutatorGenes.put("ChangeSingleTripMode", new Character('F'));
077        SupportedMutatorGenes.put("SubtourModeChoice", new Character('G'));
078        SupportedMutatorGenes.put("ChangeTripMode", new Character('H'));
079        SupportedMutatorGenes.put("TripSubtourModeChoice", new Character('J'));
080        SupportedMutatorGenes.put("TransitLocationChoice", new Character('K'));
081
082    }
083
084    public static void registerStrategiesWithControler(final Controler controler, final PlanCatcher slave, final boolean quickReplanning, final int selectionInflationFactor) {
085        for (final Map.Entry<String, Class<? extends Provider<PlanStrategy>>> e : SupportedSelectors.entrySet()) {
086            controler.addOverridingModule(new AbstractModule() {
087                @Override
088                public void install() {
089                    addPlanStrategyBinding(e.getKey()).toProvider(e.getValue());
090                    addPlanStrategyBinding(e.getKey() + SUFFIX).toProvider(new DistributedPlanSelectorStrategyFactory(slave, quickReplanning, selectionInflationFactor, controler, e.getKey()));
091                }
092            });
093
094        }
095
096        for (final Map.Entry<String, Class<? extends Provider<PlanStrategy>>> e : SupportedMutators.entrySet()) {
097            controler.addOverridingModule(new AbstractModule() {
098                @Override
099                public void install() {
100                    addPlanStrategyBinding(e.getKey()).toProvider(e.getValue());
101                    addPlanStrategyBinding(e.getKey() + SUFFIX).toProvider(new DistributedPlanMutatorStrategyFactory(slave, SupportedMutatorGenes.get(e.getKey()), TrackGenome, controler, e.getKey()));
102                }
103            });
104        }
105
106    }
107
108
109    public static boolean isStrategySupported(String name) {
110        if (SupportedSelectors.size() == 0)
111            initMaps();
112        if (SupportedSelectors.containsKey(name))
113            return true;
114        else return SupportedMutators.containsKey(name);
115    }
116
117    public static void substituteStrategies(Config config, boolean quickReplanning, int selectionInflationFactor) {
118        for (StrategyConfigGroup.StrategySettings settings : config.strategy().getStrategySettings()) {
119
120            String classname = settings.getStrategyName();
121
122            if (classname.startsWith("org.matsim.demandmodeling.plans.strategies.")) {
123                classname = classname.replace("org.matsim.demandmodeling.plans.strategies.", "");
124                settings.setStrategyName(classname);
125            }
126            if (!DistributedPlanStrategyTranslationAndRegistration.isStrategySupported(classname)) {
127                throw new RuntimeException("Strategy " + classname + " not known to be compatible with (Distributed) PSim. Exiting.");
128            } else {
129                if (SupportedMutators.containsKey(classname))
130                    settings.setStrategyName(classname + DistributedPlanStrategyTranslationAndRegistration.SUFFIX);
131                if (SupportedSelectors.containsKey(classname)) {
132                    settings.setStrategyName(classname + DistributedPlanStrategyTranslationAndRegistration.SUFFIX);
133                    //implement quick replanning by simply multiplying the selector weights by the number of PSim Iterations
134                    if (quickReplanning) {
135                        settings.setWeight(settings.getWeight() * (double) selectionInflationFactor);
136                    }
137                }
138            }
139
140
141        }
142
143    }
144}
