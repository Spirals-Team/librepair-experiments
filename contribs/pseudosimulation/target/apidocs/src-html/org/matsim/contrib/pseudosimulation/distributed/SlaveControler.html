087    public SlaveControler(String[] args) throws IOException, ClassNotFoundException, ParseException, InterruptedException {
088        lastIterationStartTime = System.currentTimeMillis();
089        System.setProperty("matsim.preferLocalDtds", "true");
090        Options options = new Options();
091        options.addOption(OptionBuilder.withLongOpt("config")
092                .withDescription("Config file location")
093                .hasArg(true)
094                .withArgName("CONFIG.XML")
095                .isRequired(false)
096                .create("c"));
097        options.addOption("h", "host", true, "Host name or IP");
098        options.addOption("p", "port", true, "Port number of MasterControler");
099        options.addOption("t", "threads", true, "Number of threads for replanning.");
100        CommandLineParser parser = new BasicParser();
101        CommandLine commandLine = parser.parse(options, args);
102
103
104        if (commandLine.hasOption("c")) {
105            try {
106                config = ConfigUtils.loadConfig(commandLine.getOptionValue("c"));
107
108            } catch (UncheckedIOException e) {
109                System.err.println("Config file not found");
110                printHelp(options);
111                System.exit(1);
112            }
113        } else if (commandLine.getArgs().length>0) {
114            try {
115                config = ConfigUtils.loadConfig(commandLine.getArgs()[0]);
116
117            } catch (UncheckedIOException e) {
118                System.err.println("Config file not found");
119                printHelp(options);
120                System.exit(1);
121            }
122        } else {
123            System.err.println("Config file not specified");
124            System.out.println(options.toString());
125            System.exit(1);
126        }
127
128
129        final DistributedSimConfigGroup distributedSimConfigGroup = ConfigUtils.addOrGetModule(this.config,DistributedSimConfigGroup.GROUP_NAME,DistributedSimConfigGroup.class);
130
131        //The following line will make the controler use the events manager that doesn't check for event order
132        config.parallelEventHandling().setSynchronizeOnSimSteps(false);
133        //if you don't set the number of threads, org.matsim.core.events.EventsUtils will just use the simstepmanager
134        int numThreads = config.global().getNumberOfThreads();
135        if (commandLine.hasOption("t"))
136            try {
137                numThreads = Integer.parseInt(commandLine.getOptionValue("t"));
138            } catch (NumberFormatException e) {
139                System.err.println("Number of threads should be int or it wasn't specced on cmd line. Taking the default of "+distributedSimConfigGroup.getDefaultNumThreadsOnSlave());
140                System.out.println(options.toString());
141                numThreads = distributedSimConfigGroup.getDefaultNumThreadsOnSlave();
142            }
143        else {
144            System.err.println("Will use the number of threads in config for simulation.");
145        }
146        config.global().setNumberOfThreads(numThreads);
147        config.parallelEventHandling().setNumberOfThreads(1);
148
149        String hostname = "localhost";
150        if (commandLine.hasOption("h")) {
151            hostname = commandLine.getOptionValue("h");
152        } else
153            System.err.println("No host or IP specified, using default (localhost)");
154
155        /*
156        * INITIALIZING COMMS
157        * */
158        Socket socket = null;
159        int socketNumber = distributedSimConfigGroup.getMasterPortNumber();
160        if (commandLine.hasOption("p")) {
161            try {
162                socketNumber = Integer.parseInt(commandLine.getOptionValue("p"));
163            } catch (NumberFormatException e) {
164                System.err.println("Port number should be integer. Defaulting to "+ distributedSimConfigGroup.getMasterPortNumber());
165                System.out.println(options.toString());
166            }
167        } else {
168            System.err.println("Will accept connections on default port number 12345");
169        }
170        boolean connected = false;
171        while (!connected) {
172            try {
173                socket = new Socket(hostname, socketNumber);
174                connected = true;
175            } catch (ConnectException e) {
176                Thread.sleep(1000);
177            }
178        }
179        this.reader = new ObjectInputStream(socket.getInputStream());
180        this.writer = new ObjectOutputStream(socket.getOutputStream());
181
182        int myNumber = reader.readInt();
183        slaveLogger = Logger.getLogger(("SLAVE_" + myNumber));
184
185        numberOfPSimIterationsPerCycle = reader.readInt();
186        numberOfPlansOnSlave = reader.readInt();
187        slaveMutationRate = reader.readDouble();
188        slaveLogger.warn("Running " + numberOfPSimIterationsPerCycle + " PSim iterations for every QSim iter");
189        config.controler().setLastIteration(reader.readInt());
190        initialRouting = reader.readBoolean();
191        boolean quickReplannning = reader.readBoolean();
192        fullTransitPerformanceTransmission = reader.readBoolean();
193        boolean trackGenome = reader.readBoolean();
194        IntelligentRouters = reader.readBoolean();
195        boolean diversityGeneratingPlanSelection = reader.readBoolean();
196
197        if (initialRouting) slaveLogger.warn("Performing initial routing.");
198
199        memoryUsageCalculator = new MemoryUsageCalculator();
200        writeMemoryStats();
201        writer.writeInt(config.global().getNumberOfThreads());
202        writer.flush();
203
204
205        config.controler().setOutputDirectory(config.controler().getOutputDirectory() + "_" + myNumber);
206        //limit IO
207        config.linkStats().setWriteLinkStatsInterval(0);
208        config.controler().setCreateGraphs(false);
209        config.controler().setWriteEventsInterval(0);
210        config.controler().setWritePlansInterval(0);
211        config.controler().setWriteSnapshotsInterval(0);
212        // don't load plans; receive them from master
213        config.plans().setInputFile(null);
214//        Important, otherwise Psim breaks
215        config.parallelEventHandling().setSynchronizeOnSimSteps(false);
216        config.parallelEventHandling().setNumberOfThreads(1);
217        if (slaveMutationRate > 0)
218            setReplanningWeights(config, slaveMutationRate);
219
220        scenario = ScenarioUtils.loadScenario(config);
221
222        //experimental, not to be used
223        DistributedPlanStrategyTranslationAndRegistration.TrackGenome = trackGenome;
224//        strategy substitution: mainly to register whether the option for quick replanning is to be used,
225//        as its original function is to mark plans for execution by PSim. But here, all plans are executed by PSim
226//        should rather be that PSim marks activities for execution in some other way
227        DistributedPlanStrategyTranslationAndRegistration.substituteStrategies(config, quickReplannning, numberOfPSimIterationsPerCycle);
228        matsimControler = new Controler(scenario);
229        DistributedPlanStrategyTranslationAndRegistration.registerStrategiesWithControler(this.matsimControler, plancatcher, quickReplannning, numberOfPSimIterationsPerCycle);
230        matsimControler.getConfig().controler().setOverwriteFileSetting(
231                true ?
232                        OutputDirectoryHierarchy.OverwriteFileSetting.overwriteExistingFiles :
233                        OutputDirectoryHierarchy.OverwriteFileSetting.failIfDirectoryExists);
234        matsimControler.addControlerListener(this);
235
236//        init traveltime for when non yet has been received from master
237        travelTime = new ReplaceableTravelTime();
238        linkTravelTimes = new FreeSpeedTravelTime();
239        travelTime.setTravelTime(linkTravelTimes);
240
241        matsimControler.addOverridingModule(new AbstractModule() {
242            @Override
243            public void install() {
244                bindMobsim().toProvider(pSimProvider);
245            }
246        });
247
248        if (config.transit().isUseTransit()) {
249
250            stopStopTimes = new StopStopTimeCalculatorSerializable(scenario.getTransitSchedule(),
251                    config.travelTimeCalculator().getTraveltimeBinSize(), (int) (config
252                    .qsim().getEndTime() - config.qsim().getStartTime())).getStopStopTimes();
253
254            waitTimes = new WaitTimeCalculatorSerializable(scenario.getTransitSchedule(),
255                    config.travelTimeCalculator().getTraveltimeBinSize(), (int) (config
256                    .qsim().getEndTime() - config.qsim().getStartTime())).getWaitTimes();
257
258            // tell PlanSerializable to record transit routes
259            PlanSerializable.isUseTransit = true;
260
261        }
262        if (IntelligentRouters)
263            matsimControler.addOverridingModule(new AbstractModule() {
264                @Override
265                public void install() {
266                    System.out.println("init routers");
267//                    transitRouterEventsWSFactory = new TransitRouterEventsWSFactory(scenario,
268//                            waitTimes,
269//                            stopStopTimes);
270                    bind(TransitRouter.class).toProvider(transitRouterEventsWSFactory);
271                }
272            });
273
274        else {
275            final RandomizingTimeDistanceTravelDisutilityFactory disutilityFactory =
276                    new RandomizingTimeDistanceTravelDisutilityFactory(TransportMode.car, config.planCalcScore());
277            matsimControler.addOverridingModule(new AbstractModule() {
278                @Override
279                public void install() {
280                    bindCarTravelDisutilityFactory().toInstance(disutilityFactory);
281                }
282            });
283            disutilityFactory.setSigma(0.1);
284
285        }
286        matsimControler.addOverridingModule(new AbstractModule() {
287            @Override
288            public void install() {
289                bind(TravelTime.class).toInstance(travelTime);
290                if (scenario.getConfig().strategy().getPlanSelectorForRemoval().equals("DiversityGeneratingPlansRemover")) {
291                    bindPlanSelectorForRemoval().toProvider(DiversityGeneratingPlansRemover.Builder.class);
292                }
293            }
294        });
295
296
297        if (trackGenome) {
298
299        }
300        if (diversityGeneratingPlanSelection)
301            matsimControler.getConfig().strategy().setPlanSelectorForRemoval("DiversityGeneratingPlansRemover");
302        //no use for this, if you don't exactly know the communicationsMode of population when something goes wrong.
303        // better to have plans written out every n successful iterations, specified in the config
304        matsimControler.getConfig().controler().setDumpDataAtEnd(false);
305        matsimControler.getConfig().strategy().setMaxAgentPlanMemorySize(numberOfPlansOnSlave);
306    }

353    @Override
354    public void notifyIterationStarts(IterationStartsEvent event) {
355        if (numberOfIterations >= 0 || initialRouting)
356            iterationTimes.add(System.currentTimeMillis() - lastIterationStartTime);
357// send plans only after the previous iteration has completed
358        if (initialRouting || (numberOfIterations > 0 && numberOfIterations % numberOfPSimIterationsPerCycle == 0)) {
359            this.totalIterationTime = getTotalIterationTime();
360            communications();
361            if (somethingWentWrong) Runtime.getRuntime().halt(0);
362            initialRouting = false;
363        }
364        this.currentIteration = event.getIteration();
365        lastIterationStartTime = System.currentTimeMillis();
366        travelTime.setTravelTime(linkTravelTimes);
367        pSimProvider.setTravelTime(linkTravelTimes);
368        if (config.transit().isUseTransit()) {
369            pSimProvider.setStopStopTime(stopStopTimes);
370            pSimProvider.setWaitTime(waitTimes);
371            pSimProvider.setTransitPerformance(transitPerformance);
372            if (transitRouterEventsWSFactory != null) {
373//                transitRouterEventsWSFactory.setStopStopTimeCalculator(stopStopTimes);
374//                transitRouterEventsWSFactory.setWaitTimeCalculator(waitTimes);
375            }
376        }
377        plancatcher.init();
378        numberOfIterations++;
379    }

381    public double getTotalIterationTime() {

385        }

396    private List<PersonSerializable> getPersonsToSend(int diff) {
397        int i = 0;
398        List<PersonSerializable> personsToSend = new ArrayList<>();
399        Set<Id<Person>> personIdsToRemove = new HashSet<>();
400        for (Id<Person> personId : matsimControler.getScenario().getPopulation().getPersons().keySet()) {
401            if (i++ >= diff) break;
402            personsToSend.add(new PersonSerializable(matsimControler.getScenario().getPopulation().getPersons().get(personId)));
403            personIdsToRemove.add(personId);
404        }
405        for (Id<Person> personId : personIdsToRemove)
406            matsimControler.getScenario().getPopulation().getPersons().remove(personId);
407        return personsToSend;
408    }

410    public void transmitPlans() throws IOException, ClassNotFoundException {
411        Map<String, PlanSerializable> tempPlansCopyForSending = new HashMap<>();
412        for (Person person : matsimControler.getScenario().getPopulation().getPersons().values()) {
413            PlanSerializable planSerializable = new PlanSerializable(person.getSelectedPlan());
414            planSerializable.pSimScore = planSerializable.getScore() == null ? 0 : planSerializable.getScore();
415            tempPlansCopyForSending.put(person.getId().toString(), planSerializable);
416        }
417        Map<String, PlanSerializable> plansCopyForSending = tempPlansCopyForSending;
418        slaveLogger.warn("Sending " + plansCopyForSending.size() + " plans...");
419        writer.writeInt(currentIteration);
420        writer.writeInt(masterCurrentIteration);
421        writer.writeObject(plansCopyForSending);
422        slaveLogger.warn("Sending completed.");
423
424    }

441    public void transmitPerformance() throws IOException {
442        if (totalIterationTime > 0) {
443            slaveLogger.warn("Spent a total of " + totalIterationTime +
444                    " running " + executedPlanCount +
445                    " person plans for " + numberOfPSimIterationsPerCycle +
446                    " PSim iterations.");
447        }
448        writer.writeDouble(totalIterationTime);
449        writer.writeInt(matsimControler.getScenario().getPopulation().getPersons().size());
450        //send memory usage fraction of max to prevent being assigned more persons
451        writeMemoryStats();

455    private double getMemoryUse() {
456        putOutTheGarbage();
457        double totalMemory = Runtime.getRuntime().totalMemory();
458        putOutTheGarbage();
459        double freeMemory = Runtime.getRuntime().freeMemory();
460        double usedMemoryEst = totalMemory - freeMemory;
461        double maxMemory = Runtime.getRuntime().maxMemory();
462        return usedMemoryEst / maxMemory;
463    }

470    private void collectGarbage() {
471        try {
472            System.gc();
473            long fSLEEP_INTERVAL = 100;
474            Thread.currentThread().sleep(fSLEEP_INTERVAL);
475            System.runFinalization();
476            Thread.currentThread().sleep(fSLEEP_INTERVAL);
477        } catch (InterruptedException ex) {
478            ex.printStackTrace();
479            somethingWentWrong = true;
480        }
481    }

492    public void poolPersons() throws IOException {
493        slaveLogger.warn("Load balancing...");
494        int diff = reader.readInt();
495        slaveLogger.warn("Received " + diff + " as lb instr from master");
496        List<PersonSerializable> personsToSend = new ArrayList<>();
497        if (diff > 0) {
498            personsToSend = getPersonsToSend(diff);
499        }
500        writer.writeObject(personsToSend);
501        slaveLogger.warn("Sent " + personsToSend.size() + " pax to master");
502    }

504    public void communications() {
505        CommunicationsMode communicationsMode = CommunicationsMode.WAIT;
506        slaveLogger.warn("Initializing communications...");
507        try {
508            while (!communicationsMode.equals(CommunicationsMode.CONTINUE)) {
509                communicationsMode = (CommunicationsMode) reader.readObject();
510                switch (communicationsMode) {
511                    case TRANSMIT_SCENARIO:
512                        distributePersons();
513                        break;
514                    case TRANSMIT_TRAVEL_TIMES:
515                        transmitTravelTimes();
516                        break;
517                    case POOL_PERSONS:
518                        poolPersons();
519                        break;
520                    case DISTRIBUTE_PERSONS:
521                        distributePersons();
522                        break;
523                    case TRANSMIT_PLANS_TO_MASTER:
524                        transmitPlans();
525                        slaveIsOKForNextIter();
526                        break;
527                    case TRANSMIT_SCORES:
528                        transmitScores();
529                        break;
530                    case TRANSMIT_PERFORMANCE:
531                        transmitPerformance();
532                        break;
533                    case CONTINUE:
534                        break;
535                    case WAIT:
536                        Thread.sleep(10);
537                        break;
538                    case DIE:
539                        slaveLogger.error("Got the kill signal from MASTER. Bye.");
540                        Runtime.getRuntime().halt(0);
541                        break;
542
543                }
544                // sending a boolean forces the thread on the master to wait
545                writer.writeBoolean(true);
546                writer.flush();
547            }
548            writer.reset();
549        } catch (ClassNotFoundException | IOException | InterruptedException e) {
550            e.printStackTrace();
551            slaveLogger.error("Something went wrong. Exiting.");
552            somethingWentWrong = true;
553            return;
554        }
555        slaveLogger.warn("Communications completed.");
556    }

578    @Override
579    public void notifyBeforeMobsim(BeforeMobsimEvent event) {
580        selectedPlanScoreMemory = new HashMap<>(scenario.getPopulation().getPersons().size());
581        if (event.getIteration() == 0) {
582            plancatcher.init();
583            for (Person person : scenario.getPopulation().getPersons().values()) {
584                plancatcher.addPlansForPsim(person.getSelectedPlan());
585            }
586        } else {
587            for (Person person : scenario.getPopulation().getPersons().values()) {
588                selectedPlanScoreMemory.put(person.getId(), person.getSelectedPlan().getScore());
589            }
590            for (Plan plan : plancatcher.getPlansForPSim()) {
591                selectedPlanScoreMemory.remove(plan.getPerson().getId());
592            }
593        }
594
595        executedPlanCount += plancatcher.getPlansForPSim().size();
596    }

598    @Override
599    public void notifyIterationEnds(IterationEndsEvent event) {
600        Iterator<Map.Entry<Id<Person>, Double>> iterator = selectedPlanScoreMemory.entrySet().iterator();
601        StopStopTimeCalculatorSerializable.printCallStatisticsAndReset();
602        WaitTimeCalculatorSerializable.printCallStatisticsAndReset();
603        while (iterator.hasNext()) {
604            Map.Entry<Id<Person>, Double> entry = iterator.next();
605            scenario.getPopulation().getPersons().get(entry.getKey()).getSelectedPlan().setScore(entry.getValue());
606        }
607    }

609    private void setReplanningWeights(Config config, double mutationRate) {
610        if (mutationRate > 1)
611            mutationRate = 0.9999;
612        List<StrategyConfigGroup.StrategySettings> strategySettings = new ArrayList<>();
613        strategySettings.addAll(config.strategy().getStrategySettings());
614        Map<Integer, Double> selectors = new HashMap<>();
615        Map<Integer, Double> mutators = new HashMap<>();
616        for (int i = 0; i < strategySettings.size(); i++) {
617            StrategyConfigGroup.StrategySettings setting = strategySettings.get(i);
618            if (DistributedPlanStrategyTranslationAndRegistration.SupportedSelectors.keySet().contains(setting.getStrategyName()))
619                selectors.put(i, setting.getWeight());
620            else {
621                mutators.put(i, setting.getWeight());
622            }
623        }
624
625        double mutatorSum = CollectionUtils.sumElements(mutators.values());
626        double selectorSum = CollectionUtils.sumElements(selectors.values());
627        // set to new weight
628        for (Map.Entry<Integer, Double> entry : selectors.entrySet()) {
629            strategySettings.get(entry.getKey()).setWeight((1 - mutationRate) * entry.getValue() / selectorSum);
630        }
631        for (Map.Entry<Integer, Double> entry : mutators.entrySet()) {
632            strategySettings.get(entry.getKey()).setWeight(mutationRate * entry.getValue() / mutatorSum);
633        }
634        //put it back in the config
635        config.strategy().clearStrategySettings();
636        for (StrategyConfigGroup.StrategySettings strategySetting : strategySettings) {
637            config.strategy().addStrategySettings(strategySetting);
638        }
639    }

644        //send the last iteratin's plans
645        this.totalIterationTime = getTotalIterationTime();
646        communications();
647        //wait for the kill signal from master
648        communications();
649    }
