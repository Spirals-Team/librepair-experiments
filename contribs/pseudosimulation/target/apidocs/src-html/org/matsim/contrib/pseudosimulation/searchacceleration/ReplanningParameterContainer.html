020package org.matsim.contrib.pseudosimulation.searchacceleration;
021
022import java.util.LinkedHashMap;
023import java.util.Map;
024
025import org.matsim.api.core.v01.Id;
026import org.matsim.api.core.v01.network.Link;
027import org.matsim.api.core.v01.network.Network;
028import org.matsim.core.router.util.TravelTime;
029
030import floetteroed.utilities.Units;
031
032/**
033 * 
034 * @author Gunnar Flötteröd
035 *
036 */
037public interface ReplanningParameterContainer {
038
039        public static Map<Id<Link>, Double> newUniformLinkWeights(final Network network) {
040                final Map<Id<Link>, Double> weights = new LinkedHashMap<>();
041                for (Link link : network.getLinks().values()) {
042                        weights.put(link.getId(), 1.0);
043                }
044                return weights;
045        }
046
047        public static Map<Id<Link>, Double> newOneOverCapacityLinkWeights(final Network network) {
048                final Map<Id<Link>, Double> weights = new LinkedHashMap<>();
049                for (Link link : network.getLinks().values()) {
050                        final double cap_veh_h = link.getFlowCapacityPerSec() * Units.VEH_H_PER_VEH_S;
051                        if (cap_veh_h <= 1e-6) {
052                                throw new RuntimeException("link " + link.getId() + " has capacity " + cap_veh_h + " veh/h");
053                        }
054                        weights.put(link.getId(), 1.0 / cap_veh_h);
055                }
056                return weights;
057        }
058
059        public double getMeanLambda(int iteration);
060
061        public double getDelta(int iteration, Double deltaN2);
062        
063        public boolean isCongested(Object locObj, int timeBin, TravelTime travelTimes);
064        
065        public double getWeight(Object locObj, int timeBin, TravelTime travelTimes);
066        
067        // public boolean isCongested(Object locObj, int time_s);
068
069}
