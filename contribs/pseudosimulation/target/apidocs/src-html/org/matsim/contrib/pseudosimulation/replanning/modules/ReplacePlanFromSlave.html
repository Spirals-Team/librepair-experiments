001package org.matsim.contrib.pseudosimulation.replanning.modules;
002
003import org.matsim.api.core.v01.population.HasPlansAndId;
004import org.matsim.api.core.v01.population.Person;
005import org.matsim.api.core.v01.population.Plan;
006import org.matsim.api.core.v01.replanning.PlanStrategyModule;
007import org.matsim.core.replanning.PlanStrategy;
008import org.matsim.core.replanning.ReplanningContext;
009
010import java.util.HashMap;
011
012public class ReplacePlanFromSlave implements PlanStrategy {
013
014        private final HashMap<String, Plan> plans;
015
016        public ReplacePlanFromSlave(HashMap<String, Plan> plans) {
017                this.plans=plans;
018        }
019
020        @Override
021        public void run(HasPlansAndId<Plan,Person> person) {
022//              leave removing of plans to the plans remover. Thsi was initially so that the master always only had one plan in memory
023//              however, this doesn't have to be the case for all scenarios.
024//              person.removePlan(person.getSelectedPlan());
025                Plan plan = plans.get(person.getId().toString());
026                person.addPlan(plan);   
027                person.setSelectedPlan(plan);
028
029        }
030
031        @Override
032        public void init(ReplanningContext replanningContext) {
033
034        }
035
036        @Override
037        public void finish() {
038
039        }
040
041}
