120    /**
121     * Removes the specified act from the plan as well as a leg according to the following rule:
122     * <ul>
123     * <li>first act: removes the act and the following leg</li>
124     * <li>last act: removes the act and the previous leg</li>
125     * <li>in-between act: removes the act, removes the previous leg's route, and removes the following leg.
126     * </ul>
127     *
128     * @param index
129     */
130    public final void removeActivity(final int index) {
131        if ((index % 2 != 0) || (index < 0) || (index > getPlanElements().size() - 1)) {
132            log.warn(this + "[index=" + index + " is wrong. nothing removed]");
133        } else if (getPlanElements().size() == 1) {
134            log.warn(this + "[index=" + index + " only one act. nothing removed]");
135        } else {
136            if (index == 0) {
137                // remove first act and first leg
138                getPlanElements().remove(index + 1); // following leg
139                getPlanElements().remove(index); // act
140            } else if (index == getPlanElements().size() - 1) {
141                // remove last act and last leg
142                getPlanElements().remove(index); // act
143                getPlanElements().remove(index - 1); // previous leg
144            } else {
145                // remove an in-between act
146                Leg prev_leg = (Leg) getPlanElements().get(index - 1); // prev leg;
147                prev_leg.setDepartureTime(Time.UNDEFINED_TIME);
148                prev_leg.setTravelTime(Time.UNDEFINED_TIME);
149                prev_leg.setTravelTime( Time.UNDEFINED_TIME - prev_leg.getDepartureTime() );
150                prev_leg.setRoute(null);
151
152                getPlanElements().remove(index + 1); // following leg
153                getPlanElements().remove(index); // act
154            }
155        }
156    }

162    /**
163     * Removes the specified leg <b>and</b> the following act, too! If the following act is not the last one,
164     * the following leg will be emptied to keep consistency (i.e. for the route)
165     *
166     * @param index
167     */
168    public final void removeLeg(final int index) {
169        if ((index % 2 == 0) || (index < 1) || (index >= getPlanElements().size() - 1)) {
170            log.warn(this + "[index=" + index + " is wrong. nothing removed]");
171        } else {
172            if (index != getPlanElements().size() - 2) {
173                // not the last leg
174                Leg next_leg = (Leg) getPlanElements().get(index + 2);
175                next_leg.setDepartureTime(Time.UNDEFINED_TIME);
176                next_leg.setTravelTime(Time.UNDEFINED_TIME);
177                next_leg.setTravelTime( Time.UNDEFINED_TIME - next_leg.getDepartureTime() );
178                next_leg.setRoute(null);
179            }
180            getPlanElements().remove(index + 1); // following act
181            getPlanElements().remove(index); // leg
182        }
183    }

253    /**
254     * loads a copy of an existing plan, but keeps the person reference
255     * <p></p>
256     * Design comments:<ul>
257     * <li> In my intuition, this is really a terrible method: (1) Plan is a data object, not a behavioral object, and thus it should be accessed
258     * from static, interface-based methods only.
259     * (2) It is not clear about the fact if it is doing a deep or a shallow copy.  The only excuse is that this is one of the oldest parts of
260     * matsim.  kai, jan'13
261     * </ul>
262     *
263     * @param in a plan who's data will be loaded into this plan
264     */
265    public final void copyFrom(final Plan in) {
266        this.getPlanElements().clear();
267        setScore(in.getScore());
268        this.setType(in.getType());
269        for (PlanElement pe : in.getPlanElements()) {
270            if (pe instanceof Activity) {
271                //no need to cast to ActivityImpl here
272                Activity a = (Activity) pe;
273                getPlanElements().add(PopulationUtils.createActivity(a));
274            } else if (pe instanceof Leg) {
275                Leg l = (Leg) pe;
276                Leg l2 = createAndAddLeg(l.getMode());
277                l2.setDepartureTime(l.getDepartureTime());
278                l2.setTravelTime(l.getTravelTime());
279                if (pe instanceof Leg) {
280                    // get the arrival time information only if available
281                    Leg r = ((Leg) pe);
282                        l2.setTravelTime( r.getDepartureTime() + r.getTravelTime() - l2.getDepartureTime() );
283                }
284                if (l.getRoute() != null) {
285                    l2.setRoute(l.getRoute().clone());
286                }
287            } else {
288                throw new IllegalArgumentException("unrecognized plan element type discovered");
289            }
290        }
291    }

293    /**
294     * Inserts a leg and a following act at position <code>pos</code> into the plan.
295     *
296     * @param pos the position where to insert the leg-act-combo. acts and legs are both counted from the beginning starting at 0.
297     * @param leg the leg to insert
298     * @param act the act to insert, following the leg
299     * @throws IllegalArgumentException If the leg and act cannot be inserted at the specified position without retaining the correct order of legs and acts.
300     */
301    public void insertLegAct(final int pos, final Leg leg, final Activity act) throws IllegalArgumentException {
302        if (pos < getPlanElements().size()) {
303            Object o = getPlanElements().get(pos);
304            if (!(o instanceof Leg)) {
305                throw new IllegalArgumentException("Position to insert leg and act is not valid (act instead of leg at position).");
306            }
307        } else if (pos > getPlanElements().size()) {
308            throw new IllegalArgumentException("Position to insert leg and act is not valid.");
309        }
310
311        getPlanElements().add(pos, act);
312        getPlanElements().add(pos, leg);
313    }

347    public int getActLegIndex(final PlanElement o) {
348        if ((o instanceof Leg) || (o instanceof Activity)) {
349            for (int i = 0; i < getPlanElements().size(); i++) {
350                if (getPlanElements().get(i).equals(o)) {
351                    return i;
352                }
353            }
354            return -1;
355        }
356        throw new IllegalArgumentException("Method call only valid with a Leg or Act instance as parameter!");
357    }
