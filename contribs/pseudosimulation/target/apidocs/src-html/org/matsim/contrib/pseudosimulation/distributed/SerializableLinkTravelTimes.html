001package org.matsim.contrib.pseudosimulation.distributed;
002
003import java.io.Serializable;
004import java.util.Collection;
005import java.util.HashMap;
006import java.util.Iterator;
007import java.util.Map;
008
009import org.matsim.api.core.v01.network.Link;
010import org.matsim.api.core.v01.population.Person;
011import org.matsim.core.router.util.TravelTime;
012import org.matsim.vehicles.Vehicle;
013
014public class SerializableLinkTravelTimes implements Serializable, TravelTime {
015
016        private final double[][] times;
017        private final Map<String, Integer> indices = new HashMap<>();
018        private final int travelTimeBinSize;
019
020        public SerializableLinkTravelTimes(TravelTime linkTravelTimes,
021                        int traveltimeBinSize, double endTime,
022                        Collection<? extends Link> links) {
023                this.travelTimeBinSize = traveltimeBinSize;
024                endTime = endTime <= 0 ? 86400 : endTime;
025                times = new double[links.size()][(int) (endTime / traveltimeBinSize)];
026                Iterator<? extends Link> iterator = links.iterator();
027                for (int i = 0; i < times.length; i++) {
028                        Link link = iterator.next();
029                        indices.put(link.getId().toString(), i);
030                        for (int j = 0; j < times[i].length; j++)
031                                times[i][j] = linkTravelTimes.getLinkTravelTime(link,
032                                                traveltimeBinSize * j, null, null);
033                }
034        }
035
036        @Override
037        public double getLinkTravelTime(Link link, double time, Person person,
038                        Vehicle vehicle) {
039                time = time % 86400;
040                try {
041                        return times[indices.get(link.getId().toString())][(int) (time / travelTimeBinSize)];
042                } catch (ArrayIndexOutOfBoundsException e) {
043                        // TODO Auto-generated catch block
044                        e.printStackTrace();
045                }
046                return time;
047        }
048
049}
