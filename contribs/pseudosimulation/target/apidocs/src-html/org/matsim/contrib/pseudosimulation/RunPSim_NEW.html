059/**
060 * @author pieterfourie
061 * @author Gunnar Flötteröd
062 * 
063 */
064public class RunPSim_NEW {
065        
066        private Config config;
067        private Scenario scenario;
068        private TransitPerformanceRecorder transitPerformanceRecorder;
069        private Controler matsimControler;
070
071        private PlanCatcher plancatcher;
072        private PSimProvider pSimProvider;
073
074        public RunPSim_NEW(final Config config, final PSimConfigGroup pSimConfigGroup,
075                        final AccelerationConfigGroup accelerationConfigGroup) {
076
077                // pSim.
078                
079                this.config = config;
080                this.scenario = ScenarioUtils.loadScenario(config);
081
082                // The following line will make the controler use the events manager that
083                // doesn't check for event order.
084                // This is essential for pseudo-simulation as the PSim module generates events
085                // on a person-basis,
086                // not a system basis
087                config.parallelEventHandling().setSynchronizeOnSimSteps(false);
088                config.parallelEventHandling().setNumberOfThreads(1);
089
090                this.matsimControler = new Controler(this.scenario);
091
092                final MobSimSwitcher mobSimSwitcher = new MobSimSwitcher(pSimConfigGroup, this.scenario);
093                this.matsimControler.addControlerListener(mobSimSwitcher);
094
095                this.matsimControler.addOverridingModule(new AbstractModule() {
096                        @Override
097                        public void install() {
098                                bind(MobSimSwitcher.class).toInstance(mobSimSwitcher);
099                                bindMobsim().toProvider(SwitchingMobsimProvider.class);
100                                bind(WaitTimeCalculator.class).to(PSimWaitTimeCalculator.class);
101                                bind(WaitTime.class).toProvider(PSimWaitTimeCalculator.class);
102                                bind(StopStopTimeCalculator.class).to(PSimStopStopTimeCalculator.class);
103                                bind(StopStopTime.class).toProvider(PSimStopStopTimeCalculator.class);
104                                bind(TravelTimeCalculator.class).to(PSimTravelTimeCalculator.class);
105                                bind(TravelTime.class).toProvider(PSimTravelTimeCalculator.class);
106                                bind(TransitRouter.class).toProvider(TransitRouterEventsWSFactory.class);
107                                bind(PlanCatcher.class).toInstance(new PlanCatcher());
108                                bind(PSimProvider.class).toInstance(new PSimProvider(scenario, matsimControler.getEvents()));
109                        }
110                });             
111                
112                if (config.transit().isUseTransit()) {
113                        if (pSimConfigGroup.isFullTransitPerformanceTransmission()) {
114                                transitPerformanceRecorder = new TransitPerformanceRecorder(matsimControler.getScenario(),
115                                                matsimControler.getEvents(), mobSimSwitcher);
116                        }
117                }
118                
119                // Re-planner selection.
120                
121                final TimeDiscretization timeDiscr = accelerationConfigGroup.getTimeDiscretization();
122                final ReplanningParameterContainer replanningParameterProvider = new ConstantReplanningParameters(
123                                accelerationConfigGroup, this.scenario.getNetwork());
124                this.matsimControler.addOverridingModule(new AbstractModule() {
125                        @Override
126                        public void install() {
127                                this.bind(TimeDiscretization.class).toInstance(timeDiscr);
128                                this.bind(ReplanningParameterContainer.class).toInstance(replanningParameterProvider);
129                                this.bind(SearchAccelerator.class).in(Singleton.class);
130                                this.addControlerListenerBinding().to(SearchAccelerator.class);
131                                this.addEventHandlerBinding().to(SearchAccelerator.class);
132                                this.addPlanStrategyBinding(AcceptIntendedReplanningStrategy.STRATEGY_NAME)
133                                                .toProvider(AcceptIntendedReplanningStragetyProvider.class);
134                        }
135                });
136        }
137
138        public void run() {
139                this.matsimControler.run();
140        }
141
142        public static void main(String args[]) {
143
144                /*
145                 * Load configuration file. Add one configuration mode for the
146                 * AcceptIntendedReplanningStrategy that only clones previously made
147                 * hypothetical re-planning decisions (evaluated within the search acceleration
148                 * framework) is added. To switch it off/almost certainly on, its weight is set
149                 * to either zero or a very large number within the accelerated simulation
150                 * process.
151                 * 
152                 * TODO It seems as if the pSim does not interpret the simulation end time
153                 * "00:00:00" (which appears to indicate "run until everyone is done") right.
154                 */
155
156                final Config config;
157                config = ConfigUtils.loadConfig("/Users/GunnarF/NoBackup/data-workspace/searchacceleration"
158                                + "/rerun-2015-11-23a_No_Toll_large/matsim-config.xml");
159
160                AcceptIntendedReplanningStrategy.addOwnStrategySettings(config);
161                config.controler().setOverwriteFileSetting(OverwriteFileSetting.deleteDirectoryIfExists);
162                config.qsim().setEndTime(30 * 3600);
163                // config.controler().setCreateGraphs(false);
164
165                /*
166                 * Run pSim/acceleration.
167                 */
168
169                final PSimConfigGroup pSimConfigGroup = ConfigUtils.addOrGetModule(config, PSimConfigGroup.class);
170                final AccelerationConfigGroup accelerationConfigGroup = ConfigUtils.addOrGetModule(config,
171                                AccelerationConfigGroup.class);
172
173                final RunPSim_NEW pSim = new RunPSim_NEW(config, pSimConfigGroup, accelerationConfigGroup);
174                pSim.run();
175        }
