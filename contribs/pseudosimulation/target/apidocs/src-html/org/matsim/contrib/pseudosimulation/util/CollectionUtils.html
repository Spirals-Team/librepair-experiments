039     * Works on arrys instead of collections
040     * @param array
041     * @param n
042     * @param <T>
043     * @return a list of arrays
044     */
045    public static <T> List<T[]> split(T[] array, int n) {
046        Class theClass = array[0].getClass();
047        if (array.length >= n) {
048            @SuppressWarnings("unchecked")
049            List<T[]> out = new ArrayList<>();
050            int minSegmentSize = (int) Math.floor(array.length / (double) n);
051
052            int start = 0;
053            int stop = minSegmentSize;
054
055            for (int i = 0; i < n - 1; i++) {
056                int segmentSize = stop - start;
057                T[] segment = (T[]) Array.newInstance(theClass,segmentSize);
058                int j=0;
059                for (int k = start; k < stop; k++) {
060                    segment[j] = array[k];
061                    j++;
062                }
063                out.add( segment);
064                start = stop;
065                stop += segmentSize;
066            }
067
068            int segmentSize = array.length - start;
069            stop = start + segmentSize;
070            T[] segment = (T[]) Array.newInstance(theClass,segmentSize);
071            int j=0;
072            for (int k = start; k < stop; k++) {
073                segment[j] = array[k];
074                j++;
075            }
076            out.add( segment);
077
078            return out;
079        } else {
080            throw new IllegalArgumentException("n must not be smaller set size!");
081        }
082    }

085    public static <T> List<T>[] split(Collection<T> set, int n) {
086        if (set.size() >= n) {
087            @SuppressWarnings("unchecked")
088            List<T>[] arrays = new List[n];
089            int minSegmentSize = (int) Math.floor(set.size() / (double) n);
090
091            int start = 0;
092            int stop = minSegmentSize;
093
094            Iterator<T> it = set.iterator();
095
096            for (int i = 0; i < n - 1; i++) {
097                int segmentSize = stop - start;
098                List<T> segment = new ArrayList<>(segmentSize);
099                for (int k = 0; k < segmentSize; k++) {
100                    segment.add(it.next());
101                }
102                arrays[i] = segment;
103                start = stop;
104                stop += segmentSize;
105            }
106
107            int segmentSize = set.size() - start;
108            List<T> segment = new ArrayList<>(segmentSize);
109            for (int k = 0; k < segmentSize; k++) {
110                segment.add(it.next());
111            }
112            arrays[n - 1] = segment;
113
114            return arrays;
115        } else {
116            throw new IllegalArgumentException("n must not be smaller set size!");
117        }
118    }

120    /**
121     * Split the collection into weights.length
122     *
123     * @param set     the colleciton to split
124     * @param weights is an array of weights that will be normalized, so sum can be larger than one, but all should be positive
125     * @return an array of lists each containing a number of objects from the collection, the number determined by the weight relative to total weight
126     */
127    public static <T> List<T>[] split(Collection<T> set, double[] weights) {
128        if (set.size() >= weights.length) {
129            @SuppressWarnings("unchecked")
130            List<T>[] arrays = new List[weights.length];
131
132            double totalweight = 0.0;
133            for (double w : weights)
134                totalweight += w;
135            double[] relativeWeight = new double[weights.length];
136            for (int i = 0; i < weights.length; i++)
137                relativeWeight[i] = weights[i] / totalweight;
138            int[] listSizes = new int[weights.length];
139            for (int i = 0; i < weights.length; i++)
140                listSizes[i] = (int) (weights[i] / totalweight * (double) set.size());
141            int allocated = 0;
142            for (int ls : listSizes)
143                allocated += ls;
144            int remainder = set.size() - allocated;
145
146            listSizes[0] += remainder;
147
148            Iterator<T> it = set.iterator();
149            for (int i = 0; i < listSizes.length; i++) {
150                List<T> segment = new ArrayList<>(listSizes[i]);
151                for (int k = 0; k < listSizes[i]; k++) {
152                    segment.add(it.next());
153                }
154                arrays[i] = segment;
155            }
156            return arrays;
157        } else {
158            throw new IllegalArgumentException("weigths.length must not be smaller than set size!");
159        }
160    }

162    public static <T> List<T>[] split(Collection<T> set, int[] weights) {
163        double[] doubleWeights = new double[weights.length];
164        for(int i=0;i<weights.length;i++){
165            doubleWeights[i] = weights[i];
166        }
167        return split(set,doubleWeights);
168    }

186    public static double sumElements(Collection<Double> values) {
187        double sum = 0;
188        for (Double value : values) {
189           sum += value ;
190        }
191
