020package org.matsim.contrib.pseudosimulation.searchacceleration.datastructures;
021
022import java.util.ArrayList;
023import java.util.List;
024
025import floetteroed.utilities.EmptyIterable;
026
027/**
028 * Stores 0/1 indicator data ("visits") that are indexed by (space, time) key
029 * tuples. The same key may be set multiple times, meaning that one can store
030 * multiple indicators per key. This class is constructed to minimize memory
031 * usage for very sparse data. "Space" is represented by the generic class L
032 * (e.g. a network link).
033 * 
034 * 
035 * @author Gunnar Flötteröd
036 * 
037 * @param <L>
038 *            the space coordinate type
039 */
040public class SpaceTimeIndicators<L> {
041
042        // -------------------- MEMBERS --------------------
043
044        // outer list: time bins; inner list: space references
045        private final List<List<L>> data;
046
047        // -------------------- CONSTRUCTION --------------------
048
049        public SpaceTimeIndicators(final int timeBinCnt) {
050                this.data = new ArrayList<List<L>>(timeBinCnt);
051                for (int bin = 0; bin < timeBinCnt; bin++) {
052                        this.data.add(null);
053                }
054        }
055
056        // -------------------- IMPLEMENTATION --------------------
057
058        public int getTimeBinCnt() {
059                return this.data.size();
060        }
061
062        public Iterable<L> getVisitedSpaceObjects(final int timeBin) {
063                if (this.data.get(timeBin) != null) {
064                        return this.data.get(timeBin);
065                } else {
066                        return new EmptyIterable<L>();
067                }
068        }
069
070        public void visit(final L spaceObj, final int timeBin) {
071                List<L> spaceObjList = this.data.get(timeBin);
072                if (spaceObjList == null) {
073                        spaceObjList = new ArrayList<L>(1);
074                        this.data.set(timeBin, spaceObjList);
075                }
076                spaceObjList.add(spaceObj);
077        }
078}
