001package org.matsim.contrib.pseudosimulation;
002
003import org.matsim.api.core.v01.Id;
004import org.matsim.api.core.v01.Scenario;
005import org.matsim.api.core.v01.population.Person;
006import org.matsim.api.core.v01.population.Plan;
007import org.matsim.core.controler.events.BeforeMobsimEvent;
008import org.matsim.core.controler.events.IterationEndsEvent;
009import org.matsim.core.controler.events.IterationStartsEvent;
010import org.matsim.core.controler.listener.BeforeMobsimListener;
011import org.matsim.core.controler.listener.IterationEndsListener;
012import org.matsim.core.controler.listener.IterationStartsListener;
013
014import javax.inject.Inject;
015import java.util.ArrayList;
016import java.util.HashMap;
017import java.util.List;
018import java.util.Map;
019
020public class QsimPlansStorage implements IterationEndsListener,IterationStartsListener {
021
022        private final MobSimSwitcher mobSimSwitcher;
023        private final Scenario scenario;
024        private Map<Id<Person>,List<Plan>> plansCollection;
025
026        @Inject
027        QsimPlansStorage(MobSimSwitcher mobSimSwitcher, Scenario scenario){
028                this.mobSimSwitcher = mobSimSwitcher;
029                this.scenario = scenario;
030        }
031
032        /*
033        at the end of a qsim iteration, write the population to cold storage
034         */
035        @Override
036        public void notifyIterationEnds(IterationEndsEvent event) {
037                if (!mobSimSwitcher.isQSimIteration())
038                        return;
039                plansCollection = new HashMap<>();
040                for (Person person : scenario.getPopulation().getPersons().values()) {
041                        ArrayList<Plan> plans = new ArrayList<>();
042                        plans.addAll(person.getPlans());
043                        plansCollection.put(person.getId(), plans);
044                }
045        }
046
047        @Override
048        public void notifyIterationStarts(IterationStartsEvent event) {
049                if (!mobSimSwitcher.isQSimIteration())
050                        return;
051                for (Person person : scenario.getPopulation().getPersons().values()) {
052                        person.getPlans().clear();
053                        for (Plan plan : plansCollection.get(person.getId())) {
054                                person.addPlan(plan);
055                        }
056                }
057        }
058}
