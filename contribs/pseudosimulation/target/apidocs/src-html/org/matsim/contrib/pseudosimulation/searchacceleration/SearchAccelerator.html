176        @Override
177        public void notifyIterationEnds(final IterationEndsEvent event) {
178
179                this.log("iteration " + event.getIteration() + " ends");
180
181                /*
182                 * OBTAIN DATA FROM MOST RECENT PHYSICAL NETWORK LOADING
183                 * 
184                 * Receive information about what happened in the network during the most recent
185                 * real network loading.
186                 */
187                final Map<Id<Person>, SpaceTimeIndicators<Id<Link>>> driverId2physicalLinkUsage = this.physicalMobsimUsageListener
188                                .getAndClearIndicators();
189                this.log("observed " + driverId2physicalLinkUsage.size() + " drivers in physical mobsim");
190
191                /*
192                 * HYPOTHETICAL REPLANNING AND PSEUDOSIM
193                 * 
194                 * Let every agent re-plan once. Execute all new plans in the pseudoSim.
195                 * Memorize the new plans. Make sure that the the actual choice sets and
196                 * currently chosen plans are NOT affected by this by immediately re-setting to
197                 * the original population state once the re-planning has been implemented.
198                 */
199                this.log("hypothetical replanning...");
200
201                // Memorize the original state.
202
203                final PopulationState originalPopulationState = new PopulationState(
204                                this.services.getScenario().getPopulation());
205
206                // Re-plan. The population enters now the hypothetical state.
207
208                this.setWeightOfHypotheticalReplanning(0.0);
209
210                Map<Id<Person>, SpaceTimeIndicators<Id<Link>>> driverId2pSimLinkUsage = null;
211                final int pSimItCnt = ConfigUtils.addOrGetModule(this.services.getConfig(), AccelerationConfigGroup.class)
212                                .getPseudoSimIterations();
213
214                for (int pSimIt = 0; pSimIt < pSimItCnt; pSimIt++) {
215
216                        this.log("pSim iteration " + (pSimIt + 1) + " / " + pSimItCnt);
217
218                        // re-plan and extract selected plan
219
220                        this.services.getStrategyManager().run(this.services.getScenario().getPopulation(), this.replanningContext);
221                        final Collection<Plan> selectedHypotheticalPlans = new ArrayList<>(
222                                        this.services.getScenario().getPopulation().getPersons().size());
223                        for (Person person : this.services.getScenario().getPopulation().getPersons().values()) {
224                                selectedHypotheticalPlans.add(person.getSelectedPlan());
225                        }
226
227                        // execute in pseudoSim (TODO don't need to track indicators each time)
228
229                        final EventsManager eventsManager = EventsUtils.createEventsManager();
230                        final LinkUsageListener pSimLinkUsageListener = new LinkUsageListener(this.timeDiscr);
231                        eventsManager.addHandler(pSimLinkUsageListener);
232                        final PSim pSim = new PSim(this.services.getScenario(), eventsManager, selectedHypotheticalPlans,
233                                        this.services.getLinkTravelTimes());
234                        pSim.run();
235                        driverId2pSimLinkUsage = pSimLinkUsageListener.getAndClearIndicators();
236                        this.log("identified " + driverId2pSimLinkUsage.size() + " drivers in pSim");
237                }
238
239                /*
240                 * FIXME There is no scoring here. Will most likely need to submit all of this to the
241                 * MATSim loop (like in the pSim), meaning that much of the functionality of this handler
242                 * is only used at the end of a (set of) pSim iterations.
243                 */
244
245                this.hypotheticalPopulationState = new PopulationState(this.services.getScenario().getPopulation());
246                this.setWeightOfHypotheticalReplanning(1e9);
247
248                // Return (for now) to original population state.
249
250                originalPopulationState.set(this.services.getScenario().getPopulation());
251
252                /*
253                 * DECIDE WHO GETS TO RE-PLAN.
254                 * 
255                 * At this point, one has (i) the link usage statistics from the previous MATSim
256                 * network loading (vehId2physLinkUsage), and (ii) the hypothetical link usage
257                 * statistics that would result from a 100% re-planning rate if network
258                 * congestion did not change (vehId2pSimLinkUsage).
259                 * 
260                 * Now solve an optimization problem that aims at balancing simulation
261                 * advancement (changing link usage patterns) and simulation stabilization
262                 * (keeping link usage patterns as they are). Most of the code below prepares
263                 * the (heuristic) solution of this problem.
264                 * 
265                 */
266
267                final ReplannerIdentifier replannerIdentifier = new ReplannerIdentifier(this.replanningParameters,
268                                this.timeDiscr, event.getIteration(), driverId2physicalLinkUsage, driverId2pSimLinkUsage,
269                                this.services.getScenario().getPopulation(), this.services.getLinkTravelTimes(),
270                                ConfigUtils.addOrGetModule(this.services.getConfig(), AccelerationConfigGroup.class).getAccelerate());
271                this.replanners = replannerIdentifier.drawReplanners();
272
273                final List<Double> bootstrap;
274                if (this.bootstrapReplications > 0) {
275                        bootstrap = replannerIdentifier.bootstrap(this.bootstrapReplications);
276                } else {
277                        bootstrap = null;
278                }
279
280                /*
281                 * 
282                 * // Extract basic statistics.
283                 * 
284                 * final double meanLambda =
285                 * this.replanningParameters.getMeanLambda(event.getIteration()); final double
286                 * delta = this.replanningParameters.getDelta(event.getIteration());
287                 * 
288                 * final DynamicData<Id<Link>> currentTotalCounts =
289                 * CountIndicatorUtils.newUnweightedCounts(
290                 * this.physicalMobsimUsageListener.getTimeDiscretization(),
291                 * driverId2physicalLinkUsage.values()); final DynamicData<Id<Link>>
292                 * currentWeightedCounts = CountIndicatorUtils.newWeightedCounts(
293                 * this.physicalMobsimUsageListener.getTimeDiscretization(),
294                 * driverId2physicalLinkUsage.values(), currentTotalCounts,
295                 * this.replanningParameters); final DynamicData<Id<Link>>
296                 * upcomingWeightedCounts = CountIndicatorUtils.newWeightedCounts(
297                 * pSimLinkUsageListener.getTimeDiscretization(),
298                 * driverId2pSimLinkUsage.values(), currentTotalCounts,
299                 * this.replanningParameters);
300                 * 
301                 * final double sumOfCurrentWeightedCounts2 =
302                 * CountIndicatorUtils.sumOfEntries2(currentWeightedCounts); if
303                 * (sumOfCurrentWeightedCounts2 < 1e-6) { throw new
304                 * RuntimeException("There is no (weighted) traffic on the network: " +
305                 * sumOfCurrentWeightedCounts2); } final double sumOfWeightedCountDifferences2 =
306                 * CountIndicatorUtils.sumOfDifferences2(currentWeightedCounts,
307                 * upcomingWeightedCounts); final double w = meanLambda / (1.0 - meanLambda) *
308                 * (sumOfWeightedCountDifferences2 + delta) / sumOfCurrentWeightedCounts2;
309                 * 
310                 * // Initialize score residuals.
311                 * 
312                 * final DynamicData<Id<Link>> interactionResiduals = CountIndicatorUtils
313                 * .newWeightedDifference(upcomingWeightedCounts, currentWeightedCounts,
314                 * meanLambda);
315                 * 
316                 * final DynamicData<Id<Link>> inertiaResiduals = new
317                 * DynamicData<>(currentWeightedCounts.getStartTime_s(),
318                 * currentWeightedCounts.getBinSize_s(), currentWeightedCounts.getBinCnt()); for
319                 * (Id<Link> locObj : currentWeightedCounts.keySet()) { for (int bin = 0; bin <
320                 * currentWeightedCounts.getBinCnt(); bin++) { inertiaResiduals.put(locObj, bin,
321                 * (1.0 - meanLambda) * currentWeightedCounts.getBinValue(locObj, bin)); } }
322                 * double regularizationResidual = meanLambda * sumOfCurrentWeightedCounts2;
323                 * 
324                 * // Go through all vehicles and decide which driver gets to re-plan.
325                 * 
326                 * this.log("identify replanners");
327                 * 
328                 * this.replanners = new LinkedHashSet<>(); double totalScoreChange = 0.0;
329                 * 
330                 * final List<Id<Person>> allPersonIdsShuffled = new
331                 * ArrayList<>(driverId2physicalLinkUsage.keySet());
332                 * Collections.shuffle(allPersonIdsShuffled);
333                 * 
334                 * for (Id<Person> driverId : allPersonIdsShuffled) {
335                 * 
336                 * this.log("  driver " + driverId);
337                 * 
338                 * final ScoreUpdater<Id<Link>> scoreUpdater = new
339                 * ScoreUpdater<>(driverId2physicalLinkUsage.get(driverId),
340                 * driverId2pSimLinkUsage.get(driverId), meanLambda, currentWeightedCounts,
341                 * sumOfCurrentWeightedCounts2, w, delta, interactionResiduals,
342                 * inertiaResiduals, regularizationResidual, this.replanningParameters,
343                 * currentTotalCounts);
344                 * 
345                 * this.log("    scoreChange if one  = " + (totalScoreChange +
346                 * scoreUpdater.getScoreChangeIfOne())); this.log("    scoreChange if zero = " +
347                 * (totalScoreChange + scoreUpdater.getScoreChangeIfZero()));
348                 * 
349                 * final double newLambda; if (scoreUpdater.getScoreChangeIfOne() <=
350                 * scoreUpdater.getScoreChangeIfZero()) { newLambda = 1.0;
351                 * this.replanners.add(driverId); this.log("  => is a replanner");
352                 * totalScoreChange += scoreUpdater.getScoreChangeIfOne();
353                 * this.personId2lastReplanIteration.put(driverId, event.getIteration()); } else
354                 * { newLambda = 0.0; this.log("  => is NOT a replanner"); totalScoreChange +=
355                 * scoreUpdater.getScoreChangeIfZero(); } this.log("  totalScoreChange = " +
356                 * totalScoreChange);
357                 * 
358                 * scoreUpdater.updateResiduals(newLambda); // Interaction- and inertiaResiduals
359                 * are updated by reference. regularizationResidual =
360                 * scoreUpdater.getUpdatedRegularizationResidual(); }
361                 * 
362                 */
363
364                // this.log("Effective replanning rate = " + (int) (((double) 100 *
365                // this.replanners.size())
366                // / this.services.getScenario().getPopulation().getPersons().size()) + "
367                // percent");
368                //
369                // final int[] referenceReplanLagHist = new int[event.getIteration() + 1];
370                // for (int i = 0; i <= event.getIteration(); i++) {
371                // referenceReplanLagHist[i] = (int) Math.round(Math.pow(1.0 - meanLambda, i) *
372                // meanLambda
373                // * this.services.getScenario().getPopulation().getPersons().size());
374                // }
375                //
376                // final int[] actualReplanLagHist = new int[event.getIteration() + 1];
377                // for (Id<Person> personId :
378                // this.services.getScenario().getPopulation().getPersons().keySet()) {
379                // final int val;
380                // if (this.personId2lastReplanIteration.containsKey(personId)) {
381                // val = this.personId2lastReplanIteration.get(personId);
382                // } else {
383                // val = 0;
384                // }
385                // actualReplanLagHist[event.getIteration() - val]++;
386                // }
387
388                // this.log("\treplanLag\treference\tactual");
389                // for (int i = 0; i <= event.getIteration(); i++) {
390                // this.log("\t" + i + "\t" + referenceReplanLagHist[i] + "\t" +
391                // actualReplanLagHist[i]);
392                // }
393                //
394
395                this.analyzer.analyze(this.services.getScenario().getPopulation().getPersons().keySet(),
396                                driverId2physicalLinkUsage, driverId2pSimLinkUsage, this.replanners, event.getIteration(), bootstrap,
397                                replannerIdentifier.getUniformReplanningObjectiveFunctionValue(),
398                                replannerIdentifier.getShareOfScoreImprovingReplanners(),
399                                replannerIdentifier.getFinalObjectiveFunctionValue(), replannerIdentifier.getUniformityExcess(),
400                                this.services.getLinkTravelTimes());
401
402                // this.log("OPTIMIZED MINUS UNIFORM REALIZED DELTA N");
403                // for (Double val : analyzer.diffList) {
404                // this.log(val);
405                // }
406                // System.exit(0);
407
408                this.statsWriter.writeToFile(this.analyzer);
409        }

411        // -------------------- REPLANNING FUNCTIONALITY --------------------
412
413        public void replan(final HasPlansAndId<Plan, Person> person) {
414                if (this.replanners.contains(person.getId())) {
415                        // TODO This replaces the entire choice set and not just the selected plan.
416                        this.hypotheticalPopulationState.set(person);
417                }
418        }

422        private void setWeightOfHypotheticalReplanning(final double weight) {
423                final StrategyManager strategyManager = this.services.getStrategyManager();
424                for (GenericPlanStrategy<Plan, Person> strategy : strategyManager.getStrategies(null)) {
425                        if (strategy instanceof AcceptIntendedReplanningStrategy) {
426                                strategyManager.changeWeightOfStrategy(strategy, null, weight);
427                        }
428                }
429        }
