139        private void setWeightOfHypotheticalReplanning(final double weight) {
140                final StrategyManager strategyManager = this.services.getStrategyManager();
141                for (GenericPlanStrategy<Plan, Person> strategy : strategyManager.getStrategies(null)) {
142                        if (strategy instanceof AcceptIntendedReplanningStrategy) {
143                                strategyManager.changeWeightOfStrategy(strategy, null, weight);
144                        }
145                }
146        }

197        // --------------- IMPLEMENTATION OF IterationEndsListener ---------------
198
199        private PopulationState lastPhysicalPopulationState = null;
200        private Map<Id<Person>, SpaceTimeIndicators<Id<Link>>> lastPhysicalLinkUsages = null;
201
202        private int pseudoSimIterationCnt = 0;
203
204        // for testing
205        private boolean nextMobsimIsExpectedToBePhysical = true;
206
207        @Override
208        public void notifyIterationEnds(final IterationEndsEvent event) {
209
210//              for (Person person : this.services.getScenario().getPopulation().getPersons().values()) {
211//                      if (person.getPlans().size() > 1) {
212//                              throw new RuntimeException("person " + person.getId() + " has " + person.getPlans().size() + " plans");
213//                      }
214//              }
215
216                if (this.mobsimSwitcher.isQSimIteration()) {
217                        this.log("physical mobsim run in iteration " + event.getIteration() + " ends");
218                        if (!this.nextMobsimIsExpectedToBePhysical) {
219                                throw new RuntimeException("Did not expect a physical mobsim run!");
220                        }
221                        this.lastPhysicalPopulationState = new PopulationState(this.services.getScenario().getPopulation());
222                        this.lastPhysicalLinkUsages = this.matsimMobsimUsageListener.getAndClearIndicators();
223                        this.pseudoSimIterationCnt = 0;
224                } else {
225                        this.log("pseudoSim run in iteration " + event.getIteration() + " ends");
226                        if (this.nextMobsimIsExpectedToBePhysical) {
227                                throw new RuntimeException("Expected a physical mobsim run!");
228                        }
229                        this.pseudoSimIterationCnt++;
230                }
231
232                if (this.pseudoSimIterationCnt == (ConfigUtils.addOrGetModule(this.services.getConfig(), PSimConfigGroup.class)
233                                .getIterationsPerCycle() - 1)) {
234
235                        /*
236                         * Extract, for each agent, the expected (hypothetical) score change.
237                         */
238                        double deltaScoreTotal = 0.0;
239                        final Map<Id<Person>, Double> personId2deltaScore = new LinkedHashMap<>();
240                        for (Person person : this.services.getScenario().getPopulation().getPersons().values()) {
241                                final double oldScore = this.lastPhysicalPopulationState.getSelectedPlan(person.getId()).getScore();
242                                final double newScore = person.getSelectedPlan().getScore();
243                                final double deltaScore = newScore - oldScore;
244                                personId2deltaScore.put(person.getId(), deltaScore);
245                                deltaScoreTotal += deltaScore;
246                        }
247
248                        /*
249                         * Extract hypothetical selected plans.
250                         */
251
252                        final Collection<Plan> selectedHypotheticalPlans = new ArrayList<>(
253                                        this.services.getScenario().getPopulation().getPersons().size());
254                        for (Person person : this.services.getScenario().getPopulation().getPersons().values()) {
255                                selectedHypotheticalPlans.add(person.getSelectedPlan());
256                        }
257
258                        /*
259                         * Execute one pSim with the full population.
260                         */
261
262                        final EventsManager eventsManager = EventsUtils.createEventsManager();
263                        final LinkUsageListener pSimLinkUsageListener = new LinkUsageListener(this.timeDiscr);
264                        eventsManager.addHandler(pSimLinkUsageListener);
265                        final PSim pSim = new PSim(this.services.getScenario(), eventsManager, selectedHypotheticalPlans,
266                                        this.services.getLinkTravelTimes());
267                        pSim.run();
268                        final Map<Id<Person>, SpaceTimeIndicators<Id<Link>>> lastPseudoSimLinkUsages = pSimLinkUsageListener
269                                        .getAndClearIndicators();
270
271                        this.log("observed " + this.lastPhysicalLinkUsages.size() + " physical drivers");
272                        this.log("observed " + lastPseudoSimLinkUsages.size() + " drivers in pSim");
273
274                        /*
275                         * Memorize the most recent hypothetical population state re-set the population
276                         * to its most recent physical state.
277                         */
278
279                        this.hypotheticalPopulationState = new PopulationState(this.services.getScenario().getPopulation());
280                        this.lastPhysicalPopulationState.set(this.services.getScenario().getPopulation());
281
282                        /*
283                         * DECIDE WHO GETS TO RE-PLAN.
284                         * 
285                         * At this point, one has (i) the link usage statistics from the last physical
286                         * MATSim network loading (lastPhysicalLinkUsages), and (ii) the hypothetical
287                         * link usage statistics that would result from a 100% re-planning rate if
288                         * network congestion did not change (lastPseudoSimLinkUsages).
289                         * 
290                         * Now solve an optimization problem that aims at balancing simulation
291                         * advancement (changing link usage patterns) and simulation stabilization
292                         * (keeping link usage patterns as they are). Most of the code below prepares
293                         * the (heuristic) solution of this problem.
294                         * 
295                         */
296
297                        final ReplannerIdentifier replannerIdentifier = new ReplannerIdentifier(this.replanningParameters,
298                                        this.timeDiscr, event.getIteration(), this.lastPhysicalLinkUsages, lastPseudoSimLinkUsages,
299                                        this.services.getScenario().getPopulation(), this.services.getLinkTravelTimes(), ConfigUtils
300                                                        .addOrGetModule(this.services.getConfig(), AccelerationConfigGroup.class).getAccelerate(),
301                                        personId2deltaScore, deltaScoreTotal);
302                        this.replanners = replannerIdentifier.drawReplanners();
303
304                        final List<Double> bootstrap;
305                        if (this.bootstrapReplications > 0) {
306                                bootstrap = replannerIdentifier.bootstrap(this.bootstrapReplications);
307                        } else {
308                                bootstrap = null;
309                        }
310
311                        this.analyzer.analyze(this.services.getScenario().getPopulation().getPersons().keySet(),
312                                        this.lastPhysicalLinkUsages, lastPseudoSimLinkUsages, this.replanners, event.getIteration(),
313                                        bootstrap, replannerIdentifier.getUniformReplanningObjectiveFunctionValue(),
314                                        replannerIdentifier.getShareOfScoreImprovingReplanners(),
315                                        replannerIdentifier.getFinalObjectiveFunctionValue(), replannerIdentifier.getUniformityExcess(),
316                                        this.services.getLinkTravelTimes(),
317                                        replannerIdentifier.getExpectedUniformSamplingObjectiveFunctionValue(),
318                                        replannerIdentifier.getSumOfWeightedCountDifferences2(),
319                                        replannerIdentifier.getSumOfUnweightedCountDifferences2());
320
321                        this.statsWriter.writeToFile(this.analyzer);
322
323                        this.nextMobsimIsExpectedToBePhysical = true;
324                        this.setWeightOfHypotheticalReplanning(1e9);
325
326                } else {
327
328                        this.nextMobsimIsExpectedToBePhysical = false;
329                        this.setWeightOfHypotheticalReplanning(0);
330
331                }
332        }
333
334        // -------------------- REPLANNING FUNCTIONALITY --------------------
335
336        public void replan(final HasPlansAndId<Plan, Person> person) {
337                if (this.replanners.contains(person.getId())) {
338                        // This replaces the entire choice set and not just the selected plan. Why not.
339                        this.hypotheticalPopulationState.set(person);
340                }
341        }
