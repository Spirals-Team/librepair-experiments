053public class RouteAndTimeCleaner {
054
055        private static double randomTime_s() {
056                return Math.random() * 3600 * 24;
057        }
058
059        public static void keepOnlySelected(final Scenario scenario) {
060                for (Person person : scenario.getPopulation().getPersons().values()) {
061                        Plan plan = person.getSelectedPlan();
062                        person.getPlans().clear();
063                        person.addPlan(plan);
064                        person.setSelectedPlan(plan);
065                }
066        }
067
068        public static void removeTimeAndRoute(final Scenario scenario, final boolean removeTime,
069                        final boolean removeRoute) {
070                for (Person person : scenario.getPopulation().getPersons().values()) {
071                        
072                        Plan plan = person.getSelectedPlan();
073
074                        final int actCnt = (plan.getPlanElements().size() + 1) / 2;
075                        double actLength_s = 3600 * 24 / actCnt;
076                        double nextActStartTime_s = 0;
077                        
078                        for (PlanElement pe : plan.getPlanElements()) {
079                                if (pe instanceof Activity) {
080                                        Activity act = (Activity) pe;
081                                        if (removeTime) {
082                                                act.setStartTime(nextActStartTime_s);
083                                                act.setMaximumDuration(actLength_s);
084                                                nextActStartTime_s += actLength_s;
085                                                act.setEndTime(nextActStartTime_s);
086                                        }
087                                } else if (pe instanceof Leg) {
088                                        Leg leg = (Leg) pe;
089                                        if (removeTime) {
090                                                leg.setDepartureTime(nextActStartTime_s);
091                                                leg.setTravelTime(0);
092                                        }
093                                        if (removeRoute) {
094                                                leg.setRoute(null);
095                                        }
096                                }
097                        }
098                }
099        }
100
101        static void keepOnlyCarUsers(Population pop) {
102
103                Set<Id<Person>> noCarUsersIds = new LinkedHashSet<>(pop.getPersons().keySet());
104                for (Person person : pop.getPersons().values()) {
105                        final Plan plan = person.getSelectedPlan();
106                        for (PlanElement pe : plan.getPlanElements()) {
107                                if (pe instanceof Leg) {
108                                        Leg leg = (Leg) pe;
109                                        if ("car".equals(leg.getMode())) {
110                                                noCarUsersIds.remove(person.getId());
111                                        }
112                                }
113                        }
114                }
115
116                for (Id<Person> nonCarUserId : noCarUsersIds) {
117                        pop.getPersons().remove(nonCarUserId);
118                }
119        }
120
121        static void keepOnlyFraction(Population pop, final double frac) {
122
123                final Set<Id<Person>> idsToRemove = new LinkedHashSet<>();
124                for (Id<Person> removePersonId : new FractionalIterable<>(pop.getPersons().keySet(), 1.0 - frac)) {
125                        idsToRemove.add(removePersonId);
126                }
127
128                for (Id<Person> removeId : idsToRemove) {
129                        pop.getPersons().remove(removeId);
130                }
131        }
132
133        public static void main(String[] args) {
134
135                System.out.println("exiting");
136                System.exit(0);
137
138                String path = "/Users/GunnarF/NoBackup/data-workspace/searchacceleration" + "/rerun-2015-11-23a_No_Toll_large/";
139                Config config = ConfigUtils.loadConfig(path + "matsim-config.xml");
140                config.plans().setInputFile(path + "400.plans.xml.gz");
141
142                Scenario scenario = ScenarioUtils.loadScenario(config);
143
144                keepOnlySelected(scenario);
145                keepOnlyFraction(scenario.getPopulation(), 0.2); // from 5 to 1 percent
146                keepOnlyCarUsers(scenario.getPopulation()); // of the maintained fraction!
147                removeTimeAndRoute(scenario, true, true);
148
149                PopulationWriter writer = new PopulationWriter(scenario.getPopulation());
150                writer.write(path + "initial-plans.1pct.xml");
151        }
