002package org.matsim.contrib.pseudosimulation.distributed;
003
004import java.io.Serializable;
005import java.util.ArrayList;
006import java.util.List;
007import java.util.TreeSet;
008
009import org.matsim.api.core.v01.Id;
010import org.matsim.api.core.v01.population.Person;
011import org.matsim.api.core.v01.population.Plan;
012import org.matsim.core.population.PersonUtils;
013import org.matsim.core.population.PopulationUtils;
014
015public class PersonSerializable implements Serializable {
016    protected List<PlanSerializable> plans = new ArrayList<>(5);
017
018    public PersonSerializable(Person p) {
019        this.id = p.getId().toString();
020        Person person = p;
021
022        for (Plan plan : person.getPlans()) {
023            PlanSerializable planSerializable = new PlanSerializable(plan);
024            plans.add(planSerializable);
025            if (plan.equals(person.getSelectedPlan()))
026                this.selectedPlan = planSerializable;
027        }
028    }
029
030    protected String id;
031
032    PlanSerializable selectedPlan = null;
033    private TreeSet<String> travelcards = null;
034
035    private Boolean isEmployed;
036
037    public Person getPerson() {
038        Person person = PopulationUtils.getFactory().createPerson(Id.createPersonId(id));
039
040        for (PlanSerializable planSer : plans) {
041            Plan plan = planSer.getPlan(person);
042            person.addPlan(plan);
043            if(planSer.equals(selectedPlan))
044                person.setSelectedPlan(plan);
045
046        }
047        return person;
048    }
049}
