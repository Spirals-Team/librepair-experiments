055        public static <L> DynamicData<L> newWeightedCounts(final TimeDiscretization timeDiscr,
056                        final Collection<SpaceTimeIndicators<L>> allIndicators, final ReplanningParameterContainer replParams,
057                        final TravelTime travelTimes) {
058                final DynamicData<L> result = new DynamicData<L>(timeDiscr);
059                for (SpaceTimeIndicators<L> indicators : allIndicators) {
060                        for (int bin = 0; bin < indicators.getTimeBinCnt(); bin++) {
061                                for (L locObj : indicators.getVisitedSpaceObjects(bin)) {
062                                        result.add(locObj, bin, replParams.getWeight(locObj, bin, travelTimes));
063                                }
064                        }
065                }
066                return result;
067        }
068
069        public static <L> double sumOfEntries2(final DynamicData<L> data) {
070                double result = 0.0;
071                for (L locObj : data.keySet()) {
072                        for (int bin = 0; bin < data.getBinCnt(); bin++) {
073                                final double val = data.getBinValue(locObj, bin);
074                                result += val * val;
075                        }
076                }
077                return result;
078        }
079
080        public static <L> double sumOfDifferences2(final DynamicData<L> counts1, final DynamicData<L> counts2) {
081                if (counts1.getBinCnt() != counts2.getBinCnt()) {
082                        throw new RuntimeException(
083                                        "counts1 has " + counts1.getBinCnt() + " bins, but counts2 has " + counts2.getBinCnt() + " bins.");
084                }
085                double result = 0.0;
086                for (L locObj : SetUtils.union(counts1.keySet(), counts2.keySet())) {
087                        for (int bin = 0; bin < counts1.getBinCnt(); bin++) {
088                                final double diff = counts1.getBinValue(locObj, bin) - counts2.getBinValue(locObj, bin);
089                                result += diff * diff;
090                        }
091                }
092                return result;
093        }
094
095        public static <L> DynamicData<L> newWeightedDifference(final DynamicData<L> data1, final DynamicData<L> data2,
096                        final double weight) {
097                if (data1.getBinCnt() != data2.getBinCnt()) {
098                        throw new RuntimeException("currentWeightedCounts has " + data1.getBinCnt()
099                                        + " bins; newWeightedCounts has " + data2.getBinCnt() + " bins.");
100                }
101                final DynamicData<L> result = new DynamicData<L>(data1.getStartTime_s(), data1.getBinSize_s(),
102                                data1.getBinCnt());
103                for (L locObj : SetUtils.union(data1.keySet(), data2.keySet())) {
104                        for (int bin = 0; bin < data1.getBinCnt(); bin++) {
105                                result.put(locObj, bin, weight * (data1.getBinValue(locObj, bin) - data2.getBinValue(locObj, bin)));
106                        }
107                }
108                return result;
109        }
