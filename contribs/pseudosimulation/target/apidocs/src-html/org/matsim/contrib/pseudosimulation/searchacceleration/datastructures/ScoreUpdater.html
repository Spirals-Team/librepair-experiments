020package org.matsim.contrib.pseudosimulation.searchacceleration.datastructures;
021
022import java.util.Map;
023
024import org.matsim.contrib.pseudosimulation.searchacceleration.ReplanningParameterContainer;
025import org.matsim.core.router.util.TravelTime;
026
027import floetteroed.utilities.DynamicData;
028import floetteroed.utilities.Tuple;
029
030/**
031 * The "score" this class refers to is the anticipated change of the search
032 * acceleration objective function resulting from setting a single agent's
033 * (possibly space-weighted) 0/1 re-planning indicator.
034 * 
035 * Implements the score used in the greedy heuristic of Merz, P. and Freisleben,
036 * B. (2002). "Greedy and local search heuristics for unconstrained binary
037 * quadratic programming." Journal of Heuristics 8:197–213.
038 * 
039 * @author Gunnar Flötteröd
040 * 
041 * @param L
042 *            the space coordinate type
043 *
044 */
045public class ScoreUpdater<L> {
046
047        // -------------------- MEMBERS --------------------
048
049        private final SpaceTimeCounts<L> currentIndividualWeightedCounts;
050
051        private final SpaceTimeCounts<L> individualWeightedChanges;
052
053        private final DynamicData<L> currentWeightedTotalCounts;
054
055        private final DynamicData<L> interactionResiduals;
056
057        private final double deltaScoreIndividual;
058
059        // private final DynamicData<L> inertiaResiduals;
060
061        private double regularizationResidual;
062
063        private final double scoreChangeIfZero;
064
065        private final double scoreChangeIfOne;
066
067        private boolean residualsUpdated = false;
068
069        // -------------------- CONSTRUCTION --------------------
070
071        public ScoreUpdater(final SpaceTimeIndicators<L> currentIndicators, final SpaceTimeIndicators<L> upcomingIndicators,
072                        final double meanLambda, final DynamicData<L> currentWeightedTotalCounts, final double w,
073                        final double delta, final DynamicData<L> interactionResiduals,
074                        // final DynamicData<L> inertiaResiduals,
075                        final double regularizationResidual, final ReplanningParameterContainer replParams,
076                        final TravelTime travelTimes,
077                        // final double numberOfRelevantLinkTimeSlots,
078                        final double deltaScoreIndividual, final double deltaScoreTotal) {
079
080                this.currentWeightedTotalCounts = currentWeightedTotalCounts;
081                this.interactionResiduals = interactionResiduals;
082                // this.inertiaResiduals = inertiaResiduals;
083                this.regularizationResidual = regularizationResidual;
084                this.deltaScoreIndividual = deltaScoreIndividual;
085
086                /*
087                 * One has to go beyond 0/1 indicator arithmetics in the following because the
088                 * same vehicle may enter the same link multiple times during one time bin.
089                 */
090
091                this.currentIndividualWeightedCounts = new SpaceTimeCounts<>(currentIndicators, replParams, travelTimes);
092                this.individualWeightedChanges = new SpaceTimeCounts<>(upcomingIndicators, replParams, travelTimes);
093                this.individualWeightedChanges.subtract(this.currentIndividualWeightedCounts);
094
095                // Update the residuals.
096
097                for (Map.Entry<Tuple<L, Integer>, Double> entry : this.individualWeightedChanges.entriesView()) {
098                        final L spaceObj = entry.getKey().getA();
099                        final int timeBin = entry.getKey().getB();
100                        final double weightedIndividualChange = entry.getValue();
101                        this.interactionResiduals.add(spaceObj, timeBin, -meanLambda * weightedIndividualChange);
102                }
103
104                this.regularizationResidual -= meanLambda * deltaScoreIndividual;
105                // for (Map.Entry<Tuple<L, Integer>, Double> entry :
106                // this.currentIndividualWeightedCounts.entriesView()) {
107                // final L spaceObj = entry.getKey().getA();
108                // final Integer timeBin = entry.getKey().getB();
109                // final double weightedCurrentIndividualCount = entry.getValue();
110                // // this.inertiaResiduals.add(spaceObj, timeBin, -(1.0 - meanLambda) *
111                // // weightedCurrentIndividualCount);
112                // // this.regularizationResiduals.add(spaceObj, timeBin, -meanLambda *
113                // weightedCurrentIndividualCount);
114                // }
115
116                // Compute individual score terms.
117
118                double sumOfWeightedIndividualChanges2 = 0.0;
119                double sumOfWeightedIndividualChangesTimesInteractionResiduals = 0.0;
120
121                for (Map.Entry<Tuple<L, Integer>, Double> entry : this.individualWeightedChanges.entriesView()) {
122                        final L spaceObj = entry.getKey().getA();
123                        final int timeBin = entry.getKey().getB();
124                        final double weightedIndividualChange = entry.getValue();
125                        sumOfWeightedIndividualChanges2 += weightedIndividualChange * weightedIndividualChange;
126                        sumOfWeightedIndividualChangesTimesInteractionResiduals += weightedIndividualChange
127                                        * this.interactionResiduals.getBinValue(spaceObj, timeBin);
128                }
129
130                // double sumOfLinkShares2 = 0;
131                // double
132                // sumOfWeightedCurrentIndividualCountsTimesInertiaResidualsOverWeightedSquareCounts
133                // = 0;
134                // double
135                // sumOfWeightedCurrentIndividualCountsTimesRegularizationResidualsOverWeigthedSquareCounts
136                // = 0;
137                // double sumOfWeightedCurrentIndividualCounts2 = 0.0;
138                // double sumOfWeightedCurrentIndividualCountsTimesWeightedCurrentTotalCounts =
139                // 0.0;
140                // double sumOfWeightedCurrentIndividualCountsTimesInertiaResiduals = 0.0;
141
142                for (Map.Entry<Tuple<L, Integer>, Double> entry : this.currentIndividualWeightedCounts.entriesView()) {
143                        final L spaceObj = entry.getKey().getA();
144                        final Integer timeBin = entry.getKey().getB();
145                        final double weightedCurrentIndividualCount = entry.getValue();
146                        // sumOfWeightedCurrentIndividualCounts2 += weightedCurrentIndividualCount *
147                        // weightedCurrentIndividualCount;
148                        // sumOfWeightedCurrentIndividualCountsTimesWeightedCurrentTotalCounts +=
149                        // weightedCurrentIndividualCount
150                        // * this.currentWeightedTotalCounts.getBinValue(spaceObj, timeBin);
151                        // sumOfWeightedCurrentIndividualCountsTimesInertiaResiduals +=
152                        // weightedCurrentIndividualCount
153                        // * this.inertiaResiduals.getBinValue(spaceObj, timeBin);
154                        final double weightedCurrentTotalCount = this.currentWeightedTotalCounts.getBinValue(spaceObj, timeBin);
155                        if (weightedCurrentTotalCount > 1e-6) {
156                                // sumOfLinkShares2 += Math.pow(weightedCurrentIndividualCount /
157                                // weightedCurrentTotalCount, 2.0);
158                                // sumOfWeightedCurrentIndividualCountsTimesInertiaResidualsOverWeightedSquareCounts
159                                // += weightedCurrentIndividualCount
160                                // * this.inertiaResiduals.getBinValue(spaceObj, timeBin) /
161                                // weightedCurrentTotalCount
162                                // / weightedCurrentTotalCount;
163                                // sumOfWeightedCurrentIndividualCountsTimesRegularizationResidualsOverWeigthedSquareCounts
164                                // += weightedCurrentIndividualCount
165                                // * this.regularizationResiduals.getBinValue(spaceObj, timeBin) /
166                                // weightedCurrentTotalCount
167                                // / weightedCurrentTotalCount;
168                        }
169                }
170
171                // Compose the actual score change.
172
173                this.factor1 = sumOfWeightedIndividualChanges2 + delta * deltaScoreIndividual * deltaScoreIndividual;
174                this.factor2 = 2.0 * sumOfWeightedIndividualChangesTimesInteractionResiduals - w * deltaScoreIndividual
175                                + 2 * delta * deltaScoreIndividual * regularizationResidual;
176                this.meanLambda = meanLambda;
177
178                this.scoreChangeIfOne = (1.0 - meanLambda * meanLambda) * this.factor1 + (1.0 - meanLambda) * this.factor2;
179                this.scoreChangeIfZero = (0.0 - meanLambda * meanLambda) * this.factor1 + (0.0 - meanLambda) * this.factor2;
180        }
181
182        private final double factor1;
183        private final double factor2;
184        private final double meanLambda;
185
186        public double getScoreChange(final double newLambda) {
187                return (newLambda * newLambda - this.meanLambda * this.meanLambda) * this.factor1
188                                + (newLambda - this.meanLambda) * this.factor2;
189        }
190
191        // -------------------- GETTERS --------------------
192
193        public double getUpdatedRegularizationResidual() {
194                if (!this.residualsUpdated) {
195                        throw new RuntimeException("Residuals have not yet updated.");
196                }
197                return this.regularizationResidual;
198        }
199
200        public double getScoreChangeIfOne() {
201                return this.scoreChangeIfOne;
202        }
203
204        public double getScoreChangeIfZero() {
205                return this.scoreChangeIfZero;
206        }
207
208        // -------------------- IMPLEMENTATION --------------------
209
210        public void updateResiduals(final double newLambda) {
211                if (this.residualsUpdated) {
212                        throw new RuntimeException("Residuals have already been updated.");
213                }
214                this.residualsUpdated = true;
215
216                for (Map.Entry<Tuple<L, Integer>, Double> entry : this.individualWeightedChanges.entriesView()) {
217                        final L spaceObj = entry.getKey().getA();
218                        final int timeBin = entry.getKey().getB();
219                        this.interactionResiduals.add(spaceObj, timeBin, newLambda * entry.getValue());
220                }
221
222                // for (Map.Entry<Tuple<L, Integer>, Double> entry :
223                // this.currentIndividualWeightedCounts.entriesView()) {
224                // final L spaceObj = entry.getKey().getA();
225                // final int timeBin = entry.getKey().getB();
226                // // this.inertiaResiduals.add(spaceObj, timeBin, (1.0 - newLambda) *
227                // // entry.getValue());
228                // this.regularizationResiduals.add(spaceObj, timeBin, newLambda *
229                // entry.getValue());
230                // }
231                this.regularizationResidual += newLambda * this.deltaScoreIndividual;
232        }
233}
