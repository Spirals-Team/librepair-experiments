027/**
028 * Created by pieterfourie on 4/20/15.
029 * A class for artemc to where you load the travel time structures with events,
030 * and psim repeatedly executes using those fixed times.
031 * so produces the plans that are optimal if each agent were the only one to adapt to the system,
032 * i.e. if their change did not affect other traffic.
033 */
034public class ChoiceGenerationControler implements BeforeMobsimListener{
035
036        final WaitTimeStuckCalculator waitTimeCalculator;
037        final StopStopTimeCalculatorImpl stopStopTimeCalculator;
038        final TravelTimeCalculator travelTimeCalculator;
039    private PSimProvider pSimProvider;
040    Config config;
041        Controler controler;
042        Scenario scenario;
043public ChoiceGenerationControler(String[] args) {
044    config = ConfigUtils.loadConfig(args[0]);
045    config.parallelEventHandling().setSynchronizeOnSimSteps(false);
046    config.parallelEventHandling().setNumberOfThreads(1);
047    config.planCalcScore().setWriteExperiencedPlans(true);
048    scenario = ScenarioUtils.loadScenario(config);
049    controler = new Controler(scenario);
050
051    waitTimeCalculator = new WaitTimeStuckCalculator(
052            controler.getScenario().getPopulation(),
053            controler.getScenario().getTransitSchedule(),
054            controler.getConfig().travelTimeCalculator().getTraveltimeBinSize(),
055            (int) (controler.getConfig().qsim().getEndTime() - controler.getConfig().qsim().getStartTime()));
056    stopStopTimeCalculator = new StopStopTimeCalculatorImpl(
057            controler.getScenario().getTransitSchedule(),
058            controler.getConfig().travelTimeCalculator().getTraveltimeBinSize(),
059            (int) (controler.getConfig().qsim().getEndTime() - controler.getConfig().qsim().getStartTime()));
060    controler.addOverridingModule(new AbstractModule() {
061        @Override
062        public void install() {
063            bind(TransitRouter.class).toProvider(new TransitRouterEventsWSFactory(controler.getScenario(),
064                    waitTimeCalculator.get(),
065                    stopStopTimeCalculator.get()));
066        }
067    });
068    //    controler.setScoringFunctionFactory(
069//            new CharyparNagelOpenTimesScoringFunctionFactory(controler.getConfig().planCalcScore(),
070//                    controler.getScenario()));
071    travelTimeCalculator = TravelTimeCalculator.create(scenario.getNetwork(), config.travelTimeCalculator());
072
073    EventsManagerImpl eventsManager = new EventsManagerImpl();
074    EventsReaderXMLv1 reader = new EventsReaderXMLv1(eventsManager);
075    eventsManager.addHandler(waitTimeCalculator);
076    eventsManager.addHandler(stopStopTimeCalculator);
077    eventsManager.addHandler(travelTimeCalculator);
078    reader.readFile(args[1]);
079
080    controler.addOverridingModule(new AbstractModule() {
081        @Override
082        public void install() {
083            bindMobsim().toProvider(new Provider<Mobsim>() {
084                @Override
085                public Mobsim get() {
086                    return pSimProvider.createMobsim(controler.getScenario(), controler.getEvents());
087                }
088            });
089        }
090    });
091    controler.addControlerListener(this);
092}
093    public void run(){
094
095        controler.run();
096    }
097    public static void main(String[] args) {
098        new ChoiceGenerationControler(args).run();
099    }
100
101
102    @Override
103    public void notifyBeforeMobsim(BeforeMobsimEvent event) {
104        Collection<Plan> plans = new ArrayList<>();
105        for(Person person:controler.getScenario().getPopulation().getPersons().values()){
106            plans.add(person.getSelectedPlan());
107        }
108        pSimProvider.setWaitTime(waitTimeCalculator.get());
109        pSimProvider.setTravelTime(travelTimeCalculator.getLinkTravelTimes());
110        pSimProvider.setStopStopTime(stopStopTimeCalculator.get());
111    }
112}
