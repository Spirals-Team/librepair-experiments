001package org.matsim.contrib.pseudosimulation.replanning.modules;
002
003import org.matsim.api.core.v01.population.Plan;
004import org.matsim.api.core.v01.replanning.PlanStrategyModule;
005import org.matsim.contrib.pseudosimulation.distributed.plans.PlanGenome;
006import org.matsim.core.controler.MatsimServices;
007import org.matsim.core.controler.events.IterationEndsEvent;
008import org.matsim.core.controler.listener.IterationEndsListener;
009import org.matsim.core.replanning.ReplanningContext;
010import org.matsim.contrib.pseudosimulation.replanning.PlanCatcher;
011
012/**
013 * Created by fouriep on 11/26/14.
014 */
015public class RegisterMutatedPlanForPSim implements PlanStrategyModule, IterationEndsListener {
016    private final char gene;
017    private final boolean trackGenome;
018    private int iterationNumber;
019
020    public RegisterMutatedPlanForPSim(PlanCatcher slave, char gene, boolean trackGenome, MatsimServices controler) {
021        this.slave = slave;
022        this.gene=gene;
023        this.trackGenome = trackGenome;
024        controler.addControlerListener(this);
025    }
026
027    private PlanCatcher slave;
028
029    @Override
030    public void prepareReplanning(ReplanningContext replanningContext) {
031
032    }
033
034    @Override
035    public void handlePlan(Plan plan) {
036        if(slave != null)
037            slave.addPlansForPsim(plan);
038        if(trackGenome)
039            ((PlanGenome) plan).appendStrategyToGenome(String.format("%s%04d",gene, iterationNumber));
040    }
041
042    @Override
043    public void finishReplanning() {
044
045    }
046
047    @Override
048    public void notifyIterationEnds(IterationEndsEvent event) {
049        iterationNumber=event.getIteration();
050    }
051
052
053
054}
