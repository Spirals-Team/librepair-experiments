001package org.matsim.contrib.pseudosimulation.distributed.listeners.events.transit;
002
003import org.matsim.core.gbl.MatsimRandom;
004
005import java.util.Random;
006
007/**
008 * Created by fouriep on 12/19/14.
009 */
010public class BoardingModelStochasticLinear implements BoardingModel {
011    private Random random;
012
013    public BoardingModelStochasticLinear() {
014        random = MatsimRandom.getLocalInstance();
015    }
016
017    @Override
018    public boolean canBoard(double occupancy) {
019        return random.nextDouble() > occupancy;
020    }
021}
