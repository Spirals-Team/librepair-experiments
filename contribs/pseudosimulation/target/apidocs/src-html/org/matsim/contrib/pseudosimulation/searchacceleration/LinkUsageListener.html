043/**
044 * Keeps track of when every single vehicle enters which link.
045 * 
046 * @author Gunnar Flötteröd
047 *
048 */
049public class LinkUsageListener implements LinkEnterEventHandler, VehicleEntersTrafficEventHandler {
050
051        // -------------------- MEMBERS --------------------
052
053        private final TimeDiscretization timeDiscretization;
054
055        private final Map<Id<Vehicle>, Id<Person>> vehicleId2driverId = new LinkedHashMap<>();
056
057        private final Map<Id<Person>, SpaceTimeIndicators<Id<Link>>> driverId2indicators = new LinkedHashMap<>();
058
059        // -------------------- CONSTRUCTION --------------------
060
061        public LinkUsageListener(final TimeDiscretization timeDiscretization) {
062                this.timeDiscretization = timeDiscretization;
063        }
064
065        // -------------------- INTERNALS --------------------
066
067        private void registerLinkEntry(final Id<Link> linkId, final Id<Vehicle> vehicleId, final double time_s,
068                        Id<Person> driverId) {
069                if (driverId != null) {
070                        // Added even if currently outside of the time window because later
071                        // link entries may be within the time window.
072                        this.vehicleId2driverId.put(vehicleId, driverId);
073                } else {
074                        driverId = this.vehicleId2driverId.get(vehicleId);
075                        if (driverId == null) {
076                                throw new RuntimeException("Driver of vehicle " + vehicleId + " is unknown.");
077                        }
078                }
079                if ((time_s >= this.timeDiscretization.getStartTime_s()) && (time_s < this.timeDiscretization.getEndTime_s())) {
080                        SpaceTimeIndicators<Id<Link>> indicators = this.driverId2indicators.get(driverId);
081                        if (indicators == null) {
082                                indicators = new SpaceTimeIndicators<Id<Link>>(this.timeDiscretization.getBinCnt());
083                                this.driverId2indicators.put(driverId, indicators);
084                        }
085                        indicators.visit(linkId, this.timeDiscretization.getBin(time_s));
086                }
087        }
088
089        // -------------------- IMPLEMENTATION --------------------
090
091        public TimeDiscretization getTimeDiscretization() {
092                return this.timeDiscretization;
093        }
094
095        Map<Id<Person>, SpaceTimeIndicators<Id<Link>>> getAndClearIndicators() {
096                final Map<Id<Person>, SpaceTimeIndicators<Id<Link>>> result = new LinkedHashMap<>(this.driverId2indicators);
097                this.vehicleId2driverId.clear();
098                this.driverId2indicators.clear();
099                return result;
100        }
101
102        // --------------- IMPLEMENTATION OF EventHandler INTERFACES ---------------
103
104        @Override
105        public void reset(int iteration) {
106                // TODO Probably better to (i) remove replace "getAndClearIndicators" by a plain
107                // getter and to clear here.
108                if (this.driverId2indicators.size() > 0) {
109                        throw new RuntimeException("veh2indicators should be empty");
110                }
111                if (this.vehicleId2driverId.size() > 0) {
112                        throw new RuntimeException("vehicleId2driverId should be empty");
113                }
114        }
115
116        @Override
117        public void handleEvent(final VehicleEntersTrafficEvent event) {
118                this.registerLinkEntry(event.getLinkId(), event.getVehicleId(), event.getTime(), event.getPersonId());
119        }
120
121        @Override
122        public void handleEvent(final LinkEnterEvent event) {
123                this.registerLinkEntry(event.getLinkId(), event.getVehicleId(), event.getTime(), null);
124        }
125
126        // -------------------- MAIN-FUNCTION, ONLY FOR TESTING --------------------
127
128        public static void main(String[] args) {
129
130                System.out.println("Started ...");
131
132                final Config config = ConfigUtils.loadConfig("./testdata/berlin_2014-08-01_car_1pct/config.xml");
133                config.controler().setOverwriteFileSetting(OverwriteFileSetting.deleteDirectoryIfExists);
134
135                final Scenario scenario = ScenarioUtils.loadScenario(config);
136
137                final Controler controler = new Controler(scenario);
138                final TimeDiscretization timeDiscr = new TimeDiscretization(0, 3600, 24);
139                final LinkUsageListener loa = new LinkUsageListener(timeDiscr);
140                controler.getEvents().addHandler(loa);
141
142                controler.run();
143
144                System.out.println("... done.");
145        }
