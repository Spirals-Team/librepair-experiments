255                int hypotheticalVisitedCongestedLinkCnt = 0;

263
264                for (Id<Person> personId : allPersonIds) {
265
266                        // hypothetical
267                        final SpaceTimeIndicators<Id<Link>> hypotheticalTravelIndicators = driverId2pseudoSimUsage.get(personId);
268                        if (hypotheticalTravelIndicators != null) {
269                                for (int timeBin = 0; timeBin < this.timeDiscr.getBinCnt(); timeBin++) {
270                                        for (Id<Link> linkId : hypotheticalTravelIndicators.getVisitedSpaceObjects(timeBin)) {
271                                                totalHypotheticalVisitedLinkCnt++;
272                                                if (this.replParams.isCongested(linkId, timeBin, travelTimes)) {
273                                                        hypotheticalVisitedCongestedLinkCnt++;
274                                                }
275                                        }
276                                }
277                        }
278
279                        // experienced
280                        final SpaceTimeIndicators<Id<Link>> experiencedTravelIndicators = driverId2physicalSimUsage.get(personId);

288                                        }
289                                }
290                        }
291
292                        // anticipated

295                                anticipatedTravelIndicators = driverId2pseudoSimUsage.get(personId);
296                        } else {

298                        }

301                                        for (Id<Link> linkId : anticipatedTravelIndicators.getVisitedSpaceObjects(timeBin)) {
302                                                totalAnticipatedVisitedLinkCnt++;
303                                                if (this.replParams.isCongested(linkId, timeBin, travelTimes)) {
304                                                        anticipatedVisitedCongestedLinkCnt++;
305                                                }
306                                        }
307                                }
308                        }
309                }
310

314                                / totalAnticipatedVisitedLinkCnt;
315                this.experiencedCongestedLinkShare = ((double) experiencedVisitedCongestedLinkCnt)
316                                / totalExperiencedVisitedLinkCnt;
