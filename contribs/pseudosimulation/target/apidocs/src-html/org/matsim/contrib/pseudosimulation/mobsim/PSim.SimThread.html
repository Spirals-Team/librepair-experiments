086    public PSim(Scenario sc, EventsManager eventsManager, Collection<Plan> plans, TravelTime carLinkTravelTimes) {
087        Logger.getLogger(getClass()).warn("Constructing PSim");
088        this.scenario = sc;
089        this.endTime = sc.getConfig().qsim().getEndTime();
090        this.eventManager = eventsManager;
091        int numThreads = Integer.parseInt(sc.getConfig().getParam("global", "numberOfThreads"));
092        threads = new SimThread[numThreads];
093        for (int i = 0; i < numThreads; i++)
094            threads[i] = new SimThread();
095
096
097        PlansCalcRouteConfigGroup pcrConfig = sc.getConfig().plansCalcRoute();
098        this.beelineDistanceFactor = pcrConfig.getModeRoutingParams().get( TransportMode.walk ).getBeelineDistanceFactor();
099        this.walkSpeed = pcrConfig.getTeleportedModeSpeeds().get(TransportMode.walk) ;
100        this.carLinkTravelTimes = carLinkTravelTimes;
101        this.plans = plans;
102    }

116    @Override
117    public void run() {
118
119        Logger.getLogger(this.getClass()).error("Executing " + plans.size() + " plans in pseudosimulation.");
120
121        Network network = scenario.getNetwork();
122
123                /*
124         * split collection in approx even segments
125                 */
126        int n = Math.min(plans.size(), threads.length);
127        List<Plan>[] segments = CollectionUtils.split(plans, n);
128                /*
129                 * submit tasks
130                 */
131        numThreads = new AtomicInteger(threads.length);
132        for (int i = 0; i < segments.length; i++) {
133            threads[i].init(segments[i], network, eventManager);
134            new Thread(threads[i]).start();
135        }
136                /*
137                 * wait for threads
138                 */
139        while (numThreads.get() > 0) {
140            try {
141                Thread.sleep(100);
142            } catch (InterruptedException e) {
143                e.printStackTrace();
144            }
145        }
146    }

148    public class SimThread implements Runnable {
149
150        private Collection<Plan> threadPlans;
151
152        private EventsManager eventManager;
153
154        private Network network;
155
156        public void init(Collection<Plan> plans, Network network, EventsManager eventManager) {
157            this.threadPlans = plans;
158            this.network = network;
159            this.eventManager = eventManager;
160        }

162        @Override
163        public void run() {
164            PLANS:
165            for (Plan plan : threadPlans) {
166                Queue<Event> eventQueue = new LinkedList<>();
167                Id personId = plan.getPerson().getId();
168                List<PlanElement> elements = plan.getPlanElements();
169
170                double prevEndTime = 0;
171                LEGS:
172                for (int idx = 0; idx < elements.size(); idx += 2) {
173                    Activity act = (Activity) elements.get(idx);
174                                        /*
175                                         * Make sure that the activity does not end before the
176                                         * previous activity.
177                                         */
178                    double actEndTime = Math.max(prevEndTime + MIN_ACT_DURATION, act.getEndTime());
179
180                    if (idx > 0) {
181                                                /*
182                                                 * If this is not the first activity, then there must
183                                                 * exist a leg before.
184                                                 */
185
186                        Leg prevLeg = (Leg) elements.get(idx - 1);
187                        Activity prevAct = (Activity) elements.get(idx - 2);
188                        double travelTime = 0.0;
189                        if (prevLeg.getMode().equals(TransportMode.car)) {
190                            try {
191                                eventQueue.add(new PersonEntersVehicleEvent(prevEndTime, personId, personId));
192                                eventQueue.add(new VehicleEntersTrafficEvent(prevEndTime,personId, prevLeg.getRoute().getStartLinkId(),personId, TransportMode.car,1.0));
193                                NetworkRoute croute = (NetworkRoute) prevLeg.getRoute();
194
195                                travelTime = calcRouteTravelTime(croute, prevEndTime, carLinkTravelTimes, network, eventQueue, personId);
196                                eventQueue.add(new VehicleLeavesTrafficEvent(prevEndTime + travelTime,personId, prevLeg.getRoute().getEndLinkId(),personId, TransportMode.car,1.0));
197                                eventQueue.add(new PersonLeavesVehicleEvent(prevEndTime + travelTime, personId, personId));
198                            } catch (NullPointerException ne) {
199                                Logger.getLogger(this.getClass()).error("No route for car leg. Continuing with next leg");
200                                continue;
201                            }
202                        } else if (prevLeg.getMode().equals(TransportMode.transit_walk)) {
203                            TransitWalkTimeAndDistance tnd = new TransitWalkTimeAndDistance(act.getCoord(), prevAct.getCoord());
204                            travelTime = tnd.time;
205                            eventQueue.add(new TeleportationArrivalEvent(prevEndTime + tnd.time, personId, tnd.distance));
206                        } else if (prevLeg.getMode().equals(TransportMode.pt)) {
207                            if (isUseTransit) {
208                                ExperimentalTransitRoute route = (ExperimentalTransitRoute) prevLeg.getRoute();
209                                Id accessStopId = route.getAccessStopId();
210                                Id egressStopId = route.getEgressStopId();
211                                Id dummyVehicleId = Id.create("dummy", TransitVehicle.class);
212                                TransitLine line = transitLines.get(route.getLineId());
213                                TransitRoute transitRoute = line.getRoutes().get(route.getRouteId());
214                                if (transitPerformance == null) {
215                                    travelTime += waitTimes.getRouteStopWaitTime(line.getId(), transitRoute.getId(), accessStopId, prevEndTime);
216                                    eventQueue.add(new PersonEntersVehicleEvent(prevEndTime + travelTime, personId, dummyVehicleId));
217                                    travelTime += findTransitTravelTime(route, prevEndTime + travelTime);
218                                    eventQueue.add(new PersonLeavesVehicleEvent(prevEndTime + travelTime, personId, dummyVehicleId));
219                                } else {
220                                    Tuple<Double, Double> routeTravelTime = transitPerformance.getRouteTravelTime(line.getId(), route.getRouteId(), accessStopId, egressStopId, prevEndTime);
221                                    travelTime += routeTravelTime.getFirst();
222                                    eventQueue.add(new PersonEntersVehicleEvent(prevEndTime + travelTime, personId, dummyVehicleId));
223                                    travelTime += routeTravelTime.getSecond();
224                                    eventQueue.add(new PersonLeavesVehicleEvent(prevEndTime + travelTime, personId, dummyVehicleId));
225                                }
226                            }
227                        } else {
228                            try {
229                                Route route = prevLeg.getRoute();
230                                travelTime = route.getTravelTime();
231                                eventQueue.add(new TeleportationArrivalEvent(prevEndTime + travelTime, personId, Double.NaN));
232                            } catch (NullPointerException e) {
233                                Logger.getLogger(this.getClass()).error("No route for this leg. Continuing with next leg");
234                                continue;
235                            }
236                        }
237
238                        travelTime = Math.max(MIN_LEG_DURATION, travelTime);
239                        double arrivalTime = travelTime + prevEndTime;
240
241                                                /*
242                                                 * Make sure that the activity does not end before the
243                                                 * agent arrives.
244                                                 */
245                        actEndTime = Math.max(arrivalTime + MIN_ACT_DURATION, actEndTime);
246                                                /*
247                                                 * If act end time is not specified...
248                                                 */
249                        if (Double.isInfinite(actEndTime)) {
250                            if(transitPerformance!=null){
251                                //this guy is stuck, will be caught in events handling outside the loop
252                                break;
253                            }else
254                                throw new RuntimeException("I think this is discuraged.");
255                        }
256                                                /*
257                                                 * Send arrival and activity start events.
258                                                 */
259                        PersonArrivalEvent arrivalEvent = new PersonArrivalEvent(arrivalTime, personId, act.getLinkId(), prevLeg.getMode());
260                        eventQueue.add(arrivalEvent);
261                        ActivityStartEvent startEvent = new ActivityStartEvent(arrivalTime, personId, act.getLinkId(), act.getFacilityId(), act.getType());
262                        eventQueue.add(startEvent);
263                    }
264
265                    if (idx < elements.size() - 1) {
266                                                /*
267                                                 * This is not the last activity, send activity end and
268                                                 * departure events.
269                                                 */
270                        Leg nextLeg = (Leg) elements.get(idx + 1);
271                        ActivityEndEvent endEvent = new ActivityEndEvent(actEndTime, personId, act.getLinkId(), act.getFacilityId(), act.getType());
272                        eventQueue.add(endEvent);
273                        PersonDepartureEvent departureEvent = new PersonDepartureEvent(actEndTime, personId, act.getLinkId(), nextLeg.getMode());
274
275                        eventQueue.add(departureEvent);
276                    }
277
278                    prevEndTime = actEndTime;
279                }
280                for (Event event : eventQueue) {
281                    if (event.getTime() > endTime) {
282                        eventManager.processEvent(new PersonStuckEvent(endTime, personId, null, null));
283                        break;
284                    }
285                    eventManager.processEvent(event);
286                }
287            }
288
289            numThreads.decrementAndGet();
290        }

292        private double findTransitTravelTime(ExperimentalTransitRoute route, double prevEndTime) {
293            double travelTime = 0;
294            double prevStopTime = prevEndTime;
295            TransitRouteImpl transitRoute = (TransitRouteImpl) transitLines.get(route.getLineId()).getRoutes().get(route.getRouteId());
296            // cannot just get the indices of the two transitstops, because
297            // sometimes routes visit the same stop facility more than once
298            // and the transitroutestop is different for each time it stops at
299            // the same facility
300            TransitRouteStop orig = transitRoute.getStop(stopFacilities.get(route.getAccessStopId()));
301            Id dest = route.getEgressStopId();
302            int i = transitRoute.getStops().indexOf(orig);
303            // int j = transitRoute.getStops().indexOf(dest);
304            // if(i>=j){
305            // throw new
306            // RuntimeException(String.format("Cannot route from origin stop %s to destination stop %s on route %s.",
307            // orig.getStopFacility().getId().toString()
308            // ,dest.getStopFacility().getId().toString()
309            // ,route.getRouteId().toString()
310            // ));
311            //
312            // }
313            boolean destinationFound = false;
314            while (i < transitRoute.getStops().size() - 1) {
315                Id fromId = transitRoute.getStops().get(i).getStopFacility().getId();
316                TransitRouteStop toStop = transitRoute.getStops().get(i + 1);
317                Id toId = toStop.getStopFacility().getId();
318                travelTime += stopStopTimes.getStopStopTime(fromId, toId, prevStopTime);
319                prevStopTime += travelTime;
320                if (toStop.getStopFacility().getId().equals(dest)) {
321                    destinationFound = true;
322                    break;
323                }
324                if (toStop.getStopFacility().getId().equals(route.getAccessStopId())) {
325                    //this is a repeating stop, for routes that loop on themselves more than once
326                    travelTime = 0;
327                    prevStopTime = prevEndTime;
328                }
329                i++;
330            }
331            if (destinationFound) {
332                return travelTime;
333            } else {
334
335                return Double.NEGATIVE_INFINITY;
336            }
337        }

339        private double calcRouteTravelTime(NetworkRoute route, double startTime, TravelTime travelTime, Network network, Queue<Event> eventQueue, Id agentId) {
340
341            double tt = 0;
342            if (route.getStartLinkId() != route.getEndLinkId()) {
343                Id<Link> startLink = route.getStartLinkId();
344                double linkEnterTime = startTime;
345                LinkEnterEvent linkEnterEvent = null;
346                LinkLeaveEvent linkLeaveEvent = new LinkLeaveEvent(++linkEnterTime, agentId, startLink);
347                eventQueue.add(linkLeaveEvent);
348                double linkLeaveTime = linkEnterTime;
349                List<Id<Link>> routeLinkIds = route.getLinkIds();
350                for (Id<Link> routeLinkId : routeLinkIds) {
351                    if (linkEnterTime > 1E16) {
352                        int mmm = 0;
353                    }
354                    linkEnterTime = linkLeaveTime;
355                    linkEnterEvent = new LinkEnterEvent(linkEnterTime, agentId, routeLinkId);
356                    eventQueue.add(linkEnterEvent);
357
358                    double linkTime = travelTime.getLinkTravelTime(network.getLinks().get(routeLinkId), linkEnterTime, null, null);
359                    tt += Math.max(linkTime, 1.0);
360
361                    linkLeaveTime = Math.max(linkEnterTime + 1, linkEnterTime + linkTime);
362                    linkLeaveEvent = new LinkLeaveEvent(linkLeaveTime, agentId, routeLinkId);
363                    eventQueue.add(linkLeaveEvent);
364                }
365                tt = linkLeaveTime - startTime;
366            }
367            LinkEnterEvent linkEnterEvent = new LinkEnterEvent(startTime + tt, agentId, route.getEndLinkId());
368            eventQueue.add(linkEnterEvent);
369            return tt + travelTime.getLinkTravelTime(network.getLinks().get(route.getEndLinkId()), tt + startTime, null, null);
370        }
371
372        private double calcRouteTravelTime(NetworkRoute route, double startTime, TravelTime travelTime, Network network) {
373            double tt = 0;
374            if (route.getStartLinkId() != route.getEndLinkId()) {
375
376                List<Id<Link>> ids = route.getLinkIds();
377                for (Id<Link> id : ids) {
378                    tt += travelTime.getLinkTravelTime(network.getLinks().get(id), startTime, null, null);
379                    tt++;// 1 sec for each node
380                }
381                tt += travelTime.getLinkTravelTime(network.getLinks().get(route.getEndLinkId()), startTime, null, null);
382            }
383
384            return tt;
385        }

389    class TransitWalkTimeAndDistance {
390        final double time;
391        final double distance;
392
393        public TransitWalkTimeAndDistance(Coord startCoord, Coord endCoord) {
394            distance = beelineDistanceFactor * CoordUtils.calcEuclideanDistance(startCoord, endCoord);
395            time = distance / walkSpeed;
396        }
