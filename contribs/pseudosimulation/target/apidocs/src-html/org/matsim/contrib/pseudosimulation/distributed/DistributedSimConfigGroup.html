001package org.matsim.contrib.pseudosimulation.distributed;
002
003import org.matsim.core.config.ReflectiveConfigGroup;
004
005/**
006 * Created by fouriep on 4/4/17.
007 */
008public class DistributedSimConfigGroup extends ReflectiveConfigGroup {
009    public static final String GROUP_NAME = "distributed";
010
011    // parameters and their defaults
012    final static String MASTER_PORT_NUMBER = "masterPortNumber";
013    private int masterPortNumber = 12345;
014    static final String SLAVE_ITERS_PER_MASTER_ITER = "slaveIterationsPerMasterIteration";
015    private int slaveIterationsPerMasterIteration = 10;
016    static final String DEFAULT_NUM_THREADS_ON_SLAVE = "defaultNumThreadsOnSlave";
017    private int defaultNumThreadsOnSlave = 1;
018    /**
019     * The number of slaves the master will distribute to from the start.
020     * Defaults to 2, each with one thread for replanning.
021     * You can still add more afterwards and load balancing should take care of it.
022     *
023     * Override the default number of threads on the slave with a
024     */
025    static final String INIT_NUM_SLAVES = "initialNumberOfSlaves";
026    private int initialNumberOfSlaves = 2;
027    /**
028     * Overall innovation rate on master.
029     */
030    static final String MASTER_MUTATION_RATE = "masterMutationRate";
031    private double masterMutationRate = 0.1;
032    /**
033     * Overall rate of innovation on slaves per slave iteration. Total number of plans produced
034     * per master iteration across all slaves is slaveMutationRate * slaveIterationsPerMasterIteration.
035     */
036    static final String SLAVE_MUTATION_RATE = "slaveMutationRate";
037    private double slaveMutationRate = 0.3;
038    /**
039     * The probability that the master will accept a plan from a slave.
040     */
041    static final String MASTER_BORROWING_RATE = "masterBorrowingRate";
042    private double masterBorrowingRate = 0.6667;
043    /**
044     * if false, randomised routing takes place. Should be modified to
045     * allow injection of custom routers...
046     */
047    static final String INTELLIGENT_ROUTERS = "intelligentRouters";
048    private boolean intelligentRouters = true;
049    /**
050     * If true, a data structure recording each vehicle's arrivals and departures at each stop
051     * in the master simulation is transmitted to the slaves, instead of an average StopStopTime
052     * that gives stop-to-stop travel time for time bins between each pair of stops.
053     */
054    static final String FULL_TRANSIT_PERFORMANCE = "fullTransitPerformanceTransmission";
055    private boolean fullTransitPerformanceTransmission = true;
056    /**
057     * If in parallel, the slaves run while the Master is sunning QSim. That means they operate
058     * on information that is 2 iterations older than the master's current iteration.
059     * If not in parallel, slaves wait or the latest travel times from the master.
060     * The master, in turn, waits for all slaves to complete the requisite number of
061     * slaveIterationsPerMasterIteration. Slaves then operate on information one iteration older than
062     * the master's current iteration, i.e. similar to default MATSim.
063     */
064    static final String PARALLEL_SIMULATION = "slavesRunInParallelToMaster";
065    private boolean slavesRunInParallelToMaster = true;
066
067    public DistributedSimConfigGroup() {
068        super(GROUP_NAME);
069    }
070    @StringGetter(MASTER_PORT_NUMBER)
071    public int getMasterPortNumber() {
072        return masterPortNumber;
073    }
074    @StringSetter(MASTER_PORT_NUMBER)
075    public void setMasterPortNumber(int masterPortNumber) {
076        this.masterPortNumber = masterPortNumber;
077    }
078    @StringGetter(SLAVE_ITERS_PER_MASTER_ITER)
079    public int getSlaveIterationsPerMasterIteration() {
080        return slaveIterationsPerMasterIteration;
081    }
082    @StringSetter(SLAVE_ITERS_PER_MASTER_ITER)
083    public void setSlaveIterationsPerMasterIteration(int slaveIterationsPerMasterIteration) {
084        this.slaveIterationsPerMasterIteration = slaveIterationsPerMasterIteration;
085    }
086    @StringGetter(INIT_NUM_SLAVES)
087    public int getInitialNumberOfSlaves() {
088        return initialNumberOfSlaves;
089    }
090    @StringSetter(INIT_NUM_SLAVES)
091    public void setInitialNumberOfSlaves(int initialNumberOfSlaves) {
092        this.initialNumberOfSlaves = initialNumberOfSlaves;
093    }
094    @StringGetter(MASTER_MUTATION_RATE)
095    public double getMasterMutationRate() {
096        return masterMutationRate;
097    }
098    @StringSetter(MASTER_MUTATION_RATE)
099    public void setMasterMutationRate(double masterMutationRate) {
100        this.masterMutationRate = masterMutationRate;
101    }
102    @StringGetter(DEFAULT_NUM_THREADS_ON_SLAVE)
103    public int getDefaultNumThreadsOnSlave() {
104        return defaultNumThreadsOnSlave;
105    }
106    @StringSetter(DEFAULT_NUM_THREADS_ON_SLAVE)
107    public void setDefaultNumThreadsOnSlave(int defaultNumThreadsOnSlave) {
108        this.defaultNumThreadsOnSlave = defaultNumThreadsOnSlave;
109    }
110    @StringGetter(SLAVE_MUTATION_RATE)
111    public double getSlaveMutationRate() {
112        return slaveMutationRate;
113    }
114    @StringSetter(SLAVE_MUTATION_RATE)
115    public void setSlaveMutationRate(double slaveMutationRate) {
116        this.slaveMutationRate = slaveMutationRate;
117    }
118    @StringGetter(MASTER_BORROWING_RATE)
119    public double getMasterBorrowingRate() {
120        return masterBorrowingRate;
121    }
122    @StringSetter(MASTER_BORROWING_RATE)
123    public void setMasterBorrowingRate(double masterBorrowingRate) {
124        this.masterBorrowingRate = masterBorrowingRate;
125    }
126    @StringGetter(INTELLIGENT_ROUTERS)
127    public boolean isIntelligentRouters() {
128        return intelligentRouters;
129    }
130    @StringSetter(INTELLIGENT_ROUTERS)
131    public void setIntelligentRouters(boolean intelligentRouters) {
132        this.intelligentRouters = intelligentRouters;
133    }
134    @StringGetter(FULL_TRANSIT_PERFORMANCE)
135    public boolean isFullTransitPerformanceTransmission() {
136        return fullTransitPerformanceTransmission;
137    }
138    @StringSetter(FULL_TRANSIT_PERFORMANCE)
139    public void setFullTransitPerformanceTransmission(boolean fullTransitPerformanceTransmission) {
140        this.fullTransitPerformanceTransmission = fullTransitPerformanceTransmission;
141    }
142    @StringGetter(PARALLEL_SIMULATION)
143    public boolean isSlavesRunInParallelToMaster() {
144        return slavesRunInParallelToMaster;
145    }
146    @StringSetter(PARALLEL_SIMULATION)
147    public void setSlavesRunInParallelToMaster(boolean slavesRunInParallelToMaster) {
148        this.slavesRunInParallelToMaster = slavesRunInParallelToMaster;
149    }
150}
