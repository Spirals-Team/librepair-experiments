001package org.matsim.contrib.pseudosimulation;
002
003
004import com.google.inject.Inject;
005import org.apache.log4j.Logger;
006import org.matsim.api.core.v01.Id;
007import org.matsim.api.core.v01.Scenario;
008import org.matsim.api.core.v01.population.Person;
009import org.matsim.api.core.v01.population.Plan;
010import org.matsim.contrib.pseudosimulation.mobsim.PSimProvider;
011import org.matsim.contrib.pseudosimulation.replanning.PlanCatcher;
012import org.matsim.core.controler.events.BeforeMobsimEvent;
013import org.matsim.core.controler.events.IterationEndsEvent;
014import org.matsim.core.controler.events.IterationStartsEvent;
015import org.matsim.core.controler.listener.BeforeMobsimListener;
016import org.matsim.core.controler.listener.IterationEndsListener;
017import org.matsim.core.controler.listener.IterationStartsListener;
018
019import java.util.ArrayList;
020import java.util.HashMap;
021import java.util.Iterator;
022import java.util.Map;
023
024
025public class MobSimSwitcher implements IterationEndsListener,
026                IterationStartsListener, BeforeMobsimListener {
027        private final ArrayList<Integer> qsimIters = new ArrayList<>();
028        private final Scenario scenario;
029        private boolean isQSimIteration = true;
030        private int psimIterationCount = 0;
031        private int iterationsPerCycle = 0;
032        private Map<Id<Person>, Double> selectedPlanScoreMemory;
033        @Inject private PlanCatcher plancatcher;
034        @Inject private PSimProvider pSimProvider;
035
036        @Inject
037        MobSimSwitcher(PSimConfigGroup pSimConfigGroup, Scenario scenario) {
038                iterationsPerCycle = pSimConfigGroup.getIterationsPerCycle();
039                this.scenario = scenario;
040        }
041
042        public PSimProvider getpSimProvider() {
043                return pSimProvider;
044        }
045
046        public boolean isQSimIteration() {
047                return isQSimIteration;
048        }
049
050        public ArrayList<Integer> getQSimIters() {
051                return qsimIters;
052        }
053
054        @Override
055        public void notifyIterationStarts(IterationStartsEvent event) {
056                if (determineIfQSimIter(event.getIteration())) {
057                        Logger.getLogger(this.getClass()).warn("Running full queue simulation");
058
059                } else {
060                        Logger.getLogger(this.getClass()).info("Running PSim");
061                        plancatcher.init();
062                        for (Person person : scenario.getPopulation().getPersons().values()) {
063                                        plancatcher.addPlansForPsim(person.getSelectedPlan());
064                        }
065
066                }
067        }
068
069        private boolean determineIfQSimIter(int iteration) {
070
071                if (iteration == scenario.getConfig().controler().getLastIteration() || iteration == scenario.getConfig().controler().getFirstIteration() ) {
072                        isQSimIteration = true;
073                        return isQSimIteration;
074                }
075                if (isQSimIteration && psimIterationCount == 0) {
076                        isQSimIteration = false;
077                        psimIterationCount++;
078                        return isQSimIteration;
079                }
080                if (psimIterationCount >= iterationsPerCycle - 1) {
081                        isQSimIteration = true;
082                        qsimIters.add(iteration);
083                        psimIterationCount = 0;
084                        return isQSimIteration;
085                }
086                if (isQSimIteration) {
087                        qsimIters.add(iteration);
088                } else {
089                        psimIterationCount++;
090
091                }
092                return isQSimIteration;
093        }
094
095        @Override
096        public void notifyBeforeMobsim(BeforeMobsimEvent event) {
097                //only for psim iterations
098                if (this.isQSimIteration())
099                        return;
100
101                for (Person person : scenario.getPopulation().getPersons().values()) {
102                                plancatcher.removeExistingPlanOrAddNewPlan(person.getSelectedPlan());
103                }
104
105                selectedPlanScoreMemory = new HashMap<>(scenario.getPopulation().getPersons().size());
106
107                for (Person person : scenario.getPopulation().getPersons().values()) {
108                        selectedPlanScoreMemory.put(person.getId(), person.getSelectedPlan().getScore());
109                }
110                for (Plan plan : plancatcher.getPlansForPSim()) {
111                        selectedPlanScoreMemory.remove(plan.getPerson().getId());
112                }
113
114        }
115
116
117        @Override
118        public void notifyIterationEnds(IterationEndsEvent event) {
119                if (this.isQSimIteration())
120                        return;
121                Iterator<Map.Entry<Id<Person>, Double>> iterator = selectedPlanScoreMemory.entrySet().iterator();
122                while (iterator.hasNext()) {
123                        Map.Entry<Id<Person>, Double> entry = iterator.next();
124                        scenario.getPopulation().getPersons().get(entry.getKey()).getSelectedPlan().setScore(entry.getValue());
125                }
126        }
127
128}
