019public class TransitPerformance implements Serializable {
020    private Map<String, DwellEventsForLine> linesToStopDwellEvents = new HashMap<>();

023        return boardingModel;

028    private BoardingModel boardingModel;
029
030    public TransitPerformance(BoardingModel boardingModel) {
031        this.boardingModel = boardingModel;
032    }
033
034    public TransitPerformance() {
035        this.boardingModel = new BoardingModelStochasticLinear();
036    }
037
038    public void setBoardingModel(BoardingModel boardingModel) {
039        this.boardingModel = boardingModel;
040    }
041
042    public void addVehicleDwellEventAtStop(Id<TransitLine> line, Id<TransitRoute> route, Id<TransitStopFacility> stopId, DwellEvent dwellEvent) {
043        DwellEventsForLine stopDwellEventsForLine = linesToStopDwellEvents.get(line.toString());
044        if (stopDwellEventsForLine == null) {
045            stopDwellEventsForLine = new DwellEventsForLine();
046            linesToStopDwellEvents.put(line.toString(),stopDwellEventsForLine);
047        }
048        stopDwellEventsForLine.addVehicleDwellEventAtStop(route, stopId, dwellEvent);
049    }
050
051
052    public Tuple<Double, Double> getRouteTravelTime(Id<TransitLine> line, Id<TransitRoute> route, Id<TransitStopFacility> originStop, Id<TransitStopFacility> destinationStop, double time) {
053        try {
054            return linesToStopDwellEvents.get(line.toString()).getRouteTravelTime(route, originStop, destinationStop, time);
055        }catch(NullPointerException ne){
056            return new Tuple(Double.POSITIVE_INFINITY,Double.POSITIVE_INFINITY);
057        }
058    }

061    private class DwellEventsAtStop implements Serializable {
062        private List<DwellEvent> dwellEvents;
063
064        public DwellEventsAtStop() {
065            this.dwellEvents = new ArrayList<>();
066
067        }
068
069        public void addVehicleDwellEventAtStop(DwellEvent dwellEvent) {
070            this.dwellEvents.add(dwellEvent);
071        }
072
073        public Tuple<Double, Double> getTravelTime(Id<TransitStopFacility> destinationStop, double time) {
074            double inVehicleTime = Double.POSITIVE_INFINITY;
075            LinkedList<Double> lastInvehicleTime = new LinkedList<>();
076            for (DwellEvent dwellEvent : dwellEvents) {
077                lastInvehicleTime.add(dwellEvent.getVehicle().getInVehicleTime(dwellEvent, destinationStop));
078                if (lastInvehicleTime.size() > 6) lastInvehicleTime.removeFirst();
079                if (dwellEvent.getArrivalTime() >= time) {
080                    //check if it's possible to board
081                    if (!boardingModel.canBoard(dwellEvent.getOccupancyAtDeparture()))
082                        continue;
083
084                    DoubleArrayList doubleArrayList = new DoubleArrayList();
085                    doubleArrayList.addAllOf(lastInvehicleTime);
086                    double mean = Descriptive.mean(doubleArrayList);
087//                    try {
088                        inVehicleTime = lastInvehicleTime.get(rand.nextInt(lastInvehicleTime.size()));
089//                        inVehicleTime = new NormalDistributionImpl(mean, Math.sqrt(Descriptive.sampleVariance(doubleArrayList,mean))).inverseCumulativeProbability(MatsimRandom.getRandom().nextDouble());
090//                    } catch (MathException e) {
091//                        inVehicleTime= mean;
092//                    }catch (MathRuntimeException re){
093//                        inVehicleTime= mean;
094//                    }
095
096//                    inVehicleTime = dwellEvent.getVehicle().getInVehicleTime(dwellEvent, destinationStop);
097                }
098                if (!Double.isInfinite(inVehicleTime)) {
099                    return new Tuple<>(dwellEvent.getArrivalTime() - time, inVehicleTime);
100                }
101            }
102            return new Tuple<>(Double.POSITIVE_INFINITY, Double.POSITIVE_INFINITY);
103        }
104    }
105
106    private class DwellEventsForRoute implements Serializable {
107
108        Map<String, DwellEventsAtStop> dwellEventsAtStops = new HashMap<>();
109
110        private void addVehicleDwellEventAtStop(Id<TransitStopFacility> stopId, DwellEvent dwellEvent) {
111            DwellEventsAtStop dwellEvents = dwellEventsAtStops.get(stopId.toString());
112            if (dwellEvents == null) {
113                dwellEvents = new DwellEventsAtStop();
114                dwellEventsAtStops.put(stopId.toString(), dwellEvents);
115            }
116            dwellEvents.addVehicleDwellEventAtStop(dwellEvent);
117        }
118
119        public Tuple<Double, Double> getTravelTime(Id<TransitStopFacility> originStop, Id<TransitStopFacility> destinationStop, double time) {
120            return dwellEventsAtStops.get(originStop.toString()).getTravelTime(destinationStop, time);
121        }
122
123    }
124
125    private class DwellEventsForLine implements Serializable{
126
127        private Map<String, DwellEventsForRoute> routesToDwellEvents = new HashMap<>();
128
129        private void addVehicleDwellEventAtStop(Id<TransitRoute> route, Id<TransitStopFacility> stopId, DwellEvent dwellEvent) {
130            DwellEventsForRoute stopDwellEventsForRoute = routesToDwellEvents.get(route.toString());
131            if (stopDwellEventsForRoute == null) {
132                stopDwellEventsForRoute = new DwellEventsForRoute();
133                routesToDwellEvents.put(route.toString(),stopDwellEventsForRoute);
134            }
135            stopDwellEventsForRoute.addVehicleDwellEventAtStop(stopId, dwellEvent);
136        }
137
138
139        public Tuple<Double, Double> getRouteTravelTime(Id<TransitRoute> route, Id<TransitStopFacility> originStop, Id<TransitStopFacility> destinationStop, double time) {
140            return routesToDwellEvents.get(route.toString()).getTravelTime(originStop, destinationStop, time);
141        }
142    }
