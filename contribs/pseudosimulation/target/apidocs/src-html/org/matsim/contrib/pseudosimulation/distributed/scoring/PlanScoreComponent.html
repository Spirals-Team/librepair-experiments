001package org.matsim.contrib.pseudosimulation.distributed.scoring;
002
003import java.io.Serializable;
004
005public class PlanScoreComponent implements Serializable{
006    private final String description;
007    private ScoreComponentType type;
008
009    public double getScore() {
010        return score;
011    }
012
013    public ScoreComponentType getType() {
014        return type;
015    }
016
017    private double score;
018
019    @Override
020    public String toString() {
021        return String.format("%s\t%s\t%.3f",type,description,score);
022    }
023
024    public PlanScoreComponent(ScoreComponentType type, double score, String description) {
025        this.type = type;
026        this.score = score;
027        this.description=description;
028    }
029
030    public String getDescription() {
031        return description;
032    }
033}
