032import java.io.*;
033import java.util.zip.Deflater;
034
035public class CompressedBlockOutputStream extends FilterOutputStream {
036    /**
037     * Buffer for input data
038     */
039    private byte[] inBuf = null;
040
041    /**
042     * Buffer for compressed data to be written
043     */
044    private byte[] outBuf = null;
045
046    /**
047     * Number of bytes in the buffer
048     */
049    private int len = 0;
050
051    /**
052     * Deflater for compressing data
053     */
054    private Deflater deflater = null;
055
056    /**
057     * Constructs a CompressedBlockOutputStream that writes to
058     * the given underlying output stream 'os' and sends a compressed
059     * block once 'size' byte have been written. The default
060     * compression strategy and level are used.
061     */
062    public CompressedBlockOutputStream(OutputStream os, int size)
063            throws IOException {
064        this(os, size,
065                Deflater.DEFAULT_COMPRESSION, Deflater.DEFAULT_STRATEGY);
066    }
067
068    /**
069     * Constructs a CompressedBlockOutputStream that writes to the
070     * given underlying output stream 'os' and sends a compressed
071     * block once 'size' byte have been written. The compression
072     * level and strategy should be specified using the constants
073     * defined in java.util.zip.Deflator.
074     */
075    public CompressedBlockOutputStream(OutputStream os, int size,
076                                       int level, int strategy) throws IOException {
077        super(os);
078        this.inBuf = new byte[size];
079        this.outBuf = new byte[size + 64];
080        this.deflater = new Deflater(level);
081        this.deflater.setStrategy(strategy);
082    }
083
084    protected void compressAndSend() throws IOException {
085        if (len > 0) {
086            deflater.setInput(inBuf, 0, len);
087            deflater.finish();
088            int size = deflater.deflate(outBuf);
089
090            // Write the size of the compressed data, followed
091            // by the size of the uncompressed data
092            out.write((size >> 24) & 0xFF);
093            out.write((size >> 16) & 0xFF);
094            out.write((size >>  8) & 0xFF);
095            out.write((size >>  0) & 0xFF);
096
097            out.write((len >> 24) & 0xFF);
098            out.write((len >> 16) & 0xFF);
099            out.write((len >>  8) & 0xFF);
100            out.write((len >>  0) & 0xFF);
101
102            out.write(outBuf, 0, size);
103            out.flush();
104
105            len = 0;
106            deflater.reset();
107        }
108    }
109
110    public void write(int b) throws IOException {
111        inBuf[len++] = (byte) b;
112        if (len == inBuf.length) {
113            compressAndSend();
114        }
115    }
116
117    public void write(byte[] b, int boff, int blen)
118            throws IOException {
119        while ((len + blen) > inBuf.length) {
120            int toCopy = inBuf.length - len;
121            System.arraycopy(b, boff, inBuf, len, toCopy);
122            len += toCopy;
123            compressAndSend();
124            boff += toCopy;
125            blen -= toCopy;
126        }
127        System.arraycopy(b, boff, inBuf, len, blen);
128        len += blen;
129    }
130
131    public void flush() throws IOException {
132        compressAndSend();
133        out.flush();
134    }
135
136    public void close() throws IOException {
137        compressAndSend();
138        out.close();
139    }
140}
