020public class GenomeAnalysis implements IterationEndsListener {
021    private final boolean writePlanScores;
022    private final boolean writeFullGenomeStats;
023    private final boolean writeScoreComponents;
024    private String outputPath;
025
026    public GenomeAnalysis(boolean writePlanScores, boolean writeFullGenomeStats, boolean writeScoreComponents) {
027        this.writePlanScores = writePlanScores;
028        this.writeFullGenomeStats = writeFullGenomeStats;
029        this.writeScoreComponents = writeScoreComponents;
030    }
031
032    @Override
033    public void notifyIterationEnds(IterationEndsEvent event) {
034        outputPath = event.getServices().getControlerIO().getOutputPath();
035        Map<String, Integer> fullGeneCount = new HashMap<>();
036        Map<String, Double> fullGeneScore = new HashMap<>();
037        Map<String, Double> fullGeneAltScore = new HashMap<>();
038        Map<Id<Person>, ? extends Person> persons = event.getServices().getScenario().getPopulation().getPersons();
039        boolean append = event.getIteration() != event.getServices().getConfig().controler().getFirstIteration();
040        try {
041            PrintWriter writer = null;
042            PrintWriter scoreComponentWriter = null;
043            if (writePlanScores) {
044                writer = getWriter(false, "planScores.csv", "iter,personid,genome,score,altscore");
045            }
046            if (writeScoreComponents) {
047                scoreComponentWriter = getWriter(false, "scoreComponents.csv", "iter,personid,simtype,component,description,score");
048            }
049            for (Person person : persons.values()) {
050                Plan plan = person.getSelectedPlan();
051                PlanGenome planGenome = (PlanGenome) plan;
052                if (writePlanScores) {
053                    writer.println(String.format("%d,%s,%s,%.3f,%.3f", event.getIteration(), person.getId(), planGenome.getGenome(), planGenome.getScore(), planGenome.getpSimScore()));
054                }
055                if (writeScoreComponents) {
056                    ArrayList<PlanScoreComponent> scoreComponents = planGenome.getScoreComponents();
057                    for (PlanScoreComponent component : scoreComponents) {
058                        String s = String.format("%d,%s,%s,%s,%s,%.3f", event.getIteration(), person.getId(), "q", component.getType().toString(), component.getDescription(), component.getScore());
059                        scoreComponentWriter.println(s);
060                    }
061                    scoreComponents = planGenome.getAltScoreComponents();
062                    for (PlanScoreComponent component : scoreComponents) {
063                        String s = String.format("%d,%s,%s,%s,%s,%.3f", event.getIteration(), person.getId(), "p", component.getType().toString(), component.getDescription(), component.getScore());
064                        scoreComponentWriter.println(s);
065                    }
066                }
067                try {
068                    int count = fullGeneCount.get(planGenome.getGenome());
069                    fullGeneCount.put(planGenome.getGenome(), count + 1);
070                    double meanScore = fullGeneScore.get(planGenome.getGenome());
071                    double dcount = (double) count;
072                    meanScore = (meanScore * dcount + planGenome.getScore()) / (dcount + 1);
073                    fullGeneScore.put(planGenome.getGenome(), meanScore);
074                    meanScore = fullGeneAltScore.get(planGenome.getGenome());
075                    meanScore = (meanScore * dcount + planGenome.getpSimScore()) / (dcount + 1);
076                    fullGeneAltScore.put(planGenome.getGenome(), meanScore);
077                } catch (NullPointerException e) {
078                    fullGeneCount.put(planGenome.getGenome(), 1);
079                    fullGeneScore.put(planGenome.getGenome(), planGenome.getScore());
080                    fullGeneAltScore.put(planGenome.getGenome(), planGenome.getpSimScore());
081                }
082            }
083            if (writePlanScores) {
084                writer.flush();
085                writer.close();
086            }
087            if (writeScoreComponents) {
088                scoreComponentWriter.flush();
089                scoreComponentWriter.close();
090            }
091        } catch (IOException e) {
092            e.printStackTrace();
093        }
094        Logger logger = Logger.getLogger(this.getClass());
095        List<String> keys = new ArrayList<>();
096        keys.addAll(fullGeneCount.keySet());
097        Collections.sort(keys);
098        if (writeFullGenomeStats) {
099            logger.info("Printing genome stats");
100            try {
101                PrintWriter writer = getWriter(append, "planEvo.csv", "iter,genome,count,score,altscore");
102                for (String key : keys) {
103                    writer.println(String.format("%d,%s,%d,%.3f,%.3f", event.getIteration(), key, fullGeneCount.get(key), fullGeneScore.get(key), fullGeneAltScore.get(key)));
104                }
105                writer.flush();
106                writer.close();
107//            Set<SimpleLink> edges = new HashSet<>();
108//            for(String key:keys){
109//                edges.add(new SimpleLink(key, fullGeneCount.get(key),fullGeneScore.get(key),fullGeneAltScore.get(key)));
110//            }
111//            writer = new PrintWriter(new BufferedWriter(new FileWriter(event.getServices().getControlerIO().getOutputPath() + "/planEvoEdges.csv", false)));
112//                writer.println(String.format("Id,Source,Target,Type,Label,Count,Score,altscore"));
113//            for (SimpleLink edge : edges) {
114//                writer.println(String.format("%s,%s,%s,Undirected,%s,%d,%.3f,%.3f", edge.to,edge.from,edge.to,edge.type, edge.count,edge.score,edge.altscore));
115//            }
116//            writer.flush();
117//            writer.close();
118            } catch (IOException e) {
119                e.printStackTrace();
120            }
121        }
122        Map<String, Integer> strategyMutationCount = new HashMap<>();
123        Map<String, Integer> simpleGeneCount = new HashMap<>();
124        Map<String, Double> simpleGeneScore = new HashMap<>();
125        Map<String, Double> simpleGeneAltScore = new HashMap<>();
126        Map<Integer, Integer> geneLengths = new TreeMap<>();
127        Map<Integer, Double> geneScoresbyLength = new TreeMap<>();
128
129        for (String s : fullGeneCount.keySet()) {
130            String[] charSplit = s.split("[0-9]*");
131            String[] numSplit = s.split("[A-Z]+");
132            int geneLength = charSplit.length / 2 + 1;
133            if (geneLengths.get(geneLength) != null) {
134                geneLengths.put(geneLength, geneLengths.get(geneLength) + fullGeneCount.get(s));
135                geneScoresbyLength.put(geneLength,
136                        (geneScoresbyLength.get(geneLength) * geneLengths.get(geneLength) +
137                                fullGeneScore.get(s) * fullGeneCount.get(s)) / (geneLengths.get(geneLength) + fullGeneCount.get(s)));
138            } else {
139                geneLengths.put(geneLength, fullGeneCount.get(s));
140                geneScoresbyLength.put(geneLength, fullGeneScore.get(s));
141            }
142            if (charSplit.length == 1) {
143                String simpleGene = "I0000";
144                strategyMutationCount.put("I", fullGeneCount.get(s));
145                simpleGeneCount.put(simpleGene, fullGeneCount.get(s));
146                simpleGeneScore.put(simpleGene, fullGeneScore.get(s));
147                simpleGeneAltScore.put(simpleGene, fullGeneAltScore.get(s));
148            } else {
149
150                for (int i = 2; i < charSplit.length; i += 2) {
151                    String simpleGene = charSplit[i] + numSplit[i / 2 + 1];
152                    try {
153                        int count = strategyMutationCount.get(charSplit[i]);
154                        strategyMutationCount.put(charSplit[i], fullGeneCount.get(s) + count);
155                        count = simpleGeneCount.get(simpleGene);
156                        simpleGeneCount.put(simpleGene, count + fullGeneCount.get(s));
157                        Double score = simpleGeneScore.get(simpleGene);
158                        score = (score * (double) count + fullGeneScore.get(s) * (double) fullGeneCount.get(s)) / (count + fullGeneCount.get(s));
159                        simpleGeneScore.put(simpleGene, score);
160                        score = simpleGeneAltScore.get(simpleGene);
161                        score = (score * (double) count + fullGeneAltScore.get(s) * (double) fullGeneCount.get(s)) / (count + fullGeneCount.get(s));
162                        simpleGeneAltScore.put(simpleGene, score);
163                    } catch (NullPointerException ne) {
164                        strategyMutationCount.put(charSplit[i], fullGeneCount.get(s));
165                        simpleGeneCount.put(simpleGene, fullGeneCount.get(s));
166                        simpleGeneScore.put(simpleGene, fullGeneScore.get(s));
167                        simpleGeneAltScore.put(simpleGene, fullGeneAltScore.get(s));
168                    }
169                }
170            }
171        }
172        logger.warn("Printing mutation counts in surviving plans");
173        PrintWriter writer = null;
174        try {
175            writer = getWriter(append, "survivorMutationCounts.csv", "iter,mutator,count");
176            Iterator<Map.Entry<String, Integer>> iterator = strategyMutationCount.entrySet().iterator();
177            while (iterator.hasNext()) {
178                Map.Entry<String, Integer> next = iterator.next();
179                writer.println(String.format("%d,%s,%d", event.getIteration(), next.getKey(), next.getValue()));
180            }
181            writer.flush();
182            writer.close();
183            logger.warn("Printing number of surviving plans by mutation and iteration");
184            keys = new ArrayList<>();
185            keys.addAll(simpleGeneCount.keySet());
186            Collections.sort(keys);
187            writer = getWriter(append, "mutationAtIterSurvivorCounts.csv", String.format("%s,%s,%s,%s,%s,%s", "iter", "mutator", "mutationIter", "survivors", "avg.score", "avg.alt.score"));
188            for (String key : keys) {
189                writer.println(String.format("%s,%s,%d,%d,%.3f,%.3f", event.getIteration(), key.substring(0, 1), Integer.parseInt(key.substring(1)), simpleGeneCount.get(key), simpleGeneScore.get(key), simpleGeneAltScore.get(key)));
190
191            }
192            writer.flush();
193            writer.close();
194            logger.warn("Gene lengths and average scores");
195            writer = getWriter(append, "geneLengths.csv", String.format("%s,%s,%s,%s", "iter", "len", "num", "avgscor"));
196            for (int gl : geneLengths.keySet()) {
197                writer.println(String.format("%d,%d,%d,%.3f", event.getIteration(), gl, geneLengths.get(gl), geneScoresbyLength.get(gl)));
198            }
199            writer.flush();
200            writer.close();
201        } catch (IOException e) {
202            e.printStackTrace();
203        }
204        logger.warn("Done");
205
206    }
207
208    public PrintWriter getWriter(boolean append, String fileName, String header) throws IOException {
209        PrintWriter writer;
210        if (append) {
211            writer = new PrintWriter(IOUtils.getAppendingBufferedWriter(outputPath + "/" + fileName));
212        } else {
213            writer = new PrintWriter(IOUtils.getBufferedWriter(outputPath + "/" + fileName));
214            writer.println(header);
215        }
216        return writer;
217    }
218}

220class SimpleLink {
221    final int count;
222    final double score;
223    final double altscore;
224    String from;
225    String to;
226    String type;
227
228    public SimpleLink(String genome, int count, double score, double altscore) {
229        to = genome;
230        from = genome.substring(0, genome.length() - 5);
231        type = genome.substring(genome.length() - 5, genome.length());
232        this.count = count;
233        this.score = score;
234        this.altscore = altscore;
235    }
