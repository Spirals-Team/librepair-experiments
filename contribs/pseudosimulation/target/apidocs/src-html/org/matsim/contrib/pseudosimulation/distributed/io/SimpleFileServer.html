001package org.matsim.contrib.pseudosimulation.distributed.io;
002
003import java.io.BufferedInputStream;
004import java.io.File;
005import java.io.FileInputStream;
006import java.io.IOException;
007import java.io.OutputStream;
008import java.net.ServerSocket;
009import java.net.Socket;
010
011public class SimpleFileServer {
012
013    public final static int SOCKET_PORT = 13267;  // you may change this
014    public final static String FILE_TO_SEND = "Siouxfalls_population.xml";  // you may change this
015
016    public static void main(String[] args) throws IOException {
017        FileInputStream fis = null;
018        BufferedInputStream bis = null;
019        OutputStream os = null;
020        ServerSocket servsock = null;
021        Socket sock = null;
022        try {
023            servsock = new ServerSocket(SOCKET_PORT);
024            while (true) {
025                System.out.println("Waiting...");
026                try {
027                    sock = servsock.accept();
028                    System.out.println("Accepted connection : " + sock);
029                    // send file
030                    File myFile = new File(FILE_TO_SEND);
031                    byte[] mybytearray = new byte[(int) myFile.length()];
032                    fis = new FileInputStream(myFile);
033                    bis = new BufferedInputStream(fis);
034                    bis.read(mybytearray, 0, mybytearray.length);
035                    os = sock.getOutputStream();
036                    System.out.println("Sending " + FILE_TO_SEND + "(" + mybytearray.length + " bytes)");
037                    os.write(mybytearray, 0, mybytearray.length);
038                    os.flush();
039                    System.out.println("Done.");
040                } finally {
041                    if (bis != null) bis.close();
042                    if (os != null) os.close();
043                    if (sock != null) sock.close();
044                }
045            }
046        } finally {
047            if (servsock != null) servsock.close();
048        }
049    }
050}
