020package org.matsim.core.mobsim.qsim.qnetsimengine;
021
022import javax.inject.Inject;
023
024import org.matsim.api.core.v01.Scenario;
025import org.matsim.api.core.v01.network.Link;
026import org.matsim.api.core.v01.network.Network;
027import org.matsim.api.core.v01.network.Node;
028import org.matsim.contrib.hybridsim.simulation.ExternalEngine;
029import org.matsim.core.api.experimental.events.EventsManager;
030import org.matsim.core.config.groups.QSimConfigGroup;
031import org.matsim.core.mobsim.framework.MobsimTimer;
032import org.matsim.core.mobsim.qsim.interfaces.AgentCounter;
033import org.matsim.core.mobsim.qsim.qnetsimengine.QNetsimEngine.NetsimInternalInterface;
034import org.matsim.core.mobsim.qsim.qnetsimengine.linkspeedcalculator.DefaultLinkSpeedCalculator;
035import org.matsim.core.mobsim.qsim.qnetsimengine.linkspeedcalculator.LinkSpeedCalculator;
036import org.matsim.vis.snapshotwriters.SnapshotLinkWidthCalculator;
037
038public class HybridNetworkFactory extends QNetworkFactory {
039        @Inject QSimConfigGroup qsimConfig ;
040        @Inject EventsManager events ;
041        @Inject Scenario scenario ;
042
043        private Network network ;
044        private NetsimEngineContext context;
045        private NetsimInternalInterface netsimEngine ;
046
047
048        private ExternalEngine externalEngine;
049
050        @Override
051        void initializeFactory(AgentCounter agentCounter, MobsimTimer mobsimTimer, NetsimInternalInterface arg2) {
052                network = arg2.getNetsimNetwork().getNetwork();
053                double effectiveCellSize = ( network).getEffectiveCellSize() ;
054
055                SnapshotLinkWidthCalculator linkWidthCalculator = new SnapshotLinkWidthCalculator();
056                linkWidthCalculator.setLinkWidthForVis( qsimConfig.getLinkWidthForVis() );
057                if (! Double.isNaN(network.getEffectiveLaneWidth())){
058                        linkWidthCalculator.setLaneWidth( network.getEffectiveLaneWidth() );
059                }
060                AbstractAgentSnapshotInfoBuilder snapshotInfoBuilder = QNetsimEngine.createAgentSnapshotInfoBuilder( scenario, linkWidthCalculator );
061
062                this.context = new NetsimEngineContext( events, effectiveCellSize, agentCounter, snapshotInfoBuilder, qsimConfig, mobsimTimer, linkWidthCalculator ) ;
063                
064                this.netsimEngine = arg2 ;
065
066        }
067
068        @Override
069        public QNodeI createNetsimNode(Node node) {
070                QNodeImpl.Builder builder = new QNodeImpl.Builder( netsimEngine, context ) ;
071                return builder.build( node ) ;
072        }
073
074
075        @Override
076        public QLinkI createNetsimLink(Link link, QNodeI queueNode) {
077                if (link.getAllowedModes().contains("2ext")) {
078                        LinkSpeedCalculator linkSpeedCalculator = new DefaultLinkSpeedCalculator() ;
079                        // yyyyyy I don't think that this would have been set correctly before I refactored this.  kai, feb'18
080                        return new QSimExternalTransitionLink(link, this.externalEngine, context, netsimEngine, queueNode, linkSpeedCalculator);
081                }
082//              QLinkImpl ret = new QLinkImpl(link, network, queueNode, linkSpeedCalculator);
083                QLinkImpl.Builder linkBuilder = new QLinkImpl.Builder(context, netsimEngine );
084                QLinkImpl ret = linkBuilder.build(link, queueNode) ;
085                if (link.getAllowedModes().contains("ext2")) {
086                        this.externalEngine.registerAdapter(new QLinkInternalIAdapter(ret));
087                }
088                return ret;
089        }
090        
091        public void setExternalEngine(ExternalEngine externalEngine) {
092                this.externalEngine = externalEngine;
093        }
094}
