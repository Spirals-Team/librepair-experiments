332        private static void enrichConfig(Config c) {
333                PlanCalcScoreConfigGroup.ActivityParams pre = new PlanCalcScoreConfigGroup.ActivityParams("origin");
334                pre.setTypicalDuration(49); // needs to be geq 49, otherwise when
335                // running a simulation one gets
336                // "java.lang.RuntimeException: zeroUtilityDuration of type pre-evac must be greater than 0.0. Did you forget to specify the typicalDuration?"
337                // the reason is the double precision. see also comment in
338                // ActivityUtilityParameters.java (gl)
339                pre.setMinimalDuration(49);
340                pre.setClosingTime(49);
341                pre.setEarliestEndTime(49);
342                pre.setLatestStartTime(49);
343                pre.setOpeningTime(49);
344
345                PlanCalcScoreConfigGroup.ActivityParams post = new PlanCalcScoreConfigGroup.ActivityParams("destination");
346                post.setTypicalDuration(49); // dito
347                post.setMinimalDuration(49);
348                post.setClosingTime(49);
349                post.setEarliestEndTime(49);
350                post.setLatestStartTime(49);
351                post.setOpeningTime(49);
352                c.planCalcScore().addActivityParams(pre);
353                c.planCalcScore().addActivityParams(post);
354
355                c.planCalcScore().setLateArrival_utils_hr(0.);
356                c.planCalcScore().setPerforming_utils_hr(0.);
357        }

390        private static void createNetwork(Scenario sc) {
391                Network net = sc.getNetwork();
392                net.setCapacityPeriod(1);
393                net.setEffectiveLaneWidth(0.71);
394                net.setEffectiveCellSize(0.26);
395                NetworkFactory fac = net.getFactory();
396                Node n0 = fac.createNode(Id.createNodeId(0), CoordUtils.createCoord(93.85, 110.1));
397                net.addNode(n0);
398                Node n1 = fac.createNode(Id.createNodeId(1), CoordUtils.createCoord(103.85, 110.1));
399                net.addNode(n1);
400                Node n2 = fac.createNode(Id.createNodeId(2), CoordUtils.createCoord(113.85, 110.1));
401                net.addNode(n2);
402                Node n3 = fac.createNode(Id.createNodeId(3), CoordUtils.createCoord(200, 110.1));
403                net.addNode(n3);
404                Node n4 = fac.createNode(Id.createNodeId(4), CoordUtils.createCoord(210, 110.1));
405                net.addNode(n4);
406                Node n5 = fac.createNode(Id.createNodeId(5), CoordUtils.createCoord(220, 110.1));
407                net.addNode(n5);
408                Link l0 = fac.createLink(Id.createLinkId(0), n0, n1);
409                net.addLink(l0);
410                Link l1 = fac.createLink(Id.createLinkId(1), n1, n2);
411                net.addLink(l1);
412                Link lJPS0 = fac.createLink(Id.createLinkId("JPS0"), n2, n3);
413                net.addLink(lJPS0);
414                Link l2 = fac.createLink(Id.createLinkId("2"), n3, n4);
415                net.addLink(l2);
416                Link l3 = fac.createLink(Id.createLinkId("3"), n4, n5);
417                net.addLink(l3);
418                Link l3r = fac.createLink(Id.createLinkId("3r"), n5, n4);
419                net.addLink(l3r);
420                Link l2r = fac.createLink(Id.createLinkId("2r"), n4, n3);
421                net.addLink(l2r);
422                Link lJPS0r = fac.createLink(Id.createLinkId("JPS0r"), n3, n2);
423                net.addLink(lJPS0r);
424                Link l1r = fac.createLink(Id.createLinkId("1r"), n2, n1);
425                net.addLink(l1r);
426                Link l0r = fac.createLink(Id.createLinkId("0r"), n1, n0);
427                net.addLink(l0r);
428
429                for (Link l : net.getLinks().values()) {
430                        l.setFreespeed(1.34);
431                        l.setCapacity(1.8886);
432                        l.setNumberOfLanes(1.12);
433                        l.setLength(CoordUtils.calcEuclideanDistance(l.getFromNode().getCoord(),l.getToNode().getCoord()));
434                }
435                Set<String> ext = new HashSet<>();
436                ext.add("car"); ext.add("ext"); ext.add("2ext");
437                lJPS0.setAllowedModes(ext); lJPS0r.setAllowedModes(ext);
438                Set<String> ext2 = new HashSet<>();
439                ext2.add("car"); ext2.add("ext2");
440                l1r.setAllowedModes(ext2);
441                l2.setAllowedModes(ext2);
442
443

445        }
