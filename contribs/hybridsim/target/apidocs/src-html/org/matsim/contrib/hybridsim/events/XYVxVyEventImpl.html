020package org.matsim.contrib.hybridsim.events;
021
022import org.matsim.api.core.v01.Id;
023import org.matsim.api.core.v01.events.Event;
024import org.matsim.api.core.v01.population.Person;
025
026
027import java.util.Map;
028
029public class XYVxVyEventImpl extends Event  {
030
031        public static final String EVENT_TYPE = "XYVxVyEvent";
032        public static final String ATTRIBUTE_X = "x";
033        public static final String ATTRIBUTE_Y = "y";
034        public static final String ATTRIBUTE_VX = "vx";
035        public static final String ATTRIBUTE_VY = "vy";
036        
037        public static final String ATTRIBUTE_PERSON = "person";
038
039        private final double x;
040        private final double y;
041        private final double vx;
042        private final double vy;
043        private final Id<Person> personId;
044
045
046        public XYVxVyEventImpl(Id<Person> id, double x, double y, double vx, double vy, double time) {
047                super(time);
048                this.personId = id;
049                this.x = x;
050                this.y = y;
051                this.vx = vx;
052                this.vy = vy;
053        }
054
055
056
057        
058        @Override
059        public Map<String, String> getAttributes() {
060                Map<String, String> attr = super.getAttributes();
061                attr.put(ATTRIBUTE_X, Double.toString(this.x));
062                attr.put(ATTRIBUTE_Y, Double.toString(this.y));
063                attr.put(ATTRIBUTE_VX, Double.toString(this.vx));
064
065                attr.put(ATTRIBUTE_PERSON, this.personId.toString());
066                attr.put(ATTRIBUTE_VY, Double.toString(this.vy));
067                return attr;
068        }
069        
070        public double getX() {
071                return this.x;
072        }
073
074        public double getY() {
075                return this.y;
076        }
077
078
079        public double getVX() {
080                return this.vx;
081        }
082
083        public double getVY() {
084                return this.vy;
085        }
086        
087        public Id<Person> getPersonId() {
088                return this.personId;
089        }
090
091        @Override
092        public String getEventType() {
093                return EVENT_TYPE;
094        }
095        
096
097}
