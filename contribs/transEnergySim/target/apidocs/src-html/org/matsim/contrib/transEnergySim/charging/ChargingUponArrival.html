107        public ChargingUponArrival(HashMap<Id<Vehicle>, Vehicle> vehicles, MatsimServices controller) {
108                this.vehicles = vehicles;
109                this.controller = controller;
110                this.setDefaultValues(chargablePowerAtActivityTypes);
111                controller.addControlerListener(this);
112                chargablePowerAtActivityTypes = new DoubleValueHashMap<String>();
113                enableLogging();
114        }

116        @Override
117        public void reset(int iteration) {
118                firstDepartureTimeOfDay = new DoubleValueHashMap<Id<Vehicle>>();
119                previousCarArrivalTime = new DoubleValueHashMap<Id<Vehicle>>();
120                firstActivityTypeAfterCarArrival = new HashMap<Id<Vehicle>, String>();
121                firstFacilityIdAfterCarArrival = new HashMap<>();
122                setLog(new StationaryChargingOutputLog());
123                previousCarArrivalLinkId = new HashMap<>();
124        }
125
126        @Override
127        public void handleEvent(PersonDepartureEvent event) {
128                if (!isVehicleWithBattery(Id.create(event.getPersonId(), Vehicle.class))) {
129                        return;
130                }
131
132                Id<Person> personId = event.getPersonId();
133                Id<Vehicle> vehicleId = Id.create(personId, Vehicle.class);
134
135                if (event.getLegMode().equals(TransportMode.car) && vehicles.containsKey(vehicleId)) {
136                        if (isFirstCarDepartureOfDay(vehicleId)) {
137                                firstDepartureTimeOfDay.put(vehicleId, event.getTime());
138                        } else {
139                                double carArrivalTime = previousCarArrivalTime.get(vehicleId);
140                                double carDepartureTime = event.getTime();
141
142                                chargeVehicle(vehicleId, carArrivalTime, carDepartureTime);
143                        }
144                }
145        }
146
147        private void chargeVehicle(Id<Vehicle> personId, double carArrivalTime, double carDepartureTime) {
148
149                double parkingDuration = GeneralLib.getIntervalDuration(carArrivalTime, carDepartureTime);
150                Double availablePowerInWatt = null;
151                String activityType = firstActivityTypeAfterCarArrival.get(personId);
152
153                availablePowerInWatt = getChargablePowerAtActivityTypes().get(activityType);
154
155                if (availablePowerInWatt == 0) {
156                        return;
157                }
158
159                if (availablePowerInWatt == null) {
160                        DebugLib.stopSystemAndReportInconsistency("power at all activity types needs to be specified, missing:" + activityType);
161                }
162
163                double chargableEnergyInJoules = availablePowerInWatt * parkingDuration;
164
165                VehicleWithBattery vehicleWithBattery = (VehicleWithBattery) vehicles.get(personId);
166                double energyToChargeInJoules = 0;
167                if (vehicleWithBattery.getRequiredEnergyInJoules() <= chargableEnergyInJoules) {
168                        energyToChargeInJoules = vehicleWithBattery.getRequiredEnergyInJoules();
169                } else {
170                        energyToChargeInJoules = chargableEnergyInJoules;
171                }
172
173                if (energyToChargeInJoules > 0) {
174                        vehicleWithBattery.addEnergyToVehicleBattery(energyToChargeInJoules);
175                        
176                        if (loggingEnabled) {
177                                Id<ActivityFacility> facilityId = firstFacilityIdAfterCarArrival.get(personId);
178                                
179                                if (facilityId != Id.create(facilityId, ActivityFacility.class)) {
180                                        System.out.println("id has wrong type");
181                                }
182                                
183                                Id<Link> linkId = controller.getScenario().getActivityFacilities().getFacilities().get(facilityId).getLinkId();
184                                
185                                if (linkId==null){
186                    linkId = NetworkUtils.getNearestLink((controller.getScenario().getNetwork()), controller.getScenario().getActivityFacilities().getFacilities().get(facilityId).getCoord()).getId();
187                                }
188                                
189                                ChargingLogRowFacilityLevel chargingLogRow = new ChargingLogRowFacilityLevel(personId, linkId,facilityId,
190                                                carArrivalTime, energyToChargeInJoules / availablePowerInWatt, energyToChargeInJoules);
191                                getLog().add(chargingLogRow);
192                        }
193                }
194
195        }
196
197        private boolean isFirstCarDepartureOfDay(Id<Vehicle> vehicleId) {
198                return !firstDepartureTimeOfDay.containsKey(vehicleId);
199        }
200
201        @Override
202        public void handleEvent(PersonArrivalEvent event) {
203                if (!isVehicleWithBattery(Id.create(event.getPersonId(), Vehicle.class))) {
204                        return;
205                }
206
207                if (event.getLegMode().equals(TransportMode.car)) {
208                        initFirstActivityAfterCarArrival(event);
209                        updateCarArrivalTime(event);
210                        updatePreviousCarArrivalLinkId(event);
211                }
212        }
213
214        private boolean isVehicleWithBattery(Id<Vehicle> vehicleId) {
215                return vehicles.get(vehicleId) instanceof VehicleWithBattery;
216        }
217
218        private void updatePreviousCarArrivalLinkId(PersonArrivalEvent event) {
219                previousCarArrivalLinkId.put(Id.create(event.getPersonId(),Vehicle.class), event.getLinkId());
220        }
221
222        private void updateCarArrivalTime(PersonArrivalEvent event) {
223                previousCarArrivalTime.put(Id.create(event.getPersonId(),Vehicle.class), event.getTime());
224        }
225
226        private void initFirstActivityAfterCarArrival(PersonArrivalEvent event) {
227                firstActivityTypeAfterCarArrival.remove(Id.create(event.getPersonId(), Vehicle.class));
228                firstFacilityIdAfterCarArrival.remove(Id.create(event.getPersonId(), Vehicle.class));
229        }
230
231        @Override
232        public void handleEvent(ActivityStartEvent event) {
233                updateFirstActivityTypeAfterCarArrival(event);
234        }
235
236        private void updateFirstActivityTypeAfterCarArrival(ActivityStartEvent event) {
237                if (!isVehicleWithBattery(Id.create(event.getPersonId(), Vehicle.class))){
238                        return;
239                }
240
241                Id<Vehicle> vehicleId = Id.create(event.getPersonId(), Vehicle.class);
242                if (!firstActivityTypeAfterCarArrival.containsKey(vehicleId)) {
243                        firstActivityTypeAfterCarArrival.put(vehicleId, event.getActType());
244                }
245
246                if (!firstFacilityIdAfterCarArrival.containsKey(vehicleId)) {
247                        firstFacilityIdAfterCarArrival.put(Id.create(event.getPersonId(),Vehicle.class), event.getFacilityId());
248                }
249        }
250
251        public void handleLastParkingOfDay() {
252                for (Id<Vehicle> vehicleId : vehicles.keySet()) {
253                        double carArrivalTime = previousCarArrivalTime.get(vehicleId);
254                        double carDepartureTime = firstDepartureTimeOfDay.get(vehicleId);
255
256                        chargeVehicle(vehicleId, carArrivalTime, carDepartureTime);
257                }
258        }

269        public void notifyAfterMobsim(AfterMobsimEvent event) {

273        public void setPowerForNonInitializedActivityTypes(ActivityFacilities facilities, double powerInWatt) {
274                this.powerAvalabilityParameters = new PowerAvalabilityParameters(facilities, powerInWatt);
275        }
276
277        public void setChargablePowerAtActivityTypes(DoubleValueHashMap<String> chargablePowerAtActivityTypes) {
278                this.chargablePowerAtActivityTypes = chargablePowerAtActivityTypes;
279        }
280
281        @Override
282        public void notifyStartup(StartupEvent event) {
283                initPowerAvailableAtStartup();
284        }
285
286        private void initPowerAvailableAtStartup() {
287                if (this.powerAvalabilityParameters != null) {
288                        for (ActivityFacility facility : this.powerAvalabilityParameters.getFacilities().getFacilities().values()) {
289                                for (ActivityOption actOption : facility.getActivityOptions().values()) {
290                                        String actType = actOption.getType();
291                                        if (!getChargablePowerAtActivityTypes().containsKey(actType)) {
292                                                getChargablePowerAtActivityTypes().put(actType, this.powerAvalabilityParameters.getPowerInWatt());
293                                        }
294                                }
295                        }
296                }
297        }
298
299        public DoubleValueHashMap<String> getChargablePowerAtActivityTypes() {
300                return chargablePowerAtActivityTypes;
301        }
302
303        public void setDefaultValues(DoubleValueHashMap<String> chargablePowerAtActivityTypes) {
304                this.setChargablePowerAtActivityTypes(chargablePowerAtActivityTypes);
305        }

310                private double powerInWatt;
311
312                public ActivityFacilities getFacilities() {
313                        return facilities;
314                }
315
316                public double getPowerInWatt() {
317                        return powerInWatt;
318                }
319
320                PowerAvalabilityParameters(ActivityFacilities facilities, double powerInWatt) {
321                        this.facilities = facilities;
322                        this.powerInWatt = powerInWatt;
323                }
324
325        }

328                loggingEnabled = true;
