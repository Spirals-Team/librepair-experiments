001package org.matsim.contrib.transEnergySim.vehicles.energyConsumption.galus;
002
003import java.util.Iterator;
004import java.util.PriorityQueue;
005
006import org.matsim.contrib.transEnergySim.vehicles.energyConsumption.EnergyConsumption;
007
008
009
010
011public class EnergyConsumptionInterpolated{
012        
013        private PriorityQueue<EnergyConsumption> queue=new PriorityQueue<EnergyConsumption>();
014        private EnergyConsumption zeroSpeedConsumption=new EnergyConsumption(0,0);
015        
016        public void add(EnergyConsumption averageConsumption){
017                queue.add(averageConsumption);
018        }
019        
020        
021        public void clearSamples(){
022                queue.clear();
023        }
024        
025        // returns for a given speed the energy consumption
026        // precondition: use this method only, if at least one average consumption added to sample
027        public double getInterpolatedEnergyConsumption(double speed, double distance){
028                assert(queue.size()>0);
029                
030                Iterator<EnergyConsumption> iter=queue.iterator();
031                while (iter.hasNext()){
032                        EnergyConsumption averageConsumption=iter.next();
033                        
034                        // if speed is equal to a sample speed
035                        if (averageConsumption.getSpeed()==speed){
036                                return averageConsumption.getEnergyConsumption()*distance;
037                        }
038                        // only if smaller than first sample speed
039                        if (averageConsumption.getSpeed()>speed){
040                                return getInterpolatedValue(zeroSpeedConsumption,averageConsumption,speed)*distance;
041                        }
042                        
043                        EnergyConsumption previousConsumption=null;
044                        
045                        while (averageConsumption.getSpeed()<speed && iter.hasNext()){
046                                previousConsumption=averageConsumption;
047                                averageConsumption=iter.next();
048                        }
049                        
050                        if (iter.hasNext()){
051                                // if there are more elements in the list, then interpolate at that point
052                                return getInterpolatedValue(previousConsumption,averageConsumption,speed)*distance;
053                        } else {
054                                // if last element, then interpolat last and first point
055                                return getInterpolatedValue(zeroSpeedConsumption,averageConsumption,speed)*distance;
056                        }
057                        
058                }
059                
060                // this case should never happen (if precondition fulfilled)
061                assert(false);
062                return 0;
063        }
064        
065        // gives for the speed an interpolated energy
066        // precondition: speed of consumption1 is smaller than consumption2 (not equal!)
067        // note: energyConsumption of consumption1 can be large than that of consumption2
068        // note: speed can be bigger than both consumption1 and consumption2 speed (but not smaller than consumption 1)
069        public static double getInterpolatedValue(EnergyConsumption consumption1, EnergyConsumption consumption2, double speed){
070                assert(consumption1.getSpeed()<consumption2.getSpeed());
071                
072                double differenceSpeed=consumption2.getSpeed()-consumption1.getSpeed();
073                double interpolationFactor=(speed-consumption1.getSpeed())/differenceSpeed;
074                
075                double differenceEnergyConsumption=consumption2.getEnergyConsumption()-consumption1.getEnergyConsumption();
076                
077                double result=consumption1.getEnergyConsumption()+interpolationFactor*differenceEnergyConsumption;
078                return result;
079        }
080        
081}
