001package org.matsim.contrib.transEnergySim.chargingInfrastructure.management;
002
003import org.matsim.contrib.transEnergySim.agents.VehicleAgent;
004import org.matsim.contrib.transEnergySim.chargingInfrastructure.stationary.ChargingPlugType;
005import org.matsim.contrib.transEnergySim.vehicles.api.Vehicle;
006
007/*
008 * A ChargingQueue manages a line of vehicles constrained by physical access to plugs which are tracked by type.
009 * 
010 * E.g. if a single charging point has 3 plus, 2 of level 2 and 1 of level 1, this class would be used to manage
011 * the total number of vehicles with access to the charging point, but as vehicle are added to the queue, they
012 * are also mapped to a particular plug type and are therefore tracked internally as separate queues, whose combined
013 * size cannot exceed the maximum queue size.
014 */
015public interface ChargingQueue {
016        
017        // Returns true if vehicle successfully added to queue and false if queue is full
018        boolean isPhysicalSiteFull();
019        VehicleAgent dequeueVehicleFromChargingQueue(ChargingPlugType plugType);
020        VehicleAgent peekAtChargingQueue(ChargingPlugType chargingPlugType);
021        int getNumAtPhysicalSite();
022        int maxSizeOfPhysicalSite();
023        boolean removeVehicleFromChargingQueue(VehicleAgent vehicle);
024        void removeVehicleFromPhysicalSite(VehicleAgent vehicle);
025        boolean addVehicleToPhysicalSite();
026        boolean addVehicleToChargingQueue(ChargingPlugType plugType, VehicleAgent vehicle);
027        int getNumInChargingQueue(ChargingPlugType chargingPlugType);
028
029}
