054/**
055 * This class should work both with jdeqsim and mobsim.
056 * 
057 * @author wrashid
058 * 
059 */
060public class InductiveStreetCharger implements PersonDepartureEventHandler, LinkEnterEventHandler, LinkLeaveEventHandler,
061                PersonArrivalEventHandler, StartupListener, VehicleEntersTrafficEventHandler, VehicleLeavesTrafficEventHandler {
062
063        private DoubleValueHashMap<Id<Link>> chargableStreets;
064        private ChargingOutputLog log;
065
066        private HashMap<Id<Vehicle>, Vehicle> vehicles;
067
068        DoubleValueHashMap<Id<Vehicle>> linkEnterTime;
069        HashMap<Id, Id> previousLinkEntered;
070
071        private Double samePowerAtAllLinks;
072        private boolean loggingEnabled;
073        
074        private Vehicle2DriverEventHandler delegate = new Vehicle2DriverEventHandler();
075
076        public InductiveStreetCharger(HashMap<Id<Vehicle>, Vehicle> vehicles, Network network, MatsimServices controller) {
077                this.setVehicles(vehicles);
078                controller.addControlerListener(this);
079                enableLogging();
080        }
081
082        private void setVehicles(HashMap<Id<Vehicle>, Vehicle> vehicles) {
083                this.vehicles = vehicles;
084        }
085
086        @Override
087        public void reset(int iteration) {
088                linkEnterTime = new DoubleValueHashMap<Id<Vehicle>>();
089                previousLinkEntered = new HashMap<Id, Id>();
090                setLog(new InductiveChargingAtRoadOutputLog());
091                delegate.reset(iteration);
092        }
093
094        @Override
095        public void handleEvent(PersonArrivalEvent event) {
096               Id<Vehicle> vehicleId = Id.create(event.getPersonId(),Vehicle.class);
097
098                if (ignoreAgent(vehicleId, event.getLinkId())) {
099                        return;
100                }
101
102                handleCharging(vehicleId, event.getLinkId(), event.getTime());
103        }
104
105        private boolean ignoreAgent(Id<Vehicle> vehicleId, Id<Link> linkId) {
106                return (samePowerAtAllLinks == null && !getChargableStreets().containsKey(linkId))
107                                && !(getVehicles().get(vehicleId) instanceof VehicleWithBattery);
108        }
109
110        @Override
111        public void handleEvent(PersonDepartureEvent event) {
112                if (ignoreAgent(Id.create(event.getPersonId(),Vehicle.class), event.getLinkId())) {
113                        return;
114                }
115
116                if (event.getLegMode().equals(TransportMode.car)) {
117                        Id<Vehicle> vehicleId = Id.create(event.getPersonId(),Vehicle.class);
118                        linkEnterTime.put(vehicleId, event.getTime());
119                }
120        }
121
122        @Override
123        public void handleEvent(LinkLeaveEvent event) {
124            Id<Vehicle> vehicleId = Id.create(delegate.getDriverOfVehicle(event.getVehicleId()),Vehicle.class);
125                if (ignoreAgent(vehicleId, event.getLinkId())) {
126                        return;
127                }
128
129                handleCharging(vehicleId, event.getLinkId(), event.getTime());
130        }
131
132        private void handleCharging(Id<Vehicle> vehicleId, Id<Link> linkId, double linkLeaveTime) {
133                if (!(getVehicles().get(vehicleId) instanceof InductivlyChargable)) {
134                        return;
135                }
136
137                double linkEnterTime = this.linkEnterTime.get(vehicleId);
138                double timeSpendOnLink = GeneralLib.getIntervalDuration(linkEnterTime, linkLeaveTime);
139
140                if (shouldLinkBeIgnored(linkEnterTime, linkLeaveTime)) {
141                        return;
142                }
143
144                double availablePowerInWatt;
145                if (samePowerAtAllLinks != null) {
146                        availablePowerInWatt = samePowerAtAllLinks;
147                } else {
148                        availablePowerInWatt = getChargableStreets().get(linkId);
149                }
150
151                double chargableEnergyInJoules = availablePowerInWatt * timeSpendOnLink;
152
153                VehicleWithBattery vehicleWithBattery = (VehicleWithBattery) getVehicles().get(vehicleId);
154                double energyToChargeInJoules = 0;
155                if (vehicleWithBattery.getRequiredEnergyInJoules() <= chargableEnergyInJoules) {
156                        energyToChargeInJoules = vehicleWithBattery.getRequiredEnergyInJoules();
157                } else {
158                        energyToChargeInJoules = chargableEnergyInJoules;
159                }
160
161                if (energyToChargeInJoules > 0) {
162                        vehicleWithBattery.addEnergyToVehicleBattery(energyToChargeInJoules);
163
164                        if (loggingEnabled) {
165                                ChargingLogRowLinkLevel chargingLogRow = new ChargingLogRowLinkLevel(vehicleId, linkId, linkEnterTime, energyToChargeInJoules
166                                                / availablePowerInWatt, energyToChargeInJoules);
167                                getLog().add(chargingLogRow);
168                        }
169                }
170
171        }
172
173        private boolean shouldLinkBeIgnored(double linkEnterTime, double linkLeaveTime) {
174                return linkEnterTime == linkLeaveTime;
175        }
176
177        @Override
178        public void handleEvent(LinkEnterEvent event) {
179                if (ignoreAgent(Id.create(delegate.getDriverOfVehicle(event.getVehicleId()), Vehicle.class), event.getLinkId())) {
180                        return;
181                }
182
183                linkEnterTime.put(Id.create(delegate.getDriverOfVehicle(event.getVehicleId()), Vehicle.class), event.getTime());
184        }
185
186        public ChargingOutputLog getLog() {
187                return log;
188        }
189
190        private void setLog(ChargingOutputLog log) {
191                this.log = log;
192        }
193
194        public HashMap<Id<Vehicle>, Vehicle> getVehicles() {
195                return vehicles;
196        }
197
198        public DoubleValueHashMap<Id<Link>> getChargableStreets() {
199                return chargableStreets;
200        }
201
202        public void setChargableStreets(DoubleValueHashMap<Id<Link>> chargableStreets) {
203                this.chargableStreets = chargableStreets;
204        }
205
206        public void setSamePowerAtAllStreets(double powerInWatt) {
207                samePowerAtAllLinks = powerInWatt;
208        }
209
210        @Override
211        public void notifyStartup(StartupEvent event) {
212                if (samePowerAtAllLinks == null) {
213                        return;
214                }
215
216                if (getChargableStreets() != null && getChargableStreets().size() > 0) {
217                        DebugLib.stopSystemAndReportInconsistency("when using method 'setSamePowerAtAllStreets', manipulation of individual roads not allowed ");
218                }
219        }
220
221        public void enableLogging() {
222                loggingEnabled = true;
223        }
224
225        public void disableLogging() {
226                loggingEnabled = false;
227        }
228
229        @Override
230        public void handleEvent(VehicleLeavesTrafficEvent event) {
231                delegate.handleEvent(event);
232        }
233
234        @Override
235        public void handleEvent(VehicleEntersTrafficEvent event) {
236                delegate.handleEvent(event);
237        }
