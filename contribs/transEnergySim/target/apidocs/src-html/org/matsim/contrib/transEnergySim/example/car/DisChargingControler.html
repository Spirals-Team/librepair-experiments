020package org.matsim.contrib.transEnergySim.example.car;
021
022import org.matsim.api.core.v01.Id;
023import org.matsim.api.core.v01.Scenario;
024import org.matsim.contrib.transEnergySim.controllers.EventHandlerGroup;
025import org.matsim.contrib.transEnergySim.vehicles.api.Vehicle;
026import org.matsim.contrib.transEnergySim.vehicles.energyConsumption.EnergyConsumptionTracker;
027import org.matsim.core.config.Config;
028import org.matsim.core.controler.AbstractModule;
029import org.matsim.core.controler.Controler;
030import org.matsim.core.scenario.ScenarioUtils;
031
032import java.util.HashMap;
033
034public class DisChargingControler {
035
036
037        private final Controler controler;
038        private HashMap<Id<Vehicle>, Vehicle> vehicles;
039        private EnergyConsumptionTracker energyConsumptionTracker;
040
041        /**
042         * @author jbischoff
043         * Sample DisChargingControler: Tracks only energy consumption of each agent at each link and dumps it to a file
044         * 
045         */
046
047        
048        public DisChargingControler(Config config,  HashMap<Id<Vehicle>, Vehicle> vehicles) {
049                Scenario scenario = ScenarioUtils.loadScenario(config);
050                this.controler = new Controler(scenario);
051                init(vehicles);
052                
053        }
054
055        private void init(HashMap<Id<Vehicle>, Vehicle> vehicles2) {
056                this.vehicles = vehicles2;
057                final EventHandlerGroup handlerGroup = new EventHandlerGroup();
058        setEnergyConsumptionTracker(new EnergyConsumptionTracker(vehicles, controler.getScenario().getNetwork()));
059                handlerGroup.addHandler(getEnergyConsumptionTracker());
060                controler.addOverridingModule(new AbstractModule() {
061                        @Override
062                        public void install() {
063                                addEventHandlerBinding().toInstance(handlerGroup);
064                        }
065                });
066        }
067        
068        
069
070        public void printStatisticsToConsole() {
071                System.out.println("energy consumption stats");
072                energyConsumptionTracker.getLog().printToConsole();
073                System.out.println("===");
074
075        }
076
077        public void writeStatisticsToFile(String filename) {
078                System.out.println("Writing energy consumption stats to" +filename);
079                energyConsumptionTracker.getLog().writeToFile(filename);
080                System.out.println("Done writing");
081
082        }
083
084
085        
086        public EnergyConsumptionTracker getEnergyConsumptionTracker() {
087                return energyConsumptionTracker;
088        }
089
090        private void setEnergyConsumptionTracker(EnergyConsumptionTracker energyConsumptionTracker) {
091                this.energyConsumptionTracker = energyConsumptionTracker;
092        }
093
094        public void run() {
095                controler.run();
096        }
097
098}
