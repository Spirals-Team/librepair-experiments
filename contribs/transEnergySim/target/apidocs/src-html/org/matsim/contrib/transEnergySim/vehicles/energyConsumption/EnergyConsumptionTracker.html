048/**
049 * This module tracks the energy consumption of vehicles based on event
050 * handling.
051 * 
052 * Special attention should be paid, when using this module in connection with
053 * parallelEventHandling, see comments regarding this on the matsim.org website. 
054 * 
055 * 
056 * 
057 * This module can handle both the energy consumption of
058 * jdeqsim and qsim.
059 * 
060 * TODO: add tests for this also
061 * 
062 * @author rashid_waraich
063 *                      jbischoff
064 * 
065 */
066
067public class EnergyConsumptionTracker implements LinkEnterEventHandler, LinkLeaveEventHandler, PersonDepartureEventHandler,
068                PersonArrivalEventHandler, VehicleEntersTrafficEventHandler, VehicleLeavesTrafficEventHandler {
069
070        private EnergyConsumptionOutputLog log;
071
072        HashMap<Id<Vehicle>, Vehicle> vehicles;
073
074        DoubleValueHashMap<Id<Vehicle>> linkEnterTime;
075//      HashMap<Id, Id> previousLinkEntered;
076
077        private final Network network;
078
079        private boolean loggingEnabled;
080        
081        private Vehicle2DriverEventHandler delegate = new Vehicle2DriverEventHandler();
082
083        public EnergyConsumptionTracker(HashMap<Id<Vehicle>, Vehicle> vehicles, Network network) {
084                this.vehicles = vehicles;
085                this.network = network;
086                enableLogging();
087                reset(-1);
088        }
089
090        @Override
091        public void reset(int iteration) {
092                linkEnterTime = new DoubleValueHashMap<Id<Vehicle>>();
093//              previousLinkEntered = new HashMap<Id, Id>();
094                setLog(new EnergyConsumptionOutputLog());
095
096                for (Vehicle vehicle : vehicles.values()) {
097                        vehicle.reset();
098                }
099                
100                delegate.reset(iteration);
101        }
102
103        @Override
104        public void handleEvent(PersonArrivalEvent event) {
105                if (event.getLegMode().equals(TransportMode.car)) {
106                        handleEnergyConsumption(Id.create(event.getPersonId(), Vehicle.class), event.getLinkId(), event.getTime());
107                            
108                }
109        }
110
111        @Override
112        public void handleEvent(PersonDepartureEvent event) {
113                if (event.getLegMode().equals(TransportMode.car)) {
114                        Id<Vehicle> vehicleId = Id.create(event.getPersonId(), Vehicle.class);
115                        linkEnterTime.put(vehicleId, event.getTime());
116                        //assumption  - Vehicle and Person Ids are the same
117                        
118                
119                        }
120        }
121
122        @Override
123        public void handleEvent(LinkLeaveEvent event) {
124                handleEnergyConsumption(Id.create(delegate.getDriverOfVehicle(event.getVehicleId()), Vehicle.class), event.getLinkId(), event.getTime());
125        }
126
127        private void handleEnergyConsumption(Id<Vehicle> vehicleId, Id<Link> linkId, double linkLeaveTime) {
128                double linkEnterTime = this.linkEnterTime.get(vehicleId);
129                double timeSpendOnLink = GeneralLib.getIntervalDuration(linkEnterTime, linkLeaveTime);
130
131                Link link = network.getLinks().get(linkId);
132                double averageSpeedDrivenInMetersPerSecond = link.getLength() / timeSpendOnLink;
133
134                if (zeroTravelTime(linkEnterTime, linkLeaveTime)) {
135                        return;
136                }
137
138                if (vehicles.containsKey(vehicleId)){
139                        
140                Vehicle vehicle = vehicles.get(vehicleId);
141                
142                double energyConsumptionInJoule=0;
143                if (averageSpeedDrivenInMetersPerSecond<=link.getFreespeed()){
144                        energyConsumptionInJoule= vehicle.updateEnergyUse(link, averageSpeedDrivenInMetersPerSecond);
145                } else {
146                        
147                        energyConsumptionInJoule= vehicle.updateEnergyUse(link.getFreespeed()*timeSpendOnLink, link.getFreespeed(), link.getFreespeed());
148                }
149                 
150
151                if (loggingEnabled) {
152                        
153                        double soc = 0;
154                        double loc = 0;
155                        if (vehicle instanceof VehicleWithBattery){
156                                VehicleWithBattery a = (VehicleWithBattery)vehicle;
157                                soc = a.getSocInJoules();
158                                loc = soc / a.getUsableBatteryCapacityInJoules();
159                        }
160                                
161//                      Id ll = new IdImpl(Math.round(link.getLength()));
162//                       getLog().add(new EnergyConsumptionLogRow(personId, ll, energyConsumptionInJoule));
163
164                        getLog().add(new EnergyConsumptionLogRow(vehicleId, linkId, energyConsumptionInJoule));
165                }
166                }
167        }
168
169        private boolean zeroTravelTime(double linkEnterTime, double linkLeaveTime) {
170                return linkEnterTime == linkLeaveTime;
171        }
172
173        @Override
174        public void handleEvent(LinkEnterEvent event) {
175                Id<Vehicle> vehicleId = Id.create(delegate.getDriverOfVehicle(event.getVehicleId()), Vehicle.class);
176                linkEnterTime.put(vehicleId, event.getTime());
177        }
178
179        public EnergyConsumptionOutputLog getLog() {
180                return log;
181        }
182
183        public void setLog(EnergyConsumptionOutputLog log) {
184                this.log = log;
185        }
186
187        public void enableLogging() {
188                loggingEnabled = true;
189        }
190
191        public void disableLogging() {
192                loggingEnabled = false;
193        }
194
195        @Override
196        public void handleEvent(VehicleLeavesTrafficEvent event) {
197                delegate.handleEvent(event);            
198        }
199
200        @Override
201        public void handleEvent(VehicleEntersTrafficEvent event) {
202                delegate.handleEvent(event);
203        }
204
205}
