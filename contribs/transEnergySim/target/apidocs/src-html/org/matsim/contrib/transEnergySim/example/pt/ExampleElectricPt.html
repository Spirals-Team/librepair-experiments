038/*
039 * This uses the input files from the following tutorial:
040 * http://matsim.org/docs/tutorials/transit
041 */
042public class ExampleElectricPt {
043
044        /**
045         * @param args
046         */
047        public static void main(String[] args) {
048                Controler controler = new Controler(args);
049                HashMap<Id<Vehicle>, PtVehicleEnergyState> ptEnergyMangementModels = new HashMap<>();
050                EnergyConsumptionModel ecm=new ConstantEnergyConsumptionModel(600);
051                double batterySize=24*3600000;
052                ptEnergyMangementModels.put(Id.create("tr_1", Vehicle.class), new PtVehicleEnergyState(batterySize, ecm, ecm));
053                ptEnergyMangementModels.put(Id.create("tr_2", Vehicle.class), new PtVehicleEnergyState(batterySize, ecm, ecm));
054                
055                DoubleValueHashMap<Id<TransitStopFacility>> chargingPowerAtStops=new DoubleValueHashMap<>();
056                chargingPowerAtStops.put(Id.create("1", TransitStopFacility.class), 1.0);
057                chargingPowerAtStops.put(Id.create("2a", TransitStopFacility.class), 10.0);
058                chargingPowerAtStops.put(Id.create("2b", TransitStopFacility.class), 5.0);
059                chargingPowerAtStops.put(Id.create("3", TransitStopFacility.class), 1.0);
060        PtVehicleEnergyControlImpl ptVehicleEnergyControl = new PtVehicleEnergyControlImpl(controler.getScenario().getNetwork(),chargingPowerAtStops);
061                
062                new ElectricPtSimModule(controler, ptEnergyMangementModels,ptVehicleEnergyControl);
063                controler.getConfig().controler().setOverwriteFileSetting(
064                                true ?
065                                                OutputDirectoryHierarchy.OverwriteFileSetting.overwriteExistingFiles :
066                                                OutputDirectoryHierarchy.OverwriteFileSetting.failIfDirectoryExists );
067                controler.run();
068        }
069        
070        private static class PtVehicleEnergyControlImpl implements PtVehicleEnergyControl{
071
072                private Network network;
073                private DoubleValueHashMap<Id<TransitStopFacility>> chargingPowerAtStops;
074
075                public PtVehicleEnergyControlImpl(Network network, DoubleValueHashMap<Id<TransitStopFacility>> chargingPowerAtStops){
076                        this.network = network;
077                        this.chargingPowerAtStops = chargingPowerAtStops;
078                }
079                
080                @Override
081                public void handleLinkTravelled(PtVehicleEnergyState ptVehicleEnergyState, Id<Link> linkId, double travelTime) {
082                        DebugLib.emptyFunctionForSettingBreakPoint();
083                        Link link = network.getLinks().get(linkId);
084                        ptVehicleEnergyState.useBattery(link.getLength(), link.getFreespeed(), link.getLength()/travelTime);
085                        DebugLib.emptyFunctionForSettingBreakPoint();
086                }
087
088                @Override
089                public void handleChargingOpportunityAtStation(PtVehicleEnergyState ptVehicleEnergyState,
090                                double durationOfStayAtStationInSeconds, Id<TransitStopFacility> stationId) {
091                        DebugLib.emptyFunctionForSettingBreakPoint();
092                        if (ptVehicleEnergyState.getSocInJoules()<ptVehicleEnergyState.getUsableBatterySize()){
093                                double charge = durationOfStayAtStationInSeconds*chargingPowerAtStops.get(stationId);
094                                if (ptVehicleEnergyState.getSocInJoules()+charge <ptVehicleEnergyState.getUsableBatterySize()){
095                                        ptVehicleEnergyState.chargeVehicle(charge);
096                                } else {
097                                        ptVehicleEnergyState.chargeVehicle(ptVehicleEnergyState.getUsableBatterySize()-ptVehicleEnergyState.getSocInJoules());
098                                }
099                        }
100                        DebugLib.emptyFunctionForSettingBreakPoint();
101                }
102                
103        }
104
105}
