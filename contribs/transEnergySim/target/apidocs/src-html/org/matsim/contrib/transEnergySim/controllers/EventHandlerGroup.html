020package org.matsim.contrib.transEnergySim.controllers;
021
022import java.util.LinkedList;
023
024import org.matsim.api.core.v01.events.ActivityEndEvent;
025import org.matsim.api.core.v01.events.ActivityStartEvent;
026import org.matsim.api.core.v01.events.PersonArrivalEvent;
027import org.matsim.api.core.v01.events.PersonDepartureEvent;
028import org.matsim.api.core.v01.events.LinkEnterEvent;
029import org.matsim.api.core.v01.events.LinkLeaveEvent;
030import org.matsim.api.core.v01.events.PersonEntersVehicleEvent;
031import org.matsim.api.core.v01.events.PersonLeavesVehicleEvent;
032import org.matsim.api.core.v01.events.VehicleEntersTrafficEvent;
033import org.matsim.api.core.v01.events.handler.ActivityEndEventHandler;
034import org.matsim.api.core.v01.events.handler.ActivityStartEventHandler;
035import org.matsim.api.core.v01.events.handler.PersonArrivalEventHandler;
036import org.matsim.api.core.v01.events.handler.PersonDepartureEventHandler;
037import org.matsim.api.core.v01.events.handler.LinkEnterEventHandler;
038import org.matsim.api.core.v01.events.handler.LinkLeaveEventHandler;
039import org.matsim.api.core.v01.events.handler.PersonEntersVehicleEventHandler;
040import org.matsim.api.core.v01.events.handler.PersonLeavesVehicleEventHandler;
041import org.matsim.api.core.v01.events.handler.VehicleEntersTrafficEventHandler;
042import org.matsim.core.events.handler.EventHandler;
043
044public class EventHandlerGroup implements ActivityStartEventHandler, PersonArrivalEventHandler,
045PersonDepartureEventHandler, LinkEnterEventHandler, LinkLeaveEventHandler, PersonEntersVehicleEventHandler, PersonLeavesVehicleEventHandler, VehicleEntersTrafficEventHandler, ActivityEndEventHandler {
046        /**
047         * All event handlers needed for electric vehicle simulation. Convenience class.
048         * 
049         */
050        protected LinkedList<EventHandler> handler = new LinkedList<EventHandler>();
051
052        public void addHandler(EventHandler handler) {
053                this.handler.add(handler);
054        }
055
056        @Override
057        public void reset(int iteration) {
058                for (EventHandler h : handler) {
059                        h.reset(iteration);
060                }
061        }
062
063        @Override
064        public void handleEvent(LinkLeaveEvent event) {
065                for (EventHandler h : handler) {
066                        if (h instanceof LinkLeaveEventHandler) {
067                                ((LinkLeaveEventHandler) h).handleEvent(event);
068                        }
069                }
070        }
071
072        @Override
073        public void handleEvent(LinkEnterEvent event) {
074                for (EventHandler h : handler) {
075                        if (h instanceof LinkEnterEventHandler) {
076                                ((LinkEnterEventHandler) h).handleEvent(event);
077                        }
078                }
079        }
080
081        @Override
082        public void handleEvent(PersonDepartureEvent event) {
083                for (EventHandler h : handler) {
084                        if (h instanceof PersonDepartureEventHandler) {
085                                ((PersonDepartureEventHandler) h).handleEvent(event);
086                        }
087                }
088        }
089
090        @Override
091        public void handleEvent(PersonArrivalEvent event) {
092                for (EventHandler h : handler) {
093                        if (h instanceof PersonArrivalEventHandler) {
094                                ((PersonArrivalEventHandler) h).handleEvent(event);
095                        }
096                }
097        }
098
099        @Override
100        public void handleEvent(ActivityStartEvent event) {
101                for (EventHandler h : handler) {
102                        if (h instanceof ActivityStartEventHandler) {
103                                ((ActivityStartEventHandler) h).handleEvent(event);
104                        }
105                }
106        }
107
108        @Override
109        public void handleEvent(PersonLeavesVehicleEvent event) {
110                for (EventHandler h : handler) {
111                        if (h instanceof PersonLeavesVehicleEventHandler) {
112                                ((PersonLeavesVehicleEventHandler) h).handleEvent(event);
113                        }
114                }
115        }
116
117        @Override
118        public void handleEvent(PersonEntersVehicleEvent event) {
119                for (EventHandler h : handler) {
120                        if (h instanceof PersonEntersVehicleEventHandler) {
121                                ((PersonEntersVehicleEventHandler) h).handleEvent(event);
122                        }
123                }
124                
125        }
126
127        @Override
128        public void handleEvent(VehicleEntersTrafficEvent event) {
129                for (EventHandler h : handler) {
130                        if (h instanceof VehicleEntersTrafficEventHandler) {
131                                ((VehicleEntersTrafficEventHandler) h).handleEvent(event);
132                        }
133                }               
134        }
135
136        @Override
137        public void handleEvent(ActivityEndEvent event) {
138                for (EventHandler h : handler) {
139                        if (h instanceof ActivityEndEventHandler) {
140                                ((ActivityEndEventHandler) h).handleEvent(event);
141                        }
142                }       
143        }
144
145
146}
