020package org.matsim.contrib.transEnergySim.vehicles.energyConsumption;
021
022import java.util.Iterator;
023import java.util.PriorityQueue;
024
025import org.matsim.api.core.v01.network.Link;
026import org.matsim.contrib.parking.parkingchoice.lib.DebugLib;
027
028public abstract class AbstractInterpolatedEnergyConsumptionModel implements EnergyConsumptionModel{
029
030        protected PriorityQueue<EnergyConsumption> queue = new PriorityQueue<EnergyConsumption>();
031        private EnergyConsumption zeroSpeedConsumption = new EnergyConsumption(0, 0);
032        
033        /**
034         * 
035         * Gives the interpolated energy consumption. Speed must be inside the
036         * original interval (borders included).
037         * 
038         * @param consumptionA
039         * @param consumptionB
040         * @param speed
041         * @return
042         */
043        protected static double getInterpolatedEnergyConsumption(EnergyConsumption consumptionA, EnergyConsumption consumptionB,
044                        double speed) {
045                EnergyConsumption smallerSpeedEC;
046                EnergyConsumption biggerSpeedEC;
047
048                if (consumptionA.getSpeed() < consumptionB.getSpeed()) {
049                        smallerSpeedEC = consumptionA;
050                        biggerSpeedEC = consumptionB;
051                } else if (consumptionA.getSpeed() < consumptionB.getSpeed()) {
052                        smallerSpeedEC = consumptionB;
053                        biggerSpeedEC = consumptionA;
054                } else {
055                        return (consumptionA.getEnergyConsumption() + consumptionB.getEnergyConsumption()) / 2;
056                }
057
058                if (speed < smallerSpeedEC.getSpeed() || speed > biggerSpeedEC.getSpeed()) {
059                        DebugLib.stopSystemAndReportInconsistency("input speed is not inside given interval");
060                }
061
062                if (speed == smallerSpeedEC.getSpeed()) {
063                        return smallerSpeedEC.getEnergyConsumption();
064                }
065
066                if (speed == biggerSpeedEC.getSpeed()) {
067                        return biggerSpeedEC.getEnergyConsumption();
068                }
069
070                double differenceSpeed = biggerSpeedEC.getSpeed() - smallerSpeedEC.getSpeed();
071                double differenceEnergyConsumption = biggerSpeedEC.getEnergyConsumption() - smallerSpeedEC.getEnergyConsumption();
072
073                double interpolationFactor = differenceEnergyConsumption / differenceSpeed;
074
075                double result = smallerSpeedEC.getEnergyConsumption() + interpolationFactor * (speed - smallerSpeedEC.getSpeed());
076                return result;
077        }
078        
079        /**
080         * @param speedInMetersPerSecond
081         * @param distanceInMeters
082         * @return
083         */
084        protected double getInterpolatedEnergyConsumption(double speedInMetersPerSecond, double distanceInMeters) {
085                Iterator<EnergyConsumption> iter = queue.iterator();
086
087                EnergyConsumption currentAverageConsumption = iter.next();
088
089                if (ifSmallerThanMimimumSpeed(speedInMetersPerSecond, currentAverageConsumption)) {
090                        return currentAverageConsumption.getEnergyConsumption() * distanceInMeters;
091                }
092                EnergyConsumption previousConsumption = null;
093
094                while (currentAverageConsumption.getSpeed() < speedInMetersPerSecond && iter.hasNext()) {
095                        previousConsumption = currentAverageConsumption;
096                        currentAverageConsumption = iter.next();
097                }
098
099                if (ifHigherThanMaxSpeed(speedInMetersPerSecond, currentAverageConsumption)) {
100                        return currentAverageConsumption.getEnergyConsumption() * distanceInMeters;
101                } else {
102                        return getInterpolatedEnergyConsumption(previousConsumption, currentAverageConsumption, speedInMetersPerSecond)
103                                        * distanceInMeters;
104                }
105        }
106        
107        private boolean ifHigherThanMaxSpeed(double speedInMetersPerSecond, EnergyConsumption currentAverageConsumption) {
108                return currentAverageConsumption.getSpeed() < speedInMetersPerSecond;
109        }
110
111        private boolean ifSmallerThanMimimumSpeed(double speedInMetersPerSecond, EnergyConsumption currentAverageConsumption) {
112                return currentAverageConsumption.getSpeed() >= speedInMetersPerSecond;
113        }
114        
115        @Override
116        public double getEnergyConsumptionForLinkInJoule(Link link, double averageSpeedDriven) {
117                return getEnergyConsumptionForLinkInJoule(link.getLength(), -1, averageSpeedDriven);
118        }
119
120        @Override
121        public double getEnergyConsumptionForLinkInJoule(double drivenDistanceInMeters, double maxSpeedOnLink, double averageSpeedDriven) {
122                return getInterpolatedEnergyConsumption(averageSpeedDriven, drivenDistanceInMeters);
123        }
124}
