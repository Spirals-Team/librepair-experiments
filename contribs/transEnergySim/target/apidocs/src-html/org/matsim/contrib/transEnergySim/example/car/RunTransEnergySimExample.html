024import java.util.HashMap;
025
026import org.apache.log4j.Logger;
027import org.matsim.api.core.v01.Id;
028import org.matsim.api.core.v01.Scenario;
029import org.matsim.api.core.v01.population.Person;
030import org.matsim.contrib.transEnergySim.vehicles.api.Vehicle;
031import org.matsim.contrib.transEnergySim.vehicles.energyConsumption.EnergyConsumptionModel;
032import org.matsim.contrib.transEnergySim.vehicles.energyConsumption.ricardoFaria2012.EnergyConsumptionModelRicardoFaria2012;
033import org.matsim.contrib.transEnergySim.vehicles.impl.BatteryElectricVehicleImpl;
034import org.matsim.core.config.Config;
035import org.matsim.core.config.ConfigUtils;
036import org.matsim.core.controler.OutputDirectoryHierarchy.OverwriteFileSetting;
037import org.matsim.core.scenario.ScenarioUtils;
038
039/**
040 * 
041 */
042/**
043 * @author jbischoff
044 * Sample main file for setting up an Electric Vehicle simulation
045 * to run this, you will need:
046 * a config, network and a population
047 * 
048 */
049
050
051public class RunTransEnergySimExample {
052        
053        private Scenario sc;
054        
055        private static final Logger log = Logger.getLogger(RunTransEnergySimExample.class);
056        private static final String CONFIG = "test/input/org/matsim/contrib/transEnergySim/controllers/TestInductiveChargingController/config.xml";
057        private static  String outputDir =  "output/"; // Simulation output 
058        private static  String estats ;
059        
060        private DisChargingControler c;
061
062        public void run(){
063                c.run();
064                c.writeStatisticsToFile(estats);
065                
066        }
067        public static void main(String[] args){
068                if (args.length==1){
069                        // use the given output if args is not null
070                        outputDir = args[0];
071                }
072                estats = outputDir + "/estats.txt";
073                RunTransEnergySimExample runner = new RunTransEnergySimExample();
074                Config config = ConfigUtils.loadConfig(CONFIG);
075                config.controler().setOverwriteFileSetting(OverwriteFileSetting.deleteDirectoryIfExists);
076                config.controler().setOutputDirectory(outputDir);
077                runner.setUpControler(config);
078                runner.run();
079        }
080
081
082
083
084
085
086        private void setUpControler(Config config) {
087                log.info("Setting up emob controler");
088                int batteryCapacityInJoules = 25*1000*3600;
089                //assuming a 25kwh battery
090                EnergyConsumptionModel faria = new EnergyConsumptionModelRicardoFaria2012();
091                this.sc = ScenarioUtils.loadScenario(config);
092                HashMap<Id<Vehicle>,Vehicle> vehicles = new HashMap<Id<Vehicle>, Vehicle>();
093                for (Person p : this.sc.getPopulation().getPersons().values()){
094                    Id<Vehicle> vid = Id.create(p.getId(), Vehicle.class);
095                        vehicles.put(vid, new BatteryElectricVehicleImpl(faria, batteryCapacityInJoules, vid));
096                        //gives every person of the population an electric car
097                }
098                
099                this.c = new DisChargingControler(sc.getConfig(), vehicles);
100
101        
102                
103        }
104        
105        
106}
