	public static byte[] convertIntToByteArray(int value) {
		int originValue = value;
		int size = 0;
		if (value < 0) {
			value = -value;
		}

		while (value > sizeTable[size])
			size++;
		size++;
		if (originValue < 0)
			size++;

		int q, r;
		byte[] buf = new byte[size];
		if (originValue < 0) {
			buf[0] = '-';
		}
		int charPos = size;

		while (value >= 65536) {
			q = value / 100;
			r = value - ((q << 6) + (q << 5) + (q << 2));
			value = q;
			buf[--charPos] = DigitOnes[r];
			buf[--charPos] = DigitTens[r];
		}

		for (;;) {
			q = (value * 52429) >>> (16 + 3);
			r = value - ((q << 3) + (q << 1));
			buf[--charPos] = digits[r];
			value = q;
			if (value == 0)
				break;
		}
		return buf;
	}

	public static byte[] buildErrorReplyBytes(String message) {

		return ("ERR " + message).getBytes();
	}

	public static int readInt(ByteBuf buffer) throws IndexOutOfBoundsException {

		long size = 0;
		boolean isNeg = false;

		byte b =buffer.readByte();
		while (b != '\r') {
			if (b == '-') {
				isNeg = true;
			} else {
				size = size * 10 + b - '0';
			}
			b = buffer.readByte();
		}

		// skip \r\n
		buffer.readByte();

		size = (isNeg ? -size : size);
		if (size > Integer.MAX_VALUE) {
			throw new RuntimeException("Cannot allocate more than " + Integer.MAX_VALUE + " bytes");
		}
		if (size < Integer.MIN_VALUE) {
			throw new RuntimeException("Cannot allocate less than " + Integer.MIN_VALUE + " bytes");
		}
		return (int) size;
	}
