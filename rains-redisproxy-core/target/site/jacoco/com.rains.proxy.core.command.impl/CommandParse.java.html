    // 解析特殊指令策略
	public static RedisRequestPolicy getPolicy(String cmd) {
		RedisRequestPolicy policy = _cmds.get( cmd );
		return policy == null ? new RedisRequestPolicy(UNKNOW_CMD, UNKNOW_CMD, UNKNOW_CMD) : policy;		
	}
